96/**
97 * \def U_ICUDATA_TYPE_LETTER
98 *
99 * This is a platform-dependent string containing one letter:
100 * - b for big-endian, ASCII-family platforms
101 * - l for little-endian, ASCII-family platforms
102 * - e for big-endian, EBCDIC-family platforms
103 * This letter is part of the common data file name.
104 * @stable ICU 2.0
105 */
106
107/**
108 * \def U_ICUDATA_TYPE_LITLETTER
109 * The non-string form of U_ICUDATA_TYPE_LETTER
110 * @stable ICU 2.0
111 */

113# if U_IS_BIG_ENDIAN
114 /* EBCDIC - should always be BE */
115# define U_ICUDATA_TYPE_LETTER "e"
116# define U_ICUDATA_TYPE_LITLETTER e
117# else
118# error "Don't know what to do with little endian EBCDIC!"
119# define U_ICUDATA_TYPE_LETTER "x"
120# define U_ICUDATA_TYPE_LITLETTER x
121# endif
122#else
123# if U_IS_BIG_ENDIAN
124 /* Big-endian ASCII */
125# define U_ICUDATA_TYPE_LETTER "b"
126# define U_ICUDATA_TYPE_LITLETTER b
127# else
128 /* Little-endian ASCII */
129# define U_ICUDATA_TYPE_LETTER "l"
130# define U_ICUDATA_TYPE_LITLETTER l
131# endif
132#endif

387/**
388 * Standard ICU4C error code type, a substitute for exceptions.
389 *
390 * Initialize the UErrorCode with U_ZERO_ERROR, and check for success or
391 * failure using U_SUCCESS() or U_FAILURE():
392 *
393 * UErrorCode errorCode = U_ZERO_ERROR;
394 * // call ICU API that needs an error code parameter.
395 * if (U_FAILURE(errorCode)) {
396 * // An error occurred. Handle it here.
397 * }
398 *
399 * C++ code should use icu::ErrorCode, available in unicode/errorcode.h, or a
400 * suitable subclass.
401 *
402 * For more information, see:
403 * http://icu-project.org/userguide/conventions
404 *
405 * Note: By convention, ICU functions that take a reference (C++) or a pointer
406 * (C) to a UErrorCode first test:
407 *
408 * if (U_FAILURE(errorCode)) { return immediately; }
409 *
410 * so that in a chain of such functions the first one that sets an error code
411 * causes the following ones to not perform any operations.
412 *
413 * @stable ICU 2.0
414 */

418 * which is not the what the code is used for
419 */

686#ifdef __cplusplus

693 /**
694 * Does the error code indicate a failure?
695 * @stable ICU 2.0
696 */
697 static
698 inline UBool U_FAILURE(UErrorCode code) { return (UBool)(code>U_ZERO_ERROR); }
699#else
700 /**
701 * Does the error code indicate success?
702 * @stable ICU 2.0
703 */
704# define U_SUCCESS(x) ((x)<=U_ZERO_ERROR)
705 /**
706 * Does the error code indicate a failure?
707 * @stable ICU 2.0
708 */
709# define U_FAILURE(x) ((x)>U_ZERO_ERROR)
