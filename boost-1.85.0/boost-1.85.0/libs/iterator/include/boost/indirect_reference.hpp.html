21
22namespace detail

24 template <class P>
25 struct smart_ptr_reference

28 };
29}

31template <class P>
32struct indirect_reference
33 : mpl::eval_if<

35 , iterator_reference<P>
36 , detail::smart_ptr_reference<P>
37 >

39};
