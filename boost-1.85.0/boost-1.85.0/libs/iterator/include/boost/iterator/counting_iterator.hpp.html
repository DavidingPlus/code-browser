36 // Try to detect numeric types at compile time in ways compatible
37 // with the limitations of the compiler and library.
38 template <class T>
39 struct is_numeric_impl
40 {
41 // For a while, this wasn't true, but we rely on it below. This is a regression assert.
42 BOOST_STATIC_ASSERT(::boost::is_integral<char>::value);
43
44# ifndef BOOST_NO_LIMITS_COMPILE_TIME_CONSTANTS
45
46 BOOST_STATIC_CONSTANT(bool, value = std::numeric_limits<T>::is_specialized);
47
48# else
49
50# if !BOOST_WORKAROUND(BOOST_BORLANDC, BOOST_TESTED_AT(0x551))
51 BOOST_STATIC_CONSTANT(
52 bool, value = (
53 boost::is_convertible<int,T>::value
54 && boost::is_convertible<T,int>::value
55 ));
56# else
57 BOOST_STATIC_CONSTANT(bool, value = ::boost::is_arithmetic<T>::value);
58# endif
59
60# endif
61 };

63 template <class T>
64 struct is_numeric
65 : boost::integral_constant<bool, ::boost::iterators::detail::is_numeric_impl<T>::value>
66 {};
67
68# if defined(BOOST_HAS_LONG_LONG)
69 template <>
70 struct is_numeric<boost::long_long_type>
71 : boost::true_type {};
72
73 template <>
74 struct is_numeric<boost::ulong_long_type>
75 : boost::true_type {};
76# endif
77
78# if defined(BOOST_HAS_INT128)
79 template <>
80 struct is_numeric<boost::int128_type>
81 : boost::true_type {};

83 template <>

114 template <class Incrementable, class CategoryOrTraversal, class Difference>
115 struct counting_iterator_base
116 {
117 typedef typename detail::ia_dflt_help<
118 CategoryOrTraversal
119 , typename boost::conditional<
120 is_numeric<Incrementable>::value
121 , boost::type_identity<random_access_traversal_tag>
122 , iterator_traversal<Incrementable>
123 >::type
124 >::type traversal;
125
126 typedef typename detail::ia_dflt_help<
127 Difference
128 , typename boost::conditional<
129 is_numeric<Incrementable>::value
130 , numeric_difference<Incrementable>
131 , iterator_difference<Incrementable>
132 >::type
133 >::type difference;
134
135 typedef iterator_adaptor<
136 counting_iterator<Incrementable, CategoryOrTraversal, Difference> // self
137 , Incrementable // Base
138 , Incrementable // Value
139# ifndef BOOST_ITERATOR_REF_CONSTNESS_KILLS_WRITABILITY
140 const // MSVC won't strip this. Instead we enable Thomas'
141 // criterion (see boost/iterator/detail/facade_iterator_category.hpp)
142# endif
143 , traversal
144 , Incrementable const& // reference
145 , difference
146 > type;
147 };

149 // Template class distance_policy_select -- choose a policy for computing the
150 // distance between counting_iterators at compile-time based on whether or not
151 // the iterator wraps an integer or an iterator, using "poor man's partial
152 // specialization".
153
154 template <bool is_integer> struct distance_policy_select;
155
156 // A policy for wrapped iterators
157 template <class Difference, class Incrementable1, class Incrementable2>
158 struct iterator_distance
159 {
160 static Difference distance(Incrementable1 x, Incrementable2 y)
161 {
162 return y - x;
163 }
164 };
165
166 // A policy for wrapped numbers
167 template <class Difference, class Incrementable1, class Incrementable2>
168 struct number_distance
169 {
170 static Difference distance(Incrementable1 x, Incrementable2 y)
171 {
172 return boost::detail::numeric_distance(x, y);
173 }
174 };
175}

177template <
178 class Incrementable
179 , class CategoryOrTraversal = use_default
180 , class Difference = use_default
181>
182class counting_iterator
183 : public detail::counting_iterator_base<
184 Incrementable, CategoryOrTraversal, Difference
185 >::type
186{
187 typedef typename detail::counting_iterator_base<
188 Incrementable, CategoryOrTraversal, Difference
189 >::type super_t;
190
191 friend class iterator_core_access;
192
193 public:
194 typedef typename super_t::difference_type difference_type;
195
196 BOOST_DEFAULTED_FUNCTION(counting_iterator(), {})
197
198 BOOST_DEFAULTED_FUNCTION(counting_iterator(counting_iterator const& rhs), : super_t(rhs.base()) {})
199
200 counting_iterator(Incrementable x)
201 : super_t(x)
202 {
203 }
204
205# if 0
206 template<class OtherIncrementable>
207 counting_iterator(
208 counting_iterator<OtherIncrementable, CategoryOrTraversal, Difference> const& t
209 , typename enable_if_convertible<OtherIncrementable, Incrementable>::type* = 0
210 )
211 : super_t(t.base())
212 {}
213# endif
214
215 BOOST_DEFAULTED_FUNCTION(counting_iterator& operator=(counting_iterator const& rhs), { *static_cast< super_t* >(this) = static_cast< super_t const& >(rhs); return *this; })
216
217 private:
218
219 typename super_t::reference dereference() const
220 {
221 return this->base_reference();
222 }
223
224 template <class OtherIncrementable>
225 difference_type
226 distance_to(counting_iterator<OtherIncrementable, CategoryOrTraversal, Difference> const& y) const
227 {
228 typedef typename boost::conditional<
229 detail::is_numeric<Incrementable>::value
230 , detail::number_distance<difference_type, Incrementable, OtherIncrementable>
231 , detail::iterator_distance<difference_type, Incrementable, OtherIncrementable>
232 >::type d;
233
234 return d::distance(this->base(), y.base());
235 }
236};

238// Manufacture a counting iterator for an arbitrary incrementable type
239template <class Incrementable>
240inline counting_iterator<Incrementable>
241make_counting_iterator(Incrementable x)
242{
243 typedef counting_iterator<Incrementable> result_t;
244 return result_t(x);
245}
