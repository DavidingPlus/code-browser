30namespace boost {

32BOOST_LOG_OPEN_NAMESPACE
33
34/*!
35 * Attribute name exception information
36 */
37typedef error_info< struct attribute_name_info_tag, attribute_name > attribute_name_info;
38
39/*!
40 * Type info exception information
41 */
42typedef error_info< struct type_info_info_tag, typeindex::type_index > type_info_info;
43
44/*!
45 * Parse position exception information
46 */
47typedef error_info< struct position_info_tag, unsigned int > position_info;
48
49/*!
50 * Current scope exception information
51 */
52typedef error_info< struct current_scope_info_tag, attributes::named_scope_list > current_scope_info;
53
54/*!
55 * The function returns an error information object that contains current stack of scopes.
56 * This information can then be attached to an exception and extracted at the catch site.
57 * The extracted scope list won't be affected by any scope changes that may happen during
58 * the exception propagation.
59 *
60 * \note See the \c named_scope attribute documentation on how to maintain scope list.
61 */
62inline current_scope_info current_scope()
63{
64 return current_scope_info(attributes::named_scope::get_scopes());
65}
