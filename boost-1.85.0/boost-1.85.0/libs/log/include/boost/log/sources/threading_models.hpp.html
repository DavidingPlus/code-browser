38//! Single thread locking model
39struct single_thread_model
40{
41 // We provide methods for the most advanced locking concept: UpgradeLockable
42 void lock_shared() const BOOST_NOEXCEPT {}
43 bool try_lock_shared() const BOOST_NOEXCEPT { return true; }
44 template< typename TimeT >
45 bool timed_lock_shared(TimeT const&) const BOOST_NOEXCEPT { return true; }
46 void unlock_shared() const BOOST_NOEXCEPT {}
47 void lock() const BOOST_NOEXCEPT {}
48 bool try_lock() const BOOST_NOEXCEPT { return true; }
49 template< typename TimeT >
50 bool timed_lock(TimeT const&) const BOOST_NOEXCEPT { return true; }
51 void unlock() const BOOST_NOEXCEPT {}
52 void lock_upgrade() const BOOST_NOEXCEPT {}
53 bool try_lock_upgrade() const BOOST_NOEXCEPT { return true; }
54 template< typename TimeT >
55 bool timed_lock_upgrade(TimeT const&) const BOOST_NOEXCEPT { return true; }
56 void unlock_upgrade() const BOOST_NOEXCEPT {}
57 void unlock_upgrade_and_lock() const BOOST_NOEXCEPT {}
58 void unlock_and_lock_upgrade() const BOOST_NOEXCEPT {}
59 void unlock_and_lock_shared() const BOOST_NOEXCEPT {}
60 void unlock_upgrade_and_lock_shared() const BOOST_NOEXCEPT {}
61
62 void swap(single_thread_model&) BOOST_NOEXCEPT {}
63};
64
65inline void swap(single_thread_model&, single_thread_model&) BOOST_NOEXCEPT
66{
67}

69#if !defined(BOOST_LOG_NO_THREADS)
70
71//! Multi-thread locking model with maximum locking capabilities
72template< typename MutexT >
73struct multi_thread_model
74{
75 multi_thread_model() BOOST_NOEXCEPT_IF(boost::has_nothrow_constructor< MutexT >::value) {}
76 multi_thread_model(multi_thread_model const&) BOOST_NOEXCEPT_IF(boost::has_nothrow_constructor< MutexT >::value) {}
77 multi_thread_model& operator= (multi_thread_model const&) BOOST_NOEXCEPT { return *this; }
78#if !defined(BOOST_NO_CXX11_RVALUE_REFERENCES)
79 multi_thread_model(multi_thread_model&&) BOOST_NOEXCEPT_IF(boost::has_nothrow_constructor< MutexT >::value) {}
80 multi_thread_model& operator= (multi_thread_model&&) BOOST_NOEXCEPT { return *this; }
81#endif
82
83 void lock_shared() const { m_Mutex.lock_shared(); }
84 bool try_lock_shared() const { return m_Mutex.try_lock_shared(); }
85 template< typename TimeT >
86 bool timed_lock_shared(TimeT const& t) const { return m_Mutex.timed_lock_shared(t); }
87 void unlock_shared() const BOOST_NOEXCEPT { m_Mutex.unlock_shared(); }
88 void lock() const { m_Mutex.lock(); }
89 bool try_lock() const { return m_Mutex.try_lock(); }
90 template< typename TimeT >
91 bool timed_lock(TimeT const& t) const { return m_Mutex.timed_lock(t); }
92 void unlock() const BOOST_NOEXCEPT { m_Mutex.unlock(); }
93 void lock_upgrade() const { m_Mutex.lock_upgrade(); }
94 bool try_lock_upgrade() const { return m_Mutex.try_lock_upgrade(); }
95 template< typename TimeT >
96 bool timed_lock_upgrade(TimeT const& t) const { return m_Mutex.timed_lock_upgrade(t); }
97 void unlock_upgrade() const BOOST_NOEXCEPT { m_Mutex.unlock_upgrade(); }
98 void unlock_upgrade_and_lock() const { m_Mutex.unlock_upgrade_and_lock(); }
99 void unlock_and_lock_upgrade() const { m_Mutex.unlock_and_lock_upgrade(); }
100 void unlock_and_lock_shared() const { m_Mutex.unlock_and_lock_shared(); }
101 void unlock_upgrade_and_lock_shared() const { m_Mutex.unlock_upgrade_and_lock_shared(); }
102
103 void swap(multi_thread_model&) BOOST_NOEXCEPT {}
104
105private:
106 //! Synchronization primitive
107 mutable MutexT m_Mutex;
108};
109
110template< typename MutexT >
111inline void swap(multi_thread_model< MutexT >&, multi_thread_model< MutexT >&) BOOST_NOEXCEPT
112{
113}
