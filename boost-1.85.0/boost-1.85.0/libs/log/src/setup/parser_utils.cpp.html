59//! Skips spaces in the beginning of the input
60const char* char_constants< char >::trim_spaces_left(const char_type* begin, const char_type* end)
61{
62 using namespace std;
63 while (begin != end && isspace(*begin))
64 ++begin;
65 return begin;
66}
67
68//! Skips spaces in the end of the input
69const char* char_constants< char >::trim_spaces_right(const char_type* begin, const char_type* end)
70{
71 using namespace std;
72 while (begin != end && isspace(*(end - 1)))
73 --end;
74 return end;
75}

77//! Scans for the attribute name placeholder in the input
78const char* char_constants< char >::scan_attr_placeholder(const char_type* begin, const char_type* end)

80 using namespace std;
81 while (begin != end)

83 char_type c = *begin;

92//! Parses an operand string (possibly quoted) from the input
93const char* char_constants< char >::parse_operand(const char_type* begin, const char_type* end, string_type& operand)
94{
95 using namespace std; // to make sure we can use C functions unqualified
96
97 const char_type* p = begin;
98 if (p == end)
99 BOOST_LOG_THROW_DESCR(parse_error, "Operand value is empty");
100
101 char_type c = *p;
102 if (c == char_quote)
103 {
104 // The value is specified as a quoted string
105 const char_type* start = ++p;
106 for (; p != end; ++p)
107 {
108 c = *p;
109 if (c == char_quote)
110 {
111 break;
112 }
113 else if (c == char_backslash)
114 {
115 ++p;
116 if (p == end)
117 BOOST_LOG_THROW_DESCR(parse_error, "Invalid escape sequence in the argument value");
118 }
119 }
120 if (p == end)
121 BOOST_LOG_THROW_DESCR(parse_error, "Unterminated quoted string in the argument value");
122
123 operand.assign(first: start, last: p);
124 translate_escape_sequences(str&: operand);
125
126 ++p; // skip the closing quote
127 }
128 else
129 {
130 // The value is specified as a single word
131 const char_type* start = p;
132 for (++p; p != end; ++p)
133 {
134 c = *p;
135 if (!isalnum(c) && c != '_' && c != '-' && c != '+' && c != '.')
136 break;
137 }
138
139 operand.assign(first: start, last: p);
140 }
141
142 return p;
143}

145//! Converts escape sequences to the corresponding characters
146void char_constants< char >::translate_escape_sequences(string_type& str)
147{
148 using namespace std; // to make sure we can use C functions unqualified
149
150 string_type::iterator it = str.begin();
151 while (it != str.end())
152 {
153 it = std::find(first: it, last: str.end(), val: '\\');
154 if (std::distance(first: it, last: str.end()) >= 2)
155 {
156 it = str.erase(position: it);
157 switch (*it)
158 {
159 case 'n':
160 *it = '\n'; break;
161 case 'r':
162 *it = '\r'; break;
163 case 'a':
164 *it = '\a'; break;
165 case '\\':
166 ++it; break;
167 case 't':
168 *it = '\t'; break;
169 case 'b':
170 *it = '\b'; break;
171 case 'x':
172 {
173 string_type::iterator b = it;
174 if (std::distance(first: ++b, last: str.end()) >= 2)
175 {
176 char_type c1 = *b++, c2 = *b++;
177 if (isxdigit(c1) && isxdigit(c2))
178 {
179 *it++ = char_type((to_number(c: c1) << 4) | to_number(c: c2));
180 it = str.erase(first: it, last: b);
181 }
182 }
183 break;
184 }
185 default:
186 {
187 if (*it >= '0' && *it <= '7')
188 {
189 string_type::iterator b = it;
190 int c = (*b++) - '0';
191 if (*b >= '0' && *b <= '7')
192 c = c * 8 + (*b++) - '0';
193 if (*b >= '0' && *b <= '7')
194 c = c * 8 + (*b++) - '0';
195
196 *it++ = char_type(c);
197 it = str.erase(first: it, last: b);
198 }
199 break;
200 }
201 }
202 }
203 }
204}

318//! Converts escape sequences to the corresponding characters
319void char_constants< wchar_t >::translate_escape_sequences(string_type& str)
320{
321 using namespace std; // to make sure we can use C functions unqualified
322
323 string_type::iterator it = str.begin();
324 while (it != str.end())
325 {
326 it = std::find(first: it, last: str.end(), val: L'\\');
327 if (std::distance(first: it, last: str.end()) >= 2)
328 {
329 it = str.erase(position: it);
330 switch (*it)
331 {
332 case L'n':
333 *it = L'\n'; break;
334 case L'r':
335 *it = L'\r'; break;
336 case L'a':
337 *it = L'\a'; break;
338 case L'\\':
339 ++it; break;
340 case L't':
341 *it = L'\t'; break;
342 case L'b':
343 *it = L'\b'; break;
344 case L'x':
345 {
346 string_type::iterator b = it;
347 if (std::distance(first: ++b, last: str.end()) >= 2)
348 {
349 char_type c1 = *b++, c2 = *b++;
350 if (iswxdigit(c: c1) && iswxdigit(c: c2))
351 {
352 *it++ = char_type((to_number(c: c1) << 4) | to_number(c: c2));
353 it = str.erase(first: it, last: b);
354 }
355 }
356 break;
357 }
358 case L'u':
359 {
360 string_type::iterator b = it;
361 if (std::distance(first: ++b, last: str.end()) >= 4)
362 {
363 char_type c1 = *b++, c2 = *b++, c3 = *b++, c4 = *b++;
364 if (iswxdigit(c: c1) && iswxdigit(c: c2) && iswxdigit(c: c3) && iswxdigit(c: c4))
365 {
366 *it++ = char_type(
367 (to_number(c: c1) << 12) |
368 (to_number(c: c2) << 8) |
369 (to_number(c: c3) << 4) |
370 to_number(c: c4));
371 it = str.erase(first: it, last: b);
372 }
373 }
374 break;
375 }
376 case L'U':
377 {
378 string_type::iterator b = it;
379 if (std::distance(first: ++b, last: str.end()) >= 8)
380 {
381 char_type c1 = *b++, c2 = *b++, c3 = *b++, c4 = *b++;
382 char_type c5 = *b++, c6 = *b++, c7 = *b++, c8 = *b++;
383 if (iswxdigit(c: c1) && iswxdigit(c: c2) && iswxdigit(c: c3) && iswxdigit(c: c4) &&
384 iswxdigit(c: c5) && iswxdigit(c: c6) && iswxdigit(c: c7) && iswxdigit(c: c8))
385 {
386 *it++ = char_type(
387 (to_number(c: c1) << 28) |
388 (to_number(c: c2) << 24) |
389 (to_number(c: c3) << 20) |
390 (to_number(c: c4) << 16) |
391 (to_number(c: c5) << 12) |
392 (to_number(c: c6) << 8) |
393 (to_number(c: c7) << 4) |
394 to_number(c: c8));
395 it = str.erase(first: it, last: b);
396 }
397 }
398 break;
399 }
400 default:
401 {
402 if (*it >= L'0' && *it <= L'7')
403 {
404 string_type::iterator b = it;
405 int c = (*b++) - L'0';
406 if (*b >= L'0' && *b <= L'7')
407 c = c * 8 + (*b++) - L'0';
408 if (*b >= L'0' && *b <= L'7')
409 c = c * 8 + (*b++) - L'0';
410
411 *it++ = char_type(c);
412 it = str.erase(first: it, last: b);
413 }
414 break;
415 }
416 }
417 }
418 }
