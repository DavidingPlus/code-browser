55//! A special filtering predicate that adopts the string operand to the attribute value character type
56template< typename RelationT >
57class string_predicate :
58 public RelationT
59{
60private:
61 template< typename StringT >
62 struct initializer
63 {
64 typedef void result_type;
65
66 explicit initializer(StringT const& val) : m_initializer(val)
67 {
68 }
69
70 template< typename T >
71 result_type operator() (T& val) const
72 {
73 try
74 {
75 log::aux::code_convert(m_initializer, val);
76 }
77 catch (...)
78 {
79 val.clear();
80 }
81 }
82
83 private:
84 StringT const& m_initializer;
85 };
86
87public:
88 typedef RelationT relation_type;
89 typedef typename relation_type::result_type result_type;
90
91 template< typename StringT >
92 string_predicate(relation_type const& rel, StringT const& operand) : relation_type(rel)
93 {
94 fusion::for_each(m_operands, initializer< StringT >(operand));
95 }
96
97 template< typename T >
98 result_type operator() (T const& val) const
99 {
100 typedef std::basic_string< typename T::value_type > operand_type;
101 return relation_type::operator() (val, fusion::at_key< operand_type >(m_operands));
102 }
103
104private:
105 fusion::set< std::string, std::wstring > m_operands;
106};

110//! A special filtering predicate that adopts the string operand to the attribute value character type
111template< typename RelationT >
112class string_predicate :
113 public RelationT
114{
115private:
116#if defined(BOOST_LOG_USE_CHAR)
117 typedef std::basic_string< char > string_type;
118#elif defined(BOOST_LOG_USE_WCHAR_T)
119 typedef std::basic_string< wchar_t > string_type;
120#else
121#error Boost.Log: Inconsistent character configuration
122#endif
123
124public:
125 typedef RelationT relation_type;
126 typedef typename relation_type::result_type result_type;
127
128public:
129 string_predicate(relation_type const& rel, string_type const& operand) : relation_type(rel), m_operand(operand)
130 {
131 }
132
133 template< typename T >
134 result_type operator() (T const& val) const
135 {
136 return relation_type::operator() (val, m_operand);
137 }
138
139private:
140 const string_type m_operand;
141};

145//! A special filtering predicate that adopts the severity level or string operand to the attribute value character type
146template< typename RelationT >
147class severity_or_string_predicate :
148 public string_predicate< RelationT >
149{
150private:
151 typedef string_predicate< RelationT > base_type;
152
153public:
154 typedef typename base_type::relation_type relation_type;
155 typedef typename base_type::result_type result_type;
156
157public:
158 template< typename StringT >
159 severity_or_string_predicate(relation_type const& rel, StringT const& string_operand, boost::log::trivial::severity_level severity_operand) :
160 base_type(rel, string_operand),
161 m_severity_operand(severity_operand)
162 {
163 }
164
165 using base_type::operator();
166
167 result_type operator() (boost::log::trivial::severity_level val) const
168 {
169 return relation_type::operator() (val, m_severity_operand);
170 }
171
172private:
173 const boost::log::trivial::severity_level m_severity_operand;
174};

176//! A filtering predicate for numeric relations
177template< typename NumericT, typename RelationT >
178class numeric_predicate :
179 public string_predicate< RelationT >
180{
181 typedef string_predicate< RelationT > base_type;
182
183public:
184 typedef NumericT numeric_type;
185 typedef typename base_type::relation_type relation_type;
186 typedef typename base_type::result_type result_type;
187
188public:
189 template< typename StringT >
190 numeric_predicate(relation_type const& rel, StringT const& string_operand, numeric_type numeric_operand) :
191 base_type(rel, string_operand),
192 m_numeric_operand(numeric_operand)
193 {
194 }
195
196 template< typename CharT, typename TraitsT, typename AllocatorT >
197 result_type operator() (std::basic_string< CharT, TraitsT, AllocatorT > const& val) const
198 {
199 return base_type::operator() (val);
200 }
201
202 template< typename CharT, typename TraitsT >
203 result_type operator() (basic_string_literal< CharT, TraitsT > const& val) const
204 {
205 return base_type::operator() (val);
206 }
207
208 template< typename T >
209 result_type operator() (T const& val) const
210 {
211 return relation_type::operator() (val, m_numeric_operand);
212 }
213
214private:
215 const numeric_type m_numeric_operand;
216};

266//! The callback for custom relation filter
267template< typename CharT >
268filter default_filter_factory< CharT >::on_custom_relation(attribute_name const& name, string_type const& rel, string_type const& arg)
269{
270 typedef log::aux::char_constants< char_type > constants;
271
272 if (rel == constants::begins_with_keyword())
273 {
274 typedef string_predicate< begins_with_fun > predicate;
275 return predicate_wrapper< log::string_types, predicate >(name, predicate(begins_with_fun(), arg));
276 }
277 else if (rel == constants::ends_with_keyword())
278 {
279 typedef string_predicate< ends_with_fun > predicate;
280 return predicate_wrapper< log::string_types, predicate >(name, predicate(ends_with_fun(), arg));
281 }
282 else if (rel == constants::contains_keyword())
283 {
284 typedef string_predicate< contains_fun > predicate;
285 return predicate_wrapper< log::string_types, predicate >(name, predicate(contains_fun(), arg));
286 }
287 else if (rel != constants::matches_keyword())
288 {
289 BOOST_LOG_THROW_DESCR(parse_error, "The custom attribute relation \"" + log::aux::to_narrow(rel) + "\" is not supported");
290 }
291
292 return parse_matches_relation(name, arg);
293}

296//! The function parses the argument value for a binary relation and constructs the corresponding filter
297template< typename CharT >
298template< typename RelationT >
299filter default_filter_factory< CharT >::parse_argument(attribute_name const& name, string_type const& arg)
300{
301 const char_type* begin = arg.c_str();
302 const char_type* const end = begin + arg.size();
303
304 double real_val = 0.0;
305 const qi::real_parser< double, qi::strict_real_policies< double > > real_;
306 if (qi::parse(begin, end, real_, real_val) && begin == end)
307 {
308 typedef numeric_predicate< double, RelationT > predicate;
309 typedef floating_point_and_string_types value_types;
310 return predicate_wrapper< value_types, predicate >(name, predicate(RelationT(), arg, real_val));
311 }
312 else
313 {
314 begin = arg.c_str();
315 long int_val = 0;
316 if (qi::parse(begin, end, qi::long_, int_val) && begin == end)
317 {
318 typedef numeric_predicate< long, RelationT > predicate;
319 typedef default_attribute_value_types value_types;
320 return predicate_wrapper< value_types, predicate >(name, predicate(RelationT(), arg, int_val));
321 }
322 else
323 {
324 boost::log::trivial::severity_level lvl;
325 if (name == boost::log::aux::default_attribute_names::severity() && boost::log::trivial::from_string(arg.data(), arg.size(), lvl))
326 {
327 typedef severity_or_string_predicate< RelationT > predicate;
328 typedef mpl::vector< boost::log::trivial::severity_level, BOOST_PP_SEQ_ENUM(BOOST_LOG_STANDARD_STRING_TYPES()) > value_types;
329 return predicate_wrapper< value_types, predicate >(name, predicate(RelationT(), arg, lvl));
330 }
331 else
332 {
333 typedef string_predicate< RelationT > predicate;
334 return predicate_wrapper< log::string_types, predicate >(name, predicate(RelationT(), arg));
335 }
336 }
337 }
