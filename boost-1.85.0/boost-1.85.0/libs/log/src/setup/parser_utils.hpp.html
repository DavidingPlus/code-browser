149 static int to_number(char_type c)
150 {
151 using namespace std; // to make sure we can use C functions unqualified
152 int n = 0;
153 if (isdigit(c))
154 n = c - '0';
155 else if (c >= 'a' && c <= 'f')
156 n = c - 'a' + 10;
157 else if (c >= 'A' && c <= 'F')
158 n = c - 'A' + 10;
159 return n;
160 }
