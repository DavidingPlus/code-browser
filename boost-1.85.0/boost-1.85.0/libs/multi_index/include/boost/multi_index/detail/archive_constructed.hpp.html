29/* constructs a stack-based object from a serialization archive */
30
31template<typename T>
32struct archive_constructed:private noncopyable
33{
34 template<class Archive>
35 archive_constructed(Archive& ar,const unsigned int version)
36 {
37 core::load_construct_data_adl(ar,&get(),version);
38 BOOST_TRY{
39 ar>>get();
40 }
41 BOOST_CATCH(...){
42 (&get())->~T();
43 BOOST_RETHROW;
44 }
45 BOOST_CATCH_END
46 }
47
48 template<class Archive>
49 archive_constructed(const char* name,Archive& ar,const unsigned int version)
50 {
51 core::load_construct_data_adl(ar,&get(),version);
52 BOOST_TRY{
53 ar>>core::make_nvp(name,get());
54 }
55 BOOST_CATCH(...){
56 (&get())->~T();
57 BOOST_RETHROW;
58 }
59 BOOST_CATCH_END
60 }
61
62 ~archive_constructed()
63 {
64 (&get())->~T();
65 }
66
67#include <boost/multi_index/detail/ignore_wstrict_aliasing.hpp>
68
69 T& get(){return *reinterpret_cast<T*>(&space);}
70
71#include <boost/multi_index/detail/restore_wstrict_aliasing.hpp>
72
73private:
74 typename aligned_storage<sizeof(T),alignment_of<T>::value>::type space;
75};
