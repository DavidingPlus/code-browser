82//
83// We disable this if the compiler is really nvcc with C++03 as it
84// doesn't actually support __int128 as of CUDA_VERSION=7500
85// even though it defines __SIZEOF_INT128__.
86// See https://svn.boost.org/trac/boost/ticket/10418
87// https://svn.boost.org/trac/boost/ticket/11852
88// Only re-enable this for nvcc if you're absolutely sure
89// of the circumstances under which it's supported.
90// Similarly __SIZEOF_INT128__ is defined when targetting msvc
91// compatibility even though the required support functions are absent.
92//

95# define BOOST_NVCC_CXX11
96# else
97# define BOOST_NVCC_CXX03
98# endif

284// clang < 3.5 has a defect with dependent type, like following.
285//
286// template <class T>
287// constexpr typename enable_if<pred<T> >::type foo(T &)
288// { } // error: no return statement in constexpr function
289//
290// This issue also affects C++11 mode, but C++11 constexpr requires return stmt.
291// Therefore we don't care such case.
292//
293// Note that we can't check Clang version directly as the numbering system changes depending who's
294// creating the Clang release (see https://github.com/boostorg/config/pull/39#issuecomment-59927873)
295// so instead verify that we have a feature that was introduced at the same time as working C++14
296// constexpr (generic lambda's in this case):
297//
298#if !__has_feature(__cxx_generic_lambdas__) || !__has_feature(__cxx_relaxed_constexpr__)
299# define BOOST_NO_CXX14_CONSTEXPR
300#endif
301
