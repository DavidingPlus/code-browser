23#if !defined( BOOST_SP_NO_SYNC_INTRINSICS ) && !defined( BOOST_SP_NO_SYNC )
24
25#if defined( __GCC_HAVE_SYNC_COMPARE_AND_SWAP_4 ) && !defined( __c2__ )
26
27# define BOOST_SP_HAS_SYNC_INTRINSICS
28
29#elif defined( __IBMCPP__ ) && ( __IBMCPP__ >= 1210 ) && !defined( __COMPILER_VER__ )
30
31# define BOOST_SP_HAS_SYNC_INTRINSICS
32
33#elif defined( __GNUC__ ) && ( __GNUC__ * 100 + __GNUC_MINOR__ >= 401 ) && !defined( __c2__ )
34
35#define BOOST_SP_HAS_SYNC_INTRINSICS

37#if defined( __arm__ ) || defined( __armel__ )
38#undef BOOST_SP_HAS_SYNC_INTRINSICS
39#endif
40
41#if defined( __hppa ) || defined( __hppa__ )
42#undef BOOST_SP_HAS_SYNC_INTRINSICS
43#endif
44
45#if defined( __m68k__ )
46#undef BOOST_SP_HAS_SYNC_INTRINSICS
47#endif
48
49#if defined( __sh__ )
50#undef BOOST_SP_HAS_SYNC_INTRINSICS
51#endif
52
53#if defined( __sparc__ )
54#undef BOOST_SP_HAS_SYNC_INTRINSICS
55#endif
56
57#if defined( __INTEL_COMPILER ) && !defined( __ia64__ ) && ( __INTEL_COMPILER < 1110 )
58#undef BOOST_SP_HAS_SYNC_INTRINSICS
59#endif
60

62#undef BOOST_SP_HAS_SYNC_INTRINSICS
