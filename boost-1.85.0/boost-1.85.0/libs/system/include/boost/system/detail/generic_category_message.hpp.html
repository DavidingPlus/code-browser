28// glibc has two incompatible strerror_r definitions

30inline char const * strerror_r_helper( char const * r, char const * ) noexcept
31{
32 return r;
33}
34
35inline char const * strerror_r_helper( int r, char const * buffer ) noexcept
36{
37 return r == 0? buffer: "Unknown error";
38}
39
40inline char const * generic_error_category_message( int ev, char * buffer, std::size_t len ) noexcept
41{
42 return strerror_r_helper( r: strerror_r( errnum: ev, buf: buffer, buflen: len ), buffer );
43}

45inline std::string generic_error_category_message( int ev )
46{
47 char buffer[ 128 ];
48 return generic_error_category_message( ev, buffer, len: sizeof( buffer ) );

65inline std::string generic_error_category_message( int ev )
66{
67 char const * m = std::strerror( ev );
68 return m? m: "Unknown error";
69}
70
71inline char const * generic_error_category_message( int ev, char * buffer, std::size_t len ) noexcept
72{
73 if( len == 0 )
74 {
75 return buffer;
76 }
77
78 if( len == 1 )
79 {
80 buffer[0] = 0;
81 return buffer;
82 }
83
84 char const * m = std::strerror( ev );
85
86 if( m == 0 ) return "Unknown error";
87
88 std::strncpy( buffer, m, len - 1 );
89 buffer[ len-1 ] = 0;
90
91 return buffer;
92}
