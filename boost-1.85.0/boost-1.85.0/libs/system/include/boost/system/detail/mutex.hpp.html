100template<class Mtx> class lock_guard
101{
102private:
103
104 Mtx& mtx_;
105
106private:
107
108 lock_guard( lock_guard const& );
109 lock_guard& operator=( lock_guard const& );
110
111public:
112
113 explicit lock_guard( Mtx& mtx ): mtx_( mtx )
114 {
115 mtx_.lock();
116 }
117
118 ~lock_guard()
119 {
120 mtx_.unlock();
121 }
