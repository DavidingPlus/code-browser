25namespace detail
26{
27
28// Misuse of the error_code object is turned into a noisy failure by
29// poisoning the reference. This particular implementation doesn't
30// produce warnings or errors from popular compilers, is very efficient
31// (as determined by inspecting generated code), and does not suffer
32// from order of initialization problems. In practice, it also seems
33// cause user function error handling implementation errors to be detected
34// very early in the development cycle.
35
36inline system::error_code* throws()
37{
38 // See github.com/boostorg/system/pull/12 by visigoth for why the return
39 // is poisoned with nonzero rather than (0). A test, test_throws_usage(),
40 // has been added to error_code_test.cpp, and as visigoth mentioned it
41 // fails on clang for release builds with a return of 0 but works fine
42 // with (1).
43 // Since the undefined behavior sanitizer (-fsanitize=undefined) does not
44 // allow a reference to be formed to the unaligned address of (1), we use
45 // (8) instead.
46
47 return reinterpret_cast<system::error_code*>(8);
48}
49
50} // namespace detail
51
52inline system::error_code& throws()
53{
54 return *detail::throws();
55}
