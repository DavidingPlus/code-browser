19namespace boost{
20namespace unordered{
21namespace detail{
22
23/* constructs a stack-based object from a serialization archive */
24
25template<typename T>
26struct archive_constructed:private noncopyable
27{
28 template<class Archive>
29 archive_constructed(const char* name,Archive& ar,unsigned int version)
30 {
31 core::load_construct_data_adl(ar,std::addressof(get()),version);
32 BOOST_TRY{
33 ar>>core::make_nvp(name,get());
34 }
35 BOOST_CATCH(...){
36 get().~T();
37 BOOST_RETHROW;
38 }
39 BOOST_CATCH_END
40 }
41
42 ~archive_constructed()
43 {
44 get().~T();
45 }
46
47#if defined(BOOST_GCC)&&(BOOST_GCC>=4*10000+6*100)
48#define BOOST_UNORDERED_IGNORE_WSTRICT_ALIASING
49#endif
50
51#if defined(BOOST_UNORDERED_IGNORE_WSTRICT_ALIASING)
52#pragma GCC diagnostic push
53#pragma GCC diagnostic ignored "-Wstrict-aliasing"
54#endif
55
56 T& get(){return *space.address();}
57
58#if defined(BOOST_UNORDERED_IGNORE_WSTRICT_ALIASING)
59#pragma GCC diagnostic pop
60#undef BOOST_UNORDERED_IGNORE_WSTRICT_ALIASING
61#endif
62
63private:
64 opt_storage<T> space;
65};

67} /* namespace detail */
