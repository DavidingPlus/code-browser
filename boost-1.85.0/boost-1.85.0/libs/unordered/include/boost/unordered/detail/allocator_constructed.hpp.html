9#ifndef BOOST_UNORDERED_DETAIL_ALLOCATOR_CONSTRUCTED_HPP
10#define BOOST_UNORDERED_DETAIL_ALLOCATOR_CONSTRUCTED_HPP
11
12#include <boost/core/allocator_traits.hpp>
13#include <boost/unordered/detail/opt_storage.hpp>
14
15namespace boost {
16 namespace unordered {
17 namespace detail {
18
19 struct allocator_policy
20 {
21 template <class Allocator, class T, class... Args>
22 static void construct(Allocator& a, T* p, Args&&... args)
23 {
24 boost::allocator_construct(a, p, std::forward<Args>(args)...);
25 }
26
27 template <class Allocator, class T>
28 static void destroy(Allocator& a, T* p)
29 {
30 boost::allocator_destroy(a, p);
31 }
32 };
33
34 /* constructs a stack-based object with the given policy and allocator */
35 template <class Allocator, class T, class Policy = allocator_policy>
36 class allocator_constructed
37 {
38 opt_storage<T> storage;
39 Allocator alloc;
40
41 public:
42 template <class... Args>
43 allocator_constructed(Allocator const& alloc_, Args&&... args)
44 : alloc(alloc_)
45 {
46 Policy::construct(
47 alloc, storage.address(), std::forward<Args>(args)...);
48 }
49
50 ~allocator_constructed() { Policy::destroy(alloc, storage.address()); }
51
52 T& value() { return *storage.address(); }
53 };
54
55 } /* namespace detail */
56 } /* namespace unordered */
57} /* namespace boost */
58
59#endif
