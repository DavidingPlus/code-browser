108 template <class T, class = void> struct is_nothrow_swappable_helper
109 {
110 constexpr static bool const value = false;
111 };

113 template <class T>
114 struct is_nothrow_swappable_helper<T,
115 void_t<decltype(swap(std::declval<T&>(), std::declval<T&>()))> >

131 // Type checkers used for the transparent member functions added by C++20

134 template <class, class = void>
135 struct is_transparent : public std::false_type
136 {
137 };
138
139 template <class T>
140 struct is_transparent<T,
141 boost::unordered::detail::void_t<typename T::is_transparent> >
142 : public std::true_type
143 {
144 };
145
146 template <class, class Hash, class KeyEqual> struct are_transparent
147 {
148 static bool const value =
149 is_transparent<Hash>::value && is_transparent<KeyEqual>::value;
150 };
151
152 template <class Key, class UnorderedMap> struct transparent_non_iterable
153 {
154 typedef typename UnorderedMap::hasher hash;
155 typedef typename UnorderedMap::key_equal key_equal;
156 typedef typename UnorderedMap::iterator iterator;
157 typedef typename UnorderedMap::const_iterator const_iterator;
158
159 static bool const value =
160 are_transparent<Key, hash, key_equal>::value &&
161 !std::is_convertible<Key, iterator>::value &&
162 !std::is_convertible<Key, const_iterator>::value;
163 };

165 template <class T>
166 using remove_cvref_t =
167 typename std::remove_cv<typename std::remove_reference<T>::type>::type;
168
169 template <class T, class U>
170 using is_similar = std::is_same<remove_cvref_t<T>, remove_cvref_t<U> >;
171
172 template <class, class...> struct is_similar_to_any : std::false_type
173 {
174 };
175 template <class T, class U, class... Us>
176 struct is_similar_to_any<T, U, Us...>
177 : std::conditional<is_similar<T, U>::value, is_similar<T, U>,
178 is_similar_to_any<T, Us...> >::type
179 {
180 };

192 constexpr static bool const value = false;
193 };
194
195 template <class A>
196 struct is_allocator<A,
197 boost::unordered::detail::void_t<typename A::value_type,
198 decltype(std::declval<A&>().allocate(std::size_t{}))> >
199 {
200 constexpr static bool const value = true;
201 };
202
203 template <class A>
204 constexpr bool const is_allocator_v = is_allocator<A>::value;

206 template <class H>
207 constexpr bool const is_hash_v =
208 !std::is_integral<H>::value && !is_allocator_v<H>;
209
210 template <class P> constexpr bool const is_pred_v = !is_allocator_v<P>;
