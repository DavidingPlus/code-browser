6#ifndef BOOST_UNORDERED_DETAIL_OPT_STORAGE_HPP

9#include <boost/config.hpp>

11#include <memory>

13namespace boost {
14 namespace unordered {
15 namespace detail {
16 template <class T> union opt_storage
17 {
18 BOOST_ATTRIBUTE_NO_UNIQUE_ADDRESS T t_;
19
20 opt_storage() {}
21 ~opt_storage() {}
22
23 T* address() noexcept { return std::addressof(t_); }
24 T const* address() const noexcept { return std::addressof(t_); }
25 };
26 } // namespace detail
27 } // namespace unordered
28} // namespace boost
