24/* Tracked address serialization to support iterator serialization as described
25 * in serialize_container.hpp. The underlying technique is to reinterpret_cast
26 * T pointers to serialization_tracker<T> pointers, which, when dereferenced
27 * and serialized, do not emit any serialization payload to the
28 * archive, but activate object tracking on the relevant addresses for later
29 * use with serialize_tracked_address().
30 */
31
32template<typename T>
33struct serialization_tracker
34{
35 /* An attempt to construct a serialization_tracker means a stray address
36 * in the archive, that is, one without a previously tracked address.
37 */
38 serialization_tracker(){throw_exception(e: bad_archive_exception());}
39
40 template<typename Archive>
41 void serialize(Archive&,unsigned int){} /* no data emitted */
42};
43
44template<typename Archive,typename Ptr>
45void track_address(Archive& ar,Ptr p)
46{
47 typedef typename boost::pointer_traits<Ptr> ptr_traits;
48 typedef typename std::remove_const<
49 typename ptr_traits::element_type>::type element_type;
50
51 if(p){
52 ar&core::make_nvp(
53 "address",
54 *reinterpret_cast<serialization_tracker<element_type>*>(
55 const_cast<element_type*>(
56 boost::to_address(p))));
57 }
58}
59
60template<typename Archive,typename Ptr>
61void serialize_tracked_address(Archive& ar,Ptr& p,std::true_type /* save */)
62{
63 typedef typename boost::pointer_traits<Ptr> ptr_traits;
64 typedef typename std::remove_const<
65 typename ptr_traits::element_type>::type element_type;
66 typedef serialization_tracker<element_type> tracker;
67
68 tracker* pt=
69 const_cast<tracker*>(
70 reinterpret_cast<const tracker*>(
71 const_cast<const element_type*>(
72 boost::to_address(p))));
73 ar<<core::make_nvp("pointer",pt);
74}
75
76template<typename Archive,typename Ptr>
77void serialize_tracked_address(Archive& ar,Ptr& p,std::false_type /* load */)
78{
79 typedef typename boost::pointer_traits<Ptr> ptr_traits;
80 typedef typename std::remove_const<
81 typename ptr_traits::element_type>::type element_type;
82 typedef serialization_tracker<element_type> tracker;
83
84 tracker* pt;
85 ar>>core::make_nvp("pointer",pt);
86 element_type* pn=const_cast<element_type*>(
87 reinterpret_cast<const element_type*>(
88 const_cast<const tracker*>(pt)));
89 p=pn?ptr_traits::pointer_to(*pn):0;
90}
91
92template<typename Archive,typename Ptr>
93void serialize_tracked_address(Archive& ar,Ptr& p)
94{
95 serialize_tracked_address(
96 ar,p,
97 std::integral_constant<bool,Archive::is_saving::value>());
98}
