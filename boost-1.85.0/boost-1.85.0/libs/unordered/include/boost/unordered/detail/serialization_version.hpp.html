9#ifndef BOOST_UNORDERED_DETAIL_SERIALIZATION_VERSION_HPP
10#define BOOST_UNORDERED_DETAIL_SERIALIZATION_VERSION_HPP
11
12#include <boost/config.hpp>
13#include <boost/core/serialization.hpp>
14
15namespace boost{
16namespace unordered{
17namespace detail{
18
19/* boost::serialization::load_construct_adl(ar,t,version) requires user code
20 * to pass the serialization version for t, when this information is really
21 * stored in the archive. serialization_version<T> circumvents this design
22 * error by acting as a regular serializable type with the same serialization
23 * version as T; loading/saving serialization_version<T> does nothing with
24 * the archive data itself but captures the stored serialization version
25 * at load() time.
26 */
27
28template<typename T>
29struct serialization_version
30{
31 serialization_version():
32 value(boost::serialization::version<serialization_version>::value){}
33
34 serialization_version& operator=(unsigned int x){value=x;return *this;};
35
36 operator unsigned int()const{return value;}
37
38private:
39 friend class boost::serialization::access;
40
41 template<class Archive>
42 void serialize(Archive& ar,unsigned int version)
43 {
44 core::split_member(ar,*this,version);
45 }
46
47 template<class Archive>
48 void save(Archive&,unsigned int)const{}
49
50 template<class Archive>
51 void load(Archive&,unsigned int version)
52 {
53 this->value=version;
54 }
55
56 unsigned int value;
57};
58
59} /* namespace detail */
60} /* namespace unordered */
61
62namespace serialization{
63
64template<typename T>
65struct version<boost::unordered::detail::serialization_version<T> >
66{
67 BOOST_STATIC_CONSTANT(int,value=version<T>::value);
68};
69
70} /* namespace serialization */
71
72} /* namespace boost */
73
74#endif
