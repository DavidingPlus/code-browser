10/*
11
12The general structure of the fast closed addressing implementation is that we
13use straight-forward separate chaining (i.e. each bucket contains its own linked
14list) and then improve iteration time by adding an array of "bucket groups".
15
16A bucket group is a constant-width view into a subsection of the buckets array,
17containing a bitmask that indicates which one of the buckets in the subsection
18contains a list of nodes. This allows the code to test N buckets for occupancy
19in a single operation. Additional speed can be found by inter-linking occupied
20bucket groups with one another in a doubly-linked list. To this end, large
21swathes of the bucket groups array no longer need to be iterated and have their
22bitmasks examined for occupancy.
23
24A bucket group iterator contains a pointer to a bucket group along with a
25pointer into the buckets array. The iterator's bucket pointer is guaranteed to
26point to a bucket within the bucket group's view of the array. To advance the
27iterator, we need to determine if we need to skip to the next bucket group or
28simply move to the next occupied bucket as denoted by the bitmask.
29
30To accomplish this, we perform something roughly equivalent to this:
31```
32bucket_iterator itb = ...
33bucket_pointer p = itb.p
34bucket_group_pointer pbg = itb.pbg
35
36offset = p - pbg->buckets
37// because we wish to see if the _next_ bit in the mask is occupied, we'll
38// generate a testing mask from the current offset + 1
39//
40testing_mask = reset_first_bits(offset + 1)
41n = ctz(pbg->bitmask & testing_mask)
42
43if (n < N) {
44 p = pbg->buckets + n
45} else {
46 pbg = pbg->next
47 p = pbg->buckets + ctz(pbg->bitmask)
48}
49```
50
51`reset_first_bits` yields an unsigned integral with the first n bits set to 0
52and then by counting the number of trailing zeroes when AND'd against the bucket
53group's bitmask, we can derive the offset into the buckets array. When the
54calculated offset is equal to N, we know we've reached the end of a bucket group
55and we can advance to the next one.
56
57This is a rough explanation for how iterator incrementation should work for a
58fixed width size of N as 3 for the bucket groups
59```
60N = 3
61p = buckets
62pbg->bitmask = 0b101
63pbg->buckets = buckets
64
65offset = p - pbg->buckets // => 0
66testing_mask = reset_first_bits(offset + 1) // reset_first_bits(1) => 0b110
67
68x = bitmask & testing_mask // => 0b101 & 0b110 => 0b100
69ctz(x) // ctz(0b100) => 2
70// 2 < 3
71=> p = pbg->buckets + 2
72
73// increment again...
74offset = p - pbg->buckets // => 2
75testing_mask = reset_first_bits(offset + 1) // reset_first_bits(3) => 0b000
76
77bitmask & testing_mask // 0b101 & 0b000 => 0b000
78ctz(0b000) => 3
79// 3 < 3 is false now
80pbg = pbg->next
81initial_offset = ctz(pbg->bitmask)
82p = pbg->buckets + initial_offset
83```
84
85For `size_` number of buckets, there are `1 + (size_ / N)` bucket groups where
86`N` is the width of a bucket group, determined at compile-time.
87
88We allocate space for `size_ + 1` buckets, using the last one as a dummy bucket
89which is kept permanently empty so it can act as a sentinel value in the
90implementation of `iterator end();`. We set the last bucket group to act as a
91sentinel.
92
93```
94num_groups = size_ / N + 1
95groups = allocate(num_groups)
96pbg = groups + (num_groups - 1)
97
98// not guaranteed to point to exactly N buckets
99pbg->buckets = buckets + N * (size_ / N)
100
101// this marks the true end of the bucket array
102buckets pbg->bitmask = set_bit(size_ % N)
103
104// links in on itself
105pbg->next = pbg->prev = pbg
106```
107
108To this end, we can devise a safe iteration scheme while also creating a useful
109sentinel to use as the end iterator.
110
111Otherwise, usage of the data structure is relatively straight-forward compared
112to normal separate chaining implementations.
113
114*/

189 inline std::size_t set_bit(std::size_t n) { return std::size_t(1) << n; }
190
191 inline std::size_t reset_bit(std::size_t n)
192 {
193 return ~(std::size_t(1) << n);
194 }
195
196 inline std::size_t reset_first_bits(std::size_t n) // n>0
197 {
198 return ~(~(std::size_t(0)) >> (sizeof(std::size_t) * 8 - n));

267 void increment() noexcept
268 {
269 std::size_t const offset = static_cast<std::size_t>(p - pbg->buckets);
270
271 std::size_t n = std::size_t(boost::core::countr_zero(
272 pbg->bitmask & reset_first_bits(n: offset + 1)));
273
274 if (n < N) {
275 p = pbg->buckets + static_cast<difference_type>(n);
276 } else {
277 pbg = pbg->next;
278
279 std::ptrdiff_t x = boost::core::countr_zero(pbg->bitmask);
280 p = pbg->buckets + x;
281 }
282 }

452 class grouped_bucket_array
453 : boost::empty_value<typename boost::allocator_rebind<Allocator,
454 node<typename boost::allocator_value_type<Allocator>::type,

463 difference_type;

467 node<allocator_value_type, void_pointer> >::type node_allocator_type;
468
469 typedef node<allocator_value_type, void_pointer> node_type;
470 typedef typename boost::allocator_pointer<node_allocator_type>::type
471 node_pointer;
472 typedef SizePolicy size_policy;
473

475 typedef typename boost::allocator_rebind<Allocator, Bucket>::type
476 bucket_allocator_type;
477 typedef typename boost::allocator_pointer<bucket_allocator_type>::type
478 bucket_pointer;
479 typedef boost::pointer_traits<bucket_pointer> bucket_pointer_traits;
480
481 typedef bucket_group<Bucket> group;
482 typedef typename boost::allocator_rebind<Allocator, group>::type
483 group_allocator_type;
484 typedef typename boost::allocator_pointer<group_allocator_type>::type
485 group_pointer;
486 typedef typename boost::pointer_traits<group_pointer>
487 group_pointer_traits;
488

490 typedef Bucket value_type;

493 typedef Allocator allocator_type;
494 typedef grouped_bucket_iterator<Bucket> iterator;
495 typedef grouped_local_bucket_iterator<node_type> local_iterator;
496 typedef const_grouped_local_bucket_iterator<node_type>
497 const_local_iterator;

499 private:
500 std::size_t size_index_, size_;
501 bucket_pointer buckets;
502 group_pointer groups;
503
504 public:
505 static std::size_t bucket_count_for(std::size_t num_buckets)
506 {
507 if (num_buckets == 0) {
508 return 0;
509 }
510 return size_policy::size(size_policy::size_index(num_buckets));
511 }
512
513 grouped_bucket_array()
514 : empty_value<node_allocator_type>(
515 empty_init_t(), node_allocator_type()),
516 size_index_(0), size_(0), buckets(), groups()
517 {
518 }
519
520 grouped_bucket_array(size_type n, const Allocator& al)
521 : empty_value<node_allocator_type>(
522 empty_init_t(), node_allocator_type(al)),
523 size_index_(0), size_(0), buckets(), groups()
524 {
525 if (n == 0) {
526 return;
527 }
528
529 size_index_ = size_policy::size_index(n);
530 size_ = size_policy::size(size_index_);
531
532 bucket_allocator_type bucket_alloc = this->get_bucket_allocator();
533 group_allocator_type group_alloc = this->get_group_allocator();
534
535 size_type const num_buckets = buckets_len();
536 size_type const num_groups = groups_len();
537
538 buckets = boost::allocator_allocate(bucket_alloc, num_buckets);
539 BOOST_TRY
540 {
541 groups = boost::allocator_allocate(group_alloc, num_groups);
542
543 bucket_type* pb = boost::to_address(buckets);
544 for (size_type i = 0; i < num_buckets; ++i) {
545 new (pb + i) bucket_type();
546 }
547
548 group* pg = boost::to_address(groups);
549 for (size_type i = 0; i < num_groups; ++i) {
550 new (pg + i) group();
551 }
552 }
553 BOOST_CATCH(...)
554 {
555 boost::allocator_deallocate(bucket_alloc, buckets, num_buckets);
556 BOOST_RETHROW
557 }
558 BOOST_CATCH_END
559
560 size_type const N = group::N;
561 group_pointer pbg =
562 groups + static_cast<difference_type>(num_groups - 1);
563
564 pbg->buckets =
565 buckets + static_cast<difference_type>(N * (size_ / N));
566 pbg->bitmask = set_bit(size_ % N);
567 pbg->next = pbg->prev = pbg;
568 }

618 void deallocate() noexcept
619 {
620 if (buckets) {
621 size_type const num_buckets = buckets_len();
622 bucket_type* pb = boost::to_address(buckets);
623 (void)pb; // VS complains when dtor is trivial
624
625 for (size_type i = 0; i < num_buckets; ++i) {
626 (pb + i)->~bucket_type();
627 }
628
629 bucket_allocator_type bucket_alloc = this->get_bucket_allocator();
630 boost::allocator_deallocate(bucket_alloc, buckets, num_buckets);
631
632 buckets = bucket_pointer();
633 }
634
635 if (groups) {
636 size_type const num_groups = groups_len();
637 group* pg = boost::to_address(groups);
638 (void)pg; // VS complains when dtor is trivial
639
640 for (size_type i = 0; i < num_groups; ++i) {
641 (pg + i)->~group();
642 }
643
644 group_allocator_type group_alloc = this->get_group_allocator();
645 boost::allocator_deallocate(group_alloc, groups, num_groups);
646
647 groups = group_pointer();
648 }
649 }

764 void append_bucket_group(iterator itb) noexcept
765 {
766 std::size_t const N = group::N;
767
768 bool const is_empty_bucket = (!itb->next);
769 if (is_empty_bucket) {
770 bucket_pointer pb = itb.p;
771 group_pointer pbg = itb.pbg;
772
773 std::size_t n =
774 static_cast<std::size_t>(boost::to_address(pb) - &buckets[0]);
775
776 bool const is_empty_group = (!pbg->bitmask);
777 if (is_empty_group) {
778 size_type const num_groups = this->groups_len();
779 group_pointer last_group =
780 groups + static_cast<difference_type>(num_groups - 1);
781
782 pbg->buckets =
783 buckets + static_cast<difference_type>(N * (n / N));
784 pbg->next = last_group->next;
785 pbg->next->prev = pbg;
786 pbg->prev = last_group;
787 pbg->prev->next = pbg;
788 }
789
790 pbg->bitmask |= set_bit(n % N);
791 }
792 }

833 void unlink_empty_buckets() noexcept
834 {
835 std::size_t const N = group::N;
836
837 group_pointer pbg = groups,
838 last = groups + static_cast<difference_type>(
839 this->groups_len() - 1);
840
841 for (; pbg != last; ++pbg) {
842 if (!pbg->buckets) {
843 continue;
844 }
845
846 for (std::size_t n = 0; n < N; ++n) {
847 bucket_pointer bs = pbg->buckets;
848 bucket_type& b = bs[static_cast<std::ptrdiff_t>(n)];
849 if (!b.next)
850 pbg->bitmask &= reset_bit(n);
851 }
852 if (!pbg->bitmask && pbg->next)
853 unlink_group(pbg);
854 }
855
856 // do not check end bucket
857 for (std::size_t n = 0; n < size_ % N; ++n) {
858 if (!pbg->buckets[static_cast<std::ptrdiff_t>(n)].next)
859 pbg->bitmask &= reset_bit(n);
860 }
861 }

866 typename iterator::bucket_group_pointer pbg = itb.pbg;

868 reset_bit(n: static_cast<std::size_t>(p - pbg->buckets))))
869 unlink_group(pbg);
