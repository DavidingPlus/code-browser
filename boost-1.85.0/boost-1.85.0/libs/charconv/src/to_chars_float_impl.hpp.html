55template <typename Real>
56inline to_chars_result to_chars_nonfinite(char* first, char* last, Real value, int classification) noexcept
57{
58 std::ptrdiff_t offset {};
59 std::ptrdiff_t buffer_size = last - first;
60
61 if (classification == FP_NAN)
62 {
63 bool is_negative = false;
64 const bool is_signaling = issignaling(value);
65
66 if (std::signbit(value))
67 {
68 is_negative = true;
69 *first++ = '-';
70 }
71
72 if (is_signaling && buffer_size >= (9 + static_cast<int>(is_negative)))
73 {
74 std::memcpy(first, "nan(snan)", 9);
75 offset = 9 + static_cast<int>(is_negative);
76 }
77 else if (is_negative && buffer_size >= 9)
78 {
79 std::memcpy(first, "nan(ind)", 8);
80 offset = 8;
81 }
82 else if (!is_negative && !is_signaling && buffer_size >= 3)
83 {
84 std::memcpy(first, "nan", 3);
85 offset = 3;
86 }
87 else
88 {
89 // Avoid buffer overflow
90 return { first, std::errc::value_too_large };
91 }
92
93 }
94 else if (classification == FP_INFINITE)
95 {
96 auto sign_bit_value = std::signbit(value);
97 if (sign_bit_value && buffer_size >= 4)
98 {
99 std::memcpy(first, "-inf", 4);
100 offset = 4;
101 }
102 else if (!sign_bit_value && buffer_size >= 3)
103 {
104 std::memcpy(first, "inf", 3);
105 offset = 3;
106 }
107 else
108 {
109 // Avoid buffer overflow
110 return { first, std::errc::value_too_large };
111 }
112 }
113 else
114 {
115 BOOST_UNREACHABLE_RETURN(first);
116 }
117
118 return { first + offset, std::errc() };
119}

121#endif // BOOST_CHARCONV_LDBL_BITS == 128

225template <typename Real, typename Unsigned_Integer>
226inline std::uint64_t extract_exp(Real, Unsigned_Integer uint_value, int significand_bits) noexcept
227{
228 return static_cast<std::uint64_t>(uint_value >> significand_bits);
229}

233// In the 80-bit long double case we need to ignore the pad on top of the high bits
234// and the 64-bits of significand in the low bits
235// This is easiest to accomplish with a mask rather than all kinds of shifting
236template <>
237inline std::uint64_t extract_exp<long double, uint128>(long double, uint128 uint_value, int) noexcept
238{
239
240 constexpr auto exp_mask = uint128{UINT64_C(0x7FFF), UINT64_C(0)};
241 auto exponent = uint_value & exp_mask;
242 return exponent.high;
243}

247template <typename Real>
248to_chars_result to_chars_hex(char* first, char* last, Real value, int precision) noexcept
249{
250 // Sanity check our bounds
251 const std::ptrdiff_t buffer_size = last - first;
252 auto real_precision = get_real_precision<Real>(precision);
253
254 if (precision != -1)
255 {
256 real_precision = precision;
257 }
258
259 if (buffer_size < real_precision || first > last)
260 {
261 return {.ptr: last, .ec: std::errc::value_too_large};
262 }
263
264 // Extract the significand and the exponent
265 using type_layout = typename std::conditional<std::is_same<Real, float>::value, ieee754_binary32,
266 typename std::conditional<std::is_same<Real, double>::value, ieee754_binary64,
267 #ifdef BOOST_CHARCONV_HAS_FLOAT128
268 typename std::conditional<std::is_same<Real, __float128>::value || BOOST_CHARCONV_LDBL_BITS == 128, ieee754_binary128, ieee754_binary80>::type
269 #elif BOOST_CHARCONV_LDBL_BITS == 128
270 ieee754_binary128
271 #elif BOOST_CHARCONV_LDBL_BITS == 80
272 ieee754_binary80
273 #else
274 ieee754_binary64
275 #endif
276 >::type>::type;
277
278 using Unsigned_Integer = typename std::conditional<std::is_same<Real, float>::value, std::uint32_t,
279 typename std::conditional<std::is_same<Real, double>::value, std::uint64_t, uint128>::type>::type;
280
281
282 Unsigned_Integer uint_value {convert_value<Unsigned_Integer>(value)};
283
284 const Unsigned_Integer denorm_mask = (Unsigned_Integer(1) << (type_layout::significand_bits)) - 1;
285 const Unsigned_Integer significand = uint_value & denorm_mask;
286 std::uint64_t exponent = extract_exp(value, uint_value, type_layout::significand_bits);
287
288 BOOST_IF_CONSTEXPR (!(std::is_same<Real, float>::value || std::is_same<Real, double>::value
289 #if BOOST_CHARCONV_LDBL_BITS == 80
290 || std::is_same<Real, long double>::value
291 #endif
292 ))
293 {
294 exponent += 2;
295 }
296
297 // Align the significand to the hexit boundaries (i.e. divisible by 4)
298 constexpr auto hex_precision = std::is_same<Real, float>::value ? 6 :
299 std::is_same<Real, double>::value ? 13
300 #if BOOST_CHARCONV_LDBL_BITS == 80
301 : std::is_same<Real, long double>::value ? 15
302 #endif
303 : 28;
304
305 constexpr auto nibble_bits = CHAR_BIT / 2;
306 constexpr auto hex_bits = hex_precision * nibble_bits;
307 const Unsigned_Integer hex_mask = (static_cast<Unsigned_Integer>(1) << hex_bits) - 1;
308
309 Unsigned_Integer aligned_significand;
310 BOOST_IF_CONSTEXPR (std::is_same<Real, float>::value)
311 {
312 aligned_significand = significand << 1;
313 }
314 else
315 {
316 aligned_significand = significand;
317 }
318
319 // Adjust the exponent based on the bias as described in IEEE 754
320 std::int64_t unbiased_exponent;
321 if (exponent == 0 && significand != 0)
322 {
323 // Subnormal value since we already handled zero
324 unbiased_exponent = 1 + type_layout::exponent_bias;
325 }
326 else
327 {
328 aligned_significand |= static_cast<Unsigned_Integer>(1) << hex_bits;
329 unbiased_exponent = exponent + type_layout::exponent_bias;
330 }
331
332 // Bounds check the exponent
333 BOOST_IF_CONSTEXPR (std::is_same<Real, float>::value)
334 {
335 if (unbiased_exponent > 127)
336 {
337 unbiased_exponent -= 256;
338 }
339 }
340 else BOOST_IF_CONSTEXPR (std::is_same<Real, double>::value)
341 {
342 if (unbiased_exponent > 1023)
343 {
344 unbiased_exponent -= 2048;
345 }
346 }
347 else BOOST_IF_CONSTEXPR (std::is_same<Real, long double>::value)
348 {
349 #if BOOST_CHARCONV_LDBL_BITS == 80
350 while (unbiased_exponent > 16383)
351 {
352 unbiased_exponent -= 32768;
353 }
354 unbiased_exponent -= 3;
355 #else
356 while (unbiased_exponent > 16383)
357 {
358 unbiased_exponent -= 32768;
359 }
360 #endif
361 }
362 else
363 {
364 while (unbiased_exponent > 16383)
365 {
366 unbiased_exponent -= 32768;
367 }
368 }
369
370 const std::uint32_t abs_unbiased_exponent = unbiased_exponent < 0 ? static_cast<std::uint32_t>(-unbiased_exponent) :
371 static_cast<std::uint32_t>(unbiased_exponent);
372
373 // Bounds check
374 const std::ptrdiff_t total_length = total_buffer_length(real_precision, abs_unbiased_exponent, (value < 0));
375 if (total_length > buffer_size)
376 {
377 return {.ptr: last, .ec: std::errc::value_too_large};
378 }
379
380 // Round if required
381 if (real_precision < hex_precision)
382 {
383 const int lost_bits = (hex_precision - real_precision) * nibble_bits;
384 const Unsigned_Integer lsb_bit = aligned_significand;
385 const Unsigned_Integer round_bit = aligned_significand << 1;
386 const Unsigned_Integer tail_bit = round_bit - 1;
387 const Unsigned_Integer round = round_bit & (tail_bit | lsb_bit) & (static_cast<Unsigned_Integer>(1) << lost_bits);
388 aligned_significand += round;
389 }
390
391 // Print the sign
392 if (value < 0)
393 {
394 *first++ = '-';
395 }
396
397 // Print the integral part
398 #if BOOST_CHARCONV_LDBL_BITS == 80
399 std::uint32_t leading_nibble;
400 BOOST_IF_CONSTEXPR (std::is_same<Real, long double>::value)
401 {
402 leading_nibble = static_cast<std::uint32_t>(significand >> hex_bits);
403 }
404 else
405 {
406 leading_nibble = static_cast<std::uint32_t>(aligned_significand >> hex_bits);
407 }
408 #else
409 const auto leading_nibble = static_cast<std::uint32_t>(aligned_significand >> hex_bits);
410 #endif
411
412 BOOST_CHARCONV_ASSERT(leading_nibble < 16);
413 *first++ = digit_table[leading_nibble];
414
415 aligned_significand &= hex_mask;
416
417 // Print the fractional part
418 if (real_precision > 0)
419 {
420 *first++ = '.';
421 std::int32_t remaining_bits = hex_bits;
422
423 while (true)
424 {
425 remaining_bits -= nibble_bits;
426 const auto current_nibble = static_cast<std::uint32_t>(aligned_significand >> remaining_bits);
427 *first++ = digit_table[current_nibble];
428
429 --real_precision;
430 if (real_precision == 0)
431 {
432 break;
433 }
434 else if (remaining_bits == 0)
435 {
436 // Do not print trailing zeros with unspecified precision
437 if (precision != -1)
438 {
439 std::memset(s: first, c: '0', n: static_cast<std::size_t>(real_precision));
440 first += real_precision;
441 }
442 break;
443 }
444
445 // Mask away the hexit we just printed
446 aligned_significand &= (static_cast<Unsigned_Integer>(1) << remaining_bits) - 1;
447 }
448 }
449
450 // Remove any trailing zeros if the precision was unspecified
451 if (precision == -1)
452 {
453 --first;
454 while (*first == '0')
455 {
456 --first;
457 }
458 ++first;
459 }
460
461 // Print the exponent
462 *first++ = 'p';
463 if (unbiased_exponent < 0)
464 {
465 *first++ = '-';
466 }
467 else
468 {
469 *first++ = '+';
470 }
471
472 return to_chars_int(first, last, value: abs_unbiased_exponent);
473}

479template <typename Real>
480to_chars_result to_chars_fixed_impl(char* first, char* last, Real value, chars_format fmt = chars_format::general, int precision = -1) noexcept
481{
482 const std::ptrdiff_t buffer_size = last - first;
483 auto real_precision = get_real_precision<Real>(precision);
484 if (buffer_size < real_precision || first > last)
485 {
486 return {.ptr: last, .ec: std::errc::value_too_large};
487 }
488
489 auto abs_value = std::abs(value);
490
491 auto value_struct = boost::charconv::detail::to_decimal(value);
492 if (value_struct.is_negative)
493 {
494 *first++ = '-';
495 }
496
497 int num_dig = 0;
498 if (precision != -1)
499 {
500 num_dig = num_digits(value_struct.significand);
501 while (num_dig > precision + 2)
502 {
503 value_struct.significand /= 10;
504 ++value_struct.exponent;
505 --num_dig;
506 }
507
508 if (num_dig == precision + 2)
509 {
510 const auto trailing_dig = value_struct.significand % 10;
511 value_struct.significand /= 10;
512 ++value_struct.exponent;
513 --num_dig;
514
515 if (trailing_dig >= 5)
516 {
517 ++value_struct.significand;
518 }
519 }
520
521 // In general formatting we remove trailing 0s
522 if (fmt == chars_format::general)
523 {
524 while (value_struct.significand % 10 == 0)
525 {
526 value_struct.significand /= 10;
527 ++value_struct.exponent;
528 --num_dig;
529 }
530 }
531 }
532
533 // Make sure the result will fit in the buffer
534 const std::ptrdiff_t total_length = total_buffer_length(num_dig, value_struct.exponent, (value < 0));
535 if (total_length > buffer_size)
536 {
537 return {.ptr: last, .ec: std::errc::value_too_large};
538 }
539
540 auto r = to_chars_integer_impl(first, last, value_struct.significand);
541 if (r.ec != std::errc())
542 {
543 return r;
544 }
545
546 // Bounds check
547 if (abs_value >= 1)
548 {
549 if (value_struct.exponent < 0 && -value_struct.exponent < buffer_size)
550 {
551 std::memmove(dest: r.ptr + value_struct.exponent + 1, src: r.ptr + value_struct.exponent,
552 n: static_cast<std::size_t>(-value_struct.exponent));
553 std::memset(s: r.ptr + value_struct.exponent, c: '.', n: 1);
554 ++r.ptr;
555 }
556
557 while (std::fmod(abs_value, 10) == 0)
558 {
559 *r.ptr++ = '0';
560 abs_value /= 10;
561 }
562 }
563 else
564 {
565 #ifdef BOOST_CHARCONV_DEBUG_FIXED
566 std::cerr << std::setprecision(std::numeric_limits<Real>::digits10) << "Value: " << value
567 << "\n Buf: " << first
568 << "\n sig: " << value_struct.significand
569 << "\n exp: " << value_struct.exponent << std::endl;
570 #endif
571
572 const auto offset_bytes = static_cast<std::size_t>(-value_struct.exponent - num_dig);
573
574 std::memmove(dest: first + 2 + static_cast<std::size_t>(value_struct.is_negative) + offset_bytes,
575 src: first + static_cast<std::size_t>(value_struct.is_negative),
576 n: static_cast<std::size_t>(-value_struct.exponent) - offset_bytes);
577
578 std::memcpy(dest: first + static_cast<std::size_t>(value_struct.is_negative), src: "0.", n: 2U);
579 first += 2;
580 r.ptr += 2;
581
582 while (num_dig < -value_struct.exponent)
583 {
584 *first++ = '0';
585 ++num_dig;
586 ++r.ptr;
587 }
588 }
589
590 return { r.ptr, std::errc() };
591}

593template <typename Real>
594to_chars_result to_chars_float_impl(char* first, char* last, Real value, chars_format fmt, int precision) noexcept
595{
596 using Unsigned_Integer = typename std::conditional<std::is_same<Real, double>::value, std::uint64_t, std::uint32_t>::type;
597
598 // Sanity check our bounds
599 if (first >= last)
600 {
601 return {.ptr: last, .ec: std::errc::value_too_large};
602 }
603
604 auto abs_value = std::abs(value);
605 constexpr auto max_fractional_value = std::is_same<Real, double>::value ? static_cast<Real>(1e16) : static_cast<Real>(1e7);
606 constexpr auto max_value = static_cast<Real>((std::numeric_limits<Unsigned_Integer>::max)());
607
608 // Unspecified precision so we always go with the shortest representation
609 if (precision == -1)
610 {
611 if (fmt == boost::charconv::chars_format::general || fmt == boost::charconv::chars_format::fixed)
612 {
613 if (abs_value >= 1 && abs_value < max_fractional_value)
614 {
615 return to_chars_fixed_impl(first, last, value, fmt, precision);
616 }
617 else if (abs_value >= max_fractional_value && abs_value < max_value)
618 {
619 if (value < 0)
620 {
621 *first++ = '-';
622 }
623 return to_chars_integer_impl(first, last, value: static_cast<std::uint64_t>(abs_value));
624 }
625 else
626 {
627 return boost::charconv::detail::dragonbox_to_chars(value, first, last, fmt);
628 }
629 }
630 else if (fmt == boost::charconv::chars_format::scientific)
631 {
632 return boost::charconv::detail::dragonbox_to_chars(value, first, last, fmt);
633 }
634 }
635 else
636 {
637 if (fmt != boost::charconv::chars_format::hex)
638 {
639 if (fmt == boost::charconv::chars_format::general)
640 {
641 constexpr int max_output_length = std::is_same<Real, double>::value ? 773 : 117;
642 constexpr int max_precision = std::is_same<Real, double>::value ? 767 : 112;
643 // We remove trailing zeros, so precision > max_precision is same as precision == max_precision.
644 if (precision > max_precision)
645 {
646 precision = max_precision;
647 }
648 char temp_buffer[max_output_length];
649 auto result = boost::charconv::detail::floff<boost::charconv::detail::main_cache_full,
650 boost::charconv::detail::extended_cache_long>(value, precision,
651 temp_buffer,
652 temp_buffer + max_output_length,
653 fmt);
654 auto output_size = static_cast<std::size_t>(result.ptr - temp_buffer);
655 if (static_cast<std::size_t>(last - first) < output_size)
656 {
657 return {.ptr: last, .ec: std::errc::value_too_large};
658 }
659 std::memcpy(dest: first, src: temp_buffer, n: output_size);
660 return {.ptr: first + output_size, .ec: std::errc()};
661
662 }
663 return boost::charconv::detail::floff<boost::charconv::detail::main_cache_full,
664 boost::charconv::detail::extended_cache_long>(value, precision,
665 first, last, fmt);
666 }
667 }
668
669 const int classification = std::fpclassify(value);
670 switch (classification)
671 {
672 case FP_INFINITE:
673 case FP_NAN:
674 // The dragonbox impl will return the correct type of NaN
675 return boost::charconv::detail::dragonbox_to_chars(value, first, last, chars_format::general);
676 case FP_ZERO:
677 if (std::signbit(value))
678 {
679 *first++ = '-';
680 }
681 std::memcpy(dest: first, src: "0p+0", n: 4); // NOLINT : No null terminator is purposeful
682 return {.ptr: first + 4, .ec: std::errc()};
683 default:
684 // Do nothing
685 (void)precision;
686 }
687
688 // Hex handles both cases already
689 return boost::charconv::detail::to_chars_hex(first, last, value, precision);
690}

694template <>
695to_chars_result to_chars_float_impl(char* first, char* last, long double value, chars_format fmt, int precision) noexcept
696{
697 static_assert(std::numeric_limits<long double>::is_iec559, "Long double must be IEEE 754 compliant");
698
699 const auto classification = std::fpclassify(x: value);
700 #if BOOST_CHARCONV_LDBL_BITS == 128
701 if (classification == FP_NAN || classification == FP_INFINITE)
702 {
703 return boost::charconv::detail::to_chars_nonfinite(first, last, value, classification);
704 }
705 #else
706 if (classification == FP_NAN || classification == FP_INFINITE)
707 {
708 const auto fd128 = boost::charconv::detail::ryu::long_double_to_fd128(d: value);
709 const auto num_chars = boost::charconv::detail::ryu::generic_to_chars(v: fd128, result: first, result_size: last - first, fmt, precision);
710
711 if (num_chars > 0)
712 {
713 return { .ptr: first + num_chars, .ec: std::errc() };
714 }
715 else
716 {
717 return {.ptr: last, .ec: std::errc::value_too_large};
718 }

767template <>
768to_chars_result to_chars_float_impl(char* first, char* last, __float128 value, chars_format fmt, int precision) noexcept
769{
770 // Sanity check our bounds
771 if (first >= last)
772 {
773 return {last, std::errc::value_too_large};
774 }
775
776 char* const original_first = first;
777
778 if (isnanq(value))
779 {
780 return boost::charconv::detail::to_chars_nonfinite(first, last, value, FP_NAN);
781 }
782 else if (isinfq(value))
783 {
784 return boost::charconv::detail::to_chars_nonfinite(first, last, value, FP_INFINITE);
785 }
786
787 // Sanity check our bounds
788 const std::ptrdiff_t buffer_size = last - first;
789 auto real_precision = boost::charconv::detail::get_real_precision<__float128>(precision);
790 if (buffer_size < real_precision || first > last)
791 {
792 return {last, std::errc::value_too_large};
793 }
794
795 if ((fmt == boost::charconv::chars_format::general || fmt == boost::charconv::chars_format::scientific))
796 {
797 const auto fd128 = boost::charconv::detail::ryu::float128_to_fd128(value);
798 const auto num_chars = boost::charconv::detail::ryu::generic_to_chars(fd128, first, last - first, fmt, precision);
799
800 if (num_chars > 0)
801 {
802 return { first + num_chars, std::errc() };
803 }
804 else if (num_chars == -1)
805 {
806 return {last, std::errc::value_too_large};
807 }
808 }
809 else if (fmt == boost::charconv::chars_format::hex)
810 {
811 return boost::charconv::detail::to_chars_hex(first, last, value, precision);
812 }
813 else if (fmt == boost::charconv::chars_format::fixed)
814 {
815 const auto fd128 = boost::charconv::detail::ryu::float128_to_fd128(value);
816 const auto num_chars = boost::charconv::detail::ryu::generic_to_chars_fixed(fd128, first, last - first, precision);
817

819 {
820 return { first + num_chars, std::errc() };

825 }
