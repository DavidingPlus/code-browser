38template <typename T>
39from_chars_result from_chars_float_impl(const char* first, const char* last, T& value, chars_format fmt) noexcept
40{
41 bool sign {};
42 std::uint64_t significand {};
43 std::int64_t exponent {};
44
45 auto r = boost::charconv::detail::parser(first, last, sign, significand, exponent, fmt);
46 if (r.ec == std::errc::value_too_large)
47 {
48 r.ec = std::errc();
49 value = sign ? -std::numeric_limits<T>::infinity() : std::numeric_limits<T>::infinity();
50 return r;
51 }
52 else if (r.ec == std::errc::not_supported)
53 {
54 r.ec = std::errc();
55 if (significand == 0)
56 {
57 value = sign ? -std::numeric_limits<T>::quiet_NaN() : std::numeric_limits<long double>::quiet_NaN();
58 }
59 else
60 {
61 value = sign ? -std::numeric_limits<T>::signaling_NaN() : std::numeric_limits<T>::signaling_NaN();
62 }
63
64 return r;
65 }
66 if (r.ec != std::errc())
67 {
68 return r;
69 }
70 else if (significand == 0)
71 {
72 value = sign ? static_cast<T>(-0.0L) : static_cast<T>(0.0L);
73 return r;
74 }
75 else if (exponent == -1)
76 {
77 // A full length significand e.g. -1985444280612224 with a power of -1 sometimes
78 // fails in compute_float64 but is trivial to calculate
79 // Found investigating GitHub issue #47
80 value = (sign ? -static_cast<T>(significand) : static_cast<T>(significand)) / 10;
81 }
82
83 bool success {};
84 T return_val {};
85 BOOST_IF_CONSTEXPR (std::is_same<T, float>::value)
86 {
87 return_val = compute_float32(power: exponent, i: significand, negative: sign, success);
88 }
89 else
90 {
91 return_val = compute_float64(power: exponent, i: significand, negative: sign, success);
92 }
93
94 if (!success)
95 {
96 if (significand == 1 && exponent == 0)
97 {
98 value = 1;
99 r.ptr = last;
100 r.ec = std::errc();
101 }
102 else
103 {
104 BOOST_IF_CONSTEXPR (std::is_same<T, float>::value)
105 {
106 #ifndef __INTEL_LLVM_COMPILER
107 if (return_val == HUGE_VALF || return_val == -HUGE_VALF)
108 #else
109 if (return_val >= std::numeric_limits<T>::max() || return_val <= std::numeric_limits<T>::lowest())
110 #endif
111 {
112 value = return_val;
113 r.ec = std::errc::result_out_of_range;
114 }
115 else if (exponent < -46)
116 {
117 value = sign ? -0.0F : 0.0;
118 r.ec = std::errc::result_out_of_range;
119 }
120 else
121 {
122 r = from_chars_strtod(first, r.ptr, value);
123 }
124 }
125 else BOOST_IF_CONSTEXPR (std::is_same<T, double>::value)
126 {
127 #ifndef __INTEL_LLVM_COMPILER
128 if (return_val == HUGE_VAL || return_val == -HUGE_VAL)
129 #else
130 if (return_val >= std::numeric_limits<T>::max() || return_val <= std::numeric_limits<T>::lowest())
131 #endif
132 {
133 value = return_val;
134 r.ec = std::errc::result_out_of_range;
135 }
136 else if (exponent < -342)
137 {
138 value = sign ? -0.0 : 0.0;
139 r.ec = std::errc::result_out_of_range;
140 }
141 else
142 {
143 r = from_chars_strtod(first, r.ptr, value);
144 }
145 }
146 else BOOST_IF_CONSTEXPR (std::is_same<T, long double>::value)
147 {
148 #ifndef __INTEL_LLVM_COMPILER
149 if (return_val == HUGE_VALL || return_val == -HUGE_VALL)
150 #else
151 if (return_val >= std::numeric_limits<T>::max() || return_val <= std::numeric_limits<T>::lowest())
152 #endif
153 {
154 value = return_val;
155 r.ec = std::errc::result_out_of_range;
156 }
157 #if BOOST_CHARCONV_LDBL_BITS == 64
158 else if (exponent < -342)
159 #else // 80 or 128 bit long doubles have same range
160 else if (exponent < -4965)
161 #endif
162 {
163 value = sign ? -0.0L : 0.0L;
164 r.ec = std::errc::result_out_of_range;
165 }
166 else
167 {
168 r = from_chars_strtod(first, r.ptr, value);
169 }
170 }
171 }
172 }
173 else
174 {
175 value = return_val;
176 }
177
178 return r;
179}
