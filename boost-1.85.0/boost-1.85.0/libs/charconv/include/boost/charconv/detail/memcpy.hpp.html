5#ifndef BOOST_CHARCONV_DETAIL_MEMCPY_HPP
6#define BOOST_CHARCONV_DETAIL_MEMCPY_HPP
7
8#include <boost/charconv/detail/config.hpp>
9#include <cstring>
10#include <cstdint>
11
12// https://gcc.gnu.org/bugzilla/show_bug.cgi?id=89689
13// GCC 10 added checks for length of memcpy which yields the following warning (converted to error with -Werror)
14// /usr/include/x86_64-linux-gnu/bits/string_fortified.h:34:33: error:
15// ‘void* __builtin___memcpy_chk(void*, const void*, long unsigned int, long unsigned int)’ specified size between
16// 18446744071562067968 and 18446744073709551615 exceeds maximum object size 9223372036854775807 [-Werror=stringop-overflow=]
17//
18// memcpy is defined as taking a size_t for the count and the largest count this will recieve is the number of digits
19// in a 128-bit int (39) so we can safely ignore
20#if defined(__GNUC__) && __GNUC__ >= 10
21# pragma GCC diagnostic push
22# pragma GCC diagnostic ignored "-Wstringop-overflow"
23# define BOOST_CHARCONV_STRINGOP_OVERFLOW_DISABLED
24#endif
25
26namespace boost { namespace charconv { namespace detail {
27
28#if !defined(BOOST_CHARCONV_NO_CONSTEXPR_DETECTION) && defined(BOOST_CXX14_CONSTEXPR)
29
30#define BOOST_CHARCONV_CONSTEXPR constexpr
31
32constexpr char* memcpy(char* dest, const char* src, std::size_t count)
33{
34 if (BOOST_CHARCONV_IS_CONSTANT_EVALUATED(count))
35 {
36 for (std::size_t i = 0; i < count; ++i)
37 {
38 *(dest + i) = *(src + i);
39 }
40
41 return dest;
42 }
43 else
44 {
45 // Workaround for GCC-11 because it does not honor GCC diagnostic ignored
46 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=53431
47 // Hopefully the optimizer turns this into memcpy
48 #if defined(__GNUC__) && __GNUC__ == 11
49 for (std::size_t i = 0; i < count; ++i)
50 {
51 *(dest + i) = *(src + i);
52 }
53
54 return dest;
55 #else
56 return static_cast<char*>(std::memcpy(dest: dest, src: src, n: count));
57 #endif
58 }
59}
60
61#else // Either not C++14 or no way of telling if we are in a constexpr context
62
63#define BOOST_CHARCONV_CONSTEXPR inline
64
65inline void* memcpy(void* dest, const void* src, std::size_t count)
66{
67 return std::memcpy(dest, src, count);
68}
69
70#endif
71
72}}} // Namespace boost::charconv::detail
73
74#ifdef BOOST_CHARCONV_STRINGOP_OVERFLOW_DISABLED

76#endif
