22// Use 128-bit integers and suppress warnings for using extensions
23#if defined(BOOST_HAS_INT128)

56// Determine endianness
57#if defined(_WIN32)
58
59#define BOOST_CHARCONV_ENDIAN_BIG_BYTE 0
60#define BOOST_CHARCONV_ENDIAN_LITTLE_BYTE 1
61
62#elif defined(__BYTE_ORDER__)
63
64#define BOOST_CHARCONV_ENDIAN_BIG_BYTE (__BYTE_ORDER__ == __ORDER_BIG_ENDIAN__)
65#define BOOST_CHARCONV_ENDIAN_LITTLE_BYTE (__BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__)
66
67#else
68
69#error Could not determine endian type. Please file an issue at https://github.com/cppalliance/charconv with your architecture
70
71#endif // Determine endianness
72
73// Inclue intrinsics if available
74#if defined(BOOST_MSVC)
75# include <intrin.h>
76# if defined(_WIN64)
77# define BOOST_CHARCONV_HAS_MSVC_64BIT_INTRINSICS
78# else
79# define BOOST_CHARCONV_HAS_MSVC_32BIT_INTRINSICS
80# endif
81#elif (defined(__x86_64__) || defined(__i386__))
82# include <x86intrin.h>
83# define BOOST_CHARCONV_HAS_X86_INTRINSICS
84#elif defined(__ARM_NEON__)
85# include <arm_neon.h>
86# define BOOST_CHARCONV_HAS_ARM_INTRINSICS
87#else
88# define BOOST_CHARCONV_HAS_NO_INTRINSICS
89#endif
90
91static_assert((BOOST_CHARCONV_ENDIAN_BIG_BYTE || BOOST_CHARCONV_ENDIAN_LITTLE_BYTE) &&
92 !(BOOST_CHARCONV_ENDIAN_BIG_BYTE && BOOST_CHARCONV_ENDIAN_LITTLE_BYTE),
93"Inconsistent endianness detected. Please file an issue at https://github.com/cppalliance/charconv with your architecture");

95// Suppress additional buffer overrun check.
96// I have no idea why MSVC thinks some functions here are vulnerable to the buffer overrun
97// attacks. No, they aren't.
98#if defined(__GNUC__) || defined(__clang__)
99 #define BOOST_CHARCONV_SAFEBUFFERS
100#elif defined(_MSC_VER)
101 #define BOOST_CHARCONV_SAFEBUFFERS __declspec(safebuffers)
102#else
103 #define BOOST_CHARCONV_SAFEBUFFERS
104#endif

155#elif defined(BOOST_CHARCONV_HAS_BUILTIN_IS_CONSTANT_EVALUATED)
156# define BOOST_CHARCONV_IS_CONSTANT_EVALUATED(x) __builtin_is_constant_evaluated()

158# define BOOST_CHARCONV_IS_CONSTANT_EVALUATED(x) __builtin_constant_p(x)
