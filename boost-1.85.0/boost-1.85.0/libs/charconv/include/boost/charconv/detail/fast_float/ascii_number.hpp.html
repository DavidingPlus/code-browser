19// Next function can be micro-optimized, but compilers are entirely
20// able to optimize it well.
21template <typename UC>
22BOOST_FORCEINLINE constexpr bool is_integer(UC c) noexcept {
23 return !(c > UC('9') || c < UC('0'));
24}

26BOOST_FORCEINLINE constexpr uint64_t byteswap(uint64_t val) {
27 return (val & 0xFF00000000000000) >> 56
28 | (val & 0x00FF000000000000) >> 40
29 | (val & 0x0000FF0000000000) >> 24
30 | (val & 0x000000FF00000000) >> 8
31 | (val & 0x00000000FF000000) << 8
32 | (val & 0x0000000000FF0000) << 24
33 | (val & 0x000000000000FF00) << 40
34 | (val & 0x00000000000000FF) << 56;
35}

37BOOST_FORCEINLINE BOOST_CHARCONV_FASTFLOAT_CONSTEXPR20
38uint64_t read_u64(const char *chars) {
39 if (cpp20_and_in_constexpr()) {
40 uint64_t val = 0;
41 for(int i = 0; i < 8; ++i) {
42 val |= uint64_t(*chars) << (i*8);
43 ++chars;
44 }
45 return val;
46 }
47 uint64_t val;
48 ::memcpy(dest: &val, src: chars, n: sizeof(uint64_t));
49#if BOOST_CHARCONV_FASTFLOAT_IS_BIG_ENDIAN == 1
50 // Need to read as-if the number was in little-endian order.
51 val = byteswap(val);
52#endif
53 return val;
54}

56BOOST_FORCEINLINE BOOST_CHARCONV_FASTFLOAT_CONSTEXPR20
57void write_u64(uint8_t *chars, uint64_t val) {
58 if (cpp20_and_in_constexpr()) {
59 for(int i = 0; i < 8; ++i) {
60 *chars = uint8_t(val);
61 val >>= 8;
62 ++chars;
63 }
64 return;
65 }
66#if BOOST_CHARCONV_FASTFLOAT_IS_BIG_ENDIAN == 1
67 // Need to read as-if the number was in little-endian order.
68 val = byteswap(val);
69#endif
70 ::memcpy(dest: chars, src: &val, n: sizeof(uint64_t));
71}

73// credit @aqrit
74BOOST_FORCEINLINE BOOST_CHARCONV_FASTFLOAT_CONSTEXPR14
75uint32_t parse_eight_digits_unrolled(uint64_t val) {
76 constexpr uint64_t mask = 0x000000FF000000FF;
77 constexpr uint64_t mul1 = 0x000F424000000064; // 100 + (1000000ULL << 32)
78 constexpr uint64_t mul2 = 0x0000271000000001; // 1 + (10000ULL << 32)
79 val -= 0x3030303030303030;
80 val = (val * 10) + (val >> 8); // val = (val * 2561) >> 8;
81 val = (((val & mask) * mul1) + (((val >> 16) & mask) * mul2)) >> 32;
82 return uint32_t(val);
83}

101BOOST_FORCEINLINE constexpr bool is_made_of_eight_digits_fast(uint64_t val) noexcept {
102 return !((((val + 0x4646464646464646) | (val - 0x3030303030303030)) & 0x8080808080808080));
103}

120template <typename UC>
121struct parsed_number_string_t {
122 int64_t exponent{0};
123 uint64_t mantissa{0};
124 UC const * lastmatch{nullptr};
125 bool negative{false};
126 bool valid{false};
127 bool too_many_digits{false};
128 // contains the range of the significant digits
129 span<const UC> integer{}; // non-nullable
130 span<const UC> fraction{}; // nullable
131};
132using byte_span = span<char>;
133using parsed_number_string = parsed_number_string_t<char>;
134// Assuming that you use no more than 19 digits, this will
135// parse an ASCII string.
136template <typename UC>
137BOOST_FORCEINLINE BOOST_CHARCONV_FASTFLOAT_CONSTEXPR20
138parsed_number_string_t<UC> parse_number_string(UC const *p, UC const * pend, parse_options_t<UC> options) noexcept {
139 chars_format const fmt = options.format;
140 UC const decimal_point = options.decimal_point;
141
142 parsed_number_string_t<UC> answer;
143 answer.valid = false;
144 answer.too_many_digits = false;
145 answer.negative = (*p == UC('-'));
146#ifdef BOOST_CHARCONV_FASTFLOAT_ALLOWS_LEADING_PLUS // disabled by default
147 if ((*p == UC('-')) || (*p == UC('+')))
148#else
149 if (*p == UC('-')) // C++17 20.19.3.(7.1) explicitly forbids '+' sign here
150#endif
151 {
152 ++p;
153 if (p == pend) {
154 return answer;
155 }
156 if (!is_integer(*p) && (*p != decimal_point)) { // a sign must be followed by an integer or the dot
157 return answer;
158 }
159 }
160 UC const * const start_digits = p;
161
162 uint64_t i = 0; // an unsigned int avoids signed overflows (which are bad)
163
164 while ((p != pend) && is_integer(*p)) {
165 // a multiplication by 10 is cheaper than an arbitrary integer
166 // multiplication
167 i = 10 * i +
168 uint64_t(*p - UC('0')); // might overflow, we will handle the overflow later
169 ++p;
170 }
171 UC const * const end_of_integer_part = p;
172 int64_t digit_count = int64_t(end_of_integer_part - start_digits);
173 answer.integer = span<const UC>(start_digits, size_t(digit_count));
174 int64_t exponent = 0;
175 if ((p != pend) && (*p == decimal_point)) {
176 ++p;
177 UC const * before = p;
178 // can occur at most twice without overflowing, but let it occur more, since
179 // for integers with many digits, digit parsing is the primary bottleneck.
180 if (std::is_same<UC,char>::value) {
181 while ((std::distance(p, pend) >= 8) && is_made_of_eight_digits_fast(p)) {
182 i = i * 100000000 + parse_eight_digits_unrolled(p); // in rare cases, this will overflow, but that's ok
183 p += 8;
184 }
185 }
186 while ((p != pend) && is_integer(*p)) {
187 uint8_t digit = uint8_t(*p - UC('0'));
188 ++p;
189 i = i * 10 + digit; // in rare cases, this will overflow, but that's ok
190 }
191 exponent = before - p;
192 answer.fraction = span<const UC>(before, size_t(p - before));
193 digit_count -= exponent;
194 }
195 // we must have encountered at least one integer!
196 if (digit_count == 0) {
197 return answer;
198 }
199 int64_t exp_number = 0; // explicit exponential part
200 if ((static_cast<unsigned>(fmt) & static_cast<unsigned>(chars_format::scientific)) && (p != pend) && ((UC('e') == *p) || (UC('E') == *p))) {
201 UC const * location_of_e = p;
202 ++p;
203 bool neg_exp = false;
204 if ((p != pend) && (UC('-') == *p)) {
205 neg_exp = true;
206 ++p;
207 } else if ((p != pend) && (UC('+') == *p)) { // '+' on exponent is allowed by C++17 20.19.3.(7.1)
208 ++p;
209 }
210 if ((p == pend) || !is_integer(*p)) {
211 if(!(static_cast<unsigned>(fmt) & static_cast<unsigned>(chars_format::fixed))) {
212 // We are in error.
213 return answer;
214 }
215 // Otherwise, we will be ignoring the 'e'.
216 p = location_of_e;
217 } else {
218 while ((p != pend) && is_integer(*p)) {
219 uint8_t digit = uint8_t(*p - UC('0'));
220 if (exp_number < 0x10000000) {
221 exp_number = 10 * exp_number + digit;
222 }
223 ++p;
224 }
225 if(neg_exp) { exp_number = - exp_number; }
226 exponent += exp_number;
227 }
228 } else {
229 // If it scientific and not fixed, we have to bail out.
230 if((static_cast<unsigned>(fmt) & static_cast<unsigned>(chars_format::scientific)) &&
231 !(static_cast<unsigned>(fmt) & static_cast<unsigned>(chars_format::fixed)))
232 {
233 return answer;
234 }
235 }
236 answer.lastmatch = p;
237 answer.valid = true;
238
239 // If we frequently had to deal with long strings of digits,
240 // we could extend our code by using a 128-bit integer instead
241 // of a 64-bit integer. However, this is uncommon.
242 //
243 // We can deal with up to 19 digits.
244 if (digit_count > 19) { // this is uncommon
245 // It is possible that the integer had an overflow.
246 // We have to handle the case where we have 0.0000somenumber.
247 // We need to be mindful of the case where we only have zeroes...
248 // E.g., 0.000000000...000.
249 UC const * start = start_digits;
250 while ((start != pend) && (*start == UC('0') || *start == decimal_point)) {
251 if(*start == UC('0')) { digit_count --; }
252 start++;
253 }
254 if (digit_count > 19) {
255 answer.too_many_digits = true;
256 // Let us start again, this time, avoiding overflows.
257 // We don't need to check if is_integer, since we use the
258 // pre-tokenized spans from above.
259 i = 0;
260 p = answer.integer.ptr;
261 UC const * int_end = p + answer.integer.len();
262 constexpr uint64_t minimal_nineteen_digit_integer{1000000000000000000};
263 while((i < minimal_nineteen_digit_integer) && (p != int_end)) {
264 i = i * 10 + uint64_t(*p - UC('0'));
265 ++p;
266 }
267 if (i >= minimal_nineteen_digit_integer) { // We have a big integers
268 exponent = end_of_integer_part - p + exp_number;
269 } else { // We have a value with a fractional component.
270 p = answer.fraction.ptr;
271 UC const * frac_end = p + answer.fraction.len();
272 while((i < minimal_nineteen_digit_integer) && (p != frac_end)) {
273 i = i * 10 + uint64_t(*p - UC('0'));
274 ++p;
275 }
276 exponent = answer.fraction.ptr - p + exp_number;
277 }
278 // We have now corrected both exponent and i, to a truncated value
279 }
280 }
281 answer.exponent = exponent;
282 answer.mantissa = i;
283 return answer;
284}
