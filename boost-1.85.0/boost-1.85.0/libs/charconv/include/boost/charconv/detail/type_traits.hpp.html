5#ifndef BOOST_CHARCONV_DETAIL_TYPE_TRAITS_HPP
6#define BOOST_CHARCONV_DETAIL_TYPE_TRAITS_HPP
7
8#include <boost/charconv/detail/config.hpp>
9#include <type_traits>
10
11namespace boost { namespace charconv { namespace detail {
12
13template <typename T>
14struct is_signed { static constexpr bool value = std::is_signed<T>::value; };
15
16#ifdef BOOST_CHARCONV_HAS_INT128
17
18template <>
19struct is_signed<boost::int128_type> { static constexpr bool value = true; };
20
21template <>
22struct is_signed<boost::uint128_type> { static constexpr bool value = false; };
23
24#endif
25
26#if defined(BOOST_NO_CXX17_INLINE_VARIABLES) && (!defined(BOOST_MSVC) || BOOST_MSVC != 1900)
27
28template <typename T>
29constexpr bool is_signed<T>::value;
30
31#endif
32
33template <typename T>
34struct make_unsigned { using type = typename std::make_unsigned<T>::type; };
35
36template <>
37struct make_unsigned<uint128> { using type = uint128; };
38
39#ifdef BOOST_CHARCONV_HAS_INT128
40
41template <>
42struct make_unsigned<boost::int128_type> { using type = boost::uint128_type; };
43
44template <>
45struct make_unsigned<boost::uint128_type> { using type = boost::uint128_type; };
46
47#endif
48
49template <typename T>
50using make_unsigned_t = typename make_unsigned<T>::type;
51
52template <typename T>
53struct make_signed { using type = typename std::make_signed<T>::type; };
54
55#ifdef BOOST_CHARCONV_HAS_INT128
56
57template <>
58struct make_signed<boost::int128_type> { using type = boost::int128_type; };
59
60template <>
61struct make_signed<boost::uint128_type> { using type = boost::int128_type; };
62
63#endif
64
65}}} // Namespaces
66
67#endif //BOOST_CHARCONV_DETAIL_TYPE_TRAITS_HPP
