62// See: https://jk-jeon.github.io/posts/2022/02/jeaiii-algorithm/
63// https://arxiv.org/abs/2101.11408
64BOOST_CHARCONV_CONSTEXPR char* decompose32(std::uint32_t value, char* buffer) noexcept
65{
66 constexpr auto mask = (std::uint64_t(1) << 57) - 1;
67 auto y = value * std::uint64_t(1441151881);
68
69 for (std::size_t i {}; i < 10; i += 2)
70 {
71 boost::charconv::detail::memcpy(dest: buffer + i, src: radix_table + static_cast<std::size_t>(y >> 57) * 2, count: 2);
72 y &= mask;
73 y *= 100;
74 }
75
76 return buffer + 10;
77}

84template <typename Integer>
85BOOST_CHARCONV_CONSTEXPR to_chars_result to_chars_integer_impl(char* first, char* last, Integer value) noexcept
86{
87 using Unsigned_Integer = typename std::make_unsigned<Integer>::type;
88 Unsigned_Integer unsigned_value {};
89
90 char buffer[10] {};
91 int converted_value_digits {};
92 const std::ptrdiff_t user_buffer_size = last - first;
93 BOOST_ATTRIBUTE_UNUSED bool is_negative = false;
94
95 if (first > last)
96 {
97 return {.ptr: last, .ec: std::errc::invalid_argument};
98 }
99
100 // Strip the sign from the value and apply at the end after parsing if the type is signed
101 BOOST_IF_CONSTEXPR (std::is_signed<Integer>::value)
102 {
103 if (value < 0)
104 {
105 is_negative = true;
106 unsigned_value = apply_sign(value);
107 }
108 else
109 {
110 unsigned_value = static_cast<Unsigned_Integer>(value);
111 }
112 }
113 else
114 {
115 unsigned_value = static_cast<Unsigned_Integer>(value);
116 }
117
118 // If the type is less than 32 bits we can use this without change
119 // If the type is greater than 32 bits we use a binary search tree to figure out how many digits
120 // are present and then decompose the value into two (or more) std::uint32_t of known length so that we
121 // don't have the issue of removing leading zeros from the least significant digits
122
123 // Yields: warning C4127: conditional expression is constant because first half of the expression is constant,
124 // but we need to short circuit to avoid UB on the second half
125 if (std::numeric_limits<Integer>::digits <= std::numeric_limits<std::uint32_t>::digits ||
126 unsigned_value <= static_cast<Unsigned_Integer>((std::numeric_limits<std::uint32_t>::max)()))
127 {
128 const auto converted_value = static_cast<std::uint32_t>(unsigned_value);
129 converted_value_digits = num_digits(x: converted_value);
130
131 if (converted_value_digits > user_buffer_size)
132 {
133 return {.ptr: last, .ec: std::errc::value_too_large};
134 }
135
136 decompose32(value: converted_value, buffer);
137
138 if (is_negative)
139 {
140 *first++ = '-';
141 }
142
143 boost::charconv::detail::memcpy(dest: first, src: buffer + (sizeof(buffer) - static_cast<unsigned>(converted_value_digits)),
144 count: static_cast<std::size_t>(converted_value_digits));
145 }
146 else if (std::numeric_limits<Integer>::digits <= std::numeric_limits<std::uint64_t>::digits ||
147 static_cast<std::uint64_t>(unsigned_value) <= (std::numeric_limits<std::uint64_t>::max)())
148 {
149 auto converted_value = static_cast<std::uint64_t>(unsigned_value);
150 converted_value_digits = num_digits(x: converted_value);
151
152 if (converted_value_digits > user_buffer_size)
153 {
154 return {.ptr: last, .ec: std::errc::value_too_large};
155 }
156
157 if (is_negative)
158 {
159 *first++ = '-';
160 }
161
162 // Only store 9 digits in each to avoid overflow
163 if (num_digits(x: converted_value) <= 18)
164 {
165 const auto x = static_cast<std::uint32_t>(converted_value / UINT64_C(1000000000));
166 const auto y = static_cast<std::uint32_t>(converted_value % UINT64_C(1000000000));
167 const int first_value_chars = num_digits(x);
168
169 decompose32(value: x, buffer);
170 boost::charconv::detail::memcpy(dest: first, src: buffer + (sizeof(buffer) - static_cast<unsigned>(first_value_chars)),
171 count: static_cast<std::size_t>(first_value_chars));
172
173 decompose32(value: y, buffer);
174 boost::charconv::detail::memcpy(dest: first + first_value_chars, src: buffer + 1, count: sizeof(buffer) - 1);
175 }
176 else
177 {
178 const auto x = static_cast<std::uint32_t>(converted_value / UINT64_C(100000000000));
179 converted_value -= x * UINT64_C(100000000000);
180 const auto y = static_cast<std::uint32_t>(converted_value / UINT64_C(100));
181 const auto z = static_cast<std::uint32_t>(converted_value % UINT64_C(100));
182
183 if (converted_value_digits == 19)
184 {
185 decompose32(value: x, buffer);
186 boost::charconv::detail::memcpy(dest: first, src: buffer + 2, count: sizeof(buffer) - 2);
187
188 decompose32(value: y, buffer);
189 boost::charconv::detail::memcpy(dest: first + 8, src: buffer + 1, count: sizeof(buffer) - 1);
190
191 // Always prints 2 digits last
192 boost::charconv::detail::memcpy(dest: first + 17, src: radix_table + z * 2, count: 2);
193 }
194 else // 20
195 {
196 decompose32(value: x, buffer);
197 boost::charconv::detail::memcpy(dest: first, src: buffer + 1, count: sizeof(buffer) - 1);
198
199 decompose32(value: y, buffer);
200 boost::charconv::detail::memcpy(dest: first + 9, src: buffer + 1, count: sizeof(buffer) - 1);
201
202 // Always prints 2 digits last
203 boost::charconv::detail::memcpy(dest: first + 18, src: radix_table + z * 2, count: 2);
204 }
205 }
206 }
207
208 return {.ptr: first + converted_value_digits, .ec: std::errc()};
209}

211// Prior to GCC 10.3 std::numeric_limits was not specialized for __int128 which breaks the above control flow
212// Here we find if the 128-bit type will fit into a 64-bit type and use the above, or we use string manipulation
213// to extract the digits
214//
215// See: https://quuxplusone.github.io/blog/2019/02/28/is-int128-integral/
216template <typename Integer>
217BOOST_CHARCONV_CONSTEXPR to_chars_result to_chars_128integer_impl(char* first, char* last, Integer value) noexcept
218{
219 #ifdef BOOST_CHARCONV_HAS_INT128
220 using Unsigned_Integer = boost::uint128_type;
221 #else
222 using Unsigned_Integer = uint128;
223 #endif
224
225 Unsigned_Integer unsigned_value {};
226
227 const std::ptrdiff_t user_buffer_size = last - first;
228 BOOST_ATTRIBUTE_UNUSED bool is_negative = false;
229
230 if (first > last)
231 {
232 return {.ptr: last, .ec: std::errc::invalid_argument};
233 }
234
235 // Strip the sign from the value and apply at the end after parsing if the type is signed
236 #ifdef BOOST_CHARCONV_HAS_INT128
237 BOOST_IF_CONSTEXPR (std::is_same<boost::int128_type, Integer>::value)
238 {
239 if (value < 0)
240 {
241 is_negative = true;
242 unsigned_value = -(static_cast<Unsigned_Integer>(value));
243 }
244 else
245 {
246 unsigned_value = static_cast<Unsigned_Integer>(value);
247 }
248 }
249 else
250 #endif
251 {
252 unsigned_value = static_cast<Unsigned_Integer>(value);
253 }
254
255 auto converted_value = static_cast<Unsigned_Integer>(unsigned_value);
256
257 const int converted_value_digits = num_digits(x: converted_value);
258
259 if (converted_value_digits > user_buffer_size)
260 {
261 return {.ptr: last, .ec: std::errc::value_too_large};
262 }
263
264 if (is_negative)
265 {
266 *first++ = '-';
267 }
268
269 // If the value fits into 64 bits use the other method of processing
270 if (converted_value < (std::numeric_limits<std::uint64_t>::max)())
271 {
272 return to_chars_integer_impl(first, last, value: static_cast<std::uint64_t>(value));
273 }
274
275 constexpr std::uint32_t ten_9 = UINT32_C(1000000000);
276 char buffer[5][10] {};
277 int num_chars[5] {};
278 int i = 0;
279
280 while (converted_value != 0)
281 {
282 auto digits = static_cast<std::uint32_t>(converted_value % ten_9);
283 num_chars[i] = num_digits(x: digits);
284 decompose32(value: digits, buffer: buffer[i]); // Always returns 10 digits (to include leading 0s) which we want
285 converted_value = (converted_value - digits) / ten_9;
286 ++i;
287 }
288
289 --i;
290 auto offset = static_cast<std::size_t>(num_chars[i]);
291 boost::charconv::detail::memcpy(dest: first, src: buffer[i] + 10 - offset, count: offset);
292
293 while (i > 0)
294 {
295 --i;
296 boost::charconv::detail::memcpy(dest: first + offset, src: buffer[i] + 1, count: 9);
297 offset += 9;
298 }
299
300 return {.ptr: first + converted_value_digits, .ec: std::errc()};
301}

312// All other bases
313// Use a simple lookup table to put together the Integer in character form
314template <typename Integer, typename Unsigned_Integer>
315BOOST_CHARCONV_CONSTEXPR to_chars_result to_chars_integer_impl(char* first, char* last, Integer value, int base) noexcept
316{
317 const std::ptrdiff_t output_length = last - first;
318
319 if (!((first <= last) && (base >= 2 && base <= 36)))
320 {
321 return {.ptr: last, .ec: std::errc::invalid_argument};
322 }
323
324 if (value == 0)
325 {
326 *first++ = '0';
327 return {.ptr: first, .ec: std::errc()};
328 }
329
330 Unsigned_Integer unsigned_value {};
331 const auto unsigned_base = static_cast<Unsigned_Integer>(base);
332
333 BOOST_IF_CONSTEXPR (std::is_signed<Integer>::value)
334 {
335 if (value < 0)
336 {
337 *first++ = '-';
338 unsigned_value = static_cast<Unsigned_Integer>(detail::apply_sign(value));
339 }
340 else
341 {
342 unsigned_value = static_cast<Unsigned_Integer>(value);
343 }
344 }
345 else
346 {
347 unsigned_value = static_cast<Unsigned_Integer>(value);
348 }
349
350 constexpr Unsigned_Integer zero = 48U; // Char for '0'
351 constexpr auto buffer_size = sizeof(Unsigned_Integer) * CHAR_BIT;
352 char buffer[buffer_size] {};
353 const char* buffer_end = buffer + buffer_size;
354 char* end = buffer + buffer_size - 1;
355
356 // Work from LSB to MSB
357 switch (base)
358 {
359 case 2:
360 while (unsigned_value != 0)
361 {
362 *end-- = static_cast<char>(zero + (unsigned_value & 1U)); // 1<<1 - 1
363 unsigned_value >>= static_cast<Unsigned_Integer>(1);
364 }
365 break;
366
367 case 4:
368 while (unsigned_value != 0)
369 {
370 *end-- = static_cast<char>(zero + (unsigned_value & 3U)); // 1<<2 - 1
371 unsigned_value >>= static_cast<Unsigned_Integer>(2);
372 }
373 break;
374
375 case 8:
376 while (unsigned_value != 0)
377 {
378 *end-- = static_cast<char>(zero + (unsigned_value & 7U)); // 1<<3 - 1
379 unsigned_value >>= static_cast<Unsigned_Integer>(3);
380 }
381 break;
382
383 case 16:
384 while (unsigned_value != 0)
385 {
386 *end-- = digit_table[unsigned_value & 15U]; // 1<<4 - 1
387 unsigned_value >>= static_cast<Unsigned_Integer>(4);
388 }
389 break;
390
391 case 32:
392 while (unsigned_value != 0)
393 {
394 *end-- = digit_table[unsigned_value & 31U]; // 1<<5 - 1
395 unsigned_value >>= static_cast<Unsigned_Integer>(5);
396 }
397 break;
398
399 default:
400 while (unsigned_value != 0)
401 {
402 *end-- = digit_table[unsigned_value % unsigned_base];
403 unsigned_value /= unsigned_base;
404 }
405 break;
406 }
407
408 const std::ptrdiff_t num_chars = buffer_end - end - 1;
409
410 if (num_chars > output_length)
411 {
412 return {.ptr: last, .ec: std::errc::value_too_large};
413 }
414
415 boost::charconv::detail::memcpy(dest: first, src: buffer + (buffer_size - static_cast<unsigned long>(num_chars)),
416 count: static_cast<std::size_t>(num_chars));
417
418 return {.ptr: first + num_chars, .ec: std::errc()};
419}

431template <typename Integer>
432BOOST_CHARCONV_CONSTEXPR to_chars_result to_chars_int(char* first, char* last, Integer value, int base = 10) noexcept
433{
434 using Unsigned_Integer = typename std::make_unsigned<Integer>::type;
435 if (base == 10)
436 {
437 return to_chars_integer_impl(first, last, value);
438 }
439
440 return to_chars_integer_impl<Integer, Unsigned_Integer>(first, last, value, base);
441}
