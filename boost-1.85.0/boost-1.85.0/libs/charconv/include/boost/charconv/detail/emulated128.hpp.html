22// Compilers might support built-in 128-bit integer types. However, it seems that
23// emulating them with a pair of 64-bit integers actually produces a better code,
24// so we avoid using those built-ins. That said, they are still useful for
25// implementing 64-bit x 64-bit -> 128-bit multiplication.
26
27// Memcpy-able temp class for uint128
28struct trivial_uint128
29{
30 #if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
31 std::uint64_t low;
32 std::uint64_t high;
33 #else
34 std::uint64_t high;
35 std::uint64_t low;
36 #endif
37};

393 // Left shift
394 #define INTEGER_BINARY_OPERATOR_LEFT_SHIFT(expr) \
395 BOOST_CHARCONV_CXX14_CONSTEXPR friend uint128 operator<<(uint128 lhs, expr rhs) noexcept \
396 { \
397 if (rhs >= 64) \
398 { \
399 return {lhs.low << (rhs - 64), 0}; \
400 } \
401 else if (rhs == 0) \
402 { \
403 return lhs; \
404 } \
405 \
406 return {(lhs.high << rhs) | (lhs.low >> (64 - rhs)), lhs.low << rhs}; \
407 } // NOLINT

729BOOST_CHARCONV_CXX14_CONSTEXPR uint128 operator*(uint128 lhs, uint128 rhs) noexcept
730{
731 const auto a = static_cast<std::uint64_t>(lhs.low >> 32);
732 const auto b = static_cast<std::uint64_t>(lhs.low & UINT32_MAX);
733 const auto c = static_cast<std::uint64_t>(rhs.low >> 32);
734 const auto d = static_cast<std::uint64_t>(rhs.low & UINT32_MAX);

736 uint128 result { lhs.high * rhs.low + lhs.low * rhs.high + a * c, b * d };

762// See: https://stackoverflow.com/questions/5386377/division-without-using
763BOOST_CHARCONV_CXX14_CONSTEXPR void div_impl(uint128 lhs, uint128 rhs, uint128& quotient, uint128& remainder) noexcept
764{
765 constexpr uint128 one {0, 1};
766
767 if (rhs > lhs)
768 {
769 quotient = 0U;
770 remainder = 0U;
771 }
772 else if (lhs == rhs)
773 {
774 quotient = 1U;
775 remainder = 0U;
776 }
777
778 uint128 denom = rhs;
779 quotient = 0U;
780
781 std::int32_t shift = high_bit(v: lhs) - high_bit(v: rhs);
782 if (shift < 0)
783 {
784 shift = 32 - shift;
785 }
786 denom <<= shift;
787
788 for (int i = 0; i <= shift; ++i)
789 {
790 quotient <<= 1;
791 if (lhs >= denom)
792 {
793 lhs -= denom;
794 quotient |= one;
795 }
796 denom >>= 1;
797 }
798
799 remainder = lhs;
800}

803{
804 uint128 quotient {0, 0};
805 uint128 remainder {0, 0};
806 div_impl(lhs, rhs, quotient, remainder);
807
808 return quotient;
809}

846// Get 128-bit result of multiplication of two 64-bit unsigned integers.
847BOOST_CHARCONV_SAFEBUFFERS inline uint128 umul128(std::uint64_t x, std::uint64_t y) noexcept
848{
849 #if defined(BOOST_CHARCONV_HAS_INT128)
850
851 auto result = static_cast<boost::uint128_type>(x) * static_cast<boost::uint128_type>(y);
852 return {static_cast<std::uint64_t>(result >> 64), static_cast<std::uint64_t>(result)};
853
854 // _umul128 is x64 only https://learn.microsoft.com/en-us/cpp/intrinsics/umul128?view=msvc-170
855 #elif defined(BOOST_CHARCONV_HAS_MSVC_64BIT_INTRINSICS) && !defined(_M_ARM64)
856
857 unsigned long long high;
858 std::uint64_t low = _umul128(x, y, &high);
859 return {static_cast<std::uint64_t>(high), low};
860
861 // https://developer.arm.com/documentation/dui0802/a/A64-General-Instructions/UMULH
862 #elif defined(_M_ARM64) && !defined(__MINGW32__)
863
864 std::uint64_t high = __umulh(x, y);
865 std::uint64_t low = x * y;
866 return {high, low};
867
868 #else
869
870 auto a = static_cast<std::uint32_t>(x >> 32);
871 auto b = static_cast<std::uint32_t>(x);
872 auto c = static_cast<std::uint32_t>(y >> 32);
873 auto d = static_cast<std::uint32_t>(y);
874
875 auto ac = umul64(a, c);
876 auto bc = umul64(b, c);
877 auto ad = umul64(a, d);
878 auto bd = umul64(b, d);
879
880 auto intermediate = (bd >> 32) + static_cast<std::uint32_t>(ad) + static_cast<std::uint32_t>(bc);
881
882 return {ac + (intermediate >> 32) + (ad >> 32) + (bc >> 32),
883 (intermediate << 32) + static_cast<std::uint32_t>(bd)};
884
885 #endif
886}

889{
890 #if defined(BOOST_CHARCONV_HAS_INT128)
