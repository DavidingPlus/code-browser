5#ifndef BOOST_GENERATE_NAN_HPP
6#define BOOST_GENERATE_NAN_HPP
7
8#include <cstdint>
9#include <cstring>
10
11#ifdef BOOST_CHARCONV_HAS_FLOAT128
12
13namespace boost {
14namespace charconv {
15namespace detail {
16
17struct words
18{
19#if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
20 std::uint64_t lo;
21 std::uint64_t hi;
22#else
23 std::uint64_t hi;
24 std::uint64_t lo;
25#endif
26};
27
28inline __float128 nans BOOST_PREVENT_MACRO_SUBSTITUTION () noexcept
29{
30 words bits;
31 bits.hi = UINT64_C(0x7FFF400000000000);
32 bits.lo = UINT64_C(0);
33
34 __float128 return_val;
35 std::memcpy(&return_val, &bits, sizeof(__float128));
36 return return_val;
37}
38
39inline __float128 nanq BOOST_PREVENT_MACRO_SUBSTITUTION () noexcept
40{
41 words bits;
42 bits.hi = UINT64_C(0x7FFF800000000000);
43 bits.lo = UINT64_C(0);
44
45 __float128 return_val;
46 std::memcpy(&return_val, &bits, sizeof(__float128));
47 return return_val;
48}
49
50} //namespace detail
51} //namespace charconv
52} //namespace boost
53
54#endif
55
56#endif //BOOST_GENERATE_NAN_HPP
