60// A floating-point traits class defines ways to interpret a bit pattern of given size as an
61// encoding of floating-point number. This is a default implementation of such a traits class,
62// supporting ways to interpret 32-bits into a binary32-encoded floating-point number and to
63// interpret 64-bits into a binary64-encoded floating-point number. Users might specialize this
64// class to change the default behavior for certain types.
65template <typename T>
66struct default_float_traits
67{
68 // I don't know if there is a truly reliable way of detecting
69 // IEEE-754 binary32/binary64 formats; I just did my best here.
70 static_assert(std::numeric_limits<T>::is_iec559 && std::numeric_limits<T>::radix == 2 &&
71 (detail::physical_bits<T>::value == 32 || detail::physical_bits<T>::value == 64),
72 "default_ieee754_traits only works for 32-bits or 64-bits types "
73 "supporting binary32 or binary64 formats!");
74
75 // The type that is being viewed.
76 using type = T;
77
78 // Refers to the format specification class.
79 using format =
80 typename std::conditional<detail::physical_bits<T>::value == 32, detail::ieee754_binary32, detail::ieee754_binary64>::type;
81
82 // Defines an unsignedeger type that is large enough to carry a variable of type T.
83 // Most of the operations will be done on this integer type.
84 using carrier_uint = typename std::conditional<detail::physical_bits<T>::value == 32, std::uint32_t, std::uint64_t>::type;
85
86 static_assert(sizeof(carrier_uint) == sizeof(T), "carrier_uint must be T");
87
88 // Number of bits in the above unsignedeger type.
89 static constexpr int carrier_bits = static_cast<int>(detail::physical_bits<carrier_uint>::value);
90
91 // Convert from carrier_uint into the original type.
92 // Depending on the floating-point encoding format, this operation might not be possible for
93 // some specific bit patterns. However, the contract is that u always denotes a
94 // valid bit pattern, so this function must be assumed to be noexcept.
95 static T carrier_to_float(carrier_uint u) noexcept
96 {
97 T x;
98 std::memcpy(dest: &x, src: &u, n: sizeof(carrier_uint));
99 return x;
100 }
101
102 // Same as above.
103 static carrier_uint float_to_carrier(T x) noexcept
104 {
105 carrier_uint u;
106 std::memcpy(dest: &u, src: &x, n: sizeof(carrier_uint));
107 return u;
108 }
109
110 // Extract exponent bits from a bit pattern.
111 // The result must be aligned to the LSB so that there is no additional zero paddings
112 // on the right. This function does not do bias adjustment.
113 static constexpr unsigned extract_exponent_bits(carrier_uint u) noexcept
114 {
115 return static_cast<unsigned>(u >> format::exponent_bits) & static_cast<unsigned>((1U << format::exponent_bits) - 1);
116 }
117
118 // Extract significand bits from a bit pattern.
119 // The result must be aligned to the LSB so that there is no additional zero paddings
120 // on the right. The result does not contain the implicit bit.
121 static constexpr carrier_uint extract_significand_bits(carrier_uint u) noexcept
122 {
123 return static_cast<carrier_uint>(u & static_cast<carrier_uint>((static_cast<carrier_uint>(1) << format::significand_bits) - 1));
124 }
125
126 // Remove the exponent bits and extract significand bits together with the sign bit.
127 static constexpr carrier_uint remove_exponent_bits(carrier_uint u, unsigned exponent_bits) noexcept
128 {
129 return u ^ (static_cast<carrier_uint>(exponent_bits) << format::significand_bits);
130 }
131
132 // Shift the obtained signed significand bits to the left by 1 to remove the sign bit.
133 static constexpr carrier_uint remove_sign_bit_and_shift(carrier_uint u) noexcept
134 {
135 return static_cast<carrier_uint>(static_cast<carrier_uint>(u) << 1);
136 }
137
138 // The actual value of exponent is obtained by adding this value to the extracted exponent bits
139 static constexpr int exponent_bias = 1 - (1 << (carrier_bits - format::significand_bits - 2));
140
141 // Obtain the actual value of the binary exponent from the extracted exponent bits.
142 static constexpr int binary_exponent(unsigned exponent_bits) noexcept
143 {
144 return exponent_bits == 0 ? format::min_exponent : static_cast<int>(exponent_bits) + format::exponent_bias;
145 }
146
147 // Obtain the actual value of the binary exponent from the extracted significand bits and
148 // exponent bits.
149 static constexpr carrier_uint binary_significand(carrier_uint significand_bits, unsigned exponent_bits) noexcept
150 {
151 return exponent_bits == 0 ? significand_bits : (significand_bits | (static_cast<carrier_uint>(1) << format::significand_bits));
152 }
153
154
155 // Various boolean observer functions
156
157 static constexpr bool is_nonzero(carrier_uint u) noexcept { return (u << 1) != 0; }
158
159 static constexpr bool is_positive(carrier_uint u) noexcept
160 {
161 return u < static_cast<carrier_uint>(1) << (format::significand_bits + format::exponent_bits);
162 }
163
164 static constexpr bool is_negative(carrier_uint u) noexcept { return !is_positive(u); }
165
166 static constexpr bool is_finite(unsigned exponent_bits) noexcept
167 {
168 //constexpr unsigned exponent_bits_all_set = (1u << format::exponent_bits) - 1;
169 return exponent_bits != (1u << format::exponent_bits) - 1;
170 }
171
172 static constexpr bool has_all_zero_significand_bits(carrier_uint u) noexcept
173 {
174 return (u << 1) == 0;
175 }
176
177 static constexpr bool has_even_significand_bits(carrier_uint u) noexcept
178 {
179 return u % 2 == 0;
180 }
181};

183// Convenient wrappers for floating-point traits classes.
184// In order to reduce the argument passing overhead, these classes should be as simple as
185// possible (e.g., no inheritance, no private non-static data member, etc.; this is an
186// unfortunate fact about common ABI convention).
187
188template <typename T, typename Traits = default_float_traits<T>>
189struct float_bits;
190
191template <typename T, typename Traits = default_float_traits<T>>
192struct signed_significand_bits;
193
194template <typename T, typename Traits>
195struct float_bits
196{
197 using type = T;
198 using traits_type = Traits;
199 using carrier_uint = typename traits_type::carrier_uint;
200
201 carrier_uint u;
202
203 float_bits() = default;
204 constexpr explicit float_bits(carrier_uint bit_pattern) noexcept : u{bit_pattern} {}
205 constexpr explicit float_bits(T float_value) noexcept : u{traits_type::float_to_carrier(float_value)} {}
206
207 constexpr T to_float() const noexcept { return traits_type::carrier_to_float(u); }
208
209 // Extract exponent bits from a bit pattern.
210 // The result must be aligned to the LSB so that there is no additional zero paddings
211 // on the right. This function does not do bias adjustment.
212 constexpr unsigned extract_exponent_bits() const noexcept
213 {
214 return traits_type::extract_exponent_bits(u);
215 }
216
217 // Extract significand bits from a bit pattern.
218 // The result must be aligned to the LSB so that there is no additional zero paddings
219 // on the right. The result does not contain the implicit bit.
220 constexpr carrier_uint extract_significand_bits() const noexcept
221 {
222 return traits_type::extract_significand_bits(u);
223 }
224
225 // Remove the exponent bits and extract significand bits together with the sign bit.
226 constexpr signed_significand_bits<type, traits_type> remove_exponent_bits(unsigned exponent_bits) const noexcept
227 {
228 return signed_significand_bits<type, traits_type>(traits_type::remove_exponent_bits(u, exponent_bits));
229 }
230
231 // Obtain the actual value of the binary exponent from the extracted exponent bits.
232 static constexpr int binary_exponent(unsigned exponent_bits) noexcept
233 {
234 return traits_type::binary_exponent(exponent_bits);
235 }
236
237 constexpr int binary_exponent() const noexcept
238 {
239 return binary_exponent(extract_exponent_bits());
240 }
241
242 // Obtain the actual value of the binary exponent from the extracted significand bits and
243 // exponent bits.
244 static constexpr carrier_uint binary_significand(carrier_uint significand_bits, unsigned exponent_bits) noexcept
245 {
246 return traits_type::binary_significand(significand_bits, exponent_bits);
247 }
248
249 constexpr carrier_uint binary_significand() const noexcept
250 {
251 return binary_significand(extract_significand_bits(), extract_exponent_bits());
252 }
253
254 constexpr bool is_nonzero() const noexcept { return traits_type::is_nonzero(u); }
255
256 constexpr bool is_positive() const noexcept { return traits_type::is_positive(u); }
257
258 constexpr bool is_negative() const noexcept { return traits_type::is_negative(u); }
259
260 constexpr bool is_finite(unsigned exponent_bits) const noexcept { return traits_type::is_finite(exponent_bits); }
261
262 constexpr bool is_finite() const noexcept { return traits_type::is_finite(extract_exponent_bits()); }
263
264 constexpr bool has_even_significand_bits() const noexcept { return traits_type::has_even_significand_bits(u); }
265};

305template <class Int, class Int2>
306BOOST_CHARCONV_CXX14_CONSTEXPR Int compute_power(Int a, Int2 exp) noexcept
307{
308 BOOST_CHARCONV_ASSERT(exp >= 0);
309
310 Int res = 1;
311 while (exp > 0)
312 {
313 if (exp % 2 != 0)
314 {
315 res *= a;
316 }
317
318 a *= a;
319 exp >>= 1;
320 }
321 return res;
322}

356// Compute floor(e * c - s).
357enum class multiply : std::uint32_t {};
358enum class subtract : std::uint32_t {};
359enum class shift : std::size_t {};
360enum class min_exponent : std::int32_t {};
361enum class max_exponent : std::int32_t {};
362
363template <multiply m, subtract f, shift k, min_exponent e_min, max_exponent e_max>
364constexpr int compute(int e) noexcept
365{
366 return static_cast<int>((std::int32_t(e) * std::int32_t(m) - std::int32_t(f)) >> std::size_t(k));
367}

369// For constexpr computation.
370// Returns -1 when n = 0.
371template <class UInt>
372BOOST_CHARCONV_CXX14_CONSTEXPR int floor_log2(UInt n) noexcept
373{
374 int count = -1;
375 while (n != 0)
376 {
377 ++count;
378 n >>= 1;
379 }
380
381 return count;
382}
