13// Layouts of floating point types as specified by IEEE 754
14// See page 23 of IEEE 754-2008
15
16namespace boost { namespace charconv { namespace detail {
17
18struct ieee754_binary32
19{
20 static constexpr int significand_bits = 23;
21 static constexpr int exponent_bits = 8;
22 static constexpr int min_exponent = -126;
23 static constexpr int max_exponent = 127;
24 static constexpr int exponent_bias = -127;
25 static constexpr int decimal_digits = 9;
26};
27
28struct ieee754_binary64
29{
30 static constexpr int significand_bits = 52;
31 static constexpr int exponent_bits = 11;
32 static constexpr int min_exponent = -1022;
33 static constexpr int max_exponent = 1023;
34 static constexpr int exponent_bias = -1023;
35 static constexpr int decimal_digits = 17;
36};
37
38// 80 bit long double (e.g. x86-64)
39#if LDBL_MANT_DIG == 64 && LDBL_MAX_EXP == 16384
40
41struct IEEEl2bits
42{
43#if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
44 std::uint64_t mantissa_l : 64;
45 std::uint32_t exponent : 15;
46 std::uint32_t sign : 1;
47 std::uint64_t pad : 48;
48#else // Big endian
49 std::uint64_t pad : 48;
50 std::uint32_t sign : 1;
51 std::uint32_t exponent : 15;
52 std::uint64_t mantissa_h : 64;
53#endif
54};
55
56struct ieee754_binary80
57{
58 static constexpr int significand_bits = 64; // Fraction is 63 and 1 integer bit
59 static constexpr int exponent_bits = 15;
60 static constexpr int min_exponent = -16382;
61 static constexpr int max_exponent = 16383;
62 static constexpr int exponent_bias = -16383;
63 static constexpr int decimal_digits = 18;
64};
65
66#define BOOST_CHARCONV_LDBL_BITS 80
67
68// 128 bit long double (e.g. s390x, ppcle64)
69#elif LDBL_MANT_DIG == 113 && LDBL_MAX_EXP == 16384
70
71struct IEEEl2bits
72{
73#if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
74 std::uint64_t mantissa_l : 64;
75 std::uint64_t mantissa_h : 48;
76 std::uint32_t exponent : 15;
77 std::uint32_t sign : 1;
78#else // Big endian
79 std::uint32_t sign : 1;
80 std::uint32_t exponent : 15;
81 std::uint64_t mantissa_h : 48;
82 std::uint64_t mantissa_l : 64;
83#endif
84};
85
86#define BOOST_CHARCONV_LDBL_BITS 128
87
88// 64 bit long double (double == long double on ARM)
89#elif LDBL_MANT_DIG == 53 && LDBL_MAX_EXP == 1024
90
91struct IEEEl2bits
92{
93#if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
94 std::uint32_t mantissa_l : 32;
95 std::uint32_t mantissa_h : 20;
96 std::uint32_t exponent : 11;
97 std::uint32_t sign : 1;
98#else // Big endian
99 std::uint32_t sign : 1;
100 std::uint32_t exponent : 11;
101 std::uint32_t mantissa_h : 20;
102 std::uint32_t mantissa_l : 32;
103#endif
104};
105
106#define BOOST_CHARCONV_LDBL_BITS 64
107
108#else // Unsupported long double representation
109# define BOOST_MATH_UNSUPPORTED_LONG_DOUBLE
110#endif
111
112struct IEEEbinary128
113{
114#if BOOST_CHARCONV_ENDIAN_LITTLE_BYTE
115 std::uint64_t mantissa_l : 64;
116 std::uint64_t mantissa_h : 48;
117 std::uint32_t exponent : 15;
118 std::uint32_t sign : 1;
119#else // Big endian
120 std::uint32_t sign : 1;
121 std::uint32_t exponent : 15;
122 std::uint64_t mantissa_h : 48;
123 std::uint64_t mantissa_l : 64;
124#endif
125};
126
127struct ieee754_binary128
128{
129 static constexpr int significand_bits = 112;
130 static constexpr int exponent_bits = 15;
131 static constexpr int min_exponent = -16382;
132 static constexpr int max_exponent = 16383;
133 static constexpr int exponent_bias = 16383;
134 static constexpr int decimal_digits = 33;
135};
