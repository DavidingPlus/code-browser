33{
34 return std::snprintf(s: first, maxlen: size, format: format, value);

37template <typename T>
38to_chars_result to_chars_printf_impl(char* first, char* last, T value, chars_format fmt, int precision)
39{
40 // v % + . + num_digits(INT_MAX) + specifier + null terminator
41 // 1 + 1 + 10 + 1 + 1
42 char format[14] {};
43 std::memcpy(dest: format, src: "%", n: 1); // NOLINT : No null terminator is purposeful
44 std::size_t pos = 1;
45
46 // precision of -1 is unspecified
47 if (precision != -1 && fmt != chars_format::fixed)
48 {
49 format[pos] = '.';
50 ++pos;
51 const auto unsigned_precision = static_cast<std::uint32_t>(precision);
52 if (unsigned_precision < 10)
53 {
54 boost::charconv::detail::print_1_digit(n: unsigned_precision, buffer: format + pos);
55 ++pos;
56 }
57 else if (unsigned_precision < 100)
58 {
59 boost::charconv::detail::print_2_digits(n: unsigned_precision, buffer: format + pos);
60 pos += 2;
61 }
62 else
63 {
64 boost::charconv::detail::to_chars_int(first: format + pos, last: format + sizeof(format), value: precision);
65 pos = std::strlen(s: format);
66 }
67 }
68 else if (fmt == chars_format::fixed)
69 {
70 // Force 0 decimal places
71 std::memcpy(dest: format + pos, src: ".0", n: 2); // NOLINT : No null terminator is purposeful
72 pos += 2;
73 }
74
75 // Add the type identifier
76 #ifdef BOOST_CHARCONV_HAS_FLOAT128
77 BOOST_CHARCONV_IF_CONSTEXPR (std::is_same<T, __float128>::value || std::is_same<T, long double>::value)
78 {
79 format[pos] = std::is_same<T, __float128>::value ? 'Q' : 'L';
80 ++pos;
81 }
82 #else
83 BOOST_CHARCONV_IF_CONSTEXPR (std::is_same<T, long double>::value)
84 {
85 format[pos] = 'L';
86 ++pos;
87 }
88 #endif
89
90 // Add the format character
91 switch (fmt)
92 {
93 case boost::charconv::chars_format::general:
94 format[pos] = 'g';
95 break;
96
97 case boost::charconv::chars_format::scientific:
98 format[pos] = 'e';
99 break;
100
101 case boost::charconv::chars_format::fixed:
102 format[pos] = 'f';
103 break;
104
105 case boost::charconv::chars_format::hex:
106 format[pos] = 'a';
107 break;
108 }
109
110 const auto rv = print_val(first, static_cast<std::size_t>(last - first), format, value);
111
112 if (rv <= 0)
113 {
114 return {.ptr: last, .ec: static_cast<std::errc>(errno)};
115 }
116
117 return {first + rv, std::errc()};
118}

135// We know that the string is in the "C" locale because it would have previously passed through our parser.
136// Convert the string into the current locale so that the strto* family of functions
137// works correctly for the given locale.
138//
139// We are operating on our own copy of the buffer, so we are free to modify it.
140inline void convert_string_locale(char* buffer) noexcept
141{
142 const auto locale_decimal_point = *std::localeconv()->decimal_point;
143 if (locale_decimal_point != '.')
144 {
145 auto p = std::strchr(s: buffer, c: '.');
146 if (p != nullptr)
147 {
148 *p = locale_decimal_point;
149 }
150 }
151}

153template <typename T>
154from_chars_result from_chars_strtod_impl(const char* first, const char* last, T& value, char* buffer) noexcept
155{
156 // For strto(f/d)
157 // Floating point value corresponding to the contents of str on success.
158 // If the converted value falls out of range of corresponding return type, range error occurs and HUGE_VAL, HUGE_VALF or HUGE_VALL is returned.
159 // If no conversion can be performed, 0 is returned and *str_end is set to str.
160
161 std::memcpy(dest: buffer, src: first, n: static_cast<std::size_t>(last - first));
162 buffer[last - first] = '\0';
163 convert_string_locale(buffer);
164
165 char* str_end;
166 T return_value {};
167 from_chars_result r {.ptr: nullptr, .ec: std::errc()};
168
169 BOOST_IF_CONSTEXPR (std::is_same<T, float>::value)
170 {
171 return_value = std::strtof(nptr: buffer, endptr: &str_end);
172
173 #ifndef __INTEL_LLVM_COMPILER
174 if (return_value == HUGE_VALF)
175 #else
176 if (return_value >= std::numeric_limits<T>::max())
177 #endif
178 {
179 r = {.ptr: last, .ec: std::errc::result_out_of_range};
180 }
181 }
182 else BOOST_IF_CONSTEXPR (std::is_same<T, double>::value)
183 {
184 return_value = std::strtod(nptr: buffer, endptr: &str_end);
185
186 #ifndef __INTEL_LLVM_COMPILER
187 if (return_value == HUGE_VAL)
188 #else
189 if (return_value >= std::numeric_limits<T>::max())
190 #endif
191 {
192 r = {.ptr: last, .ec: std::errc::result_out_of_range};
193 }
194 }
195 else BOOST_IF_CONSTEXPR (std::is_same<T, long double>::value)
196 {
197 return_value = std::strtold(nptr: buffer, endptr: &str_end);
198
199 #ifndef __INTEL_LLVM_COMPILER
200 if (return_value == HUGE_VALL)
201 #else
202 if (return_value >= std::numeric_limits<T>::max())
203 #endif
204 {
205 r = {.ptr: last, .ec: std::errc::result_out_of_range};
206 }
207 }
208 #ifdef BOOST_CHARCONV_HAS_FLOAT128
209 else
210 {
211 return_value = strtoflt128(buffer, &str_end);
212
213 if (return_value == HUGE_VALQ)
214 {
215 r = {last, std::errc::result_out_of_range};
216 }
217 }
218 #endif
219
220 // Since this is a fallback routine we are safe to check for 0
221 if (return_value == 0 && str_end == last)
222 {
223 r = {.ptr: first, .ec: std::errc::result_out_of_range};
224 }
225
226 if (r)
227 {
228 value = return_value;
229 r = {.ptr: first + (str_end - buffer), .ec: std::errc()};
230 }
231
232 return r;
233}

235template <typename T>
236inline from_chars_result from_chars_strtod(const char* first, const char* last, T& value) noexcept
237{
238 if (last - first < 1024)
239 {
240 char buffer[1024];
241 return from_chars_strtod_impl(first, last, value, buffer);
242 }
243
244 // If the string to be parsed does not fit into the 1024 byte static buffer than we have to allocate a buffer.
245 // malloc is used here because it does not throw on allocation failure.
246
247 char* buffer = static_cast<char*>(std::malloc(size: static_cast<std::size_t>(last - first + 1)));
248 if (buffer == nullptr)
249 {
250 return {.ptr: first, .ec: std::errc::not_enough_memory};
251 }
252
253 auto r = from_chars_strtod_impl(first, last, value, buffer);
254 std::free(ptr: buffer);
255
256 return r;
257}
