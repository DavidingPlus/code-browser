385// Returns e == 0 ? 1 : ceil(log_2(5^e)); requires 0 <= e <= 32768.
386static BOOST_CHARCONV_CXX14_CONSTEXPR uint32_t pow5bits(const uint32_t e) noexcept
387{
388 BOOST_CHARCONV_ASSERT(e <= 1 << 15);
389 return static_cast<uint32_t>(((e * UINT64_C(163391164108059)) >> 46) + 1);
390}

392static BOOST_CHARCONV_CXX14_CONSTEXPR
393void mul_128_256_shift(
394 const uint64_t* const a, const uint64_t* const b,
395 const uint32_t shift, const uint32_t corr,
396 uint64_t* const result) noexcept
397{
398 BOOST_CHARCONV_ASSERT(shift > 0);
399 BOOST_CHARCONV_ASSERT(shift < 256);
400 const unsigned_128_type b00 = static_cast<unsigned_128_type>(a[0]) * b[0]; // 0
401 const unsigned_128_type b01 = static_cast<unsigned_128_type>(a[0]) * b[1]; // 64
402 const unsigned_128_type b02 = static_cast<unsigned_128_type>(a[0]) * b[2]; // 128
403 const unsigned_128_type b03 = static_cast<unsigned_128_type>(a[0]) * b[3]; // 196
404 const unsigned_128_type b10 = static_cast<unsigned_128_type>(a[1]) * b[0]; // 64
405 const unsigned_128_type b11 = static_cast<unsigned_128_type>(a[1]) * b[1]; // 128
406 const unsigned_128_type b12 = static_cast<unsigned_128_type>(a[1]) * b[2]; // 196
407 const unsigned_128_type b13 = static_cast<unsigned_128_type>(a[1]) * b[3]; // 256
408
409 const unsigned_128_type s0 = b00; // 0 x
410 const unsigned_128_type s1 = b01 + b10; // 64 x
411 const unsigned_128_type c1 = s1 < b01; // 196 x
412 const unsigned_128_type s2 = b02 + b11; // 128 x
413 const unsigned_128_type c2 = s2 < b02; // 256 x
414 const unsigned_128_type s3 = b03 + b12; // 196 x
415 const unsigned_128_type c3 = s3 < b03; // 324 x
416
417 const unsigned_128_type p0 = s0 + (s1 << 64); // 0
418 const unsigned_128_type d0 = p0 < b00; // 128
419 const unsigned_128_type q1 = s2 + (s1 >> 64) + (s3 << 64); // 128
420 const unsigned_128_type d1 = q1 < s2; // 256
421 const unsigned_128_type p1 = q1 + (c1 << 64) + d0; // 128
422 const unsigned_128_type d2 = p1 < q1; // 256
423 const unsigned_128_type p2 = b13 + (s3 >> 64) + c2 + (c3 << 64) + d1 + d2; // 256
424
425 if (shift < 128)
426 {
427 const unsigned_128_type r0 = corr + ((p0 >> shift) | (p1 << (128 - shift)));
428 const unsigned_128_type r1 = ((p1 >> shift) | (p2 << (128 - shift))) + (r0 < corr);
429 result[0] = static_cast<uint64_t>(r0);
430 result[1] = static_cast<uint64_t>(r0 >> 64);
431 result[2] = static_cast<uint64_t>(r1);
432 result[3] = static_cast<uint64_t>(r1 >> 64);
433 }
434 else if (shift == 128)
435 {
436 const unsigned_128_type r0 = corr + p1;
437 const unsigned_128_type r1 = p2 + (r0 < corr);
438 result[0] = static_cast<uint64_t>(r0);
439 result[1] = static_cast<uint64_t>(r0 >> 64);
440 result[2] = static_cast<uint64_t>(r1);
441 result[3] = static_cast<uint64_t>(r1 >> 64);
442 }
443 else
444 {
445 const unsigned_128_type r0 = corr + ((p1 >> (shift - 128)) | (p2 << (256 - shift)));
446 const unsigned_128_type r1 = (p2 >> (shift - 128)) + (r0 < corr);
447 result[0] = static_cast<uint64_t>(r0);
448 result[1] = static_cast<uint64_t>(r0 >> 64);
449 result[2] = static_cast<uint64_t>(r1);
450 result[3] = static_cast<uint64_t>(r1 >> 64);
451 }
452}

454// Computes 5^i in the form required by Ryu, and stores it in the given pointer.
455static BOOST_CXX14_CONSTEXPR void generic_computePow5(const uint32_t i, uint64_t* const result) noexcept
456{
457 const uint32_t base = i / BOOST_CHARCONV_POW5_TABLE_SIZE;
458 const uint32_t base2 = base * BOOST_CHARCONV_POW5_TABLE_SIZE;
459 const uint64_t* const mul = ryu_tables::GENERIC_POW5_SPLIT[base];
460 if (i == base2)
461 {
462 result[0] = mul[0];
463 result[1] = mul[1];
464 result[2] = mul[2];
465 result[3] = mul[3];
466 }
467 else
468 {
469 const uint32_t offset = i - base2;
470 const uint64_t* const m = ryu_tables::GENERIC_POW5_TABLE[offset];
471 const uint32_t delta = pow5bits(e: i) - pow5bits(e: base2);
472 const uint32_t corr = static_cast<uint32_t>((ryu_tables::POW5_ERRORS[i / 32] >> (2 * (i % 32))) & 3);
473 mul_128_256_shift(a: m, b: mul, shift: delta, corr, result);
474 }
475}

477// Computes 5^-i in the form required by Ryu, and stores it in the given pointer.
478static BOOST_CXX14_CONSTEXPR void generic_computeInvPow5(const uint32_t i, uint64_t* const result) noexcept
479{
480 const uint32_t base = (i + BOOST_CHARCONV_POW5_TABLE_SIZE - 1) / BOOST_CHARCONV_POW5_TABLE_SIZE;
481 const uint32_t base2 = base * BOOST_CHARCONV_POW5_TABLE_SIZE;
482 const uint64_t* const mul = ryu_tables::GENERIC_POW5_INV_SPLIT[base]; // 1/5^base2
483 if (i == base2)
484 {
485 result[0] = mul[0] + 1;
486 result[1] = mul[1];
487 result[2] = mul[2];
488 result[3] = mul[3];
489 }
490 else
491 {
492 const uint32_t offset = base2 - i;
493 const uint64_t* const m = ryu_tables::GENERIC_POW5_TABLE[offset]; // 5^offset
494 const uint32_t delta = pow5bits(e: base2) - pow5bits(e: i);
495 const uint32_t corr = static_cast<uint32_t>((ryu_tables::POW5_INV_ERRORS[i / 32] >> (2 * (i % 32))) & 3) + UINT32_C(1);
496 mul_128_256_shift(a: m, b: mul, shift: delta, corr, result);
497 }
498}

500static BOOST_CXX14_CONSTEXPR uint32_t pow5Factor(unsigned_128_type value) noexcept
501{
502 for (uint32_t count = 0; value > 0; ++count)
503 {
504 if (value % 5 != 0)
505 {
506 return count;
507 }
508 value /= 5;
509 }
510 return 0;
511}

526static BOOST_CHARCONV_CXX14_CONSTEXPR
527unsigned_128_type mulShift(const unsigned_128_type m, const uint64_t* const mul, const int32_t j) noexcept
528{
529 BOOST_CHARCONV_ASSERT(j > 128);
530 uint64_t a[2] {};
531 a[0] = static_cast<uint64_t>(m);
532 a[1] = static_cast<uint64_t>(m >> 64);
533 uint64_t result[4] {};
534 mul_128_256_shift(a, b: mul, shift: static_cast<uint32_t>(j), corr: 0, result);
535 return (static_cast<unsigned_128_type>(result[1]) << 64) | result[0];
536}
