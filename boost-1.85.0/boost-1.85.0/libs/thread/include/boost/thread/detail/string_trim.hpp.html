15inline std::string string_trim( std::string const& s )
16{
17 std::size_t i = s.find_first_not_of( s: " \t\r\n" );
18
19 if( i == std::string::npos ) return std::string();
20
21 std::size_t j = s.find_last_not_of( s: " \t\r\n" );
22
23 return s.substr( pos: i, n: j + 1 - i );
24}
