8#include <string>
9#include <climits>
10
11namespace boost

13namespace thread_detail

15
16inline bool string_to_unsigned( std::string const& s, unsigned& v )
17{
18 v = 0;
19
20 if( s.empty() )
21 {
22 return false;
23 }
24
25 for( char const* p = s.c_str(); *p; ++p )
26 {
27 unsigned char ch = static_cast<unsigned char>( *p );
28
29 if( ch < '0' || ch > '9' )
30 {
31 return false;
32 }
33
34 if( v > UINT_MAX / 10 )
35 {
36 return false;
37 }
38
39 unsigned q = static_cast<unsigned>( ch - '0' );
40
41 if( v == UINT_MAX / 10 && q > UINT_MAX % 10 )
42 {
43 return false;
44 }
45
46 v = v * 10 + q;
47 }
48
49 return true;
50}
