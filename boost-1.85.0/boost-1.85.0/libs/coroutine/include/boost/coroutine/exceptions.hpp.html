72class coroutine_error : public std::logic_error
73{
74private:
75 system::error_code ec_;
76
77public:
78 coroutine_error( system::error_code ec) :
79 logic_error( ec.message() ),
80 ec_( ec)
81 {}
82
83 system::error_code const& code() const BOOST_NOEXCEPT
84 { return ec_; }
85};
