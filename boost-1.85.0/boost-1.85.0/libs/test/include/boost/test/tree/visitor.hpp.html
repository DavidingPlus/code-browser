34class BOOST_TEST_DECL test_tree_visitor {
35public:
36 // test tree visitor interface
37 virtual bool visit( test_unit const& ) { return true; }
38 virtual void visit( test_case const& tc ) { visit( (test_unit const&)tc ); }
39 virtual bool test_suite_start( test_suite const& ts ){ return visit( (test_unit const&)ts ); }
40 virtual void test_suite_finish( test_suite const& ) {}
41
42protected:
43 BOOST_TEST_PROTECTED_VIRTUAL ~test_tree_visitor() {}
44};
