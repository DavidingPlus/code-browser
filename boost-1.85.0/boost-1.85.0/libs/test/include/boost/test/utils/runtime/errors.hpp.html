40class BOOST_SYMBOL_VISIBLE param_error : public std::exception {
41public:
42 BOOST_TEST_DEFAULTED_FUNCTION(~param_error() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE, {})
43
44 const char * what() const BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE
45 {
46 return msg.c_str();
47 }
48
49 cstring param_name;
50 std::string msg;
51
52protected:
53 explicit param_error( cstring param_name_ ) : param_name( param_name_) {}
54};
55

58class BOOST_SYMBOL_VISIBLE init_error : public param_error {
59protected:
60 explicit init_error( cstring param_name ) : param_error( param_name ) {}
61 BOOST_TEST_DEFAULTED_FUNCTION(~init_error() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE, {})
62};

64class BOOST_SYMBOL_VISIBLE input_error : public param_error {
65protected:
66 explicit input_error( cstring param_name ) : param_error( param_name ) {}

70//____________________________________________________________________________//
71
72template<typename Derived, typename Base>
73class BOOST_SYMBOL_VISIBLE specific_param_error : public Base {
74protected:
75 explicit specific_param_error( cstring param_name ) : Base( param_name ) {}
76 BOOST_TEST_DEFAULTED_FUNCTION(~specific_param_error() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE, {})
77
78public:
79
80//____________________________________________________________________________//
81
82#if !defined(BOOST_NO_CXX11_RVALUE_REFERENCES) && \
83 !defined(BOOST_NO_CXX11_REF_QUALIFIERS)
84
85 Derived operator<<(char const* val) &&
86 {
87 this->msg.append( val );
88
89 return static_cast<Derived&&>(*this);
90 }
91
92 //____________________________________________________________________________//
93
94 template<typename T>
95 Derived operator<<(T const& val) &&
96 {
97 this->msg.append( unit_test::utils::string_cast( val ) );
98
99 return static_cast<Derived&&>(*this);
100 }
101
102 //____________________________________________________________________________//
103
104#else
105
106 Derived const& operator<<(char const* val) const
107 {
108 const_cast<specific_param_error<Derived, Base>&>(*this).msg.append( val );
109
110 return static_cast<Derived const&>(*this);
111 }
112
113 //____________________________________________________________________________//
114
115 template<typename T>
116 Derived const& operator<<(T const& val) const
117 {
118 const_cast<specific_param_error<Derived, Base>&>(*this).msg.append( unit_test::utils::string_cast( val ) );
119
120 return static_cast<Derived const&>(*this);
121 }
122
123 //____________________________________________________________________________//
124
125#endif
126
127};

135#define SPECIFIC_EX_TYPE( type, base ) \
136class BOOST_SYMBOL_VISIBLE type : public specific_param_error<type,base> { \
137public: \
138 explicit type( cstring param_name = cstring() ) \
139 : specific_param_error<type,base>( param_name ) \
140 {} \
141} \
142/**/
143
144SPECIFIC_EX_TYPE( invalid_cla_id, init_error );
145SPECIFIC_EX_TYPE( duplicate_param, init_error );
146SPECIFIC_EX_TYPE( conflicting_param, init_error );
147SPECIFIC_EX_TYPE( unknown_param, init_error );
148SPECIFIC_EX_TYPE( access_to_missing_argument, init_error );
149SPECIFIC_EX_TYPE( arg_type_mismatch, init_error );
150SPECIFIC_EX_TYPE( invalid_param_spec, init_error );
151
152SPECIFIC_EX_TYPE( format_error, input_error );
153SPECIFIC_EX_TYPE( duplicate_arg, input_error );
154SPECIFIC_EX_TYPE( missing_req_arg, input_error );
155
156#undef SPECIFIC_EX_TYPE

158class BOOST_SYMBOL_VISIBLE ambiguous_param : public specific_param_error<ambiguous_param, input_error> {
159public:
160#ifndef BOOST_NO_CXX11_RVALUE_REFERENCES
161 explicit ambiguous_param( std::vector<cstring>&& amb_candidates )
162 : specific_param_error<ambiguous_param,input_error>( "" )
163 , m_amb_candidates( std::move( amb_candidates ) ) {}
164#else
165 explicit ambiguous_param( std::vector<cstring> const& amb_candidates )
166 : specific_param_error<ambiguous_param,input_error>( "" )
167 , m_amb_candidates( amb_candidates ) {}
168#endif
169 BOOST_TEST_DEFAULTED_FUNCTION(~ambiguous_param() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE, {})
170
171 std::vector<cstring> m_amb_candidates;
172};
