255template <typename ErrorInfo>
256typename ErrorInfo::value_type
257extract( boost::exception const* ex )
258{
259 if( !ex )
260 return 0;
261
262 typename ErrorInfo::value_type const * val = boost::get_error_info<ErrorInfo>( *ex );
263
264 return val ? *val : 0;
265}

269static void
270BOOST_TEST_PRINTF_ATTRIBUTE_CHECK(3, 0)
271report_error( execution_exception::error_code ec, boost::exception const* be, char const* format, va_list* args )
272{
273 static const int REPORT_ERROR_BUFFER_SIZE = 4096;
274 static char buf[REPORT_ERROR_BUFFER_SIZE];
275
276 BOOST_TEST_VSNPRINTF( buf, sizeof(buf)-1, format, *args );
277 buf[sizeof(buf)-1] = 0;
278
279 va_end( *args );
280
281 BOOST_TEST_I_THROW(execution_exception( ec, buf, execution_exception::location( extract<throw_file>( be ),
282 (size_t)extract<throw_line>( be ),
283 extract<throw_function>( be ) ) ));
284}

323struct fpe_except_guard {
324 explicit fpe_except_guard( unsigned detect_fpe )
325 : m_detect_fpe( detect_fpe )
326 {
327 // prepare fp exceptions control
328 m_previously_enabled = fpe::disable( mask: fpe::BOOST_FPE_ALL );
329 if( m_previously_enabled != fpe::BOOST_FPE_INV && detect_fpe != fpe::BOOST_FPE_OFF )
330 fpe::enable( mask: detect_fpe );
331 }
332 ~fpe_except_guard()
333 {
334 if( m_detect_fpe != fpe::BOOST_FPE_OFF )
335 fpe::disable( mask: m_detect_fpe );
336 if( m_previously_enabled != fpe::BOOST_FPE_INV )
337 fpe::enable( mask: m_previously_enabled );
338 }
339
340 unsigned m_detect_fpe;
341 unsigned m_previously_enabled;
342};

372class system_signal_exception {
373public:
374 // Constructor
375 system_signal_exception()
376 : m_sig_info( 0 )
377 , m_context( 0 )
378 {}
379
380 // Access methods
381 void operator()( siginfo_t* i, void* c )
382 {
383 m_sig_info = i;
384 m_context = c;
385 }
386 void report() const;
387
388private:
389 // Data members
390 siginfo_t* m_sig_info; // system signal detailed info
391 void* m_context; // signal context
392};
393
394//____________________________________________________________________________//
395
396void
397system_signal_exception::report() const
398{
399 if( !m_sig_info )
400 return; // no error actually occur?
401
402 switch( m_sig_info->si_code ) {
403#ifdef __VXWORKS__
404// a bit of a hack to adapt code to small m_sig_info VxWorks uses
405#define si_addr si_value.sival_int
406#define si_band si_value.sival_int
407#else
408 case SI_USER:
409 report_error( ec: execution_exception::system_error,
410 format: "signal: generated by kill() (or family); uid=%d; pid=%d",
411 (int)m_sig_info->si_uid, (int)m_sig_info->si_pid );
412 break;
413#endif
414 case SI_QUEUE:
415 report_error( ec: execution_exception::system_error,
416 format: "signal: sent by sigqueue()" );
417 break;
418 case SI_TIMER:
419 report_error( ec: execution_exception::system_error,
420 format: "signal: the expiration of a timer set by timer_settimer()" );
421 break;
422// OpenBSD was missing SI_ASYNCIO and SI_MESGQ
423#ifdef SI_ASYNCIO
424 case SI_ASYNCIO:
425 report_error( ec: execution_exception::system_error,
426 format: "signal: generated by the completion of an asynchronous I/O request" );
427 break;
428#endif
429#ifdef SI_MESGQ
430 case SI_MESGQ:
431 report_error( ec: execution_exception::system_error,
432 format: "signal: generated by the the arrival of a message on an empty message queue" );
433 break;
434#endif
435 default:
436 break;
437 }
438
439 switch( m_sig_info->si_signo ) {
440 case SIGILL:
441 switch( m_sig_info->si_code ) {
442#ifndef BOOST_TEST_LIMITED_SIGNAL_DETAILS
443 case ILL_ILLOPC:
444 report_error( ec: execution_exception::system_fatal_error,
445 format: "signal: illegal opcode; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
446 (uintptr_t) m_sig_info->si_addr );
447 break;
448 case ILL_ILLTRP:
449 report_error( ec: execution_exception::system_fatal_error,
450 format: "signal: illegal trap; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
451 (uintptr_t) m_sig_info->si_addr );
452 break;
453 case ILL_PRVREG:
454 report_error( ec: execution_exception::system_fatal_error,
455 format: "signal: privileged register; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
456 (uintptr_t) m_sig_info->si_addr );
457 break;
458 case ILL_BADSTK:
459 report_error( ec: execution_exception::system_fatal_error,
460 format: "signal: internal stack error; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
461 (uintptr_t) m_sig_info->si_addr );
462 break;
463#endif
464 case ILL_ILLOPN:
465 report_error( ec: execution_exception::system_fatal_error,
466 format: "signal: illegal operand; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
467 (uintptr_t) m_sig_info->si_addr );
468 break;
469 case ILL_ILLADR:
470 report_error( ec: execution_exception::system_fatal_error,
471 format: "signal: illegal addressing mode; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
472 (uintptr_t) m_sig_info->si_addr );
473 break;
474 case ILL_PRVOPC:
475 report_error( ec: execution_exception::system_fatal_error,
476 format: "signal: privileged opcode; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
477 (uintptr_t) m_sig_info->si_addr );
478 break;
479 case ILL_COPROC:
480 report_error( ec: execution_exception::system_fatal_error,
481 format: "signal: co-processor error; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
482 (uintptr_t) m_sig_info->si_addr );
483 break;
484 default:
485 report_error( ec: execution_exception::system_fatal_error,
486 format: "signal: SIGILL, si_code: %d (illegal instruction; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR ")",
487 m_sig_info->si_code, (uintptr_t) m_sig_info->si_addr );
488 break;
489 }
490 break;
491
492 case SIGFPE:
493 switch( m_sig_info->si_code ) {
494 case FPE_INTDIV:
495 report_error( ec: execution_exception::system_error,
496 format: "signal: integer divide by zero; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
497 (uintptr_t) m_sig_info->si_addr );
498 break;
499 case FPE_INTOVF:
500 report_error( ec: execution_exception::system_error,
501 format: "signal: integer overflow; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
502 (uintptr_t) m_sig_info->si_addr );
503 break;
504 case FPE_FLTDIV:
505 report_error( ec: execution_exception::system_error,
506 format: "signal: floating point divide by zero; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
507 (uintptr_t) m_sig_info->si_addr );
508 break;
509 case FPE_FLTOVF:
510 report_error( ec: execution_exception::system_error,
511 format: "signal: floating point overflow; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
512 (uintptr_t) m_sig_info->si_addr );
513 break;
514 case FPE_FLTUND:
515 report_error( ec: execution_exception::system_error,
516 format: "signal: floating point underflow; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
517 (uintptr_t) m_sig_info->si_addr );
518 break;
519 case FPE_FLTRES:
520 report_error( ec: execution_exception::system_error,
521 format: "signal: floating point inexact result; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
522 (uintptr_t) m_sig_info->si_addr );
523 break;
524 case FPE_FLTINV:
525 report_error( ec: execution_exception::system_error,
526 format: "signal: invalid floating point operation; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
527 (uintptr_t) m_sig_info->si_addr );
528 break;
529 case FPE_FLTSUB:
530 report_error( ec: execution_exception::system_error,
531 format: "signal: subscript out of range; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR,
532 (uintptr_t) m_sig_info->si_addr );
533 break;
534 default:
535 report_error( ec: execution_exception::system_error,
536 format: "signal: SIGFPE, si_code: %d (errnoneous arithmetic operations; address of failing instruction: 0x%" BOOST_TEST_PRIxPTR ")",
537 m_sig_info->si_code, (uintptr_t) m_sig_info->si_addr );
538 break;
539 }
540 break;
541
542 case SIGSEGV:
543 switch( m_sig_info->si_code ) {
544#ifndef BOOST_TEST_LIMITED_SIGNAL_DETAILS
545 case SEGV_MAPERR:
546 report_error( ec: execution_exception::system_fatal_error,
547 format: "memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ": no mapping at fault address",
548 (uintptr_t) m_sig_info->si_addr );
549 break;
550 case SEGV_ACCERR:
551 report_error( ec: execution_exception::system_fatal_error,
552 format: "memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ": invalid permissions",
553 (uintptr_t) m_sig_info->si_addr );
554 break;
555#endif
556 default:
557 report_error( ec: execution_exception::system_fatal_error,
558 format: "signal: SIGSEGV, si_code: %d (memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ")",
559 m_sig_info->si_code, (uintptr_t) m_sig_info->si_addr );
560 break;
561 }
562 break;
563
564 case SIGBUS:
565 switch( m_sig_info->si_code ) {
566#ifndef BOOST_TEST_LIMITED_SIGNAL_DETAILS
567 case BUS_ADRALN:
568 report_error( ec: execution_exception::system_fatal_error,
569 format: "memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ": invalid address alignment",
570 (uintptr_t) m_sig_info->si_addr );
571 break;
572 case BUS_ADRERR:
573 report_error( ec: execution_exception::system_fatal_error,
574 format: "memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ": non-existent physical address",
575 (uintptr_t) m_sig_info->si_addr );
576 break;
577 case BUS_OBJERR:
578 report_error( ec: execution_exception::system_fatal_error,
579 format: "memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ": object specific hardware error",
580 (uintptr_t) m_sig_info->si_addr );
581 break;
582#endif
583 default:
584 report_error( ec: execution_exception::system_fatal_error,
585 format: "signal: SIGSEGV, si_code: %d (memory access violation at address: 0x%" BOOST_TEST_PRIxPTR ")",
586 m_sig_info->si_code, (uintptr_t) m_sig_info->si_addr );
587 break;
588 }
589 break;
590
591#if defined(BOOST_TEST_CATCH_SIGPOLL)
592
593 case SIGPOLL:
594 switch( m_sig_info->si_code ) {
595#ifndef BOOST_TEST_LIMITED_SIGNAL_DETAILS
596 case POLL_IN:
597 report_error( ec: execution_exception::system_error,
598 format: "data input available; band event %d",
599 (int)m_sig_info->si_band );
600 break;
601 case POLL_OUT:
602 report_error( ec: execution_exception::system_error,
603 format: "output buffers available; band event %d",
604 (int)m_sig_info->si_band );
605 break;
606 case POLL_MSG:
607 report_error( ec: execution_exception::system_error,
608 format: "input message available; band event %d",
609 (int)m_sig_info->si_band );
610 break;
611 case POLL_ERR:
612 report_error( ec: execution_exception::system_error,
613 format: "i/o error; band event %d",
614 (int)m_sig_info->si_band );
615 break;
616 case POLL_PRI:
617 report_error( ec: execution_exception::system_error,
618 format: "high priority input available; band event %d",
619 (int)m_sig_info->si_band );
620 break;
621#if defined(POLL_ERR) && defined(POLL_HUP) && (POLL_ERR - POLL_HUP)
622 case POLL_HUP:
623 report_error( execution_exception::system_error,
624 "device disconnected; band event %d",
625 (int)m_sig_info->si_band );
626 break;
627#endif
628#endif
629 default:
630 report_error( ec: execution_exception::system_error,
631 format: "signal: SIGPOLL, si_code: %d (asynchronous I/O event occurred; band event %d)",
632 m_sig_info->si_code, (int)m_sig_info->si_band );
633 break;
634 }
635 break;
636
637#endif
638
639 case SIGABRT:
640 report_error( ec: execution_exception::system_error,
641 format: "signal: SIGABRT (application abort requested)" );
642 break;
643
644 case SIGALRM:
645 report_error( ec: execution_exception::timeout_error,
646 format: "signal: SIGALRM (timeout while executing function)" );
647 break;
648
649 default:
650 report_error( ec: execution_exception::system_error,
651 format: "unrecognized signal %d", m_sig_info->si_signo );
652 }
653}

689//____________________________________________________________________________//
690
691signal_action::signal_action( int sig, bool install, bool attach_dbg, char* alt_stack )
692: m_sig( sig )
693, m_installed( install )
694{
695 if( !install )
696 return;
697
698 std::memset( s: &m_new_action, c: 0, n: sizeof(struct sigaction) );
699
700 BOOST_TEST_SYS_ASSERT( ::sigaction( m_sig , sigaction_ptr(), &m_new_action ) != -1 );
701
702 if( m_new_action.sa_sigaction || m_new_action.sa_handler ) {
703 m_installed = false;
704 return;
705 }
706
707 m_new_action.sa_flags |= SA_SIGINFO;
708 m_new_action.sa_sigaction = attach_dbg ? &boost_execution_monitor_attaching_signal_handler
709 : &boost_execution_monitor_jumping_signal_handler;
710 BOOST_TEST_SYS_ASSERT( sigemptyset( &m_new_action.sa_mask ) != -1 );
711
712#ifdef BOOST_TEST_USE_ALT_STACK
713 if( alt_stack )
714 m_new_action.sa_flags |= SA_ONSTACK;
715#endif
716
717 BOOST_TEST_SYS_ASSERT( ::sigaction( m_sig, &m_new_action, &m_old_action ) != -1 );
718}

721
722signal_action::~signal_action()
723{
724 if( m_installed )
725 ::sigaction( sig: m_sig, act: &m_old_action , oact: sigaction_ptr() );
726}

788signal_handler::signal_handler( bool catch_system_errors,
789 bool detect_fpe,
790 unsigned long int timeout_microseconds,
791 bool attach_dbg,
792 char* alt_stack )
793: m_prev_handler( s_active_handler )
794, m_timeout_microseconds( timeout_microseconds )
795, m_ILL_action ( SIGILL , catch_system_errors, attach_dbg, alt_stack )
796, m_FPE_action ( SIGFPE , detect_fpe , attach_dbg, alt_stack )
797, m_SEGV_action( SIGSEGV, catch_system_errors, attach_dbg, alt_stack )

801#endif
802, m_ABRT_action( SIGABRT, catch_system_errors, attach_dbg, alt_stack )
803, m_ALRM_action( SIGALRM, timeout_microseconds > 0, attach_dbg, alt_stack )
804{
805 s_active_handler = this;
806
807 if( m_timeout_microseconds > 0 ) {
808 ::alarm( seconds: 0 );
809 ::alarm( seconds: static_cast<unsigned int>(std::ceil(x: timeout_microseconds / 1E6) )); // alarm has a precision to the seconds
810 }
811
812#ifdef BOOST_TEST_USE_ALT_STACK
813 if( alt_stack ) {
814 stack_t sigstk;
815 std::memset( s: &sigstk, c: 0, n: sizeof(stack_t) );
816
817 BOOST_TEST_SYS_ASSERT( ::sigaltstack( 0, &sigstk ) != -1 );
818
819 if( sigstk.ss_flags & SS_DISABLE ) {
820 sigstk.ss_sp = alt_stack;
821 sigstk.ss_size = BOOST_TEST_ALT_STACK_SIZE;
822 sigstk.ss_flags = 0;
823 BOOST_TEST_SYS_ASSERT( ::sigaltstack( &sigstk, 0 ) != -1 );
824 }
825 }
826#endif
827}

838#ifdef BOOST_TEST_USE_ALT_STACK
839#ifdef __GNUC__

841 // but gcc warns if we don't, so add initializers for each of the

986//____________________________________________________________________________//
987
988int
989system_signal_exception::operator()( unsigned id, _EXCEPTION_POINTERS* exps )
990{
991 const unsigned MSFT_CPP_EXCEPT = 0xE06d7363; // EMSC
992
993 // C++ exception - allow to go through
994 if( id == MSFT_CPP_EXCEPT )
995 return EXCEPTION_CONTINUE_SEARCH;
996
997 // FPE detection is enabled, while system exception detection is not - check if this is actually FPE
998 if( !m_em->p_catch_system_errors ) {
999 if( !m_em->p_detect_fp_exceptions )
1000 return EXCEPTION_CONTINUE_SEARCH;
1001
1002 switch( id ) {
1003 case EXCEPTION_FLT_DIVIDE_BY_ZERO:
1004 case EXCEPTION_FLT_STACK_CHECK:
1005 case EXCEPTION_FLT_DENORMAL_OPERAND:
1006 case EXCEPTION_FLT_INEXACT_RESULT:
1007 case EXCEPTION_FLT_OVERFLOW:
1008 case EXCEPTION_FLT_UNDERFLOW:
1009 case EXCEPTION_FLT_INVALID_OPERATION:
1010 case STATUS_FLOAT_MULTIPLE_FAULTS:
1011 case STATUS_FLOAT_MULTIPLE_TRAPS:
1012 break;
1013 default:
1014 return EXCEPTION_CONTINUE_SEARCH;
1015 }
1016 }
1017
1018 if( !!m_em->p_auto_start_dbg && debug::attach_debugger( false ) ) {
1019 m_em->p_catch_system_errors.value = false;
1020#if BOOST_WORKAROUND( BOOST_MSVC, <= 1310)
1021 _set_se_translator( &seh_catch_preventer );
1022#endif
1023 return EXCEPTION_CONTINUE_EXECUTION;
1024 }
1025
1026 m_se_id = id;
1027 if( m_se_id == EXCEPTION_ACCESS_VIOLATION && exps->ExceptionRecord->NumberParameters == 2 ) {
1028 m_fault_address = (void*)exps->ExceptionRecord->ExceptionInformation[1];
1029 m_dir = exps->ExceptionRecord->ExceptionInformation[0] == 0;
1030 }
1031
1032 return EXCEPTION_EXECUTE_HANDLER;
1033}

1035//____________________________________________________________________________//
1036
1037void
1038system_signal_exception::report() const
1039{
1040 switch( m_se_id ) {
1041 // cases classified as system_fatal_error
1042 case EXCEPTION_ACCESS_VIOLATION: {
1043 if( !m_fault_address )
1044 detail::report_error( execution_exception::system_fatal_error, "memory access violation" );
1045 else
1046 detail::report_error(
1047 execution_exception::system_fatal_error,
1048 "memory access violation occurred at address 0x%" BOOST_TEST_PRIxPTR ", while attempting to %s",
1049 m_fault_address,
1050 m_dir ? " read inaccessible data"
1051 : " write to an inaccessible (or protected) address"
1052 );
1053 break;
1054 }
1055
1056 case EXCEPTION_ILLEGAL_INSTRUCTION:
1057 detail::report_error( execution_exception::system_fatal_error, "illegal instruction" );
1058 break;
1059
1060 case EXCEPTION_PRIV_INSTRUCTION:
1061 detail::report_error( execution_exception::system_fatal_error, "tried to execute an instruction whose operation is not allowed in the current machine mode" );
1062 break;
1063
1064 case EXCEPTION_IN_PAGE_ERROR:
1065 detail::report_error( execution_exception::system_fatal_error, "access to a memory page that is not present" );
1066 break;
1067
1068 case EXCEPTION_STACK_OVERFLOW:
1069 detail::report_error( execution_exception::system_fatal_error, "stack overflow" );
1070 break;
1071
1072 case EXCEPTION_NONCONTINUABLE_EXCEPTION:
1073 detail::report_error( execution_exception::system_fatal_error, "tried to continue execution after a non continuable exception occurred" );
1074 break;
1075
1076 // cases classified as (non-fatal) system_trap
1077 case EXCEPTION_DATATYPE_MISALIGNMENT:
1078 detail::report_error( execution_exception::system_error, "data misalignment" );
1079 break;
1080
1081 case EXCEPTION_INT_DIVIDE_BY_ZERO:
1082 detail::report_error( execution_exception::system_error, "integer divide by zero" );
1083 break;
1084
1085 case EXCEPTION_INT_OVERFLOW:
1086 detail::report_error( execution_exception::system_error, "integer overflow" );
1087 break;
1088
1089 case EXCEPTION_ARRAY_BOUNDS_EXCEEDED:
1090 detail::report_error( execution_exception::system_error, "array bounds exceeded" );
1091 break;
1092
1093 case EXCEPTION_FLT_DIVIDE_BY_ZERO:
1094 detail::report_error( execution_exception::system_error, "floating point divide by zero" );
1095 break;
1096
1097 case EXCEPTION_FLT_STACK_CHECK:
1098 detail::report_error( execution_exception::system_error,
1099 "stack overflowed or underflowed as the result of a floating-point operation" );
1100 break;
1101
1102 case EXCEPTION_FLT_DENORMAL_OPERAND:
1103 detail::report_error( execution_exception::system_error,
1104 "operand of floating point operation is denormal" );
1105 break;
1106
1107 case EXCEPTION_FLT_INEXACT_RESULT:
1108 detail::report_error( execution_exception::system_error,
1109 "result of a floating-point operation cannot be represented exactly" );
1110 break;
1111
1112 case EXCEPTION_FLT_OVERFLOW:
1113 detail::report_error( execution_exception::system_error,
1114 "exponent of a floating-point operation is greater than the magnitude allowed by the corresponding type" );
1115 break;
1116
1117 case EXCEPTION_FLT_UNDERFLOW:
1118 detail::report_error( execution_exception::system_error,
1119 "exponent of a floating-point operation is less than the magnitude allowed by the corresponding type" );
1120 break;
1121
1122 case EXCEPTION_FLT_INVALID_OPERATION:
1123 detail::report_error( execution_exception::system_error, "floating point error" );
1124 break;
1125
1126 case STATUS_FLOAT_MULTIPLE_FAULTS:
1127 detail::report_error( execution_exception::system_error, "multiple floating point errors" );
1128 break;
1129
1130 case STATUS_FLOAT_MULTIPLE_TRAPS:
1131 detail::report_error( execution_exception::system_error, "multiple floating point errors" );
1132 break;
1133
1134 case EXCEPTION_BREAKPOINT:
1135 detail::report_error( execution_exception::system_error, "breakpoint encountered" );
1136 break;
1137
1138 default:
1139 if( m_timeout ) {
1140 detail::report_error(execution_exception::timeout_error, "timeout while executing function");
1141 }
1142 else {
1143 detail::report_error( execution_exception::system_error, "unrecognized exception. Id: 0x%" BOOST_TEST_PRIxPTR, m_se_id );
1144 }
1145 break;
1146 }
1147}

1182// ************************************************************************** //
1183// ************** execution_monitor::catch_signals ************** //
1184// ************************************************************************** //
1185
1186int
1187execution_monitor::catch_signals( boost::function<int ()> const& F )
1188{
1189 _invalid_parameter_handler old_iph = _invalid_parameter_handler();
1190 BOOST_TEST_CRT_HOOK_TYPE old_crt_hook = 0;
1191
1192 if( p_catch_system_errors ) {
1193 old_crt_hook = BOOST_TEST_CRT_SET_HOOK( &detail::assert_reporting_function );
1194
1195 old_iph = _set_invalid_parameter_handler(
1196 reinterpret_cast<_invalid_parameter_handler>( &detail::invalid_param_handler ) );
1197 } else if( !p_detect_fp_exceptions ) {
1198#if BOOST_WORKAROUND( BOOST_MSVC, <= 1310)
1199 _set_se_translator( &detail::seh_catch_preventer );
1200#endif
1201 }
1202
1203#if defined(BOOST_TEST_WIN32_WAITABLE_TIMERS)
1204 HANDLE htimer = INVALID_HANDLE_VALUE;
1205 BOOL bTimerSuccess = FALSE;
1206
1207 if( p_timeout ) {
1208 htimer = ::CreateWaitableTimer(
1209 NULL,
1210 TRUE,
1211 NULL); // naming the timer might create collisions
1212
1213 if( htimer != INVALID_HANDLE_VALUE ) {
1214 LARGE_INTEGER liDueTime;
1215 liDueTime.QuadPart = - static_cast<LONGLONG>(p_timeout) * 10ll; // resolution of 100 ns
1216
1217 bTimerSuccess = ::SetWaitableTimer(
1218 htimer,
1219 &liDueTime,
1220 0,
1221 0,
1222 0,
1223 FALSE); // Do not restore a suspended system
1224 }
1225 }
1226#endif
1227
1228 detail::system_signal_exception SSE( this );
1229
1230 int ret_val = 0;
1231 // clang windows workaround: this not available in __finally scope
1232 bool l_catch_system_errors = p_catch_system_errors;
1233
1234 __try {
1235 __try {
1236 ret_val = detail::do_invoke( m_custom_translators, F );
1237 }
1238 __except( SSE( GetExceptionCode(), GetExceptionInformation() ) ) {
1239 throw SSE;
1240 }
1241
1242 // we check for time outs: we do not have any signaling facility on Win32
1243 // however, we signal a timeout as a hard error as for the other operating systems
1244 // and throw the signal error handler
1245 if( bTimerSuccess && htimer != INVALID_HANDLE_VALUE) {
1246 if (::WaitForSingleObject(htimer, 0) == WAIT_OBJECT_0) {
1247 SSE.set_timed_out();
1248 throw SSE;
1249 }
1250 }
1251
1252 }
1253 __finally {
1254
1255#if defined(BOOST_TEST_WIN32_WAITABLE_TIMERS)
1256 if( htimer != INVALID_HANDLE_VALUE ) {
1257 ::CloseHandle(htimer);
1258 }
1259#endif
1260
1261 if( l_catch_system_errors ) {
1262 BOOST_TEST_CRT_SET_HOOK( old_crt_hook );
1263
1264 _set_invalid_parameter_handler( old_iph );
1265 }
1266 }
1267
1268 return ret_val;
1269}

1306//____________________________________________________________________________//
1307
1308int
1309execution_monitor::execute( boost::function<int ()> const& F )
1310{
1311 if( debug::under_debugger() )
1312 p_catch_system_errors.value = false;
1313
1314 BOOST_TEST_I_TRY {
1315 detail::fpe_except_guard G( p_detect_fp_exceptions );
1316 boost::ignore_unused( G );
1317
1318 return catch_signals( F );
1319 }
1320
1321#ifndef BOOST_NO_EXCEPTIONS
1322
1323 // Catch-clause reference arguments are a bit different from function
1324 // arguments (ISO 15.3 paragraphs 18 & 19). Apparently const isn't
1325 // required. Programmers ask for const anyhow, so we supply it. That's
1326 // easier than answering questions about non-const usage.
1327
1328 catch( char const* ex )
1329 { detail::report_error( ec: execution_exception::cpp_exception_error,
1330 format: "C string: %s", ex ); }
1331 catch( std::string const& ex )
1332 { detail::report_error( ec: execution_exception::cpp_exception_error,
1333 format: "std::string: %s", ex.c_str() ); }
1334
1335 // boost::exception (before std::exception, with extended diagnostic)
1336 catch( boost::exception const& ex )
1337 { detail::report_error( ec: execution_exception::cpp_exception_error,
1338 be: &ex,
1339 format: "%s", boost::diagnostic_information(e: ex).c_str() ); }
1340
1341 // std:: exceptions
1342#if defined(BOOST_NO_TYPEID) || defined(BOOST_NO_RTTI)
1343#define CATCH_AND_REPORT_STD_EXCEPTION( ex_name ) \
1344 catch( ex_name const& ex ) \
1345 { detail::report_error( execution_exception::cpp_exception_error, \
1346 current_exception_cast<boost::exception const>(), \
1347 #ex_name ": %s", ex.what() ); } \
1348/**/
1349#else
1350#define CATCH_AND_REPORT_STD_EXCEPTION( ex_name ) \
1351 catch( ex_name const& ex ) \
1352 { detail::report_error( execution_exception::cpp_exception_error, \
1353 current_exception_cast<boost::exception const>(), \
1354 "%s: %s", detail::typeid_name(ex).c_str(), ex.what() ); } \
1355/**/
1356#endif
1357

1464unsigned
1465enable( unsigned mask )
1466{
1467 boost::ignore_unused(mask);
1468#if defined(BOOST_TEST_FPE_SUPPORT_WITH_SEH__)
1469 _clearfp();
1470
1471#if BOOST_WORKAROUND( BOOST_MSVC, <= 1310)
1472 unsigned old_cw = ::_controlfp( 0, 0 );
1473 ::_controlfp( old_cw & ~mask, BOOST_FPE_ALL );
1474#else
1475 unsigned old_cw;
1476 if( ::_controlfp_s( &old_cw, 0, 0 ) != 0 )
1477 return BOOST_FPE_INV;
1478
1479 // Set the control word
1480 if( ::_controlfp_s( 0, old_cw & ~mask, BOOST_FPE_ALL ) != 0 )
1481 return BOOST_FPE_INV;
1482#endif
1483 return ~old_cw & BOOST_FPE_ALL;
1484
1485#elif defined(BOOST_TEST_FPE_SUPPORT_WITH_GLIBC_EXTENSIONS__)
1486 // same macro definition as in execution_monitor.hpp
1487 if (BOOST_FPE_ALL == BOOST_FPE_OFF)
1488 /* Not Implemented */
1489 return BOOST_FPE_OFF;
1490 feclearexcept(BOOST_FPE_ALL);
1491 int res = feenableexcept( mask );
1492 return res == -1 ? (unsigned)BOOST_FPE_INV : (unsigned)res;
1493#else
1494 /* Not Implemented */
1495 return BOOST_FPE_OFF;
1496#endif
1497}
