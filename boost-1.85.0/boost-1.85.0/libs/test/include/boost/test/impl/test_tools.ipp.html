78void
79print_log_value<bool>::operator()( std::ostream& ostr, bool t )
80{
81 ostr << std::boolalpha << t;
82}
83
84void
85print_log_value<char>::operator()( std::ostream& ostr, char t )
86{
87 if( (std::isprint)( static_cast<unsigned char>(t) ) )
88 ostr << '\'' << t << '\'';
89 else
90 ostr << std::hex
91#if BOOST_TEST_USE_STD_LOCALE
92 << std::showbase
93#else
94 << "0x"
95#endif
96 << static_cast<int>(t);
97}

157using ::boost::unit_test::lazy_ostream;
158
159static char const* check_str [] = { " == ", " != ", " < " , " <= ", " > " , " >= " };
160static char const* rever_str [] = { " != ", " == ", " >= ", " > " , " <= ", " < " };
161
162template<typename OutStream>
163void
164format_report( OutStream& os, assertion_result const& pr, unit_test::lazy_ostream const& assertion_descr,
165 tool_level tl, check_type ct,
166 std::size_t num_args, va_list args,
167 char const* prefix, char const* suffix )
168{
169 using namespace unit_test;
170
171 switch( ct ) {
172 case CHECK_PRED:
173 os << prefix << assertion_descr << suffix;
174
175 if( !pr.has_empty_message() )
176 os << ". " << pr.message();
177 break;
178
179 case CHECK_BUILT_ASSERTION: {
180 os << prefix << assertion_descr << suffix;
181
182 if( tl != PASS ) {
183 const_string details_message = pr.message();
184
185 if( !details_message.is_empty() ) {
186 os << details_message;
187 }
188 }
189 break;
190 }
191
192 case CHECK_MSG:
193 if( tl == PASS )
194 os << prefix << "'" << assertion_descr << "'" << suffix;
195 else
196 os << assertion_descr;
197
198 if( !pr.has_empty_message() )
199 os << ". " << pr.message();
200 break;
201
202 case CHECK_EQUAL:
203 case CHECK_NE:
204 case CHECK_LT:
205 case CHECK_LE:
206 case CHECK_GT:
207 case CHECK_GE: {
208 char const* arg1_descr = va_arg( args, char const* );
209 lazy_ostream const* arg1_val = va_arg( args, lazy_ostream const* );
210 char const* arg2_descr = va_arg( args, char const* );
211 lazy_ostream const* arg2_val = va_arg( args, lazy_ostream const* );
212
213 os << prefix << arg1_descr << check_str[ct-CHECK_EQUAL] << arg2_descr << suffix;
214
215 if( tl != PASS )
216 os << " [" << *arg1_val << rever_str[ct-CHECK_EQUAL] << *arg2_val << "]" ;
217
218 if( !pr.has_empty_message() )
219 os << ". " << pr.message();
220 break;
221 }
222
223 case CHECK_CLOSE:
224 case CHECK_CLOSE_FRACTION: {
225 char const* arg1_descr = va_arg( args, char const* );
226 lazy_ostream const* arg1_val = va_arg( args, lazy_ostream const* );
227 char const* arg2_descr = va_arg( args, char const* );
228 lazy_ostream const* arg2_val = va_arg( args, lazy_ostream const* );
229 /* toler_descr = */ va_arg( args, char const* );
230 lazy_ostream const* toler_val = va_arg( args, lazy_ostream const* );
231
232 os << "difference{" << pr.message()
233 << "} between " << arg1_descr << "{" << *arg1_val
234 << "} and " << arg2_descr << "{" << *arg2_val
235 << ( tl == PASS ? "} doesn't exceed " : "} exceeds " )
236 << *toler_val;
237 if( ct == CHECK_CLOSE )
238 os << "%";
239 break;
240 }
241 case CHECK_SMALL: {
242 char const* arg1_descr = va_arg( args, char const* );
243 lazy_ostream const* arg1_val = va_arg( args, lazy_ostream const* );
244 /* toler_descr = */ va_arg( args, char const* );
245 lazy_ostream const* toler_val = va_arg( args, lazy_ostream const* );
246
247 os << "absolute value of " << arg1_descr << "{" << *arg1_val << "}"
248 << ( tl == PASS ? " doesn't exceed " : " exceeds " )
249 << *toler_val;
250
251 if( !pr.has_empty_message() )
252 os << ". " << pr.message();
253 break;
254 }
255
256 case CHECK_PRED_WITH_ARGS: {
257 std::vector< std::pair<char const*, lazy_ostream const*> > args_copy;
258 args_copy.reserve( n: num_args );
259 for( std::size_t i = 0; i < num_args; ++i ) {
260 char const* desc = va_arg( args, char const* );
261 lazy_ostream const* value = va_arg( args, lazy_ostream const* );
262 args_copy.push_back( x: std::make_pair( x&: desc, y&: value ) );
263 }
264
265 os << prefix << assertion_descr;
266
267 // print predicate call description
268 os << "( ";
269 for( std::size_t i = 0; i < num_args; ++i ) {
270 os << args_copy[i].first;
271
272 if( i != num_args-1 )
273 os << ", ";
274 }
275 os << " )" << suffix;
276
277 if( tl != PASS ) {
278 os << " for ( ";
279 for( std::size_t i = 0; i < num_args; ++i ) {
280 os << *args_copy[i].second;
281
282 if( i != num_args-1 )
283 os << ", ";
284 }
285 os << " )";
286 }
287
288 if( !pr.has_empty_message() )
289 os << ". " << pr.message();
290 break;
291 }
292
293 case CHECK_EQUAL_COLL: {
294 char const* left_begin_descr = va_arg( args, char const* );
295 char const* left_end_descr = va_arg( args, char const* );
296 char const* right_begin_descr = va_arg( args, char const* );
297 char const* right_end_descr = va_arg( args, char const* );
298
299 os << prefix << "{ " << left_begin_descr << ", " << left_end_descr << " } == { "
300 << right_begin_descr << ", " << right_end_descr << " }"
301 << suffix;
302
303 if( !pr.has_empty_message() )
304 os << ". " << pr.message();
305 break;
306 }
307
308 case CHECK_BITWISE_EQUAL: {
309 char const* left_descr = va_arg( args, char const* );
310 char const* right_descr = va_arg( args, char const* );
311
312 os << prefix << left_descr << " =.= " << right_descr << suffix;
313
314 if( !pr.has_empty_message() )
315 os << ". " << pr.message();
316 break;
317 }
318 }
319}

324#pragma warning(push)
325#pragma warning(disable : 4702) // There is intentionally unreachable code
326#endif
327
328bool
329report_assertion( assertion_result const& ar,
330 lazy_ostream const& assertion_descr,
331 const_string file_name,
332 std::size_t line_num,
333 tool_level tl,
334 check_type ct,
335 std::size_t num_args, ... )
336{
337 using namespace unit_test;
338
339 if( !framework::test_in_progress() ) {
340 // in case no test is in progress, we do not throw anything:
341 // raising an exception here may result in raising an exception in a destructor of a global fixture
342 // which will abort the process
343 // We flag this as aborted instead
344
345 //BOOST_TEST_I_ASSRT( framework::current_test_case_id() != INV_TEST_UNIT_ID,
346 // std::runtime_error( "Can't use testing tools outside of test case implementation." ) );
347
348 framework::test_aborted();
349 return false;
350 }
351
352
353 if( !!ar )
354 tl = PASS;
355
356 log_level ll;
357 char const* prefix;
358 char const* suffix;
359
360 switch( tl ) {
361 case PASS:
362 ll = log_successful_tests;
363 prefix = "check ";
364 suffix = " has passed";
365 break;
366 case WARN:
367 ll = log_warnings;
368 prefix = "condition ";
369 suffix = " is not satisfied";
370 break;
371 case CHECK:
372 ll = log_all_errors;
373 prefix = "check ";
374 suffix = " has failed";
375 break;
376 case REQUIRE:
377 ll = log_fatal_errors;
378 prefix = "critical check ";
379 suffix = " has failed";
380 break;
381 default:
382 return true;
383 }
384
385 unit_test_log << unit_test::log::begin( file_name, line_num ) << ll;
386 va_list args;
387 va_start( args, num_args );
388
389 format_report( os&: unit_test_log, pr: ar, assertion_descr, tl, ct, num_args, args, prefix, suffix );
390
391 va_end( args );
392 unit_test_log << unit_test::log::end();
393
394 switch( tl ) {
395 case PASS:
396 framework::assertion_result( ar: AR_PASSED );
397 return true;
398
399 case WARN:
400 framework::assertion_result( ar: AR_TRIGGERED );
401 return false;
402
403 case CHECK:
404 framework::assertion_result( ar: AR_FAILED );
405 return false;
406
407 case REQUIRE:
408 framework::assertion_result( ar: AR_FAILED );
409 framework::test_unit_aborted( framework::current_test_unit() );
410 BOOST_TEST_I_THROW( execution_aborted() );
411 // the previous line either throws or aborts and the return below is not reached
412 // return false;
413 BOOST_TEST_UNREACHABLE_RETURN(false);
414 }
415
416 return true;
417}

424
425assertion_result
426format_assertion_result( const_string expr_val, const_string details )
427{
428 assertion_result res(false);
429
430 bool starts_new_line = first_char( source: expr_val ) == '\n';
431
432 if( !starts_new_line && !expr_val.is_empty() )
433 res.message().stream() << " [" << expr_val << "]";
434
435 if( !details.is_empty() ) {
436 if( first_char(source: details) != '[' )
437 res.message().stream() << ": ";
438 else
439 res.message().stream() << " ";
440
441 res.message().stream() << details;
442 }
443
444 if( starts_new_line )
445 res.message().stream() << "." << expr_val;
446
447 return res;
448}

529struct output_test_stream::Impl
530{
531 std::fstream m_pattern;
532 bool m_match_or_save;
533 bool m_text_or_binary;
534 std::string m_synced_string;
535
536 char get_char()
537 {
538 char res = 0;
539 do {
540 m_pattern.get( c&: res );
541 } while( m_text_or_binary && res == '\r' && !m_pattern.fail() && !m_pattern.eof() );
542
543 return res;
544 }
545
546 void check_and_fill( assertion_result& res )
547 {
548 if( !res.p_predicate_value )
549 res.message() << "Output content: \"" << m_synced_string << '\"';
550 }
551};

555output_test_stream::output_test_stream( const_string pattern_file_name, bool match_or_save, bool text_or_binary )
556: m_pimpl( new Impl )

558 if( !pattern_file_name.is_empty() ) {
559 std::ios::openmode m = match_or_save ? std::ios::in : std::ios::out;

561 m |= std::ios::binary;
562

566 BOOST_TEST_FRAMEWORK_MESSAGE( "Can't open pattern file " << pattern_file_name << " for " << (match_or_save ? "reading" : "writing") );
567 }
568
569 m_pimpl->m_match_or_save = match_or_save;
570 m_pimpl->m_text_or_binary = text_or_binary;
571}

633std::string pretty_print_log(std::string str) {
634
635 static const std::string to_replace[] = { "\r", "\n" };
636 static const std::string replacement[] = { "\\r", "\\n" };
637
638 return unit_test::utils::replace_all_occurrences_of(
639 str,
640 first1: to_replace, last1: to_replace + sizeof(to_replace)/sizeof(to_replace[0]),
641 first2: replacement, last2: replacement + sizeof(replacement)/sizeof(replacement[0]));
642}
643
644assertion_result
645output_test_stream::match_pattern( bool flush_stream )
646{
647 const std::string::size_type n_chars_presuffix = 10;
648 sync();
649
650 assertion_result result( true );
651
652 const std::string stream_string_repr = get_stream_string_representation();
653
654 if( !m_pimpl->m_pattern.is_open() ) {
655 result = false;
656 result.message() << "Pattern file can't be opened!";
657 }
658 else {
659 if( m_pimpl->m_match_or_save ) {
660
661 int offset = 0;
662 std::vector<char> last_elements;
663 for ( std::string::size_type i = 0; static_cast<int>(i + offset) < static_cast<int>(stream_string_repr.length()); ++i ) {
664
665 char c = m_pimpl->get_char();
666
667 if( last_elements.size() <= n_chars_presuffix ) {
668 last_elements.push_back( x: c );
669 }
670 else {
671 last_elements[ i % last_elements.size() ] = c;
672 }
673
674 bool is_same = !m_pimpl->m_pattern.fail() &&
675 !m_pimpl->m_pattern.eof() &&
676 (stream_string_repr[i+offset] == c);
677
678 if( !is_same ) {
679
680 result = false;
681
682 std::string::size_type prefix_size = (std::min)( a: i + offset, b: n_chars_presuffix );
683
684 std::string::size_type suffix_size = (std::min)( a: stream_string_repr.length() - i - offset,
685 b: n_chars_presuffix );
686
687 // try to log area around the mismatch
688 std::string substr = stream_string_repr.substr(pos: 0, n: i+offset);
689 std::size_t line = std::count(first: substr.begin(), last: substr.end(), value: '\n');
690 std::size_t column = i + offset - substr.rfind(c: '\n');
691
692 result.message()
693 << "Mismatch at position " << i
694 << " (line " << line
695 << ", column " << column
696 << "): '" << pretty_print_log(str: std::string(1, stream_string_repr[i+offset])) << "' != '" << pretty_print_log(str: std::string(1, c)) << "' :\n";
697
698 // we already escape this substring because we need its actual size for the pretty print
699 // of the difference location.
700 std::string sub_str_prefix(pretty_print_log(str: stream_string_repr.substr( pos: i + offset - prefix_size, n: prefix_size )));
701
702 // we need this substring as is because we compute the best matching substrings on it.
703 std::string sub_str_suffix(stream_string_repr.substr( pos: i + offset, n: suffix_size));
704 result.message() << "... " << sub_str_prefix + pretty_print_log(str: sub_str_suffix) << " ..." << '\n';
705
706 result.message() << "... ";
707 for( std::size_t j = 0; j < last_elements.size() ; j++ )
708 result.message() << pretty_print_log(str: std::string(1, last_elements[(i + j + 1) % last_elements.size()]));
709
710 std::vector<char> last_elements_ordered;
711 last_elements_ordered.push_back(x: c);
712 for( std::string::size_type counter = 0; counter < suffix_size - 1 ; counter++ ) {
713 char c2 = m_pimpl->get_char();
714
715 if( m_pimpl->m_pattern.fail() || m_pimpl->m_pattern.eof() )
716 break;
717
718 result.message() << pretty_print_log(str: std::string(1, c2));
719
720 last_elements_ordered.push_back(x: c2);
721 }
722
723 // tries to find the best substring matching in the remainder of the
724 // two strings
725 std::size_t max_nb_char_in_common = 0;
726 std::size_t best_pattern_start_index = 0;
727 std::size_t best_stream_start_index = 0;
728 for( std::size_t pattern_start_index = best_pattern_start_index;
729 pattern_start_index < last_elements_ordered.size();
730 pattern_start_index++ ) {
731 for( std::size_t stream_start_index = best_stream_start_index;
732 stream_start_index < sub_str_suffix.size();
733 stream_start_index++ ) {
734
735 std::size_t max_size = (std::min)( a: last_elements_ordered.size() - pattern_start_index, b: sub_str_suffix.size() - stream_start_index );
736 if( max_nb_char_in_common > max_size )
737 break; // safely break to go to the outer loop
738
739 std::size_t nb_char_in_common = 0;
740 for( std::size_t k = 0; k < max_size; k++) {
741 if( last_elements_ordered[pattern_start_index + k] == sub_str_suffix[stream_start_index + k] )
742 nb_char_in_common ++;
743 else
744 break; // we take fully matching substring only
745 }
746
747 if( nb_char_in_common > max_nb_char_in_common ) {
748 max_nb_char_in_common = nb_char_in_common;
749 best_pattern_start_index = pattern_start_index;
750 best_stream_start_index = stream_start_index;
751 }
752 }
753 }
754
755 // indicates with more precision the location of the mismatchs in "ascii arts" ...
756 result.message() << " ...\n... ";
757 for( std::string::size_type j = 0; j < sub_str_prefix.size(); j++) {
758 result.message() << ' ';
759 }
760
761 result.message() << '~'; // places the first tilde at the current char that mismatches
762
763 for( std::size_t k = 1; k < (std::max)(a: best_pattern_start_index, b: best_stream_start_index); k++ ) { // 1 is for the current char c
764 std::string s1(pretty_print_log(str: std::string(1, last_elements_ordered[(std::min)(a: k, b: best_pattern_start_index)])));
765 std::string s2(pretty_print_log(str: std::string(1, sub_str_suffix[(std::min)(a: k, b: best_stream_start_index)])));
766 for( int h = static_cast<int>((std::max)(a: s1.size(), b: s2.size())); h > 0; h--)
767 result.message() << "~";
768 }
769
770 if( m_pimpl->m_pattern.eof() ) {
771 result.message() << " (reference string shorter than current stream)";
772 }
773
774 result.message() << "\n";
775
776 // no need to continue if the EOF is reached
777 if( m_pimpl->m_pattern.eof() ) {
778 break;
779 }
780
781 // first char is a replicat of c, so we do not copy it.
782 for(std::string::size_type counter = 0; counter < last_elements_ordered.size() - 1 ; counter++)
783 last_elements[ (i + 1 + counter) % last_elements.size() ] = last_elements_ordered[counter + 1];
784
785 i += last_elements_ordered.size()-1;
786 offset += best_stream_start_index - best_pattern_start_index;
787
788 }
789
790 }
791
792 // not needed anymore
793 /*
794 if(offset > 0 && false) {
795 m_pimpl->m_pattern.ignore(
796 static_cast<std::streamsize>( offset ));
797 }
798 */
799 }
800 else {
801 m_pimpl->m_pattern.write( s: stream_string_repr.c_str(),
802 n: static_cast<std::streamsize>( stream_string_repr.length() ) );
803 m_pimpl->m_pattern.flush();
804 }
805 }
806
807 if( flush_stream )
808 flush();
809
810 return result;
811}
