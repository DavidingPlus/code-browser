10#ifndef BOOST_JSON_SERIALIZE_OPTIONS_HPP
11#define BOOST_JSON_SERIALIZE_OPTIONS_HPP
12
13#include <boost/json/detail/config.hpp>
14#include <iosfwd>
15
16namespace boost {
17namespace json {
18
19/** Serialize options
20
21 This structure is used for specifying whether to allow non-standard
22 extensions. Default-constructed options specify that only standard JSON is
23 produced.
24
25 @see
26 @ref serialize,
27 @ref serializer.
28*/
29struct serialize_options
30{
31 /** Non-standard extension option
32
33 Output `Infinity`, `-Infinity` and `NaN` for positive infinity,
34 negative infinity, and "not a number" doubles.
35
36 @see
37 @ref serialize,
38 @ref serializer.
39 */
40 bool allow_infinity_and_nan = false;
41
42 /** Set JSON serialization options on input stream.
43 The function stores serialization options in the private storage of the
44 stream. If the stream fails to allocate necessary private storage,
45 `badbit` will be set on it.
46
47 @return Reference to `os`.
48
49 @par Complexity
50 Amortized constant (due to potential memory allocation by the stream).
51
52 @par Exception Safety
53 Strong guarantee.
54 The stream may throw as configured by
55 [`std::ios::exceptions`](https://en.cppreference.com/w/cpp/io/basic_ios/exceptions).
56
57 @param os The output stream.
58
59 @param opts The options to store.
60 */
61 BOOST_JSON_DECL
62 friend
63 std::ostream&
64 operator<<( std::ostream& os, serialize_options const& opts );
65};
66
67} // namespace json
68} // namespace boost
69
70#endif // BOOST_JSON_SERIALIZE_OPTIONS_HPP
