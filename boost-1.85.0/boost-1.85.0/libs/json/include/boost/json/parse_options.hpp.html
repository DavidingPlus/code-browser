10#ifndef BOOST_JSON_PARSE_OPTIONS_HPP
11#define BOOST_JSON_PARSE_OPTIONS_HPP
12
13#include <boost/json/detail/config.hpp>
14#include <iosfwd>
15
16namespace boost {
17namespace json {
18
19/** Enumeration of number parsing modes
20
21 These values are used to select the way to parse numbers.
22
23 @see
24 @ref parse_options,
25 @ref basic_parser,
26 @ref parser.
27*/
28enum class number_precision : unsigned char
29{
30 /// Fast, but potentially less precise mode.
31 imprecise,
32
33 /// Slower, but precise mode.
34 precise,
35
36 /// The fastest mode, that only validates encountered numbers without
37 /// parsing them.
38 none,
39};
40
41/** Parser options
42
43 This structure is used for specifying
44 maximum parsing depth, and whether
45 to allow various non-standard extensions.
46 Default-constructed options set maximum
47 parsing depth to 32 and specify that only
48 standard JSON is allowed,
49
50 @see
51 @ref basic_parser,
52 @ref parser.
53*/
54struct parse_options
55{
56 /** Maximum nesting level of arrays and objects.
57
58 This specifies the maximum number of nested
59 structures allowed while parsing a JSON text. If
60 this limit is exceeded during a parse, an
61 error is returned.
62
63 @see
64 @ref basic_parser,
65 @ref stream_parser.
66 */
67 std::size_t max_depth = 32;
68
69 /** Number pasing mode
70
71 This selects the way to parse numbers. The default is to parse them
72 fast, but with possible slight imprecision for floating point numbers
73 with larger mantissas. Users can also choose to parse numbers slower
74 but with full precision. Or to not parse them at all, and only validate
75 numbers. The latter mode is useful for @ref basic_parser instantiations
76 that wish to treat numbers in a custom way.
77
78 @see
79 @ref basic_parser,
80 @ref stream_parser.
81 */
82 number_precision numbers = number_precision::imprecise;
83
84 /** Non-standard extension option
85
86 Allow C and C++ style comments to appear
87 anywhere that whitespace is permissible.
88
89 @see
90 @ref basic_parser,
91 @ref stream_parser.
92 */
93 bool allow_comments = false;
94
95 /** Non-standard extension option
96
97 Allow a trailing comma to appear after
98 the last element of any array or object.
99
100 @see
101 @ref basic_parser,
102 @ref stream_parser.
103 */
104 bool allow_trailing_commas = false;
105
106 /** Non-standard extension option
107
108 Allow invalid UTF-8 sequnces to appear
109 in keys and strings.
110
111 @note This increases parsing performance.
112
113 @see
114 @ref basic_parser,
115 @ref stream_parser.
116 */
117 bool allow_invalid_utf8 = false;
118
119 /** Non-standard extension option
120
121 Allow `Infinity`, `-Infinity`, and `NaN` JSON literals. These values
122 are produced by some popular JSON implementations for positive
123 infinity, negative infinity and NaN special numbers respectively.
124
125 @see
126 @ref basic_parser,
127 @ref stream_parser.
128 */
129 bool allow_infinity_and_nan = false;
130
131 /** Set JSON parse options on input stream.
132
133 The function stores parse options in the private storage of the stream. If
134 the stream fails to allocate necessary private storage, `badbit` will be
135 set on it.
136
137 @return Reference to `is`.
138
139 @par Complexity
140 Amortized constant (due to potential memory allocation by the stream).
141
142 @par Exception Safety
143 Strong guarantee.
144 The stream may throw as configured by
145 [`std::ios::exceptions`](https://en.cppreference.com/w/cpp/io/basic_ios/exceptions).
146
147 @param is The input stream.
148
149 @param opts The options to store.
150 */
151 BOOST_JSON_DECL
152 friend
153 std::istream&
154 operator>>( std::istream& is, parse_options const& opts );
155};
156
157} // namespace json
158} // namespace boost
159
160#endif
