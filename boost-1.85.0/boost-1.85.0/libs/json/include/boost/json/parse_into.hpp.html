25/** `basic_parser` that parses into a given type
26
27 This is an alias template for @ref basic_parser instantiations that use
28 a dedicated handler that parses directly into an object provided by the
29 user instead of creating a @ref value.
30
31 Objects of type `parser_for<T>` have constructor signature equivalent to
32 `parser_for( parse_options const&, T& )`.
33
34 @tparam T the type to parse into. This type must be
35 [*DefaultConstructible*](https://en.cppreference.com/w/cpp/named_req/DefaultConstructible).
36*/
37template< class T >
38using parser_for =
39#ifndef BOOST_JSON_DOCS
40 basic_parser<detail::into_handler<T>>;
41#else
42 __see_below__;
43#endif
44
45/** Parse a JSON text into a user-defined object.
46
47 This function parses an entire string in one step and fills an object
48 provided by the user. If the buffer does not contain a complete serialized
49 JSON text, an error occurs. In this case `v` may be partially filled.
50
51 The function supports default constructible types satisfying
52 <a href="https://en.cppreference.com/w/cpp/named_req/SequenceContainer"><em>SequenceContainer</em></a>,
53 arrays, arithmetic types, `bool`, `std::tuple`, `std::pair`,
54 `std::optional`, `std::nullptr_t`, and structs and enums described using
55 Boost.Describe.
56
57 @par Complexity
58 Linear in `sv.size()`.
59
60 @par Exception Safety
61 Basic guarantee.
62 Calls to `memory_resource::allocate` may throw.
63
64 @param v The type to parse into.
65
66 @param sv The string to parse.
67
68 @param ec Set to the error, if any occurred.
69
70 @param opt The options for the parser. If this parameter
71 is omitted, the parser will accept only standard JSON.
72*/
73/** @{ */
74template<class V>
75void
76parse_into(
77 V& v,
78 string_view sv,
79 system::error_code& ec,
80 parse_options const& opt = {} );
81
82template<class V>
83void
84parse_into(
85 V& v,
86 string_view sv,
87 std::error_code& ec,
88 parse_options const& opt = {} );
89/** @} */
90

127/** Parse a JSON text into a user-defined object.
128
129 This function reads data from an input stream and fills an object provided
130 by the user. If the buffer does not contain a complete serialized JSON
131 text, or contains extra non-whitespace data, an error occurs. In this case
132 `v` may be partially filled.
133
134 The function supports default constructible types satisfying
135 <a href="https://en.cppreference.com/w/cpp/named_req/SequenceContainer"><em>SequenceContainer</em></a>,
136 arrays, arithmetic types, `bool`, `std::tuple`, `std::pair`,
137 `std::optional`, `std::nullptr_t`, and structs and enums described using
138 Boost.Describe.
139
140 @par Complexity
141 Linear in the size of consumed input.
142
143 @par Exception Safety
144 Basic guarantee.
145 Calls to `memory_resource::allocate` may throw.
146 The stream may throw as described by
147 [`std::ios::exceptions`](https://en.cppreference.com/w/cpp/io/basic_ios/exceptions).
148
149 @param v The type to parse into.
150
151 @param is The stream to read from.
152
153 @param ec Set to the error, if any occurred.
154
155 @param opt The options for the parser. If this parameter
156 is omitted, the parser will accept only standard JSON.
157*/
158/** @{ */
159template<class V>
160void
161parse_into(
162 V& v,
163 std::istream& is,
164 system::error_code& ec,
165 parse_options const& opt = {} );
166
167template<class V>
168void
169parse_into(
170 V& v,
171 std::istream& is,
172 std::error_code& ec,
173 parse_options const& opt = {} );
174/** @} */
