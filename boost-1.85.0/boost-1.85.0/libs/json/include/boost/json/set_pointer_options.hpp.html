10#ifndef BOOST_JSON_SET_POINTER_OPTIONS_HPP
11#define BOOST_JSON_SET_POINTER_OPTIONS_HPP
12
13#include <boost/json/detail/config.hpp>
14
15namespace boost {
16namespace json {
17
18/** Options for @ref value::set_at_pointer
19 *
20 * This structure is used for controlling behavior of
21 * @ref value::set_at_pointer regarding creation of intermediate elements.
22 */
23struct set_pointer_options
24{
25 /** Whether to create arrays
26
27 The option controls whether arrays are created when a pointer token is
28 a number or a past-the-end marker.
29 */
30 bool create_arrays = true;
31
32 /** Whether to create objects
33
34 The option controls whether objects are created for valid pointer
35 tokens.
36 */
37 bool create_objects = true;
38
39 /** Whether to replace non-null scalars
40
41 If the option is `true` any non-object, non-array value can be replaced
42 by either an @ref object or a @ref array. If it's `false`, only `null`
43 values can be replaced.
44 */
45 bool replace_any_scalar = false;
46
47
48 /** Maximum amount of elements added per one pointer token
49
50 When addressing @ref array elements the number represented by pointer
51 token can exceed the size of the array. In that case several elements
52 may be created to satisfy the request. This option limits the amount of
53 elements that can be added per one pointer token. This can be
54 important, because in general such operations can be dangerous, since a
55 relatively small string `"/18446744073709551616"` can request all of
56 the machine's memory. For that reason the default value for this option
57 is 1, that is no more than 1 element can be added per one pointer
58 token.
59 */
60 std::size_t max_created_elements = 1;
61};
62
63} // namespace json
64} // namespace boost
65
66#endif // BOOST_JSON_SET_POINTER_OPTIONS_HPP
