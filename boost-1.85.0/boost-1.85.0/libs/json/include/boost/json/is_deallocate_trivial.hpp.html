10#ifndef BOOST_JSON_IS_DEALLOCATE_TRIVIAL_HPP
11#define BOOST_JSON_IS_DEALLOCATE_TRIVIAL_HPP
12
13namespace boost {
14namespace json {
15
16/** Return true if a memory resource's deallocate function has no effect.
17
18 This metafunction may be specialized to indicate to the library that calls
19 to the `deallocate` function of a `boost::container::pmr::memory_resource`
20 have no effect. The implementation will elide such calls when it is safe to
21 do so. By default, the implementation assumes that all memory resources
22 require a call to `deallocate` for each memory region obtained by
23 calling `allocate`.
24
25 @par Example
26
27 This example specializes the metafuction for `my_resource`,
28 to indicate that calls to deallocate have no effect:
29
30 @code
31
32 // Forward-declaration for a user-defined memory resource
33 struct my_resource;
34
35 // It is necessary to specialize the template from
36 // inside the namespace in which it is declared:
37
38 namespace boost {
39 namespace json {
40
41 template<>
42 struct is_deallocate_trivial< my_resource >
43 {
44 static constexpr bool value = true;
45 };
46
47 } // namespace json
48 } // namespace boost
49
50 @endcode
51
52 It is usually not necessary for users to check this trait.
53 Instead, they can call @ref storage_ptr::is_deallocate_trivial
54 to determine if the pointed-to memory resource has a trivial
55 deallocate function.
56
57 @see
58 @ref storage_ptr,
59 [`boost::container::pmr::memory_resource`](https://www.boost.org/doc/libs/release/doc/html/boost/container/pmr/memory_resource.html).
60*/
61template<class T>
62struct is_deallocate_trivial
63{
64 /** A bool equal to true if calls to `T::do_deallocate` have no effect.
65
66 The primary template sets `value` to false.
67 */
68 static constexpr bool value = false;
69};
70
71} // namespace json
72} // namespace boost
73
74#endif // BOOST_JSON_IS_DEALLOCATE_TRIVIAL_HPP
