72/** Determine if `T` can be treated like a string during conversions.
73
74 Provides the member constant `value` that is equal to `true`, if `T` is
75 convertible to @ref string_view. Otherwise, `value` is equal to `false`.
76 <br>
77
78 Users can specialize the trait for their own types if they don't want them
79 to be treated like strings. For example:
80
81 @code
82 namespace boost {
83 namespace json {
84
85 template <>
86 struct is_string_like<your::string> : std::false_type
87 { };
88
89 } // namespace boost
90 } // namespace json
91 @endcode
92
93 @par Types satisfying the trait
94
95 @ref string,
96 @ref string_view,
97 <a href="https://en.cppreference.com/w/cpp/string/basic_string"><tt>std::string</tt></a>,
98 <a href="https://en.cppreference.com/w/cpp/string/basic_string_view"><tt>std::string_view</tt></a>.
99
100 @see @ref value_from, @ref value_to
101*/
102template<class T>
103struct is_string_like;

105/** Determine if `T` can be treated like `std::filesystem::path` during conversions.
106
107 Given `t`, a glvalue of type `T`, if
108
109 @li given `It`, the type denoted by `decltype(std::begin(t))`,
110 <tt>std::iterator_traits<It>::iterator_category</tt> is well-formed and
111 denotes a type; and
112
113 @li <tt>std::iterator_traits<It>::value_type</tt> is `T`; and
114
115 @li `T::value_type` is well-formed and denotes a type; and
116
117 @li `T::string_type` is well-formed, denotes a type, and is an alias for
118 `std::basic_string< T::value_type >`;
119
120 then the trait provides the member constant `value` that is equal to
121 `true`. Otherwise, `value` is equal to `false`.<br>
122
123 Users can specialize the trait for their own types if they don't want them
124 to be treated like filesystem paths. For example:
125
126 @code
127 namespace boost {
128 namespace json {
129
130 template <>
131 struct is_path_like<your::path> : std::false_type
132 { };
133
134 } // namespace boost
135 } // namespace json
136 @endcode
137
138
139 @par Types satisfying the trait
140
141 `std::filesystem::path`, `boost::filesystem::path`.
142
143 @see @ref value_from, @ref value_to
144*/
145template<class T>
146struct is_path_like;

148/** Determine if `T` can be treated like a sequence during conversions.
149
150 Given `t`, a glvalue of type `T`, if
151
152 @li given `It`, the type denoted by `decltype(std::begin(t))`,
153 <tt>std::iterator_traits<It>::iterator_category</tt> is well-formed and
154 denotes a type; and
155
156 @li `decltype(std::end(t))` also denotes the type `It`; and
157
158 @li <tt>std::iterator_traits<It>::value_type</tt> is not `T`; and
159
160 then the trait provides the member constant `value` that is equal to
161 `true`. Otherwise, `value` is equal to `false`.<br>
162
163 Users can specialize the trait for their own types if they don't want them
164 to be treated like sequences. For example:
165
166 @code
167 namespace boost {
168 namespace json {
169
170 template <>
171 struct is_sequence_like<your::container> : std::false_type
172 { };
173
174 } // namespace boost
175 } // namespace json
176 @endcode
177
178
179 @par Types satisfying the trait
180
181 Any <a href="https://en.cppreference.com/w/cpp/named_req/SequenceContainer"><em>SequenceContainer</em></a>,
182 array types.
183
184 @see @ref value_from, @ref value_to
185*/
186template<class T>
187struct is_sequence_like;

189/** Determine if `T` can be treated like a 1-to-1 mapping during
190 conversions.
191
192 Given `t`, a glvalue of type `T`, if
193
194 @li <tt>is_sequence_like<T>::value</tt> is `true`; and
195
196 @li given type `It` denoting `decltype(std::begin(t))`, and types `K`
197 and `M`, <tt>std::iterator_traits<It>::value_type</tt> denotes
198 `std::pair<K, M>`; and
199
200 @li <tt>std::is_string_like<K>::value</tt> is `true`; and
201
202 @li given `v`, a glvalue of type `V`, and `E`, the type denoted by
203 `decltype(t.emplace(v))`,
204 <tt>std::is_tuple_like<E>::value</tt> is `true`;
205
206 then the trait provides the member constant `value`
207 that is equal to `true`. Otherwise, `value` is equal to `false`.<br>
208
209 Users can specialize the trait for their own types if they don't want them
210 to be treated like mappings. For example:
211
212 @code
213 namespace boost {
214 namespace json {
215
216 template <>
217 struct is_map_like<your::map> : std::false_type
218 { };
219
220 } // namespace boost
221 } // namespace json
222 @endcode
223
224
225 @note
226
227 The restriction for `t.emplace()` return type ensures that the container
228 does not accept duplicate keys.
229
230 @par Types satisfying the trait
231
232 <a href="https://en.cppreference.com/w/cpp/container/map"><tt>std::map</tt></a>,
233 <a href="https://en.cppreference.com/w/cpp/container/unordered_map"><tt>std::unordered_map</tt></a>.
234
235 @see @ref value_from, @ref value_to
236*/
237template<class T>
238struct is_map_like;

302/** Determine if `T` should be treated as a described class
303
304 Described classes are serialised as objects with an element for each
305 described data member. A described class should not have described
306 bases or non-public members.<br>
307
308 Or more formally, given `L`, a class template
309 of the form `template<class...> struct L {};`, if
310
311 @li <tt>boost::describe::has_members<T, boost::describe::mod_public>::value</tt> is `true`; and
312
313 @li `boost::describe::describe_members<T, boost::describe::mod_private | boost::describe::mod_protected>` denotes `L<>`; and
314
315 @li `boost::describe::describe_bases<T, boost::describe::mod_any_access>` denotes `L<>`; and
316
317 @li <tt>std::is_union<T>::value</tt> is `false`;
318
319 then the trait provides the member constant `value`
320 that is equal to `true`. Otherwise, `value` is equal to `false`.<br>
321
322 Users can specialize the trait for their own types if they don't want them
323 to be treated as described classes. For example:
324
325 @code
326 namespace boost {
327 namespace json {
328
329 template <>
330 struct is_described_class<your::described_class> : std::false_type
331 { };
332
333 } // namespace boost
334 } // namespace json
335 @endcode
336
337 Users can also specialize the trait for their own types _with_ described
338 bases or described non-public data members to enable this conversion
339 implementation. In this case the class will be serialized in a flattened
340 way, that is members of bases will be serialized as direct elements of the
341 object, and no nested objects will be created for bases.
342
343 @see <a href="https://www.boost.org/doc/libs/develop/libs/describe/doc/html/describe.html">Boost.Describe</a>.
344*/
345template<class T>
346struct is_described_class;
347
348/** Determine if `T` should be treated as a described enum
349
350 Described enums are serialised as strings when their value equals to a
351 described enumerator, and as integers otherwise. The reverse operation
352 does not convert numbers to enums values, though, and instead produces
353 an error.<br>
354
355 If <tt>boost::describe::has_describe_enumerators<T>::value</tt> is `true`,
356 then the trait provides the member constant `value`
357 that is equal to `true`. Otherwise, `value` is equal to `false`.<br>
358
359 Users can specialize the trait for their own enums if they don't want them
360 to be treated as described enums. For example:
361
362 @code
363 namespace boost {
364 namespace json {
365
366 template <>
367 struct is_described_enum<your::described_enum> : std::false_type
368 { };
369
370 } // namespace boost
371 } // namespace json
372 @endcode
373
374 @see <a href="https://www.boost.org/doc/libs/develop/libs/describe/doc/html/describe.html">Boost.Describe</a>.
375*/
376template<class T>
377struct is_described_enum;

379/** Determine if `T` should be treated as a variant
380
381 Variants are serialised the same way their active alternative is
382 serialised. The opposite conversion selects the first alternative for which
383 conversion succeeds.<br>
384
385 Given `t`, a glvalue of type ` const T`, if
386 <tt>t.valueless_by_exception()</tt> is well-formed, then the trait provides
387 the member constant `value` that is equal to `true`. Otherwise, `value` is
388 equal to `false`.<br>
389
390 Users can specialize the trait for their own types if they don't want them
391 to be treated as variants. For example:
392
393 @code
394 namespace boost {
395 namespace json {
396
397 template <>
398 struct is_variant_like<your::variant> : std::false_type
399 { };
400
401 } // namespace boost
402 } // namespace json
403 @endcode
404*/
405template<class T>
406struct is_variant_like;

408/** Determine if `T` should be treated as an optional
409
410 Optionals are serialised as `null` if empty, or as the stored type
411 otherwise.<br>
412
413 Given `t`, a glvalue of type `T`, if
414
415 @li <tt>decltype( t.value() )</tt> is well-formed and isn't a void type; and
416
417 @li <tt>t.reset()</tt> is well-formed;
418
419 then the trait provides the member constant `value`
420 that is equal to `true`. Otherwise, `value` is equal to `false`.<br>
421
422 Users can specialize the trait for their own types if they don't want them
423 to be treated as optionals. For example:
424
425 @code
426 namespace boost {
427 namespace json {
428
429 template <>
430 struct is_optional_like<your::optional> : std::false_type
431 { };
432
433 } // namespace boost
434 } // namespace json
435 @endcode
436*/
437template<class T>
438struct is_optional_like;
