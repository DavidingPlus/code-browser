66/** The type of result returned by library functions
67
68 This is an alias template used as the return type for functions that can
69 either return a value, or fail with an error code. This is a brief
70 synopsis of the type:
71
72 @par Declaration
73 @code
74 template< class T >
75 class result
76 {
77 public:
78 // Return true if the result contains an error
79 constexpr bool has_error() const noexcept;
80
81 // These two return true if the result contains a value
82 constexpr bool has_value() const noexcept;
83 constexpr explicit operator bool() const noexcept;
84
85 // Return the value or throw an exception if has_value() == false
86 constexpr T& value();
87 constexpr T const& value() const;
88
89 // Return the value, assuming the result contains it
90 constexpr T& operator*();
91 constexpr T const& operator*() const;
92
93 // Return the error, which is default constructed if has_error() == false
94 constexpr error_code error() const noexcept;
95 ...more
96 };
97 @endcode
98
99 @par Usage
100 Given the function @ref try_value_to with this signature:
101
102 @code
103 template< class T>
104 result< T > try_value_to( const value& jv );
105 @endcode
106
107 The following statement captures the value in a variable upon success,
108 otherwise throws:
109 @code
110 int n = try_value_to<int>( jv ).value();
111 @endcode
112
113 This statement captures the result in a variable and inspects the error
114 condition:
115 @code
116 result< int > r = try_value_to<int>( jv );
117 if( r )
118 std::cout << *r;
119 else
120 std::cout << r.error();
121 @endcode
122
123 @note This alias is deprecated in favor of `boost::system::result`. It is
124 included for backwards compatibility and shouldn't be used in new code. It
125 will be removed completely in version 1.87.0.
126
127 @tparam T The type of value held by the result.
128*/
129template< class T >
130using
131 result
132#ifndef BOOST_MSVC
133 BOOST_DEPRECATED("Use boost::system::result instead")
134#endif
135 = boost::system::result<T>;
