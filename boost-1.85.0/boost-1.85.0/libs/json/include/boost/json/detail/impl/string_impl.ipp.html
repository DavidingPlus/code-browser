23inline
24bool
25ptr_in_range(
26 const char* first,
27 const char* last,
28 const char* ptr) noexcept
29{
30 return std::less<const char*>()(ptr, last) &&
31 std::greater_equal<const char*>()(ptr, first);
32}

34string_impl::
35string_impl() noexcept
36{
37 s_.k = short_string_;
38 s_.buf[sbo_chars_] =
39 static_cast<char>(
40 sbo_chars_);
41 s_.buf[0] = 0;
42}
43
44string_impl::
45string_impl(
46 std::size_t size,
47 storage_ptr const& sp)
48{
49 if(size <= sbo_chars_)
50 {
51 s_.k = short_string_;
52 s_.buf[sbo_chars_] =
53 static_cast<char>(
54 sbo_chars_ - size);
55 s_.buf[size] = 0;
56 }
57 else
58 {
59 s_.k = kind::string;
60 auto const n = growth(
61 new_size: size, capacity: sbo_chars_ + 1);
62 p_.t = ::new(sp->allocate(
63 bytes: sizeof(table) +
64 n + 1,
65 alignment: alignof(table))) table{
66 .size: static_cast<
67 std::uint32_t>(size),
68 .capacity: static_cast<
69 std::uint32_t>(n)};
70 data()[n] = 0;
71 }
72}

74// construct a key, unchecked
75string_impl::
76string_impl(
77 key_t,
78 string_view s,
79 storage_ptr const& sp)
80{
81 BOOST_ASSERT(
82 s.size() <= max_size());
83 k_.k = key_string_;
84 k_.n = static_cast<
85 std::uint32_t>(s.size());
86 k_.s = reinterpret_cast<char*>(
87 sp->allocate(bytes: s.size() + 1,
88 alignment: alignof(char)));
89 k_.s[s.size()] = 0; // null term
90 std::memcpy(dest: &k_.s[0],
91 src: s.data(), n: s.size());
92}

94// construct a key, unchecked
95string_impl::
96string_impl(
97 key_t,
98 string_view s1,
99 string_view s2,
100 storage_ptr const& sp)
101{
102 auto len = s1.size() + s2.size();
103 BOOST_ASSERT(len <= max_size());
104 k_.k = key_string_;
105 k_.n = static_cast<
106 std::uint32_t>(len);
107 k_.s = reinterpret_cast<char*>(
108 sp->allocate(bytes: len + 1,
109 alignment: alignof(char)));
110 k_.s[len] = 0; // null term
111 std::memcpy(dest: &k_.s[0],
112 src: s1.data(), n: s1.size());
113 std::memcpy(dest: &k_.s[s1.size()],
114 src: s2.data(), n: s2.size());
115}

117std::uint32_t
118string_impl::
119growth(
120 std::size_t new_size,
121 std::size_t capacity)
122{
123 if(new_size > max_size())
124 {
125 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
126 detail::throw_system_error( e: error::string_too_large, loc: &loc );
127 }
128 // growth factor 2
129 if( capacity >
130 max_size() - capacity)
131 return static_cast<
132 std::uint32_t>(max_size()); // overflow
133 return static_cast<std::uint32_t>(
134 (std::max)(a: capacity * 2, b: new_size));
135}

137char*
138string_impl::
139assign(
140 std::size_t new_size,
141 storage_ptr const& sp)
142{
143 if(new_size > capacity())
144 {
145 string_impl tmp(growth(
146 new_size,
147 capacity: capacity()), sp);
148 destroy(sp);
149 *this = tmp;
150 }
151 term(n: new_size);
152 return data();
153}

155char*
156string_impl::
157append(
158 std::size_t n,
159 storage_ptr const& sp)
160{
161 if(n > max_size() - size())
162 {
163 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
164 detail::throw_system_error( e: error::string_too_large, loc: &loc );
165 }
166 if(n <= capacity() - size())
167 {
168 term(n: size() + n);
169 return end() - n;
170 }
171 string_impl tmp(growth(
172 new_size: size() + n, capacity: capacity()), sp);
173 std::memcpy(
174 dest: tmp.data(), src: data(), n: size());
175 tmp.term(n: size() + n);
176 destroy(sp);
177 *this = tmp;
178 return end() - n;
179}

181void
182string_impl::
183insert(
184 std::size_t pos,
185 const char* s,
186 std::size_t n,
187 storage_ptr const& sp)
188{
189 const auto curr_size = size();
190 if(pos > curr_size)
191 {
192 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
193 detail::throw_system_error( e: error::out_of_range, loc: &loc );
194 }
195 const auto curr_data = data();
196 if(n <= capacity() - curr_size)
197 {
198 const bool inside = detail::ptr_in_range(first: curr_data, last: curr_data + curr_size, ptr: s);
199 if (!inside || (inside && ((s - curr_data) + n <= pos)))
200 {
201 std::memmove(dest: &curr_data[pos + n], src: &curr_data[pos], n: curr_size - pos + 1);
202 std::memcpy(dest: &curr_data[pos], src: s, n: n);
203 }
204 else
205 {
206 const std::size_t offset = s - curr_data;
207 std::memmove(dest: &curr_data[pos + n], src: &curr_data[pos], n: curr_size - pos + 1);
208 if (offset < pos)
209 {
210 const std::size_t diff = pos - offset;
211 std::memcpy(dest: &curr_data[pos], src: &curr_data[offset], n: diff);
212 std::memcpy(dest: &curr_data[pos + diff], src: &curr_data[pos + n], n: n - diff);
213 }
214 else
215 {
216 std::memcpy(dest: &curr_data[pos], src: &curr_data[offset + n], n: n);
217 }
218 }
219 size(n: curr_size + n);
220 }
221 else
222 {
223 if(n > max_size() - curr_size)
224 {
225 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
226 detail::throw_system_error( e: error::string_too_large, loc: &loc );
227 }
228 string_impl tmp(growth(
229 new_size: curr_size + n, capacity: capacity()), sp);
230 tmp.size(n: curr_size + n);
231 std::memcpy(
232 dest: tmp.data(),
233 src: curr_data,
234 n: pos);
235 std::memcpy(
236 dest: tmp.data() + pos + n,
237 src: curr_data + pos,
238 n: curr_size + 1 - pos);
239 std::memcpy(
240 dest: tmp.data() + pos,
241 src: s,
242 n: n);
243 destroy(sp);
244 *this = tmp;
245 }
246}

248char*
249string_impl::
250insert_unchecked(
251 std::size_t pos,
252 std::size_t n,
253 storage_ptr const& sp)
254{
255 const auto curr_size = size();
256 if(pos > curr_size)
257 {
258 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
259 detail::throw_system_error( e: error::out_of_range, loc: &loc );
260 }
261 const auto curr_data = data();
262 if(n <= capacity() - size())
263 {
264 auto const dest =
265 curr_data + pos;
266 std::memmove(
267 dest: dest + n,
268 src: dest,
269 n: curr_size + 1 - pos);
270 size(n: curr_size + n);
271 return dest;
272 }
273 if(n > max_size() - curr_size)
274 {
275 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
276 detail::throw_system_error( e: error::string_too_large, loc: &loc );
277 }
278 string_impl tmp(growth(
279 new_size: curr_size + n, capacity: capacity()), sp);
280 tmp.size(n: curr_size + n);
281 std::memcpy(
282 dest: tmp.data(),
283 src: curr_data,
284 n: pos);
285 std::memcpy(
286 dest: tmp.data() + pos + n,
287 src: curr_data + pos,
288 n: curr_size + 1 - pos);
289 destroy(sp);
290 *this = tmp;
291 return data() + pos;
292}

294void
295string_impl::
296replace(
297 std::size_t pos,
298 std::size_t n1,
299 const char* s,
300 std::size_t n2,
301 storage_ptr const& sp)
302{
303 const auto curr_size = size();
304 if (pos > curr_size)
305 {
306 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
307 detail::throw_system_error( e: error::out_of_range, loc: &loc );
308 }
309 const auto curr_data = data();
310 n1 = (std::min)(a: n1, b: curr_size - pos);
311 const auto delta = (std::max)(a: n1, b: n2) -
312 (std::min)(a: n1, b: n2);
313 // if we are shrinking in size or we have enough
314 // capacity, dont reallocate
315 if (n1 > n2 || delta <= capacity() - curr_size)
316 {
317 const bool inside = detail::ptr_in_range(first: curr_data, last: curr_data + curr_size, ptr: s);
318 // there is nothing to replace; return
319 if (inside && s == curr_data + pos && n1 == n2)
320 return;
321 if (!inside || (inside && ((s - curr_data) + n2 <= pos)))
322 {
323 // source outside
324 std::memmove(dest: &curr_data[pos + n2], src: &curr_data[pos + n1], n: curr_size - pos - n1 + 1);
325 std::memcpy(dest: &curr_data[pos], src: s, n: n2);
326 }
327 else
328 {
329 // source inside
330 const std::size_t offset = s - curr_data;
331 if (n2 >= n1)
332 {
333 // grow/unchanged
334 const std::size_t diff = offset <= pos + n1 ? (std::min)(a: (pos + n1) - offset, b: n2) : 0;
335 // shift all right of splice point by n2 - n1 to the right
336 std::memmove(dest: &curr_data[pos + n2], src: &curr_data[pos + n1], n: curr_size - pos - n1 + 1);
337 // copy all before splice point
338 std::memmove(dest: &curr_data[pos], src: &curr_data[offset], n: diff);
339 // copy all after splice point
340 std::memmove(dest: &curr_data[pos + diff], src: &curr_data[(offset - n1) + n2 + diff], n: n2 - diff);
341 }
342 else
343 {
344 // shrink
345 // copy all elements into place
346 std::memmove(dest: &curr_data[pos], src: &curr_data[offset], n: n2);
347 // shift all elements after splice point left
348 std::memmove(dest: &curr_data[pos + n2], src: &curr_data[pos + n1], n: curr_size - pos - n1 + 1);
349 }
350 }
351 size(n: (curr_size - n1) + n2);
352 }
353 else
354 {
355 if (delta > max_size() - curr_size)
356 {
357 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
358 detail::throw_system_error( e: error::string_too_large, loc: &loc );
359 }
360 // would exceed capacity, reallocate
361 string_impl tmp(growth(
362 new_size: curr_size + delta, capacity: capacity()), sp);
363 tmp.size(n: curr_size + delta);
364 std::memcpy(
365 dest: tmp.data(),
366 src: curr_data,
367 n: pos);
368 std::memcpy(
369 dest: tmp.data() + pos + n2,
370 src: curr_data + pos + n1,
371 n: curr_size - pos - n1 + 1);
372 std::memcpy(
373 dest: tmp.data() + pos,
374 src: s,
375 n: n2);
376 destroy(sp);
377 *this = tmp;
378 }
379}

381// unlike the replace overload, this function does
382// not move any characters
383char*
384string_impl::
385replace_unchecked(
386 std::size_t pos,
387 std::size_t n1,
388 std::size_t n2,
389 storage_ptr const& sp)
390{
391 const auto curr_size = size();
392 if(pos > curr_size)
393 {
394 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
395 detail::throw_system_error( e: error::out_of_range, loc: &loc );
396 }
397 const auto curr_data = data();
398 const auto delta = (std::max)(a: n1, b: n2) -
399 (std::min)(a: n1, b: n2);
400 // if the size doesn't change, we don't need to
401 // do anything
402 if (!delta)
403 return curr_data + pos;
404 // if we are shrinking in size or we have enough
405 // capacity, dont reallocate
406 if(n1 > n2 || delta <= capacity() - curr_size)
407 {
408 auto const replace_pos = curr_data + pos;
409 std::memmove(
410 dest: replace_pos + n2,
411 src: replace_pos + n1,
412 n: curr_size - pos - n1 + 1);
413 size(n: (curr_size - n1) + n2);
414 return replace_pos;
415 }
416 if(delta > max_size() - curr_size)
417 {
418 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
419 detail::throw_system_error( e: error::string_too_large, loc: &loc );
420 }
421 // would exceed capacity, reallocate
422 string_impl tmp(growth(
423 new_size: curr_size + delta, capacity: capacity()), sp);
424 tmp.size(n: curr_size + delta);
425 std::memcpy(
426 dest: tmp.data(),
427 src: curr_data,
428 n: pos);
429 std::memcpy(
430 dest: tmp.data() + pos + n2,
431 src: curr_data + pos + n1,
432 n: curr_size - pos - n1 + 1);
433 destroy(sp);
434 *this = tmp;
435 return data() + pos;
436}

438void
439string_impl::
440shrink_to_fit(
441 storage_ptr const& sp) noexcept
442{
443 if(s_.k == short_string_)
444 return;
445 auto const t = p_.t;
446 if(t->size <= sbo_chars_)
447 {
448 s_.k = short_string_;
449 std::memcpy(
450 dest: s_.buf, src: data(), n: t->size);
451 s_.buf[sbo_chars_] =
452 static_cast<char>(
453 sbo_chars_ - t->size);
454 s_.buf[t->size] = 0;
455 sp->deallocate(p: t,
456 bytes: sizeof(table) +
457 t->capacity + 1,
458 alignment: alignof(table));
459 return;
460 }
461 if(t->size >= t->capacity)
462 return;
463#ifndef BOOST_NO_EXCEPTIONS
464 try
465 {
466#endif
467 string_impl tmp(t->size, sp);
468 std::memcpy(
469 dest: tmp.data(),
470 src: data(),
471 n: size());
472 destroy(sp);
473 *this = tmp;
474#ifndef BOOST_NO_EXCEPTIONS
475 }
476 catch(std::exception const&)
477 {
478 // eat the exception
479 }
480#endif
481}
