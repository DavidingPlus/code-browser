17namespace boost {
18namespace json {
19namespace detail {
20
21class stack
22{
23 storage_ptr sp_;
24 std::size_t cap_ = 0;
25 std::size_t size_ = 0;
26 unsigned char* base_ = nullptr;
27 unsigned char* buf_ = nullptr;
28
29public:
30 BOOST_JSON_DECL
31 ~stack();
32
33 stack() = default;
34
35 stack(
36 storage_ptr sp,
37 unsigned char* buf,
38 std::size_t buf_size) noexcept;
39
40 bool
41 empty() const noexcept
42 {
43 return size_ == 0;
44 }
45
46 void
47 clear() noexcept
48 {
49 size_ = 0;
50 }
51
52 BOOST_JSON_DECL
53 void
54 reserve(std::size_t n);
55
56 template<class T>
57 void
58 push(T const& t)
59 {
60 auto const n = sizeof(T);
61 // If this assert goes off, it
62 // means the calling code did not
63 // reserve enough to prevent a
64 // reallocation.
65 //BOOST_ASSERT(cap_ >= size_ + n);
66 reserve(n: size_ + n);
67 std::memcpy(
68 dest: base_ + size_, src: &t, n: n);
69 size_ += n;
70 }
71
72 template<class T>
73 void
74 push_unchecked(T const& t)
75 {
76 auto const n = sizeof(T);
77 BOOST_ASSERT(size_ + n <= cap_);
78 std::memcpy(
79 dest: base_ + size_, src: &t, n: n);
80 size_ += n;
81 }
82
83 template<class T>
84 void
85 peek(T& t)
86 {
87 auto const n = sizeof(T);
88 BOOST_ASSERT(size_ >= n);
89 std::memcpy(dest: &t,
90 src: base_ + size_ - n, n: n);
91 }
92
93 template<class T>
94 void
95 pop(T& t)
96 {
97 auto const n = sizeof(T);
98 BOOST_ASSERT(size_ >= n);
99 size_ -= n;
100 std::memcpy(
101 dest: &t, src: base_ + size_, n: n);
102 }
103};
