11#ifdef __has_include
12#if __has_include(<version>)
13#include <version>
14#endif
15#endif
16
17#if defined(__cpp_lib_bit_cast) && __cpp_lib_bit_cast >= 201806L
18# define BOOST_JSON_HAS_BIT_CAST
19#endif
20
21#if defined(__cpp_lib_is_constant_evaluated) && __cpp_lib_is_constant_evaluated >= 201811L

23#endif
24
25// Testing for relevant C++20 constexpr library features
26#if defined(BOOST_JSON_HAS_IS_CONSTANT_EVALUATED) \
27 && defined(BOOST_JSON_HAS_BIT_CAST) \
28 && __cpp_lib_constexpr_algorithms >= 201806L /*For std::copy and std::fill*/

30#else
31#define BOOST_JSON_FASTFLOAT_CONSTEXPR20
32#endif
