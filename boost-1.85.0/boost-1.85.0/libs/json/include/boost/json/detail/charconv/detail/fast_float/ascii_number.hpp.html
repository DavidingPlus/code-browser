20// Next function can be micro-optimized, but compilers are entirely
21// able to optimize it well.
22template <typename UC>
23BOOST_FORCEINLINE constexpr bool is_integer(UC c) noexcept {
24 return !(c > UC('9') || c < UC('0'));
25}

27BOOST_FORCEINLINE constexpr uint64_t byteswap(uint64_t val) {
28 return (val & 0xFF00000000000000) >> 56
29 | (val & 0x00FF000000000000) >> 40
30 | (val & 0x0000FF0000000000) >> 24
31 | (val & 0x000000FF00000000) >> 8
32 | (val & 0x00000000FF000000) << 8
33 | (val & 0x0000000000FF0000) << 24
34 | (val & 0x000000000000FF00) << 40
35 | (val & 0x00000000000000FF) << 56;
36}

69BOOST_FORCEINLINE BOOST_JSON_CXX14_CONSTEXPR_NO_INLINE
70uint32_t parse_eight_digits_unrolled(uint64_t val) {
71 constexpr uint64_t mask = 0x000000FF000000FF;
72 constexpr uint64_t mul1 = 0x000F424000000064; // 100 + (1000000ULL << 32)
73 constexpr uint64_t mul2 = 0x0000271000000001; // 1 + (10000ULL << 32)
74 val -= 0x3030303030303030;
75 val = (val * 10) + (val >> 8); // val = (val * 2561) >> 8;
76 val = (((val & mask) * mul1) + (((val >> 16) & mask) * mul2)) >> 32;
77 return uint32_t(val);
78}

96BOOST_FORCEINLINE constexpr bool is_made_of_eight_digits_fast(uint64_t val) noexcept {
97 return !((((val + 0x4646464646464646) | (val - 0x3030303030303030)) & 0x8080808080808080));
98}

115template <typename UC>
116struct parsed_number_string_t {
117 int64_t exponent{0};
118 uint64_t mantissa{0};
119 UC const * lastmatch{nullptr};
120 bool negative{false};
121 bool valid{false};
122 bool too_many_digits{false};
123 // contains the range of the significant digits
124 span<const UC> integer{}; // non-nullable
125 span<const UC> fraction{}; // nullable
126};
127using byte_span = span<char>;
128using parsed_number_string = parsed_number_string_t<char>;
129// Assuming that you use no more than 19 digits, this will
130// parse an ASCII string.
131template <typename UC>
132BOOST_FORCEINLINE BOOST_JSON_FASTFLOAT_CONSTEXPR20
133parsed_number_string_t<UC> parse_number_string(UC const *p, UC const * pend, parse_options_t<UC> options) noexcept {
134 chars_format const fmt = options.format;
135 UC const decimal_point = options.decimal_point;
136
137 parsed_number_string_t<UC> answer;
138 answer.valid = false;
139 answer.too_many_digits = false;
140 answer.negative = (*p == UC('-'));
141 if (*p == UC('-')) // C++17 20.19.3.(7.1) explicitly forbids '+' sign here
142 {
143 ++p;
144 if (p == pend) {
145 return answer;
146 }
147 if (!is_integer(*p) && (*p != decimal_point)) { // a sign must be followed by an integer or the dot
148 return answer;
149 }
150 }
151 UC const * const start_digits = p;
152
153 uint64_t i = 0; // an unsigned int avoids signed overflows (which are bad)
154
155 while ((p != pend) && is_integer(*p)) {
156 // a multiplication by 10 is cheaper than an arbitrary integer
157 // multiplication
158 i = 10 * i +
159 uint64_t(*p - UC('0')); // might overflow, we will handle the overflow later
160 ++p;
161 }
162 UC const * const end_of_integer_part = p;
163 int64_t digit_count = int64_t(end_of_integer_part - start_digits);
164 answer.integer = span<const UC>(start_digits, size_t(digit_count));
165 int64_t exponent = 0;
166 if ((p != pend) && (*p == decimal_point)) {
167 ++p;
168 UC const * before = p;
169 // can occur at most twice without overflowing, but let it occur more, since
170 // for integers with many digits, digit parsing is the primary bottleneck.
171 if (std::is_same<UC,char>::value) {
172 while ((std::distance(p, pend) >= 8) && is_made_of_eight_digits_fast(p)) {
173 i = i * 100000000 + parse_eight_digits_unrolled(p); // in rare cases, this will overflow, but that's ok
174 p += 8;
175 }
176 }
177 while ((p != pend) && is_integer(*p)) {
178 uint8_t digit = uint8_t(*p - UC('0'));
179 ++p;
180 i = i * 10 + digit; // in rare cases, this will overflow, but that's ok
181 }
182 exponent = before - p;
183 answer.fraction = span<const UC>(before, size_t(p - before));
184 digit_count -= exponent;
185 }
186 // we must have encountered at least one integer!
187 if (digit_count == 0) {
188 return answer;
189 }
190 int64_t exp_number = 0; // explicit exponential part
191 if (((unsigned)fmt & (unsigned)chars_format::scientific) && (p != pend) && ((UC('e') == *p) || (UC('E') == *p))) {
192 UC const * location_of_e = p;
193 ++p;
194 bool neg_exp = false;
195 if ((p != pend) && (UC('-') == *p)) {
196 neg_exp = true;
197 ++p;
198 } else if ((p != pend) && (UC('+') == *p)) { // '+' on exponent is allowed by C++17 20.19.3.(7.1)
199 ++p;
200 }
201 if ((p == pend) || !is_integer(*p)) {
202 if(!((unsigned)fmt & (unsigned)chars_format::fixed)) {
203 // We are in error.
204 return answer;
205 }
206 // Otherwise, we will be ignoring the 'e'.
207 p = location_of_e;
208 } else {
209 while ((p != pend) && is_integer(*p)) {
210 uint8_t digit = uint8_t(*p - UC('0'));
211 if (exp_number < 0x10000000) {
212 exp_number = 10 * exp_number + digit;
213 }
214 ++p;
215 }
216 if(neg_exp) { exp_number = - exp_number; }
217 exponent += exp_number;
218 }
219 } else {
220 // If it scientific and not fixed, we have to bail out.
221 if(((unsigned)fmt & (unsigned)chars_format::scientific) && !((unsigned)fmt & (unsigned)chars_format::fixed))
222 {
223 return answer;
224 }
225 }
226 answer.lastmatch = p;
227 answer.valid = true;
228
229 // If we frequently had to deal with long strings of digits,
230 // we could extend our code by using a 128-bit integer instead
231 // of a 64-bit integer. However, this is uncommon.
232 //
233 // We can deal with up to 19 digits.
234 if (digit_count > 19) { // this is uncommon
235 // It is possible that the integer had an overflow.
236 // We have to handle the case where we have 0.0000somenumber.
237 // We need to be mindful of the case where we only have zeroes...
238 // E.g., 0.000000000...000.
239 UC const * start = start_digits;
240 while ((start != pend) && (*start == UC('0') || *start == decimal_point)) {
241 if(*start == UC('0')) { digit_count --; }
242 start++;
243 }
244 if (digit_count > 19) {
245 answer.too_many_digits = true;
246 // Let us start again, this time, avoiding overflows.
247 // We don't need to check if is_integer, since we use the
248 // pre-tokenized spans from above.
249 i = 0;
250 p = answer.integer.ptr;
251 UC const * int_end = p + answer.integer.len();
252 constexpr uint64_t minimal_nineteen_digit_integer{1000000000000000000};
253 while((i < minimal_nineteen_digit_integer) && (p != int_end)) {
254 i = i * 10 + uint64_t(*p - UC('0'));
255 ++p;
256 }
257 if (i >= minimal_nineteen_digit_integer) { // We have a big integers
258 exponent = end_of_integer_part - p + exp_number;
259 } else { // We have a value with a fractional component.
260 p = answer.fraction.ptr;
261 UC const * frac_end = p + answer.fraction.len();
262 while((i < minimal_nineteen_digit_integer) && (p != frac_end)) {
263 i = i * 10 + uint64_t(*p - UC('0'));
264 ++p;
265 }
266 exponent = answer.fraction.ptr - p + exp_number;
267 }
268 // We have now corrected both exponent and i, to a truncated value
269 }
270 }
271 answer.exponent = exponent;
272 answer.mantissa = i;
273 return answer;
274}
