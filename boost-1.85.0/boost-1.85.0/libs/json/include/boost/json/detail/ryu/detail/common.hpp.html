43inline uint32_t decimalLength9(const uint32_t v) {
44 // Function precondition: v is not a 10-digit number.
45 // (f2s: 9 digits are sufficient for round-tripping.)
46 // (d2fixed: We print 9-digit blocks.)
47 BOOST_ASSERT(v < 1000000000);
48 if (v >= 100000000) { return 9; }
49 if (v >= 10000000) { return 8; }
50 if (v >= 1000000) { return 7; }
51 if (v >= 100000) { return 6; }
52 if (v >= 10000) { return 5; }
53 if (v >= 1000) { return 4; }
54 if (v >= 100) { return 3; }
55 if (v >= 10) { return 2; }
56 return 1;
57}

59// Returns e == 0 ? 1 : ceil(log_2(5^e)).
60inline int32_t pow5bits(const int32_t e) {
61 // This approximation works up to the point that the multiplication overflows at e = 3529.
62 // If the multiplication were done in 64 bits, it would fail at 5^4004 which is just greater
63 // than 2^9297.
64 BOOST_ASSERT(e >= 0);
65 BOOST_ASSERT(e <= 3528);
66 return (int32_t) (((((uint32_t) e) * 1217359) >> 19) + 1);
67}
68
69// Returns floor(log_10(2^e)).
70inline uint32_t log10Pow2(const int32_t e) {
71 // The first value this approximation fails for is 2^1651 which is just greater than 10^297.
72 BOOST_ASSERT(e >= 0);
73 BOOST_ASSERT(e <= 1650);
74 return (((uint32_t) e) * 78913) >> 18;
75}
76
77// Returns floor(log_10(5^e)).
78inline uint32_t log10Pow5(const int32_t e) {
79 // The first value this approximation fails for is 5^2621 which is just greater than 10^1832.
80 BOOST_ASSERT(e >= 0);
81 BOOST_ASSERT(e <= 2620);
82 return (((uint32_t) e) * 732923) >> 20;
83}

85inline int copy_special_str(char * const result, const bool sign, const bool exponent, const bool mantissa) {
86 if (mantissa) {
87 memcpy(dest: result, src: "NaN", n: 3);
88 return 3;
89 }
90 if (sign) {
91 result[0] = '-';
92 }
93 if (exponent) {
94 memcpy(dest: result + sign, src: "Infinity", n: 8);
95 return sign + 8;
96 }
97 memcpy(dest: result + sign, src: "0E0", n: 3);
98 return sign + 3;
99}

101inline
102int
103copy_special_str_conforming(
104 char* const result, bool sign, bool exponent, bool mantissa)
105{
106 if (mantissa)
107 {
108 memcpy(dest: result, src: "null", n: 4);
109 return 4;
110 }
111
112 if (sign)
113 result[0] = '-';
114
115 if (exponent)
116 {
117 memcpy(dest: result + sign, src: "1e99999", n: 7);
118 return sign + 7;
119 }
120
121 memcpy(dest: result + sign, src: "0E0", n: 3);
122 return sign + 3;
123}
