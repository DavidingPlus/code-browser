19#ifndef BOOST_JSON_WEAK_CONSTINIT

31// this is here so that ~memory_resource
32// is emitted in the library instead of
33// the user's TU.
34default_resource::
35~default_resource() = default;
36
37void*
38default_resource::
39do_allocate(
40 std::size_t n,
41 std::size_t)
42{
43 return ::operator new(n);
44}
45
46void
47default_resource::
48do_deallocate(
49 void* p,
50 std::size_t,
51 std::size_t)
52{
53 ::operator delete(p);
54}
55
56bool
57default_resource::
58do_is_equal(
59 memory_resource const& mr) const noexcept
60{
61 return this == &mr;
62}
