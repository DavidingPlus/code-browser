17unchecked_array::
18~unchecked_array()
19{
20 if(! data_ ||
21 sp_.is_not_shared_and_deallocate_is_trivial())
22 return;
23 for(unsigned long i = 0;
24 i < size_; ++i)
25 data_[i].~value();
26}
27
28void
29unchecked_array::
30relocate(value* dest) noexcept
31{
32 if(size_ > 0)
33 std::memcpy(
34 dest: static_cast<void*>(dest),
35 src: data_, n: size_ * sizeof(value));
36 data_ = nullptr;
37}
