29template<class T>
30using has_reserve_member_helper = decltype(std::declval<T&>().reserve(0));
31template<class T>
32using has_reserve_member = mp11::mp_valid<has_reserve_member_helper, T>;
33template<class T>
34using reserve_implementation = mp11::mp_cond<
35 is_tuple_like<T>, mp11::mp_int<2>,
36 has_reserve_member<T>, mp11::mp_int<1>,
37 mp11::mp_true, mp11::mp_int<0>>;
38
39template<class T>
40error
41try_reserve(
42 T&,
43 std::size_t size,
44 mp11::mp_int<2>)
45{
46 constexpr std::size_t N = std::tuple_size<remove_cvref<T>>::value;
47 if ( N != size )
48 return error::size_mismatch;
49 return error();
50}
51
52template<typename T>
53error
54try_reserve(
55 T& cont,
56 std::size_t size,
57 mp11::mp_int<1>)
58{
59 cont.reserve(size);
60 return error();
61}
62
63template<typename T>
64error
65try_reserve(
66 T&,
67 std::size_t,
68 mp11::mp_int<0>)
69{
70 return error();
71}

205// map-like containers
206template< class T, class Ctx >
207system::result<T>
208value_to_impl(
209 map_like_conversion_tag,
210 try_value_to_tag<T>,
211 value const& jv,
212 Ctx const& ctx )
213{
214 object const* obj = jv.if_object();
215 if( !obj )
216 {
217 system::error_code ec;
218 BOOST_JSON_FAIL(ec, error::not_object);
219 return {boost::system::in_place_error, ec};
220 }
221
222 T res;
223 error const e = detail::try_reserve(
224 res, obj->size(), reserve_implementation<T>());
225 if( e != error() )
226 {
227 system::error_code ec;
228 BOOST_JSON_FAIL( ec, e );
229 return {boost::system::in_place_error, ec};
230 }
231
232 auto ins = detail::inserter(res, inserter_implementation<T>());
233 for( key_value_pair const& kv: *obj )
234 {
235 auto elem_res = try_value_to<mapped_type<T>>( kv.value(), ctx );
236 if( elem_res.has_error() )
237 return {boost::system::in_place_error, elem_res.error()};
238 *ins++ = value_type<T>{
239 key_type<T>(kv.key()),
240 std::move(*elem_res)};
241 }
242 return res;
243}

245// all other containers
246template< class T, class Ctx >
247system::result<T>
248value_to_impl(
249 sequence_conversion_tag,
250 try_value_to_tag<T>,
251 value const& jv,
252 Ctx const& ctx )
253{
254 array const* arr = jv.if_array();
255 if( !arr )
256 {
257 system::error_code ec;
258 BOOST_JSON_FAIL(ec, error::not_array);
259 return {boost::system::in_place_error, ec};
260 }
261
262 T result;
263 error const e = detail::try_reserve(
264 result, arr->size(), reserve_implementation<T>());
265 if( e != error() )
266 {
267 system::error_code ec;
268 BOOST_JSON_FAIL( ec, e );
269 return {boost::system::in_place_error, ec};
270 }
271
272 auto ins = detail::inserter(result, inserter_implementation<T>());
273 for( value const& val: *arr )
274 {
275 auto elem_res = try_value_to<value_type<T>>( val, ctx );
276 if( elem_res.has_error() )
277 return {boost::system::in_place_error, elem_res.error()};
278 *ins++ = std::move(*elem_res);
279 }
280 return result;
281}

284template< class T, class Ctx >
285system::result<T>
286try_make_tuple_elem(value const& jv, Ctx const& ctx, system::error_code& ec)
287{
288 if( ec.failed() )
289 return {boost::system::in_place_error, ec};
290
291 auto result = try_value_to<T>( jv, ctx );
292 ec = result.error();
293 return result;
294}

296template <class T, class Ctx, std::size_t... Is>
297system::result<T>
298try_make_tuple_like(
299 array const& arr, Ctx const& ctx, boost::mp11::index_sequence<Is...>)
300{
301 system::error_code ec;
302 auto items = std::make_tuple(
303 try_make_tuple_elem<
304 typename std::decay<tuple_element_t<Is, T>>::type >(
305 arr[Is], ctx, ec)
306 ...);
307 if( ec.failed() )
308 return {boost::system::in_place_error, ec};
309
310 return {
311 boost::system::in_place_value, T(std::move(*std::get<Is>(items))...)};
312}

314template< class T, class Ctx >
315system::result<T>
316value_to_impl(
317 tuple_conversion_tag,
318 try_value_to_tag<T>,
319 value const& jv,
320 Ctx const& ctx )
321{
322 system::error_code ec;
323
324 array const* arr = jv.if_array();
325 if( !arr )
326 {
327 BOOST_JSON_FAIL(ec, error::not_array);
328 return {boost::system::in_place_error, ec};
329 }
330
331 constexpr std::size_t N = std::tuple_size<remove_cvref<T>>::value;
332 if( N != arr->size() )
333 {
334 BOOST_JSON_FAIL(ec, error::size_mismatch);
335 return {boost::system::in_place_error, ec};
336 }
337
338 return try_make_tuple_like<T>(
339 *arr, ctx, boost::mp11::make_index_sequence<N>());
340}

342template< class Ctx, class T, bool non_throwing = true >
343struct to_described_member
344{
345 using Ds = described_members<T>;
346
347 using result_type = mp11::mp_eval_if_c<
348 !non_throwing, T, system::result, T>;
349
350 result_type& res;
351 object const& obj;
352 std::size_t count;
353 Ctx const& ctx;
354
355 template< class I >
356 void
357 operator()(I)
358 {
359 if( !res )
360 return;
361
362 using D = mp11::mp_at<Ds, I>;
363 using M = described_member_t<T, D>;
364
365 auto const found = obj.find(D::name);
366 if( found == obj.end() )
367 {
368 BOOST_IF_CONSTEXPR( !is_optional_like<M>::value )
369 {
370 system::error_code ec;
371 BOOST_JSON_FAIL(ec, error::unknown_name);
372 res = {boost::system::in_place_error, ec};
373 }
374 return;
375 }
376
377#if defined(__GNUC__) && BOOST_GCC_VERSION >= 80000 && BOOST_GCC_VERSION < 11000
378# pragma GCC diagnostic push
379# pragma GCC diagnostic ignored "-Wunused"
380# pragma GCC diagnostic ignored "-Wunused-variable"
381#endif
382 auto member_res = try_value_to<M>( found->value(), ctx );
383#if defined(__GNUC__) && BOOST_GCC_VERSION >= 80000 && BOOST_GCC_VERSION < 11000
384# pragma GCC diagnostic pop
385#endif
386 if( member_res )
387 {
388 (*res).* D::pointer = std::move(*member_res);
389 ++count;
390 }
391 else
392 res = {boost::system::in_place_error, member_res.error()};
393 }
394};

396// described classes
397template< class T, class Ctx >
398system::result<T>
399value_to_impl(
400 described_class_conversion_tag,
401 try_value_to_tag<T>,
402 value const& jv,
403 Ctx const& ctx )
404{
405 BOOST_STATIC_ASSERT( std::is_default_constructible<T>::value );
406 system::result<T> res;
407
408 auto* obj = jv.if_object();
409 if( !obj )
410 {
411 system::error_code ec;
412 BOOST_JSON_FAIL(ec, error::not_object);
413 res = {boost::system::in_place_error, ec};
414 return res;
415 }
416
417 to_described_member< Ctx, T > member_converter{ res, *obj, 0u, ctx };
418
419 using Ds = typename decltype(member_converter)::Ds;
420 constexpr std::size_t N = mp11::mp_size<Ds>::value;
421 mp11::mp_for_each< mp11::mp_iota_c<N> >(member_converter);
422
423 if( !res )
424 return res;
425
426 if( member_converter.count != obj->size() )
427 {
428 system::error_code ec;
429 BOOST_JSON_FAIL(ec, error::size_mismatch);
430 res = {boost::system::in_place_error, ec};
431 return res;
432 }
433
434 return res;
435}

437// described enums
438template< class T, class Ctx >
439system::result<T>
440value_to_impl(
441 described_enum_conversion_tag,
442 try_value_to_tag<T>,
443 value const& jv,
444 Ctx const& )
445{
446 T val = {};
447 (void)jv;
448#ifdef BOOST_DESCRIBE_CXX14
449 system::error_code ec;
450
451 auto str = jv.if_string();
452 if( !str )
453 {
454 BOOST_JSON_FAIL(ec, error::not_string);
455 return {system::in_place_error, ec};
456 }
457
458 if( !describe::enum_from_string(str->data(), val) )
459 {
460 BOOST_JSON_FAIL(ec, error::unknown_name);
461 return {system::in_place_error, ec};
462 }
463#endif
464
465 return {system::in_place_value, val};
466}

486using variant_construction_category = mp11::mp_cond<
487 std::is_constructible< T, variant2::in_place_index_t<I::value>, V >,
488 mp11::mp_int<2>,
489#ifndef BOOST_NO_CXX17_HDR_VARIANT

492#endif // BOOST_NO_CXX17_HDR_VARIANT

571template< class T, class Ctx >
572struct alternative_converter
573{
574 system::result<T>& res;
575 value const& jv;
576 Ctx const& ctx;
577
578 template< class I >
579 void operator()( I ) const
580 {
581 if( res )
582 return;
583
584 auto&& local_ctx = make_locally_nonthrowing_context(ctx);
585 using V = mp11::mp_at<T, I>;
586 auto attempt = try_value_to<V>(jv, local_ctx);
587 if( attempt )
588 {
589 using cat = variant_construction_category<T, V, I>;
590 res = initialize_variant<T, I>( std::move(*attempt), cat() );
591 }
592 }
593};

595template< class T, class Ctx >
596system::result<T>
597value_to_impl(
598 variant_conversion_tag,
599 try_value_to_tag<T>,
600 value const& jv,
601 Ctx const& ctx)
602{
603 system::error_code ec;
604 BOOST_JSON_FAIL(ec, error::exhausted_variants);
605
606 using Is = mp11::mp_iota< mp11::mp_size<T> >;
607
608 system::result<T> res = {system::in_place_error, ec};
609 mp11::mp_for_each<Is>( alternative_converter<T, Ctx>{res, jv, ctx} );
610 return res;
611}

814template< class T, class... Args >
815system::result<T>
816wrap_conversion_exceptions( std::true_type, value_to_tag<T>, Args&& ... args )
817{
818 return {
819 boost::system::in_place_value,
820 tag_invoke( value_to_tag<T>(), static_cast<Args&&>(args)... )};
821}
822
823template< class T, class... Args >
824system::result<T>
825wrap_conversion_exceptions( std::false_type, value_to_tag<T>, Args&& ... args )
826{
827#ifndef BOOST_NO_EXCEPTIONS
828 try
829 {
830#endif
831 return wrap_conversion_exceptions(
832 std::true_type(),
833 value_to_tag<T>(),
834 static_cast<Args&&>(args)... );
835#ifndef BOOST_NO_EXCEPTIONS
836 }
837 catch( std::bad_alloc const&)
838 {
839 throw;
840 }
841 catch( system::system_error const& e)
842 {
843 return {boost::system::in_place_error, e.code()};
844 }
845 catch( ... )
846 {
847 system::error_code ec;
848 BOOST_JSON_FAIL(ec, error::exception);
849 return {boost::system::in_place_error, ec};
850 }
851#endif
852}
