24template<int N>
25std::uint32_t
26load_little_endian(void const* p)
27{
28 std::uint32_t v = 0;
29 std::memcpy(dest: &v, src: p, n: N);
30 endian::little_to_native_inplace(x&: v);
31 return v;
32}

34inline
35uint16_t
36classify_utf8(char c)
37{
38 // 0x000 = invalid
39 // 0x102 = 2 bytes, second byte [80, BF]
40 // 0x203 = 3 bytes, second byte [A0, BF]
41 // 0x303 = 3 bytes, second byte [80, BF]
42 // 0x403 = 3 bytes, second byte [80, 9F]
43 // 0x504 = 4 bytes, second byte [90, BF]
44 // 0x604 = 4 bytes, second byte [80, BF]
45 // 0x704 = 4 bytes, second byte [80, 8F]
46 static constexpr uint16_t first[128]
47 {
48 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
49 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
50 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
51 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
52 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
53 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
54 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
55 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
56
57 0x000, 0x000, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102,
58 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102,
59 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102,
60 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102, 0x102,
61 0x203, 0x303, 0x303, 0x303, 0x303, 0x303, 0x303, 0x303,
62 0x303, 0x303, 0x303, 0x303, 0x303, 0x403, 0x303, 0x303,
63 0x504, 0x604, 0x604, 0x604, 0x704, 0x000, 0x000, 0x000,
64 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000, 0x000,
65 };
66 return first[static_cast<unsigned char>(c & 0x7F)];
67}

69inline
70bool
71is_valid_utf8(const char* p, uint16_t first)
72{
73 uint32_t v;
74 switch(first >> 8)
75 {
76 default:
77 return false;
78
79 // 2 bytes, second byte [80, BF]
80 case 1:
81 v = load_little_endian<2>(p);
82 return (v & 0xC000) == 0x8000;
83
84 // 3 bytes, second byte [A0, BF]
85 case 2:
86 v = load_little_endian<3>(p);
87 return (v & 0xC0E000) == 0x80A000;
88
89 // 3 bytes, second byte [80, BF]
90 case 3:
91 v = load_little_endian<3>(p);
92 return (v & 0xC0C000) == 0x808000;
93
94 // 3 bytes, second byte [80, 9F]
95 case 4:
96 v = load_little_endian<3>(p);
97 return (v & 0xC0E000) == 0x808000;
98
99 // 4 bytes, second byte [90, BF]
100 case 5:
101 v = load_little_endian<4>(p);
102 return (v & 0xC0C0FF00) + 0x7F7F7000 <= 0x2F00;
103
104 // 4 bytes, second byte [80, BF]
105 case 6:
106 v = load_little_endian<4>(p);
107 return (v & 0xC0C0C000) == 0x80808000;
108
109 // 4 bytes, second byte [80, 8F]
110 case 7:
111 v = load_little_endian<4>(p);
112 return (v & 0xC0C0F000) == 0x80808000;
113 }
114}

116class utf8_sequence
117{
118 char seq_[4];
119 uint16_t first_;
120 uint8_t size_;
121
122public:
123 void
124 save(
125 const char* p,
126 std::size_t remain) noexcept
127 {
128 first_ = classify_utf8(c: *p );
129 if(remain >= length())
130 size_ = length();
131 else
132 size_ = static_cast<uint8_t>(remain);
133 std::memcpy(dest: seq_, src: p, n: size_);
134 }
135
136 uint8_t
137 length() const noexcept
138 {
139 return first_ & 0xFF;
140 }
141
142 bool
143 complete() const noexcept
144 {
145 return size_ >= length();
146 }
147
148 // returns true if complete
149 bool
150 append(
151 const char* p,
152 std::size_t remain) noexcept
153 {
154 if(BOOST_JSON_UNLIKELY(needed() == 0))
155 return true;
156 if(BOOST_JSON_LIKELY(remain >= needed()))
157 {
158 std::memcpy(
159 dest: seq_ + size_, src: p, n: needed());
160 size_ = length();
161 return true;
162 }
163 if(BOOST_JSON_LIKELY(remain > 0))
164 {
165 std::memcpy(dest: seq_ + size_, src: p, n: remain);
166 size_ += static_cast<uint8_t>(remain);
167 }
168 return false;
169 }
170
171 const char*
172 data() const noexcept
173 {
174 return seq_;
175 }
176
177 uint8_t
178 needed() const noexcept
179 {
180 return length() - size_;
181 }
182
183 bool
184 valid() const noexcept
185 {
186 BOOST_ASSERT(size_ >= length());
187 return is_valid_utf8(p: seq_, first: first_);
188 }
189};
