27stack::

39void
40stack::
41reserve(std::size_t n)
42{
43 if(cap_ >= n)
44 return;
45 auto const base = static_cast<
46 unsigned char*>(sp_->allocate(bytes: n));
47 if(base_)
48 {
49 if(size_ > 0)
50 std::memcpy(dest: base, src: base_, n: size_);
51 if(base_ != buf_)
52 sp_->deallocate(p: base_, bytes: cap_);
53 }
54 base_ = base;
55 cap_ = n;
56}
