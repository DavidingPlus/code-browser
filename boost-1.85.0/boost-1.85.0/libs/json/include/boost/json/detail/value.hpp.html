165struct scalar
166{
167 storage_ptr sp; // must come first
168 kind k; // must come second
169 union
170 {
171 bool b;
172 std::int64_t i;
173 std::uint64_t u;
174 double d;
175 };
176
177 explicit
178 scalar(storage_ptr sp_ = {}) noexcept
179 : sp(std::move(sp_))
180 , k(json::kind::null)
181 {
182 }
183
184 explicit
185 scalar(bool b_,
186 storage_ptr sp_ = {}) noexcept
187 : sp(std::move(sp_))
188 , k(json::kind::bool_)
189 , b(b_)
190 {
191 }
192
193 explicit
194 scalar(std::int64_t i_,
195 storage_ptr sp_ = {}) noexcept
196 : sp(std::move(sp_))
197 , k(json::kind::int64)
198 , i(i_)
199 {
200 }
201
202 explicit
203 scalar(std::uint64_t u_,
204 storage_ptr sp_ = {}) noexcept
205 : sp(std::move(sp_))
206 , k(json::kind::uint64)
207 , u(u_)
208 {
209 }
210

212 scalar(double d_,
213 storage_ptr sp_ = {}) noexcept
214 : sp(std::move(sp_))
215 , k(json::kind::double_)
216 , d(d_)
217 {
218 }

221struct access
222{
223 template<class Value, class... Args>
224 static
225 Value&
226 construct_value(Value* p, Args&&... args)
227 {
228 return *reinterpret_cast<
229 Value*>(::new(p) Value(
230 std::forward<Args>(args)...));
231 }
232
233 template<class KeyValuePair, class... Args>
234 static
235 KeyValuePair&
236 construct_key_value_pair(
237 KeyValuePair* p, Args&&... args)
238 {
239 return *reinterpret_cast<
240 KeyValuePair*>(::new(p)
241 KeyValuePair(
242 std::forward<Args>(args)...));
243 }
244
245 template<class Value>
246 static
247 char const*
248 release_key(
249 Value& jv,
250 std::size_t& len) noexcept
251 {
252 BOOST_ASSERT(jv.is_string());
253 jv.str_.sp_.~storage_ptr();
254 return jv.str_.impl_.release_key(len);
255 }

267 template<class KeyValuePair>
268 static
269 index_t const&
270 next(KeyValuePair const& e) noexcept
271 {
272 return e.next_;
273 }
