33template<bool AllowBadUTF8>
34inline
35const char*
36count_valid(
37 char const* p,
38 const char* end) noexcept
39{
40 __m128i const q1 = _mm_set1_epi8( b: '\x22' ); // '"'
41 __m128i const q2 = _mm_set1_epi8( b: '\\' ); // '\\'
42 __m128i const q3 = _mm_set1_epi8( b: 0x1F );
43
44 while(end - p >= 16)
45 {
46 __m128i v1 = _mm_loadu_si128( p: (__m128i const*)p );
47 __m128i v2 = _mm_cmpeq_epi8( a: v1, b: q1 ); // quote
48 __m128i v3 = _mm_cmpeq_epi8( a: v1, b: q2 ); // backslash
49 __m128i v4 = _mm_or_si128( a: v2, b: v3 ); // combine quotes and backslash
50 __m128i v5 = _mm_min_epu8( a: v1, b: q3 );
51 __m128i v6 = _mm_cmpeq_epi8( a: v5, b: v1 ); // controls
52 __m128i v7 = _mm_or_si128( a: v4, b: v6 ); // combine with control
53
54 int w = _mm_movemask_epi8( a: v7 );
55
56 if( w != 0 )
57 {
58 int m;
59#if defined(__GNUC__) || defined(__clang__)
60 m = __builtin_ffs( w ) - 1;
61#else
62 unsigned long index;
63 _BitScanForward( &index, w );
64 m = index;
65#endif
66 return p + m;
67 }
68
69 p += 16;
70 }
71
72 while(p != end)
73 {
74 const unsigned char c = *p;
75 if(c == '\x22' || c == '\\' || c < 0x20)
76 break;
77 ++p;
78 }
79
80 return p;
81}

83template<>
84inline
85const char*
86count_valid<false>(
87 char const* p,
88 const char* end) noexcept
89{
90 __m128i const q1 = _mm_set1_epi8( b: '\x22' ); // '"'
91 __m128i const q2 = _mm_set1_epi8( b: '\\' );
92 __m128i const q3 = _mm_set1_epi8( b: 0x20 );
93
94 while(end - p >= 16)
95 {
96 __m128i v1 = _mm_loadu_si128( p: (__m128i const*)p );
97
98 __m128i v2 = _mm_cmpeq_epi8( a: v1, b: q1 );
99 __m128i v3 = _mm_cmpeq_epi8( a: v1, b: q2 );
100 __m128i v4 = _mm_cmplt_epi8( a: v1, b: q3 );
101
102 __m128i v5 = _mm_or_si128( a: v2, b: v3 );
103 __m128i v6 = _mm_or_si128( a: v5, b: v4 );
104
105 int w = _mm_movemask_epi8( a: v6 );
106
107 if( w != 0 )
108 {
109 int m;
110#if defined(__GNUC__) || defined(__clang__)
111 m = __builtin_ffs( w ) - 1;
112#else
113 unsigned long index;
114 _BitScanForward( &index, w );
115 m = index;
116#endif
117 p += m;
118 break;
119 }
120
121 p += 16;
122 }
123
124 while(p != end)
125 {
126 const unsigned char c = *p;
127 if(c == '\x22' || c == '\\' || c < 0x20)
128 break;
129 if(c < 0x80)
130 {
131 ++p;
132 continue;
133 }
134 // validate utf-8
135 uint16_t first = classify_utf8(c);
136 uint8_t len = first & 0xFF;
137 if(BOOST_JSON_UNLIKELY(end - p < len))
138 break;
139 if(BOOST_JSON_UNLIKELY(! is_valid_utf8(p, first)))
140 break;
141 p += len;
142 }
143
144 return p;
145}

267// assumes p..p+15 are valid
268inline int count_digits( char const* p ) noexcept
269{
270 __m128i v1 = _mm_loadu_si128( p: (__m128i const*)p );
271 v1 = _mm_add_epi8(a: v1, b: _mm_set1_epi8(b: 70));
272 v1 = _mm_cmplt_epi8(a: v1, b: _mm_set1_epi8(b: 118));
273
274 int m = _mm_movemask_epi8(a: v1);
275
276 int n;
277
278 if( m == 0 )
279 {
280 n = 16;
281 }
282 else
283 {
284#if defined(__GNUC__) || defined(__clang__)
285 n = __builtin_ffs( m ) - 1;
286#else
287 unsigned long index;
288 _BitScanForward( &index, m );
289 n = static_cast<int>(index);
290#endif
291 }
292
293 return n;
294}

298// assumes p..p+15 are valid
299inline int count_digits( char const* p ) noexcept
300{
301 int n = 0;
302
303 for( ; n < 16; ++n )
304 {
305 unsigned char const d = *p++ - '0';
306 if(d > 9) break;
307 }
308
309 return n;
310}

314// parse_unsigned
315
316inline uint64_t parse_unsigned( uint64_t r, char const * p, std::size_t n ) noexcept
317{
318 while( n >= 4 )
319 {
320 // faster on on clang for x86,
321 // slower on gcc
322#ifdef __clang__
323 r = r * 10 + p[0] - '0';
324 r = r * 10 + p[1] - '0';
325 r = r * 10 + p[2] - '0';
326 r = r * 10 + p[3] - '0';
327#else
328 uint32_t v;
329 std::memcpy( &v, p, 4 );
330 endian::native_to_little_inplace(v);
331
332 v -= 0x30303030;
333
334 unsigned w0 = v & 0xFF;
335 unsigned w1 = (v >> 8) & 0xFF;
336 unsigned w2 = (v >> 16) & 0xFF;
337 unsigned w3 = (v >> 24);
338
339 r = (((r * 10 + w0) * 10 + w1) * 10 + w2) * 10 + w3;
340#endif
341 p += 4;
342 n -= 4;
343 }
344
345 switch( n )
346 {
347 case 0:
348 break;
349 case 1:
350 r = r * 10 + p[0] - '0';
351 break;
352 case 2:
353 r = r * 10 + p[0] - '0';
354 r = r * 10 + p[1] - '0';
355 break;
356 case 3:
357 r = r * 10 + p[0] - '0';
358 r = r * 10 + p[1] - '0';
359 r = r * 10 + p[2] - '0';
360 break;
361 }
362 return r;
363}

417// count_whitespace
418
419#ifdef BOOST_JSON_USE_SSE2
420
421inline const char* count_whitespace( char const* p, const char* end ) noexcept
422{
423 if( p == end )
424 {
425 return p;
426 }
427
428 if( static_cast<unsigned char>( *p ) > 0x20 )
429 {
430 return p;
431 }
432
433 __m128i const q1 = _mm_set1_epi8( b: ' ' );
434 __m128i const q2 = _mm_set1_epi8( b: '\n' );
435 __m128i const q3 = _mm_set1_epi8( b: 4 ); // '\t' | 4 == '\r'
436 __m128i const q4 = _mm_set1_epi8( b: '\r' );
437
438 while( end - p >= 16 )
439 {
440 __m128i v0 = _mm_loadu_si128( p: (__m128i const*)p );
441
442 __m128i w0 = _mm_or_si128(
443 a: _mm_cmpeq_epi8( a: v0, b: q1 ),
444 b: _mm_cmpeq_epi8( a: v0, b: q2 ));
445 __m128i v1 = _mm_or_si128( a: v0, b: q3 );
446 __m128i w1 = _mm_cmpeq_epi8( a: v1, b: q4 );
447 __m128i w2 = _mm_or_si128( a: w0, b: w1 );
448
449 int m = _mm_movemask_epi8( a: w2 ) ^ 0xFFFF;
450
451 if( m != 0 )
452 {
453#if defined(__GNUC__) || defined(__clang__)
454 std::size_t c = __builtin_ffs( m ) - 1;
455#else
456 unsigned long index;
457 _BitScanForward( &index, m );
458 std::size_t c = index;
459#endif
460
461 p += c;
462 return p;
463 }
464
465 p += 16;
466 }
467
468 while( p != end )
469 {
470 if( *p != ' ' && *p != '\t' && *p != '\r' && *p != '\n' )
471 {
472 return p;
473 }
474
475 ++p;
476 }
477
478 return p;
479}
