10#ifndef BOOST_JSON_KIND_HPP
11#define BOOST_JSON_KIND_HPP

13#include <boost/json/detail/config.hpp>
14#include <boost/json/string_view.hpp>
15#include <iosfwd>

17namespace boost {
18namespace json {
19
20/** Constants for identifying the type of a value
21
22 These values are returned from @ref value::kind
23*/
24// Order matters
25enum class kind : unsigned char
26{
27 /// The null value.
28 null,
29
30 /// A `bool`.
31 bool_,
32
33 /// A `std::int64_t`
34 int64,
35
36 /// A `std::uint64_t`
37 uint64,
38
39 /// A `double`.
40 double_,
41
42 /// A @ref string.
43 string,
44
45 /// An @ref array.
46 array,
47
48 /// An @ref object.
49 object
50};
51
52/** Return a string representing a kind.
53
54 This provides a human-readable string
55 representing a @ref kind. This may be
56 useful for diagnostics.
57
58 @returns The string.
59
60 @param k The kind.
61*/
62BOOST_JSON_DECL
63string_view
64to_string(kind k) noexcept;
65
66/** Format a kind to an output stream.
67
68 This allows a @ref kind to be formatted as
69 a string, typically for diagnostics.
70
71 @returns The output stream.
72
73 @param os The output stream to format to.
74
75 @param k The kind to format.
76*/
77BOOST_JSON_DECL
78std::ostream&
79operator<<(std::ostream& os, kind k);
80
81/** A tag type used to select a @ref value constructor overload.
82
83 The library provides the constant @ref array_kind
84 which may be used to select the @ref value constructor
85 that creates an empty @ref array.
86
87 @see @ref array_kind
88*/
89struct array_kind_t
90{
91};
92
93/** A tag type used to select a @ref value constructor overload.
94
95 The library provides the constant @ref object_kind
96 which may be used to select the @ref value constructor
97 that creates an empty @ref object.
98
99 @see @ref object_kind
100*/
101struct object_kind_t
102{
103};
104
105/** A tag type used to select a @ref value constructor overload.
106
107 The library provides the constant @ref string_kind
108 which may be used to select the @ref value constructor
109 that creates an empty @ref string.
110
111 @see @ref string_kind
112*/
113struct string_kind_t
114{
115};

117/** A constant used to select a @ref value constructor overload.
118
119 The library provides this constant to allow efficient
120 construction of a @ref value containing an empty @ref array.
121
122 @par Example
123 @code
124 storage_ptr sp;
125 value jv( array_kind, sp ); // sp is an optional parameter
126 @endcode
127
128 @see @ref array_kind_t
129*/
130BOOST_JSON_INLINE_VARIABLE(array_kind, array_kind_t);
131
