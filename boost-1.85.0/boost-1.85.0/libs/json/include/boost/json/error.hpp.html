19/** Error codes returned by JSON operations
20
21*/
22enum class error
23{
24 //
25 // parse errors
26 //
27
28 /// syntax error
29 syntax = 1,
30
31 /// extra data
32 extra_data,
33
34 /// incomplete JSON
35 incomplete,
36
37 /// exponent too large
38 exponent_overflow,
39
40 /// too deep
41 too_deep,
42
43 /// illegal leading surrogate
44 illegal_leading_surrogate,
45
46 /// illegal trailing surrogate
47 illegal_trailing_surrogate,
48
49 /// expected hex digit
50 expected_hex_digit,
51
52 /// expected utf16 escape
53 expected_utf16_escape,
54
55 /// An object contains too many elements
56 object_too_large,
57
58 /// An array contains too many elements
59 array_too_large,
60
61 /// A key is too large
62 key_too_large,
63
64 /// A string is too large
65 string_too_large,
66
67 /// A number is too large
68 number_too_large,
69
70 /// error occured when trying to read input
71 input_error,
72
73 //
74 // generic errors
75 //
76
77 /// An exception was thrown during operation
78 exception,
79
80 /// A requested element is outside of container's range
81 out_of_range,
82
83 /// test failure
84 test_failure,
85
86 //
87 // JSON Pointer errors
88 //
89
90 /// missing slash character before token reference
91 missing_slash,
92
93 /// invalid escape sequence
94 invalid_escape,
95
96 /// token should be a number but cannot be parsed as such
97 token_not_number,
98
99 /// current value is neither an object nor an array
100 value_is_scalar,
101
102 /// current value does not contain referenced value
103 not_found,
104
105 /// token cannot be represented by std::size_t
106 token_overflow,
107
108 /// past-the-end index is not supported
109 past_the_end,
110
111 //
112 // Conversion errors
113 //
114
115 /// JSON number was expected during conversion
116 not_number,
117
118 /// number cast is not exact
119 not_exact,
120
121 /// JSON null was expected during conversion
122 not_null,
123
124 /// JSON bool was expected during conversion
125 not_bool,
126
127 /// JSON array was expected during conversion
128 not_array,
129
130 /// JSON object was expected during conversion
131 not_object,
132
133 /// JSON string was expected during conversion
134 not_string,
135
136 /// std::int64_t was expected during conversion
137 not_int64,
138
139 /// std::uint64_t was expected during conversion
140 not_uint64,
141
142 /// `double` was expected during conversion
143 not_double,
144
145 /// JSON integer was expected during conversion
146 not_integer,
147
148 /// source composite has size incompatible with target
149 size_mismatch,
150
151 /// none of the possible conversions were successful
152 exhausted_variants,
153
154 /// the key does not correspond to a known name
155 unknown_name,
156};
157
158/** Error conditions corresponding to JSON errors
159*/
160enum class condition
161{
162 /// A parser-related error
163 parse_error = 1,
164
165 /// An error related to parsing JSON pointer string
166 pointer_parse_error,
167
168 /// An error related to applying JSON pointer string to a value
169 pointer_use_error,
170
171 /// A conversion error
172 conversion_error,
173
174 /// A generic error
175 generic_error,
176};
