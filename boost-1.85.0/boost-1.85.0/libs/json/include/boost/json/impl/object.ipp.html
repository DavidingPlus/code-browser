29template<class CharRange>
30std::pair<key_value_pair*, std::size_t>
31find_in_object(
32 object const& obj,
33 CharRange key) noexcept
34{
35 BOOST_ASSERT(obj.t_->capacity > 0);
36 if(obj.t_->is_small())
37 {
38 auto it = &(*obj.t_)[0];
39 auto const last =
40 &(*obj.t_)[obj.t_->size];
41 for(;it != last; ++it)
42 if( key == it->key() )
43 return { it, 0 };
44 return { nullptr, 0 };
45 }
46 std::pair<
47 key_value_pair*,
48 std::size_t> result;
49 BOOST_ASSERT(obj.t_->salt != 0);
50 result.second = detail::digest(key.begin(), key.end(), obj.t_->salt);
51 auto i = obj.t_->bucket(
52 hash: result.second);
53 while(i != object::null_index_)
54 {
55 auto& v = (*obj.t_)[i];
56 if( key == v.key() )
57 {
58 result.first = &v;
59 return result;
60 }
61 i = access::next(e&: v);
62 }
63 result.first = nullptr;
64 return result;
65}

124object::table*
125object::table::
126allocate(
127 std::size_t capacity,
128 std::uintptr_t salt,
129 storage_ptr const& sp)
130{
131 BOOST_STATIC_ASSERT(
132 alignof(key_value_pair) >=
133 alignof(index_t));
134 BOOST_ASSERT(capacity > 0);
135 BOOST_ASSERT(capacity <= max_size());
136 table* p;
137 if(capacity <= detail::small_object_size_)
138 {
139 p = reinterpret_cast<
140 table*>(sp->allocate(
141 bytes: sizeof(table) + capacity *
142 sizeof(key_value_pair)));
143 p->capacity = static_cast<
144 std::uint32_t>(capacity);
145 }
146 else
147 {
148 p = reinterpret_cast<
149 table*>(sp->allocate(
150 bytes: sizeof(table) + capacity * (
151 sizeof(key_value_pair) +
152 sizeof(index_t))));
153 p->capacity = static_cast<
154 std::uint32_t>(capacity);
155 p->clear();
156 }
157 if(salt)
158 {
159 p->salt = salt;
160 }
161 else
162 {
163 // VFALCO This would be better if it
164 // was random, but maybe this
165 // is good enough.
166 p->salt = reinterpret_cast<
167 std::uintptr_t>(p);
168 }
169 return p;
170}

200object::
201object(detail::unchecked_object&& uo)
202 : sp_(uo.storage())
203{
204 if(uo.size() == 0)
205 {
206 t_ = &empty_;
207 return;
208 }
209 // should already be checked
210 BOOST_ASSERT(
211 uo.size() <= max_size());
212 t_ = table::allocate(
213 capacity: uo.size(), salt: 0, sp: sp_);
214
215 // insert all elements, keeping
216 // the last of any duplicate keys.
217 auto dest = begin();
218 auto src = uo.release();
219 auto const end = src + 2 * uo.size();
220 if(t_->is_small())
221 {
222 t_->size = 0;
223 while(src != end)
224 {
225 access::construct_key_value_pair(
226 p: dest, args: pilfer(t&: src[0]), args: pilfer(t&: src[1]));
227 src += 2;
228 auto result = detail::find_in_object(obj: *this, key: dest->key());
229 if(! result.first)
230 {
231 ++dest;
232 ++t_->size;
233 continue;
234 }
235 // handle duplicate
236 auto& v = *result.first;
237 // don't bother to check if
238 // storage deallocate is trivial
239 v.~key_value_pair();
240 // trivial relocate
241 std::memcpy(
242 dest: static_cast<void*>(&v),
243 src: dest, n: sizeof(v));
244 }
245 return;
246 }
247 while(src != end)
248 {
249 access::construct_key_value_pair(
250 p: dest, args: pilfer(t&: src[0]), args: pilfer(t&: src[1]));
251 src += 2;
252 auto& head = t_->bucket(key: dest->key());
253 auto i = head;
254 for(;;)
255 {
256 if(i == null_index_)
257 {
258 // end of bucket
259 access::next(
260 e&: *dest) = head;
261 head = static_cast<index_t>(
262 dest - begin());
263 ++dest;
264 break;
265 }
266 auto& v = (*t_)[i];
267 if(v.key() != dest->key())
268 {
269 i = access::next(e&: v);
270 continue;
271 }
272
273 // handle duplicate
274 access::next(e&: *dest) =
275 access::next(e&: v);
276 // don't bother to check if
277 // storage deallocate is trivial
278 v.~key_value_pair();
279 // trivial relocate
280 std::memcpy(
281 dest: static_cast<void*>(&v),
282 src: dest, n: sizeof(v));
283 break;
284 }
285 }
286 t_->size = static_cast<
287 index_t>(dest - begin());
288}

335object::
336object(
337 object const& other,
338 storage_ptr sp)
339 : sp_(std::move(sp))
340 , t_(&empty_)
341{
342 reserve(new_capacity: other.size());
343 revert_construct r(*this);
344 if(t_->is_small())
345 {
346 for(auto const& v : other)
347 {
348 ::new(end())
349 key_value_pair(v, sp_);
350 ++t_->size;
351 }
352 r.commit();
353 return;
354 }
355 for(auto const& v : other)
356 {
357 // skip duplicate checking
358 auto& head =
359 t_->bucket(key: v.key());
360 auto pv = ::new(end())
361 key_value_pair(v, sp_);
362 access::next(e&: *pv) = head;
363 head = t_->size;
364 ++t_->size;
365 }
366 r.commit();
367}

443void
444object::
445insert(
446 std::initializer_list<std::pair<
447 string_view, value_ref>> init)
448{
449 auto const n0 = size();
450 if(init.size() > max_size() - n0)
451 {
452 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
453 detail::throw_system_error( e: error::object_too_large, loc: &loc );
454 }
455 revert_insert r( *this, n0 + init.size() );
456 if(t_->is_small())
457 {
458 for(auto& iv : init)
459 {
460 auto result =
461 detail::find_in_object(obj: *this, key: iv.first);
462 if(result.first)
463 {
464 // ignore duplicate
465 continue;
466 }
467 ::new(end()) key_value_pair(
468 iv.first,
469 iv.second.make_value(sp: sp_));
470 ++t_->size;
471 }
472 r.commit();
473 return;
474 }
475 for(auto& iv : init)
476 {
477 auto& head = t_->bucket(key: iv.first);
478 auto i = head;
479 for(;;)
480 {
481 if(i == null_index_)
482 {
483 // VFALCO value_ref should construct
484 // a key_value_pair using placement
485 auto& v = *::new(end())
486 key_value_pair(
487 iv.first,
488 iv.second.make_value(sp: sp_));
489 access::next(e&: v) = head;
490 head = static_cast<index_t>(
491 t_->size);
492 ++t_->size;
493 break;
494 }
495 auto& v = (*t_)[i];
496 if(v.key() == iv.first)
497 {
498 // ignore duplicate
499 break;
500 }
501 i = access::next(e&: v);
502 }
503 }
504 r.commit();
505}

570void
571object::
572swap(object& other)
573{
574 if(*sp_ == *other.sp_)
575 {
576 t_ = detail::exchange(
577 t&: other.t_, u: t_);
578 return;
579 }
580 object temp1(
581 std::move(*this),
582 other.storage());
583 object temp2(
584 std::move(other),
585 this->storage());
586 other.~object();
587 ::new(&other) object(pilfer(t&: temp1));
588 this->~object();
589 ::new(this) object(pilfer(t&: temp2));
590}

685key_value_pair*
686object::
687insert_impl(
688 pilfered<key_value_pair> p,
689 std::size_t hash)
690{
691 BOOST_ASSERT(
692 capacity() > size());
693 if(t_->is_small())
694 {
695 auto const pv = ::new(end())
696 key_value_pair(p);
697 ++t_->size;
698 return pv;
699 }
700 auto& head =
701 t_->bucket(hash);
702 auto const pv = ::new(end())
703 key_value_pair(p);
704 access::next(e&: *pv) = head;
705 head = t_->size;
706 ++t_->size;
707 return pv;
708}

710// allocate new table, copy elements there, and rehash them
711object::table*
712object::
713reserve_impl(std::size_t new_capacity)
714{
715 BOOST_ASSERT(
716 new_capacity > t_->capacity);
717 auto t = table::allocate(
718 capacity: growth(new_size: new_capacity),
719 salt: t_->salt, sp: sp_);
720 if(! empty())
721 std::memcpy(
722 dest: static_cast<
723 void*>(&(*t)[0]),
724 src: begin(),
725 n: size() * sizeof(
726 key_value_pair));
727 t->size = t_->size;
728 std::swap(a&: t_, b&: t);
729
730 if(! t_->is_small())
731 {
732 // rebuild hash table,
733 // without dup checks
734 auto p = end();
735 index_t i = t_->size;
736 while(i-- > 0)
737 {
738 --p;
739 auto& head =
740 t_->bucket(key: p->key());
741 access::next(e&: *p) = head;
742 head = i;
743 }
744 }
745
746 return t;
747}

749bool
750object::
751equal(object const& other) const noexcept
752{
753 if(size() != other.size())
754 return false;
755 auto const end_ = other.end();
756 for(auto e : *this)
757 {
758 auto it = other.find(key: e.key());
759 if(it == end_)
760 return false;
761 if(it->value() != e.value())
762 return false;
763 }
764 return true;
765}

767std::size_t
768object::
769growth(
770 std::size_t new_size) const
771{
772 if(new_size > max_size())
773 {
774 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
775 detail::throw_system_error( e: error::object_too_large, loc: &loc );
776 }
777 std::size_t const old = capacity();
778 if(old > max_size() - old / 2)
779 return new_size;
780 std::size_t const g =
781 old + old / 2; // 1.5x
782 if(g < new_size)
783 return new_size;
784 return g;
785}

787void
788object::
789remove(
790 index_t& head,
791 key_value_pair& v) noexcept
792{
793 BOOST_ASSERT(! t_->is_small());
794 auto const i = static_cast<
795 index_t>(&v - begin());
796 if(head == i)
797 {
798 head = access::next(e&: v);
799 return;
800 }
801 auto* pn =
802 &access::next(e&: (*t_)[head]);
803 while(*pn != i)
804 pn = &access::next(e&: (*t_)[*pn]);
805 *pn = access::next(e&: v);
806}

829template<class FS, class FB>
830auto
831object::
832do_erase(
833 const_iterator pos,
834 FS small_reloc,
835 FB big_reloc) noexcept
836 -> iterator
837{
838 auto p = begin() + (pos - begin());
839 if(t_->is_small())
840 {
841 p->~value_type();
842 --t_->size;
843 if(p != end())
844 {
845 small_reloc(p);
846 }
847 return p;
848 }
849 remove(head&: t_->bucket(key: p->key()), v&: *p);
850 p->~value_type();
851 --t_->size;
852 if(p != end())
853 {
854 big_reloc(p);
855 }
856 return p;
857}

859void
860object::
861reindex_relocate(
862 key_value_pair* src,
863 key_value_pair* dst) noexcept
864{
865 BOOST_ASSERT(! t_->is_small());
866 auto& head = t_->bucket(key: src->key());
867 remove(head, v&: *src);
868 // the casts silence warnings
869 std::memcpy(
870 dest: static_cast<void*>(dst),
871 src: static_cast<void const*>(src),
872 n: sizeof(*dst));
873 access::next(e&: *dst) = head;
874 head = static_cast<
875 index_t>(dst - begin());
876}
