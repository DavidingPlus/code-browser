24struct alignas(value)
25 array::table
26{
27 std::uint32_t size = 0;
28 std::uint32_t capacity = 0;
29
30 constexpr table();
31
32 value&
33 operator[](std::size_t pos) noexcept
34 {
35 return (reinterpret_cast<

37 }

42 allocate(
43 std::size_t capacity,
44 storage_ptr const& sp);

56class array::revert_construct
57{
58 array* arr_;
59
60public:
61 explicit
62 revert_construct(
63 array& arr) noexcept
64 : arr_(&arr)
65 {
66 }
67
68 ~revert_construct()
69 {
70 if(! arr_)
71 return;
72 arr_->destroy();
73 }
74
75 void
76 commit() noexcept
77 {
78 arr_ = nullptr;
79 }
80};

84class array::revert_insert
85{
86 array* arr_;
87 std::size_t const i_;
88 std::size_t const n_;
89
90public:
91 value* p;
92
93 BOOST_JSON_DECL
94 revert_insert(
95 const_iterator pos,
96 std::size_t n,
97 array& arr);
98
99 BOOST_JSON_DECL
100 ~revert_insert();
101
102 value*
103 commit() noexcept
104 {
105 auto it =
106 arr_->data() + i_;
107 arr_ = nullptr;
108 return it;
109 }
110};

114void
115array::
116relocate(
117 value* dest,
118 value* src,
119 std::size_t n) noexcept
120{
121 if(n == 0)
122 return;
123 std::memmove(
124 dest: static_cast<void*>(dest),
125 src: static_cast<void const*>(src),
126 n: n * sizeof(value));
127}

495template<class InputIt>
496array::
497array(
498 InputIt first, InputIt last,
499 storage_ptr sp,
500 std::input_iterator_tag)
501 : sp_(std::move(sp))
502 , t_(&empty_)
503{
504 revert_construct r(*this);
505 while(first != last)
506 {
507 reserve(new_capacity: size() + 1);
508 ::new(end()) value(
509 *first++, sp_);
510 ++t_->size;
511 }
512 r.commit();
513}
514
515template<class InputIt>
516array::
517array(
518 InputIt first, InputIt last,
519 storage_ptr sp,
520 std::forward_iterator_tag)
521 : sp_(std::move(sp))
522{
523 std::size_t n =
524 std::distance(first, last);
525 if( n == 0 )
526 {
527 t_ = &empty_;
528 return;
529 }
530
531 t_ = table::allocate(capacity: n, sp: sp_);
532 t_->size = 0;
533 revert_construct r(*this);
534 while(n--)
535 {
536 ::new(end()) value(
537 *first++, sp_);
538 ++t_->size;
539 }
540 r.commit();
541}

543template<class InputIt>
544auto
545array::
546insert(
547 const_iterator pos,
548 InputIt first, InputIt last,
549 std::input_iterator_tag) ->
550 iterator
551{
552 BOOST_ASSERT(
553 pos >= begin() && pos <= end());
554 if(first == last)
555 return data() + (pos - data());
556 array temp(first, last, sp_);
557 revert_insert r(
558 pos, temp.size(), *this);
559 relocate(
560 dest: r.p,
561 src: temp.data(),
562 n: temp.size());
563 temp.t_->size = 0;
564 return r.commit();
565}
566
567template<class InputIt>
568auto
569array::
570insert(
571 const_iterator pos,
572 InputIt first, InputIt last,
573 std::forward_iterator_tag) ->
574 iterator
575{
576 std::size_t n =
577 std::distance(first, last);
578 revert_insert r(pos, n, *this);
579 while(n--)
580 {
581 ::new(r.p) value(*first++);
582 ++r.p;
583 }
584 return r.commit();
585}
