33auto
34array::
35table::
36allocate(
37 std::size_t capacity,
38 storage_ptr const& sp) ->
39 table*
40{
41 BOOST_ASSERT(capacity > 0);
42 if(capacity > array::max_size())
43 {
44 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
45 detail::throw_system_error( e: error::array_too_large, loc: &loc );
46 }
47 auto p = reinterpret_cast<
48 table*>(sp->allocate(
49 bytes: sizeof(table) +
50 capacity * sizeof(value),
51 alignment: alignof(value)));
52 p->capacity = static_cast<
53 std::uint32_t>(capacity);
54 return p;
55}

57void
58array::
59table::
60deallocate(
61 table* p,
62 storage_ptr const& sp)
63{
64 if(p->capacity == 0)
65 return;
66 sp->deallocate(p,
67 bytes: sizeof(table) +
68 p->capacity * sizeof(value),
69 alignment: alignof(value));
70}

74array::
75revert_insert::
76revert_insert(
77 const_iterator pos,
78 std::size_t n,
79 array& arr)
80 : arr_(&arr)
81 , i_(pos - arr_->data())
82 , n_(n)
83{
84 BOOST_ASSERT(
85 pos >= arr_->begin() &&
86 pos <= arr_->end());
87 if( n_ <= arr_->capacity() -
88 arr_->size())
89 {
90 // fast path
91 p = arr_->data() + i_;
92 if(n_ == 0)
93 return;
94 relocate(
95 dest: p + n_,
96 src: p,
97 n: arr_->size() - i_);
98 arr_->t_->size = static_cast<
99 std::uint32_t>(
100 arr_->t_->size + n_);
101 return;
102 }
103 if(n_ > max_size() - arr_->size())
104 {
105 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
106 detail::throw_system_error( e: error::array_too_large, loc: &loc );
107 }
108 auto t = table::allocate(
109 capacity: arr_->growth(new_size: arr_->size() + n_),
110 sp: arr_->sp_);
111 t->size = static_cast<std::uint32_t>(
112 arr_->size() + n_);
113 p = &(*t)[0] + i_;
114 relocate(
115 dest: &(*t)[0],
116 src: arr_->data(),
117 n: i_);
118 relocate(
119 dest: &(*t)[i_ + n_],
120 src: arr_->data() + i_,
121 n: arr_->size() - i_);
122 t = detail::exchange(t&: arr_->t_, u: t);
123 table::deallocate(p: t, sp: arr_->sp_);
124}
125
126array::
127revert_insert::
128~revert_insert()
129{
130 if(! arr_)
131 return;
132 BOOST_ASSERT(n_ != 0);
133 auto const pos =
134 arr_->data() + i_;
135 arr_->destroy(first: pos, last: p);
136 arr_->t_->size = static_cast<
137 std::uint32_t>(
138 arr_->t_->size - n_);
139 relocate(
140 dest: pos,
141 src: pos + n_,
142 n: arr_->size() - i_);
143}

147void
148array::
149destroy(
150 value* first, value* last) noexcept
151{
152 if(sp_.is_not_shared_and_deallocate_is_trivial())
153 return;
154 while(last-- != first)
155 last->~value();
156}
157
158void
159array::
160destroy() noexcept
161{
162 if(sp_.is_not_shared_and_deallocate_is_trivial())
163 return;
164 auto last = end();
165 auto const first = begin();
166 while(last-- != first)
167 last->~value();
168 table::deallocate(p: t_, sp: sp_);
169}

177array::
178array(detail::unchecked_array&& ua)
179 : sp_(ua.storage())
180{
181 BOOST_STATIC_ASSERT(
182 alignof(table) == alignof(value));
183 if(ua.size() == 0)

188 t_= table::allocate(

190 t_->size = static_cast<
191 std::uint32_t>(ua.size());
192 ua.relocate(dest: data());
193}

201array::
202array(
203 std::size_t count,
204 value const& v,
205 storage_ptr sp)
206 : sp_(std::move(sp))
207{
208 if(count == 0)
209 {
210 t_ = &empty_;
211 return;

213 t_= table::allocate(
214 capacity: count, sp: sp_);
215 t_->size = 0;
216 revert_construct r(*this);

282array::
283array(
284 array&& other,
285 storage_ptr sp)
286 : sp_(std::move(sp))
287{
288 if(*sp_ == *other.sp_)
289 {
290 // same resource
291 t_ = detail::exchange(
292 t&: other.t_, u: &empty_);
293 return;
294 }
295 else if(other.empty())
296 {
297 t_ = &empty_;
298 return;
299 }
300 // copy
301 t_ = table::allocate(
302 capacity: other.size(), sp: sp_);
303 t_->size = 0;
304 revert_construct r(*this);
305 auto src = other.data();
306 auto dest = data();
307 auto const n = other.size();
308 do
309 {
310 ::new(dest++) value(
311 *src++, sp_);
312 ++t_->size;
313 }
314 while(t_->size < n);
315 r.commit();
316}

377void
378array::
379shrink_to_fit() noexcept
380{
381 if(capacity() <= size())
382 return;
383 if(size() == 0)
384 {
385 table::deallocate(p: t_, sp: sp_);
386 t_ = &empty_;
387 return;
388 }
389
390#ifndef BOOST_NO_EXCEPTIONS
391 try
392 {
393#endif
394 auto t = table::allocate(
395 capacity: size(), sp: sp_);
396 relocate(
397 dest: &(*t)[0],
398 src: data(),
399 n: size());
400 t->size = static_cast<
401 std::uint32_t>(size());
402 t = detail::exchange(
403 t&: t_, u: t);
404 table::deallocate(p: t, sp: sp_);
405#ifndef BOOST_NO_EXCEPTIONS
406 }
407 catch(...)
408 {
409 // eat the exception
410 return;
411 }
412#endif
413}

497auto
498array::
499erase(
500 const_iterator first,
501 const_iterator last) noexcept ->
502 iterator
503{
504 BOOST_ASSERT(
505 first >= begin() &&
506 last >= first &&
507 last <= end());
508 std::size_t const n =
509 last - first;
510 auto const p = &(*t_)[0] +
511 (first - &(*t_)[0]);
512 destroy(first: p, last: p + n);
513 relocate(dest: p, src: p + n,
514 n: t_->size - (last -
515 &(*t_)[0]));
516 t_->size = static_cast<
517 std::uint32_t>(t_->size - n);
518 return p;
519}

544void
545array::
546resize(std::size_t count)
547{
548 if(count <= t_->size)
549 {
550 // shrink
551 destroy(
552 first: &(*t_)[0] + count,
553 last: &(*t_)[0] + t_->size);
554 t_->size = static_cast<
555 std::uint32_t>(count);
556 return;
557 }
558
559 reserve(new_capacity: count);
560 auto p = &(*t_)[t_->size];
561 auto const end = &(*t_)[count];
562 while(p != end)
563 ::new(p++) value(sp_);
564 t_->size = static_cast<
565 std::uint32_t>(count);
566}

568void
569array::
570resize(
571 std::size_t count,
572 value const& v)
573{
574 if(count <= size())
575 {
576 // shrink
577 destroy(
578 first: data() + count,
579 last: data() + size());
580 t_->size = static_cast<
581 std::uint32_t>(count);
582 return;
583 }
584 count -= size();
585 revert_insert r(
586 end(), count, *this);
587 while(count--)
588 {
589 ::new(r.p) value(v, sp_);
590 ++r.p;
591 }
592 r.commit();
593}

595void
596array::
597swap(array& other)
598{
599 if(*sp_ == *other.sp_)
600 {
601 t_ = detail::exchange(
602 t&: other.t_, u: t_);
603 return;
604 }
605 array temp1(
606 std::move(*this),
607 other.storage());
608 array temp2(
609 std::move(other),
610 this->storage());
611 this->~array();
612 ::new(this) array(
613 pilfer(t&: temp2));
614 other.~array();
615 ::new(&other) array(
616 pilfer(t&: temp1));
617}

625std::size_t
626array::
627growth(
628 std::size_t new_size) const
629{
630 if(new_size > max_size())
631 {
632 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
633 detail::throw_system_error( e: error::array_too_large, loc: &loc );
634 }
635 std::size_t const old = capacity();
636 if(old > max_size() - old / 2)
637 return new_size;
638 std::size_t const g =
639 old + old / 2; // 1.5x
640 if(g < new_size)
641 return new_size;
642 return g;
643}

645// precondition: new_capacity > capacity()
646void
647array::
648reserve_impl(
649 std::size_t new_capacity)
650{
651 BOOST_ASSERT(
652 new_capacity > t_->capacity);
653 auto t = table::allocate(
654 capacity: growth(new_size: new_capacity), sp: sp_);
655 relocate(
656 dest: &(*t)[0],
657 src: &(*t_)[0],
658 n: t_->size);
659 t->size = t_->size;
660 t = detail::exchange(t&: t_, u: t);
661 table::deallocate(p: t, sp: sp_);
662}

664// precondition: pv is not aliased
665value&
666array::
667push_back(
668 pilfered<value> pv)
669{
670 auto const n = t_->size;
671 if(n < t_->capacity)
672 {
673 // fast path
674 auto& v = *::new(
675 &(*t_)[n]) value(pv);
676 ++t_->size;
677 return v;
678 }
679 auto const t =
680 detail::exchange(t&: t_,
681 u: table::allocate(
682 capacity: growth(new_size: n + 1),
683 sp: sp_));
684 auto& v = *::new(
685 &(*t_)[n]) value(pv);
686 relocate(
687 dest: &(*t_)[0],
688 src: &(*t)[0],
689 n);
690 t_->size = n + 1;
691 table::deallocate(p: t, sp: sp_);
692 return v;
693}

695// precondition: pv is not aliased
696auto
697array::
698insert(
699 const_iterator pos,
700 pilfered<value> pv) ->
701 iterator
702{
703 BOOST_ASSERT(
704 pos >= begin() &&
705 pos <= end());
706 std::size_t const n =
707 t_->size;
708 std::size_t const i =
709 pos - &(*t_)[0];
710 if(n < t_->capacity)
711 {
712 // fast path
713 auto const p =
714 &(*t_)[i];
715 relocate(
716 dest: p + 1,
717 src: p,
718 n: n - i);
719 ::new(p) value(pv);
720 ++t_->size;
721 return p;
722 }
723 auto t =
724 table::allocate(
725 capacity: growth(new_size: n + 1), sp: sp_);
726 auto const p = &(*t)[i];
727 ::new(p) value(pv);
728 relocate(
729 dest: &(*t)[0],
730 src: &(*t_)[0],
731 n: i);
732 relocate(
733 dest: p + 1,
734 src: &(*t_)[i],
735 n: n - i);
736 t->size = static_cast<
737 std::uint32_t>(size() + 1);
738 t = detail::exchange(t&: t_, u: t);
739 table::deallocate(p: t, sp: sp_);
740 return p;
741}

745bool
746array::
747equal(
748 array const& other) const noexcept
749{
750 if(size() != other.size())
751 return false;
752 for(std::size_t i = 0; i < size(); ++i)
753 if((*this)[i] != other[i])
754 return false;
755 return true;
756}
