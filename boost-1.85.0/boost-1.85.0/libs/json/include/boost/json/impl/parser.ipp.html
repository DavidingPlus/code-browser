99std::size_t
100parser::
101write(
102 char const* data,
103 std::size_t size,
104 system::error_code& ec)
105{
106 auto const n = write_some(
107 data, size, ec);
108 if(! ec && n < size)
109 {
110 BOOST_JSON_FAIL(ec, error::extra_data);
111 p_.fail(ec);
112 }
113 return n;
114}

143value
144parser::
145release()
146{
147 if( ! p_.done())
148 {
149 // prevent undefined behavior
150 if(! p_.last_error())
151 {
152 system::error_code ec;
153 BOOST_JSON_FAIL(ec, error::incomplete);
154 p_.fail(ec);
155 }
156 detail::throw_system_error(
157 ec: p_.last_error());
158 }
159 return p_.handler().st.release();
160}
