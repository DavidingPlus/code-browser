21template<class V>
22void
23parse_into(
24 V& v,
25 string_view sv,
26 system::error_code& ec,
27 parse_options const& opt )
28{
29 parser_for<V> p( opt, &v );
30
31 std::size_t n = p.write_some( false, sv.data(), sv.size(), ec );
32
33 if( !ec && n < sv.size() )
34 {
35 BOOST_JSON_FAIL( ec, error::extra_data );
36 }
37}

65template<class V>
66void
67parse_into(
68 V& v,
69 std::istream& is,
70 system::error_code& ec,
71 parse_options const& opt )
72{
73 parser_for<V> p( opt, &v );
74
75 char read_buffer[BOOST_JSON_STACK_BUFFER_SIZE];
76 do
77 {
78 if( is.eof() )
79 {
80 p.write_some(false, nullptr, 0, ec);
81 break;
82 }
83
84 if( !is )
85 {
86 BOOST_JSON_FAIL( ec, error::input_error );
87 break;
88 }
89
90 is.read(s: read_buffer, n: sizeof(read_buffer));
91 std::size_t const consumed = static_cast<std::size_t>( is.gcount() );
92
93 std::size_t const n = p.write_some( true, read_buffer, consumed, ec );
94 if( !ec.failed() && n < consumed )
95 {
96 BOOST_JSON_FAIL( ec, error::extra_data );
97 }
98 }
99 while( !ec.failed() );
100}
