23value
24parse(
25 string_view s,
26 system::error_code& ec,
27 storage_ptr sp,
28 const parse_options& opt)
29{
30 unsigned char temp[
31 BOOST_JSON_STACK_BUFFER_SIZE];
32 parser p(storage_ptr(), opt, temp);
33 p.reset(sp: std::move(sp));
34 p.write(s, ec);
35 if(ec)
36 return nullptr;
37 return p.release();
38}

67value
68parse(
69 std::istream& is,
70 system::error_code& ec,
71 storage_ptr sp,
72 parse_options const& opt)
73{
74 unsigned char parser_buffer[BOOST_JSON_STACK_BUFFER_SIZE / 2];
75 stream_parser p(storage_ptr(), opt, parser_buffer);
76 p.reset(sp: std::move(sp));
77
78 char read_buffer[BOOST_JSON_STACK_BUFFER_SIZE / 2];
79 do
80 {
81 if( is.eof() )
82 {
83 p.finish(ec);
84 break;
85 }
86
87 if( !is )
88 {
89 BOOST_JSON_FAIL( ec, error::input_error );
90 break;
91 }
92
93 is.read(s: read_buffer, n: sizeof(read_buffer));
94 auto const consumed = is.gcount();
95
96 p.write( data: read_buffer, size: static_cast<std::size_t>(consumed), ec );
97 }
98 while( !ec.failed() );
99
100 if( ec.failed() )
101 return nullptr;
102
103 return p.release();
104}
