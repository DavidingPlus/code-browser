39bool
40value_ref::
41is_key_value_pair() const noexcept
42{
43 if(what_ != what::ini)
44 return false;
45 if(arg_.init_list_.size() != 2)
46 return false;
47 auto const& e =
48 *arg_.init_list_.begin();
49 if( e.what_ != what::str &&
50 e.what_ != what::strfunc)
51 return false;
52 return true;
53}

55bool
56value_ref::
57maybe_object(
58 std::initializer_list<
59 value_ref> init) noexcept
60{
61 for(auto const& e : init)
62 if(! e.is_key_value_pair())
63 return false;
64 return true;
65}

67string_view
68value_ref::
69get_string() const noexcept
70{
71 BOOST_ASSERT(
72 what_ == what::str ||
73 what_ == what::strfunc);
74 if (what_ == what::strfunc)
75 return *static_cast<const string*>(f_.p);
76 return arg_.str_;
77}

79value
80value_ref::
81make_value(
82 storage_ptr sp) const
83{
84 switch(what_)
85 {
86 default:
87 case what::str:
88 return string(
89 arg_.str_,
90 std::move(sp));
91
92 case what::ini:
93 return make_value(
94 init: arg_.init_list_,
95 sp: std::move(sp));
96
97 case what::func:
98 return f_.f(f_.p,
99 std::move(sp));
100
101 case what::strfunc:
102 return f_.f(f_.p,
103 std::move(sp));
104
105 case what::cfunc:
106 return cf_.f(cf_.p,
107 std::move(sp));
108 }
109}

111value
112value_ref::
113make_value(
114 std::initializer_list<
115 value_ref> init,
116 storage_ptr sp)
117{
118 if(maybe_object(init))
119 return make_object(
120 init, sp: std::move(sp));
121 return make_array(
122 init, sp: std::move(sp));
123}

125object
126value_ref::
127make_object(
128 std::initializer_list<value_ref> init,
129 storage_ptr sp)
130{
131 object obj(std::move(sp));
132 obj.reserve(new_capacity: init.size());
133 for(auto const& e : init)
134 obj.emplace(
135 key: e.arg_.init_list_.begin()[0].get_string(),
136 arg: e.arg_.init_list_.begin()[1].make_value(
137 sp: obj.storage()));
138 return obj;
139}

141array
142value_ref::
143make_array(
144 std::initializer_list<
145 value_ref> init,
146 storage_ptr sp)
147{
148 array arr(std::move(sp));
149 arr.reserve(new_capacity: init.size());
150 for(auto const& e : init)
151 arr.emplace_back(
152 arg: e.make_value(
153 sp: arr.storage()));
154 return arr;
155}

157void
158value_ref::
159write_array(
160 value* dest,
161 std::initializer_list<
162 value_ref> init,
163 storage_ptr const& sp)
164{
165 struct undo
166 {
167 value* const base;
168 value* pos;
169 ~undo()
170 {
171 if(pos)
172 while(pos > base)
173 (--pos)->~value();
174 }
175 };
176 undo u{.base: dest, .pos: dest};
177 for(auto const& e : init)
178 {
179 ::new(u.pos) value(
180 e.make_value(sp));
181 ++u.pos;
182 }
183 u.pos = nullptr;
184}
