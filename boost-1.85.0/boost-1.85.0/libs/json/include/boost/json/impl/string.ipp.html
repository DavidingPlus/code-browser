142string&
143string::
144assign(
145 string const& other)
146{
147 if(this == &other)
148 return *this;
149 return assign(
150 s: other.data(),
151 count: other.size());
152}
153
154string&
155string::
156assign(string&& other)
157{
158 if( &other == this )
159 return *this;
160
161 if(*sp_ == *other.sp_)
162 {
163 impl_.destroy(sp: sp_);
164 impl_ = other.impl_;
165 ::new(&other.impl_) detail::string_impl();
166 return *this;
167 }
168
169 // copy
170 return assign(other);
171}

313string&
314string::
315erase(
316 size_type pos,
317 size_type count)
318{
319 if(pos > impl_.size())
320 {
321 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
322 detail::throw_system_error( e: error::out_of_range, loc: &loc );
323 }
324 if( count > impl_.size() - pos)
325 count = impl_.size() - pos;
326 std::char_traits<char>::move(
327 s1: impl_.data() + pos,
328 s2: impl_.data() + pos + count,
329 n: impl_.size() - pos - count + 1);
330 impl_.term(n: impl_.size() - count);
331 return *this;
332}

357void
358string::
359resize(size_type count, char ch)
360{
361 if(count <= impl_.size())
362 {
363 impl_.term(n: count);
364 return;
365 }
366
367 reserve(new_capacity: count);
368 std::char_traits<char>::assign(
369 s: impl_.end(),
370 n: count - impl_.size(),
371 a: ch);
372 grow(n: count - size());
373}

377void
378string::
379swap(string& other)
380{
381 if(*sp_ == *other.sp_)
382 {
383 std::swap(a&: impl_, b&: other.impl_);
384 return;
385 }
386 string temp1(
387 std::move(*this), other.sp_);
388 string temp2(
389 std::move(other), sp_);
390 this->~string();
391 ::new(this) string(pilfer(t&: temp2));
392 other.~string();
393 ::new(&other) string(pilfer(t&: temp1));
394}

398void
399string::
400reserve_impl(size_type new_cap)
401{
402 BOOST_ASSERT(
403 new_cap >= impl_.capacity());
404 if(new_cap > impl_.capacity())
405 {
406 // grow
407 new_cap = detail::string_impl::growth(
408 new_size: new_cap, capacity: impl_.capacity());
409 detail::string_impl tmp(new_cap, sp_);
410 std::char_traits<char>::copy(s1: tmp.data(),
411 s2: impl_.data(), n: impl_.size() + 1);
412 tmp.size(n: impl_.size());
413 impl_.destroy(sp: sp_);
414 impl_ = tmp;
415 return;
416 }
417}
