20class pointer_token
21{
22public:
23 class iterator;
24
25 pointer_token(
26 string_view sv) noexcept
27 : b_( sv.begin() + 1 )
28 , e_( sv.end() )
29 {
30 BOOST_ASSERT( !sv.empty() );
31 BOOST_ASSERT( *sv.data() == '/' );
32 }
33
34 iterator begin() const noexcept;
35 iterator end() const noexcept;
36
37private:
38 char const* b_;
39 char const* e_;
40};
41
42class pointer_token::iterator
43{
44public:
45 using value_type = char;
46 using reference = char;
47 using pointer = value_type*;
48 using difference_type = std::ptrdiff_t;
49 using iterator_category = std::forward_iterator_tag;
50
51 explicit iterator(char const* base) noexcept
52 : base_(base)
53 {
54 }
55
56 char operator*() const noexcept
57 {
58 switch( char c = *base_ )
59 {
60 case '~':
61 c = base_[1];
62 if( '0' == c )
63 return '~';
64 BOOST_ASSERT('1' == c);
65 return '/';
66 default:
67 return c;
68 }
69 }
70
71 iterator& operator++() noexcept
72 {
73 if( '~' == *base_ )
74 base_ += 2;
75 else
76 ++base_;
77 return *this;
78 }
79
80 iterator operator++(int) noexcept
81 {
82 iterator result = *this;
83 ++(*this);
84 return result;
85 }
86
87 char const* base() const noexcept
88 {
89 return base_;
90 }
91
92private:
93 char const* base_;
94};

116bool operator==(pointer_token token, string_view sv) noexcept
117{
118 auto t_b = token.begin();
119 auto const t_e = token.end();
120 auto s_b = sv.begin();
121 auto const s_e = sv.end();
122 while( s_b != s_e )
123 {
124 if( t_e == t_b )
125 return false;
126 if( *t_b != *s_b )
127 return false;
128 ++t_b;
129 ++s_b;
130 }
131 return t_b == t_e;
132}

134bool is_invalid_zero(
135 char const* b,
136 char const* e) noexcept
137{
138 // in JSON Pointer only zero index can start character '0'
139 if( *b != '0' )
140 return false;
141
142 // if an index token starts with '0', then it should not have any more
143 // characters: either the string should end, or new token should start
144 ++b;
145 if( b == e )
146 return false;
147
148 BOOST_ASSERT( *b != '/' );
149 return true;
150}
151
152bool is_past_the_end_token(
153 char const* b,
154 char const* e) noexcept
155{
156 if( *b != '-' )
157 return false;
158
159 ++b;
160 BOOST_ASSERT( (b == e) || (*b != '/') );
161 return b == e;
162}

164std::size_t
165parse_number_token(
166 string_view sv,
167 system::error_code& ec) noexcept
168{
169 BOOST_ASSERT( !sv.empty() );
170
171 char const* b = sv.begin();
172 BOOST_ASSERT( *b == '/' );
173
174 ++b;
175 char const* const e = sv.end();
176 if( ( b == e )
177 || is_invalid_zero(b, e) )
178 {
179 BOOST_JSON_FAIL(ec, error::token_not_number);
180 return {};
181 }
182
183 if( is_past_the_end_token(b, e) )
184 {
185 ++b;
186 BOOST_JSON_FAIL(ec, error::past_the_end);
187 return {};
188 }
189
190 std::size_t result = 0;
191 for( ; b != e; ++b )
192 {
193 char const c = *b;
194 BOOST_ASSERT( c != '/' );
195
196 unsigned d = c - '0';
197 if( d > 9 )
198 {
199 BOOST_JSON_FAIL(ec, error::token_not_number);
200 return {};
201 }
202
203 std::size_t new_result = result * 10 + d;
204 if( new_result < result )
205 {
206 BOOST_JSON_FAIL(ec, error::token_overflow);
207 return {};
208 }
209
210 result = new_result;
211
212 }
213 return result;
214}

216string_view
217next_segment(
218 string_view& sv,
219 system::error_code& ec) noexcept
220{
221 if( sv.empty() )
222 return sv;
223
224 char const* const start = sv.begin();
225 char const* b = start;
226 if( *b++ != '/' )
227 {
228 BOOST_JSON_FAIL( ec, error::missing_slash );
229 return {};
230 }
231
232 char const* e = sv.end();
233 for( ; b < e; ++b )
234 {
235 char const c = *b;
236 if( '/' == c )
237 break;
238
239 if( '~' == c )
240 {
241 if( ++b == e )
242 {
243 BOOST_JSON_FAIL( ec, error::invalid_escape );
244 break;
245 }
246
247 switch (*b)
248 {
249 case '0': // fall through
250 case '1':
251 // valid escape sequence
252 continue;
253 default: {
254 BOOST_JSON_FAIL( ec, error::invalid_escape );
255 break;
256 }
257 }
258 break;
259 }
260 }
261
262 sv.remove_prefix( n: b - start );
263 return string_view( start, b );
264}

266value*
267if_contains_token(object const& obj, pointer_token token)
268{
269 if( obj.empty() )
270 return nullptr;
271
272 auto const it = detail::find_in_object(obj, key: token).first;
273 if( !it )
274 return nullptr;
275
276 return &it->value();
277}

279template<
280 class Value,
281 class OnObject,
282 class OnArray,
283 class OnScalar >
284Value*
285walk_pointer(
286 Value& jv,
287 string_view sv,
288 system::error_code& ec,
289 OnObject on_object,
290 OnArray on_array,
291 OnScalar on_scalar)
292{
293 ec.clear();
294
295 string_view segment = detail::next_segment( sv, ec );
296
297 Value* result = &jv;
298 while( true )
299 {
300 if( ec.failed() )
301 return nullptr;
302
303 if( !result )
304 {
305 BOOST_JSON_FAIL(ec, error::not_found);
306 return nullptr;
307 }
308
309 if( segment.empty() )
310 break;
311
312 switch( result->kind() )
313 {
314 case kind::object: {
315 auto& obj = result->get_object();
316
317 detail::pointer_token const token( segment );
318 segment = detail::next_segment( sv, ec );
319
320 result = on_object( obj, token );
321 break;
322 }
323 case kind::array: {
324 auto const index = detail::parse_number_token( sv: segment, ec );
325 segment = detail::next_segment( sv, ec );
326
327 auto& arr = result->get_array();
328 result = on_array( arr, index, ec );
329 break;
330 }
331 default: {
332 if( on_scalar( *result, segment ) )
333 break;
334 BOOST_JSON_FAIL( ec, error::value_is_scalar );
335 }}
336 }
337
338 BOOST_ASSERT( result );
339 return result;
340}

354value const*
355value::find_pointer( string_view sv, system::error_code& ec ) const noexcept
356{
357 return detail::walk_pointer(
358 jv: *this,
359 sv,
360 ec,
361 on_object: []( object const& obj, detail::pointer_token token )
362 {
363 return detail::if_contains_token(obj, token);
364 },
365 on_array: []( array const& arr, std::size_t index, system::error_code& ec )
366 -> value const*
367 {
368 if( ec )
369 return nullptr;
370
371 return arr.if_contains(pos: index);
372 },
373 on_scalar: []( value const&, string_view)
374 {
375 return std::false_type();
376 });
377}

402value*
403value::set_at_pointer(
404 string_view sv,
405 value_ref ref,
406 system::error_code& ec,
407 set_pointer_options const& opts )
408{
409 value* result = detail::walk_pointer(
410 jv&: *this,
411 sv,
412 ec,
413 on_object: []( object& obj, detail::pointer_token token)
414 {
415 if( !obj.empty() )
416 {
417 key_value_pair* kv = detail::find_in_object( obj, key: token ).first;
418 if( kv )
419 return &kv->value();
420 }
421
422 string key( token.begin(), token.end(), obj.storage() );
423 return &obj.emplace( key: std::move(key), arg: nullptr ).first->value();
424 },
425 on_array: [ &opts ]( array& arr, std::size_t index, system::error_code& ec ) -> value*
426 {
427 if( ec == error::past_the_end )
428 index = arr.size();
429 else if( ec.failed() )
430 return nullptr;
431
432 if( index >= arr.size() )
433 {
434 std::size_t const n = index - arr.size();
435 if( n >= opts.max_created_elements )
436 return nullptr;
437
438 arr.resize( count: arr.size() + n + 1 );
439 }
440
441 ec.clear();
442 return arr.data() + index;
443 },
444 on_scalar: [ &opts ]( value& jv, string_view segment )
445 {
446 if( jv.is_null() || opts.replace_any_scalar )
447 {
448 if( opts.create_arrays )
449 {
450 system::error_code ec;
451 detail::parse_number_token( sv: segment, ec );
452 if( !ec.failed() || ec == error::past_the_end )
453 {
454 jv = array( jv.storage() );
455 return true;
456 }
457 }
458
459 if( opts.create_objects )
460 {
461 jv = object( jv.storage() );
462 return true;
463 }
464 }
465
466 return false;
467 });
468
469 if( result )
470 *result = ref.make_value( sp: storage() );
471 return result;
472}
