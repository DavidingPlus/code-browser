31struct value_hasher
32{
33 std::size_t& seed;
34
35 template< class T >
36 void operator()( T&& t ) const noexcept
37 {
38 boost::hash_combine( seed, t );
39 }
40};

42enum class stream_parse_flags
43{
44 allow_comments = 1 << 0,
45 allow_trailing_commas = 1 << 1,
46 allow_invalid_utf8 = 1 << 2,
47};
48
49long
50to_bitmask( parse_options const& opts )
51{
52 using E = stream_parse_flags;
53 return
54 (opts.allow_comments ?
55 static_cast<long>(E::allow_comments) : 0) |
56 (opts.allow_trailing_commas ?
57 static_cast<long>(E::allow_trailing_commas) : 0) |
58 (opts.allow_invalid_utf8 ?
59 static_cast<long>(E::allow_invalid_utf8) : 0);
60}
61
62parse_options
63get_parse_options( std::istream& is )
64{
65 long const flags = is.iword(ix: parse_flags_xalloc);
66
67 using E = stream_parse_flags;
68 parse_options opts;
69 opts.allow_comments =
70 flags & static_cast<long>(E::allow_comments) ? true : false;
71 opts.allow_trailing_commas =
72 flags & static_cast<long>(E::allow_trailing_commas) ? true : false;
73 opts.allow_invalid_utf8 =
74 flags & static_cast<long>(E::allow_invalid_utf8) ? true : false;
75 return opts;
76}

386void
387value::
388swap(value& other)
389{
390 if(*storage() == *other.storage())
391 {
392 // fast path
393 union U
394 {
395 value tmp;
396 U(){}
397 ~U(){}
398 };
399 U u;
400 relocate(dest: &u.tmp, src: *this);
401 relocate(dest: this, src: other);
402 relocate(dest: &other, src: u.tmp);
403 return;
404 }
405
406 // copy
407 value temp1(
408 std::move(*this),
409 other.storage());
410 value temp2(
411 std::move(other),
412 this->storage());
413 other.~value();
414 ::new(&other) value(pilfer(t&: temp1));
415 this->~value();
416 ::new(this) value(pilfer(t&: temp2));
417}

419std::istream&
420operator>>(
421 std::istream& is,
422 value& jv)
423{
424 using Traits = std::istream::traits_type;
425
426 // sentry prepares the stream for reading and finalizes it in destructor
427 std::istream::sentry sentry(is);
428 if( !sentry )
429 return is;
430
431 parse_options opts = get_parse_options( is );
432 if( auto depth = static_cast<std::size_t>( is.iword(ix: parse_depth_xalloc) ) )
433 opts.max_depth = depth;
434
435 unsigned char parser_buf[BOOST_JSON_STACK_BUFFER_SIZE / 2];
436 stream_parser p( {}, opts, parser_buf );
437 p.reset( sp: jv.storage() );
438
439 char read_buf[BOOST_JSON_STACK_BUFFER_SIZE / 2];
440 std::streambuf& buf = *is.rdbuf();
441 std::ios::iostate err = std::ios::goodbit;
442#ifndef BOOST_NO_EXCEPTIONS
443 try
444#endif
445 {
446 while( true )
447 {
448 system::error_code ec;
449
450 // we peek the buffer; this either makes sure that there's no
451 // more input, or makes sure there's something in the internal
452 // buffer (so in_avail will return a positive number)
453 std::istream::int_type c = is.rdbuf()->sgetc();
454 // if we indeed reached EOF, we check if we parsed a full JSON
455 // document; if not, we error out
456 if( Traits::eq_int_type(c1: c, c2: Traits::eof()) )
457 {
458 err |= std::ios::eofbit;
459 p.finish(ec);
460 if( ec.failed() )
461 break;
462 }
463
464 // regardless of reaching EOF, we might have parsed a full JSON
465 // document; if so, we successfully finish
466 if( p.done() )
467 {
468 jv = p.release();
469 return is;
470 }
471
472 // at this point we definitely have more input, specifically in
473 // buf's internal buffer; we also definitely haven't parsed a whole
474 // document
475 std::streamsize available = buf.in_avail();
476 // if this assert fails, the streambuf is buggy
477 BOOST_ASSERT( available > 0 );
478
479 available = ( std::min )(
480 a: static_cast<std::size_t>(available), b: sizeof(read_buf) );
481 // we read from the internal buffer of buf into our buffer
482 available = buf.sgetn( s: read_buf, n: available );
483
484 std::size_t consumed = p.write_some(
485 data: read_buf, size: static_cast<std::size_t>(available), ec );
486 // if the parser hasn't consumed the entire input we've took from
487 // buf, we put the remaining data back; this should succeed,
488 // because we only read data from buf's internal buffer
489 while( consumed++ < static_cast<std::size_t>(available) )
490 {
491 std::istream::int_type const status = buf.sungetc();
492 BOOST_ASSERT( status != Traits::eof() );
493 (void)status;
494 }
495
496 if( ec.failed() )
497 break;
498 }
499 }
500#ifndef BOOST_NO_EXCEPTIONS
501 catch(...)
502 {
503 try
504 {
505 is.setstate(std::ios::badbit);
506 }
507 // we ignore the exception, because we need to throw the original
508 // exception instead
509 catch( std::ios::failure const& ) { }
510
511 if( is.exceptions() & std::ios::badbit )
512 throw;
513 }
514#endif
515
516 is.setstate(err | std::ios::failbit);
517 return is;
518}

536storage_ptr
537value::
538destroy() noexcept
539{
540 switch(kind())
541 {
542 case json::kind::null:
543 case json::kind::bool_:
544 case json::kind::int64:
545 case json::kind::uint64:
546 case json::kind::double_:
547 break;
548
549 case json::kind::string:
550 {
551 auto sp = str_.storage();
552 str_.~string();
553 return sp;
554 }
555
556 case json::kind::array:
557 {
558 auto sp = arr_.storage();
559 arr_.~array();
560 return sp;
561 }
562
563 case json::kind::object:
564 {
565 auto sp = obj_.storage();
566 obj_.~object();
567 return sp;
568 }
569
570 }
571 return std::move(sp_);
572}

574bool
575value::
576equal(value const& other) const noexcept
577{
578 switch(kind())
579 {
580 default: // unreachable()?
581 case json::kind::null:
582 return other.kind() == json::kind::null;
583
584 case json::kind::bool_:
585 return
586 other.kind() == json::kind::bool_ &&
587 get_bool() == other.get_bool();
588
589 case json::kind::int64:
590 switch(other.kind())
591 {
592 case json::kind::int64:
593 return get_int64() == other.get_int64();
594 case json::kind::uint64:
595 if(get_int64() < 0)
596 return false;
597 return static_cast<std::uint64_t>(
598 get_int64()) == other.get_uint64();
599 default:
600 return false;
601 }
602
603 case json::kind::uint64:
604 switch(other.kind())
605 {
606 case json::kind::uint64:
607 return get_uint64() == other.get_uint64();
608 case json::kind::int64:
609 if(other.get_int64() < 0)
610 return false;
611 return static_cast<std::uint64_t>(
612 other.get_int64()) == get_uint64();
613 default:
614 return false;
615 }
616
617 case json::kind::double_:
618 return
619 other.kind() == json::kind::double_ &&
620 get_double() == other.get_double();
621
622 case json::kind::string:
623 return
624 other.kind() == json::kind::string &&
625 get_string() == other.get_string();
626
627 case json::kind::array:
628 return
629 other.kind() == json::kind::array &&
630 get_array() == other.get_array();
631
632 case json::kind::object:
633 return
634 other.kind() == json::kind::object &&
635 get_object() == other.get_object();
636 }
637}

673 len_ = other.len_;
674 p[len_] = 0;

683std::size_t
684hash_value_impl( value const& jv ) noexcept
685{
686 std::size_t seed = 0;
687
688 kind const k = jv.kind();
689 boost::hash_combine( seed, v: k != kind::int64 ? k : kind::uint64 );
690
691 visit( v: value_hasher{.seed: seed}, jv );
692 return seed;
693}
