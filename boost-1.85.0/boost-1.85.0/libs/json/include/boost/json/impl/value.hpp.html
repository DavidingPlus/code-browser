16value&
17value::at_pointer(string_view ptr) &
18{
19 auto const& self = *this;
20 return const_cast<value&>( self.at_pointer(ptr) );
21}
22
23value&&
24value::at_pointer(string_view ptr) &&
25{
26 return std::move( this->at_pointer(ptr) );
27}
