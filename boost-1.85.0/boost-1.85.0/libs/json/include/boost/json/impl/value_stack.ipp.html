23value_stack::
24stack::
25~stack()
26{
27 clear();
28 if( begin_ != temp_ &&
29 begin_ != nullptr)
30 sp_->deallocate(
31 p: begin_,
32 bytes: (end_ - begin_) *
33 sizeof(value));
34}

36value_stack::
37stack::
38stack(
39 storage_ptr sp,
40 void* temp,
41 std::size_t size) noexcept
42 : sp_(std::move(sp))
43 , temp_(temp)
44{
45 if(size >= min_size_ *
46 sizeof(value))
47 {
48 begin_ = reinterpret_cast<
49 value*>(temp);
50 top_ = begin_;
51 end_ = begin_ +
52 size / sizeof(value);
53 }
54 else
55 {
56 begin_ = nullptr;
57 top_ = nullptr;
58 end_ = nullptr;
59 }
60}

88// destroy the values but
89// not the stack allocation.
90void
91value_stack::
92stack::
93clear() noexcept
94{
95 if(top_ != begin_)
96 {
97 if(run_dtors_)
98 for(auto it = top_;
99 it-- != begin_;)
100 it->~value();
101 top_ = begin_;
102 }
103 chars_ = 0;
104}

106void
107value_stack::
108stack::
109maybe_grow()
110{
111 if(top_ >= end_)
112 grow_one();
113}
114
115// make room for at least one more value
116void
117value_stack::
118stack::
119grow_one()
120{
121 BOOST_ASSERT(chars_ == 0);
122 std::size_t const capacity =
123 end_ - begin_;
124 std::size_t new_cap = min_size_;
125 // VFALCO check overflow here
126 while(new_cap < capacity + 1)
127 new_cap <<= 1;
128 auto const begin =
129 reinterpret_cast<value*>(
130 sp_->allocate(
131 bytes: new_cap * sizeof(value)));
132 std::size_t const cur_size = top_ - begin_;
133 if(begin_)
134 {
135 std::memcpy(
136 dest: reinterpret_cast<char*>(begin),
137 src: reinterpret_cast<char*>(begin_),
138 n: size() * sizeof(value));
139 if(begin_ != temp_)
140 sp_->deallocate(p: begin_,
141 bytes: capacity * sizeof(value));
142 }
143 // book-keeping
144 top_ = begin + cur_size;
145 end_ = begin + new_cap;
146 begin_ = begin;
147}
148
149// make room for nchars additional characters.
150void
151value_stack::
152stack::
153grow(std::size_t nchars)
154{
155 // needed capacity in values
156 std::size_t const needed =
157 size() +
158 1 +
159 ((chars_ + nchars +
160 sizeof(value) - 1) /
161 sizeof(value));
162 std::size_t const capacity =
163 end_ - begin_;
164 BOOST_ASSERT(
165 needed > capacity);
166 std::size_t new_cap = min_size_;
167 // VFALCO check overflow here
168 while(new_cap < needed)
169 new_cap <<= 1;
170 auto const begin =
171 reinterpret_cast<value*>(
172 sp_->allocate(
173 bytes: new_cap * sizeof(value)));
174 std::size_t const cur_size = top_ - begin_;
175 if(begin_)
176 {
177 std::size_t amount =
178 size() * sizeof(value);
179 if(chars_ > 0)
180 amount += sizeof(value) + chars_;
181 std::memcpy(
182 dest: reinterpret_cast<char*>(begin),
183 src: reinterpret_cast<char*>(begin_),
184 n: amount);
185 if(begin_ != temp_)
186 sp_->deallocate(p: begin_,
187 bytes: capacity * sizeof(value));
188 }
189 // book-keeping
190 top_ = begin + cur_size;
191 end_ = begin + new_cap;
192 begin_ = begin;
193}

197void
198value_stack::
199stack::
200append(string_view s)
201{
202 std::size_t const bytes_avail =
203 reinterpret_cast<
204 char const*>(end_) -
205 reinterpret_cast<
206 char const*>(top_);
207 // make sure there is room for
208 // pushing one more value without
209 // clobbering the string.
210 if(sizeof(value) + chars_ +
211 s.size() > bytes_avail)
212 grow(nchars: s.size());
213
214 // copy the new piece
215 std::memcpy(
216 dest: reinterpret_cast<char*>(
217 top_ + 1) + chars_,
218 src: s.data(), n: s.size());
219 chars_ += s.size();
220
221 // ensure a pushed value cannot
222 // clobber the released string.
223 BOOST_ASSERT(
224 reinterpret_cast<char*>(
225 top_ + 1) + chars_ <=
226 reinterpret_cast<char*>(
227 end_));
228}
229
230string_view
231value_stack::
232stack::
233release_string() noexcept
234{
235 // ensure a pushed value cannot
236 // clobber the released string.
237 BOOST_ASSERT(
238 reinterpret_cast<char*>(
239 top_ + 1) + chars_ <=
240 reinterpret_cast<char*>(
241 end_));
242 auto const n = chars_;
243 chars_ = 0;
244 return { reinterpret_cast<
245 char const*>(top_ + 1), n };
246}

248// transfer ownership of the top n
249// elements of the stack to the caller
250value*
251value_stack::
252stack::
253release(std::size_t n) noexcept
254{
255 BOOST_ASSERT(n <= size());
256 BOOST_ASSERT(chars_ == 0);
257 top_ -= n;
258 return top_;
259}

261template<class... Args>
262value&
263value_stack::
264stack::
265push(Args&&... args)
266{
267 BOOST_ASSERT(chars_ == 0);
268 if(top_ >= end_)
269 grow_one();
270 value& jv = detail::access::
271 construct_value(top_,
272 std::forward<Args>(args)...);
273 ++top_;
274 return jv;
275}
276
277template<class Unchecked>
278void
279value_stack::
280stack::
281exchange(Unchecked&& u)
282{
283 BOOST_ASSERT(chars_ == 0);
284 union U
285 {
286 value v;
287 U() {}
288 ~U() {}
289 } jv;
290 // construct value on the stack
291 // to avoid clobbering top_[0],
292 // which belongs to `u`.
293 detail::access::
294 construct_value(
295 &jv.v, std::move(u));
296 std::memcpy(
297 dest: reinterpret_cast<
298 char*>(top_),
299 src: &jv.v, n: sizeof(value));
300 ++top_;
301}

325void
326value_stack::
327reset(storage_ptr sp) noexcept
328{
329 st_.clear();
330
331 sp_.~storage_ptr();
332 ::new(&sp_) storage_ptr(
333 pilfer(t&: sp));
334
335 // `stack` needs this
336 // to clean up correctly
337 st_.run_dtors(
338 b: ! sp_.is_not_shared_and_deallocate_is_trivial());
339}
340
341value
342value_stack::
343release() noexcept
344{
345 // This means the caller did not
346 // cause a single top level element
347 // to be produced.
348 BOOST_ASSERT(st_.size() == 1);
349
350 // give up shared ownership
351 sp_ = {};
352
353 return pilfer(t&: *st_.release(n: 1));
354}

358void
359value_stack::
360push_array(std::size_t n)
361{
362 // we already have room if n > 0
363 if(BOOST_JSON_UNLIKELY(n == 0))
364 st_.maybe_grow();
365 detail::unchecked_array ua(
366 st_.release(n), n, sp_);
367 st_.exchange(u: std::move(ua));
368}

370void
371value_stack::
372push_object(std::size_t n)
373{
374 // we already have room if n > 0
375 if(BOOST_JSON_UNLIKELY(n == 0))
376 st_.maybe_grow();
377 detail::unchecked_object uo(
378 st_.release(n: n * 2), n, sp_);
379 st_.exchange(u: std::move(uo));
380}

390void
391value_stack::
392push_key(
393 string_view s)
394{
395 if(! st_.has_chars())
396 {
397 st_.push(args: detail::key_t{}, args&: s, args&: sp_);
398 return;
399 }
400 auto part = st_.release_string();
401 st_.push(args: detail::key_t{}, args&: part, args&: s, args&: sp_);
402}
403
404void
405value_stack::
406push_string(
407 string_view s)
408{
409 if(! st_.has_chars())
410 {
411 // fast path
412 st_.push(args&: s, args&: sp_);
413 return;
414 }
415
416 // VFALCO We could add a special
417 // private ctor to string that just
418 // creates uninitialized space,
419 // to reduce member function calls.
420 auto part = st_.release_string();
421 auto& str = st_.push(
422 args: string_kind, args&: sp_).get_string();
423 str.reserve(
424 new_capacity: part.size() + s.size());
425 std::memcpy(
426 dest: str.data(),
427 src: part.data(), n: part.size());
428 std::memcpy(
429 dest: str.data() + part.size(),
430 src: s.data(), n: s.size());
431 str.grow(n: part.size() + s.size());
432}
