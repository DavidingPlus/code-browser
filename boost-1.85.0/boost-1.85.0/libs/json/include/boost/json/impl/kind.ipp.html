10#ifndef BOOST_JSON_IMPL_KIND_IPP
11#define BOOST_JSON_IMPL_KIND_IPP
12
13#include <boost/json/kind.hpp>
14#include <ostream>
15
16namespace boost {
17namespace json {
18
19string_view
20to_string(kind k) noexcept
21{
22 switch(k)
23 {
24 case kind::array: return "array";
25 case kind::object: return "object";
26 case kind::string: return "string";
27 case kind::int64: return "int64";
28 case kind::uint64: return "uint64";
29 case kind::double_: return "double";
30 case kind::bool_: return "bool";
31 default: // satisfy warnings
32 case kind::null: return "null";
33 }
34}
35
36std::ostream&
37operator<<(std::ostream& os, kind k)
38{
39 os << to_string(k);
40 return os;
41}
42
43} // namespace json
44} // namespace boost

46#endif
