104inline
105BOOST_SYSTEM_CONSTEXPR
106system::error_code
107make_error_code(error e) noexcept
108{
109
110 return system::error_code(
111 static_cast<std::underlying_type<error>::type>(e),
112 detail::error_code_category );
113}

115inline
116BOOST_SYSTEM_CONSTEXPR
117system::error_condition
118make_error_condition(condition c) noexcept
119{
120 return system::error_condition(
121 static_cast<std::underlying_type<condition>::type>(c),
122 detail::error_condition_category );
123}
