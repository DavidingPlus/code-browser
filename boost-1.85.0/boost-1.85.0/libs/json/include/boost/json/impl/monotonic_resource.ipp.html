29constexpr
30std::size_t
31monotonic_resource::
32max_size()
33{
34 return std::size_t(-1) - sizeof(block);
35}
36
37// lowest power of 2 greater than or equal to n
38std::size_t
39monotonic_resource::
40round_pow2(
41 std::size_t n) noexcept
42{
43 if(n & (n - 1))
44 return next_pow2(n);
45 return n;
46}
47
48// lowest power of 2 greater than n
49std::size_t
50monotonic_resource::
51next_pow2(
52 std::size_t n) noexcept
53{
54 std::size_t result = min_size_;
55 while(result <= n)
56 {
57 if(result >= max_size() - result)
58 {
59 // overflow
60 result = max_size();
61 break;
62 }
63 result *= 2;
64 }
65 return result;
66}

99void
100monotonic_resource::
101release() noexcept
102{
103 auto p = head_;
104 while(p != &buffer_)
105 {
106 auto next = p->next;
107 upstream_->deallocate(p, bytes: p->size);
108 p = next;
109 }
110 buffer_.p = reinterpret_cast<
111 unsigned char*>(buffer_.p) - (
112 buffer_.size - buffer_.avail);
113 buffer_.avail = buffer_.size;
114 head_ = &buffer_;
115}
116
117void*
118monotonic_resource::
119do_allocate(
120 std::size_t n,
121 std::size_t align)
122{
123 auto p = alignment::align(
124 alignment: align, size: n, ptr&: head_->p, space&: head_->avail);
125 if(p)
126 {
127 head_->p = reinterpret_cast<
128 unsigned char*>(p) + n;
129 head_->avail -= n;
130 return p;
131 }
132
133 if(next_size_ < n)
134 next_size_ = round_pow2(n);
135 auto b = ::new(upstream_->allocate(
136 bytes: sizeof(block) + next_size_)) block;
137 b->p = b + 1;
138 b->avail = next_size_;
139 b->size = next_size_;
140 b->next = head_;
141 head_ = b;
142 next_size_ = next_pow2(n: next_size_);
143
144 p = alignment::align(
145 alignment: align, size: n, ptr&: head_->p, space&: head_->avail);
146 BOOST_ASSERT(p);
147 head_->p = reinterpret_cast<
148 unsigned char*>(p) + n;
149 head_->avail -= n;
150 return p;
151}
