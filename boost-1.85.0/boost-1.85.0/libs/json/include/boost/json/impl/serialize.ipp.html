22int serialize_xalloc = std::ios::xalloc();
23
24enum class serialize_stream_flags : long
25{
26 allow_infinity_and_nan = 1,
27};
28
29std::underlying_type<serialize_stream_flags>::type
30to_bitmask( serialize_options const& opts )
31{
32 using E = serialize_stream_flags;
33 using I = std::underlying_type<E>::type;
34 return (opts.allow_infinity_and_nan
35 ? static_cast<I>(E::allow_infinity_and_nan) : 0);
36}
37
38serialize_options
39get_stream_flags( std::ostream& os )
40{
41 auto const flags = os.iword(ix: serialize_xalloc);
42
43 serialize_options opts;
44 using E = serialize_stream_flags;
45 using I = std::underlying_type<E>::type;
46 opts.allow_infinity_and_nan =
47 flags & static_cast<I>(E::allow_infinity_and_nan);
48 return opts;
49}

53static
54void
55serialize_impl(
56 std::string& s,
57 serializer& sr)
58{
59 // serialize to a small buffer to avoid
60 // the first few allocations in std::string
61 char buf[BOOST_JSON_STACK_BUFFER_SIZE];
62 string_view sv;
63 sv = sr.read(dest&: buf);
64 if(sr.done())
65 {
66 // fast path
67 s.append(
68 s: sv.data(), n: sv.size());
69 return;
70 }
71 std::size_t len = sv.size();
72 s.reserve(res_arg: len * 2);
73 s.resize(n: s.capacity());
74 BOOST_ASSERT(
75 s.size() >= len * 2);
76 std::memcpy(dest: &s[0],
77 src: sv.data(), n: sv.size());
78 auto const lim =
79 s.max_size() / 2;
80 for(;;)
81 {
82 sv = sr.read(
83 dest: &s[0] + len,
84 size: s.size() - len);
85 len += sv.size();
86 if(sr.done())
87 break;
88 // growth factor 2x
89 if(s.size() < lim)
90 s.resize(n: s.size() * 2);
91 else
92 s.resize(n: 2 * lim);
93 }
94 s.resize(n: len);
95}

176//[example_operator_lt__lt_
177// Serialize a value into an output stream
178
179std::ostream&
180operator<<( std::ostream& os, value const& jv )
181{
182 // Create a serializer
183 serializer sr( get_stream_flags(os) );
184
185 // Set the serializer up for our value
186 sr.reset( p: &jv );
187
188 // Loop until all output is produced.
189 while( ! sr.done() )
190 {
191 // Use a local buffer to avoid allocation.
192 char buf[ BOOST_JSON_STACK_BUFFER_SIZE ];
193
194 // Fill our buffer with serialized characters and write it to the output stream.
195 os << sr.read( dest&: buf );
196 }
197
198 return os;
199}

202static
203void
204to_ostream(
205 std::ostream& os,
206 serializer& sr)
207{
208 while(! sr.done())
209 {
210 char buf[BOOST_JSON_STACK_BUFFER_SIZE];
211 auto s = sr.read(dest&: buf);
212 os.write(s: s.data(), n: s.size());
213 }
214}
