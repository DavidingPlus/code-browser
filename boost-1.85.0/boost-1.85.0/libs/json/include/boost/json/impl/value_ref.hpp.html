10#ifndef BOOST_JSON_IMPL_VALUE_REF_HPP
11#define BOOST_JSON_IMPL_VALUE_REF_HPP
12
13namespace boost {
14namespace json {
15
16template<class T>
17value
18value_ref::
19from_builtin(
20 void const* p,
21 storage_ptr sp) noexcept
22{
23 return value(
24 *reinterpret_cast<
25 T const*>(p),
26 std::move(sp));
27}
28
29template<class T>
30value
31value_ref::
32from_const(
33 void const* p,
34 storage_ptr sp)
35{
36 return value(
37 *reinterpret_cast<
38 T const*>(p),
39 std::move(sp));
40}
41
42template<class T>
43value
44value_ref::
45from_rvalue(
46 void* p,
47 storage_ptr sp)
48{
49 return value(
50 std::move(
51 *reinterpret_cast<T*>(p)),
52 std::move(sp));
53}
54
55} // namespace json
56} // namespace boost
