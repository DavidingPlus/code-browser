97std::size_t
98stream_parser::
99write(
100 char const* data,
101 std::size_t size,
102 system::error_code& ec)
103{
104 auto const n = write_some(
105 data, size, ec);
106 if(! ec && n < size)
107 {
108 BOOST_JSON_FAIL(ec, error::extra_data);
109 p_.fail(ec);
110 }
111 return n;
112}

141void
142stream_parser::
143finish(system::error_code& ec)
144{
145 p_.write_some(more: false, data: nullptr, size: 0, ec);
146}

148void
149stream_parser::
150finish()
151{
152 system::error_code ec;
153 finish(ec);
154 if(ec)
155 detail::throw_system_error( ec );
156}

167value
168stream_parser::
169release()
170{
171 if(! p_.done())
172 {
173 // prevent undefined behavior
174 finish();
175 }
176 return p_.handler().st.release();
177}
