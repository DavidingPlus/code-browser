24// Objects with size less than or equal
25// to this number will use a linear search
26// instead of the more expensive hash function.
27static
28constexpr
29std::size_t
30small_object_size_ = 18;
31
32BOOST_STATIC_ASSERT(
33 small_object_size_ <
34 BOOST_JSON_MAX_STRUCTURED_SIZE);

40struct alignas(key_value_pair)
41 object::table
42{
43 std::uint32_t size = 0;
44 std::uint32_t capacity = 0;
45 std::uintptr_t salt = 0;
46
47#if defined(_MSC_VER) && BOOST_JSON_ARCH == 32
48 // VFALCO If we make key_value_pair smaller,
49 // then we might want to revisit this
50 // padding.
51 BOOST_STATIC_ASSERT(
52 sizeof(key_value_pair) == 32);
53 char pad[4] = {}; // silence warnings
54#endif
55
56 constexpr table();
57
58 // returns true if we use a linear
59 // search instead of the hash table.
60 bool is_small() const noexcept
61 {
62 return capacity <=
63 detail::small_object_size_;
64 }
65
66 key_value_pair&
67 operator[](
68 std::size_t pos) noexcept
69 {
70 return reinterpret_cast<
71 key_value_pair*>(
72 this + 1)[pos];
73 }
74
75 // VFALCO This is exported for tests
76 BOOST_JSON_DECL
77 std::size_t
78 digest(string_view key) const noexcept;
79
80 inline
81 index_t&
82 bucket(std::size_t hash) noexcept;
83
84 inline
85 index_t&
86 bucket(string_view key) noexcept;
87
88 inline
89 void
90 clear() noexcept;
91
92 static
93 inline
94 table*
95 allocate(
96 std::size_t capacity,
97 std::uintptr_t salt,
98 storage_ptr const& sp);
99
100 static
101 void
102 deallocate(
103 table* p,
104 storage_ptr const& sp) noexcept
105 {
106 if(p->capacity == 0)
107 return;
108 if(! p->is_small())
109 sp->deallocate(p,
110 bytes: sizeof(table) + p->capacity * (
111 sizeof(key_value_pair) +
112 sizeof(index_t)));
113 else
114 sp->deallocate(p,
115 bytes: sizeof(table) + p->capacity *
116 sizeof(key_value_pair));
117 }
118};

154class object::revert_insert
155{
156 object* obj_;
157 table* t_ = nullptr;
158 std::size_t size_;
159
160 BOOST_JSON_DECL
161 void
162 destroy() noexcept;
163
164public:
165 explicit
166 revert_insert(
167 object& obj,
168 std::size_t capacity)
169 : obj_(&obj)
170 , size_(obj_->size())
171 {
172 if( capacity > obj_->capacity() )
173 t_ = obj_->reserve_impl(new_capacity: capacity);
174 }
175
176 ~revert_insert()
177 {
178 if(! obj_)
179 return;
180
181 destroy();
182 if( t_ )
183 {
184 table::deallocate( p: obj_->t_, sp: obj_->sp_ );
185 obj_->t_ = t_;
186 }
187 else
188 {
189 obj_->t_->size = static_cast<index_t>(size_);
190 }
191 }
192
193 void
194 commit() noexcept
195 {
196 BOOST_ASSERT(obj_);
197 if( t_ )
198 table::deallocate( p: t_, sp: obj_->sp_ );
199 obj_ = nullptr;
200 }
201};

326constexpr
327std::size_t
328object::
329max_size() noexcept
330{
331 // max_size depends on the address model
332 using min = std::integral_constant<std::size_t,
333 (std::size_t(-1) - sizeof(table)) /
334 (sizeof(key_value_pair) + sizeof(index_t))>;
335 return min::value < BOOST_JSON_MAX_STRUCTURED_SIZE ?
336 min::value : BOOST_JSON_MAX_STRUCTURED_SIZE;
337}

347void
348object::
349reserve(std::size_t new_capacity)
350{
351 if( new_capacity <= capacity() )
352 return;
353 table* const old_table = reserve_impl(new_capacity);
354 table::deallocate( p: old_table, sp: sp_ );
355}

412 std::pair<iterator, bool>
413{
414 std::pair<iterator, bool> result = emplace_impl(
415 key, key, static_cast<M&&>(m) );

441template<class InputIt>
442void
443object::
444construct(
445 InputIt first,
446 InputIt last,
447 std::size_t min_capacity,
448 std::input_iterator_tag)
449{
450 reserve(new_capacity: min_capacity);
451 revert_construct r(*this);
452 while(first != last)
453 {
454 insert(*first);
455 ++first;
456 }
457 r.commit();
458}

502template<class InputIt>
503void
504object::
505insert(
506 InputIt first,
507 InputIt last,
508 std::forward_iterator_tag)
509{
510 auto const n =
511 static_cast<std::size_t>(
512 std::distance(first, last));
513 auto const n0 = size();
514 if(n > max_size() - n0)
515 {
516 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
517 detail::throw_system_error( e: error::object_too_large, loc: &loc );
518 }
519 revert_insert r( *this, n0 + n );
520 while(first != last)
521 {
522 insert(*first);
523 ++first;
524 }
525 r.commit();
526}

528template< class... Args >
529std::pair<object::iterator, bool>
530object::
531emplace_impl( string_view key, Args&& ... args )
532{
533 std::pair<iterator, std::size_t> search_result(nullptr, 0);
534 if( !empty() )
535 {
536 search_result = detail::find_in_object(obj: *this, key);
537 if( search_result.first )
538 return { search_result.first, false };
539 }
540
541 // we create the new value before reserving, in case it is a reference to
542 // a subobject of the current object
543 key_value_pair kv( static_cast<Args&&>(args)..., sp_ );
544 // the key might get deallocated too
545 key = kv.key();
546
547 std::size_t const old_capacity = capacity();
548 reserve(new_capacity: size() + 1);
549 if( (empty() && capacity() > detail::small_object_size_)
550 || (capacity() != old_capacity) )
551 search_result.second = detail::digest(
552 b: key.begin(), e: key.end(), salt: t_->salt);
553
554 BOOST_ASSERT(
555 t_->is_small() ||
556 (search_result.second ==
557 detail::digest(key.begin(), key.end(), t_->salt)) );
558
559 return { insert_impl(p: pilfer(t&: kv), hash: search_result.second), true };
560}

566unchecked_object::
567~unchecked_object()
568{
569 if(! data_)
570 return;
571 if(sp_.is_not_shared_and_deallocate_is_trivial())
572 return;
573 value* p = data_;
574 while(size_--)
575 {
576 p[0].~value();
577 p[1].~value();
578 p += 2;
579 }
580}
