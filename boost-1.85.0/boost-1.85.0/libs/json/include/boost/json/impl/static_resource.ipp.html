13#include <boost/json/static_resource.hpp>
14#include <boost/throw_exception.hpp>
15#include <boost/align/align.hpp>
16#include <memory>
17
18namespace boost {
19namespace json {
20
21static_resource::
22static_resource(
23 unsigned char* buffer,
24 std::size_t size) noexcept
25 : p_(buffer)
26 , n_(size)
27 , size_(size)
28{
29}
30
31void
32static_resource::
33release() noexcept
34{
35 p_ = reinterpret_cast<
36 char*>(p_) - (size_ - n_);
37 n_ = size_;
38}
39
40void*
41static_resource::
42do_allocate(
43 std::size_t n,
44 std::size_t align)
45{
46 auto p = alignment::align(
47 alignment: align, size: n, ptr&: p_, space&: n_);
48 if(! p)
49 throw_exception( e: std::bad_alloc(), BOOST_CURRENT_LOCATION );
50 p_ = reinterpret_cast<char*>(p) + n;
51 n_ -= n;
52 return p;
53}
54
55void
56static_resource::
57do_deallocate(
58 void*,
59 std::size_t,
60 std::size_t)
61{
62 // do nothing
63}
64
65bool
66static_resource::
67do_is_equal(
68 memory_resource const& mr) const noexcept
69{
70 return this == &mr;
71}
