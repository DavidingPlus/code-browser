145inline
146double
147dec_to_float(
148 std::uint64_t m,
149 std::int32_t e,
150 bool neg) noexcept
151{
152 // convert to double explicitly to silence warnings
153 double x = static_cast<double>(m);
154 if(neg)
155 x = -x;
156
157 if(e < -305)
158 {
159 x *= 1e-305 ;
160 e += 305;
161 }
162
163 if(e >= -22 && e < 0)
164 return x / pow10(exp: -e);
165
166 return x * pow10(exp: e);
167}

227template<class Handler>
228void
229basic_parser<Handler>::
230reserve()
231{
232 if(BOOST_JSON_LIKELY(
233 ! st_.empty()))
234 return;
235 // Reserve the largest stack we need,
236 // to avoid reallocation during suspend.
237 st_.reserve(
238 n: sizeof(state) + // document parsing state
239 (sizeof(state) +
240 sizeof(std::size_t)) * depth() + // array and object state + size
241 sizeof(state) + // value parsing state
242 sizeof(std::size_t) + // string size
243 sizeof(state)); // comment state
244}

428template<class Handler>
429template<
430 bool StackEmpty_/*,
431 bool Terminal_*/>
432const char*
433basic_parser<Handler>::
434parse_comment(const char* p,
435 std::integral_constant<bool, StackEmpty_> stack_empty,
436 /*std::integral_constant<bool, Terminal_>*/ bool terminal)
437{
438 detail::const_stream_wrapper cs(p, end_);
439 const char* start = cs.begin();
440 std::size_t remain;
441 if(! stack_empty && ! st_.empty())
442 {
443 state st;
444 st_.pop(st);
445 switch(st)
446 {
447 default: BOOST_JSON_UNREACHABLE();
448 case state::com1: goto do_com1;
449 case state::com2: goto do_com2;
450 case state::com3: goto do_com3;
451 case state::com4: goto do_com4;
452 }
453 }
454 BOOST_ASSERT(*cs == '/');
455 ++cs;
456do_com1:
457 if(BOOST_JSON_UNLIKELY(! cs))
458 return maybe_suspend(cs.begin(), state::com1);
459 switch(*cs)
460 {
461 default:
462 {
463 BOOST_STATIC_CONSTEXPR source_location loc
464 = BOOST_CURRENT_LOCATION;
465 return fail(cs.begin(), error::syntax, &loc);
466 }
467 case '/':
468 ++cs;
469do_com2:
470 // KRYSTIAN TODO: this is a mess, we have to fix this
471 remain = cs.remain();
472 cs = remain ? static_cast<const char*>(
473 std::memchr(s: cs.begin(), c: '\n', n: remain)) : sentinel();
474 if(! cs.begin())
475 cs = sentinel();
476 if(BOOST_JSON_UNLIKELY(incomplete(cs)))
477 {
478 // if the doc does not terminate
479 // with a newline, treat it as the
480 // end of the comment
481 if(terminal && ! more_)
482 {
483 if(BOOST_JSON_UNLIKELY(! h_.on_comment(
484 {start, cs.remain(start)}, ec_)))
485 return fail(cs.end());
486 return cs.end();
487 }
488 if(BOOST_JSON_UNLIKELY(! h_.on_comment_part(
489 {start, cs.remain(start)}, ec_)))
490 return fail(cs.end());
491 if(terminal)
492 return suspend(cs.end(), state::com2);
493 return maybe_suspend(cs.end(), state::com2);
494 }
495 break;
496 case '*':
497 do
498 {
499 ++cs;
500do_com3:
501 // KRYSTIAN TODO: this is a mess, we have to fix this
502 remain = cs.remain();
503 cs = remain ? static_cast<const char*>(
504 std::memchr(s: cs.begin(), c: '*', n: remain)) : sentinel();
505 if(! cs.begin())
506 cs = sentinel();
507 // stopped inside a c comment
508 if(BOOST_JSON_UNLIKELY(incomplete(cs)))
509 {
510 if(BOOST_JSON_UNLIKELY(! h_.on_comment_part(
511 {start, cs.remain(start)}, ec_)))
512 return fail(cs.end());
513 return maybe_suspend(cs.end(), state::com3);
514 }
515 // found a asterisk, check if the next char is a slash
516 ++cs;
517do_com4:
518 if(BOOST_JSON_UNLIKELY(! cs))
519 {
520 if(BOOST_JSON_UNLIKELY(! h_.on_comment_part(
521 {start, cs.used(start)}, ec_)))
522 return fail(cs.begin());
523 return maybe_suspend(cs.begin(), state::com4);
524 }
525 }
526 while(*cs != '/');
527 }
528 ++cs;
529 if(BOOST_JSON_UNLIKELY(! h_.on_comment(
530 {start, cs.used(start)}, ec_)))
531 return fail(cs.begin());
532 return cs.begin();
533}

535template<class Handler>
536template<bool StackEmpty_>
537const char*
538basic_parser<Handler>::
539parse_document(const char* p,
540 std::integral_constant<bool, StackEmpty_> stack_empty)
541{
542 detail::const_stream_wrapper cs(p, end_);
543 if(! stack_empty && ! st_.empty())
544 {
545 state st;
546 st_.peek(st);
547 switch(st)
548 {
549 default: goto do_doc2;
550 case state::doc1:
551 st_.pop(st);
552 goto do_doc1;
553 case state::doc3:
554 st_.pop(st);
555 goto do_doc3;
556 case state::com1: case state::com2:
557 case state::com3: case state::com4:
558 goto do_doc4;
559 }
560 }
561do_doc1:
562 cs = detail::count_whitespace(p: cs.begin(), end: cs.end());
563 if(BOOST_JSON_UNLIKELY(! cs))
564 return maybe_suspend(cs.begin(), state::doc1);
565do_doc2:
566 switch(+opt_.allow_comments |
567 (opt_.allow_trailing_commas << 1) |
568 (opt_.allow_invalid_utf8 << 2))
569 {
570 // no extensions
571 default:
572 cs = parse_value(cs.begin(), stack_empty, std::false_type(), std::false_type(), std::false_type());
573 break;
574 // comments
575 case 1:
576 cs = parse_value(cs.begin(), stack_empty, std::true_type(), std::false_type(), std::false_type());
577 break;
578 // trailing
579 case 2:
580 cs = parse_value(cs.begin(), stack_empty, std::false_type(), std::true_type(), std::false_type());
581 break;
582 // comments & trailing
583 case 3:
584 cs = parse_value(cs.begin(), stack_empty, std::true_type(), std::true_type(), std::false_type());
585 break;
586 // skip validation
587 case 4:
588 cs = parse_value(cs.begin(), stack_empty, std::false_type(), std::false_type(), std::true_type());
589 break;
590 // comments & skip validation
591 case 5:
592 cs = parse_value(cs.begin(), stack_empty, std::true_type(), std::false_type(), std::true_type());
593 break;
594 // trailing & skip validation
595 case 6:
596 cs = parse_value(cs.begin(), stack_empty, std::false_type(), std::true_type(), std::true_type());
597 break;
598 // comments & trailing & skip validation
599 case 7:
600 cs = parse_value(cs.begin(), stack_empty, std::true_type(), std::true_type(), std::true_type());
601 break;
602 }
603 if(BOOST_JSON_UNLIKELY(incomplete(cs)))

805template<int Literal>
806const char*
807basic_parser<Handler>::
808parse_literal(const char* p,
809 std::integral_constant<int, Literal> literal)
810{
811 constexpr char const* literals[] = {
812 "null",
813 "true",
814 "false",
815 "Infinity",
816 "-Infinity",
817 "NaN",
818 };
819
820 constexpr std::size_t literal_sizes[] = {
821 4,
822 4,
823 5,
824 8,
825 9,
826 3,
827 };
828
829 std::size_t cur_lit;
830 std::size_t offset;
831
832 detail::const_stream_wrapper cs(p, end_);
833 BOOST_IF_CONSTEXPR( literal != detail::resume_literal )
834 {
835 BOOST_ASSERT( literal >= 0 );
836 if(BOOST_JSON_LIKELY( cs.remain() >= literal_sizes[literal] ))
837 {
838 int const cmp = std::memcmp(
839 s1: cs.begin(), s2: literals[literal], n: literal_sizes[literal] );
840 if( cmp != 0 )
841 {
842 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
843 return fail(cs.begin(), error::syntax, &loc);
844 }
845
846 BOOST_IF_CONSTEXPR( literal == detail::null_literal )
847 {
848 if(BOOST_JSON_UNLIKELY(
849 ! h_.on_null(ec_)))
850 return fail(cs.begin());
851 }
852 else BOOST_IF_CONSTEXPR( literal == detail::true_literal )
853 {
854 if(BOOST_JSON_UNLIKELY(
855 ! h_.on_bool(true, ec_)))
856 return fail(cs.begin());
857 }
858 else BOOST_IF_CONSTEXPR( literal == detail::false_literal )
859 {
860 if(BOOST_JSON_UNLIKELY(
861 ! h_.on_bool(false, ec_)))
862 return fail(cs.begin());
863 }
864 else BOOST_IF_CONSTEXPR( literal == detail::infinity_literal )
865 {
866 if(BOOST_JSON_UNLIKELY(
867 ! h_.on_double(
868 std::numeric_limits<double>::infinity(),
869 string_view(
870 literals[detail::infinity_literal],
871 literal_sizes[detail::infinity_literal]),
872 ec_)))
873 return fail(cs.begin());
874 }
875 else BOOST_IF_CONSTEXPR( literal == detail::neg_infinity_literal )
876 {
877 if(BOOST_JSON_UNLIKELY(
878 ! h_.on_double(
879 -std::numeric_limits<double>::infinity(),
880 string_view(
881 literals[detail::neg_infinity_literal],
882 literal_sizes[detail::neg_infinity_literal]),
883 ec_)))
884 return fail(cs.begin());
885 }
886 else BOOST_IF_CONSTEXPR( literal == detail::nan_literal )
887 {
888 if(BOOST_JSON_UNLIKELY(
889 ! h_.on_double(
890 std::numeric_limits<double>::quiet_NaN(),
891 string_view(
892 literals[detail::nan_literal],
893 literal_sizes[detail::nan_literal]),
894 ec_)))
895 return fail(cs.begin());
896 }
897 else
898 {
899 BOOST_JSON_UNREACHABLE();
900 }
901
902 cs += literal_sizes[literal];
903 return cs.begin();
904 }
905
906 offset = 0;
907 cur_lit = literal;
908 }
909 else
910 {
911 state st;
912 st_.pop(st);
913 BOOST_ASSERT( st == state::lit1 );
914
915 cur_lit = cur_lit_;
916 offset = lit_offset_;
917 }
918
919 std::size_t const size = (std::min)(
920 a: literal_sizes[cur_lit] - offset, b: cs.remain() );
921 int cmp = 0;
922 if(BOOST_JSON_LIKELY( cs.begin() ))
923 cmp = std::memcmp( s1: cs.begin(), s2: literals[cur_lit] + offset, n: size );
924 if( cmp != 0 )
925 {
926 BOOST_STATIC_CONSTEXPR source_location loc = BOOST_CURRENT_LOCATION;
927 return fail(cs.begin(), error::syntax, &loc);
928 }
929
930 if(BOOST_JSON_UNLIKELY( offset + size < literal_sizes[cur_lit] ))
931 {
932 BOOST_ASSERT( cur_lit < 256 );
933 cur_lit_ = static_cast<unsigned char>( cur_lit );
934 BOOST_ASSERT( offset + size < 256 );
935 lit_offset_ = static_cast<unsigned char>( offset + size );
936 return maybe_suspend(cs.begin() + size, state::lit1);
937 }
938
939 switch( cur_lit )
940 {
941 case detail::null_literal:
942 if(BOOST_JSON_UNLIKELY(
943 ! h_.on_null(ec_)))
944 return fail(cs.begin());
945 break;
946 case detail::true_literal:
947 if(BOOST_JSON_UNLIKELY(
948 ! h_.on_bool(true, ec_)))
949 return fail(cs.begin());
950 break;
951 case detail::false_literal:
952 if(BOOST_JSON_UNLIKELY(
953 ! h_.on_bool(false, ec_)))
954 return fail(cs.begin());
955 break;
956 case detail::infinity_literal:
957 if(BOOST_JSON_UNLIKELY(
958 ! h_.on_double(
959 std::numeric_limits<double>::infinity(),
960 string_view(
961 literals[detail::infinity_literal],
962 literal_sizes[detail::infinity_literal]),
963 ec_)))
964 return fail(cs.begin());
965 break;
966 case detail::neg_infinity_literal:
967 if(BOOST_JSON_UNLIKELY(
968 ! h_.on_double(
969 -std::numeric_limits<double>::infinity(),
970 string_view(
971 literals[detail::neg_infinity_literal],
972 literal_sizes[detail::neg_infinity_literal]),
973 ec_)))
974 return fail(cs.begin());
975 break;
976 case detail::nan_literal:
977 if(BOOST_JSON_UNLIKELY(
978 ! h_.on_double(
979 std::numeric_limits<double>::quiet_NaN(),
980 string_view(
981 literals[detail::nan_literal],
982 literal_sizes[detail::nan_literal]),
983 ec_)))
984 return fail(cs.begin());
985 break;
986 default: BOOST_JSON_UNREACHABLE();
987 }
988
989 cs += size;
990 return cs.begin();
991}

1038parse_unescaped(const char* p,

1040 std::integral_constant<bool, IsKey_> is_key,
1041 /*std::integral_constant<bool, AllowBadUTF8_>*/ bool allow_bad_utf8)
1042{
1043 detail::const_stream_wrapper cs(p, end_);
1044 std::size_t total;
1045 if(stack_empty || st_.empty())
1046 {
1047 BOOST_ASSERT(*cs == '\x22'); // '"'
1048 ++cs;
1049 total = 0;
1050 }
1051 else
1052 {
1053 state st;
1054 st_.pop(st);
1055 st_.pop(t&: total);
1056 }
1057 char const* start = cs.begin();
1058 cs = allow_bad_utf8?
1059 detail::count_valid<true>(p: cs.begin(), end: cs.end()):
1060 detail::count_valid<false>(p: cs.begin(), end: cs.end());
1061 std::size_t size = cs.used(p: start);
1062 if(is_key)
1063 {
1064 BOOST_ASSERT(total <= Handler::max_key_size);
1065 if(BOOST_JSON_UNLIKELY(size >
1066 Handler::max_key_size - total))
1067 {
1068 BOOST_STATIC_CONSTEXPR source_location loc
1069 = BOOST_CURRENT_LOCATION;
1070 return fail(cs.begin(), error::key_too_large, &loc);
1071 }
1072 }
1073 else
1074 {
1075 BOOST_ASSERT(total <= Handler::max_string_size);
1076 if(BOOST_JSON_UNLIKELY(size >
1077 Handler::max_string_size - total))
1078 {
1079 BOOST_STATIC_CONSTEXPR source_location loc
1080 = BOOST_CURRENT_LOCATION;
1081 return fail(cs.begin(), error::string_too_large, &loc);
1082 }
1083 }
1084 total += size;
1085 if(BOOST_JSON_UNLIKELY(! cs))
1086 {
1087 // call handler if the string isn't empty
1088 if(BOOST_JSON_LIKELY(size))
1089 {
1090 {
1091 bool r = is_key?
1092 h_.on_key_part( {start, size}, total, ec_ ):
1093 h_.on_string_part( {start, size}, total, ec_ );
1094
1095 if(BOOST_JSON_UNLIKELY(!r))
1096 {
1097 return fail(cs.begin());
1098 }
1099 }
1100 }
1101 return maybe_suspend(cs.begin(), state::str1, total);
1102 }
1103 // at this point all valid characters have been skipped, so any remaining
1104 // if there are any more characters, they are either escaped, or incomplete
1105 // utf8, or invalid utf8
1106 if(BOOST_JSON_UNLIKELY(*cs != '\x22')) // '"'
1107 {
1108 // sequence is invalid or incomplete
1109 if((*cs & 0x80) && !allow_bad_utf8)
1110 {
1111 seq_.save(p: cs.begin(), remain: cs.remain());
1112 if(BOOST_JSON_UNLIKELY(seq_.complete()))
1113 {
1114 BOOST_STATIC_CONSTEXPR source_location loc
1115 = BOOST_CURRENT_LOCATION;
1116 return fail(cs.begin(), error::syntax, &loc);
1117 }
1118 if(BOOST_JSON_LIKELY(size))
1119 {
1120 {
1121 bool r = is_key?
1122 h_.on_key_part( {start, size}, total, ec_ ):
1123 h_.on_string_part( {start, size}, total, ec_ );
1124
1125 if(BOOST_JSON_UNLIKELY(!r))
1126 {
1127 return fail(cs.begin());
1128 }
1129 }
1130 }
1131 return maybe_suspend(cs.end(), state::str8, total);
1132 }
1133 else if(BOOST_JSON_LIKELY(*cs == '\\'))
1134 {
1135 // flush unescaped run from input
1136 if(BOOST_JSON_LIKELY(size))
1137 {
1138 {
1139 bool r = is_key?
1140 h_.on_key_part( {start, size}, total, ec_ ):
1141 h_.on_string_part( {start, size}, total, ec_ );
1142
1143 if(BOOST_JSON_UNLIKELY(!r))

1145 return fail(cs.begin());
1146 }

1154 }

1156 {

1158 h_.on_key( {start, size}, total, ec_ ):
1159 h_.on_string( {start, size}, total, ec_ );
1160

1162 {

1164 }
1165 }

1171template<class Handler>
1172template<
1173 bool StackEmpty_/*,
1174 bool IsKey_,
1175 bool AllowBadUTF8_*/>
1176const char*
1177basic_parser<Handler>::
1178parse_escaped(
1179 const char* p,
1180 std::size_t total,
1181 std::integral_constant<bool, StackEmpty_> stack_empty,
1182 /*std::integral_constant<bool, IsKey_>*/ bool is_key,
1183 /*std::integral_constant<bool, AllowBadUTF8_>*/ bool allow_bad_utf8)
1184{
1185 //---------------------------------------------------------------
1186 //
1187 // To handle escapes, a local temporary buffer accumulates
1188 // the unescaped result. The algorithm attempts to fill the
1189 // buffer to capacity before invoking the handler.
1190 // In some cases the temporary buffer needs to be flushed
1191 // before it is full:
1192 // * When the closing double quote is seen
1193 // * When there in no more input (and more is expected later)
1194 // A goal of the algorithm is to call the handler as few times
1195 // as possible. Thus, when the first escape is encountered,
1196 // the algorithm attempts to fill the temporary buffer first.
1197 //
1198 auto const ev_too_large = is_key?
1199 error::key_too_large : error::string_too_large;
1200 auto const max_size = is_key?
1201 Handler::max_key_size : Handler::max_string_size;
1202 detail::clipped_const_stream cs(p, end_);
1203 detail::buffer<BOOST_JSON_STACK_BUFFER_SIZE> temp;
1204 int digit;
1205 char c;
1206 cs.clip(n: temp.max_size());
1207 if(! stack_empty && ! st_.empty())
1208 {
1209 state st;
1210 st_.pop(st);
1211 st_.pop(t&: total);
1212 switch(st)
1213 {
1214 default: BOOST_JSON_UNREACHABLE();
1215 case state::str2: goto do_str2;
1216 case state::str3: goto do_str3;
1217 case state::str4: goto do_str4;
1218 case state::str5: goto do_str5;
1219 case state::str6: goto do_str6;
1220 case state::str7: goto do_str7;
1221 case state::str8: goto do_str8;
1222 case state::sur1: goto do_sur1;
1223 case state::sur2: goto do_sur2;
1224 case state::sur3: goto do_sur3;
1225 case state::sur4: goto do_sur4;
1226 case state::sur5: goto do_sur5;
1227 case state::sur6: goto do_sur6;
1228 }
1229 }
1230 // Unescaped JSON is never larger than its escaped version.
1231 // To efficiently process only what will fit in the temporary buffer,
1232 // the size of the input stream is temporarily "clipped" to the size
1233 // of the temporary buffer.
1234 // handle escaped character
1235 BOOST_ASSERT(*cs == '\\');
1236 ++cs;
1237do_str3:
1238 if(BOOST_JSON_UNLIKELY(! cs))
1239 {
1240 if(BOOST_JSON_LIKELY(! temp.empty()))
1241 {
1242 BOOST_ASSERT(total <= max_size);
1243 if(BOOST_JSON_UNLIKELY(
1244 temp.size() > max_size - total))
1245 {
1246 BOOST_STATIC_CONSTEXPR source_location loc
1247 = BOOST_CURRENT_LOCATION;
1248 return fail(cs.begin(), ev_too_large, &loc);
1249 }
1250 total += temp.size();
1251 {
1252 bool r = is_key
1253 ? h_.on_key_part(temp.get(), total, ec_)
1254 : h_.on_string_part(temp.get(), total, ec_);
1255
1256 if(BOOST_JSON_UNLIKELY(!r))
1257 {
1258 return fail(cs.begin());
1259 }
1260 }
1261 temp.clear();
1262 }
1263 cs.clip(n: temp.max_size());
1264 if(BOOST_JSON_UNLIKELY(! cs))
1265 return maybe_suspend(cs.begin(), state::str3, total);
1266 }
1267 switch(*cs)
1268 {
1269 default:
1270 {
1271 BOOST_STATIC_CONSTEXPR source_location loc
1272 = BOOST_CURRENT_LOCATION;
1273 return fail(cs.begin(), error::syntax, &loc);
1274 }
1275 case '\x22': // '"'
1276 temp.push_back(ch: '\x22');
1277 ++cs;
1278 break;
1279 case '\\':
1280 temp.push_back(ch: '\\');
1281 ++cs;
1282 break;
1283 case '/':
1284 temp.push_back(ch: '/');
1285 ++cs;
1286 break;
1287 case 'b':
1288 temp.push_back(ch: '\x08');
1289 ++cs;
1290 break;
1291 case 'f':
1292 temp.push_back(ch: '\x0c');
1293 ++cs;
1294 break;
1295 case 'n':
1296 temp.push_back(ch: '\x0a');
1297 ++cs;
1298 break;
1299 case 'r':
1300 temp.push_back(ch: '\x0d');
1301 ++cs;
1302 break;
1303 case 't':
1304 temp.push_back(ch: '\x09');
1305 ++cs;
1306 break;
1307 case 'u':
1308 // utf16 escape
1309 //
1310 // fast path only when the buffer
1311 // is large enough for 2 surrogates
1312 if(BOOST_JSON_LIKELY(cs.remain() > 10))
1313 {
1314 // KRYSTIAN TODO: this could be done
1315 // with fewer instructions

2755 return cs.begin();
2756finish_dub:

2846 std::size_t size,
2847 system::error_code& ec)

2850 // on the last call to write_some
2851 if(! clean_)
2852 {
2853 // prevent UB
2854 if(! ec_)
2855 {
2856 BOOST_JSON_FAIL(ec_, error::exception);
2857 }
2858 }
2859 if(ec_)

2861 // error is sticky
2862 ec = ec_;

2864 }
2865 clean_ = false;
2866 more_ = more;
2867 end_ = data + size;
2868 const char* p;
2869 if(BOOST_JSON_LIKELY(st_.empty()))
2870 {
2871 // first time
2872 depth_ = opt_.max_depth;
2873 if(BOOST_JSON_UNLIKELY(
2874 ! h_.on_document_begin(ec_)))
2875 {
2876 ec = ec_;
2877 return 0;
2878 }
2879 p = parse_document(data, std::true_type());
2880 }
2881 else
2882 {
2883 p = parse_document(data, std::false_type());
2884 }
2885
2886 if(BOOST_JSON_LIKELY(p != sentinel()))
2887 {
2888 BOOST_ASSERT(! ec_);
2889 if(! done_)
2890 {
2891 done_ = true;
2892 h_.on_document_end(ec_);
2893 }
2894 }
2895 else
2896 {
2897 if(! ec_)
2898 {
2899 if(! more_)
2900 {
2901 BOOST_JSON_FAIL(ec_, error::incomplete);
2902 }
2903 else if(! st_.empty())
2904 {

2906 // the JSON document as possible, but still
2907 // consider the parse complete
2908 state st;
2909 st_.peek(st);
2910 if( st == state::doc3 &&
2911 ! done_)
2912 {
2913 done_ = true;
2914 h_.on_document_end(ec_);

2916 }
2917 }

2919 }
