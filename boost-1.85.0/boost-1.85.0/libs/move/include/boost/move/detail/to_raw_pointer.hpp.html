28namespace boost {
29namespace movelib {
30
31template <class T>
32BOOST_MOVE_FORCEINLINE T* to_raw_pointer(T* p)
33{ return p; }
34
35template <class Pointer>
36inline typename boost::movelib::pointer_element<Pointer>::type*
37to_raw_pointer(const Pointer &p)
38{ return ::boost::movelib::to_raw_pointer(p.operator->()); }
