28namespace boost {
29namespace movelib {
30
31template<class I>
32inline typename iterator_traits<I>::pointer iterator_arrow_result(const I &i)
33{ return i.operator->(); }
34
35template<class T>
36BOOST_MOVE_FORCEINLINE T * iterator_arrow_result(T *p)
37{ return p; }
38
39template<class It>
40class reverse_iterator
41{
42 public:
43 typedef typename boost::movelib::iterator_traits<It>::pointer pointer;
44 typedef typename boost::movelib::iterator_traits<It>::reference reference;
45 typedef typename boost::movelib::iterator_traits<It>::difference_type difference_type;
46 typedef typename boost::movelib::iterator_traits<It>::iterator_category iterator_category;
47 typedef typename boost::movelib::iterator_traits<It>::value_type value_type;
48
49
50 typedef It iterator_type;
51
52 inline reverse_iterator()
53 : m_current() //Value initialization to achieve "null iterators" (N3644)
54 {}
55
56 inline explicit reverse_iterator(It r)
57 : m_current(r)
58 {}
59
60 inline reverse_iterator(const reverse_iterator& r)
61 : m_current(r.base())
62 {}
63
64 template<class OtherIt>
65 inline
66 reverse_iterator( const reverse_iterator<OtherIt>& r
67 , typename boost::move_detail::enable_if_convertible<OtherIt, It>::type* =0
68 )
69 : m_current(r.base())
70 {}
71
72 inline reverse_iterator & operator=( const reverse_iterator& r)
73 { m_current = r.base(); return *this; }
74
75 template<class OtherIt>
76 inline typename boost::move_detail::enable_if_convertible<OtherIt, It, reverse_iterator &>::type
77 operator=( const reverse_iterator<OtherIt>& r)
78 { m_current = r.base(); return *this; }
79
80 inline It base() const
81 { return m_current; }
82
83 inline reference operator*() const
84 {
85 It temp(m_current);
86 --temp;
87 reference r = *temp;
88 return r;
89 }
90
91 inline pointer operator->() const
92 {
93 It temp(m_current);
94 --temp;
95 return (iterator_arrow_result)(temp);
96 }
97
98 inline reference operator[](difference_type off) const
99 {
100 return this->m_current[difference_type(-off - 1)];
101 }
102
103 inline reverse_iterator& operator++()
104 {
105 --m_current;
106 return *this;
107 }
108
109 inline reverse_iterator operator++(int)
110 {
111 reverse_iterator temp((*this));
112 --m_current;
113 return temp;
114 }
115
116 inline reverse_iterator& operator--()
117 {
118 ++m_current;
119 return *this;
120 }
121
122 inline reverse_iterator operator--(int)
123 {
124 reverse_iterator temp((*this));
125 ++m_current;
126 return temp;
127 }
128
129 inline friend bool operator==(const reverse_iterator& l, const reverse_iterator& r)
130 { return l.m_current == r.m_current; }
131
132 inline friend bool operator!=(const reverse_iterator& l, const reverse_iterator& r)
133 { return l.m_current != r.m_current; }
134
135 inline friend bool operator<(const reverse_iterator& l, const reverse_iterator& r)
136 { return l.m_current > r.m_current; }
137
138 inline friend bool operator<=(const reverse_iterator& l, const reverse_iterator& r)
139 { return l.m_current >= r.m_current; }
140
141 inline friend bool operator>(const reverse_iterator& l, const reverse_iterator& r)
142 { return l.m_current < r.m_current; }
143
144 inline friend bool operator>=(const reverse_iterator& l, const reverse_iterator& r)
145 { return l.m_current <= r.m_current; }
146
147 inline reverse_iterator& operator+=(difference_type off)
148 { m_current -= off; return *this; }
149
150 inline reverse_iterator& operator-=(difference_type off)
151 { m_current += off; return *this; }
152
153 inline friend reverse_iterator operator+(reverse_iterator l, difference_type off)
154 { return (l += off); }
155
156 inline friend reverse_iterator operator+(difference_type off, reverse_iterator r)
157 { return (r += off); }
158
159 inline friend reverse_iterator operator-(reverse_iterator l, difference_type off)
160 { return (l-= off); }
161
162 inline friend difference_type operator-(const reverse_iterator& l, const reverse_iterator& r)
163 { return r.m_current - l.m_current; }
164
165 private:
166 It m_current; // the wrapped iterator
167};
168
169template< class Iterator >
170inline reverse_iterator<Iterator> make_reverse_iterator( Iterator i )
171{ return reverse_iterator<Iterator>(i); }
