42//compatible with a pointer type T*:
43//When either Y* is convertible to T*
44//Y is U[N] and T is U cv []
45template<class U, class T>
46struct def_del_compatible_cond
47 : bmupmu::is_convertible<U*, T*>
48{};
49
50template<class U, class T, std::size_t N>
51struct def_del_compatible_cond<U[N], T[]>
52 : def_del_compatible_cond<U[], T[]>
53{};
54
55template<class U, class T, class Type = bmupmu::nat>
56struct enable_def_del
57 : bmupmu::enable_if_c<def_del_compatible_cond<U, T>::value, Type>
58{};
59

61//// enable_defdel_call
62////////////////////////////////////////
63
64//When 2nd is T[N], 1st(*)[N] shall be convertible to T(*)[N];
65//When 2nd is T[], 1st(*)[] shall be convertible to T(*)[];
66//Otherwise, 1st* shall be convertible to 2nd*.
67
68template<class U, class T, class Type = bmupmu::nat>
69struct enable_defdel_call
70 : public enable_def_del<U, T, Type>
71{};
72
73template<class U, class T, class Type>
74struct enable_defdel_call<U, T[], Type>
75 : public enable_def_del<U[], T[], Type>
76{};

78template<class U, class T, class Type, std::size_t N>
79struct enable_defdel_call<U, T[N], Type>
80 : public enable_def_del<U[N], T[N], Type>
81{};

98template<bool B>
99struct is_array_del
100{};
101
102template<class T>
103void call_delete(T *p, is_array_del<true>)
104{
105 delete [] p;
106}
107
108template<class T>
109void call_delete(T *p, is_array_del<false>)
110{
111 delete p;
112}

114template< class T, class U
115 , bool enable = def_del_compatible_cond< U, T>::value &&
116 !move_upmu::is_array<T>::value &&
117 !move_upmu::is_same<typename move_upmu::remove_cv<T>::type, void>::value &&
118 !move_upmu::is_same<typename move_upmu::remove_cv<U>::type, typename move_upmu::remove_cv<T>::type>::value
119 >
120struct missing_virtual_destructor_default_delete
121{ static const bool value = !move_upmu::has_virtual_destructor<T>::value; };
122
123template<class T, class U>
124struct missing_virtual_destructor_default_delete<T, U, false>
125{ static const bool value = false; };

149//!The class template <tt>default_delete</tt> serves as the default deleter
150//!(destruction policy) for the class template <tt>unique_ptr</tt>.
151//!
152//! \tparam T The type to be deleted. It may be an incomplete type
153template <class T>
154struct default_delete
155{
156 //! Default constructor.
157 //!
158 BOOST_CONSTEXPR default_delete()
159 //Avoid "defaulted on its first declaration must not have an exception-specification" error for GCC 4.6
160 #if !defined(BOOST_GCC) || (BOOST_GCC < 40600 && BOOST_GCC >= 40700) || defined(BOOST_MOVE_DOXYGEN_INVOKED)
161 BOOST_NOEXCEPT
162 #endif
163 #if !defined(BOOST_NO_CXX11_DEFAULTED_FUNCTIONS) || defined(BOOST_MOVE_DOXYGEN_INVOKED)
164 = default;
165 #else
166 {};
167 #endif
168
169 #if defined(BOOST_MOVE_DOXYGEN_INVOKED)
170 //! Trivial copy constructor
171 //!
172 default_delete(const default_delete&) BOOST_NOEXCEPT = default;
173 //! Trivial assignment
174 //!
175 default_delete &operator=(const default_delete&) BOOST_NOEXCEPT = default;
176 #else
177 typedef typename bmupmu::remove_extent<T>::type element_type;
178 #endif
179
180 //! <b>Effects</b>: Constructs a default_delete object from another <tt>default_delete<U></tt> object.
181 //!
182 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
183 //! - If T is not an array type and U* is implicitly convertible to T*.
184 //! - If T is an array type and U* is a more CV qualified pointer to remove_extent<T>::type.
185 template <class U>
186 default_delete(const default_delete<U>&
187 BOOST_MOVE_DOCIGN(BOOST_MOVE_I typename bmupd::enable_def_del<U BOOST_MOVE_I T>::type* =0)
188 ) BOOST_NOEXCEPT
189 {
190 //If T is not an array type, U derives from T
191 //and T has no virtual destructor, then you have a problem
192 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor<default_delete, U>::value ));
193 }
194
195 //! <b>Effects</b>: Constructs a default_delete object from another <tt>default_delete<U></tt> object.
196 //!
197 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
198 //! - If T is not an array type and U* is implicitly convertible to T*.
199 //! - If T is an array type and U* is a more CV qualified pointer to remove_extent<T>::type.
200 template <class U>
201 BOOST_MOVE_DOC1ST(default_delete&,
202 typename bmupd::enable_def_del<U BOOST_MOVE_I T BOOST_MOVE_I default_delete &>::type)
203 operator=(const default_delete<U>&) BOOST_NOEXCEPT
204 {
205 //If T is not an array type, U derives from T
206 //and T has no virtual destructor, then you have a problem
207 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor<default_delete, U>::value ));
208 return *this;
209 }
210
211 //! <b>Effects</b>: if T is not an array type, calls <tt>delete</tt> on static_cast<T*>(ptr),
212 //! otherwise calls <tt>delete[]</tt> on static_cast<remove_extent<T>::type*>(ptr).
213 //!
214 //! <b>Remarks</b>: If U is an incomplete type, the program is ill-formed.
215 //! This operator shall not participate in overload resolution unless:
216 //! - T is not an array type and U* is convertible to T*, OR
217 //! - T is an array type, and remove_cv<U>::type is the same type as
218 //! remove_cv<remove_extent<T>::type>::type and U* is convertible to remove_extent<T>::type*.
219 template <class U>
220 BOOST_MOVE_DOC1ST(void, typename bmupd::enable_defdel_call<U BOOST_MOVE_I T BOOST_MOVE_I void>::type)
221 operator()(U* ptr) const BOOST_NOEXCEPT
222 {
223 //U must be a complete type
224 BOOST_MOVE_STATIC_ASSERT(sizeof(U) > 0);
225 //If T is not an array type, U derives from T
226 //and T has no virtual destructor, then you have a problem
227 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor<default_delete, U>::value ));
228 element_type * const p = static_cast<element_type*>(ptr);
229 move_upd::call_delete(p, move_upd::is_array_del<bmupmu::is_array<T>::value>());
230 }
231
232 //! <b>Effects</b>: Same as <tt>(*this)(static_cast<element_type*>(nullptr))</tt>.
233 //!
234 void operator()(BOOST_MOVE_DOC0PTR(bmupd::nullptr_type)) const BOOST_NOEXCEPT
235 { BOOST_MOVE_STATIC_ASSERT(sizeof(element_type) > 0); }
236};
