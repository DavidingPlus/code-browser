32//!\file
33//! Describes the smart pointer unique_ptr, a drop-in replacement for std::unique_ptr,
34//! usable also from C++03 compilers.
35//!
36//! Main differences from std::unique_ptr to avoid heavy dependencies,
37//! specially in C++03 compilers:
38//! - <tt>operator < </tt> uses pointer <tt>operator < </tt>instead of <tt>std::less<common_type></tt>.
39//! This avoids dependencies on <tt>std::common_type</tt> and <tt>std::less</tt>
40//! (<tt><type_traits>/<functional></tt> headers). In C++03 this avoid pulling Boost.Typeof and other
41//! cascading dependencies. As in all Boost platforms <tt>operator <</tt> on raw pointers and
42//! other smart pointers provides strict weak ordering in practice this should not be a problem for users.
43//! - assignable from literal 0 for compilers without nullptr
44//! - <tt>unique_ptr<T[]></tt> is constructible and assignable from <tt>unique_ptr<U[]></tt> if
45//! cv-less T and cv-less U are the same type and T is more CV qualified than U.
46

67template <class D>
68struct deleter_types
69{
70 typedef typename bmupmu::add_lvalue_reference<D>::type del_ref;
71 typedef typename bmupmu::add_const_lvalue_reference<D>::type del_cref;
72 #ifndef BOOST_NO_CXX11_RVALUE_REFERENCES
73 typedef typename bmupmu::if_c
74 < bmupmu::is_lvalue_reference<D>::value, D, del_cref >::type deleter_arg_type1;
75 typedef typename bmupmu::remove_reference<D>::type && deleter_arg_type2;
76 #else
77 typedef typename bmupmu::if_c
78 < is_noncopyable<D>::value, bmupmu::nat, del_cref>::type non_ref_deleter_arg1;
79 typedef typename bmupmu::if_c< bmupmu::is_lvalue_reference<D>::value
80 , D, non_ref_deleter_arg1 >::type deleter_arg_type1;

82 #endif

88template <class P, class D, bool = bmupmu::is_unary_function<D>::value || bmupmu::is_reference<D>::value >
89struct unique_ptr_data
90{
91 typedef typename deleter_types<D>::deleter_arg_type1 deleter_arg_type1;
92 typedef typename deleter_types<D>::del_ref del_ref;
93 typedef typename deleter_types<D>::del_cref del_cref;
94
95 inline unique_ptr_data() BOOST_NOEXCEPT
96 : m_p(), d()
97 {}
98
99 inline explicit unique_ptr_data(P p) BOOST_NOEXCEPT
100 : m_p(p), d()
101 {}
102
103 inline unique_ptr_data(P p, deleter_arg_type1 d1) BOOST_NOEXCEPT
104 : m_p(p), d(d1)
105 {}
106
107 template <class U>
108 inline unique_ptr_data(P p, BOOST_FWD_REF(U) d1) BOOST_NOEXCEPT
109 : m_p(p), d(::boost::forward<U>(d1))
110 {}
111
112 inline del_ref deleter() { return d; }
113 inline del_cref deleter() const{ return d; }
114
115 P m_p;
116 D d;
117
118 private:
119 unique_ptr_data& operator=(const unique_ptr_data&);
120 unique_ptr_data(const unique_ptr_data&);
121};

123template <class P, class D>
124struct unique_ptr_data<P, D, false>
125 : private D
126{
127 typedef typename deleter_types<D>::deleter_arg_type1 deleter_arg_type1;
128 typedef typename deleter_types<D>::del_ref del_ref;
129 typedef typename deleter_types<D>::del_cref del_cref;
130
131 inline unique_ptr_data() BOOST_NOEXCEPT
132 : D(), m_p()
133 {}
134
135 inline explicit unique_ptr_data(P p) BOOST_NOEXCEPT
136 : D(), m_p(p)
137 {}
138
139 inline unique_ptr_data(P p, deleter_arg_type1 d1) BOOST_NOEXCEPT
140 : D(d1), m_p(p)
141 {}
142
143 template <class U>
144 inline unique_ptr_data(P p, BOOST_FWD_REF(U) d) BOOST_NOEXCEPT
145 : D(::boost::forward<U>(d)), m_p(p)
146 {}
147
148 inline del_ref deleter() BOOST_NOEXCEPT { return static_cast<del_ref>(*this); }
149 inline del_cref deleter() const BOOST_NOEXCEPT { return static_cast<del_cref>(*this); }
150
151 P m_p;
152
153 private:
154 unique_ptr_data& operator=(const unique_ptr_data&);
155 unique_ptr_data(const unique_ptr_data&);
156};

308//! A unique pointer is an object that owns another object and
309//! manages that other object through a pointer.
310//!
311//! More precisely, a unique pointer is an object u that stores a pointer to a second object p and will dispose
312//! of p when u is itself destroyed (e.g., when leaving block scope). In this context, u is said to own p.
313//!
314//! The mechanism by which u disposes of p is known as p's associated deleter, a function object whose correct
315//! invocation results in p's appropriate disposition (typically its deletion).
316//!
317//! Let the notation u.p denote the pointer stored by u, and let u.d denote the associated deleter. Upon request,
318//! u can reset (replace) u.p and u.d with another pointer and deleter, but must properly dispose of its owned
319//! object via the associated deleter before such replacement is considered completed.
320//!
321//! Additionally, u can, upon request, transfer ownership to another unique pointer u2. Upon completion of
322//! such a transfer, the following postconditions hold:
323//! - u2.p is equal to the pre-transfer u.p,
324//! - u.p is equal to nullptr, and
325//! - if the pre-transfer u.d maintained state, such state has been transferred to u2.d.
326//!
327//! As in the case of a reset, u2 must properly dispose of its pre-transfer owned object via the pre-transfer
328//! associated deleter before the ownership transfer is considered complete.
329//!
330//! Each object of a type U instantiated from the unique_ptr template specified in this subclause has the strict
331//! ownership semantics, specified above, of a unique pointer. In partial satisfaction of these semantics, each
332//! such U is MoveConstructible and MoveAssignable, but is not CopyConstructible nor CopyAssignable.
333//! The template parameter T of unique_ptr may be an incomplete type.
334//!
335//! The uses of unique_ptr include providing exception safety for dynamically allocated memory, passing
336//! ownership of dynamically allocated memory to a function, and returning dynamically allocated memory from
337//! a function.
338//!
339//! If T is an array type (e.g. unique_ptr<MyType[]>) the interface is slightly altered:
340//! - Pointers to types derived from T are rejected by the constructors, and by reset.
341//! - The observers <tt>operator*</tt> and <tt>operator-></tt> are not provided.
342//! - The indexing observer <tt>operator[]</tt> is provided.
343//!
344//! \tparam T Provides the type of the stored pointer.
345//! \tparam D The deleter type:
346//! - The default type for the template parameter D is default_delete. A client-supplied template argument
347//! D shall be a function object type, lvalue-reference to function, or lvalue-reference to function object type
348//! for which, given a value d of type D and a value ptr of type unique_ptr<T, D>::pointer, the expression
349//! d(ptr) is valid and has the effect of disposing of the pointer as appropriate for that deleter.
350//! - If the deleter's type D is not a reference type, D shall satisfy the requirements of Destructible.
351//! - If the type <tt>remove_reference<D>::type::pointer</tt> exists, it shall satisfy the requirements of NullablePointer.
352template <class T, class D = default_delete<T> >
353class unique_ptr
354{
355 #if defined(BOOST_MOVE_DOXYGEN_INVOKED)
356 public:
357 unique_ptr(const unique_ptr&) = delete;
358 unique_ptr& operator=(const unique_ptr&) = delete;
359 private:
360 #else
361 BOOST_MOVABLE_BUT_NOT_COPYABLE(unique_ptr)
362
363 typedef bmupmu::pointer_type<T, D > pointer_type_obtainer;
364 typedef bmupd::unique_ptr_data
365 <typename pointer_type_obtainer::type, D> data_type;
366 typedef typename bmupd::deleter_types<D>::deleter_arg_type1 deleter_arg_type1;
367 typedef typename bmupd::deleter_types<D>::deleter_arg_type2 deleter_arg_type2;
368 data_type m_data;
369 #endif
370
371 public:
372 //! If the type <tt>remove_reference<D>::type::pointer</tt> exists, then it shall be a
373 //! synonym for <tt>remove_reference<D>::type::pointer</tt>. Otherwise it shall be a
374 //! synonym for T*.
375 typedef typename BOOST_MOVE_SEEDOC(pointer_type_obtainer::type) pointer;
376 //! If T is an array type, then element_type is equal to T. Otherwise, if T is a type
377 //! in the form U[], element_type is equal to U.
378 typedef typename BOOST_MOVE_SEEDOC(bmupmu::remove_extent<T>::type) element_type;
379 typedef D deleter_type;
380
381 //! <b>Requires</b>: D shall satisfy the requirements of DefaultConstructible, and
382 //! that construction shall not throw an exception.
383 //!
384 //! <b>Effects</b>: Constructs a unique_ptr object that owns nothing, value-initializing the
385 //! stored pointer and the stored deleter.
386 //!
387 //! <b>Postconditions</b>: <tt>get() == nullptr</tt>. <tt>get_deleter()</tt> returns a reference to the stored deleter.
388 //!
389 //! <b>Remarks</b>: If this constructor is instantiated with a pointer type or reference type
390 //! for the template argument D, the program is ill-formed.
391 inline BOOST_CONSTEXPR unique_ptr() BOOST_NOEXCEPT
392 : m_data()
393 {
394 //If this constructor is instantiated with a pointer type or reference type
395 //for the template argument D, the program is ill-formed.
396 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_pointer<D>::value);
397 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_reference<D>::value);
398 }
399
400 //! <b>Effects</b>: Same as <tt>unique_ptr()</tt> (default constructor).
401 //!
402 inline BOOST_CONSTEXPR unique_ptr(BOOST_MOVE_DOC0PTR(bmupd::nullptr_type)) BOOST_NOEXCEPT
403 : m_data()
404 {
405 //If this constructor is instantiated with a pointer type or reference type
406 //for the template argument D, the program is ill-formed.
407 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_pointer<D>::value);
408 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_reference<D>::value);
409 }

411 //! <b>Requires</b>: D shall satisfy the requirements of DefaultConstructible, and
412 //! that construction shall not throw an exception.
413 //!
414 //! <b>Effects</b>: Constructs a unique_ptr which owns p, initializing the stored pointer
415 //! with p and value initializing the stored deleter.
416 //!
417 //! <b>Postconditions</b>: <tt>get() == p</tt>. <tt>get_deleter()</tt> returns a reference to the stored deleter.
418 //!
419 //! <b>Remarks</b>: If this constructor is instantiated with a pointer type or reference type
420 //! for the template argument D, the program is ill-formed.
421 //! This constructor shall not participate in overload resolution unless:
422 //! - If T is not an array type and Pointer is implicitly convertible to pointer.
423 //! - If T is an array type and Pointer is a more CV qualified pointer to element_type.
424 template<class Pointer>
425 inline explicit unique_ptr(Pointer p
426 BOOST_MOVE_DOCIGN(BOOST_MOVE_I typename bmupd::enable_up_ptr<T BOOST_MOVE_I Pointer BOOST_MOVE_I pointer>::type* =0)
427 ) BOOST_NOEXCEPT
428 : m_data(p)
429 {
430 //If T is not an array type, element_type_t<Pointer> derives from T
431 //it uses the default deleter and T has no virtual destructor, then you have a problem
432 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor
433 <D, typename bmupd::get_element_type<Pointer>::type>::value ));
434 //If this constructor is instantiated with a pointer type or reference type
435 //for the template argument D, the program is ill-formed.
436 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_pointer<D>::value);
437 BOOST_MOVE_STATIC_ASSERT(!bmupmu::is_reference<D>::value);
438 }

440 //!The signature of this constructor depends upon whether D is a reference type.
441 //! - If D is non-reference type A, then the signature is <tt>unique_ptr(pointer p, const A& d)</tt>.
442 //! - If D is an lvalue-reference type A&, then the signature is <tt>unique_ptr(pointer p, A& d)</tt>.
443 //! - If D is an lvalue-reference type const A&, then the signature is <tt>unique_ptr(pointer p, const A& d)</tt>.
444 //!
445 //!
446 //! <b>Requires</b>: Either
447 //! - D is not an lvalue-reference type and d is an lvalue or const rvalue.
448 //! D shall satisfy the requirements of CopyConstructible, and the copy constructor of D
449 //! shall not throw an exception. This unique_ptr will hold a copy of d.
450 //! - D is an lvalue-reference type and d is an lvalue. the type which D references need not be CopyConstructible nor
451 //! MoveConstructible. This unique_ptr will hold a D which refers to the lvalue d.
452 //!
453 //! <b>Effects</b>: Constructs a unique_ptr object which owns p, initializing the stored pointer with p and
454 //! initializing the deleter as described above.
455 //!
456 //! <b>Postconditions</b>: <tt>get() == p</tt>. <tt>get_deleter()</tt> returns a reference to the stored deleter. If D is a
457 //! reference type then <tt>get_deleter()</tt> returns a reference to the lvalue d.
458 //!
459 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
460 //! - If T is not an array type and Pointer is implicitly convertible to pointer.
461 //! - If T is an array type and Pointer is a more CV qualified pointer to element_type.
462 template<class Pointer>
463 inline unique_ptr(Pointer p, BOOST_MOVE_SEEDOC(deleter_arg_type1) d1
464 BOOST_MOVE_DOCIGN(BOOST_MOVE_I typename bmupd::enable_up_ptr<T BOOST_MOVE_I Pointer BOOST_MOVE_I pointer>::type* =0)
465 ) BOOST_NOEXCEPT
466 : m_data(p, d1)
467 {
468 //If T is not an array type, element_type_t<Pointer> derives from T
469 //it uses the default deleter and T has no virtual destructor, then you have a problem
470 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor
471 <D, typename bmupd::get_element_type<Pointer>::type>::value ));
472 }

474 //! <b>Effects</b>: Same effects as <tt>template<class Pointer> unique_ptr(Pointer p, deleter_arg_type1 d1)</tt>
475 //! and additionally <tt>get() == nullptr</tt>
476 inline unique_ptr(BOOST_MOVE_DOC0PTR(bmupd::nullptr_type), BOOST_MOVE_SEEDOC(deleter_arg_type1) d1) BOOST_NOEXCEPT
477 : m_data(pointer(), d1)
478 {}

480 //! The signature of this constructor depends upon whether D is a reference type.
481 //! - If D is non-reference type A, then the signature is <tt>unique_ptr(pointer p, A&& d)</tt>.
482 //! - If D is an lvalue-reference type A&, then the signature is <tt>unique_ptr(pointer p, A&& d)</tt>.
483 //! - If D is an lvalue-reference type const A&, then the signature is <tt>unique_ptr(pointer p, const A&& d)</tt>.
484 //!
485 //! <b>Requires</b>: Either
486 //! - D is not an lvalue-reference type and d is a non-const rvalue. D
487 //! shall satisfy the requirements of MoveConstructible, and the move constructor
488 //! of D shall not throw an exception. This unique_ptr will hold a value move constructed from d.
489 //! - D is an lvalue-reference type and d is an rvalue, the program is ill-formed.
490 //!
491 //! <b>Effects</b>: Constructs a unique_ptr object which owns p, initializing the stored pointer with p and
492 //! initializing the deleter as described above.
493 //!
494 //! <b>Postconditions</b>: <tt>get() == p</tt>. <tt>get_deleter()</tt> returns a reference to the stored deleter. If D is a
495 //! reference type then <tt>get_deleter()</tt> returns a reference to the lvalue d.
496 //!
497 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
498 //! - If T is not an array type and Pointer is implicitly convertible to pointer.
499 //! - If T is an array type and Pointer is a more CV qualified pointer to element_type.
500 template<class Pointer>
501 inline unique_ptr(Pointer p, BOOST_MOVE_SEEDOC(deleter_arg_type2) d2
502 BOOST_MOVE_DOCIGN(BOOST_MOVE_I typename bmupd::enable_up_ptr<T BOOST_MOVE_I Pointer BOOST_MOVE_I pointer>::type* =0)
503 ) BOOST_NOEXCEPT
504 : m_data(p, ::boost::move(d2))
505 {
506 //If T is not an array type, element_type_t<Pointer> derives from T
507 //it uses the default deleter and T has no virtual destructor, then you have a problem
508 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor
509 <D, typename bmupd::get_element_type<Pointer>::type>::value ));
510 }

512 //! <b>Effects</b>: Same effects as <tt>template<class Pointer> unique_ptr(Pointer p, deleter_arg_type2 d2)</tt>
513 //! and additionally <tt>get() == nullptr</tt>
514 inline unique_ptr(BOOST_MOVE_DOC0PTR(bmupd::nullptr_type), BOOST_MOVE_SEEDOC(deleter_arg_type2) d2) BOOST_NOEXCEPT
515 : m_data(pointer(), ::boost::move(d2))

518 //! <b>Requires</b>: If D is not a reference type, D shall satisfy the requirements of MoveConstructible.
519 //! Construction of the deleter from an rvalue of type D shall not throw an exception.
520 //!
521 //! <b>Effects</b>: Constructs a unique_ptr by transferring ownership from u to *this. If D is a reference type,
522 //! this deleter is copy constructed from u's deleter; otherwise, this deleter is move constructed from u's
523 //! deleter.
524 //!
525 //! <b>Postconditions</b>: <tt>get()</tt> yields the value u.get() yielded before the construction. <tt>get_deleter()</tt>
526 //! returns a reference to the stored deleter that was constructed from u.get_deleter(). If D is a
527 //! reference type then <tt>get_deleter()</tt> and <tt>u.get_deleter()</tt> both reference the same lvalue deleter.
528 inline unique_ptr(BOOST_RV_REF(unique_ptr) u) BOOST_NOEXCEPT
529 : m_data(u.release(), ::boost::move_if_not_lvalue_reference<D>(u.get_deleter()))
530 {}
531
532 //! <b>Requires</b>: If E is not a reference type, construction of the deleter from an rvalue of type E shall be
533 //! well formed and shall not throw an exception. Otherwise, E is a reference type and construction of the
534 //! deleter from an lvalue of type E shall be well formed and shall not throw an exception.
535 //!
536 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
537 //! - <tt>unique_ptr<U, E>::pointer</tt> is implicitly convertible to pointer,
538 //! - U is not an array type, and
539 //! - either D is a reference type and E is the same type as D, or D is not a reference type and E is
540 //! implicitly convertible to D.
541 //!
542 //! <b>Effects</b>: Constructs a unique_ptr by transferring ownership from u to *this. If E is a reference type,
543 //! this deleter is copy constructed from u's deleter; otherwise, this deleter is move constructed from u's deleter.
544 //!
545 //! <b>Postconditions</b>: <tt>get()</tt> yields the value <tt>u.get()</tt> yielded before the construction. <tt>get_deleter()</tt>
546 //! returns a reference to the stored deleter that was constructed from <tt>u.get_deleter()</tt>.
547 template <class U, class E>
548 inline unique_ptr( BOOST_RV_REF_BEG_IF_CXX11 unique_ptr<U, E> BOOST_RV_REF_END_IF_CXX11 u
549 BOOST_MOVE_DOCIGN(BOOST_MOVE_I typename bmupd::enable_up_moveconv_constr<T BOOST_MOVE_I D BOOST_MOVE_I U BOOST_MOVE_I E>::type* =0)
550 ) BOOST_NOEXCEPT
551 : m_data(u.release(), ::boost::move_if_not_lvalue_reference<E>(u.get_deleter()))
552 {
553 //If T is not an array type, U derives from T
554 //it uses the default deleter and T has no virtual destructor, then you have a problem
555 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor
556 <D, typename unique_ptr<U, E>::pointer>::value ));
557 }

559 //! <b>Requires</b>: The expression <tt>get_deleter()(get())</tt> shall be well formed, shall have well-defined behavior,
560 //! and shall not throw exceptions.
561 //!
562 //! <b>Effects</b>: If <tt>get() == nullpt1r</tt> there are no effects. Otherwise <tt>get_deleter()(get())</tt>.
563 //!
564 //! <b>Note</b>: The use of default_delete requires T to be a complete type
565 ~unique_ptr()
566 { if(m_data.m_p) m_data.deleter()(m_data.m_p); }
567
568 //! <b>Requires</b>: If D is not a reference type, D shall satisfy the requirements of MoveAssignable
569 //! and assignment of the deleter from an rvalue of type D shall not throw an exception. Otherwise, D
570 //! is a reference type; <tt>remove_reference<D>::type</tt> shall satisfy the CopyAssignable requirements and
571 //! assignment of the deleter from an lvalue of type D shall not throw an exception.
572 //!
573 //! <b>Effects</b>: Transfers ownership from u to *this as if by calling <tt>reset(u.release())</tt> followed
574 //! by <tt>get_deleter() = std::forward<D>(u.get_deleter())</tt>.
575 //!
576 //! <b>Returns</b>: *this.
577 unique_ptr& operator=(BOOST_RV_REF(unique_ptr) u) BOOST_NOEXCEPT
578 {
579 this->reset(u.release());
580 m_data.deleter() = ::boost::move_if_not_lvalue_reference<D>(u.get_deleter());
581 return *this;
582 }

585 //! well-formed and shall not throw an exception. Otherwise, E is a reference type and assignment of the

587 //!
588 //! <b>Remarks</b>: This operator shall not participate in overload resolution unless:
589 //! - <tt>unique_ptr<U, E>::pointer</tt> is implicitly convertible to pointer and

694 //! <b>Requires</b>: The expression <tt>get_deleter()(get())</tt> shall be well formed, shall have well-defined behavior,
695 //! and shall not throw exceptions.
696 //!
697 //! <b>Effects</b>: assigns p to the stored pointer, and then if the old value of the stored pointer, old_p, was not
698 //! equal to nullptr, calls <tt>get_deleter()(old_p)</tt>. Note: The order of these operations is significant
699 //! because the call to <tt>get_deleter()</tt> may destroy *this.
700 //!
701 //! <b>Postconditions</b>: <tt>get() == p</tt>. Note: The postcondition does not hold if the call to <tt>get_deleter()</tt>
702 //! destroys *this since <tt>this->get()</tt> is no longer a valid expression.
703 //!
704 //! <b>Remarks</b>: This constructor shall not participate in overload resolution unless:
705 //! - If T is not an array type and Pointer is implicitly convertible to pointer.
706 //! - If T is an array type and Pointer is a more CV qualified pointer to element_type.
707 template<class Pointer>
708 BOOST_MOVE_DOC1ST(void, typename bmupd::enable_up_ptr<T BOOST_MOVE_I Pointer BOOST_MOVE_I pointer BOOST_MOVE_I void>::type)
709 reset(Pointer p) BOOST_NOEXCEPT
710 {
711 //If T is not an array type, element_type_t<Pointer> derives from T
712 //it uses the default deleter and T has no virtual destructor, then you have a problem
713 BOOST_MOVE_STATIC_ASSERT(( !bmupd::missing_virtual_destructor
714 <D, typename bmupd::get_element_type<Pointer>::type>::value ));
715 pointer tmp = m_data.m_p;
716 m_data.m_p = p;
717 if(tmp) m_data.deleter()(tmp);
718 }
719
720 //! <b>Requires</b>: The expression <tt>get_deleter()(get())</tt> shall be well formed, shall have well-defined behavior,
721 //! and shall not throw exceptions.
722 //!
723 //! <b>Effects</b>: assigns nullptr to the stored pointer, and then if the old value of the stored pointer, old_p, was not
724 //! equal to nullptr, calls <tt>get_deleter()(old_p)</tt>. Note: The order of these operations is significant
725 //! because the call to <tt>get_deleter()</tt> may destroy *this.
726 //!
727 //! <b>Postconditions</b>: <tt>get() == p</tt>. Note: The postcondition does not hold if the call to <tt>get_deleter()</tt>
728 //! destroys *this since <tt>this->get()</tt> is no longer a valid expression.
729 void reset() BOOST_NOEXCEPT
730 { this->reset(pointer()); }
731
732 //! <b>Effects</b>: Same as <tt>reset()</tt>
733 //!
734 void reset(BOOST_MOVE_DOC0PTR(bmupd::nullptr_type)) BOOST_NOEXCEPT
735 { this->reset(); }
736
737 //! <b>Requires</b>: <tt>get_deleter()</tt> shall be swappable and shall not throw an exception under swap.
738 //!
739 //! <b>Effects</b>: Invokes swap on the stored pointers and on the stored deleters of *this and u.
740 void swap(unique_ptr& u) BOOST_NOEXCEPT
741 {
742 ::boost::adl_move_swap(m_data.m_p, u.m_data.m_p);
743 ::boost::adl_move_swap(m_data.deleter(), u.m_data.deleter());
744 }
