53bool is_backslash(uchar *p, uchar *end, int &len)
54{
55 if (*p == '\\') {
56 len = 1;
57 return true;
58 }
59 else if (*p == '?' && *(p+1) == '?' && (p+2 < end && *(p+2) == '/')) {
60 len = 3;
61 return true;
62 }
63 return false;
64}

67// Special wrapper class holding the current cursor position
68uchar_wrapper::uchar_wrapper (uchar *base_cursor, std::size_t column)
69 : base_cursor(base_cursor), column(column)
70{}
71
72uchar_wrapper& uchar_wrapper::operator++()
73{
74 ++base_cursor;
75 ++column;
76 return *this;
77}
78
79uchar_wrapper& uchar_wrapper::operator--()
80{
81 --base_cursor;
82 --column;
83 return *this;
84}
85
86uchar uchar_wrapper::operator* () const
87{
88 return *base_cursor;
89}
90
91uchar_wrapper::operator uchar *() const
92{
93 return base_cursor;
94}
95
96std::ptrdiff_t
97operator- (uchar_wrapper const& lhs, uchar_wrapper const& rhs)
98{
99 return lhs.base_cursor - rhs.base_cursor;
100}
