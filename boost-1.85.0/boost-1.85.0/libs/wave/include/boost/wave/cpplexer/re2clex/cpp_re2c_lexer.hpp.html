61template <typename IteratorT,
62 typename PositionT = boost::wave::util::file_position_type,
63 typename TokenT = lex_token<PositionT> >
64class lexer
65{
66public:
67 typedef TokenT token_type;
68 typedef typename token_type::string_type string_type;
69
70 lexer(IteratorT const &first, IteratorT const &last,
71 PositionT const &pos, boost::wave::language_support language_);
72 ~lexer();
73
74 token_type& get(token_type&);
75 void set_position(PositionT const &pos)
76 {
77 // set position has to change the file name and line number only
78 filename = pos.get_file();
79 scanner.line = pos.get_line();
80// scanner.column = scanner.curr_column = pos.get_column();
81 scanner.file_name = filename.c_str();
82 }
83#if BOOST_WAVE_SUPPORT_PRAGMA_ONCE != 0
84 bool has_include_guards(std::string& guard_name) const
85 {
86 return guards.detected(guard_name);
87 }
88#endif
89
90 // error reporting from the re2c generated lexer
91 static int report_error(Scanner<IteratorT> const* s, int code, char const *, ...);
92
93private:
94 static char const *tok_names[];
95
96 Scanner<IteratorT> scanner;
97 string_type filename;
98 string_type value;
99 bool at_eof;
100 boost::wave::language_support language;
101#if BOOST_WAVE_SUPPORT_PRAGMA_ONCE != 0
102 include_guards<token_type> guards;
103#endif
104
105#if BOOST_WAVE_SUPPORT_THREADING == 0
106 static token_cache<string_type> const cache;
107#else
108 token_cache<string_type> const cache;
109#endif
110};

112///////////////////////////////////////////////////////////////////////////////
113// initialize cpp lexer
114template <typename IteratorT, typename PositionT, typename TokenT>
115inline
116lexer<IteratorT, PositionT, TokenT>::lexer(IteratorT const &first,
117 IteratorT const &last, PositionT const &pos,
118 boost::wave::language_support language_)
119 : scanner(first, last),
120 filename(pos.get_file()), at_eof(false), language(language_)
121#if BOOST_WAVE_SUPPORT_THREADING != 0
122 , cache()
123#endif
124{
125 using namespace std; // some systems have memset in std
126 scanner.line = pos.get_line();
127 scanner.column = scanner.curr_column = pos.get_column();
128 scanner.error_proc = report_error;
129 scanner.file_name = filename.c_str();
130
131#if BOOST_WAVE_SUPPORT_MS_EXTENSIONS != 0
132 scanner.enable_ms_extensions = true;
133#else
134 scanner.enable_ms_extensions = false;
135#endif
136
137#if BOOST_WAVE_SUPPORT_VARIADICS_PLACEMARKERS != 0
138 scanner.act_in_c99_mode = boost::wave::need_c99(language: language_);
139#endif
140
141#if BOOST_WAVE_SUPPORT_IMPORT_KEYWORD != 0
142 scanner.enable_import_keyword = !boost::wave::need_c99(language_);
143#else
144 scanner.enable_import_keyword = false;
145#endif
146
147 scanner.detect_pp_numbers = boost::wave::need_prefer_pp_numbers(language: language_);
148 scanner.single_line_only = boost::wave::need_single_line(language: language_);
149
150#if BOOST_WAVE_SUPPORT_CPP0X != 0
151 scanner.act_in_cpp0x_mode = boost::wave::need_cpp0x(language: language_);
152#else
153 scanner.act_in_cpp0x_mode = false;
154#endif
155
156#if BOOST_WAVE_SUPPORT_CPP2A != 0
157 scanner.act_in_cpp2a_mode = boost::wave::need_cpp2a(language: language_);
158 scanner.act_in_cpp0x_mode = boost::wave::need_cpp2a(language: language_)
159 || boost::wave::need_cpp0x(language: language_);
160#else
161 scanner.act_in_cpp2a_mode = false;
162#endif
163}

173///////////////////////////////////////////////////////////////////////////////
174// get the next token from the input stream
175template <typename IteratorT, typename PositionT, typename TokenT>
176inline TokenT&
177lexer<IteratorT, PositionT, TokenT>::get(TokenT& result)
178{
179 if (at_eof)
180 return result = token_type(); // return T_EOI
181
182 std::size_t actline = scanner.line;
183 token_id id = token_id(scan(&scanner));
184
185 switch (id) {
186 case T_IDENTIFIER:
187 // test identifier characters for validity (throws if invalid chars found)
188 value = string_type((char const *)scanner.tok,
189 scanner.cur-scanner.tok);
190 if (!boost::wave::need_no_character_validation(language))
191 impl::validate_identifier_name(value, actline, scanner.column, filename);
192 break;
193
194 case T_STRINGLIT:
195 case T_CHARLIT:
196 case T_RAWSTRINGLIT:
197 // test literal characters for validity (throws if invalid chars found)
198 value = string_type((char const *)scanner.tok,
199 scanner.cur-scanner.tok);
200 if (boost::wave::need_convert_trigraphs(language))
201 value = impl::convert_trigraphs(value);
202 if (!boost::wave::need_no_character_validation(language))
203 impl::validate_literal(value, actline, scanner.column, filename);
204 break;
205
206 case T_PP_HHEADER:
207 case T_PP_QHEADER:
208 case T_PP_INCLUDE:
209 // convert to the corresponding ..._next token, if appropriate
210 {
211 value = string_type((char const *)scanner.tok,
212 scanner.cur-scanner.tok);
213
214#if BOOST_WAVE_SUPPORT_INCLUDE_NEXT != 0
215 // Skip '#' and whitespace and see whether we find an 'include_next' here.
216 typename string_type::size_type start = value.find("include");
217 if (value.compare(start, 12, "include_next", 12) == 0)
218 id = token_id(id | AltTokenType);
219#endif
220 break;
221 }
222
223 case T_LONGINTLIT: // supported in C++11, C99 and long_long mode
224 value = string_type((char const *)scanner.tok,
225 scanner.cur-scanner.tok);
226 if (!boost::wave::need_long_long(language)) {
227 // syntax error: not allowed in C++ mode
228 BOOST_WAVE_LEXER_THROW(lexing_exception, invalid_long_long_literal,
229 value.c_str(), actline, scanner.column, filename.c_str());
230 }
231 break;
232
233 case T_OCTALINT:
234 case T_DECIMALINT:
235 case T_HEXAINT:
236 case T_INTLIT:
237 case T_FLOATLIT:
238 case T_FIXEDPOINTLIT:
239 case T_CCOMMENT:
240 case T_CPPCOMMENT:
241 case T_SPACE:
242 case T_SPACE2:
243 case T_ANY:
244 case T_PP_NUMBER:
245 value = string_type((char const *)scanner.tok,
246 scanner.cur-scanner.tok);
247 break;
248
249 case T_EOF:
250 // T_EOF is returned as a valid token, the next call will return T_EOI,
251 // i.e. the actual end of input
252 at_eof = true;
253 value.clear();
254 break;
255
256 case T_OR_TRIGRAPH:
257 case T_XOR_TRIGRAPH:
258 case T_LEFTBRACE_TRIGRAPH:
259 case T_RIGHTBRACE_TRIGRAPH:
260 case T_LEFTBRACKET_TRIGRAPH:
261 case T_RIGHTBRACKET_TRIGRAPH:
262 case T_COMPL_TRIGRAPH:
263 case T_POUND_TRIGRAPH:
264 if (boost::wave::need_convert_trigraphs(language)) {
265 value = cache.get_token_value(BASEID_FROM_TOKEN(id));
266 }
267 else {
268 value = string_type((char const *)scanner.tok,
269 scanner.cur-scanner.tok);
270 }
271 break;
272
273 case T_ANY_TRIGRAPH:
274 if (boost::wave::need_convert_trigraphs(language)) {
275 value = impl::convert_trigraph(
276 string_type((char const *)scanner.tok,
277 scanner.cur-scanner.tok));
278 }
279 else {
280 value = string_type((char const *)scanner.tok,
281 scanner.cur-scanner.tok);
282 }
283 break;
284
285 default:
286 if (CATEGORY_FROM_TOKEN(id) != EXTCATEGORY_FROM_TOKEN(id) ||
287 IS_CATEGORY(id, UnknownTokenType))
288 {
289 value = string_type((char const *)scanner.tok,
290 scanner.cur-scanner.tok);
291 }
292 else {
293 value = cache.get_token_value(id);
294 }
295 break;
296 }
297
298// std::cerr << boost::wave::get_token_name(id) << ": " << value << std::endl;
299
300 // the re2c lexer reports the new line number for newline tokens
301 result = token_type(id, value, PositionT(filename, actline, scanner.column));
302
303#if BOOST_WAVE_SUPPORT_PRAGMA_ONCE != 0
304 return guards.detect_guard(result);
305#else
306 return result;
307#endif
308}

310template <typename IteratorT, typename PositionT, typename TokenT>
311inline int
312lexer<IteratorT, PositionT, TokenT>::report_error(Scanner<IteratorT> const *s, int errcode,
313 char const *msg, ...)
314{
315 BOOST_ASSERT(0 != s);
316 BOOST_ASSERT(0 != msg);
317
318 using namespace std; // some system have vsprintf in namespace std
319
320 char buffer[200]; // should be large enough
321 va_list params;
322 va_start(params, msg);
323 vsprintf(s: buffer, format: msg, arg: params);
324 va_end(params);
325
326 BOOST_WAVE_LEXER_THROW_VAR(lexing_exception, errcode, buffer, s->line,
327 s->column, s->file_name);
328 // BOOST_UNREACHABLE_RETURN(0);
329 return 0;
330}

338template <typename IteratorT,
339 typename PositionT = boost::wave::util::file_position_type,
340 typename TokenT = typename lexer<IteratorT, PositionT>::token_type>
341class lex_functor
342: public lex_input_interface_generator<TokenT>
343{
344public:
345 typedef TokenT token_type;
346
347 lex_functor(IteratorT const &first, IteratorT const &last,
348 PositionT const &pos, boost::wave::language_support language)
349 : re2c_lexer(first, last, pos, language)
350 {}
351 virtual ~lex_functor() {}
352
353 // get the next token from the input stream
354 token_type& get(token_type& result) BOOST_OVERRIDE { return re2c_lexer.get(result); }
355 void set_position(PositionT const &pos) BOOST_OVERRIDE { re2c_lexer.set_position(pos); }
356#if BOOST_WAVE_SUPPORT_PRAGMA_ONCE != 0
357 bool has_include_guards(std::string& guard_name) const BOOST_OVERRIDE
358 { return re2c_lexer.has_include_guards(guard_name); }
359#endif
360
361private:
362 lexer<IteratorT, PositionT, TokenT> re2c_lexer;
363};

391///////////////////////////////////////////////////////////////////////////////
392//
393// The 'new_lexer' function allows the opaque generation of a new lexer object.
394// It is coupled to the iterator type to allow to decouple the lexer/iterator
395// configurations at compile time.
396//
397// This function is declared inside the cpp_lex_token.hpp file, which is
398// referenced by the source file calling the lexer and the source file, which
399// instantiates the lex_functor. But it is defined here, so it will be
400// instantiated only while compiling the source file, which instantiates the
401// lex_functor. While the cpp_re2c_token.hpp file may be included everywhere,
402// this file (cpp_re2c_lexer.hpp) should be included only once. This allows
403// to decouple the lexer interface from the lexer implementation and reduces
404// compilation time.
405//
406///////////////////////////////////////////////////////////////////////////////
407
408template <typename IteratorT, typename PositionT, typename TokenT>
409BOOST_WAVE_RE2C_NEW_LEXER_INLINE
410lex_input_interface<TokenT> *
411new_lexer_gen<IteratorT, PositionT, TokenT>::new_lexer(IteratorT const &first,
412 IteratorT const &last, PositionT const &pos,
413 boost::wave::language_support language)
414{
415 using re2clex::lex_functor;
416 return new lex_functor<IteratorT, PositionT, TokenT>(first, last, pos, language);
417}
