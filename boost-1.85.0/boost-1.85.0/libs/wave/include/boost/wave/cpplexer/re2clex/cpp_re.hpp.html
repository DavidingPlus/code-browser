92#define RE2C_ASSERT BOOST_ASSERT

94template<typename Iterator>
95int get_one_char(Scanner<Iterator> *s)
96{
97 RE2C_ASSERT(s->first <= s->act && s->act <= s->last);
98 if (s->act < s->last)
99 return *(s->act)++;
100 return -1;
101}
102
103template<typename Iterator>
104std::ptrdiff_t rewind_stream (Scanner<Iterator> *s, int cnt)
105{
106 std::advance(s->act, cnt);
107 RE2C_ASSERT(s->first <= s->act && s->act <= s->last);
108 return std::distance(s->first, s->act);
109}
110
111template<typename Iterator>
112std::size_t get_first_eol_offset(Scanner<Iterator>* s)
113{
114 if (!AQ_EMPTY(s->eol_offsets))
115 {
116 return s->eol_offsets->queue[s->eol_offsets->head];
117 }
118 else
119 {
120 return (unsigned int)-1;
121 }
122}
123
124template<typename Iterator>
125void adjust_eol_offsets(Scanner<Iterator>* s, std::size_t adjustment)
126{
127 aq_queue q;
128 std::size_t i;
129
130 if (!s->eol_offsets)
131 s->eol_offsets = aq_create();
132
133 q = s->eol_offsets;
134
135 if (AQ_EMPTY(q))
136 return;
137
138 i = q->head;
139 while (i != q->tail)
140 {
141 if (adjustment > q->queue[i])
142 q->queue[i] = 0;
143 else
144 q->queue[i] -= adjustment;
145 ++i;
146 if (i == q->max_size)
147 i = 0;
148 }
149 if (adjustment > q->queue[i])
150 q->queue[i] = 0;
151 else
152 q->queue[i] -= adjustment;
153}

155template<typename Iterator>
156int count_backslash_newlines(Scanner<Iterator> *s, uchar *cursor)
157{
158 std::size_t diff, offset;
159 int skipped = 0;
160
161 /* figure out how many backslash-newlines skipped over unknowingly. */
162 diff = cursor - s->bot;
163 offset = get_first_eol_offset(s);
164 while (offset <= diff && offset != (unsigned int)-1)
165 {
166 skipped++;
167 aq_pop(s->eol_offsets);
168 offset = get_first_eol_offset(s);
169 }
170 return skipped;
171}

173BOOST_WAVE_DECL bool is_backslash(uchar *p, uchar *end, int &len);

175#define BOOST_WAVE_BSIZE 196608
176template<typename Iterator>
177uchar *fill(Scanner<Iterator> *s, uchar *cursor)
178{
179 using namespace std; // some systems have memcpy etc. in namespace std
180 if(!s->eof)
181 {
182 uchar* p;
183 std::ptrdiff_t cnt = s->tok - s->bot;
184 if(cnt)
185 {
186 if (NULL == s->lim)
187 s->lim = s->top;
188 size_t length = s->lim - s->tok;
189 if(length > 0){
190 memmove(s->bot, s->tok, length);
191 }
192 s->tok = s->cur = s->bot;
193 s->ptr -= cnt;
194 cursor -= cnt;
195 s->lim -= cnt;
196 adjust_eol_offsets(s, cnt);
197 }
198
199 if((s->top - s->lim) < BOOST_WAVE_BSIZE)
200 {
201 uchar *buf = (uchar*) malloc(((s->lim - s->bot) + BOOST_WAVE_BSIZE)*sizeof(uchar));
202 if (buf == 0)
203 {
204 (*s->error_proc)(s, lexing_exception::unexpected_error,
205 "Out of memory!");
206
207 /* get the scanner to stop */
208 *cursor = 0;
209 return cursor;
210 }
211
212 size_t length = s->lim - s->tok;
213 if(length > 0){
214 memmove(buf, s->tok, length);
215 }
216 s->tok = s->cur = buf;
217 s->ptr = &buf[s->ptr - s->bot];
218 cursor = &buf[cursor - s->bot];
219 s->lim = &buf[s->lim - s->bot];
220 s->top = &s->lim[BOOST_WAVE_BSIZE];
221 free(s->bot);
222 s->bot = buf;
223 }
224
225 cnt = std::distance(s->act, s->last);
226 if (cnt > BOOST_WAVE_BSIZE)
227 cnt = BOOST_WAVE_BSIZE;
228 uchar * dst = s->lim;
229 for (std::ptrdiff_t idx = 0; idx < cnt; ++idx)
230 {
231 *dst++ = *s->act++;
232 }
233
234 if (cnt != BOOST_WAVE_BSIZE)
235 {
236 s->eof = &s->lim[cnt]; *(s->eof)++ = '\0';
237 }
238
239 /* backslash-newline erasing time */
240
241 /* first scan for backslash-newline and erase them */
242 /* a backslash-newline combination can be 2 (regular) or 4 (trigraph backslash) chars */
243 /* start checking 3 chars within the old buffer, if possible */
244 for (p = (std::max)(s->lim - 3, s->cur); p < s->lim + cnt - 2; ++p)
245 {
246 int len = 0;
247 /* is there a backslash, and room afterwards for a newline? */
248 if (is_backslash(p, s->lim + cnt, len) && ((p + len) < (s->lim + cnt)))
249 {
250 if (*(p+len) == '\n')
251 {
252 int offset = len + 1;
253 memmove(p, p + offset, s->lim + cnt - p - offset);
254 cnt -= offset;
255 --p;
256 aq_enqueue(s->eol_offsets, p - s->bot + 1);
257 }
258 else if (*(p+len) == '\r')
259 {
260 /* is there also room for a newline, and is one present? */
261 if (((p + len + 1) < s->lim + cnt) && (*(p+len+1) == '\n'))
262 {
263 int offset = len + 2;
264 memmove(p, p + offset, s->lim + cnt - p - offset);
265 cnt -= offset;
266 --p;
267 }
268 else
269 {
270 int offset = len + 1;
271 memmove(p, p + offset, s->lim + cnt - p - offset);
272 cnt -= offset;
273 --p;
274 }
275 aq_enqueue(s->eol_offsets, p - s->bot + 1);
276 }
277 }
278 }
279
280 /* FIXME: the following code should be fixed to recognize correctly the
281 trigraph backslash token */
282
283 /* check to see if what we just read ends in a backslash */
284 if (cnt >= 2)
285 {
286 uchar last = s->lim[cnt-1];
287 uchar last2 = s->lim[cnt-2];
288 /* check \ EOB */
289 if (last == '\\')
290 {
291 int next = get_one_char(s);
292 /* check for \ \n or \ \r or \ \r \n straddling the border */
293 if (next == '\n')
294 {
295 --cnt; /* chop the final \, we've already read the \n. */
296 aq_enqueue(s->eol_offsets, cnt + (s->lim - s->bot));
297 }
298 else if (next == '\r')
299 {
300 int next2 = get_one_char(s);
301 if (next2 == '\n')
302 {
303 --cnt; /* skip the backslash */
304 }
305 else
306 {
307 /* rewind one, and skip one char */
308 rewind_stream(s, -1);
309 --cnt;
310 }
311 aq_enqueue(s->eol_offsets, cnt + (s->lim - s->bot));
312 }
313 else if (next != -1) /* -1 means end of file */
314 {
315 /* next was something else, so rewind the stream */
316 rewind_stream(s, -1);
317 }
318 }
319 /* check \ \r EOB */
320 else if (last == '\r' && last2 == '\\')
321 {
322 int next = get_one_char(s);
323 if (next == '\n')
324 {
325 cnt -= 2; /* skip the \ \r */
326 }
327 else
328 {
329 /* rewind one, and skip two chars */
330 rewind_stream(s, -1);
331 cnt -= 2;
332 }
333 aq_enqueue(s->eol_offsets, cnt + (s->lim - s->bot));
334 }
335 /* check \ \n EOB */
336 else if (last == '\n' && last2 == '\\')
337 {
338 cnt -= 2;
339 aq_enqueue(s->eol_offsets, cnt + (s->lim - s->bot));
340 }
341 }
342
343 s->lim += cnt;
344 if (s->eof) /* eof needs adjusting if we erased backslash-newlines */
345 {
346 s->eof = s->lim;
347 *(s->eof)++ = '\0';
348 }
349 }
350 return cursor;
351}

353///////////////////////////////////////////////////////////////////////////////
354// Special wrapper class holding the current cursor position
355struct BOOST_WAVE_DECL uchar_wrapper
356{
357 uchar_wrapper (uchar *base_cursor, std::size_t column = 1);
358
359 uchar_wrapper& operator++();
360
361 uchar_wrapper& operator--();
362
363 uchar operator* () const;
364
365 operator uchar *() const;
366
367 friend BOOST_WAVE_DECL std::ptrdiff_t
368 operator- (uchar_wrapper const& lhs, uchar_wrapper const& rhs);
369
370 uchar *base_cursor;
371 std::size_t column;
372};

376template<typename Iterator>
377boost::wave::token_id scan(Scanner<Iterator> *s)
378{
379 BOOST_ASSERT(0 != s->error_proc); // error handler must be given
380
381 uchar_wrapper cursor (s->tok = s->cur, s->column = s->curr_column);
382 uchar_wrapper marker (s->ptr);
383 uchar_wrapper limit (s->lim);

386 string_type rawstringdelim; // for use with C++11 raw string literals
