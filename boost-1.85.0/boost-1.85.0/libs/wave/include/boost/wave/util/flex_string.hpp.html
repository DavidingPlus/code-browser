138 template <class Pod, class T>
139 inline void pod_fill(Pod* b, Pod* e, T c)
140 {
141 switch ((e - b) & 7)
142 {
143 case 0:
144 while (b != e)
145 {
146 *b = c; ++b; BOOST_FALLTHROUGH;
147 case 7: *b = c; ++b; BOOST_FALLTHROUGH;
148 case 6: *b = c; ++b; BOOST_FALLTHROUGH;
149 case 5: *b = c; ++b; BOOST_FALLTHROUGH;
150 case 4: *b = c; ++b; BOOST_FALLTHROUGH;
151 case 3: *b = c; ++b; BOOST_FALLTHROUGH;
152 case 2: *b = c; ++b; BOOST_FALLTHROUGH;
153 case 1: *b = c; ++b;
154 }
155 }
156 }

305template <typename E, class A = std::allocator<E> >
306class SimpleStringStorage
307{
308 // The "public" below exists because MSVC can't do template typedefs
309public:
310 struct Data
311 {
312 Data() : pEnd_(buffer_), pEndOfMem_(buffer_) { buffer_[0] = E(0); }
313
314 E* pEnd_;
315 E* pEndOfMem_;
316 E buffer_[1];
317 };
318 static const Data emptyString_;
319
320 typedef typename boost::allocator_size_type<A>::type size_type;
321
322private:
323 Data* pData_;
324
325 void Init(size_type size, size_type capacity)
326 {
327 BOOST_ASSERT(size <= capacity);
328 if (capacity == 0)
329 {
330 pData_ = const_cast<Data*>(&emptyString_);
331 }
332 else
333 {
334 // 11-17-2000: comment added:
335 // No need to allocate (capacity + 1) to
336 // accommodate the terminating 0, because Data already
337 // has one character in there
338 pData_ = static_cast<Data*>(
339 malloc(sizeof(Data) + capacity * sizeof(E)));
340 if (!pData_) boost::throw_exception(e: std::bad_alloc());
341 pData_->pEnd_ = pData_->buffer_ + size;
342 pData_->pEndOfMem_ = pData_->buffer_ + capacity;
343 }
344 }

346private:
347 // Warning - this doesn't initialize pData_. Used in reserve()
348 SimpleStringStorage()
349 { }

351public:

353 typedef E* iterator;

355 typedef A allocator_type;

395 {
396 BOOST_ASSERT(begin() <= end());
397 if (capacity() > 0)
398 {
399 BOOST_WAVE_COMPILE_TIME_ASSERT(pData_ != &emptyString_);
400 free(pData_);
401 }

404 iterator begin()
405 { return pData_->buffer_; }

410 iterator end()
411 { return pData_->pEnd_; }
412
413 const_iterator end() const
414 { return pData_->pEnd_; }
415
416 size_type size() const
417 { return pData_->pEnd_ - pData_->buffer_; }
418
419 size_type max_size() const
420 { return std::size_t(-1) / sizeof(E) - sizeof(Data) - 1; }
421
422 size_type capacity() const
423 { return pData_->pEndOfMem_ - pData_->buffer_; }

425 void reserve(size_type res_arg)
426 {
427 size_type cap = capacity();
428 if (res_arg <= cap)
429 {
430 // @@@ insert shrinkage here if you wish
431 return;
432 }
433
434 if (cap == 0)
435 {
436 Init(size: 0, capacity: res_arg);
437 }
438 else
439 {
440 const size_type sz = size();
441
442 void* p = realloc(pData_,
443 sizeof(Data) + res_arg * sizeof(E));
444 if (!p) boost::throw_exception(e: std::bad_alloc());
445
446 if (p != pData_)
447 {
448 pData_ = static_cast<Data*>(p);
449 pData_->pEnd_ = pData_->buffer_ + sz;
450 }
451 pData_->pEndOfMem_ = pData_->buffer_ + res_arg;
452 }
453 }
454
455 void append(const E* s, size_type sz)
456 {
457 const size_type neededCapacity = size() + sz;
458
459 if (capacity() < neededCapacity)
460 {
461 const iterator b = begin();
462 static std::less_equal<const E*> le;
463 if (le(b, s) && le(s, end()))
464 {
465 // aliased
466 const size_type offset = s - b;
467 reserve(res_arg: neededCapacity);
468 s = begin() + offset;
469 }
470 else
471 {
472 reserve(res_arg: neededCapacity);
473 }
474 }
475 flex_string_details::pod_copy(s, s + sz, end());
476 pData_->pEnd_ += sz;
477 }

489 void resize(size_type newSize, E fill)
490 {
491 const int delta = int(newSize - size());
492 if (delta == 0) return;
493
494 if (delta > 0)
495 {

504 }

521 { return A(); }
