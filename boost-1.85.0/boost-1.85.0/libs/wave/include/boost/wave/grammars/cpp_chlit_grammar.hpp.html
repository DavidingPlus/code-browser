67///////////////////////////////////////////////////////////////////////////////
68//
69// compose a multibyte character literal
70//
71///////////////////////////////////////////////////////////////////////////////
72 struct compose_character_literal {
73
74 template <typename A1, typename A2, typename A3, typename A4>
75 struct result
76 {
77 typedef void type;
78 };
79
80 void
81 operator()(boost::uint32_t& value, bool long_lit, bool& overflow,
82 boost::uint32_t character) const
83 {
84 // The following assumes that wchar_t is max. 32 Bit
85 BOOST_STATIC_ASSERT(sizeof(wchar_t) <= 4);
86
87 static boost::uint32_t masks[] = {
88 0x000000ff, 0x0000ffff, 0x00ffffff, 0xffffffff
89 };
90 static boost::uint32_t overflow_masks[] = {
91 0xff000000, 0xffff0000, 0xffffff00, 0xffffffff
92 };
93
94 if (long_lit) {
95 // make sure no overflow will occur below
96 if ((value & overflow_masks[sizeof(wchar_t)-1]) != 0) {
97 overflow |= true;
98 }
99 else {
100 // calculate the new value (avoiding a warning regarding
101 // shifting count >= size of the type)
102 value <<= CHAR_BIT * (sizeof(wchar_t)-1);
103 value <<= CHAR_BIT;
104 value |= character & masks[sizeof(wchar_t)-1];
105 }
106 }
107 else {
108 // make sure no overflow will occur below
109 if ((value & overflow_masks[sizeof(char)-1]) != 0) {
110 overflow |= true;
111 }
112 else {
113 // calculate the new value
114 value <<= CHAR_BIT * sizeof(char);
115 value |= character & masks[sizeof(char)-1];
116 }
117 }
118 }
119 };
120 phoenix::function<compose_character_literal> const compose;
121
122} // namespace impl

124///////////////////////////////////////////////////////////////////////////////
125// define, whether the rule's should generate some debug output
126#define TRACE_CHLIT_GRAMMAR \
127 bool(BOOST_SPIRIT_DEBUG_FLAGS_CPP & BOOST_SPIRIT_DEBUG_FLAGS_CHLIT_GRAMMAR) \
128 /**/
129
130struct chlit_grammar :
131 public boost::spirit::classic::grammar<chlit_grammar,
132 closures::chlit_closure::context_t>
133{
134 chlit_grammar()
135 : overflow(false)
136 {
137 BOOST_SPIRIT_DEBUG_TRACE_GRAMMAR_NAME(*this, "chlit_grammar",
138 TRACE_CHLIT_GRAMMAR);
139 }
140
141 // no need for copy constructor/assignment operator
142 chlit_grammar(chlit_grammar const&);
143 chlit_grammar& operator=(chlit_grammar const&);
144
145 template <typename ScannerT>
146 struct definition
147 {
148 typedef boost::spirit::classic::rule<
149 ScannerT, closures::chlit_closure::context_t>
150 rule_t;
151
152 rule_t ch_lit;
153
154 definition(chlit_grammar const &self)
155 {
156 using namespace boost::spirit::classic;
157 namespace phx = phoenix;
158
159 // special parsers for '\x..' and L'\x....'
160 typedef uint_parser<
161 unsigned int, 16, 1, 2 * sizeof(char)
162 > hex_char_parser_type;
163 typedef uint_parser<
164 unsigned int, 16, 1, 2 * sizeof(wchar_t)
165 > hex_wchar_parser_type;
166
167 // the rule for a character literal
168 ch_lit
169 = eps_p[(self.value = phx::val(v: 0), self.long_lit = phx::val(v: false))]
170 >> !ch_p(ch: 'L')[self.long_lit = phx::val(v: true)]
171 >> ch_p(ch: '\'')
172 >> +( (
173 ch_p(ch: '\\')
174 >> ( ch_p(ch: 'a') // BEL
175 [
176 impl::compose(self.value, self.long_lit,
177 phx::var(v&: self.overflow), phx::val(v: 0x07))
178 ]
179 | ch_p(ch: 'b') // BS
180 [
181 impl::compose(self.value, self.long_lit,
182 phx::var(v&: self.overflow), phx::val(v: 0x08))
183 ]
184 | ch_p(ch: 't') // HT
185 [
186 impl::compose(self.value, self.long_lit,
187 phx::var(v&: self.overflow), phx::val(v: 0x09))
188 ]
189 | ch_p(ch: 'n') // NL
190 [
191 impl::compose(self.value, self.long_lit,
192 phx::var(v&: self.overflow), phx::val(v: 0x0a))
193 ]
194 | ch_p(ch: 'v') // VT
195 [
196 impl::compose(self.value, self.long_lit,
197 phx::var(v&: self.overflow), phx::val(v: 0x0b))
198 ]
199 | (ch_p(ch: 'e') | ch_p(ch: 'E')) // ESC
200 [
201 impl::compose(self.value, self.long_lit,
202 phx::var(v&: self.overflow), phx::val(v: 0x1b))
203 ]
204 | ch_p(ch: 'f') // FF
205 [
206 impl::compose(self.value, self.long_lit,
207 phx::var(v&: self.overflow), phx::val(v: 0x0c))
208 ]
209 | ch_p(ch: 'r') // CR
210 [
211 impl::compose(self.value, self.long_lit,
212 phx::var(v&: self.overflow), phx::val(v: 0x0d))
213 ]
214 | ch_p(ch: '?')
215 [
216 impl::compose(self.value, self.long_lit,
217 phx::var(v&: self.overflow), phx::val(v: '?'))
218 ]
219 | ch_p(ch: '\'')
220 [
221 impl::compose(self.value, self.long_lit,
222 phx::var(v&: self.overflow), phx::val(v: '\''))
223 ]
224 | ch_p(ch: '\"')
225 [
226 impl::compose(self.value, self.long_lit,
227 phx::var(v&: self.overflow), phx::val(v: '\"'))
228 ]
229 | ch_p(ch: '\\')
230 [
231 impl::compose(self.value, self.long_lit,
232 phx::var(v&: self.overflow), phx::val(v: '\\'))
233 ]
234 | ch_p(ch: 'x')
235 >> if_p(cond: self.long_lit)
236 [
237 hex_wchar_parser_type()
238 [
239 impl::compose(self.value, self.long_lit,
240 phx::var(v&: self.overflow), phx::arg1)
241 ]
242 ]
243 .else_p
244 [
245 hex_char_parser_type()
246 [
247 impl::compose(self.value, self.long_lit,
248 phx::var(v&: self.overflow), phx::arg1)
249 ]
250 ]
251 | ch_p(ch: 'u')
252 >> uint_parser<unsigned int, 16, 4, 4>()
253 [
254 impl::compose(self.value, self.long_lit,
255 phx::var(v&: self.overflow), phx::arg1)
256 ]
257 | ch_p(ch: 'U')
258 >> uint_parser<unsigned int, 16, 8, 8>()
259 [
260 impl::compose(self.value, self.long_lit,
261 phx::var(v&: self.overflow), phx::arg1)
262 ]
263 | uint_parser<unsigned int, 8, 1, 3>()
264 [
265 impl::compose(self.value, self.long_lit,
266 phx::var(v&: self.overflow), phx::arg1)
267 ]
268 )
269 )
270 | ~eps_p(ch_p(ch: '\'')) >> anychar_p
271 [
272 impl::compose(self.value, self.long_lit,
273 phx::var(v&: self.overflow), phx::arg1)
274 ]
275 )
276 >> ch_p(ch: '\'')
277 ;
278
279 BOOST_SPIRIT_DEBUG_TRACE_RULE(ch_lit, TRACE_CHLIT_GRAMMAR);
280 }
281
282 // start rule of this grammar
283 rule_t const& start() const
284 { return ch_lit; }
285 };
286
287 // flag signaling integer overflow during value composition
288 mutable bool overflow;
289};
290
291#undef TRACE_CHLIT_GRAMMAR

300#if BOOST_WAVE_SEPARATE_GRAMMAR_INSTANTIATION != 0
301#define BOOST_WAVE_CHLITGRAMMAR_GEN_INLINE
302#else
303#define BOOST_WAVE_CHLITGRAMMAR_GEN_INLINE inline

305
306template <typename IntegralResult, typename TokenT>
307BOOST_WAVE_CHLITGRAMMAR_GEN_INLINE
308IntegralResult
309chlit_grammar_gen<IntegralResult, TokenT>::evaluate(TokenT const &token, value_error &status)
310{
311 using namespace boost::spirit::classic;
312
313chlit_grammar g;
314IntegralResult result = 0;
315typename TokenT::string_type const &token_val = token.get_value();
316parse_info<typename TokenT::string_type::const_iterator> hit =
317 parse(token_val.begin(), token_val.end(), g[spirit_assign_actor(result)]);
318
319 if (!hit.hit) {
320 BOOST_WAVE_THROW(preprocess_exception, ill_formed_character_literal,
321 token_val.c_str(), token.get_position());
322 }
323 else {
324 // range check
325 if ('L' == token_val[0]) {
326 // recognized wide character
327 if (g.overflow ||
328 result > (IntegralResult)(std::numeric_limits<wchar_t>::max)())
329 {
330 // out of range
331 status = error_character_overflow;
332 }
333 }
334 else {
335 // recognized narrow ('normal') character
336 if (g.overflow ||
337 result > (IntegralResult)(std::numeric_limits<unsigned char>::max)())
338 {
339 // out of range
340 status = error_character_overflow;
341 }
342 }
343 }
344 return result;
345}
