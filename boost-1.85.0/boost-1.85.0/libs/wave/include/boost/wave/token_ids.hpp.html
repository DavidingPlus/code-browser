333///////////////////////////////////////////////////////////////////////////////
334// token_category and token_id may be used together
335constexpr token_id operator&(token_id a, token_category b)
336{
337 return static_cast<token_id>(
338 static_cast<std::uint32_t>(a) & static_cast<std::uint32_t>(b));
339}
340
341constexpr token_id operator|(token_id a, token_category b)
342{
343 return static_cast<token_id>(
344 static_cast<std::uint32_t>(a) | static_cast<std::uint32_t>(b));
345}
346
347constexpr bool operator==(token_category a, token_id b)
348{
349 return static_cast<std::uint32_t>(a) == static_cast<std::uint32_t>(b);
350}

381///////////////////////////////////////////////////////////////////////////////
382// verify whether the given token(-id) represents a valid pp_token
383inline bool is_pp_token(boost::wave::token_id id)
384{
385 return (id & boost::wave::PPTokenFlag) ? true : false;
386}
387
388template <typename TokenT>
389inline bool is_pp_token(TokenT const& tok)
390{
391 return is_pp_token(id: boost::wave::token_id(tok));
392}
