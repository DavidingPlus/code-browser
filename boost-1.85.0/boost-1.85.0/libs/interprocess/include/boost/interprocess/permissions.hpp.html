55 static winapi::interprocess_all_access_security unrestricted;

67//!The permissions class represents permissions to be set to shared memory or
68//!files, that can be constructed form usual permission representations:
69//!a SECURITY_ATTRIBUTES pointer in windows or ORed rwx chmod integer in UNIX.
70class permissions
71{
72 #if !defined(BOOST_INTERPROCESS_DOXYGEN_INVOKED)
73
74 #if defined(BOOST_INTERPROCESS_WINDOWS)
75 typedef void* os_permissions_type;
76 #else
77 typedef ::mode_t os_permissions_type;
78 #endif
79 os_permissions_type m_perm;
80
81 #endif //#ifndef BOOST_INTERPROCESS_DOXYGEN_INVOKED
82
83 public:
84 //!Constructs a permissions object from a user provided os-dependent
85 //!permissions.
86 permissions(os_permissions_type type) BOOST_NOEXCEPT
87 : m_perm(type)
88 {}
89
90 //!Constructs a default permissions object:
91 //!A null security attributes pointer for windows or 0644
92 //!for UNIX.
93 permissions() BOOST_NOEXCEPT
94 { set_default(); }
95
96 //!Sets permissions to default values:
97 //!A null security attributes pointer for windows or 0644
98 //!for UNIX.
99 void set_default() BOOST_NOEXCEPT
100 {
101 #if defined (BOOST_INTERPROCESS_WINDOWS)
102 m_perm = 0;
103 #else
104 m_perm = 0644;
105 #endif
106 }
107
108 //!Sets permissions to unrestricted access:
109 //!A null DACL for windows or 0666 for UNIX.
110 void set_unrestricted() BOOST_NOEXCEPT
111 {
112 #if defined (BOOST_INTERPROCESS_WINDOWS)
113 m_perm = &ipcdetail::unrestricted_permissions_holder<0>::unrestricted;
114 #else
115 m_perm = 0666;
116 #endif
117 }
118
119 //!Sets permissions from a user provided os-dependent
120 //!permissions.
121 void set_permissions(os_permissions_type perm) BOOST_NOEXCEPT
122 { m_perm = perm; }
123
124 //!Returns stored os-dependent
125 //!permissions
126 os_permissions_type get_permissions() const BOOST_NOEXCEPT
127 { return m_perm; }
128};
