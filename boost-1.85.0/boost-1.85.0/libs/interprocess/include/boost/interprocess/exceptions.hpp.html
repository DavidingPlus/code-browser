30namespace boost {

32namespace interprocess {

34//!This class is the base class of all exceptions
35//!thrown by boost::interprocess
36class BOOST_SYMBOL_VISIBLE interprocess_exception : public std::exception
37{
38 public:
39 interprocess_exception(const char *err) BOOST_NOEXCEPT
40 : m_err(other_error)
41 {
42 BOOST_TRY { m_str = err; }
43 BOOST_CATCH(...) {} BOOST_CATCH_END
44 }
45
46 interprocess_exception(const error_info &err_info, const char *str = 0)
47 : m_err(err_info)
48 {
49 BOOST_TRY{
50 if(m_err.get_native_error() != 0){
51 fill_system_message(system_error: m_err.get_native_error(), str&: m_str);
52 }
53 else if(str){
54 m_str = str;
55 }
56 else{
57 m_str = "boost::interprocess_exception::library_error";
58 }
59 }
60 BOOST_CATCH(...){} BOOST_CATCH_END
61 }
62
63 ~interprocess_exception() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE {}
64
65 const char * what() const BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE
66 { return m_str.c_str(); }
67
68 native_error_t get_native_error() const BOOST_NOEXCEPT { return m_err.get_native_error(); }
69
70 // Note: a value of other_error implies a library (rather than system) error
71 error_code_t get_error_code() const BOOST_NOEXCEPT { return m_err.get_error_code(); }
72
73 #if !defined(BOOST_INTERPROCESS_DOXYGEN_INVOKED)
74 private:
75 error_info m_err;
76 std::string m_str;
77 #endif //#ifndef BOOST_INTERPROCESS_DOXYGEN_INVOKED
78};

80//!This is the exception thrown by shared interprocess_mutex family when a deadlock situation
81//!is detected or when using a interprocess_condition the interprocess_mutex is not locked
82class BOOST_SYMBOL_VISIBLE lock_exception : public interprocess_exception
83{
84 public:
85 lock_exception(error_code_t err = lock_error) BOOST_NOEXCEPT
86 : interprocess_exception(err)
87 {}
88
89 const char* what() const BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE
90 { return "boost::interprocess::lock_exception"; }
91};

93
94//!This exception is thrown when a memory request can't be
95//!fulfilled.
96class BOOST_SYMBOL_VISIBLE bad_alloc : public interprocess_exception
97{
98 public:
99 bad_alloc() : interprocess_exception("::boost::interprocess::bad_alloc") {}
100
101 const char* what() const BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE
102 { return "boost::interprocess::bad_alloc"; }
103};
