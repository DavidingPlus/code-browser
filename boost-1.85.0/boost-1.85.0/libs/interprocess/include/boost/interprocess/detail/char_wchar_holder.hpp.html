25#include <cwchar>
26#include <cstring>

28namespace boost {
29namespace interprocess {
30
31class char_wchar_holder
32{
33 public:
34 char_wchar_holder()
35 : m_str(), m_is_wide()
36 {
37 m_str.n = 0;
38 }
39
40 char_wchar_holder(const char *nstr)
41 : m_str(), m_is_wide()
42 {
43 m_str.n = new char [std::strlen(s: nstr)+1];
44 std::strcpy(dest: m_str.n, src: nstr);
45 }
46
47 char_wchar_holder(const wchar_t *wstr)
48 : m_str(), m_is_wide(true)
49 {
50 m_str.w = new wchar_t [std::wcslen(s: wstr)+1];
51 std::wcscpy(dest: m_str.w, src: wstr);
52 }
53
54 char_wchar_holder& operator=(const char *nstr)
55 {
56 char *tmp = new char [std::strlen(s: nstr)+1];
57 this->delete_mem();
58 m_str.n = tmp;
59 std::strcpy(dest: m_str.n, src: nstr);
60 return *this;
61 }
62
63 char_wchar_holder& operator=(const wchar_t *wstr)
64 {
65 wchar_t *tmp = new wchar_t [std::wcslen(s: wstr)+1];
66 this->delete_mem();
67 m_str.w = tmp;
68 std::wcscpy(dest: m_str.w, src: wstr);
69 return *this;
70 }
71
72 char_wchar_holder& operator=(const char_wchar_holder &other)
73 {
74 if (other.m_is_wide)
75 *this = other.getn();
76 else
77 *this = other.getw();
78 return *this;
79 }
80
81 ~char_wchar_holder()
82 {
83 this->delete_mem();
84 }
85
86 wchar_t *getw() const
87 { return m_is_wide ? m_str.w : 0; }
88
89 char *getn() const
90 { return !m_is_wide ? m_str.n : 0; }
91
92 void swap(char_wchar_holder& other)
93 {
94 char_wchar tmp;
95 std::memcpy(dest: &tmp, src: &m_str, n: sizeof(char_wchar));
96 std::memcpy(dest: &m_str, src: &other.m_str, n: sizeof(char_wchar));
97 std::memcpy(dest: &other.m_str, src: &tmp, n: sizeof(char_wchar));
98 //
99 bool b_tmp(m_is_wide);
100 m_is_wide = other.m_is_wide;
101 other.m_is_wide = b_tmp;
102 }
103
104 private:
105
106 void delete_mem()
107 {
108 if(m_is_wide)
109 delete [] m_str.w;
110 else
111 delete [] m_str.n;
112 }
113
114 union char_wchar
115 {
116 char *n;
117 wchar_t *w;
118 } m_str;
119 bool m_is_wide;
120};
