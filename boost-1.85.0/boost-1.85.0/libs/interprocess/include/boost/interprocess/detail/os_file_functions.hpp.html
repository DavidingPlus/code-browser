125inline bool get_temporary_path(char *buffer, std::size_t buf_len, std::size_t &required_len)
126{
127 required_len = 0;
128 //std::size_t is always bigger or equal than unsigned long in Windows systems
129 //In case std::size_t is bigger than unsigned long
130 unsigned long buf = buf_len;
131 if(buf_len != buf){ //maybe overflowed
132 return false;
133 }
134 required_len = winapi::get_temp_path(buf_len, buffer);
135 const bool ret = required_len && (buf_len > required_len);
136 if(ret && buffer[required_len-1] == '\\'){
137 buffer[required_len-1] = '\0';
138 }
139 return ret;
140}

194inline bool truncate_file (file_handle_t hnd, std::size_t size)
195{
196 offset_t filesize;
197 if(!winapi::get_file_size(hnd, filesize))
198 return false;
199
200 typedef ::boost::move_detail::make_unsigned<offset_t>::type uoffset_t;
201 const uoffset_t max_filesize = uoffset_t(-1)/2u;
202 const uoffset_t uoff_size = uoffset_t(size);
203 //Avoid unused variable warnings in 32 bit systems
204 if(uoff_size > max_filesize){
205 winapi::set_last_error(winapi::error_file_too_large);
206 return false;
207 }
208
209 if(offset_t(size) > filesize){
210 if(!winapi::set_file_pointer(hnd, filesize, 0, winapi::file_begin)){
211 return false;
212 }
213 //We will write zeros in the end of the file
214 //since set_end_of_file does not guarantee this
215 for(std::size_t remaining = size - std::size_t(filesize), write_size = 0
216 ;remaining > 0
217 ;remaining -= write_size){
218 const std::size_t DataSize = 512;
219 static char data [DataSize];
220 write_size = DataSize < remaining ? DataSize : remaining;
221 unsigned long written;
222 winapi::write_file(hnd, data, (unsigned long)write_size, &written, 0);
223 if(written != write_size){
224 return false;
225 }
226 }
227 }
228 else{
229 if(!winapi::set_file_pointer(hnd, size, 0, winapi::file_begin)){
230 return false;
231 }
232 if(!winapi::set_end_of_file(hnd)){
233 return false;
234 }
235 }
236 return true;
237}

271{
272 const unsigned long len = ((unsigned long)-1);
273 winapi::interprocess_overlapped overlapped;
274 std::memset(&overlapped, 0, sizeof(overlapped));
275 if(!winapi::lock_file_ex

277 0, len, len, &overlapped)){
278 return winapi::get_last_error() == winapi::error_lock_violation ?
279 acquired = false, true : false;

281 }
282 acquired = true;
283 return true;

359template<class CharT>
360inline bool delete_subdirectories_recursive
361 (const std::basic_string<CharT> &refcstrRootDirectory, const CharT *dont_delete_this, unsigned int count)
362{
363 bool bSubdirectory = false; // Flag, indicating whether
364 // subdirectories have been found
365 void * hFile; // Handle to directory
366 std::basic_string<CharT> strFilePath; // Filepath
367 std::basic_string<CharT> strPattern; // Pattern
368 typedef os_file_traits<CharT> traits_t;
369 typename traits_t::win32_find_data_t FileInformation; // File information
370
371 //Find all files and directories
372 strPattern = refcstrRootDirectory + traits_t::any_file();
373 hFile = winapi::find_first_file(strPattern.c_str(), &FileInformation);
374 if(hFile != winapi::invalid_handle_value){
375 do{
376 //If it's not "." or ".." or the pointed root_level dont_delete_this erase it
377 if(FileInformation.cFileName[0] != traits_t::dot() &&
378 !(dont_delete_this && count == 0 && traits_t::cmp(dont_delete_this, FileInformation.cFileName) == 0)){
379 strFilePath.erase();
380 strFilePath = refcstrRootDirectory + traits_t::backslash() + FileInformation.cFileName;
381
382 //If it's a directory, go recursive
383 if(FileInformation.dwFileAttributes & winapi::file_attribute_directory){
384 // Delete subdirectory
385 if(!delete_subdirectories_recursive(strFilePath, dont_delete_this, count+1)){
386 winapi::find_close(hFile);
387 return false;
388 }
389 }
390 //If it's a file, just delete it
391 else{
392 // Set file attributes
393 //if(::SetFileAttributes(strFilePath.c_str(), winapi::file_attribute_normal) == 0)
394 //return winapi::get_last_error();
395 // Delete file
396 winapi::unlink_file(strFilePath.c_str());
397 }
398 }
399 //Go to the next file
400 } while(winapi::find_next_file(hFile, &FileInformation) == 1);
401
402 // Close handle
403 winapi::find_close(hFile);
404
405 //See if the loop has ended with an error or just because we've traversed all the files
406 if(winapi::get_last_error() != winapi::error_no_more_files){
407 return false;
408 }
409 else
410 {
411 //Erase empty subdirectories or original refcstrRootDirectory
412 if(!bSubdirectory && count)
413 {
414 // Set directory attributes
415 //if(::SetFileAttributes(refcstrRootDirectory.c_str(), FILE_ATTRIBUTE_NORMAL) == 0)
416 //return ::GetLastError();
417 // Delete directory
418 if(winapi::remove_directory(refcstrRootDirectory.c_str()) == 0)
419 return false;
420 }
421 }
422 }
423 return true;
424}

426//This function erases all the subdirectories of a directory except the one pointed by "dont_delete_this"
427template <class CharT>
428inline bool delete_subdirectories(const std::basic_string<CharT> &refcstrRootDirectory, const CharT *dont_delete_this)
429{
430 return delete_subdirectories_recursive(refcstrRootDirectory, dont_delete_this, 0u);
431}

434template<class Function>
435inline bool for_each_file_in_dir(const char *dir, Function f)
436{
437 void * hFile; // Handle to directory
438 winapi::win32_find_data_a FileInformation; // File information
439
440 //Get base directory
441 std::string str(dir);
442 const std::size_t base_root_dir_len = str.size();
443
444 //Find all files and directories
445 str += "\\*.*";
446 hFile = winapi::find_first_file(str.c_str(), &FileInformation);
447 if(hFile != winapi::invalid_handle_value){
448 do{ //Now loop every file
449 str.erase(base_root_dir_len);
450 //If it's not "." or ".." skip it
451 if(FileInformation.cFileName[0] != '.'){
452 str += "\\"; str += FileInformation.cFileName;
453 //If it's a file, apply erase logic
454 if(!(FileInformation.dwFileAttributes & winapi::file_attribute_directory)){
455 f(str.c_str(), FileInformation.cFileName);
456 }
457 }
458 //Go to the next file
459 } while(winapi::find_next_file(hFile, &FileInformation) == 1);
460
461 // Close handle and see if the loop has ended with an error
462 winapi::find_close(hFile);
463 if(winapi::get_last_error() != winapi::error_no_more_files){
464 return false;
465 }
466 }
467 return true;
468}

558inline file_handle_t create_or_open_file
559 (const char *name, mode_t mode, const permissions & perm = permissions(), bool temporary = false)
560{
561 (void)temporary;
562 int ret = -1;
563 //We need a loop to change permissions correctly using fchmod, since
564 //with "O_CREAT only" ::open we don't know if we've created or opened the file.
565 while(true){
566 ret = ::open(file: name, oflag: ((int)mode) | O_EXCL | O_CREAT, perm.get_permissions());
567 if(ret >= 0){
568 ::fchmod(fd: ret, mode: perm.get_permissions());
569 break;
570 }
571 else if(errno == EEXIST){
572 if((ret = ::open(file: name, oflag: (int)mode)) >= 0 || errno != ENOENT){
573 break;
574 }
575 }
576 else{
577 break;
578 }
579 }
580 return ret;
581}
