73 template <typename Sequence>
74 struct as_variant
75 {
76 // build a variant generator being able to generate a variant holding
77 // all of the types as given in the typelist
78 typedef typename
79 detail::as_variant_impl<fusion::result_of::size<Sequence>::value>
80 gen;
81
82 // use this generator to create the actual variant
83 typedef typename gen::template apply<
84 typename fusion::result_of::begin<Sequence>::type
85 >::type
86 type;
87 };
88}}}
