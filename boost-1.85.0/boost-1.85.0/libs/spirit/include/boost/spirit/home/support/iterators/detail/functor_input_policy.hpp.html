25 ///////////////////////////////////////////////////////////////////////////
26 // class functor_input
27 // Implementation of the InputPolicy used by multi_pass
28 // functor_input gets tokens from a functor
29 //
30 // Note: the functor must have a typedef for result_type
31 // It also must have a static variable of type result_type defined
32 // to represent EOF that is called eof.
33 //
34 ///////////////////////////////////////////////////////////////////////////
35 struct functor_input
36 {
37 ///////////////////////////////////////////////////////////////////////
38 template <typename Functor>
39 class unique : public detail::default_input_policy
40 {
41 private:
42 typedef typename Functor::result_type result_type;
43
44 protected:
45 unique() {}
46 explicit unique(Functor const& x) : ftor(x) {}
47
48 void swap(unique& x)
49 {
50 boost::core::invoke_swap(ftor, x.ftor);
51 }
52
53 public:
54 typedef result_type value_type;
55 typedef std::ptrdiff_t difference_type;
56 typedef std::ptrdiff_t distance_type;
57 typedef result_type* pointer;
58 typedef result_type& reference;
59
60 public:
61 // get the next token
62 template <typename MultiPass>
63 static typename MultiPass::reference get_input(MultiPass& mp)
64 {
65 value_type& curtok = mp.shared()->curtok;
66 if (!input_is_valid(mp, curtok))
67 curtok = mp.ftor();
68 return curtok;
69 }
70
71 template <typename MultiPass>
72 static void advance_input(MultiPass& mp)
73 {
74 // if mp.shared is NULL then this instance of the multi_pass
75 // represents a end iterator
76 BOOST_ASSERT(0 != mp.shared());
77 mp.shared()->curtok = mp.ftor();
78 }
79
80 // test, whether we reached the end of the underlying stream
81 template <typename MultiPass>
82 static bool input_at_eof(MultiPass const& mp)
83 {
84 return mp.shared()->curtok == mp.ftor.eof;
85 }
86
87 template <typename MultiPass>
88 static bool input_is_valid(MultiPass const&, value_type const& t)
89 {
90 using namespace is_valid_test_;
91 return token_is_valid(t);
92 }
93
94 Functor& get_functor() const
95 {
96 return ftor;
97 }
98
99 protected:
100 mutable Functor ftor;
101 };
102
103 ///////////////////////////////////////////////////////////////////////
104 template <typename Functor>
105 struct shared
106 {
107 explicit shared(Functor const&) : curtok(0) {}
108
109 typename Functor::result_type curtok;
110 };
111 };
