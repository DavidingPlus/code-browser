39///////////////////////////////////////////////////////////////////////////////
40//
41// tuple
42//
43// Tuples hold heterogeneous types up to a predefined maximum. Only
44// the most basic functionality needed is provided. Unlike other
45// recursive list-like tuple implementations, this tuple
46// implementation uses simple structs similar to std::pair with
47// specialization for 0 to N tuple elements.
48//
49// 1) Construction
50// Here are examples on how to construct tuples:
51//
52// typedef tuple<int, char> t1_t;
53// typedef tuple<int, std::string, double> t2_t;
54//
55// // this tuple has an int and char members
56// t1_t t1(3, 'c');
57//
58// // this tuple has an int, std::string and double members
59// t2_t t2(3, "hello", 3.14);
60//
61// Tuples can also be constructed from other tuples. The
62// source and destination tuples need not have exactly the
63// same element types. The only requirement is that the
64// source tuple have the same number of elements as the
65// destination and that each element slot in the
66// destination can be copy constructed from the source
67// element. For example:
68//
69// tuple<double, double> t3(t1); // OK. Compatible tuples
70// tuple<double, double> t4(t2); // Error! Incompatible tuples
71//
72// 2) Member access
73// A member in a tuple can be accessed using the
74// tuple's [] operator by specifying the Nth
75// tuple_index. Here are some examples:
76//
77// tuple_index<0> ix0; // 0th index == 1st item
78// tuple_index<1> ix1; // 1st index == 2nd item
79// tuple_index<2> ix2; // 2nd index == 3rd item
80//
81// t1[ix0] = 33; // sets the int member of the tuple t1
82// t2[ix2] = 6e6; // sets the double member of the tuple t2
83// t1[ix1] = 'a'; // sets the char member of the tuple t1
84//
85// There are some predefined names are provided in sub-
86// namespace tuple_index_names:
87//
88// tuple_index<0> _1;
89// tuple_index<1> _2;
90// ...
91// tuple_index<N> _N;
92//
93// These indexes may be used by 'using' namespace
94// phoenix::tuple_index_names.
95//
96// Access to out of bound indexes returns a nil_t value.
97//
98// 3) Member type inquiry
99// The type of an individual member can be queried.
100// Example:
101//
102// tuple_element<1, t2_t>::type
103//
104// Refers to the type of the second member (note zero based,
105// thus 0 = 1st item, 1 = 2nd item) of the tuple.
106//
107// Aside from tuple_element<N, T>::type, there are two
108// more types that tuple_element provides: rtype and
109// crtype. While 'type' is the plain underlying type,
110// 'rtype' is the reference type, or type& and 'crtype'
111// is the constant reference type or type const&. The
112// latter two are provided to make it easy for the
113// client in dealing with the possibility of reference
114// to reference when type is already a reference, which
115// is illegal in C++.
116//
117// Access to out of bound indexes returns a nil_t type.
118//
119// 4) Tuple length
120// The number of elements in a tuple can be queried.
121// Example:
122//
123// int n = t1.length;
124//
125// gets the number of elements in tuple t1.
126//
127// length is a static constant. Thus, TupleT::length
128// also works. Example:
129//
130// int n = t1_t::length;
131//
132///////////////////////////////////////////////////////////////////////////////
133struct nil_t {};
134using boost::remove_reference;
135using boost::call_traits;

155///////////////////////////////////////////////////////////////////////////////
156//
157// tuple_element
158//
159// A query class that gets the Nth element inside a tuple.
160// Examples:
161//
162// tuple_element<1, tuple<int, char, void*> >::type // plain
163// tuple_element<1, tuple<int, char, void*> >::rtype // ref
164// tuple_element<1, tuple<int, char, void*> >::crtype // const ref
165//
166// Has type char which is the 2nd type in the tuple
167// (note zero based, thus 0 = 1st item, 1 = 2nd item).
168//
169// Given a tuple object, the static function tuple_element<N,
170// TupleT>::get(tuple) gets the Nth element in the tuple. The
171// tuple class' tuple::operator[] uses this to get its Nth
172// element.
173//
174///////////////////////////////////////////////////////////////////////////////
175template <int N, typename TupleT>
176struct tuple_element
177{
178 typedef nil_t type;
179 typedef nil_t& rtype;
180 typedef nil_t const& crtype;
181
182 static nil_t get(TupleT const&) { return nil_t(); }
183};

464template <typename DerivedT>
465struct tuple_base {
466
467 typedef nil_t a_type;
468 typedef nil_t b_type;
469 typedef nil_t c_type;
470
471#if PHOENIX_LIMIT > 3
472 typedef nil_t d_type;
473 typedef nil_t e_type;
474 typedef nil_t f_type;
475
476#if PHOENIX_LIMIT > 6
477 typedef nil_t g_type;
478 typedef nil_t h_type;
479 typedef nil_t i_type;
480
481#if PHOENIX_LIMIT > 9
482 typedef nil_t j_type;
483 typedef nil_t k_type;
484 typedef nil_t l_type;
485
486#if PHOENIX_LIMIT > 12
487 typedef nil_t m_type;
488 typedef nil_t n_type;
489 typedef nil_t o_type;
490
491#endif
492#endif
493#endif
494#endif
495
496 template <int N>
497 typename tuple_element<N, DerivedT>::crtype
498 operator[](tuple_index<N>) const
499 {
500 return tuple_element<N, DerivedT>
501 ::get(*static_cast<DerivedT const*>(this));
502 }
503
504 template <int N>
505 typename tuple_element<N, DerivedT>::rtype
506 operator[](tuple_index<N>)
507 {
508 return tuple_element<N, DerivedT>
509 ::get(*static_cast<DerivedT*>(this));
510 }
511};
