80 ///////////////////////////////////////////////////////////////////////////
81 // Parse for literal strings
82 ///////////////////////////////////////////////////////////////////////////
83 template <typename String, bool no_attribute>
84 struct literal_string
85 : primitive_parser<literal_string<String, no_attribute> >
86 {
87 typedef typename
88 remove_const<typename traits::char_type_of<String>::type>::type
89 char_type;
90 typedef std::basic_string<char_type> string_type;
91
92 literal_string(typename add_reference<String>::type str_)
93 : str(str_)
94 {}
95
96 template <typename Context, typename Iterator>
97 struct attribute
98 {
99 typedef typename mpl::if_c<
100 no_attribute, unused_type, string_type>::type
101 type;
102 };
103
104 template <typename Iterator, typename Context
105 , typename Skipper, typename Attribute>
106 bool parse(Iterator& first, Iterator const& last
107 , Context& /*context*/, Skipper const& skipper, Attribute& attr_) const
108 {
109 qi::skip_over(first, last, skipper);
110 return detail::string_parse(str, first, last, attr_);
111 }
112
113 template <typename Context>
114 info what(Context& /*context*/) const
115 {
116 return info("literal-string", str);
117 }
118
119 String str;
120 };
121
122 template <typename String, bool no_attribute>
123 struct no_case_literal_string
124 : primitive_parser<no_case_literal_string<String, no_attribute> >
125 {
126 typedef typename
127 remove_const<typename traits::char_type_of<String>::type>::type
128 char_type;
129 typedef std::basic_string<char_type> string_type;
130
131 template <typename CharEncoding>
132 no_case_literal_string(char_type const* in, CharEncoding encoding)
133 : str_lo(in)
134 , str_hi(in)
135 {
136#if BOOST_WORKAROUND(BOOST_MSVC, BOOST_TESTED_AT(1600))
137 (void)encoding; // suppresses warning: C4100: 'encoding' : unreferenced formal parameter
138#endif
139 typename string_type::iterator loi = str_lo.begin();
140 typename string_type::iterator hii = str_hi.begin();
141
142 for (; loi != str_lo.end(); ++loi, ++hii, ++in)
143 {
144 typedef typename CharEncoding::char_type encoded_char_type;
145
146 *loi = static_cast<char_type>(encoding.tolower(encoded_char_type(*loi)));
147 *hii = static_cast<char_type>(encoding.toupper(encoded_char_type(*hii)));
148 }
149 }
150
151 template <typename Context, typename Iterator>
152 struct attribute
153 {
154 typedef typename mpl::if_c<
155 no_attribute, unused_type, string_type>::type
156 type;
157 };
158
159 template <typename Iterator, typename Context
160 , typename Skipper, typename Attribute>
161 bool parse(Iterator& first, Iterator const& last
162 , Context& /*context*/, Skipper const& skipper, Attribute& attr_) const
163 {
164 qi::skip_over(first, last, skipper);
165 return detail::string_parse(str_lo, str_hi, first, last, attr_);
166 }
167
168 template <typename Context>
169 info what(Context& /*context*/) const
170 {
171 return info("no-case-literal-string", str_lo);
172 }
173
174 string_type str_lo, str_hi;
175 };

177 ///////////////////////////////////////////////////////////////////////////
178 // Parser generators: make_xxx function (objects)
179 ///////////////////////////////////////////////////////////////////////////
180 template <typename T, typename Modifiers>
181 struct make_primitive<T, Modifiers
182 , typename enable_if<traits::is_string<T> >::type>
183 {
184 typedef has_modifier<Modifiers, tag::char_code_base<tag::no_case> > no_case;
185
186 typedef typename add_const<T>::type const_string;
187 typedef typename mpl::if_<
188 no_case
189 , no_case_literal_string<const_string, true>
190 , literal_string<const_string, true> >::type
191 result_type;
192
193 result_type operator()(
194 typename add_reference<const_string>::type str, unused_type) const
195 {
196 return op(str, no_case());
197 }
198
199 template <typename String>
200 result_type op(String const& str, mpl::false_) const
201 {
202 return result_type(str);
203 }
204
205 template <typename String>
206 result_type op(String const& str, mpl::true_) const
207 {
208 typename spirit::detail::get_encoding<Modifiers,
209 spirit::char_encoding::standard>::type encoding;
210 return result_type(traits::get_c_string(str), encoding);
211 }
212 };

216 struct make_primitive<

218 , Modifiers

220 {
221 typedef has_modifier<Modifiers, tag::char_code_base<tag::no_case> > no_case;
222
223 typedef typename add_const<A0>::type const_string;
224 typedef typename mpl::if_<
225 no_case
226 , no_case_literal_string<const_string, true>
227 , literal_string<const_string, true> >::type
228 result_type;
229
230 template <typename Terminal>
231 result_type operator()(Terminal const& term, unused_type) const
232 {
233 return op(fusion::at_c<0>(term.args), no_case());
234 }
235
236 template <typename String>
237 result_type op(String const& str, mpl::false_) const
238 {
239 return result_type(str);
240 }
241
242 template <typename String>
243 result_type op(String const& str, mpl::true_) const
244 {
245 typedef typename traits::char_encoding_from_char<
246 typename traits::char_type_of<A0>::type>::type encoding_type;
247 typename spirit::detail::get_encoding<Modifiers,
248 encoding_type>::type encoding;
249 return result_type(traits::get_c_string(str), encoding);
250 }
251 };
