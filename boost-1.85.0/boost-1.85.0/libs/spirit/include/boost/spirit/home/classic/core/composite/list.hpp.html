25 // operator% is defined as:
26 // a % b ---> a >> *(b >> a)
27 //
28 ///////////////////////////////////////////////////////////////////////////
29 template <typename A, typename B>

31 operator%(parser<A> const& a, parser<B> const& b);

36
37 template <typename B>

39 operator%(char a, parser<B> const& b);
40
41 template <typename A>
