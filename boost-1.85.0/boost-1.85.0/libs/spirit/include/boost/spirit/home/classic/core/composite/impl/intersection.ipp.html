23 template <typename A, typename B>
24 inline intersection<A, B>
25 operator&(parser<A> const& a, parser<B> const& b)
26 {
27 return intersection<A, B>(a.derived(), b.derived());
28 }
29
30 template <typename A>
31 inline intersection<A, chlit<char> >
32 operator&(parser<A> const& a, char b)
33 {
34 return intersection<A, chlit<char> >(a.derived(), b);
35 }
36
37 template <typename B>
38 inline intersection<chlit<char>, B>
39 operator&(char a, parser<B> const& b)
40 {
41 return intersection<chlit<char>, B>(a, b.derived());
42 }
