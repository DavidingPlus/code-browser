19 ///////////////////////////////////////////////////////////////////////////
20 // class illegal_backtracking
21 // thrown by buf_id_check CheckingPolicy if an instance of an iterator is
22 // used after another one has invalidated the queue
23 ///////////////////////////////////////////////////////////////////////////
24 class BOOST_SYMBOL_VISIBLE illegal_backtracking : public std::exception
25 {
26 public:
27 illegal_backtracking() BOOST_NOEXCEPT_OR_NOTHROW {}
28 ~illegal_backtracking() BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE {}
29
30 char const* what() const BOOST_NOEXCEPT_OR_NOTHROW BOOST_OVERRIDE
31 {
32 return "boost::spirit::multi_pass::illegal_backtracking";
33 }
34 };

36 ///////////////////////////////////////////////////////////////////////////////
37 // class buf_id_check
38 // Implementation of the CheckingPolicy used by multi_pass
39 // This policy is most effective when used together with the std_deque
40 // StoragePolicy.
41 //
42 // If used with the fixed_size_queue StoragePolicy, it will not detect
43 // iterator dereferences that are out of the range of the queue.
44 ///////////////////////////////////////////////////////////////////////////////
45 struct buf_id_check
46 {
47 ///////////////////////////////////////////////////////////////////////
48 struct unique //: detail::default_checking_policy
49 {
50 unique() : buf_id(0) {}
51 unique(unique const& x) : buf_id(x.buf_id) {}
52
53 void swap(unique& x)
54 {
55 boost::core::invoke_swap(left&: buf_id, right&: x.buf_id);
56 }
57
58 // called to verify that everything is ok.
59 template <typename MultiPass>
60 static void docheck(MultiPass const& mp)
61 {
62 if (mp.buf_id != mp.shared()->shared_buf_id)
63 boost::throw_exception(e: illegal_backtracking());
64 }
65
66 // called from multi_pass::clear_queue, so we can increment the count
67 template <typename MultiPass>
68 static void clear_queue(MultiPass& mp)
69 {
70 ++mp.shared()->shared_buf_id;
71 ++mp.buf_id;
72 }
73
74 template <typename MultiPass>
75 static void destroy(MultiPass&) {}
76
77 protected:
78 unsigned long buf_id;
79 };
80
81 ///////////////////////////////////////////////////////////////////////
82 struct shared
83 {
84 shared() : shared_buf_id(0) {}
85 unsigned long shared_buf_id;
86 };
87 };
