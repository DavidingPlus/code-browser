24 template <typename Context>
25 struct what_function
26 {
27 what_function(info& what_, Context& context_)
28 : what(what_), context(context_)
29 {
30 what.value = std::list<info>();
31 }
32
33 template <typename Component>
34 void operator()(Component const& component) const
35 {
36#if BOOST_WORKAROUND(BOOST_MSVC, BOOST_TESTED_AT(1600))
37 (void)component; // suppresses warning: C4100: 'component' : unreferenced formal parameter
38#endif
39 boost::get<std::list<info> >(operand&: what.value).
40 push_back(component.what(context));
41 }
42
43 info& what;
44 Context& context;
45 };
