18 ///////////////////////////////////////////////////////////////////////////
19 // class split_std_deque
20 //
21 // Implementation of the StoragePolicy used by multi_pass
22 // This stores all data in a std::vector (despite its name), and keeps an
23 // offset to the current position. It stores all the data unless there is
24 // only one iterator using the queue.
25 //
26 ///////////////////////////////////////////////////////////////////////////
27 struct split_std_deque
28 {
29 enum { threshold = 16 };
30
31 ///////////////////////////////////////////////////////////////////////
32 template <typename Value>
33 class unique //: public detail::default_storage_policy
34 {
35 private:
36 typedef std::vector<Value> queue_type;
37
38 protected:
39 unique() : queued_position(0) {}
40
41 unique(unique const& x)
42 : queued_position(x.queued_position) {}
43
44 void swap(unique& x)
45 {
46 boost::core::invoke_swap(queued_position, x.queued_position);
47 }
48
49 // This is called when the iterator is dereferenced. It's a
50 // template method so we can recover the type of the multi_pass
51 // iterator and call advance_input and input_is_valid.
52 template <typename MultiPass>
53 static typename MultiPass::reference
54 dereference(MultiPass const& mp)
55 {
56 queue_type& queue = mp.shared()->queued_elements;
57 typename queue_type::size_type size = queue.size();
58
59 BOOST_ASSERT(mp.queued_position <= size);
60
61 if (mp.queued_position == size)
62 {
63 // check if this is the only iterator
64 if (size >= threshold && MultiPass::is_unique(mp))
65 {
66 // free up the memory used by the queue.
67 queue.clear();
68 mp.queued_position = 0;
69 }
70 return MultiPass::get_input(mp);
71 }
72
73 return queue[mp.queued_position];
74 }
75
76 // This is called when the iterator is incremented. It's a template
77 // method so we can recover the type of the multi_pass iterator
78 // and call is_unique and advance_input.
79 template <typename MultiPass>
80 static void increment(MultiPass& mp)
81 {
82 queue_type& queue = mp.shared()->queued_elements;
83 typename queue_type::size_type size = queue.size();
84
85 BOOST_ASSERT(mp.queued_position <= size);
86
87// // do not increment iterator as long as the current token is
88// // invalid
89// if (size > 0 && !MultiPass::input_is_valid(mp, queue[mp.queued_position-1]))
90// return;
91
92 if (mp.queued_position == size)
93 {
94 // check if this is the only iterator
95 if (size >= threshold && MultiPass::is_unique(mp))
96 {
97 // free up the memory used by the queue. we avoid
98 // clearing the queue on every increment, though,
99 // because this would be too time consuming
100 queue.clear();
101 mp.queued_position = 0;
102 }
103 else
104 {
105 queue.push_back(MultiPass::get_input(mp));
106 ++mp.queued_position;
107 }
108 MultiPass::advance_input(mp);
109 }
110 else
111 {
112 ++mp.queued_position;
113 }
114 }
115
116 // called to forcibly clear the queue
117 template <typename MultiPass>
118 static void clear_queue(MultiPass& mp)
119 {
120 mp.shared()->queued_elements.clear();
121 mp.queued_position = 0;
122 }
123
124 // called to determine whether the iterator is an eof iterator
125 template <typename MultiPass>
126 static bool is_eof(MultiPass const& mp)
127 {
128 return mp.queued_position == mp.shared()->queued_elements.size()
129 && MultiPass::input_at_eof(mp);
130 }
131
132 // called by operator==
133 template <typename MultiPass>
134 static bool equal_to(MultiPass const& mp, MultiPass const& x)
135 {
136 return mp.queued_position == x.queued_position;
137 }
138
139 // called by operator<
140 template <typename MultiPass>
141 static bool less_than(MultiPass const& mp, MultiPass const& x)
142 {
143 return mp.queued_position < x.queued_position;
144 }
145
146 template <typename MultiPass>
147 static void destroy(MultiPass&) {}
148
149 protected:
150 mutable typename queue_type::size_type queued_position;
151 };
152
153 ///////////////////////////////////////////////////////////////////////
154 template <typename Value>
155 struct shared
156 {
157 shared()
158 {
159 queued_elements.reserve(threshold);
160 }
161
162 typedef std::vector<Value> queue_type;
163 queue_type queued_elements;
164 };
165
166 }; // split_std_deque
