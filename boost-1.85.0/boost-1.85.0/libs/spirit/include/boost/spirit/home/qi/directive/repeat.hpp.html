85 template <typename T>
86 struct exact_iterator // handles repeat(exact)[p]
87 {
88 exact_iterator(T const exact_)
89 : exact(exact_) {}
90
91 typedef T type;
92 T start() const { return 0; }
93 bool got_max(T i) const { return i >= exact; }
94 bool got_min(T i) const { return i >= exact; }
95
96 T const exact;
97 };
98
99 template <typename T>
100 struct finite_iterator // handles repeat(min, max)[p]
101 {
102 finite_iterator(T const min_, T const max_)
103 : min BOOST_PREVENT_MACRO_SUBSTITUTION (min_)
104 , max BOOST_PREVENT_MACRO_SUBSTITUTION (max_) {}
105
106 typedef T type;
107 T start() const { return 0; }
108 bool got_max(T i) const { return i >= max; }
109 bool got_min(T i) const { return i >= min; }
110
111 T const min;
112 T const max;
113 };
114
115 template <typename T>
116 struct infinite_iterator // handles repeat(min, inf)[p]
117 {
118 infinite_iterator(T const min_)
119 : min BOOST_PREVENT_MACRO_SUBSTITUTION (min_) {}
120
121 typedef T type;
122 T start() const { return 0; }
123 bool got_max(T /*i*/) const { return false; }
124 bool got_min(T i) const { return i >= min; }
125
126 T const min;
127 };

132 template <typename Subject, typename LoopIter>
133 struct repeat_parser : unary_parser<repeat_parser<Subject, LoopIter> >
134 {
135 typedef Subject subject_type;
136
137 template <typename Context, typename Iterator>
138 struct attribute
139 {
140 // Build a std::vector from the subject's attribute. Note
141 // that build_std_vector may return unused_type if the
142 // subject's attribute is an unused_type.
143 typedef typename
144 traits::build_std_vector<
145 typename traits::attribute_of<
146 Subject, Context, Iterator>::type
147 >::type
148 type;
149 };
150
151 repeat_parser(Subject const& subject_, LoopIter const& iter_)
152 : subject(subject_), iter(iter_) {}
153
154 template <typename F>
155 bool parse_container(F f) const
156 {
157 typename LoopIter::type i = iter.start();
158 for (/**/; !iter.got_min(i); ++i)
159 {
160 if (f (subject))
161 return false;
162 }
163
164 // parse some more up to the maximum specified
165 typename F::iterator_type save = f.f.first;
166 for (/**/; !iter.got_max(i); ++i)
167 {
168 if (f (subject))
169 break;
170 save = f.f.first;
171 }
172
173 f.f.first = save;
174 return true;
175 }
176
177 template <typename Iterator, typename Context
178 , typename Skipper, typename Attribute>
179 bool parse(Iterator& first, Iterator const& last
180 , Context& context, Skipper const& skipper
181 , Attribute& attr_) const
182 {
183 typedef detail::fail_function<Iterator, Context, Skipper>
184 fail_function;
185
186 // ensure the attribute is actually a container type
187 traits::make_container(attr_);
188
189 Iterator iter_local = first;
190 fail_function f(iter_local, last, context, skipper);
191 if (!parse_container(detail::make_pass_container(f, attr_)))
192 return false;
193
194 first = f.first;
195 return true;
196 }
197
198 template <typename Context>
199 info what(Context& context) const
200 {
201 return info("repeat", subject.what(context));
202 }
203
204 Subject subject;
205 LoopIter iter;
206 };

212 struct make_directive<tag::repeat, Subject, Modifiers>
213 {

218 }
219 };

221 template <typename T, typename Subject, typename Modifiers>
222 struct make_directive<
223 terminal_ex<tag::repeat, fusion::vector1<T> >, Subject, Modifiers>
224 {
225 typedef exact_iterator<T> iterator_type;
226 typedef repeat_parser<Subject, iterator_type> result_type;
227
228 template <typename Terminal>
229 result_type operator()(
230 Terminal const& term, Subject const& subject, unused_type) const
231 {
232 return result_type(subject, fusion::at_c<0>(term.args));
233 }
234 };
235
236 template <typename T, typename Subject, typename Modifiers>
237 struct make_directive<
238 terminal_ex<tag::repeat, fusion::vector2<T, T> >, Subject, Modifiers>
239 {
240 typedef finite_iterator<T> iterator_type;
241 typedef repeat_parser<Subject, iterator_type> result_type;
242
243 template <typename Terminal>
244 result_type operator()(
245 Terminal const& term, Subject const& subject, unused_type) const
246 {
247 return result_type(subject,
248 iterator_type(
249 fusion::at_c<0>(term.args)
250 , fusion::at_c<1>(term.args)
251 )
252 );
253 }
254 };
