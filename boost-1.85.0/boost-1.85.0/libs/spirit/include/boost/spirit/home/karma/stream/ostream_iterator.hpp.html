17namespace boost { namespace spirit { namespace karma

19 ///////////////////////////////////////////////////////////////////////////
20 // We need our own implementation of an ostream_iterator just to be able
21 // to access the wrapped ostream, which is necessary for the
22 // stream_generator, where we must generate the output using the original
23 // ostream to retain possibly registered facets.
24 ///////////////////////////////////////////////////////////////////////////
25 template <
26 typename T, typename Elem = char
27 , typename Traits = std::char_traits<Elem> >
28 class ostream_iterator
29 {
30 public:
31 typedef std::output_iterator_tag iterator_category;
32 typedef void value_type;
33 typedef void difference_type;
34 typedef void pointer;
35 typedef void reference;
36 typedef Elem char_type;
37 typedef Traits traits_type;
38 typedef std::basic_ostream<Elem, Traits> ostream_type;
39 typedef ostream_iterator<T, Elem, Traits> self_type;
40
41 ostream_iterator(ostream_type& os_, Elem const* delim_ = 0)
42 : os(&os_), delim(delim_) {}
43
44 self_type& operator= (T const& val)
45 {
46 *os << val;
47 if (0 != delim)
48 *os << delim;
49 return *this;
50 }
51
52 self_type& operator*() { return *this; }
53 self_type& operator++() { return *this; }
54 self_type operator++(int) { return *this; }
55
56 // expose underlying stream
57 ostream_type& get_ostream() { return *os; }
58 ostream_type const& get_ostream() const { return *os; }
59
60 // expose good bit of underlying stream object
61 bool good() const { return get_ostream().good(); }
62
63 protected:
64 ostream_type *os;
65 Elem const* delim;
66 };
67
