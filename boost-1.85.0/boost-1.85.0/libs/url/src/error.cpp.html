17namespace boost {
18namespace urls {

20namespace detail {
21
22const char*
23error_cat_type::
24name() const noexcept
25{
26 return "boost.url";
27}
28
29std::string
30error_cat_type::
31message(int code) const
32{
33 return message(code, nullptr, 0);
34}
35
36char const*
37error_cat_type::
38message(
39 int code,
40 char*,
41 std::size_t) const noexcept
42{
43 switch(static_cast<error>(code))
44 {
45case error::success: return "success";
46case error::illegal_null: return "illegal null";
47case error::illegal_reserved_char: return "illegal reserved char";
48case error::non_canonical: return "non canonical";
49
50case error::bad_pct_hexdig: return "bad hexdig in pct-encoding";
51case error::incomplete_encoding: return "incomplete pct-encoding";
52case error::missing_pct_hexdig: return "missing hexdig in pct-encoding";
53case error::no_space: return "no space";
54case error::not_a_base: return "not a base";
55 }
56 return "";
57}
58
59system::error_condition
60error_cat_type::
61default_error_condition(
62 int ev) const noexcept
63{
64 switch(static_cast<error>(ev))
65 {
66 default:
67 return {ev, *this};
68
69case error::bad_pct_hexdig:
70case error::incomplete_encoding:
71case error::missing_pct_hexdig:
72 return grammar::condition::fatal;
73 }
74}
