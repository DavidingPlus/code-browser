23ipv4_address::

30ipv4_address::
31ipv4_address(
32 bytes_type const& bytes) noexcept
33{
34 addr_ =
35(static_cast<unsigned long>(bytes[0]) << 24) |
36(static_cast<unsigned long>(bytes[1]) << 16) |
37(static_cast<unsigned long>(bytes[2]) << 8) |
38(static_cast<unsigned long>(bytes[3]));
39}

50auto
51ipv4_address::
52to_bytes() const noexcept ->
53 bytes_type
54{
55 bytes_type bytes;
56 bytes[0] = (addr_ >> 24) & 0xff;
57 bytes[1] = (addr_ >> 16) & 0xff;
58 bytes[2] = (addr_ >> 8) & 0xff;
59 bytes[3] = addr_ & 0xff;
60 return bytes;
61}

63auto
64ipv4_address::
65to_uint() const noexcept ->
66 uint_type
67{
68 return addr_;
69}

71core::string_view
72ipv4_address::
73to_buffer(
74 char* dest,
75 std::size_t dest_size) const
76{
77 if(dest_size < max_str_len)

79 auto n = print_impl(dest);

83bool
84ipv4_address::
85is_loopback() const noexcept
86{
87 return (to_uint() & 0xFF000000) ==
88 0x7F000000;
89}
90
91bool
92ipv4_address::
93is_unspecified() const noexcept
94{
95 return to_uint() == 0;
96}
97
98bool
99ipv4_address::
100is_multicast() const noexcept
101{
102 return (to_uint() & 0xF0000000) ==
103 0xE0000000;
104}

106std::size_t
107ipv4_address::
108print_impl(
109 char* dest) const noexcept
110{
111 auto const start = dest;
112 auto const write =
113 []( char*& dest,
114 unsigned char v)
115 {
116 if(v >= 100)
117 {
118 *dest++ = '0' +
119 v / 100;
120 v %= 100;
121 *dest++ = '0' +
122 v / 10;
123 v %= 10;
124 }
125 else if(v >= 10)
126 {
127 *dest++ = '0' +
128 v / 10;
129 v %= 10;
130 }
131 *dest++ = '0' + v;
132 };
133 auto const v = to_uint();
134 write(dest, (v >> 24) & 0xff);
135 *dest++ = '.';
136 write(dest, (v >> 16) & 0xff);
137 *dest++ = '.';
138 write(dest, (v >> 8) & 0xff);
139 *dest++ = '.';
140 write(dest, (v ) & 0xff);
141 return dest - start;
142}

144void
145ipv4_address::
146to_string_impl(
147 string_token::arg& t) const
148{
149 char buf[max_str_len];
150 auto const n = print_impl(dest: buf);
151 char* dest = t.prepare(n);
152 std::memcpy(dest: dest, src: buf, n: n);
153}
