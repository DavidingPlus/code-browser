23static_url_base::
24static_url_base(
25 char* buf,
26 std::size_t cap) noexcept
27{
28 s_ = buf;
29 cap_ = cap;
30 s_[0] = '\0';
31 impl_.cs_ = s_;
32}

45void
46static_url_base::
47clear_impl() noexcept
48{
49 impl_ = {from::url};
50 s_[0] = '\0';
51 impl_.cs_ = s_;
52}

54void
55static_url_base::
56reserve_impl(
57 std::size_t n,
58 op_t&)
59{
60 if(n <= cap_)
61 return;
62 detail::throw_length_error();
63}
