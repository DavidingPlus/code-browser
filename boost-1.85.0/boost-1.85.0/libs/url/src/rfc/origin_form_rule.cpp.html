13#include <boost/url/detail/config.hpp>

15#include <boost/url/rfc/query_rule.hpp>

17#include "detail/query_part_rule.hpp"
18#include <boost/url/grammar/delim_rule.hpp>
19#include <boost/url/grammar/range_rule.hpp>

22namespace boost {
23namespace urls {
24
25auto
26origin_form_rule_t::
27parse(
28 char const*& it,
29 char const* end
30 ) const noexcept ->
31 system::result<value_type>
32{
33 detail::url_impl u(detail::url_impl::from::string);
34 u.cs_ = it;
35
36 {
37 auto rv = grammar::parse(it, end,
38 r: grammar::range_rule(
39 next: grammar::tuple_rule(
40 r0: grammar::delim_rule(ch: '/'),
41 rn: detail::segment_rule),
42 N: 1));
43 if(! rv)
44 return rv.error();
45 u.apply_path(
46 rv->string(),
47 rv->size());
48 }
49
50 // [ "?" query ]
51 {
52 auto rv = grammar::parse(
53 it, end, r: detail::query_part_rule);
54 if(! rv)
55 return rv.error();
56 if(rv->has_query)
57 {
58 // map "?" to { {} }
59 u.apply_query(
60 rv->query,
61 rv->count +
62 rv->query.empty());
63 }
64 }
65
66 return u.construct();
67}
68
69} // urls
70} // boost
