25auto
26port_rule::
27parse(
28 char const*& it,
29 char const* end) const noexcept ->
30 system::result<value_type>
31{
32 value_type t;
33 auto const start = it;
34 while(
35 it != end &&
36 *it == '0')
37 {
38 ++it;
39 }
40
41 if (it != end)
42 {
43 grammar::unsigned_rule<std::uint16_t> r;
44 auto it0 = it;
45 auto rv = r.parse(it, end);
46 if (rv)
47 {
48 // number < max uint16_t
49 t.str = core::string_view(start, it);
50 t.has_number = true;
51 t.number = *rv;
52 return t;
53 }
54 it = it0;
55 if (grammar::digit_chars(*it))
56 {
57 // number > max uint16_t
58 while (
59 it != end &&
60 grammar::digit_chars(*it))
61 {
62 ++it;
63 }
64 t.str = core::string_view(start, it);
65 t.has_number = true;
66 t.number = 0;
67 return t;
68 }
69 }
70 // no digits
71 t.str = core::string_view(start, it);
72 t.has_number = it != end;
73 t.number = 0;
74 return t;
75}

77auto
78port_part_rule_t::
79parse(
80 char const*& it,
81 char const* end) const noexcept ->
82 system::result<value_type>
83{
84 value_type t;
85 if( it == end ||
86 *it != ':')
87 {
88 t.has_port = false;
89 return t;
90 }
91 ++it;
92 auto rv = grammar::parse(
93 it, end, r: port_rule{});
94 if(! rv)
95 return rv.error();
96 t.has_port = true;
97 t.port = rv->str;
98 t.has_number = rv->has_number;
99 t.port_number = rv->number;
100 return t;
101}
