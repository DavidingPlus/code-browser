19namespace boost {
20namespace urls {
21namespace detail {
22
23auto
24hier_part_rule_t::
25parse(
26 char const*& it,
27 char const* const end
28 ) const noexcept ->
29 system::result<value_type>
30{
31 value_type t;
32 if(it == end)
33 {
34 // path-empty
35 return t;
36 }
37 if(end - it == 1)
38 {
39 if(*it == '/')
40 {
41 // path-absolute
42 t.path = make_pct_string_view_unsafe(
43 data: it, size: 1, decoded_size: 1);
44 t.segment_count = 1;
45 ++it;
46 return t;
47 }
48 // path-rootless
49 auto rv = grammar::parse(
50 it, end, r: segment_rule);
51 if(! rv)
52 return rv.error();
53 t.path = *rv;
54 t.segment_count = !t.path.empty();
55 return t;
56 }
57 if( it[0] == '/' &&
58 it[1] == '/')
59 {
60 // "//" authority
61 it += 2;
62 auto rv = grammar::parse(
63 it, end, r: authority_rule);
64 if(! rv)
65 return rv.error();
66 t.authority = *rv;
67 t.has_authority = true;
68 }
69 // the authority requires an absolute path
70 // or an empty path
71 if(it == end || (
72 t.has_authority && (
73 *it != '/' &&
74 *it != '?' &&
75 *it != '#')))
76 {
77 // path-empty
78 return t;
79 }
80 auto const it0 = it;
81 std::size_t dn = 0;
82 if(*it != '/')
83 {
84 auto rv = grammar::parse(
85 it, end, r: segment_rule);
86 if(! rv)
87 return rv.error();
88 if(rv->empty())
89 return t;
90 dn += rv->decoded_size();
91 ++t.segment_count;
92 }
93 while(it != end)
94 {
95 if(*it == '/')
96 {
97 ++dn;
98 ++it;
99 ++t.segment_count;
100 continue;
101 }
102 auto rv = grammar::parse(
103 it, end, r: segment_rule);
104 if(! rv)
105 return rv.error();
106 if(rv->empty())
107 break;
108 dn += rv->decoded_size();
109 }
110 t.path = make_pct_string_view_unsafe(
111 data: it0, size: it - it0, decoded_size: dn);
112 return t;
113}
