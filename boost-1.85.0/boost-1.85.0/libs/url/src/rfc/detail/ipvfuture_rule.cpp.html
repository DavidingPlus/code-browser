28auto
29ipvfuture_rule_t::
30parse(
31 char const*& it,
32 char const* const end
33 ) const noexcept ->
34 system::result<value_type>
35{
36 static constexpr auto
37 minor_chars =
38 unreserved_chars +
39 sub_delim_chars + ':';
40 auto const it0 = it;
41 auto rv = grammar::parse(
42 it, end,
43 r: grammar::tuple_rule(
44 r0: grammar::delim_rule(ch: 'v'),
45 rn: grammar::token_rule(
46 cs: grammar::hexdig_chars),
47 rn: grammar::delim_rule(ch: '.'),
48 rn: grammar::token_rule(cs: minor_chars)));
49 if(! rv)
50 return rv.error();
51 value_type t;
52 t.major = std::get<0>(t&: *rv);
53 t.minor = std::get<1>(t&: *rv);
54 if(t.major.empty())
55 {
56 // can't be empty
57 BOOST_URL_RETURN_EC(
58 grammar::error::invalid);
59 }
60 if(t.minor.empty())
61 {
62 // can't be empty
63 BOOST_URL_RETURN_EC(
64 grammar::error::invalid);
65 }
66 t.str = core::string_view(
67 it0, it - it0);
68 return t;
69}
