21/** Rule for fragment-part
22
23 @par BNF
24 @code
25 fragment-part = [ "#" fragment ]
26
27 fragment = *( pchar / "/" / "?" )
28 @endcode
29
30 @par Specification
31 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.5"
32 >3.5. Fragment (rfc3986)</a>
33*/
34struct fragment_part_rule_t
35{
36 struct value_type
37 {
38 pct_string_view fragment;
39 bool has_fragment = false;
40 };
41
42 system::result<value_type>
43 parse(
44 char const*& it,
45 char const* end
46 ) const noexcept
47 {
48 if( it == end ||
49 *it != '#')
50 return {};
51 ++it;
52 auto rv = grammar::parse(
53 it, end, r: pct_encoded_rule(
54 cs: fragment_chars));
55 if(! rv)
56 return rv.error();
57 value_type t;
58 t.fragment = *rv;
59 t.has_fragment = true;
60 return t;
61 }
62};
63constexpr fragment_part_rule_t fragment_part_rule{};
64
65} // detail
66} // urls
67} // boost
