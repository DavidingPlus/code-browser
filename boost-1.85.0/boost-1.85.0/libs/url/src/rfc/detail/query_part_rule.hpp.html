24/** Rule for query-part
25
26 @par BNF
27 @code
28 query-part = [ "?" query ]
29 @endcode
30*/
31struct query_part_rule_t
32{
33 struct value_type
34 {
35 pct_string_view query;
36 std::size_t count = 0;
37 bool has_query = false;
38 };
39
40 auto
41 parse(
42 char const*& it,
43 char const* end
44 ) const noexcept ->
45 system::result<value_type>
46 {
47 if( it == end ||
48 *it != '?')
49 return {};
50 ++it;
51 auto rv = grammar::parse(
52 it, end, r: query_rule);
53 if(! rv)
54 return rv.error();
55 value_type t;
56 t.query = rv->buffer();
57 t.count = rv->size();
58 t.has_query = true;
59 return t;
60 }
61};
62
63constexpr query_part_rule_t query_part_rule{};
64
65} // detail
