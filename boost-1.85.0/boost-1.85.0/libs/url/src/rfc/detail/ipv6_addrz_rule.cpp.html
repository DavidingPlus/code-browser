20namespace boost {
21namespace urls {
22namespace detail {
23
24auto
25ipv6_addrz_rule_t::
26parse(
27 char const*& it,
28 char const* const end
29 ) const noexcept ->
30 system::result<value_type>
31{
32 value_type t;
33 auto rv1 = grammar::parse(
34 it, end, r: ipv6_address_rule);
35 if (! rv1)
36 return rv1.error();
37 t.ipv6 = *rv1;
38
39 // "%25"
40 auto it0 = it;
41 if (end - it < 3 ||
42 *it != '%' ||
43 *(it + 1) != '2' ||
44 *(it + 2) != '5')
45 {
46 BOOST_URL_RETURN_EC(
47 grammar::error::invalid);
48 }
49 it += 3;
50
51 // ZoneID = 1*( unreserved / pct-encoded )
52 // Parse as many (unreserved / pct-encoded)
53 // as available
54 auto rv2 = grammar::parse(
55 it, end,
56 r: pct_encoded_rule(cs: unreserved_chars));
57 if(!rv2 || rv2->empty())
58 {
59 it = it0;
60 BOOST_URL_RETURN_EC(
61 grammar::error::invalid);
62 }
63 else
64 {
65 t.zone_id = *rv2;
66 }
67 return t;
68}
