21/** Rule for h16
22
23 This parses a sixteen bit unsigned
24 hexadecimal number.
25
26 @par BNF
27 @code
28 h16 = 1*4HEXDIG
29 ; 16 bits of address represented in hexadecimal
30 @endcode
31
32 @par Specification
33 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.2.2"
34 >3.2.2. Host (rfc3986)</a>
35*/
36struct h16_rule_t
37{
38 struct value_type
39 {
40 std::uint8_t hi;
41 std::uint8_t lo;
42 };
43
44 auto
45 parse(
46 char const*& it,
47 char const* end
48 ) const noexcept ->
49 system::result<value_type>;
50};
51
52constexpr h16_rule_t h16_rule{};
53
54} // detail
55} // urls
56} // boost

58#endif
