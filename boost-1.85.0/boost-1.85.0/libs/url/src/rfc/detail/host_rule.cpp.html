10#ifndef BOOST_URL_RFC_DETAIL_IMPL_HOST_RULE_IPP
11#define BOOST_URL_RFC_DETAIL_IMPL_HOST_RULE_IPP
12
13#include <boost/url/detail/config.hpp>
14#include <boost/url/rfc/ipv4_address_rule.hpp>
15#include "host_rule.hpp"
16#include "ip_literal_rule.hpp"
17#include "reg_name_rule.hpp"
18#include <boost/url/grammar/parse.hpp>
19
20namespace boost {
21namespace urls {
22namespace detail {
23
24auto
25host_rule_t::
26parse(
27 char const*& it,
28 char const* const end
29 ) const noexcept ->
30 system::result<value_type>
31{
32 value_type t;
33
34 if(it == end)
35 {
36 // empty host
37 t.host_type =
38 urls::host_type::name;
39 return t;
40 }
41
42 auto const it0 = it;
43 if(*it == '[')
44 {
45 // IP-literal
46 auto rv = grammar::parse(
47 it, end,
48 r: detail::ip_literal_rule);
49 if(! rv)
50 return rv.error();
51 auto v = *rv;
52 if(v.is_ipv6)
53 {
54 // IPv6address
55 auto const b =
56 v.ipv6.to_bytes();
57 std::memcpy(
58 dest: t.addr,
59 src: b.data(),
60 n: b.size());
61 t.host_type =
62 urls::host_type::ipv6;
63 t.match = core::string_view(
64 it0, it - it0);
65 return t;
66 }
67
68 // IPvFuture
69 t.host_type =
70 urls::host_type::ipvfuture;
71 t.match = core::string_view(
72 it0, it - it0);
73 return t;
74 }
75
76 // IPv4address
77 {
78 auto rv = grammar::parse(
79 it, end, r: ipv4_address_rule);
80 if( rv )
81 {
82 auto it02 = it;
83 auto rv2 = grammar::parse(
84 it, end,
85 r: detail::reg_name_rule);
86 if (rv2.has_value() &&
87 !rv2->empty())
88 {
89 t.name = core::string_view(
90 it0, it - it0);
91 t.host_type =
92 urls::host_type::name;
93 t.match = core::string_view(
94 it0, it - it0);
95 return t;
96 }
97 it = it02;
98 auto const b =
99 rv->to_bytes();
100 std::memcpy(
101 dest: t.addr,
102 src: b.data(),
103 n: b.size());
104 t.host_type =
105 urls::host_type::ipv4;
106 t.match = core::string_view(
107 it0, it - it0);
108 return t;
109 }
110
111 it = it0; // rewind
112 }
113
114 // reg-name
115 {
116 auto rv = grammar::parse(
117 it, end,
118 r: detail::reg_name_rule);
119 if(! rv)
120 return rv.error();
121 t.name = *rv;
122 t.host_type =
123 urls::host_type::name;
124 t.match = core::string_view(
125 it0, it - it0);
126 return t;
127 }
128}
129
130} // detail
131} // urls
132} // boost
