24auto
25h16_rule_t::
26parse(
27 char const*& it,
28 char const* end
29 ) const noexcept ->
30 system::result<value_type>
31{
32 // VFALCO it might be impossible for
33 // this condition to be true (coverage)
34 if(it == end)
35 {
36 // end
37 BOOST_URL_RETURN_EC(
38 grammar::error::invalid);
39 }
40
41 std::uint16_t v;
42 for(;;)
43 {
44 auto d = grammar::hexdig_value(ch: *it);
45 if(d < 0)
46 {
47 // expected HEXDIG
48 BOOST_URL_RETURN_EC(
49 grammar::error::invalid);
50 }
51 v = d;
52 ++it;
53 if(it == end)
54 break;
55 d = grammar::hexdig_value(ch: *it);
56 if(d < 0)
57 break;
58 v = (16 * v) + d;
59 ++it;
60 if(it == end)
61 break;
62 d = grammar::hexdig_value(ch: *it);
63 if(d < 0)
64 break;
65 v = (16 * v) + d;
66 ++it;
67 if(it == end)
68 break;
69 d = grammar::hexdig_value(ch: *it);
70 if(d < 0)
71 break;
72 v = (16 * v) + d;
73 ++it;
74 break;
75 }
76 return value_type{
77 .hi: static_cast<
78 unsigned char>(v / 256),
79 .lo: static_cast<
80 unsigned char>(v % 256)};
81}
