10#ifndef BOOST_URL_RFC_DETAIL_REG_NAME_RULE_HPP
11#define BOOST_URL_RFC_DETAIL_REG_NAME_RULE_HPP
12
13#include "boost/url/detail/config.hpp"
14#include "boost/url/rfc/pct_encoded_rule.hpp"
15#include "boost/url/rfc/sub_delim_chars.hpp"
16#include "boost/url/rfc/unreserved_chars.hpp"
17
18namespace boost {
19namespace urls {
20namespace detail {
21
22/* VFALCO In theory we could enforce these
23 additional requirements from errata 4942:
24
25 Such a name consists of a sequence of domain
26 labels separated by ".", each domain label
27 starting and ending with an alphanumeric character
28 and possibly also containing "-" characters. The
29 rightmost domain label of a fully qualified domain
30 name in DNS may be followed by a single "." and
31 should be if it is necessary to distinguish between
32 the complete domain name and some local domain.
33*/
34
35/** Rule for reg-name
36
37 @par BNF
38 @code
39 reg-name = *( unreserved / pct-encoded / "-" / ".")
40 @endcode
41
42 @par Specification
43 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.2.2"
44 >3.2.2. Host (rfc3986)</a>
45 @li <a href="https://www.rfc-editor.org/errata/eid4942"
46 >Errata ID: 4942</a>
47*/
48constexpr auto reg_name_rule =
49 pct_encoded_rule(cs: unreserved_chars + sub_delim_chars);
50
51} // detail
52} // urls
53} // boost
54
55#endif
