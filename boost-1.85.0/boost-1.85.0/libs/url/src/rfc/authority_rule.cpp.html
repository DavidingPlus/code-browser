23namespace boost {
24namespace urls {
25
26auto
27authority_rule_t::
28parse(
29 char const*& it,
30 char const* const end
31 ) const noexcept ->
32 system::result<value_type>
33{
34 detail::url_impl u(detail::url_impl::from::authority);
35 u.cs_ = it;
36
37 // [ userinfo "@" ]
38 {
39 auto rv = grammar::parse(
40 it, end,
41 r: grammar::optional_rule(
42 r: grammar::tuple_rule(
43 r0: detail::userinfo_rule,
44 rn: grammar::squelch(
45 r: grammar::delim_rule(ch: '@')))));
46 if(! rv)
47 return rv.error();
48 if(rv->has_value())
49 {
50 u.apply_userinfo(
51 (*rv)->user,
52 (*rv)->has_password
53 ? &(*rv)->password
54 : nullptr);
55 }
56 }
57
58 // host
59 {
60 auto rv = grammar::parse(
61 it, end, r: detail::host_rule);
62 if(! rv)
63 return rv.error();
64 u.apply_host(rv->host_type,
65 rv->match, rv->addr);
66 }
67
68 // [ ":" port ]
69 {
70 auto rv = grammar::parse(
71 it, end, r: detail::port_part_rule);
72 if(! rv)
73 return rv.error();
74 if(rv->has_port)
75 u.apply_port(
76 rv->port,
77 rv->port_number);
78 }
79
80 return u.construct_authority();
81}

83} // urls
