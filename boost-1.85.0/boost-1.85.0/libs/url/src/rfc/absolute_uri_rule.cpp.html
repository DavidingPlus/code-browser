24namespace boost {
25namespace urls {
26
27auto
28absolute_uri_rule_t::
29parse(
30 char const*& it,
31 char const* const end
32 ) const noexcept ->
33 system::result<value_type>
34{
35 detail::url_impl u(detail::url_impl::from::string);
36 u.cs_ = it;
37
38 // scheme
39 {
40 auto rv = grammar::parse(
41 it, end,
42 r: grammar::tuple_rule(
43 r0: detail::scheme_rule(),
44 rn: grammar::squelch(
45 r: grammar::delim_rule(ch: ':'))));
46 if(! rv)
47 return rv.error();
48 u.apply_scheme(rv->scheme);
49 }
50
51 // hier_part
52 {
53 auto rv = grammar::parse(
54 it, end, r: detail::hier_part_rule);
55 if(! rv)
56 return rv.error();
57 if(rv->has_authority)
58 u.apply_authority(rv->authority);
59 u.apply_path(
60 rv->path,
61 rv->segment_count);
62 }
63
64 // [ "?" query ]
65 {
66 auto rv = grammar::parse(
67 it, end, r: detail::query_part_rule);
68 if(! rv)
69 return rv.error();
70 if(rv->has_query)
71 {
72 // map "?" to { {} }
73 u.apply_query(
74 rv->query,
75 rv->count +
76 rv->query.empty());
77 }
78 }
79
80 return u.construct();
81}
