20namespace boost {
21namespace urls {
22
23auto
24ipv4_address_rule_t::
25parse(
26 char const*& it,
27 char const* end
28 ) const noexcept ->
29 system::result<value_type>
30{
31 using namespace grammar;
32 auto rv = grammar::parse(
33 it, end, r: tuple_rule(
34 r0: dec_octet_rule, rn: squelch(r: delim_rule(ch: '.')),
35 rn: dec_octet_rule, rn: squelch(r: delim_rule(ch: '.')),
36 rn: dec_octet_rule, rn: squelch(r: delim_rule(ch: '.')),
37 rn: dec_octet_rule));
38 if(! rv)
39 return rv.error();
40 std::array<unsigned char, 4> v;
41 v[0] = std::get<0>(t&: *rv);
42 v[1] = std::get<1>(t&: *rv);
43 v[2] = std::get<2>(t&: *rv);
44 v[3] = std::get<3>(t&: *rv);
45 return ipv4_address(v);
46}
