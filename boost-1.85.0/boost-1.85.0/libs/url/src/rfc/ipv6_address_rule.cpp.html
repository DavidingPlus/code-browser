28// return `true` if the hex
29// word could be 0..255 if
30// interpreted as decimal
31static
32bool
33maybe_octet(
34 unsigned char const* p) noexcept
35{
36 unsigned short word =
37 static_cast<unsigned short>(
38 p[0]) * 256 +
39 static_cast<unsigned short>(
40 p[1]);
41 if(word > 0x255)
42 return false;
43 if(((word >> 4) & 0xf) > 9)
44 return false;
45 if((word & 0xf) > 9)
46 return false;
47 return true;
48}
49
50} // detail
51
52auto
53ipv6_address_rule_t::
54parse(
55 char const*& it,
56 char const* const end
57 ) const noexcept ->
58 system::result<ipv6_address>
59{
60 int n = 8; // words needed
61 int b = -1; // value of n
62 // when '::' seen
63 bool c = false; // need colon
64 auto prev = it;
65 ipv6_address::bytes_type bytes;
66 system::result<detail::h16_rule_t::value_type> rv;
67 for(;;)
68 {
69 if(it == end)
70 {
71 if(b != -1)
72 {
73 // end in "::"
74 break;
75 }
76 BOOST_ASSERT(n > 0);
77 // not enough words
78 BOOST_URL_RETURN_EC(
79 grammar::error::invalid);
80 }
81 if(*it == ':')
82 {
83 ++it;
84 if(it == end)
85 {
86 // expected ':'
87 BOOST_URL_RETURN_EC(
88 grammar::error::invalid);
89 }
90 if(*it == ':')
91 {
92 if(b == -1)
93 {
94 // first "::"
95 ++it;
96 --n;
97 b = n;
98 if(n == 0)
99 break;
100 c = false;
101 continue;
102 }
103 // extra "::" found
104 BOOST_URL_RETURN_EC(
105 grammar::error::invalid);
106 }
107 if(c)
108 {
109 prev = it;
110 rv = grammar::parse(
111 it, end,
112 r: detail::h16_rule);
113 if(! rv)
114 return rv.error();
115 bytes[2*(8-n)+0] = rv->hi;
116 bytes[2*(8-n)+1] = rv->lo;
117 --n;
118 if(n == 0)
119 break;
120 continue;
121 }
122 // expected h16
123 BOOST_URL_RETURN_EC(
124 grammar::error::invalid);
125 }
126 if(*it == '.')
127 {
128 if(b == -1 && n > 1)
129 {
130 // not enough h16
131 BOOST_URL_RETURN_EC(
132 grammar::error::invalid);
133 }
134 if(! detail::maybe_octet(
135 p: &bytes[2*(7-n)]))
136 {
137 // invalid octet
138 BOOST_URL_RETURN_EC(
139 grammar::error::invalid);
140 }
141 // rewind the h16 and
142 // parse it as ipv4
143 it = prev;
144 auto rv1 = grammar::parse(
145 it, end, r: ipv4_address_rule);
146 if(! rv1)
147 return rv1.error();
148 auto v4 = *rv1;
149 auto const b4 =
150 v4.to_bytes();
151 bytes[2*(7-n)+0] = b4[0];
152 bytes[2*(7-n)+1] = b4[1];
153 bytes[2*(7-n)+2] = b4[2];
154 bytes[2*(7-n)+3] = b4[3];
155 --n;
156 break;
157 }
158 auto d =
159 grammar::hexdig_value(ch: *it);
160 if( b != -1 &&
161 d < 0)
162 {
163 // ends in "::"
164 break;
165 }
166 if(! c)
167 {
168 prev = it;
169 rv = grammar::parse(
170 it, end,
171 r: detail::h16_rule);
172 if(! rv)
173 return rv.error();
174 bytes[2*(8-n)+0] = rv->hi;
175 bytes[2*(8-n)+1] = rv->lo;
176 --n;
177 if(n == 0)
178 break;
179 c = true;
180 continue;
181 }
182 // ':' divides a word
183 BOOST_URL_RETURN_EC(
184 grammar::error::invalid);
185 }
186 if(b == -1)
187 return ipv6_address{bytes};
188 if(b == n)
189 {
190 // "::" last
191 auto const i =
192 2 * (7 - n);
193 std::memset(
194 s: &bytes[i],
195 c: 0, n: 16 - i);
196 }
197 else if(b == 7)
198 {
199 // "::" first
200 auto const i =
201 2 * (b - n);
202 std::memmove(
203 dest: &bytes[16 - i],
204 src: &bytes[2],
205 n: i);
206 std::memset(
207 s: &bytes[0],
208 c: 0, n: 16 - i);
209 }
210 else
211 {
212 // "::" in middle
213 auto const i0 =
214 2 * (7 - b);
215 auto const i1 =
216 2 * (b - n);
217 std::memmove(
218 dest: &bytes[16 - i1],
219 src: &bytes[i0 + 2],
220 n: i1);
221 std::memset(
222 s: &bytes[i0],
223 c: 0, n: 16 - (i0 + i1));
224 }
225 return ipv6_address{bytes};
226}
