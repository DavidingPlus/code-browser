50operator*() const ->

52
53{
54 encoding_opts opt;
55 opt.space_as_plus =
56 space_as_plus_;
57 param_pct_view p =
58 it_.dereference();

60 p.key.decode(opt),

190std::size_t
191params_base::
192count(
193 core::string_view key,
194 ignore_case_param ic) const noexcept
195{
196 std::size_t n = 0;
197 auto it = find(key, ic);
198 auto const end_ = end();
199 while(it != end_)
200 {
201 ++n;
202 ++it;
203 it = find(it, key, ic);
204 }
205 return n;
206}

214detail::params_iter_impl
215params_base::
216find_impl(
217 detail::params_iter_impl it,
218 core::string_view key,
219 ignore_case_param ic) const noexcept
220{
221 detail::params_iter_impl end_(ref_, 0);
222 if(! ic)
223 {
224 for(;;)
225 {
226 if(it.equal(other: end_))
227 return it;
228 if(*it.key() == key)
229 return it;
230 it.increment();
231 }
232 }
233 for(;;)
234 {
235 if(it.equal(other: end_))
236 return it;
237 if( grammar::ci_is_equal(
238 s0: *it.key(), s1: key))
239 return it;
240 it.increment();
241 }
242}
243
244detail::params_iter_impl
245params_base::
246find_last_impl(
247 detail::params_iter_impl it,
248 core::string_view key,
249 ignore_case_param ic) const noexcept
250{
251 detail::params_iter_impl begin_(ref_);
252 if(! ic)
253 {
254 for(;;)
255 {
256 if(it.equal(other: begin_))
257 return { ref_, 0 };
258 it.decrement();
259 if(*it.key() == key)
260 return it;
261 }
262 }
263 for(;;)
264 {
265 if(it.equal(other: begin_))
266 return { ref_, 0 };
267 it.decrement();
268 if(grammar::ci_is_equal(
269 s0: *it.key(), s1: key))
270 return it;
271 }
272}
