90void
91url::
92clear_impl() noexcept
93{
94 if(s_)
95 {
96 // preserve capacity
97 impl_ = {from::url};
98 s_[0] = '\0';
99 impl_.cs_ = s_;
100 }
101 else
102 {
103 BOOST_ASSERT(impl_.cs_[0] == 0);
104 }
105}

107void
108url::
109reserve_impl(
110 std::size_t n,
111 op_t& op)
112{
113 if(n > max_size())
114 detail::throw_length_error();
115 if(n <= cap_)
116 return;
117 char* s;
118 if(s_ != nullptr)
119 {
120 // 50% growth policy
121 auto const h = cap_ / 2;
122 std::size_t new_cap;
123 if(cap_ <= max_size() - h)
124 new_cap = cap_ + h;
125 else
126 new_cap = max_size();
127 if( new_cap < n)
128 new_cap = n;
129 s = allocate(n: new_cap);
130 std::memcpy(dest: s, src: s_, n: size() + 1);
131 BOOST_ASSERT(! op.old);
132 op.old = s_;
133 s_ = s;
134 }
135 else
136 {
137 s_ = allocate(n);
138 s_[0] = '\0';
139 }
140 impl_.cs_ = s_;
141}
142
143void
144url::
145cleanup(
146 op_t& op)
147{
148 if(op.old)
149 deallocate(s: op.old);
150}

154void
155url::
156swap(url& other) noexcept
157{
158 if (this == &other)
159 return;
160 std::swap(a&: s_, b&: other.s_);
161 std::swap(a&: cap_, b&: other.cap_);
162 std::swap(a&: impl_, b&: other.impl_);
163 std::swap(a&: pi_, b&: other.pi_);
164 if (pi_ == &other.impl_)
165 pi_ = &impl_;
166 if (other.pi_ == &impl_)
167 other.pi_ = &other.impl_;
168}
