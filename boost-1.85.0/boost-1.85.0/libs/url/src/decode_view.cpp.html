23template <class T>
24int
25decoded_strcmp(decode_view s0, T s1)
26{
27 auto const n0 = s0.size();
28 auto const n1 = s1.size();
29 auto n = (std::min)(n0, n1);
30 auto it0 = s0.begin();
31 auto it1 = s1.begin();
32 while (n--)
33 {
34 const char c0 = *it0++;
35 const char c1 = *it1++;
36 if (c0 == c1)
37 continue;
38 return 1 - 2 * (static_cast<unsigned char>(c0)
39 < static_cast<unsigned char>(c1));
40 }
41 return 1 - (n0 == n1) - 2 * (n0 < n1);
42}

48auto
49decode_view::
50iterator::
51operator*() const noexcept ->
52 reference
53{
54 if (space_as_plus_ &&
55 *pos_ == '+')
56 return ' ';
57 if (*pos_ != '%')
58 return *pos_;
59 auto d0 = grammar::hexdig_value(ch: pos_[1]);
60 auto d1 = grammar::hexdig_value(ch: pos_[2]);
61 return static_cast<char>(
62 ((static_cast<
63 unsigned char>(d0) << 4) +
64 (static_cast<
65 unsigned char>(d1))));
66}

106void
107decode_view::
108remove_prefix( size_type n )
109{
110 auto it = begin();
111 auto n0 = n;
112 while (n)
113 {
114 ++it;
115 --n;
116 }
117 n_ -= (it.base() - begin().base());
118 dn_ -= n0;
119 p_ = it.base();
120}
121
122void
123decode_view::
124remove_suffix( size_type n )
125{
126 auto it = end();
127 auto n0 = n;
128 while (n)
129 {
130 --it;
131 --n;
132 }
133 n_ -= (end().base() - it.base());
134 dn_ -= n0;
135}

137bool
138decode_view::
139starts_with( core::string_view s ) const noexcept
140{
141 if (s.size() > size())
142 return false;
143 auto it0 = begin();
144 auto it1 = s.begin();
145 std::size_t n = s.size();
146 while (n)
147 {
148 if (*it0 != *it1)
149 return false;
150 ++it0;
151 ++it1;
152 --n;
153 }
154 return true;
155}
156
157bool
158decode_view::
159ends_with( core::string_view s ) const noexcept
160{
161 if (s.size() > size())
162 return false;
163 auto it0 = end();
164 auto it1 = s.end();
165 std::size_t n = s.size();
166 --it0;
167 --it1;
168 while (n - 1)
169 {
170 if (*it0 != *it1)
171 return false;
172 --it0;
173 --it1;
174 --n;
175 }
176 return *it0 == *it1;
177}

197decode_view::const_iterator
198decode_view::
199find( char ch ) const noexcept
200{
201 auto it = begin();
202 auto end = this->end();
203 while (it != end)
204 {
205 if (*it == ch)
206 return it;
207 ++it;
208 }
209 return it;
210}
211
212decode_view::const_iterator
213decode_view::
214rfind( char ch ) const noexcept
215{
216 if (empty())
217 return end();
218 auto it = end();
219 auto begin = this->begin();
220 --it;
221 while (it != begin)
222 {
223 if (*it == ch)
224 return it;
225 --it;
226 }
227 if (*it == ch)
228 return it;
229 return end();
230}
