93std::size_t
94params_ref::
95erase(
96 core::string_view key,
97 ignore_case_param ic) noexcept
98{
99 // end() can't be fully cached,
100 // since erase invalidates it.
101 iterator it;
102 {
103 auto const end_ = end();
104 it = find_last(before: end_, key, ic);
105 if(it == end_)
106 return 0;
107 }
108 std::size_t n = 0;
109 for(;;)
110 {
111 ++n;
112 // Use it->key instead of key,
113 // to handle self-intersection
114 auto prev = find_last(before: it, key: (*it).key, ic);
115 if(prev == end())
116 break;
117 erase(pos: it);
118 it = prev;
119 }
120 erase(pos: it);
121 return n;
122}

189auto
190params_ref::
191set(
192 core::string_view key,
193 core::string_view value,
194 ignore_case_param ic) ->
195 iterator
196{
197 // VFALCO we can't cache end() here
198 // because it is invalidated
199 // every time we set or erase.
200 auto it0 = find(key, ic);
201 if(it0 == end())
202 return append(p: {key, value});
203 it0 = set(pos: it0, value);
204 auto it = end();
205 for(;;)
206 {
207 it = find_last(before: it, key, ic);
208 if(it == it0)
209 return it0;
210 it = erase(pos: it);
211 }
212}

247detail::params_iter_impl
248params_ref::
249find_impl(
250 detail::params_iter_impl it,
251 core::string_view key,
252 ignore_case_param ic) const noexcept
253{
254 detail::params_iter_impl end_(u_->impl_, 0);
255 if(! ic)
256 {
257 for(;;)
258 {
259 if(it.equal(other: end_))
260 return it;
261 if(*it.key() == key)
262 return it;
263 it.increment();
264 }
265 }
266 for(;;)
267 {
268 if(it.equal(other: end_))
269 return it;
270 if( grammar::ci_is_equal(
271 s0: *it.key(), s1: key))
272 return it;
273 it.increment();
274 }
275}
276
277detail::params_iter_impl
278params_ref::
279find_last_impl(
280 detail::params_iter_impl it,
281 core::string_view key,
282 ignore_case_param ic) const noexcept
283{
284 detail::params_iter_impl begin_(u_->impl_);
285 if(! ic)
286 {
287 for(;;)
288 {
289 if(it.equal(other: begin_))
290 return { u_->impl_, 0 };
291 it.decrement();
292 if(*it.key() == key)
293 return it;
294 }
295 }
296 for(;;)
297 {
298 if(it.equal(other: begin_))
299 return { u_->impl_, 0 };
300 it.decrement();
301 if(grammar::ci_is_equal(
302 s0: *it.key(), s1: key))
303 return it;
304 }
305}
