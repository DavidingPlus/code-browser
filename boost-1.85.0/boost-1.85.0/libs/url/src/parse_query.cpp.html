21namespace boost {
22namespace urls {
23
24system::result<params_encoded_view>
25parse_query(core::string_view s) noexcept
26{
27 // Handle empty strings differently.
28 // We produce {}, versus empty but
29 // present query in URL (e.g. "http:?")
30 // which produces {{"", none}}.
31 if(s.empty())
32 return params_encoded_view(
33 detail::query_ref(
34 s.data(), 0, 0));
35 auto rv = grammar::parse(s, r: query_rule);
36 if(! rv)
37 return rv.error();
38 return params_encoded_view(
39 detail::query_ref(
40 s, s.size(), rv->size()));
41}
