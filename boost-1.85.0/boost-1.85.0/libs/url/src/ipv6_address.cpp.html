78bool
79ipv6_address::
80is_v4_mapped() const noexcept
81{
82 return
83 addr_[ 0] == 0 && addr_[ 1] == 0 &&
84 addr_[ 2] == 0 && addr_[ 3] == 0 &&
85 addr_[ 4] == 0 && addr_[ 5] == 0 &&
86 addr_[ 6] == 0 && addr_[ 7] == 0 &&
87 addr_[ 8] == 0 && addr_[ 9] == 0 &&
88 addr_[10] == 0xff &&
89 addr_[11] == 0xff;
90}

101std::size_t
102ipv6_address::
103print_impl(
104 char* dest) const noexcept
105{
106 auto const count_zeroes =
107 []( unsigned char const* first,
108 unsigned char const* const last)
109 {
110 std::size_t n = 0;
111 while(first != last)
112 {
113 if( first[0] != 0 ||
114 first[1] != 0)
115 break;
116 n += 2;
117 first += 2;
118 }
119 return n;
120 };
121 auto const print_hex =
122 []( char* dest,
123 unsigned short v)
124 {
125 char const* const dig =
126 "0123456789abcdef";
127 if(v >= 0x1000)
128 {
129 *dest++ = dig[v>>12];
130 v &= 0x0fff;
131 *dest++ = dig[v>>8];
132 v &= 0x0ff;
133 *dest++ = dig[v>>4];
134 v &= 0x0f;
135 *dest++ = dig[v];
136 }
137 else if(v >= 0x100)
138 {
139 *dest++ = dig[v>>8];
140 v &= 0x0ff;
141 *dest++ = dig[v>>4];
142 v &= 0x0f;
143 *dest++ = dig[v];
144 }
145 else if(v >= 0x10)
146 {
147 *dest++ = dig[v>>4];
148 v &= 0x0f;
149 *dest++ = dig[v];
150 }
151 else
152 {
153 *dest++ = dig[v];
154 }
155 return dest;
156 };
157 auto const dest0 = dest;
158 // find longest run of zeroes
159 std::size_t best_len = 0;
160 int best_pos = -1;
161 auto it = addr_.data();
162 auto const v4 =
163 is_v4_mapped();
164 auto const end = v4 ?
165 (it + addr_.size() - 4)
166 : it + addr_.size();
167 while(it != end)
168 {
169 auto n = count_zeroes(
170 it, end);
171 if(n == 0)
172 {
173 it += 2;
174 continue;
175 }
176 if(n > best_len)
177 {
178 best_pos = static_cast<
179 int>(it - addr_.data());
180 best_len = n;
181 }
182 it += n;
183 }
184 it = addr_.data();
185 if(best_pos != 0)
186 {
187 unsigned short v =
188 (it[0] * 256U) + it[1];
189 dest = print_hex(dest, v);
190 it += 2;
191 }
192 else
193 {
194 *dest++ = ':';
195 it += best_len;
196 if(it == end)
197 *dest++ = ':';
198 }
199 while(it != end)
200 {
201 *dest++ = ':';
202 if(it - addr_.data() ==
203 best_pos)
204 {
205 it += best_len;
206 if(it == end)
207 *dest++ = ':';
208 continue;
209 }
210 unsigned short v =
211 (it[0] * 256U) + it[1];
212 dest = print_hex(dest, v);
213 it += 2;
214 }
215 if(v4)
216 {
217 ipv4_address::bytes_type bytes;
218 bytes[0] = it[0];
219 bytes[1] = it[1];
220 bytes[2] = it[2];
221 bytes[3] = it[3];
222 ipv4_address a(bytes);
223 *dest++ = ':';
224 dest += a.print_impl(dest);
225 }
226 return dest - dest0;
227}

229void
230ipv6_address::
231to_string_impl(
232 string_token::arg& t) const
233{
234 char buf[max_str_len];
235 auto const n = print_impl(dest: buf);
236 char* dest = t.prepare(n);
237 std::memcpy(dest: dest, src: buf, n: n);
238}
