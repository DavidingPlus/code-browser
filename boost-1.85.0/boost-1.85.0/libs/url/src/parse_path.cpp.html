21namespace boost {
22namespace urls {
23
24system::result<segments_encoded_view>
25parse_path(core::string_view s) noexcept
26{
27 auto it = s.data();
28 auto const end = it + s.size();
29 std::size_t dn = 0;
30 std::size_t nseg = 0;
31 if( it != end &&
32 *it != '/')
33 ++nseg;
34 while(it != end)
35 {
36 if(*it == '/')
37 {
38 ++it;
39 ++dn;
40 ++nseg;
41 continue;
42 }
43 auto rv = grammar::parse(
44 it, end, r: detail::segment_rule);
45 if(! rv)
46 return rv.error();
47 if(rv->empty())
48 {
49 BOOST_URL_RETURN_EC(
50 grammar::error::mismatch);
51 }
52 dn += rv->decoded_size();
53 }
54 // adjust nseg
55 nseg = detail::path_segments(s, nseg);
56 return segments_encoded_view(
57 detail::path_ref(s, dn, nseg));
58}
