20scheme
21string_to_scheme(
22 core::string_view s) noexcept
23{
24 using grammar::to_lower;
25 switch(s.size())
26 {
27 case 0: // none
28 return scheme::none;
29
30 case 2: // ws
31 if( to_lower(c: s[0]) == 'w' &&
32 to_lower(c: s[1]) == 's')
33 return scheme::ws;
34 break;
35
36 case 3:
37 switch(to_lower(c: s[0]))
38 {
39 case 'w': // wss
40 if( to_lower(c: s[1]) == 's' &&
41 to_lower(c: s[2]) == 's')
42 return scheme::wss;
43 break;
44
45 case 'f': // ftp
46 if( to_lower(c: s[1]) == 't' &&
47 to_lower(c: s[2]) == 'p')
48 return scheme::ftp;
49 break;
50
51 default:
52 break;
53 }
54 break;
55
56 case 4:
57 switch(to_lower(c: s[0]))
58 {
59 case 'f': // file
60 if( to_lower(c: s[1]) == 'i' &&
61 to_lower(c: s[2]) == 'l' &&
62 to_lower(c: s[3]) == 'e')
63 return scheme::file;
64 break;
65
66 case 'h': // http
67 if( to_lower(c: s[1]) == 't' &&
68 to_lower(c: s[2]) == 't' &&
69 to_lower(c: s[3]) == 'p')
70 return scheme::http;
71 break;
72
73 default:
74 break;
75 }
76 break;
77
78 case 5: // https
79 if( to_lower(c: s[0]) == 'h' &&
80 to_lower(c: s[1]) == 't' &&
81 to_lower(c: s[2]) == 't' &&
82 to_lower(c: s[3]) == 'p' &&
83 to_lower(c: s[4]) == 's')
84 return scheme::https;
85 break;
86
87 default:
88 break;
89 }
90 return scheme::unknown;
91}
92
93core::string_view
94to_string(scheme s) noexcept
95{
96 switch(s)
97 {
98 case scheme::ftp: return "ftp";
99 case scheme::file: return "file";
100 case scheme::http: return "http";
101 case scheme::https: return "https";
102 case scheme::ws: return "ws";
103 case scheme::wss: return "wss";
104 case scheme::none: return {};
105 default:
106 break;
107 }
108 return "<unknown>";
109}
110
111std::uint16_t
112default_port(scheme s) noexcept
113{
114 switch(s)
115 {
116 case scheme::ftp:
117 return 21;
118 case scheme::http:
119 case scheme::ws:
120 return 80;
121 case scheme::https:
122 case scheme::wss:
123 return 443;
124 default:
125 break;
126 }
127 return 0;
128}
