27bool
28ci_is_equal(
29 core::string_view s0,
30 core::string_view s1) noexcept
31{
32 auto n = s0.size();
33 auto p1 = s0.data();
34 auto p2 = s1.data();
35 char a, b;
36 // fast loop
37 while(n--)
38 {
39 a = *p1++;
40 b = *p2++;
41 if(a != b)
42 goto slow;
43 }
44 return true;
45slow:
46 do
47 {
48 if( to_lower(c: a) !=
49 to_lower(c: b))
50 return false;
51 a = *p1++;
52 b = *p2++;
53 }
54 while(n--);
55 return true;
56}

60bool
61ci_is_less(
62 core::string_view s0,
63 core::string_view s1) noexcept
64{
65 auto p1 = s0.data();
66 auto p2 = s1.data();
67 for(auto n = s0.size();n--;)
68 {
69 auto c1 = to_lower(c: *p1++);
70 auto c2 = to_lower(c: *p2++);
71 if(c1 != c2)
72 return c1 < c2;
73 }
74 // equal
75 return false;
76}

82int
83ci_compare(
84 core::string_view s0,
85 core::string_view s1) noexcept
86{
87 int bias;
88 std::size_t n;
89 if( s0.size() <
90 s1.size())
91 {
92 bias = -1;
93 n = s0.size();
94 }
95 else
96 {
97 if( s0.size() >
98 s1.size())
99 bias = 1;
100 else
101 bias = 0;
102 n = s1.size();
103 }
104 auto it0 = s0.data();
105 auto it1 = s1.data();
106 while(n--)
107 {
108 auto c0 =
109 to_lower(c: *it0++);
110 auto c1 =
111 to_lower(c: *it1++);
112 if(c0 == c1)
113 continue;
114 if(c0 < c1)
115 return -1;
116 return 1;
117 }
118 return bias;
119}

123std::size_t
124ci_digest(
125 core::string_view s) noexcept
126{
127 // Only 4 and 8 byte sizes are supported
128 static_assert(
129 sizeof(std::size_t) == 4 ||
130 sizeof(std::size_t) == 8, "");
131 constexpr std::size_t prime = (
132 sizeof(std::size_t) == 8) ?
133 0x100000001B3ULL :
134 0x01000193UL;
135 constexpr std::size_t hash0 = (
136 sizeof(std::size_t) == 8) ?
137 0xcbf29ce484222325ULL :
138 0x811C9DC5UL;
139 auto hash = hash0;
140 auto p = s.data();
141 auto n = s.size();
142 for(;n--;++p)
143 {
144 // VFALCO NOTE Consider using a lossy
145 // to_lower which works 4 or 8 chars at a time.
146 hash = (to_lower(c: *p) ^ hash) * prime;
147 }
148 return hash;
149}
