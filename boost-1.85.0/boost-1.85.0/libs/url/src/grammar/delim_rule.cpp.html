21auto
22ch_delim_rule::
23parse(
24 char const*& it,
25 char const* end) const noexcept ->
26 system::result<value_type>
27{
28 if(it == end)
29 {
30 // end
31 BOOST_URL_RETURN_EC(
32 error::need_more);
33 }
34 if(*it != ch_)
35 {
36 // wrong character
37 BOOST_URL_RETURN_EC(
38 error::mismatch);
39 }
40 return core::string_view{
41 it++, 1 };
42};
