19namespace boost {
20namespace urls {
21namespace grammar {
22
23auto
24literal_rule::
25parse(
26 char const*& it,
27 char const* end) const noexcept ->
28 system::result<value_type>
29{
30 // Can't have a literal
31 // with an empty string!
32 BOOST_ASSERT(n_ > 0);
33
34 std::size_t n = end - it;
35 if(n >= n_)
36 {
37 if(std::memcmp(
38 s1: it, s2: s_, n: n_) != 0)
39 {
40 // non-match
41 BOOST_URL_RETURN_EC(
42 error::mismatch);
43 }
44 it += n_;
45 return core::string_view(
46 it - n_, it);
47 }
48 if(n > 0)
49 {
50 // short input
51 if(std::memcmp(
52 s1: it, s2: s_, n: n) != 0)
53 {
54 // non-match
55 BOOST_URL_RETURN_EC(
56 error::mismatch);
57 }
58 // prefix matches
59 BOOST_URL_RETURN_EC(
60 error::need_more);
61 }
62 // end
63 BOOST_URL_RETURN_EC(
64 error::need_more);
65}
66
67} // grammar
68} // urls
69} // boost
