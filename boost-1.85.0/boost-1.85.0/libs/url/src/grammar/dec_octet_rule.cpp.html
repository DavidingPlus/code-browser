19namespace boost {
20namespace urls {
21namespace grammar {
22
23auto
24dec_octet_rule_t::
25parse(
26 char const*& it,
27 char const* const end
28 ) const noexcept ->
29 system::result<value_type>
30{
31 if(it == end)
32 {
33 // end
34 BOOST_URL_RETURN_EC(
35 error::mismatch);
36 }
37 if(! digit_chars(*it))
38 {
39 // expected DIGIT
40 BOOST_URL_RETURN_EC(
41 error::mismatch);
42 }
43 unsigned v = *it - '0';
44 ++it;
45 if( it == end ||
46 ! digit_chars(*it))
47 {
48 return static_cast<
49 value_type>(v);
50 }
51 if(v == 0)
52 {
53 // leading '0'
54 BOOST_URL_RETURN_EC(
55 error::invalid);
56 }
57 v = (10 * v) + *it - '0';
58 ++it;
59 if( it == end ||
60 ! digit_chars(*it))
61 {
62 return static_cast<
63 value_type>(v);
64 }
65 if(v > 25)
66 {
67 // integer overflow
68 BOOST_URL_RETURN_EC(
69 error::invalid);
70 }
71 v = (10 * v) + *it - '0';
72 if(v > 255)
73 {
74 // integer overflow
75 BOOST_URL_RETURN_EC(
76 error::invalid);
77 }
78 ++it;
79 if( it != end &&
80 digit_chars(*it))
81 {
82 // integer overflow
83 BOOST_URL_RETURN_EC(
84 error::invalid);
85 }
86 return static_cast<
87 value_type>(v);
88}
