84bool
85authority_view::
86has_userinfo() const noexcept
87{
88 auto n = u_.len(id_pass);
89 if(n == 0)
90 return false;
91 BOOST_ASSERT(u_.get(
92 id_pass).ends_with('@'));
93 return true;
94}
95
96pct_string_view
97authority_view::
98encoded_userinfo() const noexcept
99{
100 auto s = u_.get(
101 id_user, id_host);
102 if(s.empty())
103 return s;
104 BOOST_ASSERT(
105 s.ends_with('@'));
106 s.remove_suffix(n: 1);
107 return make_pct_string_view_unsafe(
108 data: s.data(),
109 size: s.size(),
110 decoded_size: u_.decoded_[id_user] +
111 u_.decoded_[id_pass] +
112 has_password());
113}
114
115pct_string_view
116authority_view::
117encoded_user() const noexcept
118{
119 auto s = u_.get(id_user);
120 return make_pct_string_view_unsafe(
121 data: s.data(),
122 size: s.size(),
123 decoded_size: u_.decoded_[id_user]);
124}
125
126bool
127authority_view::
128has_password() const noexcept
129{
130 auto const n = u_.len(id_pass);
131 if(n > 1)
132 {
133 BOOST_ASSERT(u_.get(id_pass
134 ).starts_with(':'));
135 BOOST_ASSERT(u_.get(id_pass
136 ).ends_with('@'));
137 return true;
138 }
139 BOOST_ASSERT(n == 0 || u_.get(
140 id_pass).ends_with('@'));
141 return false;
142}
143
144pct_string_view
145authority_view::
146encoded_password() const noexcept
147{
148 auto s = u_.get(id_pass);
149 switch(s.size())
150 {
151 case 1:
152 BOOST_ASSERT(
153 s.starts_with('@'));
154 s.remove_prefix(n: 1);
155 BOOST_FALLTHROUGH;
156 case 0:
157 return make_pct_string_view_unsafe(
158 data: s.data(), size: s.size(), decoded_size: 0);
159 default:
160 break;
161 }
162 BOOST_ASSERT(s.ends_with('@'));
163 BOOST_ASSERT(s.starts_with(':'));
164 return make_pct_string_view_unsafe(
165 data: s.data() + 1,
166 size: s.size() - 2,
167 decoded_size: u_.decoded_[id_pass]);
168}

174//------------------------------------------------
175/*
176host_type host_type() // ipv4, ipv6, ipvfuture, name
177
178std::string host() // return encoded_host().decode()
179pct_string_view encoded_host() // return host part, as-is
180std::string host_address() // return encoded_host_address().decode()
181pct_string_view encoded_host_address() // ipv4, ipv6, ipvfut, or encoded name, no brackets
182
183ipv4_address host_ipv4_address() // return ipv4_address or {}
184ipv6_address host_ipv6_address() // return ipv6_address or {}
185core::string_view host_ipvfuture() // return ipvfuture or {}
186std::string host_name() // return decoded name or ""
187pct_string_view encoded_host_name() // return encoded host name or ""
188*/
189
190pct_string_view
191authority_view::
192encoded_host() const noexcept
193{
194 return u_.pct_get(id_host);
195}
196
197pct_string_view
198authority_view::
199encoded_host_address() const noexcept
200{
201 core::string_view s = u_.get(id_host);
202 std::size_t n;
203 switch(u_.host_type_)
204 {
205 default:
206 case urls::host_type::none:
207 BOOST_ASSERT(s.empty());
208 n = 0;
209 break;
210
211 case urls::host_type::name:
212 case urls::host_type::ipv4:
213 n = u_.decoded_[id_host];
214 break;
215
216 case urls::host_type::ipv6:
217 case urls::host_type::ipvfuture:
218 {
219 BOOST_ASSERT(
220 u_.decoded_[id_host] ==
221 s.size());
222 BOOST_ASSERT(s.size() >= 2);
223 BOOST_ASSERT(s.front() == '[');
224 BOOST_ASSERT(s.back() == ']');
225 s = s.substr(pos: 1, n: s.size() - 2);
226 n = u_.decoded_[id_host] - 2;
227 break;
228 }
229 }
230 return make_pct_string_view_unsafe(
231 data: s.data(), size: s.size(), decoded_size: n);
232}

234urls::ipv4_address
235authority_view::
236host_ipv4_address() const noexcept
237{
238 if(u_.host_type_ !=
239 urls::host_type::ipv4)
240 return {};
241 ipv4_address::bytes_type b{._M_elems: {}};
242 std::memcpy(
243 dest: &b[0], src: &u_.ip_addr_[0], n: b.size());
244 return urls::ipv4_address(b);
245}

260core::string_view
261authority_view::
262host_ipvfuture() const noexcept
263{
264 if(u_.host_type_ !=
265 urls::host_type::ipvfuture)
266 return {};
267 core::string_view s = u_.get(id_host);
268 BOOST_ASSERT(s.size() >= 6);
269 BOOST_ASSERT(s.front() == '[');
270 BOOST_ASSERT(s.back() == ']');
271 s = s.substr(pos: 1, n: s.size() - 2);
272 return s;
273}

305port() const noexcept
306{
307 auto s = u_.get(id_port);
308 if(s.empty())
309 return s;
310 BOOST_ASSERT(has_port());
311 return s.substr(pos: 1);
312}

346// Comparisons
347//
348//------------------------------------------------
349
350int
351authority_view::
352compare(const authority_view& other) const noexcept
353{
354 auto comp = static_cast<int>(has_userinfo()) -
355 static_cast<int>(other.has_userinfo());
356 if ( comp != 0 )
357 return comp;
358
359 if (has_userinfo())
360 {
361 comp = detail::compare_encoded(
362 lhs: encoded_user(),
363 rhs: other.encoded_user());
364 if ( comp != 0 )
365 return comp;
366
367 comp = static_cast<int>(has_password()) -
368 static_cast<int>(other.has_password());
369 if ( comp != 0 )
370 return comp;
371
372 if (has_password())
373 {
374 comp = detail::compare_encoded(
375 lhs: encoded_password(),
376 rhs: other.encoded_password());
377 if ( comp != 0 )
378 return comp;
379 }
380 }
381
382 comp = detail::ci_compare_encoded(
383 lhs: encoded_host(),
384 rhs: other.encoded_host());
385 if ( comp != 0 )
386 return comp;
387
388 comp = static_cast<int>(has_port()) -
389 static_cast<int>(other.has_port());
390 if ( comp != 0 )
391 return comp;
392
393 if (has_port())
394 {
395 comp = detail::compare(
396 lhs: port(),
397 rhs: other.port());
398 if ( comp != 0 )
399 return comp;
400 }
401
402 return 0;
403}
