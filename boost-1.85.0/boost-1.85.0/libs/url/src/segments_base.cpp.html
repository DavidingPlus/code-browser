62pct_string_view
63segments_base::
64buffer() const noexcept
65{
66 return ref_.buffer();
67}
68
69bool
70segments_base::
71is_absolute() const noexcept
72{
73 return ref_.buffer().starts_with(x: '/');
74}
75
76bool
77segments_base::
78empty() const noexcept
79{
80 return ref_.nseg() == 0;
81}
82
83std::size_t
84segments_base::
85size() const noexcept
86{
87 return ref_.nseg();
88}
89
90auto
91segments_base::
92begin() const noexcept ->

95 return iterator(ref_);

97
98auto
99segments_base::
100end() const noexcept ->
101 iterator

104}
