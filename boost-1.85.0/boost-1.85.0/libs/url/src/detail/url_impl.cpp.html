35void
36url_impl::
37apply_scheme(
38 core::string_view s) noexcept
39{
40 scheme_ = string_to_scheme(s);
41 set_size(id_scheme, s.size() + 1);
42}

44void
45url_impl::
46apply_userinfo(
47 pct_string_view const& user,
48 pct_string_view const* pass) noexcept
49{
50 // this function is for
51 // authority_view_rule only
52 BOOST_ASSERT(from_ == from::authority);
53
54 // userinfo
55 set_size(id_user, user.size());
56 decoded_[id_user] =
57 user.decoded_size();
58 if(pass)
59 {
60 set_size(id_pass,
61 pass->size() + 2);
62 decoded_[id_pass] =
63 pass->decoded_size();
64 }
65 else
66 {
67 // trailing '@'
68 set_size(id_pass, 1 );
69 }
70}

72void
73url_impl::
74apply_host(
75 host_type ht,
76 pct_string_view s,
77 unsigned char const* addr) noexcept
78{
79 // this function is for
80 // authority_view_rule only
81 BOOST_ASSERT(from_ == from::authority);
82
83 // host, port
84 host_type_ = ht;
85 set_size(id_host, s.size());
86 decoded_[id_host] =
87 s.decoded_size();
88 std::memcpy(
89 dest: ip_addr_,
90 src: addr,
91 n: sizeof(ip_addr_));
92}

94void
95url_impl::
96apply_port(
97 core::string_view s,
98 unsigned short pn) noexcept
99{
100 // this function is for
101 // authority_view_rule only
102 BOOST_ASSERT(from_ == from::authority);
103
104 port_number_ = pn;
105 set_size(id_port, 1 + s.size());
106}

108void
109url_impl::
110apply_authority(
111 authority_view const& a) noexcept
112{
113 BOOST_ASSERT(from_ != from::authority);
114
115 // userinfo
116 set_size(id_user,
117 a.u_.len(id_user) +
118 (from_ == from::authority ? 0 : 2));
119 set_size(id_pass, a.u_.len(id_pass));
120 decoded_[id_user] = a.u_.decoded_[id_user];
121 decoded_[id_pass] = a.u_.decoded_[id_pass];
122
123 // host, port
124 host_type_ = a.u_.host_type_;
125 port_number_ = a.u_.port_number_;
126 set_size(id_host, a.u_.len(id_host));
127 set_size(id_port, a.u_.len(id_port));
128 std::memcpy(
129 dest: ip_addr_,
130 src: a.u_.ip_addr_,
131 n: sizeof(ip_addr_));
132 decoded_[id_host] = a.u_.decoded_[id_host];
133}

166// return length of [first, last)
167auto
168url_impl::
169len(
170 int first,
171 int last) const noexcept ->
172 std::size_t
173{
174 BOOST_ASSERT(first <= last);
175 BOOST_ASSERT(last <= id_end);
176 return offset(last) - offset(first);
177}
178
179// return length of part
180auto
181url_impl::
182len(int id) const noexcept ->
183 std::size_t
184{
185 return id == id_end
186 ? zero_
187 : ( offset(id + 1) -
188 offset(id) );
189}
190
191// return offset of id
192auto
193url_impl::
194offset(int id) const noexcept ->
195 std::size_t
196{
197 return
198 id == id_scheme
199 ? zero_
200 : offset_[id];
201}

203// return id as string
204core::string_view
205url_impl::
206get(int id) const noexcept
207{
208 return {
209 cs_ + offset(id), len(id) };
210}

212// return [first, last) as string
213core::string_view
214url_impl::
215get(int first,
216 int last) const noexcept
217{
218 return { cs_ + offset(id: first),
219 offset(id: last) - offset(id: first) };
220}
221
222// return id as pct-string
223pct_string_view
224url_impl::
225pct_get(
226 int id) const noexcept
227{
228 return make_pct_string_view_unsafe(
229 data: cs_ + offset(id),
230 size: len(id),
231 decoded_size: decoded_[id]);
232}
233
234// return [first, last) as pct-string
235pct_string_view
236url_impl::
237pct_get(
238 int first,
239 int last) const noexcept
240{
241 auto const pos = offset(id: first);
242 std::size_t n = 0;
243 for(auto i = first; i < last;)
244 n += decoded_[i++];
245 return make_pct_string_view_unsafe(
246 data: cs_ + pos,
247 size: offset(id: last) - pos,
248 decoded_size: n);
249}

253// change id to size n
254void
255url_impl::
256set_size(
257 int id,
258 std::size_t n) noexcept
259{
260 auto d = n - len(id);
261 for(auto i = id + 1;
262 i <= id_end; ++i)
263 offset_[i] += d;
264}
265
266// trim id to size n,
267// moving excess into id+1
268void
269url_impl::
270split(
271 int id,
272 std::size_t n) noexcept
273{
274 BOOST_ASSERT(id < id_end - 1);
275 //BOOST_ASSERT(n <= len(id));
276 offset_[id + 1] = offset(id) + n;
277}

279// add n to [first, last]
280void
281url_impl::
282adjust_right(
283 int first,
284 int last,
285 std::size_t n) noexcept
286{
287 for(int i = first;
288 i <= last; ++i)
289 offset_[i] += n;
290}
291
292// remove n from [first, last]
293void
294url_impl::
295adjust_left(
296 int first,
297 int last,
298 std::size_t n) noexcept
299{
300 for(int i = first;
301 i <= last; ++i)
302 offset_[i] -= n;
303}
304
305// set [first, last) offset
306void
307url_impl::
308collapse(
309 int first,
310 int last,
311 std::size_t n) noexcept
312{
313 for(int i = first + 1;
314 i < last; ++i)
315 offset_[i] = n;
316}

329 if(impl.from_ == url_impl::from::url)

331 impl_ = &impl;

335 core::string_view s = impl.get(id: id_path);

424
425query_ref::
426query_ref(
427 url_impl const& impl) noexcept
428{
429 if(impl.from_ == url_impl::from::url)
430 {
431 impl_ = &impl;
432 }
433 else
434 {
435 core::string_view s = impl.get(id: id_query);
436 if (!s.empty())
437 {
438 s.remove_prefix(n: 1);
439 question_mark_ = true;
440 }
441 data_ = s.data();
442 size_ = s.size();
443 nparam_ = impl.nparam_;
444 dn_ = impl.decoded_[id_query];
445 }
446}
447
448pct_string_view
449query_ref::
450buffer() const noexcept
451{
452 if(impl_)
453 {
454 auto pos = impl_->offset_[id_query];
455 auto pos1 = impl_->offset_[id_frag];
456 if(pos < pos1)
457 {
458 ++pos; // no '?'
459 return make_pct_string_view_unsafe(
460 data: impl_->cs_ + pos,
461 size: pos1 - pos,
462 decoded_size: impl_->decoded_[id_query]);
463 }
464 // empty
465 return make_pct_string_view_unsafe(
466 data: impl_->cs_ + pos,
467 size: 0,
468 decoded_size: 0);
469 }
470 // no '?'
471 return make_pct_string_view_unsafe(
472 data: data_, size: size_, decoded_size: dn_);
473}
474
475// with '?'
476std::size_t
477query_ref::
478size() const noexcept
479{
480 if(impl_)
481 return impl_->len(id: id_query);
482 if(size_ > 0)
483 return size_ + 1;
484 return question_mark_;
485}
486
487// no '?'
488char const*
489query_ref::
490begin() const noexcept
491{
492 if(impl_)
493 {
494 // using the offset array here
495 auto pos = impl_->offset_[id_query];
496 auto pos1 = impl_->offset_[id_frag];
497 if(pos < pos1)
498 return impl_->cs_ + pos + 1; // no '?'
499 // empty
500 return impl_->cs_ + pos;
501 }
502 return data_;
503
504}
505
506char const*
507query_ref::
508end() const noexcept
509{
510 if(impl_)
511 return impl_->cs_ +
512 impl_->offset(id: id_frag);
513 return data_ + size_;
514}
