45char const*
46formatter<core::string_view>::
47parse(format_parse_context& ctx)
48{
49 char const* it = ctx.begin();
50 char const* end = ctx.end();
51 BOOST_ASSERT(it != end);
52
53 // fill / align
54 if (end - it > 2)
55 {
56 if (*it != '{' &&
57 *it != '}' &&
58 (*(it + 1) == '<' ||
59 *(it + 1) == '>' ||
60 *(it + 1) == '^'))
61 {
62 fill = *it;
63 align = *(it + 1);
64 it += 2;
65 }
66 }
67
68 // align
69 if (align == '\0' &&
70 (*it == '<' ||
71 *it == '>' ||
72 *it == '^'))
73 {
74 align = *it++;
75 }
76
77 // width
78 char const* it0 = it;
79 constexpr auto width_rule =
80 grammar::variant_rule(
81 r0: grammar::unsigned_rule<std::size_t>{},
82 rn: grammar::tuple_rule(
83 r0: grammar::squelch(
84 r: grammar::delim_rule(ch: '{')),
85 rn: grammar::optional_rule(
86 r: arg_id_rule),
87 rn: grammar::squelch(
88 r: grammar::delim_rule(ch: '}'))));
89 auto rw = grammar::parse(it, end, r: width_rule);
90 if (!rw)
91 {
92 // rewind
93 it = it0;
94 }
95 else if (align != '\0')
96 {
97 // width is ignored when align is '\0'
98 if (rw->index() == 0)
99 {
100 // unsigned_rule
101 width = variant2::get<0>(v&: *rw);
102 }
103 else
104 {
105 // arg_id: store the id idx or string
106 auto& arg_id = variant2::get<1>(v&: *rw);
107 if (!arg_id)
108 {
109 // empty arg_id, use and consume
110 // the next arg idx
111 width_idx = ctx.next_arg_id();
112 }
113 else if (arg_id->index() == 0)
114 {
115 // string identifier
116 width_name = variant2::get<0>(v&: *arg_id);
117 }
118 else
119 {
120 // integer identifier: use the
121 // idx of this format_arg
122 width_idx = variant2::get<1>(v&: *arg_id);
123 }
124 }
125 }
126
127 // type is parsed but doesn't have to
128 // be stored for strings
129 if (*it == 'c' ||
130 *it == 's')
131 {
132 ++it;
133 }
134
135 // we should have arrived at the end now
136 if (*it != '}')
137 {
138 urls::detail::throw_invalid_argument();
139 }
140
141 return it;
142}

144std::size_t
145formatter<core::string_view>::
146measure(
147 core::string_view str,
148 measure_context& ctx,
149 grammar::lut_chars const& cs) const
150{
151 std::size_t w = width;
152 if (width_idx != std::size_t(-1) ||
153 !width_name.empty())
154 {
155 get_width_from_args(
156 arg_idx: width_idx, arg_name: width_name, args: ctx.args(), w);
157 }
158
159 std::size_t n = ctx.out();
160 if (str.size() < w)
161 n += measure_one(c: fill, unreserved: cs) * (w - str.size());
162
163 return n + encoded_size(s: str, unreserved: cs);
164}

166char*
167formatter<core::string_view>::
168format(core::string_view str, format_context& ctx, grammar::lut_chars const& cs) const
169{
170 std::size_t w = width;
171 if (width_idx != std::size_t(-1) ||
172 !width_name.empty())
173 {
174 get_width_from_args(
175 arg_idx: width_idx, arg_name: width_name, args: ctx.args(), w);
176 }
177
178 std::size_t lpad = 0;
179 std::size_t rpad = 0;
180 if (str.size() < w)
181 {
182 std::size_t pad = w - str.size();
183 switch (align)
184 {
185 case '<':
186 rpad = pad;
187 break;
188 case '>':
189 lpad = pad;
190 break;
191 case '^':
192 lpad = w / 2;
193 rpad = pad - lpad;
194 break;
195 }
196 }
197
198 // unsafe `encode`, assuming `out` has
199 // enough capacity
200 char* out = ctx.out();
201 for (std::size_t i = 0; i < lpad; ++i)
202 encode_one(out, c: fill, unreserved: cs);
203 for (char c: str)
204 encode_one(out, c, unreserved: cs);
205 for (std::size_t i = 0; i < rpad; ++i)
206 encode_one(out, c: fill, unreserved: cs);
207 return out;
208}

210void
211get_width_from_args(
212 std::size_t arg_idx,
213 core::string_view arg_name,
214 format_args args,
215 std::size_t& w)
216{
217 // check arg_id
218 format_arg warg;
219 if (arg_idx != std::size_t(-1))
220 {
221 // identifier
222 warg = args.get(i: arg_idx);
223 }
224 else
225 {
226 // unsigned integer
227 warg = args.get(name: arg_name);
228 }
229
230 // get unsigned int value from that format arg
231 w = warg.value();
232}

234char const*
235integer_formatter_impl::
236parse(format_parse_context& ctx)
237{
238 char const* it = ctx.begin();
239 char const* end = ctx.end();
240 BOOST_ASSERT(it != end);
241
242 // fill / align
243 if (end - it > 2)
244 {
245 if (*it != '{' &&
246 *it != '}' &&
247 (*(it + 1) == '<' ||
248 *(it + 1) == '>' ||
249 *(it + 1) == '^'))
250 {
251 fill = *it;
252 align = *(it + 1);
253 it += 2;
254 }
255 }
256
257 // align
258 if (align == '\0' &&
259 (*it == '<' ||
260 *it == '>' ||
261 *it == '^'))
262 {
263 align = *it++;
264 }
265
266 // sign
267 if (*it == '+' ||
268 *it == '-' ||
269 *it == ' ')
270 {
271 sign = *it++;
272 }
273
274 // #
275 if (*it == '#')
276 {
277 // alternate form not supported
278 ++it;
279 }
280
281 // 0
282 if (*it == '0')
283 {
284 zeros = *it++;
285 }
286
287 // width
288 char const* it0 = it;
289 constexpr auto width_rule = grammar::variant_rule(
290 r0: grammar::unsigned_rule<std::size_t>{},
291 rn: grammar::tuple_rule(
292 r0: grammar::squelch(
293 r: grammar::delim_rule(ch: '{')),
294 rn: grammar::optional_rule(
295 r: arg_id_rule),
296 rn: grammar::squelch(
297 r: grammar::delim_rule(ch: '}'))));
298 auto rw = grammar::parse(it, end, r: width_rule);
299 if (!rw)
300 {
301 // rewind
302 it = it0;
303 }
304 else if (align != '\0')
305 {
306 // width is ignored when align is '\0'
307 if (rw->index() == 0)
308 {
309 // unsigned_rule
310 width = variant2::get<0>(v&: *rw);
311 }
312 else
313 {
314 // arg_id: store the id idx or string
315 auto& arg_id = variant2::get<1>(v&: *rw);
316 if (!arg_id)
317 {
318 // empty arg_id, use and consume
319 // the next arg idx
320 width_idx = ctx.next_arg_id();
321 }
322 else if (arg_id->index() == 0)
323 {
324 // string identifier
325 width_name = variant2::get<0>(v&: *arg_id);
326 }
327 else
328 {
329 // integer identifier: use the
330 // idx of this format_arg
331 width_idx = variant2::get<1>(v&: *arg_id);
332 }
333 }
334 }
335
336 // type is parsed but doesn't have to
337 // be stored for strings
338 if (*it == 'd')
339 {
340 // we don't include other presentation
341 // modes for integers as they are not
342 // recommended or generally used in
343 // urls
344 ++it;
345 }
346
347 // we should have arrived at the end now
348 if (*it != '}')
349 {
350 urls::detail::throw_invalid_argument();
351 }
352
353 return it;
354}

356std::size_t
357integer_formatter_impl::
358measure(
359 long long int v,
360 measure_context& ctx,
361 grammar::lut_chars const& cs) const
362{
363 std::size_t dn = 0;
364 std::size_t n = 0;
365 if (v < 0)
366 {
367 dn += measure_one(c: '-', unreserved: cs);
368 ++n;
369 v *= -1;
370 }
371 else if (sign != '-')
372 {
373 dn += measure_one(c: sign, unreserved: cs);
374 ++n;
375 }
376 do
377 {
378 int d = v % 10;
379 v /= 10;
380 dn += measure_one(c: '0' + static_cast<char>(d), unreserved: cs);
381 ++n;
382 }
383 while (v > 0);
384
385 std::size_t w = width;
386 if (width_idx != std::size_t(-1) ||
387 !width_name.empty())
388 {
389 get_width_from_args(
390 arg_idx: width_idx, arg_name: width_name, args: ctx.args(), w);
391 }
392 if (w > n)
393 {
394 if (!zeros)
395 dn += measure_one(c: fill, unreserved: cs) * (w - n);
396 else
397 dn += measure_one(c: '0', unreserved: cs) * (w - n);
398 }
399 return ctx.out() + dn;
400}

442char*
443integer_formatter_impl::
444format(
445 long long int v,
446 format_context& ctx,
447 grammar::lut_chars const& cs) const
448{
449 // get n digits
450 long long int v0 = v;
451 long long int p = 1;
452 std::size_t n = 0;
453 if (v < 0)
454 {
455 v *= - 1;
456 ++n;
457 }
458 else if (sign != '-')
459 {
460 ++n;
461 }
462 do
463 {
464 if (v >= 10)
465 p *= 10;
466 v /= 10;
467 ++n;
468 }
469 while (v > 0);
470 static constexpr auto m =
471 std::numeric_limits<long long int>::digits10;
472 BOOST_ASSERT(n <= m + 1);
473 ignore_unused(m);
474
475 // get pad
476 std::size_t w = width;
477 if (width_idx != std::size_t(-1) ||
478 !width_name.empty())
479 {
480 get_width_from_args(
481 arg_idx: width_idx, arg_name: width_name, args: ctx.args(), w);
482 }
483 std::size_t lpad = 0;
484 std::size_t rpad = 0;
485 if (w > n)
486 {
487 std::size_t pad = w - n;
488 if (zeros)
489 {
490 lpad = pad;
491 }
492 else
493 {
494 switch (align)
495 {
496 case '<':
497 rpad = pad;
498 break;
499 case '>':
500 lpad = pad;
501 break;
502 case '^':
503 lpad = pad / 2;
504 rpad = pad - lpad;
505 break;
506 }
507 }
508 }
509
510 // write
511 v = v0;
512 char* out = ctx.out();
513 if (!zeros)
514 {
515 for (std::size_t i = 0; i < lpad; ++i)
516 encode_one(out, c: fill, unreserved: cs);
517 }
518 if (v < 0)
519 {
520 encode_one(out, c: '-', unreserved: cs);
521 v *= -1;
522 --n;
523 }
524 else if (sign != '-')
525 {
526 encode_one(out, c: sign, unreserved: cs);
527 --n;
528 }
529 if (zeros)
530 {
531 for (std::size_t i = 0; i < lpad; ++i)
532 encode_one(out, c: '0', unreserved: cs);
533 }
534 while (n)
535 {
536 unsigned long long int d = v / p;
537 encode_one(out, c: '0' + static_cast<char>(d), unreserved: cs);
538 --n;
539 v %= p;
540 p /= 10;
541 }
542 if (!zeros)
543 {
544 for (std::size_t i = 0; i < rpad; ++i)
545 encode_one(out, c: fill, unreserved: cs);
546 }
547 return out;
548}
