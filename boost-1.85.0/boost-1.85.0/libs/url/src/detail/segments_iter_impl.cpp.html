46segments_iter_impl::
47segments_iter_impl(
48 url_impl const& u_,
49 std::size_t pos_,
50 std::size_t index_) noexcept
51 : ref(u_)
52 , pos(pos_)
53 , index(index_)
54{
55 if(index == 0)
56 {
57 pos = path_prefix(s: ref.buffer());
58 }
59 else if(pos != ref.size())
60 {
61 BOOST_ASSERT(
62 ref.data()[pos] == '/');
63 ++pos; // skip '/'
64 }
65 update();
66}
67
68void
69segments_iter_impl::
70update() noexcept
71{
72 auto const end = ref.end();
73 char const* const p0 =
74 ref.data() + pos;
75 dn = 0;
76 auto p = p0;
77 while(p != end)
78 {
79 if(*p == '/')
80 break;
81 if(*p != '%')
82 {
83 ++p;
84 continue;
85 }
86 p += 3;
87 dn += 2;
88 }
89 next = p - ref.data();
90 dn = p - p0 - dn;
91 s_ = make_pct_string_view_unsafe(
92 data: p0, size: p - p0, decoded_size: dn);
93}

95void
96segments_iter_impl::
97increment() noexcept
98{
99 BOOST_ASSERT(
100 index != ref.nseg());
101 ++index;
102 pos = next;
103 if(index == ref.nseg())
104 return;
105 // "/" segment
106 auto const end = ref.end();
107 auto p = ref.data() + pos;
108 BOOST_ASSERT(p != end);
109 BOOST_ASSERT(*p == '/');
110 dn = 0;
111 ++p; // skip '/'
112 auto const p0 = p;
113 while(p != end)
114 {
115 if(*p == '/')
116 break;
117 if(*p != '%')
118 {
119 ++p;
120 continue;
121 }
122 p += 3;
123 dn += 2;
124 }
125 next = p - ref.data();
126 dn = p - p0 - dn;
127 s_ = make_pct_string_view_unsafe(
128 data: p0, size: p - p0, decoded_size: dn);
129}
130
131void
132segments_iter_impl::
133decrement() noexcept
134{
135 BOOST_ASSERT(index != 0);
136 --index;
137 if(index == 0)
138 {
139 next = pos;
140 pos = path_prefix(s: ref.buffer());
141 s_ = core::string_view(
142 ref.data() + pos,
143 next - pos);
144 BOOST_ASSERT(! s_.ends_with('/'));
145 return;
146 }
147 auto const begin = ref.data() +
148 path_prefix(s: ref.buffer());
149 next = pos;
150 auto p = ref.data() + next;
151 auto const p1 = p;
152 BOOST_ASSERT(p != begin);
153 dn = 0;
154 while(p != begin)
155 {
156 --p;
157 if(*p == '/')
158 {
159 ++dn;
160 break;
161 }
162 if(*p == '%')
163 dn += 2;
164 }
165 dn = p1 - p - dn;
166 pos = p - ref.data();
167 s_ = make_pct_string_view_unsafe(
168 data: p + 1, size: p1 - p - 1, decoded_size: dn);
169}
