30segment_iter::
31segment_iter(
32 core::string_view s_) noexcept
33 : any_segments_iter(s_)
34{
35 front = s;
36 fast_nseg = 1;
37}
38
39void
40segment_iter::
41rewind() noexcept
42{
43 at_end_ = false;
44}
45
46bool
47segment_iter::
48measure(
49 std::size_t& n) noexcept
50{
51 if(at_end_)
52 return false;
53 encoding_opts opt;
54 opt.space_as_plus = false;
55 n += encoded_size(
56 s,
57 unreserved: encode_colons ?
58 nocolon_pchars :
59 pchars,
60 opt);
61 at_end_ = true;
62 return true;
63}

73 dest += encode(
74 dest,
75 size: end - dest,
76 s,
77 unreserved: encode_colons ?
78 nocolon_pchars :
79 pchars,
80 opt);
81}
