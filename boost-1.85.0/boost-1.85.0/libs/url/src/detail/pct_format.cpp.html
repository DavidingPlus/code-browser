22std::size_t
23pct_vmeasure(
24 grammar::lut_chars const& cs,
25 format_parse_context& pctx,
26 measure_context& mctx)
27{
28 auto it0 = pctx.begin();
29 auto end = pctx.end();
30 while( it0 != end )
31 {
32 // look for replacement id
33 char const* it1 = it0;
34 while(
35 it1 != end &&
36 *it1 != '{' )
37 {
38 ++it1;
39 }
40
41 // output literal prefix
42 if( it0 != it1 )
43 {
44 for (char const* i = it0; i != it1; ++i)
45 mctx.advance_to( n: mctx.out() + measure_one(c: *i, unreserved: cs));
46 }
47
48 // over
49 if( it1 == end )
50 {
51 break;
52 }
53
54 // enter replacement id
55 ++it1;
56 BOOST_ASSERT(it1 != end);
57
58 // handle escaped replacement (second '{')
59 // there's no "{{" in URL templates because
60 // '{'s are not allowed in URLs
61 BOOST_ASSERT(*it1 != '{');
62 /*
63 if( *it1 == '{' )
64 {
65 mctx.advance_to( mctx.out() + measure_one('{', cs));
66 ++it1;
67 // this was not a real replacement,
68 // so we just keep moving
69 continue;
70 }
71 */
72
73
74 // parse {id} or {id:specs}
75 char const* id_start = it1;
76 while (it1 != end &&
77 *it1 != ':' &&
78 *it1 != '}')
79 {
80 ++it1;
81 }
82 core::string_view id(id_start, it1);
83
84 // move to specs start
85 if (it1 != end &&
86 *it1 == ':')
87 ++it1;
88 pctx.advance_to( it: it1 );
89
90 // get format_arg to use
91 auto idv = grammar::parse(
92 s: id, r: grammar::unsigned_rule<std::size_t>{});
93 if (idv)
94 {
95 mctx.arg( id: *idv ).measure( pctx, mctx, cs );
96 }
97 else if (!id.empty())
98 {
99 mctx.arg( name: id ).measure( pctx, mctx, cs );
100 }
101 else
102 {
103 std::size_t arg_id = pctx.next_arg_id();
104 mctx.arg( id: arg_id ).measure( pctx, mctx, cs );
105 }
106
107
108 it1 = pctx.begin();
109 BOOST_ASSERT(*it1 == '}');
110 it0 = it1 + 1;
111 }
112
113 return mctx.out();
114}

116char*
117pct_vformat(
118 grammar::lut_chars const& cs,
119 format_parse_context& pctx,
120 format_context& fctx)
121{
122 auto it0 = pctx.begin();
123 auto end = pctx.end();
124 while( it0 != end )
125 {
126 // look for replacement id
127 char const* it1 = it0;
128 while(
129 it1 != end &&
130 *it1 != '{' )
131 {
132 ++it1;
133 }
134
135 // output literal prefix
136 if( it0 != it1 )
137 {
138 for (char const* i = it0; i != it1; ++i)
139 {
140 char* o = fctx.out();
141 encode_one(out&: o, c: *i, unreserved: cs);
142 fctx.advance_to(it: o);
143 }
144 }
145
146 // over
147 if( it1 == end )
148 {
149 break;
150 }
151
152 // enter replacement id
153 ++it1;
154 BOOST_ASSERT(it1 != end);
155
156 // handle escaped replacement (second '{')
157 // there's no "{{" in URL templates because
158 // '{'s are not allowed in URLs
159 BOOST_ASSERT(*it1 != '{');
160 /*
161 if( *it1 == '{' )
162 {
163 char* o = fctx.out();
164 encode_one(o, '{', cs);
165 fctx.advance_to(o);
166 ++it1;
167 // this was not a real replacement,
168 // so we just keep moving
169 continue;
170 }
171 */
172
173 // parse {id} or {id:specs}
174 char const* id_start = it1;
175 while (it1 != end &&
176 *it1 != ':' &&
177 *it1 != '}')
178 {
179 ++it1;
180 }
181 core::string_view id(id_start, it1);
182
183 // move to specs part
184 if (it1 != end &&
185 *it1 == ':')
186 ++it1;
187 pctx.advance_to( it: it1 );
188
189 // get format_arg to use
190 auto idv = grammar::parse(
191 s: id, r: grammar::unsigned_rule<std::size_t>{});
192 if (idv)
193 {
194 fctx.arg( id: *idv ).format( pctx, fctx, cs );
195 }
196 else if (!id.empty())
197 {
198 fctx.arg( name: id ).format( pctx, fctx, cs );
199 }
200 else
201 {
202 std::size_t arg_id = pctx.next_arg_id();
203 fctx.arg( id: arg_id ).format( pctx, fctx, cs );
204 }
205
206 it1 = pctx.begin();
207 BOOST_ASSERT(*it1 == '}');
208 it0 = it1 + 1;
209 }
210
211 return fctx.out();
212}
