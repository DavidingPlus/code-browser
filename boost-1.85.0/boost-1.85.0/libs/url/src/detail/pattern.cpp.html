30static constexpr auto lhost_chars = host_chars + ':';
31
32void
33pattern::
34apply(
35 url_base& u,
36 format_args const& args) const
37{
38 // measure total
39 struct sizes
40 {
41 std::size_t scheme = 0;
42 std::size_t user = 0;
43 std::size_t pass = 0;
44 std::size_t host = 0;
45 std::size_t port = 0;
46 std::size_t path = 0;
47 std::size_t query = 0;
48 std::size_t frag = 0;
49 };
50 sizes n;
51
52 format_parse_context pctx(nullptr, nullptr, 0);
53 measure_context mctx(args);
54 if (!scheme.empty())
55 {
56 pctx = {scheme, pctx.next_arg_id()};
57 n.scheme = pct_vmeasure(
58 cs: grammar::alpha_chars, pctx, mctx);
59 mctx.advance_to(n: 0);
60 }
61 if (has_authority)
62 {
63 if (has_user)
64 {
65 pctx = {user, pctx.next_arg_id()};
66 n.user = pct_vmeasure(
67 cs: user_chars, pctx, mctx);
68 mctx.advance_to(n: 0);
69 if (has_pass)
70 {
71 pctx = {pass, pctx.next_arg_id()};
72 n.pass = pct_vmeasure(
73 cs: password_chars, pctx, mctx);
74 mctx.advance_to(n: 0);
75 }
76 }
77 if (host.starts_with(x: '['))
78 {
79 BOOST_ASSERT(host.ends_with(']'));
80 pctx = {host.substr(pos: 1, n: host.size() - 2), pctx.next_arg_id()};
81 n.host = pct_vmeasure(
82 cs: lhost_chars, pctx, mctx) + 2;
83 mctx.advance_to(n: 0);
84 }
85 else
86 {
87 pctx = {host, pctx.next_arg_id()};
88 n.host = pct_vmeasure(
89 cs: host_chars, pctx, mctx);
90 mctx.advance_to(n: 0);
91 }
92 if (has_port)
93 {
94 pctx = {port, pctx.next_arg_id()};
95 n.port = pct_vmeasure(
96 cs: grammar::digit_chars, pctx, mctx);
97 mctx.advance_to(n: 0);
98 }
99 }
100 if (!path.empty())
101 {
102 pctx = {path, pctx.next_arg_id()};
103 n.path = pct_vmeasure(
104 cs: path_chars, pctx, mctx);
105 mctx.advance_to(n: 0);
106 }
107 if (has_query)
108 {
109 pctx = {query, pctx.next_arg_id()};
110 n.query = pct_vmeasure(
111 cs: query_chars, pctx, mctx);
112 mctx.advance_to(n: 0);
113 }
114 if (has_frag)
115 {
116 pctx = {frag, pctx.next_arg_id()};
117 n.frag = pct_vmeasure(
118 cs: fragment_chars, pctx, mctx);
119 mctx.advance_to(n: 0);
120 }
121 std::size_t const n_total =
122 n.scheme +
123 (n.scheme != 0) * 1 + // ":"
124 has_authority * 2 + // "//"
125 n.user +
126 has_pass * 1 + // ":"
127 n.pass +
128 has_user * 1 + // "@"
129 n.host +
130 has_port * 1 + // ":"
131 n.port +
132 n.path +
133 has_query * 1 + // "?"
134 n.query +
135 has_frag * 1 + // "#"
136 n.frag;
137 u.reserve(n: n_total);
138
139 // Apply
140 pctx = {nullptr, nullptr, 0};
141 format_context fctx(nullptr, args);
142 url_base::op_t op(u);
143 using parts = parts_base;
144 if (!scheme.empty())
145 {
146 auto dest = u.resize_impl(
147 parts::id_scheme,
148 n.scheme + 1, op);
149 pctx = {scheme, pctx.next_arg_id()};
150 fctx.advance_to(it: dest);
151 const char* dest1 = pct_vformat(
152 cs: grammar::alpha_chars, pctx, fctx);
153 dest[n.scheme] = ':';
154 // validate
155 if (!grammar::parse(s: {dest, dest1}, r: scheme_rule()))
156 {
157 throw_invalid_argument();
158 }
159 }
160 if (has_authority)
161 {
162 if (has_user)
163 {
164 auto dest = u.set_user_impl(
165 n: n.user, op);
166 pctx = {user, pctx.next_arg_id()};
167 fctx.advance_to(it: dest);
168 char const* dest1 = pct_vformat(
169 cs: user_chars, pctx, fctx);
170 u.impl_.decoded_[parts::id_user] =
171 pct_string_view(dest, dest1 - dest)
172 ->decoded_size();
173 if (has_pass)
174 {
175 char* destp = u.set_password_impl(
176 n: n.pass, op);
177 pctx = {pass, pctx.next_arg_id()};
178 fctx.advance_to(it: destp);
179 dest1 = pct_vformat(
180 cs: password_chars, pctx, fctx);
181 u.impl_.decoded_[parts::id_pass] =
182 pct_string_view({destp, dest1})
183 ->decoded_size() + 1;
184 }
185 }
186 auto dest = u.set_host_impl(
187 n: n.host, op);
188 if (host.starts_with(x: '['))
189 {
190 BOOST_ASSERT(host.ends_with(']'));
191 pctx = {host.substr(pos: 1, n: host.size() - 2), pctx.next_arg_id()};
192 *dest++ = '[';
193 fctx.advance_to(it: dest);
194 char* dest1 =
195 pct_vformat(cs: lhost_chars, pctx, fctx);
196 *dest1++ = ']';
197 u.impl_.decoded_[parts::id_host] =
198 pct_string_view(dest - 1, dest1 - dest)
199 ->decoded_size();
200 }
201 else
202 {
203 pctx = {host, pctx.next_arg_id()};
204 fctx.advance_to(it: dest);
205 char const* dest1 =
206 pct_vformat(cs: host_chars, pctx, fctx);
207 u.impl_.decoded_[parts::id_host] =
208 pct_string_view(dest, dest1 - dest)
209 ->decoded_size();
210 }
211 auto uh = u.encoded_host();
212 auto h = grammar::parse(s: uh, r: host_rule).value();
213 std::memcpy(
214 dest: u.impl_.ip_addr_,
215 src: h.addr,
216 n: sizeof(u.impl_.ip_addr_));
217 u.impl_.host_type_ = h.host_type;
218 if (has_port)
219 {
220 dest = u.set_port_impl(n: n.port, op);
221 pctx = {port, pctx.next_arg_id()};
222 fctx.advance_to(it: dest);
223 char const* dest1 = pct_vformat(
224 cs: grammar::digit_chars, pctx, fctx);
225 u.impl_.decoded_[parts::id_port] =
226 pct_string_view(dest, dest1 - dest)
227 ->decoded_size() + 1;
228 core::string_view up = {dest - 1, dest1};
229 auto p = grammar::parse(s: up, r: detail::port_part_rule).value();
230 if (p.has_port)
231 u.impl_.port_number_ = p.port_number;
232 }
233 }
234 if (!path.empty())
235 {
236 auto dest = u.resize_impl(
237 parts::id_path,
238 n.path, op);
239 pctx = {path, pctx.next_arg_id()};
240 fctx.advance_to(it: dest);
241 auto dest1 = pct_vformat(
242 cs: path_chars, pctx, fctx);
243 pct_string_view npath(dest, dest1 - dest);
244 u.impl_.decoded_[parts::id_path] +=
245 npath.decoded_size();
246 if (!npath.empty())
247 {
248 u.impl_.nseg_ = std::count(
249 first: npath.begin() + 1,
250 last: npath.end(), value: '/') + 1;
251 }
252 // handle edge cases
253 // 1) path is first component and the
254 // first segment contains an unencoded ':'
255 // This is impossible because the template
256 // "{}" would be a host.
257 if (u.scheme().empty() &&
258 !u.has_authority())
259 {
260 auto fseg = u.encoded_segments().front();
261 std::size_t nc = std::count(
262 first: fseg.begin(), last: fseg.end(), value: ':');
263 if (nc)
264 {
265 std::size_t diff = nc * 2;
266 u.reserve(n: n_total + diff);
267 dest = u.resize_impl(
268 parts::id_path,
269 n.path + diff, op);
270 char* dest0 = dest + diff;
271 std::memmove(dest: dest0, src: dest, n: n.path);
272 while (dest0 != dest)
273 {
274 if (*dest0 != ':')
275 {
276 *dest++ = *dest0++;
277 }
278 else
279 {
280 *dest++ = '%';
281 *dest++ = '3';
282 *dest++ = 'A';
283 dest0++;
284 }
285 }
286 }
287 }
288 // 2) url has no authority and path
289 // starts with "//"
290 if (!u.has_authority() &&
291 u.encoded_path().starts_with(x: "//"))
292 {
293 u.reserve(n: n_total + 2);
294 dest = u.resize_impl(
295 parts::id_path,
296 n.path + 2, op);
297 std::memmove(dest: dest + 2, src: dest, n: n.path);
298 *dest++ = '/';
299 *dest = '.';
300 }
301 }
302 if (has_query)
303 {
304 auto dest = u.resize_impl(
305 parts::id_query,
306 n.query + 1, op);
307 *dest++ = '?';
308 pctx = {query, pctx.next_arg_id()};
309 fctx.advance_to(it: dest);
310 auto dest1 = pct_vformat(
311 cs: query_chars, pctx, fctx);
312 pct_string_view nquery(dest, dest1 - dest);
313 u.impl_.decoded_[parts::id_query] +=
314 nquery.decoded_size() + 1;
315 if (!nquery.empty())
316 {
317 u.impl_.nparam_ = std::count(
318 first: nquery.begin(),
319 last: nquery.end(), value: '&') + 1;
320 }
321 }
322 if (has_frag)
323 {
324 auto dest = u.resize_impl(
325 parts::id_frag,
326 n.frag + 1, op);
327 *dest++ = '#';
328 pctx = {frag, pctx.next_arg_id()};
329 fctx.advance_to(it: dest);
330 auto dest1 = pct_vformat(
331 cs: fragment_chars, pctx, fctx);
332 u.impl_.decoded_[parts::id_frag] +=
333 make_pct_string_view(
334 s: core::string_view(dest, dest1 - dest))
335 ->decoded_size() + 1;
336 }
337}

339// This rule represents a pct-encoded string
340// that contains an arbitrary number of
341// replacement ids in it
342template<class CharSet>
343struct pct_encoded_fmt_string_rule_t
344{
345 using value_type = pct_string_view;
346
347 constexpr
348 pct_encoded_fmt_string_rule_t(
349 CharSet const& cs) noexcept
350 : cs_(cs)
351 {
352 }
353
354 template<class CharSet_>
355 friend
356 constexpr
357 auto
358 pct_encoded_fmt_string_rule(
359 CharSet_ const& cs) noexcept ->
360 pct_encoded_fmt_string_rule_t<CharSet_>;
361
362 system::result<value_type>
363 parse(
364 char const*& it,
365 char const* end) const noexcept
366 {
367 auto const start = it;
368 if(it == end)
369 {
370 // this might be empty
371 return {};
372 }
373
374 // consume some with literal rule
375 // this might be an empty literal
376 auto literal_rule = pct_encoded_rule(cs_);
377 auto rv = literal_rule.parse(it, end);
378 while (rv)
379 {
380 auto it0 = it;
381 // consume some with replacement id
382 // rule
383 if (!replacement_field_rule.parse(it, end))
384 {
385 it = it0;
386 break;
387 }
388 rv = literal_rule.parse(it, end);
389 }
390
391 return core::string_view(start, it - start);
392 }
393
394private:
395 CharSet cs_;
396};
397
398template<class CharSet>
399constexpr
400auto
401pct_encoded_fmt_string_rule(
402 CharSet const& cs) noexcept ->
403 pct_encoded_fmt_string_rule_t<CharSet>
404{
405 // If an error occurs here it means that
406 // the value of your type does not meet
407 // the requirements. Please check the
408 // documentation!
409 static_assert(
410 grammar::is_charset<CharSet>::value,
411 "CharSet requirements not met");
412
413 return pct_encoded_fmt_string_rule_t<CharSet>(cs);
414}

416// This rule represents a regular string with
417// only chars from the specified charset and
418// an arbitrary number of replacement ids in it
419template<class CharSet>
420struct fmt_token_rule_t
421{
422 using value_type = pct_string_view;
423
424 constexpr
425 fmt_token_rule_t(
426 CharSet const& cs) noexcept
427 : cs_(cs)
428 {
429 }
430
431 template<class CharSet_>
432 friend
433 constexpr
434 auto
435 fmt_token_rule(
436 CharSet_ const& cs) noexcept ->
437 fmt_token_rule_t<CharSet_>;
438
439 system::result<value_type>
440 parse(
441 char const*& it,
442 char const* end) const noexcept
443 {
444 auto const start = it;
445 BOOST_ASSERT(it != end);
446 /*
447 // This should never happen because
448 // all tokens are optional and will
449 // already return `none`:
450 if(it == end)
451 {
452 BOOST_URL_RETURN_EC(
453 grammar::error::need_more);
454 }
455 */
456
457 // consume some with literal rule
458 // this might be an empty literal
459 auto partial_token_rule =
460 grammar::optional_rule(
461 grammar::token_rule(cs_));
462 auto rv = partial_token_rule.parse(it, end);
463 while (rv)
464 {
465 auto it0 = it;
466 // consume some with replacement id
467 if (!replacement_field_rule.parse(it, end))
468 {
469 // no replacement and no more cs
470 // before: nothing else to consume
471 it = it0;
472 break;
473 }
474 // after {...}, consume any more chars
475 // in the charset
476 rv = partial_token_rule.parse(it, end);
477 }
478
479 if(it == start)
480 {
481 // it != end but we consumed nothing
482 BOOST_URL_RETURN_EC(
483 grammar::error::need_more);
484 }
485
486 return core::string_view(start, it - start);
487 }
488
489private:
490 CharSet cs_;
491};
492
493template<class CharSet>
494constexpr
495auto
496fmt_token_rule(
497 CharSet const& cs) noexcept ->
498 fmt_token_rule_t<CharSet>
499{
500 // If an error occurs here it means that
501 // the value of your type does not meet
502 // the requirements. Please check the
503 // documentation!
504 static_assert(
505 grammar::is_charset<CharSet>::value,
506 "CharSet requirements not met");
507
508 return fmt_token_rule_t<CharSet>(cs);
509}

511struct userinfo_template_rule_t
512{
513 struct value_type
514 {
515 core::string_view user;
516 core::string_view password;
517 bool has_password = false;
518 };
519
520 auto
521 parse(
522 char const*& it,
523 char const* end
524 ) const noexcept ->
525 system::result<value_type>
526 {
527 static constexpr auto uchars =
528 unreserved_chars +
529 sub_delim_chars;
530 static constexpr auto pwchars =
531 uchars + ':';
532
533 value_type t;
534
535 // user
536 static constexpr auto user_fmt_rule =
537 pct_encoded_fmt_string_rule(cs: uchars);
538 auto rv = grammar::parse(
539 it, end, r: user_fmt_rule);
540 BOOST_ASSERT(rv);
541 t.user = *rv;
542
543 // ':'
544 if( it == end ||
545 *it != ':')
546 {
547 t.has_password = false;
548 t.password = {};
549 return t;
550 }
551 ++it;
552
553 // pass
554 static constexpr auto pass_fmt_rule =
555 pct_encoded_fmt_string_rule(cs: grammar::ref(cs: pwchars));
556 rv = grammar::parse(
557 it, end, r: pass_fmt_rule);
558 BOOST_ASSERT(rv);
559 t.has_password = true;
560 t.password = *rv;
561
562 return t;
563 }
564};

568struct host_template_rule_t
569{
570 using value_type = core::string_view;
571
572 auto
573 parse(
574 char const*& it,
575 char const* end
576 ) const noexcept ->
577 system::result<value_type>
578 {
579 if(it == end)
580 {
581 // empty host
582 return {};
583 }
584
585 // the host type will be ultimately
586 // validated when applying the replacement
587 // strings. Any chars allowed in hosts
588 // are allowed here.
589 if (*it != '[')
590 {
591 // IPv4address and reg-name have the
592 // same char sets.
593 constexpr auto any_host_template_rule =
594 pct_encoded_fmt_string_rule(cs: host_chars);
595 auto rv = grammar::parse(
596 it, end, r: any_host_template_rule);
597 // any_host_template_rule can always
598 // be empty, so it's never invalid
599 BOOST_ASSERT(rv);
600 return detail::to_sv(s: *rv);
601 }
602 // IP-literals need to be enclosed in
603 // "[]" if using ':' in the template
604 // string, because the ':' would be
605 // ambiguous with the port in fmt string.
606 // The "[]:" can be used in replacement
607 // strings without the "[]" though.
608 constexpr auto ip_literal_template_rule =
609 pct_encoded_fmt_string_rule(cs: lhost_chars);
610 auto it0 = it;
611 auto rv = grammar::parse(
612 it, end,
613 r: grammar::optional_rule(
614 r: grammar::tuple_rule(
615 r0: grammar::squelch(
616 r: grammar::delim_rule(ch: '[')),
617 rn: ip_literal_template_rule,
618 rn: grammar::squelch(
619 r: grammar::delim_rule(ch: ']')))));
620 // ip_literal_template_rule can always
621 // be empty, so it's never invalid, but
622 // the rule might fail to match the
623 // closing "]"
624 BOOST_ASSERT(rv);
625 return core::string_view{it0, it};
626 }
627};

631struct authority_template_rule_t
632{
633 using value_type = pattern;
634
635 system::result<value_type>
636 parse(
637 char const*& it,
638 char const* end
639 ) const noexcept
640 {
641 pattern u;
642
643 // [ userinfo "@" ]
644 {
645 auto rv = grammar::parse(
646 it, end,
647 r: grammar::optional_rule(
648 r: grammar::tuple_rule(
649 r0: userinfo_template_rule,
650 rn: grammar::squelch(
651 r: grammar::delim_rule(ch: '@')))));
652 BOOST_ASSERT(rv);
653 if(rv->has_value())
654 {
655 auto& r = **rv;
656 u.has_user = true;
657 u.user = r.user;
658 u.has_pass = r.has_password;
659 u.pass = r.password;
660 }
661 }
662
663 // host
664 {
665 auto rv = grammar::parse(
666 it, end,
667 r: host_template_rule);
668 // host is allowed to be empty
669 BOOST_ASSERT(rv);
670 u.host = *rv;
671 }
672
673 // [ ":" port ]
674 {
675 constexpr auto port_template_rule =
676 grammar::optional_rule(
677 r: fmt_token_rule(cs: grammar::digit_chars));
678 auto it0 = it;
679 auto rv = grammar::parse(
680 it, end,
681 r: grammar::tuple_rule(
682 r0: grammar::squelch(
683 r: grammar::delim_rule(ch: ':')),
684 rn: port_template_rule));
685 if (!rv)
686 {
687 it = it0;
688 }
689 else
690 {
691 u.has_port = true;
692 if (rv->has_value())
693 {
694 u.port = **rv;
695 }
696 }
697 }
698
699 return u;
700 }
701};

705struct scheme_template_rule_t
706{
707 using value_type = core::string_view;
708
709 system::result<value_type>
710 parse(
711 char const*& it,
712 char const* end) const noexcept
713 {
714 auto const start = it;
715 if(it == end)
716 {
717 // scheme can't be empty
718 BOOST_URL_RETURN_EC(
719 grammar::error::mismatch);
720 }
721 if(!grammar::alpha_chars(*it) &&
722 *it != '{')
723 {
724 // expected alpha
725 BOOST_URL_RETURN_EC(
726 grammar::error::mismatch);
727 }
728
729 // it starts with replacement id or alpha char
730 if (!grammar::alpha_chars(*it))
731 {
732 if (!replacement_field_rule.parse(it, end))
733 {
734 // replacement_field_rule is invalid
735 BOOST_URL_RETURN_EC(
736 grammar::error::mismatch);
737 }
738 }
739 else
740 {
741 // skip first
742 ++it;
743 }
744
745 static
746 constexpr
747 grammar::lut_chars scheme_chars(
748 "0123456789" "+-."
749 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
750 "abcdefghijklmnopqrstuvwxyz");
751
752 // non-scheme chars might be a new
753 // replacement-id or just an invalid char
754 it = grammar::find_if_not(
755 first: it, last: end, cs: scheme_chars);
756 while (it != end)
757 {
758 auto it0 = it;
759 if (!replacement_field_rule.parse(it, end))
760 {
761 it = it0;
762 break;
763 }
764 it = grammar::find_if_not(
765 first: it, last: end, cs: scheme_chars);
766 }
767 return core::string_view(start, it - start);
768 }
769};

773// This rule should consider all url types at the
774// same time according to the format string
775// - relative urls with no scheme/authority
776// - absolute urls have no fragment
777struct pattern_rule_t
778{
779 using value_type = pattern;
780
781 system::result<value_type>
782 parse(
783 char const*& it,
784 char const* const end
785 ) const noexcept
786 {
787 pattern u;
788
789 // optional scheme
790 {
791 auto it0 = it;
792 auto rv = grammar::parse(
793 it, end,
794 r: grammar::tuple_rule(
795 r0: scheme_template_rule,
796 rn: grammar::squelch(
797 r: grammar::delim_rule(ch: ':'))));
798 if(rv)
799 u.scheme = *rv;
800 else
801 it = it0;
802 }
803
804 // hier_part (authority + path)
805 // if there are less than 2 chars left,
806 // we are parsing the path
807 if (it == end)
808 {
809 // this is over, so we can consider
810 // that a "path-empty"
811 return u;
812 }
813 if(end - it == 1)
814 {
815 // only one char left
816 // it can be a single separator "/",
817 // representing an empty absolute path,
818 // or a single-char segment
819 if(*it == '/')
820 {
821 // path-absolute
822 u.path = {it, 1};
823 ++it;
824 return u;
825 }
826 // this can be a:
827 // - path-noscheme if there's no scheme, or
828 // - path-rootless with a single char, or
829 // - path-empty (and consume nothing)
830 if (!u.scheme.empty() ||
831 *it != ':')
832 {
833 // path-rootless with a single char
834 // this needs to be a segment because
835 // the authority needs two slashes
836 // "//"
837 // path-noscheme also matches here
838 // because we already validated the
839 // first char
840 auto rv = grammar::parse(
841 it, end, r: urls::detail::segment_rule);
842 if(! rv)
843 return rv.error();
844 u.path = *rv;
845 }
846 return u;
847 }
848
849 // authority
850 if( it[0] == '/' &&
851 it[1] == '/')
852 {
853 // "//" always indicates authority
854 it += 2;
855 auto rv = grammar::parse(
856 it, end,
857 r: authority_template_rule);
858 // authority is allowed to be empty
859 BOOST_ASSERT(rv);
860 u.has_authority = true;
861 u.has_user = rv->has_user;
862 u.user = rv->user;
863 u.has_pass = rv->has_pass;
864 u.pass = rv->pass;
865 u.host = rv->host;
866 u.has_port = rv->has_port;
867 u.port = rv->port;
868 }
869
870 // the authority requires an absolute path
871 // or an empty path
872 if (it == end ||
873 (u.has_authority &&
874 (*it != '/' &&
875 *it != '?' &&
876 *it != '#')))
877 {
878 // path-empty
879 return u;
880 }
881
882 // path-abempty
883 // consume the whole path at once because
884 // we're going to count number of segments
885 // later after the replacements happen
886 static constexpr auto segment_fmt_rule =
887 pct_encoded_fmt_string_rule(cs: path_chars);
888 auto rp = grammar::parse(
889 it, end, r: segment_fmt_rule);
890 // path-abempty is allowed to be empty
891 BOOST_ASSERT(rp);
892 u.path = *rp;
893
894 // [ "?" query ]
895 {
896 static constexpr auto query_fmt_rule =
897 pct_encoded_fmt_string_rule(cs: query_chars);
898 auto rv = grammar::parse(
899 it, end,
900 r: grammar::tuple_rule(
901 r0: grammar::squelch(
902 r: grammar::delim_rule(ch: '?')),
903 rn: query_fmt_rule));
904 // query is allowed to be empty but
905 // delim rule is not
906 if (rv)
907 {
908 u.has_query = true;
909 u.query = *rv;
910 }
911 }
912
913 // [ "#" fragment ]
914 {
915 static constexpr auto frag_fmt_rule =
916 pct_encoded_fmt_string_rule(cs: fragment_chars);
917 auto rv = grammar::parse(
918 it, end,
919 r: grammar::tuple_rule(
920 r0: grammar::squelch(
921 r: grammar::delim_rule(ch: '#')),
922 rn: frag_fmt_rule));
923 // frag is allowed to be empty but
924 // delim rule is not
925 if (rv)
926 {
927 u.has_frag = true;
928 u.frag = *rv;
929 }
930 }
931
932 return u;
933 }
934};

936constexpr pattern_rule_t pattern_rule{};
937
938system::result<pattern>
939parse_pattern(
940 core::string_view s)
941{
942 return grammar::parse(
943 s, r: pattern_rule);
944}
