22// Moves characters, and adjusts any passed
23// views if they point to any moved characters.
24
25// true if s completely overlapped by buf
26inline
27bool
28is_overlapping(
29 core::string_view buf,
30 core::string_view s) noexcept
31{
32 auto const b0 = buf.data();
33 auto const e0 = b0 + buf.size();
34 auto const b1 = s.data();
35 auto const e1 = b1 + s.size();
36 auto const less_equal =
37 std::less_equal<char const*>();
38 if(less_equal(e0, b1))
39 return false;
40 if(less_equal(e1, b0))
41 return false;
42 // partial overlap is undefined
43 BOOST_ASSERT(less_equal(e1, e0));
44 BOOST_ASSERT(less_equal(b0, b1));
45 return true;
46}
47
48inline
49void
50move_chars_impl(
51 std::ptrdiff_t,
52 core::string_view const&) noexcept
53{
54}
55
56template<class... Sn>
57void
58move_chars_impl(
59 std::ptrdiff_t d,
60 core::string_view const& buf,
61 core::string_view& s,
62 Sn&... sn) noexcept
63{
64 if(is_overlapping(buf, s))
65 s = {s.data() + d, s.size()};
66 move_chars_impl(d, buf, sn...);
67}
68
69template<class... Args>
70void
71move_chars(
72 char* dest,
73 char const* src,
74 std::size_t n,
75 Args&... args) noexcept
76{
77 core::string_view buf(src, n);
78 move_chars_impl(
79 dest - src,
80 core::string_view(src, n),
81 args...);
82 std::memmove(
83 dest: dest, src: src, n: n);
84}
