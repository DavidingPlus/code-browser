28void
29pop_encoded_front(
30 core::string_view& s,
31 char& c,
32 std::size_t& n) noexcept
33{
34 if(s.front() != '%')
35 {
36 c = s.front();
37 s.remove_prefix(n: 1);
38 }
39 else
40 {
41 detail::decode_unsafe(
42 dest: &c,
43 end: &c + 1,
44 s: s.substr(pos: 0, n: 3));
45 s.remove_prefix(n: 3);
46 }
47 ++n;
48}

50int
51compare_encoded(
52 core::string_view lhs,
53 core::string_view rhs) noexcept
54{
55 std::size_t n0 = 0;
56 std::size_t n1 = 0;
57 char c0 = 0;
58 char c1 = 0;
59 while(
60 !lhs.empty() &&
61 !rhs.empty())
62 {
63 pop_encoded_front(s&: lhs, c&: c0, n&: n0);
64 pop_encoded_front(s&: rhs, c&: c1, n&: n1);
65 if (c0 < c1)
66 return -1;
67 if (c1 < c0)
68 return 1;
69 }
70 n0 += detail::decode_bytes_unsafe(s: lhs);
71 n1 += detail::decode_bytes_unsafe(s: rhs);
72 if (n0 == n1)
73 return 0;
74 if (n0 < n1)
75 return -1;
76 return 1;
77}

139int
140compare(
141 core::string_view lhs,
142 core::string_view rhs) noexcept
143{
144 auto rlen = (std::min)(a: lhs.size(), b: rhs.size());
145 for (std::size_t i = 0; i < rlen; ++i)
146 {
147 char c0 = lhs[i];
148 char c1 = rhs[i];
149 if (c0 < c1)
150 return -1;
151 if (c1 < c0)
152 return 1;
153 }
154 if ( lhs.size() == rhs.size() )
155 return 0;
156 if ( lhs.size() < rhs.size() )
157 return -1;
158 return 1;
159}

195std::size_t
196path_starts_with(
197 core::string_view lhs,
198 core::string_view rhs) noexcept
199{
200 auto consume_one = [](
201 core::string_view::iterator& it,
202 char &c)
203 {
204 if(*it != '%')
205 {
206 c = *it;
207 ++it;
208 return;
209 }
210 detail::decode_unsafe(
211 dest: &c,
212 end: &c + 1,
213 s: core::string_view(it, 3));
214 if (c != '/')
215 {
216 it += 3;
217 return;
218 }
219 c = *it;
220 ++it;
221 };
222
223 auto it0 = lhs.begin();
224 auto it1 = rhs.begin();
225 auto end0 = lhs.end();
226 auto end1 = rhs.end();
227 char c0 = 0;
228 char c1 = 0;
229 while (
230 it0 < end0 &&
231 it1 < end1)
232 {
233 consume_one(it0, c0);
234 consume_one(it1, c1);
235 if (c0 != c1)
236 return 0;
237 }
238 if (it1 == end1)
239 return it0 - lhs.begin();
240 return 0;
241}

243std::size_t
244path_ends_with(
245 core::string_view lhs,
246 core::string_view rhs) noexcept
247{
248 auto consume_last = [](
249 core::string_view::iterator& it,
250 core::string_view::iterator& end,
251 char& c)
252 {
253 if ((end - it) < 3 ||
254 *(std::prev(x: end, n: 3)) != '%')
255 {
256 c = *--end;
257 return;
258 }
259 detail::decode_unsafe(
260 dest: &c,
261 end: &c + 1,
262 s: core::string_view(std::prev(
263 x: end, n: 3), 3));
264 if (c != '/')
265 {
266 end -= 3;
267 return;
268 }
269 c = *--end;
270 };
271
272 auto it0 = lhs.begin();
273 auto it1 = rhs.begin();
274 auto end0 = lhs.end();
275 auto end1 = rhs.end();
276 char c0 = 0;
277 char c1 = 0;
278 while(
279 it0 < end0 &&
280 it1 < end1)
281 {
282 consume_last(it0, end0, c0);
283 consume_last(it1, end1, c1);
284 if (c0 != c1)
285 return 0;
286 }
287 if (it1 == end1)
288 return lhs.end() - end0;
289 return 0;
290}

292std::size_t
293remove_dot_segments(
294 char* dest0,
295 char const* end,
296 core::string_view input) noexcept
297{
298 // 1. The input buffer `s` is initialized with
299 // the now-appended path components and the
300 // output buffer `dest0` is initialized to
301 // the empty string.
302 char* dest = dest0;
303 bool const is_absolute = input.starts_with(x: '/');
304
305 // Step 2 is a loop through 5 production rules:
306 // https://www.rfc-editor.org/rfc/rfc3986#section-5.2.4
307 //
308 // There are no transitions between all rules,
309 // which enables some optimizations.
310 //
311 // Initial:
312 // - Rule A: handle initial dots
313 // If the input buffer begins with a
314 // prefix of "../" or "./", then remove
315 // that prefix from the input buffer.
316 // Rule A can only happen at the beginning.
317 // Errata 4547: Keep "../" in the beginning
318 // https://www.rfc-editor.org/errata/eid4547
319 //
320 // Then:
321 // - Rule D: ignore a final ".." or "."
322 // if the input buffer consists only of "."
323 // or "..", then remove that from the input
324 // buffer.
325 // Rule D can only happen after Rule A because:
326 // - B and C write "/" to the input
327 // - E writes "/" to input or returns
328 //
329 // Then:
330 // - Rule B: ignore ".": write "/" to the input
331 // - Rule C: apply "..": remove seg and write "/"
332 // - Rule E: copy complete segment
333 auto append =
334 [](char*& first, char const* last, core::string_view in)
335 {
336 // append `in` to `dest`
337 BOOST_ASSERT(in.size() <= std::size_t(last - first));
338 std::memmove(dest: first, src: in.data(), n: in.size());
339 first += in.size();
340 ignore_unused(last);
341 };
342
343 auto dot_starts_with = [](
344 core::string_view str, core::string_view dots, std::size_t& n)
345 {
346 // starts_with for encoded/decoded dots
347 // or decoded otherwise. return how many
348 // chars in str match the dots
349 n = 0;
350 for (char c: dots)
351 {
352 if (str.starts_with(x: c))
353 {
354 str.remove_prefix(n: 1);
355 ++n;
356 }
357 else if (str.size() > 2 &&
358 str[0] == '%' &&
359 str[1] == '2' &&
360 (str[2] == 'e' ||
361 str[2] == 'E'))
362 {
363 str.remove_prefix(n: 3);
364 n += 3;
365 }
366 else
367 {
368 n = 0;
369 return false;
370 }
371 }
372 return true;
373 };
374
375 auto dot_equal = [&dot_starts_with](
376 core::string_view str, core::string_view dots)
377 {
378 std::size_t n = 0;
379 dot_starts_with(str, dots, n);
380 return n == str.size();
381 };
382
383 // Rule A
384 std::size_t n;
385 while (!input.empty())
386 {
387 if (dot_starts_with(input, "../", n))
388 {
389 // Errata 4547
390 append(dest, end, "../");
391 input.remove_prefix(n);
392 continue;
393 }
394 else if (!dot_starts_with(input, "./", n))
395 {
396 break;
397 }
398 input.remove_prefix(n);
399 }
400
401 // Rule D
402 if( dot_equal(input, "."))
403 {
404 input = {};
405 }
406 else if( dot_equal(input, "..") )
407 {
408 // Errata 4547
409 append(dest, end, "..");
410 input = {};
411 }
412
413 // 2. While the input buffer is not empty,
414 // loop as follows:
415 while (!input.empty())
416 {
417 // Rule B
418 bool const is_dot_seg = dot_starts_with(input, "/./", n);
419 if (is_dot_seg)
420 {
421 input.remove_prefix(n: n - 1);
422 continue;
423 }
424
425 bool const is_final_dot_seg = dot_equal(input, "/.");
426 if (is_final_dot_seg)
427 {
428 // We can't remove "." from a core::string_view
429 // So what we do here is equivalent to
430 // replacing s with '/' as required
431 // in Rule B and executing the next
432 // iteration, which would append this
433 // '/' to the output, as required by
434 // Rule E
435 append(dest, end, input.substr(pos: 0, n: 1));
436 input = {};
437 break;
438 }
439
440 // Rule C
441 bool const is_dotdot_seg = dot_starts_with(input, "/../", n);
442 if (is_dotdot_seg)
443 {
444 core::string_view cur_out(dest0, dest - dest0);
445 std::size_t p = cur_out.find_last_of(c: '/');
446 bool const has_multiple_segs = p != core::string_view::npos;
447 if (has_multiple_segs)
448 {
449 // output has multiple segments
450 // "erase" [p, end] if not "/.."
451 core::string_view last_seg(dest0 + p, dest - (dest0 + p));
452 bool const prev_is_dotdot_seg = dot_equal(last_seg, "/..");
453 if (!prev_is_dotdot_seg)
454 {
455 dest = dest0 + p;
456 }
457 else
458 {
459 append(dest, end, "/..");
460 }
461 }
462 else if (dest0 != dest)
463 {
464 // Only one segment in the output: remove it
465 core::string_view last_seg(dest0, dest - dest0);
466 bool const prev_is_dotdot_seg = dot_equal(last_seg, "..");
467 if (!prev_is_dotdot_seg)
468 {
469 dest = dest0;
470 if (!is_absolute)
471 {
472 input.remove_prefix(n: 1);
473 }
474 }
475 else
476 {
477 append(dest, end, "/..");
478 }
479 }
480 else
481 {
482 // Output is empty
483 if (is_absolute)
484 {
485 append(dest, end, "/..");
486 }
487 else
488 {
489 append(dest, end, "..");
490 }
491 }
492 input.remove_prefix(n: n - 1);
493 continue;
494 }
495
496 bool const is_final_dotdot_seg = dot_equal(input, "/..");
497 if (is_final_dotdot_seg)
498 {
499 core::string_view cur_out(dest0, dest - dest0);
500 std::size_t p = cur_out.find_last_of(c: '/');
501 bool const has_multiple_segs = p != core::string_view::npos;
502 if (has_multiple_segs)
503 {
504 // output has multiple segments
505 // "erase" [p, end] if not "/.."
506 core::string_view last_seg(dest0 + p, dest - (dest0 + p));
507 bool const prev_is_dotdot_seg = dot_equal(last_seg, "/..");
508 if (!prev_is_dotdot_seg)
509 {
510 dest = dest0 + p;
511 append(dest, end, "/");
512 }
513 else
514 {
515 append(dest, end, "/..");
516 }
517 }
518 else if (dest0 != dest)
519 {
520 // Only one segment in the output: remove it
521 core::string_view last_seg(dest0, dest - dest0);
522 bool const prev_is_dotdot_seg = dot_equal(last_seg, "..");
523 if (!prev_is_dotdot_seg) {
524 dest = dest0;
525 }
526 else
527 {
528 append(dest, end, "/..");
529 }
530 }
531 else
532 {
533 // Output is empty: append dotdot
534 if (is_absolute)
535 {
536 append(dest, end, "/..");
537 }
538 else
539 {
540 append(dest, end, "..");
541 }
542 }
543 input = {};
544 break;
545 }
546
547 // Rule E
548 std::size_t p = input.find_first_of(c: '/', pos: 1);
549 if (p != core::string_view::npos)
550 {
551 append(dest, end, input.substr(pos: 0, n: p));
552 input.remove_prefix(n: p);
553 }
554 else
555 {
556 append(dest, end, input);
557 input = {};
558 }
559 }
560
561 // 3. Finally, the output buffer is set
562 // as the result of remove_dot_segments,
563 // and we return its size
564 return dest - dest0;
565}

567char
568path_pop_back( core::string_view& s )
569{
570 if (s.size() < 3 ||
571 *std::prev(x: s.end(), n: 3) != '%')
572 {
573 char c = s.back();
574 s.remove_suffix(n: 1);
575 return c;
576 }
577 char c = 0;
578 detail::decode_unsafe(
579 dest: &c, end: &c + 1, s: s.substr(pos: s.size() - 3));
580 if (c != '/')
581 {
582 s.remove_suffix(n: 3);
583 return c;
584 }
585 c = s.back();
586 s.remove_suffix(n: 1);
587 return c;
588};

590void
591pop_last_segment(
592 core::string_view& s,
593 core::string_view& c,
594 std::size_t& level,
595 bool r) noexcept
596{
597 c = {};
598 std::size_t n = 0;
599 while (!s.empty())
600 {
601 // B. if the input buffer begins with a
602 // prefix of "/./" or "/.", where "." is
603 // a complete path segment, then replace
604 // that prefix with "/" in the input
605 // buffer; otherwise,
606 n = detail::path_ends_with(lhs: s, rhs: "/./");
607 if (n)
608 {
609 c = s.substr(pos: s.size() - n);
610 s.remove_suffix(n);
611 continue;
612 }
613 n = detail::path_ends_with(lhs: s, rhs: "/.");
614 if (n)
615 {
616 c = s.substr(pos: s.size() - n, n: 1);
617 s.remove_suffix(n);
618 continue;
619 }
620
621 // C. if the input buffer begins with a
622 // prefix of "/../" or "/..", where ".."
623 // is a complete path segment, then
624 // replace that prefix with "/" in the
625 // input buffer and remove the last
626 // segment and its preceding "/"
627 // (if any) from the output buffer
628 // otherwise,
629 n = detail::path_ends_with(lhs: s, rhs: "/../");
630 if (n)
631 {
632 c = s.substr(pos: s.size() - n);
633 s.remove_suffix(n);
634 ++level;
635 continue;
636 }
637 n = detail::path_ends_with(lhs: s, rhs: "/..");
638 if (n)
639 {
640 c = s.substr(pos: s.size() - n);
641 s.remove_suffix(n);
642 ++level;
643 continue;
644 }
645
646 // E. move the first path segment in the
647 // input buffer to the end of the output
648 // buffer, including the initial "/"
649 // character (if any) and any subsequent
650 // characters up to, but not including,
651 // the next "/" character or the end of
652 // the input buffer.
653 std::size_t p = s.size() > 1
654 ? s.find_last_of(c: '/', pos: s.size() - 2)
655 : core::string_view::npos;
656 if (p != core::string_view::npos)
657 {
658 c = s.substr(pos: p + 1);
659 s.remove_suffix(n: c.size());
660 }
661 else
662 {
663 c = s;
664 s = {};
665 }
666
667 if (level == 0)
668 return;
669 if (!s.empty())
670 --level;
671 }
672 // we still need to skip n_skip + 1
673 // but the string is empty
674 if (r && level)
675 {
676 c = "/";
677 level = 0;
678 return;
679 }
680 else if (level)
681 {
682 if (c.empty())
683 c = "/..";
684 else
685 c = "/../";
686 --level;
687 return;
688 }
689 c = {};
690}

692void
693normalized_path_digest(
694 core::string_view s,
695 bool remove_unmatched,
696 fnv_1a& hasher) noexcept
697{
698 core::string_view child;
699 std::size_t level = 0;
700 do
701 {
702 pop_last_segment(
703 s, c&: child, level, r: remove_unmatched);
704 while (!child.empty())
705 {
706 char c = path_pop_back(s&: child);
707 hasher.put(c);
708 }
709 }
710 while (!s.empty());
711}

713// compare segments as if there were a normalized
714int
715segments_compare(
716 segments_encoded_view seg0,
717 segments_encoded_view seg1) noexcept
718{
719 // calculate path size as if it were normalized
720 auto normalized_size =
721 [](segments_encoded_view seg) -> std::size_t
722 {
723 if (seg.empty())
724 return seg.is_absolute();
725
726 std::size_t n = 0;
727 std::size_t skip = 0;
728 auto begin = seg.begin();
729 auto it = seg.end();
730 while (it != begin)
731 {
732 --it;
733 decode_view dseg = **it;
734 if (dseg == "..")
735 ++skip;
736 else if (dseg != ".")
737 {
738 if (skip)
739 --skip;
740 else
741 n += dseg.size() + 1;
742 }
743 }
744 n += skip * 3;
745 n -= !seg.is_absolute();
746 return n;
747 };
748
749 // find the normalized size for the comparison
750 std::size_t n0 = normalized_size(seg0);
751 std::size_t n1 = normalized_size(seg1);
752 std::size_t n00 = n0;
753 std::size_t n10 = n1;
754
755 // consume the last char from a segment range
756 auto consume_last =
757 [](
758 std::size_t& n,
759 decode_view& dseg,
760 segments_encoded_view::iterator& begin,
761 segments_encoded_view::iterator& it,
762 decode_view::iterator& cit,
763 std::size_t& skip,
764 bool& at_slash) -> char
765 {
766 if (cit != dseg.begin())
767 {
768 // return last char from current segment
769 at_slash = false;
770 --cit;
771 --n;
772 return *cit;
773 }
774
775 if (!at_slash)
776 {
777 // current segment dseg is over and
778 // previous char was not a slash
779 // so we output one
780 at_slash = true;
781 --n;
782 return '/';
783 }
784
785 // current segment dseg is over and
786 // last char was already the slash
787 // between segments, so take the
788 // next final segment to consume
789 at_slash = false;
790 while (cit == dseg.begin())
791 {
792 // take next segment
793 if (it != begin)
794 --it;
795 else
796 break;
797 if (**it == "..")
798 {
799 // skip next if this is ".."
800 ++skip;
801 }
802 else if (**it != ".")
803 {
804 if (skip)
805 {
806 // discount skips
807 --skip;
808 }
809 else
810 {
811 // or update current seg
812 dseg = **it;
813 cit = dseg.end();
814 break;
815 }
816 }
817 }
818 // consume from the new current
819 // segment
820 --n;
821 if (cit != dseg.begin())
822 {
823 // in the general case, we consume
824 // one more character from the end
825 --cit;
826 return *cit;
827 }
828
829 // nothing left to consume in the
830 // current and new segment
831 if (it == begin)
832 {
833 // if this is the first
834 // segment, the segments are
835 // over and there can only
836 // be repetitions of "../" to
837 // output
838 return "/.."[n % 3];
839 }
840 // at other segments, we need
841 // a slash to transition to the
842 // next segment
843 at_slash = true;
844 return '/';
845 };
846
847 // consume final segments from seg0 that
848 // should not influence the comparison
849 auto begin0 = seg0.begin();
850 auto it0 = seg0.end();
851 decode_view dseg0;
852 if (it0 != seg0.begin())
853 {
854 --it0;
855 dseg0 = **it0;
856 }
857 decode_view::iterator cit0 = dseg0.end();
858 std::size_t skip0 = 0;
859 bool at_slash0 = true;
860 while (n0 > n1)
861 {
862 consume_last(n0, dseg0, begin0, it0, cit0, skip0, at_slash0);
863 }
864
865 // consume final segments from seg1 that
866 // should not influence the comparison
867 auto begin1 = seg1.begin();
868 auto it1 = seg1.end();
869 decode_view dseg1;
870 if (it1 != seg1.begin())
871 {
872 --it1;
873 dseg1 = **it1;
874 }
875 decode_view::iterator cit1 = dseg1.end();
876 std::size_t skip1 = 0;
877 bool at_slash1 = true;
878 while (n1 > n0)
879 {
