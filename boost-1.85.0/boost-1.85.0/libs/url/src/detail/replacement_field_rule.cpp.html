29auto
30replacement_field_rule_t::
31parse(
32 char const*& it,
33 char const* const end) const noexcept ->
34 system::result<value_type>
35{
36 static constexpr auto replacement_field_rules =
37 grammar::tuple_rule(
38 // open
39 r0: grammar::squelch(
40 r: grammar::delim_rule(ch: '{')),
41 // id
42 rn: grammar::optional_rule(r: arg_id_rule),
43 // format options
44 rn: grammar::optional_rule(
45 r: grammar::tuple_rule(
46 r0: grammar::squelch(
47 r: grammar::delim_rule(ch: ':')),
48 rn: format_spec_rule)),
49 // close
50 rn: grammar::squelch(
51 r: grammar::delim_rule(ch: '}')));
52 auto it0 = it;
53 auto rv = grammar::parse(it, end, r: replacement_field_rules);
54 if (!rv)
55 {
56 BOOST_URL_RETURN_EC(
57 grammar::error::mismatch);
58 }
59 return core::string_view(it0, it);
60}

62auto
63identifier_rule_t::
64parse(
65 char const*& it,
66 char const* const end) const noexcept
67 -> system::result<value_type>
68{
69 static constexpr auto identifier_rules =
70 grammar::tuple_rule(
71 r0: grammar::delim_rule(
72 cs: grammar::alpha_chars +
73 grammar::lut_chars('_')),
74 rn: grammar::optional_rule(
75 r: grammar::token_rule(
76 cs: grammar::alnum_chars +
77 grammar::lut_chars('_'))));
78 char const* it0 = it;
79 auto rv = grammar::parse(it, end, r: identifier_rules);
80 if (!rv)
81 {
82 BOOST_URL_RETURN_EC(
83 grammar::error::mismatch);
84 }
85 return core::string_view(it0, it);
86}
87
88auto
89format_spec_rule_t::
90parse(
91 char const*& it,
92 char const* const end) const noexcept
93 -> system::result<value_type>
94{
95 if (it == end)
96 return {};
97
98 // any tokens allowed in fmt specs
99 static constexpr auto fmt_specs_token_rule =
100 grammar::optional_rule(
101 r: grammar::token_rule(
102 cs: grammar::vchars +
103 grammar::lut_chars(' ')
104 - "{}"));
105
106 // internal ids in the fmt specs
107 // "{" [arg_id] "}"
108 static constexpr auto internal_id_rule =
109 grammar::tuple_rule(
110 r0: grammar::squelch(
111 r: grammar::delim_rule(ch: '{')),
112 rn: grammar::optional_rule(
113 r: arg_id_rule),
114 rn: grammar::squelch(
115 r: grammar::delim_rule(ch: '}')));
116
117 auto start = it;
118 // consume fmt_spec chars
119 while (grammar::parse(it, end, r: fmt_specs_token_rule))
120 {
121 auto it0 = it;
122 // consume internal id
123 if (!grammar::parse(it, end, r: internal_id_rule))
124 {
125 // rewind
126 it = it0;
127 break;
128 }
129 }
130
131 return core::string_view(start, it);
132}
