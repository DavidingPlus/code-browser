24void
25throw_system_error(
26 system::error_code const& ec,
27 source_location const& loc)
28{
29 throw_exception(
30 e: boost::system::system_error(ec), loc);
31}
32
33void
34throw_errc(
35 boost::system::errc::errc_t ev,
36 source_location const& loc)
37{
38 throw_system_error(ec: make_error_code(e: ev), loc);
39}
40
41void
42throw_invalid_argument(
43 source_location const& loc)
44{
45 throw_errc(ev: boost::system::errc::invalid_argument, loc);
46}
47
48void
49throw_length_error(
50 source_location const& loc)
51{
52 throw_errc(ev: boost::system::errc::value_too_large, loc);
53}
