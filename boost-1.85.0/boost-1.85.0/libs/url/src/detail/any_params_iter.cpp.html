23/*
24 When a string is transformed into a range of
25 params, the empty string becomes ambiguous:
26 it can be an empty range, or a range with
27 one param. The value `not_empty` is used on
28 construction to inform the transformation
29 that the empty string should be treated as
30 a one-element range. This simplifies
31 edit_params().
32*/

49query_iter::
50query_iter(
51 core::string_view s,
52 bool ne) noexcept
53 : any_params_iter(
54 s.empty() && ! ne, s)
55{
56 rewind();
57}
58
59void
60query_iter::
61rewind() noexcept
62{
63 if(empty)
64 {
65 at_end_ = true;
66 return;
67 }
68 p_ = s0.begin();
69 if(! s0.empty())
70 {
71 auto pos =
72 s0.find_first_of(c: '&');
73 if(pos != core::string_view::npos)
74 n_ = pos;
75 else
76 n_ = s0.size();
77 }
78 else
79 {
80 n_ = 0;
81 }
82 at_end_ = false;
83}
84
85bool
86query_iter::
87measure(
88 std::size_t& n) noexcept
89{
90 if(at_end_)
91 return false;
92 // When interacting with the query as
93 // an intact string, we do not treat
94 // the plus sign as an encoded space.
95 encoding_opts opt;
96 opt.space_as_plus = false;
97 n += encoded_size(
98 s: core::string_view(p_, n_),
99 unreserved: query_chars,
100 opt);
101 increment();
102 return true;
103}
104
105void
106query_iter::
107copy(
108 char*& dest,
109 char const* end) noexcept
110{
111 BOOST_ASSERT(! at_end_);
112 // When interacting with the query as
113 // an intact string, we do not treat
114 // the plus sign as an encoded space.
115 encoding_opts opt;
116 opt.space_as_plus = false;
117 dest += encode_unsafe(
118 dest,
119 size: end - dest,
120 s: core::string_view(p_, n_),
121 unreserved: query_chars,
122 opt);
123 increment();
124}
125
126void
127query_iter::
128increment() noexcept
129{
130 p_ += n_;
131 if(p_ == s0.end())
132 {
133 at_end_ = true;
134 return;
135 }
136 ++p_;
137 core::string_view s(p_, s0.end() - p_);
138 auto pos = s.find_first_of(c: '&');
139 if(pos != core::string_view::npos)
140 n_ = pos;
141 else
142 n_ = s.size();
143}

169bool
170param_iter::
171measure(std::size_t& n) noexcept
172{
173 if(at_end_)
174 return false;
175 encoding_opts opt;
176 opt.space_as_plus = false;
177 n += encoded_size(
178 s: s0,
179 unreserved: detail::param_key_chars,
180 opt);
181 if(has_value_)
182 {
183 ++n; // '='
184 n += encoded_size(
185 s: s1,
186 unreserved: detail::param_value_chars,
187 opt);
188 }
189 at_end_ = true;
190 return true;
191}

198{

202 dest += encode(
203 dest,

205 s: s0,
206 unreserved: detail::param_key_chars,
207 opt);
208 if(has_value_)
209 {
210 *dest++ = '=';
211 dest += encode(
212 dest,
213 size: end - dest,
214 s: s1,
215 unreserved: detail::param_value_chars,
216 opt);
217 }
