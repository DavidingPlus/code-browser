23void
24pct_string_view::
25decode_impl(
26 string_token::arg& dest,
27 encoding_opts opt) const
28{
29 auto p = dest.prepare(n: dn_);
30 if(dn_ > 0)
31 detail::decode_unsafe(
32 dest: p, end: p + dn_, s: s_, opt);
33}

48system::result<pct_string_view>
49make_pct_string_view(
50 core::string_view s) noexcept
51{
52 auto p = s.begin();
53 auto const end = s.end();
54 std::size_t dn = 0;
55 if(s.size() >= 3)
56 {
57 auto const safe_end = end - 2;
58 while(p < safe_end)
59 {
60 if(*p != '%')
61 {
62 ++p;
63 }
64 else if(
65 grammar::hexdig_value(ch: p[1]) >= 0 &&
66 grammar::hexdig_value(ch: p[2]) >= 0)
67 {
68 // percent-escape
69 p += 3;
70 }
71 else
72 {
73 // invalid encoding
74 BOOST_URL_RETURN_EC(
75 error::bad_pct_hexdig);
76 }
77 ++dn;
78 }
79 }
80 auto const n = end - p;
81 if( (n >= 1 && p[0] == '%') ||
82 (n >= 2 && p[1] == '%'))
83 {
84 // invalid encoding
85 BOOST_URL_RETURN_EC(
86 error::incomplete_encoding);
87 }
88 dn += n;
89 return make_pct_string_view_unsafe(
90 data: s.data(), size: s.size(), decoded_size: dn);
91}
