43std::size_t
44url_view_base::
45digest(std::size_t salt) const noexcept
46{
47 detail::fnv_1a h(salt);
48 detail::ci_digest(s: pi_->get(id_scheme), hasher&: h);
49 detail::digest_encoded(s: pi_->get(id_user), hasher&: h);
50 detail::digest_encoded(s: pi_->get(id_pass), hasher&: h);
51 detail::ci_digest_encoded(s: pi_->get(id_host), hasher&: h);
52 h.put(s: pi_->get(id_port));
53 detail::normalized_path_digest(
54 s: pi_->get(id_path), remove_unmatched: is_path_absolute(), hasher&: h);
55 detail::digest_encoded(s: pi_->get(id_query), hasher&: h);
56 detail::digest_encoded(s: pi_->get(id_frag), hasher&: h);
57 return h.digest();
58}

83std::shared_ptr<url_view const>
84url_view_base::
85persist() const
86{
87 using T = shared_impl;
88 using Alloc = std::allocator<char>;
89 Alloc a;
90 auto p = std::allocate_shared<T>(
91 a: detail::over_allocator<T, Alloc>(
92 size(), a), args: url_view(*pi_));
93 std::memcpy(
94 dest: reinterpret_cast<char*>(
95 p.get() + 1), src: data(), n: size());
96 return p;
97}

147authority_view
148url_view_base::
149authority() const noexcept
150{
151 detail::url_impl u(from::authority);
152 u.cs_ = encoded_authority().data();
153 if(has_authority())
154 {
155 u.set_size(id_user, pi_->len(id_user) - 2);
156 u.set_size(id_pass, pi_->len(id_pass));
157 u.set_size(id_host, pi_->len(id_host));
158 u.set_size(id_port, pi_->len(id_port));
159 }
160 else
161 {
162 u.set_size(id_user, pi_->len(id_user));
163 BOOST_ASSERT(pi_->len(id_pass) == 0);
164 BOOST_ASSERT(pi_->len(id_host) == 0);
165 BOOST_ASSERT(pi_->len(id_port) == 0);
166 }
167 u.decoded_[id_user] = pi_->decoded_[id_user];
168 u.decoded_[id_pass] = pi_->decoded_[id_pass];
169 u.decoded_[id_host] = pi_->decoded_[id_host];
170 for (int i = 0; i < 16; ++i)
171 u.ip_addr_[i] = pi_->ip_addr_[i];
172 u.port_number_ = pi_->port_number_;
173 u.host_type_ = pi_->host_type_;
174 return u.construct_authority();
175}

177pct_string_view
178url_view_base::
179encoded_authority() const noexcept
180{
181 auto s = pi_->get(id_user, id_path);
182 if(! s.empty())
183 {
184 BOOST_ASSERT(has_authority());
185 s.remove_prefix(n: 2);
186 }
187 return make_pct_string_view_unsafe(
188 data: s.data(),
189 size: s.size(),
190 decoded_size: pi_->decoded_[id_user] +
191 pi_->decoded_[id_pass] +
192 pi_->decoded_[id_host] +
193 pi_->decoded_[id_port] +
194 has_password());
195}

275pct_string_view
276url_view_base::
277encoded_password() const noexcept
278{
279 auto s = pi_->get(id_pass);
280 switch(s.size())
281 {
282 case 1:
283 BOOST_ASSERT(
284 s.starts_with('@'));
285 s.remove_prefix(n: 1);
286 BOOST_FALLTHROUGH;
287 case 0:
288 return make_pct_string_view_unsafe(
289 data: s.data(), size: s.size(), decoded_size: 0);
290 default:
291 break;
292 }
293 BOOST_ASSERT(s.ends_with('@'));
294 BOOST_ASSERT(s.starts_with(':'));
295 return make_pct_string_view_unsafe(
296 data: s.data() + 1,
297 size: s.size() - 2,
298 decoded_size: pi_->decoded_[id_pass]);
299}

328pct_string_view
329url_view_base::
330encoded_host_address() const noexcept
331{
332 core::string_view s = pi_->get(id_host);
333 std::size_t n;
334 switch(pi_->host_type_)
335 {
336 default:
337 case urls::host_type::none:
338 BOOST_ASSERT(s.empty());
339 n = 0;
340 break;
341
342 case urls::host_type::name:
343 case urls::host_type::ipv4:
344 n = pi_->decoded_[id_host];
345 break;
346
347 case urls::host_type::ipv6:
348 case urls::host_type::ipvfuture:
349 {
350 BOOST_ASSERT(
351 pi_->decoded_[id_host] ==
352 s.size());
353 BOOST_ASSERT(s.size() >= 2);
354 BOOST_ASSERT(s.front() == '[');
355 BOOST_ASSERT(s.back() == ']');
356 s = s.substr(pos: 1, n: s.size() - 2);
357 n = pi_->decoded_[id_host] - 2;
358 break;
359 }
360 }
361 return make_pct_string_view_unsafe(
362 data: s.data(),
363 size: s.size(),
364 decoded_size: n);
365}

422pct_string_view
423url_view_base::
424encoded_zone_id() const noexcept
425{
426 if(pi_->host_type_ !=
427 urls::host_type::ipv6)
428 return {};
429 core::string_view s = pi_->get(id_host);
430 BOOST_ASSERT(s.front() == '[');
431 BOOST_ASSERT(s.back() == ']');
432 s = s.substr(pos: 1, n: s.size() - 2);
433 auto pos = s.find(s: "%25");
434 if (pos == core::string_view::npos)
435 return {};
436 s.remove_prefix(n: pos + 3);
437 return *make_pct_string_view(s);
438}

654//------------------------------------------------
655//
656// Comparisons
657//
658//------------------------------------------------
659
660int
661url_view_base::
662compare(const url_view_base& other) const noexcept
663{
664 int comp =
665 static_cast<int>(has_scheme()) -
666 static_cast<int>(other.has_scheme());
667 if ( comp != 0 )
668 return comp;
669
670 if (has_scheme())
671 {
672 comp = detail::ci_compare(
673 lhs: scheme(),
674 rhs: other.scheme());
675 if ( comp != 0 )
676 return comp;
677 }
678
679 comp =
680 static_cast<int>(has_authority()) -
681 static_cast<int>(other.has_authority());
682 if ( comp != 0 )
683 return comp;
684
685 if (has_authority())
686 {
687 comp = authority().compare(other: other.authority());
688 if ( comp != 0 )
689 return comp;
690 }
691
692 comp = detail::segments_compare(
693 seg0: encoded_segments(),
694 seg1: other.encoded_segments());
695 if ( comp != 0 )
696 return comp;
697
698 comp =
699 static_cast<int>(has_query()) -
700 static_cast<int>(other.has_query());
701 if ( comp != 0 )
702 return comp;
703
704 if (has_query())
705 {
706 comp = detail::compare_encoded(
707 lhs: encoded_query(),
708 rhs: other.encoded_query());
709 if ( comp != 0 )
710 return comp;
711 }
712
713 comp =
714 static_cast<int>(has_fragment()) -
715 static_cast<int>(other.has_fragment());
716 if ( comp != 0 )
717 return comp;
718
719 if (has_fragment())
720 {
721 comp = detail::compare_encoded(
722 lhs: encoded_fragment(),
723 rhs: other.encoded_fragment());
724 if ( comp != 0 )
725 return comp;
726 }
727
728 return 0;
729}
