108std::size_t
109params_encoded_ref::
110erase(
111 pct_string_view key,
112 ignore_case_param ic) noexcept
113{
114 // end() can't be fully cached,
115 // since erase invalidates it.
116 iterator it;
117 {
118 auto const end_ = end();
119 it = find_last(it: end_, key, ic);
120 if(it == end_)
121 return 0;
122 }
123 std::size_t n = 0;
124 for(;;)
125 {
126 ++n;
127 // Use it->key instead of key,
128 // to handle self-intersection
129 auto prev = find_last(it, key: it->key, ic);
130 if(prev == end())
131 break;
132 erase(pos: it);
133 it = prev;
134 }
135 erase(pos: it);
136 return n;
137}

197auto
198params_encoded_ref::
199set(
200 pct_string_view key,
201 pct_string_view value,
202 ignore_case_param ic) ->
203 iterator
204{
205 // VFALCO we can't cache end() here
206 // because it is invalidated
207 // every time we set or erase.
208 auto it0 = find(key, ic);
209 if(it0 == end())
210 return append(p: {key, value});
211 it0 = set(pos: it0, value);
212 auto it = end();
213 for(;;)
214 {
215 it = find_last(it, key, ic);
216 if(it == it0)
217 return it0;
218 it = erase(pos: it);
219 }
220}

254detail::params_iter_impl
255params_encoded_ref::
256find_impl(
257 detail::params_iter_impl it,
258 pct_string_view key,
259 ignore_case_param ic) const noexcept
260{
261 detail::params_iter_impl end_(u_->impl_, 0);
262 if(! ic)
263 {
264 for(;;)
265 {
266 if(it.equal(other: end_))
267 return it;
268 if(*it.key() == *key)
269 return it;
270 it.increment();
271 }
272 }
273 for(;;)
274 {
275 if(it.equal(other: end_))
276 return it;
277 if( grammar::ci_is_equal(
278 s0: *it.key(), s1: *key))
279 return it;
280 it.increment();
281 }
282}
283
284detail::params_iter_impl
285params_encoded_ref::
286find_last_impl(
287 detail::params_iter_impl it,
288 pct_string_view key,
289 ignore_case_param ic) const noexcept
290{
291 detail::params_iter_impl begin_(u_->impl_);
292 if(! ic)
293 {
294 for(;;)
295 {
296 if(it.equal(other: begin_))
297 return { u_->impl_, 0 };
298 it.decrement();
299 if(*it.key() == *key)
300 return it;
301 }
302 }
303 for(;;)
304 {
305 if(it.equal(other: begin_))
306 return { u_->impl_, 0 };
307 it.decrement();
308 if(grammar::ci_is_equal(
309 s0: *it.key(), s1: *key))
310 return it;
311 }
312}
