97std::size_t
98params_encoded_base::
99count(
100 pct_string_view key,
101 ignore_case_param ic) const noexcept
102{
103 std::size_t n = 0;
104 auto it = find(key, ic);
105 auto const end_ = end();
106 while(it != end_)
107 {
108 ++n;
109 ++it;
110 it = find(it, key, ic);
111 }
112 return n;
113}

121detail::params_iter_impl
122params_encoded_base::
123find_impl(
124 detail::params_iter_impl it,
125 pct_string_view key,
126 ignore_case_param ic) const noexcept
127{
128 detail::params_iter_impl end_(ref_, 0);
129 if(! ic)
130 {
131 for(;;)
132 {
133 if(it.equal(other: end_))
134 return it;
135 if(*it.key() == *key)
136 return it;
137 it.increment();
138 }
139 }
140 for(;;)
141 {
142 if(it.equal(other: end_))
143 return it;
144 if( grammar::ci_is_equal(
145 s0: *it.key(), s1: *key))
146 return it;
147 it.increment();
148 }
149}

151detail::params_iter_impl
152params_encoded_base::
153find_last_impl(
154 detail::params_iter_impl it,
155 pct_string_view key,
156 ignore_case_param ic) const noexcept
157{
158 detail::params_iter_impl begin_(ref_);
159 if(! ic)
160 {
161 for(;;)
162 {
163 if(it.equal(other: begin_))
164 return { ref_, 0 };
165 it.decrement();
166 if(*it.key() == *key)
167 return it;
168 }
169 }
170 for(;;)
171 {
172 if(it.equal(other: begin_))
173 return { ref_, 0 };
174 it.decrement();
175 if(grammar::ci_is_equal(
176 s0: *it.key(), s1: *key))
177 return it;
178 }
179}
