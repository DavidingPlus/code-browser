21/** Return a reference to a parsed URL string
22
23 This function parses a string according
24 to the grammar below and returns a view
25 referencing the passed string upon success,
26 else returns an error.
27 Ownership of the string is not transferred;
28 the caller is responsible for ensuring that
29 the lifetime of the character buffer extends
30 until the view is no longer being accessed.
31
32 @par Example
33 @code
34 system::result< url_view > rv = parse_absolute_uri( "http://example.com/index.htm?id=1" );
35 @endcode
36
37 @par BNF
38 @code
39 absolute-URI = scheme ":" hier-part [ "?" query ]
40
41 hier-part = "//" authority path-abempty
42 / path-absolute
43 / path-rootless
44 / path-empty
45 @endcode
46
47 @throw std::length_error `s.size() > url_view::max_size`
48
49 @return A @ref result containing a value or an error
50
51 @param s The string to parse
52
53 @par Specification
54 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-4.3"
55 >4.3. Absolute URI (rfc3986)</a>
56
57 @see
58 @ref parse_origin_form,
59 @ref parse_relative_ref,
60 @ref parse_uri,
61 @ref parse_uri_reference,
62 @ref url_view.
63*/
64BOOST_URL_DECL
65system::result<url_view>
66parse_absolute_uri(
67 core::string_view s);
