25/** Common functionality for containers
26
27 This base class is used by the library
28 to provide common member functions for
29 containers. This cannot be instantiated
30 directly; Instead, use one of the
31 containers or functions:
32
33 @par Containers
34 @li @ref params_ref
35 @li @ref params_view
36 @li @ref params_encoded_ref
37 @li @ref params_encoded_view
38*/
39class BOOST_URL_DECL params_base
40{
41 friend class url_view_base;
42 friend class params_ref;
43 friend class params_view;
44
45 detail::query_ref ref_;
46 encoding_opts opt_;
47
48 params_base() noexcept;
49 params_base(
50 detail::query_ref const& ref,
51 encoding_opts opt) noexcept;
52 params_base(
53 params_base const&) = default;
54 params_base& operator=(
55 params_base const&) = default;
56
57public:
58 /** A Bidirectional iterator to a query parameter
59
60 Objects of this type allow iteration
61 through the parameters in the query.
62 Any percent-escapes in returned strings
63 are decoded first.
64 The values returned are read-only;
65 changes to parameters must be made
66 through the container instead, if the
67 container supports modification.
68
69 <br>
70
71 The strings produced when iterators are
72 dereferenced belong to the iterator and
73 become invalidated when that particular
74 iterator is incremented, decremented,
75 or destroyed.
76
77 @note
78
79 The implementation may use temporary,
80 recycled storage to store decoded
81 strings. These iterators are meant
82 to be used ephemerally. That is, for
83 short durations such as within a
84 function scope. Do not store
85 iterators with static storage
86 duration or as long-lived objects.
87 */
88#ifdef BOOST_URL_DOCS
89 using iterator = __see_below__;
90#else
91 class iterator;
92#endif
93
94 /// @copydoc iterator
95 using const_iterator = iterator;
96
97 /** The value type
98
99 Values of this type represent parameters
100 whose strings retain unique ownership by
101 making a copy.
102
103 @par Example
104 @code
105 params_view::value_type qp( *url_view( "?first=John&last=Doe" ).params().find( "first" ) );
106 @endcode
107
108 @see
109 @ref param.
110 */
111 using value_type = param;
112
113 /** The reference type
114
115 This is the type of value returned when
116 iterators of the view are dereferenced.
117
118 @see
119 @ref param_view.
120 */
121 using reference = param;
122
123 /// @copydoc reference
124 using const_reference = param;
125
126 /** An unsigned integer type to represent sizes.
127 */
128 using size_type = std::size_t;
129
130 /** A signed integer type used to represent differences.
131 */
132 using difference_type = std::ptrdiff_t;
133
134 //--------------------------------------------
135 //
136 // Observers
137 //
138 //--------------------------------------------
139
140 /** Return the maximum number of characters possible
141
142 This represents the largest number of
143 characters that are possible in a path,
144 not including any null terminator.
145
146 @par Exception Safety
147 Throws nothing.
148 */
149 static
150 constexpr
151 std::size_t
152 max_size() noexcept
153 {
154 return BOOST_URL_MAX_SIZE;
155 }
156
157 /** Return the referenced character buffer.
158
159 This function returns the character
160 buffer referenced by the view.
161 The returned string may contain
162 percent escapes.
163
164 @par Example
165 @code
166 assert( url_view( "?first=John&last=Doe" ).params().buffer() == "?first=John&last=Doe" );
167 @endcode
168
169 @par Complexity
170 Constant.
171
172 @par Exception Safety
173 Throws nothing.
174 */
175 pct_string_view
176 buffer() const noexcept;
177
178 /** Return true if there are no params
179
180 @par Example
181 @code
182 assert( ! url_view( "?key=value" ).params().empty() );
183 @endcode
184
185 @par Complexity
186 Constant.
187
188 @par Exception Safety
189 Throws nothing.
190 */
191 bool
192 empty() const noexcept;
193
194 /** Return the number of params
195
196 @par Example
197 @code
198 assert( url_view( "?key=value").params().size() == 1 );
199 @endcode
200
201 @par Complexity
202 Constant.
203
204 @par Exception Safety
205 Throws nothing.
206 */
207 std::size_t
208 size() const noexcept;
209
210 /** Return an iterator to the beginning
211
212 @par Complexity
213 Linear in the size of the first param.
214
215 @par Exception Safety
216 Throws nothing.
217 */
218 iterator
219 begin() const noexcept;
220
221 /** Return an iterator to the end
222
223 @par Complexity
224 Constant.
225
226 @par Exception Safety
227 Throws nothing.
228 */
229 iterator
230 end() const noexcept;
231
232 //--------------------------------------------
233
234 /** Return true if a matching key exists
235
236 This function examines the parameters
237 in the container to find a match for
238 the specified key.
239 The comparison is performed as if all
240 escaped characters were decoded first.
241
242 @par Example
243 @code
244 assert( url_view( "?first=John&last=Doe" ).params().contains( "first" ) );
245 @endcode
246
247 @par Complexity
248 Linear in `this->buffer().size()`.
249
250 @par Exception Safety
251 Throws nothing.
252
253 @param key The key to match.
254 By default, a case-sensitive
255 comparison is used.
256
257 @param ic An optional parameter. If
258 the value @ref ignore_case is passed
259 here, the comparison is
260 case-insensitive.
261 */
262 bool
263 contains(
264 core::string_view key,
265 ignore_case_param ic = {}) const noexcept;
266
267 /** Return the number of matching keys
268
269 This function examines the
270 parameters in the container to
271 find the number of matches for
272 the specified key.
273 The comparison is performed as if all
274 escaped characters were decoded first.
275
276 @par Example
277 @code
278 assert( url_view( "?first=John&last=Doe" ).params().count( "first" ) == 1 );
279 @endcode
280
281 @par Complexity
282 Linear in `this->buffer().size()`.
283
284 @par Exception Safety
285 Throws nothing.
286
287 @param key The key to match.
288 By default, a case-sensitive
289 comparison is used.
290
291 @param ic An optional parameter. If
292 the value @ref ignore_case is passed
293 here, the comparison is
294 case-insensitive.
295 */
296 std::size_t
297 count(
298 core::string_view key,
299 ignore_case_param ic = {}) const noexcept;
300
301 /** Find a matching key
302
303 This function examines the parameters
304 in the container to find a match for
305 the specified key.
306 The comparison is performed as if all
307 escaped characters were decoded first.
308
309 <br>
310
311 The search starts from the first param
312 and proceeds forward until either the
313 key is found or the end of the range is
314 reached, in which case `end()` is
315 returned.
316
317 @par Example
318 @code
319 assert( (*url_view( "?first=John&last=Doe" ).params().find( "First", ignore_case )).value == "John" );
320 @endcode
321
322 @par Effects
323 @code
324 return this->find( this->begin(), key, ic );
325 @endcode
326
327 @par Complexity
328 Linear in `this->buffer().size()`.
329
330 @return an iterator to the param
331
332 @param key The key to match.
333 By default, a case-sensitive
334 comparison is used.
335
336 @param ic An optional parameter. If
337 the value @ref ignore_case is passed
338 here, the comparison is
339 case-insensitive.
340 */
341 iterator
342 find(
343 core::string_view key,
344 ignore_case_param ic = {}) const noexcept;
345
346 /** Find a matching key
347
348 This function examines the
349 parameters in the container to
350 find a match for the specified key.
351 The comparison is performed as if all
352 escaped characters were decoded first.
353
354 <br>
355
356 The search starts at `from`
357 and proceeds forward until either the
358 key is found or the end of the range is
359 reached, in which case `end()` is
360 returned.
361
362 @par Example
363 @code
364 url_view u( "?First=John&Last=Doe" );
365
366 assert( u.params().find( "first" ) != u.params().find( "first", ignore_case ) );
367 @endcode
368
369 @par Complexity
370 Linear in `this->buffer().size()`.
371
372 @return an iterator to the param
373
374 @param from The position to begin the
375 search from. This can be `end()`.
376
377 @param key The key to match.
378 By default, a case-sensitive
379 comparison is used.
380
381 @param ic An optional parameter. If
382 the value @ref ignore_case is passed
383 here, the comparison is
384 case-insensitive.
385 */
386 iterator
387 find(
388 iterator from,
389 core::string_view key,
390 ignore_case_param ic = {}) const noexcept;
391
392 /** Find a matching key
393
394 This function examines the
395 parameters in the container to
396 find a match for the specified key.
397 The comparison is performed as if all
398 escaped characters were decoded first.
399
400 <br>
401
402 The search starts from the last param
403 and proceeds backwards until either the
404 key is found or the beginning of the
405 range is reached, in which case `end()`
406 is returned.
407
408 @par Example
409 @code
410 assert( (*url_view( "?first=John&last=Doe" ).params().find_last( "last" )).value == "Doe" );
411 @endcode
412
413 @par Complexity
414 Linear in `this->buffer().size()`.
415
416 @return an iterator to the param
417
418 @param key The key to match.
419 By default, a case-sensitive
420 comparison is used.
421
422 @param ic An optional parameter. If
423 the value @ref ignore_case is passed
424 here, the comparison is
425 case-insensitive.
426 */
427 iterator
428 find_last(
429 core::string_view key,
430 ignore_case_param ic = {}) const noexcept;
431
432 /** Find a matching key
433
434 This function examines the
435 parameters in the container to
436 find a match for the specified key.
437 The comparison is performed as if all
438 escaped characters were decoded first.
439
440 <br>
441
442 The search starts prior to `before`
443 and proceeds backwards until either the
444 key is found or the beginning of the
445 range is reached, in which case `end()`
446 is returned.
447
448 @par Example
449 @code
450 url_view u( "?First=John&Last=Doe" );
451
452 assert( u.params().find_last( "last" ) != u.params().find_last( "last", ignore_case ) );
453 @endcode
454
455 @par Complexity
456 Linear in `this->buffer().size()`.
457
458 @return an iterator to the param
459
460 @param before One past the position
461 to begin the search from. This can
462 be `end()`.
463
464 @param key The key to match.
465 By default, a case-sensitive
466 comparison is used.
467
468 @param ic An optional parameter. If
469 the value @ref ignore_case is passed
470 here, the comparison is
471 case-insensitive.
472 */
473 iterator
474 find_last(
475 iterator before,
476 core::string_view key,
477 ignore_case_param ic = {}) const noexcept;
478
479private:
480 detail::params_iter_impl
481 find_impl(

485
486 detail::params_iter_impl
487 find_last_impl(
488 detail::params_iter_impl,
489 core::string_view,
490 ignore_case_param) const noexcept;
491};
