25/** Identifies a known URL scheme
26
27 @par Specification
28 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.1"
29 >3.1. Scheme (rfc3986)</a>
30*/
31// Made this short so it doesn't
32// show up as an ascii character
33enum class scheme : unsigned short
34{
35 /** Indicates that no scheme is present
36 */
37 none = 0,
38
39 /** Indicates the scheme is not a well-known scheme
40 */
41 unknown,
42
43 /**
44 * File Transfer Protocol (FTP)
45
46 FTP is a standard communication protocol
47 used for the transfer of computer files
48 from a server to a client on a computer
49 network.
50
51 @par Specification
52 @li <a href="https://datatracker.ietf.org/doc/html/draft-yevstifeyev-ftp-uri-scheme">
53 The 'ftp' URI Scheme</a>
54 */
55 ftp,
56
57 /**
58 * File URI Scheme
59
60 The File URI Scheme is typically used
61 to retrieve files from within one's
62 own computer.
63
64 @par Specification
65 @li <a href="https://datatracker.ietf.org/doc/html/rfc8089">
66 The "file" URI Scheme (rfc8089)</a>
67 */
68 file,
69
70 /**
71 * The Hypertext Transfer Protocol URI Scheme
72
73 URLs of this type indicate a resource which
74 is interacted with using the HTTP protocol.
75
76 @par Specification
77 @li <a href="https://datatracker.ietf.org/doc/html/rfc7230">
78 Hypertext Transfer Protocol (HTTP/1.1) (rfc7230)</a>
79 */
80 http,
81
82 /**
83 * The Secure Hypertext Transfer Protocol URI Scheme
84
85 URLs of this type indicate a resource which
86 is interacted with using the Secure HTTP
87 protocol.
88
89 @par Specification
90 @li <a href="https://datatracker.ietf.org/doc/html/rfc7230">
91 Hypertext Transfer Protocol (HTTP/1.1) (rfc7230)</a>
92 */
93 https,
94
95 /**
96 * The WebSocket URI Scheme
97
98 URLs of this type indicate a resource which
99 is interacted with using the WebSocket protocol.
100
101 @par Specification
102 @li <a href="https://datatracker.ietf.org/doc/html/rfc6455">
103 The WebSocket Protocol (rfc6455)</a>
104 */
105 ws,
106
107 /**
108 * The Secure WebSocket URI Scheme
109
110 URLs of this type indicate a resource which
111 is interacted with using the Secure WebSocket
112 protocol.
113
114 @par Specification
115 @li <a href="https://datatracker.ietf.org/doc/html/rfc6455">
116 The WebSocket Protocol (rfc6455)</a>
117 */
118 wss
119};
120
121/** Return the known scheme for a non-normalized string, if known
122
123 If the string does not identify a known
124 scheme, the value @ref scheme::unknown is
125 returned.
126
127 @par BNF
128 @code
129 scheme = ALPHA *( ALPHA / DIGIT / "+" / "-" / "." )
130 @endcode
131
132 @return The known scheme
133
134 @param s The string holding the scheme
135
136 @par Specification
137 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.1"
138 >3.1. Scheme (rfc3986)</a>
139*/
140BOOST_URL_DECL
141scheme
142string_to_scheme(core::string_view s) noexcept;
143
144/** Return the normalized string for a known scheme
145
146 @return A string representing the known scheme
147
148 @param s The known scheme constant
149*/
150BOOST_URL_DECL
151core::string_view
152to_string(scheme s) noexcept;
153
154/** Return the default port for a known scheme
155
156 This function returns the default port
157 for the known schemes. If the value does
158 not represent a known scheme or the scheme
159 does not represent a protocol, the function
160 returns zero.
161
162 The following ports are returned by the
163 function:
164
165 @li @ref scheme::ftp = 21
166 @li @ref scheme::http, @ref scheme::ws = 80
167 @li @ref scheme::https, @ref scheme::wss = 443
168
169 @return An integer with the default port number
170
171 @param s The known scheme constant
172*/
173BOOST_URL_DECL
174std::uint16_t
175default_port(scheme s) noexcept;
176
177} // urls
178} // boost
