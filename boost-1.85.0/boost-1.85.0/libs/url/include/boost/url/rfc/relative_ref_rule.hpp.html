20/** Rule for relative-ref
21
22 @par Value Type
23 @code
24 using value_type = url_view;
25 @endcode
26
27 @par Example
28 Rules are used with the function @ref grammar::parse.
29 @code
30 system::result< url_view > rv = grammar::parse( "images/dot.gif?v=hide#a", relative_ref_rule );
31 @endcode
32
33 @par BNF
34 @code
35 relative-ref = relative-part [ "?" query ] [ "#" fragment ]
36 @endcode
37
38 @par Specification
39 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-4.2"
40 >4.2. Relative Reference (rfc3986)</a>
41
42 @see
43 @ref grammar::parse,
44 @ref parse_relative_ref,
45 @ref url_view.
46*/
47#ifdef BOOST_URL_DOCS
48constexpr __implementation_defined__ relative_ref_rule;
49#else
50struct relative_ref_rule_t
51{
52 using value_type = url_view;
53
54 BOOST_URL_DECL
55 auto
56 parse(
57 char const*& it,
58 char const* end
59 ) const noexcept ->
60 system::result<value_type>;
61};
62
63constexpr relative_ref_rule_t relative_ref_rule{};
64#endif
