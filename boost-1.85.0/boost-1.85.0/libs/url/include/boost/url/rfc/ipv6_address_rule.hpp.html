20/** Rule for An IP version 6 style address
21
22 @par Value Type
23 @code
24 using value_type = ipv6_address;
25 @endcode
26
27 @par Example
28 Rules are used with the function @ref grammar::parse.
29 @code
30 system::result< ipv6_address > rv = grammar::parse( "2001:0db8:85a3:0000:0000:8a2e:0370:7334", ipv6_address_rule );
31 @endcode
32
33 @par BNF
34 @code
35 IPv6address = 6( h16 ":" ) ls32
36 / "::" 5( h16 ":" ) ls32
37 / [ h16 ] "::" 4( h16 ":" ) ls32
38 / [ *1( h16 ":" ) h16 ] "::" 3( h16 ":" ) ls32
39 / [ *2( h16 ":" ) h16 ] "::" 2( h16 ":" ) ls32
40 / [ *3( h16 ":" ) h16 ] "::" h16 ":" ls32
41 / [ *4( h16 ":" ) h16 ] "::" ls32
42 / [ *5( h16 ":" ) h16 ] "::" h16
43 / [ *6( h16 ":" ) h16 ] "::"
44
45 ls32 = ( h16 ":" h16 ) / IPv4address
46 ; least-significant 32 bits of address
47
48 h16 = 1*4HEXDIG
49 ; 16 bits of address represented in hexadecimal
50 @endcode
51
52 @par Specification
53 @li <a href="https://datatracker.ietf.org/doc/html/rfc4291"
54 >IP Version 6 Addressing Architecture (rfc4291)</a>
55 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.2.2"
56 >3.2.2. Host (rfc3986)</a>
57
58 @see
59 @ref ipv6_address,
60 @ref parse_ipv6_address,
61 @ref grammar::parse.
62*/
63#ifdef BOOST_URL_DOCS
64constexpr __implementation_defined__ ipv6_address_rule;
65#else
66struct ipv6_address_rule_t
67{
68 using value_type =
69 ipv6_address;
70
71 BOOST_URL_DECL
72 auto
73 parse(
74 char const*& it,
75 char const* end
76 ) const noexcept ->
77 system::result<ipv6_address>;
78};
79
80constexpr ipv6_address_rule_t ipv6_address_rule{};
81#endif
82
