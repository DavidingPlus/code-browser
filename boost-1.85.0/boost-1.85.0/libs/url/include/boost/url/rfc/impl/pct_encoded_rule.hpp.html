22template<class CharSet>
23auto
24parse_encoded(
25 char const*& it,
26 char const* end,
27 CharSet const& cs) noexcept ->
28 system::result<pct_string_view>
29{
30 auto const start = it;
31 std::size_t n = 0;
32 char const* it0;
33skip:
34 it0 = it;
35 it = grammar::find_if_not(
36 it0, end, cs);
37 n += it - it0;
38 if(it == end)
39 goto finish;
40 if(*it != '%')
41 goto finish;
42 for(;;)
43 {
44 ++it;
45 if(it == end)
46 {
47 // expected HEXDIG
48 BOOST_URL_RETURN_EC(
49 grammar::error::invalid);
50 }
51 auto r = grammar::hexdig_value(ch: *it);
52 if(r < 0)
53 {
54 // expected HEXDIG
55 BOOST_URL_RETURN_EC(
56 grammar::error::invalid);
57 }
58 ++it;
59 if(it == end)
60 {
61 // expected HEXDIG
62 BOOST_URL_RETURN_EC(
63 grammar::error::invalid);
64 }
65 r = grammar::hexdig_value(ch: *it);
66 if(r < 0)
67 {
68 // expected HEXDIG
69 BOOST_URL_RETURN_EC(
70 grammar::error::invalid);
71 }
72 ++n;
73 ++it;
74 if(it == end)
75 break;
76 if(*it != '%')
77 goto skip;
78 }
79finish:
80 return make_pct_string_view_unsafe(
81 data: start, size: it - start, decoded_size: n);
82}
83
84} // detail

88template<class CharSet>
89auto
90pct_encoded_rule_t<CharSet>::
91parse(
92 char const*& it,
93 char const* end) const noexcept ->
94 system::result<value_type>
95{
96 return detail::parse_encoded(
97 it, end, cs_);
98}
