11#ifndef BOOST_URL_PARSE_PATH_HPP
12#define BOOST_URL_PARSE_PATH_HPP
13
14#include <boost/url/detail/config.hpp>
15#include <boost/url/error_types.hpp>
16#include <boost/url/segments_encoded_view.hpp>
17
18namespace boost {
19namespace urls {
20
21/** Parse a string and return an encoded segment view
22
23 This function parses the string and returns the
24 corresponding path object if the string is valid,
25 otherwise returns an error.
26
27 @par BNF
28 @code
29 path = [ "/" ] segment *( "/" segment )
30 @endcode
31
32 @par Exception Safety
33 No-throw guarantee.
34
35 @return A valid view on success, otherwise an
36 error code.
37
38 @param s The string to parse
39
40 @par Specification
41 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-3.3"
42 >3.3. Path (rfc3986)</a>
43
44 @see
45 @ref segments_encoded_view.
46*/
47BOOST_URL_DECL
48system::result<segments_encoded_view>
49parse_path(core::string_view s) noexcept;
50
51} // urls
52} // boost
53
54#endif
