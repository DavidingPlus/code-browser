38// base class for the type-erased rule pair
39template<class T>
40struct range<T>::
41 any_rule
42{
43 virtual
44 ~any_rule() = default;
45
46 virtual
47 void
48 move(void* dest) noexcept
49 {
50 ::new(dest) any_rule(
51 std::move(*this));
52 }
53
54 virtual
55 void
56 copy(void* dest) const noexcept
57 {
58 ::new(dest) any_rule(*this);
59 }
60
61 virtual
62 system::result<T>
63 first(
64 char const*&,
65 char const*) const noexcept
66 {
67 return system::error_code{};
68 }
69
70 virtual
71 system::result<T>
72 next(
73 char const*&,
74 char const*) const noexcept
75 {
76 return system::error_code{};
77 }
78};

360template<class T>
361class range<T>::
362 iterator
363{
364public:
365 using value_type = T;
366 using reference = T const&;
367 using pointer = void const*;
368 using difference_type =
369 std::ptrdiff_t;
370 using iterator_category =
371 std::forward_iterator_tag;
372
373 iterator() = default;
374 iterator(
375 iterator const&) = default;
376 iterator& operator=(
377 iterator const&) = default;
378
379 reference
380 operator*() const noexcept
381 {
382 return *rv_;
383 }
384
385 bool
386 operator==(
387 iterator const& other) const noexcept
388 {
389 // can't compare iterators
390 // from different containers!
391 BOOST_ASSERT(r_ == other.r_);
392
393 return p_ == other.p_;
394 }
395
396 bool
397 operator!=(
398 iterator const& other) const noexcept
399 {
400 return !(*this == other);
401 }
402
403 iterator&
404 operator++() noexcept
405 {
406 BOOST_ASSERT(
407 p_ != nullptr);
408 auto const end =
409 r_->s_.data() +
410 r_->s_.size();
411 rv_ = r_->get().next(p_, end);
412 if( !rv_ )
413 p_ = nullptr;
414 return *this;
415 }
416
417 iterator
418 operator++(int) noexcept
419 {
420 auto tmp = *this;
421 ++*this;
422 return tmp;
423 }
424
425private:
426 friend class range<T>;
427
428 range<T> const* r_ = nullptr;
429 char const* p_ = nullptr;
430 system::result<T> rv_;
431
432 iterator(
433 range<T> const& r) noexcept
434 : r_(&r)
435 , p_(r.s_.data())
436 {
437 auto const end =
438 r_->s_.data() +
439 r_->s_.size();
440 rv_ = r_->get().first(p_, end);
441 if( !rv_ )
442 p_ = nullptr;
443 }
444
445 constexpr
446 iterator(
447 range<T> const& r,
448 int) noexcept
449 : r_(&r)
450 , p_(nullptr)
451 {
452 }
453};

502template<class T>
503range<T>::
504~range()
505{
506 get().~any_rule();
507}
508
509template<class T>
510range<T>::
511range() noexcept
512{
513 ::new(&get()) any_rule{};
514 char const* it = nullptr;
515 get().first(it, nullptr);
516 get().next(it, nullptr);
517}
518
519template<class T>
520range<T>::
521range(
522 range&& other) noexcept
523 : s_(other.s_)
524 , n_(other.n_)
525{
526 other.s_ = {};
527 other.n_ = {};
528 other.get().move(&get());
529 other.get().~any_rule();
530 ::new(&other.get()) any_rule{};
531}
532
533template<class T>
534range<T>::
535range(
536 range const& other) noexcept
537 : s_(other.s_)
538 , n_(other.n_)
539{
540 other.get().copy(&get());
541}
542
543template<class T>
544auto
545range<T>::
546operator=(
547 range&& other) noexcept ->
548 range&
549{
550 s_ = other.s_;
551 n_ = other.n_;
552 other.s_ = {};
553 other.n_ = 0;
554 // VFALCO we rely on nothrow move
555 // construction here, but if necessary we
556 // could move to a local buffer first.
557 get().~any_rule();
558 other.get().move(&get());
559 other.get().~any_rule();
560 ::new(&other.get()) any_rule{};
561 return *this;
562}
563
564template<class T>
565auto
566range<T>::
567operator=(
568 range const& other) noexcept ->
569 range&
570{
571 s_ = other.s_;
572 n_ = other.n_;
573 // VFALCO we rely on nothrow copy
574 // construction here, but if necessary we
575 // could construct to a local buffer first.
576 get().~any_rule();
577 other.get().copy(&get());
578 return *this;
579}

601template<class R>
602auto
603range_rule_t<R>::
604parse(
605 char const*& it,
606 char const* end) const ->
607 system::result<value_type>
608{
609 using T = typename R::value_type;
610
611 std::size_t n = 0;
612 auto const it0 = it;
613 auto it1 = it;
614 auto rv = (grammar::parse)(
615 it, end, next_);
616 if( !rv )
617 {
618 if(rv.error() != error::end_of_range)
619 {
620 // rewind unless error::end_of_range
621 it = it1;
622 }
623 if(n < N_)
624 {
625 // too few
626 BOOST_URL_RETURN_EC(
627 error::mismatch);
628 }
629 // good
630 return range<T>(
631 core::string_view(it0, it - it0),
632 n, next_);
633 }
634 for(;;)
635 {
636 ++n;
637 it1 = it;
638 rv = (grammar::parse)(
639 it, end, next_);
640 if( !rv )
641 {
642 if(rv.error() != error::end_of_range)
643 {
644 // rewind unless error::end_of_range
645 it = it1;
646 }
647 break;
648 }
649 if(n >= M_)
650 {
651 // too many
652 BOOST_URL_RETURN_EC(
653 error::mismatch);
654 }
655 }
656 if(n < N_)
657 {
658 // too few
659 BOOST_URL_RETURN_EC(
660 error::mismatch);
661 }
662 // good
663 return range<T>(
664 core::string_view(it0, it - it0),
665 n, next_);
666}
