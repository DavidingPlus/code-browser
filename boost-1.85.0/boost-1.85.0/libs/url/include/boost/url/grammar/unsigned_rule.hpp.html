26/** Match an unsigned decimal
27
28 Extra leading zeroes are disallowed.
29
30 @par Value Type
31 @code
32 using value_type = Unsigned;
33 @endcode
34
35 @par Example
36 Rules are used with the function @ref parse.
37 @code
38 system::result< unsigned short > rv = parse( "32767", unsigned_rule< unsigned short >{} );
39 @endcode
40
41 @par BNF
42 @code
43 unsigned = "0" / ( ["1"..."9"] *DIGIT )
44 @endcode
45
46 @tparam Unsigned The unsigned integer type used
47 to store the result.
48
49 @see
50 @ref grammar::parse.
51*/
52#ifdef BOOST_URL_DOCS
53template<class Unsigned>
54struct unsigned_rule;
55#else
56template<class Unsigned>
57struct unsigned_rule
58{
59 BOOST_STATIC_ASSERT(
60 std::numeric_limits<
61 Unsigned>::is_integer &&
62 ! std::numeric_limits<
63 Unsigned>::is_signed);
64
65 using value_type = Unsigned;
66
67 auto
68 parse(
69 char const*& it,
70 char const* end
71 ) const noexcept ->
72 system::result<value_type>;
73};
74#endif
75
76} // grammar
