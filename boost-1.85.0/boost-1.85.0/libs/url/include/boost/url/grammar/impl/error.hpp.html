38struct BOOST_SYMBOL_VISIBLE

47 std::string message(
48 int) const override;

97inline
98system::error_code
99make_error_code(
100 error ev) noexcept
101{
102 return system::error_code{
103 static_cast<std::underlying_type<
104 error>::type>(ev),
105 detail::error_cat};
106}
107
108inline
109system::error_condition
110make_error_condition(
111 condition c) noexcept
112{
113 return system::error_condition{
114 static_cast<std::underlying_type<
115 condition>::type>(c),
116 detail::condition_cat};
117}
