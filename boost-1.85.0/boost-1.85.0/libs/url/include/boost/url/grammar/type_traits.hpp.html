21/** Determine if T meets the requirements of Rule
22
23 This is an alias for `std::true_type` if
24 `T` meets the requirements, otherwise it
25 is an alias for `std::false_type`.
26
27 @par Example
28 @code
29 struct U
30 {
31 struct value_type;
32
33 auto
34 parse(
35 char const*& it,
36 char const* end) const ->
37 system::result<value_type>
38 };
39
40 static_assert( is_rule<U>::value, "Requirements not met" );
41 @endcode
42
43 @see
44 @ref parse.
45*/
46#ifdef BOOST_URL_DOCS
47template<class T>
48using is_rule = __see_below__;
49#else
50template<class T, class = void>
51struct is_rule : std::false_type {};
52
53template<class T>
54struct is_rule<T, void_t<decltype(
55 std::declval<system::result<typename T::value_type>&>() =
56 std::declval<T const&>().parse(
57 std::declval<char const*&>(),
58 std::declval<char const*>())
59 )>> : std::is_nothrow_copy_constructible<T>
60{
61};
62#endif
63
64} // grammar
65} // urls
66} // boost
67
68#endif
