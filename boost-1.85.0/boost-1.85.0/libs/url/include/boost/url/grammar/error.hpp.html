20/** Error codes returned when using rules
21
22 @see
23 @ref condition,
24 @ref parse.
25*/
26enum class error
27{
28 // VFALCO 3 space indent or
29 // else Doxygen malfunctions
30
31 //
32 // (informational)
33 //
34
35 /**
36 * More input is needed to match the rule
37 *
38 * A rule reached the end of the input,
39 * resulting in a partial match. The error
40 * is recoverable; the caller may obtain
41 * more input if possible and attempt to
42 * parse the character buffer again.
43 * Custom rules should only return this
44 * error if it is completely unambiguous
45 * that the rule cannot be matched without
46 * more input.
47 */
48 need_more = 1,
49
50 /**
51 * The rule did not match the input.
52 *
53 * This error is returned when a rule fails
54 * to match the input. The error is recoverable;
55 * the caller may rewind the input pointer and
56 * attempt to parse again using a different rule.
57 */
58 mismatch,
59
60 /**
61 * A rule reached the end of a range
62 *
63 * This indicates that the input was consumed
64 * when parsing a @ref range. The @ref range_rule
65 * avoids rewinding the input buffer when
66 * this error is returned. Thus the consumed
67 * characters are be considered part of the
68 * range without contributing additional
69 * elements.
70 */
71 end_of_range,
72
73 /**
74 * Leftover input remaining after match.
75 */
76 leftover,
77
78 //--------------------------------------------
79 //
80 // condition::fatal
81 //
82 //--------------------------------------------
83
84 /**
85 * A rule encountered unrecoverable invalid input.
86 *
87 * This error is returned when input is matching
88 * but one of the requirements is violated. For
89 * example if a percent escape is found, but
90 * one or both characters that follow are not
91 * valid hexadecimal digits. This is usually an
92 * unrecoverable error.
93 */
94 invalid,
95
96 /** An integer overflowed during parsing.
97 */
98 out_of_range,
99
100 /**
101 * An unspecified syntax error was found.
102 */
103 syntax
104};
105

108/** Error conditions for errors received from rules

114enum class condition

117 * A fatal error in syntax was encountered.

119 This indicates that parsing cannot continue.
120 */
121 fatal = 1
122};
