10#ifndef BOOST_URL_GRAMMAR_IMPL_UNSIGNED_RULE_HPP
11#define BOOST_URL_GRAMMAR_IMPL_UNSIGNED_RULE_HPP
12
13#include <boost/url/grammar/error.hpp>
14#include <boost/url/grammar/digit_chars.hpp>
15#include <algorithm> // VFALCO grr..
16
17namespace boost {
18namespace urls {
19namespace grammar {
20
21template<class U>
22auto
23unsigned_rule<U>::
24parse(
25 char const*& it,
26 char const* end
27 ) const noexcept ->
28 system::result<value_type>
29{
30 if(it == end)
31 {
32 // end
33 BOOST_URL_RETURN_EC(
34 error::mismatch);
35 }
36 if(*it == '0')
37 {
38 ++it;
39 if( it == end ||
40 ! digit_chars(*it))
41 {
42 return U(0);
43 }
44 // bad leading zero
45 BOOST_URL_RETURN_EC(
46 error::invalid);
47 }
48 if(! digit_chars(*it))
49 {
50 // expected digit
51 BOOST_URL_RETURN_EC(
52 error::mismatch);
53 }
54 static constexpr U Digits10 =
55 std::numeric_limits<
56 U>::digits10;
57 static constexpr U ten = 10;
58 char const* safe_end;
59 if(static_cast<std::size_t>(
60 end - it) >= Digits10)
61 safe_end = it + Digits10;
62 else
63 safe_end = end;
64 U u = *it - '0';
65 ++it;
66 while(it != safe_end &&
67 digit_chars(*it))
68 {
69 char const dig = *it - '0';
70 u = u * ten + dig;
71 ++it;
72 }
73 if( it != end &&
74 digit_chars(*it))
75 {
76 static constexpr U Max = (
77 std::numeric_limits<
78 U>::max)();
79 static constexpr
80 auto div = (Max / ten);
81 static constexpr
82 char rem = (Max % ten);
83 char const dig = *it - '0';
84 if( u > div || (
85 u == div && dig > rem))
86 {
87 // integer overflow
88 BOOST_URL_RETURN_EC(
89 error::invalid);
90 }
91 u = u * ten + dig;
92 ++it;
93 if( it < end &&
94 digit_chars(*it))
95 {
96 // integer overflow
97 BOOST_URL_RETURN_EC(
98 error::invalid);
99 }
100 }
101
102 return u;
103}
104
105} // grammar
106} // urls
107} // boost
