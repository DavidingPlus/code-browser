10#ifndef BOOST_URL_GRAMMAR_IMPL_OPTIONAL_RULE_HPP
11#define BOOST_URL_GRAMMAR_IMPL_OPTIONAL_RULE_HPP
12
13#include <boost/url/grammar/error.hpp>
14
15namespace boost {
16namespace urls {
17namespace grammar {
18
19template<class R>
20auto
21optional_rule_t<R>::
22parse(
23 char const*& it,
24 char const* end) const ->
25 system::result<value_type>
26{
27 if(it == end)
28 return boost::none;
29 auto const it0 = it;
30 auto rv =
31 this->get().parse(it, end);
32 if(rv)
33 return value_type(*rv);
34 it = it0;
35 return boost::none;
36}
37
38} // grammar
39} // urls
40} // boost
