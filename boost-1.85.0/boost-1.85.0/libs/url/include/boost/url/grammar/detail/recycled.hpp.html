20template<
21 std::size_t Size,
22 std::size_t Align>
23struct aligned_storage_impl
24{
25 void* addr() noexcept
26 {
27 return buf_;
28 }
29
30 void const* addr() const noexcept
31 {
32 return buf_;
33 }
34
35private:
36 alignas(Align)
37 unsigned char buf_[Size];
38};
