21template<class T>
22recycled<T>::
23~recycled()
24{
25 std::size_t n = 0;
26 // VFALCO we should probably deallocate
27 // in reverse order of allocation but
28 // that requires a doubly-linked list.
29 auto it = head_;
30 while(it)
31 {
32 ++n;
33 auto next = it->next;
34 BOOST_ASSERT(
35 it->refs == 0);
36 delete it;
37 it = next;
38 }
39 detail::recycled_remove(
40 sizeof(U) * n);
41}
42
43template<class T>
44auto
45recycled<T>::
46acquire() ->
47 U*
48{
49 U* p;
50 {
51#if !defined(BOOST_URL_DISABLE_THREADS)
52 std::lock_guard<
53 std::mutex> lock(m_);
54#endif
55 p = head_;
56 if(p)
57 {
58 // reuse
59 head_ = head_->next;
60 detail::recycled_remove(
61 sizeof(U));
62 ++p->refs;
63 }
64 else
65 {
66 p = new U;
67 }
68 }
69 BOOST_ASSERT(p->refs == 1);
70 return p;
71}
72
73template<class T>
74void
75recycled<T>::
76release(U* u) noexcept
77{
78 if(--u->refs != 0)
79 return;
80 {
81#if !defined(BOOST_URL_DISABLE_THREADS)
82 std::lock_guard<
83 std::mutex> lock(m_);
84#endif
85 u->next = head_;
86 head_ = u;
87 }
88 detail::recycled_add(
89 sizeof(U));
90}

94template<class T>
95recycled_ptr<T>::
96~recycled_ptr()
97{
98 if(p_)
99 bin_->release(p_);
100}

102template<class T>
103recycled_ptr<T>::
104recycled_ptr(
105 recycled<T>& bin)
106 : bin_(&bin)
107 , p_(bin.acquire())
108{
109}

111template<class T>
112recycled_ptr<T>::
113recycled_ptr(
114 recycled<T>& bin,

116 : bin_(&bin)

118}

120template<class T>
121recycled_ptr<T>::
122recycled_ptr()
123 : recycled_ptr(nullptr)
124{
125 p_ = bin_->acquire();
126}

128template<class T>
129recycled_ptr<T>::
130recycled_ptr(
131 std::nullptr_t) noexcept
132 : recycled_ptr([]() -> B&
133 {
134 // VFALCO need guaranteed constexpr-init
135 static B r;
136 return r;
137 }(), nullptr)
138{
139}

141template<class T>
142recycled_ptr<T>::
143recycled_ptr(
144 recycled_ptr const& other) noexcept
145 : bin_(other.bin_)
146 , p_(other.p_)
147{
148 if(p_)
149 ++p_->refs;
150}
151
152template<class T>
153recycled_ptr<T>::
154recycled_ptr(
155 recycled_ptr&& other) noexcept
156 : bin_(other.bin_)
157 , p_(other.p_)
158{
159 other.p_ = nullptr;
160}
161
162template<class T>
163auto
164recycled_ptr<T>::
165operator=(
166 recycled_ptr&& other) noexcept ->
167 recycled_ptr&
168{
169 BOOST_ASSERT(
170 bin_ == other.bin_);
171 if(p_)
172 bin_->release(p_);
173 p_ = other.p_;
174 other.p_ = nullptr;
175 return *this;
176}
177
178template<class T>
179auto
180recycled_ptr<T>::
181operator=(
182 recycled_ptr const& other) noexcept ->
183 recycled_ptr&
184{
185 BOOST_ASSERT(
186 bin_ == other.bin_);
187 if(p_)
188 bin_->release(p_);
189 p_ = other.p_;
190 if(p_)
191 ++p_->refs;
192 return *this;
193}
194
195template<class T>
196T&
197recycled_ptr<T>::
198acquire()
199{
200 if(! p_)
201 p_ = bin_->acquire();
202 return p_->t;
203}
204
205template<class T>
206void
207recycled_ptr<T>::
208release() noexcept
209{
210 if(p_)
211 {
212 bin_->release(p_);
213 p_ = nullptr;
214 }
215}
