24// must come first
25template<
26 class R0,
27 class... Rn,
28 std::size_t I>
29auto
30parse_variant(
31 char const*&,
32 char const*,
33 detail::tuple<
34 R0, Rn...> const&,
35 std::integral_constant<
36 std::size_t, I> const&,
37 std::false_type const&) ->
38 system::result<variant<
39 typename R0::value_type,
40 typename Rn::value_type...>>
41{
42 // no match
43 BOOST_URL_RETURN_EC(
44 error::mismatch);
45}
46
47template<
48 class R0,
49 class... Rn,
50 std::size_t I>
51auto
52parse_variant(
53 char const*& it,
54 char const* const end,
55 detail::tuple<
56 R0, Rn...> const& rn,
57 std::integral_constant<
58 std::size_t, I> const&,
59 std::true_type const&) ->
60 system::result<variant<
61 typename R0::value_type,
62 typename Rn::value_type...>>
63{
64 auto const it0 = it;
65 auto rv = parse(
66 it, end, get<I>(rn));
67 if( rv )
68 return variant<
69 typename R0::value_type,
70 typename Rn::value_type...>{
71 variant2::in_place_index_t<I>{}, *rv};
72 it = it0;
73 return parse_variant(
74 it, end, rn,
75 std::integral_constant<
76 std::size_t, I+1>{},
77 std::integral_constant<bool,
78 ((I + 1) < (1 +
79 sizeof...(Rn)))>{});
80}
81
82} // detail
83
84template<class R0, class... Rn>
85auto
86variant_rule_t<R0, Rn...>::
87parse(
88 char const*& it,
89 char const* end) const ->
90 system::result<value_type>
91{
92 return detail::parse_variant(
93 it, end, rn_,
94 std::integral_constant<
95 std::size_t, 0>{},
96 std::true_type{});
97}

100
101template<class R0, class... Rn>
102auto
103constexpr
104variant_rule(
105 R0 const& r0,
106 Rn const&... rn) noexcept ->
107 variant_rule_t<R0, Rn...>
108{
109 return { r0, rn... };
110}
