11#ifndef BOOST_URL_SEGMENTS_ENCODED_BASE_HPP
12#define BOOST_URL_SEGMENTS_ENCODED_BASE_HPP
13
14#include <boost/url/detail/config.hpp>
15#include <boost/url/ignore_case.hpp>
16#include <boost/url/pct_string_view.hpp>
17#include <boost/url/detail/url_impl.hpp>
18#include <iosfwd>
19
20namespace boost {
21namespace urls {
22
23/** Common functionality for containers
24
25 This base class is used by the library
26 to provide common member functions for
27 containers. This cannot be instantiated
28 directly; Instead, use one of the
29 containers or functions:
30
31 @par Containers
32 @li @ref segments_ref
33 @li @ref segments_view
34 @li @ref segments_encoded_ref
35 @li @ref segments_encoded_view
36*/
37class segments_encoded_base
38{
39 detail::path_ref ref_;
40
41 friend class url_view_base;
42 friend class segments_encoded_ref;
43 friend class segments_encoded_view;
44
45 segments_encoded_base(
46 detail::path_ref const& ref) noexcept;
47 segments_encoded_base() = default;
48 segments_encoded_base(
49 segments_encoded_base const&) = default;
50 segments_encoded_base& operator=(
51 segments_encoded_base const&) = default;
52
53public:
54 /** A Bidirectional iterator to a path segment
55
56 Objects of this type allow iteration
57 through the segments in the path.
58 Strings returned by iterators may
59 contain percent escapes.
60 The values returned are read-only;
61 changes to segments must be made
62 through the container instead, if the
63 container supports modification.
64
65 <br>
66
67 The strings produced when iterators
68 are dereferenced refer to the underlying
69 character buffer.
70 Ownership is not transferred; the caller
71 is responsible for ensuring that the
72 lifetime of the buffer extends until
73 it is no longer referenced by any
74 container or iterator.
75 */
76#ifdef BOOST_URL_DOCS
77 using iterator = __see_below__;
78#else
79 class iterator;
80#endif
81
82 /// @copydoc iterator
83 using const_iterator = iterator;
84
85 /** The value type
86
87 Values of this type represent a segment
88 where unique ownership is retained by
89 making a copy.
90
91 @par Example
92 @code
93 segments_encoded_base::value_type ps( url_view( "/path/to/file.txt" ).encoded_segments().back() );
94 @endcode
95 */
96 using value_type = std::string;
97
98 /** The reference type
99
100 This is the type of value returned when
101 iterators of the view are dereferenced.
102 */
103 using reference = pct_string_view;
104
105 /// @copydoc reference
106 using const_reference = pct_string_view;
107
108 /** An unsigned integer type used to represent size.
109 */
110 using size_type = std::size_t;
111
112 /** A signed integer type used to represent differences.
113 */
114 using difference_type = std::ptrdiff_t;
115
116 //--------------------------------------------
117 //
118 // Observers
119 //
120 //--------------------------------------------
121
122 /** Return the maximum number of characters possible
123
124 This represents the largest number of
125 characters that are possible in a path,
126 not including any null terminator.
127
128 @par Exception Safety
129 Throws nothing.
130 */
131 static
132 constexpr
133 std::size_t
134 max_size() noexcept
135 {
136 return BOOST_URL_MAX_SIZE;
137 }
138
139 /** Return the referenced character buffer.
140
141 This function returns the character
142 buffer referenced by the view.
143 The returned string may contain
144 percent escapes.
145
146 @par Example
147 @code
148 assert( url_view( "/path/to/file.txt" ).encoded_segments().buffer() == "/path/to/file.txt" );
149 @endcode
150
151 @par Complexity
152 Constant.
153
154 @par Exception Safety
155 Throws nothing.
156 */
157 BOOST_URL_DECL
158 pct_string_view
159 buffer() const noexcept;
160
161 /** Returns true if this references an absolute path.
162
163 Absolute paths always start with a
164 forward slash ('/').
165
166 @par Example
167 @code
168 assert( url_view( "/path/to/file.txt" ).encoded_segments().is_absolute() == true );
169 @endcode
170
171 @par Complexity
172 Constant.
173
174 @par Exception Safety
175 Throws nothing.
176 */
177 BOOST_URL_DECL
178 bool
179 is_absolute() const noexcept;
180
181 /** Return true if there are no segments
182
183 @par Example
184 @code
185 assert( ! url_view( "/index.htm" ).encoded_segments().empty() );
186 @endcode
187
188 @par Complexity
189 Constant.
190
191 @par Exception Safety
192 Throws nothing.
193 */
194 BOOST_URL_DECL
195 bool
196 empty() const noexcept;
197
198 /** Return the number of segments
199
200 @par Example
201 @code
202 assert( url_view( "/path/to/file.txt" ).encoded_segments().size() == 3 );
203 @endcode
204
205 @par Complexity
206 Constant.
207
208 @par Exception Safety
209 Throws nothing.
210 */
211 BOOST_URL_DECL
212 std::size_t
213 size() const noexcept;
214
215 /** Return the first segment
216
217 This function returns a string with the
218 first segment of the path without any
219 leading or trailing '/' separators.
220 The returned string may contain
221 percent escapes.
222
223 @par Preconditions
224 @code
225 this->empty() == false
226 @endcode
227
228 @par Effects
229 @code
230 return *begin();
231 @endcode
232
233 @par Example
234 @code
235 assert( url_view( "/path/to/file.txt" ).encoded_segments().front() == "path" );
236 @endcode
237
238 @par Complexity
239 Constant.
240
241 @par Exception Safety
242 Throws nothing.
243 */
244 pct_string_view
245 front() const noexcept;
246
247 /** Return the last segment
248
249 This function returns a string with the
250 last segment of the path without any
251 leading or trailing '/' separators.
252 The returned string may contain
253 percent escapes.
254
255 @par Preconditions
256 @code
257 this->empty() == false
258 @endcode
259
260 @par Example
261 @code
262 assert( url_view( "/path/to/file.txt" ).encoded_segments().back() == "file.txt" );
263 @endcode
264
265 @par Preconditions
266 @code
267 this->empty() == false
268 @endcode
269
270 @par Effects
271 @code
272 return *--end();
273 @endcode
274
275 @par Complexity
276 Constant.
277
278 @par Exception Safety
279 Throws nothing.
280 */
281 pct_string_view
282 back() const noexcept;
283
284 /** Return an iterator to the beginning
285
286 @par Complexity
287 Linear in `this->front().size()` or
288 constant if `this->empty()`.
289
290 @par Exception Safety
291 Throws nothing.
292 */
293 BOOST_URL_DECL
294 iterator
295 begin() const noexcept;
296
297 /** Return an iterator to the end
298
299 @par Complexity
300 Constant.
301
302 @par Exception Safety
303 Throws nothing.
304 */
305 BOOST_URL_DECL
306 iterator
307 end() const noexcept;
308};
309

312/** Format to an output stream
313
314 Any percent-escapes are emitted as-is;
315 no decoding is performed.
316
317 @par Complexity
318 Linear in `ps.buffer().size()`.
319
320 @par Effects
321 @code
322 return os << ps.buffer();
323 @endcode
324*/
325BOOST_URL_DECL
326std::ostream&
327operator<<(
328 std::ostream& os,
329 segments_encoded_base const& ps);
