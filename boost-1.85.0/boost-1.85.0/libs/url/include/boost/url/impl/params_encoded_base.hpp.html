25class params_encoded_base::iterator
26{
27 detail::params_iter_impl it_;
28
29 friend class params_encoded_base;
30 friend class params_encoded_ref;
31
32 iterator(detail::query_ref const& ref) noexcept;
33 iterator(detail::query_ref const& ref, int) noexcept;
34 iterator(
35 detail::params_iter_impl const& it)
36 : it_(it)
37 {
38 }
39
40public:
41 using value_type =
42 params_encoded_base::value_type;
43 using reference =
44 params_encoded_base::reference;
45 using pointer = reference;
46 using difference_type = std::ptrdiff_t;
47 using iterator_category =
48 std::bidirectional_iterator_tag;
49
50 iterator() = default;
51 iterator(iterator const&) = default;
52 iterator& operator=(
53 iterator const&) = default;
54
55 iterator&
56 operator++() noexcept
57 {
58 it_.increment();
59 return *this;
60 }
61
62 iterator
63 operator++(int) noexcept
64 {
65 auto tmp = *this;
66 ++*this;
67 return tmp;
68 }
69
70 iterator&
71 operator--() noexcept
72 {
73 it_.decrement();
74 return *this;
75 }
76
77 iterator
78 operator--(int) noexcept
79 {
80 auto tmp = *this;
81 --*this;
82 return tmp;
83 }
84
85 reference
86 operator*() const
87 {
88 return it_.dereference();
89 }
90
91 pointer
92 operator->() const
93 {
94 return it_.dereference();
95 }
96
97 friend
98 bool
99 operator==(
100 iterator const& it0,
101 iterator const& it1) noexcept
102 {
103 return it0.it_.equal(other: it1.it_);
104 }
105
106 friend
107 bool
108 operator!=(
109 iterator const& it0,
110 iterator const& it1) noexcept
111 {
112 return ! it0.it_.equal(other: it1.it_);
113 }
114};
