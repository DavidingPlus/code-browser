10#ifndef BOOST_URL_IMPL_PCT_ENCODED_VIEW_HPP
11#define BOOST_URL_IMPL_PCT_ENCODED_VIEW_HPP
12
13#include <boost/url/grammar/type_traits.hpp>
14#include <boost/static_assert.hpp>
15
16namespace boost {
17namespace urls {
18
19class decode_view::iterator
20{
21 char const* begin_ = nullptr;
22 char const* pos_ = nullptr;
23 bool space_as_plus_ = true;
24
25 friend decode_view;
26
27 iterator(
28 char const* str,
29 bool space_as_plus) noexcept
30 : begin_(str)
31 , pos_(str)
32 , space_as_plus_(
33 space_as_plus)
34 {
35 }
36
37 // end ctor
38 iterator(
39 char const* str,
40 size_type n,
41 bool space_as_plus) noexcept
42 : begin_(str)
43 , pos_(str + n)
44 , space_as_plus_(space_as_plus)
45 {
46 }
47
48public:
49 using value_type = char;
50 using reference = char;
51 using pointer = void const*;
52 using const_reference = char;
53 using size_type = std::size_t;
54 using difference_type = std::ptrdiff_t;
55 using iterator_category =
56 std::bidirectional_iterator_tag;
57
58 iterator() = default;
59
60 iterator(iterator const&) = default;
61
62 iterator&
63 operator=(iterator const&) = default;
64
65 BOOST_URL_DECL
66 reference
67 operator*() const noexcept;
68
69 iterator&
70 operator++() noexcept
71 {
72 BOOST_ASSERT(pos_ != nullptr);
73 if (*pos_ != '%')
74 ++pos_;
75 else
76 pos_ += 3;
77 return *this;
78 }
79
80 iterator&
81 operator--() noexcept
82 {
83 BOOST_ASSERT(pos_ != begin_);
84 if (pos_ - begin_ < 3 ||
85 pos_[-3] != '%')
86 --pos_;
87 else
88 pos_ -= 3;
89 return *this;
90 }
91
92 iterator
93 operator++(int) noexcept
94 {
95 auto tmp = *this;
96 ++*this;
97 return tmp;
98 }
99
100 iterator
101 operator--(int) noexcept
102 {
103 auto tmp = *this;
104 --*this;
105 return tmp;
106 }
107
108 char const*
109 base()
110 {
111 return pos_;
112 }
113
114 bool
115 operator==(
116 iterator const& other) const noexcept
117 {
118 return pos_ == other.pos_;
119 }
120
121 bool
122 operator!=(
123 iterator const& other) const noexcept
124 {
125 return !(*this == other);
126 }
127};
128
129//------------------------------------------------
130
131inline
132auto
133decode_view::
134begin() const noexcept ->
135 const_iterator
136{
137 return {p_, space_as_plus_};
138}
139
140inline
141auto
142decode_view::
143end() const noexcept ->
144 const_iterator
145{
146 return {p_, n_, space_as_plus_};
147}
148
149inline
150auto
151decode_view::
152front() const noexcept ->
153 const_reference
154{
155 BOOST_ASSERT( !empty() );
156 return *begin();
157}
158
159inline
160auto
161decode_view::
162back() const noexcept ->
163 const_reference
164{
165 BOOST_ASSERT( !empty() );
166 return *--end();
167}
