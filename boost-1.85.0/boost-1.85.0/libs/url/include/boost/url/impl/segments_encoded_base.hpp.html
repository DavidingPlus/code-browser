11#ifndef BOOST_URL_IMPL_SEGMENTS_ENCODED_BASE_HPP
12#define BOOST_URL_IMPL_SEGMENTS_ENCODED_BASE_HPP
13
14#include <boost/url/detail/segments_iter_impl.hpp>
15#include <boost/assert.hpp>
16
17namespace boost {
18namespace urls {
19
20class segments_encoded_base::iterator
21{
22 detail::segments_iter_impl it_;
23
24 friend class url_base;
25 friend class segments_encoded_base;
26 friend class segments_encoded_ref;
27
28 iterator(detail::path_ref const&) noexcept;
29 iterator(detail::path_ref const&, int) noexcept;
30
31 iterator(
32 detail::segments_iter_impl const& it) noexcept
33 : it_(it)
34 {
35 }
36
37public:
38 using value_type =
39 segments_encoded_base::value_type;
40 using reference =
41 segments_encoded_base::reference;
42 using pointer = reference;
43 using difference_type = std::ptrdiff_t;
44 using iterator_category =
45 std::bidirectional_iterator_tag;
46
47 iterator() = default;
48 iterator(iterator const&) = default;
49 iterator& operator=(
50 iterator const&) = default;
51
52 reference
53 operator*() const noexcept
54 {
55 return it_.dereference();
56 }
57
58 pointer
59 operator->() const noexcept
60 {
61 return it_.dereference();
62 }
63
64 iterator&
65 operator++() noexcept
66 {
67 it_.increment();
68 return *this;
69 }
70
71 iterator&
72 operator--() noexcept
73 {
74 it_.decrement();
75 return *this;
76 }
77
78 iterator
79 operator++(int) noexcept
80 {
81 auto tmp = *this;
82 ++*this;
83 return tmp;
84 }
85
86 iterator
87 operator--(int) noexcept
88 {
89 auto tmp = *this;
90 --*this;
91 return tmp;
92 }
93
94 bool
95 operator==(
96 iterator const& other) const noexcept
97 {
98 return it_.equal(other: other.it_);
99 }
100
101 bool
102 operator!=(
103 iterator const& other) const noexcept
104 {
105 return ! it_.equal(other: other.it_);
106 }
107};

113segments_encoded_base::
114front() const noexcept

120inline
121pct_string_view
122segments_encoded_base::
123back() const noexcept

125 BOOST_ASSERT(! empty());
126 return *--end();
127}
