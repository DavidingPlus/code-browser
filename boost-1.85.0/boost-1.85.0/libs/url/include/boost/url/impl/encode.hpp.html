27template<class CharSet>
28std::size_t
29encoded_size(
30 core::string_view s,
31 CharSet const& unreserved,
32 encoding_opts opt) noexcept
33{
34/* If you get a compile error here, it
35 means that the value you passed does
36 not meet the requirements stated in
37 the documentation.
38*/
39 static_assert(
40 grammar::is_charset<CharSet>::value,
41 "Type requirements not met");
42
43 std::size_t n = 0;
44 auto it = s.data();
45 auto const last = it + s.size();
46
47 if(! opt.space_as_plus ||
48 unreserved(' '))
49 {
50 while(it != last)
51 {
52 if(unreserved(*it))
53 n += 1;
54 else
55 n += 3;
56 ++it;
57 }
58 }
59 else
60 {
61 while(it != last)
62 {
63 auto c = *it;
64 if(unreserved(c))
65 ++n;
66 else if(c == ' ')
67 ++n;
68 else
69 n += 3;
70 ++it;
71 }
72 }
73 return n;
74}

78template<class CharSet>
79std::size_t
80encode(
81 char* dest,
82 std::size_t size,
83 core::string_view s,
84 CharSet const& unreserved,
85 encoding_opts opt)
86{
87/* If you get a compile error here, it
88 means that the value you passed does
89 not meet the requirements stated in
90 the documentation.
91*/
92 static_assert(
93 grammar::is_charset<CharSet>::value,
94 "Type requirements not met");
95
96 // '%' must be reserved
97 BOOST_ASSERT(! unreserved('%'));
98
99 char const* const hex =
100 detail::hexdigs[opt.lower_case];
101 auto const encode = [hex](
102 char*& dest,
103 unsigned char c) noexcept
104 {
105 *dest++ = '%';
106 *dest++ = hex[c>>4];
107 *dest++ = hex[c&0xf];
108 };
109
110 auto it = s.data();
111 auto const end = dest + size;
112 auto const last = it + s.size();
113 auto const dest0 = dest;
114 auto const end3 = end - 3;
115
116 if(! opt.space_as_plus)
117 {
118 while(it != last)
119 {
120 if(unreserved(*it))
121 {
122 if(dest == end)
123 return dest - dest0;
124 *dest++ = *it++;
125 continue;
126 }
127 if(dest > end3)
128 return dest - dest0;
129 encode(dest, *it++);
130 }
131 return dest - dest0;
132 }
133 else if(! unreserved(' '))
134 {
135 // VFALCO space is usually reserved,
136 // and we depend on this for an
137 // optimization. if this assert
138 // goes off we can split the loop
139 // below into two versions.
140 BOOST_ASSERT(! unreserved(' '));
141
142 while(it != last)
143 {
144 if(unreserved(*it))
145 {
146 if(dest == end)
147 return dest - dest0;
148 *dest++ = *it++;
149 continue;
150 }
151 if(*it == ' ')
152 {
153 if(dest == end)
154 return dest - dest0;
155 *dest++ = '+';
156 ++it;
157 continue;
158 }
159 if(dest > end3)
160 return dest - dest0;
161 encode(dest, *it++);
162 }
163 }
164 return dest - dest0;
165}

169// unsafe encode just
170// asserts on the output buffer
171//
172template<class CharSet>
173std::size_t
174encode_unsafe(
175 char* dest,
176 std::size_t size,
177 core::string_view s,
178 CharSet const& unreserved,
179 encoding_opts opt)
180{
181 // '%' must be reserved
182 BOOST_ASSERT(! unreserved('%'));
183
184 auto it = s.data();
185 auto const last = it + s.size();
186 auto const end = dest + size;
187 ignore_unused(end);
188
189 char const* const hex =
190 detail::hexdigs[opt.lower_case];
191 auto const encode = [end, hex](
192 char*& dest,
193 unsigned char c) noexcept
194 {
195 ignore_unused(end);
196 *dest++ = '%';
197 BOOST_ASSERT(dest != end);
198 *dest++ = hex[c>>4];
199 BOOST_ASSERT(dest != end);
200 *dest++ = hex[c&0xf];
201 };
202
203 auto const dest0 = dest;
204 if(! opt.space_as_plus)
205 {
206 while(it != last)
207 {
208 BOOST_ASSERT(dest != end);
209 if(unreserved(*it))
210 *dest++ = *it++;
211 else
212 encode(dest, *it++);
213 }
214 }
215 else
216 {
217 // VFALCO space is usually reserved,
218 // and we depend on this for an
219 // optimization. if this assert
220 // goes off we can split the loop
221 // below into two versions.
222 BOOST_ASSERT(! unreserved(' '));
223
224 while(it != last)
225 {
226 BOOST_ASSERT(dest != end);
227 if(unreserved(*it))
228 {
229 *dest++ = *it++;
230 }
231 else if(*it == ' ')
232 {
233 *dest++ = '+';
234 ++it;
235 }
236 else
237 {
238 encode(dest, *it++);
239 }
240 }
241 }
242 return dest - dest0;
243}

247template<
248 class StringToken,
249 class CharSet>
250BOOST_URL_STRTOK_RETURN
251encode(
252 core::string_view s,
253 CharSet const& unreserved,
254 encoding_opts opt,
255 StringToken&& token) noexcept
256{
257/* If you get a compile error here, it
258 means that the value you passed does
259 not meet the requirements stated in
260 the documentation.
261*/
262 static_assert(
263 grammar::is_charset<CharSet>::value,
264 "Type requirements not met");
265
266 auto const n = encoded_size(
267 s, unreserved, opt);
268 auto p = token.prepare(n);
269 if(n > 0)
270 encode_unsafe(
271 p, n, s, unreserved, opt);
272 return token.result();
273}
