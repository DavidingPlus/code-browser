26// state of the format string. It basically keeps
27// track of where we are in the format string.
28class format_parse_context
29{
30 char const* begin_;
31 char const* end_;
32 std::size_t arg_id_ = 0;
33
34public:
35 constexpr
36 format_parse_context(
37 char const* first,
38 char const* last,
39 std::size_t arg_id = 0)
40 : begin_( first )
41 , end_( last )
42 , arg_id_( arg_id )
43 {}
44
45 constexpr
46 format_parse_context(
47 core::string_view fmt,
48 std::size_t arg_id = 0)
49 : format_parse_context(
50 fmt.data(),
51 fmt.data() + fmt.size(),
52 arg_id )
53 {}
54
55 constexpr
56 char const*
57 begin() const noexcept
58 {
59 return begin_;
60 }
61
62 constexpr
63 char const*
64 end() const noexcept
65 {
66 return end_;
67 }
68
69 BOOST_CXX14_CONSTEXPR
70 void
71 advance_to( char const* it )
72 {
73 begin_ = it;
74 }
75
76 std::size_t
77 next_arg_id()
78 {
79 return arg_id_++;
80 }
81};

100// A type erased format argument
101class format_arg
102{
103 void const* arg_;
104 void (*measure_)(
105 format_parse_context&,
106 measure_context&,
107 grammar::lut_chars const&,
108 void const* );
109 void (*fmt_)(
110 format_parse_context&,
111 format_context&,
112 grammar::lut_chars const&,
113 void const* );
114 core::string_view name_;
115 std::size_t value_ = 0;
116 bool ignore_ = false;
117
118 template <class A>
119 static
120 void
121 measure_impl(
122 format_parse_context& pctx,
123 measure_context& mctx,
124 grammar::lut_chars const& cs,
125 void const* a );
126
127 template <class A>
128 static
129 void
130 format_impl(
131 format_parse_context& pctx,
132 format_context& fctx,
133 grammar::lut_chars const& cs,
134 void const* a );
135
136public:
137 template<class A>
138 format_arg( A&& a );
139
140 template<class A>
141 format_arg( named_arg<A>&& a );
142
143 template<class A>
144 format_arg( core::string_view name, A&& a );
145
146 format_arg()
147 : format_arg(ignore_format{})
148 {}
149
150 explicit
151 operator bool() const noexcept
152 {
153 return !ignore_;
154 }
155
156 void
157 measure(
158 format_parse_context& pctx,
159 measure_context& mctx,
160 grammar::lut_chars const& cs)
161 {
162 measure_( pctx, mctx, cs, arg_ );
163 }
164
165 void
166 format(
167 format_parse_context& pctx,
168 format_context& fctx,
169 grammar::lut_chars const& cs )
170 {
171 fmt_( pctx, fctx, cs, arg_ );
172 }
173
174 core::string_view
175 name() const
176 {
177 return name_;
178 }
179
180 std::size_t
181 value() const
182 {
183 return value_;
184 }
185};

187// create temp stack storage for type erased args
188template< class... Args >
189std::array<format_arg, sizeof...(Args)>
190make_format_args( Args&&... args )
191{
192 return {{ std::forward<Args>(args)... }};

195// reference to an array of format_args
196class format_args
197{
198 format_arg const* p_{nullptr};
199 std::size_t n_{0};
200
201public:
202 format_args(
203 detail::format_arg const* first,
204 detail::format_arg const* last ) noexcept
205 : p_(first)
206 , n_(static_cast<std::size_t>(last - first))
207 {}
208
209 template < std::size_t N >
210 format_args( std::array<format_arg, N> const& store ) noexcept
211 : p_(store.data())
212 , n_(store.size())
213 {}
214
215 format_arg
216 get( std::size_t i ) const noexcept
217 {
218 if (i < n_)
219 return p_[i];
220 return {};
221 }
222
223 format_arg
224 get( core::string_view name ) const noexcept
225 {
226 for (std::size_t i = 0; i < n_; ++i)
227 {
228 if (p_[i].name() == name)
229 return p_[i];
230 }
231 return {};
232 }
233};
