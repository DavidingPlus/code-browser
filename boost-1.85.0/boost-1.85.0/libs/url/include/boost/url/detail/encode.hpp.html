31// re-encode is to percent-encode a
32// string that can already contain
33// escapes. Characters not in the
34// unreserved set are escaped, and
35// escapes are passed through unchanged.
36//
37template<class CharSet>
38std::size_t
39re_encoded_size_unsafe(
40 core::string_view s,
41 CharSet const& unreserved,
42 encoding_opts opt) noexcept
43{
44 std::size_t n = 0;
45 auto const end = s.end();
46 auto it = s.begin();
47 if(opt.space_as_plus)
48 {
49 while(it != end)
50 {
51 if(*it != '%')
52 {
53 if( unreserved(*it)
54 || *it == ' ')
55 n += 1;
56 else
57 n += 3;
58 ++it;
59 }
60 else
61 {
62 BOOST_ASSERT(end - it >= 3);
63 BOOST_ASSERT(
64 grammar::hexdig_value(
65 it[1]) >= 0);
66 BOOST_ASSERT(
67 grammar::hexdig_value(
68 it[2]) >= 0);
69 n += 3;
70 it += 3;
71 }
72 }
73 }
74 else
75 {
76 while(it != end)
77 {
78 if(*it != '%')
79 {
80 if(unreserved(*it))
81 n += 1;
82 else
83 n += 3;
84 ++it;
85 }
86 else
87 {
88 BOOST_ASSERT(end - it >= 3);
89 BOOST_ASSERT(
90 grammar::hexdig_value(
91 it[1]) >= 0);
92 BOOST_ASSERT(
93 grammar::hexdig_value(
94 it[2]) >= 0);
95 n += 3;
96 it += 3;
97 }
98 }
99 }
100 return n;
101}

103// unchecked
104// returns decoded size
105template<class CharSet>
106std::size_t
107re_encode_unsafe(
108 char*& dest_,
109 char const* const end,
110 core::string_view s,
111 CharSet const& unreserved,
112 encoding_opts opt) noexcept
113{
114 char const* const hex =
115 detail::hexdigs[opt.lower_case];
116 auto const encode = [end, hex](
117 char*& dest,
118 char c0) noexcept
119 {
120 auto c = static_cast<unsigned char>(c0);
121 ignore_unused(end);
122 *dest++ = '%';
123 BOOST_ASSERT(dest != end);
124 *dest++ = hex[c>>4];
125 BOOST_ASSERT(dest != end);
126 *dest++ = hex[c&0xf];
127 };
128 ignore_unused(end);
129
130 auto dest = dest_;
131 auto const dest0 = dest;
132 auto const last = s.end();
133 std::size_t dn = 0;
134 auto it = s.begin();
135
136 if(opt.space_as_plus)
137 {
138 while(it != last)
139 {
140 BOOST_ASSERT(dest != end);
141 if(*it != '%')
142 {
143 if(*it == ' ')
144 {
145 *dest++ = '+';
146 }
147 else if(unreserved(*it))
148 {
149 *dest++ = *it;
150 }
151 else
152 {
153 encode(dest, *it);
154 dn += 2;
155 }
156 ++it;
157 }
158 else
159 {
160 *dest++ = *it++;
161 BOOST_ASSERT(dest != end);
162 *dest++ = *it++;
163 BOOST_ASSERT(dest != end);
164 *dest++ = *it++;
165 dn += 2;
166 }
167 }
168 }
169 else
170 {
171 while(it != last)
172 {
173 BOOST_ASSERT(dest != end);
174 if(*it != '%')
175 {
176 if(unreserved(*it))
177 {
178 *dest++ = *it;
179 }
180 else
181 {
182 encode(dest, *it);
183 dn += 2;
184 }
185 ++it;
186 }
187 else
188 {
189 *dest++ = *it++;
190 BOOST_ASSERT(dest != end);
191 *dest++ = *it++;
192 BOOST_ASSERT(dest != end);
193 *dest++ = *it++;
194 dn += 2;
195 }
196 }
197 }
198 dest_ = dest;
199 return dest - dest0 - dn;
200}
