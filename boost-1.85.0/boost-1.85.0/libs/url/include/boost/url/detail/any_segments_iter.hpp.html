23struct BOOST_SYMBOL_VISIBLE
24 any_segments_iter
25{
26protected:
27 explicit
28 any_segments_iter(
29 core::string_view s_ = {}) noexcept
30 : s(s_)
31 {
32 }
33
34 virtual ~any_segments_iter() = default;
35
36public:
37 // this is adjusted
38 // when self-intersecting
39 core::string_view s;
40
41 // the first segment,
42 // to handle special cases
43 core::string_view front;
44
45 // quick number of segments
46 // 0 = zero
47 // 1 = one
48 // 2 = two, or more
49 int fast_nseg = 0;
50
51 // whether the segments should encode colons
52 // when we measure and copy. the calling
53 // function uses this for the first
54 // segment in some cases, such as:
55 // "x:y:z" -> remove_scheme -> "y%3Az"
56 // as "y:z" would no longer represent a path
57 bool encode_colons = false;
58
59 // Rewind the iterator to the beginning
60 virtual void rewind() noexcept = 0;
61
62 // Measure and increment the current
63 // element. n is increased by the
64 // encoded size. Returns false on
65 // end of range.
66 virtual bool measure(std::size_t& n) = 0;
67
68 // Copy and increment the current
69 // element, encoding as needed.
70 virtual void copy(char*& dest,
71 char const* end) noexcept = 0;
72};

116// iterates segments in a
117// plain segment range
118template<class FwdIt>
119struct segments_iter
120 : any_segments_iter
121 , segments_iter_base
122{
123 BOOST_STATIC_ASSERT(
124 std::is_convertible<
125 typename std::iterator_traits<
126 FwdIt>::reference,
127 core::string_view>::value);
128
129 segments_iter(
130 FwdIt first,
131 FwdIt last) noexcept
132 : it_(first)
133 , it0_(first)
134 , end_(last)
135 {
136 if(first != last)
137 {
138 front = *first;
139 auto it = first;
140 if(++it == last)
141 fast_nseg = 1;
142 else
143 fast_nseg = 2;
144 }
145 else
146 {
147 fast_nseg = 0;
148 }
149 }
150
151private:
152 FwdIt it_;
153 FwdIt it0_;
154 FwdIt end_;
155
156 void
157 rewind() noexcept override
158 {
159 it_ = it0_;
160 }
161
162 bool
163 measure(
164 std::size_t& n) noexcept override
165 {
166 if(it_ == end_)
167 return false;
168 measure_impl(n,
169 detail::to_sv(*it_),
170 encode_colons);
171 ++it_;
172 return true;
173 }
174
175 void
176 copy(
177 char*& dest,
178 char const* end) noexcept override
179 {
180 copy_impl(dest, end,
181 detail::to_sv(*it_++),
182 encode_colons);
183 }
184};

230// iterates segments in an
231// encoded segment range
232template<class FwdIt>
233struct segments_encoded_iter
234 : public any_segments_iter
235 , public segments_encoded_iter_base
236{
237 BOOST_STATIC_ASSERT(
238 std::is_convertible<
239 typename std::iterator_traits<
240 FwdIt>::reference,
241 core::string_view>::value);
242
243 segments_encoded_iter(
244 FwdIt first,
245 FwdIt last)
246 : it_(first)
247 , it0_(first)
248 , end_(last)
249 {
250 if(it_ != end_)
251 {
252 // throw on invalid input
253 front = pct_string_view(
254 detail::to_sv(*first));
255 auto it = first;
256 if(++it == last)
257 fast_nseg = 1;
258 else
259 fast_nseg = 2;
260 }
261 else
262 {
263 fast_nseg = 0;
264 }
265 }
266
267private:
268 FwdIt it_;
269 FwdIt it0_;
270 FwdIt end_;
271
272 void
273 rewind() noexcept override
274 {
275 it_ = it0_;
276 }
277
278 bool
279 measure(
280 std::size_t& n) override
281 {
282 if(it_ == end_)
283 return false;
284 // throw on invalid input
285 measure_impl(n,
286 pct_string_view(
287 detail::to_sv(*it_++)),
288 encode_colons);
289 return true;
290 }
291
292 void
293 copy(
294 char*& dest,
295 char const* end) noexcept override
296 {
297 copy_impl(dest, end,
298 detail::to_sv(*it_++),
299 encode_colons);
300 }
301};
