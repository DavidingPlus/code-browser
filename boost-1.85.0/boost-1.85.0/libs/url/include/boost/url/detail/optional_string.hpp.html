19#ifndef BOOST_URL_DOCS
20struct no_value_t;

23namespace detail {
24struct optional_string
25{
26 core::string_view s;
27 bool b = false;
28};
29
30template <class String>
31typename std::enable_if<
32 std::is_convertible<String, core::string_view>::value,
33 optional_string>::type
34get_optional_string(
35 String const& s)
36{
37 optional_string r;
38 r.s = s;
39 r.b = true;
40 return r;
41}
42
43template <class T, class = void>
44struct is_dereferenceable : std::false_type
45{};
46
47template <class T>
48struct is_dereferenceable<
49 T,
50 void_t<
51 decltype(*std::declval<T>())
52 >> : std::true_type
53{};
54
55template <class OptionalString>
56typename std::enable_if<
57 !std::is_convertible<OptionalString, core::string_view>::value,
58 optional_string>::type
59get_optional_string(
60 OptionalString const& opt)
61{
62 // If this goes off, it means the rule
63 // passed in did not meet the requirements.
64 // Please check the documentation of functions
65 // that call get_optional_string.
66 static_assert(
67 is_dereferenceable<OptionalString>::value &&
68 std::is_constructible<bool, OptionalString>::value &&
69 !std::is_convertible<OptionalString, core::string_view>::value &&
70 std::is_convertible<typename std::decay<decltype(*std::declval<OptionalString>())>::type, core::string_view>::value,
71 "OptionalString requirements not met");
72 optional_string r;
73 r.s = opt ? detail::to_sv(*opt) : core::string_view{};
74 r.b = static_cast<bool>(opt);
75 return r;
76}
77
78inline
79optional_string
80get_optional_string(
81 std::nullptr_t)
82{
83 return {};
84}
85
86inline
87optional_string
88get_optional_string(
89 no_value_t const&)
90{
91 return {};
92}
