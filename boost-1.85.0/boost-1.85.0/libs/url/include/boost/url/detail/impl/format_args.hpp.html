97// define the type-erased implementations that
98// depends on everything: the context types,
99// formatters, and type erased args
100template <class A>
101void
102format_arg::
103measure_impl(
104 format_parse_context& pctx,
105 measure_context& mctx,
106 grammar::lut_chars const& cs,
107 void const* a )
108{
109 using ref_t = typename std::remove_cv<
110 typename std::remove_reference<A>::type>::type;
111 A const& ref = *static_cast<ref_t*>(
112 const_cast<void*>( a ) );
113 formatter<ref_t> f;
114 pctx.advance_to( it: f.parse(pctx) );
115 mctx.advance_to( n: f.measure( ref, mctx, cs ) );
116}
117
118template <class A>
119void
120format_arg::
121format_impl(
122 format_parse_context& pctx,
123 format_context& fctx,
124 grammar::lut_chars const& cs,
125 void const* a )
126{
127 using ref_t = typename std::remove_cv<
128 typename std::remove_reference<A>::type>::type;
129 A const& ref = *static_cast<ref_t*>(
130 const_cast<void*>( a ) );
131 formatter<ref_t> f;
132 pctx.advance_to( it: f.parse(pctx) );
133 fctx.advance_to( it: f.format( ref, fctx, cs ) );
134}

136// We point to formatter<ignore_format> where
137// the format_arg variant would store monostate
138template <>
139struct formatter<ignore_format>
140{
141public:
142 char const*
143 parse(format_parse_context& ctx) const
144 {
145 return parse_empty_spec(
146 it: ctx.begin(), end: ctx.end());
147 }
148
149 std::size_t
150 measure(
151 ignore_format,
152 measure_context& ctx,
153 grammar::lut_chars const&) const
154 {
155 return ctx.out();
156 }
157
158 char*
159 format(
160 ignore_format,
161 format_context& ctx,
162 grammar::lut_chars const&) const
163 {
164 return ctx.out();
165 }
166
167 // We ignore the modifiers in all replacements
168 // for now
169 static
170 char const*
171 parse_empty_spec(
172 char const* it,
173 char const* end)
174 {
175 // [it, end] -> "} suffix"
176 BOOST_ASSERT(it != end);
177 ignore_unused(end);
178 // Should be always empty/valid as an
179 // implementation detail
180 BOOST_ASSERT(*it == '}');
181 /*
182 if (*it != '}')
183 urls::detail::throw_invalid_argument();
184 */
185 return it;
186 }
187};

189inline
190std::size_t
191measure_one(
192 char c,
193 grammar::lut_chars const& unreserved)
194{
195 // '%' must be reserved
196 BOOST_ASSERT(! unreserved('%'));
197 return 1 + !unreserved(c) * 2;
198}
199
200inline
201void
202encode_one(
203 char*& out,
204 char c,
205 grammar::lut_chars const& unreserved)
206{
207 // '%' must be reserved
208 BOOST_ASSERT(! unreserved('%'));
209 if(unreserved(c))
210 {
211 *out++ = c;
212 return;
213 }
214 *out++ = '%';
215 *out++ = urls::detail::hexdigs[0][c>>4];
216 *out++ = urls::detail::hexdigs[0][c&0xf];
217}
