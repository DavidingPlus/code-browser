10#ifndef BOOST_URL_DETAIL_REPLACEMENT_FIELD_RULE_HPP
11#define BOOST_URL_DETAIL_REPLACEMENT_FIELD_RULE_HPP
12
13#include <boost/url/error.hpp>
14#include <boost/core/detail/string_view.hpp>
15#include <boost/url/grammar/variant_rule.hpp>
16#include <boost/url/grammar/unsigned_rule.hpp>
17
18namespace boost {
19namespace urls {
20namespace detail {
21
22// replacement_field ::= "{" [arg_id] [":" format_spec "}"
23struct replacement_field_rule_t
24{
25 using value_type = core::string_view;
26
27 BOOST_URL_DECL
28 system::result<value_type>
29 parse(
30 char const*& it,
31 char const* end) const noexcept;
32};
33
34constexpr replacement_field_rule_t replacement_field_rule{};
35
36// identifier ::= id_start id_continue*
37// id_start ::= "a"..."z" | "A"..."Z" | "_"
38// id_continue ::= id_start | digit
39struct identifier_rule_t
40{
41 using value_type = core::string_view;
42
43 BOOST_URL_DECL
44 system::result<value_type>
45 parse(
46 char const*& it,
47 char const* end) const noexcept;
48};
49
50constexpr identifier_rule_t identifier_rule{};
51
52// arg_id ::= integer | identifier
53// integer ::= digit+
54// digit ::= "0"..."9"
55static constexpr auto arg_id_rule =
56 grammar::variant_rule(
57 r0: identifier_rule,
58 rn: grammar::unsigned_rule<std::size_t>{});
59
60struct format_spec_rule_t
61{
62 using value_type = core::string_view;
63
64 system::result<value_type>
65 parse(
66 char const*& it,
67 char const* end) const noexcept;
68};
69
70constexpr format_spec_rule_t format_spec_rule{};
71
72} // detail

74} // boost
