23// Determine if compiling as a dynamic library
24#if (defined(BOOST_URL_DYN_LINK) || defined(BOOST_ALL_DYN_LINK)) && !defined(BOOST_URL_STATIC_LINK)
25# define BOOST_URL_BUILD_DLL
26#endif
27
28// Set visibility flags
29#if !defined(BOOST_URL_BUILD_DLL)
30# define BOOST_URL_DECL /* static library */
31#elif defined(BOOST_URL_SOURCE)
32# define BOOST_URL_DECL BOOST_SYMBOL_EXPORT /* source: dllexport/visibility */
33#else
34# define BOOST_URL_DECL BOOST_SYMBOL_IMPORT /* header: dllimport */
35#endif
36
37// Set up auto-linker
38# if !defined(BOOST_URL_SOURCE) && !defined(BOOST_ALL_NO_LIB) && !defined(BOOST_URL_NO_LIB)
39# define BOOST_LIB_NAME boost_url
40# if defined(BOOST_ALL_DYN_LINK) || defined(BOOST_URL_DYN_LINK)
41# define BOOST_DYN_LINK
42# endif
43# include <boost/config/auto_link.hpp>
44# endif
45
46// Set up SSE2
47#if ! defined(BOOST_URL_NO_SSE2) && \
48 ! defined(BOOST_URL_USE_SSE2)
49# if (defined(_M_IX86) && _M_IX86_FP == 2) || \
50 defined(_M_X64) || defined(__SSE2__)
51# define BOOST_URL_USE_SSE2
52# endif
53#endif
54
55// constexpr
56#if BOOST_WORKAROUND( BOOST_GCC_VERSION, <= 72000 ) || \
57 BOOST_WORKAROUND( BOOST_CLANG_VERSION, <= 35000 )
58# define BOOST_URL_CONSTEXPR
59#else
60# define BOOST_URL_CONSTEXPR constexpr
61#endif
62
63// Add source location to error codes
64#ifdef BOOST_URL_NO_SOURCE_LOCATION
65# define BOOST_URL_ERR(ev) (::boost::system::error_code(ev))
66# define BOOST_URL_RETURN_EC(ev) return (ev)
67# define BOOST_URL_POS ::boost::source_location()
68#else
69# define BOOST_URL_ERR(ev) (::boost::system::error_code( (ev), [] { \
70 static constexpr auto loc((BOOST_CURRENT_LOCATION)); \
71 return &loc; }()))
72# define BOOST_URL_RETURN_EC(ev) \
73 static constexpr auto loc ## __LINE__((BOOST_CURRENT_LOCATION)); \
74 return ::boost::system::error_code((ev), &loc ## __LINE__)
75# define BOOST_URL_POS (BOOST_CURRENT_LOCATION)
76#endif
77
78// String token parameters
79#ifndef BOOST_URL_STRTOK_TPARAM
80#define BOOST_URL_STRTOK_TPARAM class StringToken = string_token::return_string
81#endif
82#ifndef BOOST_URL_STRTOK_RETURN
83#define BOOST_URL_STRTOK_RETURN typename StringToken::result_type
84#endif
85#ifndef BOOST_URL_STRTOK_ARG
86#define BOOST_URL_STRTOK_ARG(name) StringToken&& token = {}
87#endif
88
89// Move
90#if BOOST_WORKAROUND( BOOST_GCC_VERSION, < 80000 ) || \
91 BOOST_WORKAROUND( BOOST_CLANG_VERSION, < 30900 )
92#define BOOST_URL_RETURN(x) return std::move((x))
93#else
94#define BOOST_URL_RETURN(x) return (x)
95#endif
96
97// Limit tests
98#ifndef BOOST_URL_MAX_SIZE
99// we leave room for a null,
100// and still fit in size_t
101#define BOOST_URL_MAX_SIZE ((std::size_t(-1))-1)
102#endif
103
104// noinline attribute
105#ifdef BOOST_GCC
106#define BOOST_URL_NO_INLINE [[gnu::noinline]]
107#else
108#define BOOST_URL_NO_INLINE
109#endif
110
111// libstdcxx copy-on-write strings
112#ifndef BOOST_URL_COW_STRINGS
113#if defined(BOOST_LIBSTDCXX_VERSION) && (BOOST_LIBSTDCXX_VERSION < 60000 || (defined(_GLIBCXX_USE_CXX11_ABI) && _GLIBCXX_USE_CXX11_ABI == 0))
114#define BOOST_URL_COW_STRINGS
115#endif
116#endif
117
118// detect 32/64 bit
119#if UINTPTR_MAX == UINT64_MAX
120# define BOOST_URL_ARCH 64
121#elif UINTPTR_MAX == UINT32_MAX
122# define BOOST_URL_ARCH 32
123#else
124# error Unknown or unsupported architecture, please open an issue
125#endif

131#define BOOST_URL_DEPRECATED(msg) BOOST_DEPRECATED(msg)
132#endif
