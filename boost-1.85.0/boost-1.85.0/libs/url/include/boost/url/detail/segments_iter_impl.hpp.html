23struct segments_iter_impl

42 segments_iter_impl(

46 segments_iter_impl(

48 int) noexcept;

56 void update() noexcept;

59 void
60 increment() noexcept;
61
62 BOOST_URL_DECL
63 void
64 decrement() noexcept;
65
66 pct_string_view

72 bool
