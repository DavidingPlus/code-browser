23/** Return the buffer size needed for percent-encoding
24
25 This function returns the exact number
26 of bytes necessary to store the result
27 of applying percent-encoding to the
28 string using the given options and
29 character set.
30 No encoding is actually performed.
31
32 @par Example
33 @code
34 assert( encoded_size( "My Stuff", pchars ) == 10 );
35 @endcode
36
37 @par Exception Safety
38 Throws nothing.
39
40 @return The number of bytes needed,
41 excluding any null terminator.
42
43 @param s The string to measure.
44
45 @param unreserved The set of characters
46 that is not percent-encoded.
47
48 @param opt The options for encoding. If
49 this parameter is omitted, the default
50 options are be used.
51
52 @par Specification
53 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-2.1"
54 >2.1. Percent-Encoding (rfc3986)</a>
55
56 @see
57 @ref encode,
58 @ref encoding_opts,
59 @ref make_pct_string_view.
60*/
61template<class CharSet>
62std::size_t
63encoded_size(
64 core::string_view s,
65 CharSet const& unreserved,
66 encoding_opts opt = {}) noexcept;

70/** Apply percent-encoding to a string
71
72 This function applies percent-encoding
73 to the string using the given options and
74 character set. The destination buffer
75 provided by the caller is used to store
76 the result, which may be truncated if
77 there is insufficient space.
78
79 @par Example
80 @code
81 char buf[100];
82 assert( encode( buf, sizeof(buf), "Program Files", pchars ) == 15 );
83 @endcode
84
85 @par Exception Safety
86 Throws nothing.
87
88 @return The number of characters written
89 to the destination buffer.
90
91 @param dest The destination buffer
92 to write to.
93
94 @param size The number of writable
95 characters pointed to by `dest`.
96 If this is less than `encoded_size(s)`,
97 the result is truncated.
98
99 @param s The string to encode.
100
101 @param unreserved The set of characters
102 that is not percent-encoded.
103
104 @param opt The options for encoding. If
105 this parameter is omitted, the default
106 options are used.
107
108 @par Specification
109 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-2.1"
110 >2.1. Percent-Encoding (rfc3986)</a>
111
112 @see
113 @ref encode,
114 @ref encoded_size,
115 @ref make_pct_string_view.
116*/
117template<class CharSet>
118std::size_t
119encode(
120 char* dest,
121 std::size_t size,
122 core::string_view s,
123 CharSet const& unreserved,
124 encoding_opts opt = {});

140/** Return a percent-encoded string
141
142 This function applies percent-encoding
143 to the string using the given options and
144 character set, and returns the result as
145 a string when called with default arguments.
146
147 @par Example
148 @code
149 encoding_opts opt;
150 opt.space_as_plus = true;
151 std::string s = encode( "My Stuff", opt, pchars );
152
153 assert( s == "My+Stuff" );
154 @endcode
155
156 @par Exception Safety
157 Calls to allocate may throw.
158
159 @return The string
160
161 @param s The string to encode.
162
163 @param unreserved The set of characters
164 that is not percent-encoded.
165
166 @param opt The options for encoding. If
167 this parameter is omitted, the default
168 options are used.
169
170 @param token A string token.
171
172 @par Specification
173 @li <a href="https://datatracker.ietf.org/doc/html/rfc3986#section-2.1"
174 >2.1. Percent-Encoding (rfc3986)</a>
175
176 @see
177 @ref encode,
178 @ref encoded_size,
179 @ref encoding_opts,
180*/
181template<
182 BOOST_URL_STRTOK_TPARAM,
183 class CharSet>
184BOOST_URL_STRTOK_RETURN
185encode(
186 core::string_view s,
187 CharSet const& unreserved,
188 encoding_opts opt = {},
189 BOOST_URL_STRTOK_ARG(token)) noexcept;
