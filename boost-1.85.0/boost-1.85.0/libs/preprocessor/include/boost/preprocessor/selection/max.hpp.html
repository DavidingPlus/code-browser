21# /* BOOST_PP_MAX */
22#
23# if ~BOOST_PP_CONFIG_FLAGS() & BOOST_PP_CONFIG_EDG()
24# define BOOST_PP_MAX(x, y) BOOST_PP_IIF(BOOST_PP_LESS_EQUAL(x, y), y, x)
25# else
26# define BOOST_PP_MAX(x, y) BOOST_PP_MAX_I(x, y)
27# define BOOST_PP_MAX_I(x, y) BOOST_PP_IIF(BOOST_PP_LESS_EQUAL(x, y), y, x)
28# endif
29#
30# /* BOOST_PP_MAX_D */
31#
32# if ~BOOST_PP_CONFIG_FLAGS() & BOOST_PP_CONFIG_EDG()
33# define BOOST_PP_MAX_D(d, x, y) BOOST_PP_IIF(BOOST_PP_LESS_EQUAL_D(d, x, y), y, x)
34# else
35# define BOOST_PP_MAX_D(d, x, y) BOOST_PP_MAX_D_I(d, x, y)
36# define BOOST_PP_MAX_D_I(d, x, y) BOOST_PP_IIF(BOOST_PP_LESS_EQUAL_D(d, x, y), y, x)
37# endif
