34 //////////////////////////////////////////////////////////////////////
35 // primary template
36 //////////////////////////////////////////////////////////////////////
37 template< typename C >
38 BOOST_CONSTEXPR inline BOOST_DEDUCED_TYPENAME range_iterator<C>::type
39 range_end( C& c )
40 {
41 //
42 // If you get a compile-error here, it is most likely because
43 // you have not implemented range_begin() properly in
44 // the namespace of C
45 //
46 return c.end();
47 }
48
49 //////////////////////////////////////////////////////////////////////
50 // pair
51 //////////////////////////////////////////////////////////////////////
52
53 template< typename Iterator >
54 BOOST_CONSTEXPR inline Iterator range_end( const std::pair<Iterator,Iterator>& p )
55 {
56 return p.second;
57 }
58
59 template< typename Iterator >
60 BOOST_CONSTEXPR inline Iterator range_end( std::pair<Iterator,Iterator>& p )
61 {
62 return p.second;
63 }
64
65 //////////////////////////////////////////////////////////////////////
66 // array
67 //////////////////////////////////////////////////////////////////////
68
69 template< typename T, std::size_t sz >
70 BOOST_CONSTEXPR inline const T* range_end( const T (&a)[sz] ) BOOST_NOEXCEPT
71 {
72 return range_detail::array_end<T,sz>( a );
73 }
74
75 template< typename T, std::size_t sz >
76 BOOST_CONSTEXPR inline T* range_end( T (&a)[sz] ) BOOST_NOEXCEPT
77 {
78 return range_detail::array_end<T,sz>( a );
79 }
