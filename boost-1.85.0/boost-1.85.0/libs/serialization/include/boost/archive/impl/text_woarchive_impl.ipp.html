31// implementation of woarchive functions
32//
33template<class Archive>
34BOOST_WARCHIVE_DECL void
35text_woarchive_impl<Archive>::save(const char *s)
36{
37 // note: superfluous local variable fixes borland warning
38 const std::size_t size = std::strlen(s: s);
39 * this->This() << size;
40 this->This()->newtoken();
41 while(*s != '\0')
42 os.put(c: os.widen(c: *s++));
43}
44
45template<class Archive>
46BOOST_WARCHIVE_DECL void
47text_woarchive_impl<Archive>::save(const std::string &s)
48{
49 const std::size_t size = s.size();
50 * this->This() << size;
51 this->This()->newtoken();
52 const char * cptr = s.data();
53 for(std::size_t i = size; i-- > 0;)
54 os.put(c: os.widen(c: *cptr++));
55}

57#ifndef BOOST_NO_INTRINSIC_WCHAR_T
58template<class Archive>
59BOOST_WARCHIVE_DECL void
60text_woarchive_impl<Archive>::save(const wchar_t *ws)
61{
62 const std::size_t size = std::wostream::traits_type::length(s: ws);
63 * this->This() << size;
64 this->This()->newtoken();
65 os.write(s: ws, n: size);
66}
67#endif

73{
74 const std::size_t size = ws.length();
75 * this->This() << size;
76 this->This()->newtoken();
77 os.write(s: ws.data(), n: size);
78}
