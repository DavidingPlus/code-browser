30template<class Archive>
31BOOST_ARCHIVE_OR_WARCHIVE_DECL void
32basic_binary_oarchive<Archive>::init(){
33 // write signature in an archive version independent manner
34 const std::string file_signature(BOOST_ARCHIVE_SIGNATURE());
35 * this->This() << file_signature;
36 // write library version
37 const boost::serialization::library_version_type v(BOOST_ARCHIVE_VERSION());
