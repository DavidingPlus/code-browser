33// note: referred to as Curiously Recurring Template Patter (CRTP)
34template<class Archive>
35
36class BOOST_SYMBOL_VISIBLE common_oarchive :
37 public basic_oarchive,
38 public interface_oarchive<Archive>
39{
40 friend class interface_oarchive<Archive>;
41 friend class basic_oarchive;
42private:
43 void vsave(const version_type t) BOOST_OVERRIDE {
44 * this->This() << t;
45 }
46 void vsave(const object_id_type t) BOOST_OVERRIDE {
47 * this->This() << t;
48 }
49 void vsave(const object_reference_type t) BOOST_OVERRIDE {
50 * this->This() << t;
51 }
52 void vsave(const class_id_type t) BOOST_OVERRIDE {
53 * this->This() << t;
54 }
55 void vsave(const class_id_reference_type t) BOOST_OVERRIDE {
56 * this->This() << t;
57 }
58 void vsave(const class_id_optional_type t) BOOST_OVERRIDE {
59 * this->This() << t;
60 }
61 void vsave(const class_name_type & t) BOOST_OVERRIDE {
62 * this->This() << t;
63 }
64 void vsave(const tracking_type t) BOOST_OVERRIDE {
65 * this->This() << t;
66 }
67protected:
68 // default processing - invoke serialization library
69 template<class T>
70 void save_override(T & t){
71 archive::save(* this->This(), t);
72 }
73 void save_start(const char * /*name*/){}
74 void save_end(const char * /*name*/){}
75 common_oarchive(unsigned int flags = 0) :
76 basic_oarchive(flags),
77 interface_oarchive<Archive>()
78 {}
79};
