95BOOST_ARCHIVE_DECL const basic_serializer *
96basic_serializer_map::find(
97 const boost::serialization::extended_type_info & eti
98) const {
99 const basic_serializer_arg bs(eti);
100 map_type::const_iterator it;
101 it = m_map.find(x: & bs);
102 if(it == m_map.end()){
103 BOOST_ASSERT(false);
104 return 0;
105 }
106 return *it;
107}
