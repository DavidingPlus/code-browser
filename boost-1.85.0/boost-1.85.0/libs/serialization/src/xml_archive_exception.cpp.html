28BOOST_ARCHIVE_DECL
29xml_archive_exception::xml_archive_exception(
30 exception_code c,
31 const char * e1,
32 const char * e2
33 ) :
34 archive_exception(other_exception, e1, e2)
35 {
36 switch(c){
37 case xml_archive_parsing_error:
38 archive_exception::append(l: 0, a: "unrecognized XML syntax");
39 break;
40 case xml_archive_tag_mismatch:{
41 unsigned int l;
42 l = archive_exception::append(l: 0, a: "XML start/end tag mismatch");
43 if(NULL != e1){
44 l = archive_exception::append(l, a: " - ");
45 archive_exception::append(l, a: e1);
46 }
47 break;
48 }
49 case xml_archive_tag_name_error:
50 archive_exception::append(l: 0, a: "Invalid XML tag name");
51 break;
52 default:
53 BOOST_ASSERT(false);
54 archive_exception::append(l: 0, a: "programming error");
55 break;
56 }
57 }
