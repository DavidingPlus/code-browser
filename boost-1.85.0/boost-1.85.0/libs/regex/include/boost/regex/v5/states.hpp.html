25/*** mask_type *******************************************************
26Whenever we have a choice of two alternatives, we use an array of bytes
27to indicate which of the two alternatives it is possible to take for any
28given input character. If mask_take is set, then we can take the next
29state, and if mask_skip is set then we can take the alternative.
30***********************************************************************/
31enum mask_type
32{
33 mask_take = 1,
34 mask_skip = 2,
35 mask_init = 4,
36 mask_any = mask_skip | mask_take,
37 mask_all = mask_any
38};

40/*** helpers **********************************************************
41These helpers let us use function overload resolution to detect whether
42we have narrow or wide character strings:
43***********************************************************************/
44struct _narrow_type{};
45struct _wide_type{};
46template <class charT> struct is_byte;
47template<> struct is_byte<char> { typedef _narrow_type width_type; };
48template<> struct is_byte<unsigned char>{ typedef _narrow_type width_type; };
49template<> struct is_byte<signed char> { typedef _narrow_type width_type; };
50template <class charT> struct is_byte { typedef _wide_type width_type; };

52/*** enum syntax_element_type ******************************************
53Every record in the state machine falls into one of the following types:
54***********************************************************************/
55enum syntax_element_type
56{
57 // start of a marked sub-expression, or perl-style (?...) extension
58 syntax_element_startmark = 0,
59 // end of a marked sub-expression, or perl-style (?...) extension
60 syntax_element_endmark = syntax_element_startmark + 1,
61 // any sequence of literal characters
62 syntax_element_literal = syntax_element_endmark + 1,
63 // start of line assertion: ^
64 syntax_element_start_line = syntax_element_literal + 1,
65 // end of line assertion $
66 syntax_element_end_line = syntax_element_start_line + 1,
67 // match any character: .
68 syntax_element_wild = syntax_element_end_line + 1,
69 // end of expression: we have a match when we get here
70 syntax_element_match = syntax_element_wild + 1,
71 // perl style word boundary: \b
72 syntax_element_word_boundary = syntax_element_match + 1,
73 // perl style within word boundary: \B
74 syntax_element_within_word = syntax_element_word_boundary + 1,
75 // start of word assertion: \<
76 syntax_element_word_start = syntax_element_within_word + 1,
77 // end of word assertion: \>
78 syntax_element_word_end = syntax_element_word_start + 1,
79 // start of buffer assertion: \`
80 syntax_element_buffer_start = syntax_element_word_end + 1,
81 // end of buffer assertion: \'
82 syntax_element_buffer_end = syntax_element_buffer_start + 1,
83 // backreference to previously matched sub-expression
84 syntax_element_backref = syntax_element_buffer_end + 1,
85 // either a wide character set [..] or one with multicharacter collating elements:
86 syntax_element_long_set = syntax_element_backref + 1,
87 // narrow character set: [...]
88 syntax_element_set = syntax_element_long_set + 1,
89 // jump to a new state in the machine:
90 syntax_element_jump = syntax_element_set + 1,
91 // choose between two production states:
92 syntax_element_alt = syntax_element_jump + 1,
93 // a repeat
94 syntax_element_rep = syntax_element_alt + 1,
95 // match a combining character sequence
96 syntax_element_combining = syntax_element_rep + 1,
97 // perl style soft buffer end: \z
98 syntax_element_soft_buffer_end = syntax_element_combining + 1,
99 // perl style continuation: \G
100 syntax_element_restart_continue = syntax_element_soft_buffer_end + 1,
101 // single character repeats:
102 syntax_element_dot_rep = syntax_element_restart_continue + 1,
103 syntax_element_char_rep = syntax_element_dot_rep + 1,
104 syntax_element_short_set_rep = syntax_element_char_rep + 1,
105 syntax_element_long_set_rep = syntax_element_short_set_rep + 1,
106 // a backstep for lookbehind repeats:
107 syntax_element_backstep = syntax_element_long_set_rep + 1,
108 // an assertion that a mark was matched:
109 syntax_element_assert_backref = syntax_element_backstep + 1,
110 syntax_element_toggle_case = syntax_element_assert_backref + 1,
111 // a recursive expression:
112 syntax_element_recurse = syntax_element_toggle_case + 1,
113 // Verbs:
114 syntax_element_fail = syntax_element_recurse + 1,
115 syntax_element_accept = syntax_element_fail + 1,
116 syntax_element_commit = syntax_element_accept + 1,
117 syntax_element_then = syntax_element_commit + 1
118};

127/*** union offset_type ************************************************
128Points to another state in the machine. During machine construction
129we use integral offsets, but these are converted to pointers before
130execution of the machine.
131***********************************************************************/
132union offset_type
133{
134 re_syntax_base* p;
135 std::ptrdiff_t i;
136};
137
138/*** struct re_syntax_base ********************************************
139Base class for all states in the machine.
140***********************************************************************/
141struct re_syntax_base
142{
143 syntax_element_type type; // what kind of state this is
144 offset_type next; // next state in the machine
145};
146
147/*** struct re_brace **************************************************
148A marked parenthesis.
149***********************************************************************/
150struct re_brace : public re_syntax_base
151{
152 // The index to match, can be zero (don't mark the sub-expression)
153 // or negative (for perl style (?...) extensions):
154 int index;
155 bool icase;
156};
157
158/*** struct re_dot **************************************************
159Match anything.
160***********************************************************************/
161enum
162{
163 dont_care = 1,
164 force_not_newline = 0,
165 force_newline = 2,
166
167 test_not_newline = 2,
168 test_newline = 3
169};
170struct re_dot : public re_syntax_base
171{
172 unsigned char mask;
173};
174
175/*** struct re_literal ************************************************
176A string of literals, following this structure will be an
177array of characters: charT[length]
178***********************************************************************/
179struct re_literal : public re_syntax_base
180{
181 unsigned int length;
182};
183
184/*** struct re_case ************************************************
185Indicates whether we are moving to a case insensive block or not
186***********************************************************************/
187struct re_case : public re_syntax_base
188{
189 bool icase;
190};
191
192/*** struct re_set_long ***********************************************
193A wide character set of characters, following this structure will be
194an array of type charT:
195First csingles null-terminated strings
196Then 2 * cranges NULL terminated strings
197Then cequivalents NULL terminated strings
198***********************************************************************/
199template <class mask_type>
200struct re_set_long : public re_syntax_base
201{
202 unsigned int csingles, cranges, cequivalents;
203 mask_type cclasses;
204 mask_type cnclasses;
205 bool isnot;
206 bool singleton;
207};
208
209/*** struct re_set ****************************************************
210A set of narrow-characters, matches any of _map which is none-zero
211***********************************************************************/
212struct re_set : public re_syntax_base
213{
214 unsigned char _map[1 << CHAR_BIT];
215};
216
217/*** struct re_jump ***************************************************
218Jump to a new location in the machine (not next).
219***********************************************************************/
220struct re_jump : public re_syntax_base
221{
222 offset_type alt; // location to jump to
223};
224
225/*** struct re_alt ***************************************************
226Jump to a new location in the machine (possibly next).
227***********************************************************************/
228struct re_alt : public re_jump
229{
230 unsigned char _map[1 << CHAR_BIT]; // which characters can take the jump
231 unsigned int can_be_null; // true if we match a NULL string
232};
233
234/*** struct re_repeat *************************************************
235Repeat a section of the machine
236***********************************************************************/
237struct re_repeat : public re_alt
238{
239 std::size_t min, max; // min and max allowable repeats
240 int state_id; // Unique identifier for this repeat
241 bool leading; // True if this repeat is at the start of the machine (lets us optimize some searches)
242 bool greedy; // True if this is a greedy repeat
243};
244
245/*** struct re_recurse ************************************************
246Recurse to a particular subexpression.
247**********************************************************************/
248struct re_recurse : public re_jump
249{
250 int state_id; // identifier of first nested repeat within the recursion.
251};
252
253/*** struct re_commit *************************************************
254Used for the PRUNE, SKIP and COMMIT verbs which basically differ only in what happens
255if no match is found and we start searching forward.
256**********************************************************************/
257enum commit_type
258{
259 commit_prune,
260 commit_skip,
261 commit_commit
262};
263struct re_commit : public re_syntax_base
264{
265 commit_type action;
266};
267
268/*** enum re_jump_size_type *******************************************
269Provides compiled size of re_jump structure (allowing for trailing alignment).
270We provide this so we know how manybytes to insert when constructing the machine
271(The value of padding_mask is defined in regex_raw_buffer.hpp).
272***********************************************************************/
273enum re_jump_size_type
274{
275 re_jump_size = (sizeof(re_jump) + padding_mask) & ~(padding_mask),
276 re_repeater_size = (sizeof(re_repeat) + padding_mask) & ~(padding_mask),
277 re_alt_size = (sizeof(re_alt) + padding_mask) & ~(padding_mask)
278};
