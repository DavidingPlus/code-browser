31//////////////////////////////////////////////////////////////////////////
32// function template apply_visitor(visitor, visitable1, visitable2)
33//
34// Visits visitable1 and visitable2 such that their values (which we
35// shall call x and y, respectively) are used as arguments in the
36// expression visitor(x, y).
37//
38
39namespace detail { namespace variant {
40
41template <typename Visitor, typename Value1, bool MoveSemantics>
42class apply_visitor_binary_invoke
43{
44public: // visitor typedefs
45
46 typedef typename Visitor::result_type
47 result_type;
48
49private: // representation
50
51 Visitor& visitor_;
52 Value1& value1_;
53
54public: // structors
55
56 apply_visitor_binary_invoke(Visitor& visitor, Value1& value1) BOOST_NOEXCEPT
57 : visitor_(visitor)
58 , value1_(value1)
59 {
60 }
61
62public: // visitor interfaces
63
64 template <typename Value2>
65 typename enable_if_c<MoveSemantics && is_same<Value2, Value2>::value, result_type>::type
66 operator()(Value2&& value2)
67 {
68 return visitor_(std::move(value1_), std::forward<Value2>(value2));
69 }
70
71 template <typename Value2>
72 typename disable_if_c<MoveSemantics && is_same<Value2, Value2>::value, result_type>::type
73 operator()(Value2&& value2)
74 {
75 return visitor_(value1_, std::forward<Value2>(value2));
76 }
77
78private:
79 apply_visitor_binary_invoke& operator=(const apply_visitor_binary_invoke&);
80};
81
82template <typename Visitor, typename Visitable2, bool MoveSemantics>
83class apply_visitor_binary_unwrap
84{
85public: // visitor typedefs
86
87 typedef typename Visitor::result_type
88 result_type;
89
90private: // representation
91
92 Visitor& visitor_;
93 Visitable2& visitable2_;
94
95public: // structors
96
97 apply_visitor_binary_unwrap(Visitor& visitor, Visitable2& visitable2) BOOST_NOEXCEPT
98 : visitor_(visitor)
99 , visitable2_(visitable2)
100 {
101 }
102
103public: // visitor interfaces
104
105 template <typename Value1>
106 typename enable_if_c<MoveSemantics && is_same<Value1, Value1>::value, result_type>::type
107 operator()(Value1&& value1)
108 {
109 apply_visitor_binary_invoke<
110 Visitor
111 , Value1
112 , ! ::boost::is_lvalue_reference<Value1>::value
113 > invoker(visitor_, value1);
114
115 return boost::apply_visitor(invoker, std::move(visitable2_));
116 }
117
118 template <typename Value1>
119 typename disable_if_c<MoveSemantics && is_same<Value1, Value1>::value, result_type>::type
120 operator()(Value1&& value1)
121 {
122 apply_visitor_binary_invoke<
123 Visitor
124 , Value1
125 , ! ::boost::is_lvalue_reference<Value1>::value
126 > invoker(visitor_, value1);
127
128 return boost::apply_visitor(invoker, visitable2_);
129 }
130
131private:
132 apply_visitor_binary_unwrap& operator=(const apply_visitor_binary_unwrap&);
133
134};

138//
139// nonconst-visitor version:
140//
141
142template <typename Visitor, typename Visitable1, typename Visitable2>
143inline typename Visitor::result_type
144apply_visitor( Visitor& visitor, Visitable1&& visitable1, Visitable2&& visitable2)
145{
146 ::boost::detail::variant::apply_visitor_binary_unwrap<
147 Visitor, Visitable2, ! ::boost::is_lvalue_reference<Visitable2>::value
148 > unwrapper(visitor, visitable2);
149
150 return boost::apply_visitor(unwrapper, std::forward<Visitable1>(visitable1));
151}
