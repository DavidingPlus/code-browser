30///////////////////////////////////////////////////////////////////////////////
31// (detail) metafunction substitute
32//
33// Substitutes one type for another in the given type expression.
34//
35
36//
37// primary template
38//
39template <
40 typename T, typename Dest, typename Source
41 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(
42 typename Arity /* = ... (see substitute_fwd.hpp) */
43 )
44 >
45struct substitute
46{
47 typedef T type;
48};
49
50//
51// tag substitution specializations
52//
53
54#define BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG(CV_) \
55 template <typename Dest, typename Source> \
56 struct substitute< \
57 CV_ Source \
58 , Dest \
59 , Source \
60 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(mpl::int_<-1>) \
61 > \
62 { \
63 typedef CV_ Dest type; \
64 }; \
65 /**/
66
67BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG( BOOST_PP_EMPTY() )
68BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG(const)
69BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG(volatile)
70BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG(const volatile)
71
72#undef BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_SUBSTITUTE_TAG
73
74//
75// pointer specializations
76//
77#define BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER(CV_) \
78 template <typename T, typename Dest, typename Source> \
79 struct substitute< \
80 T * CV_ \
81 , Dest \
82 , Source \
83 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(mpl::int_<-1>) \
84 > \
85 { \
86 typedef typename substitute< \
87 T, Dest, Source \
88 >::type * CV_ type; \
89 }; \
90 /**/
91
92BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER( BOOST_PP_EMPTY() )
93BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER(const)
94BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER(volatile)
95BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER(const volatile)
96
97#undef BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL_HANDLE_POINTER
98
99//
100// reference specializations
101//
102template <typename T, typename Dest, typename Source>
103struct substitute<
104 T&
105 , Dest
106 , Source
107 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(mpl::int_<-1>)
108 >
109{
110 typedef typename substitute<
111 T, Dest, Source
112 >::type & type;
113};
114
115//
116// template expression (i.e., F<...>) specializations
117//
118
119template <
120 template <typename...> class F
121 , typename... Ts
122 , typename Dest
123 , typename Source
124 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(typename Arity)
125 >
126struct substitute<
127 F<Ts...>
128 , Dest
129 , Source
130 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(Arity)
131 >
132{
133 typedef F<typename substitute<
134 Ts, Dest, Source
135 >::type...> type;
136};
137
138//
139// function specializations
140//
141template <
142 typename R
143 , typename... A
144 , typename Dest
145 , typename Source
146 >
147struct substitute<
148 R (*)(A...)
149 , Dest
150 , Source
151 BOOST_MPL_AUX_LAMBDA_ARITY_PARAM(mpl::int_<-1>)
152 >
153{
154private:
155 typedef typename substitute< R, Dest, Source >::type r;
156
157public:
158 typedef r (*type)(typename substitute<
159 A, Dest, Source
160 >::type...);
161};
