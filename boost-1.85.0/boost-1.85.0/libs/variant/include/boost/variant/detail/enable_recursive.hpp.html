43///////////////////////////////////////////////////////////////////////////////
44// (detail) metafunction enable_recursive
45//
46// See boost/variant/detail/enable_recursive_fwd.hpp for more information.
47//
48
49
50template <typename T, typename RecursiveVariant, typename NoWrapper>
51struct enable_recursive
52 : BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL(
53 T, RecursiveVariant, ::boost::recursive_variant_
54 )
55{
56};
57
58template <typename T, typename RecursiveVariant>
59struct enable_recursive< T,RecursiveVariant,mpl::false_ >
60{
61private: // helpers, for metafunction result (below)
62
63 typedef typename BOOST_VARIANT_AUX_ENABLE_RECURSIVE_IMPL(
64 T, RecursiveVariant, ::boost::recursive_variant_
65 )::type t_;
66
67public: // metafunction result
68
69 // [Wrap with recursive_wrapper only if rebind really changed something:]
70 typedef typename mpl::if_<
71 mpl::or_<
72 is_same< t_,T >
73 , is_reference<t_>
74 , is_pointer<t_>
75 >
76 , t_
77 , boost::recursive_wrapper<t_>
78 >::type type;
79
80};
