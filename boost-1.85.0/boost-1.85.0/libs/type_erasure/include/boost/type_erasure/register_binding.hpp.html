51template<class Map>
52struct append_to_key_static {
53 append_to_key_static(key_type* k) : key(k) {}
54 template<class P>
55 void operator()(P) {
56#ifndef BOOST_TYPE_ERASURE_USE_MP11
57 key->push_back(&typeid(typename ::boost::mpl::at<Map, P>::type));
58#else
59 key->push_back(x: &typeid(::boost::mp11::mp_second< ::boost::mp11::mp_map_find<Map, P> >));
60#endif
61 }
62 key_type* key;
63};

65// This placeholder exists solely to create a normalized
66// representation of a primitive concept. For the moment
67// I'm going to be conservative and require a bijection
68// between the original placeholders and the normalized
69// placeholders. It should be safe to map everything
70// to a single placeholder, though, as long as the
71// key includes every instance of each placeholder
72// as a separate element. i.e. we should be able to
73// turn addable<_a, _b> into addable<_, _> and
74// addable<_a, _a> into addable<_, _> as well if we always
75// add typeids for both arguments to the search key.
76template<int N>
77struct _ : ::boost::type_erasure::placeholder {};

79struct counting_map_appender
80{
81 template<class State, class Key>
82 struct apply
83 {
84 typedef typename ::boost::mpl::insert<
85 State,
86 ::boost::mpl::pair<
87 Key,
88 ::boost::type_erasure::detail::_<
89 ::boost::mpl::size<State>::value
90 >
91 >
92 >::type type;
93 };
94};

96template<class Map>
97struct register_function {
98 template<class F>
99 void operator()(F) {
100 key_type key;
101#ifndef BOOST_TYPE_ERASURE_USE_MP11
102 typedef typename ::boost::type_erasure::detail::get_placeholders<F, ::boost::mpl::set0<> >::type placeholders;
103#else
104 typedef typename ::boost::type_erasure::detail::get_placeholders<F, ::boost::mp11::mp_list<> >::type placeholders;
105#endif
106 typedef typename ::boost::mpl::fold<
107 placeholders,
108 ::boost::mpl::map0<>,
109 ::boost::type_erasure::detail::counting_map_appender
110 >::type placeholder_map;
111 key.push_back(x: &typeid(typename ::boost::type_erasure::detail::rebind_placeholders<F, placeholder_map>::type));
112 ::boost::mpl::for_each<placeholders>(append_to_key_static<Map>(&key));
113 value_type fn = reinterpret_cast<value_type>(&::boost::type_erasure::detail::rebind_placeholders<F, Map>::type::value);
114 ::boost::type_erasure::detail::register_function_impl(key, fn);
115 }
116};

120/**
121 * Registers a model of a concept to allow downcasting @ref any
122 * via \dynamic_any_cast.
123 */
124template<class Concept, class Map>
125void register_binding(const static_binding<Map>&)
126{
127 typedef typename ::boost::type_erasure::detail::normalize_concept<
128 Concept
129 >::type normalized;
130 typedef typename ::boost::mpl::transform<normalized,
131 ::boost::type_erasure::detail::maybe_adapt_to_vtable< ::boost::mpl::_1>
132 >::type actual_concept;
133 typedef typename ::boost::type_erasure::detail::get_placeholder_normalization_map<
134 Concept
135 >::type placeholder_subs;
136 typedef typename ::boost::type_erasure::detail::add_deductions<Map, placeholder_subs>::type actual_map;
137 ::boost::mpl::for_each<actual_concept>(::boost::type_erasure::detail::register_function<actual_map>());
138}

140/**
141 * \overload
142 */
143template<class Concept, class T>
144void register_binding()
145{
146 // Find all placeholders
147 typedef typename ::boost::type_erasure::detail::normalize_concept_impl<Concept>::type normalized;
148 typedef typename normalized::first basic;
149 typedef typename ::boost::mpl::fold<
150 basic,
151#ifndef BOOST_TYPE_ERASURE_USE_MP11
152 ::boost::mpl::set0<>,
153#else
154 ::boost::mp11::mp_list<>,
155#endif
156 ::boost::type_erasure::detail::get_placeholders< ::boost::mpl::_2, ::boost::mpl::_1>
157 >::type all_placeholders;
158 // remove deduced placeholders
159 typedef typename ::boost::mpl::fold<
160 typename normalized::second,
161 ::boost::mpl::set0<>,
162 ::boost::mpl::insert< ::boost::mpl::_1, ::boost::mpl::second< ::boost::mpl::_2> >
163 >::type xtra_deduced;
164 typedef typename ::boost::mpl::remove_if<
165 all_placeholders,
166 ::boost::mpl::or_<
167 ::boost::type_erasure::detail::is_deduced< ::boost::mpl::_1>,
168 ::boost::mpl::has_key<xtra_deduced, ::boost::mpl::_1>
169 >,
170 ::boost::mpl::back_inserter< ::boost::mpl::vector0<> >
171 >::type unknown_placeholders;
172 // Bind the single remaining placeholder to T
173 BOOST_MPL_ASSERT((boost::mpl::equal_to<boost::mpl::size<unknown_placeholders>, boost::mpl::int_<1> >));
174 register_binding<Concept>(::boost::type_erasure::make_binding<
175 ::boost::mpl::map< ::boost::mpl::pair<typename ::boost::mpl::front<unknown_placeholders>::type, T> > >());
176}
