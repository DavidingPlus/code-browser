20namespace boost {
21namespace type_erasure {

23#ifndef BOOST_TYPE_ERASURE_DOXYGEN
24template<class Concept, class T>
25class any;
26#endif
27
28/**
29 * A metafunction that changes the @ref placeholder of
30 * an @ref any. If @c T is not a placeholder,
31 * returns @c T unchanged. This class is intended
32 * to be used in @ref concept_interface to deduce
33 * the argument types from the arguments of the concept.
34 *
35 * @pre Any must be a specialization of @ref any or a base
36 * class of such a specialization.
37 *
38 * \code
39 * rebind_any<any<Concept>, _a>::type -> any<Concept, _a>
40 * rebind_any<any<Concept>, _b&>::type -> any<Concept, _b&>
41 * rebind_any<any<Concept>, _c&&>::type -> any<Concept, _c&&>
42 * rebind_any<any<Concept>, int>::type -> int
43 * \endcode
44 *
45 * @see derived, as_param
46 */
47template<class Any, class T>
48struct rebind_any
49{
50#ifdef BOOST_TYPE_ERASURE_DOXYGEN
51 typedef detail::unspecified type;
52#else
53 typedef typename ::boost::mpl::if_<
54 ::boost::type_erasure::is_placeholder<
55 typename ::boost::remove_cv<
56 typename ::boost::remove_reference<T>::type
57 >::type
58 >,
59 ::boost::type_erasure::any<
60 typename ::boost::type_erasure::concept_of<Any>::type,
61 T
62 >,
63 T
64 >::type type;
65#endif
66};
67
68#ifndef BOOST_NO_CXX11_TEMPLATE_ALIASES
69
70template<class Any, class T>
71using rebind_any_t = typename ::boost::type_erasure::rebind_any<Any, T>::type;
72
73#endif
74
75}
76}

78#endif
