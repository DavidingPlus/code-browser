63template<class T>
64struct rebind_placeholders_in_argument_impl
65{
66 template<class Bindings>
67 using apply = ::boost::type_erasure::detail::eval_if<
68 ::boost::mp11::mp_map_contains<Bindings, T>::value,
69 ::boost::type_erasure::detail::rebind_one_placeholder,
70 ::boost::type_erasure::detail::first,
71 T,
72 Bindings
73 >;
74};

178template<class T, class Bindings>
179struct rebind_placeholders_in_argument
180{
181 BOOST_MPL_ASSERT((boost::mpl::or_<
182 ::boost::mpl::not_< ::boost::type_erasure::is_placeholder<T> >,
183 ::boost::mpl::has_key<Bindings, T>
184 >));
185 typedef typename ::boost::mpl::eval_if<
186 ::boost::type_erasure::is_placeholder<T>,
187 ::boost::mpl::at<Bindings, T>,
188 ::boost::type_erasure::detail::identity<T>
189 >::type type;
190};
