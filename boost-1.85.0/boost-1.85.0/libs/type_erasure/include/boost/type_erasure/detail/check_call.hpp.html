39template<class T, class Enable = void>
40struct qualified_placeholder
41{
42 typedef void type;
43};
44
45template<class T>
46struct qualified_placeholder<T&, typename T::_boost_type_erasure_is_any>
47{
48 typedef typename ::boost::type_erasure::placeholder_of<T>::type placeholder;
49 typedef typename ::boost::remove_reference<placeholder>::type unref;
50 typedef typename ::boost::mpl::if_< ::boost::is_const<T>,
51 const unref,
52 unref
53 >::type add_const;
54 typedef typename ::boost::mpl::if_< ::boost::is_reference<placeholder>,
55 placeholder,
56 add_const&
57 >::type type;
58};

133template<class FormalArg, class ActualArg>

135 ::boost::type_erasure::detail::eval_if<

137 ::boost::remove_cv_t<
138 ::boost::remove_reference_t<FormalArg>

163template<class FormalArg, class ActualArg>
164struct check_arg
165{
166 typedef typename ::boost::mpl::eval_if<
167 is_placeholder<
168 typename ::boost::remove_cv<
169 typename ::boost::remove_reference<FormalArg>::type
170 >::type
171 >,
172 ::boost::type_erasure::detail::check_placeholder_arg<FormalArg, ActualArg>,
173#ifdef BOOST_TYPE_ERASURE_BROKEN_RVALUE_IS_CONVERTIBLE
174 ::boost::mpl::true_
175#else
176 ::boost::is_convertible<ActualArg, FormalArg>
177#endif
178 >::type type;
179};

222#define BOOST_TYPE_ERASURE_CHECK_ARG(z, n, data) \
223 typedef typename ::boost::type_erasure::detail::check_arg< \
224 BOOST_PP_CAT(T, n), \
225 BOOST_PP_CAT(U, n) \
226 >::type BOOST_PP_CAT(check, n); \
227 typedef typename ::boost::mpl::and_< \
228 BOOST_PP_CAT(type, n), \
229 BOOST_PP_CAT(check, n) \
230 >::type BOOST_PP_CAT(type, BOOST_PP_INC(n));
231
232template<
233 class R
234 BOOST_PP_ENUM_TRAILING_PARAMS(N, class T)
235 BOOST_PP_ENUM_TRAILING_PARAMS(N, class U)
236>
237struct check_call<R(BOOST_PP_ENUM_PARAMS(N, T)), void(BOOST_PP_ENUM_BINARY_PARAMS(N, U, u))> {
238 typedef ::boost::mpl::true_ type0;
239 BOOST_PP_REPEAT(N, BOOST_TYPE_ERASURE_CHECK_ARG, ~)
240 typedef BOOST_PP_CAT(type, N) type;
241};
