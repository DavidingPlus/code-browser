57//! The class template rbtree is an intrusive red-black tree container, that
58//! is used to construct intrusive set and multiset containers. The no-throw
59//! guarantee holds only, if the key_compare object
60//! doesn't throw.
61//!
62//! The template parameter \c T is the type to be managed by the container.
63//! The user can specify additional options and if no options are provided
64//! default options are used.
65//!
66//! The container supports the following options:
67//! \c base_hook<>/member_hook<>/value_traits<>,
68//! \c constant_time_size<>, \c size_type<> and
69//! \c compare<>.

71template<class T, class ...Options>
72#else

74#endif

466//! Helper metafunction to define a \c rbtree that yields to the same type when the
467//! same options (either explicitly or implicitly) are used.
468#if defined(BOOST_INTRUSIVE_DOXYGEN_INVOKED) || defined(BOOST_INTRUSIVE_VARIADIC_TEMPLATES)
469template<class T, class ...Options>
470#else
471template<class T, class O1 = void, class O2 = void
472 , class O3 = void, class O4 = void
473 , class O5 = void, class O6 = void>
474#endif
475struct make_rbtree
476{
477 /// @cond
478 typedef typename pack_options
479 < rbtree_defaults,
480 #if !defined(BOOST_INTRUSIVE_VARIADIC_TEMPLATES)
481 O1, O2, O3, O4, O5, O6
482 #else
483 Options...
484 #endif
485 >::type packed_options;
486
487 typedef typename detail::get_value_traits
488 <T, typename packed_options::proto_value_traits>::type value_traits;
489
490 typedef rbtree_impl
491 < value_traits
492 , typename packed_options::key_of_value
493 , typename packed_options::compare
494 , typename packed_options::size_type
495 , packed_options::constant_time_size
496 , typename packed_options::header_holder_type
497 > implementation_defined;
498 /// @endcond
499 typedef implementation_defined type;
500};
