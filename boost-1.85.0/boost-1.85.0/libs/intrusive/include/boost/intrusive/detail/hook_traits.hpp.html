35template<class T, class NodePtr, class Tag, unsigned int Type>
36struct bhtraits_base
37{
38 public:
39 typedef NodePtr node_ptr;
40 typedef typename pointer_traits<node_ptr>::element_type node;
41 typedef node_holder<node, Tag, Type> node_holder_type;
42 typedef T value_type;
43 typedef typename pointer_traits<node_ptr>::
44 template rebind_pointer<const node>::type const_node_ptr;
45 typedef typename pointer_traits<node_ptr>::
46 template rebind_pointer<T>::type pointer;
47 typedef typename pointer_traits<node_ptr>::
48 template rebind_pointer<const T>::type const_pointer;
49 typedef T & reference;
50 typedef const T & const_reference;
51 typedef node_holder_type & node_holder_reference;
52 typedef const node_holder_type & const_node_holder_reference;
53 typedef node& node_reference;
54 typedef const node & const_node_reference;
55
56 inline static pointer to_value_ptr(node_ptr n)
57 {
58 pointer p = pointer_traits<pointer>::pointer_to
59 (static_cast<reference>(static_cast<node_holder_reference>(*n)));
60 return p;
61 }
62
63 inline static const_pointer to_value_ptr(const_node_ptr n)
64 {
65 const_pointer p = pointer_traits<const_pointer>::pointer_to
66 (static_cast<const_reference>(static_cast<const_node_holder_reference>(*n)));
67 return p;
68 }
69
70 inline static node_ptr to_node_ptr(reference value)
71 {
72 node_ptr p = pointer_traits<node_ptr>::pointer_to
73 (static_cast<node_reference>(static_cast<node_holder_reference>(value)));
74 return p;
75 }
76
77 inline static const_node_ptr to_node_ptr(const_reference value)
78 {
79 const_node_ptr p = pointer_traits<const_node_ptr>::pointer_to
80 (static_cast<const_node_reference>(static_cast<const_node_holder_reference>(value)));
81 return p;
82 }
83};

85template<class T, class NodeTraits, link_mode_type LinkMode, class Tag, unsigned int Type>
86struct bhtraits
87 : public bhtraits_base<T, typename NodeTraits::node_ptr, Tag, Type>
88{
89 static const link_mode_type link_mode = LinkMode;
90 typedef NodeTraits node_traits;
91};

94template<class T, class Hook, Hook T::* P>
95struct mhtraits
96{
97 public:
98 typedef Hook hook_type;
99 typedef typename hook_type::hooktags::node_traits node_traits;
100 typedef typename node_traits::node node;
101 typedef T value_type;
102 typedef typename node_traits::node_ptr node_ptr;
103 typedef typename node_traits::const_node_ptr const_node_ptr;
104 typedef typename pointer_traits<node_ptr>::
105 template rebind_pointer<T>::type pointer;
106 typedef typename pointer_traits<node_ptr>::
107 template rebind_pointer<const T>::type const_pointer;
108 typedef T & reference;
109 typedef const T & const_reference;
110 typedef node& node_reference;
111 typedef const node & const_node_reference;
112 typedef hook_type& hook_reference;
113 typedef const hook_type & const_hook_reference;
114
115 static const link_mode_type link_mode = Hook::hooktags::link_mode;
116
117 inline static node_ptr to_node_ptr(reference value)
118 {
119 return pointer_traits<node_ptr>::pointer_to
120 (static_cast<node_reference>(static_cast<hook_reference>(value.*P)));
121 }
122
123 inline static const_node_ptr to_node_ptr(const_reference value)
124 {
125 return pointer_traits<const_node_ptr>::pointer_to
126 (static_cast<const_node_reference>(static_cast<const_hook_reference>(value.*P)));
127 }
128
129 inline static pointer to_value_ptr(node_ptr n)
130 {
131 return pointer_traits<pointer>::pointer_to
132 (*detail::parent_from_member<T, Hook>
133 (static_cast<Hook*>(boost::movelib::to_raw_pointer(n)), P));
134 }
135
136 inline static const_pointer to_value_ptr(const_node_ptr n)
137 {
138 return pointer_traits<const_pointer>::pointer_to
139 (*detail::parent_from_member<T, Hook>
140 (static_cast<const Hook*>(boost::movelib::to_raw_pointer(n)), P));
141 }
142};

145template<class Functor>
146struct fhtraits
147{
148 public:
149 typedef typename Functor::hook_type hook_type;
150 typedef typename Functor::hook_ptr hook_ptr;
151 typedef typename Functor::const_hook_ptr const_hook_ptr;
152 typedef typename hook_type::hooktags::node_traits node_traits;
153 typedef typename node_traits::node node;
154 typedef typename Functor::value_type value_type;
155 typedef typename node_traits::node_ptr node_ptr;
156 typedef typename node_traits::const_node_ptr const_node_ptr;
157 typedef typename pointer_traits<node_ptr>::
158 template rebind_pointer<value_type>::type pointer;
159 typedef typename pointer_traits<node_ptr>::
160 template rebind_pointer<const value_type>::type const_pointer;
161 typedef value_type & reference;
162 typedef const value_type & const_reference;
163 static const link_mode_type link_mode = hook_type::hooktags::link_mode;
164
165 inline static node_ptr to_node_ptr(reference value)
166 { return static_cast<node*>(boost::movelib::to_raw_pointer(Functor::to_hook_ptr(value))); }
167
168 inline static const_node_ptr to_node_ptr(const_reference value)
169 { return static_cast<const node*>(boost::movelib::to_raw_pointer(Functor::to_hook_ptr(value))); }
170
171 inline static pointer to_value_ptr(node_ptr n)
172 { return Functor::to_value_ptr(to_hook_ptr(n)); }
173
174 inline static const_pointer to_value_ptr(const_node_ptr n)
175 { return Functor::to_value_ptr(to_hook_ptr(n)); }
176
177 private:
178 inline static hook_ptr to_hook_ptr(node_ptr n)
179 { return hook_ptr(&*static_cast<hook_type*>(&*n)); }
180
181 inline static const_hook_ptr to_hook_ptr(const_node_ptr n)
182 { return const_hook_ptr(&*static_cast<const hook_type*>(&*n)); }
183};
