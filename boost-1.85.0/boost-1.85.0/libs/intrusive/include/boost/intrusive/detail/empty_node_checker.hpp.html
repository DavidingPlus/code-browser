28template<class ValueTraits>
29struct empty_node_checker
30{
31 typedef ValueTraits value_traits;
32 typedef typename value_traits::node_traits node_traits;
33 typedef typename node_traits::const_node_ptr const_node_ptr;
34
35 struct return_type {};
36
37 void operator () (const_node_ptr, const return_type&, const return_type&, return_type&) {}
38};
