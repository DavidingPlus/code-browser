161template<typename T, typename Tag = void, bool = is_unary_or_binary_function<T>::value>
162class ebo_functor_holder
163{
164 BOOST_COPYABLE_AND_MOVABLE(ebo_functor_holder)
165
166 public:
167 typedef T functor_type;
168
169 inline ebo_functor_holder()
170 : t_()
171 {}
172
173 inline explicit ebo_functor_holder(const T &t)
174 : t_(t)
175 {}
176
177 inline explicit ebo_functor_holder(BOOST_RV_REF(T) t)
178 : t_(::boost::move(t))
179 {}
180
181 template<class Arg1, class Arg2>
182 inline ebo_functor_holder(BOOST_FWD_REF(Arg1) arg1, BOOST_FWD_REF(Arg2) arg2)
183 : t_(::boost::forward<Arg1>(arg1), ::boost::forward<Arg2>(arg2))
184 {}
185
186 inline ebo_functor_holder(const ebo_functor_holder &x)
187 : t_(x.t_)
188 {}
189
190 inline ebo_functor_holder(BOOST_RV_REF(ebo_functor_holder) x)
191 : t_(x.t_)
192 {}
193
194 inline ebo_functor_holder& operator=(BOOST_COPY_ASSIGN_REF(ebo_functor_holder) x)
195 {
196 this->get() = x.get();
197 return *this;
198 }
199
200 inline ebo_functor_holder& operator=(BOOST_RV_REF(ebo_functor_holder) x)
201 {
202 this->get() = ::boost::move(x.get());
203 return *this;
204 }
205
206 inline ebo_functor_holder& operator=(const T &x)
207 {
208 this->get() = x;
209 return *this;
210 }
211
212 inline ebo_functor_holder& operator=(BOOST_RV_REF(T) x)
213 {
214 this->get() = ::boost::move(x);
215 return *this;
216 }
217
218 inline T& get(){return t_;}
219 inline const T& get()const{return t_;}
220
221 private:
222 T t_;
223};

225template<typename T, typename Tag>
226class ebo_functor_holder<T, Tag, false>
227 : public T
228{
229 BOOST_COPYABLE_AND_MOVABLE(ebo_functor_holder)
230
231 public:
232 typedef T functor_type;
233
234 inline ebo_functor_holder()
235 : T()
236 {}
237
238 inline explicit ebo_functor_holder(const T &t)
239 : T(t)
240 {}
241
242 inline explicit ebo_functor_holder(BOOST_RV_REF(T) t)
243 : T(::boost::move(t))
244 {}
245
246 template<class Arg1, class Arg2>
247 inline ebo_functor_holder(BOOST_FWD_REF(Arg1) arg1, BOOST_FWD_REF(Arg2) arg2)
248 : T(::boost::forward<Arg1>(arg1), ::boost::forward<Arg2>(arg2))
249 {}
250
251 inline ebo_functor_holder(const ebo_functor_holder &x)
252 : T(static_cast<const T&>(x))
253 {}
254
255 inline ebo_functor_holder(BOOST_RV_REF(ebo_functor_holder) x)
256 : T(BOOST_MOVE_BASE(T, x))
257 {}
258
259 inline ebo_functor_holder& operator=(BOOST_COPY_ASSIGN_REF(ebo_functor_holder) x)
260 {
261 const ebo_functor_holder&r = x;
262 this->get() = r;
263 return *this;
264 }
265
266 inline ebo_functor_holder& operator=(BOOST_RV_REF(ebo_functor_holder) x)
267 {
268 this->get() = ::boost::move(x.get());
269 return *this;
270 }
271
272 inline ebo_functor_holder& operator=(const T &x)
273 {
274 this->get() = x;
275 return *this;
276 }
277
278 inline ebo_functor_holder& operator=(BOOST_RV_REF(T) x)
279 {
280 this->get() = ::boost::move(x);
281 return *this;
282 }
283
284 inline T& get(){return *this;}
285 inline const T& get()const{return *this;}
286};
