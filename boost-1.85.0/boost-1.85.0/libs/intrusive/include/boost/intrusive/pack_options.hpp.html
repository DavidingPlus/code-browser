126template<std::size_t I, typename Tuple>
127struct typelist_element;
128
129template<std::size_t I, typename Head, typename... Tail>
130struct typelist_element<I, typelist<Head, Tail...> >
131{
132 typedef typename typelist_element<I-1, typelist<Tail...> >::type type;
133};

135template<typename Head, typename... Tail>
136struct typelist_element<0, typelist<Head, Tail...> >
137{
138 typedef Head type;
139};

141template<std::size_t ...Ints, class ...Types>
142typelist<typename typelist_element<(sizeof...(Types) - 1) - Ints, typelist<Types...> >::type...>
143 inverted_typelist(index_tuple<Ints...>, typelist<Types...>)
144{
145 return typelist<typename typelist_element<(sizeof...(Types) - 1) - Ints, typelist<Types...> >::type...>();
146}

163template<class Typelist, std::size_t ...Ints>
164struct invert_typelist_impl< Typelist, index_tuple<Ints...> >
165{
166 static const std::size_t last_idx = sizeof_typelist<Typelist>::value - 1;
167 typedef typelist
168 <typename typelist_element<last_idx - Ints, Typelist>::type...> type;
169};

171template<class Typelist, std::size_t Int>
172struct invert_typelist_impl< Typelist, index_tuple<Int> >
173{
174 typedef Typelist type;

183//invert_typelist
184template<class Typelist>
185struct invert_typelist;
186
187template<class ...Types>
188struct invert_typelist< typelist<Types...> >
189{
190 typedef typelist<Types...> typelist_t;
191 typedef typename build_number_seq<sizeof...(Types)>::type indexes_t;
192 typedef typename invert_typelist_impl<typelist_t, indexes_t>::type type;
193};

239template< class TYPE> \
240struct OPTION_NAME \
241{ \
242 template<class Base> \
243 struct pack : Base \
244 { \
245 typedef TYPEDEF_EXPR TYPEDEF_NAME; \

247}; \

266//! This class is a utility that takes:
267//! - a default options class defining initial static constant
268//! and typedefs
269//! - several options defined with BOOST_INTRUSIVE_OPTION_CONSTANT and
270//! BOOST_INTRUSIVE_OPTION_TYPE
271//!
272//! and packs them together in a new type that defines all options as
273//! member typedefs or static constant values. Given options of form:
274//!
275//! \code
276//! BOOST_INTRUSIVE_OPTION_TYPE(my_pointer, VoidPointer, VoidPointer, my_pointer_type)
277//! BOOST_INTRUSIVE_OPTION_CONSTANT(incremental, bool, Enabled, is_incremental)
278//! \endcode
279//!
280//! the following expression
281//!
282//! \code
283//!
284//! struct default_options
285//! {
286//! typedef long int_type;
287//! static const int int_constant = -1;
288//! };
289//!
290//! pack_options< default_options, my_pointer<void*>, incremental<true> >::type
291//! \endcode
292//!
293//! will create a type that will contain the following typedefs/constants
294//!
295//! \code
296//! struct unspecified_type
297//! {
298//! //Default options
299//! typedef long int_type;
300//! static const int int_constant = -1;
301//!
302//! //Packed options (will ovewrite any default option)
303//! typedef void* my_pointer_type;
304//! static const bool is_incremental = true;
305//! };
306//! \endcode
307//!
308//! If an option is specified in the default options argument and later
309//! redefined as an option, the last definition will prevail.
310template<class DefaultOptions, class ...Options>
311struct pack_options
312{
313 typedef unspecified_type type;
314};
315
316//! Defines an option class of name OPTION_NAME that can be used to specify a type
317//! of type TYPE...
318//!
319//! \code
320//! struct OPTION_NAME<class TYPE>
321//! { unspecified_content };
322//! \endcode
323//!
324//! ...that after being combined with
325//! <code>boost::intrusive::pack_options</code>,
326//! will typedef TYPE as a typedef of name TYPEDEF_NAME. Example:
327//!
328//! \code
329//! //[includes and namespaces omitted for brevity]
330//!
331//! //This macro will create the following class:
332//! // template<class VoidPointer>
333//! // struct my_pointer
334//! // { unspecified_content };
335//! BOOST_INTRUSIVE_OPTION_TYPE(my_pointer, VoidPointer, boost::remove_pointer<VoidPointer>::type, my_pointer_type)
336//!
337//! struct empty_default{};
338//!
339//! typedef pack_options< empty_default, typename my_pointer<void*> >::type::my_pointer_type type;
340//!
341//! BOOST_INTRUSIVE_STATIC_ASSERT(( boost::is_same<type, void>::value ));
342//!
343//! \endcode
344#define BOOST_INTRUSIVE_OPTION_TYPE(OPTION_NAME, TYPE, TYPEDEF_EXPR, TYPEDEF_NAME)
345
346//! Defines an option class of name OPTION_NAME that can be used to specify a constant
347//! of type TYPE with value VALUE...
348//!
349//! \code
350//! struct OPTION_NAME<TYPE VALUE>
351//! { unspecified_content };
352//! \endcode
353//!
354//! ...that after being combined with
355//! <code>boost::intrusive::pack_options</code>,
356//! will contain a CONSTANT_NAME static constant of value VALUE. Example:
357//!
358//! \code
359//! //[includes and namespaces omitted for brevity]
360//!
361//! //This macro will create the following class:
362//! // template<bool Enabled>
363//! // struct incremental
364//! // { unspecified_content };
365//! BOOST_INTRUSIVE_OPTION_CONSTANT(incremental, bool, Enabled, is_incremental)
366//!
367//! struct empty_default{};
368//!
369//! const bool is_incremental = pack_options< empty_default, incremental<true> >::type::is_incremental;
370//!
371//! BOOST_INTRUSIVE_STATIC_ASSERT(( is_incremental == true ));
372//!
373//! \endcode
374#define BOOST_INTRUSIVE_OPTION_CONSTANT(OPTION_NAME, TYPE, VALUE, CONSTANT_NAME)
