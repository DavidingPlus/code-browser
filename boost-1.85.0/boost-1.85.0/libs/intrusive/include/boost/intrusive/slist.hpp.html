96//! The class template slist is an intrusive container, that encapsulates
97//! a singly-linked list. You can use such a list to squeeze the last bit
98//! of performance from your application. Unfortunately, the little gains
99//! come with some huge drawbacks. A lot of member functions can't be
100//! implemented as efficiently as for standard containers. To overcome
101//! this limitation some other member functions with rather unusual semantics
102//! have to be introduced.
103//!
104//! The template parameter \c T is the type to be managed by the container.
105//! The user can specify additional options and if no options are provided
106//! default options are used.
107//!
108//! The container supports the following options:
109//! \c base_hook<>/member_hook<>/value_traits<>,
110//! \c constant_time_size<>, \c size_type<>,
111//! \c linear<> and \c cache_last<>.
112//!
113//! The iterators of slist are forward iterators. slist provides a static
114//! function called "previous" to compute the previous iterator of a given iterator.
115//! This function has linear complexity. To improve the usability esp. with
116//! the '*_after' functions, ++end() == begin() and previous(begin()) == end()
117//! are defined. An new special function "before_begin()" is defined, which returns
118//! an iterator that points one less the beginning of the list: ++before_begin() == begin()
119#if defined(BOOST_INTRUSIVE_DOXYGEN_INVOKED)
120template<class T, class ...Options>
121#else
122template<class ValueTraits, class SizeType, std::size_t BoolFlags, typename HeaderHolder>
123#endif

267 //! <b>Requires</b>: f and before_l belong to another slist.
268 //!
269 //! <b>Effects</b>: Transfers the range [f, before_l] to this
270 //! list, after the element pointed by prev_pos.
271 //! No destructors or copy constructors are called.
272 //!
273 //! <b>Throws</b>: Nothing.
274 //!
275 //! <b>Complexity</b>: Linear to the number of elements transferred
276 //! if constant_time_size is true. Constant-time otherwise.
277 //!
278 //! <b>Note</b>: Iterators of values obtained from list x now point to elements of this
279 //! list. Iterators of this list and all the references are not invalidated.
280 //!
281 //! <b>Warning</b>: Experimental function, don't use it!
282 slist_impl( node_ptr f, node_ptr before_l
283 , size_type n, const value_traits &v_traits = value_traits())
284 : data_(v_traits)
285 {
286 if(n){
287 this->priv_size_traits().set_size(n);
288 BOOST_IF_CONSTEXPR(cache_last){
289 this->set_last_node(before_l);
290 }
291 node_traits::set_next(this->get_root_node(), f);
292 node_traits::set_next(before_l, this->get_end_node());
293 }
294 else{
295 this->set_default_constructed_state();
296 }
297 }

338 //! <b>Effects</b>: Constructs a container moving resources from another container.
339 //! Internal value traits are move constructed and
340 //! nodes belonging to x (except the node representing the "end") are linked to *this.
341 //!
342 //! <b>Complexity</b>: Constant.
343 //!
344 //! <b>Throws</b>: If value_traits::node_traits::node's
345 //! move constructor throws (this does not happen with predefined Boost.Intrusive hooks)

360 //! <b>Effects</b>: If it's a safe-mode
361 //! or auto-unlink value, the destructor does nothing
362 //! (ie. no code is generated). Otherwise it detaches all elements from this.

377 //! <b>Effects</b>: Erases all the elements of the container.
378 //!
379 //! <b>Throws</b>: Nothing.

382 //! if it's a safe-mode or auto-unlink value_type. Constant time otherwise.

395 //! <b>Requires</b>: Disposer::operator()(pointer) shouldn't throw.
396 //!
397 //! <b>Effects</b>: Erases all the elements of the container
398 //! Disposer::operator()(pointer) is called for the removed elements.
399 //!
400 //! <b>Throws</b>: Nothing.
401 //!
402 //! <b>Complexity</b>: Linear to the number of elements of the list.
403 //!
404 //! <b>Note</b>: Invalidates the iterators to the erased elements.
405 template <class Disposer>
406 void clear_and_dispose(Disposer disposer) BOOST_NOEXCEPT
407 {
408 node_algorithms::detach_and_dispose
409 ( this->get_root_node()
410 , detail::node_disposer<Disposer, value_traits, CommonSListAlgorithms>
411 (disposer, &this->priv_value_traits()) );
412 this->set_default_constructed_state();
413 }

415 //! <b>Requires</b>: value must be an lvalue.
416 //!
417 //! <b>Effects</b>: Inserts the value in the front of the list.
418 //! No copy constructors are called.
419 //!
420 //! <b>Throws</b>: Nothing.
421 //!
422 //! <b>Complexity</b>: Constant.
423 //!
424 //! <b>Note</b>: Does not affect the validity of iterators and references.
425 void push_front(reference value) BOOST_NOEXCEPT
426 {
427 node_ptr to_insert = priv_value_traits().to_node_ptr(value);
428 BOOST_INTRUSIVE_SAFE_HOOK_DEFAULT_ASSERT(!safemode_or_autounlink || node_algorithms::inited(to_insert));
429 BOOST_IF_CONSTEXPR(cache_last){
430 if(this->empty()){
431 this->set_last_node(to_insert);
432 }
433 }
434 node_algorithms::link_after(this->get_root_node(), to_insert);
435 this->priv_size_traits().increment();
436 }

439 //!
440 //! <b>Effects</b>: Inserts the value in the back of the list.
441 //! No copy constructors are called.
442 //!
443 //! <b>Throws</b>: Nothing.
444 //!
445 //! <b>Complexity</b>: Constant.

485 node_ptr to_erase = node_traits::get_next(this->get_root_node());

487 this->priv_size_traits().decrement();
488 BOOST_IF_CONSTEXPR(safemode_or_autounlink)
489 node_algorithms::init(to_erase);
490 disposer(priv_value_traits().to_value_ptr(to_erase));
491 BOOST_IF_CONSTEXPR(cache_last){
492 if(this->empty()){
493 this->set_last_node(this->get_root_node());
494 }
495 }
496 }
