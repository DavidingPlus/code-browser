30//! circular_list_algorithms provides basic algorithms to manipulate nodes
31//! forming a circular doubly linked list. An empty circular list is formed by a node
32//! whose pointers point to itself.
33//!
34//! circular_list_algorithms is configured with a NodeTraits class, which encapsulates the
35//! information about the node to be manipulated. NodeTraits must support the
36//! following interface:
37//!
38//! <b>Typedefs</b>:
39//!
40//! <tt>node</tt>: The type of the node that forms the circular list
41//!
42//! <tt>node_ptr</tt>: A pointer to a node
43//!
44//! <tt>const_node_ptr</tt>: A pointer to a const node
45//!
46//! <b>Static functions</b>:
47//!
48//! <tt>static node_ptr get_previous(const_node_ptr n);</tt>
49//!
50//! <tt>static void set_previous(node_ptr n, node_ptr prev);</tt>
51//!
52//! <tt>static node_ptr get_next(const_node_ptr n);</tt>
53//!
54//! <tt>static void set_next(node_ptr n, node_ptr next);</tt>
55template<class NodeTraits>
56class circular_list_algorithms
57{
58 public:
59 typedef typename NodeTraits::node node;
60 typedef typename NodeTraits::node_ptr node_ptr;
61 typedef typename NodeTraits::const_node_ptr const_node_ptr;
62 typedef NodeTraits node_traits;
63
64 //! <b>Effects</b>: Constructs an non-used list element, so that
65 //! inited(this_node) == true
66 //!
67 //! <b>Complexity</b>: Constant
68 //!
69 //! <b>Throws</b>: Nothing.
70 static void init(node_ptr this_node) BOOST_NOEXCEPT
71 {
72 const node_ptr null_node = node_ptr();
73 NodeTraits::set_next(this_node, null_node);
74 NodeTraits::set_previous(this_node, null_node);
75 }
76
77 //! <b>Effects</b>: Returns true is "this_node" is in a non-used state
78 //! as if it was initialized by the "init" function.
79 //!
80 //! <b>Complexity</b>: Constant
81 //!
82 //! <b>Throws</b>: Nothing.
83 inline static bool inited(const_node_ptr this_node) BOOST_NOEXCEPT
84 { return !NodeTraits::get_next(this_node); }
85
86 //! <b>Effects</b>: Constructs an empty list, making this_node the only
87 //! node of the circular list:
88 //! <tt>NodeTraits::get_next(this_node) == NodeTraits::get_previous(this_node)
89 //! == this_node</tt>.
90 //!
91 //! <b>Complexity</b>: Constant
92 //!
93 //! <b>Throws</b>: Nothing.
94 static void init_header(node_ptr this_node) BOOST_NOEXCEPT
95 {
96 NodeTraits::set_next(this_node, this_node);
97 NodeTraits::set_previous(this_node, this_node);
98 }
99
100 //! <b>Effects</b>: Returns true if this_node_points to an empty list.
101 //!
102 //! <b>Complexity</b>: Constant
103 //!
104 //! <b>Throws</b>: Nothing.
105 inline static bool is_empty(const_node_ptr this_node) BOOST_NOEXCEPT
106 {
107 return NodeTraits::get_next(this_node) == this_node;
108 }
109
110 //! <b>Requires</b>: this_node must be in a circular list or be an empty circular list.
111 //!
112 //! <b>Effects</b>: Returns true is "this_node" is the only node of a circular list:
113 //! <tt>return NodeTraits::get_next(this_node) == this_node</tt>
114 //!
115 //! <b>Complexity</b>: Constant
116 //!
117 //! <b>Throws</b>: Nothing.
118 static bool unique(const_node_ptr this_node) BOOST_NOEXCEPT
119 {
120 node_ptr next = NodeTraits::get_next(this_node);
121 return !next || next == this_node;
122 }
123
124 //! <b>Requires</b>: this_node must be in a circular list or be an empty circular list.
125 //!
126 //! <b>Effects</b>: Returns the number of nodes in a circular list. If the circular list
127 //! is empty, returns 1.
128 //!
129 //! <b>Complexity</b>: Linear
130 //!
131 //! <b>Throws</b>: Nothing.
132 static std::size_t count(const_node_ptr this_node) BOOST_NOEXCEPT
133 {
134 std::size_t result = 0;
135 const_node_ptr p = this_node;
136 do{
137 p = NodeTraits::get_next(p);
138 ++result;
139 }while (p != this_node);
140 return result;
141 }
142
143 //! <b>Requires</b>: this_node must be in a circular list or be an empty circular list.
144 //!
145 //! <b>Effects</b>: Unlinks the node from the circular list.
146 //!
147 //! <b>Complexity</b>: Constant
148 //!
149 //! <b>Throws</b>: Nothing.
150 static node_ptr unlink(node_ptr this_node) BOOST_NOEXCEPT
151 {
152 node_ptr next(NodeTraits::get_next(this_node));
153 node_ptr prev(NodeTraits::get_previous(this_node));
154 NodeTraits::set_next(prev, next);
155 NodeTraits::set_previous(next, prev);
156 return next;
157 }
158
159 //! <b>Requires</b>: b and e must be nodes of the same circular list or an empty range.
160 //!
161 //! <b>Effects</b>: Unlinks the node [b, e) from the circular list.
162 //!
163 //! <b>Complexity</b>: Constant
164 //!
165 //! <b>Throws</b>: Nothing.
166 static void unlink(node_ptr b, node_ptr e) BOOST_NOEXCEPT
167 {
168 if (b != e) {
169 node_ptr prevb(NodeTraits::get_previous(b));
170 NodeTraits::set_previous(e, prevb);
171 NodeTraits::set_next(prevb, e);
172 }
173 }
174
175 //! <b>Requires</b>: nxt_node must be a node of a circular list.
176 //!
177 //! <b>Effects</b>: Links this_node before nxt_node in the circular list.
178 //!
179 //! <b>Complexity</b>: Constant
180 //!
181 //! <b>Throws</b>: Nothing.
182 static void link_before(node_ptr nxt_node, node_ptr this_node) BOOST_NOEXCEPT
183 {
184 node_ptr prev(NodeTraits::get_previous(nxt_node));
185 NodeTraits::set_previous(this_node, prev);
186 NodeTraits::set_next(this_node, nxt_node);
187 //nxt_node might be an alias for prev->next_
188 //so use it before NodeTraits::set_next(prev, ...)
189 //is called and the reference changes its value
190 NodeTraits::set_previous(nxt_node, this_node);
191 NodeTraits::set_next(prev, this_node);
192 }
193
194 //! <b>Requires</b>: prev_node must be a node of a circular list.
195 //!
196 //! <b>Effects</b>: Links this_node after prev_node in the circular list.
197 //!
198 //! <b>Complexity</b>: Constant
199 //!
200 //! <b>Throws</b>: Nothing.
201 static void link_after(node_ptr prev_node, node_ptr this_node) BOOST_NOEXCEPT
202 {
203 node_ptr next(NodeTraits::get_next(prev_node));
204 NodeTraits::set_previous(this_node, prev_node);
205 NodeTraits::set_next(this_node, next);
206 //prev_node might be an alias for next->next_
207 //so use it before update it before NodeTraits::set_previous(next, ...)
208 //is called and the reference changes it's value
209 NodeTraits::set_next(prev_node, this_node);
210 NodeTraits::set_previous(next, this_node);
211 }
212
213 //! <b>Requires</b>: this_node and other_node must be nodes inserted
214 //! in circular lists or be empty circular lists.
215 //!
216 //! <b>Effects</b>: Swaps the position of the nodes: this_node is inserted in
217 //! other_nodes position in the second circular list and the other_node is inserted
218 //! in this_node's position in the first circular list.
219 //!
220 //! <b>Complexity</b>: Constant
221 //!
222 //! <b>Throws</b>: Nothing.
223 static void swap_nodes(node_ptr this_node, node_ptr other_node) BOOST_NOEXCEPT
224 {
225 if (other_node == this_node)
226 return;
227 bool this_inited = inited(this_node);
228 bool other_inited = inited(this_node: other_node);
229 if(this_inited){
230 init_header(this_node);
231 }
232 if(other_inited){
233 init_header(this_node: other_node);
234 }
235
236 node_ptr next_this(NodeTraits::get_next(this_node));
237 node_ptr prev_this(NodeTraits::get_previous(this_node));
238 node_ptr next_other(NodeTraits::get_next(other_node));
239 node_ptr prev_other(NodeTraits::get_previous(other_node));
240 //these first two swaps must happen before the other two
241 swap_prev(this_node: next_this, other_node: next_other);
242 swap_next(this_node: prev_this, other_node: prev_other);
243 swap_next(this_node, other_node);
244 swap_prev(this_node, other_node);
245
246 if(this_inited){
247 init(this_node: other_node);
248 }
249 if(other_inited){
250 init(this_node);
251 }
252 }
253
254 //! <b>Requires</b>: b and e must be nodes of the same circular list or an empty range.
255 //! and p must be a node of a different circular list or may not be an iterator in
256 // [b, e).
257 //!
258 //! <b>Effects</b>: Removes the nodes from [b, e) range from their circular list and inserts
259 //! them before p in p's circular list.
260 //!
261 //! <b>Complexity</b>: Constant
262 //!
263 //! <b>Throws</b>: Nothing.
264 static void transfer(node_ptr p, node_ptr b, node_ptr e) BOOST_NOEXCEPT
265 {
266 if (b != e && p != b && p != e) {
267 node_ptr prev_p(NodeTraits::get_previous(p));
268 node_ptr prev_b(NodeTraits::get_previous(b));
269 node_ptr prev_e(NodeTraits::get_previous(e));
270 NodeTraits::set_next(prev_e, p);
271 NodeTraits::set_previous(p, prev_e);
272 NodeTraits::set_next(prev_b, e);
273 NodeTraits::set_previous(e, prev_b);
274 NodeTraits::set_next(prev_p, b);
275 NodeTraits::set_previous(b, prev_p);
276 }
277 }
278
279 //! <b>Requires</b>: i must a node of a circular list
280 //! and p must be a node of a different circular list.
281 //!
282 //! <b>Effects</b>: Removes the node i from its circular list and inserts
283 //! it before p in p's circular list.
284 //! If p == i or p == NodeTraits::get_next(i), this function is a null operation.
285 //!
286 //! <b>Complexity</b>: Constant
287 //!
288 //! <b>Throws</b>: Nothing.
289 static void transfer(node_ptr p, node_ptr i) BOOST_NOEXCEPT
290 {
291 node_ptr n(NodeTraits::get_next(i));
292 if(n != p && i != p){
293 node_ptr prev_p(NodeTraits::get_previous(p));
294 node_ptr prev_i(NodeTraits::get_previous(i));
295 NodeTraits::set_next(prev_p, i);
296 NodeTraits::set_previous(i, prev_p);
297 NodeTraits::set_next(i, p);
298 NodeTraits::set_previous(p, i);
299 NodeTraits::set_previous(n, prev_i);
300 NodeTraits::set_next(prev_i, n);
301
302 }
303 }
304
305 //! <b>Effects</b>: Reverses the order of elements in the list.
306 //!
307 //! <b>Throws</b>: Nothing.
308 //!
309 //! <b>Complexity</b>: This function is linear time.
310 static void reverse(node_ptr p) BOOST_NOEXCEPT
311 {
312 node_ptr f(NodeTraits::get_next(p));
313 node_ptr i(NodeTraits::get_next(f)), e(p);
314
315 while(i != e) {
316 node_ptr n = i;
317 i = NodeTraits::get_next(i);
318 transfer(f, n, i);
319 f = n;
320 }
321 }
322
323 //! <b>Effects</b>: Moves the node p n positions towards the end of the list.
324 //!
325 //! <b>Throws</b>: Nothing.
326 //!
327 //! <b>Complexity</b>: Linear to the number of moved positions.
328 static void move_backwards(node_ptr p, std::size_t n) BOOST_NOEXCEPT
329 {
330 //Null shift, nothing to do
331 if(!n) return;
332 node_ptr first = NodeTraits::get_next(p);
333 //size() == 0 or 1, nothing to do
334 if(first == NodeTraits::get_previous(p)) return;
335 unlink(p);
336 //Now get the new first node
337 while(n--){
338 first = NodeTraits::get_next(first);
339 }
340 link_before(nxt_node: first, this_node: p);
341 }
342
343 //! <b>Effects</b>: Moves the node p n positions towards the beginning of the list.
344 //!
345 //! <b>Throws</b>: Nothing.
346 //!
347 //! <b>Complexity</b>: Linear to the number of moved positions.
348 static void move_forward(node_ptr p, std::size_t n) BOOST_NOEXCEPT
349 {
350 //Null shift, nothing to do
351 if(!n) return;
352 node_ptr last = NodeTraits::get_previous(p);
353 //size() == 0 or 1, nothing to do
354 if(last == NodeTraits::get_next(p)) return;
355
356 unlink(p);
357 //Now get the new last node
358 while(n--){
359 last = NodeTraits::get_previous(last);
360 }
361 link_after(prev_node: last, this_node: p);
362 }
363
364 //! <b>Requires</b>: f and l must be in a circular list.
365 //!
366 //! <b>Effects</b>: Returns the number of nodes in the range [f, l).
367 //!
368 //! <b>Complexity</b>: Linear
369 //!
370 //! <b>Throws</b>: Nothing.
371 static std::size_t distance(const_node_ptr f, const_node_ptr l) BOOST_NOEXCEPT
372 {
373 std::size_t result = 0;
374 while(f != l){
375 f = NodeTraits::get_next(f);
376 ++result;
377 }
378 return result;
379 }

381 struct stable_partition_info
382 {
383 std::size_t num_1st_partition;
384 std::size_t num_2nd_partition;
385 node_ptr beg_2st_partition;
386 };
387
388 template<class Pred>
389 static void stable_partition(node_ptr beg, node_ptr end, Pred pred, stable_partition_info &info)
390 {
391 node_ptr bcur = node_traits::get_previous(beg);
392 node_ptr cur = beg;
393 node_ptr new_f = end;
394
395 std::size_t num1 = 0, num2 = 0;
396 while(cur != end){
397 if(pred(cur)){
398 ++num1;
399 bcur = cur;
400 cur = node_traits::get_next(cur);
401 }
402 else{
403 ++num2;
404 node_ptr last_to_remove = bcur;
405 new_f = cur;
406 bcur = cur;
407 cur = node_traits::get_next(cur);
408 BOOST_INTRUSIVE_TRY{
409 //Main loop
410 while(cur != end){
411 if(pred(cur)){ //Might throw
412 ++num1;
413 //Process current node
414 node_traits::set_next (last_to_remove, cur);
415 node_traits::set_previous(cur, last_to_remove);
416 last_to_remove = cur;
417 node_ptr nxt = node_traits::get_next(cur);
418 node_traits::set_next (bcur, nxt);
419 node_traits::set_previous(nxt, bcur);
420 cur = nxt;
421 }
422 else{
423 ++num2;
424 bcur = cur;
425 cur = node_traits::get_next(cur);
426 }
427 }
428 }
429 BOOST_INTRUSIVE_CATCH(...){
430 node_traits::set_next (last_to_remove, new_f);
431 node_traits::set_previous(new_f, last_to_remove);
432 BOOST_INTRUSIVE_RETHROW;
433 }
434 BOOST_INTRUSIVE_CATCH_END
435 node_traits::set_next(last_to_remove, new_f);
436 node_traits::set_previous(new_f, last_to_remove);
437 break;
438 }
439 }
440 info.num_1st_partition = num1;
441 info.num_2nd_partition = num2;
442 info.beg_2st_partition = new_f;
443 }
444
445 private:
446 static void swap_prev(node_ptr this_node, node_ptr other_node) BOOST_NOEXCEPT
447 {
448 node_ptr temp(NodeTraits::get_previous(this_node));
449 NodeTraits::set_previous(this_node, NodeTraits::get_previous(other_node));
450 NodeTraits::set_previous(other_node, temp);
451 }
452
453 static void swap_next(node_ptr this_node, node_ptr other_node) BOOST_NOEXCEPT
454 {
455 node_ptr temp(NodeTraits::get_next(this_node));
456 NodeTraits::set_next(this_node, NodeTraits::get_next(other_node));
457 NodeTraits::set_next(other_node, temp);
458 }
459};
