6// Define fiber_properties, a base class from which a library consumer can
7// derive a subclass with specific properties important to a user-coded
8// scheduler.
9
10#ifndef BOOST_FIBERS_PROPERTIES_HPP
11#define BOOST_FIBERS_PROPERTIES_HPP

25namespace boost {
26namespace fibers {
27
28class context;
29
30namespace algo {
31
32class algorithm;
33
34}
35
36class BOOST_FIBERS_DECL fiber_properties {
37protected:
38 // initialized by constructor
39 context * ctx_;
40 // set every time this fiber becomes READY
41 algo::algorithm * algo_{ nullptr };
42
43 // Inform the relevant algorithm instance that something important
44 // has changed, so it can (presumably) adjust its data structures
45 // accordingly.
46 void notify() noexcept;
47
48public:
49 // Any specific property setter method, after updating the relevant
50 // instance variable, can/should call notify().
51
52 // fiber_properties, and by implication every subclass, must accept a back
53 // pointer to its context.
54
55 // For fiber_properties passed to fiber constructors, nullptr must be
56 // used here.
57 explicit fiber_properties( context * ctx) noexcept :
58 ctx_{ ctx } {
59 }
60
61 // We need a virtual destructor (hence a vtable) because fiber_properties
62 // is stored polymorphically (as fiber_properties*) in context, and
63 // destroyed via that pointer.
64 virtual ~fiber_properties() = default;
65
66 // not really intended for public use, but algorithm_with_properties
67 // must be able to call this
68 void set_algorithm( algo::algorithm * algo) noexcept {
69 algo_ = algo;
70 }
71
72 // not really intended for public use, but required to set properties
73 // on fiber/context construction.
74 void set_context( context* ctx ) noexcept {
75 BOOST_ASSERT( ctx_ == nullptr );
76 BOOST_ASSERT( ctx != nullptr );
77 ctx_ = ctx;
78 }
79};
80
81}} // namespace boost::fibers
