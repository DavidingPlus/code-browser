430template< typename Fn, typename ... Arg >
431class worker_context final : public context {
432private:
433 typename std::decay< Fn >::type fn_;
434 std::tuple< Arg ... > arg_;
435
436 boost::context::fiber
437 run_( boost::context::fiber && c) {
438 {
439 // fn and tpl must be destroyed before calling terminate()
440 auto fn = std::move( fn_);
441 auto arg = std::move( arg_);
442#if (defined(BOOST_USE_UCONTEXT)||defined(BOOST_USE_WINFIB))
443 std::move( c).resume();
444#else
445 boost::ignore_unused(c);
446#endif
447#if defined(BOOST_NO_CXX17_STD_APPLY)
448 boost::context::detail::apply( std::move( fn), std::move( arg) );
449#else
450 std::apply( std::move( fn), std::move( arg) );
451#endif
452 }
453 // terminate context
454 return terminate();
455 }
456
457public:
458 template< typename StackAlloc >
459 worker_context( launch policy,
460 fiber_properties* properties,
461 boost::context::preallocated const& palloc, StackAlloc && salloc,
462 Fn && fn, Arg ... arg) :
463 context{ 1, type::worker_context, policy },
464 fn_( std::forward< Fn >( fn) ),
465 arg_( std::forward< Arg >( arg) ... ) {
466 if ( properties != nullptr ) {
467 set_properties(properties);
468 properties->set_context(this);
469 }
470 c_ = boost::context::fiber{ std::allocator_arg, palloc, std::forward< StackAlloc >( salloc),
471 std::bind( & worker_context::run_, this, std::placeholders::_1) };
472#if (defined(BOOST_USE_UCONTEXT)||defined(BOOST_USE_WINFIB))
473 c_ = std::move( c_).resume();
474#endif
475 }
476
477 template< typename StackAlloc >
478 worker_context( launch policy,
479 boost::context::preallocated const& palloc, StackAlloc && salloc,
480 Fn && fn, Arg ... arg) :
481 worker_context( policy, palloc, salloc, nullptr, std::forward<Fn>( fn ), std::forward<Arg>( arg ) ... ){
482 }
483};

486template< typename StackAlloc, typename Fn, typename ... Arg >
487static intrusive_ptr< context > make_worker_context_with_properties( launch policy,
488 fiber_properties* properties,
489 StackAlloc && salloc,
490 Fn && fn, Arg ... arg) {
491 typedef worker_context< Fn, Arg ... > context_t;
492
493 auto sctx = salloc.allocate();
494 // reserve space for control structure
495 void * storage = reinterpret_cast< void * >(
496 ( reinterpret_cast< uintptr_t >( sctx.sp) - static_cast< uintptr_t >( sizeof( context_t) ) )
497 & ~ static_cast< uintptr_t >( 0xff) );
498 void * stack_bottom = reinterpret_cast< void * >(
499 reinterpret_cast< uintptr_t >( sctx.sp) - static_cast< uintptr_t >( sctx.size) );
500 const std::size_t size = reinterpret_cast< uintptr_t >( storage) - reinterpret_cast< uintptr_t >( stack_bottom);
501 // placement new of context on top of fiber's stack
502 return intrusive_ptr< context >{
503 new ( storage) context_t{
504 policy,
505 properties,
506 boost::context::preallocated{ storage, size, sctx },
507 std::forward< StackAlloc >( salloc),
508 std::forward< Fn >( fn),
509 std::forward< Arg >( arg) ... } };
510}
511
512template< typename StackAlloc, typename Fn, typename ... Arg >
513static intrusive_ptr< context > make_worker_context( launch policy,
514 StackAlloc && salloc,
515 Fn && fn, Arg ... arg){
516 return make_worker_context_with_properties( policy, nullptr, std::forward<StackAlloc>(salloc),
517 std::forward<Fn>( fn ), std::forward<Arg>( arg ) ... );
518}
