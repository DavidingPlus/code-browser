28class spinlock_ttas_futex {
29private:
30 template< typename FBSplk >
31 friend class spinlock_rtm;
32
33 std::atomic< std::int32_t > value_{ 0 };
34
35public:
36 spinlock_ttas_futex() = default;
37
38 spinlock_ttas_futex( spinlock_ttas_futex const&) = delete;
39 spinlock_ttas_futex & operator=( spinlock_ttas_futex const&) = delete;
40
41 void lock() noexcept {
42 static thread_local std::minstd_rand generator{ std::random_device{}() };
43 std::int32_t collisions = 0, retries = 0, expected = 0;
44 // after max. spins or collisions suspend via futex
45 while ( retries++ < BOOST_FIBERS_RETRY_THRESHOLD) {
46 // avoid using multiple pause instructions for a delay of a specific cycle count
47 // the delay of cpu_relax() (pause on Intel) depends on the processor family
48 // the cycle count can not guaranteed from one system to the next
49 // -> check the shared variable 'value_' in between each cpu_relax() to prevent
50 // unnecessarily long delays on some systems
51 // test shared variable 'status_'
52 // first access to 'value_' -> chache miss
53 // sucessive acccess to 'value_' -> cache hit
54 // if 'value_' was released by other fiber
55 // cached 'value_' is invalidated -> cache miss
56 if ( 0 != ( expected = value_.load( m: std::memory_order_relaxed) ) ) {
57#if !defined(BOOST_FIBERS_SPIN_SINGLE_CORE)
58 if ( BOOST_FIBERS_SPIN_BEFORE_SLEEP0 > retries) {
59 // give CPU a hint that this thread is in a "spin-wait" loop
60 // delays the next instruction's execution for a finite period of time (depends on processor family)
61 // the CPU is not under demand, parts of the pipeline are no longer being used
62 // -> reduces the power consumed by the CPU
63 // -> prevent pipeline stalls
64 cpu_relax();
65 } else if ( BOOST_FIBERS_SPIN_BEFORE_YIELD > retries) {
66 // std::this_thread::sleep_for( 0us) has a fairly long instruction path length,
67 // combined with an expensive ring3 to ring 0 transition costing about 1000 cycles
68 // std::this_thread::sleep_for( 0us) lets give up this_thread the remaining part of its time slice
69 // if and only if a thread of equal or greater priority is ready to run
70 static constexpr std::chrono::microseconds us0{ 0 };
71 std::this_thread::sleep_for( rtime: us0);
72 } else {
73 // std::this_thread::yield() allows this_thread to give up the remaining part of its time slice,
74 // but only to another thread on the same processor
75 // instead of constant checking, a thread only checks if no other useful work is pending
76 std::this_thread::yield();
77 }
78#else
79 // std::this_thread::yield() allows this_thread to give up the remaining part of its time slice,
80 // but only to another thread on the same processor
81 // instead of constant checking, a thread only checks if no other useful work is pending
82 std::this_thread::yield();
83#endif
84 } else if ( ! value_.compare_exchange_strong( i1&: expected, i2: 1, m: std::memory_order_acquire) ) {
85 // spinlock now contended
86 // utilize 'Binary Exponential Backoff' algorithm
87 // linear_congruential_engine is a random number engine based on Linear congruential generator (LCG)
88 std::uniform_int_distribution< std::int32_t > distribution{
89 0, static_cast< std::int32_t >( 1) << (std::min)(a: collisions, b: static_cast< std::int32_t >( BOOST_FIBERS_CONTENTION_WINDOW_THRESHOLD)) };
90 const std::int32_t z = distribution( generator);
91 ++collisions;
92 for ( std::int32_t i = 0; i < z; ++i) {
93 // -> reduces the power consumed by the CPU
94 // -> prevent pipeline stalls
95 cpu_relax();
96 }
97 } else {
98 // success, lock acquired
99 return;
100 }
101 }
102 // failure, lock not acquired
103 // pause via futex
104 if ( 2 != expected) {
105 expected = value_.exchange( i: 2, m: std::memory_order_acquire);
106 }
107 while ( 0 != expected) {
108 futex_wait( addr: & value_, x: 2);
109 expected = value_.exchange( i: 2, m: std::memory_order_acquire);
110 }
111 }
112
113 bool try_lock() noexcept {
114 std::int32_t expected = 0;
115 return value_.compare_exchange_strong( i1&: expected, i2: 1, m: std::memory_order_acquire);
116 }
117
118 void unlock() noexcept {
119 if ( 1 != value_.fetch_sub( i: 1, m: std::memory_order_acquire) ) {
120 value_.store( i: 0, m: std::memory_order_release);
121 futex_wake( addr: & value_);
122 }
123 }
124};
