23namespace boost {
24namespace fibers {
25namespace detail {
26
27class thread_barrier {
28private:
29 std::size_t initial_;
30 std::size_t current_;
31 bool cycle_{ true };
32 std::mutex mtx_{};
33 std::condition_variable cond_{};
34
35public:
36 explicit thread_barrier( std::size_t initial) :
37 initial_{ initial },
38 current_{ initial_ } {
39 BOOST_ASSERT ( 0 != initial);
40 }
41
42 thread_barrier( thread_barrier const&) = delete;
43 thread_barrier & operator=( thread_barrier const&) = delete;
44
45 bool wait() {
46 std::unique_lock< std::mutex > lk( mtx_);
47 const bool cycle = cycle_;
48 if ( 0 == --current_) {
49 cycle_ = ! cycle_;
50 current_ = initial_;
51 lk.unlock(); // no pessimization
52 cond_.notify_all();
53 return true;
54 }
55 cond_.wait( lock&: lk, p: [&](){ return cycle != cycle_; });
56 return false;
57 }
58};
59
60}}}
