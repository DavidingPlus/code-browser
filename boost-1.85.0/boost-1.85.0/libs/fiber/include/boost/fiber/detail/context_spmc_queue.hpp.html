38namespace boost {
39namespace fibers {
40namespace detail {
41
42class context_spmc_queue {
43private:
44 class array {
45 private:
46 typedef std::atomic< context * > atomic_type;
47 typedef atomic_type storage_type;
48
49 std::size_t capacity_;
50 storage_type * storage_;
51
52 public:
53 array( std::size_t capacity) :
54 capacity_{ capacity },
55 storage_{ new storage_type[capacity_] } {
56 for ( std::size_t i = 0; i < capacity_; ++i) {
57 ::new ( static_cast< void * >( std::addressof( r&: storage_[i]) ) ) atomic_type{ nullptr };
58 }
59 }
60
61 ~array() {
62 for ( std::size_t i = 0; i < capacity_; ++i) {
63 reinterpret_cast< atomic_type * >( std::addressof( r&: storage_[i]) )->~atomic_type();
64 }
65 delete [] storage_;
66 }
67
68 std::size_t capacity() const noexcept {
69 return capacity_;
70 }
71
72 void push( std::size_t bottom, context * ctx) noexcept {
73 reinterpret_cast< atomic_type * >(
74 std::addressof( r&: storage_[bottom % capacity_]) )
75 ->store( p: ctx, m: std::memory_order_relaxed);
76 }
77
78 context * pop( std::size_t top) noexcept {
79 return reinterpret_cast< atomic_type * >(
80 std::addressof( r&: storage_[top % capacity_]) )
81 ->load( m: std::memory_order_relaxed);
82 }
83
84 array * resize( std::size_t bottom, std::size_t top) {
85 std::unique_ptr< array > tmp{ new array{ 2 * capacity_ } };
86 for ( std::size_t i = top; i != bottom; ++i) {
87 tmp->push( bottom: i, ctx: pop( top: i) );
88 }
89 return tmp.release();
90 }
91 };
92
93 std::atomic< std::size_t > top_{ 0 };
94 std::atomic< std::size_t > bottom_{ 0 };
95 std::atomic< array * > array_;
96 std::vector< array * > old_arrays_{};
97 char padding_[cacheline_length];
98
99public:
100 context_spmc_queue( std::size_t capacity = 4096) :
101 array_{ new array{ capacity } } {
102 old_arrays_.reserve( n: 32);
103 }
104
105 ~context_spmc_queue() {
106 for ( array * a : old_arrays_) {
107 delete a;
108 }
109 delete array_.load();
110 }
111
112 context_spmc_queue( context_spmc_queue const&) = delete;
113 context_spmc_queue & operator=( context_spmc_queue const&) = delete;
114
115 bool empty() const noexcept {
116 std::size_t bottom = bottom_.load( m: std::memory_order_relaxed);
117 std::size_t top = top_.load( m: std::memory_order_relaxed);
118 return bottom <= top;
119 }
120
121 void push( context * ctx) {
122 std::size_t bottom = bottom_.load( m: std::memory_order_relaxed);
123 std::size_t top = top_.load( m: std::memory_order_acquire);
124 array * a = array_.load( m: std::memory_order_relaxed);
125 if ( (a->capacity() - 1) < (bottom - top) ) {
126 // queue is full
127 // resize
128 array * tmp = a->resize( bottom, top);
129 old_arrays_.push_back( x: a);
130 std::swap( a&: a, b&: tmp);
131 array_.store( p: a, m: std::memory_order_relaxed);
132 }
133 a->push( bottom, ctx);
134 std::atomic_thread_fence( m: std::memory_order_release);
135 bottom_.store( i: bottom + 1, m: std::memory_order_relaxed);
136 }
137
138 context * pop() {
139 std::size_t bottom = bottom_.load( m: std::memory_order_relaxed) - 1;
140 array * a = array_.load( m: std::memory_order_relaxed);
141 bottom_.store( i: bottom, m: std::memory_order_relaxed);
142 std::atomic_thread_fence( m: std::memory_order_seq_cst);
143 std::size_t top = top_.load( m: std::memory_order_relaxed);
144 context * ctx = nullptr;
145 if ( top <= bottom) {
146 // queue is not empty
147 ctx = a->pop( top: bottom);
148 BOOST_ASSERT( nullptr != ctx);
149 if ( top == bottom) {
150 // last element dequeued
151 if ( ! top_.compare_exchange_strong( i1&: top, i2: top + 1,
152 m1: std::memory_order_seq_cst,
153 m2: std::memory_order_relaxed) ) {
154 // lose the race
155 ctx = nullptr;
156 }
157 bottom_.store( i: bottom + 1, m: std::memory_order_relaxed);
158 }
159 } else {
160 // queue is empty
161 bottom_.store( i: bottom + 1, m: std::memory_order_relaxed);
162 }
163 return ctx;
164 }
165
166 context * steal() {
167 std::size_t top = top_.load( m: std::memory_order_acquire);
168 std::atomic_thread_fence( m: std::memory_order_seq_cst);
169 std::size_t bottom = bottom_.load( m: std::memory_order_acquire);
170 context * ctx = nullptr;
171 if ( top < bottom) {
172 // queue is not empty
173 array * a = array_.load( m: std::memory_order_consume);
174 ctx = a->pop( top);
175 BOOST_ASSERT( nullptr != ctx);
176 // do not steal pinned context (e.g. main-/dispatcher-context)
177 if ( ctx->is_context( t: type::pinned_context) ) {
178 return nullptr;
179 }
180 if ( ! top_.compare_exchange_strong( i1&: top, i2: top + 1,
181 m1: std::memory_order_seq_cst,
182 m2: std::memory_order_relaxed) ) {
183 // lose the race
184 return nullptr;
185 }
186 }
187 return ctx;
188 }
189};
