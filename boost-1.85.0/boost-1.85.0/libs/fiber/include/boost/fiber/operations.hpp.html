38template< typename Clock, typename Duration >
39void sleep_until( std::chrono::time_point< Clock, Duration > const& sleep_time_) {
40 std::chrono::steady_clock::time_point sleep_time = boost::fibers::detail::convert( sleep_time_);
41 fibers::context * active_ctx = fibers::context::active();
42 active_ctx->wait_until( sleep_time);
43}
44
45template< typename Rep, typename Period >
46void sleep_for( std::chrono::duration< Rep, Period > const& timeout_duration) {
47 fibers::context * active_ctx = fibers::context::active();
48 active_ctx->wait_until( std::chrono::steady_clock::now() + timeout_duration);
49}

51template< typename PROPS >
52PROPS & properties() {
53 fibers::fiber_properties * props = fibers::context::active()->get_properties();
54 if ( BOOST_LIKELY( nullptr == props) ) {
55 // props could be nullptr if the thread's main fiber has not yet
56 // yielded (not yet passed through algorithm_with_properties::
57 // awakened()). Address that by yielding right now.
58 yield();
59 // Try again to obtain the fiber_properties subclass instance ptr.
60 // Walk through the whole chain again because who knows WHAT might
61 // have happened while we were yielding!
62 props = fibers::context::active()->get_properties();
63 // Could still be hosed if the running manager isn't a subclass of
64 // algorithm_with_properties.
65 BOOST_ASSERT_MSG( props, "this_fiber::properties not set");
66 }
67 return dynamic_cast< PROPS & >( * props );
68}

79// Returns true if the thread could be initialize, false otherwise (it was already initialized previously).
80inline bool initialize_thread(algo::algorithm::ptr_t algo, stack_allocator_wrapper&& salloc) noexcept {
81 return boost::fibers::context::initialize_thread(algo, salloc: std::move(salloc));
82}
83
84template< typename SchedAlgo, typename ... Args >
85void use_scheduling_algorithm( Args && ... args) noexcept {
86 initialize_thread(new SchedAlgo(std::forward< Args >( args) ... ), make_stack_allocator_wrapper<boost::fibers::default_stack>());
87}
