40class BOOST_FIBERS_DECL fiber {
41private:
42 friend class context;
43
44 using ptr_t = intrusive_ptr<context>;
45
46 ptr_t impl_{};
47
48 void start_() noexcept;
49
50public:
51 using id = context::id;
52
53 fiber() = default;
54
55 template< typename Fn,
56 typename ... Arg,
57 typename = detail::disable_overload< fiber, Fn >,
58 typename = detail::disable_overload< launch, Fn >,
59 typename = detail::disable_overload< std::allocator_arg_t, Fn >
60 >
61#if BOOST_COMP_GNUC < 50000000
62 explicit fiber( Fn && fn, Arg && ... arg) :
63#else
64 fiber( Fn && fn, Arg ... arg) :
65#endif
66 fiber{ launch::post,
67 std::allocator_arg, default_stack(),
68 std::forward< Fn >( fn), std::forward< Arg >( arg) ... } {
69 }
70
71 template< typename Fn,
72 typename ... Arg,
73 typename = detail::disable_overload< fiber, Fn >
74 >
75#if BOOST_COMP_GNUC < 50000000
76 fiber( launch policy, Fn && fn, Arg && ... arg) :
77#else
78 fiber( launch policy, Fn && fn, Arg ... arg) :
79#endif
80 fiber{ policy,
81 std::allocator_arg, default_stack(),
82 std::forward< Fn >( fn), std::forward< Arg >( arg) ... } {
83 }

89#if BOOST_COMP_GNUC < 50000000
90 fiber( std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg && ... arg) :
91#else
92 fiber( std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg ... arg) :
93#endif
94 fiber{ launch::post,

99 template< typename StackAllocator,
100 typename Fn,

103#if BOOST_COMP_GNUC < 50000000
104 fiber( launch policy, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg && ... arg) :
105#else
106 fiber( launch policy, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg ... arg) :
107#endif

115 typename ... Arg,

146 template< typename StackAllocator,
147 typename Fn,
148 typename ... Arg

150#if BOOST_COMP_GNUC < 50000000
151 fiber( fiber_properties* properties, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg && ... arg) :
152#else
153 fiber( fiber_properties* properties, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg ... arg) :
154#endif
155 fiber{ launch::post,
156 properties,
157 std::allocator_arg, std::forward< StackAllocator >( salloc),

160
161 template< typename StackAllocator,
162 typename Fn,
163 typename ... Arg
164 >
165#if BOOST_COMP_GNUC < 50000000
166 fiber( launch policy, fiber_properties* properties, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg && ... arg) :
167#else
168 fiber( launch policy, fiber_properties* properties, std::allocator_arg_t, StackAllocator && salloc, Fn && fn, Arg ... arg) :
169#endif
170 impl_{ make_worker_context_with_properties( policy, properties, std::forward< StackAllocator >( salloc), std::forward< Fn >( fn), std::forward< Arg >( arg) ... ) } {
171 start_();
172 }
173
174 ~fiber() {
175 if ( joinable() ) {
176 std::terminate();
177 }
178 }
179
180 fiber( fiber const&) = delete;
181 fiber & operator=( fiber const&) = delete;
182
183 fiber( fiber && other) noexcept :
184 impl_{} {
185 swap( other);
186 }
187
188 fiber & operator=( fiber && other) noexcept {
189 if ( joinable() ) {
190 std::terminate();
191 }
192 if ( BOOST_UNLIKELY( this == & other) ) {
193 return * this;
194 }
195 impl_.swap( rhs&: other.impl_);
196 return * this;
197 }
198
199 void swap( fiber & other) noexcept {
200 impl_.swap( rhs&: other.impl_);
201 }
202
203 id get_id() const noexcept {
204 return impl_ ? impl_->get_id() : id();
205 }
206
207 bool joinable() const noexcept {
208 return nullptr != impl_;
209 }
210
211 void join();
212
213 void detach();
214
215 template< typename PROPS >
216 PROPS & properties() {
217 auto props = impl_->get_properties();
218 BOOST_ASSERT_MSG( props, "fiber::properties not set");
219 return dynamic_cast< PROPS & >( * props );
220 }
221};
222
223inline
224bool operator<( fiber const& l, fiber const& r) noexcept {
225 return l.get_id() < r.get_id();
226}
227
228inline
229void swap( fiber & l, fiber & r) noexcept {
230 return l.swap( other&: r);
231}
