25bool
26wait_queue::suspend_and_wait_until( detail::spinlock_lock & lk,
27 context * active_ctx,
28 std::chrono::steady_clock::time_point const& timeout_time) {
29 waker_with_hook w{ active_ctx->create_waker() };
30 slist_.push_back(value&: w);
31 // suspend this fiber
32 if ( ! active_ctx->wait_until( timeout_time, lk, waker(w)) ) {
33 // relock local lk
34 for(;;) {
35 if(lk.try_lock())
36 break;
37 active_ctx->yield();
38 }
39 // remove from waiting-queue
40 if ( w.is_linked()) {
41 slist_.remove( value: w);
42 }
43 lk.unlock();
44 return false;
45 }
46 return true;
47}

50wait_queue::notify_one() {
