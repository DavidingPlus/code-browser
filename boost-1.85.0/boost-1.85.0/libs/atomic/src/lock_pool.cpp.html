102//! \c find_address generic implementation
103std::size_t find_address_generic(const volatile void* addr, const volatile void* const* addrs, std::size_t size)
104{
105 for (std::size_t i = 0u; i < size; ++i)
106 {
107 if (addrs[i] == addr)
108 return i;
109 }
110
111 return size;
112}

143#if defined(BOOST_ATOMIC_DETAIL_X86_USE_RUNTIME_DISPATCH)
144
145std::size_t find_address_dispatch(const volatile void* addr, const volatile void* const* addrs, std::size_t size)
146{
147 find_address_t* find_addr = &find_address_generic;
148
149#if defined(BOOST_ATOMIC_USE_SSE2)
150 // First, check the max available cpuid function
151 uint32_t eax = 0u, ebx = 0u, ecx = 0u, edx = 0u;
152 atomics::detail::cpuid(eax, ebx, ecx, edx);
153
154 const uint32_t max_cpuid_function = eax;
155 if (max_cpuid_function >= 1u)
156 {
157 // Obtain CPU features
158 eax = 1u;
159 ebx = ecx = edx = 0u;
160 atomics::detail::cpuid(eax, ebx, ecx, edx);
161
162 if ((edx & (1u << 26)) != 0u)
163 find_addr = &find_address_sse2;
164
165#if defined(BOOST_ATOMIC_USE_SSE41) && BOOST_ATOMIC_DETAIL_SIZEOF_POINTER == 8
166 if ((ecx & (1u << 19)) != 0u)
167 find_addr = &find_address_sse41;
168#endif
169 }
170#endif // defined(BOOST_ATOMIC_USE_SSE2)
171
172 find_address_ptr ptr = {};
173 ptr.as_ptr = find_addr;
174 func_ptr_operations::store(storage&: g_find_address.as_storage, v: ptr.as_storage, order: boost::memory_order_relaxed);
175
176 return find_addr(addr, addrs, size);
177}
178
179#endif // defined(BOOST_ATOMIC_DETAIL_X86_USE_RUNTIME_DISPATCH)

181inline std::size_t find_address(const volatile void* addr, const volatile void* const* addrs, std::size_t size)
182{
183 find_address_ptr ptr;
184 ptr.as_storage = func_ptr_operations::load(storage: g_find_address.as_storage, order: boost::memory_order_relaxed);
185 return ptr.as_ptr(addr, addrs, size);
186}

218//! List of wait states. Must be a POD structure.
219struct wait_state_list
220{
221 //! List header
222 struct header
223 {
224 //! List size
225 std::size_t size;
226 //! List capacity
227 std::size_t capacity;
228 };
229
230 /*!
231 * \brief Pointer to the list header
232 *
233 * The list buffer consists of three adjacent areas: header object, array of atomic pointers and array of pointers to the wait_state structures.
234 * Each of the arrays have header.capacity elements, of which the first header.size elements correspond to the currently ongoing wait operations
235 * and the rest are spare elements. Spare wait_state structures may still be allocated (in which case the wait_state pointer is not null) and
236 * can be reused on future requests. Spare atomic pointers are null and unused.
237 *
238 * This memory layout was designed to optimize wait state lookup by atomic address and also support memory pooling to reduce dynamic memory allocations.
239 */
240 header* m_header;
241 //! The flag indicates that memory pooling is disabled. Set on process cleanup.
242 bool m_free_memory;
243
244 //! Buffer alignment, in bytes
245 static BOOST_CONSTEXPR_OR_CONST std::size_t buffer_alignment = 16u;
246 //! Alignment of pointer arrays in the buffer, in bytes. This should align atomic pointers to the vector size used in \c find_address implementation.
247 static BOOST_CONSTEXPR_OR_CONST std::size_t entries_alignment = atomics::detail::alignment_of< void* >::value < 16u ? 16u : atomics::detail::alignment_of< void* >::value;
248 //! Offset from the list header to the beginning of the array of atomic pointers in the buffer, in bytes
249 static BOOST_CONSTEXPR_OR_CONST std::size_t entries_offset = (sizeof(header) + entries_alignment - 1u) & ~static_cast< std::size_t >(entries_alignment - 1u);
250 //! Initial buffer capacity, in elements. This should be at least as large as a vector size used in \c find_address implementation.
251 static BOOST_CONSTEXPR_OR_CONST std::size_t initial_capacity = (16u / sizeof(void*)) < 2u ? 2u : (16u / sizeof(void*));
252
253 //! Returns a pointer to the array of atomic pointers
254 static const volatile void** get_atomic_pointers(header* p) BOOST_NOEXCEPT
255 {
256 BOOST_ASSERT(p != NULL);
257 return reinterpret_cast< const volatile void** >(reinterpret_cast< unsigned char* >(p) + entries_offset);
258 }
259
260 //! Returns a pointer to the array of atomic pointers
261 const volatile void** get_atomic_pointers() const BOOST_NOEXCEPT
262 {
263 return get_atomic_pointers(p: m_header);
264 }
265
266 //! Returns a pointer to the array of pointers to the wait states
267 static wait_state** get_wait_states(const volatile void** ptrs, std::size_t capacity) BOOST_NOEXCEPT
268 {
269 return reinterpret_cast< wait_state** >(const_cast< void** >(ptrs + capacity));
270 }
271
272 //! Returns a pointer to the array of pointers to the wait states
273 static wait_state** get_wait_states(header* p) BOOST_NOEXCEPT
274 {
275 return get_wait_states(ptrs: get_atomic_pointers(p), capacity: p->capacity);
276 }
277
278 //! Returns a pointer to the array of pointers to the wait states
279 wait_state** get_wait_states() const BOOST_NOEXCEPT
280 {
281 return get_wait_states(p: m_header);
282 }
283
284 //! Finds an element with the given pointer to the atomic object
285 wait_state* find(const volatile void* addr) const BOOST_NOEXCEPT
286 {
287 wait_state* ws = NULL;
288 if (BOOST_LIKELY(m_header != NULL))
289 {
290 const volatile void* const* addrs = get_atomic_pointers();
291 const std::size_t size = m_header->size;
292 std::size_t pos = find_address(addr, addrs, size);
293 if (pos < size)
294 ws = get_wait_states()[pos];
295 }
296
297 return ws;
298 }
299
300 //! Finds an existing element with the given pointer to the atomic object or allocates a new one. Returns NULL in case of failure.
301 wait_state* find_or_create(const volatile void* addr) BOOST_NOEXCEPT;
302 //! Releases the previously created wait state
303 void erase(wait_state* w) BOOST_NOEXCEPT;
304
305 //! Deallocates spare entries and the list buffer if no allocated entries are left
306 void free_spare() BOOST_NOEXCEPT;
307 //! Allocates new buffer for the list entries. Returns NULL in case of failure.
308 static header* allocate_buffer(std::size_t new_capacity, header* old_header = NULL) BOOST_NOEXCEPT;
309};

436//! Lock pool entry
437struct lock_state
438{
439 //! Mutex futex
440 BOOST_ATOMIC_DETAIL_ALIGNED_VAR(futex_operations::storage_alignment, futex_operations::storage_type, m_mutex);
441 //! Wait states
442 wait_state_list m_wait_states;
443
444 //! Locks the mutex for a short duration
445 void short_lock() BOOST_NOEXCEPT
446 {
447 long_lock();
448 }
449
450 //! Locks the mutex for a long duration
451 void long_lock() BOOST_NOEXCEPT
452 {
453 for (unsigned int i = 0u; i < 10u; ++i)
454 {
455 futex_operations::storage_type prev_state = futex_operations::load(storage: m_mutex, order: boost::memory_order_relaxed);
456 if (BOOST_LIKELY((prev_state & mutex_bits::locked) == 0u))
457 {
458 futex_operations::storage_type new_state = prev_state | mutex_bits::locked;
459 if (BOOST_LIKELY(futex_operations::compare_exchange_strong(m_mutex, prev_state, new_state, boost::memory_order_acquire, boost::memory_order_relaxed)))
460 return;
461 }
462
463 atomics::detail::pause();
464 }
465
466 lock_slow_path();
467 }
468
469 //! Locks the mutex for a long duration
470 void lock_slow_path() BOOST_NOEXCEPT
471 {
472 futex_operations::storage_type prev_state = futex_operations::load(storage: m_mutex, order: boost::memory_order_relaxed);
473 while (true)
474 {
475 if (BOOST_LIKELY((prev_state & mutex_bits::locked) == 0u))
476 {
477 futex_operations::storage_type new_state = prev_state | mutex_bits::locked;
478 if (BOOST_LIKELY(futex_operations::compare_exchange_weak(m_mutex, prev_state, new_state, boost::memory_order_acquire, boost::memory_order_relaxed)))
479 return;
480 }
481 else
482 {
483 futex_operations::storage_type new_state = prev_state | mutex_bits::contended;
484 if (BOOST_LIKELY(futex_operations::compare_exchange_weak(m_mutex, prev_state, new_state, boost::memory_order_relaxed, boost::memory_order_relaxed)))
485 {
486 atomics::detail::futex_wait_private(pval: &m_mutex, expected: new_state);
487 prev_state = futex_operations::load(storage: m_mutex, order: boost::memory_order_relaxed);
488 }
489 }
490 }
491 }
492
493 //! Unlocks the mutex
494 void unlock() BOOST_NOEXCEPT
495 {
496 futex_operations::storage_type prev_state = futex_operations::load(storage: m_mutex, order: boost::memory_order_relaxed);
497 futex_operations::storage_type new_state;
498 while (true)
499 {
500 new_state = (prev_state & (~mutex_bits::locked)) + mutex_bits::counter_one;
501 if (BOOST_LIKELY(futex_operations::compare_exchange_weak(m_mutex, prev_state, new_state, boost::memory_order_release, boost::memory_order_relaxed)))
502 break;
503 }
504
505 if ((prev_state & mutex_bits::contended) != 0u)
506 {
507 int woken_count = atomics::detail::futex_signal_private(pval: &m_mutex);
508 if (woken_count == 0)
509 {
510 prev_state = new_state;
511 new_state &= ~mutex_bits::contended;
512 futex_operations::compare_exchange_strong(storage&: m_mutex, expected&: prev_state, desired: new_state, success_order: boost::memory_order_relaxed, failure_order: boost::memory_order_relaxed);
513 }
514 }
515 }
516};

524//! Blocks in the wait operation until notified
525inline void wait_state::wait(lock_state& state) BOOST_NOEXCEPT
526{
527 const futex_operations::storage_type prev_cond = m_cond;
528 ++m_waiter_count;
529
530 state.unlock();
531
532 while (true)
533 {
534 int err = atomics::detail::futex_wait_private(pval: &m_cond, expected: prev_cond);
535 if (BOOST_LIKELY(err != EINTR))
536 break;
537 }
538
539 state.long_lock();
540
541 --m_waiter_count;
542}

544//! Wakes up one thread blocked in the wait operation
545inline void wait_state::notify_one(lock_state& state) BOOST_NOEXCEPT
546{
547 ++m_cond;
548
549 if (BOOST_LIKELY(m_waiter_count > 0u))
550 {
551 // Move one blocked thread to the mutex futex and mark the mutex contended so that the thread is unblocked on unlock()
552 atomics::detail::futex_requeue_private(pval1: &m_cond, pval2: &state.m_mutex, wake_count: 0u, requeue_count: 1u);
553 futex_extra_operations::opaque_or(storage&: state.m_mutex, v: mutex_bits::contended, boost::memory_order_relaxed);
554 }
555}
556
557//! Wakes up all threads blocked in the wait operation
558inline void wait_state::notify_all(lock_state& state) BOOST_NOEXCEPT
559{
560 ++m_cond;
561
562 if (BOOST_LIKELY(m_waiter_count > 0u))
563 {
564 // Move blocked threads to the mutex futex and mark the mutex contended so that a thread is unblocked on unlock()
565 atomics::detail::futex_requeue_private(pval1: &m_cond, pval2: &state.m_mutex, wake_count: 0u);
566 futex_extra_operations::opaque_or(storage&: state.m_mutex, v: mutex_bits::contended, boost::memory_order_relaxed);
567 }
568}

652//! State of a wait operation associated with an atomic object
653struct wait_state :
654 public wait_state_base
655{
656 /*!
657 * \brief A semaphore used to block one or more threads
658 *
659 * A semaphore can be used to block a thread if it has no ongoing notifications (i.e. \c m_notify_count is 0).
660 * If there is no such semaphore, the thread has to allocate a new one to block on. This is to guarantee
661 * that a thread that is blocked after a notification is not immediately released by the semaphore while
662 * there are previously blocked threads.
663 *
664 * Semaphores are organized in a circular doubly linked list. A single semaphore object represents a list
665 * of one semaphore and is said to be "singular".
666 */
667 struct semaphore
668 {
669 //! Pointer to the next semaphore in the list
670 semaphore* m_next;
671 //! Pointer to the previous semaphore in the list
672 semaphore* m_prev;
673
674 //! Semaphore handle
675 boost::winapi::HANDLE_ m_semaphore;
676 //! Number of threads blocked on the semaphore
677 boost::winapi::ULONG_ m_waiter_count;
678 //! Number of threads released by notifications
679 boost::winapi::ULONG_ m_notify_count;
680
681 semaphore() BOOST_NOEXCEPT :
682 m_semaphore(boost::winapi::create_anonymous_semaphore(NULL, 0, (std::numeric_limits< boost::winapi::LONG_ >::max)())),
683 m_waiter_count(0u),
684 m_notify_count(0u)
685 {
686 m_next = m_prev = this;
687 }
688
689 ~semaphore() BOOST_NOEXCEPT
690 {
691 BOOST_ASSERT(is_singular());
692
693 if (BOOST_LIKELY(m_semaphore != boost::winapi::invalid_handle_value))
694 boost::winapi::CloseHandle(m_semaphore);
695 }
696
697 //! Creates a new semaphore or returns null in case of failure
698 static semaphore* create() BOOST_NOEXCEPT
699 {
700 semaphore* p = new (std::nothrow) semaphore();
701 if (BOOST_UNLIKELY(p != NULL && p->m_semaphore == boost::winapi::invalid_handle_value))
702 {
703 delete p;
704 p = NULL;
705 }
706 return p;
707 }
708
709 //! Returns \c true if the semaphore is the single element of the list
710 bool is_singular() const BOOST_NOEXCEPT
711 {
712 return m_next == this /* && m_prev == this */;
713 }
714
715 //! Inserts the semaphore list after the specified other semaphore
716 void link_after(semaphore* that) BOOST_NOEXCEPT
717 {
718 link_before(that->m_next);
719 }
720
721 //! Inserts the semaphore list before the specified other semaphore
722 void link_before(semaphore* that) BOOST_NOEXCEPT
723 {
724 semaphore* prev = that->m_prev;
725 that->m_prev = m_prev;
726 m_prev->m_next = that;
727 m_prev = prev;
728 prev->m_next = this;
729 }
730
731 //! Removes the semaphore from the list
732 void unlink() BOOST_NOEXCEPT
733 {
734 // Load pointers beforehand, in case we are the only element in the list
735 semaphore* next = m_next;
736 semaphore* prev = m_prev;
737 prev->m_next = next;
738 next->m_prev = prev;
739 m_next = m_prev = this;
740 }
741
742 BOOST_DELETED_FUNCTION(semaphore(semaphore const&))
743 BOOST_DELETED_FUNCTION(semaphore& operator= (semaphore const&))
744 };

746 //! Doubly linked circular list of semaphores
747 class semaphore_list
748 {
749 private:
750 semaphore* m_head;
751
752 public:
753 semaphore_list() BOOST_NOEXCEPT :
754 m_head(NULL)
755 {
756 }
757
758 //! Returns \c true if the list is empty
759 bool empty() const BOOST_NOEXCEPT
760 {
761 return m_head == NULL;
762 }
763
764 //! Returns the first semaphore in the list
765 semaphore* front() const BOOST_NOEXCEPT
766 {
767 return m_head;
768 }
769
770 //! Returns the first semaphore in the list and leaves the list empty
771 semaphore* eject() BOOST_NOEXCEPT
772 {
773 semaphore* sem = m_head;
774 m_head = NULL;
775 return sem;
776 }
777
778 //! Inserts the semaphore at the beginning of the list
779 void push_front(semaphore* sem) BOOST_NOEXCEPT
780 {
781 if (m_head)
782 sem->link_before(m_head);
783
784 m_head = sem;
785 }
786
787 //! Removes the first semaphore from the beginning of the list
788 semaphore* pop_front() BOOST_NOEXCEPT
789 {
790 BOOST_ASSERT(!empty());
791 semaphore* sem = m_head;
792 erase(sem);
793 return sem;
794 }
795
796 //! Removes the semaphore from the list
797 void erase(semaphore* sem) BOOST_NOEXCEPT
798 {
799 if (sem->is_singular())
800 {
801 BOOST_ASSERT(m_head == sem);
802 m_head = NULL;
803 }
804 else
805 {
806 if (m_head == sem)
807 m_head = sem->m_next;
808 sem->unlink();
809 }
810 }
811
812 BOOST_DELETED_FUNCTION(semaphore_list(semaphore_list const&))
813 BOOST_DELETED_FUNCTION(semaphore_list& operator= (semaphore_list const&))
814 };

816 //! List of semaphores used for notifying. Here, every semaphore has m_notify_count > 0 && m_waiter_count > 0.
817 semaphore_list m_notify_semaphores;
818 //! List of semaphores used for waiting. Here, every semaphore has m_notify_count == 0 && m_waiter_count > 0.
819 semaphore_list m_wait_semaphores;
820 //! List of free semaphores. Here, every semaphore has m_notify_count == 0 && m_waiter_count == 0.
821 semaphore_list m_free_semaphores;
822
823 explicit wait_state(std::size_t index) BOOST_NOEXCEPT :
824 wait_state_base(index)
825 {
826 }
827
828 ~wait_state() BOOST_NOEXCEPT
829 {
830 // All wait and notification operations must have been completed
831 BOOST_ASSERT(m_notify_semaphores.empty());
832 BOOST_ASSERT(m_wait_semaphores.empty());
833
834 semaphore* sem = m_free_semaphores.eject();
835 if (sem)
836 {
837 while (true)
838 {
839 bool was_last = sem->is_singular();
840 semaphore* next = sem->m_next;
841 sem->unlink();
842
843 delete sem;
844
845 if (was_last)
846 break;
847
848 sem = next;
849 }
850 }
851 }
852
853 //! Blocks in the wait operation until notified
854 void wait(lock_state& state) BOOST_NOEXCEPT;
855 //! Fallback implementation of wait
856 void wait_fallback(lock_state& state) BOOST_NOEXCEPT;
857
858 //! Wakes up one thread blocked in the wait operation
859 void notify_one(lock_state&) BOOST_NOEXCEPT
860 {
861 if (m_notify_semaphores.empty())
862 {
863 if (m_wait_semaphores.empty())
864 return;
865
866 // Move the semaphore with waiters to the notify list
867 m_notify_semaphores.push_front(m_wait_semaphores.pop_front());
868 }
869
870 semaphore* sem = m_notify_semaphores.front();
871 ++sem->m_notify_count;
872
873 if (sem->m_notify_count == sem->m_waiter_count)
874 {
875 // Remove this semaphore from the list. The waiter will re-insert it into the waiter or free list once there are no more pending notifications in it.
876 m_notify_semaphores.erase(sem);
877 }
878
879 boost::winapi::ReleaseSemaphore(sem->m_semaphore, 1, NULL);
880 }
881
882 //! Wakes up all threads blocked in the wait operation
883 void notify_all(lock_state&) BOOST_NOEXCEPT
884 {
885 // Combine all notify and waiter semaphores in one list
886 semaphore* sem = m_notify_semaphores.eject();
887 if (sem)
888 {
889 if (!m_wait_semaphores.empty())
890 {
891 m_wait_semaphores.eject()->link_before(sem);
892 }
893 }
894 else
895 {
896 sem = m_wait_semaphores.eject();
897 }
898
899 if (sem)
900 {
901 while (true)
902 {
903 bool was_last = sem->is_singular();
904 semaphore* next = sem->m_next;
905 sem->unlink();
906
907 boost::winapi::ULONG_ count = sem->m_waiter_count - sem->m_notify_count;
908 sem->m_notify_count += count;
909
910 boost::winapi::ReleaseSemaphore(sem->m_semaphore, count, NULL);
911
912 if (was_last)
913 break;
914
915 sem = next;
916 }
917 }
918 }
919};

921//! Lock pool entry
922struct lock_state
923{
924 //! Mutex
925 boost::winapi::CRITICAL_SECTION_ m_mutex;
926 //! Fallback mutex. Used as indicator of critical section initialization state and a fallback mutex, if critical section cannot be initialized.
927 BOOST_ATOMIC_DETAIL_ALIGNED_VAR(mutex_operations::storage_alignment, mutex_operations::storage_type, m_mutex_fallback);
928 //! Wait states
929 wait_state_list m_wait_states;
930
931 //! Locks the mutex for a short duration
932 void short_lock() BOOST_NOEXCEPT
933 {
934 long_lock();
935 }
936
937 //! Locks the mutex for a long duration
938 void long_lock() BOOST_NOEXCEPT
939 {
940 mutex_operations::storage_type fallback_state = mutex_operations::load(m_mutex_fallback, boost::memory_order_relaxed);
941 while (true)
942 {
943 if (BOOST_LIKELY(fallback_state == fallback_mutex_bits::critical_section_initialized))
944 {
945 lock_cs:
946 boost::winapi::EnterCriticalSection(&m_mutex);
947 return;
948 }
949
950 while (fallback_state == 0u)
951 {
952 if (!mutex_operations::compare_exchange_weak(m_mutex_fallback, fallback_state, fallback_mutex_bits::locked, boost::memory_order_acquire, boost::memory_order_relaxed))
953 continue;
954
955 if (BOOST_LIKELY(!!boost::winapi::InitializeCriticalSectionAndSpinCount(&m_mutex, 100u)))
956 {
957 mutex_operations::store(m_mutex_fallback, fallback_mutex_bits::critical_section_initialized, boost::memory_order_release);
958 goto lock_cs;
959 }
960
961 // We failed to init the critical section, leave the fallback mutex locked and return
962 return;
963 }
964
965 if (fallback_state == fallback_mutex_bits::locked)
966 {
967 // Wait intil the fallback mutex is unlocked
968 boost::winapi::SwitchToThread();
969 fallback_state = mutex_operations::load(m_mutex_fallback, boost::memory_order_relaxed);
970 }
971 }
972 }
973
974 //! Unlocks the mutex
975 void unlock() BOOST_NOEXCEPT
976 {
977 mutex_operations::storage_type fallback_state = mutex_operations::load(m_mutex_fallback, boost::memory_order_relaxed);
978 if (BOOST_LIKELY(fallback_state == fallback_mutex_bits::critical_section_initialized))
979 {
980 boost::winapi::LeaveCriticalSection(&m_mutex);
981 return;
982 }
983
984 mutex_operations::store(m_mutex_fallback, 0u, boost::memory_order_release);
985 }
986};

994//! Blocks in the wait operation until notified
995inline void wait_state::wait(lock_state& state) BOOST_NOEXCEPT
996{
997 // Find a semaphore to block on
998 semaphore* sem = m_wait_semaphores.front();
999 if (sem)
1000 {
1001 while (sem->m_waiter_count >= static_cast< boost::winapi::ULONG_ >((std::numeric_limits< boost::winapi::LONG_ >::max)()))
1002 {
1003 if (sem->m_next == m_wait_semaphores.front())
1004 {
1005 sem = NULL;
1006 break;
1007 }
1008
1009 sem = sem->m_next;
1010 }
1011 }
1012
1013 if (!sem)
1014 {
1015 if (BOOST_LIKELY(!m_free_semaphores.empty()))
1016 {
1017 sem = m_free_semaphores.pop_front();
1018 }
1019 else
1020 {
1021 sem = semaphore::create();
1022 if (BOOST_UNLIKELY(!sem))
1023 {
1024 wait_fallback(state);
1025 return;
1026 }
1027 }
1028
1029 m_wait_semaphores.push_front(sem);
1030 }
1031
1032 ++sem->m_waiter_count;
1033
1034 state.unlock();
1035
1036 boost::winapi::WaitForSingleObject(sem->m_semaphore, boost::winapi::infinite);
1037
1038 state.long_lock();
1039
1040 --sem->m_waiter_count;
1041
1042 if (sem->m_notify_count > 0u)
1043 {
1044 // This semaphore is either in the notify list or not in a list at all
1045 if (--sem->m_notify_count == 0u)
1046 {
1047 if (!sem->is_singular() || sem == m_notify_semaphores.front())
1048 m_notify_semaphores.erase(sem);
1049
1050 semaphore_list* list = sem->m_waiter_count == 0u ? &m_free_semaphores : &m_wait_semaphores;
1051 list->push_front(sem);
1052 }
1053 }
1054 else if (sem->m_waiter_count == 0u)
1055 {
1056 // Move the semaphore to the free list
1057 m_wait_semaphores.erase(sem);
1058 m_free_semaphores.push_front(sem);
1059 }
1060}

1064{
1065 state.unlock();
1066
1067 boost::winapi::Sleep(0);
1068
1069 state.long_lock();
1070}

1318BOOST_ATOMIC_DECL void* allocate_wait_state(void* vls, const volatile void* addr) BOOST_NOEXCEPT
1319{
1320 BOOST_ASSERT(vls != NULL);
1321
1322 lock_state* ls = static_cast< lock_state* >(vls);
1323
1324 // Note: find_or_create may fail to allocate memory. However, C++20 specifies that wait/notify operations
1325 // are noexcept, so allocate_wait_state must succeed. To implement this we return NULL in case of failure and test for NULL
1326 // in other wait/notify functions so that all of them become nop (which is a conforming, though inefficient behavior).
1327 wait_state* ws = ls->m_wait_states.find_or_create(addr);
1328
1329 if (BOOST_LIKELY(ws != NULL))
1330 ++ws->m_ref_count;
1331
1332 return ws;
1333}
