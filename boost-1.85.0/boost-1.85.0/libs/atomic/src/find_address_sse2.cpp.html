38BOOST_FORCEINLINE __m128i mm_pand_si128(__m128i mm1, __m128i mm2)
39{
40 // As of 2020, gcc, clang and icc prefer to generate andps instead of pand if the surrounding
41 // instructions pertain to FP domain, even if we use the _mm_and_si128 intrinsic. In our
42 // algorithm implementation, the FP instruction happen to be shufps, which is not actually
43 // restricted to FP domain (it is actually implemented in a separate MMX EU in Pentium 4 or
44 // a shuffle EU in INT domain in Core 2; on AMD K8/K10 all SSE instructions are implemented in
45 // FADD, FMUL and FMISC EUs regardless of INT/FP data types, and shufps is implemented in FADD/FMUL).
46 // In other words, there should be no domain bypass penalty between shufps and pand.
47 //
48 // This would usually not pose a problem since andps and pand have the same latency and throughput
49 // on most architectures of that age (before SSE4.1). However, it is possible that a newer architecture
50 // runs the SSE2 code path (e.g. because some weird compiler doesn't support SSE4.1 or because
51 // a hypervisor blocks SSE4.1 detection), and there pand may have a better throughput. For example,
52 // Sandy Bridge can execute 3 pand instructions per cycle, but only one andps. For this reason
53 // we prefer to generate pand and not andps.
54#if defined(__GNUC__)
55#if defined(__AVX__)
56 // Generate VEX-coded variant if the code is compiled for AVX and later.
57 __asm__("vpand %1, %0, %0\n\t" : "+x" (mm1) : "x" (mm2));
58#else
59 __asm__("pand %1, %0\n\t" : "+x" (mm1) : "x" (mm2));
60#endif
61#else
62 mm1 = _mm_and_si128(mm1, mm2);
63#endif
64 return mm1;
65}

70//! SSE2 implementation of the \c find_address algorithm
71std::size_t find_address_sse2(const volatile void* addr, const volatile void* const* addrs, std::size_t size)
72{
73#if BOOST_ATOMIC_DETAIL_SIZEOF_POINTER == 8
74
75 if (size < 12u)
76 return find_address_generic(addr, addrs, size);
77
78 const __m128i mm_addr = mm_set1_epiptr(ptr: (uintptr_t)addr);
79 std::size_t pos = 0u;
80 const std::size_t n = (size + 1u) & ~static_cast< std::size_t >(1u);
81 for (std::size_t m = n & ~static_cast< std::size_t >(15u); pos < m; pos += 16u)
82 {
83 __m128i mm1 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos));
84 __m128i mm2 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 2u));
85 __m128i mm3 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 4u));
86 __m128i mm4 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 6u));
87 __m128i mm5 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 8u));
88 __m128i mm6 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 10u));
89 __m128i mm7 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 12u));
90 __m128i mm8 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 14u));
91
92 mm1 = _mm_cmpeq_epi32(a: mm1, b: mm_addr);
93 mm2 = _mm_cmpeq_epi32(a: mm2, b: mm_addr);
94 mm3 = _mm_cmpeq_epi32(a: mm3, b: mm_addr);
95 mm4 = _mm_cmpeq_epi32(a: mm4, b: mm_addr);
96 mm5 = _mm_cmpeq_epi32(a: mm5, b: mm_addr);
97 mm6 = _mm_cmpeq_epi32(a: mm6, b: mm_addr);
98 mm7 = _mm_cmpeq_epi32(a: mm7, b: mm_addr);
99 mm8 = _mm_cmpeq_epi32(a: mm8, b: mm_addr);
100
101 __m128i mm_mask1_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(2, 0, 2, 0)));
102 __m128i mm_mask1_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(3, 1, 3, 1)));
103
104 __m128i mm_mask2_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm3), _mm_castsi128_ps(mm4), _MM_SHUFFLE(2, 0, 2, 0)));
105 __m128i mm_mask2_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm3), _mm_castsi128_ps(mm4), _MM_SHUFFLE(3, 1, 3, 1)));
106
107 __m128i mm_mask3_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm5), _mm_castsi128_ps(mm6), _MM_SHUFFLE(2, 0, 2, 0)));
108 __m128i mm_mask3_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm5), _mm_castsi128_ps(mm6), _MM_SHUFFLE(3, 1, 3, 1)));
109
110 __m128i mm_mask4_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm7), _mm_castsi128_ps(mm8), _MM_SHUFFLE(2, 0, 2, 0)));
111 __m128i mm_mask4_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm7), _mm_castsi128_ps(mm8), _MM_SHUFFLE(3, 1, 3, 1)));
112
113 mm_mask1_lo = mm_pand_si128(mm1: mm_mask1_lo, mm2: mm_mask1_hi);
114 mm_mask2_lo = mm_pand_si128(mm1: mm_mask2_lo, mm2: mm_mask2_hi);
115 mm_mask3_lo = mm_pand_si128(mm1: mm_mask3_lo, mm2: mm_mask3_hi);
116 mm_mask4_lo = mm_pand_si128(mm1: mm_mask4_lo, mm2: mm_mask4_hi);
117
118 mm_mask1_lo = _mm_packs_epi32(a: mm_mask1_lo, b: mm_mask2_lo);
119 mm_mask3_lo = _mm_packs_epi32(a: mm_mask3_lo, b: mm_mask4_lo);
120
121 mm_mask1_lo = _mm_packs_epi16(a: mm_mask1_lo, b: mm_mask3_lo);
122
123 uint32_t mask = _mm_movemask_epi8(a: mm_mask1_lo);
124 if (mask)
125 {
126 pos += atomics::detail::count_trailing_zeros(x: mask);
127 goto done;
128 }
129 }
130
131 if ((n - pos) >= 8u)
132 {
133 __m128i mm1 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos));
134 __m128i mm2 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 2u));
135 __m128i mm3 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 4u));
136 __m128i mm4 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 6u));
137
138 mm1 = _mm_cmpeq_epi32(a: mm1, b: mm_addr);
139 mm2 = _mm_cmpeq_epi32(a: mm2, b: mm_addr);
140 mm3 = _mm_cmpeq_epi32(a: mm3, b: mm_addr);
141 mm4 = _mm_cmpeq_epi32(a: mm4, b: mm_addr);
142
143 __m128i mm_mask1_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(2, 0, 2, 0)));
144 __m128i mm_mask1_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(3, 1, 3, 1)));
145
146 __m128i mm_mask2_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm3), _mm_castsi128_ps(mm4), _MM_SHUFFLE(2, 0, 2, 0)));
147 __m128i mm_mask2_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm3), _mm_castsi128_ps(mm4), _MM_SHUFFLE(3, 1, 3, 1)));
148
149 mm_mask1_lo = mm_pand_si128(mm1: mm_mask1_lo, mm2: mm_mask1_hi);
150 mm_mask2_lo = mm_pand_si128(mm1: mm_mask2_lo, mm2: mm_mask2_hi);
151
152 mm_mask1_lo = _mm_packs_epi32(a: mm_mask1_lo, b: mm_mask2_lo);
153
154 uint32_t mask = _mm_movemask_epi8(a: mm_mask1_lo);
155 if (mask)
156 {
157 pos += atomics::detail::count_trailing_zeros(x: mask) / 2u;
158 goto done;
159 }
160
161 pos += 8u;
162 }
163
164 if ((n - pos) >= 4u)
165 {
166 __m128i mm1 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos));
167 __m128i mm2 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos + 2u));
168
169 mm1 = _mm_cmpeq_epi32(a: mm1, b: mm_addr);
170 mm2 = _mm_cmpeq_epi32(a: mm2, b: mm_addr);
171
172 __m128i mm_mask1_lo = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(2, 0, 2, 0)));
173 __m128i mm_mask1_hi = _mm_castps_si128(_mm_shuffle_ps(_mm_castsi128_ps(mm1), _mm_castsi128_ps(mm2), _MM_SHUFFLE(3, 1, 3, 1)));
174
175 mm_mask1_lo = mm_pand_si128(mm1: mm_mask1_lo, mm2: mm_mask1_hi);
176
177 uint32_t mask = _mm_movemask_ps(a: _mm_castsi128_ps(a: mm_mask1_lo));
178 if (mask)
179 {
180 pos += atomics::detail::count_trailing_zeros(x: mask);
181 goto done;
182 }
183
184 pos += 4u;
185 }
186
187 if (pos < n)
188 {
189 __m128i mm1 = _mm_load_si128(p: reinterpret_cast< const __m128i* >(addrs + pos));
190
191 mm1 = _mm_cmpeq_epi32(a: mm1, b: mm_addr);
192 __m128i mm_mask = _mm_shuffle_epi32(mm1, _MM_SHUFFLE(2, 3, 0, 1));
193 mm_mask = mm_pand_si128(mm1: mm_mask, mm2: mm1);
194
195 uint32_t mask = _mm_movemask_pd(a: _mm_castsi128_pd(a: mm_mask));
196 if (mask)
197 {
198 pos += atomics::detail::count_trailing_zeros(x: mask);
199 goto done;
200 }
201
202 pos += 2u;
203 }
204
205done:
206 return pos;
207
208#else // BOOST_ATOMIC_DETAIL_SIZEOF_POINTER == 8
209
210 if (size < 10u)
211 return find_address_generic(addr, addrs, size);
212
213 const __m128i mm_addr = _mm_set1_epi32((uintptr_t)addr);
214 std::size_t pos = 0u;
215 const std::size_t n = (size + 3u) & ~static_cast< std::size_t >(3u);
216 for (std::size_t m = n & ~static_cast< std::size_t >(15u); pos < m; pos += 16u)
217 {
218 __m128i mm1 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos));
219 __m128i mm2 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos + 4u));
220 __m128i mm3 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos + 8u));
221 __m128i mm4 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos + 12u));
222
223 mm1 = _mm_cmpeq_epi32(mm1, mm_addr);
224 mm2 = _mm_cmpeq_epi32(mm2, mm_addr);
225 mm3 = _mm_cmpeq_epi32(mm3, mm_addr);
226 mm4 = _mm_cmpeq_epi32(mm4, mm_addr);
227
228 mm1 = _mm_packs_epi32(mm1, mm2);
229 mm3 = _mm_packs_epi32(mm3, mm4);
230
231 mm1 = _mm_packs_epi16(mm1, mm3);
232
233 uint32_t mask = _mm_movemask_epi8(mm1);
234 if (mask)
235 {
236 pos += atomics::detail::count_trailing_zeros(mask);
237 goto done;
238 }
239 }
240
241 if ((n - pos) >= 8u)
242 {
243 __m128i mm1 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos));
244 __m128i mm2 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos + 4u));
245
246 mm1 = _mm_cmpeq_epi32(mm1, mm_addr);
247 mm2 = _mm_cmpeq_epi32(mm2, mm_addr);
248
249 mm1 = _mm_packs_epi32(mm1, mm2);
250
251 uint32_t mask = _mm_movemask_epi8(mm1);
252 if (mask)
253 {
254 pos += atomics::detail::count_trailing_zeros(mask) / 2u;
255 goto done;
256 }
257
258 pos += 8u;
259 }
260
261 if (pos < n)
262 {
263 __m128i mm1 = _mm_load_si128(reinterpret_cast< const __m128i* >(addrs + pos));
264
265 mm1 = _mm_cmpeq_epi32(mm1, mm_addr);
266
267 uint32_t mask = _mm_movemask_ps(_mm_castsi128_ps(mm1));
268 if (mask)
269 {
270 pos += atomics::detail::count_trailing_zeros(mask);
271 goto done;
272 }
273
274 pos += 4u;
275 }
276
277done:
278 return pos;
279
280#endif // BOOST_ATOMIC_DETAIL_SIZEOF_POINTER == 8
