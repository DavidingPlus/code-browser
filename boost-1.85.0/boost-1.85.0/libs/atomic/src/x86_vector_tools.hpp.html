29namespace boost {
30namespace atomics {
31namespace detail {
32
33BOOST_FORCEINLINE __m128i mm_set1_epiptr(uintptr_t ptr)
34{
35#if !defined(_MSC_VER) || _MSC_FULL_VER >= 190024210
36 return _mm_set1_epi64x(q: ptr);
37#else
38 // MSVC up until 14.0 update 3 doesn't provide _mm_set1_epi64x
39 uint32_t lo = static_cast< uint32_t >(ptr), hi = static_cast< uint32_t >(ptr >> 32);
40 return _mm_set_epi32(hi, lo, hi, lo);
41#endif
42}
43
44} // namespace detail
