35//! Counts trailing zero bits
36BOOST_FORCEINLINE unsigned int count_trailing_zeros(unsigned int x)
37{
38#if defined(__GNUC__)
39 return __builtin_ctz(x);
40#elif defined(_MSC_VER)
41 unsigned long index;
42 _BitScanForward(&index, x);
43 return static_cast< unsigned int >(index);
44#else
45 unsigned int index = 0u;
46 if ((x & 0xFFFF) == 0u)
47 {
48 x >>= 16;
49 index += 16u;
50 }
51 if ((x & 0xFF) == 0u)
52 {
53 x >>= 8;
54 index += 8u;
55 }
56 if ((x & 0xF) == 0u)
57 {
58 x >>= 4;
59 index += 4u;
60 }
61 if ((x & 0x3) == 0u)
62 {
63 x >>= 2;
64 index += 2u;
65 }
66 if ((x & 0x1) == 0u)
67 {
68 index += 1u;
69 }
70 return index;
71#endif
72}
