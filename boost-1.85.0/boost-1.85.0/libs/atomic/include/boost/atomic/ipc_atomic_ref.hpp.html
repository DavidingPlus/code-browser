31namespace boost {
32namespace atomics {
33
34//! Atomic reference to external object for inter-process communication
35template< typename T >
36class ipc_atomic_ref :
37 public atomics::detail::base_atomic_ref< T, typename atomics::detail::classify< T >::type, true >
38{
39private:
40 typedef atomics::detail::base_atomic_ref< T, typename atomics::detail::classify< T >::type, true > base_type;
41 typedef typename base_type::value_arg_type value_arg_type;
42
43public:
44 typedef typename base_type::value_type value_type;
45
46 static_assert(sizeof(value_type) > 0u, "boost::ipc_atomic_ref<T> requires T to be a complete type");
47#if !defined(BOOST_ATOMIC_DETAIL_NO_CXX11_IS_TRIVIALLY_COPYABLE)
48 static_assert(atomics::detail::is_trivially_copyable< value_type >::value, "boost::ipc_atomic_ref<T> requires T to be a trivially copyable type");
49#endif
50
51private:
52 typedef typename base_type::storage_type storage_type;
53
54public:
55 BOOST_DEFAULTED_FUNCTION(ipc_atomic_ref(ipc_atomic_ref const& that) BOOST_ATOMIC_DETAIL_DEF_NOEXCEPT_DECL, BOOST_ATOMIC_DETAIL_DEF_NOEXCEPT_IMPL : base_type(static_cast< base_type const& >(that)) {})
56 BOOST_FORCEINLINE explicit ipc_atomic_ref(value_type& v) BOOST_NOEXCEPT : base_type(v)
57 {
58 // Check that referenced object alignment satisfies required alignment
59 BOOST_ASSERT((((atomics::detail::uintptr_t)this->m_value) & (base_type::required_alignment - 1u)) == 0u);
60 }
61
62 BOOST_FORCEINLINE value_type operator= (value_arg_type v) const BOOST_NOEXCEPT
63 {
64 this->store(v);
65 return v;
66 }
67
68 BOOST_FORCEINLINE operator value_type() const BOOST_NOEXCEPT
69 {
70 return this->load();
71 }
72
73 BOOST_DELETED_FUNCTION(ipc_atomic_ref& operator= (ipc_atomic_ref const&))
74};
75
76#if !defined(BOOST_NO_CXX17_DEDUCTION_GUIDES)
77template< typename T >
78ipc_atomic_ref(T&) -> ipc_atomic_ref< T >;
79#endif // !defined(BOOST_NO_CXX17_DEDUCTION_GUIDES)
80
81//! IPC atomic reference factory function
82template< typename T >
83BOOST_FORCEINLINE ipc_atomic_ref< T > make_ipc_atomic_ref(T& value) BOOST_NOEXCEPT
84{
85 return ipc_atomic_ref< T >(value);
86}
