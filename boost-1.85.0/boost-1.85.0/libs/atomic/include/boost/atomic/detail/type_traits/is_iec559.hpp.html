11 * This header defines \c is_iec559 type trait

13
14#ifndef BOOST_ATOMIC_DETAIL_TYPE_TRAITS_IS_IEC559_HPP_INCLUDED_
15#define BOOST_ATOMIC_DETAIL_TYPE_TRAITS_IS_IEC559_HPP_INCLUDED_
16
17#include <limits>
18#include <boost/atomic/detail/config.hpp>
19
20#ifdef BOOST_HAS_PRAGMA_ONCE
21#pragma once
22#endif
23
24namespace boost {
25namespace atomics {
26namespace detail {
27
28template< typename T >
29struct is_iec559
30{
31 static BOOST_CONSTEXPR_OR_CONST bool value = !!std::numeric_limits< T >::is_iec559;
32};
33
34#if defined(BOOST_HAS_FLOAT128)
35// libstdc++ does not specialize numeric_limits for __float128
36template< >
37struct is_iec559< boost::float128_type >
38{
39 static BOOST_CONSTEXPR_OR_CONST bool value = true;
40};
41#endif // defined(BOOST_HAS_FLOAT128)
42
43} // namespace detail
44} // namespace atomics
45} // namespace boost
46
47#endif // BOOST_ATOMIC_DETAIL_TYPE_TRAITS_IS_IEC559_HPP_INCLUDED_
