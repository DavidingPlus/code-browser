25namespace boost {
26namespace atomics {
27namespace detail {
28
29//! Fence operations for x86
30struct fence_arch_operations_gcc_x86
31{
32 static BOOST_FORCEINLINE void thread_fence(memory_order order) BOOST_NOEXCEPT
33 {
34 if (order == memory_order_seq_cst)
35 {
36 // We could generate mfence for a seq_cst fence here, but a dummy lock-prefixed instruction is enough
37 // and is faster than mfence on most modern x86 CPUs (as of 2020).
38 // Note that we want to apply the atomic operation on any location so that:
39 // - It is not shared with other threads. A variable on the stack suits this well.
40 // - It is likely in cache. Being close to the top of the stack fits this well.
41 // - It does not alias existing data on the stack, so that we don't introduce a false data dependency.
42 // See some performance data here: https://shipilev.net/blog/2014/on-the-fence-with-dependencies/
43 // Unfortunately, to make tools like valgrind happy, we have to initialize the dummy, which is
44 // otherwise not needed.
45 unsigned char dummy = 0u;
46 __asm__ __volatile__ ("lock; notb %0" : "+m" (dummy) : : "memory");
47 }
48 else if ((static_cast< unsigned int >(order) & (static_cast< unsigned int >(memory_order_acquire) | static_cast< unsigned int >(memory_order_release))) != 0u)
49 {
50 __asm__ __volatile__ ("" ::: "memory");
51 }
52 }
53
54 static BOOST_FORCEINLINE void signal_fence(memory_order order) BOOST_NOEXCEPT
55 {
56 if (order != memory_order_relaxed)
57 __asm__ __volatile__ ("" ::: "memory");
58 }
59};
60
61typedef fence_arch_operations_gcc_x86 fence_arch_operations;
