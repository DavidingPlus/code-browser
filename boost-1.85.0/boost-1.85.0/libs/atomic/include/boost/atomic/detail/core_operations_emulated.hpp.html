33template< std::size_t Size, std::size_t Alignment, bool = Alignment >= storage_traits< Size >::native_alignment >
34struct core_operations_emulated_base
35{
36 typedef typename storage_traits< Size >::type storage_type;
37};
38
39template< std::size_t Size, std::size_t Alignment >
40struct core_operations_emulated_base< Size, Alignment, false >
41{
42 typedef buffer_storage< Size, Alignment > storage_type;
43};
44
45//! Emulated implementation of core atomic operations
46template< std::size_t Size, std::size_t Alignment, bool Signed, bool Interprocess >
47struct core_operations_emulated :
48 public core_operations_emulated_base< Size, Alignment >
49{
50 typedef core_operations_emulated_base< Size, Alignment > base_type;
51
52 // Define storage_type to have alignment not greater than Alignment. This will allow operations to work with value_types
53 // that possibly have weaker alignment requirements than storage_traits< Size >::type would. This is important for atomic_ref<>.
54 // atomic<> will allow higher alignment requirement than its value_type.
55 // Note that storage_type should be an integral type, if possible, so that arithmetic and bitwise operations are possible.
56 typedef typename base_type::storage_type storage_type;
57
58 static BOOST_CONSTEXPR_OR_CONST std::size_t storage_size = Size;
59 static BOOST_CONSTEXPR_OR_CONST std::size_t storage_alignment = Alignment >= storage_traits< Size >::alignment ? storage_traits< Size >::alignment : Alignment;
60
61 static BOOST_CONSTEXPR_OR_CONST bool is_signed = Signed;
62 static BOOST_CONSTEXPR_OR_CONST bool is_interprocess = Interprocess;
63 static BOOST_CONSTEXPR_OR_CONST bool full_cas_based = false;
64
65 static BOOST_CONSTEXPR_OR_CONST bool is_always_lock_free = false;
66
67 typedef lock_pool::scoped_lock< storage_alignment > scoped_lock;
68
69 static void store(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
70 {
71 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
72 scoped_lock lock(&storage);
73 const_cast< storage_type& >(storage) = v;
74 }
75
76 static storage_type load(storage_type const volatile& storage, memory_order) BOOST_NOEXCEPT
77 {
78 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
79 scoped_lock lock(&storage);
80 return const_cast< storage_type const& >(storage);
81 }
82
83 static storage_type fetch_add(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
84 {
85 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
86 storage_type& s = const_cast< storage_type& >(storage);
87 scoped_lock lock(&storage);
88 storage_type old_val = s;
89 s += v;
90 return old_val;
91 }
92
93 static storage_type fetch_sub(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
94 {
95 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
96 storage_type& s = const_cast< storage_type& >(storage);
97 scoped_lock lock(&storage);
98 storage_type old_val = s;
99 s -= v;
100 return old_val;
101 }
102
103 static storage_type exchange(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
104 {
105 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
106 storage_type& s = const_cast< storage_type& >(storage);
107 scoped_lock lock(&storage);
108 storage_type old_val = s;
109 s = v;
110 return old_val;
111 }
112
113 static bool compare_exchange_strong(
114 storage_type volatile& storage, storage_type& expected, storage_type desired, memory_order, memory_order) BOOST_NOEXCEPT
115 {
116 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
117 storage_type& s = const_cast< storage_type& >(storage);
118 scoped_lock lock(&storage);
119 storage_type old_val = s;
120 const bool res = old_val == expected;
121 if (res)
122 s = desired;
123 expected = old_val;
124
125 return res;
126 }
127
128 static bool compare_exchange_weak(
129 storage_type volatile& storage, storage_type& expected, storage_type desired, memory_order, memory_order) BOOST_NOEXCEPT
130 {
131 // Note: This function is the exact copy of compare_exchange_strong. The reason we're not just forwarding the call
132 // is that MSVC-12 ICEs in this case.
133 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
134 storage_type& s = const_cast< storage_type& >(storage);
135 scoped_lock lock(&storage);
136 storage_type old_val = s;
137 const bool res = old_val == expected;
138 if (res)
139 s = desired;
140 expected = old_val;
141
142 return res;
143 }
144
145 static storage_type fetch_and(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
146 {
147 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
148 storage_type& s = const_cast< storage_type& >(storage);
149 scoped_lock lock(&storage);
150 storage_type old_val = s;
151 s &= v;
152 return old_val;
153 }
154
155 static storage_type fetch_or(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
156 {
157 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
158 storage_type& s = const_cast< storage_type& >(storage);
159 scoped_lock lock(&storage);
160 storage_type old_val = s;
161 s |= v;
162 return old_val;
163 }
164
165 static storage_type fetch_xor(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
166 {
167 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
168 storage_type& s = const_cast< storage_type& >(storage);
169 scoped_lock lock(&storage);
170 storage_type old_val = s;
171 s ^= v;
172 return old_val;
173 }
174
175 static BOOST_FORCEINLINE bool test_and_set(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
176 {
177 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
178 return !!exchange(storage, v: (storage_type)1, order);
179 }
180
181 static BOOST_FORCEINLINE void clear(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
182 {
183 static_assert(!is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
184 store(storage, v: (storage_type)0, order);
185 }
186};
