25namespace boost {
26namespace atomics {
27namespace detail {
28
29BOOST_FORCEINLINE BOOST_CONSTEXPR memory_order deduce_failure_order(memory_order order) BOOST_NOEXCEPT
30{
31 return order == memory_order_acq_rel ? memory_order_acquire : (order == memory_order_release ? memory_order_relaxed : order);
32}
33
34BOOST_FORCEINLINE BOOST_CONSTEXPR bool cas_failure_order_must_not_be_stronger_than_success_order(memory_order success_order, memory_order failure_order) BOOST_NOEXCEPT
35{
36 // 15 == (memory_order_seq_cst | memory_order_consume), see memory_order.hpp
37 // Given the enum values we can test the strength of memory order requirements with this single condition.
38 return (static_cast< unsigned int >(failure_order) & 15u) <= (static_cast< unsigned int >(success_order) & 15u);
39}
