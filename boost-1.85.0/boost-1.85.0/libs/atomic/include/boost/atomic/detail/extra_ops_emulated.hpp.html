32//! Emulated implementation of extra operations
33template< typename Base, std::size_t Size, bool Signed >
34struct extra_operations_emulated :
35 public Base
36{
37 typedef Base base_type;
38 typedef typename base_type::storage_type storage_type;
39 typedef typename base_type::scoped_lock scoped_lock;
40
41 static storage_type fetch_negate(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
42 {
43 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
44 storage_type& s = const_cast< storage_type& >(storage);
45 scoped_lock lock(&storage);
46 storage_type old_val = s;
47 s = static_cast< storage_type >(-old_val);
48 return old_val;
49 }
50
51 static storage_type negate(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
52 {
53 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
54 storage_type& s = const_cast< storage_type& >(storage);
55 scoped_lock lock(&storage);
56 storage_type new_val = static_cast< storage_type >(-s);
57 s = new_val;
58 return new_val;
59 }
60
61 static storage_type add(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
62 {
63 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
64 storage_type& s = const_cast< storage_type& >(storage);
65 scoped_lock lock(&storage);
66 storage_type new_val = s;
67 new_val += v;
68 s = new_val;
69 return new_val;
70 }
71

75 storage_type& s = const_cast< storage_type& >(storage);
76 scoped_lock lock(&storage);
77 storage_type new_val = s;

79 s = new_val;
80 return new_val;
81 }

85 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
86 storage_type& s = const_cast< storage_type& >(storage);

89 new_val &= v;
90 s = new_val;
91 return new_val;
92 }

94 static storage_type bitwise_or(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
95 {
96 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
97 storage_type& s = const_cast< storage_type& >(storage);
98 scoped_lock lock(&storage);
99 storage_type new_val = s;
100 new_val |= v;
101 s = new_val;
102 return new_val;
103 }
104
105 static storage_type bitwise_xor(storage_type volatile& storage, storage_type v, memory_order) BOOST_NOEXCEPT
106 {
107 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
108 storage_type& s = const_cast< storage_type& >(storage);
109 scoped_lock lock(&storage);
110 storage_type new_val = s;
111 new_val ^= v;
112 s = new_val;
113 return new_val;
114 }
115
116 static storage_type fetch_complement(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
117 {
118 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
119 storage_type& s = const_cast< storage_type& >(storage);
120 scoped_lock lock(&storage);
121 storage_type old_val = s;
122 s = static_cast< storage_type >(~old_val);
123 return old_val;
124 }
125
126 static storage_type bitwise_complement(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
127 {
128 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
129 storage_type& s = const_cast< storage_type& >(storage);
130 scoped_lock lock(&storage);
131 storage_type new_val = static_cast< storage_type >(~s);
132 s = new_val;
133 return new_val;
134 }

136 static BOOST_FORCEINLINE void opaque_add(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
137 {
138 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
139 base_type::fetch_add(storage, v, order);
140 }
141

143 {
144 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
145 base_type::fetch_sub(storage, v, order);
146 }

151 fetch_negate(storage, order);
152 }

156 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");

178 static BOOST_FORCEINLINE bool add_and_test(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
179 {

214 static BOOST_FORCEINLINE bool complement_and_test(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
215 {
216 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
217 return !!bitwise_complement(storage, order);
218 }
219
220 static BOOST_FORCEINLINE bool bit_test_and_set(storage_type volatile& storage, unsigned int bit_number, memory_order order) BOOST_NOEXCEPT
221 {
222 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
223 storage_type mask = static_cast< storage_type >(static_cast< storage_type >(1u) << bit_number);
224 storage_type old_val = base_type::fetch_or(storage, mask, order);
225 return !!(old_val & mask);
226 }
227
228 static BOOST_FORCEINLINE bool bit_test_and_reset(storage_type volatile& storage, unsigned int bit_number, memory_order order) BOOST_NOEXCEPT
229 {
230 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
231 storage_type mask = static_cast< storage_type >(static_cast< storage_type >(1u) << bit_number);
232 storage_type old_val = base_type::fetch_and(storage, ~mask, order);
233 return !!(old_val & mask);
234 }
235
236 static BOOST_FORCEINLINE bool bit_test_and_complement(storage_type volatile& storage, unsigned int bit_number, memory_order order) BOOST_NOEXCEPT
237 {
238 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
239 storage_type mask = static_cast< storage_type >(static_cast< storage_type >(1u) << bit_number);
240 storage_type old_val = base_type::fetch_xor(storage, mask, order);
241 return !!(old_val & mask);
242 }
243};
