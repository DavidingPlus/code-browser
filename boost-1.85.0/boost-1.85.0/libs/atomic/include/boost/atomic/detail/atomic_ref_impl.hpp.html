54template< typename T, bool Signed, bool Interprocess >
55struct is_atomic_ref_lock_free
56{
57 typedef T value_type;
58 typedef atomics::detail::core_operations< sizeof(value_type), Signed, Interprocess > core_operations;
59 typedef typename core_operations::storage_type storage_type;
60
61 static BOOST_CONSTEXPR_OR_CONST bool value = sizeof(value_type) == sizeof(storage_type) && core_operations::is_always_lock_free;
62};
63
64template< typename T, bool Signed, bool Interprocess >
65class base_atomic_ref_common
66{
67public:
68 typedef T value_type;
69
70protected:
71 typedef typename atomics::detail::remove_cv< value_type >::type unqualified_value_type;
72 typedef typename atomics::detail::conditional<
73 atomics::detail::is_atomic_ref_lock_free< T, Signed, Interprocess >::value,
74 atomics::detail::core_operations< sizeof(value_type), Signed, Interprocess >,
75 atomics::detail::core_operations_emulated< sizeof(value_type), atomics::detail::alignment_of< value_type >::value, Signed, Interprocess >
76 >::type core_operations;
77 typedef atomics::detail::wait_operations< core_operations > wait_operations;
78 typedef typename atomics::detail::conditional< sizeof(value_type) <= sizeof(void*), value_type, value_type const& >::type value_arg_type;
79 typedef typename core_operations::storage_type storage_type;
80 static_assert(sizeof(storage_type) == sizeof(value_type), "Boost.Atomic internal error: atomic_ref storage size doesn't match the value size");
81
82public:
83 static BOOST_CONSTEXPR_OR_CONST std::size_t required_alignment = atomics::detail::alignment_of< value_type >::value <= core_operations::storage_alignment ? core_operations::storage_alignment : atomics::detail::alignment_of< value_type >::value;
84 static BOOST_CONSTEXPR_OR_CONST bool is_always_lock_free = core_operations::is_always_lock_free;
85 static BOOST_CONSTEXPR_OR_CONST bool always_has_native_wait_notify = wait_operations::always_has_native_wait_notify;
86
87protected:
88 value_type* m_value;
89
90public:
91 BOOST_FORCEINLINE explicit base_atomic_ref_common(value_type& v) BOOST_NOEXCEPT : m_value(atomics::detail::addressof(v))
92 {
93 BOOST_ATOMIC_DETAIL_CLEAR_PADDING(const_cast< unqualified_value_type* >(m_value));
94 }
95
96 BOOST_FORCEINLINE value_type& value() const BOOST_NOEXCEPT { return *m_value; }
97
98protected:
99 BOOST_FORCEINLINE storage_type& storage() const BOOST_NOEXCEPT
100 {
101 return *reinterpret_cast< storage_type* >(const_cast< unqualified_value_type* >(m_value));
102 }
103
104public:
105 BOOST_FORCEINLINE bool is_lock_free() const BOOST_NOEXCEPT
106 {
107 // C++20 specifies that is_lock_free returns true if operations on *all* objects of the atomic_ref<T> type are lock-free.
108 // This does not allow to return true or false depending on the referenced object runtime alignment. Currently, Boost.Atomic
109 // follows this specification, although we may support runtime alignment checking in the future.
110 return is_always_lock_free;
111 }
112
113 BOOST_FORCEINLINE bool has_native_wait_notify() const BOOST_NOEXCEPT
114 {
115 return wait_operations::has_native_wait_notify(this->storage());
116 }
117
118 BOOST_FORCEINLINE void notify_one() const BOOST_NOEXCEPT
119 {
120 wait_operations::notify_one(this->storage());
121 }
122
123 BOOST_FORCEINLINE void notify_all() const BOOST_NOEXCEPT
124 {
125 wait_operations::notify_all(this->storage());
126 }
127};
