10/*!
11 * \file atomic/detail/caps_arch_gcc_x86.hpp
12 *
13 * This header defines feature capabilities macros
14 */
15
16#ifndef BOOST_ATOMIC_DETAIL_CAPS_ARCH_GCC_X86_HPP_INCLUDED_
17#define BOOST_ATOMIC_DETAIL_CAPS_ARCH_GCC_X86_HPP_INCLUDED_
18
19#include <boost/atomic/detail/config.hpp>
20
21#ifdef BOOST_HAS_PRAGMA_ONCE
22#pragma once
23#endif
24
25#if defined(__GNUC__)
26
27#if defined(__i386__) &&\
28 (\
29 defined(__GCC_HAVE_SYNC_COMPARE_AND_SWAP_8) ||\
30 defined(__i586__) || defined(__i686__) || defined(__SSE__)\
31 )
32#define BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG8B 1
33#endif
34
35#if defined(__x86_64__) && defined(__GCC_HAVE_SYNC_COMPARE_AND_SWAP_16)
36#define BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG16B 1
37#endif
38
39#if defined(__x86_64__) || defined(__SSE2__)
40// Use mfence only if SSE2 is available
41#define BOOST_ATOMIC_DETAIL_X86_HAS_MFENCE 1
42#endif
43
44#else // defined(__GNUC__)
45
46#if defined(__i386__) && !defined(BOOST_ATOMIC_NO_CMPXCHG8B)
47#define BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG8B 1
48#endif
49
50#if defined(__x86_64__) && !defined(BOOST_ATOMIC_NO_CMPXCHG16B)
51#define BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG16B 1
52#endif
53
54#if !defined(BOOST_ATOMIC_NO_MFENCE)
55#define BOOST_ATOMIC_DETAIL_X86_HAS_MFENCE 1
56#endif
57
58#endif // defined(__GNUC__)
59
60#define BOOST_ATOMIC_INT8_LOCK_FREE 2
61#define BOOST_ATOMIC_INT16_LOCK_FREE 2
62#define BOOST_ATOMIC_INT32_LOCK_FREE 2
63#if defined(__x86_64__) || defined(BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG8B)
64#define BOOST_ATOMIC_INT64_LOCK_FREE 2
65#endif
66#if defined(BOOST_ATOMIC_DETAIL_X86_HAS_CMPXCHG16B)
67#define BOOST_ATOMIC_INT128_LOCK_FREE 2
68#endif
69#define BOOST_ATOMIC_POINTER_LOCK_FREE 2
70
71#define BOOST_ATOMIC_THREAD_FENCE 2
72#define BOOST_ATOMIC_SIGNAL_FENCE 2
73
74#endif // BOOST_ATOMIC_DETAIL_CAPS_ARCH_GCC_X86_HPP_INCLUDED_
