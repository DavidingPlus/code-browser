14#ifndef BOOST_ATOMIC_DETAIL_CLASSIFY_HPP_INCLUDED_
15#define BOOST_ATOMIC_DETAIL_CLASSIFY_HPP_INCLUDED_

17#include <boost/atomic/detail/config.hpp>
18#include <boost/atomic/detail/type_traits/is_enum.hpp>
19#include <boost/atomic/detail/type_traits/is_integral.hpp>
20#include <boost/atomic/detail/type_traits/is_function.hpp>
21#include <boost/atomic/detail/type_traits/is_floating_point.hpp>
22#include <boost/atomic/detail/header.hpp>
23
24#ifdef BOOST_HAS_PRAGMA_ONCE
25#pragma once
26#endif
27
28namespace boost {
29namespace atomics {
30namespace detail {
31
32template< typename T, bool IsFunction = atomics::detail::is_function< T >::value >
33struct classify_pointer
34{
35 typedef void* type;
36};
37
38template< typename T >
39struct classify_pointer< T, true >
40{
41 typedef void type;
42};
43
44template<
45 typename T,
46 bool IsInt = atomics::detail::is_integral< T >::value,
47 bool IsFloat = atomics::detail::is_floating_point< T >::value,
48 bool IsEnum = atomics::detail::is_enum< T >::value
49>
50struct classify
51{
52 typedef void type;
53};
54
55template< typename T >
56struct classify< T, true, false, false > { typedef int type; };
57
58#if !defined(BOOST_ATOMIC_NO_FLOATING_POINT)
59template< typename T >
60struct classify< T, false, true, false > { typedef float type; };
61#endif
62
63template< typename T >
64struct classify< T, false, false, true > { typedef const int type; };
65
66template< typename T >
67struct classify< T*, false, false, false > { typedef typename classify_pointer< T >::type type; };
68
69template< >
70struct classify< void*, false, false, false > { typedef void type; };
71
72template< >
73struct classify< const void*, false, false, false > { typedef void type; };
74
75template< >
76struct classify< volatile void*, false, false, false > { typedef void type; };
77
78template< >
79struct classify< const volatile void*, false, false, false > { typedef void type; };
80
81template< typename T, typename U >
82struct classify< T U::*, false, false, false > { typedef void type; };
83
84} // namespace detail
85} // namespace atomics
86} // namespace boost
