34/*!
35 * \brief The type trait returns the size of the value of the specified floating point type
36 *
37 * This size may be less than <tt>sizeof(T)</tt> if the implementation uses padding bytes for a particular FP type. This is
38 * often the case with 80-bit extended double, which is stored in 12 or 16 initial bytes with tail padding filled with garbage.
39 */
40template< typename T >
41struct value_size_of
42{
43 static BOOST_CONSTEXPR_OR_CONST std::size_t value = sizeof(T);
44};
45
46#if defined(BOOST_ATOMIC_DETAIL_SIZEOF_FLOAT_VALUE)
47template< >
48struct value_size_of< float >
49{
50 static BOOST_CONSTEXPR_OR_CONST std::size_t value = BOOST_ATOMIC_DETAIL_SIZEOF_FLOAT_VALUE;
51};
52#endif
53
54#if defined(BOOST_ATOMIC_DETAIL_SIZEOF_DOUBLE_VALUE)
55template< >
56struct value_size_of< double >
57{
58 static BOOST_CONSTEXPR_OR_CONST std::size_t value = BOOST_ATOMIC_DETAIL_SIZEOF_DOUBLE_VALUE;
59};
60#endif
61
62#if defined(BOOST_ATOMIC_DETAIL_SIZEOF_LONG_DOUBLE_VALUE)
63template< >
64struct value_size_of< long double >
65{
66 static BOOST_CONSTEXPR_OR_CONST std::size_t value = BOOST_ATOMIC_DETAIL_SIZEOF_LONG_DOUBLE_VALUE;
67};
68#endif
69
70template< typename T >
71struct value_size_of< const T > : value_size_of< T > {};
72
73template< typename T >
74struct value_size_of< volatile T > : value_size_of< T > {};
75
76template< typename T >
77struct value_size_of< const volatile T > : value_size_of< T > {};

80#if !defined(BOOST_ATOMIC_NO_CLEAR_PADDING)
81// BOOST_ATOMIC_DETAIL_CLEAR_PADDING, which is used in bitwise_cast, will clear the tail padding bits in the source object.
82// We don't need to specify the actual value size to avoid redundant zeroing of the tail padding.
83#define BOOST_ATOMIC_DETAIL_BITWISE_FP_CAST_VALUE_SIZE_OF(x) sizeof(x)
84#else
85#define BOOST_ATOMIC_DETAIL_BITWISE_FP_CAST_VALUE_SIZE_OF(x) atomics::detail::value_size_of< x >::value
86#endif

88#if defined(BOOST_ATOMIC_DETAIL_BIT_CAST)
89
90//! Similar to bitwise_cast, but either \c From or \c To is expected to be a floating point type. Attempts to detect the actual value size in the source object and considers the rest of the object as padding.
91template< typename To, typename From >
92BOOST_FORCEINLINE BOOST_ATOMIC_DETAIL_CONSTEXPR_BITWISE_CAST To bitwise_fp_cast(From const& from) BOOST_NOEXCEPT
93{
94 // For floating point types, has_unique_object_representations is typically false even if the type contains no padding bits.
95 // Here, we rely on our detection of the actual value size to select constexpr bit_cast implementation when possible. We assume
96 // here that floating point value bits are contiguous.
97 return atomics::detail::bitwise_cast_impl< To, BOOST_ATOMIC_DETAIL_BITWISE_FP_CAST_VALUE_SIZE_OF(From) >(from, atomics::detail::integral_constant< bool,
98 atomics::detail::value_size_of< From >::value == sizeof(From) && atomics::detail::value_size_of< From >::value == sizeof(To) >());
99}
100
101#else // defined(BOOST_ATOMIC_DETAIL_BIT_CAST)
102
103//! Similar to bitwise_cast, but either \c From or \c To is expected to be a floating point type. Attempts to detect the actual value size in the source object and considers the rest of the object as padding.
104template< typename To, typename From >
105BOOST_FORCEINLINE BOOST_ATOMIC_DETAIL_CONSTEXPR_BITWISE_CAST To bitwise_fp_cast(From const& from) BOOST_NOEXCEPT
106{
107 return atomics::detail::bitwise_cast< To, BOOST_ATOMIC_DETAIL_BITWISE_FP_CAST_VALUE_SIZE_OF(From) >(from);
108}

110#endif // defined(BOOST_ATOMIC_DETAIL_BIT_CAST)
