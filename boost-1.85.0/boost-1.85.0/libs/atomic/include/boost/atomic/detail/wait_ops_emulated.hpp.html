32//! Emulated implementation of waiting and notifying operations
33template< typename Base >
34struct wait_operations_emulated :
35 public Base
36{
37 typedef Base base_type;
38 typedef typename base_type::storage_type storage_type;
39 typedef lock_pool::scoped_lock< base_type::storage_alignment, true > scoped_lock;
40 typedef lock_pool::scoped_wait_state< base_type::storage_alignment > scoped_wait_state;
41
42 static BOOST_CONSTEXPR_OR_CONST bool always_has_native_wait_notify = false;
43
44 static BOOST_FORCEINLINE bool has_native_wait_notify(storage_type const volatile&) BOOST_NOEXCEPT
45 {
46 return false;
47 }
48
49 static
50#if defined(BOOST_MSVC) && BOOST_MSVC < 1500
51 // In some cases, when this function is inlined, MSVC-8 (VS2005) x64 generates broken code that returns a bogus value from this function.
52 BOOST_NOINLINE
53#endif
54 storage_type wait(storage_type const volatile& storage, storage_type old_val, memory_order) BOOST_NOEXCEPT
55 {
56 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
57 storage_type const& s = const_cast< storage_type const& >(storage);
58 scoped_wait_state wait_state(&storage);
59 storage_type new_val = s;
60 while (new_val == old_val)
61 {
62 wait_state.wait();
63 new_val = s;
64 }
65
66 return new_val;
67 }
68
69 static void notify_one(storage_type volatile& storage) BOOST_NOEXCEPT
70 {
71 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
72 scoped_lock lock(&storage);
73 lock_pool::notify_one(ls: lock.get_lock_state(), addr: &storage);
74 }
75
76 static void notify_all(storage_type volatile& storage) BOOST_NOEXCEPT
77 {
78 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
79 scoped_lock lock(&storage);
80 lock_pool::notify_all(ls: lock.get_lock_state(), addr: &storage);
81 }
82};
83
84template< typename Base, std::size_t Size, bool Interprocess >
85struct wait_operations< Base, Size, false, Interprocess > :
86 public wait_operations_emulated< Base >
87{
88};
