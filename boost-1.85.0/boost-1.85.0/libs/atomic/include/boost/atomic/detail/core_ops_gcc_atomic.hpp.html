51template< std::size_t Size, bool Signed, bool Interprocess >
52struct core_operations_gcc_atomic
53{
54 typedef typename storage_traits< Size >::type storage_type;
55
56 static BOOST_CONSTEXPR_OR_CONST std::size_t storage_size = Size;
57 static BOOST_CONSTEXPR_OR_CONST std::size_t storage_alignment = storage_traits< Size >::alignment;
58 static BOOST_CONSTEXPR_OR_CONST bool is_signed = Signed;
59 static BOOST_CONSTEXPR_OR_CONST bool is_interprocess = Interprocess;
60 static BOOST_CONSTEXPR_OR_CONST bool full_cas_based = false;
61
62 // Note: In the current implementation, core_operations_gcc_atomic are used only when the particularly sized __atomic
63 // intrinsics are always lock-free (i.e. the corresponding LOCK_FREE macro is 2). Therefore it is safe to
64 // always set is_always_lock_free to true here.
65 static BOOST_CONSTEXPR_OR_CONST bool is_always_lock_free = true;
66
67 static BOOST_FORCEINLINE void store(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
68 {
69#if defined(BOOST_GCC) && BOOST_GCC < 100100 && (defined(__x86_64__) || defined(__i386__))
70 // gcc up to 10.1 generates mov + mfence for seq_cst stores, which is slower than xchg
71 if (order != memory_order_seq_cst)
72 __atomic_store_n(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
73 else
74 __atomic_exchange_n(&storage, v, __ATOMIC_SEQ_CST);
75#else
76 __atomic_store_n(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
77#endif
78 }
79
80 static BOOST_FORCEINLINE storage_type load(storage_type const volatile& storage, memory_order order) BOOST_NOEXCEPT
81 {
82#if defined(BOOST_ATOMIC_DETAIL_AARCH64_HAS_RCPC)
83 // At least gcc 9.3 and clang 10 do not generate relaxed ldapr instructions that are available in ARMv8.3-RCPC extension.
84 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=95751
85 typedef atomics::detail::core_arch_operations< storage_size, is_signed, is_interprocess > core_arch_operations;
86 return core_arch_operations::load(storage, order);
87#else
88 return __atomic_load_n(&storage, atomics::detail::convert_memory_order_to_gcc(order));
89#endif
90 }
91
92 static BOOST_FORCEINLINE storage_type fetch_add(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
93 {
94 return __atomic_fetch_add(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
95 }
96
97 static BOOST_FORCEINLINE storage_type fetch_sub(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
98 {
99 return __atomic_fetch_sub(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
100 }
101
102 static BOOST_FORCEINLINE storage_type exchange(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
103 {
104 return __atomic_exchange_n(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
105 }
106
107 static BOOST_FORCEINLINE bool compare_exchange_strong(
108 storage_type volatile& storage, storage_type& expected, storage_type desired, memory_order success_order, memory_order failure_order) BOOST_NOEXCEPT
109 {
110 return __atomic_compare_exchange_n
111 (
112 &storage, &expected, desired, false,
113 atomics::detail::convert_memory_order_to_gcc(order: success_order),
114 atomics::detail::convert_memory_order_to_gcc(order: failure_order)
115 );
116 }
117
118 static BOOST_FORCEINLINE bool compare_exchange_weak(
119 storage_type volatile& storage, storage_type& expected, storage_type desired, memory_order success_order, memory_order failure_order) BOOST_NOEXCEPT
120 {
121 return __atomic_compare_exchange_n
122 (
123 &storage, &expected, desired, true,
124 atomics::detail::convert_memory_order_to_gcc(order: success_order),
125 atomics::detail::convert_memory_order_to_gcc(order: failure_order)
126 );
127 }
128
129 static BOOST_FORCEINLINE storage_type fetch_and(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
130 {
131 return __atomic_fetch_and(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
132 }
133
134 static BOOST_FORCEINLINE storage_type fetch_or(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
135 {
136 return __atomic_fetch_or(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
137 }
138
139 static BOOST_FORCEINLINE storage_type fetch_xor(storage_type volatile& storage, storage_type v, memory_order order) BOOST_NOEXCEPT
140 {
141 return __atomic_fetch_xor(&storage, v, atomics::detail::convert_memory_order_to_gcc(order));
142 }
143
144 static BOOST_FORCEINLINE bool test_and_set(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
145 {
146 return __atomic_test_and_set(&storage, atomics::detail::convert_memory_order_to_gcc(order));
147 }
148
149 static BOOST_FORCEINLINE void clear(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
150 {
151 __atomic_clear(const_cast< storage_type* >(&storage), atomics::detail::convert_memory_order_to_gcc(order));
152 }
153};

155// We want to only enable __atomic* intrinsics when the corresponding BOOST_ATOMIC_DETAIL_GCC_ATOMIC_*_LOCK_FREE macro indicates
156// the same or better lock-free guarantees as the BOOST_ATOMIC_*_LOCK_FREE macro. Otherwise, we want to leave core_operations
157// unspecialized, so that core_arch_operations is used instead.

162struct core_operations< 16u, Signed, Interprocess > :

165};
