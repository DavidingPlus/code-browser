28namespace boost {
29namespace atomics {
30namespace detail {

32//! Emulated implementation of extra floating point operations
33template< typename Base, typename Value, std::size_t Size >
34struct extra_fp_operations_emulated :
35 public Base
36{
37 typedef Base base_type;
38 typedef typename base_type::storage_type storage_type;
39 typedef Value value_type;
40 typedef typename base_type::scoped_lock scoped_lock;
41
42 static value_type fetch_negate(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
43 {
44 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
45 storage_type& s = const_cast< storage_type& >(storage);
46 scoped_lock lock(&storage);
47 value_type old_val = atomics::detail::bitwise_fp_cast< value_type >(s);
48 value_type new_val = -old_val;
49 s = atomics::detail::bitwise_fp_cast< storage_type >(new_val);
50 return old_val;
51 }
52
53 static value_type negate(storage_type volatile& storage, memory_order) BOOST_NOEXCEPT
54 {
55 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
56 storage_type& s = const_cast< storage_type& >(storage);
57 scoped_lock lock(&storage);
58 value_type old_val = atomics::detail::bitwise_fp_cast< value_type >(s);
59 value_type new_val = -old_val;
60 s = atomics::detail::bitwise_fp_cast< storage_type >(new_val);
61 return new_val;
62 }
63
64 static value_type add(storage_type volatile& storage, value_type v, memory_order) BOOST_NOEXCEPT
65 {
66 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
67 storage_type& s = const_cast< storage_type& >(storage);
68 scoped_lock lock(&storage);
69 value_type old_val = atomics::detail::bitwise_fp_cast< value_type >(s);
70 value_type new_val = old_val + v;
71 s = atomics::detail::bitwise_fp_cast< storage_type >(new_val);
72 return new_val;
73 }
74
75 static value_type sub(storage_type volatile& storage, value_type v, memory_order) BOOST_NOEXCEPT
76 {
77 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
78 storage_type& s = const_cast< storage_type& >(storage);
79 scoped_lock lock(&storage);
80 value_type old_val = atomics::detail::bitwise_fp_cast< value_type >(s);
81 value_type new_val = old_val - v;
82 s = atomics::detail::bitwise_fp_cast< storage_type >(new_val);
83 return new_val;
84 }
85
86 static BOOST_FORCEINLINE void opaque_negate(storage_type volatile& storage, memory_order order) BOOST_NOEXCEPT
87 {
88 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
89 fetch_negate(storage, order);
90 }
91
92 static BOOST_FORCEINLINE void opaque_add(storage_type volatile& storage, value_type v, memory_order order) BOOST_NOEXCEPT
93 {
94 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
95 base_type::fetch_add(storage, v, order);
96 }
97
98 static BOOST_FORCEINLINE void opaque_sub(storage_type volatile& storage, value_type v, memory_order order) BOOST_NOEXCEPT
99 {
100 static_assert(!base_type::is_interprocess, "Boost.Atomic: operation invoked on a non-lock-free inter-process atomic object");
101 base_type::fetch_sub(storage, v, order);
102 }
103};
104
105template< typename Base, typename Value, std::size_t Size >
106struct extra_fp_operations< Base, Value, Size, false > :
107 public extra_fp_operations_emulated< Base, Value, Size >
108{
109};
