111//----------------------------------------------------------------------------//
112// //
113// 20.9.5 Clocks [time.clock] //
114// //
115//----------------------------------------------------------------------------//
116
117// If you're porting, clocks are the system-specific (non-portable) part.
118// You'll need to know how to get the current time and implement that under now().
119// You'll need to know what units (tick period) and representation makes the most
120// sense for your clock and set those accordingly.
121// If you know how to map this clock to time_t (perhaps your clock is std::time, which
122// makes that trivial), then you can fill out system_clock's to_time_t() and from_time_t().
123
124//----------------------------------------------------------------------------//
125// 20.9.5.1 Class system_clock [time.clock.system] //
126//----------------------------------------------------------------------------//
127
128 class BOOST_CHRONO_DECL system_clock
129 {
130 public:
131 typedef BOOST_SYSTEM_CLOCK_DURATION duration;
132 typedef duration::rep rep;
133 typedef duration::period period;
134 typedef chrono::time_point<system_clock> time_point;
135 BOOST_STATIC_CONSTEXPR bool is_steady = false;
136
137 static BOOST_CHRONO_INLINE time_point now() BOOST_NOEXCEPT;
138#if !defined BOOST_CHRONO_DONT_PROVIDE_HYBRID_ERROR_HANDLING
139 static BOOST_CHRONO_INLINE time_point now(system::error_code & ec);
140#endif
141
142 static BOOST_CHRONO_INLINE std::time_t to_time_t(const time_point& t) BOOST_NOEXCEPT;
143 static BOOST_CHRONO_INLINE time_point from_time_t(std::time_t t) BOOST_NOEXCEPT;
144 };
145
146//----------------------------------------------------------------------------//
147// 20.9.5.2 Class steady_clock [time.clock.steady] //
148//----------------------------------------------------------------------------//
149
150// As permitted by [time.clock.steady]
151// The class steady_clock is conditionally supported.
152
153#ifdef BOOST_CHRONO_HAS_CLOCK_STEADY
154 class BOOST_CHRONO_DECL steady_clock
155 {
156 public:
157 typedef nanoseconds duration;
158 typedef duration::rep rep;
159 typedef duration::period period;
160 typedef chrono::time_point<steady_clock> time_point;
161 BOOST_STATIC_CONSTEXPR bool is_steady = true;
162
163 static BOOST_CHRONO_INLINE time_point now() BOOST_NOEXCEPT;
164#if !defined BOOST_CHRONO_DONT_PROVIDE_HYBRID_ERROR_HANDLING
165 static BOOST_CHRONO_INLINE time_point now(system::error_code & ec);
166#endif
167 };
168#endif
