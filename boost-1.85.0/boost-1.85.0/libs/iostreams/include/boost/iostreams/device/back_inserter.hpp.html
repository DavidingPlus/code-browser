18namespace boost { namespace iostreams {
19
20template<typename Container>
21class back_insert_device {
22public:
23 typedef typename Container::value_type char_type;
24 typedef sink_tag category;
25 back_insert_device(Container& cnt) : container(&cnt) { }
26 std::streamsize write(const char_type* s, std::streamsize n)
27 {
28 container->insert(container->end(), s, s + n);
29 return n;
30 }
31protected:
32 Container* container;
33};
34
35template<typename Container>
36back_insert_device<Container> back_inserter(Container& cnt)
37{ return back_insert_device<Container>(cnt); }
38
39} } // End namespaces iostreams, boost.
