19namespace hash_detail

22template<class It, class T, class S>
23 std::integral_constant< bool, std::is_same<typename std::iterator_traits<It>::value_type, T>::value && std::is_integral<S>::value >
24 is_contiguous_range_check( It first, It last, T const*, T const*, S );
25
26template<class T> decltype( is_contiguous_range_check( std::declval<T const&>().begin(), std::declval<T const&>().end(), std::declval<T const&>().data(), std::declval<T const&>().data() + std::declval<T const&>().size(), std::declval<T const&>().size() ) ) is_contiguous_range_( int );
27template<class T> std::false_type is_contiguous_range_( ... );
28
29template<class T> struct is_contiguous_range: decltype( hash_detail::is_contiguous_range_<T>( 0 ) )
30{
31};
32
33} // namespace hash_detail
