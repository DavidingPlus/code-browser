189 size_type copy(charT* s, size_type n, size_type pos=0) const {
190 if (pos > size())
191 BOOST_THROW_EXCEPTION(std::out_of_range("string_view::copy" ));
192 size_type rlen = (std::min)(a: n, b: len_ - pos);
193 traits_type::copy(s, data() + pos, rlen);
194 return rlen;
195 }

201 BOOST_CXX14_CONSTEXPR basic_string_view substr(size_type pos, size_type n=npos) const {
202 if ( pos > size())

204 return basic_string_view(data() + pos, (std::min)(size() - pos, n));

207 BOOST_CXX14_CONSTEXPR int compare(basic_string_view x) const BOOST_NOEXCEPT {
208 const int cmp = traits::compare(ptr_, x.ptr_, (std::min)(len_, x.len_));
209 return cmp != 0 ? cmp : (len_ == x.len_ ? 0 : len_ < x.len_ ? -1 : 1);
210 }

265 // find
266 BOOST_CXX14_CONSTEXPR size_type find(basic_string_view s, size_type pos = 0) const BOOST_NOEXCEPT {
267 if (pos > size())
268 return npos;
269 if (s.empty())
270 return pos;
271 if (s.size() > size() - pos)
272 return npos;
273 const charT* cur = ptr_ + pos;
274 const charT* last = cend() - s.size() + 1;
275 for (; cur != last ; ++cur) {
276 cur = traits::find(cur, last - cur, s[0]);
277 if (!cur)
278 return npos;
279 if (traits::compare(cur, s.cbegin(), s.size()) == 0)
280 return cur - ptr_;
281 }
282 return npos;
283 }
284 BOOST_CXX14_CONSTEXPR size_type find(charT c, size_type pos = 0) const BOOST_NOEXCEPT {
285 if (pos > size())
286 return npos;
287 const charT* ret_ptr = traits::find(ptr_ + pos, len_ - pos, c);
288 if (ret_ptr)
289 return ret_ptr - ptr_;
290 return npos;
291 }
292 BOOST_CXX14_CONSTEXPR size_type find(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
293 { return find(basic_string_view(s, n), pos); }
294 BOOST_CXX14_CONSTEXPR size_type find(const charT* s, size_type pos = 0) const BOOST_NOEXCEPT
295 { return find(basic_string_view(s), pos); }
296
297 // rfind
298 BOOST_CXX14_CONSTEXPR size_type rfind(basic_string_view s, size_type pos = npos) const BOOST_NOEXCEPT {
299 if (len_ < s.len_)
300 return npos;
301 if (pos > len_ - s.len_)
302 pos = len_ - s.len_;
303 if (s.len_ == 0u) // an empty string is always found
304 return pos;
305 for (const charT* cur = ptr_ + pos; ; --cur) {
306 if (traits::compare(cur, s.ptr_, s.len_) == 0)
307 return cur - ptr_;
308 if (cur == ptr_)
309 return npos;
310 };
311 }
312 BOOST_CXX14_CONSTEXPR size_type rfind(charT c, size_type pos = npos) const BOOST_NOEXCEPT
313 { return rfind(basic_string_view(&c, 1), pos); }
314 BOOST_CXX14_CONSTEXPR size_type rfind(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
315 { return rfind(basic_string_view(s, n), pos); }

319 // find_first_of
320 BOOST_CXX14_CONSTEXPR size_type find_first_of(basic_string_view s, size_type pos = 0) const BOOST_NOEXCEPT {
321 if (pos >= len_ || s.len_ == 0)
322 return npos;
323 const_iterator iter = std::find_first_of
324 (this->cbegin () + pos, this->cend (), s.cbegin (), s.cend (), traits::eq);
325 return iter == this->cend () ? npos : std::distance ( this->cbegin (), iter );
326 }
327 BOOST_CXX14_CONSTEXPR size_type find_first_of(charT c, size_type pos = 0) const BOOST_NOEXCEPT
328 { return find(c, pos); }
329 BOOST_CXX14_CONSTEXPR size_type find_first_of(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
330 { return find_first_of(basic_string_view(s, n), pos); }

334 // find_last_of
335 BOOST_CXX14_CONSTEXPR size_type find_last_of(basic_string_view s, size_type pos = npos) const BOOST_NOEXCEPT {
336 if (s.len_ == 0u)
337 return npos;
338 if (pos >= len_)
339 pos = 0;
340 else
341 pos = len_ - (pos+1);
342 const_reverse_iterator iter = std::find_first_of
343 ( this->crbegin () + pos, this->crend (), s.cbegin (), s.cend (), traits::eq );
344 return iter == this->crend () ? npos : reverse_distance ( this->crbegin (), iter);
345 }
346 BOOST_CXX14_CONSTEXPR size_type find_last_of(charT c, size_type pos = npos) const BOOST_NOEXCEPT
347 { return find_last_of(basic_string_view(&c, 1), pos); }
348 BOOST_CXX14_CONSTEXPR size_type find_last_of(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
349 { return find_last_of(basic_string_view(s, n), pos); }
350 BOOST_CXX14_CONSTEXPR size_type find_last_of(const charT* s, size_type pos = npos) const BOOST_NOEXCEPT
351 { return find_last_of(basic_string_view(s), pos); }

353 // find_first_not_of
354 BOOST_CXX14_CONSTEXPR size_type find_first_not_of(basic_string_view s, size_type pos = 0) const BOOST_NOEXCEPT {
355 if (pos >= len_)
356 return npos;
357 if (s.len_ == 0)
358 return pos;
359 const_iterator iter = find_not_of ( this->cbegin () + pos, this->cend (), s );
360 return iter == this->cend () ? npos : std::distance ( this->cbegin (), iter );
361 }
362 BOOST_CXX14_CONSTEXPR size_type find_first_not_of(charT c, size_type pos = 0) const BOOST_NOEXCEPT
363 { return find_first_not_of(basic_string_view(&c, 1), pos); }
364 BOOST_CXX14_CONSTEXPR size_type find_first_not_of(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
365 { return find_first_not_of(basic_string_view(s, n), pos); }
366 BOOST_CXX14_CONSTEXPR size_type find_first_not_of(const charT* s, size_type pos = 0) const BOOST_NOEXCEPT
367 { return find_first_not_of(basic_string_view(s), pos); }
368
369 // find_last_not_of
370 BOOST_CXX14_CONSTEXPR size_type find_last_not_of(basic_string_view s, size_type pos = npos) const BOOST_NOEXCEPT {
371 if (pos >= len_)
372 pos = len_ - 1;
373 if (s.len_ == 0u)
374 return pos;
375 pos = len_ - (pos+1);
376 const_reverse_iterator iter = find_not_of ( this->crbegin () + pos, this->crend (), s );
377 return iter == this->crend () ? npos : reverse_distance ( this->crbegin (), iter );
378 }
379 BOOST_CXX14_CONSTEXPR size_type find_last_not_of(charT c, size_type pos = npos) const BOOST_NOEXCEPT
380 { return find_last_not_of(basic_string_view(&c, 1), pos); }
381 BOOST_CXX14_CONSTEXPR size_type find_last_not_of(const charT* s, size_type pos, size_type n) const BOOST_NOEXCEPT
382 { return find_last_not_of(basic_string_view(s, n), pos); }
383 BOOST_CXX14_CONSTEXPR size_type find_last_not_of(const charT* s, size_type pos = npos) const BOOST_NOEXCEPT
384 { return find_last_not_of(basic_string_view(s), pos); }
385
386 private:
387 template <typename r_iter>
388 size_type reverse_distance(r_iter first, r_iter last) const BOOST_NOEXCEPT {
389 // Portability note here: std::distance is not NOEXCEPT, but calling it with a string_view::reverse_iterator will not throw.
390 return len_ - 1 - std::distance ( first, last );
391 }
392
393 template <typename Iterator>
394 Iterator find_not_of(Iterator first, Iterator last, basic_string_view s) const BOOST_NOEXCEPT {
395 for (; first != last ; ++first)
396 if ( 0 == traits::find(s.ptr_, s.len_, *first))
397 return first;
398 return last;
399 }
400
401 const charT *ptr_;
402 std::size_t len_;
403 };
