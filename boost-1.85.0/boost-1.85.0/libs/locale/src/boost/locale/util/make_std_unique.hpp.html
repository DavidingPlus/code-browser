7#ifndef BOOST_LOCALE_MAKE_STD_UNIQUE_HPP
8#define BOOST_LOCALE_MAKE_STD_UNIQUE_HPP
9
10#include <boost/locale/config.hpp>
11#include <memory>
12#include <type_traits>
13#include <utility>
14
15namespace boost { namespace locale {
16 template<class T, class... Args>
17 std::unique_ptr<T> make_std_unique(Args&&... args)
18 {
19 static_assert(!std::is_array<T>::value, "Must not be an array");
20 return std::unique_ptr<T>(new T(std::forward<Args>(args)...));
21 }
22}} // namespace boost::locale
23
24#endif
