22 locale_data::locale_data(const std::string& locale_name)
23 {
24 if(!parse(locale_name))
25 throw std::invalid_argument("Failed to parse locale name: " + locale_name);

28 void locale_data::reset()
29 {
30 language_ = "C";
31 country_.clear();
32 encoding_ = "US-ASCII";
33 variant_.clear();
34 utf8_ = false;
35 }
36
37 std::string locale_data::to_string() const
38 {
39 std::string result = language_;
40 if(!country_.empty())
41 (result += '_') += country_;
42 if(!encoding_.empty() && !util::are_encodings_equal(l: encoding_, r: "US-ASCII"))
43 (result += '.') += encoding_;
44 if(!variant_.empty())
45 (result += '@') += variant_;
46 return result;
47 }

49 bool locale_data::parse(const std::string& locale_name)
50 {
51 reset();
52 return parse_from_lang(input: locale_name);
53 }
54
55 bool locale_data::parse_from_lang(const std::string& input)
56 {
57 const auto end = input.find_first_of(s: "-_@.");
58 std::string tmp = input.substr(pos: 0, n: end);
59 if(tmp.empty())
60 return false;
61 // lowercase ASCII
62 for(char& c : tmp) {
63 if(is_upper_ascii(c))
64 c += 'a' - 'A';
65 else if(!is_lower_ascii(c))
66 return false;
67 }
68 if(tmp != "c" && tmp != "posix") // Keep default
69 language_ = tmp;
70
71 if(end >= input.size())
72 return true;
73 else if(input[end] == '-' || input[end] == '_')
74 return parse_from_country(input: input.substr(pos: end + 1));
75 else if(input[end] == '.')
76 return parse_from_encoding(input: input.substr(pos: end + 1));
77 else {
78 BOOST_ASSERT_MSG(input[end] == '@', "Unexpected delimiter");
79 return parse_from_variant(input: input.substr(pos: end + 1));
80 }
81 }

83 bool locale_data::parse_from_country(const std::string& input)
84 {
85 if(language_ == "C")
86 return false;
87
88 const auto end = input.find_first_of(s: "@.");
89 std::string tmp = input.substr(pos: 0, n: end);
90 if(tmp.empty())
91 return false;
92
93 // Make uppercase
94 for(char& c : tmp) {
95 if(util::is_lower_ascii(c))
96 c += 'A' - 'a';
97 }
98 // If it's ALL uppercase ASCII, assume ISO 3166 country id
99 if(std::find_if_not(first: tmp.begin(), last: tmp.end(), pred: util::is_upper_ascii) != tmp.end()) {
100 // else handle special cases:
101 // - en_US_POSIX is an alias for C
102 // - M49 country code: 3 digits
103 if(language_ == "en" && tmp == "US_POSIX") {
104 language_ = "C";
105 tmp.clear();
106 } else if(tmp.size() != 3u || std::find_if_not(first: tmp.begin(), last: tmp.end(), pred: util::is_numeric_ascii) != tmp.end())
107 return false;
108 }
109
110 country_ = tmp;
111 if(end >= input.size())
112 return true;
113 else if(input[end] == '.')
114 return parse_from_encoding(input: input.substr(pos: end + 1));
115 else {
116 BOOST_ASSERT_MSG(input[end] == '@', "Unexpected delimiter");
117 return parse_from_variant(input: input.substr(pos: end + 1));
118 }
119 }

121 bool locale_data::parse_from_encoding(const std::string& input)
122 {
123 const auto end = input.find_first_of(c: '@');
124 std::string tmp = input.substr(pos: 0, n: end);
125 if(tmp.empty())
126 return false;
127 // No assumptions, but uppercase
128 encoding(new_encoding: std::move(tmp));
129 if(end >= input.size())
130 return true;
131 else {
132 BOOST_ASSERT_MSG(input[end] == '@', "Unexpected delimiter");
133 return parse_from_variant(input: input.substr(pos: end + 1));
134 }
135 }
136
137 bool locale_data::parse_from_variant(const std::string& input)
138 {
139 if(language_ == "C")
140 return false;
141 if(input.empty())
142 return false;
143 variant_ = input;
144 // No assumptions, just make it lowercase
145 for(char& c : variant_) {
146 if(util::is_upper_ascii(c))
147 c += 'a' - 'A';
148 }
149 return true;
150 }
151
152 locale_data& locale_data::encoding(std::string new_encoding, const bool uppercase)
153 {
154 if(uppercase) {
155 for(char& c : new_encoding) {
156 if(util::is_lower_ascii(c))
157 c += 'A' - 'a';
158 }
159 }
160 encoding_ = std::move(new_encoding);
161 utf8_ = util::normalize_encoding(encoding: encoding_) == "utf8";
162 return *this;
163 }
