18#if BOOST_LOCALE_USE_WIN32_API
19// Get information about the user default locale and put it into the buffer.
20// Return true on success
21template<size_t N>
22static bool get_user_default_locale_info(LCTYPE lcType, char (&buf)[N])
23{
24 return GetLocaleInfoA(LOCALE_USER_DEFAULT, lcType, buf, N) != 0;
25}
26#endif

28namespace boost { namespace locale { namespace util {
29 std::string get_system_locale(bool use_utf8_on_windows)
30 {
31 const char* lang = nullptr;
32 if(!lang || !*lang)
33 lang = getenv(name: "LC_ALL");
34 if(!lang || !*lang)
35 lang = getenv(name: "LC_CTYPE");
36 if(!lang || !*lang)
37 lang = getenv(name: "LANG");
38#if !BOOST_LOCALE_USE_WIN32_API
39 (void)use_utf8_on_windows; // not relevant for non-windows
40 if(!lang || !*lang)
41 lang = "C";
42 return lang;
43#else
44 if(lang && *lang)
45 return lang;
46
47 char buf[10]{};
48 if(!get_user_default_locale_info(LOCALE_SISO639LANGNAME, buf))
49 return "C";
50 std::string lc_name = buf;
51 if(get_user_default_locale_info(LOCALE_SISO3166CTRYNAME, buf)) {
52 lc_name += "_";
53 lc_name += buf;
54 }
55 if(use_utf8_on_windows || !get_user_default_locale_info(LOCALE_IDEFAULTANSICODEPAGE, buf))
56 lc_name += ".UTF-8";
57 else {
58 if(std::find_if_not(buf, str_end(buf), is_numeric_ascii) != str_end(buf))
59 lc_name += ".UTF-8";
60 else
61 lc_name.append(".windows-").append(buf);
62 }
63 return lc_name;
64
65#endif
66 }
