29 bool is_leap(int year)
30 {
31 if(year % 400 == 0)
32 return true;
33 if(year % 100 == 0)
34 return false;
35 if(year % 4 == 0)
36 return true;
37 return false;
38 }

40 int days_in_month(int year, int month)
41 {
42 constexpr int tbl[2][12] = {{31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31},
43 {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}};
44 return tbl[is_leap(year)][month - 1];
45 }
46
47 constexpr int days_from_0_impl(int year_m1)
48 {
49 return 365 * year_m1 + (year_m1 / 400) - (year_m1 / 100) + (year_m1 / 4);
50 }
51 constexpr int days_from_0(int year)
52 {
53 return days_from_0_impl(year_m1: year - 1);
54 }
55
56 constexpr int days_from_0_to_1970 = days_from_0(year: 1970);
57 constexpr int days_from_1970(int year)
58 {
59 return days_from_0(year) - days_from_0_to_1970;
60 }
61
62 int days_from_1jan(int year, int month, int day)
63 {
64 constexpr int days[2][12] = {{0, 31, 59, 90, 120, 151, 181, 212, 243, 273, 304, 334},
65 {0, 31, 60, 91, 121, 152, 182, 213, 244, 274, 305, 335}};
66 return days[is_leap(year)][month - 1] + day - 1;
67 }

69 std::time_t internal_timegm(const std::tm* t)
70 {
71 int year = t->tm_year + 1900;
72 int month = t->tm_mon;
73 if(month > 11) {
74 year += month / 12;
75 month %= 12;
76 } else if(month < 0) {
77 int years_diff = (-month + 11) / 12;
78 year -= years_diff;
79 month += 12 * years_diff;
80 }
81 month++;
82 int day = t->tm_mday;
83 int day_of_year = days_from_1jan(year, month, day);
84 int days_since_epoch = days_from_1970(year) + day_of_year;
85
86 std::time_t seconds_in_day = 3600 * 24;
87 std::time_t result = seconds_in_day * days_since_epoch + 3600 * t->tm_hour + 60 * t->tm_min + t->tm_sec;
88
89 return result;
90 }

111 int first_day_of_week(const char* terr)
112 {
113 constexpr const char* sat[] = {"AE", "AF", "BH", "DJ", "DZ", "EG", "ER", "ET", "IQ", "IR", "JO", "KE",
114 "KW", "LY", "MA", "OM", "QA", "SA", "SD", "SO", "SY", "TN", "YE"};
115 constexpr const char* sunday[] = {"AR", "AS", "AZ", "BW", "CA", "CN", "FO", "GE", "GL", "GU", "HK", "IL",
116 "IN", "JM", "JP", "KG", "KR", "LA", "MH", "MN", "MO", "MP", "MT", "NZ",
117 "PH", "PK", "SG", "TH", "TT", "TW", "UM", "US", "UZ", "VI", "ZW"};
118 if(strcmp(s1: terr, s2: "MV") == 0)
119 return 5; // fri
120 if(std::binary_search<const char* const*>(first: sat, last: std::end(arr: sat), val: terr, comp: comparator))
121 return 6; // sat
122 if(std::binary_search<const char* const*>(first: sunday, last: std::end(arr: sunday), val: terr, comp: comparator))
123 return 0; // sun
124 // default
125 return 1; // mon
126 }

143 /// Set specific \a value for period \a p, note not all values are settable.
144 void set_value(period::marks::period_mark m, int value) override
145 {
146 using namespace period::marks;
147 switch(m) {
148 case era: ///< Era i.e. AC, BC in Gregorian and Julian calendar, range [0,1]
149 return;
150 case year: ///< Year, it is calendar specific
151 case extended_year: ///< Extended year for Gregorian/Julian calendars, where 1 BC == 0, 2 BC == -1.
152 tm_updated_.tm_year = value - 1900;
153 break;
154 case month: tm_updated_.tm_mon = value; break;
155 case day: tm_updated_.tm_mday = value; break;
156 case hour: ///< 24 clock hour [0..23]
157 tm_updated_.tm_hour = value;
158 break;
159 case hour_12: ///< 12 clock hour [0..11]
160 tm_updated_.tm_hour = tm_updated_.tm_hour / 12 * 12 + value;
161 break;
162 case am_pm: ///< am or pm marker, [0..1]
163 tm_updated_.tm_hour = 12 * value + tm_updated_.tm_hour % 12;
164 break;
165 case minute: ///< minute [0..59]
166 tm_updated_.tm_min = value;
167 break;
168 case second: tm_updated_.tm_sec = value; break;
169 case day_of_year:
170 normalize();
171 tm_updated_.tm_mday += (value - (tm_updated_.tm_yday + 1));
172 break;
173 case day_of_week: ///< Day of week, starting from Sunday, [1..7]
174 if(value < 1) // make sure it is positive
175 value += (-value / 7) * 7 + 7;
176 // convert to local DOW
177 value = (value - 1 - first_day_of_week_ + 14) % 7 + 1;
178 BOOST_FALLTHROUGH;
179 case day_of_week_local: ///< Local day of week, for example in France Monday is 1, in US Sunday is 1,
180 ///< [1..7]
181 normalize();
182 tm_updated_.tm_mday += (value - 1) - (tm_updated_.tm_wday - first_day_of_week_ + 7) % 7;
183 break;
184 case day_of_week_in_month: ///< Original number of the day of the week in month. (1st sunday, 2nd sunday
185 ///< etc)
186 case week_of_year: ///< The week number in the year, 4 is the minimal number of days to be in month
187 case week_of_month: ///< The week number within current month
188 {
189 normalize();
190 int current_week = get_value(m, v: current);
191 int diff = 7 * (value - current_week);
192 tm_updated_.tm_mday += diff;
193 } break;
194 ///< For example Sunday in US, Monday in France
195 case period::marks::first_day_of_week: // LCOV_EXCL_LINE
196 throw std::invalid_argument("Can't change first day of week"); // LCOV_EXCL_LINE
197 case invalid: BOOST_LOCALE_CASE_INVALID(return ); // LCOV_EXCL_LINE
198 }
199 normalized_ = false;
200 }

202 void normalize() override
203 {
204 if(!normalized_) {
205 std::tm val = tm_updated_;
206 val.tm_isdst = -1;
207 val.tm_wday = -1; // indicator of error
208 std::time_t point = -1;
209 if(is_local_) {
210 point = std::mktime(tp: &val);
211 if(point == static_cast<std::time_t>(-1)) {
212#ifndef BOOST_WINDOWS
213 // windows does not handle negative time_t, under other plaforms
214 // it may be actually valid value in 1969-12-31 23:59:59
215 // so we check that a field was updated - does not happen in case of error
216 if(val.tm_wday == -1)
217#endif
218 throw date_time_error("boost::locale::gregorian_calendar: invalid time"); // LCOV_EXCL_LINE
219 }
220 } else {
221 point = internal_timegm(t: &val);
222#ifdef BOOST_WINDOWS
223 // Windows uses TLS, thread safe
224 std::tm* revert_point = nullptr;
225 if(point < 0 || (revert_point = gmtime(&point)) == nullptr)
226 throw date_time_error("boost::locale::gregorian_calendar time is out of range");
227 val = *revert_point;
228#else
229 if(!gmtime_r(timer: &point, tp: &val))
230 throw date_time_error("boost::locale::gregorian_calendar invalid time");
231#endif
232 }
233
234 time_ = point - tzoff_;
235 tm_ = val;
236 tm_updated_ = val;
237 normalized_ = true;
238 }

241 int get_week_number(int day, int wday) const
242 {
243 /// This is the number of days that are considered within
244 /// period such that the week belongs there
245 constexpr int days_in_full_week = 4;
246
247 // Always use local week start
248 int current_dow = (wday - first_day_of_week_ + 7) % 7;
249 // Calculate local week day of Jan 1st.
250 int first_week_day = (current_dow + 700 - day) % 7;
251 // adding something big dividable by 7
252
253 int start_of_period_in_weeks;
254 if(first_week_day < days_in_full_week)
255 start_of_period_in_weeks = -first_week_day;
256 else
257 start_of_period_in_weeks = 7 - first_week_day;
258 int week_number_in_days = day - start_of_period_in_weeks;
259 if(week_number_in_days < 0)
260 return -1;
261 return week_number_in_days / 7 + 1;
262 }

264 /// Get specific value for period \a p according to a value_type \a v
265 int get_value(period::marks::period_mark m, value_type v) const override
266 {
267 using namespace period::marks;
268 switch(m) {
269 case era: return 1;
270 case year:
271 case extended_year:
272 switch(v) {
273 case absolute_minimum:
274 case greatest_minimum:
275 case actual_minimum:
276#ifdef BOOST_WINDOWS
277 return 1970; // Unix epoch windows can't handle negative time_t
278#else
279 if(sizeof(std::time_t) == 4)
280 return 1901; // minimal year with 32 bit time_t
281 else
282 return 1;
283#endif
284 case absolute_maximum:
285 case least_maximum:
286 case actual_maximum:
287 BOOST_LOCALE_START_CONST_CONDITION
288 if(sizeof(std::time_t) == 4)
289 return 2038; // Y2K38 - maximal with 32 bit time_t
290 else
291 return std::numeric_limits<int>::max()
292 / 365; // Reasonably large but we can still get the days without overflowing
293 BOOST_LOCALE_END_CONST_CONDITION
294 case current: return tm_.tm_year + 1900;
295 };
296 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
297 case month:
298 switch(v) {
299 case absolute_minimum:
300 case greatest_minimum:
301 case actual_minimum: return 0;
302 case absolute_maximum:
303 case least_maximum:
304 case actual_maximum: return 11;
305 case current: return tm_.tm_mon;
306 };
307 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
308 case day:
309 switch(v) {
310 case absolute_minimum:
311 case greatest_minimum:
312 case actual_minimum: return 1;
313 case absolute_maximum: return 31;
314 case least_maximum: return 28;
315 case actual_maximum: return days_in_month(year: tm_.tm_year + 1900, month: tm_.tm_mon + 1);
316 case current: return tm_.tm_mday;
317 };
318 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
319 case day_of_year: ///< The number of day in year, starting from 1
320 switch(v) {
321 case absolute_minimum:
322 case greatest_minimum:
323 case actual_minimum: return 1;
324 case absolute_maximum: return 366;
325 case least_maximum: return 365;
326 case actual_maximum: return is_leap(year: tm_.tm_year + 1900) ? 366 : 365;
327 case current: return tm_.tm_yday + 1;
328 }
329 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
330 case day_of_week: ///< Day of week, starting from Sunday, [1..7]
331 switch(v) {
332 case absolute_minimum:
333 case greatest_minimum:
334 case actual_minimum: return 1;
335 case absolute_maximum:
336 case least_maximum:
337 case actual_maximum: return 7;
338 case current: return tm_.tm_wday + 1;
339 }
340 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
341 case day_of_week_local: ///< Local day of week, for example in France Monday is 1, in US Sunday is 1,
342 ///< [1..7]
343 switch(v) {
344 case absolute_minimum:
345 case greatest_minimum:
346 case actual_minimum: return 1;
347 case absolute_maximum:
348 case least_maximum:
349 case actual_maximum: return 7;
350 case current: return (tm_.tm_wday - first_day_of_week_ + 7) % 7 + 1;
351 }
352 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
353 case hour: ///< 24 clock hour [0..23]
354 switch(v) {
355 case absolute_minimum:
356 case greatest_minimum:
357 case actual_minimum: return 0;
358 case absolute_maximum:
359 case least_maximum:
360 case actual_maximum: return 23;
361 case current: return tm_.tm_hour;
362 }
363 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
364 case hour_12: ///< 12 clock hour [0..11]
365 switch(v) {
366 case absolute_minimum:
367 case greatest_minimum:
368 case actual_minimum: return 0;
369 case absolute_maximum:
370 case least_maximum:
371 case actual_maximum: return 11;
372 case current: return tm_.tm_hour % 12;
373 }
374 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
375 case am_pm: ///< am or pm marker, [0..1]
376 switch(v) {
377 case absolute_minimum:
378 case greatest_minimum:
379 case actual_minimum: return 0;
380 case absolute_maximum:
381 case least_maximum:
382 case actual_maximum: return 1;
383 case current: return tm_.tm_hour >= 12 ? 1 : 0;
384 }
385 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
386 case minute: ///< minute [0..59]
387 switch(v) {
388 case absolute_minimum:
389 case greatest_minimum:
390 case actual_minimum: return 0;
391 case absolute_maximum:
392 case least_maximum:
393 case actual_maximum: return 59;
394 case current: return tm_.tm_min;
395 }
396 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
397 case second: ///< second [0..59]
398 switch(v) {
399 case absolute_minimum:
400 case greatest_minimum:
401 case actual_minimum: return 0;
402 case absolute_maximum:
403 case least_maximum:
404 case actual_maximum: return 59;
405 case current: return tm_.tm_sec;
406 }
407 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
408 case period::marks::first_day_of_week: ///< For example Sunday in US, Monday in France
409 return first_day_of_week_ + 1;
410
411 case week_of_year: ///< The week number in the year
412 switch(v) {
413 case absolute_minimum:
414 case greatest_minimum:
415 case actual_minimum: return 1;
416 case absolute_maximum: return 53;
417 case least_maximum: return 52;
418 case actual_maximum: {
419 int year = tm_.tm_year + 1900;
420 int end_of_year_days = (is_leap(year) ? 366 : 365) - 1;
421 int dow_of_end_of_year = (end_of_year_days - tm_.tm_yday + tm_.tm_wday) % 7;
422 return get_week_number(day: end_of_year_days, wday: dow_of_end_of_year);
423 }
424 case current: {
425 const int val = get_week_number(day: tm_.tm_yday, wday: tm_.tm_wday);
426 return (val < 0) ? 53 : val;
427 }
428 } // LCOV_EXCL_LINE
429 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
430 case week_of_month: ///< The week number within current month
431 switch(v) {
432 case absolute_minimum:
433 case greatest_minimum:
434 case actual_minimum: return 1;
435 case absolute_maximum: return 5;
436 case least_maximum: return 4;
437 case actual_maximum: {
438 int end_of_month_days = days_in_month(year: tm_.tm_year + 1900, month: tm_.tm_mon + 1);
439 int dow_of_end_of_month = (end_of_month_days - tm_.tm_mday + tm_.tm_wday) % 7;
440 return get_week_number(day: end_of_month_days, wday: dow_of_end_of_month);
441 }
442 case current: {
443 const int val = get_week_number(day: tm_.tm_mday, wday: tm_.tm_wday);
444 return (val < 0) ? 5 : val;
445 }
446 } // LCOV_EXCL_LINE
447 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
448 case day_of_week_in_month: ///< Original number of the day of the week in month.
449 switch(v) {
450 case absolute_minimum:
451 case greatest_minimum:
452 case actual_minimum: return 1;
453 case absolute_maximum: return 5;
454 case least_maximum: return 4;
455 case actual_maximum:
456 if(tm_.tm_mon == 1 && !is_leap(year: tm_.tm_year + 1900)) {
457 // only february in non leap year is 28 days, the rest
458 // has more then 4 weeks
459 return 4;
460 }
461 return 5;
462 case current: return (tm_.tm_mday - 1) / 7 + 1;
463 }
464 throw std::invalid_argument("Invalid abstract_calendar::value_type"); // LCOV_EXCL_LINE
465 case invalid: BOOST_LOCALE_CASE_INVALID(break); // LCOV_EXCL_LINE
466 }
467 throw std::invalid_argument("Invalid period_mark"); // LCOV_EXCL_LINE
468 }

504 /// Adjust period's \a p value by \a difference items using a update_type \a u.
505 /// Note: not all values are adjustable
506 void adjust_value(period::marks::period_mark m, update_type u, int difference) override
507 {
508 if(u == move) {
509 using namespace period::marks;
510 switch(m) {
511 case year: ///< Year, it is calendar specific
512 case extended_year: ///< Extended year for Gregorian/Julian calendars, where 1 BC == 0, 2 BC == -1
513 tm_updated_.tm_year += difference;
514 break;
515 case month: tm_updated_.tm_mon += difference; break;
516 case day:
517 case day_of_year:
518 case day_of_week: ///< starting from Sunday, [1..7]
519 case day_of_week_local: ///< Local DoW, e.g. in France Monday=1, in US Sunday=1, [1..7]
520 tm_updated_.tm_mday += difference;
521 break;
522 case hour: ///< 24 clock hour [0..23]
523 case hour_12: ///< 12 clock hour [0..11]
524 tm_updated_.tm_hour += difference;
525 break;
526 case am_pm: ///< am or pm marker, [0..1]
527 tm_updated_.tm_hour += 12 * difference;
528 break;
529 case minute: ///< minute [0..59]
530 tm_updated_.tm_min += difference;
531 break;
532 case second: tm_updated_.tm_sec += difference; break;
533 case week_of_year: ///< The week number in the year
534 case week_of_month: ///< The week number within current month
535 case day_of_week_in_month: ///< Original number of the day of the week in month.
536 tm_updated_.tm_mday += difference * 7;
537 break;
538 case era: throw std::invalid_argument("era not adjustable"); // LCOV_EXCL_LINE
539 case period::marks::first_day_of_week: // LCOV_EXCL_LINE
540 throw std::invalid_argument("Can't change first day of week"); // LCOV_EXCL_LINE
541 case invalid: BOOST_LOCALE_CASE_INVALID(return ); // LCOV_EXCL_LINE
542 }
543 normalized_ = false;
544 normalize();
545 } else {
546 BOOST_ASSERT(u == roll);
547 const int cur_min = get_value(m, v: actual_minimum);
548 const int cur_max = get_value(m, v: actual_maximum);
549 BOOST_ASSERT(cur_max >= cur_min);
550 const int range = cur_max - cur_min + 1;
551 int value = get_value(m, v: current) - cur_min;
552 BOOST_ASSERT(value >= 0 && value < range);
553 BOOST_ASSERT_MSG(difference <= std::numeric_limits<int>::max(), "Input is to large");
554 value = (value + difference) % range;
555 // If the sum above was negative the result of the modulo operation "can" be negative too.
556 if(value < 0)
557 value += range;
558 BOOST_ASSERT(value >= 0 && value < range);
559 set_value(m, value: value + cur_min);
560 normalize();
561 }
562 }

564 int get_diff(period::marks::period_mark m, int diff, const gregorian_calendar* other) const
565 {
566 if(diff == 0)
567 return 0;
568 hold_ptr<gregorian_calendar> self(clone());
569 self->adjust_value(m, u: move, difference: diff);
570 if(diff > 0) {
571 if(self->time_ > other->time_)
572 return diff - 1;
573 else
574 return diff;
575 } else {
576 if(self->time_ < other->time_)
577 return diff + 1;
578 else
579 return diff;
580 }
581 }

583 /// Calculate the difference between this calendar and \a other in \a p units
584 int difference(const abstract_calendar& other_cal, period::marks::period_mark m) const override
585 {
586 hold_ptr<gregorian_calendar> keeper;
587 const gregorian_calendar* other = dynamic_cast<const gregorian_calendar*>(&other_cal);
588 if(!other) {
589 keeper.reset(p: clone());
590 keeper->set_time(other_cal.get_time());
591 other = keeper.get();
592 }
593
594 int factor = 1; // for weeks vs days handling
595
596 using namespace period::marks;
597 switch(m) {
598 case era: return 0;
599 case year:
600 case extended_year: {
601 int diff = other->tm_.tm_year - tm_.tm_year;
602 return get_diff(m: period::marks::year, diff, other);
603 }
604 case month: {
605 int diff = 12 * (other->tm_.tm_year - tm_.tm_year) + other->tm_.tm_mon - tm_.tm_mon;
606 return get_diff(m: period::marks::month, diff, other);
607 }
608 case day_of_week_in_month:
609 case week_of_month:
610 case week_of_year: factor = 7; BOOST_FALLTHROUGH;
611 case day:
612 case day_of_year:
613 case day_of_week:
614 case day_of_week_local: {
615 int diff = other->tm_.tm_yday - tm_.tm_yday;
616 if(other->tm_.tm_year != tm_.tm_year)
617 diff += days_from_0(year: other->tm_.tm_year + 1900) - days_from_0(year: tm_.tm_year + 1900);
618 return get_diff(m: period::marks::day, diff, other) / factor;
619 }
620 case am_pm: return static_cast<int>((other->time_ - time_) / (3600 * 12));
621 case hour:
622 case hour_12: return static_cast<int>((other->time_ - time_) / 3600);
623 case minute: return static_cast<int>((other->time_ - time_) / 60);
624 case second: return static_cast<int>(other->time_ - time_);
625 case invalid: // LCOV_EXCL_LINE
626 BOOST_LOCALE_CASE_INVALID(return 0); // LCOV_EXCL_LINE
627 // Not adjustable
628 case period::marks::first_day_of_week: return 0; // LCOV_EXCL_LINE
629 }
630 throw std::invalid_argument("Invalid period_mark"); // LCOV_EXCL_LINE
631 }

635 {
636 if(tz.empty()) {
637 is_local_ = true;
638 tzoff_ = 0;
639 } else {
640 is_local_ = false;
641 tzoff_ = parse_tz(tz);
642 }
643 from_time(point: time_);
644 time_zone_name_ = tz;
645 }

652 {
653 const gregorian_calendar* gcal = dynamic_cast<const gregorian_calendar*>(other);
654 if(!gcal)
655 return false;
656 return gcal->tzoff_ == tzoff_ && gcal->is_local_ == is_local_
657 && gcal->first_day_of_week_ == first_day_of_week_;
658 }

660 private:
661 void from_time(std::time_t point)
662 {
663 std::time_t real_point = point + tzoff_;
664 std::tm* t;
665#ifdef BOOST_WINDOWS
666 // Windows uses TLS, thread safe
667 t = is_local_ ? localtime(&real_point) : gmtime(&real_point);
668#else
669 std::tm tmp_tm;
670 t = is_local_ ? localtime_r(timer: &real_point, tp: &tmp_tm) : gmtime_r(timer: &real_point, tp: &tmp_tm);
671#endif
672 if(!t)
673 throw date_time_error("boost::locale::gregorian_calendar: invalid time point");
674 tm_ = *t;
675 tm_updated_ = *t;
676 normalized_ = true;
677 time_ = point;
678 }
679 int first_day_of_week_;

681 std::tm tm_;
682 std::tm tm_updated_;
683 bool normalized_;

685 int tzoff_;
