29 class utf8_converter : public base_converter {
30 public:
31 int max_len() const override { return 4; }
32
33 utf8_converter* clone() const override { return new utf8_converter(); }
34
35 bool is_thread_safe() const override { return true; }
36
37 utf::code_point to_unicode(const char*& begin, const char* end) override
38 {
39 const char* p = begin;
40
41 const utf::code_point c = utf::utf_traits<char>::decode(p, e: end);
42
43 if(c == utf::illegal)
44 return illegal;
45
46 if(c == utf::incomplete)
47 return incomplete;
48
49 begin = p;
50 return c;
51 }
52
53 utf::len_or_error from_unicode(utf::code_point u, char* begin, const char* end) override
54 {
55 if(!utf::is_valid_codepoint(v: u))
56 return illegal;
57 int width = utf::utf_traits<char>::width(value: u);
58 std::ptrdiff_t d = end - begin;
59 if(d < width)
60 return incomplete;
61 utf::utf_traits<char>::encode(value: u, out: begin);
62 return width;
63 }
64 }; // utf8_converter

66 class simple_converter_impl {
67 public:
68 static constexpr int hash_table_size = 1024;
69
70 simple_converter_impl(const std::string& encoding)
71 {
72 for(unsigned i = 0; i < 128; i++)
73 to_unicode_tbl_[i] = i;
74 const conv::utf_encoder<wchar_t> to_utf(encoding, conv::skip);
75 for(unsigned i = 128; i < 256; i++) {
76 char buf[2] = {util::to_char(c: i), 0};
77 uint32_t uchar = utf::illegal;
78 try {
79 std::wstring const tmp = to_utf.convert(begin: buf, end: buf + 1);
80 if(tmp.size() == 1)
81 uchar = tmp[0];
82 else
83 uchar = utf::illegal;
84 } catch(const conv::conversion_error&) { // LCOV_EXCL_LINE
85 uchar = utf::illegal; // LCOV_EXCL_LINE
86 }
87 to_unicode_tbl_[i] = uchar;
88 }
89 for(int i = 0; i < hash_table_size; i++)
90 from_unicode_tbl_[i] = 0;
91 for(unsigned i = 1; i < 256; i++) {
92 if(to_unicode_tbl_[i] != utf::illegal) {
93 unsigned pos = to_unicode_tbl_[i] % hash_table_size;
94 while(from_unicode_tbl_[pos] != 0)
95 pos = (pos + 1) % hash_table_size;
96 from_unicode_tbl_[pos] = i;
97 }
98 }
99 }
100
101 utf::code_point to_unicode(const char*& begin, const char* end) const
102 {
103 if(begin == end)
104 return utf::incomplete;
105 unsigned char c = *begin++;
106 return to_unicode_tbl_[c];
107 }
108 utf::len_or_error from_unicode(utf::code_point u, char* begin, const char* end) const
109 {
110 if(begin == end)
111 return utf::incomplete;
112 if(u == 0) {
113 *begin = 0;
114 return 1;
115 }
116 unsigned pos = u % hash_table_size;
117 unsigned char c;
118 while((c = from_unicode_tbl_[pos]) != 0 && to_unicode_tbl_[c] != u)
119 pos = (pos + 1) % hash_table_size;
120 if(c == 0)
121 return utf::illegal;
122 *begin = to_char(c);
123 return 1;
124 }
125
126 private:
127 utf::code_point to_unicode_tbl_[256];
128 unsigned char from_unicode_tbl_[hash_table_size];
129 };

223 template<typename CharType, bool ThreadSafe>
224 class code_converter : public generic_codecvt<CharType, code_converter<CharType, ThreadSafe>> {
225 public:
226 typedef std::unique_ptr<base_converter> base_converter_ptr;
227 typedef base_converter_ptr state_type;
228
229 code_converter(base_converter_ptr cvt, size_t refs = 0) :
230 generic_codecvt<CharType, code_converter<CharType, ThreadSafe>>(refs), cvt_(std::move(cvt))
231 {}
232
233 int max_encoding_length() const { return cvt_->max_len(); }
234
235 base_converter_ptr initial_state(generic_codecvt_base::initial_convertion_state /* unused */) const
236 {
237 base_converter_ptr r;
238 if(!ThreadSafe)
239 r.reset(p: cvt_->clone());
240 return r;
241 }
242
243 utf::code_point to_unicode(base_converter_ptr& ptr, const char*& begin, const char* end) const
244 {
245 if(ThreadSafe)
246 return cvt_->to_unicode(begin, end);
247 else
248 return ptr->to_unicode(begin, end);
249 }
250
251 utf::len_or_error from_unicode(base_converter_ptr& ptr, utf::code_point u, char* begin, const char* end) const
252 {
253 if(ThreadSafe)
254 return cvt_->from_unicode(u, begin, end);
255 else
256 return ptr->from_unicode(u, begin, end);
257 }
258
259 private:
260 base_converter_ptr cvt_;
261 };

263 template<typename CharType>
264 static std::locale do_create_codecvt(const std::locale& in, std::unique_ptr<base_converter> cvt)
265 {
266 return cvt->is_thread_safe() ? std::locale(in, new code_converter<CharType, true>(std::move(cvt))) :
267 std::locale(in, new code_converter<CharType, false>(std::move(cvt)));
268 }

270 std::locale create_codecvt(const std::locale& in, std::unique_ptr<base_converter> cvt, char_facet_t type)
271 {
272 if(!cvt)
273 cvt.reset(p: new base_converter());
274 switch(type) {
275 case char_facet_t::nochar: break;
276 case char_facet_t::char_f: return do_create_codecvt<char>(in, cvt: std::move(cvt));
277 case char_facet_t::wchar_f: return do_create_codecvt<wchar_t>(in, cvt: std::move(cvt));
278#ifdef __cpp_char8_t
279 case char_facet_t::char8_f: break; // No std facet
280#endif
281#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
282 case char_facet_t::char16_f: return do_create_codecvt<char16_t>(in, std::move(cvt));
283#endif
284#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
285 case char_facet_t::char32_f: return do_create_codecvt<char32_t>(in, std::move(cvt));
286#endif
287 }
288 return in;
289 }

291 /// Install utf8 codecvt to UTF-16 or UTF-32 into locale \a in and return
292 /// new locale that is based on \a in and uses new facet.
293 std::locale create_utf8_codecvt(const std::locale& in, char_facet_t type)
294 {
295 switch(type) {
296 case char_facet_t::nochar: break;
297 case char_facet_t::char_f: return std::locale(in, new utf8_codecvt<char>());
298 case char_facet_t::wchar_f: return std::locale(in, new utf8_codecvt<wchar_t>());
299#ifdef __cpp_char8_t
300 case char_facet_t::char8_f: break; // No std facet
301#endif
302#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
303 case char_facet_t::char16_f: return std::locale(in, new utf8_codecvt<char16_t>());
304#endif
305#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
306 case char_facet_t::char32_f: return std::locale(in, new utf8_codecvt<char32_t>());
307#endif
308 }
309 return in;
310 }

312 std::locale create_simple_codecvt(const std::locale& in, const std::string& encoding, char_facet_t type)
313 {
314 if(!is_simple_encoding(encoding))
315 throw boost::locale::conv::invalid_charset_error("Invalid simple encoding " + encoding);
316
317 switch(type) {
318 case char_facet_t::nochar: break;
319 case char_facet_t::char_f: return std::locale(in, new simple_codecvt<char>(encoding));
320 case char_facet_t::wchar_f: return std::locale(in, new simple_codecvt<wchar_t>(encoding));
321#ifdef __cpp_char8_t
322 case char_facet_t::char8_f: break; // No std facet
323#endif
324#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
325 case char_facet_t::char16_f: return std::locale(in, new simple_codecvt<char16_t>(encoding));
326#endif
327#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
328 case char_facet_t::char32_f: return std::locale(in, new simple_codecvt<char32_t>(encoding));
329#endif
330 }
331 return in;
332 }
