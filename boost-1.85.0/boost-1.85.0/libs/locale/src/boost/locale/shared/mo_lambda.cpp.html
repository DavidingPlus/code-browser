24 template<class TExp, typename... Ts>

45 template<class Functor, bool returnZeroOnZero2ndArg = false>
46 struct binary final : expr, Functor {
47 binary(expr_ptr p1, expr_ptr p2) : op1(std::move(p1)), op2(std::move(p2)) {}
48 value_type operator()(value_type n) const override
49 {
50 const auto v1 = op1(n);
51 const auto v2 = op2(n);
52 BOOST_LOCALE_START_CONST_CONDITION
53 if(returnZeroOnZero2ndArg && v2 == 0)
54 return 0;
55 BOOST_LOCALE_END_CONST_CONDITION
56 return Functor::operator()(v1, v2);
57 }
58
59 protected:
60 sub_expr_type op1, op2;
61 };

81 using token_t = int;
82 enum : token_t { END = 0, GTE = 256, LTE, EQ, NEQ, AND, OR, NUM, VARIABLE };
83
84 expr_ptr bin_factory(const token_t value, expr_ptr left, expr_ptr right)
85 {
86#define BINOP_CASE(match, cls) \
87 case match: return make_expr<cls>(std::move(left), std::move(right))
88 // Special cases: Avoid division by zero
89 using divides = binary<std::divides<expr::value_type>, true>;
90 using modulus = binary<std::modulus<expr::value_type>, true>;
91 switch(value) {
92 BINOP_CASE('/', divides);
93 BINOP_CASE('*', binary<std::multiplies<expr::value_type>>);
94 BINOP_CASE('%', modulus);
95 BINOP_CASE('+', binary<std::plus<expr::value_type>>);
96 BINOP_CASE('-', binary<std::minus<expr::value_type>>);
97 BINOP_CASE('>', binary<std::greater<expr::value_type>>);
98 BINOP_CASE('<', binary<std::less<expr::value_type>>);
99 BINOP_CASE(GTE, binary<std::greater_equal<expr::value_type>>);
100 BINOP_CASE(LTE, binary<std::less_equal<expr::value_type>>);
101 BINOP_CASE(EQ, binary<std::equal_to<expr::value_type>>);
102 BINOP_CASE(NEQ, binary<std::not_equal_to<expr::value_type>>);
103 BINOP_CASE(AND, binary<std::logical_and<expr::value_type>>);
104 BINOP_CASE(OR, binary<std::logical_or<expr::value_type>>);
105 default: throw std::logic_error("Unexpected binary operator"); // LCOV_EXCL_LINE
106 }
107#undef BINOP_CASE
108 }

110 template<size_t size>
111 bool is_in(const token_t token, const token_t (&tokens)[size])
112 {
113 for(const auto el : tokens) {
114 if(token == el)
115 return true;
116 }
117 return false;
118 }
119
120 class tokenizer {
121 public:
122 tokenizer(const char* s) : text_(s), next_tocken_(0), numeric_value_(0) { step(); }
123 token_t get(long long* val = nullptr)
124 {
125 const token_t res = next(val);
126 step();
127 return res;
128 }
129 token_t next(long long* val = nullptr) const
130 {
131 if(val && next_tocken_ == NUM)
132 *val = numeric_value_;
133 return next_tocken_;
134 }
135
136 private:
137 const char* text_;
138 token_t next_tocken_;
139 long long numeric_value_;
140
141 static constexpr bool is_blank(char c) { return c == ' ' || c == '\r' || c == '\n' || c == '\t'; }
142 static constexpr bool is_digit(char c) { return '0' <= c && c <= '9'; }
143 template<size_t size>
144 static bool is(const char* s, const char (&search)[size])
145 {
146 return strncmp(s, search, size - 1) == 0;
147 }
148 void step()
149 {
150 while(is_blank(c: *text_))
151 text_++;
152 const char* text = text_;
153 if(is(s: text, search: "&&")) {
154 text_ += 2;
155 next_tocken_ = AND;
156 } else if(is(s: text, search: "||")) {
157 text_ += 2;
158 next_tocken_ = OR;
159 } else if(is(s: text, search: "<=")) {
160 text_ += 2;
161 next_tocken_ = LTE;
162 } else if(is(s: text, search: ">=")) {
163 text_ += 2;
164 next_tocken_ = GTE;
165 } else if(is(s: text, search: "==")) {
166 text_ += 2;
167 next_tocken_ = EQ;
168 } else if(is(s: text, search: "!=")) {
169 text_ += 2;
170 next_tocken_ = NEQ;
171 } else if(*text == 'n') {
172 text_++;
173 next_tocken_ = VARIABLE;
174 } else if(is_digit(c: *text)) {
175 char* tmp_ptr;
176 // strtoll not always available -> parse as unsigned long
177 const auto value = std::strtoul(nptr: text, endptr: &tmp_ptr, base: 10);
178 // Saturate in case long=long long
179 numeric_value_ = std::min<unsigned long long>(a: std::numeric_limits<long long>::max(), b: value);
180 text_ = tmp_ptr;
181 next_tocken_ = NUM;
182 } else if(*text == '\0')
183 next_tocken_ = END;
184 else {
185 next_tocken_ = *text;
186 text_++;
187 }
188 }
189 };

191 constexpr token_t level6[] = {'*', '/', '%'};
192 constexpr token_t level5[] = {'+', '-'};
193 constexpr token_t level4[] = {'<', '>', GTE, LTE};
194 constexpr token_t level3[] = {EQ, NEQ};
195 constexpr token_t level2[] = {AND};
196 constexpr token_t level1[] = {OR};
197
198 class parser {
199 public:
200 parser(const char* str) : t(str) {}
201
202 expr_ptr compile()
203 {
204 expr_ptr res = cond_expr();
205 if(res && t.next() != END)
206 return expr_ptr();
207 return res;
208 }
209
210 private:
211 expr_ptr value_expr()
212 {
213 expr_ptr op;
214 if(t.next() == '(') {
215 t.get();
216 if(!(op = cond_expr()))
217 return expr_ptr();
218 if(t.get() != ')')
219 return expr_ptr();
220 return op;
221 } else if(t.next() == NUM) {
222 expr::value_type value;
223 t.get(val: &value);
224 return make_expr<number>(ts&: value);
225 } else if(t.next() == VARIABLE) {
226 t.get();
227 return make_expr<identity>();
228 }
229 return expr_ptr();
230 }
231
232 expr_ptr unary_expr()
233 {
234 constexpr token_t level_unary[] = {'!', '-'};
235 if(is_in(token: t.next(), tokens: level_unary)) {
236 const token_t op = t.get();
237 expr_ptr op1 = unary_expr();
238 if(!op1)
239 return expr_ptr();
240 if(BOOST_LIKELY(op == '!'))
241 return make_expr<unary<std::logical_not<expr::value_type>>>(ts: std::move(op1));
242 else {
243 BOOST_ASSERT(op == '-');
244 return make_expr<unary<std::negate<expr::value_type>>>(ts: std::move(op1));
245 }
246 } else
247 return value_expr();
248 }
249
250#define BINARY_EXPR(lvl, nextLvl, list) \
251 expr_ptr lvl() \
252 { \
253 expr_ptr op1 = nextLvl(); \
254 if(!op1) \
255 return expr_ptr(); \
256 while(is_in(t.next(), list)) { \
257 const token_t o = t.get(); \
258 expr_ptr op2 = nextLvl(); \
259 if(!op2) \
260 return expr_ptr(); \
261 op1 = bin_factory(o, std::move(op1), std::move(op2)); \
262 } \
263 return op1; \
264 }
265
266 BINARY_EXPR(l6, unary_expr, level6);
267 BINARY_EXPR(l5, l6, level5);
268 BINARY_EXPR(l4, l5, level4);
269 BINARY_EXPR(l3, l4, level3);
270 BINARY_EXPR(l2, l3, level2);
271 BINARY_EXPR(l1, l2, level1);
272#undef BINARY_EXPR
273
274 expr_ptr cond_expr()
275 {
276 expr_ptr cond;
277 if(!(cond = l1()))
278 return expr_ptr();
279 if(t.next() != '?')
280 return cond;
281 t.get();
282 expr_ptr case1, case2;
283 if(!(case1 = cond_expr()))
284 return expr_ptr();
285 if(t.get() != ':')
286 return expr_ptr();
287 if(!(case2 = cond_expr()))
288 return expr_ptr();
289 return make_expr<conditional>(ts: std::move(cond), ts: std::move(case1), ts: std::move(case2));
290 }
291
292 tokenizer t;
293 };
