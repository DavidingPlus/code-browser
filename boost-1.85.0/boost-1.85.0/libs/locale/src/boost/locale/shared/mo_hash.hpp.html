7#include <cstdint>

9namespace boost { namespace locale { namespace gnu_gettext {

11 struct pj_winberger_hash {
12 typedef uint32_t state_type;
13
14 static constexpr state_type initial_state = 0;
15
16 static state_type update_state(state_type value, char c)
17 {
18 value = (value << 4) + static_cast<unsigned char>(c);
19 uint32_t high = (value & 0xF0000000U);
20 if(high != 0)
21 value = (value ^ (high >> 24)) ^ high;
22 return value;
23 }
24 static state_type update_state(state_type value, const char* ptr)
25 {
26 while(*ptr)
27 value = update_state(value, c: *ptr++);
28 return value;
29 }
30 static state_type update_state(state_type value, const char* begin, const char* end)
31 {
32 while(begin != end)
33 value = update_state(value, c: *begin++);
34 return value;
35 }
36 };
37
38 inline pj_winberger_hash::state_type pj_winberger_hash_function(const char* ptr)
39 {
40 pj_winberger_hash::state_type state = pj_winberger_hash::initial_state;
41 state = pj_winberger_hash::update_state(value: state, ptr);
42 return state;
43 }
44
45 inline pj_winberger_hash::state_type pj_winberger_hash_function(const char* begin, const char* end)
46 {
47 pj_winberger_hash::state_type state = pj_winberger_hash::initial_state;
48 state = pj_winberger_hash::update_state(value: state, begin, end);
49 return state;
50 }
