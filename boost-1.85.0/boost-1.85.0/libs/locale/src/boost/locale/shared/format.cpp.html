17 struct format_parser::data {
18 unsigned position;
19 std::streamsize precision;
20 std::ios_base::fmtflags flags;
21 ios_info info;
22 std::locale saved_locale;
23 bool restore_locale;
24 void* cookie;
25 void (*imbuer)(void*, const std::locale&);
26 };
27
28 format_parser::format_parser(std::ios_base& ios, void* cookie, void (*imbuer)(void*, const std::locale&)) :
29 ios_(ios), d(new data)
30 {
31 d->position = std::numeric_limits<unsigned>::max();
32 d->precision = ios.precision();
33 d->flags = ios.flags();
34 d->info = ios_info::get(ios);
35 d->saved_locale = ios.getloc();
36 d->restore_locale = false;
37 d->cookie = cookie;
38 d->imbuer = imbuer;
39 }
40
41 void format_parser::imbue(const std::locale& l)
42 {
43 d->imbuer(d->cookie, l);
44 }
45
46 format_parser::~format_parser() = default;
47
48 void format_parser::restore()
49 {
50 ios_info::get(ios&: ios_) = d->info;
51 ios_.width(wide: 0);
52 ios_.flags(fmtfl: d->flags);
53 if(d->restore_locale)
54 imbue(l: d->saved_locale);
55 }
56
57 unsigned format_parser::get_position()
58 {
59 return d->position;
60 }

62 void format_parser::set_one_flag(const std::string& key, const std::string& value)
63 {
64 if(key.empty())
65 return;
66 int position;
67 if(util::try_to_int(s: key, res&: position) && position > 0) {
68 static_assert(sizeof(unsigned) <= sizeof(decltype(d->position)), "Possible lossy conversion");
69 d->position = static_cast<unsigned>(position - 1);
70 } else if(key == "num" || key == "number") {
71 as::number(ios&: ios_);
72
73 if(value == "hex")
74 ios_.setf(fmtfl: std::ios_base::hex, mask: std::ios_base::basefield);
75 else if(value == "oct")
76 ios_.setf(fmtfl: std::ios_base::oct, mask: std::ios_base::basefield);
77 else if(value == "sci" || value == "scientific")
78 ios_.setf(fmtfl: std::ios_base::scientific, mask: std::ios_base::floatfield);
79 else if(value == "fix" || value == "fixed")
80 ios_.setf(fmtfl: std::ios_base::fixed, mask: std::ios_base::floatfield);
81 } else if(key == "cur" || key == "currency") {
82 as::currency(ios&: ios_);
83 if(value == "iso")
84 as::currency_iso(ios&: ios_);
85 else if(value == "nat" || value == "national")
86 as::currency_national(ios&: ios_);
87 } else if(key == "per" || key == "percent")
88 as::percent(ios&: ios_);
89 else if(key == "date") {
90 as::date(ios&: ios_);
91 if(value == "s" || value == "short")
92 as::date_short(ios&: ios_);
93 else if(value == "m" || value == "medium")
94 as::date_medium(ios&: ios_);
95 else if(value == "l" || value == "long")
96 as::date_long(ios&: ios_);
97 else if(value == "f" || value == "full")
98 as::date_full(ios&: ios_);
99 } else if(key == "time") {
100 as::time(ios&: ios_);
101 if(value == "s" || value == "short")
102 as::time_short(ios&: ios_);
103 else if(value == "m" || value == "medium")
104 as::time_medium(ios&: ios_);
105 else if(value == "l" || value == "long")
106 as::time_long(ios&: ios_);
107 else if(value == "f" || value == "full")
108 as::time_full(ios&: ios_);
109 } else if(key == "dt" || key == "datetime") {
110 as::datetime(ios&: ios_);
111 if(value == "s" || value == "short") {
112 as::date_short(ios&: ios_);
113 as::time_short(ios&: ios_);
114 } else if(value == "m" || value == "medium") {
115 as::date_medium(ios&: ios_);
116 as::time_medium(ios&: ios_);
117 } else if(value == "l" || value == "long") {
118 as::date_long(ios&: ios_);
119 as::time_long(ios&: ios_);
120 } else if(value == "f" || value == "full") {
121 as::date_full(ios&: ios_);
122 as::time_full(ios&: ios_);
123 }
124 } else if(key == "spell" || key == "spellout")
125 as::spellout(ios&: ios_);
126 else if(key == "ord" || key == "ordinal")
127 as::ordinal(ios&: ios_);
128 else if(key == "left" || key == "<")
129 ios_.setf(fmtfl: std::ios_base::left, mask: std::ios_base::adjustfield);
130 else if(key == "right" || key == ">")
131 ios_.setf(fmtfl: std::ios_base::right, mask: std::ios_base::adjustfield);
132 else if(key == "gmt")
133 as::gmt(ios&: ios_);
134 else if(key == "local")
135 as::local_time(ios&: ios_);
136 else if(key == "timezone" || key == "tz")
137 ios_info::get(ios&: ios_).time_zone(value);
138 else if(key == "w" || key == "width") {
139 int v;
140 if(util::try_to_int(s: value, res&: v))
141 ios_.width(wide: v);
142 } else if(key == "p" || key == "precision") {
143 int v;
144 if(util::try_to_int(s: value, res&: v))
145 ios_.precision(prec: v);
146 } else if(key == "locale") {
147 if(!d->restore_locale) {
148 d->saved_locale = ios_.getloc();
149 d->restore_locale = true;
150 }
151
152 generator gen;
153 gen.categories(cats: category_t::formatting);
154
155 std::locale new_loc;
156 if(value.find(c: '.') == std::string::npos)
157 new_loc = gen(value + "." + std::use_facet<info>(loc: d->saved_locale).encoding());
158 else
159 new_loc = gen(value);
160
161 imbue(l: new_loc);
162 }
163 }
