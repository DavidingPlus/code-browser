39 std::vector<std::string> messages_info::get_lang_folders() const
40 {
41 // List of fallbacks: en_US@euro, en@euro, en_US, en.
42 std::vector<std::string> result;
43 if(!language.empty()) {
44 if(!variant.empty() && !country.empty())
45 result.push_back(x: language + "_" + country + "@" + variant);
46
47 if(!variant.empty())
48 result.push_back(x: language + "@" + variant);
49
50 if(!country.empty())
51 result.push_back(x: language + "_" + country);
52
53 result.push_back(x: language);
54 }
55 return result;
56 }

58 std::vector<std::string> messages_info::get_catalog_paths() const
59 {
60 const auto lang_folders = get_lang_folders();
61 std::vector<std::string> result;
62 result.reserve(n: lang_folders.size() * paths.size());
63 for(const std::string& lang_folder : lang_folders) {
64 for(const std::string& search_path : paths)
65 result.push_back(x: search_path + "/" + lang_folder + "/" + locale_category);
66 }
67 return result;

105 std::vector<char> read_file(FILE* file)
106 {
107 fseek(stream: file, off: 0, SEEK_END);
108 const auto len = ftell(stream: file);
109 if(BOOST_UNLIKELY(len < 0))
110 throw std::runtime_error("Wrong file object"); // LCOV_EXCL_LINE
111 else {
112 fseek(stream: file, off: 0, SEEK_SET);
113 std::vector<char> data(len);
114 if(BOOST_LIKELY(!data.empty()) && fread(ptr: data.data(), size: 1, n: data.size(), stream: file) != data.size())
115 throw std::runtime_error("Failed to read file"); // LCOV_EXCL_LINE
116 return data;
117 }

120 class mo_file {
121 public:
122 mo_file(std::vector<char> data) : data_(std::move(data))
123 {
124 if(data_.size() < 4)
125 throw std::runtime_error("invalid 'mo' file format - the file is too short");
126 uint32_t magic;
127 static_assert(sizeof(magic) == 4, "!");
128 memcpy(dest: &magic, src: data_.data(), n: sizeof(magic));
129 if(magic == 0x950412de)
130 native_byteorder_ = true;
131 else if(magic == 0xde120495)
132 native_byteorder_ = false;
133 else
134 throw std::runtime_error("Invalid file format - invalid magic number");
135
136 // Read all format sizes
137 size_ = get(offset: 8);
138 keys_offset_ = get(offset: 12);
139 translations_offset_ = get(offset: 16);
140 hash_size_ = get(offset: 20);
141 hash_offset_ = get(offset: 24);
142 }
143
144 string_view find(const char* context_in, const char* key_in) const
145 {
146 if(!has_hash())
147 return {};
148
149 pj_winberger_hash::state_type st = pj_winberger_hash::initial_state;
150 if(context_in) {
151 st = pj_winberger_hash::update_state(value: st, ptr: context_in);
152 st = pj_winberger_hash::update_state(value: st, c: '\4'); // EOT
153 }
154 st = pj_winberger_hash::update_state(value: st, ptr: key_in);
155 uint32_t hkey = st;
156 const uint32_t incr = 1 + hkey % (hash_size_ - 2);
157 hkey %= hash_size_;
158 const uint32_t orig_hkey = hkey;
159
160 do {
161 const uint32_t idx = get(offset: hash_offset_ + 4 * hkey);
162 // Not found
163 if(idx == 0)
164 return {};
165 // If equal values return translation
166 if(key_equals(real_key: key(id: idx - 1), cntx: context_in, key: key_in))
167 return value(id: idx - 1);
168 // Rehash
169 hkey = (hkey + incr) % hash_size_;
170 } while(hkey != orig_hkey);
171 return {};
172 }
173
174 static bool key_equals(const char* real_key, const char* cntx, const char* key)
175 {
176 if(!cntx)
177 return strcmp(s1: real_key, s2: key) == 0;
178 else {
179 const size_t real_key_len = strlen(s: real_key);
180 const size_t cntx_len = strlen(s: cntx);
181 const size_t key_len = strlen(s: key);
182 if(cntx_len + 1 + key_len != real_key_len)
183 return false;
184 return memcmp(s1: real_key, s2: cntx, n: cntx_len) == 0 && real_key[cntx_len] == '\4'
185 && memcmp(s1: real_key + cntx_len + 1, s2: key, n: key_len) == 0;
186 }
187 }
188
189 const char* key(unsigned id) const
190 {
191 const uint32_t off = get(offset: keys_offset_ + id * 8 + 4);
192 return data_.data() + off;
193 }
194
195 string_view value(unsigned id) const
196 {
197 const uint32_t len = get(offset: translations_offset_ + id * 8);
198 const uint32_t off = get(offset: translations_offset_ + id * 8 + 4);
199 if(len > data_.size() || off > data_.size() - len)
200 throw std::runtime_error("Bad mo-file format");
201 return string_view(&data_[off], len);
202 }
203
204 bool has_hash() const { return hash_size_ != 0; }
205
206 size_t size() const { return size_; }
207
208 bool empty() { return size_ == 0; }
209
210 private:
211 uint32_t get(unsigned offset) const
212 {
213 if(offset > data_.size() - 4)
214 throw std::runtime_error("Bad mo-file format");
215 uint32_t v;
216 memcpy(dest: &v, src: &data_[offset], n: 4);
217 if(!native_byteorder_)
218 v = ((v & 0xFF) << 24) | ((v & 0xFF00) << 8) | ((v & 0xFF0000) >> 8) | ((v & 0xFF000000) >> 24);
219
220 return v;
221 }
222
223 uint32_t keys_offset_;
224 uint32_t translations_offset_;
225 uint32_t hash_size_;
226 uint32_t hash_offset_;
227
228 const std::vector<char> data_;
229 bool native_byteorder_;
230 size_t size_;
231 };

284 template<typename CharType>
285 struct message_key {
286 typedef std::basic_string<CharType> string_type;
287
288 message_key(const string_type& c = string_type()) : c_context_(nullptr), c_key_(nullptr)
289 {
290 const size_t pos = c.find(CharType(4));
291 if(pos == string_type::npos)
292 key_ = c;
293 else {
294 context_ = c.substr(0, pos);
295 key_ = c.substr(pos + 1);
296 }
297 }
298 message_key(const CharType* c, const CharType* k) : c_key_(k)
299 {
300 static const CharType empty = 0;
301 if(c != nullptr)
302 c_context_ = c;
303 else
304 c_context_ = &empty;
305 }
306 bool operator<(const message_key& other) const
307 {
308 const int cc = compare(l: context(), r: other.context());
309 if(cc != 0)
310 return cc < 0;
311 return compare(l: key(), r: other.key()) < 0;
312 }
313 bool operator==(const message_key& other) const
314 {
315 return compare(l: context(), r: other.context()) == 0 && compare(l: key(), r: other.key()) == 0;
316 }
317 bool operator!=(const message_key& other) const { return !(*this == other); }
318 const CharType* context() const
319 {
320 if(c_context_)
321 return c_context_;
322 return context_.c_str();
323 }
324 const CharType* key() const
325 {
326 if(c_key_)
327 return c_key_;
328 return key_.c_str();
329 }
330
331 private:
332 static int compare(const CharType* l, const CharType* r)
333 {
334 typedef std::char_traits<CharType> traits_type;
335 for(;;) {
336 const CharType cl = *l++;
337 const CharType cr = *r++;
338 if(cl == 0 && cr == 0)
339 return 0;
340 if(traits_type::lt(cl, cr))
341 return -1;
342 if(traits_type::lt(cr, cl))
343 return 1;
344 }
345 }
346 string_type context_;
347 string_type key_;
348 const CharType* c_context_;
349 const CharType* c_key_;
350 };

352 template<typename CharType>
353 struct hash_function {
354 size_t operator()(const message_key<CharType>& msg) const
355 {
356 pj_winberger_hash::state_type state = pj_winberger_hash::initial_state;
357 const CharType* p = msg.context();
358 if(*p != 0) {
359 const CharType* e = util::str_end(p);
360 state = pj_winberger_hash::update_state(value: state,
361 begin: reinterpret_cast<const char*>(p),
362 end: reinterpret_cast<const char*>(e));
363 state = pj_winberger_hash::update_state(value: state, c: '\4');
364 }
365 p = msg.key();
366 const CharType* e = util::str_end(p);
367 state = pj_winberger_hash::update_state(value: state,
368 begin: reinterpret_cast<const char*>(p),
369 end: reinterpret_cast<const char*>(e));
370 return state;
371 }
372 };

401 template<typename CharType>
402 class mo_message : public message_format<CharType> {
403 typedef std::basic_string<CharType> string_type;
404 typedef message_key<CharType> key_type;
405 typedef std::unordered_map<key_type, string_type, hash_function<CharType>> catalog_type;
406 struct domain_data_type {
407 std::unique_ptr<mo_file> mo_catalog; /// Message catalog (.mo file) if it can be directly used
408 catalog_type catalog; /// Converted message catalog when .mo file cannot be directly used
409 lambda::plural_expr plural_form; /// Expression to determine the plural form index
410 };
411
412 public:
413 using string_view_type = typename mo_file_use_traits<CharType>::string_view_type;
414
415 const CharType* get(int domain_id, const CharType* context, const CharType* in_id) const override
416 {
417 const auto result = get_string(domain_id, context, in_id);
418 return result.empty() ? nullptr : result.data();
419 }
420
421 const CharType*
422 get(int domain_id, const CharType* context, const CharType* single_id, count_type n) const override
423 {
424 auto result = get_string(domain_id, context, in_id: single_id);
425 if(result.empty())
426 return nullptr;
427
428 // domain_id is already checked by get_string -> Would return a null-pair
429 BOOST_ASSERT(domain_id >= 0 && static_cast<size_t>(domain_id) < domain_data_.size());
430 lambda::expr::value_type plural_idx;
431 if(domain_data_[domain_id].plural_form)
432 plural_idx = domain_data_[domain_id].plural_form(n);
433 else
434 plural_idx = n == 1 ? 0 : 1; // Fallback to English plural form
435
436 for(decltype(plural_idx) i = 0; i < plural_idx; ++i) {
437 const auto pos = result.find(CharType(0));
438 if(BOOST_UNLIKELY(pos == string_view_type::npos))
439 return nullptr;
440 result.remove_prefix(pos + 1);
441 }
442 return result.empty() ? nullptr : result.data();
443 }
444
445 int domain(const std::string& domain) const override
446 {
447 const auto p = domains_.find(x: domain);
448 if(p == domains_.end())
449 return -1;
450 return p->second;
451 }

453 mo_message(const messages_info& inf) : key_conversion_required_(false)
454 {
455 const std::vector<messages_info::domain>& domains = inf.domains;
456 domain_data_.resize(domains.size());
457
458 const auto catalog_paths = inf.get_catalog_paths();
459 for(unsigned i = 0; i < domains.size(); i++) {
460 const auto& domain = domains[i];
461 domains_[domain.name] = i;
462 const std::string filename = domain.name + ".mo";
463 for(std::string path : catalog_paths) {
464 path += "/" + filename;
465 if(load_file(file_name: path, locale_encoding: inf.encoding, key_encoding: domain.encoding, data&: domain_data_[i], callback: inf.callback))
466 break;
467 }
468 }
469 }
470
471 const CharType* convert(const CharType* msg, string_type& buffer) const override
472 {
473 return runtime_conversion<CharType>(msg, buffer, key_conversion_required_, locale_encoding_, key_encoding_);
474 }

476 private:
477 bool load_file(const std::string& file_name,
478 const std::string& locale_encoding,
479 const std::string& key_encoding,
480 domain_data_type& data,
481 const messages_info::callback_type& callback)
482 {
483 locale_encoding_ = util::is_char8_t<CharType>::value ? "UTF-8" : locale_encoding;
484 key_encoding_ = key_encoding;
485
486 key_conversion_required_ =
487 sizeof(CharType) == 1 && !util::are_encodings_equal(l: locale_encoding, r: key_encoding);
488
489 std::unique_ptr<mo_file> mo;
490
491 {
492 std::vector<char> file_data;
493 if(callback)
494 file_data = callback(file_name, locale_encoding);
495 else {
496 c_file the_file(file_name, locale_encoding);
497 if(!the_file.handle)
498 return false;
499 file_data = read_file(file: the_file.handle);
500 }
501 if(file_data.empty())
502 return false;
503 mo.reset(p: new mo_file(std::move(file_data)));
504 }
505
506 const std::string plural = extract(meta: mo->value(id: 0), key: "plural=", separators: "\r\n;");
507 const std::string mo_encoding = extract(meta: mo->value(id: 0), key: "charset=", separators: " \r\n;");
508
509 if(mo_encoding.empty())
510 throw std::runtime_error("Invalid mo-format, encoding is not specified");
511
512 if(!plural.empty())
513 data.plural_form = lambda::compile(c_expression: plural.c_str());
514
515 if(mo_useable_directly(mo_encoding, mo: *mo))
516 data.mo_catalog = std::move(mo);
517 else {
518 converter<CharType> cvt_value(locale_encoding, mo_encoding);
519 converter<CharType> cvt_key(key_encoding, mo_encoding);
520 for(unsigned i = 0; i < mo->size(); i++) {
521 const char* ckey = mo->key(id: i);
522 const key_type key(cvt_key(ckey));
523
524 data.catalog[key] = cvt_value(mo->value(id: i));
525 }
526 }
527 return true;
528 }

530 // Check if the mo file as-is is useful
531 // 1. It is char and not wide character
532 // 2. The locale encoding and mo encoding is same
533 // 3. The source strings encoding and mo encoding is same or all
534 // mo key strings are US-ASCII
535 bool mo_useable_directly(const std::string& mo_encoding, const mo_file& mo)
536 {
537 BOOST_LOCALE_START_CONST_CONDITION
538 if(sizeof(CharType) != 1)
539 return false;
540 BOOST_LOCALE_END_CONST_CONDITION
541 if(!mo.has_hash())
542 return false;
543 if(!util::are_encodings_equal(l: mo_encoding, r: locale_encoding_))
544 return false;
545 if(util::are_encodings_equal(l: mo_encoding, r: key_encoding_))
546 return true;
547 for(unsigned i = 0; i < mo.size(); i++) {
548 if(!detail::is_us_ascii_string(msg: mo.key(id: i)))
549 return false;
550 }
551 return true;
552 }
553
554 static std::string extract(boost::string_view meta, const std::string& key, const boost::string_view separators)
555 {
556 const size_t pos = meta.find(s: key);
557 if(pos == boost::string_view::npos)
558 return "";
559 meta.remove_prefix(n: pos + key.size());
560 const size_t end_pos = meta.find_first_of(s: separators);
561 return std::string(meta.substr(pos: 0, n: end_pos));
562 }
563
564 string_view_type get_string(int domain_id, const CharType* context, const CharType* in_id) const
565 {
566 if(domain_id < 0 || static_cast<size_t>(domain_id) >= domain_data_.size())
567 return {};
568 const auto& data = domain_data_[domain_id];
569
570 BOOST_LOCALE_START_CONST_CONDITION
571 if(mo_file_use_traits<CharType>::in_use && data.mo_catalog) {
572 BOOST_LOCALE_END_CONST_CONDITION
573 return mo_file_use_traits<CharType>::use(*data.mo_catalog, context, in_id);
574 } else {
575 const key_type key(context, in_id);
576 const catalog_type& cat = data.catalog;
577 const auto p = cat.find(key);
578 if(p == cat.end())
579 return {};
580 return p->second;
581 }
582 }

605namespace boost { namespace locale { namespace detail {
606 std::locale install_message_facet(const std::locale& in,
607 const char_facet_t type,
608 const util::locale_data& data,
609 const std::vector<std::string>& domains,
610 const std::vector<std::string>& paths)
611 {
612 gnu_gettext::messages_info minf;
613 minf.language = data.language();
614 minf.country = data.country();
615 minf.variant = data.variant();
616 minf.encoding = data.encoding();
617 minf.domains = gnu_gettext::messages_info::domains_type(domains.begin(), domains.end());
618 minf.paths = paths;
619 switch(type) {
620 case char_facet_t::nochar: break;
621 case char_facet_t::char_f: return std::locale(in, gnu_gettext::create_messages_facet<char>(info: minf));
622 case char_facet_t::wchar_f: return std::locale(in, gnu_gettext::create_messages_facet<wchar_t>(info: minf));
623#ifndef BOOST_LOCALE_NO_CXX20_STRING8
624 case char_facet_t::char8_f: return std::locale(in, gnu_gettext::create_messages_facet<char8_t>(minf));
625#elif defined(__cpp_char8_t)
626 case char_facet_t::char8_f: break;
627#endif
628#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
629 case char_facet_t::char16_f: return std::locale(in, gnu_gettext::create_messages_facet<char16_t>(minf));
630#endif
631#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
632 case char_facet_t::char32_f: return std::locale(in, gnu_gettext::create_messages_facet<char32_t>(minf));
633#endif
634 }
635 return in;
636 }
