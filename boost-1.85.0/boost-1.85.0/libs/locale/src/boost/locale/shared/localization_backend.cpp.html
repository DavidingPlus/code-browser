38 class localization_backend_manager::impl {
39 public:
40 impl(const impl& other) : default_backends_(other.default_backends_)
41 {
42 for(const auto& i : other.all_backends_)
43 all_backends_.push_back(x: {i.first, clone(backend: *i.second)});
44 }
45 impl() : default_backends_(32, -1) {}
46
47 impl& operator=(const impl&) = delete;
48
49 localization_backend* create() const
50 {
51 std::vector<std::reference_wrapper<const localization_backend>> backends;
52 for(const auto& i : all_backends_)
53 backends.push_back(x: std::cref(t: *i.second));
54 return new actual_backend(backends, default_backends_);
55 }
56
57 int find_backend(const std::string& name) const
58 {
59 int id = 0;
60 for(const auto& i : all_backends_) {
61 if(i.first == name)
62 return id;
63 ++id;
64 }
65 return -1;
66 }
67
68 void add_backend(const std::string& name, std::unique_ptr<localization_backend> ptr)
69 {
70 if(all_backends_.empty())
71 std::fill(default_backends_.begin(), default_backends_.end(), 0);
72 if(BOOST_LIKELY(find_backend(name) < 0))
73 all_backends_.push_back(x: std::make_pair(x: name, y: std::move(ptr)));
74 }
75
76 void select(const std::string& backend_name, category_t category = all_categories)
77 {
78 const int id = find_backend(name: backend_name);
79 if(id >= 0) {
80 category_t flag = category_first;
81 for(int& defBackend : default_backends_) {
82 if(category & flag)
83 defBackend = id;
84 ++flag;
85 }
86 }
87 }
88
89 void remove_all_backends()
90 {
91 all_backends_.clear();
92 std::fill(default_backends_.begin(), default_backends_.end(), -1);
93 }
94 std::vector<std::string> get_all_backends() const
95 {
96 std::vector<std::string> res;
97 for(const auto& i : all_backends_)
98 res.push_back(x: i.first);
99 return res;
100 }

102 private:
103 class actual_backend : public localization_backend {
104 public:
105 actual_backend(const std::vector<std::reference_wrapper<const localization_backend>>& backends,
106 const std::vector<int>& index) :
107 index_(index)
108 {
109 for(const localization_backend& b : backends)
110 backends_.push_back(x: boost::locale::clone(backend: b));
111 }
112 actual_backend* clone() const override
113 {
114 std::vector<std::reference_wrapper<const localization_backend>> backends;
115 for(const auto& b : backends_)
116 backends.push_back(x: std::cref(t: *b));
117 return new actual_backend(backends, index_);
118 }
119 void set_option(const std::string& name, const std::string& value) override
120 {
121 for(const auto& b : backends_)
122 b->set_option(name, value);
123 }
124 void clear_options() override
125 {
126 for(const auto& b : backends_)
127 b->clear_options();
128 }
129 std::locale install(const std::locale& l, category_t category, char_facet_t type) override
130 {
131 unsigned id = 0;
132 for(category_t v = category_first; v != category; ++v, ++id) {
133 if(v == category_last)
134 return l;
135 }
136 if(id >= index_.size() || index_[id] == -1)
137 return l;
138 return backends_[index_[id]]->install(base: l, category, type);
139 }

141 private:
142 std::vector<std::unique_ptr<localization_backend>> backends_;
143 std::vector<int> index_;
144 };

148 };
