305 void date_time::time(double v)
306 {
307 constexpr int64_t ns_in_s = static_cast<int64_t>(1000) * 1000 * 1000;
308
309 double seconds;
310 const double fract_seconds = std::modf(x: v, iptr: &seconds); // v = seconds + fract_seconds
311 posix_time ptime;
312 ptime.seconds = static_cast<int64_t>(seconds);
313 int64_t nano = static_cast<int64_t>(fract_seconds * ns_in_s);
314
315 if(nano < 0) {
316 // Add 1s from seconds to nano to make nano positive
317 ptime.seconds -= 1;
318 nano = std::max(a: int64_t(0), b: nano + ns_in_s); // std::max to handle rounding issues
319 } else if(nano >= ns_in_s) // Unlikely rounding issue, when fract_seconds is close to 1.
320 nano = ns_in_s - 1; // LCOV_EXCL_LINE
321
322 BOOST_ASSERT(nano < ns_in_s);
323 static_assert(ns_in_s <= std::numeric_limits<uint32_t>::max(), "Insecure cast");
324 ptime.nanoseconds = static_cast<uint32_t>(nano);
325 impl_->set_time(ptime);
326 }

333 namespace {
334 int compare(const posix_time& left, const posix_time& right)
335 {
336 if(left.seconds < right.seconds)
337 return -1;
338 if(left.seconds > right.seconds)
339 return 1;
340 if(left.nanoseconds < right.nanoseconds)
341 return -1;
342 if(left.nanoseconds > right.nanoseconds)
343 return 1;
344 return 0;
345 }
