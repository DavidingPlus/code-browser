7#ifndef BOOST_LOCALE_STD_COLLATE_ADAPTER_HPP
8#define BOOST_LOCALE_STD_COLLATE_ADAPTER_HPP
9
10#include <boost/locale/collator.hpp>
11#include <locale>
12#include <utility>
13
14namespace boost { namespace locale { namespace impl {
15
16 template<typename CharT, class Base>
17 class BOOST_SYMBOL_VISIBLE std_collate_adapter : public std::collate<CharT> {
18 public:
19 using typename std::collate<CharT>::string_type;
20
21 template<typename... TArgs>
22 explicit std_collate_adapter(TArgs&&... args) : base_(std::forward<TArgs>(args)...)
23 {}
24
25 protected:
26 int do_compare(const CharT* beg1, const CharT* end1, const CharT* beg2, const CharT* end2) const override
27 {
28 return base_.compare(collate_level::identical, beg1, end1, beg2, end2);
29 }
30
31 string_type do_transform(const CharT* beg, const CharT* end) const override
32 {
33 return base_.transform(collate_level::identical, beg, end);
34 }
35 long do_hash(const CharT* beg, const CharT* end) const override
36 {
37 return base_.hash(collate_level::identical, beg, end);
38 }
39 Base base_;
40 };
41
42 template<typename CharType, class CollatorImpl, typename... TArgs>
43 static std::locale create_collators(const std::locale& in, TArgs&&... args)
44 {
45 static_assert(std::is_base_of<collator<CharType>, CollatorImpl>::value, "Must be a collator implementation");
46 std::locale res(in, new CollatorImpl(args...));
47 return std::locale(res, new std_collate_adapter<CharType, CollatorImpl>(args...));
48 }
49
50 template<typename CharType, template<typename> class CollatorImpl, typename... TArgs>
51 static std::locale create_collators(const std::locale& in, TArgs&&... args)
52 {
53 return create_collators<CharType, CollatorImpl<CharType>>(in, args...);
54 }
55
56}}} // namespace boost::locale::impl
57
58#endif
