49 formatters_cache::formatters_cache(const icu::Locale& locale) : locale_(locale)
50 {
51#define BOOST_LOCALE_ARRAY_SIZE(T) std::extent<typename std::remove_reference<decltype(T)>::type>::value
52 constexpr icu::DateFormat::EStyle styles[]{icu::DateFormat::kShort,
53 icu::DateFormat::kMedium,
54 icu::DateFormat::kLong,
55 icu::DateFormat::kFull};
56 constexpr int num_styles = BOOST_LOCALE_ARRAY_SIZE(styles);
57
58 static_assert(num_styles == BOOST_LOCALE_ARRAY_SIZE(date_format_), "!");
59 for(int i = 0; i < num_styles; i++)
60 get_icu_pattern(fmt: icu::DateFormat::createDateInstance(style: styles[i], aLocale: locale), out_str&: date_format_[i]);
61
62 static_assert(num_styles == BOOST_LOCALE_ARRAY_SIZE(time_format_), "!");
63 for(int i = 0; i < num_styles; i++)
64 get_icu_pattern(fmt: icu::DateFormat::createTimeInstance(style: styles[i], aLocale: locale), out_str&: time_format_[i]);
65
66 static_assert(num_styles == BOOST_LOCALE_ARRAY_SIZE(date_time_format_), "!");
67 static_assert(num_styles == BOOST_LOCALE_ARRAY_SIZE(date_time_format_[0]), "!");
68 for(int i = 0; i < num_styles; i++) {
69 for(int j = 0; j < num_styles; j++) {
70 get_icu_pattern(fmt: icu::DateFormat::createDateTimeInstance(dateStyle: styles[i], timeStyle: styles[j], aLocale: locale),
71 out_str&: date_time_format_[i][j]);
72 }
73 }
74#undef BOOST_LOCALE_ARRAY_SIZE
75
76 const auto get_str_or = [](const icu::UnicodeString& str, const char* default_str) {
77 return str.isEmpty() ? default_str : str;
78 };
79 default_date_format_ = get_str_or(date_format(f: format_len::Medium), "yyyy-MM-dd");
80 default_time_format_ = get_str_or(time_format(f: format_len::Medium), "HH:mm:ss");
81 default_date_time_format_ =
82 get_str_or(date_time_format(d: format_len::Full, t: format_len::Full), "yyyy-MM-dd HH:mm:ss");
83 }

85 icu::NumberFormat* formatters_cache::create_number_format(num_fmt_type type, UErrorCode& err) const
86 {
87 switch(type) {
88 case num_fmt_type::number: return icu::NumberFormat::createInstance(inLocale: locale_, err); break;
89 case num_fmt_type::sci: return icu::NumberFormat::createScientificInstance(inLocale: locale_, err); break;
90#if BOOST_LOCALE_ICU_VERSION >= 408
91 case num_fmt_type::curr_nat: return icu::NumberFormat::createInstance(desiredLocale: locale_, style: UNUM_CURRENCY, errorCode&: err); break;
92 case num_fmt_type::curr_iso:
93 return icu::NumberFormat::createInstance(desiredLocale: locale_, style: UNUM_CURRENCY_ISO, errorCode&: err);
94 break;
95#elif BOOST_LOCALE_ICU_VERSION >= 402
96 case num_fmt_type::curr_nat:
97 return icu::NumberFormat::createInstance(locale_, icu::NumberFormat::kCurrencyStyle, err);
98 break;
99 case num_fmt_type::curr_iso:
100 return icu::NumberFormat::createInstance(locale_, icu::NumberFormat::kIsoCurrencyStyle, err);
101 break;
102#else
103 case num_fmt_type::curr_nat:
104 case num_fmt_type::curr_iso: return icu::NumberFormat::createCurrencyInstance(locale_, err); break;
105#endif
106 case num_fmt_type::percent: return icu::NumberFormat::createPercentInstance(inLocale: locale_, err); break;
107 case num_fmt_type::spell: return new icu::RuleBasedNumberFormat(icu::URBNF_SPELLOUT, locale_, err); break;
108 case num_fmt_type::ordinal: return new icu::RuleBasedNumberFormat(icu::URBNF_ORDINAL, locale_, err); break;
109 }
110 throw std::logic_error("locale::internal error should not get there"); // LCOV_EXCL_LINE
111 }

113 icu::NumberFormat& formatters_cache::number_format(num_fmt_type type) const
114 {
115 icu::NumberFormat* result = number_format_[int(type)].get();
116 if(!result) {
117 UErrorCode err = U_ZERO_ERROR;
118 std::unique_ptr<icu::NumberFormat> new_ptr(create_number_format(type, err));
119 check_and_throw_icu_error(err, desc: "Failed to create a formatter");
120 result = new_ptr.get();
121 BOOST_ASSERT(result);
122 number_format_[int(type)].reset(new_value: new_ptr.release());
123 }
124 return *result;
125 }

127 icu::SimpleDateFormat* formatters_cache::date_formatter() const
128 {
129 icu::SimpleDateFormat* result = date_formatter_.get();
130 if(!result) {
131 std::unique_ptr<icu::DateFormat> fmt(
132 icu::DateFormat::createDateTimeInstance(dateStyle: icu::DateFormat::kMedium, timeStyle: icu::DateFormat::kMedium, aLocale: locale_));
133
134 result = icu_cast<icu::SimpleDateFormat>(p: fmt.get());
135 if(result) {
136 fmt.release();
137 date_formatter_.reset(new_value: result);
138 }
139 }
140 return result;
141 }
