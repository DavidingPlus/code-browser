63 // Under BSD, Linux and Mac OS X dirent has normal size
64 // so no issues with readdir_r
65
66 class directory {
67 public:
68 directory(const char* name) : d(0), read_result(0)
69 {
70 d = opendir(name);
71 if(!d)
72 return;
73 }
74 ~directory()
75 {
76 if(d)
77 closedir(d);
78 }
79 bool is_open() { return d; }
80 const char* next()
81 {
82 if(d && readdir_r(d, &de, &read_result) == 0 && read_result != 0)
83 return de.d_name;
84 return nullptr;
85 }
86
87 private:
88 DIR* d;
89 struct dirent de;
90 struct dirent* read_result;
91 };

93 bool files_equal(const std::string& left, const std::string& right)
94 {
95 char l[256], r[256];
96 std::ifstream ls(left);
97 if(!ls)
98 return false;
99 std::ifstream rs(right);
100 if(!rs)
101 return false;
102 do {
103 ls.read(l, sizeof(l));
104 rs.read(r, sizeof(r));
105 size_t n;
106 if((n = ls.gcount()) != size_t(rs.gcount()))
107 return false;
108 if(memcmp(l, r, n) != 0)
109 return false;
110 } while(!ls.eof() || !rs.eof());
111 if(bool(ls.eof()) != bool(rs.eof()))
112 return false;
113 return true;
114 }

116 std::string find_file_in(const std::string& ref, size_t size, const std::string& dir)
117 {
118 directory d(dir.c_str());
119 if(!d.is_open())
120 return std::string();
121
122 const char* name = nullptr;
123 while((name = d.next()) != 0) {
124 std::string file_name = name;
125 if(file_name == "." || file_name == ".." || file_name == "posixrules" || file_name == "localtime")
126 continue;
127 struct stat st;
128 std::string path = dir + "/" + file_name;
129 if(stat(path.c_str(), &st) == 0) {
130 if(S_ISDIR(st.st_mode)) {
131 std::string res = find_file_in(ref, size, path);
132 if(!res.empty())
133 return file_name + "/" + res;
134 } else {
135 if(size_t(st.st_size) == size && files_equal(path, ref))
136 return file_name;
137 }
138 }
139 }
140 return std::string();
141 }

143 // This actually emulates ICU's search
144 // algorithm... just it ignores localtime
145 std::string detect_correct_time_zone()
146 {
147 const char* tz_dir = "/usr/share/zoneinfo";
148 const char* tz_file = "/etc/localtime";
149
150 struct stat st;
151 if(::stat(tz_file, &st) != 0)
152 return std::string();
153 size_t size = st.st_size;
154 std::string r = find_file_in(tz_file, size, tz_dir);
155 if(r.empty())
156 return r;
157 if(r.compare(0, 6, "posix/") == 0 || r.compare(0, 6, "right/", 6) == 0)
158 return r.substr(6);
159 return r;
160 }

162 // Using pthread as:
163 // - This bug is relevant for only Linux, BSD, Mac OS X and
164 // pthreads are native threading API
165 // - The dependency on boost.thread may be removed when using
166 // more recent ICU versions (so TLS would not be needed)
167 //
168 // This the dependency on Boost.Thread is eliminated
169
170 pthread_once_t init_tz = PTHREAD_ONCE_INIT;
171 std::string default_time_zone_name;
172
173 extern "C" {
174 static void init_tz_proc()
175 {
176 try {
177 default_time_zone_name = detect_correct_time_zone();
178 } catch(...) {
179 }
180 }
181 }
182
183 std::string get_time_zone_name()
184 {
185 pthread_once(&init_tz, init_tz_proc);
186 return default_time_zone_name;
187 }

191 icu::TimeZone* get_time_zone(const std::string& time_zone)
192 {
193 if(!time_zone.empty())
194 return icu::TimeZone::createTimeZone(time_zone.c_str());
195 hold_ptr<icu::TimeZone> tz(icu::TimeZone::createDefault());
196 icu::UnicodeString id;
197 tz->getID(id);
198 // Check if there is a bug?
199 if(id != icu::UnicodeString("localtime"))
200 return tz.release();
201 // Now let's deal with the bug and run the fixed
202 // search loop as that of ICU
203 std::string real_id = get_time_zone_name();
204 if(real_id.empty()) {
205 // if we failed fallback to ICU's time zone
206 return tz.release();
207 }
208 return icu::TimeZone::createTimeZone(real_id.c_str());
209 }
210#endif // bug workaround
