31 class icu_handle {
32 UConverter* h_;
33 void close()
34 {
35 if(h_)
36 ucnv_close(converter: h_);

38
39 public:
40 explicit icu_handle(UConverter* h = nullptr) : h_(h) {}
41
42 icu_handle(const icu_handle& rhs) = delete;
43 icu_handle(icu_handle&& rhs) noexcept : h_(exchange(t&: rhs.h_, u: nullptr)) {}
44
45 icu_handle& operator=(const icu_handle& rhs) = delete;
46 icu_handle& operator=(icu_handle&& rhs) noexcept
47 {
48 h_ = exchange(t&: rhs.h_, u: nullptr);
49 return *this;
50 }

52 {
53 close();
54 h_ = h;
55 return *this;
56 }

61 };

63 enum class cpcvt_type { skip, stop };
64
65 struct uconv {
66 uconv(const uconv& other) = delete;
67 void operator=(const uconv& other) = delete;
68
69 uconv(const std::string& charset, cpcvt_type cvt_type = cpcvt_type::skip)
70 {
71 UErrorCode err = U_ZERO_ERROR;
72 cvt_ = ucnv_open(converterName: charset.c_str(), err: &err);
73 if(!cvt_ || U_FAILURE(code: err))
74 throw conv::invalid_charset_error(charset);
75
76 if(cvt_type == cpcvt_type::skip) {
77 ucnv_setFromUCallBack(converter: cvt_, UCNV_FROM_U_CALLBACK_SKIP, newContext: nullptr, oldAction: nullptr, oldContext: nullptr, err: &err);
78 ucnv_setToUCallBack(converter: cvt_, UCNV_TO_U_CALLBACK_SKIP, newContext: nullptr, oldAction: nullptr, oldContext: nullptr, err: &err);
79 check_and_throw_icu_error(err);
80 } else {
81 ucnv_setFromUCallBack(converter: cvt_, UCNV_FROM_U_CALLBACK_STOP, newContext: nullptr, oldAction: nullptr, oldContext: nullptr, err: &err);
82 ucnv_setToUCallBack(converter: cvt_, UCNV_TO_U_CALLBACK_STOP, newContext: nullptr, oldAction: nullptr, oldContext: nullptr, err: &err);
83 check_and_throw_icu_error(err);
84 }
85 }

87 int max_char_size() const { return ucnv_getMaxCharSize(converter: cvt_); }

89 template<typename U8Char = char>
90 std::basic_string<U8Char> go(const UChar* buf, int length, int max_size) const
91 {
92 static_assert(sizeof(U8Char) == sizeof(char), "Not an UTF-8 char type");
93 std::basic_string<U8Char> res;
94 res.resize(UCNV_GET_MAX_BYTES_FOR_STRING(length, max_size));
95 char* ptr = reinterpret_cast<char*>(&res[0]);
96 UErrorCode err = U_ZERO_ERROR;
97 int n = ucnv_fromUChars(cvt_, ptr, res.size(), buf, length, &err);
98 check_and_throw_icu_error(err);
99 res.resize(n);
100 return res;
101 }
102
103 size_t cut(size_t n, const char* begin, const char* end) const
104 {
105 const char* saved = begin;
106 while(n > 0 && begin < end) {
107 UErrorCode err = U_ZERO_ERROR;
108 ucnv_getNextUChar(converter: cvt_, source: &begin, sourceLimit: end, err: &err);
109 if(U_FAILURE(code: err))
110 return 0;
111 n--;
112 }
113 return begin - saved;
114 }
115
116 UConverter* cvt() const { return cvt_; }
117
118 private:
119 icu_handle cvt_;
120 };

125 template<typename CharType>
126 class icu_std_converter<CharType, 1> {
127 public:
128 typedef std::basic_string<CharType> string_type;
129
130 icu::UnicodeString icu_checked(const CharType* vb, const CharType* ve) const
131 {
132 return icu(vb, ve); // Already done
133 }
134 icu::UnicodeString icu(const CharType* vb, const CharType* ve) const
135 {
136 const char* begin = reinterpret_cast<const char*>(vb);
137 const char* end = reinterpret_cast<const char*>(ve);
138 UErrorCode err = U_ZERO_ERROR;
139 icu::UnicodeString tmp(begin, end - begin, cvt_.cvt(), err);
140 check_and_throw_icu_error(err);
141 return tmp;
142 }
143
144 string_type std(const icu::UnicodeString& str) const
145 {
146 return cvt_.go<CharType>(str.getBuffer(), str.length(), max_len_);
147 }
148
149 icu_std_converter(const std::string& charset, cpcvt_type cvt_type = cpcvt_type::skip) :
150 cvt_(charset, cvt_type), max_len_(cvt_.max_char_size())
151 {}
152
153 size_t cut(const icu::UnicodeString& str,
154 const CharType* begin,
155 const CharType* end,
156 size_t n,
157 size_t from_u = 0,
158 size_t from_char = 0) const
159 {
160 size_t code_points = str.countChar32(start: from_u, length: n);
161 return cvt_.cut(n: code_points,
162 begin: reinterpret_cast<const char*>(begin) + from_char,
163 end: reinterpret_cast<const char*>(end));
164 }
165
166 private:
167 uconv cvt_;
168 const int max_len_;
169 };

171 template<typename CharType>
172 class icu_std_converter<CharType, 2> {
173 public:
174 typedef std::basic_string<CharType> string_type;
175
176 icu::UnicodeString icu_checked(const CharType* begin, const CharType* end) const
177 {
178 icu::UnicodeString tmp(end - begin, 0, 0); // make initial capacity
179 while(begin != end) {
180 UChar cl = *begin++;
181 if(U16_IS_SINGLE(cl))
182 tmp.append(srcChar: static_cast<UChar32>(cl));
183 else if(U16_IS_LEAD(cl)) {
184 if(begin == end)
185 throw_if_needed();
186 else {
187 UChar ct = *begin++;
188 if(!U16_IS_TRAIL(ct))
189 throw_if_needed();
190 else {
191 UChar32 c = U16_GET_SUPPLEMENTARY(cl, ct);
192 tmp.append(srcChar: c);
193 }
194 }
195 } else
196 throw_if_needed();
197 }
198 return tmp;
199 }
200 void throw_if_needed() const
201 {
202 if(mode_ == cpcvt_type::stop)
203 throw conv::conversion_error();
204 }
205 icu::UnicodeString icu(const CharType* vb, const CharType* ve) const
206 {
207 static_assert(sizeof(CharType) == sizeof(UChar), "Size mismatch!");
208 const UChar* begin = reinterpret_cast<const UChar*>(vb);
209 const UChar* end = reinterpret_cast<const UChar*>(ve);
210 icu::UnicodeString tmp(begin, end - begin);
211 return tmp;
212 }
213
214 string_type std(const icu::UnicodeString& str) const
215 {
216 static_assert(sizeof(CharType) == sizeof(UChar), "Size mismatch!");
217 const CharType* ptr = reinterpret_cast<const CharType*>(str.getBuffer());
218 return string_type(ptr, str.length());
219 }

236 template<typename CharType>
237 class icu_std_converter<CharType, 4> {
238 public:
239 typedef std::basic_string<CharType> string_type;
240
241 icu::UnicodeString icu_checked(const CharType* begin, const CharType* end) const
242 {
243 // Fast path checking if the full string is already valid
244 {
245 UErrorCode err = U_ZERO_ERROR;
246 static_assert(sizeof(CharType) == sizeof(UChar32), "Size mismatch!");
247 u_strFromUTF32(nullptr, 0, nullptr, reinterpret_cast<const UChar32*>(begin), end - begin, &err);
248 if(err != U_INVALID_CHAR_FOUND)
249 return icu::UnicodeString::fromUTF32(utf32: reinterpret_cast<const UChar32*>(begin), length: end - begin);
250 }
251 // Any char is invalid
252 throw_if_needed();
253 // If not thrown skip invalid chars
254 icu::UnicodeString tmp(end - begin, 0, 0); // make initial capacity
255 while(begin != end) {
256 const UChar32 c = static_cast<UChar32>(*begin++);
257 // Maybe simply: UCHAR_MIN_VALUE <= c && c <= UCHAR_MAX_VALUE && !U_IS_SURROGATE(c)
258 UErrorCode err = U_ZERO_ERROR;
259 u_strFromUTF32(dest: nullptr, destCapacity: 0, pDestLength: nullptr, src: &c, srcLength: 1, pErrorCode: &err);
260 if(err != U_INVALID_CHAR_FOUND)
261 tmp.append(srcChar: c);
262 }
263 return tmp;
264 }
265 void throw_if_needed() const
266 {
267 if(mode_ == cpcvt_type::stop)
268 throw conv::conversion_error();
269 }
270
271 icu::UnicodeString icu(const CharType* begin, const CharType* end) const
272 {
273 icu::UnicodeString tmp(end - begin, 0, 0); // make initial capacity
274 while(begin != end) {
275 UChar32 c = static_cast<UChar32>(*begin++);
276 tmp.append(srcChar: c);
277 }
278 return tmp;
279 }
280
281 string_type std(const icu::UnicodeString& str) const
282 {
283 string_type tmp;
284 tmp.resize(str.length());
285 UChar32* ptr = reinterpret_cast<UChar32*>(&tmp[0]);
286 int32_t len = 0;
287 UErrorCode code = U_ZERO_ERROR;
288 u_strToUTF32(ptr, tmp.size(), &len, str.getBuffer(), str.length(), &code);
289
290 check_and_throw_icu_error(err: code);
291
292 tmp.resize(len);
293
294 return tmp;
295 }

303 {
304 return str.countChar32(start: from_u, length: n);
