7#include <boost/locale/util.hpp>
8#include "boost/locale/std/all_generator.hpp"
9#include <locale>

11namespace boost { namespace locale { namespace impl_std {
12 template<typename CharType>
13 std::locale codecvt_bychar(const std::locale& in, const std::string& locale_name)
14 {
15 return std::locale(in, new std::codecvt_byname<CharType, char, std::mbstate_t>(locale_name));
16 }
17
18 std::locale
19 create_codecvt(const std::locale& in, const std::string& locale_name, char_facet_t type, utf8_support utf)
20 {
21#if defined(BOOST_WINDOWS)
22 // This isn't fully correct:
23 // It will treat the 2-Byte wchar_t as UTF-16 encoded while it may be UCS-2
24 // std::basic_filebuf explicitely disallows using suche multi-byte codecvts
25 // but it works in practice so far, so use it instead of failing for codepoints above U+FFFF
26 if(utf != utf8_support::none)
27 return util::create_utf8_codecvt(in, type);
28#endif
29 if(utf == utf8_support::from_wide)
30 return util::create_utf8_codecvt(in, type);
31 switch(type) {
32 case char_facet_t::nochar: break;
33 case char_facet_t::char_f: return codecvt_bychar<char>(in, locale_name);
34 case char_facet_t::wchar_f: return codecvt_bychar<wchar_t>(in, locale_name);
35#ifdef __cpp_char8_t
36 case char_facet_t::char8_f: break; // std-facet not available (yet)
37#endif
38#if defined(BOOST_LOCALE_ENABLE_CHAR16_T)
39 case char_facet_t::char16_f: return codecvt_bychar<char16_t>(in, locale_name);
40#endif
41#if defined(BOOST_LOCALE_ENABLE_CHAR32_T)
42 case char_facet_t::char32_f: return codecvt_bychar<char32_t>(in, locale_name);
43#endif
44 }
45 return in;
46 }
47
48}}} // namespace boost::locale::impl_std
