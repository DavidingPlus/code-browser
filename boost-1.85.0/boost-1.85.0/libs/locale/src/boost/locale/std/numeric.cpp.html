21 /// Forwarding time_put facet
22 /// Almost the same as `std::time_put_byname` but replaces the locale of the `ios_base` in `do_put` so that e.g.
23 /// weekday names are translated and formatting is as per the language of the base locale
24 template<typename CharType>
25 class time_put_from_base : public std::time_put<CharType> {
26 public:
27 using iter_type = typename std::time_put<CharType>::iter_type;
28
29 time_put_from_base(const std::locale& base) :
30 base_facet_(std::use_facet<std::time_put<CharType>>(base)), base_ios_(nullptr)
31 {
32 // Imbue the ios with the base locale so the facets `do_put` uses its formatting information
33 base_ios_.imbue(base);
34 }
35
36 iter_type do_put(iter_type out,
37 std::ios_base& /*ios*/,
38 CharType fill,
39 const std::tm* tm,
40 char format,
41 char modifier) const override
42 {
43 return base_facet_.put(out, base_ios_, fill, tm, format, modifier);
44 }
45
46 private:
47 const std::time_put<CharType>& base_facet_;
48 mutable std::basic_ios<CharType> base_ios_;
49 };

51 class utf8_time_put_from_wide : public std::time_put<char> {
52 public:
53 utf8_time_put_from_wide(const std::locale& base, size_t refs = 0) : std::time_put<char>(refs), base_(base) {}
54 iter_type do_put(iter_type out,
55 std::ios_base& /*ios*/,
56 char fill,
57 const std::tm* tm,
58 char format,
59 char modifier = 0) const override
60 {
61 std::basic_ostringstream<wchar_t> wtmps;
62 wtmps.imbue(loc: base_);
63 std::use_facet<std::time_put<wchar_t>>(loc: base_)
64 .put(s: wtmps, io&: wtmps, fill: wchar_t(fill), tm: tm, format: wchar_t(format), mod: wchar_t(modifier));
65 const std::string tmp = conv::utf_to_utf<char>(str: wtmps.str());
66 return std::copy(tmp.begin(), tmp.end(), out);
67 }
68
69 private:
70 std::locale base_;
71 };

73 class utf8_numpunct_from_wide : public std::numpunct<char> {
74 public:
75 utf8_numpunct_from_wide(const std::locale& base, size_t refs = 0) : std::numpunct<char>(refs)
76 {
77 typedef std::numpunct<wchar_t> wfacet_type;
78 const wfacet_type& wfacet = std::use_facet<wfacet_type>(loc: base);
79
80 truename_ = conv::utf_to_utf<char>(str: wfacet.truename());
81 falsename_ = conv::utf_to_utf<char>(str: wfacet.falsename());
82
83 wchar_t tmp_decimal_point = wfacet.decimal_point();
84 wchar_t tmp_thousands_sep = wfacet.thousands_sep();
85 std::string tmp_grouping = wfacet.grouping();
86
87 if(32 <= tmp_thousands_sep && tmp_thousands_sep <= 126 && 32 <= tmp_decimal_point
88 && tmp_decimal_point <= 126) {
89 thousands_sep_ = static_cast<char>(tmp_thousands_sep);
90 decimal_point_ = static_cast<char>(tmp_decimal_point);
91 grouping_ = tmp_grouping;
92 } else if(32 <= tmp_decimal_point && tmp_decimal_point <= 126 && tmp_thousands_sep == 0xA0) {
93 // workaround common bug - substitute NBSP with ordinary space
94 thousands_sep_ = ' ';
95 decimal_point_ = static_cast<char>(tmp_decimal_point);
96 grouping_ = tmp_grouping;
97 } else if(32 <= tmp_decimal_point && tmp_decimal_point <= 126) {
98 thousands_sep_ = ',';
99 decimal_point_ = static_cast<char>(tmp_decimal_point);
100 grouping_ = std::string();
101 } else {
102 thousands_sep_ = ',';
103 decimal_point_ = '.';
104 grouping_ = std::string();
105 }
106 }
107
108 char do_decimal_point() const override { return decimal_point_; }
109 char do_thousands_sep() const override { return thousands_sep_; }
110 std::string do_grouping() const override { return grouping_; }
111 std::string do_truename() const override { return truename_; }
112 std::string do_falsename() const override { return falsename_; }
113
114 private:
115 std::string truename_;
116 std::string falsename_;
117 char thousands_sep_;
118 char decimal_point_;
119 std::string grouping_;
120 };

188 class utf8_numpunct : public std::numpunct_byname<char> {
189 public:
190 typedef std::numpunct_byname<char> base_type;
191 utf8_numpunct(const std::string& name, size_t refs = 0) : std::numpunct_byname<char>(name, refs) {}
192 char do_thousands_sep() const override
193 {
194 unsigned char bs = base_type::do_thousands_sep();
195 if(bs > 127)
196 if(bs == 0xA0)
197 return ' ';
198 else
199 return 0;
200 else
201 return bs;
202 }
203 std::string do_grouping() const override
204 {
205 unsigned char bs = base_type::do_thousands_sep();
206 if(bs > 127 && bs != 0xA0)
207 return std::string();
208 return base_type::do_grouping();
209 }
210 };

237 template<typename CharType>
238 std::locale create_basic_parsing(const std::locale& in, const std::string& locale_name)
239 {
240 std::locale tmp = std::locale(in, new std::numpunct_byname<CharType>(locale_name));
241 tmp = std::locale(tmp, new std::moneypunct_byname<CharType, true>(locale_name));
242 tmp = std::locale(tmp, new std::moneypunct_byname<CharType, false>(locale_name));
243 tmp = std::locale(tmp, new std::ctype_byname<CharType>(locale_name));
244 return std::locale(tmp, new util::base_num_parse<CharType>());
245 }

247 template<typename CharType>
248 std::locale create_basic_formatting(const std::locale& in, const std::string& locale_name)
249 {
250 std::locale tmp = create_basic_parsing<CharType>(in, locale_name);
251 tmp = std::locale(tmp, new time_put_from_base<CharType>(std::locale(locale_name)));
252 return std::locale(tmp, new util::base_num_format<CharType>());
253 }

255 std::locale
256 create_formatting(const std::locale& in, const std::string& locale_name, char_facet_t type, utf8_support utf)
257 {
258 switch(type) {
259 case char_facet_t::nochar: break;
260 case char_facet_t::char_f:
261 if(utf != utf8_support::none) {
262 const std::locale base(locale_name);
263 std::time_put<char>* time_put;
264 if(utf == utf8_support::from_wide)
265 time_put = new utf8_time_put_from_wide(base);
266 else
267 time_put = new time_put_from_base<char>(base);
268 std::locale tmp(in, time_put);
269 // Fix possibly invalid UTF-8
270 tmp = std::locale(tmp, new utf8_numpunct_from_wide(base));
271 tmp = std::locale(tmp, new utf8_moneypunct_from_wide<true>(base));
272 tmp = std::locale(tmp, new utf8_moneypunct_from_wide<false>(base));
273 return std::locale(tmp, new util::base_num_format<char>());
274 } else
275 return create_basic_formatting<char>(in, locale_name);
276 case char_facet_t::wchar_f: return create_basic_formatting<wchar_t>(in, locale_name);
277#ifdef __cpp_char8_t
278 case char_facet_t::char8_f: break; // std-facet not available (yet)
279#endif
280#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
281 case char_facet_t::char16_f: return create_basic_formatting<char16_t>(in, locale_name);
282#endif
283#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
284 case char_facet_t::char32_f: return create_basic_formatting<char32_t>(in, locale_name);
285#endif
286 }
287 return in;
288 }
