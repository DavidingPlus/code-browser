25 template<typename CharType>
26 struct case_traits;
27
28 template<>
29 struct case_traits<char> {
30 static char lower(char c, locale_t lc) { return tolower_l(c: c, l: lc); }
31 static char upper(char c, locale_t lc) { return toupper_l(c: c, l: lc); }
32 };
33
34 template<>
35 struct case_traits<wchar_t> {
36 static wchar_t lower(wchar_t c, locale_t lc) { return towlower_l(wc: c, locale: lc); }
37 static wchar_t upper(wchar_t c, locale_t lc) { return towupper_l(wc: c, locale: lc); }
38 };
39
40 template<typename CharType>
41 class std_converter : public converter<CharType> {
42 public:
43 typedef std::basic_string<CharType> string_type;
44 typedef std::ctype<CharType> ctype_type;
45 std_converter(std::shared_ptr<locale_t> lc, size_t refs = 0) : converter<CharType>(refs), lc_(std::move(lc)) {}
46 string_type convert(converter_base::conversion_type how,
47 const CharType* begin,
48 const CharType* end,
49 int /*flags*/ = 0) const override
50 {
51 switch(how) {
52 case converter_base::upper_case: {
53 string_type res;
54 res.reserve(end - begin);
55 while(begin != end)
56 res += case_traits<CharType>::upper(*begin++, *lc_);
57 return res;
58 }
59 case converter_base::lower_case:
60 case converter_base::case_folding: {
61 string_type res;
62 res.reserve(end - begin);
63 while(begin != end)
64 res += case_traits<CharType>::lower(*begin++, *lc_);
65 return res;
66 }
67 case converter_base::normalization:
68 case converter_base::title_case: break;
69 }
70 return string_type(begin, end - begin);
71 }
72
73 private:
74 std::shared_ptr<locale_t> lc_;
75 };

77 template<typename U8Char>
78 class utf8_converter : public converter<U8Char> {
79 public:
80 static_assert(sizeof(U8Char) == sizeof(char), "Not an UTF-8 char type");
81
82 utf8_converter(std::shared_ptr<locale_t> lc, size_t refs = 0) : converter<U8Char>(refs), lc_(std::move(lc)) {}
83 std::basic_string<U8Char> convert(converter_base::conversion_type how,
84 const U8Char* begin,
85 const U8Char* end,
86 int /*flags*/ = 0) const override
87 {
88 using conversion_type = converter_base::conversion_type;
89 switch(how) {
90 case conversion_type::upper_case: {
91 const std::wstring tmp = conv::utf_to_utf<wchar_t>(begin, end);
92 std::wstring wres;
93 wres.reserve(res_arg: tmp.size());
94 for(const wchar_t c : tmp)
95 wres += towupper_l(wc: c, locale: *lc_);
96 return conv::utf_to_utf<U8Char>(wres);
97 }
98
99 case conversion_type::lower_case:
100 case conversion_type::case_folding: {
101 const std::wstring tmp = conv::utf_to_utf<wchar_t>(begin, end);
102 std::wstring wres;
103 wres.reserve(res_arg: tmp.size());
104 for(const wchar_t c : tmp)
105 wres += towlower_l(wc: c, locale: *lc_);
106 return conv::utf_to_utf<U8Char>(wres);
107 }
108 case conversion_type::normalization:
109 case conversion_type::title_case: break;
110 }
111 return std::basic_string<U8Char>(begin, end - begin);
112 }
113
114 private:
115 std::shared_ptr<locale_t> lc_;
116 };

118 std::locale create_convert(const std::locale& in, std::shared_ptr<locale_t> lc, char_facet_t type)
119 {
120 switch(type) {
121 case char_facet_t::nochar: break;
122 case char_facet_t::char_f: {
123 if(util::normalize_encoding(encoding: nl_langinfo_l(CODESET, l: *lc)) == "utf8")
124 return std::locale(in, new utf8_converter<char>(std::move(lc)));
125 return std::locale(in, new std_converter<char>(std::move(lc)));
126 }
127 case char_facet_t::wchar_f: return std::locale(in, new std_converter<wchar_t>(std::move(lc)));
128#ifndef BOOST_LOCALE_NO_CXX20_STRING8
129 case char_facet_t::char8_f: return std::locale(in, new utf8_converter<char8_t>(std::move(lc)));
130#elif defined(__cpp_char8_t)
131 case char_facet_t::char8_f: break;
132#endif
133#ifdef BOOST_LOCALE_ENABLE_CHAR16_T
134 case char_facet_t::char16_f: return std::locale(in, new std_converter<char16_t>(std::move(lc)));
135#endif
136#ifdef BOOST_LOCALE_ENABLE_CHAR32_T
137 case char_facet_t::char32_f: return std::locale(in, new std_converter<char32_t>(std::move(lc)));
138#endif
139 }
140 return in;
141 }
