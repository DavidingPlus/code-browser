7#ifndef BOOST_LOCALE_HOLD_PTR_H
8#define BOOST_LOCALE_HOLD_PTR_H
9
10#include <boost/locale/config.hpp>
11#include <boost/core/exchange.hpp>
12
13namespace boost { namespace locale {
14 /// \brief a smart pointer similar to std::unique_ptr but the
15 /// underlying object has the same constness as the pointer itself (unlike an ordinary pointer).
16 template<typename T>
17 class hold_ptr {
18 public:
19 /// Create new empty pointer
20 hold_ptr() : ptr_(nullptr) {}
21
22 /// Create a pointer that holds \a v, ownership is transferred to smart pointer
23 explicit hold_ptr(T* v) : ptr_(v) {}
24
25 /// Destroy smart pointer and the object it owns.
26 ~hold_ptr() { delete ptr_; }
27
28 // Non-copyable
29 hold_ptr(const hold_ptr&) = delete;
30 hold_ptr& operator=(const hold_ptr&) = delete;
31 // Movable
32 hold_ptr(hold_ptr&& other) noexcept : ptr_(exchange(other.ptr_, nullptr)) {}
33 hold_ptr& operator=(hold_ptr&& other) noexcept
34 {
35 swap(other);
36 return *this;
37 }
38
39 /// Get a const pointer to the object
40 T const* get() const { return ptr_; }
41 /// Get a mutable pointer to the object
42 T* get() { return ptr_; }
43
44 /// Explicitly convertible to bool. Returns: get() != nullptr
45 explicit operator bool() const { return ptr_ != nullptr; }
46
47 /// Get a const reference to the object
48 T const& operator*() const { return *ptr_; }
49 /// Get a mutable reference to the object
50 T& operator*() { return *ptr_; }
51
52 /// Get a const pointer to the object
53 T const* operator->() const { return ptr_; }
54 /// Get a mutable pointer to the object
55 T* operator->() { return ptr_; }
56
57 /// Transfer ownership of the pointer to user
58 T* release() { return exchange(ptr_, nullptr); }
59
60 /// Set new value to pointer, previous object is destroyed, ownership of new object is transferred
61 void reset(T* p = nullptr)
62 {
63 if(ptr_)
64 delete ptr_;
65 ptr_ = p;
66 }
67
68 /// Swap two pointers
69 void swap(hold_ptr& other) noexcept { ptr_ = exchange(other.ptr_, ptr_); }
70
71 private:
72 T* ptr_;
73 };
74
75}} // namespace boost::locale
76
77#endif
