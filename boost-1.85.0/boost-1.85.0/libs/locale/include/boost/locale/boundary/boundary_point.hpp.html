15 /// \addtogroup boundary
16 /// @{
17
18 /// \brief This class represents a boundary point in the text.
19 ///
20 /// It represents a pair - an iterator and a rule that defines this
21 /// point.
22 ///
23 /// This type of object is dereferenced by the iterators of boundary_point_index. Using a rule()
24 /// member function you can get the reason why this specific boundary point was selected.
25 ///
26 /// For example, when you use sentence boundary analysis, the (rule() & \ref sentence_term) != 0 means
27 /// that this boundary point was selected because a sentence terminator (like .?!) was spotted
28 /// and the (rule() & \ref sentence_sep)!=0 means that a separator like line feed or carriage
29 /// return was observed.
30 ///
31 /// \note
32 ///
33 /// - The beginning of the analyzed range is always considered a boundary point and its rule is always 0.
34 /// - When using word boundary analysis, the returned rule relates to a chunk of text preceding
35 /// this point.
36 ///
37 /// \see
38 ///
39 /// - \ref boundary_point_index
40 /// - \ref segment
41 /// - \ref segment_index
42 ///
43 template<typename IteratorType>
44 class boundary_point {
45 public:
46 /// The type of the base iterator that iterates the original text
47 typedef IteratorType iterator_type;
48
49 /// Empty default constructor
50 boundary_point() : rule_(0) {}
51
52 /// Create a new boundary_point using iterator \p and a rule \a r
53 boundary_point(iterator_type p, rule_type r) : iterator_(p), rule_(r) {}
54
55 /// Set an new iterator value \a i
56 void iterator(iterator_type i) { iterator_ = i; }
57 /// Fetch an iterator
58 iterator_type iterator() const { return iterator_; }
59
60 /// Set an new rule value \a r
61 void rule(rule_type r) { rule_ = r; }
62 /// Fetch a rule
63 rule_type rule() const { return rule_; }
64
65 /// Check if two boundary points are the same
66 bool operator==(const boundary_point& other) const
67 {
68 return iterator_ == other.iterator_ && rule_ = other.rule_;
69 }
70 /// Check if two boundary points are different
71 bool operator!=(const boundary_point& other) const { return !(*this == other); }
72
73 /// Check if the boundary point points to same location as an iterator \a other
74 bool operator==(const iterator_type& other) const { return iterator_ == other; }
75 /// Check if the boundary point points to different location from an iterator \a other
76 bool operator!=(const iterator_type& other) const { return iterator_ != other; }
77
78 /// Automatic cast to the iterator it represents
79 operator iterator_type() const { return iterator_; }
80
81 private:
82 iterator_type iterator_;
83 rule_type rule_;
84 };
85
86 /// Check if the boundary point \a r points to same location as an iterator \a l
87 template<typename BaseIterator>
88 bool operator==(const BaseIterator& l, const boundary_point<BaseIterator>& r)
89 {
90 return r == l;
91 }
92 /// Check if the boundary point \a r points to different location from an iterator \a l
93 template<typename BaseIterator>
94 bool operator!=(const BaseIterator& l, const boundary_point<BaseIterator>& r)
95 {
96 return r != l;
97 }
98
99 /// @}
