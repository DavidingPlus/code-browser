11// Type encoding:
12//
13// bit 0: callable builtin
14// bit 1: non member
15// bit 2: naked function
16// bit 3: pointer
17// bit 4: reference
18// bit 5: member pointer
19// bit 6: member function pointer
20// bit 7: member object pointer
21
22#define BOOST_FT_type_mask 0x000000ff // 1111 1111
23#define BOOST_FT_callable_builtin 0x00000001 // 0000 0001
24#define BOOST_FT_non_member 0x00000002 // 0000 0010
25#define BOOST_FT_function 0x00000007 // 0000 0111
26#define BOOST_FT_pointer 0x0000000b // 0000 1011
27#define BOOST_FT_reference 0x00000013 // 0001 0011
28#define BOOST_FT_non_member_callable_builtin 0x00000003 // 0000 0011
29#define BOOST_FT_member_pointer 0x00000020 // 0010 0000
30#define BOOST_FT_member_function_pointer 0x00000061 // 0110 0001
31#define BOOST_FT_member_object_pointer 0x000000a3 // 1010 0001
32#define BOOST_FT_member_object_pointer_flags 0x000002a3
33
34#define BOOST_FT_variadic 0x00000100
35#define BOOST_FT_non_variadic 0x00000200
36#define BOOST_FT_variadic_mask 0x00000300
37
38#define BOOST_FT_const 0x00000400
39#define BOOST_FT_volatile 0x00000800
40
41#define BOOST_FT_default_cc 0x00008000
42#define BOOST_FT_cc_mask 0x00ff8000
43
44#define BOOST_FT_kind_mask 0x000000fc
45
46#define BOOST_FT_flags_mask 0x00000fff
47#define BOOST_FT_full_mask 0x00ff0fff
48
49#define BOOST_FT_arity_shift 24
50#define BOOST_FT_arity_mask 0x7f000000
