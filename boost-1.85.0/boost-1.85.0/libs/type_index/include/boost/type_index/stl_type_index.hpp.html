12/// \file stl_type_index.hpp
13/// \brief Contains boost::typeindex::stl_type_index class.
14///
15/// boost::typeindex::stl_type_index class can be used as a drop-in replacement
16/// for std::type_index.
17///
18/// It is used in situations when RTTI is enabled or typeid() method is available.
19/// When typeid() is disabled or BOOST_TYPE_INDEX_FORCE_NO_RTTI_COMPATIBILITY macro

42/// \class stl_type_index
43/// This class is a wrapper around std::type_info, that workarounds issues and provides
44/// much more rich interface. \b For \b description \b of \b functions \b see type_index_facade.
45///
46/// This class requires typeid() to work. For cases when RTTI is disabled see ctti_type_index.
47class stl_type_index
48 : public type_index_facade<
49 stl_type_index,
50 #ifdef BOOST_NO_STD_TYPEINFO
51 type_info
52 #else
53 std::type_info
54 #endif
55 >
56{
57public:
58#ifdef BOOST_NO_STD_TYPEINFO
59 typedef type_info type_info_t;
60#else
61 typedef std::type_info type_info_t;
62#endif
63
64private:
65 const type_info_t* data_;
66
67public:
68 inline stl_type_index() noexcept
69 : data_(&typeid(void))
70 {}
71
72 inline stl_type_index(const type_info_t& data) noexcept
73 : data_(&data)
74 {}
75
76 inline const type_info_t& type_info() const noexcept;
77
78 inline const char* raw_name() const noexcept;
79 inline const char* name() const noexcept;
80 inline std::string pretty_name() const;
81
82 inline std::size_t hash_code() const noexcept;
83 inline bool equal(const stl_type_index& rhs) const noexcept;
84 inline bool before(const stl_type_index& rhs) const noexcept;

90 inline static stl_type_index type_id_with_cvr() noexcept;
91
92 template <class T>
93 inline static stl_type_index type_id_runtime(const T& value) noexcept;
94};

113inline std::string stl_type_index::pretty_name() const {
114 static const char cvr_saver_name[] = "boost::typeindex::detail::cvr_saver<";
115 static BOOST_CONSTEXPR_OR_CONST std::string::size_type cvr_saver_name_len = sizeof(cvr_saver_name) - 1;
116
117 // In case of MSVC demangle() is a no-op, and name() already returns demangled name.
118 // In case of GCC and Clang (on non-Windows systems) name() returns mangled name and demangle() undecorates it.
119 const boost::core::scoped_demangled_name demangled_name(data_->name());
120
121 const char* begin = demangled_name.get();
122 if (!begin) {
123 boost::throw_exception(e: std::runtime_error("Type name demangling failed"));
124 }
125
126 const std::string::size_type len = std::strlen(s: begin);
127 const char* end = begin + len;
128
129 if (len > cvr_saver_name_len) {
130 const char* b = std::strstr(haystack: begin, needle: cvr_saver_name);
131 if (b) {
132 b += cvr_saver_name_len;
133
134 // Trim leading spaces
135 while (*b == ' ') { // the string is zero terminated, we won't exceed the buffer size
136 ++ b;
137 }
138
139 // Skip the closing angle bracket
140 const char* e = end - 1;
141 while (e > b && *e != '>') {
142 -- e;
143 }
144
145 // Trim trailing spaces
146 while (e > b && *(e - 1) == ' ') {
147 -- e;
148 }
149
150 if (b < e) {
151 // Parsing seems to have succeeded, the type name is not empty
152 begin = b;
153 end = e;
154 }
155 }
156 }
157
158 return std::string(begin, end);
159}

170// comparisons don't work, so we are using typeid(x).name() instead.
171# if (defined(__GNUC__) && (__GNUC__ < 4 || (__GNUC__ == 4 && __GNUC_MINOR__ < 5))) \

181inline bool stl_type_index::equal(const stl_type_index& rhs) const noexcept {
182#ifdef BOOST_TYPE_INDEX_CLASSINFO_COMPARE_BY_NAMES
183 return raw_name() == rhs.raw_name() || !std::strcmp(s1: raw_name(), s2: rhs.raw_name());
184#else
185 return !!(*data_ == *rhs.data_);
186#endif
187}
188
189inline bool stl_type_index::before(const stl_type_index& rhs) const noexcept {
190#ifdef BOOST_TYPE_INDEX_CLASSINFO_COMPARE_BY_NAMES
191 return raw_name() != rhs.raw_name() && std::strcmp(s1: raw_name(), s2: rhs.raw_name()) < 0;
192#else
193 return !!data_->before(*rhs.data_);
194#endif
195}
196
197#undef BOOST_TYPE_INDEX_CLASSINFO_COMPARE_BY_NAMES

200template <class T>
201inline stl_type_index stl_type_index::type_id() noexcept {
202 typedef typename std::remove_reference<T>::type no_ref_t;
203 typedef typename std::remove_cv<no_ref_t>::type no_cvr_t;
204 return typeid(no_cvr_t);
205}
206
207namespace detail {
208 template <class T> class cvr_saver{};
209}
210
211template <class T>
212inline stl_type_index stl_type_index::type_id_with_cvr() noexcept {
213 typedef typename std::conditional<
214 std::is_reference<T>::value || std::is_const<T>::value || std::is_volatile<T>::value,
215 detail::cvr_saver<T>,
216 T
217 >::type type;
218
219 return typeid(type);
220}

223template <class T>
224inline stl_type_index stl_type_index::type_id_runtime(const T& value) noexcept {
225#ifdef BOOST_NO_RTTI
226 return value.boost_type_index_type_id_runtime_();
227#else
228 return typeid(value);
229#endif
230}
