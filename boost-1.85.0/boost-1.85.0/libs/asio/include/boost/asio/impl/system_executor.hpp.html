44template <typename Blocking, typename Relationship, typename Allocator>
45template <typename Function>
46inline void
47basic_system_executor<Blocking, Relationship, Allocator>::do_execute(
48 Function&& f, execution::blocking_t::possibly_t) const
49{
50 // Obtain a non-const instance of the function.
51 detail::non_const_lvalue<Function> f2(f);
52
53#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
54 try
55 {
56#endif// !defined(BOOST_ASIO_NO_EXCEPTIONS)
57 detail::fenced_block b(detail::fenced_block::full);
58 static_cast<decay_t<Function>&&>(f2.value)();
59#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
60 }
61 catch (...)
62 {
63 std::terminate();
64 }
65#endif// !defined(BOOST_ASIO_NO_EXCEPTIONS)
66}

92template <typename Blocking, typename Relationship, typename Allocator>
93template <typename Function>
94void basic_system_executor<Blocking, Relationship, Allocator>::do_execute(
95 Function&& f, execution::blocking_t::never_t) const
96{
97 system_context& ctx = detail::global<system_context>();
98
99 // Allocate and construct an operation to wrap the function.
100 typedef detail::executor_op<decay_t<Function>, Allocator> op;
101 typename op::ptr p = { detail::addressof(allocator_),
102 op::ptr::allocate(allocator_), 0 };
103 p.p = new (p.v) op(static_cast<Function&&>(f), allocator_);
104
105 if (is_same<Relationship, execution::relationship_t::continuation_t>::value)
106 {
107 BOOST_ASIO_HANDLER_CREATION((ctx, *p.p,
108 "system_executor", &ctx, 0, "execute(blk=never,rel=cont)"));
109 }
110 else
111 {
112 BOOST_ASIO_HANDLER_CREATION((ctx, *p.p,
113 "system_executor", &ctx, 0, "execute(blk=never,rel=fork)"));
114 }
115
116 ctx.scheduler_.post_immediate_completion(op: p.p,
117 is_continuation: is_same<Relationship, execution::relationship_t::continuation_t>::value);
118 p.v = p.p = 0;
119}

137template <typename Blocking, typename Relationship, typename Allocator>
138template <typename Function, typename OtherAllocator>
139void basic_system_executor<Blocking, Relationship, Allocator>::post(
140 Function&& f, const OtherAllocator& a) const
141{
142 system_context& ctx = detail::global<system_context>();
143
144 // Allocate and construct an operation to wrap the function.
145 typedef detail::executor_op<decay_t<Function>, OtherAllocator> op;
146 typename op::ptr p = { detail::addressof(a), op::ptr::allocate(a), 0 };
147 p.p = new (p.v) op(static_cast<Function&&>(f), a);
148
149 BOOST_ASIO_HANDLER_CREATION((ctx, *p.p,
150 "system_executor", &this->context(), 0, "post"));
151
152 ctx.scheduler_.post_immediate_completion(op: p.p, is_continuation: false);
153 p.v = p.p = 0;
154}
