30std::string host_name()
31{
32 char name[1024];
33 boost::system::error_code ec;
34 if (boost::asio::detail::socket_ops::gethostname(name, namelen: sizeof(name), ec) != 0)
35 {
36 boost::asio::detail::throw_error(err: ec);
37 return std::string();
38 }
39 return std::string(name);
40}
41
42std::string host_name(boost::system::error_code& ec)
43{
44 char name[1024];
45 if (boost::asio::detail::socket_ops::gethostname(name, namelen: sizeof(name), ec) != 0)
46 return std::string();
47 return std::string(name);
48}
