30 std::basic_ostream<Elem, Traits>& os,
31 const basic_endpoint<InternetProtocol>& endpoint)
32{
33 boost::asio::ip::detail::endpoint tmp_ep(endpoint.address(), endpoint.port());
34 return os << tmp_ep.to_string().c_str();
35}
