32posix_thread::~posix_thread()
33{
34 if (!joined_)
35 ::pthread_detach(th: thread_);
36}
37
38void posix_thread::join()
39{
40 if (!joined_)
41 {
42 ::pthread_join(th: thread_, thread_return: 0);
43 joined_ = true;
44 }
45}

47std::size_t posix_thread::hardware_concurrency()
48{
49#if defined(_SC_NPROCESSORS_ONLN)
50 long result = sysconf(_SC_NPROCESSORS_ONLN);
51 if (result > 0)
52 return result;
53#endif // defined(_SC_NPROCESSORS_ONLN)
54 return 0;
55}

57void posix_thread::start_thread(func_base* arg)
58{
59 int error = ::pthread_create(newthread: &thread_, attr: 0,
60 start_routine: boost_asio_detail_posix_thread_function, arg: arg);
61 if (error != 0)
62 {
63 delete arg;
64 boost::system::error_code ec(error,
65 boost::asio::error::get_system_category());
66 boost::asio::detail::throw_error(err: ec, location: "thread");
67 }
68}
69
70void* boost_asio_detail_posix_thread_function(void* arg)
71{
72 posix_thread::auto_func_base_ptr func = {
73 .ptr: static_cast<posix_thread::func_base*>(arg) };
74 func.ptr->run();
75 return 0;
76}
