32posix_event::posix_event()
33 : state_(0)
34{
35#if (defined(__MACH__) && defined(__APPLE__)) \
36 || (defined(__ANDROID__) && (__ANDROID_API__ < 21))
37 int error = ::pthread_cond_init(&cond_, 0);
38#else // (defined(__MACH__) && defined(__APPLE__))
39 // || (defined(__ANDROID__) && (__ANDROID_API__ < 21))
40 ::pthread_condattr_t attr;
41 int error = ::pthread_condattr_init(attr: &attr);
42 if (error == 0)
43 {
44 error = ::pthread_condattr_setclock(attr: &attr, CLOCK_MONOTONIC);
45 if (error == 0)
46 error = ::pthread_cond_init(cond: &cond_, cond_attr: &attr);
47 ::pthread_condattr_destroy(attr: &attr);
48 }
49#endif // (defined(__MACH__) && defined(__APPLE__))
50 // || (defined(__ANDROID__) && (__ANDROID_API__ < 21))
51
52 boost::system::error_code ec(error,
53 boost::asio::error::get_system_category());
54 boost::asio::detail::throw_error(err: ec, location: "event");
55}
