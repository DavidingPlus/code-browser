26namespace boost {
27namespace asio {
28namespace detail {
29
30#if !defined(BOOST_ASIO_HAS_THREADS)
31typedef long atomic_count;
32inline void increment(atomic_count& a, long b) { a += b; }
33inline void decrement(atomic_count& a, long b) { a -= b; }
34inline void ref_count_up(atomic_count& a) { ++a; }
35inline bool ref_count_down(atomic_count& a) { return --a == 0; }
36#else // !defined(BOOST_ASIO_HAS_THREADS)
37typedef std::atomic<long> atomic_count;
38inline void increment(atomic_count& a, long b) { a += b; }
39inline void decrement(atomic_count& a, long b) { a -= b; }
40
41inline void ref_count_up(atomic_count& a)
42{
43 a.fetch_add(i: 1, m: std::memory_order_relaxed);
44}
45
46inline bool ref_count_down(atomic_count& a)
47{
48 if (a.fetch_sub(i: 1, m: std::memory_order_release) == 1)
49 {
50 std::atomic_thread_fence(m: std::memory_order_acquire);
51 return true;
52 }
53 return false;
54}
55#endif // !defined(BOOST_ASIO_HAS_THREADS)
56
57} // namespace detail
58} // namespace asio
59} // namespace boost
