54template <typename Executor, typename CandidateExecutor = void,
55 typename IoContext = io_context,
56 typename PolymorphicExecutor = executor, typename = void>
57class handler_work_base
58{
59public:
60 explicit handler_work_base(int, int, const Executor& ex) noexcept
61 : executor_(boost::asio::prefer(ex, execution::outstanding_work.tracked))
62 {
63 }
64
65 template <typename OtherExecutor>
66 handler_work_base(bool /*base1_owns_work*/, const Executor& ex,
67 const OtherExecutor& /*candidate*/) noexcept
68 : executor_(boost::asio::prefer(ex, execution::outstanding_work.tracked))
69 {
70 }
71
72 handler_work_base(const handler_work_base& other) noexcept
73 : executor_(other.executor_)
74 {
75 }
76
77 handler_work_base(handler_work_base&& other) noexcept
78 : executor_(static_cast<executor_type&&>(other.executor_))
79 {
80 }
81
82 bool owns_work() const noexcept
83 {
84 return true;
85 }
86
87 template <typename Function, typename Handler>
88 void dispatch(Function& function, Handler& handler)
89 {
90 boost::asio::prefer(executor_,
91 execution::allocator((get_associated_allocator)(handler))
92 ).execute(static_cast<Function&&>(function));
93 }
94
95private:
96 typedef decay_t<
97 prefer_result_t<Executor, execution::outstanding_work_t::tracked_t>
98 > executor_type;
99
100 executor_type executor_;
101};

103template <typename Executor, typename CandidateExecutor,
104 typename IoContext, typename PolymorphicExecutor>
105class handler_work_base<Executor, CandidateExecutor,
106 IoContext, PolymorphicExecutor,
107 enable_if_t<
108 !execution::is_executor<Executor>::value
109 && (!is_same<Executor, PolymorphicExecutor>::value
110 || !is_same<CandidateExecutor, void>::value)
111 >
112 >
113{
114public:
115 explicit handler_work_base(int, int, const Executor& ex) noexcept
116 : executor_(ex),
117 owns_work_(true)
118 {
119 executor_.on_work_started();
120 }
121
122 handler_work_base(bool /*base1_owns_work*/, const Executor& ex,
123 const Executor& candidate) noexcept
124 : executor_(ex),
125 owns_work_(ex != candidate)
126 {
127 if (owns_work_)
128 executor_.on_work_started();
129 }
130
131 template <typename OtherExecutor>
132 handler_work_base(bool /*base1_owns_work*/, const Executor& ex,
133 const OtherExecutor& /*candidate*/) noexcept
134 : executor_(ex),
135 owns_work_(true)
136 {
137 executor_.on_work_started();
138 }
139
140 handler_work_base(const handler_work_base& other) noexcept
141 : executor_(other.executor_),
142 owns_work_(other.owns_work_)
143 {
144 if (owns_work_)
145 executor_.on_work_started();
146 }
147
148 handler_work_base(handler_work_base&& other) noexcept
149 : executor_(static_cast<Executor&&>(other.executor_)),
150 owns_work_(other.owns_work_)
151 {
152 other.owns_work_ = false;
153 }
154
155 ~handler_work_base()
156 {
157 if (owns_work_)
158 executor_.on_work_finished();
159 }
160
161 bool owns_work() const noexcept
162 {
163 return owns_work_;
164 }
165
166 template <typename Function, typename Handler>
167 void dispatch(Function& function, Handler& handler)
168 {
169 executor_.dispatch(static_cast<Function&&>(function),
170 boost::asio::get_associated_allocator(handler));
171 }
172
173private:
174 Executor executor_;
175 bool owns_work_;
176};

179class handler_work_base<Executor, void, IoContext, PolymorphicExecutor,

182 Executor,
183 typename IoContext::executor_type
184 >::value
185 >
186 >
187{
188public:
189 explicit handler_work_base(int, int, const Executor&)

191 }

194 {
195 return false;
196 }

198 template <typename Function, typename Handler>
199 void dispatch(Function& function, Handler&)
200 {
201 // When using a native implementation, I/O completion handlers are
202 // already dispatched according to the execution context's executor's
203 // rules. We can call the function directly.
204 static_cast<Function&&>(function)();
205 }

408template <typename Handler, typename IoExecutor, typename = void>
409class handler_work :
410 handler_work_base<IoExecutor>,
411 handler_work_base<associated_executor_t<Handler, IoExecutor>, IoExecutor>
412{
413public:
414 typedef handler_work_base<IoExecutor> base1_type;
415 typedef handler_work_base<associated_executor_t<Handler, IoExecutor>,
416 IoExecutor> base2_type;
417
418 handler_work(Handler& handler, const IoExecutor& io_ex) noexcept
419 : base1_type(0, 0, io_ex),
420 base2_type(base1_type::owns_work(),
421 boost::asio::get_associated_executor(handler, io_ex), io_ex)
422 {
423 }
424
425 template <typename Function>
426 void complete(Function& function, Handler& handler)
427 {
428 if (!base1_type::owns_work() && !base2_type::owns_work())
429 {
430 // When using a native implementation, I/O completion handlers are
431 // already dispatched according to the execution context's executor's
432 // rules. We can call the function directly.
433 static_cast<Function&&>(function)();
434 }
435 else
436 {
437 base2_type::dispatch(function, handler);
438 }
439 }
440};

479template <typename Handler, typename IoExecutor>
480class immediate_handler_work
481{
482public:
483 typedef handler_work<Handler, IoExecutor> handler_work_type;
484
485 explicit immediate_handler_work(handler_work_type&& w)
486 : handler_work_(static_cast<handler_work_type&&>(w))
487 {
488 }
489
490 template <typename Function>
491 void complete(Function& function, Handler& handler, const void* io_ex)
492 {
493 typedef associated_immediate_executor_t<Handler, IoExecutor>
494 immediate_ex_type;
495
496 immediate_ex_type immediate_ex = (get_associated_immediate_executor)(
497 handler, *static_cast<const IoExecutor*>(io_ex));
498
499 (initiate_dispatch_with_executor<immediate_ex_type>(immediate_ex))(
500 static_cast<Function&&>(function));
501 }
502
503private:
504 handler_work_type handler_work_;
505};
