27namespace boost {
28namespace asio {
29namespace detail {
30
31template <typename Handler, typename IoExecutor>
32class completion_handler : public operation
33{
34public:
35 BOOST_ASIO_DEFINE_HANDLER_PTR(completion_handler);
36
37 completion_handler(Handler& h, const IoExecutor& io_ex)
38 : operation(&completion_handler::do_complete),
39 handler_(static_cast<Handler&&>(h)),
40 work_(handler_, io_ex)
41 {
42 }
43
44 static void do_complete(void* owner, operation* base,
45 const boost::system::error_code& /*ec*/,
46 std::size_t /*bytes_transferred*/)
47 {
48 // Take ownership of the handler object.
49 completion_handler* h(static_cast<completion_handler*>(base));
50 ptr p = { boost::asio::detail::addressof(h->handler_), h, h };
51
52 BOOST_ASIO_HANDLER_COMPLETION((*h));
53
54 // Take ownership of the operation's outstanding work.
55 handler_work<Handler, IoExecutor> w(
56 static_cast<handler_work<Handler, IoExecutor>&&>(
57 h->work_));
58
59 // Make a copy of the handler so that the memory can be deallocated before
60 // the upcall is made. Even if we're not about to make an upcall, a
61 // sub-object of the handler may be the true owner of the memory associated
62 // with the handler. Consequently, a local copy of the handler is required
63 // to ensure that any owning sub-object remains valid until after we have
64 // deallocated the memory here.
65 Handler handler(static_cast<Handler&&>(h->handler_));
66 p.h = boost::asio::detail::addressof(handler);
67 p.reset();
68
69 // Make the upcall if required.
70 if (owner)
71 {
72 fenced_block b(fenced_block::half);
73 BOOST_ASIO_HANDLER_INVOCATION_BEGIN(());
74 w.complete(handler, handler);
75 BOOST_ASIO_HANDLER_INVOCATION_END;
76 }
77 }
78
79private:
80 Handler handler_;
81 handler_work<Handler, IoExecutor> work_;
82};
