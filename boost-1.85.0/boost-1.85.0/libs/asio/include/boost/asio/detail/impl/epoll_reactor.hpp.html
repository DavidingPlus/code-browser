28inline void epoll_reactor::post_immediate_completion(
29 operation* op, bool is_continuation) const
30{
31 scheduler_.post_immediate_completion(op, is_continuation);

46template <typename Time_Traits>
47void epoll_reactor::schedule_timer(timer_queue<Time_Traits>& queue,
48 const typename Time_Traits::time_type& time,
49 typename timer_queue<Time_Traits>::per_timer_data& timer, wait_op* op)
50{
51 mutex::scoped_lock lock(mutex_);
52
53 if (shutdown_)
54 {
55 scheduler_.post_immediate_completion(op, is_continuation: false);
56 return;
57 }
58
59 bool earliest = queue.enqueue_timer(time, timer, op);
60 scheduler_.work_started();
61 if (earliest)
62 update_timeout();
63}

65template <typename Time_Traits>
66std::size_t epoll_reactor::cancel_timer(timer_queue<Time_Traits>& queue,
67 typename timer_queue<Time_Traits>::per_timer_data& timer,
68 std::size_t max_cancelled)
69{
70 mutex::scoped_lock lock(mutex_);
71 op_queue<operation> ops;
72 std::size_t n = queue.cancel_timer(timer, ops, max_cancelled);
73 lock.unlock();
74 scheduler_.post_deferred_completions(ops);
75 return n;
76}

78template <typename Time_Traits>
79void epoll_reactor::cancel_timer_by_key(timer_queue<Time_Traits>& queue,
80 typename timer_queue<Time_Traits>::per_timer_data* timer,
81 void* cancellation_key)
82{
83 mutex::scoped_lock lock(mutex_);
84 op_queue<operation> ops;
85 queue.cancel_timer_by_key(timer, ops, cancellation_key);
86 lock.unlock();
87 scheduler_.post_deferred_completions(ops);
88}

90template <typename Time_Traits>
91void epoll_reactor::move_timer(timer_queue<Time_Traits>& queue,
92 typename timer_queue<Time_Traits>::per_timer_data& target,
93 typename timer_queue<Time_Traits>::per_timer_data& source)
94{
95 mutex::scoped_lock lock(mutex_);
96 op_queue<operation> ops;
97 queue.cancel_timer(target, ops);
98 queue.move_timer(target, source);
99 lock.unlock();
100 scheduler_.post_deferred_completions(ops);
101}
