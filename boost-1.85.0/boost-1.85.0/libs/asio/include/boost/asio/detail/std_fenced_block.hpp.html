24namespace boost {
25namespace asio {
26namespace detail {
27
28class std_fenced_block
29 : private noncopyable
30{
31public:
32 enum half_t { half };
33 enum full_t { full };
34
35 // Constructor for a half fenced block.
36 explicit std_fenced_block(half_t)
37 {
38 }
39
40 // Constructor for a full fenced block.
41 explicit std_fenced_block(full_t)
42 {
43 std::atomic_thread_fence(m: std::memory_order_acquire);
44 }
45
46 // Destructor.
47 ~std_fenced_block()
48 {
49 std::atomic_thread_fence(m: std::memory_order_release);
50 }
51};
