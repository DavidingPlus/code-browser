74inline void* aligned_new(std::size_t align, std::size_t size)
75{
76#if defined(BOOST_ASIO_HAS_STD_ALIGNED_ALLOC)
77 align = (align < BOOST_ASIO_DEFAULT_ALIGN) ? BOOST_ASIO_DEFAULT_ALIGN : align;
78 size = (size % align == 0) ? size : size + (align - size % align);
79 void* ptr = std::aligned_alloc(align, size);
80 if (!ptr)
81 {
82 std::bad_alloc ex;
83 boost::asio::detail::throw_exception(ex);
84 }
85 return ptr;
86#elif defined(BOOST_ASIO_HAS_BOOST_ALIGN)
87 align = (align < BOOST_ASIO_DEFAULT_ALIGN) ? BOOST_ASIO_DEFAULT_ALIGN : align;
88 size = (size % align == 0) ? size : size + (align - size % align);
89 void* ptr = boost::alignment::aligned_alloc(alignment: align, size);
90 if (!ptr)
91 {
92 std::bad_alloc ex;
93 boost::asio::detail::throw_exception(e: ex);
94 }
95 return ptr;
96#elif defined(BOOST_ASIO_MSVC)
97 align = (align < BOOST_ASIO_DEFAULT_ALIGN) ? BOOST_ASIO_DEFAULT_ALIGN : align;
98 size = (size % align == 0) ? size : size + (align - size % align);
99 void* ptr = _aligned_malloc(size, align);
100 if (!ptr)
101 {
102 std::bad_alloc ex;
103 boost::asio::detail::throw_exception(ex);
104 }
105 return ptr;
106#else // defined(BOOST_ASIO_MSVC)
107 (void)align;
108 return ::operator new(size);
109#endif // defined(BOOST_ASIO_MSVC)
110}

112inline void aligned_delete(void* ptr)
113{
114#if defined(BOOST_ASIO_HAS_STD_ALIGNED_ALLOC)
115 std::free(ptr);
116#elif defined(BOOST_ASIO_HAS_BOOST_ALIGN)
117 boost::alignment::aligned_free(ptr);
118#elif defined(BOOST_ASIO_MSVC)
119 _aligned_free(ptr);
120#else // defined(BOOST_ASIO_MSVC)
121 ::operator delete(ptr);
122#endif // defined(BOOST_ASIO_MSVC)
123}
