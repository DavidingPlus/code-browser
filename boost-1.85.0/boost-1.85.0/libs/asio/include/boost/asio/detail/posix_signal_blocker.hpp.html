29namespace boost {
30namespace asio {
31namespace detail {
32
33class posix_signal_blocker
34 : private noncopyable
35{
36public:
37 // Constructor blocks all signals for the calling thread.
38 posix_signal_blocker()
39 : blocked_(false)
40 {
41 sigset_t new_mask;
42 sigfillset(set: &new_mask);
43 blocked_ = (pthread_sigmask(SIG_BLOCK, newmask: &new_mask, oldmask: &old_mask_) == 0);
44 }
45
46 // Destructor restores the previous signal mask.
47 ~posix_signal_blocker()
48 {
49 if (blocked_)
50 pthread_sigmask(SIG_SETMASK, newmask: &old_mask_, oldmask: 0);
51 }
52
53 // Block all signals for the calling thread.
54 void block()
55 {
56 if (!blocked_)
57 {
58 sigset_t new_mask;
59 sigfillset(set: &new_mask);
60 blocked_ = (pthread_sigmask(SIG_BLOCK, newmask: &new_mask, oldmask: &old_mask_) == 0);
61 }
62 }
63
64 // Restore the previous signal mask.
65 void unblock()
66 {
67 if (blocked_)
68 blocked_ = (pthread_sigmask(SIG_SETMASK, newmask: &old_mask_, oldmask: 0) != 0);
69 }
70
71private:
72 // Have signals been blocked.
73 bool blocked_;
74
75 // The previous signal mask.
76 sigset_t old_mask_;
77};
