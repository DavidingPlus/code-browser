181template <typename T, typename Buffer>
182struct is_buffer_sequence_class
183 : integral_constant<bool,
184 sizeof(buffer_sequence_begin_helper<T>(0, 0)) != 1 &&
185 sizeof(buffer_sequence_end_helper<T>(0, 0)) != 1 &&
186 sizeof(buffer_sequence_element_type_helper<T, Buffer>(0, 0)) == 1>
187{
188};
189
190template <typename T, typename Buffer>
191struct is_buffer_sequence
192 : conditional<is_class<T>::value,
193 is_buffer_sequence_class<T, Buffer>,
194 false_type>::type
195{
196};

246template <typename T>
247struct is_dynamic_buffer_class_v1
248 : integral_constant<bool,
249 sizeof(size_memfn_helper<T>(0)) != 1 &&
250 sizeof(max_size_memfn_helper<T>(0)) != 1 &&
251 sizeof(capacity_memfn_helper<T>(0)) != 1 &&
252 sizeof(data_memfn_helper<T>(0)) != 1 &&
253 sizeof(consume_memfn_helper<T>(0)) != 1 &&
254 sizeof(prepare_memfn_helper<T>(0)) != 1 &&
255 sizeof(commit_memfn_helper<T>(0)) != 1 &&
256 sizeof(const_buffers_type_typedef_helper<T>(0)) == 1 &&
257 sizeof(mutable_buffers_type_typedef_helper<T>(0)) == 1>
258{
259};
