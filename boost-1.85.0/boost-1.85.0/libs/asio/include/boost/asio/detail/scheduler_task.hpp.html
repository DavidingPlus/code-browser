22namespace boost {
23namespace asio {
24namespace detail {
25
26class scheduler_operation;
27
28// Base class for all tasks that may be run by a scheduler.
29class scheduler_task
30{
31public:
32 // Run the task once until interrupted or events are ready to be dispatched.
33 virtual void run(long usec, op_queue<scheduler_operation>& ops) = 0;
34
35 // Interrupt the task.
36 virtual void interrupt() = 0;
37
38protected:
39 // Prevent deletion through this type.
40 ~scheduler_task()
41 {
42 }
43};
44
45} // namespace detail
