32template <typename Handler, typename IoExecutor>
33class reactive_null_buffers_op : public reactor_op
34{
35public:
36 typedef Handler handler_type;
37 typedef IoExecutor io_executor_type;
38
39 BOOST_ASIO_DEFINE_HANDLER_PTR(reactive_null_buffers_op);
40
41 reactive_null_buffers_op(const boost::system::error_code& success_ec,
42 Handler& handler, const IoExecutor& io_ex)
43 : reactor_op(success_ec, &reactive_null_buffers_op::do_perform,
44 &reactive_null_buffers_op::do_complete),
45 handler_(static_cast<Handler&&>(handler)),
46 work_(handler_, io_ex)
47 {
48 }
49
50 static status do_perform(reactor_op*)
51 {
52 return done;
53 }
54
55 static void do_complete(void* owner, operation* base,
56 const boost::system::error_code& /*ec*/,
57 std::size_t /*bytes_transferred*/)
58 {
59 // Take ownership of the handler object.
60 BOOST_ASIO_ASSUME(base != 0);
61 reactive_null_buffers_op* o(static_cast<reactive_null_buffers_op*>(base));
62 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
63
64 BOOST_ASIO_HANDLER_COMPLETION((*o));
65
66 // Take ownership of the operation's outstanding work.
67 handler_work<Handler, IoExecutor> w(
68 static_cast<handler_work<Handler, IoExecutor>&&>(
69 o->work_));
70
71 // Make a copy of the handler so that the memory can be deallocated before
72 // the upcall is made. Even if we're not about to make an upcall, a
73 // sub-object of the handler may be the true owner of the memory associated
74 // with the handler. Consequently, a local copy of the handler is required
75 // to ensure that any owning sub-object remains valid until after we have
76 // deallocated the memory here.
77 detail::binder2<Handler, boost::system::error_code, std::size_t>
78 handler(o->handler_, o->ec_, o->bytes_transferred_);
79 p.h = boost::asio::detail::addressof(handler.handler_);
80 p.reset();
81
82 // Make the upcall if required.
83 if (owner)
84 {
85 fenced_block b(fenced_block::half);
86 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
87 w.complete(handler, handler.handler_);
88 BOOST_ASIO_HANDLER_INVOCATION_END;
89 }
90 }
91
92 static void do_immediate(operation* base, bool, const void* io_ex)
93 {
94 // Take ownership of the handler object.
95 BOOST_ASIO_ASSUME(base != 0);
96 reactive_null_buffers_op* o(static_cast<reactive_null_buffers_op*>(base));
97 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
98
99 BOOST_ASIO_HANDLER_COMPLETION((*o));
100
101 // Take ownership of the operation's outstanding work.
102 immediate_handler_work<Handler, IoExecutor> w(
103 static_cast<handler_work<Handler, IoExecutor>&&>(
104 o->work_));
105
106 // Make a copy of the handler so that the memory can be deallocated before
107 // the upcall is made. Even if we're not about to make an upcall, a
108 // sub-object of the handler may be the true owner of the memory associated
109 // with the handler. Consequently, a local copy of the handler is required
110 // to ensure that any owning sub-object remains valid until after we have
111 // deallocated the memory here.
112 detail::binder2<Handler, boost::system::error_code, std::size_t>
113 handler(o->handler_, o->ec_, o->bytes_transferred_);
114 p.h = boost::asio::detail::addressof(handler.handler_);
115 p.reset();
116
117 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, handler.arg2_));
118 w.complete(handler, handler.handler_, io_ex);
119 BOOST_ASIO_HANDLER_INVOCATION_END;
120 }
121
122private:
123 Handler handler_;
124 handler_work<Handler, IoExecutor> work_;
125};
