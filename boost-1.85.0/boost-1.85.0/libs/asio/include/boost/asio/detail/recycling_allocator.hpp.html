29template <typename T, typename Purpose = thread_info_base::default_tag>
30class recycling_allocator
31{
32public:
33 typedef T value_type;
34
35 template <typename U>
36 struct rebind
37 {
38 typedef recycling_allocator<U, Purpose> other;
39 };
40
41 recycling_allocator()
42 {
43 }
44
45 template <typename U>
46 recycling_allocator(const recycling_allocator<U, Purpose>&)
47 {
48 }
49
50 T* allocate(std::size_t n)
51 {
52 void* p = thread_info_base::allocate(Purpose(),
53 thread_context::top_of_thread_call_stack(),
54 sizeof(T) * n, alignof(T));
55 return static_cast<T*>(p);
56 }
57
58 void deallocate(T* p, std::size_t n)
59 {
60 thread_info_base::deallocate(Purpose(),
61 thread_context::top_of_thread_call_stack(), p, sizeof(T) * n);
62 }
63};
64
65template <typename Purpose>
66class recycling_allocator<void, Purpose>
67{
68public:
69 typedef void value_type;
70
71 template <typename U>
72 struct rebind
73 {
74 typedef recycling_allocator<U, Purpose> other;
75 };
76
77 recycling_allocator()
78 {
79 }
80
81 template <typename U>
82 recycling_allocator(const recycling_allocator<U, Purpose>&)
83 {
84 }
85};
86
87template <typename Allocator, typename Purpose>
88struct get_recycling_allocator
89{
90 typedef Allocator type;
91 static type get(const Allocator& a) { return a; }
92};
93
94template <typename T, typename Purpose>
95struct get_recycling_allocator<std::allocator<T>, Purpose>
96{
97 typedef recycling_allocator<T, Purpose> type;
98 static type get(const std::allocator<T>&) { return type(); }
99};
