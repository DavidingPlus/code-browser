23namespace boost {
24namespace asio {
25namespace detail {
26
27template <typename T>
28struct non_const_lvalue
29{
30 explicit non_const_lvalue(T& t)
31 : value(static_cast<conditional_t<
32 is_same<T, decay_t<T>>::value, decay_t<T>&, T&&>>(t))
33 {
34 }
35
36 conditional_t<is_same<T, decay_t<T>>::value, decay_t<T>&, decay_t<T>> value;
37};
