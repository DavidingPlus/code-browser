23namespace boost {
24namespace asio {
25namespace detail {
26
27class reactor_op
28 : public operation
29{
30public:
31 // The error code to be passed to the completion handler.
32 boost::system::error_code ec_;
33
34 // The operation key used for targeted cancellation.
35 void* cancellation_key_;
36
37 // The number of bytes transferred, to be passed to the completion handler.
38 std::size_t bytes_transferred_;
39
40 // Status returned by perform function. May be used to decide whether it is
41 // worth performing more operations on the descriptor immediately.
42 enum status { not_done, done, done_and_exhausted };
43
44 // Perform the operation. Returns true if it is finished.
45 status perform()
46 {
47 return perform_func_(this);
48 }
49
50protected:
51 typedef status (*perform_func_type)(reactor_op*);
52
53 reactor_op(const boost::system::error_code& success_ec,
54 perform_func_type perform_func, func_type complete_func)
55 : operation(complete_func),
56 ec_(success_ec),
57 cancellation_key_(0),
58 bytes_transferred_(0),
59 perform_func_(perform_func)
60 {
61 }
62
63private:
64 perform_func_type perform_func_;
65};
66
67} // namespace detail
68} // namespace asio
69} // namespace boost
