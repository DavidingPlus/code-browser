57struct scheduler::task_cleanup
58{
59 ~task_cleanup()
60 {
61 if (this_thread_->private_outstanding_work > 0)
62 {
63 boost::asio::detail::increment(
64 a&: scheduler_->outstanding_work_,
65 b: this_thread_->private_outstanding_work);
66 }
67 this_thread_->private_outstanding_work = 0;
68
69 // Enqueue the completed operations and reinsert the task at the end of
70 // the operation queue.
71 lock_->lock();
72 scheduler_->task_interrupted_ = true;
73 scheduler_->op_queue_.push(q&: this_thread_->private_op_queue);
74 scheduler_->op_queue_.push(h: &scheduler_->task_operation_);
75 }
76
77 scheduler* scheduler_;
78 mutex::scoped_lock* lock_;
79 thread_info* this_thread_;
80};
81
82struct scheduler::work_cleanup
83{
84 ~work_cleanup()
85 {
86 if (this_thread_->private_outstanding_work > 1)
87 {
88 boost::asio::detail::increment(
89 a&: scheduler_->outstanding_work_,
90 b: this_thread_->private_outstanding_work - 1);
91 }
92 else if (this_thread_->private_outstanding_work < 1)
93 {
94 scheduler_->work_finished();
95 }
96 this_thread_->private_outstanding_work = 0;
97
98#if defined(BOOST_ASIO_HAS_THREADS)
99 if (!this_thread_->private_op_queue.empty())
100 {
101 lock_->lock();
102 scheduler_->op_queue_.push(q&: this_thread_->private_op_queue);
103 }
104#endif // defined(BOOST_ASIO_HAS_THREADS)
105 }
106
107 scheduler* scheduler_;
108 mutex::scoped_lock* lock_;
109 thread_info* this_thread_;
110};

112scheduler::scheduler(boost::asio::execution_context& ctx,
113 int concurrency_hint, bool own_thread, get_task_func_type get_task)
114 : boost::asio::detail::execution_context_service_base<scheduler>(ctx),
115 one_thread_(concurrency_hint == 1
116 || !BOOST_ASIO_CONCURRENCY_HINT_IS_LOCKING(
117 SCHEDULER, concurrency_hint)
118 || !BOOST_ASIO_CONCURRENCY_HINT_IS_LOCKING(
119 REACTOR_IO, concurrency_hint)),
120 mutex_(BOOST_ASIO_CONCURRENCY_HINT_IS_LOCKING(
121 SCHEDULER, concurrency_hint)),
122 task_(0),
123 get_task_(get_task),
124 task_interrupted_(true),
125 outstanding_work_(0),
126 stopped_(false),
127 shutdown_(false),
128 concurrency_hint_(concurrency_hint),
129 thread_(0)
130{
131 BOOST_ASIO_HANDLER_TRACKING_INIT;
132
133 if (own_thread)
134 {
135 ++outstanding_work_;
136 boost::asio::detail::signal_blocker sb;
137 thread_ = new boost::asio::detail::thread(thread_function(this));
138 }
139}

141scheduler::~scheduler()
142{
143 if (thread_)
144 {
145 mutex::scoped_lock lock(mutex_);
146 shutdown_ = true;
147 stop_all_threads(lock);
148 lock.unlock();
149 thread_->join();
150 delete thread_;
151 }
152}

154void scheduler::shutdown()
155{
156 mutex::scoped_lock lock(mutex_);
157 shutdown_ = true;
158 if (thread_)
159 stop_all_threads(lock);
160 lock.unlock();
161
162 // Join thread to ensure task operation is returned to queue.
163 if (thread_)
164 {
165 thread_->join();
166 delete thread_;
167 thread_ = 0;
168 }
169
170 // Destroy handler objects.
171 while (!op_queue_.empty())
172 {
173 operation* o = op_queue_.front();
174 op_queue_.pop();
175 if (o != &task_operation_)
176 o->destroy();
177 }
178
179 // Reset to initial state.
180 task_ = 0;
181}

183void scheduler::init_task()
184{
185 mutex::scoped_lock lock(mutex_);

188 task_ = get_task_(this->context());
189 op_queue_.push(h: &task_operation_);
190 wake_one_thread_and_unlock(lock);
191 }

194std::size_t scheduler::run(boost::system::error_code& ec)
195{
196 ec = boost::system::error_code();
197 if (outstanding_work_ == 0)
198 {
199 stop();
200 return 0;
201 }
202
203 thread_info this_thread;
204 this_thread.private_outstanding_work = 0;
205 thread_call_stack::context ctx(this, this_thread);
206
207 mutex::scoped_lock lock(mutex_);
208
209 std::size_t n = 0;
210 for (; do_run_one(lock, this_thread, ec); lock.lock())
211 if (n != (std::numeric_limits<std::size_t>::max)())
212 ++n;
213 return n;
214}

252std::size_t scheduler::poll(boost::system::error_code& ec)
253{
254 ec = boost::system::error_code();
255 if (outstanding_work_ == 0)
256 {
257 stop();
258 return 0;
259 }
260
261 thread_info this_thread;
262 this_thread.private_outstanding_work = 0;
263 thread_call_stack::context ctx(this, this_thread);
264
265 mutex::scoped_lock lock(mutex_);
266
267#if defined(BOOST_ASIO_HAS_THREADS)
268 // We want to support nested calls to poll() and poll_one(), so any handlers
269 // that are already on a thread-private queue need to be put on to the main
270 // queue now.
271 if (one_thread_)
272 if (thread_info* outer_info = static_cast<thread_info*>(ctx.next_by_key()))
273 op_queue_.push(q&: outer_info->private_op_queue);
274#endif // defined(BOOST_ASIO_HAS_THREADS)
275
276 std::size_t n = 0;
277 for (; do_poll_one(lock, this_thread, ec); lock.lock())
278 if (n != (std::numeric_limits<std::size_t>::max)())
279 ++n;
280 return n;
281}

361#endif // defined(BOOST_ASIO_HAS_THREADS)

364 mutex::scoped_lock lock(mutex_);
365 op_queue_.push(h: op);

449std::size_t scheduler::do_run_one(mutex::scoped_lock& lock,
450 scheduler::thread_info& this_thread,
451 const boost::system::error_code& ec)
452{
453 while (!stopped_)
454 {
455 if (!op_queue_.empty())
456 {
457 // Prepare to execute first handler from queue.
458 operation* o = op_queue_.front();
459 op_queue_.pop();
460 bool more_handlers = (!op_queue_.empty());
461
462 if (o == &task_operation_)
463 {
464 task_interrupted_ = more_handlers;
465
466 if (more_handlers && !one_thread_)
467 wakeup_event_.unlock_and_signal_one(lock);
468 else
469 lock.unlock();
470
471 task_cleanup on_exit = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
472 (void)on_exit;
473
474 // Run the task. May throw an exception. Only block if the operation
475 // queue is empty and we're not polling, otherwise we want to return
476 // as soon as possible.
477 task_->run(usec: more_handlers ? 0 : -1, ops&: this_thread.private_op_queue);
478 }
479 else
480 {
481 std::size_t task_result = o->task_result_;
482
483 if (more_handlers && !one_thread_)
484 wake_one_thread_and_unlock(lock);
485 else
486 lock.unlock();
487
488 // Ensure the count of outstanding work is decremented on block exit.
489 work_cleanup on_exit = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
490 (void)on_exit;
491
492 // Complete the operation. May throw an exception. Deletes the object.
493 o->complete(owner: this, ec, bytes_transferred: task_result);
494 this_thread.rethrow_pending_exception();
495
496 return 1;
497 }
498 }
499 else
500 {
501 wakeup_event_.clear(lock);
502 wakeup_event_.wait(lock);
503 }
504 }
505
506 return 0;
507}

509std::size_t scheduler::do_wait_one(mutex::scoped_lock& lock,
510 scheduler::thread_info& this_thread, long usec,
511 const boost::system::error_code& ec)
512{
513 if (stopped_)
514 return 0;
515
516 operation* o = op_queue_.front();
517 if (o == 0)
518 {
519 wakeup_event_.clear(lock);
520 wakeup_event_.wait_for_usec(lock, usec);
521 usec = 0; // Wait at most once.
522 o = op_queue_.front();
523 }
524
525 if (o == &task_operation_)
526 {
527 op_queue_.pop();
528 bool more_handlers = (!op_queue_.empty());
529
530 task_interrupted_ = more_handlers;
531
532 if (more_handlers && !one_thread_)
533 wakeup_event_.unlock_and_signal_one(lock);
534 else
535 lock.unlock();
536
537 {
538 task_cleanup on_exit = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
539 (void)on_exit;
540
541 // Run the task. May throw an exception. Only block if the operation
542 // queue is empty and we're not polling, otherwise we want to return
543 // as soon as possible.
544 task_->run(usec: more_handlers ? 0 : usec, ops&: this_thread.private_op_queue);
545 }
546
547 o = op_queue_.front();
548 if (o == &task_operation_)
549 {
550 if (!one_thread_)
551 wakeup_event_.maybe_unlock_and_signal_one(lock);
552 return 0;
553 }
554 }
555
556 if (o == 0)
557 return 0;
558
559 op_queue_.pop();
560 bool more_handlers = (!op_queue_.empty());
561
562 std::size_t task_result = o->task_result_;
563
564 if (more_handlers && !one_thread_)
565 wake_one_thread_and_unlock(lock);
566 else
567 lock.unlock();
568
569 // Ensure the count of outstanding work is decremented on block exit.
570 work_cleanup on_exit = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
571 (void)on_exit;
572
573 // Complete the operation. May throw an exception. Deletes the object.
574 o->complete(owner: this, ec, bytes_transferred: task_result);
575 this_thread.rethrow_pending_exception();
576
577 return 1;
578}

580std::size_t scheduler::do_poll_one(mutex::scoped_lock& lock,
581 scheduler::thread_info& this_thread,
582 const boost::system::error_code& ec)
583{
584 if (stopped_)
585 return 0;
586
587 operation* o = op_queue_.front();
588 if (o == &task_operation_)
589 {
590 op_queue_.pop();
591 lock.unlock();
592
593 {
594 task_cleanup c = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
595 (void)c;
596
597 // Run the task. May throw an exception. Only block if the operation
598 // queue is empty and we're not polling, otherwise we want to return
599 // as soon as possible.
600 task_->run(usec: 0, ops&: this_thread.private_op_queue);
601 }
602
603 o = op_queue_.front();
604 if (o == &task_operation_)
605 {
606 wakeup_event_.maybe_unlock_and_signal_one(lock);
607 return 0;
608 }
609 }
610
611 if (o == 0)
612 return 0;
613
614 op_queue_.pop();
615 bool more_handlers = (!op_queue_.empty());
616
617 std::size_t task_result = o->task_result_;
618
619 if (more_handlers && !one_thread_)
620 wake_one_thread_and_unlock(lock);
621 else
622 lock.unlock();
623
624 // Ensure the count of outstanding work is decremented on block exit.
625 work_cleanup on_exit = { .scheduler_: this, .lock_: &lock, .this_thread_: &this_thread };
626 (void)on_exit;
627
628 // Complete the operation. May throw an exception. Deletes the object.
629 o->complete(owner: this, ec, bytes_transferred: task_result);
630 this_thread.rethrow_pending_exception();
631
632 return 1;
633}

635void scheduler::stop_all_threads(
636 mutex::scoped_lock& lock)
637{
638 stopped_ = true;
639 wakeup_event_.signal_all(lock);
640
641 if (!task_interrupted_ && task_)
642 {
643 task_interrupted_ = true;
644 task_->interrupt();
645 }
646}
647
648void scheduler::wake_one_thread_and_unlock(
649 mutex::scoped_lock& lock)
650{
651 if (!wakeup_event_.maybe_unlock_and_signal_one(lock))
652 {
653 if (!task_interrupted_ && task_)
654 {
655 task_interrupted_ = true;
656 task_->interrupt();
657 }
658 lock.unlock();
659 }
