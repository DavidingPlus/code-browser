37struct is_work_dispatcher_required : true_type

41template <typename Handler, typename Executor>
42struct is_work_dispatcher_required<Handler, Executor,
43 enable_if_t<
44 is_same<
45 typename associated_executor<Handler,
46 Executor>::asio_associated_executor_is_unspecialised,
47 void
48 >::value
49 >> : false_type
50{
51};

53template <typename Handler, typename Executor, typename = void>
54class work_dispatcher
55{
56public:
57 template <typename CompletionHandler>
58 work_dispatcher(CompletionHandler&& handler,
59 const Executor& handler_ex)
60 : handler_(static_cast<CompletionHandler&&>(handler)),
61 executor_(boost::asio::prefer(handler_ex,
62 execution::outstanding_work.tracked))
63 {
64 }
65
66 work_dispatcher(const work_dispatcher& other)
67 : handler_(other.handler_),
68 executor_(other.executor_)
69 {
70 }
71
72 work_dispatcher(work_dispatcher&& other)
73 : handler_(static_cast<Handler&&>(other.handler_)),
74 executor_(static_cast<work_executor_type&&>(other.executor_))
75 {
76 }
77
78 void operator()()
79 {
80 associated_allocator_t<Handler> alloc((get_associated_allocator)(handler_));
81 boost::asio::prefer(executor_, execution::allocator(alloc)).execute(
82 boost::asio::detail::bind_handler(
83 static_cast<Handler&&>(handler_)));
84 }
85
86private:
87 typedef decay_t<
88 prefer_result_t<const Executor&,
89 execution::outstanding_work_t::tracked_t
90 >
91 > work_executor_type;
92
93 Handler handler_;
94 work_executor_type executor_;
95};

97#if !defined(BOOST_ASIO_NO_TS_EXECUTORS)
98
99template <typename Handler, typename Executor>
100class work_dispatcher<Handler, Executor,
101 enable_if_t<!execution::is_executor<Executor>::value>>
102{
103public:
104 template <typename CompletionHandler>
105 work_dispatcher(CompletionHandler&& handler, const Executor& handler_ex)
106 : work_(handler_ex),
107 handler_(static_cast<CompletionHandler&&>(handler))
108 {
109 }
110
111 work_dispatcher(const work_dispatcher& other)
112 : work_(other.work_),
113 handler_(other.handler_)
114 {
115 }
116
117 work_dispatcher(work_dispatcher&& other)
118 : work_(static_cast<executor_work_guard<Executor>&&>(other.work_)),
119 handler_(static_cast<Handler&&>(other.handler_))
120 {
121 }
122
123 void operator()()
124 {
125 associated_allocator_t<Handler> alloc((get_associated_allocator)(handler_));
126 work_.get_executor().dispatch(
127 boost::asio::detail::bind_handler(
128 static_cast<Handler&&>(handler_)), alloc);
129 work_.reset();
130 }
131
132private:
133 executor_work_guard<Executor> work_;
134 Handler handler_;
135};
