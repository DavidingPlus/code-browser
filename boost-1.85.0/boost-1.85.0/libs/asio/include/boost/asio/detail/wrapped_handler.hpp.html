27struct is_continuation_delegated
28{
29 template <typename Dispatcher, typename Handler>
30 bool operator()(Dispatcher&, Handler& handler) const
31 {
32 return boost_asio_handler_cont_helpers::is_continuation(handler);
33 }
34};
35
36struct is_continuation_if_running
37{
38 template <typename Dispatcher, typename Handler>
39 bool operator()(Dispatcher& dispatcher, Handler&) const
40 {
41 return dispatcher.running_in_this_thread();
42 }
43};
44
45template <typename Dispatcher, typename Handler,
46 typename IsContinuation = is_continuation_delegated>
47class wrapped_handler
48{
49public:
50 typedef void result_type;
51
52 wrapped_handler(Dispatcher dispatcher, Handler& handler)
53 : dispatcher_(dispatcher),
54 handler_(static_cast<Handler&&>(handler))
55 {
56 }
57
58 wrapped_handler(const wrapped_handler& other)
59 : dispatcher_(other.dispatcher_),
60 handler_(other.handler_)
61 {
62 }
63
64 wrapped_handler(wrapped_handler&& other)
65 : dispatcher_(other.dispatcher_),
66 handler_(static_cast<Handler&&>(other.handler_))
67 {
68 }
69
70 void operator()()
71 {
72 dispatcher_.dispatch(static_cast<Handler&&>(handler_));
73 }
74
75 void operator()() const
76 {
77 dispatcher_.dispatch(handler_);
78 }
79
80 template <typename Arg1>
81 void operator()(const Arg1& arg1)
82 {
83 dispatcher_.dispatch(detail::bind_handler(handler_, arg1));
84 }
85
86 template <typename Arg1>
87 void operator()(const Arg1& arg1) const
88 {
89 dispatcher_.dispatch(detail::bind_handler(handler_, arg1));
90 }
91
92 template <typename Arg1, typename Arg2>
93 void operator()(const Arg1& arg1, const Arg2& arg2)
94 {
95 dispatcher_.dispatch(detail::bind_handler(handler_, arg1, arg2));
96 }
97
98 template <typename Arg1, typename Arg2>
99 void operator()(const Arg1& arg1, const Arg2& arg2) const
100 {
101 dispatcher_.dispatch(detail::bind_handler(handler_, arg1, arg2));
102 }
103
104 template <typename Arg1, typename Arg2, typename Arg3>
105 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3)
106 {
107 dispatcher_.dispatch(detail::bind_handler(handler_, arg1, arg2, arg3));
108 }
109
110 template <typename Arg1, typename Arg2, typename Arg3>
111 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3) const
112 {
113 dispatcher_.dispatch(detail::bind_handler(handler_, arg1, arg2, arg3));
114 }
115
116 template <typename Arg1, typename Arg2, typename Arg3, typename Arg4>
117 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3,
118 const Arg4& arg4)
119 {
120 dispatcher_.dispatch(
121 detail::bind_handler(handler_, arg1, arg2, arg3, arg4));
122 }
123
124 template <typename Arg1, typename Arg2, typename Arg3, typename Arg4>
125 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3,
126 const Arg4& arg4) const
127 {
128 dispatcher_.dispatch(
129 detail::bind_handler(handler_, arg1, arg2, arg3, arg4));
130 }
131
132 template <typename Arg1, typename Arg2, typename Arg3, typename Arg4,
133 typename Arg5>
134 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3,
135 const Arg4& arg4, const Arg5& arg5)
136 {
137 dispatcher_.dispatch(
138 detail::bind_handler(handler_, arg1, arg2, arg3, arg4, arg5));
139 }
140
141 template <typename Arg1, typename Arg2, typename Arg3, typename Arg4,
142 typename Arg5>
143 void operator()(const Arg1& arg1, const Arg2& arg2, const Arg3& arg3,
144 const Arg4& arg4, const Arg5& arg5) const
145 {
146 dispatcher_.dispatch(
147 detail::bind_handler(handler_, arg1, arg2, arg3, arg4, arg5));
148 }
149
150//private:
151 Dispatcher dispatcher_;
152 Handler handler_;
153};

155template <typename Handler, typename Context>
156class rewrapped_handler
157{
158public:
159 explicit rewrapped_handler(Handler& handler, const Context& context)
160 : context_(context),
161 handler_(static_cast<Handler&&>(handler))
162 {
163 }
164
165 explicit rewrapped_handler(const Handler& handler, const Context& context)
166 : context_(context),
167 handler_(handler)
168 {
169 }
170
171 rewrapped_handler(const rewrapped_handler& other)
172 : context_(other.context_),
173 handler_(other.handler_)
174 {
175 }
176
177 rewrapped_handler(rewrapped_handler&& other)
178 : context_(static_cast<Context&&>(other.context_)),
179 handler_(static_cast<Handler&&>(other.handler_))
180 {
181 }
182
183 void operator()()
184 {
185 handler_();
186 }
187
188 void operator()() const
189 {
190 handler_();
191 }
192
193//private:
194 Context context_;
195 Handler handler_;
196};
197
198template <typename Dispatcher, typename Handler, typename IsContinuation>
199inline bool asio_handler_is_continuation(
200 wrapped_handler<Dispatcher, Handler, IsContinuation>* this_handler)
201{
202 return IsContinuation()(this_handler->dispatcher_, this_handler->handler_);
203}
204
205template <typename Dispatcher, typename Context>
206inline bool asio_handler_is_continuation(
207 rewrapped_handler<Dispatcher, Context>* this_handler)
208{
209 return boost_asio_handler_cont_helpers::is_continuation(
210 this_handler->context_);
211}
