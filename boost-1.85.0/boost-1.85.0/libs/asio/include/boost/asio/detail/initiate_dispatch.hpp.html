32class initiate_dispatch
33{
34public:
35 template <typename CompletionHandler>
36 void operator()(CompletionHandler&& handler,
37 enable_if_t<
38 execution::is_executor<
39 associated_executor_t<decay_t<CompletionHandler>>
40 >::value
41 >* = 0) const
42 {
43 associated_executor_t<decay_t<CompletionHandler>> ex(
44 (get_associated_executor)(handler));
45
46 associated_allocator_t<decay_t<CompletionHandler>> alloc(
47 (get_associated_allocator)(handler));
48
49 boost::asio::prefer(ex, execution::allocator(alloc)).execute(
50 boost::asio::detail::bind_handler(
51 static_cast<CompletionHandler&&>(handler)));
52 }
53
54 template <typename CompletionHandler>
55 void operator()(CompletionHandler&& handler,
56 enable_if_t<
57 !execution::is_executor<
58 associated_executor_t<decay_t<CompletionHandler>>
59 >::value
60 >* = 0) const
61 {
62 associated_executor_t<decay_t<CompletionHandler>> ex(
63 (get_associated_executor)(handler));
64
65 associated_allocator_t<decay_t<CompletionHandler>> alloc(
66 (get_associated_allocator)(handler));
67
68 ex.dispatch(boost::asio::detail::bind_handler(
69 static_cast<CompletionHandler&&>(handler)), alloc);
70 }
71};

73template <typename Executor>
74class initiate_dispatch_with_executor
75{
76public:
77 typedef Executor executor_type;
78
79 explicit initiate_dispatch_with_executor(const Executor& ex)
80 : ex_(ex)
81 {
82 }
83
84 executor_type get_executor() const noexcept
85 {
86 return ex_;
87 }
88
89 template <typename CompletionHandler>
90 void operator()(CompletionHandler&& handler,
91 enable_if_t<
92 execution::is_executor<
93 conditional_t<true, executor_type, CompletionHandler>
94 >::value
95 >* = 0,
96 enable_if_t<
97 !detail::is_work_dispatcher_required<
98 decay_t<CompletionHandler>,
99 Executor
100 >::value
101 >* = 0) const
102 {
103 associated_allocator_t<decay_t<CompletionHandler>> alloc(
104 (get_associated_allocator)(handler));
105
106 boost::asio::prefer(ex_, execution::allocator(alloc)).execute(
107 boost::asio::detail::bind_handler(
108 static_cast<CompletionHandler&&>(handler)));
109 }
110
111 template <typename CompletionHandler>
112 void operator()(CompletionHandler&& handler,
113 enable_if_t<
114 execution::is_executor<
115 conditional_t<true, executor_type, CompletionHandler>
116 >::value
117 >* = 0,
118 enable_if_t<
119 detail::is_work_dispatcher_required<
120 decay_t<CompletionHandler>,
121 Executor
122 >::value
123 >* = 0) const
124 {
125 typedef decay_t<CompletionHandler> handler_t;
126
127 typedef associated_executor_t<handler_t, Executor> handler_ex_t;
128 handler_ex_t handler_ex((get_associated_executor)(handler, ex_));
129
130 associated_allocator_t<handler_t> alloc(
131 (get_associated_allocator)(handler));
132
133 boost::asio::prefer(ex_, execution::allocator(alloc)).execute(
134 detail::work_dispatcher<handler_t, handler_ex_t>(
135 static_cast<CompletionHandler&&>(handler), handler_ex));
136 }
137
138 template <typename CompletionHandler>
139 void operator()(CompletionHandler&& handler,
140 enable_if_t<
141 !execution::is_executor<
142 conditional_t<true, executor_type, CompletionHandler>
143 >::value
144 >* = 0,
145 enable_if_t<
146 !detail::is_work_dispatcher_required<
147 decay_t<CompletionHandler>,
148 Executor
149 >::value
150 >* = 0) const
151 {
152 associated_allocator_t<decay_t<CompletionHandler>> alloc(
153 (get_associated_allocator)(handler));
154
155 ex_.dispatch(boost::asio::detail::bind_handler(
156 static_cast<CompletionHandler&&>(handler)), alloc);
157 }
158
159 template <typename CompletionHandler>
160 void operator()(CompletionHandler&& handler,
161 enable_if_t<
162 !execution::is_executor<
163 conditional_t<true, executor_type, CompletionHandler>
164 >::value
165 >* = 0,
166 enable_if_t<
167 detail::is_work_dispatcher_required<
168 decay_t<CompletionHandler>,
169 Executor
170 >::value
171 >* = 0) const
172 {
173 typedef decay_t<CompletionHandler> handler_t;
174
175 typedef associated_executor_t<handler_t, Executor> handler_ex_t;
176 handler_ex_t handler_ex((get_associated_executor)(handler, ex_));
177
178 associated_allocator_t<handler_t> alloc(
179 (get_associated_allocator)(handler));
180
181 ex_.dispatch(detail::work_dispatcher<handler_t, handler_ex_t>(
182 static_cast<CompletionHandler&&>(handler), handler_ex), alloc);
183 }
184
185private:
186 Executor ex_;
187};
