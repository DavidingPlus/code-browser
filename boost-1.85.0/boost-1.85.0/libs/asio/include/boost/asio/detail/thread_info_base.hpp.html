35#ifndef BOOST_ASIO_RECYCLING_ALLOCATOR_CACHE_SIZE

39class thread_info_base
40 : private noncopyable
41{
42public:
43 struct default_tag
44 {
45 enum
46 {
47 cache_size = BOOST_ASIO_RECYCLING_ALLOCATOR_CACHE_SIZE,
48 begin_mem_index = 0,
49 end_mem_index = cache_size
50 };
51 };

53 struct awaitable_frame_tag
54 {
55 enum
56 {
57 cache_size = BOOST_ASIO_RECYCLING_ALLOCATOR_CACHE_SIZE,
58 begin_mem_index = default_tag::end_mem_index,
59 end_mem_index = begin_mem_index + cache_size
60 };
61 };

83 struct parallel_group_tag
84 {
85 enum
86 {
87 cache_size = BOOST_ASIO_RECYCLING_ALLOCATOR_CACHE_SIZE,

89 end_mem_index = begin_mem_index + cache_size

93 enum { max_mem_index = parallel_group_tag::end_mem_index };
94
95 thread_info_base()
96#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
97 : has_pending_exception_(0)
98#endif // !defined(BOOST_ASIO_NO_EXCEPTIONS)
99 {
100 for (int i = 0; i < max_mem_index; ++i)
101 reusable_memory_[i] = 0;
102 }
103
104 ~thread_info_base()
105 {
106 for (int i = 0; i < max_mem_index; ++i)
107 {
108 // The following test for non-null pointers is technically redundant, but
109 // it is significantly faster when using a tight io_context::poll() loop
110 // in latency sensitive applications.
111 if (reusable_memory_[i])
112 aligned_delete(ptr: reusable_memory_[i]);
113 }
114 }

116 static void* allocate(thread_info_base* this_thread,
117 std::size_t size, std::size_t align = BOOST_ASIO_DEFAULT_ALIGN)
118 {
119 return allocate(default_tag(), this_thread, size, align);
120 }
121
122 static void deallocate(thread_info_base* this_thread,
123 void* pointer, std::size_t size)
124 {
125 deallocate(default_tag(), this_thread, pointer, size);
126 }
127
128 template <typename Purpose>
129 static void* allocate(Purpose, thread_info_base* this_thread,
130 std::size_t size, std::size_t align = BOOST_ASIO_DEFAULT_ALIGN)
131 {
132 std::size_t chunks = (size + chunk_size - 1) / chunk_size;
133
134 if (this_thread)
135 {
136 for (int mem_index = Purpose::begin_mem_index;
137 mem_index < Purpose::end_mem_index; ++mem_index)
138 {
139 if (this_thread->reusable_memory_[mem_index])
140 {
141 void* const pointer = this_thread->reusable_memory_[mem_index];
142 unsigned char* const mem = static_cast<unsigned char*>(pointer);
143 if (static_cast<std::size_t>(mem[0]) >= chunks
144 && reinterpret_cast<std::size_t>(pointer) % align == 0)
145 {
146 this_thread->reusable_memory_[mem_index] = 0;
147 mem[size] = mem[0];
148 return pointer;
149 }
150 }
151 }
152
153 for (int mem_index = Purpose::begin_mem_index;
154 mem_index < Purpose::end_mem_index; ++mem_index)
155 {
156 if (this_thread->reusable_memory_[mem_index])
157 {
158 void* const pointer = this_thread->reusable_memory_[mem_index];
159 this_thread->reusable_memory_[mem_index] = 0;
160 aligned_delete(ptr: pointer);
161 break;
162 }
163 }
164 }
165
166 void* const pointer = aligned_new(align, size: chunks * chunk_size + 1);
167 unsigned char* const mem = static_cast<unsigned char*>(pointer);
168 mem[size] = (chunks <= UCHAR_MAX) ? static_cast<unsigned char>(chunks) : 0;
169 return pointer;
170 }
171
172 template <typename Purpose>
173 static void deallocate(Purpose, thread_info_base* this_thread,
174 void* pointer, std::size_t size)
175 {
176 if (size <= chunk_size * UCHAR_MAX)
177 {
178 if (this_thread)
179 {
180 for (int mem_index = Purpose::begin_mem_index;
181 mem_index < Purpose::end_mem_index; ++mem_index)
182 {
183 if (this_thread->reusable_memory_[mem_index] == 0)
184 {
185 unsigned char* const mem = static_cast<unsigned char*>(pointer);
186 mem[0] = mem[size];
187 this_thread->reusable_memory_[mem_index] = pointer;
188 return;
189 }
190 }
191 }
192 }
193
194 aligned_delete(ptr: pointer);
195 }

197 void capture_current_exception()
198 {
199#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
200 switch (has_pending_exception_)
201 {
202 case 0:
203 has_pending_exception_ = 1;
204 pending_exception_ = std::current_exception();
205 break;
206 case 1:
207 has_pending_exception_ = 2;
208 pending_exception_ =
209 std::make_exception_ptr<multiple_exceptions>(
210 ex: multiple_exceptions(pending_exception_));
211 break;
212 default:
213 break;
214 }
215#endif // !defined(BOOST_ASIO_NO_EXCEPTIONS)
216 }
217
218 void rethrow_pending_exception()
219 {
220#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
221 if (has_pending_exception_ > 0)
222 {
223 has_pending_exception_ = 0;
224 std::exception_ptr ex(
225 static_cast<std::exception_ptr&&>(
226 pending_exception_));
227 std::rethrow_exception(ex);
228 }
229#endif // !defined(BOOST_ASIO_NO_EXCEPTIONS)
230 }
231
232private:
233#if defined(BOOST_ASIO_HAS_IO_URING)
234 enum { chunk_size = 8 };
235#else // defined(BOOST_ASIO_HAS_IO_URING)
236 enum { chunk_size = 4 };
237#endif // defined(BOOST_ASIO_HAS_IO_URING)
238 void* reusable_memory_[max_mem_index];
239
240#if !defined(BOOST_ASIO_NO_EXCEPTIONS)
241 int has_pending_exception_;
242 std::exception_ptr pending_exception_;
243#endif // !defined(BOOST_ASIO_NO_EXCEPTIONS)
244};
