34template <typename Socket, typename Protocol>
35class reactive_socket_accept_op_base : public reactor_op
36{
37public:
38 reactive_socket_accept_op_base(const boost::system::error_code& success_ec,
39 socket_type socket, socket_ops::state_type state, Socket& peer,
40 const Protocol& protocol, typename Protocol::endpoint* peer_endpoint,
41 func_type complete_func)
42 : reactor_op(success_ec,
43 &reactive_socket_accept_op_base::do_perform, complete_func),
44 socket_(socket),
45 state_(state),
46 peer_(peer),
47 protocol_(protocol),
48 peer_endpoint_(peer_endpoint),
49 addrlen_(peer_endpoint ? peer_endpoint->capacity() : 0)
50 {
51 }
52
53 static status do_perform(reactor_op* base)
54 {
55 BOOST_ASIO_ASSUME(base != 0);
56 reactive_socket_accept_op_base* o(
57 static_cast<reactive_socket_accept_op_base*>(base));
58
59 socket_type new_socket = invalid_socket;
60 status result = socket_ops::non_blocking_accept(s: o->socket_,
61 state: o->state_, addr: o->peer_endpoint_ ? o->peer_endpoint_->data() : 0,
62 addrlen: o->peer_endpoint_ ? &o->addrlen_ : 0, ec&: o->ec_, new_socket)
63 ? done : not_done;
64 o->new_socket_.reset(new_socket);
65
66 BOOST_ASIO_HANDLER_REACTOR_OPERATION((*o, "non_blocking_accept", o->ec_));
67
68 return result;
69 }
70
71 void do_assign()
72 {
73 if (new_socket_.get() != invalid_socket)
74 {
75 if (peer_endpoint_)
76 peer_endpoint_->resize(addrlen_);
77 peer_.assign(protocol_, new_socket_.get(), ec_);
78 if (!ec_)
79 new_socket_.release();
80 }
81 }
82
83private:
84 socket_type socket_;
85 socket_ops::state_type state_;
86 socket_holder new_socket_;
87 Socket& peer_;
88 Protocol protocol_;
89 typename Protocol::endpoint* peer_endpoint_;
90 std::size_t addrlen_;
91};

93template <typename Socket, typename Protocol,
94 typename Handler, typename IoExecutor>
95class reactive_socket_accept_op :
96 public reactive_socket_accept_op_base<Socket, Protocol>
97{
98public:
99 typedef Handler handler_type;
100 typedef IoExecutor io_executor_type;
101
102 BOOST_ASIO_DEFINE_HANDLER_PTR(reactive_socket_accept_op);
103
104 reactive_socket_accept_op(const boost::system::error_code& success_ec,
105 socket_type socket, socket_ops::state_type state, Socket& peer,
106 const Protocol& protocol, typename Protocol::endpoint* peer_endpoint,
107 Handler& handler, const IoExecutor& io_ex)
108 : reactive_socket_accept_op_base<Socket, Protocol>(
109 success_ec, socket, state, peer, protocol, peer_endpoint,
110 &reactive_socket_accept_op::do_complete),
111 handler_(static_cast<Handler&&>(handler)),
112 work_(handler_, io_ex)
113 {
114 }
115
116 static void do_complete(void* owner, operation* base,
117 const boost::system::error_code& /*ec*/,
118 std::size_t /*bytes_transferred*/)
119 {
120 // Take ownership of the handler object.
121 BOOST_ASIO_ASSUME(base != 0);
122 reactive_socket_accept_op* o(static_cast<reactive_socket_accept_op*>(base));
123 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
124
125 // On success, assign new connection to peer socket object.
126 if (owner)
127 o->do_assign();
128
129 BOOST_ASIO_HANDLER_COMPLETION((*o));
130
131 // Take ownership of the operation's outstanding work.
132 handler_work<Handler, IoExecutor> w(
133 static_cast<handler_work<Handler, IoExecutor>&&>(
134 o->work_));
135
136 BOOST_ASIO_ERROR_LOCATION(o->ec_);
137
138 // Make a copy of the handler so that the memory can be deallocated before
139 // the upcall is made. Even if we're not about to make an upcall, a
140 // sub-object of the handler may be the true owner of the memory associated
141 // with the handler. Consequently, a local copy of the handler is required
142 // to ensure that any owning sub-object remains valid until after we have
143 // deallocated the memory here.
144 detail::binder1<Handler, boost::system::error_code>
145 handler(o->handler_, o->ec_);
146 p.h = boost::asio::detail::addressof(handler.handler_);
147 p.reset();
148
149 // Make the upcall if required.
150 if (owner)
151 {
152 fenced_block b(fenced_block::half);
153 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
154 w.complete(handler, handler.handler_);
155 BOOST_ASIO_HANDLER_INVOCATION_END;
156 }
157 }
158
159 static void do_immediate(operation* base, bool, const void* io_ex)
160 {
161 // Take ownership of the handler object.
162 BOOST_ASIO_ASSUME(base != 0);
163 reactive_socket_accept_op* o(static_cast<reactive_socket_accept_op*>(base));
164 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
165
166 // On success, assign new connection to peer socket object.
167 o->do_assign();
168
169 BOOST_ASIO_HANDLER_COMPLETION((*o));
170
171 // Take ownership of the operation's outstanding work.
172 immediate_handler_work<Handler, IoExecutor> w(
173 static_cast<handler_work<Handler, IoExecutor>&&>(
174 o->work_));
175
176 BOOST_ASIO_ERROR_LOCATION(o->ec_);
177
178 // Make a copy of the handler so that the memory can be deallocated before
179 // the upcall is made. Even if we're not about to make an upcall, a
180 // sub-object of the handler may be the true owner of the memory associated
181 // with the handler. Consequently, a local copy of the handler is required
182 // to ensure that any owning sub-object remains valid until after we have
183 // deallocated the memory here.
184 detail::binder1<Handler, boost::system::error_code>
185 handler(o->handler_, o->ec_);
186 p.h = boost::asio::detail::addressof(handler.handler_);
187 p.reset();
188
189 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_));
190 w.complete(handler, handler.handler_, io_ex);
191 BOOST_ASIO_HANDLER_INVOCATION_END;
192 }
193
194private:
195 Handler handler_;
196 handler_work<Handler, IoExecutor> work_;
197};

199template <typename Protocol, typename PeerIoExecutor,
200 typename Handler, typename IoExecutor>
201class reactive_socket_move_accept_op :
202 private Protocol::socket::template rebind_executor<PeerIoExecutor>::other,
203 public reactive_socket_accept_op_base<
204 typename Protocol::socket::template rebind_executor<PeerIoExecutor>::other,
205 Protocol>
206{
207public:
208 typedef Handler handler_type;
209 typedef IoExecutor io_executor_type;
210
211 BOOST_ASIO_DEFINE_HANDLER_PTR(reactive_socket_move_accept_op);
212
213 reactive_socket_move_accept_op(const boost::system::error_code& success_ec,
214 const PeerIoExecutor& peer_io_ex, socket_type socket,
215 socket_ops::state_type state, const Protocol& protocol,
216 typename Protocol::endpoint* peer_endpoint, Handler& handler,
217 const IoExecutor& io_ex)
218 : peer_socket_type(peer_io_ex),
219 reactive_socket_accept_op_base<peer_socket_type, Protocol>(
220 success_ec, socket, state, *this, protocol, peer_endpoint,
221 &reactive_socket_move_accept_op::do_complete),
222 handler_(static_cast<Handler&&>(handler)),
223 work_(handler_, io_ex)
224 {
225 }

235 ptr p = { boost::asio::detail::addressof(o->handler_), o, o };
236
237 // On success, assign new connection to peer socket object.
238 if (owner)
239 o->do_assign();

261 p.reset();
262
263 // Make the upcall if required.
264 if (owner)
265 {
266 fenced_block b(fenced_block::half);
267 BOOST_ASIO_HANDLER_INVOCATION_BEGIN((handler.arg1_, "..."));
268 w.complete(handler, handler.handler_);
269 BOOST_ASIO_HANDLER_INVOCATION_END;
270 }
271 }
