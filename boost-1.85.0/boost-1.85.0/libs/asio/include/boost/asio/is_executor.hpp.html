26/// The is_executor trait detects whether a type T meets the Executor type
27/// requirements.
28/**
29 * Class template @c is_executor is a UnaryTypeTrait that is derived from @c
30 * true_type if the type @c T meets the syntactic requirements for Executor,
31 * otherwise @c false_type.
32 */
33template <typename T>
34struct is_executor
35#if defined(GENERATING_DOCUMENTATION)
36 : integral_constant<bool, automatically_determined>
37#else // defined(GENERATING_DOCUMENTATION)
38 : boost::asio::detail::is_executor<T>
39#endif // defined(GENERATING_DOCUMENTATION)
40{
41};
