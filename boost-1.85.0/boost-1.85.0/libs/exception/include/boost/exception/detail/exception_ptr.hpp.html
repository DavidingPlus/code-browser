104 operator!=( exception_ptr const & other ) const

173 template <class Exception>
174 exception_ptr
175 get_static_exception_object()
176 {
177 Exception ba;
178 exception_detail::clone_impl<Exception> c(ba);
179#ifndef BOOST_EXCEPTION_DISABLE
180 c <<
181 throw_function(BOOST_CURRENT_FUNCTION) <<
182 throw_file(__FILE__) <<
183 throw_line(__LINE__);
184#endif
185 static exception_ptr ep(shared_ptr<exception_detail::clone_base const>(new exception_detail::clone_impl<Exception>(c)));
186 return ep;
187 }

356 inline
357 exception_ptr
358 current_exception_impl()
359 {
360 exception_detail::clone_base const * e=0;
361 switch(
362 exception_detail::clone_current_exception(cloned&: e) )
363 {
364 case exception_detail::clone_current_exception_result::
365 success:
366 {
367 BOOST_ASSERT(e!=0);
368 return exception_ptr(shared_ptr<exception_detail::clone_base const>(e));
369 }
370 case exception_detail::clone_current_exception_result::
371 bad_alloc:
372 {
373 BOOST_ASSERT(!e);
374 return exception_detail::exception_ptr_static_exception_object<bad_alloc_>::e;
375 }
376 case exception_detail::clone_current_exception_result::
377 bad_exception:
378 {
379 BOOST_ASSERT(!e);
380 return exception_detail::exception_ptr_static_exception_object<bad_exception_>::e;
381 }
382 default:
383 BOOST_ASSERT(0);
384 case exception_detail::clone_current_exception_result::
385 not_supported:
386 {
387 BOOST_ASSERT(!e);
388 try
389 {
390 throw;
391 }
392 catch(
393 exception_detail::clone_base & e )
394 {
395 return exception_ptr(shared_ptr<exception_detail::clone_base const>(e.clone()));
396 }

398#ifdef BOOST_NO_CXX11_HDR_EXCEPTION

400 catch(
401 std::domain_error & e )
402 {
403 return exception_detail::current_exception_std_exception(e);

536 rethrow_exception_( exception_ptr const & p )
537 {
538 BOOST_ASSERT(p);
539#if defined( BOOST_NO_CXX11_HDR_EXCEPTION ) || defined( BOOST_NO_EXCEPTIONS )
540 p.ptr_->rethrow();

551 }
552#endif
553 }

571 inline
572 std::string
573 diagnostic_information( exception_ptr const & p, bool verbose=true )
574 {
575 if( p )
576#ifdef BOOST_NO_EXCEPTIONS
577 return "<unavailable> due to BOOST_NO_EXCEPTIONS";
578#else
579 try
580 {
581 rethrow_exception(p);
582 }
583 catch(
584 ... )
585 {
586 return current_exception_diagnostic_information(verbose);
587 }
588#endif
589 return "<empty>";
590 }

592 inline
593 std::string
594 to_string( exception_ptr const & p )
595 {
596 std::string s='\n'+diagnostic_information(p);
597 std::string padding(" ");
598 std::string r;
599 bool f=false;
600 for( std::string::const_iterator i=s.begin(),e=s.end(); i!=e; ++i )
601 {
602 if( f )
603 r+=padding;
604 char c=*i;
605 r+=c;
606 f=(c=='\n');
607 }
608 return r;
609 }
