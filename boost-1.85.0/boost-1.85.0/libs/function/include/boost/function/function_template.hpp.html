520 template<typename FunctionObj,typename Allocator>
521 void
522 assign_functor_a(FunctionObj f, function_buffer& functor, Allocator a, false_type) const
523 {
524 typedef functor_wrapper<FunctionObj,Allocator> functor_wrapper_type;
525
526 using wrapper_allocator_type = typename std::allocator_traits<Allocator>::template rebind_alloc<functor_wrapper_type>;
527 using wrapper_allocator_pointer_type = typename std::allocator_traits<wrapper_allocator_type>::pointer;
528
529 wrapper_allocator_type wrapper_allocator(a);
530 wrapper_allocator_pointer_type copy = wrapper_allocator.allocate(1);
531 std::allocator_traits<wrapper_allocator_type>::construct(wrapper_allocator, copy, functor_wrapper_type(f,a));
532
533 functor_wrapper_type* new_f = static_cast<functor_wrapper_type*>(copy);
534 functor.members.obj_ptr = new_f;
535 }

786 void assign_to_own(const function_n& f)
787 {
788 if (!f.empty()) {
789 this->vtable = f.vtable;
790 if (this->has_trivial_copy_and_destroy()) {
791 // Don't operate on storage directly since union type doesn't relax
792 // strict aliasing rules, despite of having member char type.
793# if defined(BOOST_GCC) && (BOOST_GCC >= 40700)
794# pragma GCC diagnostic push
795 // This warning is technically correct, but we don't want to pay the price for initializing
796 // just to silence a warning: https://github.com/boostorg/function/issues/27
797# pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
798# if (BOOST_GCC >= 110000)
799 // GCC 11.3, 12 emit a different warning: https://github.com/boostorg/function/issues/42
800# pragma GCC diagnostic ignored "-Wuninitialized"
801# endif
802# endif
803 std::memcpy(dest: this->functor.data, src: f.functor.data, n: sizeof(boost::detail::function::function_buffer));
804# if defined(BOOST_GCC) && (BOOST_GCC >= 40700)
805# pragma GCC diagnostic pop
806# endif
807 } else
808 get_vtable()->base.manager(f.functor, this->functor,
809 boost::detail::function::clone_functor_tag);
810 }

813 template<typename Functor>
814 void assign_to(Functor f)
815 {
816 using boost::detail::function::vtable_base;
817
818 typedef typename boost::detail::function::get_function_tag<Functor>::type tag;
819 typedef boost::detail::function::get_invoker<tag> get_invoker;
820 typedef typename get_invoker::
821 template apply<Functor, R,
822 T...>
823 handler_type;
824
825 typedef typename handler_type::invoker_type invoker_type;
826 typedef typename handler_type::manager_type manager_type;
827
828 // Note: it is extremely important that this initialization use
829 // static initialization. Otherwise, we will have a race
830 // condition here in multi-threaded code. See
831 // http://thread.gmane.org/gmane.comp.lib.boost.devel/164902/.
832 static const vtable_type stored_vtable =
833 { { &manager_type::manage }, &invoker_type::invoke };
834
835 if (stored_vtable.assign_to(std::move(f), functor)) {
836 std::size_t value = reinterpret_cast<std::size_t>(&stored_vtable.base);
837 // coverity[pointless_expression]: suppress coverity warnings on apparant if(const).
838 if (boost::has_trivial_copy_constructor<Functor>::value &&
839 boost::has_trivial_destructor<Functor>::value &&
840 boost::detail::function::function_allows_small_object_optimization<Functor>::value)
841 value |= static_cast<std::size_t>(0x01);
842 vtable = reinterpret_cast<boost::detail::function::vtable_base *>(value);
843 } else
844 vtable = 0;
845 }

847 template<typename Functor,typename Allocator>
848 void assign_to_a(Functor f,Allocator a)
849 {
850 using boost::detail::function::vtable_base;
851
852 typedef typename boost::detail::function::get_function_tag<Functor>::type tag;
853 typedef boost::detail::function::get_invoker<tag> get_invoker;
854 typedef typename get_invoker::
855 template apply_a<Functor, Allocator, R,
856 T...>
857 handler_type;
858
859 typedef typename handler_type::invoker_type invoker_type;
860 typedef typename handler_type::manager_type manager_type;
861
862 // Note: it is extremely important that this initialization use
863 // static initialization. Otherwise, we will have a race
864 // condition here in multi-threaded code. See
865 // http://thread.gmane.org/gmane.comp.lib.boost.devel/164902/.
866 static const vtable_type stored_vtable =
867 { { &manager_type::manage }, &invoker_type::invoke };
868
869 if (stored_vtable.assign_to_a(std::move(f), functor, a)) {
870 std::size_t value = reinterpret_cast<std::size_t>(&stored_vtable.base);
871 // coverity[pointless_expression]: suppress coverity warnings on apparant if(const).
872 if (boost::has_trivial_copy_constructor<Functor>::value &&
873 boost::has_trivial_destructor<Functor>::value &&
874 boost::detail::function::function_allows_small_object_optimization<Functor>::value)
875 value |= static_cast<std::size_t>(0x01);
876 vtable = reinterpret_cast<boost::detail::function::vtable_base *>(value);
877 } else
878 vtable = 0;
879 }

881 // Moves the value from the specified argument to *this. If the argument
882 // has its function object allocated on the heap, move_assign will pass
883 // its buffer to *this, and set the argument's buffer pointer to NULL.
884 void move_assign(function_n& f)
885 {
886 if (&f == this)
887 return;
888
889 BOOST_TRY {
890 if (!f.empty()) {
891 this->vtable = f.vtable;
892 if (this->has_trivial_copy_and_destroy()) {
893 // Don't operate on storage directly since union type doesn't relax
894 // strict aliasing rules, despite of having member char type.
895# if defined(BOOST_GCC) && (BOOST_GCC >= 40700)
896# pragma GCC diagnostic push
897 // This warning is technically correct, but we don't want to pay the price for initializing
898 // just to silence a warning: https://github.com/boostorg/function/issues/27
899# pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
900# if (BOOST_GCC >= 120000)
901 // GCC 12 emits a different warning: https://github.com/boostorg/function/issues/42
902# pragma GCC diagnostic ignored "-Wuninitialized"
903# endif
904# endif
905 std::memcpy(dest: this->functor.data, src: f.functor.data, n: sizeof(this->functor.data));
906# if defined(BOOST_GCC) && (BOOST_GCC >= 40700)
907# pragma GCC diagnostic pop
908# endif
909 } else
910#if defined(BOOST_GCC) && (__GNUC__ >= 11)
911# pragma GCC diagnostic push
912// False positive in GCC 11/12 for empty function objects (function_n_test.cpp:673)
913# pragma GCC diagnostic ignored "-Wmaybe-uninitialized"
914#endif
915 get_vtable()->base.manager(f.functor, this->functor,
916 boost::detail::function::move_functor_tag);
917#if defined(BOOST_GCC) && (__GNUC__ >= 11)
918# pragma GCC diagnostic pop
919#endif
920 f.vtable = 0;
921 } else {
922 clear();
923 }
924 } BOOST_CATCH (...) {
925 vtable = 0;
926 BOOST_RETHROW;
927 }
928 BOOST_CATCH_END
929 }
