20 int console_output_buffer_base::overflow(int c)
21 {
22 const auto n = static_cast<int>(pptr() - pbase());
23 int r = 0;
24
25 if(n > 0 && (r = write(p: pbase(), n)) < 0)
26 return -1;
27 if(r < n)
28 std::memmove(dest: pbase(), src: pbase() + r, n: n - r);
29 setp(pbeg: buffer_, pend: buffer_ + buffer_size);
30 pbump(n: n - r);
31 if(c != traits_type::eof())
32 sputc(c: traits_type::to_char_type(c: c));
33 return 0;
34 }

36 int console_output_buffer_base::write(const char* p, int n)
37 {
38 const char* b = p;
39 const char* e = p + n;
40 // Should be impossible unless someone messes with the pointers
41 if(n > buffer_size)
42 return -1; // LCOV_EXCL_LINE
43 wchar_t* out = wbuffer_;
44 utf::code_point c;
45 size_t decoded = 0;
46 while((c = decoder::decode(p, e)) != utf::incomplete)
47 {
48 if(c == utf::illegal)
49 c = BOOST_NOWIDE_REPLACEMENT_CHARACTER;
50 assert(out - wbuffer_ + encoder::width(c) <= static_cast<int>(wbuffer_size));
51 out = encoder::encode(u: c, out);
52 decoded = p - b;
53 }
54 std::size_t num_chars_written = 0;
55 if(!do_write(buffer: wbuffer_, num_chars_to_write: static_cast<std::size_t>(out - wbuffer_), num_chars_written))
56 return -1;
57 return static_cast<int>(decoded);
58 }

62 int console_input_buffer_base::sync()
63 {
64 wsize_ = 0;
65 was_newline_ = true;
66 pback_buffer_.clear();
67 setg(gbeg: 0, gnext: 0, gend: 0);
68 return 0;
69 }

71 int console_input_buffer_base::pbackfail(int c)
72 {
73 if(c == traits_type::eof())
74 return traits_type::eof();
75
76 if(gptr() != eback())
77 {
78 gbump(n: -1);
79 *gptr() = traits_type::to_char_type(c: c);
80 return 0;
81 }
82
83 if(pback_buffer_.empty())
84 pback_buffer_.push_back(x: traits_type::to_char_type(c: c));
85 else
86 pback_buffer_.insert(position: pback_buffer_.begin(), x: traits_type::to_char_type(c: c));
87
88 char* pFirst = pback_buffer_.data();
89 char* pLast = pFirst + pback_buffer_.size();
90 setg(gbeg: pFirst, gnext: pFirst, gend: pLast);
91
92 return 0;
93 }

95 int console_input_buffer_base::underflow()
96 {
97 if(!pback_buffer_.empty())
98 pback_buffer_.clear();
99
100 const size_t n = read();
101 setg(gbeg: buffer_, gnext: buffer_, gend: buffer_ + n);
102 if(n == 0)
103 return traits_type::eof();
104 return std::char_traits<char>::to_int_type(c: *gptr());
105 }

107 size_t console_input_buffer_base::read()
108 {
109 std::size_t read_wchars = 0;
110 if(!do_read(buffer: wbuffer_ + wsize_, num_chars_to_read: wbuffer_size - wsize_, num_chars_read&: read_wchars))
111 return 0;
112 wsize_ += read_wchars;
113 char* out = buffer_;
114 const wchar_t* cur_input_ptr = wbuffer_;
115 const wchar_t* const end_input_ptr = wbuffer_ + wsize_;
116 while(cur_input_ptr != end_input_ptr)
117 {
118 const wchar_t* const prev_input_ptr = cur_input_ptr;
119 utf::code_point c = decoder::decode(current&: cur_input_ptr, last: end_input_ptr);
120 // If incomplete restore to beginning of incomplete char to use on next buffer
121 if(c == utf::incomplete)
122 {
123 cur_input_ptr = prev_input_ptr;
124 break;
125 }
126 if(c == utf::illegal)
127 c = BOOST_NOWIDE_REPLACEMENT_CHARACTER;
128 assert(out - buffer_ + encoder::width(c) <= static_cast<int>(buffer_size));
129 // Skip \r chars as std::cin does
130 if(c != '\r')
131 out = encoder::encode(value: c, out);
132 }
133
134 wsize_ = end_input_ptr - cur_input_ptr;
135 if(wsize_ > 0)
136 std::memmove(dest: wbuffer_, src: end_input_ptr - wsize_, n: sizeof(wchar_t) * wsize_);
137
138 // A CTRL+Z at the start of the line should be treated as EOF
139 if(was_newline_ && out > buffer_ && buffer_[0] == '\x1a')
140 {
141 sync();
142 return 0;
143 }
144 was_newline_ = out == buffer_ || out[-1] == '\n';
145
146 return out - buffer_;
147 }
