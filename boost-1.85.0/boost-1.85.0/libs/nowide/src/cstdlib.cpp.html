53 char* getenv(const char* key)
54 {
55 static stackstring value;
56
57 const wshort_stackstring name(key);
58
59 static const size_t buf_size = 64;
60 wchar_t buf[buf_size];
61 std::vector<wchar_t> tmp;
62 wchar_t* ptr = buf;
63 size_t n = GetEnvironmentVariableW(name.get(), buf, buf_size);
64 if(n == 0 && GetLastError() == 203) // ERROR_ENVVAR_NOT_FOUND
65 return 0;
66 if(n >= buf_size)
67 {
68 tmp.resize(n + 1, L'\0');
69 n = GetEnvironmentVariableW(name.get(), &tmp[0], static_cast<unsigned>(tmp.size() - 1));
70 // The size may have changed
71 if(n >= tmp.size() - 1)
72 return 0;
73 ptr = &tmp[0];
74 }
75 value.convert(ptr);
76 return value.get();
77 }

79 int setenv(const char* key, const char* value, int overwrite)
80 {
81 const wshort_stackstring name(key);
82 if(!overwrite)
83 {
84 wchar_t unused[2];
85 if(GetEnvironmentVariableW(name.get(), unused, 2) != 0 || GetLastError() != 203) // ERROR_ENVVAR_NOT_FOUND
86 return 0;
87 }
88 const wstackstring wval(value);
89 if(SetEnvironmentVariableW(name.get(), wval.get()))
90 return 0;
91 return -1;
92 }
93
94 int unsetenv(const char* key)
95 {
96 const wshort_stackstring name(key);
97 if(SetEnvironmentVariableW(name.get(), 0))
98 return 0;
99 return -1;
100 }
101
102 int putenv(char* string)
103 {
104 const char* key = string;
105 const char* key_end = string;
106 while(*key_end != '=' && *key_end != '\0')
107 key_end++;
108 if(*key_end == '\0')
109 return -1;
110 const wshort_stackstring wkey(key, key_end);
111 const wstackstring wvalue(key_end + 1);
112
113 if(SetEnvironmentVariableW(wkey.get(), wvalue.get()))
114 return 0;
115 return -1;
116 }

118 int system(const char* cmd)
119 {
120 if(!cmd)
121 return _wsystem(0);
122 const wstackstring wcmd(cmd);
123 return _wsystem(wcmd.get());
124 }
