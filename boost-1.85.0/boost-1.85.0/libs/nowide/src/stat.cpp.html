25namespace boost {
26namespace nowide {
27 namespace detail {
28 int stat(const char* path, posix_stat_t* buffer, size_t buffer_size)
29 {
30 if(sizeof(*buffer) != buffer_size)
31 {
32 errno = EINVAL;
33 return EINVAL;
34 }
35 const wstackstring wpath(path);
36 return _wstat(wpath.get(), buffer);
37 }
38 int stat(const char* path, stat_t* buffer, size_t buffer_size)
39 {
40 if(sizeof(*buffer) != buffer_size)
41 {
42 errno = EINVAL;
43 return EINVAL;
44 }
45 const wstackstring wpath(path);
46 return _wstat64(wpath.get(), buffer);
47 }
48 } // namespace detail
49} // namespace nowide
50} // namespace boost
