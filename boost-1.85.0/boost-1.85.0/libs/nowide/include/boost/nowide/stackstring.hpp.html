13#include <cstring>

15namespace boost {
16namespace nowide {

18 ///
19 /// \brief A class that allows to create a temporary wide or narrow UTF strings from
20 /// wide or narrow UTF source.
21 ///
22 /// It uses a stack buffer if the string is short enough
23 /// otherwise allocates a buffer on the heap.
24 ///
25 /// Invalid UTF characters are replaced by the substitution character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
26 ///
27 /// If a NULL pointer is passed to the constructor or convert method, NULL will be returned by c_str.
28 /// Similarly a default constructed stackstring will return NULL on calling c_str.
29 ///
30 template<typename CharOut = wchar_t, typename CharIn = char, size_t BufferSize = 256>
31 class basic_stackstring
32 {
33 public:
34 /// Size of the stack buffer
35 static const size_t buffer_size = BufferSize;
36 /// Type of the output character (converted to)
37 using output_char = CharOut;
38 /// Type of the input character (converted from)
39 using input_char = CharIn;
40
41 /// Creates a NULL stackstring
42 basic_stackstring()
43 {
44 buffer_[0] = 0;
45 }
46 /// Convert the NULL terminated string input and store in internal buffer
47 /// If input is NULL, nothing will be stored
48 explicit basic_stackstring(const input_char* input)
49 {
50 convert(input);
51 }
52 /// Convert the sequence [begin, end) and store in internal buffer
53 /// If begin is NULL, nothing will be stored
54 basic_stackstring(const input_char* begin, const input_char* end)
55 {
56 convert(begin, end);
57 }
58 /// Copy construct from other
59 basic_stackstring(const basic_stackstring& other)
60 {
61 *this = other;
62 }
63 /// Copy assign from other
64 basic_stackstring& operator=(const basic_stackstring& other)
65 {
66 if(this != &other)
67 {
68 clear();
69 const size_t len = other.length();
70 if(other.uses_stack_memory())
71 data_ = buffer_;
72 else if(other.data_)
73 data_ = new output_char[len + 1];
74 else
75 {
76 data_ = nullptr;
77 return *this;
78 }
79 std::memcpy(dest: data_, src: other.data_, n: sizeof(output_char) * (len + 1));
80 }
81 return *this;
82 }
83
84 ~basic_stackstring()
85 {
86 clear();
87 }
88
89 /// Convert the NULL terminated string input and store in internal buffer
90 /// If input is NULL, the current buffer will be reset to NULL
91 output_char* convert(const input_char* input)
92 {
93 if(input)
94 return convert(input, input + utf::strlen(input));
95 clear();
96 return get();
97 }
98 /// Convert the sequence [begin, end) and store in internal buffer
99 /// If begin is NULL, the current buffer will be reset to NULL
100 output_char* convert(const input_char* begin, const input_char* end)
101 {
102 clear();
103
104 if(begin)
105 {
106 const size_t input_len = end - begin;
107 // Minimum size required: 1 output char per input char + trailing NULL
108 const size_t min_output_size = input_len + 1;
109 // If there is a chance the converted string fits on stack, try it
110 if(min_output_size <= buffer_size && utf::convert_buffer(buffer_, buffer_size, begin, end))
111 data_ = buffer_;
112 else
113 {
114 // Fallback: Allocate a buffer that is surely large enough on heap
115 // Max size: Every input char is transcoded to the output char with maximum with + trailing NULL
116 const size_t max_output_size = input_len * utf::utf_traits<output_char>::max_width + 1;
117 data_ = new output_char[max_output_size];
118 const bool success = utf::convert_buffer(data_, max_output_size, begin, end) == data_;
119 assert(success);
120 (void)success;
121 }
122 }
123 return get();
124 }
125 /// Return the converted, NULL-terminated string or NULL if no string was converted
126 output_char* get()
127 {
128 return data_;
129 }
130 /// Return the converted, NULL-terminated string or NULL if no string was converted
131 const output_char* get() const
132 {
133 return data_;
134 }
135 /// Reset the internal buffer to NULL
136 void clear()
137 {
138 if(!uses_stack_memory())
139 delete[] data_;
140 data_ = nullptr;
141 }
142 /// Swap lhs with rhs
143 friend void swap(basic_stackstring& lhs, basic_stackstring& rhs)
144 {
145 if(lhs.uses_stack_memory())
146 {
147 if(rhs.uses_stack_memory())
148 {
149 for(size_t i = 0; i < buffer_size; i++)
150 std::swap(lhs.buffer_[i], rhs.buffer_[i]);
151 } else
152 {
153 lhs.data_ = rhs.data_;
154 rhs.data_ = rhs.buffer_;
155 for(size_t i = 0; i < buffer_size; i++)
156 rhs.buffer_[i] = lhs.buffer_[i];
157 }
158 } else if(rhs.uses_stack_memory())
159 {
160 rhs.data_ = lhs.data_;
161 lhs.data_ = lhs.buffer_;
162 for(size_t i = 0; i < buffer_size; i++)
163 lhs.buffer_[i] = rhs.buffer_[i];
164 } else
165 std::swap(lhs.data_, rhs.data_);
166 }
167
168 protected:
169 /// True if the stack memory is used
170 bool uses_stack_memory() const
171 {
172 return data_ == buffer_;
173 }
174 /// Return the current length of the string excluding the NULL terminator
175 /// If NULL is stored returns NULL
176 size_t length() const
177 {
178 if(!data_)
179 return 0;
180 size_t len = 0;
181 while(data_[len])
182 len++;
183 return len;
184 }
185
186 private:
187 output_char buffer_[buffer_size];
188 output_char* data_ = nullptr;
189 }; // basic_stackstring
