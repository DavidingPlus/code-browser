18 ///
19 /// Convert wide string (UTF-16/32) in range [begin,end) to NULL terminated narrow string (UTF-8)
20 /// stored in \a output of size \a output_size (including NULL)
21 ///
22 /// If there is not enough room NULL is returned, else output is returned.
23 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
24 ///
25 inline char* narrow(char* output, size_t output_size, const wchar_t* begin, const wchar_t* end)
26 {
27 return utf::convert_buffer(buffer: output, buffer_size: output_size, source_begin: begin, source_end: end);
28 }
29 ///
30 /// Convert NULL terminated wide string (UTF-16/32) to NULL terminated narrow string (UTF-8)
31 /// stored in \a output of size \a output_size (including NULL)
32 ///
33 /// If there is not enough room NULL is returned, else output is returned.
34 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
35 ///
36 inline char* narrow(char* output, size_t output_size, const wchar_t* source)
37 {
38 return narrow(output, output_size, begin: source, end: source + utf::strlen(s: source));
39 }

41 ///
42 /// Convert narrow string (UTF-8) in range [begin,end) to NULL terminated wide string (UTF-16/32)
43 /// stored in \a output of size \a output_size (including NULL)
44 ///
45 /// If there is not enough room NULL is returned, else output is returned.
46 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
47 ///
48 inline wchar_t* widen(wchar_t* output, size_t output_size, const char* begin, const char* end)
49 {
50 return utf::convert_buffer(buffer: output, buffer_size: output_size, source_begin: begin, source_end: end);
51 }
52 ///
53 /// Convert NULL terminated narrow string (UTF-8) to NULL terminated wide string (UTF-16/32)
54 /// most output_size (including NULL)
55 ///
56 /// If there is not enough room NULL is returned, else output is returned.
57 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
58 ///
59 inline wchar_t* widen(wchar_t* output, size_t output_size, const char* source)
60 {
61 return widen(output, output_size, begin: source, end: source + utf::strlen(s: source));
62 }

64 ///
65 /// Convert wide string (UTF-16/32) to narrow string (UTF-8).
66 ///
67 /// \param s Input string
68 /// \param count Number of characters to convert
69 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
70 ///
71 template<typename T_Char, typename = detail::requires_wide_char<T_Char>>
72 inline std::string narrow(const T_Char* s, size_t count)
73 {
74 return utf::convert_string<char>(s, s + count);
75 }
76 ///
77 /// Convert wide string (UTF-16/32) to narrow string (UTF-8).
78 ///
79 /// \param s NULL terminated input string
80 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
81 ///
82 template<typename T_Char, typename = detail::requires_wide_char<T_Char>>
83 inline std::string narrow(const T_Char* s)
84 {
85 return narrow(s, utf::strlen(s));
86 }
87 ///
88 /// Convert wide string (UTF-16/32) to narrow string (UTF-8).
89 ///
90 /// \param s Input string
91 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
92 ///
93 template<typename StringOrStringView, typename = detail::requires_wide_string_container<StringOrStringView>>
94 inline std::string narrow(const StringOrStringView& s)
95 {
96 return utf::convert_string<char>(s.data(), s.data() + s.size());
97 }

99 ///
100 /// Convert narrow string (UTF-8) to wide string (UTF-16/32).

107 inline std::wstring widen(const T_Char* s, size_t count)
