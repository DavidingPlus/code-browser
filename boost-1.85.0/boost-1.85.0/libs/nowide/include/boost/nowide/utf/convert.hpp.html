21 /// Return the length of the given string in code units.
22 /// That is the number of elements of type Char until the first NULL character.
23 /// Equivalent to `std::strlen(s)` but can handle wide-strings
24 template<typename Char>
25 size_t strlen(const Char* s)
26 {
27 const Char* end = s;
28 while(*end)
29 end++;
30 return end - s;
31 }

33 /// Convert a buffer of UTF sequences in the range [source_begin, source_end)
34 /// from \a CharIn to \a CharOut to the output \a buffer of size \a buffer_size.
35 ///
36 /// \return original buffer containing the NULL terminated string or NULL
37 ///
38 /// If there is not enough room in the buffer NULL is returned, and the content of the buffer is undefined.
39 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
40 template<typename CharOut, typename CharIn>
41 CharOut*
42 convert_buffer(CharOut* buffer, size_t buffer_size, const CharIn* source_begin, const CharIn* source_end)
43 {
44 CharOut* rv = buffer;
45 if(buffer_size == 0)
46 return nullptr;
47 buffer_size--;
48 while(source_begin != source_end)
49 {
50 code_point c = utf_traits<CharIn>::decode(source_begin, source_end);
51 if(c == illegal || c == incomplete)
52 {
53 c = BOOST_NOWIDE_REPLACEMENT_CHARACTER;
54 }
55 size_t width = utf_traits<CharOut>::width(c);
56 if(buffer_size < width)
57 {
58 rv = nullptr;
59 break;
60 }
61 buffer = utf_traits<CharOut>::encode(c, buffer);
62 buffer_size -= width;
63 }
64 *buffer++ = 0;
65 return rv;
66 }

68 /// Convert the UTF sequences in range [begin, end) from \a CharIn to \a CharOut
69 /// and return it as a string
70 ///
71 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
72 /// \tparam CharOut Output character type
73 template<typename CharOut, typename CharIn>
74 std::basic_string<CharOut> convert_string(const CharIn* begin, const CharIn* end)
75 {
76 std::basic_string<CharOut> result;
77 result.reserve(end - begin);
78 using inserter_type = std::back_insert_iterator<std::basic_string<CharOut>>;
79 inserter_type inserter(result);
80 code_point c;
81 while(begin != end)
82 {
83 c = utf_traits<CharIn>::decode(begin, end);
84 if(c == illegal || c == incomplete)
85 {
86 c = BOOST_NOWIDE_REPLACEMENT_CHARACTER;
87 }
88 utf_traits<CharOut>::encode(c, inserter);
89 }
90 return result;
91 }
92
93 /// Convert the UTF sequence in the input string from \a CharIn to \a CharOut
94 /// and return it as a string
95 ///
96 /// Any illegal sequences are replaced with the replacement character, see #BOOST_NOWIDE_REPLACEMENT_CHARACTER
97 /// \tparam CharOut Output character type
98 template<typename CharOut, typename CharIn>
99 std::basic_string<CharOut> convert_string(const std::basic_string<CharIn>& s)
100 {
101 return convert_string<CharOut>(s.data(), s.data() + s.size());
102 }
