73# define BOOST_MPL_PP_AUX_TAIL_PARAM_FUNC(unused, i, op) \
74 , BOOST_PP_CAT( \
75 BOOST_PP_TUPLE_ELEM(3, 1, op) \
76 , BOOST_PP_ADD_D(1, i, BOOST_PP_INC(BOOST_PP_TUPLE_ELEM(3, 0, op))) \
77 ) BOOST_PP_TUPLE_ELEM(3, 2, op)() \
