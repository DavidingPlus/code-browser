144 template <class T>
145 bool shl_char(T ch) {
146 static_assert(sizeof(T) <= sizeof(CharT),
147 "boost::lexical_cast does not support narrowing of char types."
148 "Use boost::locale instead" );
149#ifndef BOOST_LEXICAL_CAST_ASSUME_C_LOCALE
150 std::locale loc;
151 CharT const w = BOOST_USE_FACET(std::ctype<CharT>, loc).widen(ch);
152#else
153 CharT const w = static_cast<CharT>(ch);
154#endif
155 Traits::assign(buffer[0], w);
156 finish = start + 1;
157 return true;
158 }
159#endif

161 bool shl_char_array(CharT const* str_value) noexcept {
162 start = str_value;
163 finish = start + Traits::length(str_value);
164 return true;
165 }

167 bool shl_char_array_limited(CharT const* str, std::size_t max_size) noexcept {
168 start = str;
169 finish = start;
170 const auto zero = Traits::to_char_type(0);
171 while (finish < start + max_size && zero != *finish) {
172 ++ finish;
173 }
174 return true;
175 }

177 template <class T>
178 inline bool shl_unsigned(const T n) {
179 CharT* tmp_finish = buffer + CharacterBufferSize;
180 start = lcast_put_unsigned<Traits, T, CharT>(n, tmp_finish).convert();
181 finish = tmp_finish;
182 return true;
183 }

185 template <class T>
186 inline bool shl_signed(const T n) {
187 CharT* tmp_finish = buffer + CharacterBufferSize;
188 typedef typename boost::make_unsigned<T>::type utype;
189 CharT* tmp_start = lcast_put_unsigned<Traits, utype, CharT>(lcast_to_unsigned(n), tmp_finish).convert();
190 if (n < 0) {
191 --tmp_start;
192 CharT const minus = lcast_char_constants<CharT>::minus;
193 Traits::assign(*tmp_start, minus);
194 }
195 start = tmp_start;
196 finish = tmp_finish;
197 return true;
198 }

391 template<typename InputStreamable>
392 bool shl_input_streamable(InputStreamable& input) {
393#if defined(BOOST_NO_STRINGSTREAM) || defined(BOOST_NO_STD_LOCALE)
394 // If you have compilation error at this point, than your STL library
395 // does not support such conversions. Try updating it.
396 static_assert(boost::is_same<char, CharT>::value, "");
397#endif
398
399#ifndef BOOST_NO_EXCEPTIONS
400 out_stream.exceptions(std::ios::badbit);
401 try {
402#endif
403 bool const result = !(out_stream << input).fail();
404 const auto* const p = get_rdbuf();
405 start = p->pbase();
406 finish = p->pptr();
407 return result;
408#ifndef BOOST_NO_EXCEPTIONS
409 } catch (const ::std::ios_base::failure& /*f*/) {
410 return false;
411 }
412#endif
413 }

474 template <class CharT, class Traits>
475 class to_target_stream {
476 //`[start, finish)` is the range to output by `operator >>`
477 const CharT* start;
478 const CharT* const finish;
479
480 public:
481 to_target_stream(to_target_stream&&) = delete;
482 to_target_stream(const to_target_stream&) = delete;
483 to_target_stream& operator=(to_target_stream&&) = delete;
484 to_target_stream& operator=(const to_target_stream&) = delete;
485
486 to_target_stream(const CharT* begin, const CharT* end) noexcept
487 : start(begin)
488 , finish(end)
489 {}
490
491 private:
492 template <typename Type>
493#if defined(__clang__) && (__clang_major__ > 3 || __clang_minor__ > 6)
494 __attribute__((no_sanitize("unsigned-integer-overflow")))
495#endif
496 bool shr_unsigned(Type& output) {
497 if (start == finish) return false;
498 CharT const minus = lcast_char_constants<CharT>::minus;
499 CharT const plus = lcast_char_constants<CharT>::plus;
500 bool const has_minus = Traits::eq(minus, *start);
501
502 /* We won`t use `start' any more, so no need in decrementing it after */
503 if (has_minus || Traits::eq(plus, *start)) {
504 ++start;
505 }
506
507 bool const succeed = lcast_ret_unsigned<Traits, Type, CharT>(output, start, finish).convert();
508
509 if (has_minus) {
510 output = static_cast<Type>(0u - output);
511 }
512
513 return succeed;
514 }
515
516 template <typename Type>
517#if defined(__clang__) && (__clang_major__ > 3 || __clang_minor__ > 6)
518 __attribute__((no_sanitize("unsigned-integer-overflow")))
519#endif
520 bool shr_signed(Type& output) {
521 if (start == finish) return false;
522 CharT const minus = lcast_char_constants<CharT>::minus;
523 CharT const plus = lcast_char_constants<CharT>::plus;
524 typedef typename make_unsigned<Type>::type utype;
525 utype out_tmp = 0;
526 bool const has_minus = Traits::eq(minus, *start);
527
528 /* We won`t use `start' any more, so no need in decrementing it after */
529 if (has_minus || Traits::eq(plus, *start)) {
530 ++start;
531 }
532
533 bool succeed = lcast_ret_unsigned<Traits, utype, CharT>(out_tmp, start, finish).convert();
534 if (has_minus) {
535 utype const comp_val = (static_cast<utype>(1) << std::numeric_limits<Type>::digits);
536 succeed = succeed && out_tmp<=comp_val;
537 output = static_cast<Type>(0u - out_tmp);
538 } else {
539 utype const comp_val = static_cast<utype>((std::numeric_limits<Type>::max)());
540 succeed = succeed && out_tmp<=comp_val;
541 output = static_cast<Type>(out_tmp);
542 }
543 return succeed;
544 }

546 template<typename InputStreamable>
547 bool shr_using_base_class(InputStreamable& output)
548 {
549 static_assert(
550 !boost::is_pointer<InputStreamable>::value,
551 "boost::lexical_cast can not convert to pointers"
552 );
553
554#if defined(BOOST_NO_STRINGSTREAM) || defined(BOOST_NO_STD_LOCALE)
555 static_assert(boost::is_same<char, CharT>::value,
556 "boost::lexical_cast can not convert, because your STL library does not "
557 "support such conversions. Try updating it."
558 );
559#endif
560
561#if defined(BOOST_NO_STRINGSTREAM)
562 std::istrstream stream(start, static_cast<std::istrstream::streamsize>(finish - start));
563#else
564 typedef detail::lcast::buffer_t<CharT, Traits> buffer_t;
565 buffer_t buf;
566 // Usually `istream` and `basic_istream` do not modify
567 // content of buffer; `buffer_t` assures that this is true
568 buf.setbuf(const_cast<CharT*>(start), static_cast<typename buffer_t::streamsize>(finish - start));
569#if defined(BOOST_NO_STD_LOCALE)
570 std::istream stream(&buf);
571#else
572 std::basic_istream<CharT, Traits> stream(&buf);
573#endif // BOOST_NO_STD_LOCALE
574#endif // BOOST_NO_STRINGSTREAM
575
576#ifndef BOOST_NO_EXCEPTIONS
577 stream.exceptions(std::ios::badbit);
578 try {
579#endif
580 stream.unsetf(std::ios::skipws);
581 lcast_set_precision(stream, static_cast<InputStreamable*>(0));
582
583 return (stream >> output)
584 && (stream.get() == Traits::eof());
585
586#ifndef BOOST_NO_EXCEPTIONS
587 } catch (const ::std::ios_base::failure& /*f*/) {
588 return false;
589 }
590#endif
591 }

593 template<class T>
594 inline bool shr_xchar(T& output) noexcept {
595 static_assert(sizeof(CharT) == sizeof(T),
596 "boost::lexical_cast does not support narrowing of character types."
597 "Use boost::locale instead" );
598 bool const ok = (finish - start == 1);
599 if (ok) {
600 CharT out;
601 Traits::assign(out, *start);
602 output = static_cast<T>(out);
603 }
604 return ok;
605 }
606
607 template <std::size_t N, class ArrayT>
608 bool shr_std_array(ArrayT& output) noexcept {
609 const std::size_t size = static_cast<std::size_t>(finish - start);
610 if (size > N - 1) { // `-1` because we need to store \0 at the end
611 return false;
612 }
613
614 std::memcpy(dest: &output[0], src: start, n: size * sizeof(CharT));
615 output[size] = Traits::to_char_type(0);
616 return true;
617 }

673 bool stream_out(bool& output) noexcept {
674 output = false; // Suppress warning about uninitalized variable
675
676 if (start == finish) return false;
677 CharT const zero = lcast_char_constants<CharT>::zero;
678 CharT const plus = lcast_char_constants<CharT>::plus;
679 CharT const minus = lcast_char_constants<CharT>::minus;
680
681 const CharT* const dec_finish = finish - 1;
682 output = Traits::eq(*dec_finish, zero + 1);
683 if (!output && !Traits::eq(*dec_finish, zero)) {
684 return false; // Does not ends on '0' or '1'
685 }
686
687 if (start == dec_finish) return true;
688
689 // We may have sign at the beginning
690 if (Traits::eq(plus, *start) || (Traits::eq(minus, *start) && !output)) {
691 ++ start;
692 }
693
694 // Skipping zeros
695 while (start != dec_finish) {
696 if (!Traits::eq(zero, *start)) {
697 return false; // Not a zero => error
698 }
699
700 ++ start;
701 }
702
703 return true;
704 }

706 private:
707 // Not optimised converter
708 template <class T>
709 bool float_types_converter_internal(T& output) {
710 if (parse_inf_nan(start, finish, output)) return true;
711 bool const return_value = shr_using_base_class(output);
712
713 /* Some compilers and libraries successfully
714 * parse 'inf', 'INFINITY', '1.0E', '1.0E-'...
715 * We are trying to provide a unified behaviour,
716 * so we just forbid such conversions (as some
717 * of the most popular compilers/libraries do)
718 * */
719 CharT const minus = lcast_char_constants<CharT>::minus;
720 CharT const plus = lcast_char_constants<CharT>::plus;
721 CharT const capital_e = lcast_char_constants<CharT>::capital_e;
722 CharT const lowercase_e = lcast_char_constants<CharT>::lowercase_e;
723 if ( return_value &&
724 (
725 Traits::eq(*(finish-1), lowercase_e) // 1.0e
726 || Traits::eq(*(finish-1), capital_e) // 1.0E
727 || Traits::eq(*(finish-1), minus) // 1.0e- or 1.0E-
728 || Traits::eq(*(finish-1), plus) // 1.0e+ or 1.0E+
729 )
730 ) return false;
731
732 return return_value;
733 }
