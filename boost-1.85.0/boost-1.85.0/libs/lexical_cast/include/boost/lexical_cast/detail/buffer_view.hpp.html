17namespace boost { namespace conversion { namespace detail {
18
19 template < typename CharT >
20 struct buffer_view {
21 const CharT* begin;
22 const CharT* end;
23 };
24
25 template < typename CharT >
26 buffer_view<CharT> make_buffer_view(const CharT* begin, const CharT* end) {
27 return buffer_view<CharT>{begin, end};
28 }
29
30 inline buffer_view<char> make_buffer_view(const signed char* begin, const signed char* end) {
31 return buffer_view<char>{
32 .begin: reinterpret_cast<const char*>(begin),
33 .end: reinterpret_cast<const char*>(end)
34 };
35 }
36
37 inline buffer_view<char> make_buffer_view(const unsigned char* begin, const unsigned char* end) {
38 return buffer_view<char>{
39 .begin: reinterpret_cast<const char*>(begin),
40 .end: reinterpret_cast<const char*>(end)
41 };
42 }
43
44 template< typename CharT, typename Elem, typename Traits >
45 std::basic_ostream<Elem,Traits>& operator<<(
46 std::basic_ostream<Elem, Traits>& os,
47 buffer_view<CharT> r)
48 {
49 while (r.begin != r.end) {
50 os << r.begin[0];
51 ++r.begin;
52 }
53 return os;
54 }
