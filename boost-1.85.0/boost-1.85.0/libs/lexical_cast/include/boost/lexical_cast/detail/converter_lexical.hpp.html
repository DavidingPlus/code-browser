209 namespace detail // deduce_source_char_impl<T>
210 {
211 // If type T is `deduce_character_type_later` type, then tries to deduce
212 // character type using streaming metafunctions.
213 // Otherwise supplied type T is a character type, that must be normalized
214 // using normalize_single_byte_char<Char>.
215 // Executed at Stage 2 (See deduce_source_char<T> and deduce_target_char<T>)
216 template < class Char >
217 struct deduce_source_char_impl
218 {
219 typedef typename boost::detail::normalize_single_byte_char< Char >::type type;
220 };
221
222 template < class T >
223 struct deduce_source_char_impl< deduce_character_type_later< T > >
224 {
225 template <class U>
226 static auto left_shift_type(long)
227 -> decltype( std::declval<std::basic_ostream< char >&>() << std::declval<const U&>(), char{});
228
229#if !defined(BOOST_LCAST_NO_WCHAR_T)
230 template <class U>
231 static auto left_shift_type(int)
232 -> decltype( std::declval<std::basic_ostream< wchar_t >&>() << std::declval<const U&>(), wchar_t{});
233#endif
234
235 template <class U>
236 static void left_shift_type(...);
237
238 using type = decltype(left_shift_type<T>(1L));
239
240 static_assert(!std::is_same<type, void>::value,
241#if defined(BOOST_LCAST_NO_WCHAR_T)
242 "Source type is not std::ostream`able and std::wostream`s are "
243 "not supported by your STL implementation"
244#else
245 "Source type is neither std::ostream`able nor std::wostream`able"
246#endif
247 );
248 };
249 }

251 namespace detail // deduce_target_char_impl<T>
252 {
253 // If type T is `deduce_character_type_later` type, then tries to deduce
254 // character type using boost::has_right_shift<T> metafunction.
255 // Otherwise supplied type T is a character type, that must be normalized
256 // using normalize_single_byte_char<Char>.
257 // Executed at Stage 2 (See deduce_source_char<T> and deduce_target_char<T>)
258 template < class Char >
259 struct deduce_target_char_impl
260 {
261 typedef typename normalize_single_byte_char< Char >::type type;
262 };
263
264 template < class T >
265 struct deduce_target_char_impl< deduce_character_type_later<T> >
266 {
267 template <class U>
268 static auto right_shift_type(long)
269 -> decltype( std::declval<std::basic_istream< char >&>() >> std::declval<U&>(), char{});
270
271#if !defined(BOOST_LCAST_NO_WCHAR_T)
272 template <class U>
273 static auto right_shift_type(int)
274 -> decltype( std::declval<std::basic_istream< wchar_t >&>() >> std::declval<U&>(), wchar_t{});
275#endif
276
277 template <class U>
278 static void right_shift_type(...);
279
280 using type = decltype(right_shift_type<T>(1L));
281
282 static_assert(!std::is_same<type, void>::value,
283#if defined(BOOST_LCAST_NO_WCHAR_T)
284 "Target type is not std::istream`able and std::wistream`s are "
285 "not supported by your STL implementation"
286#else
287 "Target type is neither std::istream`able nor std::wistream`able"
288#endif
289 );
290 };
291 }

293 namespace detail // deduce_target_char<T> and deduce_source_char<T>
294 {
295 // We deduce stream character types in two stages.
296 //
297 // Stage 1 is common for Target and Source. At Stage 1 we get
298 // non normalized character type (may contain unsigned/signed char)
299 // or deduce_character_type_later<T> where T is the original type.
300 // Stage 1 is executed by stream_char_common<T>
301 //
302 // At Stage 2 we normalize character types or try to deduce character
303 // type using metafunctions.
304 // Stage 2 is executed by deduce_target_char_impl<T> and
305 // deduce_source_char_impl<T>
306 //
307 // deduce_target_char<T> and deduce_source_char<T> functions combine
308 // both stages
309
310 template < class T >
311 struct deduce_target_char
312 {
313 typedef typename stream_char_common< T >::type stage1_type;
314 typedef typename deduce_target_char_impl< stage1_type >::type type;
315 };
316
317 template < class T >
318 struct deduce_source_char
319 {
320 typedef typename stream_char_common< T >::type stage1_type;
321 typedef typename deduce_source_char_impl< stage1_type >::type type;
322 };
323 }

340 namespace detail // lcast_src_length
341 {
342 // Return max. length of string representation of Source;
343 template< class Source, // Source type of lexical_cast.
344 class Enable = void // helper type
345 >
346 struct lcast_src_length
347 {
348 BOOST_STATIC_CONSTANT(std::size_t, value = 1);
349 };
350
351 // Helper for integral types.
352 // Notes on length calculation:
353 // Max length for 32bit int with grouping "\1" and thousands_sep ',':
354 // "-2,1,4,7,4,8,3,6,4,7"
355 // ^ - is_signed
356 // ^ - 1 digit not counted by digits10
357 // ^^^^^^^^^^^^^^^^^^ - digits10 * 2
358 //
359 // Constant is_specialized is used instead of constant 1
360 // to prevent buffer overflow in a rare case when
361 // <boost/limits.hpp> doesn't add missing specialization for
362 // numeric_limits<T> for some integral type T.
363 // When is_specialized is false, the whole expression is 0.
364 template <class Source>
365 struct lcast_src_length<
366 Source, typename boost::enable_if<boost::is_integral<Source> >::type
367 >
368 {
369 BOOST_STATIC_CONSTANT(std::size_t, value =
370 std::numeric_limits<Source>::is_signed +
371 std::numeric_limits<Source>::is_specialized + /* == 1 */
372 std::numeric_limits<Source>::digits10 * 2
373 );
374 };
375
376 // Helper for floating point types.
377 // -1.23456789e-123456
378 // ^ sign
379 // ^ leading digit
380 // ^ decimal point
381 // ^^^^^^^^ lcast_precision<Source>::value
382 // ^ "e"
383 // ^ exponent sign
384 // ^^^^^^ exponent (assumed 6 or less digits)
385 // sign + leading digit + decimal point + "e" + exponent sign == 5
386 template<class Source>
387 struct lcast_src_length<
388 Source, typename boost::enable_if<boost::is_float<Source> >::type
389 >
390 {
391 static_assert(
392 std::numeric_limits<Source>::max_exponent10 <= 999999L &&
393 std::numeric_limits<Source>::min_exponent10 >= -999999L
394 , "");
395
396 BOOST_STATIC_CONSTANT(std::size_t, value =
397 5 + lcast_precision<Source>::value + 6
398 );
399 };
400 }

402 namespace detail // lexical_cast_stream_traits<Source, Target>
403 {
404 template <class Source, class Target>
405 struct lexical_cast_stream_traits {
406 typedef typename boost::detail::array_to_pointer_decay<Source>::type src;
407 typedef typename boost::remove_cv<src>::type no_cv_src;
408
409 typedef boost::detail::deduce_source_char<no_cv_src> deduce_src_char_metafunc;
410 typedef typename deduce_src_char_metafunc::type src_char_t;
411 typedef typename boost::detail::deduce_target_char<Target>::type target_char_t;
412
413 typedef typename boost::detail::widest_char<
414 target_char_t, src_char_t
415 >::type char_type;
416
417#if !defined(BOOST_NO_CXX11_CHAR16_T) && defined(BOOST_NO_CXX11_UNICODE_LITERALS)
418 static_assert(!boost::is_same<char16_t, src_char_t>::value
419 && !boost::is_same<char16_t, target_char_t>::value,
420 "Your compiler does not have full support for char16_t" );
421#endif
422#if !defined(BOOST_NO_CXX11_CHAR32_T) && defined(BOOST_NO_CXX11_UNICODE_LITERALS)
423 static_assert(!boost::is_same<char32_t, src_char_t>::value
424 && !boost::is_same<char32_t, target_char_t>::value,
425 "Your compiler does not have full support for char32_t" );
426#endif
427
428 typedef std::char_traits<char_type> traits;
429
430 typedef boost::detail::lcast_src_length<no_cv_src> len_t;
431 };
432 }

434 namespace detail
435 {
436 template<typename Target, typename Source>
437 struct lexical_converter_impl
438 {
439 typedef lexical_cast_stream_traits<Source, Target> stream_trait;
440
441 typedef detail::lcast::optimized_src_stream<
442 typename stream_trait::char_type,
443 typename stream_trait::traits,
444 stream_trait::len_t::value + 1
445 > optimized_src_stream;
446
447 template <class T>
448 static auto detect_type(int)
449 -> decltype(std::declval<optimized_src_stream&>().stream_in(std::declval<lcast::exact<T>>()), optimized_src_stream{});
450
451 template <class T>
452 static lcast::ios_src_stream<typename stream_trait::char_type, typename stream_trait::traits> detect_type(...);
453
454 using from_src_stream = decltype(detect_type<Source>(1));
455
456 typedef detail::lcast::to_target_stream<
457 typename stream_trait::char_type,
458 typename stream_trait::traits
459 > to_target_stream;
460
461 static inline bool try_convert(const Source& arg, Target& result) {
462 from_src_stream src_stream;
463 if (!src_stream.stream_in(lcast::exact<Source>{arg}))
464 return false;
465
466 to_target_stream out(src_stream.cbegin(), src_stream.cend());
467 if (!out.stream_out(result))
468 return false;
469
470 return true;
471 }
472 };
