55// Translate incoming UTF-8 into UCS-4
56BOOST_UTF8_DECL std::codecvt_base::result utf8_codecvt_facet::do_in(
57 std::mbstate_t& /*state*/,
58 const char * from,
59 const char * from_end,
60 const char * & from_next,
61 wchar_t * to,
62 wchar_t * to_end,
63 wchar_t * & to_next
64) const {
65 // Basic algorithm: The first octet determines how many
66 // octets total make up the UCS-4 character. The remaining
67 // "continuing octets" all begin with "10". To convert, subtract
68 // the amount that specifies the number of octets from the first
69 // octet. Subtract 0x80 (1000 0000) from each continuing octet,
70 // then mash the whole lot together. Note that each continuing
71 // octet only uses 6 bits as unique values, so only shift by
72 // multiples of 6 to combine.
73 const wchar_t * const octet1_modifier_table = detail::get_octet1_modifier_table();
74 while (from != from_end && to != to_end) {
75
76 // Error checking on the first octet
77 if (invalid_leading_octet(octet_1: *from)) {
78 from_next = from;
79 to_next = to;
80 return std::codecvt_base::error;
81 }
82
83 // The first octet is adjusted by a value dependent upon
84 // the number of "continuing octets" encoding the character
85 const int cont_octet_count = get_cont_octet_count(lead_octet: *from);
86
87 // The unsigned char conversion is necessary in case char is
88 // signed (I learned this the hard way)
89 wchar_t ucs_result =
90 (unsigned char)(*from++) - octet1_modifier_table[cont_octet_count];
91
92 // Invariants:
93 // 1) At the start of the loop, 'i' continuing characters have been
94 // processed
95 // 2) *from points to the next continuing character to be processed.
96 int i = 0;
97 while (i != cont_octet_count && from != from_end) {
98
99 // Error checking on continuing characters
100 if (invalid_continuing_octet(octet_1: *from)) {
101 from_next = from;
102 to_next = to;
103 return std::codecvt_base::error;
104 }
105
106 ucs_result *= (1 << 6);
107
108 // each continuing character has an extra (10xxxxxx)b attached to
109 // it that must be removed.
110 ucs_result += (unsigned char)(*from++) - 0x80;
111 ++i;
112 }
113
114 // If the buffer ends with an incomplete unicode character...
115 if (from == from_end && i != cont_octet_count) {
116 // rewind "from" to before the current character translation
117 from_next = from - (i + 1);
118 to_next = to;
119 return std::codecvt_base::partial;
120 }
121 *to++ = ucs_result;
122 }
123 from_next = from;
124 to_next = to;
125
126 // Were we done converting or did we run out of destination space?
127 if (from == from_end)
128 return std::codecvt_base::ok;
129 else
130 return std::codecvt_base::partial;
131}

133BOOST_UTF8_DECL std::codecvt_base::result utf8_codecvt_facet::do_out(
134 std::mbstate_t& /*state*/,
135 const wchar_t * from,
136 const wchar_t * from_end,
137 const wchar_t * & from_next,
138 char * to,
139 char * to_end,
140 char * & to_next
141) const
142{
143 const wchar_t * const octet1_modifier_table = detail::get_octet1_modifier_table();
144 wchar_t max_wchar = (std::numeric_limits<wchar_t>::max)();
145 while (from != from_end && to != to_end) {
146
147 // Check for invalid UCS-4 character
148 if (*from > max_wchar) {
149 from_next = from;
150 to_next = to;
151 return std::codecvt_base::error;
152 }
153
154 int cont_octet_count = get_cont_octet_out_count(word: *from);
155
156 // RG - comment this formula better
157 int shift_exponent = cont_octet_count * 6;
158
159 // Process the first character
160 *to++ = static_cast<char>(octet1_modifier_table[cont_octet_count] +
161 (unsigned char)(*from / (1 << shift_exponent)));
162
163 // Process the continuation characters
164 // Invariants: At the start of the loop:
165 // 1) 'i' continuing octets have been generated
166 // 2) '*to' points to the next location to place an octet
167 // 3) shift_exponent is 6 more than needed for the next octet
168 int i = 0;
169 while (i != cont_octet_count && to != to_end) {
170 shift_exponent -= 6;
171 *to++ = static_cast<char>(0x80 + ((*from / (1 << shift_exponent)) % (1 << 6)));
172 ++i;
173 }
174 // If we filled up the out buffer before encoding the character
175 if (to == to_end && i != cont_octet_count) {
176 from_next = from;
177 to_next = to - (i + 1);
178 return std::codecvt_base::partial;
179 }
180 ++from;
181 }
182 from_next = from;
183 to_next = to;
184
185 // Were we done or did we run out of destination space
186 if (from == from_end)
187 return std::codecvt_base::ok;
188 else
189 return std::codecvt_base::partial;
190}

192// How many char objects can I process to get <= max_limit
193// wchar_t objects?
194BOOST_UTF8_DECL int utf8_codecvt_facet::do_length(
195 std::mbstate_t &,
196 const char * from,
197 const char * from_end,
198 std::size_t max_limit
199) const
200#if BOOST_WORKAROUND(__IBMCPP__, BOOST_TESTED_AT(600))
201 throw()
202#endif
203{
204 const char * from_next = from;
205 for (std::size_t char_count = 0u; char_count < max_limit && from_next < from_end; ++char_count) {
206 unsigned int octet_count = get_octet_count(lead_octet: *from_next);
207 // The buffer may represent incomplete characters, so terminate early if one is found
208 if (octet_count > static_cast<std::size_t>(from_end - from_next))
209 break;
210 from_next += octet_count;
211 }
212
213 return static_cast<int>(from_next - from);
214}

216BOOST_UTF8_DECL unsigned int utf8_codecvt_facet::get_octet_count(
217 unsigned char lead_octet
218) {
219 // if the 0-bit (MSB) is 0, then 1 character
220 if (lead_octet <= 0x7f) return 1;
221
222 // Otherwise the count number of consecutive 1 bits starting at MSB
223// assert(0xc0 <= lead_octet && lead_octet <= 0xfd);
224
225 if (0xc0 <= lead_octet && lead_octet <= 0xdf) return 2;
226 else if (0xe0 <= lead_octet && lead_octet <= 0xef) return 3;
227 else if (0xf0 <= lead_octet && lead_octet <= 0xf7) return 4;
228 else if (0xf8 <= lead_octet && lead_octet <= 0xfb) return 5;
229 else return 6;
230}

234template<std::size_t s>
235inline int get_cont_octet_out_count_impl(wchar_t word) {
236 if (word < 0x80) {
237 return 0;
238 }
239 if (word < 0x800) {
240 return 1;
241 }
242 return 2;
243}
244
245template<>
246inline int get_cont_octet_out_count_impl<4>(wchar_t word) {
247 if (word < 0x80) {
248 return 0;
249 }
250 if (word < 0x800) {
251 return 1;
252 }
253
254 // Note that the following code will generate warnings on some platforms
255 // where wchar_t is defined as UCS2. The warnings are superfluous as the
256 // specialization is never instantitiated with such compilers, but this
257 // can cause problems if warnings are being treated as errors, so we guard
258 // against that. Including <boost/detail/utf8_codecvt_facet.hpp> as we do
259 // should be enough to get WCHAR_MAX defined.
260#if !defined(WCHAR_MAX)
261# error WCHAR_MAX not defined!
262#endif
263 // cope with VC++ 7.1 or earlier having invalid WCHAR_MAX
264#if defined(_MSC_VER) && _MSC_VER <= 1310 // 7.1 or earlier
265 return 2;
266#elif WCHAR_MAX > 0x10000
267
268 if (word < 0x10000) {
269 return 2;
270 }
271 if (word < 0x200000) {
272 return 3;
273 }
274 if (word < 0x4000000) {
275 return 4;
276 }
277 return 5;
278
279#else
280 return 2;
281#endif
282}

286// How many "continuing octets" will be needed for this word
287// == total octets - 1.
288BOOST_UTF8_DECL int utf8_codecvt_facet::get_cont_octet_out_count(
289 wchar_t word

291 return detail::get_cont_octet_out_count_impl<sizeof(wchar_t)>(word);
