15template < class K, class C, class A, class T >
16bool set_contains(const std::set< K, C, A >& s, const T& x)
17{
18 return s.find(x) != s.end();
19}

21template < class K, class H, class C, class A, class T >
22bool set_contains(const boost::unordered_set< K, H, C, A >& s, const T& x)
23{
24 return s.find(x) != s.end();
25}

27template < class K, class C, class A >
28bool set_equal(const std::set< K, C, A >& x, const std::set< K, C, A >& y)
29{
30 return x == y;

33// Not the same as lexicographical_compare_3way applied to std::set.
34// this is equivalent semantically to bitset::operator<()
35template < class K, class C, class A >
36int set_lex_order(const std::set< K, C, A >& x, const std::set< K, C, A >& y)
37{
38 typename std::set< K, C, A >::iterator xi = x.begin(), yi = y.begin(),
39 xend = x.end(), yend = y.end();
40 for (; xi != xend && yi != yend; ++xi, ++yi)
41 {
42 if (*xi < *yi)
43 return 1;
44 else if (*yi < *xi)
45 return -1;
46 }
47 if (xi == xend)
48 return (yi == yend) ? 0 : -1;
49 else
50 return 1;
51}

73 x.erase(a);

76template < class K, class C, class A >
77void set_intersect(const std::set< K, C, A >& x, const std::set< K, C, A >& y,
78 std::set< K, C, A >& z)
79{
80 z.clear();
81 std::set_intersection(
82 x.begin(), x.end(), y.begin(), y.end(), std::inserter(z));
83}
84
85template < class K, class C, class A >
86void set_union(const std::set< K, C, A >& x, const std::set< K, C, A >& y,
87 std::set< K, C, A >& z)
88{
89 z.clear();
90 std::set_union(x.begin(), x.end(), y.begin(), y.end(), std::inserter(z));
91}
92
93template < class K, class C, class A >
94void set_difference(const std::set< K, C, A >& x, const std::set< K, C, A >& y,
95 std::set< K, C, A >& z)
96{
97 z.clear();
98 std::set_difference(
99 x.begin(), x.end(), y.begin(), y.end(), std::inserter(z, z.begin()));
100}

102template < class K, class C, class A >
103bool set_subset(const std::set< K, C, A >& x, const std::set< K, C, A >& y)
104{
105 return std::includes(x.begin(), x.end(), y.begin(), y.end());
106}
