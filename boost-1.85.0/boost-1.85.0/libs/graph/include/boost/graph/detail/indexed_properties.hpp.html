34 template < typename Derived, typename Property, typename Descriptor,
35 typename IndexMap >
36 class indexed_vertex_properties
37 {
38 public:
39 typedef no_property vertex_property_type;
40 typedef Property vertex_bundled;
41 typedef iterator_property_map<
42 typename std::vector< Property >::iterator, IndexMap >
43 vertex_map_type;
44 typedef iterator_property_map<
45 typename std::vector< Property >::const_iterator, IndexMap >
46 const_vertex_map_type;
47
48 // Directly access a vertex or edge bundle
49 Property& operator[](Descriptor v)
50 {
51 return m_vertex_properties[get(vertex_index, derived(), v)];
52 }
53
54 const Property& operator[](Descriptor v) const
55 {
56 return m_vertex_properties[get(vertex_index, derived(), v)];
57 }
58
59 vertex_map_type get_vertex_bundle(
60 const IndexMap& index_map = IndexMap())
61 {
62 return vertex_map_type(m_vertex_properties.begin(), index_map);
63 }
64
65 const_vertex_map_type get_vertex_bundle(
66 const IndexMap& index_map = IndexMap()) const
67 {
68 return const_vertex_map_type(
69 m_vertex_properties.begin(), index_map);
70 }
71
72 protected:
73 // Default-construct with no property values
74 indexed_vertex_properties() {}
75
76 // Initialize with n default-constructed property values
77 indexed_vertex_properties(std::size_t n) : m_vertex_properties(n) {}
78
79 public:
80 // Clear the properties vector
81 void clear() { m_vertex_properties.clear(); }
82
83 // Resize the properties vector
84 void resize(std::size_t n) { m_vertex_properties.resize(n); }
85
86 // Reserve space in the vector of properties
87 void reserve(std::size_t n) { m_vertex_properties.reserve(n); }
88
89 // Add a new property value to the back
90 void push_back(const Property& prop)
91 {
92 m_vertex_properties.push_back(prop);
93 }
94
95 // Write an element by raw index
96 void write_by_index(std::size_t idx, const Property& prop)
97 {
98 m_vertex_properties[idx] = prop;
99 }
100
101 // Access to the derived object
102 Derived& derived() { return *static_cast< Derived* >(this); }
103
104 const Derived& derived() const
105 {
106 return *static_cast< const Derived* >(this);
107 }
108
109 public: // should be private, but friend templates not portable
110 std::vector< Property > m_vertex_properties;
111 };

116 struct secret

118 };
119
120 public:
121 typedef no_property vertex_property_type;
122 typedef void vertex_bundled;
123 typedef secret vertex_map_type;
124 typedef secret const_vertex_map_type;
125
126 secret operator[](secret) { return secret(); }
127
128 vertex_map_type get_vertex_bundle() const { return vertex_map_type(); }

134

138 void reserve(std::size_t) {}
139 };

141 template < typename Derived, typename Property, typename Descriptor,
142 typename IndexMap >
143 class indexed_edge_properties
144 {
145 public:
146 typedef no_property edge_property_type;
147 typedef Property edge_bundled;
148 typedef Property edge_push_back_type;
149 typedef iterator_property_map<
150 typename std::vector< Property >::iterator, IndexMap >
151 edge_map_type;
152 typedef iterator_property_map<
153 typename std::vector< Property >::const_iterator, IndexMap >
154 const_edge_map_type;
155
156 // Directly access a edge or edge bundle
157 Property& operator[](Descriptor v)
158 {
159 return m_edge_properties[get(edge_index, derived(), v)];
160 }
161
162 const Property& operator[](Descriptor v) const
163 {
164 return m_edge_properties[get(edge_index, derived(), v)];
165 }
166
167 edge_map_type get_edge_bundle(const IndexMap& index_map = IndexMap())
168 {
169 return edge_map_type(m_edge_properties.begin(), index_map);
170 }
171
172 const_edge_map_type get_edge_bundle(
173 const IndexMap& index_map = IndexMap()) const
174 {
175 return const_edge_map_type(m_edge_properties.begin(), index_map);
176 }
177
178 protected:
179 // Default-construct with no property values
180 indexed_edge_properties() {}
181
182 // Initialize with n default-constructed property values
183 indexed_edge_properties(std::size_t n) : m_edge_properties(n) {}
184
185 // Get the size of the properties vector
186 std::size_t size() const { return m_edge_properties.size(); }
187
188 // Clear the properties vector
189 void clear() { m_edge_properties.clear(); }
190
191 // Resize the properties vector
192 void resize(std::size_t n) { m_edge_properties.resize(n); }
193
194 // Reserve space in the vector of properties
195 void reserve(std::size_t n) { m_edge_properties.reserve(n); }
196
197 // Write an element by raw index
198 void write_by_index(std::size_t idx, const Property& prop)
199 {
200 m_edge_properties[idx] = prop;
201 }
202
203 public:
204 // Add a new property value to the back
205 void push_back(const Property& prop)
206 {
207 m_edge_properties.push_back(prop);
208 }
209
210 // Move range of properties backwards
211 void move_range(
212 std::size_t src_begin, std::size_t src_end, std::size_t dest_begin)
213 {
214 std::copy_backward(m_edge_properties.begin() + src_begin,
215 m_edge_properties.begin() + src_end,
216 m_edge_properties.begin() + dest_begin + (src_end - src_begin));
217 }
218
219 typedef typename std::vector< Property >::iterator iterator;
220 iterator begin() { return m_edge_properties.begin(); }
221 iterator end() { return m_edge_properties.end(); }
222
223 private:
224 // Access to the derived object
225 Derived& derived() { return *static_cast< Derived* >(this); }
226
227 const Derived& derived() const
228 {
229 return *static_cast< const Derived* >(this);
230 }
231
232 public: // should be private, but friend templates not portable
233 std::vector< Property > m_edge_properties;
234 };

236 struct dummy_no_property_iterator
237 : public boost::iterator_facade< dummy_no_property_iterator, no_property,
238 std::random_access_iterator_tag >
239 {
240 mutable no_property prop;
241 no_property& dereference() const { return prop; }
242 bool equal(const dummy_no_property_iterator&) const { return true; }
243 void increment() {}
244 void decrement() {}
245 void advance(std::ptrdiff_t) {}
246 std::ptrdiff_t distance_to(const dummy_no_property_iterator) const
247 {
248 return 0;
249 }
250 };

252 template < typename Derived, typename Descriptor, typename IndexMap >
253 class indexed_edge_properties< Derived, void, Descriptor, IndexMap >
254 {
255 struct secret
256 {
257 };
258
259 public:
260 typedef no_property edge_property_type;
261 typedef void edge_bundled;
262 typedef void* edge_push_back_type;
263 typedef secret edge_map_type;
264 typedef secret const_edge_map_type;
265
266 secret operator[](secret) { return secret(); }
267 void write_by_index(std::size_t /*idx*/, const no_property& /*prop*/) {}
268
269 edge_map_type get_edge_bundle(const IndexMap& = IndexMap()) const
270 {
271 return edge_map_type();
272 }
273
274 protected:
275 // All operations do nothing.
276 indexed_edge_properties() {}
277 indexed_edge_properties(std::size_t) {}
278 std::size_t size() const { return 0; }
279 void clear() {}
280 void resize(std::size_t) {}
281 void reserve(std::size_t) {}
282
283 public:
284 void push_back(const edge_push_back_type&) {}
285 void move_range(std::size_t /*src_begin*/, std::size_t /*src_end*/,
286 std::size_t /*dest_begin*/)
287 {
288 }
289
290 typedef dummy_no_property_iterator iterator;
291 iterator begin() { return dummy_no_property_iterator(); }
292 iterator end() { return dummy_no_property_iterator(); }
293 };
