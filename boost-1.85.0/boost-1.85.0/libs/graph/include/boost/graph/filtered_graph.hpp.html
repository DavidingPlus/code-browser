25struct keep_all
26{
27 template < typename T > bool operator()(const T&) const { return true; }
28};

30// Keep residual edges (used in maximum-flow algorithms).
31template < typename ResidualCapacityEdgeMap > struct is_residual_edge
32{
33 is_residual_edge() {}
34 is_residual_edge(ResidualCapacityEdgeMap rcap) : m_rcap(rcap) {}
35 template < typename Edge > bool operator()(const Edge& e) const
36 {
37 return 0 < get(m_rcap, e);
38 }
39 ResidualCapacityEdgeMap m_rcap;
40};
41
42template < typename Set > struct is_in_subset
43{
44 is_in_subset() : m_s(0) {}
45 is_in_subset(const Set& s) : m_s(&s) {}
46
47 template < typename Elt > bool operator()(const Elt& x) const
48 {
49 return set_contains(*m_s, x);
50 }
51 const Set* m_s;
52};
53
54template < typename Set > struct is_not_in_subset
55{
56 is_not_in_subset() : m_s(0) {}
57 is_not_in_subset(const Set& s) : m_s(&s) {}
58
59 template < typename Elt > bool operator()(const Elt& x) const
60 {
61 return !set_contains(*m_s, x);
62 }
63 const Set* m_s;
64};

69 template < typename EdgePredicate, typename VertexPredicate,
70 typename Graph >
71 struct out_edge_predicate
72 {
73 out_edge_predicate() {}
74 out_edge_predicate(EdgePredicate ep, VertexPredicate vp, const Graph& g)
75 : m_edge_pred(ep), m_vertex_pred(vp), m_g(&g)
76 {
77 }
78
79 template < typename Edge > bool operator()(const Edge& e) const
80 {
81 return m_edge_pred(e) && m_vertex_pred(target(e, *m_g));
82 }
83 EdgePredicate m_edge_pred;
84 VertexPredicate m_vertex_pred;
85 const Graph* m_g;
86 };
87
88 template < typename EdgePredicate, typename VertexPredicate,
89 typename Graph >
90 struct in_edge_predicate
91 {
92 in_edge_predicate() {}
93 in_edge_predicate(EdgePredicate ep, VertexPredicate vp, const Graph& g)
94 : m_edge_pred(ep), m_vertex_pred(vp), m_g(&g)
95 {
96 }
97
98 template < typename Edge > bool operator()(const Edge& e) const
99 {
100 return m_edge_pred(e) && m_vertex_pred(source(e, *m_g));
101 }
102 EdgePredicate m_edge_pred;
103 VertexPredicate m_vertex_pred;
104 const Graph* m_g;
105 };
106
107 template < typename EdgePredicate, typename VertexPredicate,
108 typename Graph >
109 struct edge_predicate
110 {
111 edge_predicate() {}
112 edge_predicate(EdgePredicate ep, VertexPredicate vp, const Graph& g)
113 : m_edge_pred(ep), m_vertex_pred(vp), m_g(&g)
114 {
115 }
116
117 template < typename Edge > bool operator()(const Edge& e) const
118 {
119 return m_edge_pred(e) && m_vertex_pred(source(e, *m_g))
120 && m_vertex_pred(target(e, *m_g));
121 }
122 EdgePredicate m_edge_pred;
123 VertexPredicate m_vertex_pred;
124 const Graph* m_g;
125 };

136// This base class is a stupid hack to change overload resolution
137// rules for the source and target functions so that they are a
138// worse match than the source and target functions defined for
139// pairs in graph_traits.hpp. I feel dirty. -JGS
140template < class G > struct filtered_graph_base
141{
142 typedef graph_traits< G > Traits;
143 typedef typename Traits::vertex_descriptor vertex_descriptor;
144 typedef typename Traits::edge_descriptor edge_descriptor;
145 filtered_graph_base(const G& g) : m_g(g) {}
146 // protected:
147 const G& m_g;
148};

150template < typename Graph, typename EdgePredicate,
151 typename VertexPredicate = keep_all >
152class filtered_graph : public filtered_graph_base< Graph >
153{
154 typedef filtered_graph_base< Graph > Base;
155 typedef graph_traits< Graph > Traits;
156 typedef filtered_graph self;
157
158public:
159 typedef Graph graph_type;
160 typedef detail::out_edge_predicate< EdgePredicate, VertexPredicate, self >
161 OutEdgePred;
162 typedef detail::in_edge_predicate< EdgePredicate, VertexPredicate, self >
163 InEdgePred;
164 typedef detail::edge_predicate< EdgePredicate, VertexPredicate, self >
165 EdgePred;
166
167 // Constructors
168 filtered_graph(const Graph& g, EdgePredicate ep) : Base(g), m_edge_pred(ep)
169 {
170 }
171
172 filtered_graph(const Graph& g, EdgePredicate ep, VertexPredicate vp)
173 : Base(g), m_edge_pred(ep), m_vertex_pred(vp)
174 {
175 }
176
177 // Graph requirements
178 typedef typename Traits::vertex_descriptor vertex_descriptor;
179 typedef typename Traits::edge_descriptor edge_descriptor;
180 typedef typename Traits::directed_category directed_category;
181 typedef typename Traits::edge_parallel_category edge_parallel_category;
182 typedef typename Traits::traversal_category traversal_category;
183
184 // IncidenceGraph requirements
185 typedef filter_iterator< OutEdgePred, typename Traits::out_edge_iterator >
186 out_edge_iterator;
187
188 typedef typename Traits::degree_size_type degree_size_type;
189
190 // AdjacencyGraph requirements
191 typedef typename adjacency_iterator_generator< self, vertex_descriptor,
192 out_edge_iterator >::type adjacency_iterator;
193
194 // BidirectionalGraph requirements
195 typedef filter_iterator< InEdgePred, typename Traits::in_edge_iterator >
196 in_edge_iterator;
197
198 // VertexListGraph requirements
199 typedef filter_iterator< VertexPredicate, typename Traits::vertex_iterator >
200 vertex_iterator;
201 typedef typename Traits::vertices_size_type vertices_size_type;
202
203 // EdgeListGraph requirements
204 typedef filter_iterator< EdgePred, typename Traits::edge_iterator >
205 edge_iterator;
206 typedef typename Traits::edges_size_type edges_size_type;
207
208 typedef filtered_graph_tag graph_tag;
209
210 // Bundled properties support
211 template < typename Descriptor >
212 typename graph::detail::bundled_result< Graph, Descriptor >::type&
213 operator[](Descriptor x)
214 {
215 return const_cast< Graph& >(this->m_g)[x];
216 }
217
218 template < typename Descriptor >
219 typename graph::detail::bundled_result< Graph, Descriptor >::type const&
220 operator[](Descriptor x) const
221 {
222 return this->m_g[x];
223 }
224
225 static vertex_descriptor null_vertex() { return Traits::null_vertex(); }
226
227 // private:
228 EdgePredicate m_edge_pred;
229 VertexPredicate m_vertex_pred;
230};

306template < typename G, typename EP, typename VP >
307std::pair< typename filtered_graph< G, EP, VP >::vertex_iterator,
308 typename filtered_graph< G, EP, VP >::vertex_iterator >
309vertices(const filtered_graph< G, EP, VP >& g)
310{
311 typedef filtered_graph< G, EP, VP > Graph;
312 typename graph_traits< G >::vertex_iterator f, l;
313 boost::tie(f, l) = vertices(g.m_g);
314 typedef typename Graph::vertex_iterator iter;
315 return std::make_pair(
316 iter(g.m_vertex_pred, f, l), iter(g.m_vertex_pred, l, l));
317}
318
319template < typename G, typename EP, typename VP >
320std::pair< typename filtered_graph< G, EP, VP >::edge_iterator,
321 typename filtered_graph< G, EP, VP >::edge_iterator >
322edges(const filtered_graph< G, EP, VP >& g)
323{
324 typedef filtered_graph< G, EP, VP > Graph;
325 typename Graph::EdgePred pred(g.m_edge_pred, g.m_vertex_pred, g);
326 typename graph_traits< G >::edge_iterator f, l;
327 boost::tie(f, l) = edges(g.m_g);
328 typedef typename Graph::edge_iterator iter;
329 return std::make_pair(iter(pred, f, l), iter(pred, l, l));
330}

332// An alternative for num_vertices() and num_edges() would be to
333// count the number in the filtered graph. This is problematic
334// because of the interaction with the vertex indices... they would
335// no longer go from 0 to num_vertices(), which would cause trouble
336// for algorithms allocating property storage in an array. We could
337// try to create a mapping to new recalibrated indices, but I don't
338// see an efficient way to do this.
339//
340// However, the current solution is still unsatisfactory because
341// the following semantic constraints no longer hold:
342// boost::tie(vi, viend) = vertices(g);
343// assert(std::distance(vi, viend) == num_vertices(g));
344
345template < typename G, typename EP, typename VP >
346typename filtered_graph< G, EP, VP >::vertices_size_type num_vertices(
347 const filtered_graph< G, EP, VP >& g)
348{
349 return num_vertices(g.m_g);
350}

375template < typename G, typename EP, typename VP >
376std::pair< typename filtered_graph< G, EP, VP >::out_edge_iterator,
377 typename filtered_graph< G, EP, VP >::out_edge_iterator >
378out_edges(typename filtered_graph< G, EP, VP >::vertex_descriptor u,
379 const filtered_graph< G, EP, VP >& g)
380{
381 typedef filtered_graph< G, EP, VP > Graph;
382 typename Graph::OutEdgePred pred(g.m_edge_pred, g.m_vertex_pred, g);
383 typedef typename Graph::out_edge_iterator iter;
384 typename graph_traits< G >::out_edge_iterator f, l;
385 boost::tie(f, l) = out_edges(u, g.m_g);
386 return std::make_pair(iter(pred, f, l), iter(pred, l, l));
387}

389template < typename G, typename EP, typename VP >
390typename filtered_graph< G, EP, VP >::degree_size_type out_degree(
391 typename filtered_graph< G, EP, VP >::vertex_descriptor u,
392 const filtered_graph< G, EP, VP >& g)
393{
394 typename filtered_graph< G, EP, VP >::degree_size_type n = 0;
395 typename filtered_graph< G, EP, VP >::out_edge_iterator f, l;
396 for (boost::tie(f, l) = out_edges(u, g); f != l; ++f)
397 ++n;
398 return n;
399}

401template < typename G, typename EP, typename VP >
402std::pair< typename filtered_graph< G, EP, VP >::adjacency_iterator,
403 typename filtered_graph< G, EP, VP >::adjacency_iterator >
404adjacent_vertices(typename filtered_graph< G, EP, VP >::vertex_descriptor u,
405 const filtered_graph< G, EP, VP >& g)
406{
407 typedef filtered_graph< G, EP, VP > Graph;
408 typedef typename Graph::adjacency_iterator adjacency_iterator;
409 typename Graph::out_edge_iterator f, l;
410 boost::tie(f, l) = out_edges(u, g);
411 return std::make_pair(adjacency_iterator(f, const_cast< Graph* >(&g)),
412 adjacency_iterator(l, const_cast< Graph* >(&g)));
413}

459template < typename G, typename EP, typename VP >
460std::pair< typename filtered_graph< G, EP, VP >::edge_descriptor, bool > edge(
461 typename filtered_graph< G, EP, VP >::vertex_descriptor u,
462 typename filtered_graph< G, EP, VP >::vertex_descriptor v,
463 const filtered_graph< G, EP, VP >& g)
464{
465 typename graph_traits< G >::edge_descriptor e;
466 bool exists;
467 boost::tie(e, exists) = edge(u, v, g.m_g);
468 return std::make_pair(e, exists && g.m_edge_pred(e));
469}
