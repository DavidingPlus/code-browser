66template <class FirstType>
67struct select1st
68{
69 typedef FirstType type;
70
71 template<class T>
72 BOOST_CONTAINER_FORCEINLINE const type& operator()(const T& x) const
73 { return x.first; }
74
75 template<class T>
76 BOOST_CONTAINER_FORCEINLINE type& operator()(T& x)
77 { return const_cast<type&>(x.first); }
78};

111// Trait to detect Allocator-like types.
112template<typename Allocator, typename = void>
113struct is_allocator
114{
115 static const bool value = false;
116};
117
118template <typename T>
119T&& ctad_declval();
120
121template<typename Allocator>
122struct is_allocator < Allocator,
123 variadic_void_t< typename Allocator::value_type
124 , decltype(ctad_declval<Allocator&>().allocate(size_t{})) >>
125{
126 static const bool value = true;
127};
128
129template<class T>
130using require_allocator_t = typename enable_if_c<is_allocator<T>::value, T>::type;
131
132template<class T>
133using require_nonallocator_t = typename enable_if_c<!is_allocator<T>::value, T>::type;
