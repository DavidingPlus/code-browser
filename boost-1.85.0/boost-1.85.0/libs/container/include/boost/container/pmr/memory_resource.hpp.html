25namespace boost {
26namespace container {
27namespace pmr {
28
29//! The memory_resource class is an abstract interface to an
30//! unbounded set of classes encapsulating memory resources.
31class memory_resource
32{
33 public:
34 // For exposition only
35 static BOOST_CONSTEXPR_OR_CONST std::size_t max_align =
36 boost::move_detail::alignment_of<boost::move_detail::max_align_t>::value;
37
38 //! <b>Effects</b>: Destroys
39 //! this memory_resource.
40 virtual ~memory_resource(){}
41
42 //! <b>Effects</b>: Equivalent to
43 //! `return do_allocate(bytes, alignment);`
44 void* allocate(std::size_t bytes, std::size_t alignment = max_align)
45 {
46 //Obtain a pointer to enough storage and initialize the lifetime
47 //of an array object of the given size in the address
48 return ::operator new(bytes, p: this->do_allocate(bytes, alignment), boost_container_new_t());
49 }
50
51 //! <b>Effects</b>: Equivalent to
52 //! `return do_deallocate(bytes, alignment);`
53 void deallocate(void* p, std::size_t bytes, std::size_t alignment = max_align)
54 { return this->do_deallocate(p, bytes, alignment); }
55
56 //! <b>Effects</b>: Equivalent to
57 //! `return do_is_equal(other);`
58 bool is_equal(const memory_resource& other) const BOOST_NOEXCEPT
59 { return this->do_is_equal(other); }
60
61 #if !defined(BOOST_EMBTC)
62
63 //! <b>Returns</b>:
64 //! `&a == &b || a.is_equal(b)`.
65 friend bool operator==(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT
66 { return &a == &b || a.is_equal(other: b); }
67
68 //! <b>Returns</b>:
69 //! !(a == b).
70 friend bool operator!=(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT
71 { return !(a == b); }
72
73 #else
74
75 //! <b>Returns</b>:
76 //! `&a == &b || a.is_equal(b)`.
77 friend bool operator==(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT;
78
79 //! <b>Returns</b>:
80 //! !(a == b).
81 friend bool operator!=(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT;
82
83 #endif
84
85 protected:
86 //! <b>Requires</b>: Alignment shall be a power of two.
87 //!
88 //! <b>Returns</b>: A derived class shall implement this function to return a pointer
89 //! to allocated storage with a size of at least bytes. The returned storage is
90 //! aligned to the specified alignment, if such alignment is supported; otherwise
91 //! it is aligned to max_align.
92 //!
93 //! <b>Throws</b>: A derived class implementation shall throw an appropriate exception if
94 //! it is unable to allocate memory with the requested size and alignment.
95 virtual void* do_allocate(std::size_t bytes, std::size_t alignment) = 0;
96
97 //! <b>Requires</b>: p shall have been returned from a prior call to
98 //! `allocate(bytes, alignment)` on a memory resource equal to *this, and the storage
99 //! at p shall not yet have been deallocated.
100 //!
101 //! <b>Effects</b>: A derived class shall implement this function to dispose of allocated storage.
102 //!
103 //! <b>Throws</b>: Nothing.
104 virtual void do_deallocate(void* p, std::size_t bytes, std::size_t alignment) = 0;
105
106 //! <b>Returns</b>: A derived class shall implement this function to return true if memory
107 //! allocated from this can be deallocated from other and vice-versa; otherwise it shall
108 //! return false. <i>[Note: The most-derived type of other might not match the type of this.
109 //! For a derived class, D, a typical implementation of this function will compute
110 //! `dynamic_cast<const D*>(&other)` and go no further (i.e., return false)
111 //! if it returns nullptr. - end note]</i>.
112 virtual bool do_is_equal(const memory_resource& other) const BOOST_NOEXCEPT = 0;
113};
114
115#if defined(BOOST_EMBTC)
116
117//! <b>Returns</b>:
118//! `&a == &b || a.is_equal(b)`.
119inline bool operator==(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT
120{ return &a == &b || a.is_equal(b); }
121
122//! <b>Returns</b>:
123//! !(a == b).
124inline bool operator!=(const memory_resource& a, const memory_resource& b) BOOST_NOEXCEPT
125{ return !(a == b); }
126
127#endif
