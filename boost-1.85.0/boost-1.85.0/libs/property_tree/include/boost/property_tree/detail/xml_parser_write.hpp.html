23 template<class Str>
24 void write_xml_indent(std::basic_ostream<typename Str::value_type> &stream,
25 int indent,
26 const xml_writer_settings<Str> & settings
27 )
28 {
29 stream << std::basic_string<typename Str::value_type>(indent * settings.indent_count, settings.indent_char);
30 }
31
32 template<class Str>
33 void write_xml_comment(std::basic_ostream<typename Str::value_type> &stream,
34 const Str &s,
35 int indent,
36 bool separate_line,
37 const xml_writer_settings<Str> & settings
38 )
39 {
40 typedef typename Str::value_type Ch;
41 if (separate_line)
42 write_xml_indent(stream,indent,settings);
43 stream << Ch('<') << Ch('!') << Ch('-') << Ch('-');
44 stream << s;
45 stream << Ch('-') << Ch('-') << Ch('>');
46 if (separate_line)
47 stream << Ch('\n');
48 }
49
50 template<class Str>
51 void write_xml_text(std::basic_ostream<typename Str::value_type> &stream,
52 const Str &s,
53 int indent,
54 bool separate_line,
55 const xml_writer_settings<Str> & settings
56 )
57 {
58 typedef typename Str::value_type Ch;
59 if (separate_line)
60 write_xml_indent(stream,indent,settings);
61 stream << encode_char_entities(s);
62 if (separate_line)
63 stream << Ch('\n');
64 }

66 template<class Ptree>
67 void write_xml_element(std::basic_ostream<typename Ptree::key_type::value_type> &stream,
68 const typename Ptree::key_type &key,
69 const Ptree &pt,
70 int indent,
71 const xml_writer_settings<typename Ptree::key_type> & settings)
72 {
73 typedef typename Ptree::key_type::value_type Ch;
74 typedef typename Ptree::key_type Str;
75 typedef typename Ptree::const_iterator It;
76
77 bool want_pretty = settings.indent_count > 0;
78 // Find if elements present
79 bool has_elements = false;
80 bool has_attrs_only = pt.data().empty();
81 for (It it = pt.begin(), end = pt.end(); it != end; ++it)
82 {
83 if (it->first != xmlattr<Str>() )
84 {
85 has_attrs_only = false;
86 if (it->first != xmltext<Str>())
87 {
88 has_elements = true;
89 break;
90 }
91 }
92 }
93
94 // Write element
95 if (pt.data().empty() && pt.empty()) // Empty key
96 {
97 if (indent >= 0)
98 {
99 write_xml_indent(stream,indent,settings);
100 stream << Ch('<') << key <<
101 Ch('/') << Ch('>');
102 if (want_pretty)
103 stream << Ch('\n');
104 }
105 }
106 else // Nonempty key
107 {
108 // Write opening tag, attributes and data
109 if (indent >= 0)
110 {
111 // Write opening brace and key
112 write_xml_indent(stream,indent,settings);
113 stream << Ch('<') << key;
114
115 // Write attributes
116 if (optional<const Ptree &> attribs = pt.get_child_optional(xmlattr<Str>()))
117 for (It it = attribs.get().begin(); it != attribs.get().end(); ++it)
118 stream << Ch(' ') << it->first << Ch('=')
119 << Ch('"')
120 << encode_char_entities(
121 it->second.template get_value<Str>())
122 << Ch('"');
123
124 if ( has_attrs_only )
125 {
126 // Write closing brace
127 stream << Ch('/') << Ch('>');
128 if (want_pretty)
129 stream << Ch('\n');
130 }
131 else
132 {
133 // Write closing brace
134 stream << Ch('>');
135
136 // Break line if needed and if we want pretty-printing
137 if (has_elements && want_pretty)
138 stream << Ch('\n');
139 }
140 }
141
142 // Write data text, if present
143 if (!pt.data().empty())
144 write_xml_text(stream,
145 pt.template get_value<Str>(),
146 indent + 1, has_elements && want_pretty, settings);
147
148 // Write elements, comments and texts
149 for (It it = pt.begin(); it != pt.end(); ++it)
150 {
151 if (it->first == xmlattr<Str>())
152 continue;
153 else if (it->first == xmlcomment<Str>())
154 write_xml_comment(stream,
155 it->second.template get_value<Str>(),
156 indent + 1, want_pretty, settings);
157 else if (it->first == xmltext<Str>())
158 write_xml_text(stream,
159 it->second.template get_value<Str>(),
160 indent + 1, has_elements && want_pretty, settings);
161 else
162 write_xml_element(stream, it->first, it->second,
163 indent + 1, settings);
164 }
165
166 // Write closing tag
167 if (indent >= 0 && !has_attrs_only)
168 {
169 if (has_elements)
170 write_xml_indent(stream,indent,settings);
171 stream << Ch('<') << Ch('/') << key << Ch('>');
172 if (want_pretty)
173 stream << Ch('\n');
174 }
175
176 }
177 }

179 template<class Ptree>
180 void write_xml_internal(std::basic_ostream<typename Ptree::key_type::value_type> &stream,
181 const Ptree &pt,
182 const std::string &filename,
183 const xml_writer_settings<typename Ptree::key_type> & settings)
184 {
185 typedef typename Ptree::key_type Str;
186 stream << detail::widen<Str>("<?xml version=\"1.0\" encoding=\"")
187 << settings.encoding
188 << detail::widen<Str>("\"?>\n");
189 write_xml_element(stream, Str(), pt, -1, settings);
190 stream.flush();
191 if (!stream)
192 BOOST_PROPERTY_TREE_THROW(xml_parser_error("write error", filename, 0));
193 }
194
195} } }
