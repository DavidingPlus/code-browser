24 // Helper for preparing what string in ptree_bad_path exception
25 template<class P> inline
26 std::string prepare_bad_path_what(const std::string &what,
27 const P &path)
28 {
29 return what + " (" + path.dump() + ")";
30 }
31
32 }

45 template<class D> inline

48 {
49 }
50
51 template<class D> inline
52 D ptree_bad_data::data() const

55 }

63 {
64
65 }
66
