46inline bool tn_remove_prefix( std::string& str, char const* prefix )
47{
48 std::size_t n = std::strlen( s: prefix );
49
50 if( str.substr( pos: 0, n: n ) == prefix )
51 {
52 str = str.substr( pos: n );
53 return true;
54 }
55 else
56 {
57 return false;
58 }
59}

63// typeid_name
64
65inline std::string fix_typeid_name( char const* n )
66{
67 std::string r = boost::core::demangle( name: n );
68
69#if defined(_MSC_VER)
70
71 tn_remove_prefix( r, "class " );
72 tn_remove_prefix( r, "struct " );
73 tn_remove_prefix( r, "enum " );
74
75#endif
76
77 // libc++ inline namespace
78
79 if( tn_remove_prefix( str&: r, prefix: "std::__1::" ) )
80 {
81 r = "std::" + r;
82 }
83
84 // libstdc++ inline namespace
85
86 if( tn_remove_prefix( str&: r, prefix: "std::__cxx11::" ) )
87 {
88 r = "std::" + r;
89 }
90
91#if defined(BOOST_MSVC) && BOOST_MSVC == 1600
92
93 // msvc-10.0 puts TR1 things in std::tr1
94
95 if( tn_remove_prefix( r, "std::tr1::" ) )
96 {
97 r = "std::" + r;
98 }
99
100#endif
101
102 return r;
103}

107{

199// Use snprintf if available as some compilers (clang 14.0) issue deprecation warnings for sprintf
200#if ( defined(_MSC_VER) && _MSC_VER < 1900 ) || ( defined(__MINGW32__) && !defined(__MINGW64_VERSION_MAJOR) )
201# define BOOST_CORE_DETAIL_SNPRINTF(buffer, format, arg) std::sprintf(buffer, format, arg)
202#else
203# define BOOST_CORE_DETAIL_SNPRINTF(buffer, format, arg) std::snprintf(buffer, sizeof(buffer)/sizeof(buffer[0]), format, arg)
204#endif
205
206inline std::string tn_to_string( std::size_t n )
207{
208 char buffer[ 32 ];
209 BOOST_CORE_DETAIL_SNPRINTF( buffer, "%lu", static_cast< unsigned long >( n ) );
210
211 return buffer;
212}
213
214#undef BOOST_CORE_DETAIL_SNPRINTF

506// tn_add_each
507
508template<class T> int tn_add_each_impl( std::string& st )
509{
510 if( !st.empty() ) st += ", ";
511 st += tn_holder<T>::type_name( "" );
512 return 0;
513}
514
515template<class... T> std::string tn_add_each()
516{
517 std::string st;
518
519 typedef int A[ sizeof...(T) + 1 ];
520 (void)A{ 0, tn_add_each_impl<T>( st )... };
521
522 return st;
523}

525template<class R, class... A> std::string function_type_name( tn_identity<R(A...)>, std::string const& trailer, std::string const& suffix )
526{
527 std::string r = tn_holder<R>::type_name( "" );
528
529 if( !suffix.empty() )
530 {
531 r += '(';
532
533 if( suffix[ 0 ] == ' ' )
534 {
535 r += suffix.substr( pos: 1 );
536 }
537 else
538 {
539 r += suffix;
540 }
541
542 r += ')';
543 }
544
545 r += '(' + tn_add_each<A...>() + ')';
546 r += trailer;
547
548 return r;
549}

774template<class T, std::size_t N> std::pair<std::string, std::string> array_prefix_suffix( tn_identity<T[N]> )
775{
776 std::pair<std::string, std::string> r = detail::array_prefix_suffix( tn_identity<T>() );
777
778 r.second = '[' + tn_to_string( n: N ) + ']' + r.second;
779
780 return r;
781}

783template<class T> std::string array_type_name( tn_identity<T[]>, std::string const& suffix )
784{
785 std::pair<std::string, std::string> r = detail::array_prefix_suffix( tn_identity<T>() );
786
787 if( suffix.empty() )
788 {
789 return r.first + "[]" + r.second;
790 }
791 else
792 {
793 return r.first + '(' + suffix + ")[]" + r.second;
794 }
795}
