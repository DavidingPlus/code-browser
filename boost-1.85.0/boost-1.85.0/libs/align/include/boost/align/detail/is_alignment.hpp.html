10
11#include <boost/config.hpp>
12#include <cstddef>
13
14namespace boost {
15namespace alignment {
16namespace detail {
17
18BOOST_CONSTEXPR inline bool
19is_alignment(std::size_t value) BOOST_NOEXCEPT
20{
21 return (value > 0) && ((value & (value - 1)) == 0);
22}
