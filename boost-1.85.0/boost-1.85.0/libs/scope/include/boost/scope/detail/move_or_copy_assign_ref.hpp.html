26namespace scope {
27namespace detail {

29//! The type trait produces an rvalue reference to \a From if \a To has a non-throwing assignment from a \a From rvalue and an lvalue reference otherwise.
30template< typename From, typename To = From >
31struct move_or_copy_assign_ref
32{
33 using type = typename std::conditional<
34 std::is_nothrow_assignable< To, From >::value,
35 From&&,
36 From const&
37 >::type;
38};
39
40template< typename From, typename To >
41struct move_or_copy_assign_ref< From&, To >
42{
43 using type = From&;
44};
