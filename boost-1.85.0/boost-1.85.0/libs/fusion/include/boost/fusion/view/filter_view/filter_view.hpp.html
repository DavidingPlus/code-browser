30namespace boost { namespace fusion
31{
32 struct filter_view_tag;
33 struct forward_traversal_tag;
34 struct fusion_sequence_tag;
35
36 template <typename Sequence, typename Pred>
37 struct filter_view : sequence_base<filter_view<Sequence, Pred> >
38 {
39 typedef filter_view_tag fusion_tag;
40 typedef fusion_sequence_tag tag; // this gets picked up by MPL
41 typedef typename
42 mpl::eval_if<
43 traits::is_associative<Sequence>
44 , mpl::inherit2<forward_traversal_tag,associative_tag>
45 , mpl::identity<forward_traversal_tag>
46 >::type
47 category;
48 typedef mpl::true_ is_view;
49
50 typedef typename result_of::begin<Sequence>::type first_type;
51 typedef typename result_of::end<Sequence>::type last_type;
52 typedef Pred pred_type;
53
54 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED
55 filter_view(Sequence& in_seq)
56 : seq(in_seq)
57 {}
58
59 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED
60 first_type first() const { return fusion::begin(seq); }
61 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED
62 last_type last() const { return fusion::end(seq); }
63 typename mpl::if_<traits::is_view<Sequence>, Sequence, Sequence&>::type seq;
64 };
65}}
