34namespace boost { namespace fusion
35{
36 struct filter_view_iterator_tag;
37 struct forward_traversal_tag;
38
39 template <typename Category, typename First, typename Last, typename Pred>
40 struct filter_iterator : iterator_base<filter_iterator<Category, First, Last, Pred> >
41 {
42 typedef convert_iterator<First> first_converter;
43 typedef typename first_converter::type first_iter;
44 typedef convert_iterator<Last> last_converter;
45 typedef typename last_converter::type last_iter;
46
47 typedef filter_view_iterator_tag fusion_tag;
48 typedef Category category;
49 typedef
50 detail::static_find_if<
51 first_iter
52 , last_iter
53 , mpl::bind1<
54 typename mpl::lambda<Pred>::type
55 , mpl::bind1<mpl::quote1<result_of::value_of>,mpl::_1>
56 >
57 >
58 filter;
59 typedef typename filter::type first_type;
60 typedef last_iter last_type;
61 typedef Pred pred_type;
62
63 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED
64 filter_iterator(First const& in_first)
65 : first(filter::iter_call(first_converter::call(in_first))) {}
66
67 first_type first;
68 };
69}}
