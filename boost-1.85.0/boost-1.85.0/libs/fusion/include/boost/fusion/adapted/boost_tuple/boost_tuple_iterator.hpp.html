52 template <typename Cons = tuples::null_type>
53 struct boost_tuple_iterator
54 : iterator_facade<boost_tuple_iterator<Cons>, forward_traversal_tag>
55 {
56 typedef Cons cons_type;
57
58 typedef boost_tuple_iterator_identity<
59 typename add_const<Cons>::type> identity;
60
61 BOOST_FUSION_GPU_ENABLED
62 explicit boost_tuple_iterator(Cons& in_cons)
63 : cons(in_cons) {}
64 Cons& cons;
65
66 template <typename Iterator>
67 struct value_of : mpl::identity<typename Iterator::cons_type::head_type> {};
68
69 template <typename Iterator>
70 struct deref
71 {
72 typedef typename value_of<Iterator>::type element;
73
74 typedef typename
75 mpl::if_<
76 is_const<typename Iterator::cons_type>
77 , typename tuples::access_traits<element>::const_type
78 , typename tuples::access_traits<element>::non_const_type
79 >::type
80 type;
81
82 BOOST_FUSION_GPU_ENABLED
83 static type
84 call(Iterator const& iter)
85 {
86 return iter.cons.get_head();
87 }
88 };
89
90 template <typename Iterator>
91 struct next
92 {
93 typedef typename Iterator::cons_type cons_type;
94 typedef typename cons_type::tail_type tail_type;
95
96 typedef boost_tuple_iterator<
97 typename mpl::eval_if<
98 is_const<cons_type>
99 , add_const<tail_type>
100 , mpl::identity<tail_type>
101 >::type>
102 type;
103
104 BOOST_FUSION_GPU_ENABLED
105 static type
106 call(Iterator const& iter)
107 {
108 return type(iter.cons.get_tail());
109 }
110 };

112 template <typename I1, typename I2>
113 struct distance;

115 // detail
116 template <typename I1, typename I2>
117 struct lazy_next_distance
118 {
119 typedef
120 typename mpl::plus<
121 mpl::int_<1>,
122 typename distance<
123 typename next<I1>::type,
124 I2
125 >::type
126 >::type type;
127 };
128
129 template <typename I1, typename I2>
130 struct distance
131 {
132 typedef typename mpl::eval_if<
133 boost::is_same<I1, I2>,
134 mpl::int_<0>,
135 lazy_next_distance<I1, I2>
136 >::type type;
137
138 BOOST_FUSION_GPU_ENABLED
139 static type
140 call(I1 const&, I2 const&)
141 {
142 return type();
143 }
144 };

164 template <typename I1, typename I2>
165 struct equal_to
166 : mpl::or_<
167 is_same<I1, I2>
168 , mpl::and_<
