40#define BOOST_FUSION_ADAPT_STRUCT_WRAP_ATTR(X, Y) \
41 BOOST_PP_IIF(BOOST_MPL_PP_TOKEN_EQUAL(auto, BOOST_PP_EXPAND(X)), \
