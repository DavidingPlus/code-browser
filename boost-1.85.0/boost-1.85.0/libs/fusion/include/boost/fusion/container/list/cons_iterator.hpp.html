33 template <typename Cons = nil_>
34 struct cons_iterator : iterator_base<cons_iterator<Cons> >
35 {
36 typedef cons_iterator_tag fusion_tag;
37 typedef forward_traversal_tag category;
38 typedef Cons cons_type;
39 typedef cons_iterator_identity<
40 typename add_const<Cons>::type>
41 identity;
42
43 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED
44 explicit cons_iterator(cons_type& in_cons) BOOST_NOEXCEPT
45 : cons(in_cons) {}
46
47 cons_type& cons;
48 };

50# pragma warning(pop)

53 struct nil_iterator : iterator_base<nil_iterator>
54 {

62 BOOST_CONSTEXPR BOOST_FUSION_GPU_ENABLED

66 };
