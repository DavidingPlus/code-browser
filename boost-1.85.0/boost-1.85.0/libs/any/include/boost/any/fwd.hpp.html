31namespace detail {
32
33 template <class T>
34 struct is_basic_any: public std::false_type {};

36 template<std::size_t OptimizeForSize, std::size_t OptimizeForAlignment>
37 struct is_basic_any<boost::anys::basic_any<OptimizeForSize, OptimizeForAlignment> > : public std::true_type {};
38
39 template <class T>
40 struct is_some_any: public is_basic_any<T> {};
41
42 template <>
43 struct is_some_any<boost::any>: public std::true_type {};

45 template <>
