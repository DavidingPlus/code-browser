24#include <boost/detail/workaround.hpp>
25#include <boost/type_traits/is_member_function_pointer.hpp>

27namespace boost {
28
29#if defined( BOOST_CODEGEARC )
30template <class T> struct is_member_pointer : public integral_constant<bool, __is_member_pointer(T)>{};
31#else
32template <class T> struct is_member_pointer : public integral_constant<bool, ::boost::is_member_function_pointer<T>::value>{};
33template <class T, class U> struct is_member_pointer<U T::* > : public true_type{};
34
35#if !BOOST_WORKAROUND(__MWERKS__,<=0x3003) && !BOOST_WORKAROUND(__IBMCPP__, <=600)
36template <class T, class U> struct is_member_pointer<U T::*const> : public true_type{};
37template <class T, class U> struct is_member_pointer<U T::*const volatile> : public true_type{};
38template <class T, class U> struct is_member_pointer<U T::*volatile> : public true_type{};
39#endif
