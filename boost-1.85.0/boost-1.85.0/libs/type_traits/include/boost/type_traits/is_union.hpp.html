11#ifndef BOOST_TT_IS_UNION_HPP_INCLUDED

14#include <boost/type_traits/intrinsics.hpp>
15#include <boost/type_traits/integral_constant.hpp>
16
17namespace boost {
18
19#ifdef BOOST_IS_UNION
20template <class T> struct is_union : public integral_constant<bool, BOOST_IS_UNION(T)> {};
21#else
22template <class T> struct is_union : public integral_constant<bool, false> {};
23#endif
24
25template <class T> struct is_union<T const> : public is_union<T>{};
26template <class T> struct is_union<T volatile const> : public is_union<T>{};
27template <class T> struct is_union<T volatile> : public is_union<T>{};

29} // namespace boost
