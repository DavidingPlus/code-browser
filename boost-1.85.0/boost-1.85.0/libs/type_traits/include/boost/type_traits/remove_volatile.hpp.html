11#ifndef BOOST_TT_REMOVE_VOLATILE_HPP_INCLUDED
12#define BOOST_TT_REMOVE_VOLATILE_HPP_INCLUDED
13
14#include <boost/config.hpp>
15#include <boost/detail/workaround.hpp>
16#include <cstddef> // size_t
17
18namespace boost {
19
20 // convert a type T to a non-cv-qualified type - remove_volatile<T>
21 template <class T> struct remove_volatile{ typedef T type; };
22 template <class T> struct remove_volatile<T volatile>{ typedef T type; };
23
24#if !defined(BOOST_NO_ARRAY_TYPE_SPECIALIZATIONS)
25 template <class T, std::size_t N> struct remove_volatile<T volatile[N]>{ typedef T type[N]; };
26#if !BOOST_WORKAROUND(BOOST_BORLANDC, < 0x600) && !defined(__IBMCPP__) && !BOOST_WORKAROUND(__DMC__, BOOST_TESTED_AT(0x840))
27 template <class T> struct remove_volatile<T volatile[]>{ typedef T type[]; };
28#endif
29#endif
30
31#if !defined(BOOST_NO_CXX11_TEMPLATE_ALIASES)
32
33 template <class T> using remove_volatile_t = typename remove_volatile<T>::type;
34
35#endif
36
37} // namespace boost
38
39#endif // BOOST_TT_REMOVE_VOLATILE_HPP_INCLUDED
