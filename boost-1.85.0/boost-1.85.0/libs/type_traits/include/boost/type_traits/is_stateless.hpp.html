14#include <boost/type_traits/has_trivial_destructor.hpp>

19namespace boost {
20
21template <typename T>
22struct is_stateless
23 : public integral_constant<bool,
24 (::boost::has_trivial_constructor<T>::value
25 && ::boost::has_trivial_copy<T>::value
26 && ::boost::has_trivial_destructor<T>::value
27 && ::boost::is_class<T>::value
28 && ::boost::is_empty<T>::value)>
29{};
30
31} // namespace boost
