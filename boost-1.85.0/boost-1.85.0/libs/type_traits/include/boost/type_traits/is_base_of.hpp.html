9#ifndef BOOST_TT_IS_BASE_OF_HPP_INCLUDED
10#define BOOST_TT_IS_BASE_OF_HPP_INCLUDED
11
12#include <boost/type_traits/is_base_and_derived.hpp>
13#include <boost/type_traits/is_same.hpp>
14#include <boost/type_traits/is_class.hpp>
15
16namespace boost {
17
18 namespace detail{
19 template <class B, class D>
20 struct is_base_of_imp
21 {
22 typedef typename remove_cv<B>::type ncvB;
23 typedef typename remove_cv<D>::type ncvD;
24 BOOST_STATIC_CONSTANT(bool, value = (
25 (::boost::detail::is_base_and_derived_impl<ncvB,ncvD>::value) ||
26 (::boost::is_same<ncvB,ncvD>::value && ::boost::is_class<ncvB>::value)));
27 };
28 }
29
30 template <class Base, class Derived> struct is_base_of
31 : public integral_constant<bool, (::boost::detail::is_base_of_imp<Base, Derived>::value)> {};
32
33 template <class Base, class Derived> struct is_base_of<Base, Derived&> : false_type{};
34 template <class Base, class Derived> struct is_base_of<Base&, Derived&> : false_type{};
35 template <class Base, class Derived> struct is_base_of<Base&, Derived> : false_type{};
36
37} // namespace boost

39#endif // BOOST_TT_IS_BASE_AND_DERIVED_HPP_INCLUDED
