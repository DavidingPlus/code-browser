29template<template<class...> class L, class... T, template<class...> class P> struct mp_copy_if_impl<L<T...>, P>
30{
31#if BOOST_MP11_WORKAROUND( BOOST_MP11_MSVC, < 1920 )
32 template<class U> struct _f { using type = mp_if<P<U>, mp_list<U>, mp_list<>>; };
33 using type = mp_append<L<>, typename _f<T>::type...>;
34#else
35 template<class U> using _f = mp_if<P<U>, mp_list<U>, mp_list<>>;
36 using type = mp_append<L<>, _f<T>...>;
37#endif
38};
39
40} // namespace detail

42template<class L, template<class...> class P> using mp_copy_if = typename detail::mp_copy_if_impl<L, P>::type;
43template<class L, class Q> using mp_copy_if_q = mp_copy_if<L, Q::template fn>;
