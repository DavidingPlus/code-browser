206// mp_less<T1, T2>
207template<class T1, class T2> using mp_less = mp_bool<(T1::value < 0 && T2::value >= 0) || ((T1::value < T2::value) && !(T1::value >= 0 && T2::value < 0))>;
208
209#if BOOST_MP11_GCC
210# pragma GCC diagnostic pop
211#endif
