66template<class S1, class S2> struct append_integer_sequence;

68template<class T, T... I, T... J> struct append_integer_sequence<integer_sequence<T, I...>, integer_sequence<T, J...>>

73template<class T, T N> struct make_integer_sequence_impl;
74
75template<class T, T N> struct make_integer_sequence_impl_
76{
77private:
78
79 static_assert( N >= 0, "make_integer_sequence<T, N>: N must not be negative" );
80
81 static T const M = N / 2;
82 static T const R = N % 2;
83
84 using S1 = typename make_integer_sequence_impl<T, M>::type;
85 using S2 = typename append_integer_sequence<S1, S1>::type;
86 using S3 = typename make_integer_sequence_impl<T, R>::type;
87 using S4 = typename append_integer_sequence<S2, S3>::type;
88
89public:
90
91 using type = S4;
92};
93
94template<class T, T N> struct make_integer_sequence_impl: iseq_if_c<N == 0, iseq_identity<integer_sequence<T>>, iseq_if_c<N == 1, iseq_identity<integer_sequence<T, 0>>, make_integer_sequence_impl_<T, N> > >
95{
96};
