36template<class F, class Tp, std::size_t... J> BOOST_MP11_CONSTEXPR auto tuple_apply_impl( F && f, Tp && tp, integer_sequence<std::size_t, J...> )
37 -> decltype( std::forward<F>(f)( get<J>(std::forward<Tp>(tp))... ) )
38{
39 return std::forward<F>(f)( get<J>(std::forward<Tp>(tp))... );
40}

42} // namespace detail

44template<class F, class Tp,
45 class Seq = make_index_sequence<std::tuple_size<typename std::remove_reference<Tp>::type>::value>>
46BOOST_MP11_CONSTEXPR auto tuple_apply( F && f, Tp && tp )
47 -> decltype( detail::tuple_apply_impl( std::forward<F>(f), std::forward<Tp>(tp), Seq() ) )
48{
49 return detail::tuple_apply_impl( std::forward<F>(f), std::forward<Tp>(tp), Seq() );

70// tuple_for_each
71namespace detail
72{
73
74template<class Tp, std::size_t... J, class F> BOOST_MP11_CONSTEXPR F tuple_for_each_impl( Tp && tp, integer_sequence<std::size_t, J...>, F && f )
75{
76 using A = int[sizeof...(J)];
77 return (void)A{ ((void)f(get<J>(std::forward<Tp>(tp))), 0)... }, std::forward<F>(f);
78}
79
80template<class Tp, class F> BOOST_MP11_CONSTEXPR F tuple_for_each_impl( Tp && /*tp*/, integer_sequence<std::size_t>, F && f )
81{
82 return std::forward<F>(f);
83}
84
85} // namespace detail

87template<class Tp, class F> BOOST_MP11_CONSTEXPR F tuple_for_each( Tp && tp, F && f )
88{
89 using seq = make_index_sequence<std::tuple_size<typename std::remove_reference<Tp>::type>::value>;
90 return detail::tuple_for_each_impl( std::forward<Tp>(tp), seq(), std::forward<F>(f) );
91}

98// std::forward_as_tuple is not constexpr in C++11 or libstdc++ 5.x
99template<class... T> BOOST_MP11_CONSTEXPR auto tp_forward_r( T&&... t ) -> std::tuple<T&&...>
100{
101 return std::tuple<T&&...>( std::forward<T>( t )... );
102}
103
104template<class... T> BOOST_MP11_CONSTEXPR auto tp_forward_v( T&&... t ) -> std::tuple<T...>
105{
106 return std::tuple<T...>( std::forward<T>( t )... );
107}
108
109template<std::size_t J, class... Tp>
110BOOST_MP11_CONSTEXPR auto tp_extract( Tp&&... tp )
111 -> decltype( tp_forward_r( get<J>( std::forward<Tp>( tp ) )... ) )
112{
113 return tp_forward_r( get<J>( std::forward<Tp>( tp ) )... );
114}
