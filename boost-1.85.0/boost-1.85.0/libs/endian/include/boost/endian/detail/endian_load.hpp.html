32// Requires:
33//
34// sizeof(T) must be 1, 2, 4, or 8
35// 1 <= N <= sizeof(T)
36// T is TriviallyCopyable
37// if N < sizeof(T), T is integral or enum
38
39template<class T, std::size_t N, order Order>
40inline T endian_load( unsigned char const * p ) BOOST_NOEXCEPT
41{
42 BOOST_ENDIAN_STATIC_ASSERT( sizeof(T) == 1 || sizeof(T) == 2 || sizeof(T) == 4 || sizeof(T) == 8 );
43 BOOST_ENDIAN_STATIC_ASSERT( N >= 1 && N <= sizeof(T) );
44
45 return detail::endian_load_impl<T, sizeof(T), order::native, N, Order>()( p );
46}

51// same endianness, same size
52
53template<class T, std::size_t N, order O> struct endian_load_impl<T, N, O, N, O>
54{
55 inline T operator()( unsigned char const * p ) const BOOST_NOEXCEPT
56 {
57 BOOST_ENDIAN_STATIC_ASSERT( is_trivially_copyable<T>::value );
58
59 T t;
60 std::memcpy( dest: &t, src: p, n: N );
61 return t;
62 }
63};
64
65// same size, reverse endianness
66
67template<class T, std::size_t N, order O1, order O2> struct endian_load_impl<T, N, O1, N, O2>
68{
69 inline T operator()( unsigned char const * p ) const BOOST_NOEXCEPT
70 {
71 BOOST_ENDIAN_STATIC_ASSERT( is_trivially_copyable<T>::value );
72
73 typename integral_by_size<N>::type tmp;
74 std::memcpy( dest: &tmp, src: p, n: N );
75
76 endian_reverse_inplace( tmp );
77
78 T t;
79 std::memcpy( dest: &t, src: &tmp, n: N );
80 return t;
81 }
82};

86template<class T, order Order> struct endian_load_impl<T, 2, Order, 1, order::little>
87{
88 inline T operator()( unsigned char const * p ) const BOOST_NOEXCEPT
89 {
90 BOOST_ENDIAN_STATIC_ASSERT( std::is_integral<T>::value || std::is_enum<T>::value );
91
92 unsigned char tmp[ 2 ];
93
94 tmp[0] = p[0];
95 tmp[1] = std::is_signed<T>::value && ( p[0] & 0x80 )? 0xFF: 0x00;
96
97 return boost::endian::endian_load<T, 2, order::little>( tmp );
98 }
99};
