24// http://ericniebler.com/2013/08/07/universal-references-and-the-copy-constructo/
25template< typename X, typename Y >
26using disable_overload =
27 typename std::enable_if<
28 ! std::is_base_of<
29 X,
30 typename std::decay< Y >::type
31 >::value
32 >::type;
33
34}}}
