49template <typename T>
50using identity_t = T;
51
52template <typename Pattern>
53struct ThisRefAdjuster {
54 template <typename T>
55 using AdjustT = typename std::conditional<
56 std::is_const<typename std::remove_reference<Pattern>::type>::value,
57 typename std::conditional<std::is_lvalue_reference<Pattern>::value,
58 const T&, const T&&>::type,
59 typename std::conditional<std::is_lvalue_reference<Pattern>::value, T&,
60 T&&>::type>::type;
61
62 template <typename MockType>
63 static AdjustT<MockType> Adjust(const MockType& mock) {
64 return static_cast<AdjustT<MockType>>(const_cast<MockType&>(mock));
65 }
66};

68constexpr bool PrefixOf(const char* a, const char* b) {
69 return *a == 0 || (*a == *b && internal::PrefixOf(a: a + 1, b: b + 1));
70}
71
72template <int N, int M>
73constexpr bool StartsWith(const char (&prefix)[N], const char (&str)[M]) {
74 return N <= M && internal::PrefixOf(a: prefix, b: str);
75}
76
77template <int N, int M>
78constexpr bool EndsWith(const char (&suffix)[N], const char (&str)[M]) {
79 return N <= M && internal::PrefixOf(a: suffix, b: str + M - N);
80}
81
82template <int N, int M>
83constexpr bool Equals(const char (&a)[N], const char (&b)[M]) {
84 return N == M && internal::PrefixOf(a, b);
85}
86
87template <int N>
88constexpr bool ValidateSpec(const char (&spec)[N]) {
89 return internal::Equals("const", spec) ||
90 internal::Equals("override", spec) ||
91 internal::Equals("final", spec) ||
92 internal::Equals("noexcept", spec) ||
93 (internal::StartsWith("noexcept(", spec) &&
94 internal::EndsWith(")", spec)) ||
95 internal::Equals("ref(&)", spec) ||
96 internal::Equals("ref(&&)", spec) ||
97 (internal::StartsWith("Calltype(", spec) &&
98 internal::EndsWith(")", spec));
99}

146#define GMOCK_INTERNAL_WRONG_ARITY(...) \
147 static_assert( \
148 false, \
149 "MOCK_METHOD must be called with 3 or 4 arguments. _Ret, " \
150 "_MethodName, _Args and optionally _Spec. _Args and _Spec must be " \
151 "enclosed in parentheses. If _Ret is a type with unprotected commas, " \
152 "it must also be enclosed in parentheses.")

154#define GMOCK_INTERNAL_ASSERT_PARENTHESIS(_Tuple) \

159#define GMOCK_INTERNAL_ASSERT_VALID_SIGNATURE(_N, ...) \
160 static_assert( \
161 std::is_function<__VA_ARGS__>::value, \
162 "Signature must be a function type, maybe return type contains " \
163 "unprotected comma."); \
164 static_assert( \
165 ::testing::tuple_size<typename ::testing::internal::Function< \
166 __VA_ARGS__>::ArgumentTuple>::value == _N, \
167 "This method does not take " GMOCK_PP_STRINGIZE( \
168 _N) " arguments. Parenthesize all types with unprotected commas.")

171 GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_ASSERT_VALID_SPEC_ELEMENT, ~, _Spec)

173#define GMOCK_INTERNAL_MOCK_METHOD_IMPL(_N, _MethodName, _Constness, \
174 _Override, _Final, _NoexceptSpec, \
175 _CallType, _RefSpec, _Signature) \
176 typename ::testing::internal::Function<GMOCK_PP_REMOVE_PARENS( \
177 _Signature)>::Result \
178 GMOCK_INTERNAL_EXPAND(_CallType) \
179 _MethodName(GMOCK_PP_REPEAT(GMOCK_INTERNAL_PARAMETER, _Signature, _N)) \
180 GMOCK_PP_IF(_Constness, const, ) _RefSpec _NoexceptSpec \
181 GMOCK_PP_IF(_Override, override, ) GMOCK_PP_IF(_Final, final, ) { \
182 GMOCK_MOCKER_(_N, _Constness, _MethodName) \
183 .SetOwnerAndName(this, #_MethodName); \
184 return GMOCK_MOCKER_(_N, _Constness, _MethodName) \
185 .Invoke(GMOCK_PP_REPEAT(GMOCK_INTERNAL_FORWARD_ARG, _Signature, _N)); \
186 } \
187 ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \
188 GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_PARAMETER, _Signature, _N)) \
189 GMOCK_PP_IF(_Constness, const, ) _RefSpec { \
190 GMOCK_MOCKER_(_N, _Constness, _MethodName).RegisterOwner(this); \
191 return GMOCK_MOCKER_(_N, _Constness, _MethodName) \
192 .With(GMOCK_PP_REPEAT(GMOCK_INTERNAL_MATCHER_ARGUMENT, , _N)); \
193 } \
194 ::testing::MockSpec<GMOCK_PP_REMOVE_PARENS(_Signature)> gmock_##_MethodName( \
195 const ::testing::internal::WithoutMatchers&, \
196 GMOCK_PP_IF(_Constness, const, )::testing::internal::Function< \
197 GMOCK_PP_REMOVE_PARENS(_Signature)>*) const _RefSpec _NoexceptSpec { \
198 return ::testing::internal::ThisRefAdjuster<GMOCK_PP_IF( \
199 _Constness, const, ) int _RefSpec>::Adjust(*this) \
200 .gmock_##_MethodName(GMOCK_PP_REPEAT( \
201 GMOCK_INTERNAL_A_MATCHER_ARGUMENT, _Signature, _N)); \
202 } \
203 mutable ::testing::FunctionMocker<GMOCK_PP_REMOVE_PARENS(_Signature)> \
204 GMOCK_MOCKER_(_N, _Constness, _MethodName)

297// Note: The use of `identity_t` here allows _Ret to represent return types that
298// would normally need to be specified in a different way. For example, a method
299// returning a function pointer must be written as
300//
301// fn_ptr_return_t (*method(method_args_t...))(fn_ptr_args_t...)
302//
303// But we only support placing the return type at the beginning. To handle this,
304// we wrap all calls in identity_t, so that a declaration will be expanded to
305//
306// identity_t<fn_ptr_return_t (*)(fn_ptr_args_t...)> method(method_args_t...)
307//
308// This allows us to work around the syntactic oddities of function/method
309// types.
310#define GMOCK_INTERNAL_SIGNATURE(_Ret, _Args) \
311 ::testing::internal::identity_t<GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_Ret), \
312 GMOCK_PP_REMOVE_PARENS, \
313 GMOCK_PP_IDENTITY)(_Ret)>( \
314 GMOCK_PP_FOR_EACH(GMOCK_INTERNAL_GET_TYPE, _, _Args))
315
316#define GMOCK_INTERNAL_GET_TYPE(_i, _, _elem) \
317 GMOCK_PP_COMMA_IF(_i) \
318 GMOCK_PP_IF(GMOCK_PP_IS_BEGIN_PARENS(_elem), GMOCK_PP_REMOVE_PARENS, \
319 GMOCK_PP_IDENTITY) \
320 (_elem)
321
322#define GMOCK_INTERNAL_PARAMETER(_i, _Signature, _) \
323 GMOCK_PP_COMMA_IF(_i) \
324 GMOCK_INTERNAL_ARG_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature)) \
325 gmock_a##_i
326
327#define GMOCK_INTERNAL_FORWARD_ARG(_i, _Signature, _) \
328 GMOCK_PP_COMMA_IF(_i) \
329 ::std::forward<GMOCK_INTERNAL_ARG_O( \
330 _i, GMOCK_PP_REMOVE_PARENS(_Signature))>(gmock_a##_i)
331
332#define GMOCK_INTERNAL_MATCHER_PARAMETER(_i, _Signature, _) \
333 GMOCK_PP_COMMA_IF(_i) \
334 GMOCK_INTERNAL_MATCHER_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature)) \
335 gmock_a##_i
336
337#define GMOCK_INTERNAL_MATCHER_ARGUMENT(_i, _1, _2) \
338 GMOCK_PP_COMMA_IF(_i) \
339 gmock_a##_i
340
341#define GMOCK_INTERNAL_A_MATCHER_ARGUMENT(_i, _Signature, _) \
342 GMOCK_PP_COMMA_IF(_i) \
343 ::testing::A<GMOCK_INTERNAL_ARG_O(_i, GMOCK_PP_REMOVE_PARENS(_Signature))>()
