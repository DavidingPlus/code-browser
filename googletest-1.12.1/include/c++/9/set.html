74#if __cplusplus >= 201703L
75namespace std _GLIBCXX_VISIBILITY(default)
76{
77_GLIBCXX_BEGIN_NAMESPACE_VERSION
78 namespace pmr
79 {
80 template<typename _Tp> class polymorphic_allocator;
81 template<typename _Key, typename _Cmp = std::less<_Key>>
82 using set = std::set<_Key, _Cmp, polymorphic_allocator<_Key>>;
83 template<typename _Key, typename _Cmp = std::less<_Key>>
84 using multiset = std::multiset<_Key, _Cmp, polymorphic_allocator<_Key>>;
85 } // namespace pmr
86_GLIBCXX_END_NAMESPACE_VERSION
87} // namespace std
88#endif // C++17

90#if __cplusplus > 201703L
91namespace std _GLIBCXX_VISIBILITY(default)
92{
93_GLIBCXX_BEGIN_NAMESPACE_VERSION
94 template<typename _Key, typename _Compare, typename _Alloc,
95 typename _Predicate>
96 inline typename set<_Key, _Compare, _Alloc>::size_type
97 erase_if(set<_Key, _Compare, _Alloc>& __cont, _Predicate __pred)
98 { return __detail::__erase_nodes_if(__cont, __pred); }
99
100 template<typename _Key, typename _Compare, typename _Alloc,
101 typename _Predicate>
102 inline typename multiset<_Key, _Compare, _Alloc>::size_type
103 erase_if(multiset<_Key, _Compare, _Alloc>& __cont, _Predicate __pred)
104 { return __detail::__erase_nodes_if(__cont, __pred); }
105_GLIBCXX_END_NAMESPACE_VERSION
106} // namespace std
107#endif // C++20
