108#define __cpp_lib_erase_if 202002L
109
110 template<typename _Tp, typename _Alloc, typename _Predicate>
111 inline typename vector<_Tp, _Alloc>::size_type
112 erase_if(vector<_Tp, _Alloc>& __cont, _Predicate __pred)
113 {
114 const auto __osz = __cont.size();
115 __cont.erase(std::remove_if(__cont.begin(), __cont.end(), __pred),
116 __cont.end());
117 return __osz - __cont.size();
118 }
119
120 template<typename _Tp, typename _Alloc, typename _Up>
121 inline typename vector<_Tp, _Alloc>::size_type
122 erase(vector<_Tp, _Alloc>& __cont, const _Up& __value)
123 {
124 const auto __osz = __cont.size();
125 __cont.erase(std::remove(__cont.begin(), __cont.end(), __value),
126 __cont.end());
127 return __osz - __cont.size();
128 }
