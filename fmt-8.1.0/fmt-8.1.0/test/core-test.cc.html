55// Check string_view's comparison operator.
56template <template <typename> class Op> void check_op() {
57 const char* inputs[] = {"foo", "fop", "fo"};
58 size_t num_inputs = sizeof(inputs) / sizeof(*inputs);
59 for (size_t i = 0; i < num_inputs; ++i) {
60 for (size_t j = 0; j < num_inputs; ++j) {
61 string_view lhs(inputs[i]), rhs(inputs[j]);
62 EXPECT_EQ(Op<int>()(lhs.compare(rhs), 0), Op<string_view>()(lhs, rhs));
63 }
64 }
65}

71 EXPECT_GT(string_view("foo").compare(string_view("fo")), 0);
72 EXPECT_LT(string_view("fo").compare(string_view("foo")), 0);
73 check_op<std::equal_to>();
74 check_op<std::not_equal_to>();
75 check_op<std::less>();
76 check_op<std::less_equal>();
77 check_op<std::greater>();
78 check_op<std::greater_equal>();

151template <typename T> struct mock_buffer final : buffer<T> {
152 MOCK_METHOD1(do_grow, size_t(size_t capacity));
153
154 void grow(size_t capacity) override {
155 this->set(this->data(), do_grow(gmock_a0: capacity));
156 }
157
158 mock_buffer(T* data = nullptr, size_t buf_capacity = 0) {
159 this->set(data, buf_capacity);
160 ON_CALL(*this, do_grow(_)).WillByDefault(Invoke([](size_t capacity) {
161 return capacity;
162 }));
163 }
164};

367#define CHECK_ARG(Char, expected, value) \
368 { \
369 testing::StrictMock<mock_visitor<decltype(expected)>> visitor; \
370 EXPECT_CALL(visitor, visit(expected)); \
371 using iterator = std::back_insert_iterator<buffer<Char>>; \
372 fmt::visit_format_arg( \
373 visitor, \
374 fmt::detail::make_arg<fmt::basic_format_context<iterator, Char>>( \
375 value)); \
376 }
377
378#define CHECK_ARG_SIMPLE(value) \
379 { \
380 using value_type = decltype(value); \
381 typename visit_type<value_type>::type expected = value; \
382 CHECK_ARG(char, expected, value) \
383 CHECK_ARG(wchar_t, expected, value) \
384 }
