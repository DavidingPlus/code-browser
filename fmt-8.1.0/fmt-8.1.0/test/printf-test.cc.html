25// Makes format string argument positional.
26static std::string make_positional(fmt::string_view format) {
27 std::string s(format.data(), format.size());
28 s.replace(pos: s.find(c: '%'), n1: 1, s: "%1$");
29 return s;
30}
31
32static std::wstring make_positional(fmt::basic_string_view<wchar_t> format) {
33 std::wstring s(format.data(), format.size());
34 s.replace(pos: s.find(c: L'%'), n1: 1, s: L"%1$");
35 return s;
36}

38// A wrapper around fmt::sprintf to workaround bogus warnings about invalid
39// format strings in MSVC.
40template <typename... Args>
41std::string test_sprintf(fmt::string_view format, const Args&... args) {
42 return fmt::sprintf(format, args...);
43}
44template <typename... Args>
45std::wstring test_sprintf(fmt::basic_string_view<wchar_t> format,
46 const Args&... args) {
47 return fmt::sprintf(format, args...);
48}
49
50#define EXPECT_PRINTF(expected_output, format, arg) \
51 EXPECT_EQ(expected_output, test_sprintf(format, arg)) \
52 << "format: " << format; \
53 EXPECT_EQ(expected_output, fmt::sprintf(make_positional(format), arg))

96 "cannot switch from manual to automatic argument indexing");

100 "cannot switch from manual to automatic argument indexing");
101
102 EXPECT_THROW_MSG(test_sprintf("%d%1$", 1, 2), format_error,
103 "cannot switch from automatic to manual argument indexing");
104 EXPECT_THROW_MSG(test_sprintf(format("%d%{}$d", big_num), 1, 2), format_error,
105 "cannot switch from automatic to manual argument indexing");
106 EXPECT_THROW_MSG(test_sprintf("%d%1$d", 1, 2), format_error,
107 "cannot switch from automatic to manual argument indexing");
108
109 // Indexing errors override width errors.

328// Test length format specifier ``length_spec``.
329template <typename T, typename U>
330void test_length(const char* length_spec, U value) {
331 long long signed_value = 0;
332 unsigned long long unsigned_value = 0;
333 // Apply integer promotion to the argument.
334 unsigned long long max = max_value<U>();
335 using fmt::detail::const_check;
336 if (const_check(max <= static_cast<unsigned>(max_value<int>()))) {
337 signed_value = static_cast<int>(value);
338 unsigned_value = static_cast<unsigned long long>(value);
339 } else if (const_check(max <= max_value<unsigned>())) {
340 signed_value = static_cast<unsigned>(value);
341 unsigned_value = static_cast<unsigned long long>(value);
342 }
343 if (sizeof(U) <= sizeof(int) && sizeof(int) < sizeof(T)) {
344 signed_value = static_cast<long long>(value);
345 unsigned_value = static_cast<unsigned long long>(
346 static_cast<typename std::make_unsigned<unsigned>::type>(value));
347 } else {
348 signed_value = static_cast<typename make_signed<T>::type>(value);
349 unsigned_value = static_cast<typename std::make_unsigned<T>::type>(value);
350 }
351 std::ostringstream os;
352 os << signed_value;
353 EXPECT_PRINTF(os.str(), fmt::format("%{}d", length_spec), value);
354 EXPECT_PRINTF(os.str(), fmt::format("%{}i", length_spec), value);
355 os.str(s: "");
356 os << unsigned_value;
357 EXPECT_PRINTF(os.str(), fmt::format("%{}u", length_spec), value);
358 os.str(s: "");
359 os << std::oct << unsigned_value;
360 EXPECT_PRINTF(os.str(), fmt::format("%{}o", length_spec), value);
361 os.str(s: "");
362 os << std::hex << unsigned_value;
363 EXPECT_PRINTF(os.str(), fmt::format("%{}x", length_spec), value);
364 os.str(s: "");
365 os << std::hex << std::uppercase << unsigned_value;
366 EXPECT_PRINTF(os.str(), fmt::format("%{}X", length_spec), value);
367}

369template <typename T> void test_length(const char* length_spec) {
370 T min = std::numeric_limits<T>::min(), max = max_value<T>();
371 test_length<T>(length_spec, 42);
372 test_length<T>(length_spec, -42);
373 test_length<T>(length_spec, min);
374 test_length<T>(length_spec, max);
375 long long long_long_min = std::numeric_limits<long long>::min();
376 if (static_cast<long long>(min) > long_long_min)
377 test_length<T>(length_spec, static_cast<long long>(min) - 1);
378 unsigned long long long_long_max = max_value<long long>();
379 if (static_cast<unsigned long long>(max) < long_long_max)
380 test_length<T>(length_spec, static_cast<long long>(max) + 1);
381 test_length<T>(length_spec, std::numeric_limits<short>::min());
382 test_length<T>(length_spec, max_value<unsigned short>());
383 test_length<T>(length_spec, std::numeric_limits<int>::min());
384 test_length<T>(length_spec, max_value<int>());
385 test_length<T>(length_spec, std::numeric_limits<unsigned>::min());
386 test_length<T>(length_spec, max_value<unsigned>());
387 test_length<T>(length_spec, std::numeric_limits<long long>::min());
388 test_length<T>(length_spec, max_value<long long>());
389 test_length<T>(length_spec, std::numeric_limits<unsigned long long>::min());
390 test_length<T>(length_spec, max_value<unsigned long long>());
391}
