14output_redirect::output_redirect(FILE* f) : file_(f) {
15 flush();
16 int fd = FMT_POSIX(fileno(f));
17 // Create a file object referring to the original file.
18 original_ = file::dup(fd);
19 // Create a pipe.
20 file write_end;
21 file::pipe(read_end&: read_end_, write_end);
22 // Connect the passed FILE object to the write end of the pipe.
23 write_end.dup2(fd);
24}
25
26output_redirect::~output_redirect() FMT_NOEXCEPT {
27 try {
28 restore();
29 } catch (const std::exception& e) {
30 std::fputs(s: e.what(), stderr);
31 }
32}

34void output_redirect::flush() {
35 int result = 0;
36 do {
37 result = fflush(stream: file_);
38 } while (result == EOF && errno == EINTR);
39 if (result != 0) throw fmt::system_error(errno, fmt: "cannot flush stream");
40}
41
42void output_redirect::restore() {
43 if (original_.descriptor() == -1) return; // Already restored.
44 flush();
45 // Restore the original file.
46 original_.dup2(FMT_POSIX(fileno(file_)));
47 original_.close();
48}

50std::string output_redirect::restore_and_read() {
51 // Restore output.
52 restore();
53
54 // Read everything from the pipe.
55 std::string content;
56 if (read_end_.descriptor() == -1) return content; // Already read.
57 enum { BUFFER_SIZE = 4096 };
58 char buffer[BUFFER_SIZE];
59 size_t count = 0;
60 do {
61 count = read_end_.read(buffer, count: BUFFER_SIZE);
62 content.append(s: buffer, n: count);
63 } while (count != 0);
64 read_end_.close();
65 return content;
66}
67
68std::string read(file& f, size_t count) {
69 std::string buffer(count, '\0');
70 size_t n = 0, offset = 0;
71 do {
72 n = f.read(buffer: &buffer[offset], count: count - offset);
73 // We can't read more than size_t bytes since count has type size_t.
74 offset += n;
75 } while (offset < count && n != 0);
76 buffer.resize(n: offset);
77 return buffer;
78}
