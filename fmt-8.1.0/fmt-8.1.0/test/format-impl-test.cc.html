32TEST(bigint_test, compare) {
33 bigint n1(42);
34 bigint n2(42);
35 EXPECT_EQ(compare(n1, n2), 0);

206TEST(fp_test, get_cached_power) {
207 using limits = std::numeric_limits<double>;
208 for (auto exp = limits::min_exponent; exp <= limits::max_exponent; ++exp) {
209 int dec_exp = 0;
210 auto fp = fmt::detail::get_cached_power(min_exponent: exp, pow10_exponent&: dec_exp);
211 bigint exact, cache(fp.f);
212 if (dec_exp >= 0) {
213 exact.assign_pow10(exp: dec_exp);
214 if (fp.e <= 0)
215 exact <<= -fp.e;
216 else
217 cache <<= fp.e;
218 exact.align(other: cache);
219 cache.align(other: exact);
220 auto exact_str = fmt::format(fmt: "{}", args&: exact);
221 auto cache_str = fmt::format(fmt: "{}", args&: cache);
222 EXPECT_EQ(exact_str.size(), cache_str.size());
223 EXPECT_EQ(exact_str.substr(0, 15), cache_str.substr(0, 15));
224 int diff = cache_str[15] - exact_str[15];
225 if (diff == 1)
226 EXPECT_GT(exact_str[16], '8');
227 else
228 EXPECT_EQ(diff, 0);
229 } else {
230 cache.assign_pow10(exp: -dec_exp);
231 cache *= fp.f + 1; // Inexact check.
232 exact.assign(n: 1);
233 exact <<= -fp.e;
234 exact.align(other: cache);
235 auto exact_str = fmt::format(fmt: "{}", args&: exact);
236 auto cache_str = fmt::format(fmt: "{}", args&: cache);
237 EXPECT_EQ(exact_str.size(), cache_str.size());
238 EXPECT_EQ(exact_str.substr(0, 16), cache_str.substr(0, 16));
239 }
240 }

256TEST(fp_test, get_round_direction) {
257 using fmt::detail::get_round_direction;
258 using fmt::detail::round_direction;
259 EXPECT_EQ(round_direction::down, get_round_direction(100, 50, 0));
260 EXPECT_EQ(round_direction::up, get_round_direction(100, 51, 0));
261 EXPECT_EQ(round_direction::down, get_round_direction(100, 40, 10));
262 EXPECT_EQ(round_direction::up, get_round_direction(100, 60, 10));
263 for (size_t i = 41; i < 60; ++i)
264 EXPECT_EQ(round_direction::unknown, get_round_direction(100, i, 10));
265 uint64_t max = max_value<uint64_t>();
266 EXPECT_THROW(get_round_direction(100, 100, 0), assertion_failure);
267 EXPECT_THROW(get_round_direction(100, 0, 100), assertion_failure);
268 EXPECT_THROW(get_round_direction(100, 0, 50), assertion_failure);
269 // Check that remainder + error doesn't overflow.
270 EXPECT_EQ(round_direction::up, get_round_direction(max, max - 1, 2));
271 // Check that 2 * (remainder + error) doesn't overflow.
272 EXPECT_EQ(round_direction::unknown,
273 get_round_direction(max, max / 2 + 1, max / 2));
274 // Check that remainder - error doesn't overflow.
275 EXPECT_EQ(round_direction::unknown, get_round_direction(100, 40, 41));
276 // Check that 2 * (remainder - error) doesn't overflow.
277 EXPECT_EQ(round_direction::up, get_round_direction(max, max - 1, 1));
278}

280TEST(fp_test, fixed_handler) {
281 struct handler : fmt::detail::gen_digits_handler {
282 char buffer[10];
283 handler(int prec = 0) : fmt::detail::gen_digits_handler() {
284 buf = buffer;
285 precision = prec;
286 }
287 };
288 handler().on_digit(digit: '0', divisor: 100, remainder: 99, error: 0, integral: false);
289 EXPECT_THROW(handler().on_digit('0', 100, 100, 0, false), assertion_failure);
290 namespace digits = fmt::detail::digits;
291 EXPECT_EQ(handler(1).on_digit('0', 100, 10, 10, false), digits::error);
292 // Check that divisor - error doesn't overflow.
293 EXPECT_EQ(handler(1).on_digit('0', 100, 10, 101, false), digits::error);
294 // Check that 2 * error doesn't overflow.
295 uint64_t max = max_value<uint64_t>();
296 EXPECT_EQ(handler(1).on_digit('0', max, 10, max - 1, false), digits::error);
297}

304TEST(format_impl_test, format_error_code) {
305 std::string msg = "error 42", sep = ": ";
306 {
307 fmt::memory_buffer buffer;
308 format_to(out: fmt::appender(buffer), fmt: "garbage");
309 fmt::detail::format_error_code(out&: buffer, error_code: 42, message: "test");
310 EXPECT_EQ("test: " + msg, to_string(buffer));
311 }
312 {
313 fmt::memory_buffer buffer;
314 auto prefix =
315 std::string(fmt::inline_buffer_size - msg.size() - sep.size() + 1, 'x');
316 fmt::detail::format_error_code(out&: buffer, error_code: 42, message: prefix);
317 EXPECT_EQ(msg, to_string(buffer));
318 }
319 int codes[] = {42, -1};
320 for (size_t i = 0, n = sizeof(codes) / sizeof(*codes); i < n; ++i) {
321 // Test maximum buffer size.
322 msg = fmt::format(fmt: "error {}", args&: codes[i]);
323 fmt::memory_buffer buffer;
324 auto prefix =
325 std::string(fmt::inline_buffer_size - msg.size() - sep.size(), 'x');
326 fmt::detail::format_error_code(out&: buffer, error_code: codes[i], message: prefix);
327 EXPECT_EQ(prefix + sep + msg, to_string(buffer));
328 size_t size = fmt::inline_buffer_size;
329 EXPECT_EQ(size, buffer.size());
330 buffer.resize(count: 0);
331 // Test with a message that doesn't fit into the buffer.
332 prefix += 'x';
333 fmt::detail::format_error_code(out&: buffer, error_code: codes[i], message: prefix);
334 EXPECT_EQ(msg, to_string(buffer));
335 }
336}

345// Tests fmt::detail::count_digits for integer type Int.
346template <typename Int> void test_count_digits() {
347 for (Int i = 0; i < 10; ++i) EXPECT_EQ(1u, fmt::detail::count_digits(i));
348 for (Int i = 1, n = 1, end = max_value<Int>() / 10; n <= end; ++i) {
349 n *= 10;
350 EXPECT_EQ(i, fmt::detail::count_digits(n - 1));
351 EXPECT_EQ(i + 1, fmt::detail::count_digits(n));
352 }
353}
