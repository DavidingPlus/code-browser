79#if defined(__cpp_if_constexpr) && defined(__cpp_return_type_deduction)
80struct test_formattable {};
81
82FMT_BEGIN_NAMESPACE
83template <> struct formatter<test_formattable> : formatter<const char*> {
84 char word_spec = 'f';
85 constexpr auto parse(format_parse_context& ctx) {
86 auto it = ctx.begin(), end = ctx.end();
87 if (it == end || *it == '}') return it;
88 if (it != end && (*it == 'f' || *it == 'b')) word_spec = *it++;
89 if (it != end && *it != '}') throw format_error("invalid format");
90 return it;
91 }
92 template <typename FormatContext>
93 constexpr auto format(test_formattable, FormatContext& ctx) const
94 -> decltype(ctx.out()) {
95 return formatter<const char*>::format(word_spec == 'f' ? "foo" : "bar",
96 ctx);
97 }
98};
99FMT_END_NAMESPACE

139TEST(compile_test, manual_ordering) {
140 EXPECT_EQ("42", fmt::format(FMT_COMPILE("{0}"), 42));
141 EXPECT_EQ(" -42", fmt::format(FMT_COMPILE("{0:4}"), -42));
142 EXPECT_EQ("41 43", fmt::format(FMT_COMPILE("{0} {1}"), 41, 43));
143 EXPECT_EQ("41 43", fmt::format(FMT_COMPILE("{1} {0}"), 43, 41));
144 EXPECT_EQ("41 43", fmt::format(FMT_COMPILE("{0} {2}"), 41, 42, 43));
145 EXPECT_EQ(" 41 43", fmt::format(FMT_COMPILE("{1:{2}} {0:4}"), 43, 41, 4));
146 EXPECT_EQ("42 1.2 ms ",
147 fmt::format(FMT_COMPILE("{0} {1:7.1%Q %q}"), 42,
148 std::chrono::duration<double, std::milli>(1.234)));
149 EXPECT_EQ(
150 "true 42 42 foo 0x1234 foo",
151 fmt::format(FMT_COMPILE("{0} {1} {2} {3} {4} {5}"), true, 42, 42.0f,
152 "foo", reinterpret_cast<void*>(0x1234), test_formattable()));
153 EXPECT_EQ(L"42", fmt::format(FMT_COMPILE(L"{0}"), 42));
154}

156TEST(compile_test, named) {
157 auto runtime_named_field_compiled =
158 fmt::detail::compile<decltype(fmt::arg("arg", 42))>(FMT_COMPILE("{arg}"));
159 static_assert(std::is_same_v<decltype(runtime_named_field_compiled),
160 fmt::detail::runtime_named_field<char>>);
161
162 EXPECT_EQ("42", fmt::format(FMT_COMPILE("{}"), fmt::arg("arg", 42)));
163 EXPECT_EQ("41 43", fmt::format(FMT_COMPILE("{} {}"), fmt::arg("arg", 41),
164 fmt::arg("arg", 43)));
165
166 EXPECT_EQ("foobar",
167 fmt::format(FMT_COMPILE("{a0}{a1}"), fmt::arg("a0", "foo"),
168 fmt::arg("a1", "bar")));
169 EXPECT_EQ("foobar", fmt::format(FMT_COMPILE("{}{a1}"), fmt::arg("a0", "foo"),
170 fmt::arg("a1", "bar")));
171 EXPECT_EQ("foofoo", fmt::format(FMT_COMPILE("{a0}{}"), fmt::arg("a0", "foo"),
172 fmt::arg("a1", "bar")));
173 EXPECT_EQ("foobar", fmt::format(FMT_COMPILE("{0}{a1}"), fmt::arg("a0", "foo"),
174 fmt::arg("a1", "bar")));
175 EXPECT_EQ("foobar", fmt::format(FMT_COMPILE("{a0}{1}"), fmt::arg("a0", "foo"),
176 fmt::arg("a1", "bar")));
177
178 EXPECT_EQ("foobar",
179 fmt::format(FMT_COMPILE("{}{a1}"), "foo", fmt::arg("a1", "bar")));
180 EXPECT_EQ("foobar",
181 fmt::format(FMT_COMPILE("{a0}{a1}"), fmt::arg("a1", "bar"),
182 fmt::arg("a2", "baz"), fmt::arg("a0", "foo")));
183 EXPECT_EQ(" bar foo ",
184 fmt::format(FMT_COMPILE(" {foo} {bar} "), fmt::arg("foo", "bar"),
185 fmt::arg("bar", "foo")));
186
187 EXPECT_THROW(fmt::format(FMT_COMPILE("{invalid}"), fmt::arg("valid", 42)),
188 fmt::format_error);
189
190# if FMT_USE_NONTYPE_TEMPLATE_PARAMETERS
191 using namespace fmt::literals;
192 auto statically_named_field_compiled =
193 fmt::detail::compile<decltype("arg"_a = 42)>(FMT_COMPILE("{arg}"));
194 static_assert(std::is_same_v<decltype(statically_named_field_compiled),
195 fmt::detail::field<char, int, 0>>);
196
197 EXPECT_EQ("41 43",
198 fmt::format(FMT_COMPILE("{a0} {a1}"), "a0"_a = 41, "a1"_a = 43));
199 EXPECT_EQ("41 43",
200 fmt::format(FMT_COMPILE("{a1} {a0}"), "a0"_a = 43, "a1"_a = 41));
201# endif
202}

235TEST(compile_test, unknown_format_fallback) {
236 EXPECT_EQ(" 42 ",
237 fmt::format(FMT_COMPILE("{name:^4}"), fmt::arg("name", 42)));
238
239 std::vector<char> v;
240 fmt::format_to(std::back_inserter(v), FMT_COMPILE("{name:^4}"),
241 fmt::arg("name", 42));
242 EXPECT_EQ(" 42 ", fmt::string_view(v.data(), v.size()));
243
244 char buffer[4];
245 auto result = fmt::format_to_n(buffer, 4, FMT_COMPILE("{name:^5}"),
246 fmt::arg("name", 42));
247 EXPECT_EQ(5u, result.size);
248 EXPECT_EQ(buffer + 4, result.out);
249 EXPECT_EQ(" 42 ", fmt::string_view(buffer, 4));
250}

292#if __cplusplus >= 202002L || \
293 (__cplusplus >= 201709L && FMT_GCC_VERSION >= 1002)
294template <size_t max_string_length, typename Char = char> struct test_string {
295 template <typename T> constexpr bool operator==(const T& rhs) const noexcept {
296 return fmt::basic_string_view<Char>(rhs).compare(buffer) == 0;
297 }
298 Char buffer[max_string_length]{};
299};
300
301template <size_t max_string_length, typename Char = char, typename... Args>
302consteval auto test_format(auto format, const Args&... args) {
303 test_string<max_string_length, Char> string{};
304 fmt::format_to(string.buffer, format, args...);
305 return string;
306}

322 EXPECT_EQ("+42", test_format<4>(FMT_COMPILE("{:+}"), 42));
323 EXPECT_EQ("42", test_format<3>(FMT_COMPILE("{:-}"), 42));
