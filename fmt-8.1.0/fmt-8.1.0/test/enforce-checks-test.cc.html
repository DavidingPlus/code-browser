18// Exercise the API to verify that everything we expect to can compile.
19void test_format_api() {
20 (void)fmt::format(FMT_STRING("{}"), args: 42);
21 (void)fmt::format(FMT_STRING(L"{}"), args: 42);
22 (void)fmt::format(FMT_STRING("noop"));
23
24 (void)fmt::to_string(value: 42);
25 (void)fmt::to_wstring(value: 42);
26
27 std::vector<char> out;
28 fmt::format_to(out: std::back_inserter(x&: out), FMT_STRING("{}"), args: 42);
29
30 char buffer[4];
31 fmt::format_to_n(out: buffer, n: 3, FMT_STRING("{}"), args: 12345);
32
33 wchar_t wbuffer[4];
34 fmt::format_to_n(out: wbuffer, n: 3, FMT_STRING(L"{}"), args: 12345);
35}

42void test_text_style() {
43 fmt::print(ts: fg(foreground: fmt::rgb(255, 20, 30)), FMT_STRING("{}"), args: "rgb(255,20,30)");
44 (void)fmt::format(ts: fg(foreground: fmt::rgb(255, 20, 30)), FMT_STRING("{}"),
45 args: "rgb(255,20,30)");
46
47 fmt::text_style ts = fg(foreground: fmt::rgb(255, 20, 30));
48 std::string out;
49 fmt::format_to(out: std::back_inserter(x&: out), ts,
50 FMT_STRING("rgb(255,20,30){}{}{}"), args: 1, args: 2, args: 3);
51}
