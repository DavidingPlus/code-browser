11#if defined(__cpp_lib_bit_cast) && __cpp_lib_bit_cast >= 201806 && \
12 defined(__cpp_constexpr) && __cpp_constexpr >= 201907 && \
13 defined(__cpp_constexpr_dynamic_alloc) && \
14 __cpp_constexpr_dynamic_alloc >= 201907 && __cplusplus >= 202002L
15template <size_t max_string_length, typename Char = char> struct test_string {
16 template <typename T> constexpr bool operator==(const T& rhs) const noexcept {
17 return fmt::basic_string_view<Char>(rhs).compare(buffer) == 0;
18 }
19 Char buffer[max_string_length]{};
20};
21
22template <size_t max_string_length, typename Char = char, typename... Args>
23consteval auto test_format(auto format, const Args&... args) {
24 test_string<max_string_length, Char> string{};
25 fmt::format_to(string.buffer, format, args...);
26 return string;
27}
28
29TEST(compile_time_formatting_test, floating_point) {
30 EXPECT_EQ("0", test_format<2>(FMT_COMPILE("{}"), 0.0f));
31 EXPECT_EQ("392.500000", test_format<11>(FMT_COMPILE("{0:f}"), 392.5f));
32
33 EXPECT_EQ("0", test_format<2>(FMT_COMPILE("{:}"), 0.0));
34 EXPECT_EQ("0.000000", test_format<9>(FMT_COMPILE("{:f}"), 0.0));
35 EXPECT_EQ("0", test_format<2>(FMT_COMPILE("{:g}"), 0.0));
36 EXPECT_EQ("392.65", test_format<7>(FMT_COMPILE("{:}"), 392.65));
37 EXPECT_EQ("392.65", test_format<7>(FMT_COMPILE("{:g}"), 392.65));
38 EXPECT_EQ("392.65", test_format<7>(FMT_COMPILE("{:G}"), 392.65));
39 EXPECT_EQ("4.9014e+06", test_format<11>(FMT_COMPILE("{:g}"), 4.9014e6));
40 EXPECT_EQ("-392.650000", test_format<12>(FMT_COMPILE("{:f}"), -392.65));
41 EXPECT_EQ("-392.650000", test_format<12>(FMT_COMPILE("{:F}"), -392.65));
42
43 EXPECT_EQ("3.926500e+02", test_format<13>(FMT_COMPILE("{0:e}"), 392.65));
44 EXPECT_EQ("3.926500E+02", test_format<13>(FMT_COMPILE("{0:E}"), 392.65));
45 EXPECT_EQ("+0000392.6", test_format<11>(FMT_COMPILE("{0:+010.4g}"), 392.65));
46 EXPECT_EQ("9223372036854775808.000000",
47 test_format<27>(FMT_COMPILE("{:f}"), 9223372036854775807.0));
48
49 constexpr double nan = std::numeric_limits<double>::quiet_NaN();
50 EXPECT_EQ("nan", test_format<4>(FMT_COMPILE("{}"), nan));
51 EXPECT_EQ("+nan", test_format<5>(FMT_COMPILE("{:+}"), nan));
52 if (std::signbit(-nan))
53 EXPECT_EQ("-nan", test_format<5>(FMT_COMPILE("{}"), -nan));
54 else
55 fmt::print("Warning: compiler doesn't handle negative NaN correctly");
56
57 constexpr double inf = std::numeric_limits<double>::infinity();
58 EXPECT_EQ("inf", test_format<4>(FMT_COMPILE("{}"), inf));
59 EXPECT_EQ("+inf", test_format<5>(FMT_COMPILE("{:+}"), inf));
60 EXPECT_EQ("-inf", test_format<5>(FMT_COMPILE("{}"), -inf));
61}
