20TEST(unicode_test, legacy_locale) {
21 auto loc = get_locale(name: "ru_RU.CP1251", alt_name: "Russian_Russia.1251");
22 if (loc == std::locale::classic()) return;
23
24 auto s = std::string();
25 try {
26 s = fmt::format(loc, fmt: "День недели: {:L}", args: fmt::weekday(1));
27 } catch (const fmt::format_error& e) {
28 // Formatting can fail due to an unsupported encoding.
29 fmt::print(fmt: "Format error: {}\n", args: e.what());
30 return;
31 }
32
33#if !FMT_GCC_VERSION || FMT_GCC_VERSION >= 500
34 auto&& os = std::ostringstream();
35 os.imbue(loc: loc);
36 auto tm = std::tm();
37 tm.tm_wday = 1;
38 os << std::put_time(tmb: &tm, fmt: "%a");
39 auto wd = os.str();
40 if (wd == "??") {
41 EXPECT_EQ(s, "День недели: ??");
42 fmt::print(fmt: "std::locale gives ?? as a weekday.\n");
43 return;
44 }
45#endif
46 EXPECT_THAT((std::vector<std::string>{"День недели: пн", "День недели: Пн"}),
47 Contains(s));
48}
