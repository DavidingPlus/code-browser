272std::wstring system_wcsftime(const std::wstring& format, const std::tm* timeptr,
273 std::locale* locptr = nullptr) {
274 auto loc = locptr ? *locptr : std::locale::classic();
275 auto& facet = std::use_facet<std::time_put<wchar_t>>(loc: loc);
276 std::wostringstream os;
277 os.imbue(loc: loc);
278 facet.put(s: os, io&: os, fill: L' ', tm: timeptr, beg: format.c_str(),
279 end: format.c_str() + format.size());
280#ifdef _WIN32
281 // Workaround a bug in older versions of Universal CRT.
282 auto str = os.str();
283 if (str == L"-0000") str = L"+0000";
284 return str;
285#else
286 return os.str();
287#endif
288}

293 std::vector<std::wstring> spec_list = {

295 L"%EC", L"%G", L"%g", L"%b", L"%h", L"%B", L"%m", L"%Om", L"%U",

297 L"%Oe", L"%a", L"%A", L"%w", L"%Ow", L"%u", L"%Ou", L"%H", L"%OH",
298 L"%I", L"%OI", L"%M", L"%OM", L"%S", L"%OS", L"%x", L"%Ex", L"%X",
299 L"%EX", L"%D", L"%F", L"%R", L"%T", L"%p", L"%z", L"%Z"};
300 spec_list.push_back(x: L"%Y-%m-%d %H:%M:%S");
301#ifndef _WIN32

313 auto fmt_spec = fmt::format(format_str: L"{{:{}}}", args: spec);

316 }

333template <typename Char> struct numpunct : std::numpunct<Char> {
334 protected:
335 Char do_decimal_point() const override { return '?'; }
336 std::string do_grouping() const override { return "\03"; }
337 Char do_thousands_sep() const override { return '~'; }
338};
339
340template <typename Char> struct no_grouping : std::numpunct<Char> {
341 protected:
342 Char do_decimal_point() const override { return '.'; }
343 std::string do_grouping() const override { return ""; }
344 Char do_thousands_sep() const override { return ','; }
345};
346
347template <typename Char> struct special_grouping : std::numpunct<Char> {
348 protected:
349 Char do_decimal_point() const override { return '.'; }
350 std::string do_grouping() const override { return "\03\02"; }
351 Char do_thousands_sep() const override { return ','; }
352};
353
354template <typename Char> struct small_grouping : std::numpunct<Char> {
355 protected:
356 Char do_decimal_point() const override { return '.'; }
357 std::string do_grouping() const override { return "\01"; }
358 Char do_thousands_sep() const override { return ','; }
359};

442template <class charT> struct formatter<std::complex<double>, charT> {
443 private:
444 detail::dynamic_format_specs<char> specs_;
445
446 public:
447 FMT_CONSTEXPR typename basic_format_parse_context<charT>::iterator parse(
448 basic_format_parse_context<charT>& ctx) {
449 using handler_type =
450 detail::dynamic_specs_handler<basic_format_parse_context<charT>>;
451 detail::specs_checker<handler_type> handler(handler_type(specs_, ctx),
452 detail::type::string_type);
453 auto it = parse_format_specs(ctx.begin(), ctx.end(), handler);
454 detail::parse_float_type_spec(specs_, ctx.error_handler());
455 return it;
456 }
457
458 template <class FormatContext>
459 typename FormatContext::iterator format(const std::complex<double>& c,
460 FormatContext& ctx) {
461 detail::handle_dynamic_spec<detail::precision_checker>(
462 specs_.precision, specs_.precision_ref, ctx);
463 auto specs = std::string();
464 if (specs_.precision > 0) specs = fmt::format(fmt: ".{}", args&: specs_.precision);
465 if (specs_.type == presentation_type::fixed_lower) specs += 'f';
466 auto real = fmt::format(ctx.locale().template get<std::locale>(),
467 fmt::runtime(s: "{:" + specs + "}"), c.real());
468 auto imag = fmt::format(ctx.locale().template get<std::locale>(),
469 fmt::runtime(s: "{:" + specs + "}"), c.imag());
470 auto fill_align_width = std::string();
471 if (specs_.width > 0) fill_align_width = fmt::format(fmt: ">{}", args&: specs_.width);
472 return format_to(ctx.out(), runtime(s: "{:" + fill_align_width + "}"),
473 c.real() != 0 ? fmt::format("({}+{}i)", real, imag)
474 : fmt::format("{}i", imag));
475 }
476};
477FMT_END_NAMESPACE
