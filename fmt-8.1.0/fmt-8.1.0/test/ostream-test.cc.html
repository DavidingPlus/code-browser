67TEST(ostream_test, format_specs) {
68 using fmt::format_error;
69 EXPECT_EQ("def ", fmt::format("{0:<5}", test_string("def")));
70 EXPECT_EQ(" def", fmt::format("{0:>5}", test_string("def")));
71 EXPECT_EQ(" def ", fmt::format("{0:^5}", test_string("def")));
72 EXPECT_EQ("def**", fmt::format("{0:*<5}", test_string("def")));
73 EXPECT_THROW_MSG((void)fmt::format(runtime("{0:+}"), test_string()),
74 format_error, "format specifier requires numeric argument");
75 EXPECT_THROW_MSG((void)fmt::format(runtime("{0:-}"), test_string()),
76 format_error, "format specifier requires numeric argument");
77 EXPECT_THROW_MSG((void)fmt::format(runtime("{0: }"), test_string()),
78 format_error, "format specifier requires numeric argument");
79 EXPECT_THROW_MSG((void)fmt::format(runtime("{0:#}"), test_string()),
80 format_error, "format specifier requires numeric argument");
81 EXPECT_THROW_MSG((void)fmt::format(runtime("{0:05}"), test_string()),
82 format_error, "format specifier requires numeric argument");
83 EXPECT_EQ("test ", fmt::format("{0:13}", test_string("test")));
84 EXPECT_EQ("test ", fmt::format("{0:{1}}", test_string("test"), 13));
85 EXPECT_EQ("te", fmt::format("{0:.2}", test_string("test")));
86 EXPECT_EQ("te", fmt::format("{0:.{1}}", test_string("test"), 2));
87}

111TEST(ostream_test, write_to_ostream_max_size) {
112 auto max_size = fmt::detail::max_value<size_t>();
113 auto max_streamsize = fmt::detail::max_value<std::streamsize>();
114 if (max_size <= fmt::detail::to_unsigned(value: max_streamsize)) return;
115
116 struct test_buffer final : fmt::detail::buffer<char> {
117 explicit test_buffer(size_t size)
118 : fmt::detail::buffer<char>(nullptr, size, size) {}
119 void grow(size_t) override {}
120 } buffer(max_size);
121
122 struct mock_streambuf : std::streambuf {
123 MOCK_METHOD2(xsputn, std::streamsize(const void* s, std::streamsize n));
124 std::streamsize xsputn(const char* s, std::streamsize n) override {
125 const void* v = s;
126 return xsputn(gmock_a0: v, gmock_a1: n);
127 }
128 } streambuf;
129
130 struct test_ostream : std::ostream {
131 explicit test_ostream(mock_streambuf& output_buffer)
132 : std::ostream(&output_buffer) {}
133 } os(streambuf);
134
135 testing::InSequence sequence;
136 const char* data = nullptr;
137 using ustreamsize = std::make_unsigned<std::streamsize>::type;
138 ustreamsize size = max_size;
139 do {
140 auto n = std::min(size, fmt::detail::to_unsigned(value: max_streamsize));
141 EXPECT_CALL(streambuf, xsputn(data, static_cast<std::streamsize>(n)))
142 .WillOnce(action: testing::Return(value: max_streamsize));
143 data += n;
144 size -= n;
145 } while (size != 0);
146 fmt::detail::write_buffer(os, buf&: buffer);
147}

193TEST(ostream_test, format_to_n) {
194 char buffer[4];
195 buffer[3] = 'x';
196 auto result = fmt::format_to_n(out: buffer, n: 3, fmt: "{}", args: fmt_test::abc());
197 EXPECT_EQ(3u, result.size);
198 EXPECT_EQ(buffer + 3, result.out);
199 EXPECT_EQ("abcx", fmt::string_view(buffer, 4));
200 result = fmt::format_to_n(out: buffer, n: 3, fmt: "x{}y", args: fmt_test::abc());
201 EXPECT_EQ(5u, result.size);
202 EXPECT_EQ(buffer + 3, result.out);
203 EXPECT_EQ("xabx", fmt::string_view(buffer, 4));
204}
