44std::string system_strftime(const std::string& format, const std::tm* timeptr,
45 std::locale* locptr = nullptr) {
46 auto loc = locptr ? *locptr : std::locale::classic();
47 auto& facet = std::use_facet<std::time_put<char>>(loc: loc);
48 std::ostringstream os;
49 os.imbue(loc: loc);
50 facet.put(s: os, io&: os, fill: ' ', tm: timeptr, beg: format.c_str(),
51 end: format.c_str() + format.size());
52#ifdef _WIN32
53 // Workaround a bug in older versions of Universal CRT.
54 auto str = os.str();
55 if (str == "-0000") str = "+0000";
56 return str;
57#else
58 return os.str();
59#endif
60}

62FMT_CONSTEXPR std::tm make_tm(int year, int mon, int mday, int hour, int min,
63 int sec) {
64 auto tm = std::tm();
65 tm.tm_sec = sec;
66 tm.tm_min = min;
67 tm.tm_hour = hour;
68 tm.tm_mday = mday;
69 tm.tm_mon = mon - 1;
70 tm.tm_year = year - 1900;
71 return tm;
72}

74TEST(chrono_test, format_tm) {
75 auto tm = std::tm();
76 tm.tm_year = 116;
77 tm.tm_mon = 3;
78 tm.tm_mday = 25;
79 tm.tm_hour = 11;
80 tm.tm_min = 22;
81 tm.tm_sec = 33;
82 EXPECT_EQ(fmt::format("The date is {:%Y-%m-%d %H:%M:%S}.", tm),
83 "The date is 2016-04-25 11:22:33.");
84 EXPECT_EQ(fmt::format("{:%Y}", tm), "2016");
85 EXPECT_EQ(fmt::format("{:%C}", tm), "20");
86 EXPECT_EQ(fmt::format("{:%C%y}", tm), fmt::format("{:%Y}", tm));
87 EXPECT_EQ(fmt::format("{:%e}", tm), "25");
88 EXPECT_EQ(fmt::format("{:%D}", tm), "04/25/16");
89 EXPECT_EQ(fmt::format("{:%F}", tm), "2016-04-25");
90 EXPECT_EQ(fmt::format("{:%T}", tm), "11:22:33");
91
92 // Short year
93 tm.tm_year = 999 - 1900;
94 tm.tm_mon = 0; // for %G
95 tm.tm_mday = 2; // for %G
96 tm.tm_wday = 3; // for %G
97 tm.tm_yday = 1; // for %G
98 EXPECT_EQ(fmt::format("{:%Y}", tm), "0999");
99 EXPECT_EQ(fmt::format("{:%C%y}", tm), "0999");
100 EXPECT_EQ(fmt::format("{:%G}", tm), "0999");
101
102 tm.tm_year = 27 - 1900;
103 EXPECT_EQ(fmt::format("{:%Y}", tm), "0027");
104 EXPECT_EQ(fmt::format("{:%C%y}", tm), "0027");
105
106 // Overflow year
107 tm.tm_year = 2147483647;
108 EXPECT_EQ(fmt::format("{:%Y}", tm), "2147485547");
109
110 tm.tm_year = -2147483648;
111 EXPECT_EQ(fmt::format("{:%Y}", tm), "-2147481748");
112
113 // for week on the year
114 // https://www.cl.cam.ac.uk/~mgk25/iso-time.html
115 std::vector<std::tm> tm_list = {
116 make_tm(year: 1975, mon: 12, mday: 29, hour: 12, min: 14, sec: 16), // W01
117 make_tm(year: 1977, mon: 1, mday: 2, hour: 12, min: 14, sec: 16), // W53
118 make_tm(year: 1999, mon: 12, mday: 27, hour: 12, min: 14, sec: 16), // W52
119 make_tm(year: 1999, mon: 12, mday: 31, hour: 12, min: 14, sec: 16), // W52
120 make_tm(year: 2000, mon: 1, mday: 1, hour: 12, min: 14, sec: 16), // W52
121 make_tm(year: 2000, mon: 1, mday: 2, hour: 12, min: 14, sec: 16), // W52
122 make_tm(year: 2000, mon: 1, mday: 3, hour: 12, min: 14, sec: 16) // W1
123 };
124 const std::string iso_week_spec = "%Y-%m-%d: %G %g %V";
125 for (auto ctm : tm_list) {
126 // Calculate tm_yday, tm_wday, etc.
127 std::time_t t = std::mktime(tp: &ctm);
128 tm = *std::localtime(timer: &t);
129
130 auto fmt_spec = fmt::format(fmt: "{{:{}}}", args: iso_week_spec);
131 EXPECT_EQ(system_strftime(iso_week_spec, &tm),
132 fmt::format(fmt::runtime(fmt_spec), tm));
133 }
134
135 // Every day from 1970-01-01
136 std::time_t time_now = std::time(timer: nullptr);
137 for (std::time_t t = 6 * 3600; t < time_now; t += 86400) {
138 tm = *std::localtime(timer: &t);
139
140 auto fmt_spec = fmt::format(fmt: "{{:{}}}", args: iso_week_spec);
141 EXPECT_EQ(system_strftime(iso_week_spec, &tm),
142 fmt::format(fmt::runtime(fmt_spec), tm));
143 }
144}

242template <typename TimePoint> auto strftime_full(TimePoint tp) -> std::string {
243 auto t = std::chrono::system_clock::to_time_t(t: tp);
244 auto tm = *std::localtime(timer: &t);
245 return system_strftime("%Y-%m-%d %H:%M:%S", &tm);
246}

585TEST(chrono_test, cpp20_duration_subsecond_support) {
586 using attoseconds = std::chrono::duration<long long, std::atto>;
587 // Check that 18 digits of subsecond precision are supported.
588 EXPECT_EQ(fmt::format("{:%S}", attoseconds{999999999999999999}),
589 "00.999999999999999999");
590 EXPECT_EQ(fmt::format("{:%S}", attoseconds{673231113420148734}),
591 "00.673231113420148734");
592 EXPECT_EQ(fmt::format("{:%S}", attoseconds{-673231113420148734}),
593 "-00.673231113420148734");
594 EXPECT_EQ(fmt::format("{:%S}", std::chrono::nanoseconds{13420148734}),
595 "13.420148734");
596 EXPECT_EQ(fmt::format("{:%S}", std::chrono::nanoseconds{-13420148734}),
597 "-13.420148734");
598 EXPECT_EQ(fmt::format("{:%S}", std::chrono::milliseconds{1234}), "01.234");
599 {
600 // Check that {:%H:%M:%S} is equivalent to {:%T}.
601 auto dur = std::chrono::milliseconds{3601234};
602 auto formatted_dur = fmt::format(fmt: "{:%T}", args&: dur);
603 EXPECT_EQ(formatted_dur, "01:00:01.234");
604 EXPECT_EQ(fmt::format("{:%H:%M:%S}", dur), formatted_dur);
605 }
606 using nanoseconds_dbl = std::chrono::duration<double, std::nano>;
607 EXPECT_EQ(fmt::format("{:%S}", nanoseconds_dbl{-123456789}), "-00.123456789");
608 EXPECT_EQ(fmt::format("{:%S}", nanoseconds_dbl{9123456789}), "09.123456789");
609 // Verify that only the seconds part is extracted and printed.
610 EXPECT_EQ(fmt::format("{:%S}", nanoseconds_dbl{99123456789}), "39.123456789");
611 EXPECT_EQ(fmt::format("{:%S}", nanoseconds_dbl{99123000000}), "39.123000000");
612 {
613 // Now the hour is printed, and we also test if negative doubles work.
614 auto dur = nanoseconds_dbl{-99123456789};
615 auto formatted_dur = fmt::format(fmt: "{:%T}", args&: dur);
616 EXPECT_EQ(formatted_dur, "-00:01:39.123456789");
617 EXPECT_EQ(fmt::format("{:%H:%M:%S}", dur), formatted_dur);
618 }
619 // Check that durations with precision greater than std::chrono::seconds have
620 // fixed precision, and print zeros even if there is no fractional part.
621 EXPECT_EQ(fmt::format("{:%S}", std::chrono::microseconds{7000000}),
622 "07.000000");
623}
