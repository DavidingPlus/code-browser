78#ifdef FMT_RANGES_TEST_ENABLE_FORMAT_STRUCT
79struct tuple_like {
80 int i;
81 std::string str;
82
83 template <size_t N> fmt::enable_if_t<N == 0, int> get() const noexcept {
84 return i;
85 }
86 template <size_t N>
87 fmt::enable_if_t<N == 1, fmt::string_view> get() const noexcept {
88 return str;
89 }
90};
91
92template <size_t N>
93auto get(const tuple_like& t) noexcept -> decltype(t.get<N>()) {
94 return t.get<N>();
95}
96
97namespace std {
98template <>
99struct tuple_size<tuple_like> : std::integral_constant<size_t, 2> {};
100
101template <size_t N> struct tuple_element<N, tuple_like> {
102 using type = decltype(std::declval<tuple_like>().get<N>());
103};
104} // namespace std
105
106TEST(ranges_test, format_struct) {
107 auto t = tuple_like{.i: 42, .str: "foo"};
108 EXPECT_EQ(fmt::format("{}", t), "(42, \"foo\")");

143// A range that provides non-const only begin()/end() to test fmt::join handles
144// that.
145//
146// Some ranges (e.g. those produced by range-v3's views::filter()) can cache
147// information during iteration so they only provide non-const begin()/end().
148template <typename T> class non_const_only_range {
149 private:
150 std::vector<T> vec;
151
152 public:
153 using const_iterator = typename ::std::vector<T>::const_iterator;
154
155 template <typename... Args>
156 explicit non_const_only_range(Args&&... args)
157 : vec(std::forward<Args>(args)...) {}
158
159 const_iterator begin() { return vec.begin(); }
160 const_iterator end() { return vec.end(); }
161};
162
163template <typename T> class noncopyable_range {
164 private:
165 std::vector<T> vec;
166
167 public:
168 using const_iterator = typename ::std::vector<T>::const_iterator;
169
170 template <typename... Args>
171 explicit noncopyable_range(Args&&... args)
172 : vec(std::forward<Args>(args)...) {}
173
174 noncopyable_range(noncopyable_range const&) = delete;
175 noncopyable_range(noncopyable_range&) = delete;
176
177 const_iterator begin() const { return vec.begin(); }
178 const_iterator end() const { return vec.end(); }
179};
180
181TEST(ranges_test, range) {
182 noncopyable_range<int> w(3u, 0);
183 EXPECT_EQ(fmt::format("{}", w), "[0, 0, 0]");
184 EXPECT_EQ(fmt::format("{}", noncopyable_range<int>(3u, 0)), "[0, 0, 0]");
185
186 non_const_only_range<int> x(3u, 0);
187 EXPECT_EQ(fmt::format("{}", x), "[0, 0, 0]");
188 EXPECT_EQ(fmt::format("{}", non_const_only_range<int>(3u, 0)), "[0, 0, 0]");
189
190 auto y = std::vector<int>(3u, 0);
191 EXPECT_EQ(fmt::format("{}", y), "[0, 0, 0]");
192 EXPECT_EQ(fmt::format("{}", std::vector<int>(3u, 0)), "[0, 0, 0]");
193
194 const auto z = std::vector<int>(3u, 0);
195 EXPECT_EQ(fmt::format("{}", z), "[0, 0, 0]");
196}

214#ifdef FMT_RANGES_TEST_ENABLE_JOIN
215TEST(ranges_test, join_tuple) {
216 // Value tuple args.
217 auto t1 = std::tuple<char, int, float>('a', 1, 2.0f);
218 EXPECT_EQ(fmt::format("({})", fmt::join(t1, ", ")), "(a, 1, 2)");
219
220 // Testing lvalue tuple args.
221 int x = 4;
222 auto t2 = std::tuple<char, int&>('b', x);
223 EXPECT_EQ(fmt::format("{}", fmt::join(t2, " + ")), "b + 4");
224
225 // Empty tuple.
226 auto t3 = std::tuple<>();
227 EXPECT_EQ(fmt::format("{}", fmt::join(t3, "|")), "");
228
229 // Single element tuple.
230 auto t4 = std::tuple<float>(4.0f);
231 EXPECT_EQ(fmt::format("{}", fmt::join(t4, "/")), "4");
232
233# if FMT_TUPLE_JOIN_SPECIFIERS
234 // Specs applied to each element.
235 auto t5 = std::tuple<int, int, long>(-3, 100, 1);
236 EXPECT_EQ(fmt::format("{:+03}", fmt::join(t5, ", ")), "-03, +100, +01");
237
238 auto t6 = std::tuple<float, double, long double>(3, 3.14, 3.1415);
239 EXPECT_EQ(fmt::format("{:5.5f}", fmt::join(t6, ", ")),
240 "3.00000, 3.14000, 3.14150");
241
242 // Testing lvalue tuple args.
243 int y = -1;
244 auto t7 = std::tuple<int, int&, const int&>(3, y, y);
245 EXPECT_EQ(fmt::format("{:03}", fmt::join(t7, ", ")), "003, -01, -01");
246# endif
247}

255struct zstring_sentinel {};
256
257bool operator==(const char* p, zstring_sentinel) { return *p == '\0'; }
258bool operator!=(const char* p, zstring_sentinel) { return *p != '\0'; }
259
260struct zstring {
261 const char* p;
262 const char* begin() const { return p; }
263 zstring_sentinel end() const { return {}; }
264};

266# ifdef __cpp_lib_ranges
267struct cpp20_only_range {
268 struct iterator {
269 int val = 0;
270
271 using value_type = int;
272 using difference_type = std::ptrdiff_t;
273 using iterator_concept = std::input_iterator_tag;
274
275 iterator() = default;
276 iterator(int i) : val(i) {}
277 int operator*() const { return val; }
278 iterator& operator++() {
279 ++val;
280 return *this;
281 }
282 void operator++(int) { ++*this; }
283 bool operator==(const iterator& rhs) const { return val == rhs.val; }
284 };
285
286 int lo;
287 int hi;
288
289 iterator begin() const { return iterator(lo); }
290 iterator end() const { return iterator(hi); }
291};
292
293static_assert(std::input_iterator<cpp20_only_range::iterator>);
294# endif

338TEST(ranges_test, escape_string) {
339 using vec = std::vector<std::string>;
340 EXPECT_EQ(fmt::format("{}", vec{"\n\r\t\"\\"}), "[\"\\n\\r\\t\\\"\\\\\"]");
341 EXPECT_EQ(fmt::format("{}", vec{"\x07"}), "[\"\\x07\"]");
342 EXPECT_EQ(fmt::format("{}", vec{"\x7f"}), "[\"\\x7f\"]");
343 EXPECT_EQ(fmt::format("{}", vec{"n\xcc\x83"}), "[\"n\xcc\x83\"]");
344
345 if (fmt::detail::is_utf8()) {
346 EXPECT_EQ(fmt::format("{}", vec{"\xcd\xb8"}), "[\"\\u0378\"]");
347 // Unassigned Unicode code points.
348 EXPECT_EQ(fmt::format("{}", vec{"\xf0\xaa\x9b\x9e"}), "[\"\\U0002a6de\"]");
349 EXPECT_EQ(fmt::format("{}", vec{"\xf4\x8f\xbf\xc0"}),
350 "[\"\\xf4\\x8f\\xbf\\xc0\"]");
351 }
