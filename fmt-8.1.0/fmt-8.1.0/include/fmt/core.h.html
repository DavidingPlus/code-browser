427/**
428 An implementation of ``std::basic_string_view`` for pre-C++17. It provides a
429 subset of the API. ``fmt::basic_string_view`` is used for format strings even
430 if ``std::string_view`` is available to prevent issues when a library is
431 compiled with a different ``-std`` option than the client code (which is not
432 recommended).
433 */
434template <typename Char> class basic_string_view {
435 private:
436 const Char* data_;
437 size_t size_;
438
439 public:
440 using value_type = Char;
441 using iterator = const Char*;
442
443 constexpr basic_string_view() FMT_NOEXCEPT : data_(nullptr), size_(0) {}
444
445 /** Constructs a string reference object from a C string and a size. */
446 constexpr basic_string_view(const Char* s, size_t count) FMT_NOEXCEPT
447 : data_(s),
448 size_(count) {}
449
450 /**
451 \rst
452 Constructs a string reference object from a C string computing
453 the size with ``std::char_traits<Char>::length``.
454 \endrst
455 */
456 FMT_CONSTEXPR_CHAR_TRAITS
457 FMT_INLINE
458 basic_string_view(const Char* s)
459 : data_(s),
460 size_(detail::const_check(std::is_same<Char, char>::value &&
461 !detail::is_constant_evaluated(default_value: true))
462 ? std::strlen(s: reinterpret_cast<const char*>(s))
463 : std::char_traits<Char>::length(s)) {}
464
465 /** Constructs a string reference from a ``std::basic_string`` object. */
466 template <typename Traits, typename Alloc>
467 FMT_CONSTEXPR basic_string_view(
468 const std::basic_string<Char, Traits, Alloc>& s) FMT_NOEXCEPT
469 : data_(s.data()),
470 size_(s.size()) {}
471
472 template <typename S, FMT_ENABLE_IF(std::is_same<
473 S, detail::std_string_view<Char>>::value)>
474 FMT_CONSTEXPR basic_string_view(S s) FMT_NOEXCEPT : data_(s.data()),
475 size_(s.size()) {}
476
477 /** Returns a pointer to the string data. */
478 constexpr auto data() const FMT_NOEXCEPT -> const Char* { return data_; }
479
480 /** Returns the string size. */
481 constexpr auto size() const FMT_NOEXCEPT -> size_t { return size_; }
482
483 constexpr auto begin() const FMT_NOEXCEPT -> iterator { return data_; }
484 constexpr auto end() const FMT_NOEXCEPT -> iterator { return data_ + size_; }
485
486 constexpr auto operator[](size_t pos) const FMT_NOEXCEPT -> const Char& {
487 return data_[pos];
488 }
489
490 FMT_CONSTEXPR void remove_prefix(size_t n) FMT_NOEXCEPT {
491 data_ += n;
492 size_ -= n;
493 }
494
495 // Lexicographically compare this string reference to other.
496 FMT_CONSTEXPR_CHAR_TRAITS auto compare(basic_string_view other) const -> int {
497 size_t str_size = size_ < other.size_ ? size_ : other.size_;
498 int result = std::char_traits<Char>::compare(data_, other.data_, str_size);
499 if (result == 0)
500 result = size_ == other.size_ ? 0 : (size_ < other.size_ ? -1 : 1);
501 return result;
502 }
503
504 FMT_CONSTEXPR_CHAR_TRAITS friend auto operator==(basic_string_view lhs,
505 basic_string_view rhs)
506 -> bool {
507 return lhs.compare(rhs) == 0;
508 }
509 friend auto operator!=(basic_string_view lhs, basic_string_view rhs) -> bool {
510 return lhs.compare(rhs) != 0;
511 }
512 friend auto operator<(basic_string_view lhs, basic_string_view rhs) -> bool {
513 return lhs.compare(rhs) < 0;
514 }
515 friend auto operator<=(basic_string_view lhs, basic_string_view rhs) -> bool {
516 return lhs.compare(rhs) <= 0;
517 }
518 friend auto operator>(basic_string_view lhs, basic_string_view rhs) -> bool {
519 return lhs.compare(rhs) > 0;
520 }
521 friend auto operator>=(basic_string_view lhs, basic_string_view rhs) -> bool {
522 return lhs.compare(rhs) >= 0;
523 }
524};

653 /**
654 Reports an error if using the manual argument indexing; otherwise returns
655 the next argument index and switches to the automatic indexing.
656 */
657 FMT_CONSTEXPR auto next_arg_id() -> int {
658 // Don't check if the argument id is valid to avoid overhead and because it
659 // will be checked during formatting anyway.

661 on_error(message: "cannot switch from manual to automatic argument indexing");

759 \rst
760 A contiguous memory buffer with an optional growing ability. It is an internal
761 class and shouldn't be used directly, only via `~fmt::basic_memory_buffer`.
762 \endrst

764template <typename T> class buffer {
765 private:
766 T* ptr_;
767 size_t size_;
768 size_t capacity_;
769
770 protected:
771 // Don't initialize ptr_ since it is not accessed to save a few cycles.
772 FMT_MSC_WARNING(suppress : 26495)
773 buffer(size_t sz) FMT_NOEXCEPT : size_(sz), capacity_(sz) {}
774
775 FMT_CONSTEXPR20 buffer(T* p = nullptr, size_t sz = 0,
776 size_t cap = 0) FMT_NOEXCEPT : ptr_(p),
777 size_(sz),
778 capacity_(cap) {}
779
780 FMT_CONSTEXPR20 ~buffer() = default;
781 buffer(buffer&&) = default;
782
783 /** Sets the buffer data and capacity. */
784 FMT_CONSTEXPR void set(T* buf_data, size_t buf_capacity) FMT_NOEXCEPT {
785 ptr_ = buf_data;
786 capacity_ = buf_capacity;
787 }
788
789 /** Increases the buffer capacity to hold at least *capacity* elements. */
790 virtual FMT_CONSTEXPR20 void grow(size_t capacity) = 0;
791
792 public:
793 using value_type = T;
794 using const_reference = const T&;
795
796 buffer(const buffer&) = delete;
797 void operator=(const buffer&) = delete;
798
799 auto begin() FMT_NOEXCEPT -> T* { return ptr_; }
800 auto end() FMT_NOEXCEPT -> T* { return ptr_ + size_; }
801
802 auto begin() const FMT_NOEXCEPT -> const T* { return ptr_; }
803 auto end() const FMT_NOEXCEPT -> const T* { return ptr_ + size_; }
804
805 /** Returns the size of this buffer. */
806 constexpr auto size() const FMT_NOEXCEPT -> size_t { return size_; }
807
808 /** Returns the capacity of this buffer. */
809 constexpr auto capacity() const FMT_NOEXCEPT -> size_t { return capacity_; }
810
811 /** Returns a pointer to the buffer data. */
812 FMT_CONSTEXPR auto data() FMT_NOEXCEPT -> T* { return ptr_; }
813
814 /** Returns a pointer to the buffer data. */
815 FMT_CONSTEXPR auto data() const FMT_NOEXCEPT -> const T* { return ptr_; }
816
817 /** Clears this buffer. */
818 void clear() { size_ = 0; }
819
820 // Tries resizing the buffer to contain *count* elements. If T is a POD type
821 // the new elements may not be initialized.
822 FMT_CONSTEXPR20 void try_resize(size_t count) {
823 try_reserve(new_capacity: count);
824 size_ = count <= capacity_ ? count : capacity_;
825 }
826
827 // Tries increasing the buffer capacity to *new_capacity*. It can increase the
828 // capacity by a smaller amount than requested but guarantees there is space
829 // for at least one additional element either by increasing the capacity or by
830 // flushing the buffer if it is full.
831 FMT_CONSTEXPR20 void try_reserve(size_t new_capacity) {
832 if (new_capacity > capacity_) grow(capacity: new_capacity);
833 }
834
835 FMT_CONSTEXPR20 void push_back(const T& value) {
836 try_reserve(new_capacity: size_ + 1);
837 ptr_[size_++] = value;
838 }
839
840 /** Appends data to the end of the buffer. */
841 template <typename U> void append(const U* begin, const U* end);
842
843 template <typename I> FMT_CONSTEXPR auto operator[](I index) -> T& {
844 return ptr_[index];
845 }
846 template <typename I>
847 FMT_CONSTEXPR auto operator[](I index) const -> const T& {
848 return ptr_[index];
849 }

873// A buffer that writes to an output iterator when flushed.
874template <typename OutputIt, typename T, typename Traits = buffer_traits>
875class iterator_buffer final : public Traits, public buffer<T> {
876 private:
877 OutputIt out_;
878 enum { buffer_size = 256 };
879 T data_[buffer_size];
880
881 protected:
882 FMT_CONSTEXPR20 void grow(size_t) override {
883 if (this->size() == buffer_size) flush();
884 }
885
886 void flush() {
887 auto size = this->size();
888 this->clear();
889 out_ = copy_str<T>(data_, data_ + this->limit(size), out_);
890 }
891
892 public:
893 explicit iterator_buffer(OutputIt out, size_t n = buffer_size)
894 : Traits(n), buffer<T>(data_, 0, buffer_size), out_(out) {}
895 iterator_buffer(iterator_buffer&& other)
896 : Traits(other), buffer<T>(data_, 0, buffer_size), out_(other.out_) {}
897 ~iterator_buffer() { flush(); }
898
899 auto out() -> OutputIt {
900 flush();
901 return out_;
902 }
903 auto count() const -> size_t { return Traits::count() + this->size(); }

1566/**
1567 \rst
1568 Visits an argument dispatching to the appropriate visit method based on
1569 the argument type. For example, if the argument type is ``double`` then
1570 ``vis(value)`` will be called with the value of type ``double``.
1571 \endrst
1572 */
1573template <typename Visitor, typename Context>
1574FMT_CONSTEXPR FMT_INLINE auto visit_format_arg(
1575 Visitor&& vis, const basic_format_arg<Context>& arg) -> decltype(vis(0)) {
1576 switch (arg.type_) {
1577 case detail::type::none_type:
1578 break;
1579 case detail::type::int_type:
1580 return vis(arg.value_.int_value);
1581 case detail::type::uint_type:
1582 return vis(arg.value_.uint_value);
1583 case detail::type::long_long_type:
1584 return vis(arg.value_.long_long_value);
1585 case detail::type::ulong_long_type:
1586 return vis(arg.value_.ulong_long_value);
1587 case detail::type::int128_type:
1588 return vis(detail::convert_for_visit(arg.value_.int128_value));
1589 case detail::type::uint128_type:
1590 return vis(detail::convert_for_visit(arg.value_.uint128_value));
1591 case detail::type::bool_type:
1592 return vis(arg.value_.bool_value);
1593 case detail::type::char_type:
1594 return vis(arg.value_.char_value);
1595 case detail::type::float_type:
1596 return vis(arg.value_.float_value);
1597 case detail::type::double_type:
1598 return vis(arg.value_.double_value);
1599 case detail::type::long_double_type:
1600 return vis(arg.value_.long_double_value);
1601 case detail::type::cstring_type:
1602 return vis(arg.value_.string.data);
1603 case detail::type::string_type:
1604 using sv = basic_string_view<typename Context::char_type>;
1605 return vis(sv(arg.value_.string.data, arg.value_.string.size));
1606 case detail::type::pointer_type:
1607 return vis(arg.value_.pointer);
1608 case detail::type::custom_type:
1609 return vis(typename basic_format_arg<Context>::handle(arg.value_.custom));
1610 }
1611 return vis(monostate());
1612}

2565 FMT_CONSTEXPR void operator()() { arg_id = handler.on_arg_id(); }
2566 FMT_CONSTEXPR void operator()(int id) { arg_id = handler.on_arg_id(id); }
2567 FMT_CONSTEXPR void operator()(basic_string_view<Char> id) {
2568 arg_id = handler.on_arg_id(id);
2569 }
2570 FMT_CONSTEXPR void on_error(const char* message) {
2571 if (message) handler.on_error(message);
2572 }
2573 };
2574
2575 ++begin;
2576 if (begin == end) return handler.on_error("invalid format string"), end;
2577 if (*begin == '}') {
2578 handler.on_replacement_field(handler.on_arg_id(), begin);
2579 } else if (*begin == '{') {
2580 handler.on_text(begin, begin + 1);
2581 } else {
2582 auto adapter = id_adapter{handler, 0};
2583 begin = parse_arg_id(begin, end, adapter);
2584 Char c = begin != end ? *begin : Char();
2585 if (c == '}') {
2586 handler.on_replacement_field(adapter.arg_id, begin);
2587 } else if (c == ':') {
2588 begin = handler.on_format_specs(adapter.arg_id, begin + 1, end);
2589 if (begin == end || *begin != '}')
2590 return handler.on_error("unknown format specifier"), end;
2591 } else {
2592 return handler.on_error("missing '}' in format string"), end;
2593 }
2594 }
2595 return begin + 1;
2596}

2598template <bool IS_CONSTEXPR, typename Char, typename Handler>
2599FMT_CONSTEXPR FMT_INLINE void parse_format_string(
2600 basic_string_view<Char> format_str, Handler&& handler) {
2601 // Workaround a name-lookup bug in MSVC's modules implementation.
2602 using detail::find;
2603
2604 auto begin = format_str.data();
2605 auto end = begin + format_str.size();
2606 if (end - begin < 32) {
2607 // Use a simple loop instead of memchr for small strings.
2608 const Char* p = begin;
2609 while (p != end) {
2610 auto c = *p++;
2611 if (c == '{') {
2612 handler.on_text(begin, p - 1);
2613 begin = p = parse_replacement_field(p - 1, end, handler);
2614 } else if (c == '}') {
2615 if (p == end || *p != '}')
2616 return handler.on_error("unmatched '}' in format string");
2617 handler.on_text(begin, p);
2618 begin = ++p;
2619 }
2620 }
2621 handler.on_text(begin, end);
2622 return;
2623 }
2624 struct writer {
2625 FMT_CONSTEXPR void operator()(const Char* pbegin, const Char* pend) {
2626 if (pbegin == pend) return;
2627 for (;;) {
2628 const Char* p = nullptr;
2629 if (!find<IS_CONSTEXPR>(pbegin, pend, Char('}'), p))
2630 return handler_.on_text(pbegin, pend);
2631 ++p;
2632 if (p == pend || *p != '}')
2633 return handler_.on_error("unmatched '}' in format string");
2634 handler_.on_text(pbegin, p);
2635 pbegin = p + 1;
2636 }
2637 }
2638 Handler& handler_;
2639 } write{handler};
2640 while (begin != end) {
2641 // Doing two passes with memchr (one for '{' and another for '}') is up to
2642 // 2.5x faster than the naive one-pass implementation on big format strings.
2643 const Char* p = begin;
2644 if (*begin != '{' && !find<IS_CONSTEXPR>(begin + 1, end, Char('{'), p))
2645 return write(begin, end);
2646 write(begin, p);
2647 begin = parse_replacement_field(p, end, handler);
2648 }
2649}
