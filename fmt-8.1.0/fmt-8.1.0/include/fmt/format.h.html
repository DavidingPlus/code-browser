203inline auto clz(uint32_t x) -> int {
204 unsigned long r = 0;
205 _BitScanReverse(&r, x);
206 FMT_ASSERT(x != 0, "");
207 // Static analysis complains about using uninitialized data
208 // "r", but the only way that can happen is if "x" is 0,
209 // which the callers guarantee to not happen.
210 FMT_MSC_WARNING(suppress : 6102)
211 return 31 ^ static_cast<int>(r);
212}
213# define FMT_BUILTIN_CLZ(n) detail::clz(n)
214
215inline auto clzll(uint64_t x) -> int {
216 unsigned long r = 0;
217# ifdef _WIN64
218 _BitScanReverse64(&r, x);
219# else
220 // Scan the high 32 bits.
221 if (_BitScanReverse(&r, static_cast<uint32_t>(x >> 32))) return 63 ^ (r + 32);
222 // Scan the low 32 bits.
223 _BitScanReverse(&r, static_cast<uint32_t>(x));
224# endif
225 FMT_ASSERT(x != 0, "");
226 FMT_MSC_WARNING(suppress : 6102) // Suppress a bogus static analysis warning.
227 return 63 ^ static_cast<int>(r);
228}
229# define FMT_BUILTIN_CLZLL(n) detail::clzll(n)
230
231inline auto ctz(uint32_t x) -> int {
232 unsigned long r = 0;
233 _BitScanForward(&r, x);
234 FMT_ASSERT(x != 0, "");
235 FMT_MSC_WARNING(suppress : 6102) // Suppress a bogus static analysis warning.
236 return static_cast<int>(r);
237}
238# define FMT_BUILTIN_CTZ(n) detail::ctz(n)
239
240inline auto ctzll(uint64_t x) -> int {
241 unsigned long r = 0;
242 FMT_ASSERT(x != 0, "");
243 FMT_MSC_WARNING(suppress : 6102) // Suppress a bogus static analysis warning.
244# ifdef _WIN64
245 _BitScanForward64(&r, x);
246# else
247 // Scan the low 32 bits.
248 if (_BitScanForward(&r, static_cast<uint32_t>(x))) return static_cast<int>(r);
249 // Scan the high 32 bits.
250 _BitScanForward(&r, static_cast<uint32_t>(x >> 32));
251 r += 32;
252# endif
253 return static_cast<int>(r);
254}

269template <typename Streambuf> class formatbuf : public Streambuf {
270 private:
271 using char_type = typename Streambuf::char_type;
272 using streamsize = decltype(std::declval<Streambuf>().sputn(nullptr, 0));
273 using int_type = typename Streambuf::int_type;
274 using traits_type = typename Streambuf::traits_type;
275
276 buffer<char_type>& buffer_;
277
278 public:
279 explicit formatbuf(buffer<char_type>& buf) : buffer_(buf) {}
280
281 protected:
282 // The put area is always empty. This makes the implementation simpler and has
283 // the advantage that the streambuf and the buffer are always in sync and
284 // sputc never writes into uninitialized memory. A disadvantage is that each
285 // call to sputc always results in a (virtual) call to overflow. There is no
286 // disadvantage here for sputn since this always results in a call to xsputn.
287
288 auto overflow(int_type ch) -> int_type override {
289 if (!traits_type::eq_int_type(ch, traits_type::eof()))
290 buffer_.push_back(static_cast<char_type>(ch));
291 return ch;
292 }
293
294 auto xsputn(const char_type* s, streamsize count) -> streamsize override {
295 buffer_.append(s, s + count);
296 return count;
297 }
298};

480// A public domain branchless UTF-8 decoder by Christopher Wellons:
481// https://github.com/skeeto/branchless-utf8
482/* Decode the next character, c, from s, reporting errors in e.
483 *
484 * Since this is a branchless decoder, four bytes will be read from the
485 * buffer regardless of the actual length of the next character. This
486 * means the buffer _must_ have at least three bytes of zero padding
487 * following the end of the data stream.
488 *
489 * Errors are reported in e, which will be non-zero if the parsed
490 * character was somehow invalid: invalid byte sequence, non-canonical
491 * encoding, or a surrogate half.
492 *
493 * The function returns a pointer to the next character. When an error
494 * occurs, this pointer will be a guess that depends on the particular
495 * error, but it will always advance at least one byte.
496 */
497FMT_CONSTEXPR inline auto utf8_decode(const char* s, uint32_t* c, int* e)
498 -> const char* {
499 constexpr const int masks[] = {0x00, 0x7f, 0x1f, 0x0f, 0x07};
500 constexpr const uint32_t mins[] = {4194304, 0, 128, 2048, 65536};
501 constexpr const int shiftc[] = {0, 18, 12, 6, 0};
502 constexpr const int shifte[] = {0, 6, 4, 2, 0};
503
504 int len = code_point_length(begin: s);
505 const char* next = s + len;
506
507 // Assume a four-byte character and load four bytes. Unused bits are
508 // shifted out.
509 *c = uint32_t(s[0] & masks[len]) << 18;
510 *c |= uint32_t(s[1] & 0x3f) << 12;
511 *c |= uint32_t(s[2] & 0x3f) << 6;
512 *c |= uint32_t(s[3] & 0x3f) << 0;
513 *c >>= shiftc[len];
514
515 // Accumulate the various error conditions.
516 using uchar = unsigned char;
517 *e = (*c < mins[len]) << 6; // non-canonical encoding
518 *e |= ((*c >> 11) == 0x1b) << 7; // surrogate half?
519 *e |= (*c > 0x10FFFF) << 8; // out of range?
520 *e |= (uchar(s[1]) & 0xc0) >> 2;
521 *e |= (uchar(s[2]) & 0xc0) >> 4;
522 *e |= uchar(s[3]) >> 6;
523 *e ^= 0x2a; // top two bits of each tail byte correct?
524 *e >>= shifte[len];
525
526 return next;
527}

531// Invokes f(cp, sv) for every code point cp in s with sv being the string view
532// corresponding to the code point. cp is invalid_code_point on error.
533template <typename F>
534FMT_CONSTEXPR void for_each_codepoint(string_view s, F f) {
535 auto decode = [f](const char* buf_ptr, const char* ptr) {
536 auto cp = uint32_t();
537 auto error = 0;
538 auto end = utf8_decode(s: buf_ptr, c: &cp, e: &error);
539 bool result = f(error ? invalid_code_point : cp,
540 string_view(ptr, to_unsigned(value: end - buf_ptr)));
541 return result ? end : nullptr;
542 };
543 auto p = s.data();
544 const size_t block_size = 4; // utf8_decode always reads blocks of 4 chars.
545 if (s.size() >= block_size) {
546 for (auto end = p + s.size() - block_size + 1; p < end;) {
547 p = decode(p, p);
548 if (!p) return;
549 }
550 }
551 if (auto num_chars_left = s.data() + s.size() - p) {
552 char buf[2 * block_size - 1] = {};
553 copy_str<char>(begin: p, end: p + num_chars_left, out: buf);
554 const char* buf_ptr = buf;
555 do {
556 auto end = decode(buf_ptr, p);
557 if (!end) return;
558 p += end - buf_ptr;
559 buf_ptr = end;
560 } while (buf_ptr - buf < num_chars_left);
561 }
562}

569// Computes approximate display width of a UTF-8 string.
570FMT_CONSTEXPR inline size_t compute_width(string_view s) {
571 size_t num_code_points = 0;
572 // It is not a lambda for compatibility with C++14.
573 struct count_code_points {
574 size_t* count;
575 FMT_CONSTEXPR auto operator()(uint32_t cp, string_view) const -> bool {
576 *count += detail::to_unsigned(
577 value: 1 +
578 (cp >= 0x1100 &&
579 (cp <= 0x115f || // Hangul Jamo init. consonants
580 cp == 0x2329 || // LEFT-POINTING ANGLE BRACKET
581 cp == 0x232a || // RIGHT-POINTING ANGLE BRACKET
582 // CJK ... Yi except IDEOGRAPHIC HALF FILL SPACE:
583 (cp >= 0x2e80 && cp <= 0xa4cf && cp != 0x303f) ||
584 (cp >= 0xac00 && cp <= 0xd7a3) || // Hangul Syllables
585 (cp >= 0xf900 && cp <= 0xfaff) || // CJK Compatibility Ideographs
586 (cp >= 0xfe10 && cp <= 0xfe19) || // Vertical Forms
587 (cp >= 0xfe30 && cp <= 0xfe6f) || // CJK Compatibility Forms
588 (cp >= 0xff00 && cp <= 0xff60) || // Fullwidth Forms
589 (cp >= 0xffe0 && cp <= 0xffe6) || // Fullwidth Forms
590 (cp >= 0x20000 && cp <= 0x2fffd) || // CJK
591 (cp >= 0x30000 && cp <= 0x3fffd) ||
592 // Miscellaneous Symbols and Pictographs + Emoticons:
593 (cp >= 0x1f300 && cp <= 0x1f64f) ||
594 // Supplemental Symbols and Pictographs:
595 (cp >= 0x1f900 && cp <= 0x1f9ff))));
596 return true;
597 }
598 };
599 for_each_codepoint(s, f: count_code_points{.count: &num_code_points});
600 return num_code_points;
601}

614// Calculates the index of the nth code point in a UTF-8 string.
615inline auto code_point_index(basic_string_view<char8_type> s, size_t n)
616 -> size_t {
617 const char8_type* data = s.data();
618 size_t num_code_points = 0;
619 for (size_t i = 0, size = s.size(); i != size; ++i) {
620 if ((data[i] & 0xc0) != 0x80 && ++num_code_points > n) return i;

660/**
661 \rst
662 A dynamically growing memory buffer for trivially copyable/constructible types
663 with the first ``SIZE`` elements stored in the object itself.
664
665 You can use the ``memory_buffer`` type alias for ``char`` instead.
666
667 **Example**::
668
669 auto out = fmt::memory_buffer();
670 format_to(std::back_inserter(out), "The answer is {}.", 42);
671
672 This will append the following output to the ``out`` object:
673
674 .. code-block:: none
675
676 The answer is 42.
677
678 The output can be converted to an ``std::string`` with ``to_string(out)``.
679 \endrst
680 */
681template <typename T, size_t SIZE = inline_buffer_size,
682 typename Allocator = std::allocator<T>>
683class basic_memory_buffer final : public detail::buffer<T> {
684 private:
685 T store_[SIZE];
686
687 // Don't inherit from Allocator avoid generating type_info for it.
688 Allocator alloc_;
689
690 // Deallocate memory allocated by the buffer.
691 FMT_CONSTEXPR20 void deallocate() {
692 T* data = this->data();
693 if (data != store_) alloc_.deallocate(data, this->capacity());
694 }
695
696 protected:
697 FMT_CONSTEXPR20 void grow(size_t size) override;
698
699 public:
700 using value_type = T;
701 using const_reference = const T&;
702
703 FMT_CONSTEXPR20 explicit basic_memory_buffer(
704 const Allocator& alloc = Allocator())
705 : alloc_(alloc) {
706 this->set(store_, SIZE);
707 if (detail::is_constant_evaluated()) {
708 detail::fill_n(store_, SIZE, T{});
709 }
710 }
711 FMT_CONSTEXPR20 ~basic_memory_buffer() { deallocate(); }
712
713 private:
714 // Move data from other to this buffer.
715 FMT_CONSTEXPR20 void move(basic_memory_buffer& other) {
716 alloc_ = std::move(other.alloc_);
717 T* data = other.data();
718 size_t size = other.size(), capacity = other.capacity();
719 if (data == other.store_) {
720 this->set(store_, capacity);
721 if (detail::is_constant_evaluated()) {
722 detail::copy_str<T>(other.store_, other.store_ + size,
723 detail::make_checked(store_, capacity));
724 } else {
725 std::uninitialized_copy(other.store_, other.store_ + size,
726 detail::make_checked(store_, capacity));
727 }
728 } else {
729 this->set(data, capacity);
730 // Set pointer to the inline array so that delete is not called
731 // when deallocating.
732 other.set(other.store_, 0);
733 }
734 this->resize(size);
735 }
736
737 public:
738 /**
739 \rst
740 Constructs a :class:`fmt::basic_memory_buffer` object moving the content
741 of the other object to it.
742 \endrst
743 */
744 FMT_CONSTEXPR20 basic_memory_buffer(basic_memory_buffer&& other)
745 FMT_NOEXCEPT {
746 move(other);
747 }
748
749 /**
750 \rst
751 Moves the content of the other ``basic_memory_buffer`` object to this one.
752 \endrst
753 */
754 auto operator=(basic_memory_buffer&& other) FMT_NOEXCEPT
755 -> basic_memory_buffer& {
756 FMT_ASSERT(this != &other, "");
757 deallocate();
758 move(other);
759 return *this;
760 }
761
762 // Returns a copy of the allocator associated with this buffer.
763 auto get_allocator() const -> Allocator { return alloc_; }
764
765 /**
766 Resizes the buffer to contain *count* elements. If T is a POD type new
767 elements may not be initialized.
768 */
769 FMT_CONSTEXPR20 void resize(size_t count) { this->try_resize(count); }
770
771 /** Increases the buffer capacity to *new_capacity*. */
772 void reserve(size_t new_capacity) { this->try_reserve(new_capacity); }
773
774 // Directly append data into the buffer
775 using detail::buffer<T>::append;
776 template <typename ContiguousRange>
777 void append(const ContiguousRange& range) {
778 append(range.data(), range.data() + range.size());
779 }
780};

782template <typename T, size_t SIZE, typename Allocator>
783FMT_CONSTEXPR20 void basic_memory_buffer<T, SIZE, Allocator>::grow(
784 size_t size) {
785#ifdef FMT_FUZZ
786 if (size > 5000) throw std::runtime_error("fuzz mode - won't grow that much");
787#endif
788 const size_t max_size = std::allocator_traits<Allocator>::max_size(alloc_);
789 size_t old_capacity = this->capacity();
790 size_t new_capacity = old_capacity + old_capacity / 2;
791 if (size > new_capacity)
792 new_capacity = size;
793 else if (new_capacity > max_size)
794 new_capacity = size > max_size ? size : max_size;
795 T* old_data = this->data();
796 T* new_data =
797 std::allocator_traits<Allocator>::allocate(alloc_, new_capacity);
798 // The following code doesn't throw, so the raw pointer above doesn't leak.
799 std::uninitialized_copy(old_data, old_data + this->size(),
800 detail::make_checked(new_data, new_capacity));
801 this->set(new_data, new_capacity);
802 // deallocate must not throw according to the standard, but even if it does,
803 // the buffer already uses the new storage and will deallocate it in
804 // destructor.
805 if (old_data != store_) alloc_.deallocate(old_data, old_capacity);
806}

942template <typename T> FMT_CONSTEXPR auto count_digits_fallback(T n) -> int {
943 int count = 1;
944 for (;;) {
945 // Integer division is slow so do it for a group of four digits instead
946 // of for every digit. The idea comes from the talk by Alexandrescu
947 // "Three Optimization Tips for C++". See speed-test for a comparison.
948 if (n < 10) return count;
949 if (n < 100) return count + 1;
950 if (n < 1000) return count + 2;
951 if (n < 10000) return count + 3;
952 n /= 10000u;
953 count += 4;
954 }
955}

1108// Formats a decimal unsigned integer value writing into out pointing to a
1109// buffer of specified size. The caller must ensure that the buffer is large
1110// enough.
1111template <typename Char, typename UInt>
1112FMT_CONSTEXPR20 auto format_decimal(Char* out, UInt value, int size)
1113 -> format_decimal_result<Char*> {
1114 FMT_ASSERT(size >= count_digits(value), "invalid digit count");
1115 out += size;
1116 Char* end = out;
1117 while (value >= 100) {
1118 // Integer division is slow so do it for a group of two digits instead
1119 // of for every digit. The idea comes from the talk by Alexandrescu
1120 // "Three Optimization Tips for C++". See speed-test for a comparison.
1121 out -= 2;
1122 copy2(out, digits2(value: static_cast<size_t>(value % 100)));
1123 value /= 100;
1124 }
1125 if (value < 10) {
1126 *--out = static_cast<Char>('0' + value);
1127 return {out, end};
1128 }
1129 out -= 2;
1130 copy2(out, digits2(value: static_cast<size_t>(value)));
1131 return {out, end};
1132}

1144template <unsigned BASE_BITS, typename Char, typename UInt>
1145FMT_CONSTEXPR auto format_uint(Char* buffer, UInt value, int num_digits,
1146 bool upper = false) -> Char* {
1147 buffer += num_digits;
1148 Char* end = buffer;
1149 do {
1150 const char* digits = upper ? "0123456789ABCDEF" : "0123456789abcdef";
1151 unsigned digit = (value & ((1 << BASE_BITS) - 1));
1152 *--buffer = static_cast<Char>(BASE_BITS < 4 ? static_cast<char>('0' + digit)
1153 : digits[digit]);
1154 } while ((value >>= BASE_BITS) != 0);
1155 return end;
1156}

1158template <unsigned BASE_BITS, typename Char>
1159auto format_uint(Char* buffer, detail::fallback_uintptr n, int num_digits,
1160 bool = false) -> Char* {
1161 auto char_digits = std::numeric_limits<unsigned char>::digits / 4;
1162 int start = (num_digits + char_digits - 1) / char_digits - 1;
1163 if (int start_digits = num_digits % char_digits) {
1164 unsigned value = n.value[start--];
1165 buffer = format_uint<BASE_BITS>(buffer, value, start_digits);
1166 }
1167 for (; start >= 0; --start) {
1168 unsigned value = n.value[start];
1169 buffer += char_digits;
1170 auto p = buffer;
1171 for (int i = 0; i < char_digits; ++i) {
1172 unsigned digit = (value & ((1 << BASE_BITS) - 1));
1173 *--p = static_cast<Char>("0123456789abcdef"[digit]);
1174 value >>= BASE_BITS;
1175 }
1176 }
1177 return buffer;
1178}

1181inline auto format_uint(It out, UInt value, int num_digits, bool upper = false)
1182 -> It {

1331// Writes the output of f, padded according to format specifications in specs.
1332// size: output size in code units.
1333// width: output display width in (terminal) column positions.
1334template <align::type align = align::left, typename OutputIt, typename Char,
1335 typename F>
1336FMT_CONSTEXPR auto write_padded(OutputIt out,
1337 const basic_format_specs<Char>& specs,
1338 size_t size, size_t width, F&& f) -> OutputIt {
1339 static_assert(align == align::left || align == align::right, "");
1340 unsigned spec_width = to_unsigned(specs.width);
1341 size_t padding = spec_width > width ? spec_width - width : 0;
1342 // Shifts are encoded as string literals because static constexpr is not
1343 // supported in constexpr functions.
1344 auto* shifts = align == align::left ? "\x1f\x1f\x00\x01" : "\x00\x1f\x00\x01";
1345 size_t left_padding = padding >> shifts[specs.align];
1346 size_t right_padding = padding - left_padding;
1347 auto it = reserve(out, size + padding * specs.fill.size());
1348 if (left_padding != 0) it = fill(it, left_padding, specs.fill);
1349 it = f(it);
1350 if (right_padding != 0) it = fill(it, right_padding, specs.fill);
1351 return base_iterator(out, it);
1352}

1426// Writes an integer in the format
1427// <left-padding><prefix><numeric-padding><digits><right-padding>
1428// where <digits> are written by write_digits(it).
1429// prefix contains chars in three lower bytes and the size in the fourth byte.
1430template <typename OutputIt, typename Char, typename W>
1431FMT_CONSTEXPR FMT_INLINE auto write_int(OutputIt out, int num_digits,
1432 unsigned prefix,
1433 const basic_format_specs<Char>& specs,
1434 W write_digits) -> OutputIt {
1435 // Slightly faster check for specs.width == 0 && specs.precision == -1.
1436 if ((specs.width | (specs.precision + 1)) == 0) {
1437 auto it = reserve(out, to_unsigned(value: num_digits) + (prefix >> 24));

1439 for (unsigned p = prefix & 0xffffff; p != 0; p >>= 8)
1440 *it++ = static_cast<Char>(p & 0xff);

1442 return base_iterator(out, write_digits(it));

1444 auto data = write_int_data<Char>(num_digits, prefix, specs);
1445 return write_padded<align::right>(
1446 out, specs, data.size, [=](reserve_iterator<OutputIt> it) {
1447 for (unsigned p = prefix & 0xffffff; p != 0; p >>= 8)
1448 *it++ = static_cast<Char>(p & 0xff);
1449 it = detail::fill_n(it, data.padding, static_cast<Char>('0'));

1451 });

1454template <typename Char> class digit_grouping {
1455 private:
1456 thousands_sep_result<Char> sep_;
1457
1458 struct next_state {
1459 std::string::const_iterator group;
1460 int pos;
1461 };
1462 next_state initial_state() const { return {sep_.grouping.begin(), 0}; }
1463
1464 // Returns the next digit group separator position.
1465 int next(next_state& state) const {
1466 if (!sep_.thousands_sep) return max_value<int>();
1467 if (state.group == sep_.grouping.end())
1468 return state.pos += sep_.grouping.back();
1469 if (*state.group <= 0 || *state.group == max_value<char>())
1470 return max_value<int>();
1471 state.pos += *state.group++;
1472 return state.pos;
1473 }
1474
1475 public:
1476 explicit digit_grouping(locale_ref loc, bool localized = true) {
1477 if (localized)
1478 sep_ = thousands_sep<Char>(loc);
1479 else
1480 sep_.thousands_sep = Char();
1481 }
1482 explicit digit_grouping(thousands_sep_result<Char> sep) : sep_(sep) {}
1483
1484 Char separator() const { return sep_.thousands_sep; }
1485
1486 int count_separators(int num_digits) const {
1487 int count = 0;
1488 auto state = initial_state();
1489 while (num_digits > next(state)) ++count;
1490 return count;
1491 }
1492
1493 // Applies grouping to digits and write the output to out.
1494 template <typename Out, typename C>
1495 Out apply(Out out, basic_string_view<C> digits) const {
1496 auto num_digits = static_cast<int>(digits.size());
1497 auto separators = basic_memory_buffer<int>();
1498 separators.push_back(value: 0);
1499 auto state = initial_state();
1500 while (int i = next(state)) {
1501 if (i >= num_digits) break;
1502 separators.push_back(value: i);
1503 }
1504 for (int i = 0, sep_index = static_cast<int>(separators.size() - 1);
1505 i < num_digits; ++i) {
1506 if (num_digits - i == separators[sep_index]) {
1507 *out++ = separator();
1508 --sep_index;
1509 }
1510 *out++ = static_cast<Char>(digits[to_unsigned(value: i)]);
1511 }
1512 return out;
1513 }

1568template <typename Char, typename OutputIt, typename T>
1569FMT_CONSTEXPR FMT_INLINE auto write_int(OutputIt out, write_int_arg<T> arg,
1570 const basic_format_specs<Char>& specs,
1571 locale_ref loc) -> OutputIt {
1572 static_assert(std::is_same<T, uint32_or_64_or_128_t<T>>::value, "");
1573 auto abs_value = arg.abs_value;
1574 auto prefix = arg.prefix;
1575 switch (specs.type) {
1576 case presentation_type::none:
1577 case presentation_type::dec: {
1578 if (specs.localized &&
1579 write_int_localized(out, static_cast<uint64_or_128_t<T>>(abs_value),
1580 prefix, specs, loc)) {
1581 return out;
1582 }
1583 auto num_digits = count_digits(abs_value);
1584 return write_int(
1585 out, num_digits, prefix, specs, [=](reserve_iterator<OutputIt> it) {
1586 return format_decimal<Char>(it, abs_value, num_digits).end;
1587 });
1588 }
1589 case presentation_type::hex_lower:
1590 case presentation_type::hex_upper: {
1591 bool upper = specs.type == presentation_type::hex_upper;
1592 if (specs.alt)
1593 prefix_append(prefix, unsigned(upper ? 'X' : 'x') << 8 | '0');
1594 int num_digits = count_digits<4>(abs_value);
1595 return write_int(
1596 out, num_digits, prefix, specs, [=](reserve_iterator<OutputIt> it) {
1597 return format_uint<4, Char>(it, abs_value, num_digits, upper);
1598 });
1599 }
1600 case presentation_type::bin_lower:
1601 case presentation_type::bin_upper: {
1602 bool upper = specs.type == presentation_type::bin_upper;
1603 if (specs.alt)
1604 prefix_append(prefix, unsigned(upper ? 'B' : 'b') << 8 | '0');
1605 int num_digits = count_digits<1>(abs_value);
1606 return write_int(out, num_digits, prefix, specs,
1607 [=](reserve_iterator<OutputIt> it) {
1608 return format_uint<1, Char>(it, abs_value, num_digits);
1609 });
1610 }
1611 case presentation_type::oct: {
1612 int num_digits = count_digits<3>(abs_value);
1613 // Octal prefix '0' is counted as a digit, so only add it if precision
1614 // is not greater than the number of digits.
1615 if (specs.alt && specs.precision <= num_digits && abs_value != 0)
1616 prefix_append(prefix, '0');
1617 return write_int(out, num_digits, prefix, specs,
1618 [=](reserve_iterator<OutputIt> it) {
1619 return format_uint<3, Char>(it, abs_value, num_digits);
1620 });
1621 }
1622 case presentation_type::chr:
1623 return write_char(out, static_cast<Char>(abs_value), specs);
1624 default:
1625 throw_format_error(message: "invalid type specifier");
1626 }
1627 return out;
1628}

1763 *--out = decimal_point;
1764 format_decimal(out - integral_size, significand, integral_size);

1810template <typename OutputIt, typename DecimalFP, typename Char,
1811 typename Grouping = digit_grouping<Char>>
1812FMT_CONSTEXPR20 auto do_write_float(OutputIt out, const DecimalFP& fp,
1813 const basic_format_specs<Char>& specs,
1814 float_specs fspecs, locale_ref loc)
1815 -> OutputIt {
1816 auto significand = fp.significand;
1817 int significand_size = get_significand_size(fp);
1818 constexpr Char zero = static_cast<Char>('0');
1819 auto sign = fspecs.sign;
1820 size_t size = to_unsigned(value: significand_size) + (sign ? 1 : 0);
1821 using iterator = reserve_iterator<OutputIt>;
1822
1823 Char decimal_point =
1824 fspecs.locale ? detail::decimal_point<Char>(loc) : static_cast<Char>('.');
1825
1826 int output_exp = fp.exponent + significand_size - 1;
1827 auto use_exp_format = [=]() {
1828 if (fspecs.format == float_format::exp) return true;
1829 if (fspecs.format != float_format::general) return false;
1830 // Use the fixed notation if the exponent is in [exp_lower, exp_upper),
1831 // e.g. 0.0001 instead of 1e-04. Otherwise use the exponent notation.
1832 const int exp_lower = -4, exp_upper = 16;
1833 return output_exp < exp_lower ||
1834 output_exp >= (fspecs.precision > 0 ? fspecs.precision : exp_upper);
1835 };
1836 if (use_exp_format()) {
1837 int num_zeros = 0;
1838 if (fspecs.showpoint) {
1839 num_zeros = fspecs.precision - significand_size;
1840 if (num_zeros < 0) num_zeros = 0;
1841 size += to_unsigned(value: num_zeros);
1842 } else if (significand_size == 1) {
1843 decimal_point = Char();
1844 }
1845 auto abs_output_exp = output_exp >= 0 ? output_exp : -output_exp;
1846 int exp_digits = 2;
1847 if (abs_output_exp >= 100) exp_digits = abs_output_exp >= 1000 ? 4 : 3;
1848
1849 size += to_unsigned((decimal_point ? 1 : 0) + 2 + exp_digits);
1850 char exp_char = fspecs.upper ? 'E' : 'e';
1851 auto write = [=](iterator it) {
1852 if (sign) *it++ = detail::sign<Char>(sign);
1853 // Insert a decimal point after the first digit and add an exponent.
1854 it = write_significand(it, significand, significand_size, 1,
1855 decimal_point);
1856 if (num_zeros > 0) it = detail::fill_n(it, num_zeros, zero);
1857 *it++ = static_cast<Char>(exp_char);
1858 return write_exponent<Char>(output_exp, it);
1859 };
1860 return specs.width > 0 ? write_padded<align::right>(out, specs, size, write)
1861 : base_iterator(out, write(reserve(out, size)));
1862 }
1863
1864 int exp = fp.exponent + significand_size;
1865 if (fp.exponent >= 0) {
1866 // 1234e5 -> 123400000[.0+]
1867 size += to_unsigned(fp.exponent);
1868 int num_zeros = fspecs.precision - exp;
1869#ifdef FMT_FUZZ
1870 if (num_zeros > 5000)
1871 throw std::runtime_error("fuzz mode - avoiding excessive cpu use");
1872#endif
1873 if (fspecs.showpoint) {
1874 if (num_zeros <= 0 && fspecs.format != float_format::fixed) num_zeros = 1;
1875 if (num_zeros > 0) size += to_unsigned(value: num_zeros) + 1;
1876 }
1877 auto grouping = Grouping(loc, fspecs.locale);
1878 size += to_unsigned(grouping.count_separators(significand_size));
1879 return write_padded<align::right>(out, specs, size, [&](iterator it) {
1880 if (sign) *it++ = detail::sign<Char>(sign);
1881 it = write_significand<Char>(it, significand, significand_size,
1882 fp.exponent, grouping);
1883 if (!fspecs.showpoint) return it;
1884 *it++ = decimal_point;
1885 return num_zeros > 0 ? detail::fill_n(it, num_zeros, zero) : it;
1886 });
1887 } else if (exp > 0) {
1888 // 1234e-2 -> 12.34[0+]
1889 int num_zeros = fspecs.showpoint ? fspecs.precision - significand_size : 0;
1890 size += 1 + to_unsigned(value: num_zeros > 0 ? num_zeros : 0);
1891 auto grouping = Grouping(loc, fspecs.locale);
1892 size += to_unsigned(grouping.count_separators(significand_size));
1893 return write_padded<align::right>(out, specs, size, [&](iterator it) {
1894 if (sign) *it++ = detail::sign<Char>(sign);
1895 it = write_significand(it, significand, significand_size, exp,
1896 decimal_point, grouping);
1897 return num_zeros > 0 ? detail::fill_n(it, num_zeros, zero) : it;
1898 });
1899 }
1900 // 1234e-6 -> 0.001234
1901 int num_zeros = -exp;
1902 if (significand_size == 0 && fspecs.precision >= 0 &&
1903 fspecs.precision < num_zeros) {
1904 num_zeros = fspecs.precision;
1905 }
1906 bool pointy = num_zeros != 0 || significand_size != 0 || fspecs.showpoint;
1907 size += 1 + (pointy ? 1 : 0) + to_unsigned(value: num_zeros);
1908 return write_padded<align::right>(out, specs, size, [&](iterator it) {
1909 if (sign) *it++ = detail::sign<Char>(sign);
1910 *it++ = zero;
1911 if (!pointy) return it;
1912 *it++ = decimal_point;
1913 it = detail::fill_n(it, num_zeros, zero);
1914 return write_significand<Char>(it, significand, significand_size);
1915 });
1916}
