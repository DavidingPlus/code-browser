59template <typename T>
60int format_float(char* buf, std::size_t size, const char* format, int precision,
61 T value) {
62#ifdef FMT_FUZZ
63 if (precision > 100000)
64 throw std::runtime_error(
65 "fuzz mode - avoid large allocation inside snprintf");
66#endif
67 // Suppress the warning about nonliteral format string.
68 int (*snprintf_ptr)(char*, size_t, const char*, ...) = FMT_SNPRINTF;
69 return precision < 0 ? snprintf_ptr(buf, size, format, value)
70 : snprintf_ptr(buf, size, format, precision, value);
71}
