64static Suite *
65make_suite(void) {
66 Suite *s = suite_create(name: "basic");
67
68 make_basic_test_case(s);
69 make_namespace_test_case(s);
70 make_miscellaneous_test_case(s);
71 make_alloc_test_case(s);
72 make_nsalloc_test_case(s);
73#if XML_GE == 1
74 make_accounting_test_case(s);
75#endif
76
77 return s;
78}
79
80int
81main(int argc, char *argv[]) {
82 int i, nf;
83 int verbosity = CK_NORMAL;
84 Suite *s = make_suite();
85 SRunner *sr = srunner_create(suite: s);
86
87 for (i = 1; i < argc; ++i) {
88 char *opt = argv[i];
89 if (strcmp(s1: opt, s2: "-v") == 0 || strcmp(s1: opt, s2: "--verbose") == 0)
90 verbosity = CK_VERBOSE;
91 else if (strcmp(s1: opt, s2: "-q") == 0 || strcmp(s1: opt, s2: "--quiet") == 0)
92 verbosity = CK_SILENT;
93 else {
94 fprintf(stderr, format: "runtests: unknown option '%s'\n", opt);
95 return 2;
96 }
97 }
98 if (verbosity != CK_SILENT)
99 printf(format: "Expat version: %" XML_FMT_STR "\n", XML_ExpatVersion());
100
101 for (g_chunkSize = 0; g_chunkSize <= 5; g_chunkSize++) {
102 for (int enabled = 0; enabled <= 1; ++enabled) {
103 char context[100];
104 g_reparseDeferralEnabledDefault = enabled;
105 snprintf(s: context, maxlen: sizeof(context), format: "chunksize=%d deferral=%d",
106 g_chunkSize, enabled);
107 context[sizeof(context) - 1] = '\0';
108 srunner_run_all(runner: sr, context, verbosity);
109 }
110 }
111 srunner_summarize(runner: sr, verbosity);
112 nf = srunner_ntests_failed(runner: sr);
113 srunner_free(runner: sr);
114
115 return (nf == 0) ? EXIT_SUCCESS : EXIT_FAILURE;
116}
