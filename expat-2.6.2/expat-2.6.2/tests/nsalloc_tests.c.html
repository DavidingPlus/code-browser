57static void
58nsalloc_setup(void) {
59 XML_Memory_Handling_Suite memsuite = {duff_allocator, duff_reallocator, free};
60 XML_Char ns_sep[2] = {' ', '\0'};
61
62 /* Ensure the parser creation will go through */
63 g_allocation_count = ALLOC_ALWAYS_SUCCEED;
64 g_reallocation_count = REALLOC_ALWAYS_SUCCEED;
65 g_parser = XML_ParserCreate_MM(NULL, memsuite: &memsuite, namespaceSeparator: ns_sep);
66 if (g_parser == NULL)
67 fail("Parser not created");
68}

75/* Test the effects of allocation failure in simple namespace parsing.
76 * Based on test_ns_default_with_empty_uri()
77 */
78START_TEST(test_nsalloc_xmlns) {
79 const char *text = "<doc xmlns='http://example.org/'>\n"
80 " <e xmlns=''/>\n"
81 "</doc>";
82 unsigned int i;
83 const unsigned int max_alloc_count = 30;
84
85 for (i = 0; i < max_alloc_count; i++) {
86 g_allocation_count = i;
87 /* Exercise more code paths with a default handler */
88 XML_SetDefaultHandler(parser: g_parser, handler: dummy_default_handler);
89 if (_XML_Parse_SINGLE_BYTES(parser: g_parser, s: text, len: (int)strlen(s: text), XML_TRUE)
90 != XML_STATUS_ERROR)
91 break;
92 /* Resetting the parser is insufficient, because some memory
93 * allocations are cached within the parser. Instead we use
94 * the teardown and setup routines to ensure that we have the
95 * right sort of parser back in our hands.
96 */
97 nsalloc_teardown();
98 nsalloc_setup();
99 }
100 if (i == 0)
101 fail("Parsing worked despite failing allocations");
102 else if (i == max_alloc_count)
103 fail("Parsing failed even at maximum allocation count");
104}
105END_TEST

107/* Test XML_ParseBuffer interface with namespace and a dicky allocator */
108START_TEST(test_nsalloc_parse_buffer) {
109 const char *text = "<doc>Hello</doc>";
110 void *buffer;
111
112 /* Try a parse before the start of the world */
113 /* (Exercises new code path) */
114 if (XML_ParseBuffer(parser: g_parser, len: 0, XML_FALSE) != XML_STATUS_ERROR)
115 fail("Pre-init XML_ParseBuffer not faulted");
116 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_NO_BUFFER)
117 fail("Pre-init XML_ParseBuffer faulted for wrong reason");
118
119 buffer = XML_GetBuffer(parser: g_parser, len: 1 /* any small number greater than 0 */);
120 if (buffer == NULL)
121 fail("Could not acquire parse buffer");
122
123 g_allocation_count = 0;
124 if (XML_ParseBuffer(parser: g_parser, len: 0, XML_FALSE) != XML_STATUS_ERROR)
125 fail("Pre-init XML_ParseBuffer not faulted");
126 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_NO_MEMORY)
127 fail("Pre-init XML_ParseBuffer faulted for wrong reason");
128
129 /* Now with actual memory allocation */
130 g_allocation_count = ALLOC_ALWAYS_SUCCEED;
131 if (XML_ParseBuffer(parser: g_parser, len: 0, XML_FALSE) != XML_STATUS_OK)
132 xml_failure(g_parser);
133
134 /* Check that resuming an unsuspended parser is faulted */
135 if (XML_ResumeParser(parser: g_parser) != XML_STATUS_ERROR)
136 fail("Resuming unsuspended parser not faulted");
137 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_NOT_SUSPENDED)
138 xml_failure(g_parser);
139
140 /* Get the parser into suspended state */
141 XML_SetCharacterDataHandler(parser: g_parser, handler: clearing_aborting_character_handler);
142 g_resumable = XML_TRUE;
143 buffer = XML_GetBuffer(parser: g_parser, len: (int)strlen(s: text));
144 if (buffer == NULL)
145 fail("Could not acquire parse buffer");
146 assert(buffer != NULL);
147 memcpy(dest: buffer, src: text, n: strlen(s: text));
148 if (XML_ParseBuffer(parser: g_parser, len: (int)strlen(s: text), XML_TRUE)
149 != XML_STATUS_SUSPENDED)
150 xml_failure(g_parser);
151 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_NONE)
152 xml_failure(g_parser);
153 if (XML_ParseBuffer(parser: g_parser, len: (int)strlen(s: text), XML_TRUE)
154 != XML_STATUS_ERROR)
155 fail("Suspended XML_ParseBuffer not faulted");
156 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_SUSPENDED)
157 xml_failure(g_parser);
158 if (XML_GetBuffer(parser: g_parser, len: (int)strlen(s: text)) != NULL)
159 fail("Suspended XML_GetBuffer not faulted");
160
161 /* Get it going again and complete the world */
162 XML_SetCharacterDataHandler(parser: g_parser, NULL);
163 if (XML_ResumeParser(parser: g_parser) != XML_STATUS_OK)
164 xml_failure(g_parser);
165 if (XML_ParseBuffer(parser: g_parser, len: (int)strlen(s: text), XML_TRUE)
166 != XML_STATUS_ERROR)
167 fail("Post-finishing XML_ParseBuffer not faulted");
168 if (XML_GetErrorCode(parser: g_parser) != XML_ERROR_FINISHED)
169 xml_failure(g_parser);
170 if (XML_GetBuffer(parser: g_parser, len: (int)strlen(s: text)) != NULL)
171 fail("Post-finishing XML_GetBuffer not faulted");
172}
