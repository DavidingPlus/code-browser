43#ifdef XML_LARGE_SIZE
44# define XML_FMT_INT_MOD "ll"
45#else
46# define XML_FMT_INT_MOD "l"
47#endif
48
49#ifdef XML_UNICODE_WCHAR_T
50# define XML_FMT_STR "ls"
51#else
52# define XML_FMT_STR "s"
53#endif

55static void
56usage(const char *prog, int rc) {
57 fprintf(stderr, format: "usage: %s [-n] filename bufferSize nr_of_loops\n", prog);
58 exit(status: rc);
59}
60
61int
62main(int argc, char *argv[]) {
63 XML_Parser parser;
64 char *XMLBuf, *XMLBufEnd, *XMLBufPtr;
65 FILE *fd;
66 struct stat fileAttr;
67 int nrOfLoops, bufferSize, i, isFinal;
68 size_t fileSize;
69 int j = 0, ns = 0;
70 clock_t tstart, tend;
71 double cpuTime = 0.0;
72
73 if (argc > 1) {
74 if (argv[1][0] == '-') {
75 if (argv[1][1] == 'n' && argv[1][2] == '\0') {
76 ns = 1;
77 j = 1;
78 } else
79 usage(prog: argv[0], rc: 1);
80 }
81 }
82
83 if (argc != j + 4)
84 usage(prog: argv[0], rc: 1);
85
86 if (stat(path: argv[j + 1], statbuf: &fileAttr) != 0) {
87 fprintf(stderr, format: "could not access file '%s'\n", argv[j + 1]);
88 return 2;
89 }
90
91 fd = fopen(filename: argv[j + 1], modes: "r");
92 if (! fd) {
93 fprintf(stderr, format: "could not open file '%s'\n", argv[j + 1]);
94 exit(status: 2);
95 }
96
97 bufferSize = atoi(nptr: argv[j + 2]);
98 nrOfLoops = atoi(nptr: argv[j + 3]);
99 if (bufferSize <= 0 || nrOfLoops <= 0) {
100 fprintf(stderr, format: "buffer size and nr of loops must be greater than zero.\n");
101 exit(status: 3);
102 }
103
104 XMLBuf = malloc(size: fileAttr.st_size);
105 fileSize = fread(ptr: XMLBuf, size: sizeof(char), n: fileAttr.st_size, stream: fd);
106 fclose(stream: fd);
107
108 if (ns)
109 parser = XML_ParserCreateNS(NULL, namespaceSeparator: '!');
110 else
111 parser = XML_ParserCreate(NULL);
112
113 i = 0;
114 XMLBufEnd = XMLBuf + fileSize;
115 while (i < nrOfLoops) {
116 XMLBufPtr = XMLBuf;
117 isFinal = 0;
118 tstart = clock();
119 do {
120 ptrdiff_t parseBufferSize = XMLBufEnd - XMLBufPtr;
121 if (parseBufferSize <= (ptrdiff_t)bufferSize)
122 isFinal = 1;
123 else
124 parseBufferSize = bufferSize;
125 assert(parseBufferSize <= (ptrdiff_t)bufferSize);
126 if (! XML_Parse(parser, s: XMLBufPtr, len: (int)parseBufferSize, isFinal)) {
127 fprintf(stderr,
128 format: "error '%" XML_FMT_STR "' at line %" XML_FMT_INT_MOD
129 "u character %" XML_FMT_INT_MOD "u\n",
130 XML_ErrorString(code: XML_GetErrorCode(parser)),
131 XML_GetCurrentLineNumber(parser),
132 XML_GetCurrentColumnNumber(parser));
133 free(ptr: XMLBuf);
134 XML_ParserFree(parser);
135 exit(status: 4);
136 }
137 XMLBufPtr += bufferSize;
138 } while (! isFinal);
139 tend = clock();
140 cpuTime += ((double)(tend - tstart)) / CLOCKS_PER_SEC;
141 XML_ParserReset(parser, NULL);
142 i++;
143 }
144
145 XML_ParserFree(parser);
146 free(ptr: XMLBuf);
147
148 printf(format: "%d loops, with buffer size %d. Average time per loop: %f\n",
149 nrOfLoops, bufferSize, cpuTime / (double)nrOfLoops);
150 return 0;
151}
