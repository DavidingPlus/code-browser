37#ifndef XML_STRUCTDATA_H

40# include "expat.h"

42typedef struct {
43 const XML_Char *str;
44 int data0;
45 int data1;
46 int data2;
47} StructDataEntry;
48
49typedef struct {
50 int count; /* Number of entries used */
51 int max_count; /* Number of StructDataEntry items in `entries` */
52 StructDataEntry *entries;
53} StructData;
54
55void StructData_Init(StructData *storage);
56
57void StructData_AddItem(StructData *storage, const XML_Char *s, int data0,
58 int data1, int data2);
59
60void StructData_CheckItems(StructData *storage, const StructDataEntry *expected,
61 int count);
62
63void StructData_Dispose(StructData *storage);
64
65#endif /* XML_STRUCTDATA_H */
