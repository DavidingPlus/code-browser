94static void XMLCALL
95characterData(void *userData, const XML_Char *s, int len) {
96 FILE *fp = ((XmlwfUserData *)userData)->fp;
97 for (; len > 0; --len, ++s) {
98 switch (*s) {
99 case T('&'):
100 fputts(T("&amp;"), stream: fp);
101 break;
102 case T('<'):
103 fputts(T("&lt;"), stream: fp);
104 break;
105 case T('>'):
106 fputts(T("&gt;"), stream: fp);
107 break;
108#ifdef W3C14N
109 case 13:
110 fputts(T("&#xD;"), fp);
111 break;
112#else
113 case T('"'):
114 fputts(T("&quot;"), stream: fp);
115 break;
116 case 9:
117 case 10:
118 case 13:
119 ftprintf(stream: fp, T("&#%d;"), *s);
120 break;
121#endif
122 default:
123 puttc(c: *s, stream: fp);
124 break;
125 }
126 }
127}

129static void
130attributeValue(FILE *fp, const XML_Char *s) {
131 puttc(T('='), stream: fp);
132 puttc(T('"'), stream: fp);
133 assert(s);
134 for (;;) {
135 switch (*s) {
136 case 0:
137 case NSSEP:
138 puttc(T('"'), stream: fp);
139 return;
140 case T('&'):
141 fputts(T("&amp;"), stream: fp);
142 break;
143 case T('<'):
144 fputts(T("&lt;"), stream: fp);
145 break;
146 case T('"'):
147 fputts(T("&quot;"), stream: fp);
148 break;
149#ifdef W3C14N
150 case 9:
151 fputts(T("&#x9;"), fp);
152 break;
153 case 10:
154 fputts(T("&#xA;"), fp);
155 break;
156 case 13:
157 fputts(T("&#xD;"), fp);
158 break;
159#else
160 case T('>'):
161 fputts(T("&gt;"), stream: fp);
162 break;
163 case 9:
164 case 10:
165 case 13:
166 ftprintf(stream: fp, T("&#%d;"), *s);
167 break;
168#endif
169 default:
170 puttc(c: *s, stream: fp);
171 break;
172 }
173 s++;
174 }

185static void XMLCALL
186startElement(void *userData, const XML_Char *name, const XML_Char **atts) {
187 int nAtts;
188 const XML_Char **p;
189 FILE *fp = ((XmlwfUserData *)userData)->fp;
190 puttc(T('<'), stream: fp);
191 fputts(s: name, stream: fp);
192
193 p = atts;
194 while (*p)
195 ++p;
196 nAtts = (int)((p - atts) >> 1);
197 if (nAtts > 1)
198 qsort(base: (void *)atts, nmemb: nAtts, size: sizeof(XML_Char *) * 2, compar: attcmp);
199 while (*atts) {
200 puttc(T(' '), stream: fp);
201 fputts(s: *atts++, stream: fp);
202 attributeValue(fp, s: *atts);
203 atts++;
204 }
205 puttc(T('>'), stream: fp);
206}

217static int
218nsattcmp(const void *p1, const void *p2) {
219 const XML_Char *att1 = *(const XML_Char *const *)p1;
220 const XML_Char *att2 = *(const XML_Char *const *)p2;
221 int sep1 = (tcsrchr(s: att1, NSSEP) != 0);
222 int sep2 = (tcsrchr(s: att2, NSSEP) != 0);
223 if (sep1 != sep2)
224 return sep1 - sep2;
225 return tcscmp(s1: att1, s2: att2);
226}

228static void XMLCALL
229startElementNS(void *userData, const XML_Char *name, const XML_Char **atts) {
230 int nAtts;
231 int nsi;
232 const XML_Char **p;
233 FILE *fp = ((XmlwfUserData *)userData)->fp;
234 const XML_Char *sep;
235 puttc(T('<'), stream: fp);
236
237 sep = tcsrchr(s: name, NSSEP);
238 if (sep) {
239 fputts(T("n1:"), stream: fp);
240 fputts(s: sep + 1, stream: fp);
241 fputts(T(" xmlns:n1"), stream: fp);
242 attributeValue(fp, s: name);
243 nsi = 2;
244 } else {
245 fputts(s: name, stream: fp);
246 nsi = 1;
247 }
248
249 p = atts;
250 while (*p)
251 ++p;
252 nAtts = (int)((p - atts) >> 1);
253 if (nAtts > 1)
254 qsort(base: (void *)atts, nmemb: nAtts, size: sizeof(XML_Char *) * 2, compar: nsattcmp);
255 while (*atts) {
256 name = *atts++;
257 sep = tcsrchr(s: name, NSSEP);
258 puttc(T(' '), stream: fp);
259 if (sep) {
260 ftprintf(stream: fp, T("n%d:"), nsi);
261 fputts(s: sep + 1, stream: fp);
262 } else
263 fputts(s: name, stream: fp);
264 attributeValue(fp, s: *atts);
265 if (sep) {
266 ftprintf(stream: fp, T(" xmlns:n%d"), nsi++);
267 attributeValue(fp, s: name);
268 }
269 atts++;
270 }
271 puttc(T('>'), stream: fp);
272}

304static XML_Char *
305xcsdup(const XML_Char *s) {
306 XML_Char *result;
307 int count = 0;
308 int numBytes;
309
310 /* Get the length of the string, including terminator */
311 while (s[count++] != 0) {
312 /* Do nothing */
313 }
314 numBytes = count * sizeof(XML_Char);
315 result = malloc(size: numBytes);
316 if (result == NULL)
317 return NULL;
318 memcpy(dest: result, src: s, n: numBytes);
319 return result;
320}

334freeNotations(XmlwfUserData *data) {
335 NotationList *notationListHead = data->notationListHead;
336

355static int
356xcscmp(const XML_Char *xs, const XML_Char *xt) {
357 while (*xs != 0 && *xt != 0) {
358 if (*xs < *xt)
359 return -1;
360 if (*xs > *xt)
361 return 1;
362 xs++;
363 xt++;
364 }
365 if (*xs < *xt)
366 return -1;
367 if (*xs > *xt)
368 return 1;
369 return 0;
370}

372static int
373notationCmp(const void *a, const void *b) {
374 const NotationList *const n1 = *(const NotationList *const *)a;
375 const NotationList *const n2 = *(const NotationList *const *)b;
376
377 return xcscmp(xs: n1->notationName, xt: n2->notationName);
378}

380static void XMLCALL
381endDoctypeDecl(void *userData) {
382 XmlwfUserData *data = (XmlwfUserData *)userData;
383 NotationList **notations;
384 int notationCount = 0;
385 NotationList *p;
386 int i;
387
388 /* How many notations do we have? */
389 for (p = data->notationListHead; p != NULL; p = p->next)
390 notationCount++;
391 if (notationCount == 0) {
392 /* Nothing to report */
393 free(ptr: (void *)data->currentDoctypeName);
394 data->currentDoctypeName = NULL;
395 return;
396 }
397
398 notations = malloc(size: notationCount * sizeof(NotationList *));
399 if (notations == NULL) {
400 fprintf(stderr, format: "Unable to sort notations");
401 freeNotations(data);
402 return;
403 }
404
405 for (p = data->notationListHead, i = 0; i < notationCount; p = p->next, i++) {
406 notations[i] = p;
407 }
408 qsort(base: notations, nmemb: notationCount, size: sizeof(NotationList *), compar: notationCmp);
409
410 /* Output the DOCTYPE header */
411 fputts(T("<!DOCTYPE "), stream: data->fp);
412 fputts(s: data->currentDoctypeName, stream: data->fp);
413 fputts(T(" [\n"), stream: data->fp);
414
415 /* Now the NOTATIONs */
416 for (i = 0; i < notationCount; i++) {
417 fputts(T("<!NOTATION "), stream: data->fp);
418 fputts(s: notations[i]->notationName, stream: data->fp);
419 if (notations[i]->publicId != NULL) {
420 fputts(T(" PUBLIC '"), stream: data->fp);
421 fputts(s: notations[i]->publicId, stream: data->fp);
422 puttc(T('\''), stream: data->fp);
423 if (notations[i]->systemId != NULL) {
424 puttc(T(' '), stream: data->fp);
425 puttc(T('\''), stream: data->fp);
426 fputts(s: notations[i]->systemId, stream: data->fp);
427 puttc(T('\''), stream: data->fp);
428 }
429 } else if (notations[i]->systemId != NULL) {
430 fputts(T(" SYSTEM '"), stream: data->fp);
431 fputts(s: notations[i]->systemId, stream: data->fp);
432 puttc(T('\''), stream: data->fp);
433 }
434 puttc(T('>'), stream: data->fp);
435 puttc(T('\n'), stream: data->fp);
436 }
437
438 /* Finally end the DOCTYPE */
439 fputts(T("]>\n"), stream: data->fp);
440
441 free(ptr: notations);
442 freeNotations(data);
443 free(ptr: (void *)data->currentDoctypeName);
444 data->currentDoctypeName = NULL;
445}

447static void XMLCALL
448notationDecl(void *userData, const XML_Char *notationName, const XML_Char *base,
449 const XML_Char *systemId, const XML_Char *publicId) {
450 XmlwfUserData *data = (XmlwfUserData *)userData;
451 NotationList *entry = malloc(size: sizeof(NotationList));
452 const char *errorMessage = "Unable to store NOTATION for output\n";
453
454 UNUSED_P(base);
455 if (entry == NULL) {
456 fputs(s: errorMessage, stderr);
457 return; /* Nothing we can really do about this */
458 }
459 entry->notationName = xcsdup(s: notationName);
460 if (entry->notationName == NULL) {
461 fputs(s: errorMessage, stderr);
462 free(ptr: entry);
463 return;
464 }
465 if (systemId != NULL) {
466 entry->systemId = xcsdup(s: systemId);
467 if (entry->systemId == NULL) {
468 fputs(s: errorMessage, stderr);
469 free(ptr: (void *)entry->notationName);
470 free(ptr: entry);
471 return;
472 }
473 } else {
474 entry->systemId = NULL;
475 }
476 if (publicId != NULL) {
477 entry->publicId = xcsdup(s: publicId);
478 if (entry->publicId == NULL) {
479 fputs(s: errorMessage, stderr);
480 free(ptr: (void *)entry->systemId); /* Safe if it's NULL */
481 free(ptr: (void *)entry->notationName);
482 free(ptr: entry);
483 return;
484 }
485 } else {
486 entry->publicId = NULL;
487 }
488
489 entry->next = data->notationListHead;
490 data->notationListHead = entry;
491}

586static void XMLCALL
587metaStartElement(void *userData, const XML_Char *name, const XML_Char **atts) {
588 XML_Parser parser = (XML_Parser)userData;
589 XmlwfUserData *data = (XmlwfUserData *)XML_GetUserData(parser);
590 FILE *fp = data->fp;
591 const XML_Char **specifiedAttsEnd
592 = atts + XML_GetSpecifiedAttributeCount(parser);
593 const XML_Char **idAttPtr;
594 int idAttIndex = XML_GetIdAttributeIndex(parser);
595 if (idAttIndex < 0)
596 idAttPtr = 0;
597 else
598 idAttPtr = atts + idAttIndex;
599
600 ftprintf(stream: fp, T("<starttag name=\"%s\""), name);
601 metaLocation(parser);
602 if (*atts) {
603 fputts(T(">\n"), stream: fp);
604 do {
605 ftprintf(stream: fp, T("<attribute name=\"%s\" value=\""), atts[0]);
606 characterData(userData: data, s: atts[1], len: (int)tcslen(s: atts[1]));
607 if (atts >= specifiedAttsEnd)
608 fputts(T("\" defaulted=\"yes\"/>\n"), stream: fp);
609 else if (atts == idAttPtr)
610 fputts(T("\" id=\"yes\"/>\n"), stream: fp);
611 else
612 fputts(T("\"/>\n"), stream: fp);
613 } while (*(atts += 2));
614 fputts(T("</starttag>\n"), stream: fp);
615 } else
616 fputts(T("/>\n"), stream: fp);
617}

803static int XMLCALL
804unknownEncoding(void *userData, const XML_Char *name, XML_Encoding *info) {
805 int cp;
806 static const XML_Char prefixL[] = T("windows-");
807 static const XML_Char prefixU[] = T("WINDOWS-");
808 int i;
809
810 UNUSED_P(userData);
811 for (i = 0; prefixU[i]; i++)
812 if (name[i] != prefixU[i] && name[i] != prefixL[i])
813 return 0;
814
815 cp = 0;
816 for (; name[i]; i++) {
817 static const XML_Char digits[] = T("0123456789");
818 const XML_Char *s = tcschr(s: digits, c: name[i]);
819 if (! s)
820 return 0;
821 cp *= 10;
822 cp += (int)(s - digits);
823 if (cp >= 0x10000)
824 return 0;
825 }
826 if (! codepageMap(cp, map: info->map))
827 return 0;
828 info->convert = unknownEncodingConvert;
829 /* We could just cast the code page integer to a void *,
830 and avoid the use of release. */
831 info->release = free;
832 info->data = malloc(size: sizeof(int));
833 if (! info->data)
834 return 0;
835 *(int *)info->data = cp;
836 return 1;
837}
