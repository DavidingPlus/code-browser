83static void
84reportError(XML_Parser parser, const XML_Char *filename) {
85 enum XML_Error code = XML_GetErrorCode(parser);
86 const XML_Char *message = XML_ErrorString(code);
87 if (message)
88 ftprintf(stdout,
89 T("%s") T(":%") T(XML_FMT_INT_MOD) T("u") T(":%")
90 T(XML_FMT_INT_MOD) T("u") T(": %s\n"),
91 filename, XML_GetErrorLineNumber(parser),
92 XML_GetErrorColumnNumber(parser), message);
93 else
94 ftprintf(stderr, T("%s: (unknown message %d)\n"), filename, code);
95}

97/* This implementation will give problems on files larger than INT_MAX. */
98static void
99processFile(const void *data, size_t size, const XML_Char *filename,
100 void *args) {
101 XML_Parser parser = ((PROCESS_ARGS *)args)->parser;
102 int *retPtr = ((PROCESS_ARGS *)args)->retPtr;
103 if (XML_Parse(parser, s: (const char *)data, len: (int)size, isFinal: 1) == XML_STATUS_ERROR) {
104 reportError(parser, filename);
105 *retPtr = 0;
106 } else
107 *retPtr = 1;
108}

119static const XML_Char *
120resolveSystemId(const XML_Char *base, const XML_Char *systemId,
121 XML_Char **toFree) {
122 XML_Char *s;
123 *toFree = 0;
124 if (! base || *systemId == T('/')
125#if defined(_WIN32)
126 || *systemId == T('\\')
127 || (isAsciiLetter(systemId[0]) && systemId[1] == T(':'))
128#endif
129 )
130 return systemId;
131 *toFree = (XML_Char *)malloc(size: (tcslen(s: base) + tcslen(s: systemId) + 2)
132 * sizeof(XML_Char));
133 if (! *toFree)
134 return systemId;
135 tcscpy(dest: *toFree, src: base);
136 s = *toFree;
137 if (tcsrchr(s: s, T('/')))
138 s = tcsrchr(s: s, T('/')) + 1;
139#if defined(_WIN32)
140 if (tcsrchr(s, T('\\')))
141 s = tcsrchr(s, T('\\')) + 1;
142#endif
143 tcscpy(dest: s, src: systemId);
144 return *toFree;
145}

147static int
148externalEntityRefFilemap(XML_Parser parser, const XML_Char *context,
149 const XML_Char *base, const XML_Char *systemId,
150 const XML_Char *publicId) {
151 int result;
152 XML_Char *s;
153 const XML_Char *filename;
154 XML_Parser entParser = XML_ExternalEntityParserCreate(parser, context, encoding: 0);
155 int filemapRes;
156 PROCESS_ARGS args;
157 UNUSED_P(publicId);
158 args.retPtr = &result;
159 args.parser = entParser;
160 filename = resolveSystemId(base, systemId, toFree: &s);
161 XML_SetBase(parser: entParser, base: filename);
162 filemapRes = filemap(name: filename, processor: processFile, arg: &args);
163 switch (filemapRes) {
164 case 0:
165 result = 0;
166 break;
167 case 2:
168 ftprintf(stderr,
169 T("%s: file too large for memory-mapping")
170 T(", switching to streaming\n"),
171 filename);
172 result = processStream(filename, parser: entParser);
173 break;
174 }
175 free(ptr: s);
176 XML_ParserFree(parser: entParser);
177 return result;
178}

180static int
181processStream(const XML_Char *filename, XML_Parser parser) {
182 /* passing NULL for filename means read input from stdin */
183 int fd = 0; /* 0 is the fileno for stdin */
184
185 if (filename != NULL) {
186 fd = topen(file: filename, O_BINARY | O_RDONLY);
187 if (fd < 0) {
188 tperror(s: filename);
189 return 0;
190 }
191 }
192 for (;;) {
193 int nread;
194 char *buf = (char *)XML_GetBuffer(parser, len: g_read_size_bytes);
195 if (! buf) {
196 if (filename != NULL)
197 close(fd: fd);
198 ftprintf(stderr, T("%s: out of memory\n"),
199 filename != NULL ? filename : T("xmlwf"));
200 return 0;
201 }
202 nread = read(fd: fd, buf: buf, nbytes: g_read_size_bytes);
203 if (nread < 0) {
204 tperror(s: filename != NULL ? filename : T("STDIN"));
205 if (filename != NULL)
206 close(fd: fd);
207 return 0;
208 }
209 if (XML_ParseBuffer(parser, len: nread, isFinal: nread == 0) == XML_STATUS_ERROR) {
210 reportError(parser, filename: filename != NULL ? filename : T("STDIN"));
211 if (filename != NULL)
212 close(fd: fd);
213 return 0;
214 }
215 if (nread == 0) {
216 if (filename != NULL)
217 close(fd: fd);
218 break;
219 ;
220 }
221 }
222 return 1;
223}

225static int
226externalEntityRefStream(XML_Parser parser, const XML_Char *context,
227 const XML_Char *base, const XML_Char *systemId,
228 const XML_Char *publicId) {
229 XML_Char *s;
230 const XML_Char *filename;
231 int ret;
232 XML_Parser entParser = XML_ExternalEntityParserCreate(parser, context, encoding: 0);
233 UNUSED_P(publicId);
234 filename = resolveSystemId(base, systemId, toFree: &s);
235 XML_SetBase(parser: entParser, base: filename);
236 ret = processStream(filename, parser: entParser);
237 free(ptr: s);
238 XML_ParserFree(parser: entParser);
239 return ret;
240}

242int
243XML_ProcessFile(XML_Parser parser, const XML_Char *filename, unsigned flags) {
244 int result;
245
246 if (! XML_SetBase(parser, base: filename)) {
247 ftprintf(stderr, T("%s: out of memory"), filename);
248 exit(status: 1);
249 }
250
251 if (flags & XML_EXTERNAL_ENTITIES)
252 XML_SetExternalEntityRefHandler(parser, handler: (flags & XML_MAP_FILE)
253 ? externalEntityRefFilemap
254 : externalEntityRefStream);
255 if (flags & XML_MAP_FILE) {
256 int filemapRes;
257 PROCESS_ARGS args;
258 args.retPtr = &result;
259 args.parser = parser;
260 filemapRes = filemap(name: filename, processor: processFile, arg: &args);
261 switch (filemapRes) {
262 case 0:
263 result = 0;
264 break;
265 case 2:
266 ftprintf(stderr,
267 T("%s: file too large for memory-mapping")
268 T(", switching to streaming\n"),
269 filename);
270 result = processStream(filename, parser);
271 break;
272 }
273 } else
274 result = processStream(filename, parser);
275 return result;
276}
