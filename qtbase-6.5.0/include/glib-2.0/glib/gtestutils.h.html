41/* assertion API */
42#define g_assert_cmpstr(s1, cmp, s2) G_STMT_START { \
43 const char *__s1 = (s1), *__s2 = (s2); \
44 if (g_strcmp0 (__s1, __s2) cmp 0) ; else \
45 g_assertion_message_cmpstr (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
46 #s1 " " #cmp " " #s2, __s1, #cmp, __s2); \
47 } G_STMT_END
48#define g_assert_cmpint(n1, cmp, n2) G_STMT_START { \
49 gint64 __n1 = (n1), __n2 = (n2); \
50 if (__n1 cmp __n2) ; else \
51 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
52 #n1 " " #cmp " " #n2, (long double) __n1, #cmp, (long double) __n2, 'i'); \
53 } G_STMT_END
54#define g_assert_cmpuint(n1, cmp, n2) G_STMT_START { \
55 guint64 __n1 = (n1), __n2 = (n2); \
56 if (__n1 cmp __n2) ; else \
57 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
58 #n1 " " #cmp " " #n2, (long double) __n1, #cmp, (long double) __n2, 'i'); \
59 } G_STMT_END
60#define g_assert_cmphex(n1, cmp, n2) G_STMT_START {\
61 guint64 __n1 = (n1), __n2 = (n2); \
62 if (__n1 cmp __n2) ; else \
63 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
64 #n1 " " #cmp " " #n2, (long double) __n1, #cmp, (long double) __n2, 'x'); \
65 } G_STMT_END
66#define g_assert_cmpfloat(n1,cmp,n2) G_STMT_START { \
67 long double __n1 = (long double) (n1), __n2 = (long double) (n2); \
68 if (__n1 cmp __n2) ; else \
69 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
70 #n1 " " #cmp " " #n2, (long double) __n1, #cmp, (long double) __n2, 'f'); \
71 } G_STMT_END
72#define g_assert_cmpfloat_with_epsilon(n1,n2,epsilon) \
73 G_STMT_START { \
74 double __n1 = (n1), __n2 = (n2), __epsilon = (epsilon); \
75 if (G_APPROX_VALUE (__n1, __n2, __epsilon)) ; else \
76 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
77 #n1 " == " #n2 " (+/- " #epsilon ")", __n1, "==", __n2, 'f'); \
78 } G_STMT_END
79#define g_assert_cmpmem(m1, l1, m2, l2) G_STMT_START {\
80 gconstpointer __m1 = m1, __m2 = m2; \
81 int __l1 = l1, __l2 = l2; \
82 if (__l1 != 0 && __m1 == NULL) \
83 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
84 "assertion failed (" #l1 " == 0 || " #m1 " != NULL)"); \
85 else if (__l2 != 0 && __m2 == NULL) \
86 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
87 "assertion failed (" #l2 " == 0 || " #m2 " != NULL)"); \
88 else if (__l1 != __l2) \
89 g_assertion_message_cmpnum (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
90 #l1 " (len(" #m1 ")) == " #l2 " (len(" #m2 "))", \
91 (long double) __l1, "==", (long double) __l2, 'i'); \
92 else if (__l1 != 0 && __m2 != NULL && memcmp (__m1, __m2, __l1) != 0) \
93 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
94 "assertion failed (" #m1 " == " #m2 ")"); \
95 } G_STMT_END
96#define g_assert_cmpvariant(v1, v2) \
97 G_STMT_START \
98 { \
99 GVariant *__v1 = (v1), *__v2 = (v2); \
100 if (!g_variant_equal (__v1, __v2)) \
101 { \
102 gchar *__s1, *__s2, *__msg; \
103 __s1 = g_variant_print (__v1, TRUE); \
104 __s2 = g_variant_print (__v2, TRUE); \
105 __msg = g_strdup_printf ("assertion failed (" #v1 " == " #v2 "): %s does not equal %s", __s1, __s2); \
106 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, __msg); \
107 g_free (__s1); \
108 g_free (__s2); \
109 g_free (__msg); \
110 } \
111 } \
112 G_STMT_END
113#define g_assert_no_error(err) G_STMT_START { \
114 if (err) \
115 g_assertion_message_error (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
116 #err, err, 0, 0); \
117 } G_STMT_END
118#define g_assert_error(err, dom, c) G_STMT_START { \
119 if (!err || (err)->domain != dom || (err)->code != c) \
120 g_assertion_message_error (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
121 #err, err, dom, c); \
122 } G_STMT_END
123#define g_assert_true(expr) G_STMT_START { \
124 if G_LIKELY (expr) ; else \
125 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
126 "'" #expr "' should be TRUE"); \
127 } G_STMT_END
128#define g_assert_false(expr) G_STMT_START { \
129 if G_LIKELY (!(expr)) ; else \
130 g_assertion_message (G_LOG_DOMAIN, __FILE__, __LINE__, G_STRFUNC, \
131 "'" #expr "' should be FALSE"); \
132 } G_STMT_END

196/**
197 * G_TEST_OPTION_ISOLATE_DIRS:
198 *
199 * Creates a unique temporary directory for each unit test and uses
200 * g_set_user_dirs() to set XDG directories to point into subdirectories of it
201 * for the duration of the unit test. The directory tree is cleaned up after the
202 * test finishes successfully. Note that this doesn’t take effect until
203 * g_test_run() is called, so calls to (for example) g_get_user_home_dir() will
204 * return the system-wide value when made in a test program’s main() function.
205 *
206 * The following functions will return subdirectories of the temporary directory
207 * when this option is used. The specific subdirectory paths in use are not
208 * guaranteed to be stable API — always use a getter function to retrieve them.
209 *
210 * - g_get_home_dir()
211 * - g_get_user_cache_dir()
212 * - g_get_system_config_dirs()
213 * - g_get_user_config_dir()
214 * - g_get_system_data_dirs()
215 * - g_get_user_data_dir()
216 * - g_get_user_runtime_dir()
217 *
218 * The subdirectories may not be created by the test harness; as with normal
219 * calls to functions like g_get_user_cache_dir(), the caller must be prepared
220 * to create the directory if it doesn’t exist.
221 *
222 * Since: 2.60
223 */
224#define G_TEST_OPTION_ISOLATE_DIRS "isolate_dirs"

226/* While we discourage its use, g_assert() is often used in unit tests
227 * (especially in legacy code). g_assert_*() should really be used instead.
228 * g_assert() can be disabled at client program compile time, which can render
229 * tests useless. Highlight that to the user. */
230#ifdef G_DISABLE_ASSERT
231#if defined(G_HAVE_ISO_VARARGS)
232#define g_test_init(argc, argv, ...) \
233 G_STMT_START { \
234 g_printerr ("Tests were compiled with G_DISABLE_ASSERT and are likely no-ops. Aborting.\n"); \
235 exit (1); \
236 } G_STMT_END
237#elif defined(G_HAVE_GNUC_VARARGS)
238#define g_test_init(argc, argv...) \
239 G_STMT_START { \
240 g_printerr ("Tests were compiled with G_DISABLE_ASSERT and are likely no-ops. Aborting.\n"); \
241 exit (1); \
242 } G_STMT_END
243#else /* no varargs */
244 /* do nothing */
245#endif /* varargs support */
246#endif /* G_DISABLE_ASSERT */

291/**
292 * g_test_add:
293 * @testpath: The test path for a new test case.
294 * @Fixture: The type of a fixture data structure.
295 * @tdata: Data argument for the test functions.
296 * @fsetup: The function to set up the fixture data.
297 * @ftest: The actual test function.
298 * @fteardown: The function to tear down the fixture data.
299 *
300 * Hook up a new test case at @testpath, similar to g_test_add_func().
301 * A fixture data structure with setup and teardown functions may be provided,
302 * similar to g_test_create_case().
303 *
304 * g_test_add() is implemented as a macro, so that the fsetup(), ftest() and
305 * fteardown() callbacks can expect a @Fixture pointer as their first argument
306 * in a type safe manner. They otherwise have type #GTestFixtureFunc.
307 *
308 * Since: 2.16
309 */
310#define g_test_add(testpath, Fixture, tdata, fsetup, ftest, fteardown) \
311 G_STMT_START { \
312 void (*add_vtable) (const char*, \
313 gsize, \
314 gconstpointer, \
315 void (*) (Fixture*, gconstpointer), \
316 void (*) (Fixture*, gconstpointer), \
317 void (*) (Fixture*, gconstpointer)) = (void (*) (const gchar *, gsize, gconstpointer, void (*) (Fixture*, gconstpointer), void (*) (Fixture*, gconstpointer), void (*) (Fixture*, gconstpointer))) g_test_add_vtable; \
318 add_vtable \
319 (testpath, sizeof (Fixture), tdata, fsetup, ftest, fteardown); \
320 } G_STMT_END

348/**
349 * GTestTrapFlags:
350 * @G_TEST_TRAP_SILENCE_STDOUT: Redirect stdout of the test child to
351 * `/dev/null` so it cannot be observed on the console during test
352 * runs. The actual output is still captured though to allow later
353 * tests with g_test_trap_assert_stdout().
354 * @G_TEST_TRAP_SILENCE_STDERR: Redirect stderr of the test child to
355 * `/dev/null` so it cannot be observed on the console during test
356 * runs. The actual output is still captured though to allow later
357 * tests with g_test_trap_assert_stderr().
358 * @G_TEST_TRAP_INHERIT_STDIN: If this flag is given, stdin of the
359 * child process is shared with stdin of its parent process.
360 * It is redirected to `/dev/null` otherwise.
361 *
362 * Test traps are guards around forked tests.
363 * These flags determine what traps to set.
364 *
365 * Deprecated: 2.38: #GTestTrapFlags is used only with g_test_trap_fork(),
366 * which is deprecated. g_test_trap_subprocess() uses
367 * #GTestSubprocessFlags.
368 */
369typedef enum {
370 G_TEST_TRAP_SILENCE_STDOUT = 1 << 7,
371 G_TEST_TRAP_SILENCE_STDERR = 1 << 8,
372 G_TEST_TRAP_INHERIT_STDIN = 1 << 9
373} GTestTrapFlags GLIB_DEPRECATED_TYPE_IN_2_38_FOR(GTestSubprocessFlags);
