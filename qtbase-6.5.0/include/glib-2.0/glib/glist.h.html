151void g_clear_list (GList **list_ptr,
152 GDestroyNotify destroy);
153
154#define g_clear_list(list_ptr, destroy) \
155 G_STMT_START { \
156 GList *_list; \
157 \
158 _list = *(list_ptr); \
159 if (_list) \
160 { \
161 *list_ptr = NULL; \
162 \
163 if ((destroy) != NULL) \
164 g_list_free_full (_list, (destroy)); \
165 else \
166 g_list_free (_list); \
167 } \
168 } G_STMT_END \
169 GLIB_AVAILABLE_MACRO_IN_2_64
