38/* GRand - a good and fast random number generator: Mersenne Twister
39 * see http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html for more info.
40 * The range functions return a value in the intervall [begin, end).
41 * int -> [0..2^32-1]
42 * int_range -> [begin..end-1]
43 * double -> [0..1)
44 * double_range -> [begin..end)
45 */
46

51 guint seed_length);
