80/**
81 * G_DATALIST_FLAGS_MASK:
82 *
83 * A bitmask that restricts the possible flags passed to
84 * g_datalist_set_flags(). Passing a flags value where
85 * flags & ~G_DATALIST_FLAGS_MASK != 0 is an error.
86 */
