24static inline void
25g_autoptr_cleanup_generic_gfree (void *p)
26{
27 void **pp = (void**)p;
28 g_free (mem: *pp);
29}
30
31static inline void
32g_autoptr_cleanup_gstring_free (GString *string)
33{
34 if (string)
35 g_string_free (string, TRUE);
36}
