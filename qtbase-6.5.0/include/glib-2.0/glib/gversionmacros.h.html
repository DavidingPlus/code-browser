238/* evaluates to the current stable version; for development cycles,
239 * this means the next stable target

241#if (GLIB_MINOR_VERSION % 2)
242#define GLIB_VERSION_CUR_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION + 1))
243#else
244#define GLIB_VERSION_CUR_STABLE (G_ENCODE_VERSION (GLIB_MAJOR_VERSION, GLIB_MINOR_VERSION))
245#endif

254/**
255 * GLIB_VERSION_MIN_REQUIRED:
256 *
257 * A macro that should be defined by the user prior to including
258 * the glib.h header.
259 * The definition should be one of the predefined GLib version
260 * macros: %GLIB_VERSION_2_26, %GLIB_VERSION_2_28,...
261 *
262 * This macro defines the earliest version of GLib that the package is
263 * required to be able to compile against.
264 *
265 * If the compiler is configured to warn about the use of deprecated
266 * functions, then using functions that were deprecated in version
267 * %GLIB_VERSION_MIN_REQUIRED or earlier will cause warnings (but
268 * using functions deprecated in later releases will not).
269 *
270 * Since: 2.32
271 */
272/* If the package sets GLIB_VERSION_MIN_REQUIRED to some future
273 * GLIB_VERSION_X_Y value that we don't know about, it will compare as
274 * 0 in preprocessor tests.
275 */
276#ifndef GLIB_VERSION_MIN_REQUIRED
277# define GLIB_VERSION_MIN_REQUIRED (GLIB_VERSION_CUR_STABLE)
278#elif GLIB_VERSION_MIN_REQUIRED == 0
279# undef GLIB_VERSION_MIN_REQUIRED
280# define GLIB_VERSION_MIN_REQUIRED (GLIB_VERSION_CUR_STABLE + 2)
281#endif
282
283/**
284 * GLIB_VERSION_MAX_ALLOWED:
285 *
286 * A macro that should be defined by the user prior to including
287 * the glib.h header.
288 * The definition should be one of the predefined GLib version
289 * macros: %GLIB_VERSION_2_26, %GLIB_VERSION_2_28,...
290 *
291 * This macro defines the latest version of the GLib API that the
292 * package is allowed to make use of.
293 *
294 * If the compiler is configured to warn about the use of deprecated
295 * functions, then using functions added after version
296 * %GLIB_VERSION_MAX_ALLOWED will cause warnings.
297 *
298 * Unless you are using GLIB_CHECK_VERSION() or the like to compile
299 * different code depending on the GLib version, then this should be
300 * set to the same value as %GLIB_VERSION_MIN_REQUIRED.
301 *
302 * Since: 2.32
303 */
304#if !defined (GLIB_VERSION_MAX_ALLOWED) || (GLIB_VERSION_MAX_ALLOWED == 0)
305# undef GLIB_VERSION_MAX_ALLOWED
306# define GLIB_VERSION_MAX_ALLOWED (GLIB_VERSION_CUR_STABLE)
307#endif
308
309/* sanity checks */
310#if GLIB_VERSION_MIN_REQUIRED > GLIB_VERSION_CUR_STABLE
311#error "GLIB_VERSION_MIN_REQUIRED must be <= GLIB_VERSION_CUR_STABLE"
312#endif
313#if GLIB_VERSION_MAX_ALLOWED < GLIB_VERSION_MIN_REQUIRED
314#error "GLIB_VERSION_MAX_ALLOWED must be >= GLIB_VERSION_MIN_REQUIRED"
315#endif
316#if GLIB_VERSION_MIN_REQUIRED < GLIB_VERSION_2_26
317#error "GLIB_VERSION_MIN_REQUIRED must be >= GLIB_VERSION_2_26"
318#endif
