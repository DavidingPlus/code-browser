39/**
40 * GFileInputStream:
41 *
42 * A subclass of GInputStream for opened files. This adds
43 * a few file-specific operations and seeking.
44 *
45 * #GFileInputStream implements #GSeekable.
46 **/
47typedef struct _GFileInputStreamClass GFileInputStreamClass;
48typedef struct _GFileInputStreamPrivate GFileInputStreamPrivate;
49
50struct _GFileInputStream
51{
52 GInputStream parent_instance;
53
54 /*< private >*/
55 GFileInputStreamPrivate *priv;
56};
57
58struct _GFileInputStreamClass
59{
60 GInputStreamClass parent_class;
61
62 goffset (* tell) (GFileInputStream *stream);
63 gboolean (* can_seek) (GFileInputStream *stream);
64 gboolean (* seek) (GFileInputStream *stream,
65 goffset offset,
66 GSeekType type,
67 GCancellable *cancellable,
68 GError **error);
69 GFileInfo * (* query_info) (GFileInputStream *stream,
70 const char *attributes,
71 GCancellable *cancellable,
72 GError **error);
73 void (* query_info_async) (GFileInputStream *stream,
74 const char *attributes,
75 int io_priority,
76 GCancellable *cancellable,
77 GAsyncReadyCallback callback,
78 gpointer user_data);
79 GFileInfo * (* query_info_finish) (GFileInputStream *stream,
80 GAsyncResult *result,
81 GError **error);
82
83 /*< private >*/
84 /* Padding for future expansion */
85 void (*_g_reserved1) (void);
86 void (*_g_reserved2) (void);
87 void (*_g_reserved3) (void);
88 void (*_g_reserved4) (void);
89 void (*_g_reserved5) (void);
90};
