39/**
40 * GFileIOStream:
41 *
42 * A subclass of GIOStream for opened files. This adds
43 * a few file-specific operations and seeking and truncating.
44 *
45 * #GFileIOStream implements GSeekable.
46 **/
47typedef struct _GFileIOStreamClass GFileIOStreamClass;
48typedef struct _GFileIOStreamPrivate GFileIOStreamPrivate;
49
50struct _GFileIOStream
51{
52 GIOStream parent_instance;
53
54 /*< private >*/
55 GFileIOStreamPrivate *priv;
56};
57
58struct _GFileIOStreamClass
59{
60 GIOStreamClass parent_class;
61
62 goffset (* tell) (GFileIOStream *stream);
63 gboolean (* can_seek) (GFileIOStream *stream);
64 gboolean (* seek) (GFileIOStream *stream,
65 goffset offset,
66 GSeekType type,
67 GCancellable *cancellable,
68 GError **error);
69 gboolean (* can_truncate) (GFileIOStream *stream);
70 gboolean (* truncate_fn) (GFileIOStream *stream,
71 goffset size,
72 GCancellable *cancellable,
73 GError **error);
74 GFileInfo * (* query_info) (GFileIOStream *stream,
75 const char *attributes,
76 GCancellable *cancellable,
77 GError **error);
78 void (* query_info_async) (GFileIOStream *stream,
79 const char *attributes,
80 int io_priority,
81 GCancellable *cancellable,
82 GAsyncReadyCallback callback,
83 gpointer user_data);
84 GFileInfo * (* query_info_finish) (GFileIOStream *stream,
85 GAsyncResult *result,
86 GError **error);
87 char * (* get_etag) (GFileIOStream *stream);
88
89 /* Padding for future expansion */
90 void (*_g_reserved1) (void);
91 void (*_g_reserved2) (void);
92 void (*_g_reserved3) (void);
93 void (*_g_reserved4) (void);
94 void (*_g_reserved5) (void);
95};
