40typedef struct _GActionMapInterface GActionMapInterface;

45 GTypeInterface g_iface;

47 GAction * (* lookup_action) (GActionMap *action_map,

49 void (* add_action) (GActionMap *action_map,

56{
57 const gchar *name;
58
59 void (* activate) (GSimpleAction *action,

62

64
65 const gchar *state;
66
67 void (* change_state) (GSimpleAction *action,
68 GVariant *value,
69 gpointer user_data);
70
71 /*< private >*/

73};
