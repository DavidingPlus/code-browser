35/**
36 * GSocketConnectable:
37 *
38 * Interface for objects that contain or generate a #GSocketAddress.
39 */
40typedef struct _GSocketConnectableIface GSocketConnectableIface;

42/**
43 * GSocketConnectableIface:
44 * @g_iface: The parent interface.
45 * @enumerate: Creates a #GSocketAddressEnumerator
46 * @proxy_enumerate: Creates a #GProxyAddressEnumerator
47 * @to_string: Format the connectableâ€™s address as a string for debugging.
48 * Implementing this is optional. (Since: 2.48)
49 *
50 * Provides an interface for returning a #GSocketAddressEnumerator
51 * and #GProxyAddressEnumerator
52 */
53struct _GSocketConnectableIface
54{
55 GTypeInterface g_iface;
56
57 /* Virtual Table */
58
59 GSocketAddressEnumerator * (* enumerate) (GSocketConnectable *connectable);
60
61 GSocketAddressEnumerator * (* proxy_enumerate) (GSocketConnectable *connectable);
62
63 gchar * (* to_string) (GSocketConnectable *connectable);
64};
