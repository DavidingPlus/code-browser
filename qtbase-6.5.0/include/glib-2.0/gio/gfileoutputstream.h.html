39/**
40 * GFileOutputStream:
41 *
42 * A subclass of GOutputStream for opened files. This adds
43 * a few file-specific operations and seeking and truncating.
44 *
45 * #GFileOutputStream implements GSeekable.
46 **/
47typedef struct _GFileOutputStreamClass GFileOutputStreamClass;
48typedef struct _GFileOutputStreamPrivate GFileOutputStreamPrivate;
49
50struct _GFileOutputStream
51{
52 GOutputStream parent_instance;
53
54 /*< private >*/
55 GFileOutputStreamPrivate *priv;
56};
57
58struct _GFileOutputStreamClass
59{
60 GOutputStreamClass parent_class;
61
62 goffset (* tell) (GFileOutputStream *stream);
63 gboolean (* can_seek) (GFileOutputStream *stream);
64 gboolean (* seek) (GFileOutputStream *stream,
65 goffset offset,
66 GSeekType type,
67 GCancellable *cancellable,
68 GError **error);
69 gboolean (* can_truncate) (GFileOutputStream *stream);
70 gboolean (* truncate_fn) (GFileOutputStream *stream,
71 goffset size,
72 GCancellable *cancellable,
73 GError **error);
74 GFileInfo * (* query_info) (GFileOutputStream *stream,
75 const char *attributes,
76 GCancellable *cancellable,
77 GError **error);
78 void (* query_info_async) (GFileOutputStream *stream,
79 const char *attributes,
80 int io_priority,
81 GCancellable *cancellable,
82 GAsyncReadyCallback callback,
83 gpointer user_data);
84 GFileInfo * (* query_info_finish) (GFileOutputStream *stream,
85 GAsyncResult *result,
86 GError **error);
87 char * (* get_etag) (GFileOutputStream *stream);
88
89 /* Padding for future expansion */
90 void (*_g_reserved1) (void);
91 void (*_g_reserved2) (void);
92 void (*_g_reserved3) (void);
93 void (*_g_reserved4) (void);
94 void (*_g_reserved5) (void);
95};
