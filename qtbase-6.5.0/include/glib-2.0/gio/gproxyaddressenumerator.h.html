39/**
40 * GProxyAddressEnumerator:
41 *
42 * A subclass of #GSocketAddressEnumerator that takes another address
43 * enumerator and wraps each of its results in a #GProxyAddress as
44 * directed by the default #GProxyResolver.
45 */
46
47typedef struct _GProxyAddressEnumeratorClass GProxyAddressEnumeratorClass;
48typedef struct _GProxyAddressEnumeratorPrivate GProxyAddressEnumeratorPrivate;
49
50struct _GProxyAddressEnumerator
51{
52 /*< private >*/
53 GSocketAddressEnumerator parent_instance;
54 GProxyAddressEnumeratorPrivate *priv;
55};

58 * GProxyAddressEnumeratorClass:
59 *
60 * Class structure for #GProxyAddressEnumerator.
61 */
62struct _GProxyAddressEnumeratorClass
63{
64 /*< private >*/
65 GSocketAddressEnumeratorClass parent_class;
66
67 void (*_g_reserved1) (void);

69 void (*_g_reserved3) (void);
70 void (*_g_reserved4) (void);
71 void (*_g_reserved5) (void);
72 void (*_g_reserved6) (void);
73 void (*_g_reserved7) (void);
74};
