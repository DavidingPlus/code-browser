37/**
38 * GLoadableIcon:
39 *
40 * Generic type for all kinds of icons that can be loaded
41 * as a stream.
42 **/
43typedef struct _GLoadableIconIface GLoadableIconIface;
44
45/**
46 * GLoadableIconIface:
47 * @g_iface: The parent interface.
48 * @load: Loads an icon.
49 * @load_async: Loads an icon asynchronously.
50 * @load_finish: Finishes an asynchronous icon load.
51 *
52 * Interface for icons that can be loaded as a stream.
53 **/
54struct _GLoadableIconIface
55{
56 GTypeInterface g_iface;
57
58 /* Virtual Table */
59
60 GInputStream * (* load) (GLoadableIcon *icon,
61 int size,
62 char **type,
63 GCancellable *cancellable,
64 GError **error);
65 void (* load_async) (GLoadableIcon *icon,
66 int size,
67 GCancellable *cancellable,
68 GAsyncReadyCallback callback,
69 gpointer user_data);
70 GInputStream * (* load_finish) (GLoadableIcon *icon,
71 GAsyncResult *res,
72 char **type,
73 GError **error);
74};
75

92GInputStream *g_loadable_icon_load_finish (GLoadableIcon *icon,

95 GError **error);
