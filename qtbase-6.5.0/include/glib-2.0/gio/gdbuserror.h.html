32/**
33 * G_DBUS_ERROR:
34 *
35 * Error domain for errors generated by a remote message bus. Errors
36 * in this domain will be from the #GDBusError enumeration. See
37 * #GError for more information on error domains.
38 *
39 * Note that this error domain is intended only for
40 * returning errors from a remote message bus process. Errors
41 * generated locally in-process by e.g. #GDBusConnection should use the
42 * %G_IO_ERROR domain.
43 *
44 * Since: 2.26
45 */
46#define G_DBUS_ERROR g_dbus_error_quark()
47
48GLIB_AVAILABLE_IN_ALL
49GQuark g_dbus_error_quark (void);
50
51/* Used by applications to check, get and strip the D-Bus error name */
52GLIB_AVAILABLE_IN_ALL
53gboolean g_dbus_error_is_remote_error (const GError *error);
54GLIB_AVAILABLE_IN_ALL
55gchar *g_dbus_error_get_remote_error (const GError *error);
56GLIB_AVAILABLE_IN_ALL
57gboolean g_dbus_error_strip_remote_error (GError *error);

59/**
60 * GDBusErrorEntry:
61 * @error_code: An error code.
62 * @dbus_error_name: The D-Bus error name to associate with @error_code.
63 *
64 * Struct used in g_dbus_error_register_error_domain().
65 *
66 * Since: 2.26
67 */
68struct _GDBusErrorEntry
69{
70 gint error_code;
71 const gchar *dbus_error_name;
72};
