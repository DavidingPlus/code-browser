75/**
76 * Clear a variable or struct member that contains a #DBusPendingCall.
77 * If it does not contain #NULL, the pending call that was previously
78 * there is unreferenced with dbus_pending_call_unref().
79 *
80 * This is very similar to dbus_clear_connection(): see that function
81 * for more details.
82 *
83 * @param pointer_to_pending_call A pointer to a variable or struct member.
84 * pointer_to_pending_call must not be #NULL, but *pointer_to_pending_call
85 * may be #NULL.
86 */
87static inline void
88dbus_clear_pending_call (DBusPendingCall **pointer_to_pending_call)
89{
90 _dbus_clear_pointer_impl (DBusPendingCall, pointer_to_pending_call,
91 dbus_pending_call_unref);
92}
