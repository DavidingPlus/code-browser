2/*
3 * fscrypt user API
4 *
5 * These ioctls can be used on filesystems that support fscrypt. See the
6 * "User API" section of Documentation/filesystems/fscrypt.rst.
7 */
8#ifndef _LINUX_FSCRYPT_H
9#define _LINUX_FSCRYPT_H
10
11#include <linux/types.h>
12
13/* Encryption policy flags */
14#define FSCRYPT_POLICY_FLAGS_PAD_4 0x00
15#define FSCRYPT_POLICY_FLAGS_PAD_8 0x01
16#define FSCRYPT_POLICY_FLAGS_PAD_16 0x02
17#define FSCRYPT_POLICY_FLAGS_PAD_32 0x03
18#define FSCRYPT_POLICY_FLAGS_PAD_MASK 0x03
19#define FSCRYPT_POLICY_FLAG_DIRECT_KEY 0x04
20
21/* Encryption algorithms */
22#define FSCRYPT_MODE_AES_256_XTS 1
23#define FSCRYPT_MODE_AES_256_CTS 4
24#define FSCRYPT_MODE_AES_128_CBC 5
25#define FSCRYPT_MODE_AES_128_CTS 6
26#define FSCRYPT_MODE_ADIANTUM 9
27/* If adding a mode number > 9, update FSCRYPT_MODE_MAX in fscrypt_private.h */
28
29/*
30 * Legacy policy version; ad-hoc KDF and no key verification.
31 * For new encrypted directories, use fscrypt_policy_v2 instead.
32 *
33 * Careful: the .version field for this is actually 0, not 1.
34 */
35#define FSCRYPT_POLICY_V1 0
36#define FSCRYPT_KEY_DESCRIPTOR_SIZE 8
37struct fscrypt_policy_v1 {
38 __u8 version;
39 __u8 contents_encryption_mode;
40 __u8 filenames_encryption_mode;
41 __u8 flags;
42 __u8 master_key_descriptor[FSCRYPT_KEY_DESCRIPTOR_SIZE];
43};
44#define fscrypt_policy fscrypt_policy_v1
45
46/*
47 * Process-subscribed "logon" key description prefix and payload format.
48 * Deprecated; prefer FS_IOC_ADD_ENCRYPTION_KEY instead.
49 */
50#define FSCRYPT_KEY_DESC_PREFIX "fscrypt:"
51#define FSCRYPT_KEY_DESC_PREFIX_SIZE 8
52#define FSCRYPT_MAX_KEY_SIZE 64
53struct fscrypt_key {
54 __u32 mode;
55 __u8 raw[FSCRYPT_MAX_KEY_SIZE];
56 __u32 size;
57};
58
59/*
60 * New policy version with HKDF and key verification (recommended).
61 */
62#define FSCRYPT_POLICY_V2 2
63#define FSCRYPT_KEY_IDENTIFIER_SIZE 16
64struct fscrypt_policy_v2 {
65 __u8 version;
66 __u8 contents_encryption_mode;
67 __u8 filenames_encryption_mode;
68 __u8 flags;
69 __u8 __reserved[4];
70 __u8 master_key_identifier[FSCRYPT_KEY_IDENTIFIER_SIZE];
71};
72
73/* Struct passed to FS_IOC_GET_ENCRYPTION_POLICY_EX */
74struct fscrypt_get_policy_ex_arg {
75 __u64 policy_size; /* input/output */
76 union {
77 __u8 version;
78 struct fscrypt_policy_v1 v1;
79 struct fscrypt_policy_v2 v2;
80 } policy; /* output */
81};
82
83/*
84 * v1 policy keys are specified by an arbitrary 8-byte key "descriptor",
85 * matching fscrypt_policy_v1::master_key_descriptor.
86 */
87#define FSCRYPT_KEY_SPEC_TYPE_DESCRIPTOR 1
88
89/*
90 * v2 policy keys are specified by a 16-byte key "identifier" which the kernel
91 * calculates as a cryptographic hash of the key itself,
92 * matching fscrypt_policy_v2::master_key_identifier.
93 */
94#define FSCRYPT_KEY_SPEC_TYPE_IDENTIFIER 2
95
96/*
97 * Specifies a key, either for v1 or v2 policies. This doesn't contain the
98 * actual key itself; this is just the "name" of the key.
99 */
100struct fscrypt_key_specifier {
101 __u32 type; /* one of FSCRYPT_KEY_SPEC_TYPE_* */
102 __u32 __reserved;
103 union {
104 __u8 __reserved[32]; /* reserve some extra space */
105 __u8 descriptor[FSCRYPT_KEY_DESCRIPTOR_SIZE];
106 __u8 identifier[FSCRYPT_KEY_IDENTIFIER_SIZE];
107 } u;
108};
109
110/* Struct passed to FS_IOC_ADD_ENCRYPTION_KEY */
111struct fscrypt_add_key_arg {
112 struct fscrypt_key_specifier key_spec;
113 __u32 raw_size;
114 __u32 __reserved[9];
115 __u8 raw[];
116};
117
118/* Struct passed to FS_IOC_REMOVE_ENCRYPTION_KEY */
119struct fscrypt_remove_key_arg {
120 struct fscrypt_key_specifier key_spec;
121#define FSCRYPT_KEY_REMOVAL_STATUS_FLAG_FILES_BUSY 0x00000001
122#define FSCRYPT_KEY_REMOVAL_STATUS_FLAG_OTHER_USERS 0x00000002
123 __u32 removal_status_flags; /* output */
124 __u32 __reserved[5];
125};
126
127/* Struct passed to FS_IOC_GET_ENCRYPTION_KEY_STATUS */
128struct fscrypt_get_key_status_arg {
129 /* input */
130 struct fscrypt_key_specifier key_spec;
131 __u32 __reserved[6];
132
133 /* output */
134#define FSCRYPT_KEY_STATUS_ABSENT 1
135#define FSCRYPT_KEY_STATUS_PRESENT 2
136#define FSCRYPT_KEY_STATUS_INCOMPLETELY_REMOVED 3
137 __u32 status;
138#define FSCRYPT_KEY_STATUS_FLAG_ADDED_BY_SELF 0x00000001
139 __u32 status_flags;
140 __u32 user_count;
141 __u32 __out_reserved[13];
142};
143
144#define FS_IOC_SET_ENCRYPTION_POLICY _IOR('f', 19, struct fscrypt_policy)
145#define FS_IOC_GET_ENCRYPTION_PWSALT _IOW('f', 20, __u8[16])
146#define FS_IOC_GET_ENCRYPTION_POLICY _IOW('f', 21, struct fscrypt_policy)
147#define FS_IOC_GET_ENCRYPTION_POLICY_EX _IOWR('f', 22, __u8[9]) /* size + version */
148#define FS_IOC_ADD_ENCRYPTION_KEY _IOWR('f', 23, struct fscrypt_add_key_arg)
149#define FS_IOC_REMOVE_ENCRYPTION_KEY _IOWR('f', 24, struct fscrypt_remove_key_arg)
150#define FS_IOC_REMOVE_ENCRYPTION_KEY_ALL_USERS _IOWR('f', 25, struct fscrypt_remove_key_arg)
151#define FS_IOC_GET_ENCRYPTION_KEY_STATUS _IOWR('f', 26, struct fscrypt_get_key_status_arg)
