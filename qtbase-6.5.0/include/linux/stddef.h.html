8/**
9 * __DECLARE_FLEX_ARRAY() - Declare a flexible array usable in a union
10 *
11 * @TYPE: The type of each flexible array element
12 * @NAME: The name of the flexible array member
13 *
14 * In order to have a flexible array member in a union or alone in a
15 * struct, it needs to be wrapped in an anonymous struct with at least 1
16 * named member, but that member can be empty.
17 */
18#define __DECLARE_FLEX_ARRAY(TYPE, NAME) \
19 struct { \
20 struct { } __empty_ ## NAME; \
21 TYPE NAME[]; \
22 }
