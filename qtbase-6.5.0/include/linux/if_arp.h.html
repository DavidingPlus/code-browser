35#define ARPHRD_CHAOS 5 /* Chaosnet */

37#define ARPHRD_ARCNET 7 /* ARCnet */

43#define ARPHRD_EUI64 27 /* EUI-64 */

59#define ARPHRD_HDLC ARPHRD_CISCO

65#define ARPHRD_TUNNEL 768 /* IPIP tunnel */

102#define ARPHRD_VOID 0xFFFF /* Void type, nothing is known */
103#define ARPHRD_NONE 0xFFFE /* zero header length */

105/* ARP protocol opcodes. */
106#define ARPOP_REQUEST 1 /* ARP request */
107#define ARPOP_REPLY 2 /* ARP reply */
108#define ARPOP_RREQUEST 3 /* RARP request */
109#define ARPOP_RREPLY 4 /* RARP reply */
110#define ARPOP_InREQUEST 8 /* InARP request */
111#define ARPOP_InREPLY 9 /* InARP reply */
112#define ARPOP_NAK 10 /* (ATM)ARP NAK */

114
115/* ARP ioctl request. */
116struct arpreq {
117 struct sockaddr arp_pa; /* protocol address */
118 struct sockaddr arp_ha; /* hardware address */
119 int arp_flags; /* flags */
120 struct sockaddr arp_netmask; /* netmask (only for proxy arps) */
121 char arp_dev[IFNAMSIZ];
122};
123
124struct arpreq_old {
125 struct sockaddr arp_pa; /* protocol address */
126 struct sockaddr arp_ha; /* hardware address */
127 int arp_flags; /* flags */
128 struct sockaddr arp_netmask; /* netmask (only for proxy arps) */
129};
130
131/* ARP Flag values. */
132#define ATF_COM 0x02 /* completed entry (ha valid) */
133#define ATF_PERM 0x04 /* permanent entry */
134#define ATF_PUBL 0x08 /* publish entry */
135#define ATF_USETRAILERS 0x10 /* has requested trailers */
136#define ATF_NETMASK 0x20 /* want to use a netmask (only
137 for proxy entries) */
138#define ATF_DONTPUB 0x40 /* don't answer this addresses */
139
140/*
141 * This structure defines an ethernet arp header.
142 */
143
144struct arphdr {
145 __be16 ar_hrd; /* format of hardware address */
146 __be16 ar_pro; /* format of protocol address */
147 unsigned char ar_hln; /* length of hardware address */
148 unsigned char ar_pln; /* length of protocol address */
149 __be16 ar_op; /* ARP opcode (command) */
