785/* Define macros that generate width-specific names from generic versions. The
786three-level macro scheme is necessary to get the macros expanded when we want
787them to be. First we get the width from PCRE2_LOCAL_WIDTH, which is used for
788generating three versions of everything below. After that, PCRE2_SUFFIX will be
789re-defined to use PCRE2_CODE_UNIT_WIDTH, for use when macros such as
790pcre2_compile are called by application code. */
791
792#define PCRE2_JOIN(a,b) a ## b
793#define PCRE2_GLUE(a,b) PCRE2_JOIN(a,b)
794#define PCRE2_SUFFIX(a) PCRE2_GLUE(a,PCRE2_LOCAL_WIDTH)
