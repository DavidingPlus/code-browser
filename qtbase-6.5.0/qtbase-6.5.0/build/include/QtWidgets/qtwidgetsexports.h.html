4#include <QtCore/qglobal.h>
5
6#ifndef QTWIDGETSEXPORTS_H
7#define QTWIDGETSEXPORTS_H
8
9#if defined(QT_SHARED) || !defined(QT_STATIC)
10# if defined(QT_BUILD_WIDGETS_LIB)
11# define Q_WIDGETS_EXPORT Q_DECL_EXPORT
12# else
13# define Q_WIDGETS_EXPORT Q_DECL_IMPORT
14# endif
15#else
16# define Q_WIDGETS_EXPORT
17#endif
18
19#if !defined(QT_BUILD_WIDGETS_LIB) && !defined(QT_STATIC)
20/* outside library → inline decl + defi */
21/* static builds treat everything as part of the library, so they never inline */
22# define QT_WIDGETS_INLINE_SINCE(major, minor) inline
23# define QT_WIDGETS_INLINE_IMPL_SINCE(major, minor) 1
24#elif defined(QT_WIDGETS_BUILD_REMOVED_API)
25/* inside library, inside removed_api.cpp:
26 * keep deprecated API → non-inline decl;
27 * remove deprecated API → inline decl;
28 * definition is always available */
29# define QT_WIDGETS_INLINE_SINCE(major, minor) \
30 QT_IF_DEPRECATED_SINCE(major, minor, inline, /* not inline */)
31# define QT_WIDGETS_INLINE_IMPL_SINCE(major, minor) 1
32#else
33/* inside library, outside removed_api.cpp:
34 * keep deprecated API → non-inline decl, no defi;
35 * remove deprecated API → inline decl, defi */
36# define QT_WIDGETS_INLINE_SINCE(major, minor) \
37 QT_IF_DEPRECATED_SINCE(major, minor, inline, /* not inline */)
38# define QT_WIDGETS_INLINE_IMPL_SINCE(major, minor) \
39 QT_IF_DEPRECATED_SINCE(major, minor, 1, 0)
40#endif
41
42#ifdef QT_WIDGETS_BUILD_REMOVED_API
43# define QT_WIDGETS_REMOVED_SINCE(major, minor) QT_DEPRECATED_SINCE(major, minor)
44#else
45# define QT_WIDGETS_REMOVED_SINCE(major, minor) 0
46#endif
47
48#endif // QTWIDGETSEXPORTS_H
