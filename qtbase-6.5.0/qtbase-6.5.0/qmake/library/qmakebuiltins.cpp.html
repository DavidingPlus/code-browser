85QMakeBuiltin::QMakeBuiltin(const QMakeBuiltinInit &d)
86 : index(d.func), minArgs(qMax(a: 0, b: d.min_args)), maxArgs(d.max_args)
87{
88 static const char * const nstr[6] = { "no", "one", "two", "three", "four", "five" };
89 // For legacy reasons, there is actually no such thing as "no arguments"
90 // - there is only "empty first argument", which needs to be mapped back.
91 // -1 means "one, which may be empty", which is effectively zero, except
92 // for the error message if there are too many arguments.
93 int dmin = qAbs(t: d.min_args);
94 int dmax = d.max_args;
95 if (dmax == QMakeBuiltinInit::VarArgs) {
96 Q_ASSERT_X(dmin < 2, "init", d.name);
97 if (dmin == 1) {
98 Q_ASSERT_X(d.args != nullptr, "init", d.name);
99 usage = fL1S("%1(%2) requires at least one argument.")
100 .arg(fL1S(d.name), fL1S(d.args));
101 }
102 return;
103 }
104 int arange = dmax - dmin;
105 Q_ASSERT_X(arange >= 0, "init", d.name);
106 Q_ASSERT_X(d.args != nullptr, "init", d.name);
107 usage = arange > 1
108 ? fL1S("%1(%2) requires %3 to %4 arguments.")
109 .arg(fL1S(d.name), fL1S(d.args), fL1S(nstr[dmin]), fL1S(nstr[dmax]))
110 : arange > 0
111 ? fL1S("%1(%2) requires %3 or %4 arguments.")
112 .arg(fL1S(d.name), fL1S(d.args), fL1S(nstr[dmin]), fL1S(nstr[dmax]))
113 : dmax != 1
114 ? fL1S("%1(%2) requires %3 arguments.")
115 .arg(fL1S(d.name), fL1S(d.args), fL1S(nstr[dmax]))
116 : fL1S("%1(%2) requires one argument.")
117 .arg(fL1S(d.name), fL1S(d.args));
118}

223bool
224QMakeEvaluator::getMemberArgs(const ProKey &func, int srclen, const ProStringList &args,
225 int *start, int *end)
226{
227 *start = 0, *end = 0;
228 if (args.size() >= 2) {
229 bool ok = true;
230 const ProString &start_str = args.at(i: 1);
231 *start = start_str.toInt(ok: &ok);
232 if (!ok) {
233 if (args.size() == 2) {
234 int dotdot = start_str.indexOf(s: statics.strDotDot);
235 if (dotdot != -1) {
236 *start = start_str.left(len: dotdot).toInt(ok: &ok);
237 if (ok)
238 *end = start_str.mid(off: dotdot+2).toInt(ok: &ok);
239 }
240 }
241 if (!ok) {
242 ProStringRoUser u1(func, m_tmp1);
243 ProStringRoUser u2(start_str, m_tmp2);
244 evalError(fL1S("%1() argument 2 (start) '%2' invalid.").arg(args&: u1.str(), args&: u2.str()));
245 return false;
246 }
247 } else {
248 *end = *start;
249 if (args.size() == 3)
250 *end = args.at(i: 2).toInt(ok: &ok);
251 if (!ok) {
252 ProStringRoUser u1(func, m_tmp1);
253 ProStringRoUser u2(args.at(i: 2), m_tmp2);
254 evalError(fL1S("%1() argument 3 (end) '%2' invalid.").arg(args&: u1.str(), args&: u2.str()));
255 return false;
256 }
257 }
258 }
259 if (*start < 0)
260 *start += srclen;
261 if (*end < 0)
262 *end += srclen;
263 if (*start < 0 || *start >= srclen || *end < 0 || *end >= srclen)
264 return false;
265 return true;
266}

268QString
269QMakeEvaluator::quoteValue(const ProString &val)
270{
271 QString ret;
272 ret.reserve(asize: val.size());
273 const QChar *chars = val.constData();
274 bool quote = val.isEmpty();
275 bool escaping = false;
276 for (int i = 0, l = val.size(); i < l; i++) {
277 QChar c = chars[i];
278 ushort uc = c.unicode();
279 if (uc < 32) {
280 if (!escaping) {
281 escaping = true;
282 ret += QLatin1String("$$escape_expand(");
283 }
284 switch (uc) {
285 case '\r':
286 ret += QLatin1String("\\\\r");
287 break;
288 case '\n':
289 ret += QLatin1String("\\\\n");
290 break;
291 case '\t':
292 ret += QLatin1String("\\\\t");
293 break;
294 default:
295 ret += QString::fromLatin1(ba: "\\\\x%1").arg(a: uc, fieldWidth: 2, base: 16, fillChar: QLatin1Char('0'));
296 break;
297 }
298 } else {
299 if (escaping) {
300 escaping = false;
301 ret += QLatin1Char(')');
302 }
303 switch (uc) {
304 case '\\':
305 ret += QLatin1String("\\\\");
306 break;
307 case '"':
308 ret += QLatin1String("\\\"");
309 break;
310 case '\'':
311 ret += QLatin1String("\\'");
312 break;
313 case '$':
314 ret += QLatin1String("\\$");
315 break;
316 case '#':
317 ret += QLatin1String("$${LITERAL_HASH}");
318 break;
319 case 32:
320 quote = true;
321 Q_FALLTHROUGH();
322 default:
323 ret += c;
324 break;
325 }
326 }
327 }
328 if (escaping)
329 ret += QLatin1Char(')');
330 if (quote) {
331 ret.prepend(c: QLatin1Char('"'));
332 ret.append(c: QLatin1Char('"'));
333 }
334 return ret;
335}

369static void addJsonValue(const QJsonValue &value, const QString &keyPrefix, ProValueMap *map)
370{
371 switch (value.type()) {
372 case QJsonValue::Bool:
373 insertJsonKeyValue(key: keyPrefix, values: QStringList() << (value.toBool() ? QLatin1String("true") : QLatin1String("false")), map);
374 break;
375 case QJsonValue::Double:
376 insertJsonKeyValue(key: keyPrefix, values: QStringList() << QString::number(value.toDouble()), map);
377 break;
378 case QJsonValue::String:
379 insertJsonKeyValue(key: keyPrefix, values: QStringList() << value.toString(), map);

397static ErrorPosition calculateErrorPosition(const QByteArray &json, int offset)
398{
399 ErrorPosition pos = { .line: 0, .column: 0 };
400 offset--; // offset is 1-based, switching to 0-based
401 for (int i = 0; i < offset; ++i) {
402 switch (json.at(i)) {
403 case '\n':
404 pos.line++;
405 pos.column = 0;
406 break;
407 case '\r':
408 break;
409 case '\t':
410 pos.column = (pos.column + 8) & ~7;
411 break;
412 default:
413 pos.column++;
414 break;
415 }
416 }

418 pos.line++;
