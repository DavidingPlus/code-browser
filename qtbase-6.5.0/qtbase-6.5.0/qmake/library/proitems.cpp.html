15size_t ProString::hash(const QChar *p, int n)
16{
17 size_t h = 0;
18
19 while (n--) {
20 h = (h << 4) + (*p++).unicode();
21 h ^= (h & 0xf0000000) >> 23;
22 h &= 0x0fffffff;
23 }
24 return h;
25}

188// If pending != 0, prefix with space if appending to non-empty non-pending
189ProString &ProString::append(const ProString &other, bool *pending)
190{
191 if (other.m_length) {
192 if (!m_length) {
193 *this = other;
194 } else {
195 if (m_length != m_string.size())
196 m_string = toQString();
197 if (pending && !*pending) {
198 m_string += QLatin1Char(' ') + other.toQStringView();
199 } else {
200 m_string += other.toQStringView();
201 }
202 m_length = m_string.size();
203 m_offset = 0;
204 if (other.m_file)
205 m_file = other.m_file;
206 m_hash = 0x80000000;
207 }
208 if (pending)
209 *pending = true;
210 }
211 return *this;
212}

214ProString &ProString::append(const ProStringList &other, bool *pending, bool skipEmpty1st)
215{
216 if (const int sz = other.size()) {
217 int startIdx = 0;
218 if (pending && !*pending && skipEmpty1st && other.at(i: 0).isEmpty()) {
219 if (sz == 1)
220 return *this;
221 startIdx = 1;
222 }
223 if (!m_length && sz == startIdx + 1) {
224 *this = other.at(i: startIdx);
225 } else {
226 int totalLength = sz - startIdx;
227 for (int i = startIdx; i < sz; ++i)
228 totalLength += other.at(i).size();
229 bool putSpace = false;
230 if (pending && !*pending && m_length)
231 putSpace = true;
232 else
233 totalLength--;
234
235 m_string = toQString();
236 m_offset = 0;
237 for (int i = startIdx; i < sz; ++i) {
238 if (putSpace)
239 m_string += QLatin1Char(' ');
240 else
241 putSpace = true;
242 const ProString &str = other.at(i);
243 m_string += str.toQStringView();
244 }
245 m_length = m_string.size();
246 if (other.last().m_file)
247 m_file = other.last().m_file;
248 m_hash = 0x80000000;
249 }
250 if (pending)
251 *pending = true;
252 }
253 return *this;
254}

256QString operator+(const ProString &one, const ProString &two)
257{
258 if (two.m_length) {
259 if (!one.m_length) {
260 return two.toQString();
261 } else {
262 QString neu(one.m_length + two.m_length, Qt::Uninitialized);
263 ushort *ptr = (ushort *)neu.constData();
264 memcpy(dest: ptr, src: one.m_string.constData() + one.m_offset, n: one.m_length * 2);
265 memcpy(dest: ptr + one.m_length, src: two.m_string.constData() + two.m_offset, n: two.m_length * 2);
266 return neu;
267 }
268 }
269 return one.toQString();
270}

273ProString ProString::mid(int off, int len) const
274{
275 ProString ret(*this, NoHash);
276 if (off > m_length)
277 off = m_length;
278 ret.m_offset += off;
279 ret.m_length -= off;
280 if ((uint)ret.m_length > (uint)len) // Unsigned comparison to interpret < 0 as infinite
281 ret.m_length = len;
282 return ret;
283}

285ProString ProString::trimmed() const
286{
287 ProString ret(*this, NoHash);
288 int cur = m_offset;
289 int end = cur + m_length;
290 const QChar *data = m_string.constData();
291 for (; cur < end; cur++)
292 if (!data[cur].isSpace()) {
293 // No underrun check - we know there is at least one non-whitespace
294 while (data[end - 1].isSpace())
295 end--;
296 break;
297 }
298 ret.m_offset = cur;
299 ret.m_length = end - cur;
300 return ret;
301}

309static QString ProStringList_join(const ProStringList &this_, const QChar *sep, const int sepSize)
310{
311 int totalLength = 0;
312 const int sz = this_.size();
313
314 for (int i = 0; i < sz; ++i)
315 totalLength += this_.at(i).size();
316
317 if (sz)
318 totalLength += sepSize * (sz - 1);
319
320 QString res(totalLength, Qt::Uninitialized);
321 QChar *ptr = (QChar *)res.constData();
322 for (int i = 0; i < sz; ++i) {
323 if (i) {
324 memcpy(dest: ptr, src: sep, n: sepSize * sizeof(QChar));
325 ptr += sepSize;
326 }
327 const ProString &str = this_.at(i);
328 memcpy(dest: ptr, src: str.constData(), n: str.size() * sizeof(QChar));
329 ptr += str.size();
330 }
331 return res;
332}
