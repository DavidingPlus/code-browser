10void qmakePropertyPrinter(const QList<QPair<QString, QString>> &values)
11{
12 // Assume single property request
13 if (values.size() == 1) {
14 std::cout << qPrintable(values.at(0).second) << std::endl;
15 return;
16 }
17
18 for (const auto &val : values) {
19 std::cout << qPrintable(val.first) << ":" << qPrintable(val.second) << std::endl;
20 }
21}
22
23void jsonPropertyPrinter(const QList<QPair<QString, QString>> &values)
24{
25 std::cout << "{\n";
26 for (const auto &val : values) {
27 std::cout << "\"" << qPrintable(val.first) << "\":\"" << qPrintable(val.second) << "\",\n";
28 }
29 std::cout << "}\n";
30}
