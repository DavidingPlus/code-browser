296struct ReplaceExtraCompilerCacheKey
297{
298 mutable size_t hash;
299 QString var, in, out, pwd;
300 MakefileGenerator::ReplaceFor forShell;
301 ReplaceExtraCompilerCacheKey(const QString &v, const QStringList &i, const QStringList &o, MakefileGenerator::ReplaceFor s);
302 bool operator==(const ReplaceExtraCompilerCacheKey &f) const;
303 inline size_t hashCode() const {
304 if (!hash)
305 hash = (size_t)forShell ^ qHash(key: var) ^ qHash(key: in) ^ qHash(key: out) /*^ qHash(pwd)*/;
306 return hash;
307 }
308};
309inline size_t qHash(const ReplaceExtraCompilerCacheKey &f) { return f.hashCode(); }
