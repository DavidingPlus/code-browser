14ProStringList UnixMakefileGenerator::libdirToFlags(const ProKey &key)
15{
16 ProStringList results;
17 for (const auto &libdir : std::as_const(t&: project->values(v: key))) {
18 if (!project->isEmpty(v: "QMAKE_LFLAGS_RPATH") && project->isActiveConfig(config: "rpath_libdirs"))
19 project->values(v: "QMAKE_LFLAGS") += var(var: "QMAKE_LFLAGS_RPATH") + libdir;
20 results.append(t: "-L" + escapeFilePath(path: libdir));
21 }
22 return results;
23}

25void
26UnixMakefileGenerator::init()
27{
28 ProStringList &configs = project->values(v: "CONFIG");
29 if(project->isEmpty(v: "ICON") && !project->isEmpty(v: "RC_FILE"))
30 project->values(v: "ICON") = project->values(v: "RC_FILE");
31 if(project->isEmpty(v: "QMAKE_EXTENSION_PLUGIN"))
32 project->values(v: "QMAKE_EXTENSION_PLUGIN").append(t: project->first(variableName: "QMAKE_EXTENSION_SHLIB"));
33
34 project->values(v: "QMAKE_ORIG_TARGET") = project->values(v: "TARGET");
35
36 //version handling
37 if (project->isEmpty(v: "VERSION")) {
38 project->values(v: "VERSION").append(
39 t: "1.0." + (project->isEmpty(v: "VER_PAT") ? QString("0") : project->first(variableName: "VER_PAT")));
40 }
41 QStringList l = project->first(variableName: "VERSION").toQString().split(sep: '.');
42 l << "0" << "0"; //make sure there are three
43 project->values(v: "VER_MAJ").append(t: l[0]);
44 project->values(v: "VER_MIN").append(t: l[1]);
45 project->values(v: "VER_PAT").append(t: l[2]);
46
47 QString sroot = project->sourceRoot();
48 for (const ProString &iif : project->values(v: "QMAKE_INTERNAL_INCLUDED_FILES")) {
49 if (iif == project->cacheFile())
50 continue;
51 if (iif.startsWith(sub: sroot) && iif.at(i: sroot.size()) == QLatin1Char('/'))
52 project->values(v: "DISTFILES") += fileFixify(file: iif.toQString(), fix: FileFixifyRelative);
53 }
54
55 /* this should probably not be here, but I'm using it to wrap the .t files */
56 if(project->first(variableName: "TEMPLATE") == "app")
57 project->values(v: "QMAKE_APP_FLAG").append(t: "1");
58 else if(project->first(variableName: "TEMPLATE") == "lib")
59 project->values(v: "QMAKE_LIB_FLAG").append(t: "1");
60 else if(project->first(variableName: "TEMPLATE") == "subdirs") {
61 MakefileGenerator::init();
62 if(project->isEmpty(v: "MAKEFILE"))
63 project->values(v: "MAKEFILE").append(t: "Makefile");
64 return; /* subdirs is done */
65 }
66
67 project->values(v: "QMAKE_ORIG_DESTDIR") = project->values(v: "DESTDIR");
68 if((!project->isEmpty(v: "QMAKE_LIB_FLAG") && !project->isActiveConfig(config: "staticlib")) ||
69 (project->isActiveConfig(config: "qt") && project->isActiveConfig(config: "plugin"))) {
70 if(configs.indexOf(t: "dll") == -1) configs.append(t: "dll");
71 } else if(!project->isEmpty(v: "QMAKE_APP_FLAG") || project->isActiveConfig(config: "dll")) {
72 configs.removeAll(str: "staticlib");
73 }
74 if(!project->isEmpty(v: "QMAKE_INCREMENTAL"))
75 project->values(v: "QMAKE_LFLAGS") += project->values(v: "QMAKE_LFLAGS_INCREMENTAL");
76 else if(!project->isEmpty(v: "QMAKE_LFLAGS_PREBIND") &&
77 !project->values(v: "QMAKE_LIB_FLAG").isEmpty() &&
78 project->isActiveConfig(config: "dll"))
79 project->values(v: "QMAKE_LFLAGS") += project->values(v: "QMAKE_LFLAGS_PREBIND");
80 project->values(v: "QMAKE_INCDIR") += project->values(v: "QMAKE_INCDIR_POST");
81 project->values(v: "QMAKE_RPATHDIR") += project->values(v: "QMAKE_RPATHDIR_POST");
82 project->values(v: "QMAKE_RPATHLINKDIR") += project->values(v: "QMAKE_RPATHLINKDIR_POST");
83 if(!project->isEmpty(v: "QMAKE_INCDIR"))
84 project->values(v: "INCLUDEPATH") += project->values(v: "QMAKE_INCDIR");
85 // The order of the next two lines is relevant due to side effect on QMAKE_LFLAGS.
86 ProStringList ldadd = project->values(v: "QMAKE_LIBDIR_FLAGS") + libdirToFlags(key: "QMAKE_LIBDIR");
87 ProStringList ldaddpost = libdirToFlags(key: "QMAKE_LIBDIR_POST");
88 if (project->isActiveConfig(config: "mac")) {
89 if (!project->isEmpty(v: "QMAKE_FRAMEWORKPATH")) {
90 const ProStringList &fwdirs = project->values(v: "QMAKE_FRAMEWORKPATH");
91 for (int i = 0; i < fwdirs.size(); ++i)
92 project->values(v: "QMAKE_FRAMEWORKPATH_FLAGS") += "-F" + escapeFilePath(path: fwdirs[i]);
93 }
94 ldadd += project->values(v: "QMAKE_FRAMEWORKPATH_FLAGS");
95 }
96 ProStringList &qmklibs = project->values(v: "LIBS");
97 qmklibs = ldadd + qmklibs;
98 ProStringList &qmklibspost = project->values(v: "QMAKE_LIBS");
99 qmklibspost = ldaddpost + qmklibspost;
100 if (!project->isEmpty(v: "QMAKE_RPATHDIR") && !project->isEmpty(v: "QMAKE_LFLAGS_RPATH")) {
101 const ProStringList &rpathdirs = project->values(v: "QMAKE_RPATHDIR");
102 for (int i = 0; i < rpathdirs.size(); ++i) {
103 QString rpathdir = rpathdirs[i].toQString();
104 if (rpathdir.size() > 1 && rpathdir.at(i: 0) == '$' && rpathdir.at(i: 1) != '(') {
105 rpathdir.replace(i: 0, len: 1, after: "\\$$"); // Escape from make and the shell
106 } else if (!rpathdir.startsWith(c: '@') && fileInfo(file: rpathdir).isRelative()) {
107 QString rpathbase = project->first(variableName: "QMAKE_REL_RPATH_BASE").toQString();
108 if (rpathbase.isEmpty()) {
109 fprintf(stderr, format: "Error: This platform does not support relative paths in QMAKE_RPATHDIR (%s)\n",
110 rpathdir.toLatin1().constData());
111 continue;
112 }
113 if (rpathbase.startsWith(c: '$'))
114 rpathbase.replace(i: 0, len: 1, after: "\\$$"); // Escape from make and the shell
115 if (rpathdir == ".")
116 rpathdir = rpathbase;
117 else
118 rpathdir.prepend(s: rpathbase + '/');
119 project->values(v: "QMAKE_LFLAGS").insertUnique(value: project->values(v: "QMAKE_LFLAGS_REL_RPATH"));
120 }
121 project->values(v: "QMAKE_LFLAGS") += var(var: "QMAKE_LFLAGS_RPATH") + escapeFilePath(path: rpathdir);
122 }
123 }
124 if (!project->isEmpty(v: "QMAKE_RPATHLINKDIR")) {
125 const ProStringList &rpathdirs = project->values(v: "QMAKE_RPATHLINKDIR");
126 for (int i = 0; i < rpathdirs.size(); ++i) {
127 if (!project->isEmpty(v: "QMAKE_LFLAGS_RPATHLINK"))
128 project->values(v: "QMAKE_LFLAGS") += var(var: "QMAKE_LFLAGS_RPATHLINK") + escapeFilePath(path: QFileInfo(rpathdirs[i].toQString()).absoluteFilePath());
129 }
130 }
131
132 if(project->isActiveConfig(config: "GNUmake") && !project->isEmpty(v: "QMAKE_CFLAGS_DEPS"))
133 include_deps = true; //do not generate deps
134
135 MakefileGenerator::init();
136
137 if (project->isActiveConfig(config: "objective_c"))
138 project->values(v: "QMAKE_BUILTIN_COMPILERS") << "OBJC" << "OBJCXX";
139
140 for (const ProString &compiler : project->values(v: "QMAKE_BUILTIN_COMPILERS")) {
141 QString compile_flag = var(var: "QMAKE_COMPILE_FLAG");
142 if(compile_flag.isEmpty())
143 compile_flag = "-c";
144
145 if(doPrecompiledHeaders() && !project->isEmpty(v: "PRECOMPILED_HEADER")) {
146 QString pchFlags = var(var: ProKey("QMAKE_" + compiler + "FLAGS_USE_PRECOMPILE"));
147
148 QString pchBaseName;
149 if(!project->isEmpty(v: "PRECOMPILED_DIR")) {
150 pchBaseName = Option::fixPathToTargetOS(in: project->first(variableName: "PRECOMPILED_DIR").toQString());
151 if(!pchBaseName.endsWith(s: Option::dir_sep))
152 pchBaseName += Option::dir_sep;
153 }
154 pchBaseName += project->first(variableName: "QMAKE_ORIG_TARGET").toQString();
155
156 // replace place holders
157 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_INPUT}"),
158 after: escapeFilePath(path: project->first(variableName: "PRECOMPILED_HEADER").toQString()));
159 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_OUTPUT_BASE}"), after: escapeFilePath(path: pchBaseName));
160 if (project->isActiveConfig(config: "icc_pch_style")) {
161 // icc style
162 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_OUTPUT}"),
163 after: escapeFilePath(path: pchBaseName + project->first(variableName: "QMAKE_PCH_OUTPUT_EXT")));
164 const ProStringList pchArchs = project->values(v: "QMAKE_PCH_ARCHS");
165 for (const ProString &arch : pchArchs) {
166 QString suffix = project->first(variableName: "QMAKE_PCH_OUTPUT_EXT").toQString();
167 suffix.replace(before: QLatin1String("${QMAKE_PCH_ARCH}"), after: arch.toQString());
168 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_OUTPUT_") + arch + QLatin1Char('}'),
169 after: escapeFilePath(path: pchBaseName + suffix));
170 }
171 } else {
172 // gcc style (including clang_pch_style)
173 QString headerSuffix;
174 if (project->isActiveConfig(config: "clang_pch_style"))
175 headerSuffix = project->first(variableName: "QMAKE_PCH_OUTPUT_EXT").toQString();
176
177 pchBaseName += project->first(variableName: "QMAKE_PCH_OUTPUT_EXT").toQString();
178 pchBaseName += Option::dir_sep;
179
180 ProString language = project->first(variableName: ProKey("QMAKE_LANGUAGE_" + compiler));
181 if (!language.isEmpty()) {
182 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_OUTPUT}"),
183 after: escapeFilePath(path: pchBaseName + language + headerSuffix));
184 const ProStringList pchArchs = project->values(v: "QMAKE_PCH_ARCHS");
185 for (const ProString &arch : pchArchs) {
186 QString file = pchBaseName + language + headerSuffix;
187 file.replace(before: QLatin1String("${QMAKE_PCH_ARCH}"), after: arch.toQString());
188 if (project->isActiveConfig(config: "clang_pch_style")
189 && (file.endsWith(s: QLatin1String(".pch"))
190 || file.endsWith(s: QLatin1String(".gch")))) {
191 file.chop(n: 4); // must omit header suffix for -include to recognize the PCH
192 }
193 pchFlags.replace(before: QLatin1String("${QMAKE_PCH_OUTPUT_") + arch + QLatin1Char('}'),
194 after: escapeFilePath(path: file));
195 }
196 }
197 }
198
199 if (!pchFlags.isEmpty())
200 compile_flag += " " + pchFlags;
201 }
202
203 QString compilerExecutable;
204 if (compiler == "C" || compiler == "OBJC") {
205 compilerExecutable = "$(CC)";
206 compile_flag += " $(CFLAGS)";
207 } else {
208 compilerExecutable = "$(CXX)";
209 compile_flag += " $(CXXFLAGS)";
210 }
211
212 compile_flag += " $(INCPATH)";
213
214 ProString compilerVariable = compiler;
215 if (compilerVariable == "C")
216 compilerVariable = ProString("CC");
217
218 const ProKey runComp("QMAKE_RUN_" + compilerVariable);
219 if(project->isEmpty(v: runComp))
220 project->values(v: runComp).append(t: compilerExecutable + " " + compile_flag + " " + var(var: "QMAKE_CC_O_FLAG") + "$obj $src");
221 const ProKey runCompImp("QMAKE_RUN_" + compilerVariable + "_IMP");
222 if(project->isEmpty(v: runCompImp))
223 project->values(v: runCompImp).append(t: compilerExecutable + " " + compile_flag + " " + var(var: "QMAKE_CC_O_FLAG") + "\"$@\" \"$<\"");
224 }
225
226 if (project->isActiveConfig(config: "mac") && !project->isEmpty(v: "TARGET") &&
227 ((project->isActiveConfig(config: "build_pass") || project->isEmpty(v: "BUILDS")))) {
228 ProString bundle;
229 if(project->isActiveConfig(config: "bundle") && !project->isEmpty(v: "QMAKE_BUNDLE_EXTENSION")) {
230 bundle = project->first(variableName: "TARGET");
231 if(!project->isEmpty(v: "QMAKE_BUNDLE_NAME"))
232 bundle = project->first(variableName: "QMAKE_BUNDLE_NAME");
233 if(!bundle.endsWith(sub: project->first(variableName: "QMAKE_BUNDLE_EXTENSION")))
234 bundle += project->first(variableName: "QMAKE_BUNDLE_EXTENSION");
235 } else if(project->first(variableName: "TEMPLATE") == "app" && project->isActiveConfig(config: "app_bundle")) {
236 bundle = project->first(variableName: "TARGET");
237 if(!project->isEmpty(v: "QMAKE_APPLICATION_BUNDLE_NAME"))
238 bundle = project->first(variableName: "QMAKE_APPLICATION_BUNDLE_NAME");
239 if(!bundle.endsWith(sub: ".app"))
240 bundle += ".app";
241 if(project->isEmpty(v: "QMAKE_BUNDLE_LOCATION"))
242 project->values(v: "QMAKE_BUNDLE_LOCATION").append(t: "Contents/MacOS");
243 project->values(v: "QMAKE_PKGINFO").append(t: project->first(variableName: "DESTDIR") + bundle + "/Contents/PkgInfo");
244 project->values(v: "QMAKE_BUNDLE_RESOURCE_FILE").append(t: project->first(variableName: "DESTDIR") + bundle + "/Contents/Resources/empty.lproj");
245 } else if(project->first(variableName: "TEMPLATE") == "lib" && !project->isActiveConfig(config: "staticlib") &&
246 ((!project->isActiveConfig(config: "plugin") && project->isActiveConfig(config: "lib_bundle")) ||
247 (project->isActiveConfig(config: "plugin") && project->isActiveConfig(config: "plugin_bundle")))) {
248 bundle = project->first(variableName: "TARGET");
249 if(project->isActiveConfig(config: "plugin")) {
250 if(!project->isEmpty(v: "QMAKE_PLUGIN_BUNDLE_NAME"))
251 bundle = project->first(variableName: "QMAKE_PLUGIN_BUNDLE_NAME");
252 if (project->isEmpty(v: "QMAKE_BUNDLE_EXTENSION"))
253 project->values(v: "QMAKE_BUNDLE_EXTENSION").append(t: ".plugin");
254 if (!bundle.endsWith(sub: project->first(variableName: "QMAKE_BUNDLE_EXTENSION")))
255 bundle += project->first(variableName: "QMAKE_BUNDLE_EXTENSION");
256 if(project->isEmpty(v: "QMAKE_BUNDLE_LOCATION"))
257 project->values(v: "QMAKE_BUNDLE_LOCATION").append(t: "Contents/MacOS");
258 } else {
259 if(!project->isEmpty(v: "QMAKE_FRAMEWORK_BUNDLE_NAME"))
260 bundle = project->first(variableName: "QMAKE_FRAMEWORK_BUNDLE_NAME");
261 if (project->isEmpty(v: "QMAKE_BUNDLE_EXTENSION"))
262 project->values(v: "QMAKE_BUNDLE_EXTENSION").append(t: ".framework");
263 if (!bundle.endsWith(sub: project->first(variableName: "QMAKE_BUNDLE_EXTENSION")))
264 bundle += project->first(variableName: "QMAKE_BUNDLE_EXTENSION");
265 }
266 }
267 if(!bundle.isEmpty()) {
268 project->values(v: "QMAKE_BUNDLE") = ProStringList(bundle);
269 } else {
270 project->values(v: "QMAKE_BUNDLE").clear();
271 project->values(v: "QMAKE_BUNDLE_LOCATION").clear();
272 }
273 } else { //no bundling here
274 project->values(v: "QMAKE_BUNDLE").clear();
275 project->values(v: "QMAKE_BUNDLE_LOCATION").clear();
276 }
277
278 init2();
279 ProString target = project->first(variableName: "TARGET");
280 int slsh = target.lastIndexOf(s: Option::dir_sep);
281 if (slsh != -1)
282 target.chopFront(n: slsh + 1);
283 project->values(v: "LIB_TARGET").prepend(t: target);
284}

286QStringList
287&UnixMakefileGenerator::findDependencies(const QString &f)
288{
289 QStringList &ret = MakefileGenerator::findDependencies(file: f);
290 if (doPrecompiledHeaders() && !project->isEmpty(v: "PRECOMPILED_HEADER")) {
291 ProString file = f;
292 QString header_prefix;
293 if(!project->isEmpty(v: "PRECOMPILED_DIR"))
294 header_prefix = project->first(variableName: "PRECOMPILED_DIR").toQString();
295 header_prefix += project->first(variableName: "QMAKE_ORIG_TARGET").toQString();
296 header_prefix += project->first(variableName: "QMAKE_PCH_OUTPUT_EXT").toQString();
297 if (project->isActiveConfig(config: "icc_pch_style")) {
298 // icc style
299 ProStringList pchArchs = project->values(v: "QMAKE_PCH_ARCHS");
300 if (pchArchs.isEmpty())
301 pchArchs << ProString(); // normal single-arch PCH
302 for (const ProString &arch : std::as_const(t&: pchArchs)) {
303 auto pfx = header_prefix;
304 if (!arch.isEmpty())
305 pfx.replace(before: QLatin1String("${QMAKE_PCH_ARCH}"), after: arch.toQString());
306 for (QStringList::Iterator it = Option::cpp_ext.begin();
307 it != Option::cpp_ext.end(); ++it) {
308 if (file.endsWith(sub: *it)) {
309 ret += pfx;
310 break;
311 }
312 }
313 }
314 } else {
315 // gcc style (including clang_pch_style)
316 QString header_suffix = project->isActiveConfig(config: "clang_pch_style")
317 ? project->first(variableName: "QMAKE_PCH_OUTPUT_EXT").toQString() : "";
318 header_prefix += Option::dir_sep + project->first(variableName: "QMAKE_PRECOMP_PREFIX");
319
320 for (const ProString &compiler : project->values(v: "QMAKE_BUILTIN_COMPILERS")) {
321 if (project->isEmpty(v: ProKey("QMAKE_" + compiler + "FLAGS_PRECOMPILE")))
322 continue;
323
324 ProString language = project->first(variableName: ProKey("QMAKE_LANGUAGE_" + compiler));
325 if (language.isEmpty())
326 continue;
327
328 // Unfortunately we were not consistent about the C++ naming
329 ProString extensionSuffix = compiler;
330 if (extensionSuffix == "CXX")
331 extensionSuffix = ProString("CPP");
332
333 for (const ProString &extension : project->values(v: ProKey("QMAKE_EXT_" + extensionSuffix))) {
334 if (!file.endsWith(sub: extension.toQString()))
335 continue;
336
337 ProStringList pchArchs = project->values(v: "QMAKE_PCH_ARCHS");
338 if (pchArchs.isEmpty())
339 pchArchs << ProString(); // normal single-arch PCH
340 for (const ProString &arch : std::as_const(t&: pchArchs)) {
341 QString precompiledHeader = header_prefix + language + header_suffix;
342 if (!arch.isEmpty()) {
343 precompiledHeader.replace(before: QLatin1String("${QMAKE_PCH_ARCH}"),
344 after: arch.toQString());
345 }
346 if (!ret.contains(str: precompiledHeader))
347 ret += precompiledHeader;
348 }
349
350 goto foundPrecompiledDependency;
351 }
352 }
353 foundPrecompiledDependency:
354 ; // Hurray!!
355 }
356 }
357 return ret;
358}

366bool
367UnixMakefileGenerator::findLibraries(bool linkPrl, bool mergeLflags)
368{
369 QList<QMakeLocalFileName> libdirs, frameworkdirs;
370 int libidx = 0, fwidx = 0;
371 for (const ProString &dlib : project->values(v: "QMAKE_DEFAULT_LIBDIRS"))
372 libdirs.append(t: QMakeLocalFileName(dlib.toQString()));
373 frameworkdirs.append(t: QMakeLocalFileName("/System/Library/Frameworks"));
374 frameworkdirs.append(t: QMakeLocalFileName("/Library/Frameworks"));
375 ProStringList extens;
376 extens << project->first(variableName: "QMAKE_EXTENSION_SHLIB") << "a";
377 static const char * const lflags[] = { "LIBS", "LIBS_PRIVATE",
378 "QMAKE_LIBS", "QMAKE_LIBS_PRIVATE", nullptr };
379 for (int i = 0; lflags[i]; i++) {
380 ProStringList &l = project->values(v: lflags[i]);
381 for (ProStringList::Iterator it = l.begin(); it != l.end(); ) {
382 QString opt = (*it).toQString();
383 if(opt.startsWith(s: "-")) {
384 if(opt.startsWith(s: "-L")) {
385 QString lib = opt.mid(position: 2);
386 QMakeLocalFileName f(lib);
387 int idx = libdirs.indexOf(t: f);
388 if (idx >= 0 && idx < libidx) {
389 it = l.erase(pos: it);
390 continue;
391 }
392 libdirs.insert(i: libidx++, t: f);
393 } else if(opt.startsWith(s: "-l")) {
394 QString lib = opt.mid(position: 2);
395 for (const QMakeLocalFileName &libdir : std::as_const(t&: libdirs)) {
396 QString libBase = libdir.local() + '/'
397 + project->first(variableName: "QMAKE_PREFIX_SHLIB") + lib;
398 if (linkPrl && processPrlFile(libBase, baseOnly: true))
399 goto found;
400 for (ProStringList::Iterator extit = extens.begin(); extit != extens.end(); ++extit) {
401 if (exists(file: libBase + '.' + (*extit)))
402 goto found;
403 }
404 }
405 found: ;
406 } else if (target_mode == TARG_MAC_MODE && opt.startsWith(s: "-F")) {
407 QMakeLocalFileName f(opt.mid(position: 2));
408 if (!frameworkdirs.contains(t: f))
409 frameworkdirs.insert(i: fwidx++, t: f);
410 } else if (target_mode == TARG_MAC_MODE && opt == "-framework") {
411 if (linkPrl) {
412 opt = (*++it).toQString();
413 static const QChar suffixMarker = ',';
414 const int suffixPosition = opt.indexOf(c: suffixMarker);
415 const bool hasSuffix = suffixPosition >= 0;
416 QString frameworkName = opt;
417 if (hasSuffix) {
418 frameworkName.truncate(pos: suffixPosition);
419 opt.remove(c: suffixMarker); // Apply suffix by removing marker
420 }
421 for (const QMakeLocalFileName &dir : std::as_const(t&: frameworkdirs)) {
422 auto processPrlIfFound = [&](QString directory) {
423 QString suffixedPrl = directory + opt;
424 if (processPrlFile(suffixedPrl, baseOnly: true))
425 return true;
426 if (hasSuffix) {
427 QString unsuffixedPrl = directory + frameworkName;
428 if (processPrlFile(unsuffixedPrl, baseOnly: true))
429 return true;
430 }
431 return false;
432 };
433 QString frameworkDirectory = dir.local() + "/" + frameworkName + + ".framework/";
434 if (processPrlIfFound(frameworkDirectory + "Resources/")
435 || processPrlIfFound(frameworkDirectory))
436 break;
437 }
438 } else {
439 if (opt.size() == 10)
440 ++it;
441 // Skip
442 }
443 }
444 } else if (linkPrl) {
445 processPrlFile(opt, baseOnly: false);
446 }
447
448 ProStringList &prl_libs = project->values(v: "QMAKE_CURRENT_PRL_LIBS");
449 for (int prl = 0; prl < prl_libs.size(); ++prl)
450 it = l.insert(before: ++it, t: prl_libs.at(i: prl));
451 prl_libs.clear();
452 ++it;
453 }
454
455 if (mergeLflags) {
456 QHash<ProKey, ProStringList> lflags;
457 for(int lit = 0; lit < l.size(); ++lit) {
458 ProKey arch("default");
459 ProString opt = l.at(i: lit);
460 if (opt.startsWith(c: '-')) {
461 if (target_mode == TARG_MAC_MODE && opt.startsWith(sub: "-Xarch")) {
462 if (opt.length() > 7) {
463 arch = opt.mid(off: 7).toKey();
464 opt = l.at(i: ++lit);
465 }
466 }
467
468 if (opt.startsWith(sub: "-L")
469 || (target_mode == TARG_MAC_MODE && opt.startsWith(sub: "-F"))) {
470 if (!lflags[arch].contains(str: opt))
471 lflags[arch].append(t: opt);
472 } else if (opt.startsWith(sub: "-l") || opt == "-pthread") {
473 // Make sure we keep the dependency order of libraries
474 lflags[arch].removeAll(str: opt);
475 lflags[arch].append(t: opt);
476 } else if (target_mode == TARG_MAC_MODE
477 && (opt == "-framework" || opt == "-force_load")) {
478 // Handle space separated options
479 ProString dashOpt = opt;
480 opt = l.at(i: ++lit);
481 if (opt.startsWith(sub: "-Xarch"))
482 opt = l.at(i: ++lit); // The user has done the right thing and prefixed each part
483 for(int x = 0; x < lflags[arch].size(); ++x) {
484 if (lflags[arch].at(i: x) == dashOpt && lflags[arch].at(i: ++x) == opt) {
485 lflags[arch].remove(i: x - 1, n: 2);
486 break;
487 }
488 }
489 lflags[arch].append(t: dashOpt);
490 lflags[arch].append(t: opt);
491 } else {
492 lflags[arch].append(t: opt);
493 }
494 } else if(!opt.isNull()) {
495 for (const ProString &ext : extens) {
496 if (opt.size() > ext.size() && opt.endsWith(sub: ext)
497 && opt.at(i: opt.size() - ext.size() - 1) == '.') {
498 // Make sure we keep the dependency order of libraries
499 lflags[arch].removeAll(str: opt);
500 lflags[arch].append(t: opt);
501 goto found2;
502 }
503 }
504 if(!lflags[arch].contains(str: opt))
505 lflags[arch].append(t: opt);
506 found2: ;
507 }
508 }
509
510 l = lflags.take(key: "default");
511
512 // Process architecture specific options (Xarch)
513 QHash<ProKey, ProStringList>::const_iterator archIterator = lflags.constBegin();
514 while (archIterator != lflags.constEnd()) {
515 const ProStringList &archOptions = archIterator.value();
516 for (int i = 0; i < archOptions.size(); ++i) {
517 l.append(t: QLatin1String("-Xarch_") + archIterator.key());
518 l.append(t: archOptions.at(i));
519 }
520 ++archIterator;
521 }
522 }
523 }
524 return false;
525}

527#ifdef Q_OS_WIN // MinGW x-compiling for QNX
528QString UnixMakefileGenerator::installRoot() const
529{
530 /*
531 We include a magic prefix on the path to bypass mingw-make's "helpful"
532 intervention in the environment, recognising variables that look like
533 paths and adding the msys system root as prefix, which we don't want.
534 Once this hack has smuggled INSTALL_ROOT into make's variable space, we
535 can trivially strip the magic prefix back off to get the path we meant.
536 */
537 return QStringLiteral("$(INSTALL_ROOT:@msyshack@%=%)");
538}

541QString
542UnixMakefileGenerator::defaultInstall(const QString &t)
543{
544 if(t != "target" || project->first(variableName: "TEMPLATE") == "subdirs")
545 return QString();
546
547 enum { NoBundle, SolidBundle, SlicedBundle } bundle = NoBundle;
548 bool isAux = (project->first(variableName: "TEMPLATE") == "aux");
549 const QString root = installRoot();
550 ProStringList &uninst = project->values(v: ProKey(t + ".uninstall"));
551 QString ret, destdir = project->first(variableName: "DESTDIR").toQString();
552 if(!destdir.isEmpty() && destdir.right(n: 1) != Option::dir_sep)
553 destdir += Option::dir_sep;
554 QString targetdir = fileFixify(file: project->first(variableName: "target.path").toQString(), fix: FileFixifyAbsolute);
555 if(targetdir.right(n: 1) != Option::dir_sep)
556 targetdir += Option::dir_sep;
557
558 ProStringList links;
559 QString target="$(TARGET)";
560 const ProStringList &targets = project->values(v: ProKey(t + ".targets"));
561 if(!project->isEmpty(v: "QMAKE_BUNDLE")) {
562 target = project->first(variableName: "QMAKE_BUNDLE").toQString();
563 bundle = project->isActiveConfig(config: "sliced_bundle") ? SlicedBundle : SolidBundle;
564 } else if(project->first(variableName: "TEMPLATE") == "app") {
565 target = "$(QMAKE_TARGET)";
566 } else if(project->first(variableName: "TEMPLATE") == "lib") {
567 if (!project->isActiveConfig(config: "staticlib")
568 && !project->isActiveConfig(config: "plugin")
569 && !project->isActiveConfig(config: "unversioned_libname")) {
570 if(project->isEmpty(v: "QMAKE_HPUX_SHLIB")) {
571 links << "$(TARGET0)" << "$(TARGET1)" << "$(TARGET2)";
572 } else {
573 links << "$(TARGET0)";
574 }
575 }
576 }
577 for(int i = 0; i < targets.size(); ++i) {
578 QString src = targets.at(i).toQString(),
579 dst = escapeFilePath(path: filePrefixRoot(root, targetdir + src.section(asep: '/', astart: -1)));
580 if(!ret.isEmpty())
581 ret += "\n\t";
582 ret += "$(QINSTALL) " + escapeFilePath(path: Option::fixPathToTargetOS(in: src, fix_env: false)) + ' ' + dst;
583 if(!uninst.isEmpty())
584 uninst.append(t: "\n\t");
585 uninst.append(t: "-$(DEL_FILE) " + dst);
586 }
587
588 {
589 QString src_targ = target;
590 if(!destdir.isEmpty())
591 src_targ = Option::fixPathToTargetOS(in: destdir + target, fix_env: false);
592 QString plain_targ = filePrefixRoot(root, fileFixify(file: targetdir + target, fix: FileFixifyAbsolute));
593 QString dst_targ = plain_targ;
594 plain_targ = escapeFilePath(path: plain_targ);
595 if (bundle != NoBundle) {
596 QString suffix;
597 if (project->first(variableName: "TEMPLATE") == "lib") {
598 if (!project->isActiveConfig(config: "shallow_bundle"))
599 suffix += "/Versions/" + project->first(variableName: "QMAKE_FRAMEWORK_VERSION");
600 suffix += "/$(TARGET)";
601 } else {
602 suffix = "/" + project->first(variableName: "QMAKE_BUNDLE_LOCATION") + "/$(QMAKE_TARGET)";
603 }
604 dst_targ += suffix;
605 if (bundle == SolidBundle) {
606 if (!ret.isEmpty())
607 ret += "\n\t";
608 ret += "$(DEL_FILE) -r " + plain_targ + "\n\t";
609 } else {
610 src_targ += suffix;
611 }
612 }
613 src_targ = escapeFilePath(path: src_targ);
614 dst_targ = escapeFilePath(path: dst_targ);
615
616 QString copy_cmd;
617 if (bundle == SolidBundle) {
618 copy_cmd += "$(QINSTALL) " + src_targ + ' ' + plain_targ;
619 } else if (project->first(variableName: "TEMPLATE") == "lib" && project->isActiveConfig(config: "staticlib")) {
620 copy_cmd += "$(QINSTALL) " + src_targ + ' ' + dst_targ;
621 } else if (!isAux) {
622 if (bundle == SlicedBundle) {
623 if (!ret.isEmpty())
624 ret += "\n\t";
625 ret += mkdir_p_asstring(dir: "\"`dirname " + dst_targ + "`\"", escape: false);
626 }
627 copy_cmd += "$(QINSTALL_PROGRAM) " + src_targ + ' ' + dst_targ;
628 }
629 if(project->first(variableName: "TEMPLATE") == "lib" && !project->isActiveConfig(config: "staticlib")
630 && project->values(v: ProKey(t + ".CONFIG")).indexOf(t: "fix_rpath") != -1) {
631 if (!ret.isEmpty())
632 ret += "\n\t";
633 if(!project->isEmpty(v: "QMAKE_FIX_RPATH")) {
634 ret += copy_cmd;
635 ret += "\n\t-" + var(var: "QMAKE_FIX_RPATH") + ' ' + dst_targ + ' ' + dst_targ;
636 } else if(!project->isEmpty(v: "QMAKE_LFLAGS_RPATH")) {
637 ret += "-$(LINK) $(LFLAGS) " + var(var: "QMAKE_LFLAGS_RPATH") + targetdir + " -o " +
638 dst_targ + " $(OBJECTS) $(LIBS) $(OBJCOMP)";
639 } else {
640 ret += copy_cmd;
641 }
642 } else if (!copy_cmd.isEmpty()) {
643 if (!ret.isEmpty())
644 ret += "\n\t";
645 ret += copy_cmd;
646 }
647
648 if (isAux) {
649 } else if (project->first(variableName: "TEMPLATE") == "lib" && project->isActiveConfig(config: "staticlib")) {
650 if(!project->isEmpty(v: "QMAKE_RANLIB"))
651 ret += QString("\n\t$(RANLIB) ") + dst_targ;
652 } else if (!project->isActiveConfig(config: "debug_info") && !project->isActiveConfig(config: "nostrip")
653 && !project->isEmpty(v: "QMAKE_STRIP")) {
654 ret += "\n\t-$(STRIP)";
655 if (project->first(variableName: "TEMPLATE") == "lib") {
656 if (!project->isEmpty(v: "QMAKE_STRIPFLAGS_LIB"))
657 ret += " " + var(var: "QMAKE_STRIPFLAGS_LIB");
658 } else if (project->first(variableName: "TEMPLATE") == "app") {
659 if (!project->isEmpty(v: "QMAKE_STRIPFLAGS_APP"))
660 ret += " " + var(var: "QMAKE_STRIPFLAGS_APP");
661 }
662 ret += ' ' + dst_targ;
663 }
664 if(!uninst.isEmpty())
665 uninst.append(t: "\n\t");
666 if (bundle == SolidBundle)
667 uninst.append(t: "-$(DEL_FILE) -r " + plain_targ);
668 else if (!isAux)
669 uninst.append(t: "-$(DEL_FILE) " + dst_targ);
670 if (bundle == SlicedBundle) {
671 int dstlen = project->first(variableName: "DESTDIR").length();
672 for (const ProString &src : project->values(v: "QMAKE_BUNDLED_FILES")) {
673 ProString file = src.mid(off: dstlen);
674 QString dst = escapeFilePath(
675 path: filePrefixRoot(root, fileFixify(file: targetdir + file, fix: FileFixifyAbsolute)));
676 if (!ret.isEmpty())
677 ret += "\n\t";
678 ret += mkdir_p_asstring(dir: "\"`dirname " + dst + "`\"", escape: false) + "\n\t";
679 ret += "-$(DEL_FILE) " + dst + "\n\t"; // Can't overwrite symlinks to directories
680 ret += "$(QINSTALL) " + escapeFilePath(path: src) + " " + dst;
681 if (!uninst.isEmpty())
682 uninst.append(t: "\n\t");
683 uninst.append(t: "-$(DEL_FILE) " + dst);
684 }
685 }
686 if(!links.isEmpty()) {
687 for(int i = 0; i < links.size(); ++i) {
688 if (target_mode == TARG_UNIX_MODE || target_mode == TARG_MAC_MODE) {
689 QString link = Option::fixPathToTargetOS(in: destdir + links[i], fix_env: false);
690 int lslash = link.lastIndexOf(s: Option::dir_sep);
691 if(lslash != -1)
692 link = link.right(n: link.size() - (lslash + 1));
693 QString dst_link = escapeFilePath(
694 path: filePrefixRoot(root, fileFixify(file: targetdir + link, fix: FileFixifyAbsolute)));
695 ret += "\n\t-$(SYMLINK) $(TARGET) " + dst_link;
696 if(!uninst.isEmpty())
697 uninst.append(t: "\n\t");
698 uninst.append(t: "-$(DEL_FILE) " + dst_link);
699 }
700 }
701 }
702 }
703 if (isAux || project->first(variableName: "TEMPLATE") == "lib") {
704 QStringList types;
705 types << "prl" << "libtool" << "pkgconfig";
706 for(int i = 0; i < types.size(); ++i) {
707 const QString type = types.at(i);
708 QString meta;
709 if(type == "prl" && project->isActiveConfig(config: "create_prl") && !project->isActiveConfig(config: "no_install_prl") &&
710 !project->isEmpty(v: "QMAKE_INTERNAL_PRL_FILE"))
711 meta = prlFileName(fixify: false);
712 if (type == "libtool" && project->isActiveConfig(config: "create_libtool"))
713 meta = libtoolFileName(fixify: false);
714 if(type == "pkgconfig" && project->isActiveConfig(config: "create_pc"))
715 meta = pkgConfigFileName(fixify: false);
716 if(!meta.isEmpty()) {
717 QString src_meta = meta;
718 if(!destdir.isEmpty())
719 src_meta = Option::fixPathToTargetOS(in: destdir + meta, fix_env: false);
720 QString dst_meta = filePrefixRoot(root, fileFixify(file: targetdir + meta, fix: FileFixifyAbsolute));
721 if(!uninst.isEmpty())
722 uninst.append(t: "\n\t");
723 uninst.append(t: "-$(DEL_FILE) " + escapeFilePath(path: dst_meta));
724 const QString dst_meta_dir = fileInfo(file: dst_meta).path();
725 if(!dst_meta_dir.isEmpty()) {
726 if(!ret.isEmpty())
727 ret += "\n\t";
728 ret += mkdir_p_asstring(dir: dst_meta_dir, escape: true);
729 }
730 if (!ret.isEmpty())
731 ret += "\n\t";
732 ret += installMetaFile(replace_rule: ProKey("QMAKE_" + type.toUpper() + "_INSTALL_REPLACE"), src: src_meta, dst: dst_meta);
733 }
734 }
735 }
736 return ret;
737}

739QString
740UnixMakefileGenerator::escapeFilePath(const QString &path) const
741{
742 QString ret = path;
743 if(!ret.isEmpty()) {
744 ret.replace(c: QLatin1Char(' '), after: QLatin1String("\\ "))
745 .replace(c: QLatin1Char('\t'), after: QLatin1String("\\\t"));
746 debug_msg(level: 2, fmt: "EscapeFilePath: %s -> %s", path.toLatin1().constData(), ret.toLatin1().constData());
747 }
748 return ret;
