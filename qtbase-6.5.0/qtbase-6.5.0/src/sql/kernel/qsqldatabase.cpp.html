110void QSqlDatabasePrivate::cleanConnections()
111{
112 QConnectionDict *dict = dbDict();
113 Q_ASSERT(dict);
114 QWriteLocker locker(&dict->lock);
115
116 QConnectionDict::iterator it = dict->begin();
117 while (it != dict->end()) {
118 invalidateDb(db: it.value(), name: it.key(), doWarn: false);
119 ++it;
120 }
121 dict->clear();
122}

150void QSqlDatabasePrivate::invalidateDb(const QSqlDatabase &db, const QString &name, bool doWarn)
151{
152 if (db.d->ref.loadRelaxed() != 1 && doWarn) {
153 qWarning(msg: "QSqlDatabasePrivate::removeDatabase: connection '%s' is still in use, "
154 "all queries will cease to work.", name.toLocal8Bit().constData());
155 db.d->disable();
156 db.d->connName.clear();
157 }
158}

160void QSqlDatabasePrivate::removeDatabase(const QString &name)
161{
162 QConnectionDict *dict = dbDict();
163 Q_ASSERT(dict);
164 QWriteLocker locker(&dict->lock);
165
166 if (!dict->contains(key: name))
167 return;
168
169 invalidateDb(db: dict->take(key: name), name);
170}
171
172void QSqlDatabasePrivate::addDatabase(const QSqlDatabase &db, const QString &name)
173{
174 QConnectionDict *dict = dbDict();
175 Q_ASSERT(dict);
176 QWriteLocker locker(&dict->lock);
177
178 if (dict->contains(key: name)) {
179 invalidateDb(db: dict->take(key: name), name);
180 qWarning(msg: "QSqlDatabasePrivate::addDatabase: duplicate connection name '%s', old "
181 "connection removed.", name.toLocal8Bit().data());
182 }
183 dict->insert(key: name, value: db);
184 db.d->connName = name;
185}

187/*! \internal
188*/
189QSqlDatabase QSqlDatabasePrivate::database(const QString& name, bool open)
190{
191 const QConnectionDict *dict = dbDict();
192 Q_ASSERT(dict);
193
194 dict->lock.lockForRead();
195 QSqlDatabase db = dict->value(key: name);
196 dict->lock.unlock();
197 if (!db.isValid())
198 return db;
199 if (db.driver()->thread() != QThread::currentThread()) {
200 qWarning(msg: "QSqlDatabasePrivate::database: requested database does not belong to the calling thread.");
201 return QSqlDatabase();
202 }
203
204 if (open && !db.isOpen()) {
205 if (!db.open())
206 qWarning() << "QSqlDatabasePrivate::database: unable to open database:" << db.lastError().text();
207
208 }
209 return db;
210}

287/*!
288 \class QSqlDatabase
289 \brief The QSqlDatabase class handles a connection to
290 a database.
291
292 \ingroup database
293
294 \inmodule QtSql
295
296 The QSqlDatabase class provides an interface for accessing a
297 database through a connection. An instance of QSqlDatabase
298 represents the connection. The connection provides access to the
299 database via one of the \l{SQL Database Drivers#Supported
300 Databases} {supported database drivers}, which are derived from
301 QSqlDriver. Alternatively, you can subclass your own database
302 driver from QSqlDriver. See \l{How to Write Your Own Database
303 Driver} for more information.
304
305 Create a connection (i.e., an instance of QSqlDatabase) by calling
306 one of the static addDatabase() functions, where you specify
307 \l{SQL Database Drivers#Supported Databases} {the driver or type
308 of driver} to use (depending on the type of database)
309 and a connection name. A connection is known by its own name,
310 \e{not} by the name of the database it connects to. You can have
311 multiple connections to one database. QSqlDatabase also supports
312 the concept of a \e{default} connection, which is the unnamed
313 connection. To create the default connection, don't pass the
314 connection name argument when you call addDatabase().
315 Subsequently, the default connection will be assumed if you call
316 any static member function without specifying the connection name.
317 The following snippet shows how to create and open a default connection
318 to a PostgreSQL database:
319
320 \snippet sqldatabase/sqldatabase.cpp 0
321
322 Once the QSqlDatabase object has been created, set the connection
323 parameters with setDatabaseName(), setUserName(), setPassword(),
324 setHostName(), setPort(), and setConnectOptions(). Then call
325 open() to activate the physical connection to the database. The
326 connection is not usable until you open it.
327
328 The connection defined above will be the \e{default} connection,
329 because we didn't give a connection name to \l{QSqlDatabase::}
330 {addDatabase()}. Subsequently, you can get the default connection
331 by calling database() without the connection name argument:
332
333 \snippet sqldatabase/sqldatabase.cpp 1
334
335 QSqlDatabase is a value class. Changes made to a database
336 connection via one instance of QSqlDatabase will affect other
337 instances of QSqlDatabase that represent the same connection. Use
338 cloneDatabase() to create an independent database connection based
339 on an existing one.
340
341 \warning It is highly recommended that you do not keep a copy of the
342 QSqlDatabase around as a member of a class, as this will prevent the
343 instance from being correctly cleaned up on shutdown. If you need to
344 access an existing QSqlDatabase, it should be accessed with database().
345 If you chose to have a QSqlDatabase member variable, this needs to be
346 deleted before the QCoreApplication instance is deleted, otherwise it
347 may lead to undefined behavior.
348
349 If you create multiple database connections, specify a unique
350 connection name for each one, when you call addDatabase(). Use
351 database() with a connection name to get that connection. Use
352 removeDatabase() with a connection name to remove a connection.
353 QSqlDatabase outputs a warning if you try to remove a connection
354 referenced by other QSqlDatabase objects. Use contains() to see if
355 a given connection name is in the list of connections.
356
357 \table
358 \header
359 \li {2,1}Some utility methods:
360 \row
361 \li tables()
362 \li returns the list of tables
363 \row
364 \li primaryIndex()
365 \li returns a table's primary index
366 \row
367 \li record()
368 \li returns meta-information about a table's fields
369 \row
370 \li transaction()
371 \li starts a transaction
372 \row
373 \li commit()
374 \li saves and completes a transaction
375 \row
376 \li rollback()
377 \li cancels a transaction
378 \row
379 \li hasFeature()
380 \li checks if a driver supports transactions
381 \row
382 \li lastError()
383 \li returns information about the last error
384 \row
385 \li drivers()
386 \li returns the names of the available SQL drivers
387 \row
388 \li isDriverAvailable()
389 \li checks if a particular driver is available
390 \row
391 \li registerSqlDriver()
392 \li registers a custom-made driver
393 \endtable
394
395 \note QSqlDatabase::exec() is deprecated. Use QSqlQuery::exec()
396 instead.
397
398 \note When using transactions, you must start the
399 transaction before you create your query.
400
401 \sa QSqlDriver, QSqlQuery, {Qt SQL}, {Threads and the SQL Module}
402*/
403
404/*! \fn QSqlDatabase QSqlDatabase::addDatabase(const QString &type, const QString &connectionName)
405 \threadsafe
406
407 Adds a database to the list of database connections using the
408 driver \a type and the connection name \a connectionName. If
409 there already exists a database connection called \a
410 connectionName, that connection is removed.
411
412 The database connection is referred to by \a connectionName. The
413 newly added database connection is returned.
414
415 If \a type is not available or could not be loaded, isValid() returns \c false.
416
417 If \a connectionName is not specified, the new connection becomes
418 the default connection for the application, and subsequent calls
419 to database() without the connection name argument will return the
420 default connection. If a \a connectionName is provided here, use
421 database(\a connectionName) to retrieve the connection.
422
423 \warning If you add a connection with the same name as an existing
424 connection, the new connection replaces the old one. If you call
425 this function more than once without specifying \a connectionName,
426 the default connection will be the one replaced.
427
428 Before using the connection, it must be initialized. e.g., call
429 some or all of setDatabaseName(), setUserName(), setPassword(),
430 setHostName(), setPort(), and setConnectOptions(), and, finally,
431 open().
432
433 \sa database(), removeDatabase(), {Threads and the SQL Module}
434*/
435QSqlDatabase QSqlDatabase::addDatabase(const QString &type, const QString &connectionName)
436{
437 QSqlDatabase db(type);
438 QSqlDatabasePrivate::addDatabase(db, name: connectionName);
439 return db;
440}

499QStringList QSqlDatabase::drivers()
500{
501 QStringList list;
502
503 if (QFactoryLoader *fl = loader()) {
504 typedef QMultiMap<int, QString> PluginKeyMap;
505 typedef PluginKeyMap::const_iterator PluginKeyMapConstIterator;
506
507 const PluginKeyMap keyMap = fl->keyMap();
508 const PluginKeyMapConstIterator cend = keyMap.constEnd();
509 for (PluginKeyMapConstIterator it = keyMap.constBegin(); it != cend; ++it)
510 if (!list.contains(str: it.value()))
511 list << it.value();
512 }
513
514 DriverDict dict = QSqlDatabasePrivate::driverDict();
515 for (DriverDict::const_iterator i = dict.constBegin(); i != dict.constEnd(); ++i) {
516 if (!list.contains(str: i.key()))
517 list << i.key();
518 }
519
520 return list;
521}

641/*!
642 \internal
643
644 Create the actual driver instance \a type.
645*/
646
647void QSqlDatabasePrivate::init(const QString &type)
648{
649 drvName = type;
650
651 if (!driver) {
652 DriverDict dict = QSqlDatabasePrivate::driverDict();
653 for (DriverDict::const_iterator it = dict.constBegin();
654 it != dict.constEnd() && !driver; ++it) {
655 if (type == it.key()) {
656 driver = ((QSqlDriverCreatorBase*)(*it))->createObject();
657 }
658 }
659 }
660
661 if (!driver && loader())
662 driver = qLoadPlugin<QSqlDriver, QSqlDriverPlugin>(loader: loader(), key: type);
663
664 if (!driver) {
665 qWarning(msg: "QSqlDatabase: %s driver not loaded", type.toLatin1().data());
666 qWarning(msg: "QSqlDatabase: available drivers: %s",
667 QSqlDatabase::drivers().join(sep: u' ').toLatin1().data());
668 if (QCoreApplication::instance() == nullptr)
669 qWarning(msg: "QSqlDatabase: an instance of QCoreApplication is required for loading driver plugins");
670 driver = shared_null()->driver;
671 }
672}

817/*!
818 Rolls back a transaction on the database, if the driver supports
819 transactions and a transaction() has been started. Returns \c{true}
820 if the operation succeeded. Otherwise it returns \c{false}.
821
822 \note For some databases, the rollback will fail and return
823 \c{false} if there is an \l{QSqlQuery::isActive()} {active query}
824 using the database for a \c{SELECT}. Make the query
