48/*!
49 \class QSqlRecord
50 \brief The QSqlRecord class encapsulates a database record.
51
52 \ingroup database
53 \ingroup shared
54 \inmodule QtSql
55
56 The QSqlRecord class encapsulates the functionality and
57 characteristics of a database record (usually a row in a table or
58 view within the database). QSqlRecord supports adding and
59 removing fields as well as setting and retrieving field values.
60
61 The values of a record's fields can be set by name or position
62 with setValue(); if you want to set a field to null use
63 setNull(). To find the position of a field by name use indexOf(),
64 and to find the name of a field at a particular position use
65 fieldName(). Use field() to retrieve a QSqlField object for a
66 given field. Use contains() to see if the record contains a
67 particular field name.
68
69 When queries are generated to be executed on the database only
70 those fields for which isGenerated() is true are included in the
71 generated SQL.
72
73 A record can have fields added with append() or insert(), replaced
74 with replace(), and removed with remove(). All the fields can be
75 removed with clear(). The number of fields is given by count();
76 all their values can be cleared (to null) using clearValues().
77
78 \sa QSqlField, QSqlQuery::record()
79*/
80

188/*!
189 Returns the position of the field called \a name within the
190 record, or -1 if it cannot be found. Field names are not
191 case-sensitive. If more than one field matches, the first one is
192 returned.
193
194 \sa fieldName()
195*/
196
197int QSqlRecord::indexOf(const QString& name) const
198{
199 QStringView tableName;
200 QStringView fieldName(name);
201 const qsizetype idx = name.indexOf(c: u'.');
202 if (idx != -1) {
203 tableName = fieldName.left(n: idx);
204 fieldName = fieldName.mid(pos: idx + 1);
205 }
206 const int cnt = count();
207 for (int i = 0; i < cnt; ++i) {
208 // Check the passed in name first in case it is an alias using a dot.
209 // Then check if both the table and field match when there is a table name specified.
210 const auto &currentField = d->fields.at(i);
211 const auto &currentFieldName = currentField.name();
212 if (currentFieldName.compare(s: name, cs: Qt::CaseInsensitive) == 0
213 || (idx != -1 && currentFieldName.compare(s: fieldName, cs: Qt::CaseInsensitive) == 0
214 && currentField.tableName().compare(s: tableName, cs: Qt::CaseInsensitive) == 0)) {
215 return i;
216 }
217 }
218 return -1;
219}

493#ifndef QT_NO_DEBUG_STREAM
494QDebug operator<<(QDebug dbg, const QSqlRecord &r)
495{
496 QDebugStateSaver saver(dbg);
497 dbg.nospace();
498 const int count = r.count();
499 dbg << "QSqlRecord(" << count << ')';
500 for (int i = 0; i < count; ++i) {
501 dbg.nospace();
502 dbg << '\n' << qSetFieldWidth(width: 2) << Qt::right << i << Qt::left << qSetFieldWidth(width: 0) << ':';
503 dbg.space();
504 dbg << r.field(index: i) << r.value(index: i).toString();
505 }
506 return dbg;
507}
508#endif

510/*!
511 \since 5.1
512 Returns a record containing the fields represented in \a keyFields set to values
513 that match by field name.
514*/
515QSqlRecord QSqlRecord::keyValues(const QSqlRecord &keyFields) const
516{
517 QSqlRecord retValues(keyFields);
518
519 for (int i = retValues.count() - 1; i >= 0; --i)
520 retValues.setValue(index: i, val: value(name: retValues.fieldName(index: i)));
521
522 return retValues;
523}
