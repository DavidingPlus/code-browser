34static bool qIsAlnum(QChar ch)
35{
36 uint u = uint(ch.unicode());
37 // matches [a-zA-Z0-9_]
38 return u - 'a' < 26 || u - 'A' < 26 || u - '0' < 10 || u == '_';
39}

41QString QSqlResultPrivate::positionalToNamedBinding(const QString &query) const
42{
43 int n = query.size();
44
45 QString result;
46 result.reserve(asize: n * 5 / 4);
47 QChar closingQuote;
48 int count = 0;
49 bool ignoreBraces = (sqldriver->dbmsType() == QSqlDriver::PostgreSQL);
50
51 for (int i = 0; i < n; ++i) {
52 QChar ch = query.at(i);
53 if (!closingQuote.isNull()) {
54 if (ch == closingQuote) {
55 if (closingQuote == u']'
56 && i + 1 < n && query.at(i: i + 1) == closingQuote) {
57 // consume the extra character. don't close.
58 ++i;
59 result += ch;
60 } else {
61 closingQuote = QChar();
62 }
63 }
64 result += ch;
65 } else {
66 if (ch == u'?') {
67 result += fieldSerial(i: count++);
68 } else {
69 if (ch == u'\'' || ch == u'"' || ch == u'`')
70 closingQuote = ch;
71 else if (!ignoreBraces && ch == u'[')
72 closingQuote = u']';
73 result += ch;
74 }
75 }
76 }
77 result.squeeze();
78 return result;
79}

81QString QSqlResultPrivate::namedToPositionalBinding(const QString &query)
82{
83 // In the Interbase case if it is an EXECUTE BLOCK then it is up to the
84 // caller to make sure that it is not using named bindings for the wrong
85 // parts of the query since Interbase uses them literally
86 if (sqldriver->dbmsType() == QSqlDriver::Interbase &&
87 query.trimmed().startsWith(s: "EXECUTE BLOCK"_L1, cs: Qt::CaseInsensitive))
88 return query;
89
90 int n = query.size();
91
92 QString result;
93 result.reserve(asize: n);
94 QChar closingQuote;
95 int count = 0;
96 int i = 0;
97 bool ignoreBraces = (sqldriver->dbmsType() == QSqlDriver::PostgreSQL);
98
99 while (i < n) {
100 QChar ch = query.at(i);
101 if (!closingQuote.isNull()) {
102 if (ch == closingQuote) {
103 if (closingQuote == u']'
104 && i + 1 < n && query.at(i: i + 1) == closingQuote) {
105 // consume the extra character. don't close.
106 ++i;
107 result += ch;
108 } else {
109 closingQuote = QChar();
110 }
111 }
112 result += ch;
113 ++i;
114 } else {
115 if (ch == u':'
116 && (i == 0 || query.at(i: i - 1) != u':')
117 && (i + 1 < n && qIsAlnum(ch: query.at(i: i + 1)))) {
118 int pos = i + 2;
119 while (pos < n && qIsAlnum(ch: query.at(i: pos)))
120 ++pos;
121 QString holder(query.mid(position: i, n: pos - i));
122 indexes[holder].append(t: count++);
123 holders.append(t: QHolder(holder, i));
124 result += u'?';
125 i = pos;
126 } else {
127 if (ch == u'\'' || ch == u'"' || ch == u'`')
128 closingQuote = ch;
129 else if (!ignoreBraces && ch == u'[')
130 closingQuote = u']';
131 result += ch;
132 ++i;
133 }
134 }
135 }
136 result.squeeze();
137 values.resize(size: holders.size());
138 return result;
139}

144 accessing data from specific SQL databases.
145
146 \ingroup database

156 virtual functions that you need.

158 \sa QSqlDriver

514/*!
515 Sets forward only mode to \a forward. If \a forward is true, only
516 fetchNext() is allowed for navigating the results. Forward only
517 mode needs much less memory since results do not have to be
518 cached. By default, this feature is disabled.
519
520 Setting forward only to false is a suggestion to the database engine,
521 which has the final say on whether a result set is forward only or
522 scrollable. isForwardOnly() will always return the correct status of
523 the result set.
524
525 \note Calling setForwardOnly after execution of the query will result
526 in unexpected results at best, and crashes at worst.
527
528 \note To make sure the forward-only query completed successfully,
529 the application should check lastError() for an error not only after
530 executing the query, but also after navigating the query results.
531
532 \warning PostgreSQL: While navigating the query results in forward-only
533 mode, do not execute any other SQL command on the same database
534 connection. This will cause the query results to be lost.
535
536 \sa isForwardOnly(), fetchNext(), QSqlQuery::setForwardOnly()
537*/

544/*!
545 Prepares the given \a query, using the underlying database
546 functionality where possible. Returns \c true if the query is
547 prepared successfully; otherwise returns \c false.
548
549 Note: This method should have been called "safePrepare()".
550
551 \sa prepare()
552*/
553bool QSqlResult::savePrepare(const QString& query)
554{
555 Q_D(QSqlResult);
556 if (!driver())
557 return false;
558 d->clear();
559 d->sql = query;
560 if (!driver()->hasFeature(f: QSqlDriver::PreparedQueries))
561 return prepare(query);
562
563 // parse the query to memorize parameter location
564 d->executedQuery = d->namedToPositionalBinding(query);
565
566 if (driver()->hasFeature(f: QSqlDriver::NamedPlaceholders))
567 d->executedQuery = d->positionalToNamedBinding(query);
568
569 return prepare(query: d->executedQuery);
570}

572/*!
573 Prepares the given \a query for execution; the query will normally
574 use placeholders so that it can be executed repeatedly. Returns
575 true if the query is prepared successfully; otherwise returns \c false.
576
577 \sa exec()
578*/
579bool QSqlResult::prepare(const QString& query)
580{
581 Q_D(QSqlResult);
582 d->sql = query;
583 if (d->holders.isEmpty()) {
584 // parse the query to memorize parameter location
585 d->namedToPositionalBinding(query);
586 }
587 return true; // fake prepares should always succeed
588}

590bool QSqlResultPrivate::isVariantNull(const QVariant &variant)
591{
592 if (variant.isNull())
593 return true;
594
595 switch (variant.typeId()) {
596 case qMetaTypeId<QString>():
597 return static_cast<const QString*>(variant.constData())->isNull();
598 case qMetaTypeId<QByteArray>():
599 return static_cast<const QByteArray*>(variant.constData())->isNull();
600 case qMetaTypeId<QDateTime>():
601 // We treat invalid date-time as null, since its ISODate would be empty.
602 return !static_cast<const QDateTime*>(variant.constData())->isValid();
603 case qMetaTypeId<QDate>():
604 return static_cast<const QDate*>(variant.constData())->isNull();
605 case qMetaTypeId<QTime>():
606 // As for QDateTime, QTime can be invalid without being null.
607 return !static_cast<const QTime*>(variant.constData())->isValid();
608 case qMetaTypeId<QUuid>():
609 return static_cast<const QUuid*>(variant.constData())->isNull();
610 default:
611 break;
612 }
613
614 return false;
615}

617/*!
618 Executes the query, returning true if successful; otherwise returns
619 false.
620
621 \sa prepare()
622*/
623bool QSqlResult::exec()
624{
625 Q_D(QSqlResult);
626 bool ret;
627 // fake preparation - just replace the placeholders..
628 QString query = lastQuery();
629 if (d->binds == NamedBinding) {
630 int i;
631 QVariant val;
632 QString holder;
633 for (i = d->holders.size() - 1; i >= 0; --i) {
634 holder = d->holders.at(i).holderName;
635 val = d->values.value(i: d->indexes.value(key: holder).value(i: 0,defaultValue: -1));
636 QSqlField f(""_L1, val.metaType());
637 if (QSqlResultPrivate::isVariantNull(variant: val))
638 f.setValue(QVariant());
639 else
640 f.setValue(val);
641 query = query.replace(i: d->holders.at(i).holderPos,
642 len: holder.size(), after: driver()->formatValue(field: f));
643 }
644 } else {
645 QString val;
646 qsizetype i = 0;
647 int idx = 0;
648 for (idx = 0; idx < d->values.size(); ++idx) {
649 i = query.indexOf(c: u'?', from: i);
650 if (i == -1)
651 continue;
652 QVariant var = d->values.value(i: idx);
653 QSqlField f(""_L1, var.metaType());
654 if (QSqlResultPrivate::isVariantNull(variant: var))
655 f.clear();
656 else
657 f.setValue(var);
658 val = driver()->formatValue(field: f);
659 query = query.replace(i, len: 1, after: driver()->formatValue(field: f));
660 i += val.size();
661 }
662 }
663
664 // have to retain the original query with placeholders
665 QString orig = lastQuery();
666 ret = reset(sqlquery: query);
667 d->executedQuery = query;
668 setQuery(orig);
669 d->resetBindCount();
670 return ret;
671}

673/*!
674 Binds the value \a val of parameter type \a paramType to position \a index
675 in the current record (row).
676
677 \sa addBindValue()
678*/
679void QSqlResult::bindValue(int index, const QVariant& val, QSql::ParamType paramType)
680{
681 Q_D(QSqlResult);
682 d->binds = PositionalBinding;
683 QList<int> &indexes = d->indexes[d->fieldSerial(i: index)];
684 if (!indexes.contains(t: index))
685 indexes.append(t: index);
686 if (d->values.size() <= index)
687 d->values.resize(size: index + 1);
688 d->values[index] = val;
689 if (paramType != QSql::In || !d->types.isEmpty())
690 d->types[index] = paramType;
691}

693/*!
694 \overload
695
696 Binds the value \a val of parameter type \a paramType to the \a
697 placeholder name in the current record (row).
698
699 \note Binding an undefined placeholder will result in undefined behavior.
700
701 \sa QSqlQuery::bindValue()
702*/
703void QSqlResult::bindValue(const QString& placeholder, const QVariant& val,
704 QSql::ParamType paramType)
705{
706 Q_D(QSqlResult);
707 d->binds = NamedBinding;
708 // if the index has already been set when doing emulated named
709 // bindings - don't reset it
710 const QList<int> indexes = d->indexes.value(key: placeholder);
711 for (int idx : indexes) {
712 if (d->values.size() <= idx)
713 d->values.resize(size: idx + 1);
714 d->values[idx] = val;
715 if (paramType != QSql::In || !d->types.isEmpty())
716 d->types[idx] = paramType;
717 }
718}

918/*! \internal
919 \since 4.2
920
921 Executes a prepared query in batch mode if the driver supports it,
922 otherwise emulates a batch execution using bindValue() and exec().
923 QSqlDriver::hasFeature() can be used to find out whether a driver
924 supports batch execution.
925
926 Batch execution can be faster for large amounts of data since it
927 reduces network roundtrips.
928
929 For batch executions, bound values have to be provided as lists
930 of variants (QVariantList).
931
932 Each list must contain values of the same type. All lists must
933 contain equal amount of values (rows).
934
935 NULL values are passed in as typed QVariants, for example
936 \c {QVariant(QMetaType::fromType<int>())} for an integer NULL value.
937
938 Example:
939
940 \snippet code/src_sql_kernel_qsqlresult.cpp 0
941
942 Here, we insert two rows into a SQL table, with each row containing three values.
943
944 \sa exec(), QSqlDriver::hasFeature()
945*/
946bool QSqlResult::execBatch(bool arrayBind)
947{
948 Q_UNUSED(arrayBind);
949 Q_D(QSqlResult);
950
951 QList<QVariant> values = d->values;
952 if (values.size() == 0)
953 return false;
954 for (int i = 0; i < values.at(i: 0).toList().size(); ++i) {
955 for (int j = 0; j < values.size(); ++j)
956 bindValue(index: j, val: values.at(i: j).toList().at(i), paramType: QSql::In);
957 if (!exec())
958 return false;
959 }
960 return true;
961}
