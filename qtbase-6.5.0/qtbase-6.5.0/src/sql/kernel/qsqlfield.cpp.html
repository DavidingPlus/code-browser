10class QSqlFieldPrivate
11{
12public:
13 QSqlFieldPrivate(const QString &name,
14 QMetaType type, const QString &tableName) :
15 ref(1), nm(name), table(tableName), def(QVariant()), type(type),
16 req(QSqlField::Unknown), len(-1), prec(-1), tp(-1),
17 ro(false), gen(true), autoval(false)
18 {}
19
20 QSqlFieldPrivate(const QSqlFieldPrivate &other)
21 : ref(1),
22 nm(other.nm),
23 table(other.table),
24 def(other.def),
25 type(other.type),
26 req(other.req),
27 len(other.len),
28 prec(other.prec),
29 tp(other.tp),
30 ro(other.ro),
31 gen(other.gen),
32 autoval(other.autoval)
33 {}
34
35 bool operator==(const QSqlFieldPrivate& other) const
36 {
37 return (nm == other.nm
38 && table == other.table
39 && def == other.def
40 && type == other.type
41 && req == other.req
42 && len == other.len
43 && prec == other.prec
44 && ro == other.ro
45 && gen == other.gen
46 && autoval == other.autoval);
47 }
48
49 QAtomicInt ref;
50 QString nm;
51 QString table;
52 QVariant def;
53 QMetaType type;
54 QSqlField::RequiredStatus req;
55 int len;
56 int prec;
57 int tp;
58 bool ro: 1;
59 bool gen: 1;
60 bool autoval: 1;
61};

64/*!
65 \class QSqlField
66 \brief The QSqlField class manipulates the fields in SQL database tables
67 and views.
68
69 \ingroup database
70 \ingroup shared
71 \inmodule QtSql
72
73 QSqlField represents the characteristics of a single column in a
74 database table or view, such as the data type and column name. A
75 field also contains the value of the database column, which can be
76 viewed or changed.
77
78 Field data values are stored as QVariants. Using an incompatible
79 type is not permitted. For example:
80
81 \snippet sqldatabase/sqldatabase.cpp 2
82
83 However, the field will attempt to cast certain data types to the
84 field data type where possible:
85
86 \snippet sqldatabase/sqldatabase.cpp 3
87
88 QSqlField objects are rarely created explicitly in application
89 code. They are usually accessed indirectly through \l{QSqlRecord}s
90 that already contain a list of fields. For example:
91
92 \snippet sqldatabase/sqldatabase.cpp 4
93 \dots
94 \snippet sqldatabase/sqldatabase.cpp 5
95 \snippet sqldatabase/sqldatabase.cpp 6
96
97 A QSqlField object can provide some meta-data about the field, for
98 example, its name(), variant type(), length(), precision(),
99 defaultValue(), typeID(), and its requiredStatus(),
100 isGenerated() and isReadOnly(). The field's data can be
101 checked to see if it isNull(), and its value() retrieved. When
102 editing the data can be set with setValue() or set to NULL with
103 clear().
104
105 \sa QSqlRecord
106*/
107
108/*!
109 \enum QSqlField::RequiredStatus
110
111 Specifies whether the field is required or optional.
112
113 \value Required The field must be specified when inserting records.
114 \value Optional The fields doesn't have to be specified when inserting records.
115 \value Unknown The database driver couldn't determine whether the field is required or
116 optional.
117
118 \sa requiredStatus()
119*/

121/*!
122 \fn QSqlField::QSqlField(const QString &fieldName, QVariant::Type type, const QString &table)
123 \deprecated [6.0] Use the constructor taking a QMetaType instead.
124 \overload
125
126 Constructs an empty field called \a fieldName of variant type \a

128
129 \sa setRequiredStatus(), setLength(), setPrecision(), setDefaultValue(),
130 setGenerated(), setReadOnly()
131*/
132
133/*!
134 \since 6.0
135
136 \overload
137 Constructs an empty field called \a fieldName of type \a
138 type in \a table.
139
140 \sa setRequiredStatus(), setLength(), setPrecision(), setDefaultValue(),
141 setGenerated(), setReadOnly()

143QSqlField::QSqlField(const QString &fieldName, QMetaType type, const QString &table)
144{
145 d = new QSqlFieldPrivate(fieldName, type, table);
146 val = QVariant(QMetaType(type), nullptr);
147}

177/*!

181bool QSqlField::operator==(const QSqlField& other) const

280/*!
281 Sets the value of the field to \a value. If the field is read-only
282 (isReadOnly() returns \c true), nothing happens.
283
284 If the data type of \a value differs from the field's current
285 data type, an attempt is made to cast it to the proper type. This
286 preserves the data type of the field in the case of assignment,
287 e.g. a QString to an integer data type.
288
289 To set the value to NULL, use clear().
290
291 \sa value(), isReadOnly(), defaultValue()
292*/
293
294void QSqlField::setValue(const QVariant& value)
295{
296 if (isReadOnly())
297 return;
298 val = value;
299}
300
301/*!
302 Clears the value of the field and sets it to NULL.
303 If the field is read-only, nothing happens.
304
305 \sa setValue(), isReadOnly(), requiredStatus()
306*/
307
308void QSqlField::clear()
309{
310 if (isReadOnly())
311 return;
312 val = QVariant(d->type, nullptr);
313}

518#ifndef QT_NO_DEBUG_STREAM
519QDebug operator<<(QDebug dbg, const QSqlField &f)
520{
521 QDebugStateSaver saver(dbg);
522 dbg.nospace();
523 dbg << "QSqlField(" << f.name() << ", " << f.metaType().name();
524 dbg << ", tableName: " << (f.tableName().isEmpty() ? QStringLiteral("(not specified)") : f.tableName());
525 if (f.length() >= 0)
526 dbg << ", length: " << f.length();
527 if (f.precision() >= 0)
528 dbg << ", precision: " << f.precision();
529 if (f.requiredStatus() != QSqlField::Unknown)
530 dbg << ", required: "
531 << (f.requiredStatus() == QSqlField::Required ? "yes" : "no");
532 dbg << ", generated: " << (f.isGenerated() ? "yes" : "no");
533 if (f.typeID() >= 0)
534 dbg << ", typeID: " << f.typeID();
535 if (!f.defaultValue().isNull())
536 dbg << ", defaultValue: \"" << f.defaultValue() << '\"';
537 dbg << ", autoValue: " << f.isAutoValue()
538 << ", readOnly: " << f.isReadOnly() << ')';
539 return dbg;
540}
541#endif
