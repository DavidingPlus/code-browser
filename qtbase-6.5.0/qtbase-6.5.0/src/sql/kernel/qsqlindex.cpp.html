107/*!
108 Returns \c true if field \a i in the index is sorted in descending
109 order; otherwise returns \c false.
110*/
111
112bool QSqlIndex::isDescending(int i) const
113{
114 if (i >= 0 && i < sorts.size())
115 return sorts[i];
116 return false;
117}
118
119/*!
120 If \a desc is true, field \a i is sorted in descending order.
121 Otherwise, field \a i is sorted in ascending order (the default).
122 If the field does not exist, nothing happens.
123*/
124
125void QSqlIndex::setDescending(int i, bool desc)
126{
127 if (i >= 0 && i < sorts.size())
128 sorts[i] = desc;
129}
130
131/*! \internal
132
133 Creates a string representing the field number \a i using prefix \a
134 prefix. If \a verbose is true, ASC or DESC is included in the field
135 description if the field is sorted in ASCending or DESCending order.
136*/
137
138QString QSqlIndex::createField(int i, const QString& prefix, bool verbose) const
139{
140 QString f;
141 if (!prefix.isEmpty())
142 f += prefix + u'.';
143 f += field(i).name();
144 if (verbose)
145 f += u' ' + QString((isDescending(i) ? "DESC"_L1 : "ASC"_L1));
146 return f;
147}
