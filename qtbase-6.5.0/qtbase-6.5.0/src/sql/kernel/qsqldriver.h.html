7#include <QtSql/qtsqlglobal.h>
8#include <QtCore/qobject.h>
9#include <QtCore/qstring.h>
10#include <QtCore/qstringlist.h>
11
12QT_BEGIN_NAMESPACE
13
14
15class QSqlDatabase;
16class QSqlDriverPrivate;
17class QSqlError;
18class QSqlField;
19class QSqlIndex;
20class QSqlRecord;
21class QSqlResult;
22class QVariant;
23
24class Q_SQL_EXPORT QSqlDriver : public QObject
25{
26 friend class QSqlDatabase;
27 friend class QSqlResultPrivate;
28 Q_OBJECT
29 Q_DECLARE_PRIVATE(QSqlDriver)
30
31public:
32 enum DriverFeature { Transactions, QuerySize, BLOB, Unicode, PreparedQueries,
33 NamedPlaceholders, PositionalPlaceholders, LastInsertId,
34 BatchOperations, SimpleLocking, LowPrecisionNumbers,
35 EventNotifications, FinishQuery, MultipleResultSets, CancelQuery };
36
37 enum StatementType { WhereStatement, SelectStatement, UpdateStatement,
38 InsertStatement, DeleteStatement };
39
40 enum IdentifierType { FieldName, TableName };
41
42 enum NotificationSource { UnknownSource, SelfSource, OtherSource };
43
44 enum DbmsType {
45 UnknownDbms,
46 MSSqlServer,
47 MySqlServer,
48 PostgreSQL,
49 Oracle,
50 Sybase,
51 SQLite,
52 Interbase,
53 DB2
54 };
55
56 explicit QSqlDriver(QObject *parent = nullptr);
57 ~QSqlDriver();
58 virtual bool isOpen() const;
59 bool isOpenError() const;
60
61 virtual bool beginTransaction();
62 virtual bool commitTransaction();
63 virtual bool rollbackTransaction();
64 virtual QStringList tables(QSql::TableType tableType) const;
65 virtual QSqlIndex primaryIndex(const QString &tableName) const;
66 virtual QSqlRecord record(const QString &tableName) const;
67 virtual QString formatValue(const QSqlField &field, bool trimStrings = false) const;
68
69 virtual QString escapeIdentifier(const QString &identifier, IdentifierType type) const;
70 virtual QString sqlStatement(StatementType type, const QString &tableName,
71 const QSqlRecord &rec, bool preparedStatement) const;
72
73 QSqlError lastError() const;
74
75 virtual QVariant handle() const;
76 virtual bool hasFeature(DriverFeature f) const = 0;
77 virtual void close() = 0;
78 virtual QSqlResult *createResult() const = 0;
79
80 virtual bool open(const QString &db,
81 const QString &user = QString(),
82 const QString &password = QString(),
83 const QString &host = QString(),
84 int port = -1,
85 const QString &connOpts = QString()) = 0;
86 virtual bool subscribeToNotification(const QString &name);
87 virtual bool unsubscribeFromNotification(const QString &name);
88 virtual QStringList subscribedToNotifications() const;
89
90 virtual bool isIdentifierEscaped(const QString &identifier, IdentifierType type) const;
91 virtual QString stripDelimiters(const QString &identifier, IdentifierType type) const;
92
93 void setNumericalPrecisionPolicy(QSql::NumericalPrecisionPolicy precisionPolicy);
94 QSql::NumericalPrecisionPolicy numericalPrecisionPolicy() const;
95
96 DbmsType dbmsType() const;
97 virtual int maximumIdentifierLength(IdentifierType type) const;
98public Q_SLOTS:
99 virtual bool cancelQuery();
100
101Q_SIGNALS:
102 void notification(const QString &name, QSqlDriver::NotificationSource source, const QVariant &payload);
103
104protected:
105 QSqlDriver(QSqlDriverPrivate &dd, QObject *parent = nullptr);
106 virtual void setOpen(bool o);
107 virtual void setOpenError(bool e);
108 virtual void setLastError(const QSqlError& e);
109
110
111private:
112 Q_DISABLE_COPY(QSqlDriver)
113};
