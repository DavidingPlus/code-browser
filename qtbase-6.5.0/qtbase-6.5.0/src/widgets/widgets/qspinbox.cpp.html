78/*!
79 \class QSpinBox
80 \brief The QSpinBox class provides a spin box widget.
81
82 \ingroup basicwidgets
83 \inmodule QtWidgets
84
85 \image windows-spinbox.png
86
87 QSpinBox is designed to handle integers and discrete sets of
88 values (e.g., month names); use QDoubleSpinBox for floating point
89 values.
90
91 QSpinBox allows the user to choose a value by clicking the up/down
92 buttons or pressing up/down on the keyboard to increase/decrease
93 the value currently displayed. The user can also type the value in
94 manually. The spin box supports integer values but can be extended to
95 use different strings with validate(), textFromValue() and valueFromText().
96
97 Every time the value changes QSpinBox emits valueChanged() and
98 textChanged() signals, the former providing a int and the latter
99 a QString. The textChanged() signal provides the value with both
100 prefix() and suffix().
101 The current value can be fetched with value() and set with setValue().
102
103 Clicking the up/down buttons or using the keyboard accelerator's
104 up and down arrows will increase or decrease the current value in
105 steps of size singleStep(). If you want to change this behaviour you
106 can reimplement the virtual function stepBy(). The minimum and
107 maximum value and the step size can be set using one of the
108 constructors, and can be changed later with setMinimum(),
109 setMaximum() and setSingleStep().
110
111 Most spin boxes are directional, but QSpinBox can also operate as
112 a circular spin box, i.e. if the range is 0-99 and the current
113 value is 99, clicking "up" will give 0 if wrapping() is set to
114 true. Use setWrapping() if you want circular behavior.
115
116 The displayed value can be prepended and appended with arbitrary
117 strings indicating, for example, currency or the unit of
118 measurement. See setPrefix() and setSuffix(). The text in the spin
119 box is retrieved with text() (which includes any prefix() and
120 suffix()), or with cleanText() (which has no prefix(), no suffix()
121 and no leading or trailing whitespace).
122
123 It is often desirable to give the user a special (often default)
124 choice in addition to the range of numeric values. See
125 setSpecialValueText() for how to do this with QSpinBox.
126
127 \section1 Subclassing QSpinBox
128
129 If using prefix(), suffix(), and specialValueText() don't provide
130 enough control, you subclass QSpinBox and reimplement
131 valueFromText() and textFromValue(). For example, here's the code
132 for a custom spin box that allows the user to enter icon sizes
133 (e.g., "32 x 32"):
134
135 \snippet widgets/icons/iconsizespinbox.cpp 1
136 \codeline
137 \snippet widgets/icons/iconsizespinbox.cpp 2
138
139 See the \l{widgets/icons}{Icons} example for the full source
140 code.
141
142 \sa QDoubleSpinBox, QDateTimeEdit, QSlider, {Spin Boxes Example}
143*/
144

390/*!
391 Sets the step type for the spin box to \a stepType, which is single
392 step or adaptive decimal step.
393
394 Adaptive decimal step means that the step size will continuously be
395 adjusted to one power of ten below the current \l value. So when
396 the value is 1100, the step is set to 100, so stepping up once
397 increases it to 1200. For 1200 stepping up takes it to 1300. For
398 negative values, stepping down from -1100 goes to -1200.
399
400 Step direction is taken into account to handle edges cases, so
401 that stepping down from 100 takes the value to 99 instead of 90.
402 Thus a step up followed by a step down -- or vice versa -- always
403 lands on the starting value; 99 -> 100 -> 99.
404
405 Setting this will cause the spin box to disregard the value of
406 \l singleStep, although it is preserved so that \l singleStep
407 comes into effect if adaptive decimal step is later turned off.
408
409 \since 5.12
410*/
411
412void QSpinBox::setStepType(QAbstractSpinBox::StepType stepType)
413{
414 Q_D(QSpinBox);
415 d->stepType = stepType;
416}

448void QSpinBox::setDisplayIntegerBase(int base)

463/*!
464 This virtual function is used by the spin box whenever it needs to
465 display the given \a value. The default implementation returns a
466 string containing \a value printed in the standard way using
467 QWidget::locale().toString(), but with the thousand separator
468 removed unless setGroupSeparatorShown() is set. Reimplementations may
469 return anything. (See the example in the detailed description.)
470
471 Note: QSpinBox does not call this function for specialValueText()
472 and that neither prefix() nor suffix() should be included in the
473 return value.
474
475 If you reimplement this, you may also need to reimplement
476 valueFromText() and validate()
477
478 \sa valueFromText(), validate(), QLocale::groupSeparator()
479*/
480
481QString QSpinBox::textFromValue(int value) const
482{
483 Q_D(const QSpinBox);
484 QString str;
485
486 if (d->displayIntegerBase != 10) {
487 const auto prefix = value < 0 ? "-"_L1 : ""_L1;
488 str = prefix + QString::number(qAbs(t: value), base: d->displayIntegerBase);
489 } else {
490 str = locale().toString(i: value);
491 if (!d->showGroupSeparator && (qAbs(t: value) >= 1000 || value == INT_MIN)) {
492 str.remove(s: locale().groupSeparator());
493 }
494 }
495
496 return str;
497}

499/*!
500 \fn int QSpinBox::valueFromText(const QString &text) const
501
502 This virtual function is used by the spin box whenever it needs to
503 interpret \a text entered by the user as a value.
504
505 Subclasses that need to display spin box values in a non-numeric
506 way need to reimplement this function.
507
508 Note: QSpinBox handles specialValueText() separately; this
509 function is only concerned with the other values.
510
511 \sa textFromValue(), validate()
512*/
513
514int QSpinBox::valueFromText(const QString &text) const
515{
516 Q_D(const QSpinBox);
517
518 QString copy = text;
519 int pos = d->edit->cursorPosition();
520 QValidator::State state = QValidator::Acceptable;
521 return d->validateAndInterpret(input&: copy, pos, state).toInt();
522}

528{
529 Q_D(const QSpinBox);
530
531 QValidator::State state;
532 d->validateAndInterpret(input&: text, pos, state);
533 return state;
534}

867/*!
868 Sets the step type for the spin box to \a stepType, which is single
869 step or adaptive decimal step.
870
871 Adaptive decimal step means that the step size will continuously be
872 adjusted to one power of ten below the current \l value. So when
873 the value is 1100, the step is set to 100, so stepping up once
874 increases it to 1200. For 1200 stepping up takes it to 1300. For
875 negative values, stepping down from -1100 goes to -1200.
876
877 It also works for any decimal values, 0.041 is increased to 0.042
878 by stepping once.
879
880 Step direction is taken into account to handle edges cases, so
881 that stepping down from 100 takes the value to 99 instead of 90.
882 Thus a step up followed by a step down -- or vice versa -- always
883 lands on the starting value; 99 -> 100 -> 99.
884

886 \l singleStep, although it is preserved so that \l singleStep
887 comes into effect if adaptive decimal step is later turned off.

890*/

1071 \internal Multi purpose function that parses input, sets state to
1072 the appropriate state and returns the value it will be interpreted
1073 as.
1074*/
1075
1076QVariant QSpinBoxPrivate::validateAndInterpret(QString &input, int &pos,
1077 QValidator::State &state) const
1078{
1079 if (cachedText == input && !input.isEmpty()) {
1080 state = cachedState;
1081 QSBDEBUG() << "cachedText was '" << cachedText << "' state was "
1082 << state << " and value was " << cachedValue;
1083
1084 return cachedValue;
1085 }
1086 const int max = maximum.toInt();
1087 const int min = minimum.toInt();
1088
1089 QString copy = stripped(text: input, pos: &pos);
1090 QSBDEBUG() << "input" << input << "copy" << copy;
1091 state = QValidator::Acceptable;
1092 int num = min;
1093
1094 if (max != min && (copy.isEmpty()
1095 || (min < 0 && copy == "-"_L1)
1096 || (max >= 0 && copy == "+"_L1))) {
1097 state = QValidator::Intermediate;
1098 QSBDEBUG() << __FILE__ << __LINE__<< "num is set to" << num;
1099 } else if (copy.startsWith(c: u'-') && min >= 0) {
1100 state = QValidator::Invalid; // special-case -0 will be interpreted as 0 and thus not be invalid with a range from 0-100
1101 } else {
1102 bool ok = false;
1103 if (displayIntegerBase != 10) {
1104 num = copy.toInt(ok: &ok, base: displayIntegerBase);
1105 } else {
1106 num = locale.toInt(s: copy, ok: &ok);
1107 if (!ok && (max >= 1000 || min <= -1000)) {
1108 const QString sep(locale.groupSeparator());
1109 const QString doubleSep = sep + sep;
1110 if (copy.contains(s: sep) && !copy.contains(s: doubleSep)) {
1111 QString copy2 = copy;
1112 copy2.remove(s: sep);
1113 num = locale.toInt(s: copy2, ok: &ok);
1114 }
1115 }
1116 }
1117 QSBDEBUG() << __FILE__ << __LINE__<< "num is set to" << num;
1118 if (!ok) {
1119 state = QValidator::Invalid;
1120 } else if (num >= min && num <= max) {
1121 state = QValidator::Acceptable;
1122 } else if (max == min) {
1123 state = QValidator::Invalid;
1124 } else {
1125 if ((num >= 0 && num > max) || (num < 0 && num < min)) {
1126 state = QValidator::Invalid;
1127 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1128 } else {
1129 state = QValidator::Intermediate;
1130 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Intermediate";
1131 }
1132 }
1133 }
1134 if (state != QValidator::Acceptable)
1135 num = max > 0 ? min : max;
1136 input = prefix + copy + suffix;
1137 cachedText = input;
1138 cachedState = state;
1139 cachedValue = QVariant((int)num);
1140
1141 QSBDEBUG() << "cachedText is set to '" << cachedText << "' state is set to "
1142 << state << " and value is set to " << cachedValue;
1143 return cachedValue;
1144}

1146QVariant QSpinBoxPrivate::calculateAdaptiveDecimalStep(int steps) const
1147{
1148 const int intValue = value.toInt();
1149 const int absValue = qAbs(t: intValue);
1150
1151 if (absValue < 100)
1152 return 1;
1153
1154 const bool valueNegative = intValue < 0;
1155 const bool stepsNegative = steps < 0;
1156 const int signCompensation = (valueNegative == stepsNegative) ? 0 : 1;
1157
1158 const int log = static_cast<int>(std::log10(x: absValue - signCompensation)) - 1;
1159 return static_cast<int>(std::pow(x: 10, y: log));
1160}

1223/*!
1224 \internal Multi purpose function that parses input, sets state to
1225 the appropriate state and returns the value it will be interpreted
1226 as.
1227*/
1228
1229QVariant QDoubleSpinBoxPrivate::validateAndInterpret(QString &input, int &pos,
1230 QValidator::State &state) const
1231{
1232 if (cachedText == input && !input.isEmpty()) {
1233 state = cachedState;
1234 QSBDEBUG() << "cachedText was '" << cachedText << "' state was "
1235 << state << " and value was " << cachedValue;
1236 return cachedValue;
1237 }
1238 const double max = maximum.toDouble();
1239 const double min = minimum.toDouble();
1240
1241 QString copy = stripped(text: input, pos: &pos);
1242 QSBDEBUG() << "input" << input << "copy" << copy;
1243 int len = copy.size();
1244 double num = min;
1245 const bool plus = max >= 0;
1246 const bool minus = min <= 0;
1247
1248 const QString group(locale.groupSeparator());
1249 const uint groupUcs = (group.isEmpty() ? 0 :
1250 (group.size() > 1 && group.at(i: 0).isHighSurrogate()
1251 ? QChar::surrogateToUcs4(high: group.at(i: 0), low: group.at(i: 1))
1252 : group.at(i: 0).unicode()));
1253 switch (len) {
1254 case 0:
1255 state = max != min ? QValidator::Intermediate : QValidator::Invalid;
1256 goto end;
1257 case 1:
1258 if (copy.at(i: 0) == locale.decimalPoint()
1259 || (plus && copy.at(i: 0) == u'+')
1260 || (minus && copy.at(i: 0) == u'-')) {
1261 state = QValidator::Intermediate;
1262 goto end;
1263 }
1264 break;
1265 case 2:
1266 if (copy.at(i: 1) == locale.decimalPoint()
1267 && ((plus && copy.at(i: 0) == u'+') || (minus && copy.at(i: 0) == u'-'))) {
1268 state = QValidator::Intermediate;
1269 goto end;
1270 }
1271 break;
1272 default: break;
1273 }
1274
1275 if (groupUcs && copy.startsWith(s: group)) {
1276 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1277 state = QValidator::Invalid;
1278 goto end;
1279 } else if (len > 1) {
1280 const int dec = copy.indexOf(s: locale.decimalPoint());
1281 if (dec != -1) {
1282 if (dec + 1 < copy.size() && copy.at(i: dec + 1) == locale.decimalPoint() && pos == dec + 1) {
1283 copy.remove(i: dec + 1, len: 1); // typing a delimiter when you are on the delimiter
1284 } // should be treated as typing right arrow
1285
1286 if (copy.size() - dec > decimals + 1) {
1287 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1288 state = QValidator::Invalid;
1289 goto end;
1290 }
1291 for (int i = dec + 1; i < copy.size(); ++i) {
1292 if (copy.at(i).isSpace()
1293 || (groupUcs && QStringView{copy}.sliced(pos: i).startsWith(s: group))) {
1294 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1295 state = QValidator::Invalid;
1296 goto end;
1297 }
1298 }
1299 } else {
1300 const QChar last = copy.back();
1301 const bool groupEnd = groupUcs && copy.endsWith(s: group);
1302 const QStringView head(copy.constData(), groupEnd ? len - group.size() : len - 1);
1303 const QChar secondLast = head.back();
1304 if ((groupEnd || last.isSpace())
1305 && ((groupUcs && head.endsWith(s: group)) || secondLast.isSpace())) {
1306 state = QValidator::Invalid;
1307 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1308 goto end;
1309 } else if (last.isSpace() && (!QChar::isSpace(ucs4: groupUcs) || secondLast.isSpace())) {
1310 state = QValidator::Invalid;
1311 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1312 goto end;
1313 }
1314 }
1315 }
1316
1317 {
1318 bool ok = false;
1319 num = locale.toDouble(s: copy, ok: &ok);
1320 QSBDEBUG() << __FILE__ << __LINE__ << locale << copy << num << ok;
1321
1322 if (!ok) {
1323 if (QChar::isPrint(ucs4: groupUcs)) {
1324 if (max < 1000 && min > -1000 && groupUcs && copy.contains(s: group)) {
1325 state = QValidator::Invalid;
1326 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1327 goto end;
1328 }
1329
1330 const int len = copy.size();
1331 for (int i = 0; i < len - 1;) {
1332 if (groupUcs && QStringView{copy}.sliced(pos: i).startsWith(s: group)) {
1333 if (QStringView(copy).mid(pos: i + group.size()).startsWith(s: group)) {
1334 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1335 state = QValidator::Invalid;
1336 goto end;
1337 }
1338 i += group.size();
1339 } else {
1340 i++;
1341 }
1342 }
1343
1344 QString copy2 = copy;
1345 if (groupUcs)
1346 copy2.remove(s: group);
1347 num = locale.toDouble(s: copy2, ok: &ok);
1348 QSBDEBUG() << group << num << copy2 << ok;
1349
1350 if (!ok) {
1351 state = QValidator::Invalid;
1352 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1353 goto end;
1354 }
1355 }
1356 }
1357
1358 if (!ok) {
1359 state = QValidator::Invalid;
1360 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1361 } else if (num >= min && num <= max) {
1362 state = QValidator::Acceptable;
1363 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Acceptable";
1364 } else if (max == min) { // when max and min is the same the only non-Invalid input is max (or min)
1365 state = QValidator::Invalid;
1366 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1367 } else {
1368 if ((num >= 0 && num > max) || (num < 0 && num < min)) {
1369 state = QValidator::Invalid;
1370 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Invalid";
1371 } else {
1372 state = QValidator::Intermediate;
1373 QSBDEBUG() << __FILE__ << __LINE__<< "state is set to Intermediate";
1374 }
1375 }
1376 }
1377
1378end:
1379 if (state != QValidator::Acceptable) {
1380 num = max > 0 ? min : max;
1381 }
1382
1383 input = prefix + copy + suffix;
1384 cachedText = input;
1385 cachedState = state;
1386 cachedValue = QVariant(num);
1387 return QVariant(num);
1388}

1401QVariant QDoubleSpinBoxPrivate::calculateAdaptiveDecimalStep(int steps) const
1402{
1403 const double doubleValue = value.toDouble();
1404 const double minStep = std::pow(x: 10, y: -decimals);
1405 double absValue = qAbs(t: doubleValue);
1406
1407 if (absValue < minStep)
1408 return minStep;
1409
1410 const bool valueNegative = doubleValue < 0;
1411 const bool stepsNegative = steps < 0;
1412 if (valueNegative != stepsNegative)
1413 absValue /= 1.01;
1414
1415 const double shift = std::pow(x: 10, y: 1 - std::floor(x: std::log10(x: absValue)));
1416 const double absRounded = round(value: absValue * shift) / shift;
1417 const double log = floorf(x: std::log10(x: absRounded)) - 1;
1418
1419 return std::max(a: minStep, b: std::pow(x: 10, y: log));
1420}
