70Q_WIDGETS_EXPORT void qt_setMainWindowTitleWidget(QMainWindow *mainWindow, Qt::DockWidgetArea area, QWidget *widget)
71{
72 QGridLayout *topLayout = qobject_cast<QGridLayout *>(mainWindow->layout());
73 Q_ASSERT(topLayout);
74
75 int row = 0;
76 int column = 0;
77
78 switch (area) {
79 case Qt::LeftDockWidgetArea:
80 row = 1;
81 column = 0;
82 break;
83 case Qt::TopDockWidgetArea:
84 row = 0;
85 column = 1;
86 break;
87 case Qt::BottomDockWidgetArea:
88 row = 2;
89 column = 1;
90 break;
91 case Qt::RightDockWidgetArea:
92 row = 1;
93 column = 2;
94 break;
95 default:
96 Q_ASSERT_X(false, "qt_setMainWindowTitleWidget", "Unknown area");
97 return;
98 }
99
100 if (QLayoutItem *oldItem = topLayout->itemAtPosition(row, column))
101 delete oldItem->widget();
102 topLayout->addWidget(widget, row, column);
103}
104#endif

158/*!
159 \class QMainWindow
160 \brief The QMainWindow class provides a main application
161 window.
162 \ingroup mainwindow-classes
163 \inmodule QtWidgets
164
165 \tableofcontents
166
167 \section1 Qt Main Window Framework
168
169 A main window provides a framework for building an
170 application's user interface. Qt has QMainWindow and its \l{Main
171 Window and Related Classes}{related classes} for main window
172 management. QMainWindow has its own layout to which you can add
173 \l{QToolBar}s, \l{QDockWidget}s, a
174 QMenuBar, and a QStatusBar. The layout has a center area that can
175 be occupied by any kind of widget. You can see an image of the
176 layout below.
177
178 \image mainwindowlayout.png
179
180 \section1 Creating Main Window Components
181
182 A central widget will typically be a standard Qt widget such
183 as a QTextEdit or a QGraphicsView. Custom widgets can also be
184 used for advanced applications. You set the central widget with \c
185 setCentralWidget().
186
187 Main windows have either a single (SDI) or multiple (MDI)
188 document interface. You create MDI applications in Qt by using a
189 QMdiArea as the central widget.
190
191 We will now examine each of the other widgets that can be
192 added to a main window. We give examples on how to create and add
193 them.
194
195 \section2 Creating Menus
196
197 Qt implements menus in QMenu and QMainWindow keeps them in a
198 QMenuBar. \l{QAction}{QAction}s are added to the menus, which
199 display them as menu items.
200
201 You can add new menus to the main window's menu bar by calling
202 \c menuBar(), which returns the QMenuBar for the window, and then
203 add a menu with QMenuBar::addMenu().
204
205 QMainWindow comes with a default menu bar, but you can also
206 set one yourself with \c setMenuBar(). If you wish to implement a
207 custom menu bar (i.e., not use the QMenuBar widget), you can set it
208 with \c setMenuWidget().
209
210 An example of how to create menus follows:
211
212 \snippet code/src_widgets_widgets_qmainwindow.cpp 0
213
214 The \c createPopupMenu() function creates popup menus when the
215 main window receives context menu events. The default
216 implementation generates a menu with the checkable actions from
217 the dock widgets and toolbars. You can reimplement \c
218 createPopupMenu() for a custom menu.
219
220 \section2 Creating Toolbars
221
222 Toolbars are implemented in the QToolBar class. You add a
223 toolbar to a main window with \c addToolBar().
224
225 You control the initial position of toolbars by assigning them
226 to a specific Qt::ToolBarArea. You can split an area by inserting
227 a toolbar break - think of this as a line break in text editing -
228 with \c addToolBarBreak() or \c insertToolBarBreak(). You can also
229 restrict placement by the user with QToolBar::setAllowedAreas()
230 and QToolBar::setMovable().
231
232 The size of toolbar icons can be retrieved with \c iconSize().
233 The sizes are platform dependent; you can set a fixed size with \c
234 setIconSize(). You can alter the appearance of all tool buttons in
235 the toolbars with \c setToolButtonStyle().
236
237 An example of toolbar creation follows:
238
239 \snippet code/src_widgets_widgets_qmainwindow.cpp 1
240
241 \section2 Creating Dock Widgets
242
243 Dock widgets are implemented in the QDockWidget class. A dock
244 widget is a window that can be docked into the main window. You
245 add dock widgets to a main window with \c addDockWidget().
246
247 There are four dock widget areas as given by the
248 Qt::DockWidgetArea enum: left, right, top, and bottom. You can
249 specify which dock widget area that should occupy the corners
250 where the areas overlap with \c setCorner(). By default
251 each area can only contain one row (vertical or horizontal) of
252 dock widgets, but if you enable nesting with \c
253 setDockNestingEnabled(), dock widgets can be added in either
254 direction.
255
256 Two dock widgets may also be stacked on top of each other. A
257 QTabBar is then used to select which of the widgets should be
258 displayed.
259
260 We give an example of how to create and add dock widgets to a
261 main window:
262
263 \snippet mainwindowsnippet.cpp 0
264
265 \section2 The Status Bar
266
267 You can set a status bar with \c setStatusBar(), but one is
268 created the first time \c statusBar() (which returns the main
269 window's status bar) is called. See QStatusBar for information on
270 how to use it.
271
272 \section1 Storing State
273
274 QMainWindow can store the state of its layout with \c
275 saveState(); it can later be retrieved with \c restoreState(). It
276 is the position and size (relative to the size of the main window)
277 of the toolbars and dock widgets that are stored.
278
279 \sa QMenuBar, QToolBar, QStatusBar, QDockWidget,
280 {Qt Widgets - Application Example}, {Dock Widgets Example},
281 {MDI Example}, {Menus Example}
282*/

350 \brief the docking behavior of QMainWindow
351 \since 4.3
352
353 The default value is AnimatedDocks | AllowTabbedDocks.
354*/

356/*!
357 \enum QMainWindow::DockOption
358 \since 4.3
359
360 This enum contains flags that specify the docking behavior of QMainWindow.
361
362 \value AnimatedDocks Identical to the \l animated property.
363
364 \value AllowNestedDocks Identical to the \l dockNestingEnabled property.
365
366 \value AllowTabbedDocks The user can drop one dock widget "on top" of
367 another. The two widgets are stacked and a tab
368 bar appears for selecting which one is visible.
369
370 \value ForceTabbedDocks Each dock area contains a single stack of tabbed
371 dock widgets. In other words, dock widgets cannot
372 be placed next to each other in a dock area. If
373 this option is set, AllowNestedDocks has no effect.
374
375 \value VerticalTabs The two vertical dock areas on the sides of the
376 main window show their tabs vertically. If this
377 option is not set, all dock areas show their tabs
378 at the bottom. Implies AllowTabbedDocks. See also
379 \l setTabPosition().
380
381 \value GroupedDragging When dragging the titlebar of a dock, all the tabs
382 that are tabbed with it are going to be dragged.
383 Implies AllowTabbedDocks. Does not work well if
384 some QDockWidgets have restrictions in which area
385 they are allowed. (This enum value was added in Qt
386 5.6.)
387
388 These options only control how dock widgets may be dropped in a QMainWindow.
389 They do not re-arrange the dock widgets to conform with the specified
390 options. For this reason they should be set before any dock widgets
391 are added to the main window. Exceptions to this are the AnimatedDocks and
392 VerticalTabs options, which may be set at any time.
393*/

474/*!
475 Sets the menu bar for the main window to \a menuBar.
476
477 Note: QMainWindow takes ownership of the \a menuBar pointer and
478 deletes it at the appropriate time.
479
480 \sa menuBar()
481*/
482void QMainWindow::setMenuBar(QMenuBar *menuBar)
483{
484 QLayout *topLayout = layout();
485
486 if (QWidget *existingMenuBar = topLayout->menuBar(); existingMenuBar && existingMenuBar != menuBar) {
487 // Reparent corner widgets before we delete the old menu bar.
488 QMenuBar *oldMenuBar = qobject_cast<QMenuBar *>(object: existingMenuBar);
489 if (oldMenuBar && menuBar) {
490 // TopLeftCorner widget.
491 QWidget *cornerWidget = oldMenuBar->cornerWidget(corner: Qt::TopLeftCorner);
492 if (cornerWidget)
493 menuBar->setCornerWidget(w: cornerWidget, corner: Qt::TopLeftCorner);
494 // TopRightCorner widget.
495 cornerWidget = oldMenuBar->cornerWidget(corner: Qt::TopRightCorner);
496 if (cornerWidget)
497 menuBar->setCornerWidget(w: cornerWidget, corner: Qt::TopRightCorner);
498 }
499
500 existingMenuBar->hide();
501 existingMenuBar->setParent(nullptr);
502 existingMenuBar->deleteLater();
503 }
504 topLayout->setMenuBar(menuBar);
505}

706/*!
707 Adds the \a toolbar into the specified \a area in this main
708 window. The \a toolbar is placed at the end of the current tool
709 bar block (i.e. line). If the main window already manages \a toolbar
710 then it will only move the toolbar to \a area.
711
712 \sa insertToolBar(), addToolBarBreak(), insertToolBarBreak()
713*/
714void QMainWindow::addToolBar(Qt::ToolBarArea area, QToolBar *toolbar)
715{
716 if (!checkToolBarArea(area, where: "QMainWindow::addToolBar"))
717 return;
718
719 Q_D(QMainWindow);
720
721 disconnect(sender: this, SIGNAL(iconSizeChanged(QSize)),
722 receiver: toolbar, SLOT(_q_updateIconSize(QSize)));
723 disconnect(sender: this, SIGNAL(toolButtonStyleChanged(Qt::ToolButtonStyle)),
724 receiver: toolbar, SLOT(_q_updateToolButtonStyle(Qt::ToolButtonStyle)));
725
726 if (toolbar->d_func()->state && toolbar->d_func()->state->dragging) {
727 //removing a toolbar which is dragging will cause crash
728#if QT_CONFIG(dockwidget)
729 bool animated = isAnimated();
730 setAnimated(false);
731#endif
732 toolbar->d_func()->endDrag();
733#if QT_CONFIG(dockwidget)
734 setAnimated(animated);
735#endif
736 }
737
738 d->layout->removeToolBar(toolbar);
739
740 toolbar->d_func()->_q_updateIconSize(sz: d->iconSize);
741 toolbar->d_func()->_q_updateToolButtonStyle(style: d->toolButtonStyle);
742 connect(sender: this, SIGNAL(iconSizeChanged(QSize)),
743 receiver: toolbar, SLOT(_q_updateIconSize(QSize)));
744 connect(sender: this, SIGNAL(toolButtonStyleChanged(Qt::ToolButtonStyle)),
745 receiver: toolbar, SLOT(_q_updateToolButtonStyle(Qt::ToolButtonStyle)));
746
747 d->layout->addToolBar(area, toolbar);
748}

1227/*!

1229 dockwidgets. Also restores the corner settings too. The
1230 \a version number is compared with that stored in \a state.
1231 If they do not match, the mainwindow's state is left
1232 unchanged, and this function returns \c false; otherwise, the state
1233 is restored, and this function returns \c true.

1239

1241 QWidget::restoreGeometry(), restoreDockWidget()

1243bool QMainWindow::restoreState(const QByteArray &state, int version)
1244{

1246 return false;
1247 QByteArray sd = state;
1248 QDataStream stream(&sd, QIODevice::ReadOnly);
1249 stream.setVersion(QDataStream::Qt_5_0);

1252 stream >> v;

1431 dock widgets present in the main window. If there are no toolbars and
1432 dock widgets present, this function returns \nullptr.
1433

1435 activates a context menu, typically by right-clicking on a toolbar or a dock
1436 widget.

1439 return a newly-created popup menu. Ownership of the popup menu is transferred
1440 to the caller.

1445{

1448#if QT_CONFIG(dockwidget)

1450 if (dockwidgets.size()) {
1451 menu = new QMenu(this);
1452 for (int i = 0; i < dockwidgets.size(); ++i) {
1453 QDockWidget *dockWidget = dockwidgets.at(i);
1454 // filter to find out if we own this QDockWidget
1455 if (dockWidget->parentWidget() == this) {
1456 if (d->layout->layoutState.dockAreaLayout.indexOf(dockWidget).isEmpty())
1457 continue;
1458 } else if (QDockWidgetGroupWindow *dwgw =
1459 qobject_cast<QDockWidgetGroupWindow *>(object: dockWidget->parentWidget())) {
1460 if (dwgw->parentWidget() != this)
1461 continue;
1462 if (dwgw->layoutInfo()->indexOf(widget: dockWidget).isEmpty())
1463 continue;

1468 }

1470 }
1471#endif // QT_CONFIG(dockwidget)
