50void QToolBarPrivate::init()
51{
52 Q_Q(QToolBar);
53 q->setSizePolicy(QSizePolicy(QSizePolicy::Preferred, QSizePolicy::Fixed));
54 q->setBackgroundRole(QPalette::Button);
55 q->setAttribute(Qt::WA_Hover);
56 q->setAttribute(Qt::WA_X11NetWmWindowTypeToolBar);
57
58 QStyle *style = q->style();
59 int e = style->pixelMetric(metric: QStyle::PM_ToolBarIconSize, option: nullptr, widget: q);
60 iconSize = QSize(e, e);
61
62 layout = new QToolBarLayout(q);
63 layout->updateMarginAndSpacing();
64
65 toggleViewAction = new QAction(q);
66 toggleViewAction->setCheckable(true);
67 q->setMovable(q->style()->styleHint(stylehint: QStyle::SH_ToolBar_Movable, opt: nullptr, widget: q ));
68 QObject::connect(sender: toggleViewAction, SIGNAL(triggered(bool)), receiver: q, SLOT(_q_toggleView(bool)));
69}

114void QToolBarPrivate::setWindowState(bool floating, bool unplug, const QRect &rect)
115{
116 Q_Q(QToolBar);
117 bool visible = !q->isHidden();
118 bool wasFloating = q->isFloating(); // ...is also currently using popup menus
119
120 q->hide();
121
122 updateWindowFlags(floating, unplug);
123
124 if (floating != wasFloating)
125 layout->checkUsePopupMenu();
126
127 if (!rect.isNull())
128 q->setGeometry(rect);
129
130 if (visible)
131 q->show();
132
133 if (floating != wasFloating)
134 emit q->topLevelChanged(topLevel: floating);
135}

137void QToolBarPrivate::initDrag(const QPoint &pos)
138{
139 Q_Q(QToolBar);
140
141 if (state != nullptr)
142 return;
143
144 QMainWindow *win = qobject_cast<QMainWindow*>(object: parent);
145 Q_ASSERT(win != nullptr);
146 QMainWindowLayout *layout = qt_mainwindow_layout(window: win);
147 Q_ASSERT(layout != nullptr);
148 if (layout->pluggingWidget != nullptr) // the main window is animating a docking operation
149 return;
150
151 state = new DragState;
152 state->pressPos = pos;
153 state->dragging = false;
154 state->moving = false;
155 state->widgetItem = nullptr;
156
157 if (q->isRightToLeft())
158 state->pressPos = QPoint(q->width() - state->pressPos.x(), state->pressPos.y());
159}

161void QToolBarPrivate::startDrag(bool moving)
162{
163 Q_Q(QToolBar);
164
165 Q_ASSERT(state != nullptr);
166
167 if ((moving && state->moving) || state->dragging)
168 return;
169
170 QMainWindow *win = qobject_cast<QMainWindow*>(object: parent);
171 Q_ASSERT(win != nullptr);
172 QMainWindowLayout *layout = qt_mainwindow_layout(window: win);
173 Q_ASSERT(layout != nullptr);
174
175 if (!moving) {
176 state->widgetItem = layout->unplug(widget: q);
177 Q_ASSERT(state->widgetItem != nullptr);
178 }
179 state->dragging = !moving;
180 state->moving = moving;
181}
182
183void QToolBarPrivate::endDrag()
184{
185 Q_Q(QToolBar);
186 Q_ASSERT(state != nullptr);
187
188 q->releaseMouse();
189
190 if (state->dragging) {
191 QMainWindowLayout *layout = qt_mainwindow_layout(window: qobject_cast<QMainWindow *>(object: q->parentWidget()));
192 Q_ASSERT(layout != nullptr);
193
194 if (!layout->plug(widgetItem: state->widgetItem)) {
195 if (q->isFloatable()) {
196 layout->restore();
197 setWindowState(floating: true); // gets rid of the X11BypassWindowManager window flag
198 // and activates the resizer
199 q->activateWindow();
200 } else {
201 layout->revert(widgetItem: state->widgetItem);
202 }
203 }
204 }
205
206 delete state;
207 state = nullptr;
208}

210bool QToolBarPrivate::mousePressEvent(QMouseEvent *event)
211{
212 Q_Q(QToolBar);
213 QStyleOptionToolBar opt;
214 q->initStyleOption(option: &opt);
215 if (q->style()->subElementRect(subElement: QStyle::SE_ToolBarHandle, option: &opt, widget: q).contains(p: event->position().toPoint()) == false) {
216#ifdef Q_OS_MACOS
217 // When using the unified toolbar on OS X, the user can click and
218 // drag between toolbar contents to move the window. Make this work by
219 // implementing the standard mouse-dragging code and then call
220 // window->move() in mouseMoveEvent below.
221 if (QMainWindow *mainWindow = qobject_cast<QMainWindow *>(parent)) {
222 if (mainWindow->toolBarArea(q) == Qt::TopToolBarArea
223 && mainWindow->unifiedTitleAndToolBarOnMac()
224 && q->childAt(event->pos()) == 0) {
225 macWindowDragging = true;
226 macWindowDragPressPosition = event->pos();
227 return true;
228 }
229 }
230#endif
231 return false;
232 }
233
234 if (event->button() != Qt::LeftButton)
235 return true;
236
237 if (!layout->movable())
238 return true;
239
240 initDrag(pos: event->position().toPoint());
241 return true;
242}

244bool QToolBarPrivate::mouseReleaseEvent(QMouseEvent*)
245{
246 if (state != nullptr) {
247 endDrag();
248 return true;
249 } else {
250#ifdef Q_OS_MACOS
251 if (!macWindowDragging)
252 return false;
253 macWindowDragging = false;
254 macWindowDragPressPosition = QPoint();
255 return true;
256#endif
257 return false;
258 }

261bool QToolBarPrivate::mouseMoveEvent(QMouseEvent *event)
262{
263 Q_Q(QToolBar);
264
265 if (!state) {
266#ifdef Q_OS_MACOS
267 if (!macWindowDragging)
268 return false;
269 QWidget *w = q->window();
270 const QPoint delta = event->pos() - macWindowDragPressPosition;
271 w->move(w->pos() + delta);
272 return true;
273#endif
274 return false;
275 }
276
277 QMainWindow *win = qobject_cast<QMainWindow*>(object: parent);
278 if (win == nullptr)
279 return true;
280
281 QMainWindowLayout *layout = qt_mainwindow_layout(window: win);
282 Q_ASSERT(layout != nullptr);
283
284 if (layout->pluggingWidget == nullptr
285 && (event->position().toPoint() - state->pressPos).manhattanLength() > QApplication::startDragDistance()) {
286 const bool wasDragging = state->dragging;
287 const bool moving = !q->isWindow() && (orientation == Qt::Vertical ?
288 event->position().toPoint().x() >= 0 && event->position().toPoint().x() < q->width() :
289 event->position().toPoint().y() >= 0 && event->position().toPoint().y() < q->height());
290
291 startDrag(moving);
292 if (!moving && !wasDragging)
293 q->grabMouse();
294 }
295
296 if (state->dragging) {
297 QPoint pos = event->globalPosition().toPoint();
298 // if we are right-to-left, we move so as to keep the right edge the same distance
299 // from the mouse
300 if (q->isLeftToRight())
301 pos -= state->pressPos;
302 else
303 pos += QPoint(state->pressPos.x() - q->width(), -state->pressPos.y());
304
305 q->move(pos);
306 layout->hover(hoverTarget: state->widgetItem, mousePos: event->globalPosition().toPoint());
307 } else if (state->moving) {
308
309 const QPoint rtl(q->width() - state->pressPos.x(), state->pressPos.y()); //for RTL
310 const QPoint globalPressPos = q->mapToGlobal(q->isRightToLeft() ? rtl : state->pressPos);
311 int pos = 0;
312
313 const QWindow *handle = q->window() ? q->window()->windowHandle() : nullptr;
314 const QPoint delta = handle
315 ? QHighDpi::fromNativePixels(value: event->globalPosition(), context: handle).toPoint()
316 - QHighDpi::fromNativePixels(value: globalPressPos, context: handle)
317 : event->globalPosition().toPoint() - globalPressPos;
318
319 if (orientation == Qt::Vertical) {
320 pos = q->y() + delta.y();
321 } else {
322 if (q->isRightToLeft()) {
323 pos = win->width() - q->width() - q->x() - delta.x();
324 } else {
325 pos = q->x() + delta.x();
326 }
327 }
328
329 layout->moveToolBar(toolbar: q, pos);
330 }
331 return true;
332}

352/*!
353 \class QToolBar
354
355 \brief The QToolBar class provides a movable panel that contains a
356 set of controls.
357
358 \ingroup mainwindow-classes
359 \inmodule QtWidgets
360
361 Toolbar buttons are added by adding \e actions, using addAction()
362 or insertAction(). Groups of buttons can be separated using
363 addSeparator() or insertSeparator(). If a toolbar button is not
364 appropriate, a widget can be inserted instead using addWidget() or
365 insertWidget(). Examples of suitable widgets are QSpinBox,
366 QDoubleSpinBox, and QComboBox. When a toolbar button is pressed, it
367 emits the actionTriggered() signal.
368
369 A toolbar can be fixed in place in a particular area (e.g., at the
370 top of the window), or it can be movable between toolbar areas;
371 see setMovable(), isMovable(), allowedAreas() and isAreaAllowed().
372
373 When a toolbar is resized in such a way that it is too small to
374 show all the items it contains, an extension button will appear as
375 the last item in the toolbar. Pressing the extension button will
376 pop up a menu containing the items that do not currently fit in
377 the toolbar.
378
379 When a QToolBar is not a child of a QMainWindow, it loses the ability
380 to populate the extension pop up with widgets added to the toolbar using
381 addWidget(). Please use widget actions created by inheriting QWidgetAction
382 and implementing QWidgetAction::createWidget() instead.
383
384 \sa QToolButton, QMenu, QAction, {Qt Widgets - Application Example}
385*/

637void QToolBar::setIconSize(const QSize &iconSize)
638{
639 Q_D(QToolBar);
640 QSize sz = iconSize;
641 if (!sz.isValid()) {
642 QMainWindow *mw = qobject_cast<QMainWindow *>(object: parentWidget());
643 if (mw && mw->layout()) {
644 QLayout *layout = mw->layout();
645 int i = 0;
646 QLayoutItem *item = nullptr;
647 do {
648 item = layout->itemAt(index: i++);
649 if (item && (item->widget() == this))
650 sz = mw->iconSize();
651 } while (!sz.isValid() && item != nullptr);
652 }
653 }
654 if (!sz.isValid()) {
655 const int metric = style()->pixelMetric(metric: QStyle::PM_ToolBarIconSize, option: nullptr, widget: this);
656 sz = QSize(metric, metric);
657 }
658 if (d->iconSize != sz) {
659 d->iconSize = sz;
660 setMinimumSize(minw: 0, minh: 0);
661 emit iconSizeChanged(iconSize: d->iconSize);
662 }
663 d->explicitIconSize = iconSize.isValid();
664
665 d->layout->invalidate();
666}

820void QToolBar::actionEvent(QActionEvent *event)
821{
822 Q_D(QToolBar);
823 auto action = static_cast<QAction *>(event->action());
824 QWidgetAction *widgetAction = qobject_cast<QWidgetAction *>(object: action);
825
826 switch (event->type()) {
827 case QEvent::ActionAdded: {
828 Q_ASSERT_X(widgetAction == nullptr || d->layout->indexOf(widgetAction) == -1,
829 "QToolBar", "widgets cannot be inserted multiple times");
830
831 // reparent the action to this toolbar if it has been created
832 // using the addAction(text) etc. convenience functions, to
833 // preserve Qt 4.1.x behavior. The widget is already
834 // reparented to us due to the createWidget call inside
835 // createItem()
836 if (widgetAction != nullptr && widgetAction->d_func()->autoCreated)
837 widgetAction->setParent(this);
838
839 int index = d->layout->count();
840 if (event->before()) {
841 index = d->layout->indexOf(action: event->before());
842 Q_ASSERT_X(index != -1, "QToolBar::insertAction", "internal error");
843 }
844 d->layout->insertAction(index, action);
845 break;
846 }
847
848 case QEvent::ActionChanged:
849 d->layout->invalidate();
850 break;
851
852 case QEvent::ActionRemoved: {
853 int index = d->layout->indexOf(action);
854 if (index != -1) {
855 delete d->layout->takeAt(index);
856 }
857 break;
858 }
859
860 default:
861 Q_ASSERT_X(false, "QToolBar::actionEvent", "internal error");
862 }
863}

913/*
914 Checks if an expanded toolbar has to wait for this popup to close before
915 the toolbar collapses. This is true if
916 1) the popup has the toolbar in its parent chain,
917 2) the popup is a menu whose menuAction is somewhere in the toolbar.
918*/
919static bool waitForPopup(QToolBar *tb, QWidget *popup)
920{
921 if (popup == nullptr || popup->isHidden())
922 return false;
923
924 QWidget *w = popup;
925 while (w != nullptr) {
926 if (w == tb)
927 return true;
928 w = w->parentWidget();
929 }
930
931 QMenu *menu = qobject_cast<QMenu*>(object: popup);
932 if (menu == nullptr)
933 return false;
934
935 const QAction *action = menu->menuAction();
936 for (auto object : action->associatedObjects()) {
937 if (QWidget *widget = qobject_cast<QWidget*>(o: object)) {
938 if (waitForPopup(tb, popup: widget))
939 return true;
940 }
941 }
942
943 return false;
944}

963/*! \reimp */
964bool QToolBar::event(QEvent *event)
965{
966 Q_D(QToolBar);
967
968 switch (event->type()) {
969 case QEvent::Timer:
970 if (d->waitForPopupTimer.timerId() == static_cast<QTimerEvent*>(event)->timerId()) {
971 QWidget *w = QApplication::activePopupWidget();
972 if (!waitForPopup(tb: this, popup: w)) {
973 d->waitForPopupTimer.stop();
974 if (!this->underMouse())
975 d->layout->setExpanded(false);
976 }
977 }
978 break;
979 case QEvent::Hide:
980 if (!isHidden())
981 break;
982 Q_FALLTHROUGH();
983 case QEvent::Show:
984 d->toggleViewAction->setChecked(event->type() == QEvent::Show);
985#ifdef Q_OS_MACOS
986 enableMacToolBar(this, event->type() == QEvent::Show);
987#endif
988 emit visibilityChanged(visible: event->type() == QEvent::Show);
989 break;
990 case QEvent::ParentChange:
991 d->layout->checkUsePopupMenu();
992 break;
993
994 case QEvent::MouseButtonPress: {
995 if (d->mousePressEvent(event: static_cast<QMouseEvent*>(event)))
996 return true;
997 break;
998 }
999 case QEvent::MouseButtonRelease:
1000 if (d->mouseReleaseEvent(static_cast<QMouseEvent*>(event)))
1001 return true;
1002 break;
1003 case QEvent::HoverEnter:
1004 case QEvent::HoverLeave:
1005 // there's nothing special to do here and we don't want to update the whole widget
1006 return true;
1007 case QEvent::HoverMove: {
1008#ifndef QT_NO_CURSOR
1009 QHoverEvent *e = static_cast<QHoverEvent*>(event);
1010 QStyleOptionToolBar opt;
1011 initStyleOption(option: &opt);
1012 if (style()->subElementRect(subElement: QStyle::SE_ToolBarHandle, option: &opt, widget: this).contains(p: e->position().toPoint()))
1013 setCursor(Qt::SizeAllCursor);
1014 else
1015 unsetCursor();
1016#endif
1017 break;
1018 }
1019 case QEvent::MouseMove:
1020 if (d->mouseMoveEvent(event: static_cast<QMouseEvent*>(event)))
1021 return true;
1022 break;
1023 case QEvent::Leave:
1024 if (d->state != nullptr && d->state->dragging) {
1025#ifdef Q_OS_WIN
1026 // This is a workaround for losing the mouse on Vista.
1027 QPoint pos = QCursor::pos();
1028 QMouseEvent fake(QEvent::MouseMove, mapFromGlobal(pos), pos, Qt::NoButton,
1029 QGuiApplication::mouseButtons(), QGuiApplication::keyboardModifiers());
1030 d->mouseMoveEvent(&fake);
1031#endif
1032 } else {
1033 if (!d->layout->expanded)
1034 break;
1035
1036 QWidget *w = QApplication::activePopupWidget();
1037 if (waitForPopup(tb: this, popup: w)) {
1038 d->waitForPopupTimer.start(POPUP_TIMER_INTERVAL, obj: this);
1039 break;
1040 }
1041
1042 d->waitForPopupTimer.stop();
1043 d->layout->setExpanded(false);
1044 break;
1045 }
1046 default:
1047 break;
1048 }
1049 return QWidget::event(event);
1050}
