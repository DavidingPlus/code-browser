20/*!
21 \class QDialogButtonBox
22 \since 4.2
23 \brief The QDialogButtonBox class is a widget that presents buttons in a
24 layout that is appropriate to the current widget style.
25
26 \ingroup dialog-classes
27 \inmodule QtWidgets
28
29 Dialogs and message boxes typically present buttons in a layout that
30 conforms to the interface guidelines for that platform. Invariably,
31 different platforms have different layouts for their dialogs.
32 QDialogButtonBox allows a developer to add buttons to it and will
33 automatically use the appropriate layout for the user's desktop
34 environment.
35
36 Most buttons for a dialog follow certain roles. Such roles include:
37
38 \list
39 \li Accepting or rejecting the dialog.
40 \li Asking for help.
41 \li Performing actions on the dialog itself (such as resetting fields or
42 applying changes).
43 \endlist
44
45 There can also be alternate ways of dismissing the dialog which may cause
46 destructive results.
47
48 Most dialogs have buttons that can almost be considered standard (e.g.
49 \uicontrol OK and \uicontrol Cancel buttons). It is sometimes convenient to create these
50 buttons in a standard way.
51
52 There are a couple ways of using QDialogButtonBox. One ways is to create
53 the buttons (or button texts) yourself and add them to the button box,
54 specifying their role.
55
56 \snippet dialogs/extension/finddialog.cpp 1
57
58 Alternatively, QDialogButtonBox provides several standard buttons (e.g. OK, Cancel, Save)
59 that you can use. They exist as flags so you can OR them together in the constructor.
60
61 \snippet dialogs/tabdialog/tabdialog.cpp 2
62
63 You can mix and match normal buttons and standard buttons.
64
65 Currently the buttons are laid out in the following way if the button box is horizontal:
66 \table
67 \row \li \inlineimage buttonbox-gnomelayout-horizontal.png GnomeLayout Horizontal
68 \li Button box laid out in horizontal GnomeLayout
69 \row \li \inlineimage buttonbox-kdelayout-horizontal.png KdeLayout Horizontal
70 \li Button box laid out in horizontal KdeLayout
71 \row \li \inlineimage buttonbox-maclayout-horizontal.png MacLayout Horizontal
72 \li Button box laid out in horizontal MacLayout
73 \row \li \inlineimage buttonbox-winlayout-horizontal.png WinLayout Horizontal
74 \li Button box laid out in horizontal WinLayout
75 \endtable
76
77 The buttons are laid out the following way if the button box is vertical:
78
79 \table
80 \row \li GnomeLayout
81 \li KdeLayout
82 \li MacLayout
83 \li WinLayout
84 \row \li \inlineimage buttonbox-gnomelayout-vertical.png GnomeLayout Vertical
85 \li \inlineimage buttonbox-kdelayout-vertical.png KdeLayout Vertical
86 \li \inlineimage buttonbox-maclayout-vertical.png MacLayout Vertical
87 \li \inlineimage buttonbox-winlayout-vertical.png WinLayout Vertical
88 \endtable
89
90 Additionally, button boxes that contain only buttons with ActionRole or
91 HelpRole can be considered modeless and have an alternate look on \macos:
92
93 \table
94 \row \li modeless horizontal MacLayout
95 \li \inlineimage buttonbox-mac-modeless-horizontal.png Screenshot of modeless horizontal MacLayout
96 \row \li modeless vertical MacLayout
97 \li \inlineimage buttonbox-mac-modeless-vertical.png Screenshot of modeless vertical MacLayout
98 \endtable
99
100 When a button is clicked in the button box, the clicked() signal is emitted
101 for the actual button is that is pressed. For convenience, if the button
102 has an AcceptRole, RejectRole, or HelpRole, the accepted(), rejected(), or
103 helpRequested() signals are emitted respectively.
104
105 If you want a specific button to be default you need to call
106 QPushButton::setDefault() on it yourself. However, if there is no default
107 button set and to preserve which button is the default button across
108 platforms when using the QPushButton::autoDefault property, the first push
109 button with the accept role is made the default button when the
110 QDialogButtonBox is shown,
111
112 \sa QMessageBox, QPushButton, QDialog
113*/

149void QDialogButtonBoxPrivate::initLayout()
150{
151 Q_Q(QDialogButtonBox);
152 layoutPolicy = QDialogButtonBox::ButtonLayout(q->style()->styleHint(stylehint: QStyle::SH_DialogButtonLayout, opt: nullptr, widget: q));
153 bool createNewLayout = buttonLayout == nullptr
154 || (orientation == Qt::Horizontal && qobject_cast<QVBoxLayout *>(object: buttonLayout) != 0)
155 || (orientation == Qt::Vertical && qobject_cast<QHBoxLayout *>(object: buttonLayout) != 0);
156 if (createNewLayout) {
157 delete buttonLayout;
158 if (orientation == Qt::Horizontal)
159 buttonLayout = new QHBoxLayout(q);
160 else
161 buttonLayout = new QVBoxLayout(q);
162 }
163
164 int left, top, right, bottom;
165 setLayoutItemMargins(element: QStyle::SE_PushButtonLayoutItem);
166 getLayoutItemMargins(left: &left, top: &top, right: &right, bottom: &bottom);
167 buttonLayout->setContentsMargins(left: -left, top: -top, right: -right, bottom: -bottom);
168
169 if (!q->testAttribute(attribute: Qt::WA_WState_OwnSizePolicy)) {
170 QSizePolicy sp(QSizePolicy::Expanding, QSizePolicy::Fixed, QSizePolicy::ButtonBox);
171 if (orientation == Qt::Vertical)
172 sp.transpose();
173 q->setSizePolicy(sp);
174 q->setAttribute(Qt::WA_WState_OwnSizePolicy, on: false);
175 }
176}

185void QDialogButtonBoxPrivate::addButtonsToLayout(const QList<QAbstractButton *> &buttonList,
186 bool reverse)
187{
188 int start = reverse ? buttonList.size() - 1 : 0;
189 int end = reverse ? -1 : buttonList.size();
190 int step = reverse ? -1 : 1;
191
192 for (int i = start; i != end; i += step) {
193 QAbstractButton *button = buttonList.at(i);
194 buttonLayout->addWidget(button);
195 button->show();
196 }
197}

199void QDialogButtonBoxPrivate::layoutButtons()
200{
201 Q_Q(QDialogButtonBox);
202 const int MacGap = 36 - 8; // 8 is the default gap between a widget and a spacer item
203
204 for (int i = buttonLayout->count() - 1; i >= 0; --i) {
205 QLayoutItem *item = buttonLayout->takeAt(i);
206 if (QWidget *widget = item->widget())
207 widget->hide();
208 delete item;
209 }
210
211 int tmpPolicy = layoutPolicy;
212
213 static const int M = 5;
214 static const int ModalRoles[M] = { QPlatformDialogHelper::AcceptRole, QPlatformDialogHelper::RejectRole,
215 QPlatformDialogHelper::DestructiveRole, QPlatformDialogHelper::YesRole, QPlatformDialogHelper::NoRole };
216 if (tmpPolicy == QDialogButtonBox::MacLayout) {
217 bool hasModalButton = false;
218 for (int i = 0; i < M; ++i) {
219 if (!buttonLists[ModalRoles[i]].isEmpty()) {
220 hasModalButton = true;
221 break;
222 }
223 }
224 if (!hasModalButton)
225 tmpPolicy = 4; // Mac modeless
226 }
227
228 const int *currentLayout = QPlatformDialogHelper::buttonLayout(
229 orientation, policy: static_cast<QPlatformDialogHelper::ButtonLayout>(tmpPolicy));
230
231 if (center)
232 buttonLayout->addStretch();
233
234 const QList<QAbstractButton *> &acceptRoleList = buttonLists[QPlatformDialogHelper::AcceptRole];
235
236 while (*currentLayout != QPlatformDialogHelper::EOL) {
237 int role = (*currentLayout & ~QPlatformDialogHelper::Reverse);
238 bool reverse = (*currentLayout & QPlatformDialogHelper::Reverse);
239
240 switch (role) {
241 case QPlatformDialogHelper::Stretch:
242 if (!center)
243 buttonLayout->addStretch();
244 break;
245 case QPlatformDialogHelper::AcceptRole: {
246 if (acceptRoleList.isEmpty())
247 break;
248 // Only the first one
249 QAbstractButton *button = acceptRoleList.first();
250 buttonLayout->addWidget(button);
251 button->show();
252 }
253 break;
254 case QPlatformDialogHelper::AlternateRole:
255 if (acceptRoleList.size() > 1)
256 addButtonsToLayout(buttonList: acceptRoleList.mid(pos: 1), reverse);
257 break;
258 case QPlatformDialogHelper::DestructiveRole:
259 {
260 const QList<QAbstractButton *> &list = buttonLists[role];
261
262 /*
263 Mac: Insert a gap on the left of the destructive
264 buttons to ensure that they don't get too close to
265 the help and action buttons (but only if there are
266 some buttons to the left of the destructive buttons
267 (and the stretch, whence buttonLayout->count() > 1
268 and not 0)).
269 */
270 if (tmpPolicy == QDialogButtonBox::MacLayout
271 && !list.isEmpty() && buttonLayout->count() > 1)
272 buttonLayout->addSpacing(size: MacGap);
273
274 addButtonsToLayout(buttonList: list, reverse);
275
276 /*
277 Insert a gap between the destructive buttons and the
278 accept and reject buttons.
279 */
280 if (tmpPolicy == QDialogButtonBox::MacLayout && !list.isEmpty())
281 buttonLayout->addSpacing(size: MacGap);
282 }
283 break;
284 case QPlatformDialogHelper::RejectRole:
285 case QPlatformDialogHelper::ActionRole:
286 case QPlatformDialogHelper::HelpRole:
287 case QPlatformDialogHelper::YesRole:
288 case QPlatformDialogHelper::NoRole:
289 case QPlatformDialogHelper::ApplyRole:
290 case QPlatformDialogHelper::ResetRole:
291 addButtonsToLayout(buttonList: buttonLists[role], reverse);
292 }
293 ++currentLayout;
294 }
295
296 QWidget *lastWidget = nullptr;
297 q->setFocusProxy(nullptr);
298 for (int i = 0; i < buttonLayout->count(); ++i) {
299 QLayoutItem *item = buttonLayout->itemAt(i);
300 if (QWidget *widget = item->widget()) {
301 if (lastWidget)
302 QWidget::setTabOrder(lastWidget, widget);
303 else
304 q->setFocusProxy(widget);
305 lastWidget = widget;
306 }
307 }
308
309 if (center)
310 buttonLayout->addStretch();
311}

313QPushButton *QDialogButtonBoxPrivate::createButton(QDialogButtonBox::StandardButton sbutton,
314 bool doLayout)
315{
316 Q_Q(QDialogButtonBox);
317 int icon = 0;
318
319 switch (sbutton) {
320 case QDialogButtonBox::Ok:
321 icon = QStyle::SP_DialogOkButton;
322 break;
323 case QDialogButtonBox::Save:
324 icon = QStyle::SP_DialogSaveButton;
325 break;
326 case QDialogButtonBox::Open:
327 icon = QStyle::SP_DialogOpenButton;
328 break;
329 case QDialogButtonBox::Cancel:
330 icon = QStyle::SP_DialogCancelButton;
331 break;
332 case QDialogButtonBox::Close:
333 icon = QStyle::SP_DialogCloseButton;
334 break;
335 case QDialogButtonBox::Apply:
336 icon = QStyle::SP_DialogApplyButton;
337 break;
338 case QDialogButtonBox::Reset:
339 icon = QStyle::SP_DialogResetButton;
340 break;
341 case QDialogButtonBox::Help:
342 icon = QStyle::SP_DialogHelpButton;
343 break;
344 case QDialogButtonBox::Discard:
345 icon = QStyle::SP_DialogDiscardButton;
346 break;
347 case QDialogButtonBox::Yes:
348 icon = QStyle::SP_DialogYesButton;
349 break;
350 case QDialogButtonBox::No:
351 icon = QStyle::SP_DialogNoButton;
352 break;
353 case QDialogButtonBox::YesToAll:
354 icon = QStyle::SP_DialogYesToAllButton;
355 break;
356 case QDialogButtonBox::NoToAll:
357 icon = QStyle::SP_DialogNoToAllButton;
358 break;
359 case QDialogButtonBox::SaveAll:
360 icon = QStyle::SP_DialogSaveAllButton;
361 break;
362 case QDialogButtonBox::Abort:
363 icon = QStyle::SP_DialogAbortButton;
364 break;
365 case QDialogButtonBox::Retry:
366 icon = QStyle::SP_DialogRetryButton;
367 break;
368 case QDialogButtonBox::Ignore:
369 icon = QStyle::SP_DialogIgnoreButton;
370 break;
371 case QDialogButtonBox::RestoreDefaults:
372 icon = QStyle::SP_RestoreDefaultsButton;
373 break;
374 case QDialogButtonBox::NoButton:
375 return nullptr;
376 ;
377 }
378 QPushButton *button = new QPushButton(QGuiApplicationPrivate::platformTheme()->standardButtonText(button: sbutton), q);
379 QStyle *style = q->style();
380 if (style->styleHint(stylehint: QStyle::SH_DialogButtonBox_ButtonsHaveIcons, opt: nullptr, widget: q) && icon != 0)
381 button->setIcon(style->standardIcon(standardIcon: QStyle::StandardPixmap(icon), option: nullptr, widget: q));
382 if (style != QApplication::style()) // Propagate style
383 button->setStyle(style);
384 standardButtonHash.insert(key: button, value: sbutton);
385 QPlatformDialogHelper::ButtonRole role = QPlatformDialogHelper::buttonRole(button: static_cast<QPlatformDialogHelper::StandardButton>(sbutton));
386 if (Q_UNLIKELY(role == QPlatformDialogHelper::InvalidRole))
387 qWarning(msg: "QDialogButtonBox::createButton: Invalid ButtonRole, button not added");
388 else
389 addButton(button, role: static_cast<QDialogButtonBox::ButtonRole>(role), doLayout);
390#if QT_CONFIG(shortcut)
391 const QKeySequence standardShortcut = QGuiApplicationPrivate::platformTheme()->standardButtonShortcut(button: sbutton);
392 if (!standardShortcut.isEmpty())
393 button->setShortcut(standardShortcut);
394#endif
395 return button;
396}

398void QDialogButtonBoxPrivate::addButton(QAbstractButton *button, QDialogButtonBox::ButtonRole role,
399 bool doLayout)
400{
401 Q_Q(QDialogButtonBox);
402 QObject::connect(sender: button, SIGNAL(clicked()), receiver: q, SLOT(_q_handleButtonClicked()));
403 QObject::connect(sender: button, SIGNAL(destroyed()), receiver: q, SLOT(_q_handleButtonDestroyed()));
404 buttonLists[role].append(t: button);
405 if (doLayout)
406 layoutButtons();
407}

409void QDialogButtonBoxPrivate::createStandardButtons(QDialogButtonBox::StandardButtons buttons)
410{
411 uint i = QDialogButtonBox::FirstButton;
412 while (i <= QDialogButtonBox::LastButton) {
413 if (i & buttons) {
414 createButton(sbutton: QDialogButtonBox::StandardButton(i), doLayout: false);
415 }
416 i = i << 1;
417 }
418 layoutButtons();
419}

421void QDialogButtonBoxPrivate::retranslateStrings()
422{
423 typedef QHash<QPushButton *, QDialogButtonBox::StandardButton>::iterator Iterator;
424
425 const Iterator end = standardButtonHash.end();
426 for (Iterator it = standardButtonHash.begin(); it != end; ++it) {
427 const QString text = QGuiApplicationPrivate::platformTheme()->standardButtonText(button: it.value());
428 if (!text.isEmpty())
429 it.key()->setText(text);
430 }
431}

487/*!
488 \enum QDialogButtonBox::ButtonRole
489
490//! [buttonrole-enum]
491 This enum describes the roles that can be used to describe buttons in
492 the button box. Combinations of these roles are as flags used to
493 describe different aspects of their behavior.
494
495 \value InvalidRole The button is invalid.
496 \value AcceptRole Clicking the button causes the dialog to be accepted
497 (e.g. OK).
498 \value RejectRole Clicking the button causes the dialog to be rejected
499 (e.g. Cancel).
500 \value DestructiveRole Clicking the button causes a destructive change
501 (e.g. for Discarding Changes) and closes the dialog.
502 \value ActionRole Clicking the button causes changes to the elements within
503 the dialog.
504 \value HelpRole The button can be clicked to request help.
505 \value YesRole The button is a "Yes"-like button.
506 \value NoRole The button is a "No"-like button.
507 \value ApplyRole The button applies current changes.
508 \value ResetRole The button resets the dialog's fields to default values.
509
510 \omitvalue NRoles
511
512 \sa StandardButton
513//! [buttonrole-enum]
514*/
515

626/*!
627 Clears the button box, deleting all buttons within it.
628
629 \sa removeButton(), addButton()
630*/
631void QDialogButtonBox::clear()
632{
633 Q_D(QDialogButtonBox);
634 // Remove the created standard buttons, they should be in the other lists, which will
635 // do the deletion
636 d->standardButtonHash.clear();
637 for (int i = 0; i < NRoles; ++i) {
638 QList<QAbstractButton *> &list = d->buttonLists[i];
639 while (list.size()) {
640 QAbstractButton *button = list.takeAt(i: 0);
641 QObject::disconnect(sender: button, SIGNAL(destroyed()), receiver: this, SLOT(_q_handleButtonDestroyed()));
642 delete button;
643 }
644 }
645}

664/*!
665 Returns the button role for the specified \a button. This function returns
666 \l InvalidRole if \a button is \nullptr or has not been added to the button box.

669*/
670QDialogButtonBox::ButtonRole QDialogButtonBox::buttonRole(QAbstractButton *button) const

676 if (list.at(i: j) == button)
677 return ButtonRole(i);
678 }
679 }
680 return InvalidRole;

683/*!
684 Removes \a button from the button box without deleting it and sets its parent to zero.
685
686 \sa clear(), buttons(), addButton()
687*/
688void QDialogButtonBox::removeButton(QAbstractButton *button)
689{
690 Q_D(QDialogButtonBox);
691
692 if (!button)
693 return;
694
695 // Remove it from the standard button hash first and then from the roles
696 d->standardButtonHash.remove(key: reinterpret_cast<QPushButton *>(button));
697 for (int i = 0; i < NRoles; ++i) {
698 QList<QAbstractButton *> &list = d->buttonLists[i];
699 for (int j = 0; j < list.size(); ++j) {
700 if (list.at(i: j) == button) {
701 list.takeAt(i: j);
702 if (!d->internalRemove) {
703 disconnect(sender: button, SIGNAL(clicked()), receiver: this, SLOT(_q_handleButtonClicked()));
704 disconnect(sender: button, SIGNAL(destroyed()), receiver: this, SLOT(_q_handleButtonDestroyed()));
705 }
706 break;
707 }
708 }
709 }
710 if (!d->internalRemove)
711 button->setParent(nullptr);
712}

714/*!
715 Adds the given \a button to the button box with the specified \a role.
716 If the role is invalid, the button is not added.
717
718 If the button has already been added, it is removed and added again with the
719 new role.
720
721 \note The button box takes ownership of the button.
722
723 \sa removeButton(), clear()

823void QDialogButtonBoxPrivate::_q_handleButtonClicked()
824{
825 Q_Q(QDialogButtonBox);
826 if (QAbstractButton *button = qobject_cast<QAbstractButton *>(object: q->sender())) {
827 // Can't fetch this *after* emitting clicked, as clicked may destroy the button
828 // or change its role. Now changing the role is not possible yet, but arguably
829 // both clicked and accepted/rejected/etc. should be emitted "atomically"
830 // depending on whatever role the button had at the time of the click.
831 const QDialogButtonBox::ButtonRole buttonRole = q->buttonRole(button);
832 QPointer<QDialogButtonBox> guard(q);
833
834 emit q->clicked(button);
835
836 if (!guard)
837 return;
838
839 switch (QPlatformDialogHelper::ButtonRole(buttonRole)) {
840 case QPlatformDialogHelper::AcceptRole:
841 case QPlatformDialogHelper::YesRole:
842 emit q->accepted();
843 break;
844 case QPlatformDialogHelper::RejectRole:
845 case QPlatformDialogHelper::NoRole:
846 emit q->rejected();
847 break;
848 case QPlatformDialogHelper::HelpRole:
849 emit q->helpRequested();
850 break;
851 default:
852 break;
853 }
854 }
855}

895void QDialogButtonBox::changeEvent(QEvent *event)
896{
897 typedef QHash<QPushButton *, QDialogButtonBox::StandardButton> StandardButtonHash;
898
899 Q_D(QDialogButtonBox);
900 switch (event->type()) {
901 case QEvent::StyleChange: // Propagate style
902 if (!d->standardButtonHash.empty()) {
903 QStyle *newStyle = style();
904 const StandardButtonHash::iterator end = d->standardButtonHash.end();
905 for (StandardButtonHash::iterator it = d->standardButtonHash.begin(); it != end; ++it)
906 it.key()->setStyle(newStyle);
907 }
908#ifdef Q_OS_MAC
909 Q_FALLTHROUGH();
910 case QEvent::MacSizeChange:
911#endif
912 d->resetLayout();
913 QWidget::changeEvent(event);
914 break;
915 default:
916 QWidget::changeEvent(event);
917 break;
918 }
919}

921/*!
922 \reimp
923*/
924bool QDialogButtonBox::event(QEvent *event)
925{
926 Q_D(QDialogButtonBox);
927 if (event->type() == QEvent::Show) {
928 QList<QAbstractButton *> acceptRoleList = d->buttonLists[AcceptRole];
929 QPushButton *firstAcceptButton = acceptRoleList.isEmpty() ? 0 : qobject_cast<QPushButton *>(object: acceptRoleList.at(i: 0));
930 bool hasDefault = false;
931 QWidget *dialog = nullptr;
932 QWidget *p = this;
933 while (p && !p->isWindow()) {
934 p = p->parentWidget();
935 if ((dialog = qobject_cast<QDialog *>(object: p)))
936 break;
937 }
938
939 const auto pbs = (dialog ? dialog : this)->findChildren<QPushButton *>();
940 for (QPushButton *pb : pbs) {
941 if (pb->isDefault() && pb != firstAcceptButton) {
942 hasDefault = true;
943 break;
944 }
945 }
946 if (!hasDefault && firstAcceptButton)
947 firstAcceptButton->setDefault(true);
948 }else if (event->type() == QEvent::LanguageChange) {
949 d->retranslateStrings();
950 }
951 return QWidget::event(event);
952}
