89static QFontDatabase::WritingSystem writingSystemFromLocale()
90{
91 QStringList uiLanguages = QLocale::system().uiLanguages();
92 QLocale::Script script;
93 if (!uiLanguages.isEmpty())
94 script = QLocale(uiLanguages.at(i: 0)).script();
95 else
96 script = QLocale::system().script();
97
98 return writingSystemFromScript(script);
99}

101static QFontDatabase::WritingSystem writingSystemForFont(const QFont &font, bool *hasLatin)
102{
103 QList<QFontDatabase::WritingSystem> writingSystems = QFontDatabase::writingSystems(family: font.families().first());
104// qDebug() << font.families().first() << writingSystems;
105
106 // this just confuses the algorithm below. Vietnamese is Latin with lots of special chars
107 writingSystems.removeOne(t: QFontDatabase::Vietnamese);
108 *hasLatin = writingSystems.removeOne(t: QFontDatabase::Latin);
109
110 if (writingSystems.isEmpty())
111 return QFontDatabase::Any;
112
113 QFontDatabase::WritingSystem system = writingSystemFromLocale();
114
115 if (writingSystems.contains(t: system))
116 return system;
117
118 if (system == QFontDatabase::TraditionalChinese
119 && writingSystems.contains(t: QFontDatabase::SimplifiedChinese)) {
120 return QFontDatabase::SimplifiedChinese;
121 }
122
123 if (system == QFontDatabase::SimplifiedChinese
124 && writingSystems.contains(t: QFontDatabase::TraditionalChinese)) {
125 return QFontDatabase::TraditionalChinese;
126 }
127
128 system = writingSystems.constLast();
129
130 if (!*hasLatin) {
131 // we need to show something
132 return system;
133 }
134
135 if (writingSystems.size() == 1 && system > QFontDatabase::Cyrillic)
136 return system;
137
138 if (writingSystems.size() <= 2 && system > QFontDatabase::Armenian && system < QFontDatabase::Vietnamese)
139 return system;
140
141 if (writingSystems.size() <= 5 && system >= QFontDatabase::SimplifiedChinese && system <= QFontDatabase::Korean)
142 return system;
143
144 return QFontDatabase::Any;
145}

193void QFontFamilyDelegate::paint(QPainter *painter,
194 const QStyleOptionViewItem &option,
195 const QModelIndex &index) const
196{
197 QString text = index.data(arole: Qt::DisplayRole).toString();
198 QFont font(option.font);
199 font.setPointSize(QFontInfo(font).pointSize() * 3 / 2);
200 QFont font2 = font;
201 font2.setFamilies(QStringList{text});
202
203 bool hasLatin;
204 QFontDatabase::WritingSystem system = writingSystemForFont(font: font2, hasLatin: &hasLatin);
205 if (hasLatin)
206 font = font2;
207
208 font = comboPrivate->displayFontForFontFamily.value(key: text, defaultValue: font);
209
210 QRect r = option.rect;
211
212 if (option.state & QStyle::State_Selected) {
213 painter->save();
214 painter->setBrush(option.palette.highlight());
215 painter->setPen(Qt::NoPen);
216 painter->drawRect(r: option.rect);
217 painter->setPen(QPen(option.palette.highlightedText(), 0));
218 }
219
220 const QIcon *icon = &bitmap;
221 if (QFontDatabase::isSmoothlyScalable(family: text)) {
222 icon = &truetype;
223 }
224 const QSize actualSize = icon->actualSize(size: r.size());
225 const QRect iconRect = QStyle::alignedRect(direction: option.direction, alignment: option.displayAlignment,
226 size: actualSize, rectangle: r);
227 icon->paint(painter, rect: iconRect, alignment: Qt::AlignLeft|Qt::AlignVCenter);
228 if (option.direction == Qt::RightToLeft)
229 r.setRight(r.right() - actualSize.width() - 4);
230 else
231 r.setLeft(r.left() + actualSize.width() + 4);
232
233 QFont old = painter->font();
234 painter->setFont(font);
235
236 const Qt::Alignment textAlign = QStyle::visualAlignment(direction: option.direction, alignment: option.displayAlignment);
237 // If the ascent of the font is larger than the height of the rect,
238 // we will clip the text, so it's better to align the tight bounding rect in this case
239 // This is specifically for fonts where the ascent is very large compared to
240 // the descent, like certain of the Stix family.
241 QFontMetricsF fontMetrics(font);
242 if (fontMetrics.ascent() > r.height()) {
243 QRectF tbr = fontMetrics.tightBoundingRect(text);
244 QRect textRect(r);
245 textRect.setHeight(textRect.height() + (r.height() - tbr.height()));
246 painter->drawText(r: textRect, flags: Qt::AlignBottom|Qt::TextSingleLine|textAlign, text);
247 } else {
248 painter->drawText(r, flags: Qt::AlignVCenter|Qt::TextSingleLine|textAlign, text);
249 }
250
251 if (writingSystem != QFontDatabase::Any)
252 system = writingSystem;
253
254 const QString sampleText = comboPrivate->sampleTextForFontFamily.value(key: text, defaultValue: comboPrivate->sampleTextForWritingSystem.value(key: system));
255 if (system != QFontDatabase::Any || !sampleText.isEmpty()) {
256 int w = painter->fontMetrics().horizontalAdvance(text + " "_L1);
257 painter->setFont(font2);
258 const QString sample = !sampleText.isEmpty() ? sampleText : QFontDatabase::writingSystemSample(writingSystem: system);
259 if (option.direction == Qt::RightToLeft)
260 r.setRight(r.right() - w);
261 else
262 r.setLeft(r.left() + w);
263 painter->drawText(r, flags: Qt::AlignVCenter|Qt::TextSingleLine|textAlign, text: sample);
264 }
265 painter->setFont(old);
266
267 if (option.state & QStyle::State_Selected)
268 painter->restore();
269
270}

284void QFontComboBoxPrivate::_q_updateModel()
285{
286 Q_Q(QFontComboBox);
287
288 if (QCoreApplication::closingDown())
289 return;
290
291 const int scalableMask = (QFontComboBox::ScalableFonts | QFontComboBox::NonScalableFonts);
292 const int spacingMask = (QFontComboBox::ProportionalFonts | QFontComboBox::MonospacedFonts);
293
294 QStringListModel *m = qobject_cast<QStringListModel *>(object: q->model());
295 if (!m)
296 return;
297 QFontFamilyDelegate *delegate = qobject_cast<QFontFamilyDelegate *>(object: q->view()->itemDelegate());
298 QFontDatabase::WritingSystem system = delegate ? delegate->writingSystem : QFontDatabase::Any;
299
300 QStringList list = QFontDatabase::families(writingSystem: system);
301 QStringList result;
302
303 int offset = 0;
304 QFontInfo fi(currentFont);
305
306 for (int i = 0; i < list.size(); ++i) {
307 if (QFontDatabase::isPrivateFamily(family: list.at(i)))
308 continue;
309
310 if ((filters & scalableMask) && (filters & scalableMask) != scalableMask) {
311 if (bool(filters & QFontComboBox::ScalableFonts) != QFontDatabase::isSmoothlyScalable(family: list.at(i)))
312 continue;
313 }
314 if ((filters & spacingMask) && (filters & spacingMask) != spacingMask) {
315 if (bool(filters & QFontComboBox::MonospacedFonts) != QFontDatabase::isFixedPitch(family: list.at(i)))
316 continue;
317 }
318 result += list.at(i);
319 if (list.at(i) == fi.family() || list.at(i).startsWith(s: fi.family() + " ["_L1))
320 offset = result.size() - 1;
321 }
322 list = result;
323
324 //we need to block the signals so that the model doesn't emit reset
325 //this prevents the current index from changing
326 //it will be updated just after this
327 ///TODO: we should finda way to avoid blocking signals and have a real update of the model
328 {
329 const QSignalBlocker blocker(m);
330 m->setStringList(list);
331 }
332
333 if (list.isEmpty()) {
334 if (currentFont != QFont()) {
335 currentFont = QFont();
336 emit q->currentFontChanged(f: currentFont);
337 }
338 } else {
339 q->setCurrentIndex(offset);
340 }
341}

344void QFontComboBoxPrivate::_q_currentChanged(const QString &text)
345{
346 Q_Q(QFontComboBox);
347 QStringList families = currentFont.families();
348 if (families.isEmpty() || families.first() != text) {
349 currentFont.setFamilies(QStringList{text});
350 emit q->currentFontChanged(f: currentFont);
351 }

354/*!
355 \class QFontComboBox
356 \brief The QFontComboBox widget is a combobox that lets the user
357 select a font family.
358
359 \since 4.2
360 \ingroup basicwidgets
361 \inmodule QtWidgets
362
363 The combobox is populated with an alphabetized list of font
364 family names, such as Arial, Helvetica, and Times New Roman.
365 Family names are displayed using the actual font when possible.
366 For fonts such as Symbol, where the name is not representable in
367 the font itself, a sample of the font is displayed next to the
368 family name.
369
370 QFontComboBox is often used in toolbars, in conjunction with a
371 QComboBox for controlling the font size and two \l{QToolButton}s
372 for bold and italic.
373
374 When the user selects a new font, the currentFontChanged() signal
375 is emitted in addition to currentIndexChanged().
376
377 Call setWritingSystem() to tell QFontComboBox to show only fonts
378 that support a given writing system, and setFontFilters() to
379 filter out certain types of fonts as e.g. non scalable fonts or
380 monospaced fonts.
381
382 \image windowsvista-fontcombobox.png Screenshot of QFontComboBox on Windows Vista
383
384 \sa QComboBox, QFont, QFontInfo, QFontMetrics, QFontDatabase, {Character Map Example}
385*/
386
387/*!
388 Constructs a font combobox with the given \a parent.
389*/
390QFontComboBox::QFontComboBox(QWidget *parent)
391 : QComboBox(*new QFontComboBoxPrivate, parent)
392{
393 Q_D(QFontComboBox);
394 d->currentFont = font();
395 setEditable(true);
396
397 QStringListModel *m = new QStringListModel(this);
398 setModel(m);
399 setItemDelegate(new QFontFamilyDelegate(this, d));
400 QListView *lview = qobject_cast<QListView*>(object: view());
401 if (lview)
402 lview->setUniformItemSizes(true);
403 setWritingSystem(QFontDatabase::Any);
404
405 connect(sender: this, SIGNAL(currentTextChanged(QString)),
406 receiver: this, SLOT(_q_currentChanged(QString)));
407
408 connect(qApp, SIGNAL(fontDatabaseChanged()),
409 receiver: this, SLOT(_q_updateModel()));
410}

495 Q_D(QFontComboBox);

497 d->currentFont = font;
