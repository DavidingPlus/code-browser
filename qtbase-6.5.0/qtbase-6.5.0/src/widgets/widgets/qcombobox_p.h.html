48class QComboBoxListView : public QListView
49{
50 Q_OBJECT
51public:
52 QComboBoxListView(QComboBox *cmb = nullptr) : combo(cmb) {}
53
54protected:
55 void resizeEvent(QResizeEvent *event) override
56 {
57 resizeContents(width: viewport()->width(), height: contentsSize().height());
58 QListView::resizeEvent(e: event);
59 }
60
61 void initViewItemOption(QStyleOptionViewItem *option) const override
62 {
63 QListView::initViewItemOption(option);
64 option->showDecorationSelected = true;
65 if (combo)
66 option->font = combo->font();
67 }
68
69 void paintEvent(QPaintEvent *e) override
70 {
71 if (combo) {
72 QStyleOptionComboBox opt;
73 opt.initFrom(w: combo);
74 opt.editable = combo->isEditable();
75 if (combo->style()->styleHint(stylehint: QStyle::SH_ComboBox_Popup, opt: &opt, widget: combo)) {
76 //we paint the empty menu area to avoid having blank space that can happen when scrolling
77 QStyleOptionMenuItem menuOpt;
78 menuOpt.initFrom(w: this);
79 menuOpt.palette = palette();
80 menuOpt.state = QStyle::State_None;
81 menuOpt.checkType = QStyleOptionMenuItem::NotCheckable;
82 menuOpt.menuRect = e->rect();
83 menuOpt.maxIconWidth = 0;
84 menuOpt.reservedShortcutWidth = 0;
85 QPainter p(viewport());
86 combo->style()->drawControl(element: QStyle::CE_MenuEmptyArea, opt: &menuOpt, p: &p, w: this);
87 }
88 }
89 QListView::paintEvent(e);
90 }
91
92private:
93 QComboBox *combo;
94};

96class Q_AUTOTEST_EXPORT QComboBoxPrivateScroller : public QWidget
97{
98 Q_OBJECT
99
100public:
101 QComboBoxPrivateScroller(QAbstractSlider::SliderAction action, QWidget *parent)
102 : QWidget(parent), sliderAction(action)
103 {
104 setSizePolicy(hor: QSizePolicy::Minimum, ver: QSizePolicy::Fixed);
105 setAttribute(Qt::WA_NoMousePropagation);
106 }
107 QSize sizeHint() const override {
108 return QSize(20, style()->pixelMetric(metric: QStyle::PM_MenuScrollerHeight, option: nullptr, widget: this));
109 }
110
111protected:
112 inline void stopTimer() {
113 timer.stop();
114 }
115
116 inline void startTimer() {
117 timer.start(msec: 100, obj: this);
118 fast = false;
119 }
120
121 void enterEvent(QEnterEvent *) override {
122 startTimer();
123 }
124
125 void leaveEvent(QEvent *) override {
126 stopTimer();
127 }
128 void timerEvent(QTimerEvent *e) override {
129 if (e->timerId() == timer.timerId()) {
130 emit doScroll(action: sliderAction);
131 if (fast) {
132 emit doScroll(action: sliderAction);
133 emit doScroll(action: sliderAction);
134 }
135 }
136 }
137 void hideEvent(QHideEvent *) override {
138 stopTimer();
139 }
140
141 void mouseMoveEvent(QMouseEvent *e) override
142 {
143 // Enable fast scrolling if the cursor is directly above or below the popup.
144 const int mouseX = e->position().toPoint().x();
145 const int mouseY = e->position().toPoint().y();
146 const bool horizontallyInside = pos().x() < mouseX && mouseX < rect().right() + 1;
147 const bool verticallyOutside = (sliderAction == QAbstractSlider::SliderSingleStepAdd) ?
148 rect().bottom() + 1 < mouseY : mouseY < pos().y();
149
150 fast = horizontallyInside && verticallyOutside;
151 }
152
153 void paintEvent(QPaintEvent *) override {
154 QPainter p(this);
155 QStyleOptionMenuItem menuOpt;
156 menuOpt.initFrom(w: this);
157 menuOpt.checkType = QStyleOptionMenuItem::NotCheckable;
158 menuOpt.menuRect = rect();
159 menuOpt.maxIconWidth = 0;
160 menuOpt.reservedShortcutWidth = 0;
161 menuOpt.menuItemType = QStyleOptionMenuItem::Scroller;
162 if (sliderAction == QAbstractSlider::SliderSingleStepAdd)
163 menuOpt.state |= QStyle::State_DownArrow;
164 p.eraseRect(rect: rect());
165 style()->drawControl(element: QStyle::CE_MenuScroller, opt: &menuOpt, p: &p);
166 }
167
168Q_SIGNALS:
169 void doScroll(int action);
170
171private:
172 QAbstractSlider::SliderAction sliderAction;
173 QBasicTimer timer;
174 bool fast = false;
175};

261class Q_AUTOTEST_EXPORT QComboBoxDelegate : public QStyledItemDelegate
262{
263 Q_OBJECT
264public:
265 QComboBoxDelegate(QObject *parent, QComboBox *cmb)
266 : QStyledItemDelegate(parent), mCombo(cmb)
267 {}
268
269 static bool isSeparator(const QModelIndex &index) {
270 return index.data(arole: Qt::AccessibleDescriptionRole).toString()
271 == QLatin1StringView("separator");
272 }
273 static void setSeparator(QAbstractItemModel *model, const QModelIndex &index) {
274 model->setData(index, value: QString::fromLatin1(ba: "separator"), role: Qt::AccessibleDescriptionRole);
275 if (QStandardItemModel *m = qobject_cast<QStandardItemModel*>(object: model))
276 if (QStandardItem *item = m->itemFromIndex(index))
277 item->setFlags(item->flags() & ~(Qt::ItemIsSelectable|Qt::ItemIsEnabled));
278 }
279
280protected:
281 void paint(QPainter *painter,
282 const QStyleOptionViewItem &option,
283 const QModelIndex &index) const override {
284 if (isSeparator(index)) {
285 QRect rect = option.rect;
286 if (const QAbstractItemView *view = qobject_cast<const QAbstractItemView*>(object: option.widget))
287 rect.setWidth(view->viewport()->width());
288 QStyleOption opt;
289 opt.rect = rect;
290 mCombo->style()->drawPrimitive(pe: QStyle::PE_IndicatorToolBarSeparator, opt: &opt, p: painter, w: mCombo);
291 } else {
292 QStyledItemDelegate::paint(painter, option, index);
293 }
294 }
295
296 QSize sizeHint(const QStyleOptionViewItem &option,
297 const QModelIndex &index) const override {
298 if (isSeparator(index)) {
299 int pm = mCombo->style()->pixelMetric(metric: QStyle::PM_DefaultFrameWidth, option: nullptr, widget: mCombo);
300 return QSize(pm, pm);
301 }
302 return QStyledItemDelegate::sizeHint(option, index);
303 }
304private:
305 QComboBox *mCombo;
306};
