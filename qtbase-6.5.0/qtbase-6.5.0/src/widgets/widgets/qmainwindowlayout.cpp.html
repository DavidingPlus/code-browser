153// QDockWidgetGroupWindow is the floating window containing several QDockWidgets floating together.
154// (QMainWindow::GroupedDragging feature)
155// QDockWidgetGroupLayout is the layout of that window and use a QDockAreaLayoutInfo to layout
156// the QDockWidgets inside it.
157// If there is only one QDockWidgets, or all QDockWidgets are tabbed together, it is equivalent
158// of a floating QDockWidget (the title of the QDockWidget is the title of the window). But if there
159// are nested QDockWidget, an additional title bar is there.
160#if QT_CONFIG(dockwidget)
161class QDockWidgetGroupLayout : public QLayout,
162 public QMainWindowLayoutSeparatorHelper<QDockWidgetGroupLayout>
163{
164 QWidgetResizeHandler *resizer;
165public:
166 QDockWidgetGroupLayout(QDockWidgetGroupWindow* parent) : QLayout(parent) {
167 setSizeConstraint(QLayout::SetMinAndMaxSize);
168 resizer = new QWidgetResizeHandler(parent);
169 }
170 ~QDockWidgetGroupLayout() {
171 layoutState.deleteAllLayoutItems();
172 }
173
174 void addItem(QLayoutItem*) override { Q_UNREACHABLE(); }
175 int count() const override { return 0; }
176 QLayoutItem* itemAt(int index) const override
177 {
178 int x = 0;
179 return layoutState.itemAt(x: &x, index);
180 }
181 QLayoutItem* takeAt(int index) override
182 {
183 int x = 0;
184 QLayoutItem *ret = layoutState.takeAt(x: &x, index);
185 if (savedState.rect.isValid() && ret->widget()) {
186 // we need to remove the item also from the saved state to prevent crash
187 QList<int> path = savedState.indexOf(widget: ret->widget());
188 if (!path.isEmpty())
189 savedState.remove(path);
190 // Also, the item may be contained several times as a gap item.
191 path = layoutState.indexOf(widget: ret->widget());
192 if (!path.isEmpty())
193 layoutState.remove(path);
194 }
195 return ret;
196 }
197 QSize sizeHint() const override
198 {
199 int fw = frameWidth();
200 return layoutState.sizeHint() + QSize(fw, fw);
201 }
202 QSize minimumSize() const override
203 {
204 int fw = frameWidth();
205 return layoutState.minimumSize() + QSize(fw, fw);
206 }
207 QSize maximumSize() const override
208 {
209 int fw = frameWidth();
210 return layoutState.maximumSize() + QSize(fw, fw);
211 }
212 void setGeometry(const QRect&r) override
213 {
214 groupWindow()->destroyOrHideIfEmpty();
215 QDockAreaLayoutInfo *li = dockAreaLayoutInfo();
216 if (li->isEmpty())
217 return;
218 int fw = frameWidth();
219#if QT_CONFIG(tabbar)
220 li->reparentWidgets(p: parentWidget());
221#endif
222 li->rect = r.adjusted(xp1: fw, yp1: fw, xp2: -fw, yp2: -fw);
223 li->fitItems();
224 li->apply(animate: false);
225 if (savedState.rect.isValid())
226 savedState.rect = li->rect;
227 resizer->setEnabled(!nativeWindowDeco());
228 }
229
230 QDockAreaLayoutInfo *dockAreaLayoutInfo() { return &layoutState; }
231
232 bool nativeWindowDeco() const
233 {
234 return groupWindow()->hasNativeDecos();
235 }
236
237 int frameWidth() const
238 {
239 return nativeWindowDeco() ? 0 :
240 parentWidget()->style()->pixelMetric(metric: QStyle::PM_DockWidgetFrameWidth, option: nullptr, widget: parentWidget());
241 }
242
243 QDockWidgetGroupWindow *groupWindow() const
244 {
245 return static_cast<QDockWidgetGroupWindow *>(parent());
246 }

248 QDockAreaLayoutInfo layoutState;
249 QDockAreaLayoutInfo savedState;

252bool QDockWidgetGroupWindow::event(QEvent *e)
253{
254 auto lay = static_cast<QDockWidgetGroupLayout *>(layout());
255 if (lay && lay->windowEvent(e))
256 return true;
257
258 switch (e->type()) {
259 case QEvent::Close:
260#if QT_CONFIG(tabbar)
261 // Forward the close to the QDockWidget just as if its close button was pressed
262 if (QDockWidget *dw = activeTabbedDockWidget()) {
263 dw->close();
264 adjustFlags();
265 }
266#endif
267 return true;
268 case QEvent::Move:
269#if QT_CONFIG(tabbar)
270 // Let QDockWidgetPrivate::moseEvent handle the dragging
271 if (QDockWidget *dw = activeTabbedDockWidget())
272 static_cast<QDockWidgetPrivate *>(QObjectPrivate::get(o: dw))->moveEvent(event: static_cast<QMoveEvent*>(e));
273#endif
274 return true;
275 case QEvent::NonClientAreaMouseMove:
276 case QEvent::NonClientAreaMouseButtonPress:
277 case QEvent::NonClientAreaMouseButtonRelease:
278 case QEvent::NonClientAreaMouseButtonDblClick:
279#if QT_CONFIG(tabbar)
280 // Let the QDockWidgetPrivate of the currently visible dock widget handle the drag and drop
281 if (QDockWidget *dw = activeTabbedDockWidget())
282 static_cast<QDockWidgetPrivate *>(QObjectPrivate::get(o: dw))->nonClientAreaMouseEvent(event: static_cast<QMouseEvent*>(e));
283#endif
284 return true;
285 case QEvent::ChildAdded:
286 if (qobject_cast<QDockWidget *>(object: static_cast<QChildEvent*>(e)->child()))
287 adjustFlags();
288 break;
289 case QEvent::LayoutRequest:
290 // We might need to show the widget again
291 destroyOrHideIfEmpty();
292 break;
293 case QEvent::Resize:
294 updateCurrentGapRect();
295 emit resized();
296 default:
297 break;
298 }
299 return QWidget::event(event: e);
300}

320#if QT_CONFIG(tabbar)
321/*! \internal
322 If this is a floating tab bar returns the currently the QDockWidgetGroupWindow that contains
323 tab, otherwise, return nullptr;
324 \note: if there is only one QDockWidget, it's still considered as a floating tab
325 */
326const QDockAreaLayoutInfo *QDockWidgetGroupWindow::tabLayoutInfo() const
327{
328 const QDockAreaLayoutInfo *info = layoutInfo();
329 while (info && !info->tabbed) {
330 // There should be only one tabbed subinfo otherwise we are not a floating tab but a real
331 // window
332 const QDockAreaLayoutInfo *next = nullptr;
333 bool isSingle = false;
334 for (const auto &item : info->item_list) {
335 if (item.skip() || (item.flags & QDockAreaLayoutItem::GapItem))
336 continue;
337 if (next || isSingle) // Two visible things
338 return nullptr;
339 if (item.subinfo)
340 next = item.subinfo;
341 else if (item.widgetItem)
342 isSingle = true;
343 }
344 if (isSingle)
345 return info;
346 info = next;
347 }
348 return info;
349}

351/*! \internal
352 If this is a floating tab bar returns the currently active QDockWidget, otherwise nullptr
353 */
354QDockWidget *QDockWidgetGroupWindow::activeTabbedDockWidget() const
355{
356 QDockWidget *dw = nullptr;
357 const QDockAreaLayoutInfo *info = tabLayoutInfo();
358 if (!info)
359 return nullptr;
360 if (info->tabBar && info->tabBar->currentIndex() >= 0) {
361 int i = info->tabIndexToListIndex(info->tabBar->currentIndex());
362 if (i >= 0) {
363 const QDockAreaLayoutItem &item = info->item_list.at(i);
364 if (item.widgetItem)
365 dw = qobject_cast<QDockWidget *>(object: item.widgetItem->widget());
366 }
367 }
368 if (!dw) {
369 for (int i = 0; !dw && i < info->item_list.size(); ++i) {
370 const QDockAreaLayoutItem &item = info->item_list.at(i);
371 if (item.skip())
372 continue;
373 if (!item.widgetItem)
374 continue;
375 dw = qobject_cast<QDockWidget *>(object: item.widgetItem->widget());
376 }
377 }
378 return dw;
379}

382/*! \internal
383 Destroy or hide this window if there is no more QDockWidget in it.
384 Otherwise make sure it is shown.
385 */
386void QDockWidgetGroupWindow::destroyOrHideIfEmpty()
387{
388 const QDockAreaLayoutInfo *info = layoutInfo();
389 if (!info->isEmpty()) {
390 show(); // It might have been hidden,
391 return;
392 }
393 // There might still be placeholders
394 if (!info->item_list.isEmpty()) {
395 hide();
396 return;
397 }
398
399 // Make sure to reparent the possibly floating or hidden QDockWidgets to the parent
400 const auto dockWidgets = findChildren<QDockWidget *>(options: Qt::FindDirectChildrenOnly);
401 for (QDockWidget *dw : dockWidgets) {
402 const bool wasFloating = dw->isFloating();
403 const bool wasHidden = dw->isHidden();
404 dw->setParent(parentWidget());
405 qCDebug(lcQpaDockWidgets) << "Reparented:" << dw << "to" << parentWidget() << "by" << this;
406 if (wasFloating) {
407 dw->setFloating(true);
408 } else {
409 // maybe it was hidden, we still have to put it back in the main layout.
410 QMainWindowLayout *ml =
411 qt_mainwindow_layout(window: static_cast<QMainWindow *>(parentWidget()));
412 Qt::DockWidgetArea area = ml->dockWidgetArea(widget: this);
413 if (area == Qt::NoDockWidgetArea)
414 area = Qt::LeftDockWidgetArea; // FIXME: DockWidget doesn't save original docking area
415 static_cast<QMainWindow *>(parentWidget())->addDockWidget(area, dockwidget: dw);
416 qCDebug(lcQpaDockWidgets) << "Redocked to Mainwindow:" << area << dw << "by" << this;
417 }
418 if (!wasHidden)
419 dw->show();
420 }
421#if QT_CONFIG(tabbar)
422 const auto tabBars = findChildren<QTabBar *>(options: Qt::FindDirectChildrenOnly);
423 for (QTabBar *tb : tabBars)
424 tb->setParent(parentWidget());
425#endif
426 deleteLater();
427}

429/*! \internal
430 Sets the flags of this window in accordance to the capabilities of the dock widgets
431 */
432void QDockWidgetGroupWindow::adjustFlags()
433{
434 Qt::WindowFlags oldFlags = windowFlags();
435 Qt::WindowFlags flags = oldFlags;
436
437#if QT_CONFIG(tabbar)
438 QDockWidget *top = activeTabbedDockWidget();
439#else
440 QDockWidget *top = nullptr;
441#endif
442 if (!top) { // nested tabs, show window decoration
443 flags =
444 ((oldFlags & ~Qt::FramelessWindowHint) | Qt::CustomizeWindowHint | Qt::WindowTitleHint);
445 } else if (static_cast<QDockWidgetGroupLayout *>(layout())->nativeWindowDeco()) {
446 flags |= Qt::CustomizeWindowHint | Qt::WindowTitleHint;
447 flags.setFlag(flag: Qt::WindowCloseButtonHint, on: top->features() & QDockWidget::DockWidgetClosable);
448 flags &= ~Qt::FramelessWindowHint;
449 } else {
450 flags &= ~(Qt::WindowCloseButtonHint | Qt::CustomizeWindowHint | Qt::WindowTitleHint);
451 flags |= Qt::FramelessWindowHint;
452 }
453
454 if (oldFlags != flags) {
455 if (!windowHandle())
456 create(); // The desired geometry is forgotten if we call setWindowFlags before having a window
457 setWindowFlags(flags);
458 const bool gainedNativeDecos = (oldFlags & Qt::FramelessWindowHint) && !(flags & Qt::FramelessWindowHint);
459 const bool lostNativeDecos = !(oldFlags & Qt::FramelessWindowHint) && (flags & Qt::FramelessWindowHint);
460
461 // Adjust the geometry after gaining/losing decos, so that the client area appears always
462 // at the same place when tabbing
463 if (lostNativeDecos) {
464 QRect newGeometry = geometry();
465 newGeometry.setTop(frameGeometry().top());
466 const int bottomFrame = geometry().top() - frameGeometry().top();
467 m_removedFrameSize = QSize((frameSize() - size()).width(), bottomFrame);
468 setGeometry(newGeometry);
469 } else if (gainedNativeDecos && m_removedFrameSize.isValid()) {
470 QRect r = geometry();
471 r.adjust(dx1: -m_removedFrameSize.width() / 2, dy1: 0,
472 dx2: -m_removedFrameSize.width() / 2, dy2: -m_removedFrameSize.height());
473 setGeometry(r);
474 m_removedFrameSize = QSize();
475 }
476
477 show(); // setWindowFlags hides the window
478 }
479
480 QWidget *titleBarOf = top ? top : parentWidget();
481 setWindowTitle(titleBarOf->windowTitle());
482 setWindowIcon(titleBarOf->windowIcon());
483}

501/*
502 The given widget is hovered over this floating group.
503 This function will save the state and create a gap in the actual state.
504 currentGapRect and currentGapPos will be set.
505 One must call restore() or apply() after this function.
506 Returns true if there was any change in the currentGapPos
507 */
508bool QDockWidgetGroupWindow::hover(QLayoutItem *widgetItem, const QPoint &mousePos)
509{
510 QDockAreaLayoutInfo &savedState = static_cast<QDockWidgetGroupLayout *>(layout())->savedState;
511 if (savedState.isEmpty())
512 savedState = *layoutInfo();
513
514 QMainWindow::DockOptions opts = static_cast<QMainWindow *>(parentWidget())->dockOptions();
515 QDockAreaLayoutInfo newState = savedState;
516 bool nestingEnabled =
517 (opts & QMainWindow::AllowNestedDocks) && !(opts & QMainWindow::ForceTabbedDocks);
518 QDockAreaLayoutInfo::TabMode tabMode =
519#if !QT_CONFIG(tabbar)
520 QDockAreaLayoutInfo::NoTabs;
521#else
522 nestingEnabled ? QDockAreaLayoutInfo::AllowTabs : QDockAreaLayoutInfo::ForceTabs;
523 if (auto group = qobject_cast<QDockWidgetGroupWindow *>(object: widgetItem->widget())) {
524 if (!group->tabLayoutInfo())
525 tabMode = QDockAreaLayoutInfo::NoTabs;
526 }
527 if (newState.tabbed) {
528 // insertion into a top-level tab
529 newState.item_list = { QDockAreaLayoutItem(new QDockAreaLayoutInfo(newState)) };
530 newState.item_list.first().size = pick(o: savedState.o, size: savedState.rect.size());
531 newState.tabbed = false;
532 newState.tabBar = nullptr;
533 }
534#endif
535
536 auto newGapPos = newState.gapIndex(pos: mousePos, nestingEnabled, tabMode);
537 Q_ASSERT(!newGapPos.isEmpty());
538 if (newGapPos == currentGapPos)
539 return false; // gap is already there
540 currentGapPos = newGapPos;
541 newState.insertGap(path: currentGapPos, dockWidgetItem: widgetItem);
542 newState.fitItems();
543 *layoutInfo() = std::move(newState);
544 updateCurrentGapRect();
545 layoutInfo()->apply(animate: opts & QMainWindow::AnimatedDocks);
546 return true;
547}

666#endif

689void QMainWindowLayoutState::fitLayout()
690{
691 QRect r;
692#if !QT_CONFIG(toolbar)
693 r = rect;

1153bool QMainWindowLayoutState::restoreState(QDataStream &_stream,
1154 const QMainWindowLayoutState &oldState)
1155{
1156 //make a copy of the data so that we can read it more than once
1157 QByteArray copy;
1158 while(!_stream.atEnd()) {
1159 int length = 1024;
1160 QByteArray ba(length, '\0');
1161 length = _stream.readRawData(ba.data(), len: ba.size());
1162 ba.resize(size: length);
1163 copy += ba;
1164 }
1165
1166 QDataStream ds(copy);
1167 ds.setVersion(_stream.version());
1168 if (!checkFormat(stream&: ds))
1169 return false;
1170
1171 QDataStream stream(copy);
1172 stream.setVersion(_stream.version());
1173
1174 while (!stream.atEnd()) {
1175 uchar marker;
1176 stream >> marker;
1177 switch(marker)
1178 {
1179#if QT_CONFIG(dockwidget)
1180 case QDockAreaLayout::DockWidgetStateMarker:
1181 {
1182 const auto dockWidgets = allMyDockWidgets(mainWindow);
1183 if (!dockAreaLayout.restoreState(stream, widgets: dockWidgets))
1184 return false;
1185
1186 for (int i = 0; i < dockWidgets.size(); ++i) {
1187 QDockWidget *w = dockWidgets.at(i);
1188 QList<int> path = dockAreaLayout.indexOf(dockWidget: w);
1189 if (path.isEmpty()) {
1190 QList<int> oldPath = oldState.dockAreaLayout.indexOf(dockWidget: w);
1191 if (oldPath.isEmpty()) {
1192 continue;
1193 }
1194 QDockAreaLayoutInfo *info = dockAreaLayout.info(path: oldPath);
1195 if (info == nullptr) {
1196 continue;
1197 }
1198 info->item_list.append(t: QDockAreaLayoutItem(new QDockWidgetItem(w)));
1199 }
1200 }
1201 }
1202 break;
1203#if QT_CONFIG(tabwidget)
1204 case QDockAreaLayout::FloatingDockWidgetTabMarker:
1205 {
1206 auto dockWidgets = allMyDockWidgets(mainWindow);
1207 QDockWidgetGroupWindow* floatingTab = qt_mainwindow_layout(window: mainWindow)->createTabbedDockWindow();
1208 *floatingTab->layoutInfo() = QDockAreaLayoutInfo(
1209 &dockAreaLayout.sep, QInternal::LeftDock, // FIXME: DockWidget doesn't save original docking area
1210 Qt::Horizontal, QTabBar::RoundedSouth, mainWindow);
1211 QRect geometry;
1212 stream >> geometry;
1213 QDockAreaLayoutInfo *info = floatingTab->layoutInfo();
1214 if (!info->restoreState(stream, widgets&: dockWidgets, testing: false))
1215 return false;
1216 geometry = QDockAreaLayout::constrainedRect(rect: geometry, widget: floatingTab);
1217 floatingTab->move(geometry.topLeft());
1218 floatingTab->resize(geometry.size());
1219
1220 // Don't show an empty QDockWidgetGroupWindow if no dock widget is available yet.
1221 // reparentWidgets() would be triggered by show(), so do it explicitly here.
1222 if (info->onlyHasPlaceholders())
1223 info->reparentWidgets(p: floatingTab);
1224 else
1225 floatingTab->show();
1226 }
1227 break;
1228#endif // QT_CONFIG(tabwidget)
1229#endif // QT_CONFIG(dockwidget)
1230
1231#if QT_CONFIG(toolbar)
1232 case QToolBarAreaLayout::ToolBarStateMarker:
1233 case QToolBarAreaLayout::ToolBarStateMarkerEx:
1234 {
1235 QList<QToolBar *> toolBars = findChildrenHelper<QToolBar*>(o: mainWindow);
1236 if (!toolBarAreaLayout.restoreState(stream, toolBars, tmarker: marker))
1237 return false;
1238
1239 for (int i = 0; i < toolBars.size(); ++i) {
1240 QToolBar *w = toolBars.at(i);
1241 QList<int> path = toolBarAreaLayout.indexOf(toolBar: w);
1242 if (path.isEmpty()) {
1243 QList<int> oldPath = oldState.toolBarAreaLayout.indexOf(toolBar: w);
1244 if (oldPath.isEmpty()) {
1245 continue;
1246 }
1247 toolBarAreaLayout.docks[oldPath.at(i: 0)].insertToolBar(before: nullptr, toolBar: w);
1248 }
1249 }
1250 }
1251 break;
1252#endif // QT_CONFIG(toolbar)
1253 default:
1254 return false;
1255 }// switch
1256 } //while
1257
1258
1259 return true;
1260}

1737 // When QTabBarPrivate::dragInProgress is true, and that the mouse is outside of a region
