81/*! \class QPlainTextDocumentLayout
82 \since 4.4
83 \brief The QPlainTextDocumentLayout class implements a plain text layout for QTextDocument.
84
85 \ingroup richtext-processing
86 \inmodule QtWidgets
87
88 A QPlainTextDocumentLayout is required for text documents that can
89 be display or edited in a QPlainTextEdit. See
90 QTextDocument::setDocumentLayout().
91
92 QPlainTextDocumentLayout uses the QAbstractTextDocumentLayout API
93 that QTextDocument requires, but redefines it partially in order to
94 support plain text better. For instances, it does not operate on
95 vertical pixels, but on paragraphs (called blocks) instead. The
96 height of a document is identical to the number of paragraphs it
97 contains. The layout also doesn't support tables or nested frames,
98 or any sort of advanced text layout that goes beyond a list of
99 paragraphs with syntax highlighting.
100
101*/

256/*! \reimp
257 */
258void QPlainTextDocumentLayout::documentChanged(int from, int charsRemoved, int charsAdded)
259{
260 Q_D(QPlainTextDocumentLayout);
261 QTextDocument *doc = document();
262 int newBlockCount = doc->blockCount();
263 int charsChanged = charsRemoved + charsAdded;
264
265 QTextBlock changeStartBlock = doc->findBlock(pos: from);
266 QTextBlock changeEndBlock = doc->findBlock(pos: qMax(a: 0, b: from + charsChanged - 1));
267 bool blockVisibilityChanged = false;
268
269 if (changeStartBlock == changeEndBlock && newBlockCount == d->blockCount) {
270 QTextBlock block = changeStartBlock;
271 if (block.isValid() && block.length()) {
272 QRectF oldBr = blockBoundingRect(block);
273 layoutBlock(block);
274 QRectF newBr = blockBoundingRect(block);
275 if (newBr.height() == oldBr.height()) {
276 if (!d->blockUpdate)
277 emit updateBlock(block);
278 return;
279 }
280 }
281 } else {
282 QTextBlock block = changeStartBlock;
283 do {
284 block.clearLayout();
285 if (block.isVisible()
286 ? (block.lineCount() == 0)
287 : (block.lineCount() > 0)) {
288 blockVisibilityChanged = true;
289 block.setLineCount(block.isVisible() ? 1 : 0);
290 }
291 if (block == changeEndBlock)
292 break;
293 block = block.next();
294 } while(block.isValid());
295 }
296
297 if (newBlockCount != d->blockCount || blockVisibilityChanged) {
298 int changeEnd = changeEndBlock.blockNumber();
299 int blockDiff = newBlockCount - d->blockCount;
300 int oldChangeEnd = changeEnd - blockDiff;
301
302 if (d->maximumWidthBlockNumber > oldChangeEnd)
303 d->maximumWidthBlockNumber += blockDiff;
304
305 d->blockCount = newBlockCount;
306 if (d->blockCount == 1)
307 d->maximumWidth = blockWidth(block: doc->firstBlock());
308
309 if (!d->blockDocumentSizeChanged)
310 emit documentSizeChanged(newSize: documentSize());
311
312 if (blockDiff == 1 && changeEnd == newBlockCount -1 ) {
313 if (!d->blockUpdate) {
314 QTextBlock b = changeStartBlock;
315 for(;;) {
316 emit updateBlock(block: b);
317 if (b == changeEndBlock)
318 break;
319 b = b.next();
320 }
321 }
322 return;
323 }
324 }
325
326 if (!d->blockUpdate)
327 emit update(QRectF(0., -doc->documentMargin(), 1000000000., 1000000000.)); // optimization potential
328}

331void QPlainTextDocumentLayout::layoutBlock(const QTextBlock &block)
332{
333 Q_D(QPlainTextDocumentLayout);
334 QTextDocument *doc = document();
335 qreal margin = doc->documentMargin();
336 qreal blockMaximumWidth = 0;
337
338 qreal height = 0;
339 QTextLayout *tl = block.layout();
340 QTextOption option = doc->defaultTextOption();
341 tl->setTextOption(option);
342
343 int extraMargin = 0;
344 if (option.flags() & QTextOption::AddSpaceForLineAndParagraphSeparators) {
345 QFontMetrics fm(block.charFormat().font());
346 extraMargin += fm.horizontalAdvance(QChar(0x21B5));
347 }
348 tl->beginLayout();
349 qreal availableWidth = d->width;
350 if (availableWidth <= 0) {
351 availableWidth = qreal(INT_MAX); // similar to text edit with pageSize.width == 0
352 }
353 availableWidth -= 2*margin + extraMargin;
354 while (1) {
355 QTextLine line = tl->createLine();
356 if (!line.isValid())
357 break;
358 line.setLeadingIncluded(true);
359 line.setLineWidth(availableWidth);
360 line.setPosition(QPointF(margin, height));
361 height += line.height();
362 if (line.leading() < 0)
363 height += qCeil(v: line.leading());
364 blockMaximumWidth = qMax(a: blockMaximumWidth, b: line.naturalTextWidth() + 2*margin);
365 }
366 tl->endLayout();
367
368 int previousLineCount = doc->lineCount();
369 const_cast<QTextBlock&>(block).setLineCount(block.isVisible() ? tl->lineCount() : 0);
370 int lineCount = doc->lineCount();
371
372 bool emitDocumentSizeChanged = previousLineCount != lineCount;
373 if (blockMaximumWidth > d->maximumWidth) {
374 // new longest line
375 d->maximumWidth = blockMaximumWidth;
376 d->maximumWidthBlockNumber = block.blockNumber();
377 emitDocumentSizeChanged = true;
378 } else if (block.blockNumber() == d->maximumWidthBlockNumber && blockMaximumWidth < d->maximumWidth) {
379 // longest line shrinking
380 QTextBlock b = doc->firstBlock();
381 d->maximumWidth = 0;
382 QTextBlock maximumBlock;
383 while (b.isValid()) {
384 qreal blockMaximumWidth = blockWidth(block: b);
385 if (blockMaximumWidth > d->maximumWidth) {
386 d->maximumWidth = blockMaximumWidth;
387 maximumBlock = b;
388 }
389 b = b.next();
390 }
391 if (maximumBlock.isValid()) {
392 d->maximumWidthBlockNumber = maximumBlock.blockNumber();
393 emitDocumentSizeChanged = true;
394 }
395 }
396 if (emitDocumentSizeChanged && !d->blockDocumentSizeChanged)
397 emit documentSizeChanged(newSize: documentSize());
398}

400qreal QPlainTextDocumentLayout::blockWidth(const QTextBlock &block)
401{
402 QTextLayout *layout = block.layout();
403 if (!layout->lineCount())
404 return 0; // only for layouted blocks
405 qreal blockWidth = 0;
406 for (int i = 0; i < layout->lineCount(); ++i) {
407 QTextLine line = layout->lineAt(i);
408 blockWidth = qMax(a: line.naturalTextWidth() + 8, b: blockWidth);
409 }
410 return blockWidth;
411}

460qreal QPlainTextEditPrivate::verticalOffset(int topBlock, int topLine) const
461{
462 qreal offset = 0;
463 QTextDocument *doc = control->document();
464
465 if (topLine) {
466 QTextBlock currentBlock = doc->findBlockByNumber(blockNumber: topBlock);
467 QPlainTextDocumentLayout *documentLayout = qobject_cast<QPlainTextDocumentLayout*>(object: doc->documentLayout());
468 Q_ASSERT(documentLayout);
469 QRectF r = documentLayout->blockBoundingRect(block: currentBlock);
470 Q_UNUSED(r);
471 QTextLayout *layout = currentBlock.layout();
472 if (layout && topLine <= layout->lineCount()) {
473 QTextLine line = layout->lineAt(i: topLine - 1);
474 const QRectF lr = line.naturalTextRect();
475 offset = lr.bottom();
476 }
477 }
478 if (topBlock == 0 && topLine == 0)
479 offset -= doc->documentMargin(); // top margin
480 return offset;
481}

496int QPlainTextEditControl::hitTest(const QPointF &point, Qt::HitTestAccuracy ) const {
497 int currentBlockNumber = topBlock;
498 QTextBlock currentBlock = document()->findBlockByNumber(blockNumber: currentBlockNumber);
499 if (!currentBlock.isValid())
500 return -1;
501
502 QPlainTextDocumentLayout *documentLayout = qobject_cast<QPlainTextDocumentLayout*>(object: document()->documentLayout());
503 Q_ASSERT(documentLayout);
504
505 QPointF offset;
506 QRectF r = documentLayout->blockBoundingRect(block: currentBlock);
507 while (currentBlock.next().isValid() && r.bottom() + offset.y() <= point.y()) {
508 offset.ry() += r.height();
509 currentBlock = currentBlock.next();
510 ++currentBlockNumber;
511 r = documentLayout->blockBoundingRect(block: currentBlock);
512 }
513 while (currentBlock.previous().isValid() && r.top() + offset.y() > point.y()) {
514 offset.ry() -= r.height();
515 currentBlock = currentBlock.previous();
516 --currentBlockNumber;
517 r = documentLayout->blockBoundingRect(block: currentBlock);
518 }
519
520
521 if (!currentBlock.isValid())
522 return -1;
523 QTextLayout *layout = currentBlock.layout();
524 int off = 0;
525 QPointF pos = point - offset;
526 for (int i = 0; i < layout->lineCount(); ++i) {
527 QTextLine line = layout->lineAt(i);
528 const QRectF lr = line.naturalTextRect();
529 if (lr.top() > pos.y()) {
530 off = qMin(a: off, b: line.textStart());
531 } else if (lr.bottom() <= pos.y()) {
532 off = qMax(a: off, b: line.textStart() + line.textLength());
533 } else {
534 off = line.xToCursor(x: pos.x(), overwriteMode() ?
535 QTextLine::CursorOnCharacter : QTextLine::CursorBetweenCharacters);
536 break;
537 }
538 }
539
540 return currentBlock.position() + off;
541}

543QRectF QPlainTextEditControl::blockBoundingRect(const QTextBlock &block) const {
544 int currentBlockNumber = topBlock;
545 int blockNumber = block.blockNumber();
546 QTextBlock currentBlock = document()->findBlockByNumber(blockNumber: currentBlockNumber);
547 if (!currentBlock.isValid())
548 return QRectF();
549 Q_ASSERT(currentBlock.blockNumber() == currentBlockNumber);
550 QTextDocument *doc = document();
551 QPlainTextDocumentLayout *documentLayout = qobject_cast<QPlainTextDocumentLayout*>(object: doc->documentLayout());
552 Q_ASSERT(documentLayout);
553
554 QPointF offset;
555 if (!block.isValid())
556 return QRectF();
557 QRectF r = documentLayout->blockBoundingRect(block: currentBlock);
558 int maxVerticalOffset = r.height();
559 while (currentBlockNumber < blockNumber && offset.y() - maxVerticalOffset <= 2* textEdit->viewport()->height()) {
560 offset.ry() += r.height();
561 currentBlock = currentBlock.next();
562 ++currentBlockNumber;
563 if (!currentBlock.isVisible()) {
564 currentBlock = doc->findBlockByLineNumber(blockNumber: currentBlock.firstLineNumber());
565 currentBlockNumber = currentBlock.blockNumber();
566 }
567 r = documentLayout->blockBoundingRect(block: currentBlock);
568 }
569 while (currentBlockNumber > blockNumber && offset.y() + maxVerticalOffset >= -textEdit->viewport()->height()) {
570 currentBlock = currentBlock.previous();
571 --currentBlockNumber;
572 while (!currentBlock.isVisible()) {
573 currentBlock = currentBlock.previous();
574 --currentBlockNumber;
575 }
576 if (!currentBlock.isValid())
577 break;
578
579 r = documentLayout->blockBoundingRect(block: currentBlock);
580 offset.ry() -= r.height();
581 }
582
583 if (currentBlockNumber != blockNumber) {
584 // fallback for blocks out of reach. Give it some geometry at
585 // least, and ensure the layout is up to date.
586 r = documentLayout->blockBoundingRect(block);
587 if (currentBlockNumber > blockNumber)
588 offset.ry() -= r.height();
589 }
590 r.translate(p: offset);
591 return r;
592}

608void QPlainTextEditPrivate::setTopBlock(int blockNumber, int lineNumber, int dx)
609{
610 Q_Q(QPlainTextEdit);
611 blockNumber = qMax(a: 0, b: blockNumber);
612 lineNumber = qMax(a: 0, b: lineNumber);
613 QTextDocument *doc = control->document();
614 QTextBlock block = doc->findBlockByNumber(blockNumber);
615
616 int newTopLine = block.firstLineNumber() + lineNumber;
617 int maxTopLine = vbar->maximum();
618
619 if (newTopLine > maxTopLine) {
620 block = doc->findBlockByLineNumber(blockNumber: maxTopLine);
621 blockNumber = block.blockNumber();
622 lineNumber = maxTopLine - block.firstLineNumber();
623 }
624
625 {
626 const QSignalBlocker blocker(vbar);
627 vbar->setValue(newTopLine);
628 }
629
630 if (!dx && blockNumber == control->topBlock && lineNumber == topLine)
631 return;
632
633 if (viewport->updatesEnabled() && viewport->isVisible()) {
634 int dy = 0;
635 if (doc->findBlockByNumber(blockNumber: control->topBlock).isValid()) {
636 qreal realdy = -q->blockBoundingGeometry(block).y()
637 + verticalOffset() - verticalOffset(topBlock: blockNumber, topLine: lineNumber);
638 dy = (int)realdy;
639 topLineFracture = realdy - dy;
640 }
641 control->topBlock = blockNumber;
642 topLine = lineNumber;
643
644 {
645 const QSignalBlocker blocker(vbar);
646 vbar->setValue(block.firstLineNumber() + lineNumber);
647 }
648
649 if (dx || dy) {
650 viewport->scroll(dx: q->isRightToLeft() ? -dx : dx, dy);
651 QGuiApplication::inputMethod()->update(queries: Qt::ImCursorRectangle | Qt::ImAnchorRectangle);
652 } else {
653 viewport->update();
654 topLineFracture = 0;
655 }
656 emit q->updateRequest(rect: viewport->rect(), dy);
657 } else {
658 control->topBlock = blockNumber;
659 topLine = lineNumber;
660 topLineFracture = 0;
661 }
662
663}

667void QPlainTextEditPrivate::ensureVisible(int position, bool center, bool forceCenter) {
668 Q_Q(QPlainTextEdit);
669 QRectF visible = QRectF(viewport->rect()).translated(p: -q->contentOffset());
670 QTextBlock block = control->document()->findBlock(pos: position);
671 if (!block.isValid())
672 return;
673 QRectF br = control->blockBoundingRect(block);
674 if (!br.isValid())
675 return;
676 QTextLine line = block.layout()->lineForTextPosition(pos: position - block.position());
677 Q_ASSERT(line.isValid());
678 QRectF lr = line.naturalTextRect().translated(p: br.topLeft());
679
680 if (lr.bottom() >= visible.bottom() || (center && lr.top() < visible.top()) || forceCenter){
681
682 qreal height = visible.height();
683 if (center)
684 height /= 2;
685
686 qreal h = center ? line.naturalTextRect().center().y() : line.naturalTextRect().bottom();
687
688 QTextBlock previousVisibleBlock = block;
689 while (h < height && block.previous().isValid()) {
690 previousVisibleBlock = block;
691 do {
692 block = block.previous();
693 } while (!block.isVisible() && block.previous().isValid());
694 h += q->blockBoundingRect(block).height();
695 }
696
697 int l = 0;
698 int lineCount = block.layout()->lineCount();
699 qreal voffset = verticalOffset(topBlock: block.blockNumber(), topLine: 0);
700 while (l < lineCount) {
701 QRectF lineRect = block.layout()->lineAt(i: l).naturalTextRect();
702 if (h - voffset - lineRect.top() <= height)
703 break;
704 ++l;
705 }
706
707 if (l >= lineCount) {
708 block = previousVisibleBlock;
709 l = 0;
710 }
711 setTopBlock(blockNumber: block.blockNumber(), lineNumber: l);
712 } else if (lr.top() < visible.top()) {
713 setTopBlock(blockNumber: block.blockNumber(), lineNumber: line.lineNumber());
714 }

829{
830
831 Q_Q(QPlainTextEdit);
832
833 QTextCursor cursor = control->textCursor();
834 if (moveCursor) {
835 ensureCursorVisible();
836 if (!pageUpDownLastCursorYIsValid)
837 pageUpDownLastCursorY = control->cursorRect(cursor).top() - verticalOffset();
838 }
839
840 qreal lastY = pageUpDownLastCursorY;
841
842
843 if (op == QTextCursor::Down) {
844 QRectF visible = QRectF(viewport->rect()).translated(p: -q->contentOffset());
845 QTextBlock firstVisibleBlock = q->firstVisibleBlock();
846 QTextBlock block = firstVisibleBlock;
847 QRectF br = q->blockBoundingRect(block);
848 qreal h = 0;
849 int atEnd = false;
850 while (h + br.height() <= visible.bottom()) {
851 if (!block.next().isValid()) {
852 atEnd = true;
853 lastY = visible.bottom(); // set cursor to last line
854 break;
855 }
856 h += br.height();
857 block = block.next();
858 br = q->blockBoundingRect(block);
859 }
860
861 if (!atEnd) {
862 int line = 0;
863 qreal diff = visible.bottom() - h;
864 int lineCount = block.layout()->lineCount();
865 while (line < lineCount - 1) {
866 if (block.layout()->lineAt(i: line).naturalTextRect().bottom() > diff) {
867 // the first line that did not completely fit the screen
868 break;
869 }
870 ++line;
871 }
872 setTopBlock(blockNumber: block.blockNumber(), lineNumber: line);
873 }
874
875 if (moveCursor) {
876 // move using movePosition to keep the cursor's x
877 lastY += verticalOffset();
878 bool moved = false;
879 do {
880 moved = cursor.movePosition(op, moveMode);
881 } while (moved && control->cursorRect(cursor).top() < lastY);
882 }
883
884 } else if (op == QTextCursor::Up) {
885
886 QRectF visible = QRectF(viewport->rect()).translated(p: -q->contentOffset());
887 visible.translate(dx: 0, dy: -visible.height()); // previous page
888 QTextBlock block = q->firstVisibleBlock();
889 qreal h = 0;
890 while (h >= visible.top()) {
891 if (!block.previous().isValid()) {
892 if (control->topBlock == 0 && topLine == 0) {
893 lastY = 0; // set cursor to first line
894 }
895 break;
896 }
897 block = block.previous();
898 QRectF br = q->blockBoundingRect(block);
899 h -= br.height();
900 }
901
902 int line = 0;
903 if (block.isValid()) {
904 qreal diff = visible.top() - h;
905 int lineCount = block.layout()->lineCount();
906 while (line < lineCount) {
907 if (block.layout()->lineAt(i: line).naturalTextRect().top() >= diff)
908 break;
909 ++line;
910 }
911 if (line == lineCount) {
912 if (block.next().isValid() && block.next() != q->firstVisibleBlock()) {
913 block = block.next();
914 line = 0;
915 } else {
916 --line;
917 }
918 }
919 }
920 setTopBlock(blockNumber: block.blockNumber(), lineNumber: line);
921

923 cursor.setVisualNavigation(true);
924 // move using movePosition to keep the cursor's x
925 lastY += verticalOffset();

927 do {
928 moved = cursor.movePosition(op, moveMode);
929 } while (moved && control->cursorRect(cursor).top() > lastY);
930 }
931 }

934 control->setTextCursor(cursor, selectionClipboard: moveMode == QTextCursor::KeepAnchor);

1023/*!
1024 \class QPlainTextEdit
1025 \since 4.4
1026 \brief The QPlainTextEdit class provides a widget that is used to edit and display
1027 plain text.
1028
1029 \ingroup richtext-processing
1030 \inmodule QtWidgets
1031
1032 \tableofcontents
1033
1034 \section1 Introduction and Concepts
1035
1036 QPlainTextEdit is an advanced viewer/editor supporting plain
1037 text. It is optimized to handle large documents and to respond
1038 quickly to user input.
1039
1040 QPlainText uses very much the same technology and concepts as
1041 QTextEdit, but is optimized for plain text handling.
1042
1043 QPlainTextEdit works on paragraphs and characters. A paragraph is
1044 a formatted string which is word-wrapped to fit into the width of
1045 the widget. By default when reading plain text, one newline
1046 signifies a paragraph. A document consists of zero or more
1047 paragraphs. Paragraphs are separated by hard line breaks. Each
1048 character within a paragraph has its own attributes, for example,
1049 font and color.
1050
1051 The shape of the mouse cursor on a QPlainTextEdit is
1052 Qt::IBeamCursor by default. It can be changed through the
1053 viewport()'s cursor property.
1054
1055 \section1 Using QPlainTextEdit as a Display Widget
1056
1057 The text is set or replaced using setPlainText() which deletes the
1058 existing text and replaces it with the text passed to setPlainText().
1059
1060 Text can be inserted using the QTextCursor class or using the
1061 convenience functions insertPlainText(), appendPlainText() or
1062 paste().
1063
1064 By default, the text edit wraps words at whitespace to fit within
1065 the text edit widget. The setLineWrapMode() function is used to
1066 specify the kind of line wrap you want, \l WidgetWidth or \l
1067 NoWrap if you don't want any wrapping. If you use word wrap to
1068 the widget's width \l WidgetWidth, you can specify whether to
1069 break on whitespace or anywhere with setWordWrapMode().
1070
1071 The find() function can be used to find and select a given string
1072 within the text.
1073
1074 If you want to limit the total number of paragraphs in a
1075 QPlainTextEdit, as it is for example useful in a log viewer, then
1076 you can use the maximumBlockCount property. The combination of
1077 setMaximumBlockCount() and appendPlainText() turns QPlainTextEdit
1078 into an efficient viewer for log text. The scrolling can be
1079 reduced with the centerOnScroll() property, making the log viewer
1080 even faster. Text can be formatted in a limited way, either using
1081 a syntax highlighter (see below), or by appending html-formatted
1082 text with appendHtml(). While QPlainTextEdit does not support
1083 complex rich text rendering with tables and floats, it does
1084 support limited paragraph-based formatting that you may need in a
1085 log viewer.

1885{

1901 painter.setBrushOrigin(offset);
1902

1905 - document()->documentMargin() + cursorWidth();

1969 }

1980 drawCursorAsBlock = false;
1981 } else {
1982 QTextLayout::FormatRange o;
1983 o.start = context.cursorPosition - blpos;
1984 o.length = 1;
1985 o.format.setForeground(palette().base());
1986 o.format.setBackground(palette().text());

1988 }
1989 }
1990
1991 layout->draw(p: &painter, pos: offset, selections, clip: er);
1992

1996 int cpos = context.cursorPosition;

1998 cpos = layout->preeditAreaPosition() - (cpos + 2);
1999 else
2000 cpos -= blpos;

2002 }

2007 break;

2009 }
2010
2011 if (backgroundVisible() && !block.isValid() && offset.y() <= er.bottom()
2012 && (centerOnScroll() || verticalScrollBar()->maximum() == verticalScrollBar()->minimum())) {
2013 painter.fillRect(QRect(QPoint((int)er.left(), (int)offset.y()), er.bottomRight()), palette().window());
