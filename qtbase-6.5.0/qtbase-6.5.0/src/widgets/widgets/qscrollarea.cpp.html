18/*!
19 \class QScrollArea
20
21 \brief The QScrollArea class provides a scrolling view onto
22 another widget.
23
24 \ingroup basicwidgets
25 \inmodule QtWidgets
26
27 A scroll area is used to display the contents of a child widget
28 within a frame. If the widget exceeds the size of the frame, the
29 view can provide scroll bars so that the entire area of the child
30 widget can be viewed. The child widget must be specified with
31 setWidget(). For example:
32
33 \snippet code/src_gui_widgets_qscrollarea.cpp 0
34
35 The code above creates a scroll area (shown in the images below)
36 containing an image label. When scaling the image, the scroll area
37 can provide the necessary scroll bars:
38
39 \table
40 \row
41 \li \inlineimage qscrollarea-noscrollbars.png
42 \li \inlineimage qscrollarea-onescrollbar.png
43 \li \inlineimage qscrollarea-twoscrollbars.png
44 \endtable
45
46 The scroll bars appearance depends on the currently set \l
47 {Qt::ScrollBarPolicy}{scroll bar policies}. You can control the
48 appearance of the scroll bars using the inherited functionality
49 from QAbstractScrollArea.
50
51 For example, you can set the
52 QAbstractScrollArea::horizontalScrollBarPolicy and
53 QAbstractScrollArea::verticalScrollBarPolicy properties. Or if you
54 want the scroll bars to adjust dynamically when the contents of
55 the scroll area changes, you can use the \l
56 {QAbstractScrollArea::horizontalScrollBar()}{horizontalScrollBar()}
57 and \l
58 {QAbstractScrollArea::verticalScrollBar()}{verticalScrollBar()}
59 functions (which enable you to access the scroll bars) and set the
60 scroll bars' values whenever the scroll area's contents change,
61 using the QScrollBar::setValue() function.
62
63 You can retrieve the child widget using the widget() function. The
64 view can be made to be resizable with the setWidgetResizable()
65 function. The alignment of the widget can be specified with
66 setAlignment().
67
68 Two convenience functions ensureVisible() and
69 ensureWidgetVisible() ensure a certain region of the contents is
70 visible inside the viewport, by scrolling the contents if
71 necessary.
72
73 \section1 Size Hints and Layouts
74
75 When using a scroll area to display the contents of a custom
76 widget, it is important to ensure that the
77 \l{QWidget::sizeHint}{size hint} of the child widget is set to a
78 suitable value. If a standard QWidget is used for the child
79 widget, it may be necessary to call QWidget::setMinimumSize() to
80 ensure that the contents of the widget are shown correctly within
81 the scroll area.
82
83 If a scroll area is used to display the contents of a widget that
84 contains child widgets arranged in a layout, it is important to
85 realize that the size policy of the layout will also determine the
86 size of the widget. This is especially useful to know if you intend
87 to dynamically change the contents of the layout. In such cases,
88 setting the layout's \l{QLayout::sizeConstraint}{size constraint}
89 property to one which provides constraints on the minimum and/or
90 maximum size of the layout (e.g., QLayout::SetMinAndMaxSize) will
91 cause the size of the scroll area to be updated whenever the
92 contents of the layout changes.
93
94 For a complete example using the QScrollArea class, see the \l
95 {widgets/imageviewer}{Image Viewer} example. The example shows how
96 to combine QLabel and QScrollArea to display an image.
97
98 \sa QAbstractScrollArea, QScrollBar, {Image Viewer Example}
99*/

139void QScrollAreaPrivate::updateWidgetPosition()
140{
141 Q_Q(QScrollArea);
142 Qt::LayoutDirection dir = q->layoutDirection();
143 QRect scrolled = QStyle::visualRect(direction: dir, boundingRect: viewport->rect(), logicalRect: QRect(QPoint(-hbar->value(), -vbar->value()), widget->size()));
144 QRect aligned = QStyle::alignedRect(direction: dir, alignment, size: widget->size(), rectangle: viewport->rect());
145 widget->move(ax: widget->width() < viewport->width() ? aligned.x() : scrolled.x(),
146 ay: widget->height() < viewport->height() ? aligned.y() : scrolled.y());
147}
148
149void QScrollAreaPrivate::updateScrollBars()
150{
151 Q_Q(QScrollArea);
152 if (!widget)
153 return;
154 QSize p = viewport->size();
155 QSize m = q->maximumViewportSize();
156
157 QSize min = qSmartMinSize(w: widget);
158 QSize max = qSmartMaxSize(w: widget);
159
160 if (resizable) {
161 if ((widget->layout() ? widget->layout()->hasHeightForWidth() : widget->sizePolicy().hasHeightForWidth())) {
162 QSize p_hfw = p.expandedTo(otherSize: min).boundedTo(otherSize: max);
163 int h = widget->heightForWidth(p_hfw.width());
164 // If the height we calculated requires a vertical scrollbar,
165 // then we need to constrain the width and calculate the height again,
166 // otherwise we end up flipping the scrollbar on and off all the time.
167 if (vbarpolicy == Qt::ScrollBarAsNeeded) {
168 int vbarWidth = vbar->sizeHint().width();
169 QSize m_hfw = m.expandedTo(otherSize: min).boundedTo(otherSize: max);
170 // is there any point in searching?
171 if (widget->heightForWidth(m_hfw.width() - vbarWidth) <= m.height()) {
172 while (h > m.height() && vbarWidth) {
173 --vbarWidth;
174 --m_hfw.rwidth();
175 h = widget->heightForWidth(m_hfw.width());
176 }
177 }
178 max = QSize(m_hfw.width(), qMax(a: m_hfw.height(), b: h));
179 }
180 min = QSize(p_hfw.width(), qMax(a: p_hfw.height(), b: h));
181 }
182 }
183
184 if ((resizable && m.expandedTo(otherSize: min) == m && m.boundedTo(otherSize: max) == m)
185 || (!resizable && m.expandedTo(otherSize: widget->size()) == m))
186 p = m; // no scroll bars needed
187
188 if (resizable)
189 widget->resize(p.expandedTo(otherSize: min).boundedTo(otherSize: max));
190 QSize v = widget->size();
191
192 hbar->setRange(min: 0, max: v.width() - p.width());
193 hbar->setPageStep(p.width());
194 vbar->setRange(min: 0, max: v.height() - p.height());
195 vbar->setPageStep(p.height());
196 updateWidgetPosition();
197
198}

212/*!
213 \fn void QScrollArea::setWidget(QWidget *widget)
214
215 Sets the scroll area's \a widget.
216
217 The \a widget becomes a child of the scroll area, and will be
218 destroyed when the scroll area is deleted or when a new widget is
219 set.
220
221 The widget's \l{QWidget::setAutoFillBackground()}{autoFillBackground}
222 property will be set to \c{true}.
223
224 If the scroll area is visible when the \a widget is
225 added, you must \l{QWidget::}{show()} it explicitly.
226
227 Note that You must add the layout of \a widget before you call
228 this function; if you add it later, the \a widget will not be
229 visible - regardless of when you \l{QWidget::}{show()} the scroll
230 area. In this case, you can also not \l{QWidget::}{show()} the \a
231 widget later.
232
233 \sa widget()
234*/
235void QScrollArea::setWidget(QWidget *widget)
236{
237 Q_D(QScrollArea);
238 if (widget == d->widget || !widget)
239 return;
240
241 delete d->widget;
242 d->widget = nullptr;
243 d->hbar->setValue(0);
244 d->vbar->setValue(0);
245 if (widget->parentWidget() != d->viewport)
246 widget->setParent(d->viewport);
247 if (!widget->testAttribute(attribute: Qt::WA_Resized))
248 widget->resize(widget->sizeHint());
249 d->widget = widget;
250 d->widget->setAutoFillBackground(true);
251 widget->installEventFilter(filterObj: this);
252 d->widgetSize = QSize();
253 d->updateScrollBars();
254 d->widget->show();
255
256}

274/*!
275 \reimp
276 */
277bool QScrollArea::event(QEvent *e)
278{
279 Q_D(QScrollArea);
280 if (e->type() == QEvent::StyleChange || e->type() == QEvent::LayoutRequest) {
281 d->updateScrollBars();
282 }
283#ifdef QT_KEYPAD_NAVIGATION
284 else if (QApplicationPrivate::keypadNavigationEnabled()) {
285 if (e->type() == QEvent::Show)
286 QApplication::instance()->installEventFilter(this);
287 else if (e->type() == QEvent::Hide)
288 QApplication::instance()->removeEventFilter(this);
289 }
290#endif
291 return QAbstractScrollArea::event(e);
292}

295/*!
296 \reimp
297 */
298bool QScrollArea::eventFilter(QObject *o, QEvent *e)
299{
300 Q_D(QScrollArea);
301#ifdef QT_KEYPAD_NAVIGATION
302 if (d->widget && o != d->widget && e->type() == QEvent::FocusIn
303 && QApplicationPrivate::keypadNavigationEnabled()) {
304 if (o->isWidgetType())
305 ensureWidgetVisible(static_cast<QWidget *>(o));
306 }
307#endif
308 if (o == d->widget && e->type() == QEvent::Resize)
309 d->updateScrollBars();
310
311 return QAbstractScrollArea::eventFilter(o, e);
312}

336/*!
337 \property QScrollArea::widgetResizable
338 \brief whether the scroll area should resize the view widget
339
340 If this property is set to false (the default), the scroll area
341 honors the size of its widget. Regardless of this property, you
342 can programmatically resize the widget using widget()->resize(),
343 and the scroll area will automatically adjust itself to the new
344 size.
345
346 If this property is set to true, the scroll area will
347 automatically resize the widget in order to avoid scroll bars
348 where they can be avoided, or to take advantage of extra space.
349*/
350bool QScrollArea::widgetResizable() const
351{
352 Q_D(const QScrollArea);
353 return d->resizable;
354}

364/*!
365 \reimp
366 */
367QSize QScrollArea::sizeHint() const
368{
369 Q_D(const QScrollArea);
370 int f = 2 * d->frameWidth;
371 QSize sz(f, f);
372 int h = fontMetrics().height();
373 if (d->widget) {
374 if (!d->widgetSize.isValid())
375 d->widgetSize = d->resizable ? d->widget->sizeHint() : d->widget->size();
376 sz += d->widgetSize;
377 } else {
378 sz += QSize(12 * h, 8 * h);
379 }
380 if (d->vbarpolicy == Qt::ScrollBarAlwaysOn)
381 sz.setWidth(sz.width() + d->vbar->sizeHint().width());
382 if (d->hbarpolicy == Qt::ScrollBarAlwaysOn)
383 sz.setHeight(sz.height() + d->hbar->sizeHint().height());
384 return sz.boundedTo(otherSize: QSize(36 * h, 24 * h));
385}

394 return d->resizable ? d->widget->sizeHint() : d->widget->size();
395 }
396 const int h = fontMetrics().height();
397 return QSize(6 * h, 4 * h);

414/*!
415 Scrolls the contents of the scroll area so that the point (\a x, \a y) is visible
416 inside the region of the viewport with margins specified in pixels by \a xmargin and
417 \a ymargin. If the specified point cannot be reached, the contents are scrolled to
418 the nearest valid position. The default value for both margins is 50 pixels.
419*/
420void QScrollArea::ensureVisible(int x, int y, int xmargin, int ymargin)
421{
422 Q_D(QScrollArea);
423
424 int logicalX = QStyle::visualPos(direction: layoutDirection(), boundingRect: d->viewport->rect(), logicalPos: QPoint(x, y)).x();
425
426 if (logicalX - xmargin < d->hbar->value()) {
427 d->hbar->setValue(qMax(a: 0, b: logicalX - xmargin));
428 } else if (logicalX > d->hbar->value() + d->viewport->width() - xmargin) {
429 d->hbar->setValue(qMin(a: logicalX - d->viewport->width() + xmargin, b: d->hbar->maximum()));
430 }
431
432 if (y - ymargin < d->vbar->value()) {
433 d->vbar->setValue(qMax(a: 0, b: y - ymargin));
434 } else if (y > d->vbar->value() + d->viewport->height() - ymargin) {
435 d->vbar->setValue(qMin(a: y - d->viewport->height() + ymargin, b: d->vbar->maximum()));
436 }
437}

439/*!
440 \since 4.2
441
442 Scrolls the contents of the scroll area so that the \a childWidget
443 of QScrollArea::widget() is visible inside the viewport with
444 margins specified in pixels by \a xmargin and \a ymargin. If the
445 specified point cannot be reached, the contents are scrolled to
446 the nearest valid position. The default value for both margins is
447 50 pixels.
448
449*/
450void QScrollArea::ensureWidgetVisible(QWidget *childWidget, int xmargin, int ymargin)
451{
452 Q_D(QScrollArea);
453
454 if (!d->widget->isAncestorOf(child: childWidget))
455 return;
456
457 const QRect microFocus = childWidget->inputMethodQuery(Qt::ImCursorRectangle).toRect();
458 const QRect defaultMicroFocus =
459 childWidget->QWidget::inputMethodQuery(Qt::ImCursorRectangle).toRect();
460 QRect focusRect = (microFocus != defaultMicroFocus)
461 ? QRect(childWidget->mapTo(d->widget, microFocus.topLeft()), microFocus.size())
462 : QRect(childWidget->mapTo(d->widget, QPoint(0,0)), childWidget->size());
463 const QRect visibleRect(-d->widget->pos(), d->viewport->size());
464
465 if (visibleRect.contains(r: focusRect))
466 return;
467
468 focusRect.adjust(dx1: -xmargin, dy1: -ymargin, dx2: xmargin, dy2: ymargin);
469
470 if (focusRect.width() > visibleRect.width())
471 d->hbar->setValue(focusRect.center().x() - d->viewport->width() / 2);
472 else if (focusRect.right() > visibleRect.right())
473 d->hbar->setValue(focusRect.right() - d->viewport->width() + 1);
474 else if (focusRect.left() < visibleRect.left())
475 d->hbar->setValue(focusRect.left());
476
477 if (focusRect.height() > visibleRect.height())
478 d->vbar->setValue(focusRect.center().y() - d->viewport->height() / 2);
479 else if (focusRect.bottom() > visibleRect.bottom())
480 d->vbar->setValue(focusRect.bottom() - d->viewport->height() + 1);
481 else if (focusRect.top() < visibleRect.top())
482 d->vbar->setValue(focusRect.top());
483}
