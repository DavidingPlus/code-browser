82void QTabBarPrivate::updateMacBorderMetrics()
83{
84#if defined(Q_OS_MACOS)
85 Q_Q(QTabBar);
86 // Extend the unified title and toolbar area to cover the tab bar iff
87 // 1) the tab bar is in document mode
88 // 2) the tab bar is directly below an "unified" area.
89 // The extending itself is done in the Cocoa platform plugin and Mac style,
90 // this function registers geometry and visibility state for the tab bar.
91
92 // Calculate geometry
93 int upper, lower;
94 if (documentMode) {
95 QPoint windowPos = q->mapTo(q->window(), QPoint(0,0));
96 upper = windowPos.y();
97 int tabStripHeight = q->tabSizeHint(0).height();
98 int pixelTweak = -3;
99 lower = upper + tabStripHeight + pixelTweak;
100 } else {
101 upper = 0;
102 lower = 0;
103 }
104
105 QPlatformNativeInterface *nativeInterface = QGuiApplication::platformNativeInterface();
106 if (!nativeInterface)
107 return;
108 quintptr identifier = reinterpret_cast<quintptr>(q);
109
110 // Set geometry
111 QPlatformNativeInterface::NativeResourceForIntegrationFunction function =
112 nativeInterface->nativeResourceFunctionForIntegration("registerContentBorderArea");
113 if (!function)
114 return; // Not Cocoa platform plugin.
115 typedef void (*RegisterContentBorderAreaFunction)(QWindow *window, quintptr identifier, int upper, int lower);
116 (reinterpret_cast<RegisterContentBorderAreaFunction>(function))(q->window()->windowHandle(), identifier, upper, lower);
117
118 // Set visibility state
119 function = nativeInterface->nativeResourceFunctionForIntegration("setContentBorderAreaEnabled");
120 if (!function)
121 return;
122 typedef void (*SetContentBorderAreaEnabledFunction)(QWindow *window, quintptr identifier, bool enable);
123 (reinterpret_cast<SetContentBorderAreaEnabledFunction>(function))(q->window()->windowHandle(), identifier, q->isVisible());
124#endif
125}

133void QTabBarPrivate::initBasicStyleOption(QStyleOptionTab *option, int tabIndex) const
134{
135 Q_Q(const QTabBar);
136 const int totalTabs = tabList.size();
137
138 if (!option || (tabIndex < 0 || tabIndex >= totalTabs))
139 return;
140
141 const QTabBarPrivate::Tab &tab = *tabList.at(i: tabIndex);
142 option->initFrom(w: q);
143 option->state &= ~(QStyle::State_HasFocus | QStyle::State_MouseOver);
144 option->rect = q->tabRect(index: tabIndex);
145 const bool isCurrent = tabIndex == currentIndex;
146 option->row = 0;
147 if (tabIndex == pressedIndex)
148 option->state |= QStyle::State_Sunken;
149 if (isCurrent)
150 option->state |= QStyle::State_Selected;
151 if (isCurrent && q->hasFocus())
152 option->state |= QStyle::State_HasFocus;
153 if (!tab.enabled)
154 option->state &= ~QStyle::State_Enabled;
155 if (q->isActiveWindow())
156 option->state |= QStyle::State_Active;
157 if (!dragInProgress && option->rect == hoverRect)
158 option->state |= QStyle::State_MouseOver;
159 option->shape = shape;
160 option->text = tab.text;
161
162 if (tab.textColor.isValid())
163 option->palette.setColor(acr: q->foregroundRole(), acolor: tab.textColor);
164 option->icon = tab.icon;
165 option->iconSize = q->iconSize(); // Will get the default value then.
166
167 option->leftButtonSize = tab.leftWidget ? tab.leftWidget->size() : QSize();
168 option->rightButtonSize = tab.rightWidget ? tab.rightWidget->size() : QSize();
169 option->documentMode = documentMode;
170
171 if (tabIndex > 0 && tabIndex - 1 == currentIndex)
172 option->selectedPosition = QStyleOptionTab::PreviousIsSelected;
173 else if (tabIndex + 1 < totalTabs && tabIndex + 1 == currentIndex)
174 option->selectedPosition = QStyleOptionTab::NextIsSelected;
175 else
176 option->selectedPosition = QStyleOptionTab::NotAdjacent;
177
178 const bool paintBeginning = (tabIndex == firstVisible) || (dragInProgress && tabIndex == pressedIndex + 1);
179 const bool paintEnd = (tabIndex == lastVisible) || (dragInProgress && tabIndex == pressedIndex - 1);
180 if (paintBeginning) {
181 if (paintEnd)
182 option->position = QStyleOptionTab::OnlyOneTab;
183 else
184 option->position = QStyleOptionTab::Beginning;
185 } else if (paintEnd) {
186 option->position = QStyleOptionTab::End;
187 } else {
188 option->position = QStyleOptionTab::Middle;
189 }
190
191#if QT_CONFIG(tabwidget)
192 if (const QTabWidget *tw = qobject_cast<const QTabWidget *>(object: q->parentWidget())) {
193 option->features |= QStyleOptionTab::HasFrame;
194 if (tw->cornerWidget(corner: Qt::TopLeftCorner) || tw->cornerWidget(corner: Qt::BottomLeftCorner))
195 option->cornerWidgets |= QStyleOptionTab::LeftCornerWidget;
196 if (tw->cornerWidget(corner: Qt::TopRightCorner) || tw->cornerWidget(corner: Qt::BottomRightCorner))
197 option->cornerWidgets |= QStyleOptionTab::RightCornerWidget;
198 }
199#endif
200 option->tabIndex = tabIndex;
201}

220/*!
221 \class QTabBar
222 \brief The QTabBar class provides a tab bar, e.g. for use in tabbed dialogs.
223
224 \ingroup basicwidgets
225 \inmodule QtWidgets
226
227 QTabBar is straightforward to use; it draws the tabs using one of
228 the predefined \l{QTabBar::Shape}{shapes}, and emits a
229 signal when a tab is selected. It can be subclassed to tailor the
230 look and feel. Qt also provides a ready-made \l{QTabWidget}.
231
232 Each tab has a tabText(), an optional tabIcon(), an optional
233 tabToolTip(), optional tabWhatsThis() and optional tabData().
234 The tabs's attributes can be changed with setTabText(), setTabIcon(),
235 setTabToolTip(), setTabWhatsThis and setTabData(). Each tabs can be
236 enabled or disabled individually with setTabEnabled().
237
238 Each tab can display text in a distinct color. The current text color
239 for a tab can be found with the tabTextColor() function. Set the text
240 color for a particular tab with setTabTextColor().
241
242 Tabs are added using addTab(), or inserted at particular positions
243 using insertTab(). The total number of tabs is given by
244 count(). Tabs can be removed from the tab bar with
245 removeTab(). Combining removeTab() and insertTab() allows you to
246 move tabs to different positions.
247
248 The \l shape property defines the tabs' appearance. The choice of
249 shape is a matter of taste, although tab dialogs (for preferences
250 and similar) invariably use \l RoundedNorth.
251 Tab controls in windows other than dialogs almost
252 always use either \l RoundedSouth or \l TriangularSouth. Many
253 spreadsheets and other tab controls in which all the pages are
254 essentially similar use \l TriangularSouth, whereas \l
255 RoundedSouth is used mostly when the pages are different (e.g. a
256 multi-page tool palette). The default in QTabBar is \l
257 RoundedNorth.
258
259 The most important part of QTabBar's API is the currentChanged()
260 signal. This is emitted whenever the current tab changes (even at
261 startup, when the current tab changes from 'none'). There is also
262 a slot, setCurrentIndex(), which can be used to select a tab
263 programmatically. The function currentIndex() returns the index of
264 the current tab, \l count holds the number of tabs.
265
266 QTabBar creates automatic mnemonic keys in the manner of QAbstractButton;
267 e.g. if a tab's label is "\&Graphics", Alt+G becomes a shortcut
268 key for switching to that tab.
269
270 The following virtual functions may need to be reimplemented in
271 order to tailor the look and feel or store extra data with each
272 tab:
273
274 \list
275 \li tabSizeHint() calcuates the size of a tab.
276 \li tabInserted() notifies that a new tab was added.
277 \li tabRemoved() notifies that a tab was removed.
278 \li tabLayoutChange() notifies that the tabs have been re-laid out.
279 \li paintEvent() paints all tabs.
280 \endlist
281
282 For subclasses, you might also need the tabRect() functions which
283 returns the visual geometry of a single tab.
284
285 \table 100%
286 \row \li \inlineimage fusion-tabbar.png Screenshot of a Fusion style tab bar
287 \li A tab bar shown in the \l{Qt Widget Gallery}{Fusion widget style}.
288 \row \li \inlineimage fusion-tabbar-truncated.png Screenshot of a truncated Fusion tab bar
289 \li A truncated tab bar shown in the Fusion widget style.
290 \endtable
291
292 \sa QTabWidget
293*/
294

313 the Excel spreadsheet, for example
314
315 \value TriangularWest Triangular tabs on the left of the pages.
316
317 \value TriangularEast Triangular tabs on the right of the pages.
318*/

413void QTabBarPrivate::layoutTabs()
414{
415 Q_Q(QTabBar);
416 layoutDirty = false;
417 QSize size = q->size();
418 int last, available;
419 int maxExtent;
420 bool vertTabs = verticalTabs(shape);
421 int tabChainIndex = 0;
422 int hiddenTabs = 0;
423
424 Qt::Alignment tabAlignment = Qt::Alignment(q->style()->styleHint(stylehint: QStyle::SH_TabBar_Alignment, opt: nullptr, widget: q));
425 QList<QLayoutStruct> tabChain(tabList.size() + 2);
426
427 // We put an empty item at the front and back and set its expansive attribute
428 // depending on tabAlignment and expanding.
429 tabChain[tabChainIndex].init();
430 tabChain[tabChainIndex].expansive = (!expanding)
431 && (tabAlignment != Qt::AlignLeft)
432 && (tabAlignment != Qt::AlignJustify);
433 tabChain[tabChainIndex].empty = true;
434 ++tabChainIndex;
435
436 // We now go through our list of tabs and set the minimum size and the size hint
437 // This will allow us to elide text if necessary. Since we don't set
438 // a maximum size, tabs will EXPAND to fill up the empty space.
439 // Since tab widget is rather *ahem* strict about keeping the geometry of the
440 // tab bar to its absolute minimum, this won't bleed through, but will show up
441 // if you use tab bar on its own (a.k.a. not a bug, but a feature).
442 // Update: if expanding is false, we DO set a maximum size to prevent the tabs
443 // being wider than necessary.
444 if (!vertTabs) {
445 int minx = 0;
446 int x = 0;
447 int maxHeight = 0;
448 for (int i = 0; i < tabList.size(); ++i) {
449 const auto tab = tabList.at(i);
450 if (!tab->visible) {
451 ++hiddenTabs;
452 continue;
453 }
454 QSize sz = q->tabSizeHint(index: i);
455 tab->maxRect = QRect(x, 0, sz.width(), sz.height());
456 x += sz.width();
457 maxHeight = qMax(a: maxHeight, b: sz.height());
458 sz = q->minimumTabSizeHint(index: i);
459 tab->minRect = QRect(minx, 0, sz.width(), sz.height());
460 minx += sz.width();
461 tabChain[tabChainIndex].init();
462 tabChain[tabChainIndex].sizeHint = tab->maxRect.width();
463 tabChain[tabChainIndex].minimumSize = sz.width();
464 tabChain[tabChainIndex].empty = false;
465 tabChain[tabChainIndex].expansive = true;
466
467 if (!expanding)
468 tabChain[tabChainIndex].maximumSize = tabChain[tabChainIndex].sizeHint;
469 ++tabChainIndex;
470 }
471
472 last = minx;
473 available = size.width();
474 maxExtent = maxHeight;
475 } else {
476 int miny = 0;
477 int y = 0;
478 int maxWidth = 0;
479 for (int i = 0; i < tabList.size(); ++i) {
480 auto tab = tabList.at(i);
481 if (!tab->visible) {
482 ++hiddenTabs;
483 continue;
484 }
485 QSize sz = q->tabSizeHint(index: i);
486 tab->maxRect = QRect(0, y, sz.width(), sz.height());
487 y += sz.height();
488 maxWidth = qMax(a: maxWidth, b: sz.width());
489 sz = q->minimumTabSizeHint(index: i);
490 tab->minRect = QRect(0, miny, sz.width(), sz.height());
491 miny += sz.height();
492 tabChain[tabChainIndex].init();
493 tabChain[tabChainIndex].sizeHint = tab->maxRect.height();
494 tabChain[tabChainIndex].minimumSize = sz.height();
495 tabChain[tabChainIndex].empty = false;
496 tabChain[tabChainIndex].expansive = true;
497
498 if (!expanding)
499 tabChain[tabChainIndex].maximumSize = tabChain[tabChainIndex].sizeHint;
500 ++tabChainIndex;
501 }
502
503 last = miny;
504 available = size.height();
505 maxExtent = maxWidth;
506 }
507
508 // Mirror our front item.
509 tabChain[tabChainIndex].init();
510 tabChain[tabChainIndex].expansive = (!expanding)
511 && (tabAlignment != Qt::AlignRight)
512 && (tabAlignment != Qt::AlignJustify);
513 tabChain[tabChainIndex].empty = true;
514 Q_ASSERT(tabChainIndex == tabChain.size() - 1 - hiddenTabs); // add an assert just to make sure.
515
516 // Do the calculation
517 qGeomCalc(chain&: tabChain, start: 0, count: tabChain.size(), pos: 0, space: qMax(a: available, b: last), spacer: 0);
518
519 // Use the results
520 hiddenTabs = 0;
521 for (int i = 0; i < tabList.size(); ++i) {
522 auto tab = tabList.at(i);
523 if (!tab->visible) {
524 tab->rect = QRect();
525 ++hiddenTabs;
526 continue;
527 }
528 const QLayoutStruct &lstruct = tabChain.at(i: i + 1 - hiddenTabs);
529 if (!vertTabs)
530 tab->rect.setRect(ax: lstruct.pos, ay: 0, aw: lstruct.size, ah: maxExtent);
531 else
532 tab->rect.setRect(ax: 0, ay: lstruct.pos, aw: maxExtent, ah: lstruct.size);
533 }
534
535 if (useScrollButtons && tabList.size() && last > available) {
536 const QRect scrollRect = normalizedScrollRect(index: 0);
537 scrollOffset = -scrollRect.left();
538
539 Q_Q(QTabBar);
540 QStyleOption opt;
541 opt.initFrom(w: q);
542 QRect scrollButtonLeftRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarScrollLeftButton, option: &opt, widget: q);
543 QRect scrollButtonRightRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarScrollRightButton, option: &opt, widget: q);
544 int scrollButtonWidth = q->style()->pixelMetric(metric: QStyle::PM_TabBarScrollButtonWidth, option: &opt, widget: q);
545
546 // Normally SE_TabBarScrollLeftButton should have the same width as PM_TabBarScrollButtonWidth.
547 // But if that is not the case, we set the actual button width to PM_TabBarScrollButtonWidth, and
548 // use the extra space from SE_TabBarScrollLeftButton as margins towards the tabs.
549 if (vertTabs) {
550 scrollButtonLeftRect.setHeight(scrollButtonWidth);
551 scrollButtonRightRect.setY(scrollButtonRightRect.bottom() + 1 - scrollButtonWidth);
552 scrollButtonRightRect.setHeight(scrollButtonWidth);
553 leftB->setArrowType(Qt::UpArrow);
554 rightB->setArrowType(Qt::DownArrow);
555 } else if (q->layoutDirection() == Qt::RightToLeft) {
556 scrollButtonRightRect.setWidth(scrollButtonWidth);
557 scrollButtonLeftRect.setX(scrollButtonLeftRect.right() + 1 - scrollButtonWidth);
558 scrollButtonLeftRect.setWidth(scrollButtonWidth);
559 leftB->setArrowType(Qt::RightArrow);
560 rightB->setArrowType(Qt::LeftArrow);
561 } else {
562 scrollButtonLeftRect.setWidth(scrollButtonWidth);
563 scrollButtonRightRect.setX(scrollButtonRightRect.right() + 1 - scrollButtonWidth);
564 scrollButtonRightRect.setWidth(scrollButtonWidth);
565 leftB->setArrowType(Qt::LeftArrow);
566 rightB->setArrowType(Qt::RightArrow);
567 }
568
569 leftB->setGeometry(scrollButtonLeftRect);
570 leftB->setEnabled(false);
571 leftB->show();
572
573 rightB->setGeometry(scrollButtonRightRect);
574 rightB->setEnabled(last - scrollOffset > scrollRect.x() + scrollRect.width());
575 rightB->show();
576 } else {
577 scrollOffset = 0;
578 rightB->hide();
579 leftB->hide();
580 }
581
582 layoutWidgets();
583 q->tabLayoutChange();
584}

586QRect QTabBarPrivate::normalizedScrollRect(int index)
587{
588 // "Normalized scroll rect" means return the free space on the tab bar
589 // that doesn't overlap with scroll buttons or tear indicators, and
590 // always return the rect as horizontal Qt::LeftToRight, even if the
591 // tab bar itself is in a different orientation.
592
593 Q_Q(QTabBar);
594 QStyleOptionTab opt;
595 q->initStyleOption(option: &opt, tabIndex: currentIndex);
596 opt.rect = q->rect();
597
598 QRect scrollButtonLeftRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarScrollLeftButton, option: &opt, widget: q);
599 QRect scrollButtonRightRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarScrollRightButton, option: &opt, widget: q);
600 QRect tearLeftRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarTearIndicatorLeft, option: &opt, widget: q);
601 QRect tearRightRect = q->style()->subElementRect(subElement: QStyle::SE_TabBarTearIndicatorRight, option: &opt, widget: q);
602
603 if (verticalTabs(shape)) {
604 int topEdge, bottomEdge;
605 bool leftButtonIsOnTop = scrollButtonLeftRect.y() < q->height() / 2;
606 bool rightButtonIsOnTop = scrollButtonRightRect.y() < q->height() / 2;
607
608 if (leftButtonIsOnTop && rightButtonIsOnTop) {
609 topEdge = scrollButtonRightRect.bottom() + 1;
610 bottomEdge = q->height();
611 } else if (!leftButtonIsOnTop && !rightButtonIsOnTop) {
612 topEdge = 0;
613 bottomEdge = scrollButtonLeftRect.top();
614 } else {
615 topEdge = scrollButtonLeftRect.bottom() + 1;
616 bottomEdge = scrollButtonRightRect.top();
617 }
618
619 bool tearTopVisible = index != 0 && topEdge != -scrollOffset;
620 bool tearBottomVisible = index != tabList.size() - 1 && bottomEdge != tabList.constLast()->rect.bottom() + 1 - scrollOffset;
621 if (tearTopVisible && !tearLeftRect.isNull())
622 topEdge = tearLeftRect.bottom() + 1;
623 if (tearBottomVisible && !tearRightRect.isNull())
624 bottomEdge = tearRightRect.top();
625
626 return QRect(topEdge, 0, bottomEdge - topEdge, q->height());
627 } else {
628 if (q->layoutDirection() == Qt::RightToLeft) {
629 scrollButtonLeftRect = QStyle::visualRect(direction: Qt::RightToLeft, boundingRect: q->rect(), logicalRect: scrollButtonLeftRect);
630 scrollButtonRightRect = QStyle::visualRect(direction: Qt::RightToLeft, boundingRect: q->rect(), logicalRect: scrollButtonRightRect);
631 tearLeftRect = QStyle::visualRect(direction: Qt::RightToLeft, boundingRect: q->rect(), logicalRect: tearLeftRect);
632 tearRightRect = QStyle::visualRect(direction: Qt::RightToLeft, boundingRect: q->rect(), logicalRect: tearRightRect);
633 }
634
635 int leftEdge, rightEdge;
636 bool leftButtonIsOnLeftSide = scrollButtonLeftRect.x() < q->width() / 2;
637 bool rightButtonIsOnLeftSide = scrollButtonRightRect.x() < q->width() / 2;
638
639 if (leftButtonIsOnLeftSide && rightButtonIsOnLeftSide) {
640 leftEdge = scrollButtonRightRect.right() + 1;
641 rightEdge = q->width();
642 } else if (!leftButtonIsOnLeftSide && !rightButtonIsOnLeftSide) {
643 leftEdge = 0;
644 rightEdge = scrollButtonLeftRect.left();
645 } else {
646 leftEdge = scrollButtonLeftRect.right() + 1;
647 rightEdge = scrollButtonRightRect.left();
648 }
649
650 bool tearLeftVisible = index != 0 && leftEdge != -scrollOffset;
651 bool tearRightVisible = index != tabList.size() - 1 && rightEdge != tabList.constLast()->rect.right() + 1 - scrollOffset;
652 if (tearLeftVisible && !tearLeftRect.isNull())
653 leftEdge = tearLeftRect.right() + 1;
654 if (tearRightVisible && !tearRightRect.isNull())
655 rightEdge = tearRightRect.left();
656
657 return QRect(leftEdge, 0, rightEdge - leftEdge, q->height());
658 }

670void QTabBarPrivate::makeVisible(int index)
671{
672 Q_Q(QTabBar);
673 if (!validIndex(index) || leftB->isHidden())
674 return;
675
676 const QRect tabRect = tabList.at(i: index)->rect;
677 const int oldScrollOffset = scrollOffset;
678 const bool horiz = !verticalTabs(shape);
679 const int tabStart = horiz ? tabRect.left() : tabRect.top();
680 const int tabEnd = horiz ? tabRect.right() : tabRect.bottom();
681 const int lastTabEnd = horiz ? tabList.constLast()->rect.right() : tabList.constLast()->rect.bottom();
682 const QRect scrollRect = normalizedScrollRect(index);
683 const int scrolledTabBarStart = qMax(a: 1, b: scrollRect.left() + scrollOffset);
684 const int scrolledTabBarEnd = qMin(a: lastTabEnd - 1, b: scrollRect.right() + scrollOffset);
685
686 if (tabStart < scrolledTabBarStart) {
687 // Tab is outside on the left, so scroll left.
688 scrollOffset = tabStart - scrollRect.left();
689 } else if (tabEnd > scrolledTabBarEnd) {
690 // Tab is outside on the right, so scroll right.
691 scrollOffset = tabEnd - scrollRect.right();
692 }
693
694 leftB->setEnabled(scrollOffset > -scrollRect.left());
695 rightB->setEnabled(scrollOffset < lastTabEnd - scrollRect.right());
696
697 if (oldScrollOffset != scrollOffset) {
698 q->update();
699 layoutWidgets();
700 }

937/*!\overload
938
939 Inserts a new tab with icon \a icon and text \a text at position
940 \a index. If \a index is out of range, the new tab is
941 appended. Returns the new tab's index.
942
943 If the QTabBar was empty before this function is called, the inserted tab
944 becomes the current tab.
945
946 Inserting a new tab at an index less than or equal to the current index
947 will increment the current index, but keep the current tab.
948*/
949int QTabBar::insertTab(int index, const QIcon& icon, const QString &text)
950{
951 Q_D(QTabBar);
952 if (!d->validIndex(index)) {
953 index = d->tabList.size();
954 d->tabList.append(t: new QTabBarPrivate::Tab(icon, text));
955 } else {
956 d->tabList.insert(i: index, t: new QTabBarPrivate::Tab(icon, text));
957 }
958#ifndef QT_NO_SHORTCUT
959 d->tabList.at(i: index)->shortcutId = grabShortcut(key: QKeySequence::mnemonic(text));
960#endif
961 d->firstVisible = qMax(a: qMin(a: index, b: d->firstVisible), b: 0);
962 d->refresh();
963 if (d->tabList.size() == 1)
964 setCurrentIndex(index);
965 else if (index <= d->currentIndex)
966 ++d->currentIndex;
967
968 if (index <= d->lastVisible)
969 ++d->lastVisible;
970 else
971 d->lastVisible = index;
972
973 if (d->closeButtonOnTabs) {
974 QStyleOptionTab opt;
975 initStyleOption(option: &opt, tabIndex: index);
976 ButtonPosition closeSide = (ButtonPosition)style()->styleHint(stylehint: QStyle::SH_TabBar_CloseButtonPosition, opt: nullptr, widget: this);
977 QAbstractButton *closeButton = new CloseButton(this);
978 connect(sender: closeButton, SIGNAL(clicked()), receiver: this, SLOT(_q_closeTab()));
979 setTabButton(index, position: closeSide, widget: closeButton);
980 }
981
982 for (const auto tab : std::as_const(t&: d->tabList)) {
983 if (tab->lastTab >= index)
984 ++tab->lastTab;
985 }
986
987 if (tabAt(pos: d->mousePosition) == index) {
988 d->hoverIndex = index;
989 d->hoverRect = tabRect(index);
990 }
991

993 d->autoHideTabs();

995}

998/*!
999 Removes the tab at position \a index.
1000
1001 \sa SelectionBehavior
1002 */
1003void QTabBar::removeTab(int index)
1004{
1005 Q_D(QTabBar);
1006 if (d->validIndex(index)) {
1007 auto removedTab = d->tabList.at(i: index);
1008 if (d->dragInProgress)
1009 d->moveTabFinished(index: d->pressedIndex);
1010
1011#ifndef QT_NO_SHORTCUT
1012 releaseShortcut(id: d->tabList.at(i: index)->shortcutId);
1013#endif
1014 if (removedTab->leftWidget) {
1015 removedTab->leftWidget->hide();
1016 removedTab->leftWidget->deleteLater();
1017 removedTab->leftWidget = nullptr;
1018 }
1019 if (removedTab->rightWidget) {
1020 removedTab->rightWidget->hide();
1021 removedTab->rightWidget->deleteLater();
1022 removedTab->rightWidget = nullptr;
1023 }
1024
1025 int newIndex = removedTab->lastTab;
1026 d->tabList.removeAt(i: index);
1027 delete removedTab;
1028 for (auto tab : std::as_const(t&: d->tabList)) {
1029 if (tab->lastTab == index)
1030 tab->lastTab = -1;
1031 if (tab->lastTab > index)
1032 --tab->lastTab;
1033 }
1034
1035 d->calculateFirstLastVisible(index, visible: false, remove: true);
1036
1037 if (index == d->currentIndex) {
1038 // The current tab is going away, in order to make sure
1039 // we emit that "current has changed", we need to reset this
1040 // around.
1041 d->currentIndex = -1;
1042 if (d->tabList.size() > 0) {
1043 switch(d->selectionBehaviorOnRemove) {
1044 case SelectPreviousTab:
1045 if (newIndex > index)
1046 newIndex--;
1047 if (d->validIndex(index: newIndex) && d->tabList.at(i: newIndex)->visible)
1048 break;
1049 Q_FALLTHROUGH();
1050 case SelectRightTab:
1051 newIndex = qBound(min: d->firstVisible, val: index, max: d->lastVisible);
1052 break;
1053 case SelectLeftTab:
1054 newIndex = qBound(min: d->firstVisible, val: index-1, max: d->lastVisible);
1055 break;
1056 default:
1057 break;
1058 }
1059
1060 if (d->validIndex(index: newIndex)) {
1061 // don't loose newIndex's old through setCurrentIndex
1062 int bump = d->tabList.at(i: newIndex)->lastTab;
1063 setCurrentIndex(newIndex);
1064 d->tabList.at(i: newIndex)->lastTab = bump;
1065 } else {
1066 // we had a valid current index, but there are no visible tabs left
1067 emit currentChanged(index: -1);
1068 }
1069 } else {
1070 emit currentChanged(index: -1);
1071 }
1072 } else if (index < d->currentIndex) {
1073 setCurrentIndex(d->currentIndex - 1);
1074 }
1075 d->refresh();
1076 d->autoHideTabs();
1077 if (d->hoverRect.isValid()) {
1078 update(d->hoverRect);
1079 d->hoverIndex = tabAt(pos: d->mousePosition);
1080 if (d->validIndex(index: d->hoverIndex)) {
1081 d->hoverRect = tabRect(index: d->hoverIndex);
1082 update(d->hoverRect);
1083 } else {
1084 d->hoverRect = QRect();
1085 }
1086 }
1087 tabRemoved(index);
1088 }
1089}

1363{

1366 && tabRect(index: d->currentIndex).contains(p: position)) {
1367 return d->currentIndex;

1986{
1987 Q_D(QTabBar);
1988 if (from == to
1989 || !d->validIndex(index: from)
1990 || !d->validIndex(index: to))
1991 return;

1993 auto &fromTab = *d->tabList.at(i: from);

1997 int oldPressedPosition = 0;
1998 if (d->pressedIndex != -1) {

2000 oldPressedPosition = vertical ? d->tabList.at(i: d->pressedIndex)->rect.y()
2001 : d->tabList.at(i: d->pressedIndex)->rect.x();

2018 tab.rect.moveLeft(pos: tab.rect.x() + width);

2021 direction *= -1;

2023 tab.dragOffset += (direction * width);

2027 if (from < to)
2028 fromTab.rect.moveTop(pos: toTab.rect.bottom() + 1);
2029 else
2030 fromTab.rect.moveTop(pos: toTab.rect.top() - width);
2031 } else {
