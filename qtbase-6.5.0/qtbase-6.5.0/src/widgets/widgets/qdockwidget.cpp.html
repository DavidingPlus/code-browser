148QSize QDockWidgetTitleButton::dockButtonIconSize() const
149{
150 if (m_iconSize < 0) {
151 m_iconSize = style()->pixelMetric(metric: QStyle::PM_SmallIconSize, option: nullptr, widget: this);
152 // Dock Widget title buttons on Windows where historically limited to size 10
153 // (from small icon size 16) since only a 10x10 XPM was provided.
154 // Adding larger pixmaps to the icons thus caused the icons to grow; limit
155 // this to qpiScaled(10) here.
156 if (isWindowsStyle(style: style()))
157 m_iconSize = qMin(a: (10 * logicalDpiX()) / 96, b: m_iconSize);
158 }

212 opt.arrowType = Qt::NoArrow;

317QSize QDockWidgetLayout::sizeFromContent(const QSize &content, bool floating) const
318{
319 QSize result = content;
320
321 if (verticalTitleBar) {
322 result.setHeight(qMax(a: result.height(), b: minimumTitleWidth()));
323 result.setWidth(qMax(a: content.width(), b: 0));
324 } else {
325 result.setHeight(qMax(a: result.height(), b: 0));
326 result.setWidth(qMax(a: content.width(), b: minimumTitleWidth()));
327 }
328
329 QDockWidget *w = qobject_cast<QDockWidget*>(object: parentWidget());
330 const bool nativeDeco = nativeWindowDeco(floating);
331
332 int fw = floating && !nativeDeco
333 ? w->style()->pixelMetric(metric: QStyle::PM_DockWidgetFrameWidth, option: nullptr, widget: w)
334 : 0;
335
336 const int th = titleHeight();
337 if (!nativeDeco) {
338 if (verticalTitleBar)
339 result += QSize(th + 2*fw, 2*fw);
340 else
341 result += QSize(2*fw, th + 2*fw);
342 }
343
344 result.setHeight(qMin(a: result.height(), b: (int) QWIDGETSIZE_MAX));
345 result.setWidth(qMin(a: result.width(), b: (int) QWIDGETSIZE_MAX));
346
347 if (content.width() < 0)
348 result.setWidth(-1);
349 if (content.height() < 0)
350 result.setHeight(-1);
351
352 const QMargins margins = w->contentsMargins();
353 //we need to subtract the contents margin (it will be added by the caller)
354 QSize min = w->minimumSize().shrunkBy(m: margins);
355 QSize max = w->maximumSize().shrunkBy(m: margins);
356
357 /* A floating dockwidget will automatically get its minimumSize set to the layout's
358 minimum size + deco. We're *not* interested in this, we only take minimumSize()
359 into account if the user set it herself. Otherwise we end up expanding the result
360 of a calculation for a non-floating dock widget to a floating dock widget's
361 minimum size + window decorations. */
362
363 uint explicitMin = 0;
364 uint explicitMax = 0;
365 if (w->d_func()->extra != nullptr) {
366 explicitMin = w->d_func()->extra->explicitMinSize;
367 explicitMax = w->d_func()->extra->explicitMaxSize;
368 }
369
370 if (!(explicitMin & Qt::Horizontal) || min.width() == 0)
371 min.setWidth(-1);
372 if (!(explicitMin & Qt::Vertical) || min.height() == 0)
373 min.setHeight(-1);
374
375 if (!(explicitMax & Qt::Horizontal))
376 max.setWidth(QWIDGETSIZE_MAX);
377 if (!(explicitMax & Qt::Vertical))
378 max.setHeight(QWIDGETSIZE_MAX);
379
380 return result.boundedTo(otherSize: max).expandedTo(otherSize: min);
381}

456int QDockWidgetLayout::minimumTitleWidth() const
457{
458 QDockWidget *q = qobject_cast<QDockWidget*>(object: parentWidget());
459
460 if (QWidget *title = widgetForRole(r: TitleBar))
461 return pick(vertical: verticalTitleBar, size: title->minimumSizeHint());
462
463 QSize closeSize(0, 0);
464 QSize floatSize(0, 0);
465 if (hasFeature(dockwidget: q, feature: QDockWidget::DockWidgetClosable)) {
466 if (QLayoutItem *item = item_list[CloseButton])
467 closeSize = item->widget()->sizeHint();
468 }
469 if (hasFeature(dockwidget: q, feature: QDockWidget::DockWidgetFloatable)) {
470 if (QLayoutItem *item = item_list[FloatButton])
471 floatSize = item->widget()->sizeHint();
472 }
473
474 int titleHeight = this->titleHeight();
475
476 int mw = q->style()->pixelMetric(metric: QStyle::PM_DockWidgetTitleMargin, option: nullptr, widget: q);
477 int fw = q->style()->pixelMetric(metric: QStyle::PM_DockWidgetFrameWidth, option: nullptr, widget: q);
478
479 return pick(vertical: verticalTitleBar, size: closeSize)
480 + pick(vertical: verticalTitleBar, size: floatSize)
481 + titleHeight + 2*fw + 3*mw;
482}

484int QDockWidgetLayout::titleHeight() const
485{
486 QDockWidget *q = qobject_cast<QDockWidget*>(object: parentWidget());
487
488 if (QWidget *title = widgetForRole(r: TitleBar))
489 return perp(vertical: verticalTitleBar, size: title->sizeHint());
490
491 QSize closeSize(0, 0);
492 QSize floatSize(0, 0);
493 if (QLayoutItem *item = item_list[CloseButton])
494 closeSize = item->widget()->sizeHint();
495 if (QLayoutItem *item = item_list[FloatButton])
496 floatSize = item->widget()->sizeHint();
497
498 int buttonHeight = qMax(a: perp(vertical: verticalTitleBar, size: closeSize),
499 b: perp(vertical: verticalTitleBar, size: floatSize));
500
501 QFontMetrics titleFontMetrics = q->fontMetrics();
502 int mw = q->style()->pixelMetric(metric: QStyle::PM_DockWidgetTitleMargin, option: nullptr, widget: q);
503
504 return qMax(a: buttonHeight + 2, b: titleFontMetrics.height() + 2*mw);

507void QDockWidgetLayout::setGeometry(const QRect &geometry)
508{
509 QDockWidget *q = qobject_cast<QDockWidget*>(object: parentWidget());
510
511 bool nativeDeco = nativeWindowDeco();
512
513 int fw = q->isFloating() && !nativeDeco
514 ? q->style()->pixelMetric(metric: QStyle::PM_DockWidgetFrameWidth, option: nullptr, widget: q)
515 : 0;
516
517 if (nativeDeco) {
518 if (QLayoutItem *item = item_list[Content])
519 item->setGeometry(geometry);
520 } else {
521 int titleHeight = this->titleHeight();
522
523 if (verticalTitleBar) {
524 _titleArea = QRect(QPoint(fw, fw),
525 QSize(titleHeight, geometry.height() - (fw * 2)));
526 } else {
527 _titleArea = QRect(QPoint(fw, fw),
528 QSize(geometry.width() - (fw * 2), titleHeight));
529 }
530
531 if (QLayoutItem *item = item_list[TitleBar]) {
532 item->setGeometry(_titleArea);
533 } else {
534 QStyleOptionDockWidget opt;
535 q->initStyleOption(option: &opt);
536
537 if (QLayoutItem *item = item_list[CloseButton]) {
538 if (!item->isEmpty()) {
539 QRect r = q->style()
540 ->subElementRect(subElement: QStyle::SE_DockWidgetCloseButton,
541 option: &opt, widget: q);
542 if (!r.isNull())
543 item->setGeometry(r);
544 }
545 }
546
547 if (QLayoutItem *item = item_list[FloatButton]) {
548 if (!item->isEmpty()) {
549 QRect r = q->style()
550 ->subElementRect(subElement: QStyle::SE_DockWidgetFloatButton,
551 option: &opt, widget: q);
552 if (!r.isNull())
553 item->setGeometry(r);
554 }
555 }
556 }
557
558 if (QLayoutItem *item = item_list[Content]) {
559 QRect r = geometry;
560 if (verticalTitleBar) {
561 r.setLeft(_titleArea.right() + 1);
562 r.adjust(dx1: 0, dy1: fw, dx2: -fw, dy2: -fw);
563 } else {
564 r.setTop(_titleArea.bottom() + 1);
565 r.adjust(dx1: fw, dy1: 0, dx2: -fw, dy2: -fw);
566 }
567 item->setGeometry(r);
568 }
569 }

693void QDockWidgetPrivate::updateButtons()
694{
695 Q_Q(QDockWidget);
696 QDockWidgetLayout *dwLayout = qobject_cast<QDockWidgetLayout*>(object: layout);
697
698 QStyleOptionDockWidget opt;
699 q->initStyleOption(option: &opt);
700
701 bool customTitleBar = dwLayout->widgetForRole(r: QDockWidgetLayout::TitleBar) != nullptr;
702 bool nativeDeco = dwLayout->nativeWindowDeco();
703 bool hideButtons = nativeDeco || customTitleBar;
704
705 bool canClose = hasFeature(priv: this, feature: QDockWidget::DockWidgetClosable);
706 bool canFloat = hasFeature(priv: this, feature: QDockWidget::DockWidgetFloatable);
707
708 QAbstractButton *button
709 = qobject_cast<QAbstractButton*>(object: dwLayout->widgetForRole(r: QDockWidgetLayout::FloatButton));
710 button->setIcon(q->style()->standardIcon(standardIcon: QStyle::SP_TitleBarNormalButton, option: &opt, widget: q));
711 button->setVisible(canFloat && !hideButtons);
712#if QT_CONFIG(accessibility)
713 //: Accessible name for button undocking a dock widget (floating state)
714 button->setAccessibleName(QDockWidget::tr(s: "Float"));
715 button->setAccessibleDescription(QDockWidget::tr(s: "Undocks and re-attaches the dock widget"));

801/*! \internal
802 Ends the drag end drop operation of the QDockWidget.
803 The \a abort parameter specifies that it ends because of programmatic state
804 reset rather than mouse release event.
805 */
806void QDockWidgetPrivate::endDrag(bool abort)
807{
808 Q_Q(QDockWidget);
809 Q_ASSERT(state != nullptr);
810
811 q->releaseMouse();
812
813 if (state->dragging) {
814 const QMainWindow *mainWindow = mainwindow_from_dock(dock: q);
815 Q_ASSERT(mainWindow != nullptr);
816 QMainWindowLayout *mwLayout = qt_mainwindow_layout(window: mainWindow);
817
818 if (abort || !mwLayout->plug(widgetItem: state->widgetItem)) {
819 if (hasFeature(priv: this, feature: QDockWidget::DockWidgetFloatable)) {
820 // This QDockWidget will now stay in the floating state.
821 if (state->ownWidgetItem) {
822 delete state->widgetItem;
823 state->widgetItem = nullptr;
824 }
825 mwLayout->restore();
826 QDockWidgetLayout *dwLayout = qobject_cast<QDockWidgetLayout*>(object: layout);
827 if (!dwLayout->nativeWindowDeco()) {
828 // get rid of the X11BypassWindowManager window flag and activate the resizer
829 Qt::WindowFlags flags = q->windowFlags();
830 flags &= ~Qt::X11BypassWindowManagerHint;
831 q->setWindowFlags(flags);
832 setResizerActive(q->isFloating());
833 q->show();
834 } else {
835 setResizerActive(false);
836 }
837 if (q->isFloating()) { // Might not be floating when dragging a QDockWidgetGroupWindow
838 undockedGeometry = q->geometry();
839#if QT_CONFIG(tabwidget)
840 // is the widget located within the mainwindow?
841 const Qt::DockWidgetArea area = mainWindow->dockWidgetArea(dockwidget: q);
842 if (area != Qt::NoDockWidgetArea) {
843 tabPosition = mwLayout->tabPosition(area);
844 } else if (auto dwgw = qobject_cast<QDockWidgetGroupWindow *>(object: q->parent())) {
845 // DockWidget wasn't found in one of the docks within mainwindow
846 // => derive tabPosition from parent
847 tabPosition = mwLayout->tabPosition(area: toDockWidgetArea(pos: dwgw->layoutInfo()->dockPos));
848 }
849#endif
850 }
851 q->activateWindow();
852 } else {
853 // The tab was not plugged back in the QMainWindow but the QDockWidget cannot
854 // stay floating, revert to the previous state.
855 mwLayout->revert(widgetItem: state->widgetItem);
856 }
857 }
858 }
859 delete state;
860 state = nullptr;
861}

947bool QDockWidgetPrivate::mouseMoveEvent(QMouseEvent *event)
948{
949 bool ret = false;
950#if QT_CONFIG(mainwindow)
951 Q_Q(QDockWidget);
952
953 if (!state)
954 return ret;
955
956 QDockWidgetLayout *dwlayout
957 = qobject_cast<QDockWidgetLayout *>(object: layout);
958 QMainWindowLayout *mwlayout = qt_mainwindow_layout_from_dock(dock: q);
959 if (!dwlayout->nativeWindowDeco()) {
960 if (!state->dragging
961 && mwlayout->pluggingWidget == nullptr
962 && (event->position().toPoint() - state->pressPos).manhattanLength()
963 > QApplication::startDragDistance()) {
964
965#ifdef Q_OS_MACOS
966 if (windowHandle() && !q->isFloating()) {
967 // When using native widgets on mac, we have not yet been successful in
968 // starting a drag on an NSView that belongs to one window (QMainWindow),
969 // but continue the drag on another (QDockWidget). This is what happens if
970 // we try to make this widget floating during a drag. So as a fall back
971 // solution, we simply make this widget floating instead, when we would
972 // otherwise start a drag.
973 q->setFloating(true);
974 } else
975#endif
976 {
977 startDrag();
978 q->grabMouse();
979 ret = true;
980 }
981 }
982 }
983
984 if (state && state->dragging && !state->nca) {
985 QMargins windowMargins = q->window()->windowHandle()->frameMargins();
986 QPoint windowMarginOffset = QPoint(windowMargins.left(), windowMargins.top());
987
988 // TODO maybe use QScreen API (if/when available) to simplify the below code.
989 const QScreen *orgWdgScreen = QGuiApplication::screenAt(point: state->widgetInitialPos);
990 const QScreen *screenFrom = QGuiApplication::screenAt(point: state->globalPressPos);
991 const QScreen *screenTo = QGuiApplication::screenAt(point: event->globalPosition().toPoint());
992 const QScreen *wdgScreen = q->screen();
993
994 QPoint pos;
995 if (Q_LIKELY(screenFrom && screenTo && wdgScreen && orgWdgScreen)) {
996 const QPoint nativeWdgOrgPos = QHighDpiScaling::mapPositionToNative(
997 pos: state->widgetInitialPos, platformScreen: orgWdgScreen->handle());
998 const QPoint nativeTo = QHighDpiScaling::mapPositionToNative(
999 pos: event->globalPosition().toPoint(), platformScreen: screenTo->handle());
1000 const QPoint nativeFrom = QHighDpiScaling::mapPositionToNative(pos: state->globalPressPos,
1001 platformScreen: screenFrom->handle());
1002
1003 // Calculate new nativePos based on startPos + mouse delta move.
1004 const QPoint nativeNewPos = nativeWdgOrgPos + (nativeTo - nativeFrom);
1005 pos = QHighDpiScaling::mapPositionFromNative(pos: nativeNewPos, platformScreen: wdgScreen->handle())
1006 - windowMarginOffset;
1007 } else {
1008 // Fallback in the unlikely case that source and target screens could not be established
1009 qCDebug(lcQpaDockWidgets)
1010 << "QDockWidget failed to find relevant screen info. screenFrom:" << screenFrom
1011 << "screenTo:" << screenTo << " wdgScreen:" << wdgScreen << "orgWdgScreen"
1012 << orgWdgScreen;
1013 pos = event->globalPosition().toPoint() - state->pressPos - windowMarginOffset;
1014 }
1015
1016 // If the newly floating dock widget has got a native title bar,
1017 // offset the position by the native title bar's height or width
1018 const int dx = q->geometry().x() - q->x();
1019 const int dy = q->geometry().y() - q->y();
1020 pos.rx() += dx;
1021 pos.ry() += dy;
1022
1023 QDockWidgetGroupWindow *floatingTab = qobject_cast<QDockWidgetGroupWindow*>(object: parent);
1024 if (floatingTab && !q->isFloating())
1025 floatingTab->move(pos);
1026 else
1027 q->move(pos);
1028 if (state && !state->ctrlDrag)
1029 mwlayout->hover(hoverTarget: state->widgetItem, mousePos: event->globalPosition().toPoint());
1030
1031 ret = true;
1032 }
1033
1034#endif // QT_CONFIG(mainwindow)
1035 return ret;
1036}

1052{
1053 Q_Q(QDockWidget);
1054
1055 int fw = q->style()->pixelMetric(metric: QStyle::PM_DockWidgetFrameWidth, option: nullptr, widget: q);
1056
1057 QWidget *tl = q->topLevelWidget();
1058 QRect geo = tl->geometry();

1061 titleRect.setLeft(geo.left());
1062 titleRect.setRight(geo.right());
1063 titleRect.setBottom(geo.top() - 1);
1064 titleRect.adjust(dx1: 0, dy1: fw, dx2: 0, dy2: 0);
1065 }
1066

1068 case QEvent::NonClientAreaMouseButtonPress:
1069 if (!titleRect.contains(p: event->globalPosition().toPoint()))
1070 break;
1071 if (state != nullptr)
1072 break;
1073 if (qobject_cast<QMainWindow*>(object: parent) == nullptr && qobject_cast<QDockWidgetGroupWindow*>(object: parent) == nullptr)
1074 break;
1075 if (isAnimating())
1076 break;
1077 initDrag(pos: event->position().toPoint(), nca: true);
1078 if (state == nullptr)
1079 break;

1081 (!hasFeature(priv: this, feature: QDockWidget::DockWidgetMovable) && q->isFloating());
1082 startDrag();
1083 break;
1084 case QEvent::NonClientAreaMouseMove:
1085 if (state == nullptr || !state->dragging)

1089 if (state->nca) {

1093 break;
1094 case QEvent::NonClientAreaMouseButtonRelease:
1095#if defined(Q_OS_MAC) || defined(Q_OS_WASM)
1096 if (state)
1097 endDrag();
1098#endif
1099 break;
1100 case QEvent::NonClientAreaMouseButtonDblClick:
1101 _q_toggleTopLevel();
1102 break;
1103 default:
1104 break;
1105 }

1541/*! \reimp */
1542bool QDockWidget::event(QEvent *event)
1543{
1544 Q_D(QDockWidget);
1545
1546 QMainWindow *win = qobject_cast<QMainWindow*>(object: parentWidget());
1547 QMainWindowLayout *layout = qt_mainwindow_layout_from_dock(dock: this);
1548
1549 switch (event->type()) {
1550#ifndef QT_NO_ACTION
1551 case QEvent::Hide:
1552 if (layout != nullptr)
1553 layout->keepSize(w: this);
1554 d->toggleViewAction->setChecked(false);
1555 emit visibilityChanged(visible: false);
1556 break;
1557 case QEvent::Show: {
1558 d->toggleViewAction->setChecked(true);
1559 QPoint parentTopLeft(0, 0);
1560 if (isWindow()) {
1561 const QScreen *screen = d->associatedScreen();
1562 parentTopLeft = screen
1563 ? screen->availableVirtualGeometry().topLeft()
1564 : QGuiApplication::primaryScreen()->availableVirtualGeometry().topLeft();
1565 }
1566 emit visibilityChanged(visible: geometry().right() >= parentTopLeft.x() && geometry().bottom() >= parentTopLeft.y());
1567}
1568 break;
1569#endif
1570 case QEvent::ApplicationLayoutDirectionChange:
1571 case QEvent::LayoutDirectionChange:
1572 case QEvent::StyleChange:
1573 case QEvent::ParentChange:
1574 d->updateButtons();
1575 break;
1576 case QEvent::ZOrderChange: {
1577 bool onTop = false;
1578 if (win != nullptr) {
1579 const QObjectList &siblings = win->children();
1580 onTop = siblings.size() > 0 && siblings.last() == (QObject*)this;
1581 }
1582#if QT_CONFIG(tabbar)
1583 if (!isFloating() && layout != nullptr && onTop)
1584 layout->raise(widget: this);
1585#endif
1586 break;
1587 }
1588 case QEvent::WindowActivate:
1589 case QEvent::WindowDeactivate:
1590 update(qobject_cast<QDockWidgetLayout *>(object: this->layout())->titleArea());
1591 break;
1592 case QEvent::ContextMenu:
1593 if (d->state) {
1594 event->accept();
1595 return true;
1596 }
1597 break;
1598 // return true after calling the handler since we don't want
1599 // them to be passed onto the default handlers
1600 case QEvent::MouseButtonPress:
1601 if (d->mousePressEvent(event: static_cast<QMouseEvent *>(event)))
1602 return true;
1603 break;
1604 case QEvent::MouseButtonDblClick:
1605 if (d->mouseDoubleClickEvent(event: static_cast<QMouseEvent *>(event)))
1606 return true;
1607 break;
1608 case QEvent::MouseMove:
1609 if (d->mouseMoveEvent(event: static_cast<QMouseEvent *>(event)))
1610 return true;
1611 break;
1612 case QEvent::MouseButtonRelease:
1613 if (d->mouseReleaseEvent(event: static_cast<QMouseEvent *>(event)))
1614 return true;
1615 break;
1616 case QEvent::NonClientAreaMouseMove:
1617 case QEvent::NonClientAreaMouseButtonPress:
1618 case QEvent::NonClientAreaMouseButtonRelease:
1619 case QEvent::NonClientAreaMouseButtonDblClick:
1620 d->nonClientAreaMouseEvent(event: static_cast<QMouseEvent*>(event));
1621 return true;
1622 case QEvent::Move:
1623 d->moveEvent(event: static_cast<QMoveEvent*>(event));
1624 break;
1625 case QEvent::Resize:
1626 // if the mainwindow is plugging us, we don't want to update undocked geometry
1627 if (isFloating() && layout != nullptr && layout->pluggingWidget != this)
1628 d->undockedGeometry = geometry();
1629
1630 // Usually the window won't get resized while it's being moved, but it can happen,
1631 // for example on Windows when moving to a screen with bigger scale factor
1632 // If that happens we should update state->pressPos, otherwise it will be outside
1633 // the window when the window shrinks.
1634 if (d->state && d->state->dragging)
1635 d->recalculatePressPos(event: static_cast<QResizeEvent*>(event));
1636 break;
1637 default:
1638 break;
1639 }
1640 return QWidget::event(event);
1641}
