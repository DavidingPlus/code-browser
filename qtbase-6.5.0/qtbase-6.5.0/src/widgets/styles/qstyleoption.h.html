663template <typename T>
664T qstyleoption_cast(const QStyleOption *opt)
665{
666 typedef typename std::remove_cv<typename std::remove_pointer<T>::type>::type Opt;
667 if (opt && opt->version >= Opt::Version && (opt->type == Opt::Type
668 || int(Opt::Type) == QStyleOption::SO_Default
669 || (int(Opt::Type) == QStyleOption::SO_Complex
670 && opt->type > QStyleOption::SO_Complex)))
671 return static_cast<T>(opt);
672 return nullptr;
673}
