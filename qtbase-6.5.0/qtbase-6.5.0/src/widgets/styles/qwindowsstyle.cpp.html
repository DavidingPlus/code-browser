103/*!
104 \reimp
105*/
106bool QWindowsStyle::eventFilter(QObject *o, QEvent *e)
107{
108 // Records Alt- and Focus events
109 if (!o->isWidgetType())
110 return QObject::eventFilter(watched: o, event: e);
111
112 QWidget *widget = qobject_cast<QWidget*>(o);
113 Q_D(QWindowsStyle);
114 switch(e->type()) {
115 case QEvent::KeyPress:
116 if (static_cast<QKeyEvent *>(e)->key() == Qt::Key_Alt) {
117 widget = widget->window();
118
119 // Alt has been pressed - find all widgets that care
120 QList<QWidget *> l = widget->findChildren<QWidget *>();
121 auto ignorable = [](QWidget *w) {
122 return w->isWindow() || !w->isVisible()
123 || w->style()->styleHint(stylehint: SH_UnderlineShortcut, opt: nullptr, widget: w);
124 };
125 l.removeIf(pred: ignorable);
126 // Update states before repainting
127 d->seenAlt.append(t: widget);
128 d->alt_down = true;
129
130 // Repaint all relevant widgets
131 for (int pos = 0; pos < l.size(); ++pos)
132 l.at(i: pos)->update();
133 }
134 break;
135 case QEvent::KeyRelease:
136 if (static_cast<QKeyEvent*>(e)->key() == Qt::Key_Alt) {
137 widget = widget->window();
138
139 // Update state and repaint the menu bars.
140 d->alt_down = false;
141#if QT_CONFIG(menubar)
142 QList<QMenuBar *> l = widget->findChildren<QMenuBar *>();
143 for (int i = 0; i < l.size(); ++i)
144 l.at(i)->update();
145#endif
146 }
147 break;
148 case QEvent::Close:
149 // Reset widget when closing
150 d->seenAlt.removeAll(t: widget);
151 d->seenAlt.removeAll(t: widget->window());
152 break;
153 default:
154 break;
155 }
156 return QCommonStyle::eventFilter(watched: o, event: e);
157}

261int QWindowsStylePrivate::pixelMetricFromSystemDp(QStyle::PixelMetric pm, const QStyleOption *, const QWidget *widget)
262{
263#if defined(Q_OS_WIN)
264 switch (pm) {
265 case QStyle::PM_DockWidgetFrameWidth:
266 return GetSystemMetrics(SM_CXFRAME);
267
268 case QStyle::PM_TitleBarHeight:
269 if (widget && (widget->windowType() == Qt::Tool)) {
270 // MS always use one less than they say
271 return GetSystemMetrics(SM_CYSMCAPTION) - 1;
272 }
273 return GetSystemMetrics(SM_CYCAPTION) - 1;
274
275 case QStyle::PM_ScrollBarExtent:
276 {
277 NONCLIENTMETRICS ncm;
278 ncm.cbSize = FIELD_OFFSET(NONCLIENTMETRICS, lfMessageFont) + sizeof(LOGFONT);
279 if (SystemParametersInfo(SPI_GETNONCLIENTMETRICS, sizeof(NONCLIENTMETRICS), &ncm, 0))
280 return qMax(ncm.iScrollHeight, ncm.iScrollWidth);
281 }
282 break;
283
284 case QStyle::PM_MdiSubWindowFrameWidth:
285 return GetSystemMetrics(SM_CYFRAME);
286
287 default:
288 break;
289 }
290#else // Q_OS_WIN
291 Q_UNUSED(pm);
292 Q_UNUSED(widget);
293#endif
294 return QWindowsStylePrivate::InvalidMetric;
295}

356// Calculate the overall scale factor to obtain Qt Device Independent
357// Pixels from a native Windows size. Divide by devicePixelRatio
358// and account for secondary screens with differing logical DPI.
359qreal QWindowsStylePrivate::nativeMetricScaleFactor(const QWidget *widget)
360{
361 const QPlatformScreen *screen = screenOf(w: widget)->handle();
362 const qreal scale = screen ? (screen->logicalDpi().first / screen->logicalBaseDpi().first)
363 : QWindowsStylePrivate::appDevicePixelRatio();
364 qreal result = qreal(1) / scale;
365 if (QGuiApplicationPrivate::screen_list.size() > 1) {
366 const QScreen *primaryScreen = QGuiApplication::primaryScreen();
367 const QScreen *screen = screenOf(w: widget);
368 if (screen != primaryScreen) {
369 const qreal primaryLogicalDpi = primaryScreen->handle()->logicalDpi().first;
370 const qreal logicalDpi = screen->handle()->logicalDpi().first;
371 if (!qFuzzyCompare(p1: primaryLogicalDpi, p2: logicalDpi))
372 result *= logicalDpi / primaryLogicalDpi;
373 }
374 }
375 return result;
376}

378/*!
379 \reimp
380*/
381int QWindowsStyle::pixelMetric(PixelMetric pm, const QStyleOption *opt, const QWidget *widget) const
382{
383 int ret = QWindowsStylePrivate::pixelMetricFromSystemDp(pm, opt, widget);
384 if (ret != QWindowsStylePrivate::InvalidMetric)
385 return qRound(d: qreal(ret) * QWindowsStylePrivate::nativeMetricScaleFactor(widget));
386
387 ret = QWindowsStylePrivate::fixedPixelMetric(pm);
388 if (ret != QWindowsStylePrivate::InvalidMetric)
389 return int(QStyleHelper::dpiScaled(value: ret, option: opt));
390
391 ret = 0;
392
393 switch (pm) {
394 case PM_MaximumDragDistance:
395 ret = QCommonStyle::pixelMetric(m: PM_MaximumDragDistance);
396 if (ret == -1)
397 ret = 60;
398 break;
399
400#if QT_CONFIG(slider)
401 // Returns the number of pixels to use for the business part of the
402 // slider (i.e., the non-tickmark portion). The remaining space is shared
403 // equally between the tickmark regions.
404 case PM_SliderControlThickness:
405 if (const QStyleOptionSlider *sl = qstyleoption_cast<const QStyleOptionSlider *>(opt)) {
406 int space = (sl->orientation == Qt::Horizontal) ? sl->rect.height() : sl->rect.width();
407 int ticks = sl->tickPosition;
408 int n = 0;
409 if (ticks & QSlider::TicksAbove)
410 ++n;
411 if (ticks & QSlider::TicksBelow)
412 ++n;
413 if (!n) {
414 ret = space;
415 break;
416 }
417
418 int thick = 6; // Magic constant to get 5 + 16 + 5
419 if (ticks != QSlider::TicksBothSides && ticks != QSlider::NoTicks)
420 thick += proxy()->pixelMetric(metric: PM_SliderLength, option: sl, widget) / 4;
421
422 space -= thick;
423 if (space > 0)
424 thick += (space * 2) / (n + 2);
425 ret = thick;
426 }
427 break;
428#endif // QT_CONFIG(slider)
429
430 case PM_IconViewIconSize:
431 ret = proxy()->pixelMetric(metric: PM_LargeIconSize, option: opt, widget);
432 break;
433
434 case PM_SplitterWidth:
435 ret = QStyleHelper::dpiScaled(value: 4, option: opt);
436 break;
437
438 default:
439 ret = QCommonStyle::pixelMetric(m: pm, opt, widget);
440 break;
441 }
442
443 return ret;
444}

495/*! \reimp */
496int QWindowsStyle::styleHint(StyleHint hint, const QStyleOption *opt, const QWidget *widget,
497 QStyleHintReturn *returnData) const
498{
499 int ret = 0;
500
501 switch (hint) {
502 case SH_EtchDisabledText: {
503 const QPalette pal = opt ? opt->palette
504 : widget ? widget->palette()
505 : QPalette();
506 ret = pal.window().color().lightness() > pal.text().color().lightness()
507 ? 1 : 0;
508 break;
509 }
510 case SH_Slider_SnapToValue:
511 case SH_PrintDialog_RightAlignButtons:
512 case SH_FontDialog_SelectAssociatedText:
513 case SH_Menu_AllowActiveAndDisabled:
514 case SH_MenuBar_AltKeyNavigation:
515 case SH_MenuBar_MouseTracking:
516 case SH_Menu_MouseTracking:
517 case SH_ComboBox_ListMouseTracking:
518 case SH_Slider_StopMouseOverSlider:
519 case SH_MainWindow_SpaceBelowMenuBar:
520 ret = 1;
521
522 break;
523 case SH_ItemView_ShowDecorationSelected:
524#if QT_CONFIG(listview)
525 if (qobject_cast<const QListView*>(object: widget))
526 ret = 1;
527#endif
528 break;
529 case SH_ItemView_ChangeHighlightOnFocus:
530 ret = 1;
531 break;
532 case SH_ToolBox_SelectedPageTitleBold:
533 ret = 0;
534 break;
535
536#if defined(Q_OS_WIN)
537 case SH_UnderlineShortcut:
538 {
539 ret = 1;
540 BOOL cues = false;
541 SystemParametersInfo(SPI_GETKEYBOARDCUES, 0, &cues, 0);
542 ret = int(cues);
543 // Do nothing if we always paint underlines
544 Q_D(const QWindowsStyle);
545 if (!ret && widget && d) {
546#if QT_CONFIG(menubar)
547 const QMenuBar *menuBar = qobject_cast<const QMenuBar *>(widget);
548 if (!menuBar && qobject_cast<const QMenu *>(widget)) {
549 QWidget *w = QApplication::activeWindow();
550 if (w && w != widget)
551 menuBar = w->findChild<QMenuBar *>();
552 }
553 // If we paint a menu bar draw underlines if is in the keyboardState
554 if (menuBar) {
555 if (menuBar->d_func()->keyboardState || d->altDown())
556 ret = 1;
557 // Otherwise draw underlines if the toplevel widget has seen an alt-press
558 } else
559#endif // QT_CONFIG(menubar)
560 if (d->hasSeenAlt(widget)) {
561 ret = 1;
562 }
563 }
564#if QT_CONFIG(accessibility)
565 if (!ret && opt && opt->type == QStyleOption::SO_MenuItem
566 && QStyleHelper::isInstanceOf(opt->styleObject, QAccessible::MenuItem)
567 && opt->styleObject->property("_q_showUnderlined").toBool())
568 ret = 1;
569#endif // QT_CONFIG(accessibility)
570 break;
571 }
572#endif // Q_OS_WIN
573 case SH_Menu_SubMenuSloppyCloseTimeout:
574 case SH_Menu_SubMenuPopupDelay: {
575#if defined(Q_OS_WIN)
576 DWORD delay;
577 if (SystemParametersInfo(SPI_GETMENUSHOWDELAY, 0, &delay, 0))
578 ret = delay;
579 else
580#endif // Q_OS_WIN
581 ret = 400;
582 break;
583 }
584#if QT_CONFIG(rubberband)
585 case SH_RubberBand_Mask:
586 if (const QStyleOptionRubberBand *rbOpt = qstyleoption_cast<const QStyleOptionRubberBand *>(opt)) {
587 ret = 0;
588 if (rbOpt->shape == QRubberBand::Rectangle) {
589 ret = true;
590 if (QStyleHintReturnMask *mask = qstyleoption_cast<QStyleHintReturnMask*>(hint: returnData)) {
591 mask->region = opt->rect;
592 int size = 1;
593 if (widget && widget->isWindow())
594 size = 4;
595 mask->region -= opt->rect.adjusted(xp1: size, yp1: size, xp2: -size, yp2: -size);
596 }
597 }
598 }
599 break;
600#endif // QT_CONFIG(rubberband)
601#if QT_CONFIG(wizard)
602 case SH_WizardStyle:
603 ret = QWizard::ModernStyle;
604 break;
605#endif
606 case SH_ItemView_ArrowKeysNavigateIntoChildren:
607 ret = true;
608 break;
609 case SH_DialogButtonBox_ButtonsHaveIcons:
610 ret = 0;
611 break;
612 default:
613 ret = QCommonStyle::styleHint(sh: hint, opt, w: widget, shret: returnData);
614 break;
615 }
616 return ret;
617}

632 QPen oldPen = p->pen();

635 p->setPen(QPen(opt->palette.dark().color()));
636 p->drawLine(x1: rect.bottomLeft().x() + offset,
637 y1: rect.bottomLeft().y() - margin,
638 x2: rect.topLeft().x() + offset,
639 y2: rect.topLeft().y() + margin);
640 p->setPen(QPen(opt->palette.light().color()));
641 p->drawLine(x1: rect.bottomLeft().x() + offset + 1,
642 y1: rect.bottomLeft().y() - margin,
643 x2: rect.topLeft().x() + offset + 1,
644 y2: rect.topLeft().y() + margin);
645 }
646 else{ //Draw vertical separator
647 const int offset = rect.height()/2;
648 p->setPen(QPen(opt->palette.dark().color()));
649 p->drawLine(x1: rect.topLeft().x() + margin ,
650 y1: rect.topLeft().y() + offset,
651 x2: rect.topRight().x() - margin,
652 y2: rect.topRight().y() + offset);
653 p->setPen(QPen(opt->palette.light().color()));
654 p->drawLine(x1: rect.topLeft().x() + margin ,
655 y1: rect.topLeft().y() + offset + 1,
656 x2: rect.topRight().x() - margin,
657 y2: rect.topRight().y() + offset + 1);
658 }
659 p->setPen(oldPen);
660 }
661 break;
662 case PE_IndicatorToolBarHandle:
663 p->save();
664 p->translate(dx: opt->rect.x(), dy: opt->rect.y());
665 if (opt->state & State_Horizontal) {
666 int x = opt->rect.width() / 2 - 4;
667 if (opt->direction == Qt::RightToLeft)
668 x -= 2;
669 if (opt->rect.height() > 4) {
670 qDrawShadePanel(p, x, y: 2, w: 3, h: opt->rect.height() - 4,
671 pal: opt->palette, sunken: false, lineWidth: 1, fill: nullptr);
672 qDrawShadePanel(p, x: x + 3, y: 2, w: 3, h: opt->rect.height() - 4,
673 pal: opt->palette, sunken: false, lineWidth: 1, fill: nullptr);
674 }
675 } else {
676 if (opt->rect.width() > 4) {
677 int y = opt->rect.height() / 2 - 4;
678 qDrawShadePanel(p, x: 2, y, w: opt->rect.width() - 4, h: 3,
679 pal: opt->palette, sunken: false, lineWidth: 1, fill: nullptr);
680 qDrawShadePanel(p, x: 2, y: y + 3, w: opt->rect.width() - 4, h: 3,
681 pal: opt->palette, sunken: false, lineWidth: 1, fill: nullptr);
682 }
683 }
684 p->restore();
685 break;
686
687#endif // QT_CONFIG(toolbar)
688 case PE_FrameButtonTool:
689 case PE_PanelButtonTool: {
690 QPen oldPen = p->pen();
691#if QT_CONFIG(dockwidget)
692 if (w && w->inherits(classname: "QDockWidgetTitleButton")) {
693 if (const QWidget *dw = w->parentWidget())
694 if (dw->isWindow()){
695 qDrawWinButton(p, r: opt->rect.adjusted(xp1: 1, yp1: 1, xp2: 0, yp2: 0), pal: opt->palette, sunken: opt->state & (State_Sunken | State_On),
696 fill: &opt->palette.button());
697
698 return;
699 }
700 }
701#endif // QT_CONFIG(dockwidget)
702 QBrush fill;
703 bool stippled;
704 bool panel = (pe == PE_PanelButtonTool);
