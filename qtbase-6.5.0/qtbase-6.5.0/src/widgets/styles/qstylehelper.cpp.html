28QString uniqueName(const QString &key, const QStyleOption *option, const QSize &size)
29{
30 const QStyleOptionComplex *complexOption = qstyleoption_cast<const QStyleOptionComplex *>(opt: option);
31 QString tmp = key % HexString<uint>(option->state)
32 % HexString<uint>(option->direction)
33 % HexString<uint>(complexOption ? uint(complexOption->activeSubControls) : 0u)
34 % HexString<quint64>(option->palette.cacheKey())
35 % HexString<uint>(size.width())
36 % HexString<uint>(size.height());
37
38#if QT_CONFIG(spinbox)
39 if (const QStyleOptionSpinBox *spinBox = qstyleoption_cast<const QStyleOptionSpinBox *>(opt: option)) {
40 tmp = tmp % HexString<uint>(spinBox->buttonSymbols)
41 % HexString<uint>(spinBox->stepEnabled)
42 % QChar(spinBox->frame ? u'1' : u'0');
43 }
44#endif // QT_CONFIG(spinbox)
45
46 return tmp;
47}

55Q_WIDGETS_EXPORT qreal dpi(const QStyleOption *option)
56{
57#ifndef Q_OS_DARWIN
58 // Prioritize the application override, except for on macOS where
59 // we have historically not supported the AA_Use96Dpi flag.
60 if (QCoreApplication::testAttribute(attribute: Qt::AA_Use96Dpi))
61 return 96;
62#endif
63
64 // Expect that QStyleOption::QFontMetrics::QFont has the correct DPI set
65 if (option)
66 return option->fontMetrics.fontDpi();
67
68 // Fall back to historical Qt behavior: hardocded 72 DPI on mac,
69 // primary screen DPI on other platforms.
70#ifdef Q_OS_DARWIN
71 return qstyleBaseDpi;
72#else
73 return qt_defaultDpiX();
74#endif
75}

101// Searches for an ancestor of a particular accessible role
102bool hasAncestor(QObject *obj, QAccessible::Role role)
103{
104 bool found = false;
105 QObject *parent = obj ? obj->parent() : nullptr;
106 while (parent && !found) {
107 if (isInstanceOf(obj: parent, role))
108 found = true;
109 parent = parent->parent();
110 }
111 return found;
112}

118int calcBigLineSize(int radius)
119{
120 int bigLineSize = radius / 6;
121 if (bigLineSize < 4)
122 bigLineSize = 4;
123 if (bigLineSize > radius / 2)
124 bigLineSize = radius / 2;
125 return bigLineSize;
126}

128static QPointF calcRadialPos(const QStyleOptionSlider *dial, qreal offset)
129{
130 const int width = dial->rect.width();
131 const int height = dial->rect.height();
132 const int r = qMin(a: width, b: height) / 2;
133 const int currentSliderPosition = dial->upsideDown ? dial->sliderPosition : (dial->maximum - dial->sliderPosition);
134 qreal a = 0;
135 if (dial->maximum == dial->minimum)
136 a = Q_PI / 2;
137 else if (dial->dialWrapping)
138 a = Q_PI * 3 / 2 - (currentSliderPosition - dial->minimum) * 2 * Q_PI
139 / (dial->maximum - dial->minimum);
140 else
141 a = (Q_PI * 8 - (currentSliderPosition - dial->minimum) * 10 * Q_PI
142 / (dial->maximum - dial->minimum)) / 6;
143 qreal xc = width / 2.0;
144 qreal yc = height / 2.0;
145 qreal len = r - QStyleHelper::calcBigLineSize(radius: r) - 3;
146 qreal back = offset * len;
147 QPointF pos(QPointF(xc + back * qCos(v: a), yc - back * qSin(v: a)));
148 return pos + dial->rect.topLeft();
149}

151qreal angle(const QPointF &p1, const QPointF &p2)
152{
153 static const qreal rad_factor = 180 / Q_PI;
154 qreal _angle = 0;
155
156 if (p1.x() == p2.x()) {
157 if (p1.y() < p2.y())
158 _angle = 270;
159 else
160 _angle = 90;
161 } else {
162 qreal x1, x2, y1, y2;
163
164 if (p1.x() <= p2.x()) {
165 x1 = p1.x(); y1 = p1.y();
166 x2 = p2.x(); y2 = p2.y();
167 } else {
168 x2 = p1.x(); y2 = p1.y();
169 x1 = p2.x(); y1 = p2.y();
170 }
171
172 qreal m = -(y2 - y1) / (x2 - x1);
173 _angle = qAtan(v: m) * rad_factor;
174
175 if (p1.x() < p2.x())
176 _angle = 180 - _angle;
177 else
178 _angle = -_angle;
179 }
180 return _angle;
181}

183QPolygonF calcLines(const QStyleOptionSlider *dial)
184{
185 QPolygonF poly;
186 int width = dial->rect.width();
187 int height = dial->rect.height();
188 qreal r = qMin(a: width, b: height) / 2;
189 int bigLineSize = calcBigLineSize(radius: int(r));
190
191 qreal xc = width / 2 + 0.5;
192 qreal yc = height / 2 + 0.5;
193 const int ns = dial->tickInterval;
194 if (!ns) // Invalid values may be set by Qt Designer.
195 return poly;
196 int notches = (dial->maximum + ns - 1 - dial->minimum) / ns;
197 if (notches <= 0)
198 return poly;
199 if (dial->maximum < dial->minimum || dial->maximum - dial->minimum > 1000) {
200 int maximum = dial->minimum + 1000;
201 notches = (maximum + ns - 1 - dial->minimum) / ns;
202 }
203
204 poly.resize(size: 2 + 2 * notches);
205 int smallLineSize = bigLineSize / 2;
206 for (int i = 0; i <= notches; ++i) {
207 qreal angle = dial->dialWrapping ? Q_PI * 3 / 2 - i * 2 * Q_PI / notches
208 : (Q_PI * 8 - i * 10 * Q_PI / notches) / 6;
209 qreal s = qSin(v: angle);
210 qreal c = qCos(v: angle);
211 if (i == 0 || (((ns * i) % (dial->pageStep ? dial->pageStep : 1)) == 0)) {
212 poly[2 * i] = QPointF(xc + (r - bigLineSize) * c,
213 yc - (r - bigLineSize) * s);
214 poly[2 * i + 1] = QPointF(xc + r * c, yc - r * s);
215 } else {
216 poly[2 * i] = QPointF(xc + (r - 1 - smallLineSize) * c,
217 yc - (r - 1 - smallLineSize) * s);
218 poly[2 * i + 1] = QPointF(xc + (r - 1) * c, yc -(r - 1) * s);
219 }
220 }
221 return poly.translated(offset: dial->rect.topLeft());
222}

224// This will draw a nice and shiny QDial for us. We don't want
225// all the shinyness in QWindowsStyle, hence we place it here

227void drawDial(const QStyleOptionSlider *option, QPainter *painter)
228{
229 const QPalette pal = option->palette;
230 QColor buttonColor = pal.button().color();
231 const int width = option->rect.width();
232 const int height = option->rect.height();
233 const bool enabled = option->state & QStyle::State_Enabled;
234 qreal r = qMin(a: width, b: height) / 2;
235 r -= r/50;
236 const qreal penSize = r/20.0;
237
238 painter->save();
239 painter->setRenderHint(hint: QPainter::Antialiasing);
240
241 // Draw notches
242 if (option->subControls & QStyle::SC_DialTickmarks) {
243 const bool inverted = pal.window().color().lightness() < pal.text().color().lightness()
244 && pal.light().color().lightness() > pal.dark().color().lightness();
245 const QColor notchColor = inverted ? pal.light().color().lighter(f: 120)
246 : pal.dark().color().darker(f: 120);
247 painter->setPen(notchColor);
248 painter->drawLines(pointPairs: QStyleHelper::calcLines(dial: option));
249 }
250
251 // setting color before BEGIN_STYLE_PIXMAPCACHE since
252 // otherwise it is not set when the image is in the cache
253 buttonColor.setHsv(h: buttonColor .hue(),
254 s: qMin(a: 140, b: buttonColor .saturation()),
255 v: qMax(a: 180, b: buttonColor.value()));
256
257 // Cache dial background
258 BEGIN_STYLE_PIXMAPCACHE(QString::fromLatin1("qdial"));
259 p->setRenderHint(hint: QPainter::Antialiasing);
260
261 const qreal d_ = r / 6;
262 const qreal dx = d_ + (width - 2 * r) / 2 + 1;
263 const qreal dy = d_ + (height - 2 * r) / 2 + 1;
264
265 QRectF br = QRectF(dx + 0.5, dy + 0.5,
266 int(r * 2 - 2 * d_ - 2),
267 int(r * 2 - 2 * d_ - 2));
268
269 if (enabled) {
270 // Drop shadow
271 qreal shadowSize = qMax(a: 1.0, b: penSize/2.0);
272 QRectF shadowRect= br.adjusted(xp1: -2*shadowSize, yp1: -2*shadowSize,
273 xp2: 2*shadowSize, yp2: 2*shadowSize);
274 QRadialGradient shadowGradient(shadowRect.center().x(),
275 shadowRect.center().y(), shadowRect.width()/2.0,
276 shadowRect.center().x(), shadowRect.center().y());
277 shadowGradient.setColorAt(pos: qreal(0.91), color: QColor(0, 0, 0, 40));
278 shadowGradient.setColorAt(pos: qreal(1.0), color: Qt::transparent);
279 p->setBrush(shadowGradient);
280 p->setPen(Qt::NoPen);
281 p->translate(dx: shadowSize, dy: shadowSize);
282 p->drawEllipse(r: shadowRect);
283 p->translate(dx: -shadowSize, dy: -shadowSize);
284
285 // Main gradient
286 QRadialGradient gradient(br.center().x() - br.width()/3, dy,
287 br.width()*1.3, br.center().x(),
288 br.center().y() - br.height()/2);
289 gradient.setColorAt(pos: 0, color: buttonColor.lighter(f: 110));
290 gradient.setColorAt(pos: qreal(0.5), color: buttonColor);
291 gradient.setColorAt(pos: qreal(0.501), color: buttonColor.darker(f: 102));
292 gradient.setColorAt(pos: 1, color: buttonColor.darker(f: 115));
293 p->setBrush(gradient);
294 } else {
295 p->setBrush(Qt::NoBrush);
296 }
297
298 p->setPen(QPen(buttonColor.darker(f: 280)));
299 p->drawEllipse(r: br);
300 p->setBrush(Qt::NoBrush);
301 p->setPen(buttonColor.lighter(f: 110));
302 p->drawEllipse(r: br.adjusted(xp1: 1, yp1: 1, xp2: -1, yp2: -1));
303
304 if (option->state & QStyle::State_HasFocus) {
305 QColor highlight = pal.highlight().color();
306 highlight.setHsv(h: highlight.hue(),
307 s: qMin(a: 160, b: highlight.saturation()),
308 v: qMax(a: 230, b: highlight.value()));
309 highlight.setAlpha(127);
310 p->setPen(QPen(highlight, 2.0));
311 p->setBrush(Qt::NoBrush);
312 p->drawEllipse(r: br.adjusted(xp1: -1, yp1: -1, xp2: 1, yp2: 1));
313 }
314
315 END_STYLE_PIXMAPCACHE
316
317 QPointF dp = calcRadialPos(dial: option, offset: qreal(0.70));
318 buttonColor = buttonColor.lighter(f: 104);
319 buttonColor.setAlphaF(0.8f);
320 const qreal ds = r/qreal(7.0);
321 QRectF dialRect(dp.x() - ds, dp.y() - ds, 2*ds, 2*ds);
322 QRadialGradient dialGradient(dialRect.center().x() + dialRect.width()/2,
323 dialRect.center().y() + dialRect.width(),
324 dialRect.width()*2,
325 dialRect.center().x(), dialRect.center().y());
326 dialGradient.setColorAt(pos: 1, color: buttonColor.darker(f: 140));
327 dialGradient.setColorAt(pos: qreal(0.4), color: buttonColor.darker(f: 120));
328 dialGradient.setColorAt(pos: 0, color: buttonColor.darker(f: 110));
329 if (penSize > 3.0) {
330 painter->setPen(QPen(QColor(0, 0, 0, 25), penSize));
331 painter->drawLine(p1: calcRadialPos(dial: option, offset: qreal(0.90)), p2: calcRadialPos(dial: option, offset: qreal(0.96)));
332 }
333
334 painter->setBrush(dialGradient);
335 painter->setPen(QColor(255, 255, 255, 150));
336 painter->drawEllipse(r: dialRect.adjusted(xp1: -1, yp1: -1, xp2: 1, yp2: 1));
337 painter->setPen(QColor(0, 0, 0, 80));
338 painter->drawEllipse(r: dialRect);
339 painter->restore();
340}

343void drawBorderPixmap(const QPixmap &pixmap, QPainter *painter, const QRect &rect,
344 int left, int top, int right,
345 int bottom)
346{
347 QSize size = pixmap.size();
348 //painter->setRenderHint(QPainter::SmoothPixmapTransform);
349
350 //top
351 if (top > 0) {
352 painter->drawPixmap(targetRect: QRect(rect.left() + left, rect.top(), rect.width() -right - left, top), pixmap,
353 sourceRect: QRect(left, 0, size.width() -right - left, top));
354
355 //top-left
356 if (left > 0)
357 painter->drawPixmap(targetRect: QRect(rect.left(), rect.top(), left, top), pixmap,
358 sourceRect: QRect(0, 0, left, top));
359
360 //top-right
361 if (right > 0)
362 painter->drawPixmap(targetRect: QRect(rect.left() + rect.width() - right, rect.top(), right, top), pixmap,
363 sourceRect: QRect(size.width() - right, 0, right, top));
364 }
365
366 //left
367 if (left > 0)
368 painter->drawPixmap(targetRect: QRect(rect.left(), rect.top()+top, left, rect.height() - top - bottom), pixmap,
369 sourceRect: QRect(0, top, left, size.height() - bottom - top));
370
371 //center
372 painter->drawPixmap(targetRect: QRect(rect.left() + left, rect.top()+top, rect.width() -right - left,
373 rect.height() - bottom - top), pixmap,
374 sourceRect: QRect(left, top, size.width() -right -left,
375 size.height() - bottom - top));
376 //right
377 if (right > 0)
378 painter->drawPixmap(targetRect: QRect(rect.left() +rect.width() - right, rect.top()+top, right, rect.height() - top - bottom), pixmap,
379 sourceRect: QRect(size.width() - right, top, right, size.height() - bottom - top));
380
381 //bottom
382 if (bottom > 0) {
383 painter->drawPixmap(targetRect: QRect(rect.left() +left, rect.top() + rect.height() - bottom,
384 rect.width() - right - left, bottom), pixmap,
385 sourceRect: QRect(left, size.height() - bottom,
386 size.width() - right - left, bottom));
387 //bottom-left
388 if (left > 0)
389 painter->drawPixmap(targetRect: QRect(rect.left(), rect.top() + rect.height() - bottom, left, bottom), pixmap,
390 sourceRect: QRect(0, size.height() - bottom, left, bottom));
391
392 //bottom-right
393 if (right > 0)
394 painter->drawPixmap(targetRect: QRect(rect.left() + rect.width() - right, rect.top() + rect.height() - bottom, right, bottom), pixmap,
395 sourceRect: QRect(size.width() - right, size.height() - bottom, right, bottom));
396
397 }

400QColor backgroundColor(const QPalette &pal, const QWidget* widget)
401{
402#if QT_CONFIG(scrollarea)
403 if (qobject_cast<const QScrollBar *>(object: widget) && widget->parent() &&
404 qobject_cast<const QAbstractScrollArea *>(object: widget->parent()->parent()))
405 return widget->parentWidget()->parentWidget()->palette().color(cr: QPalette::Base);
406#else
407 Q_UNUSED(widget);
408#endif
409 return pal.color(cr: QPalette::Base);
410}

412WidgetSizePolicy widgetSizePolicy(const QWidget *widget, const QStyleOption *opt)
413{
414 while (widget) {
415 if (widget->testAttribute(attribute: Qt::WA_MacMiniSize)) {
416 return SizeMini;
417 } else if (widget->testAttribute(attribute: Qt::WA_MacSmallSize)) {
418 return SizeSmall;
419 } else if (widget->testAttribute(attribute: Qt::WA_MacNormalSize)) {
420 return SizeLarge;
421 }
422 widget = widget->parentWidget();
423 }
424
425 if (opt && opt->state & QStyle::State_Mini)
426 return SizeMini;
427 else if (opt && opt->state & QStyle::State_Small)
428 return SizeSmall;
429
430 return SizeDefault;
431}
