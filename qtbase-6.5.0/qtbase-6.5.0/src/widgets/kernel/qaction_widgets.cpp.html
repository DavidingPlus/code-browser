26// we can't do this in the destructor, as it would only be called by ~QObject
27void QtWidgetsActionPrivate::destroy()
28{
29 Q_Q(QAction);
30 const auto objects = associatedObjects;
31 for (int i = objects.size()-1; i >= 0; --i) {
32 QObject *object = objects.at(i);
33 if (QWidget *widget = qobject_cast<QWidget*>(o: object))
34 widget->removeAction(action: q);
35#if QT_CONFIG(graphicsview)
36 else if (QGraphicsWidget *graphicsWidget = qobject_cast<QGraphicsWidget*>(object))
37 graphicsWidget->removeAction(action: q);
38#endif
39 }
40}

55void QtWidgetsActionPrivate::setMenu(QObject *menu)
56{
57 Q_Q(QAction);
58 QMenu *theMenu = qobject_cast<QMenu*>(object: menu);
59 Q_ASSERT_X(!menu || theMenu, "QAction::setMenu",
60 "QAction::setMenu expects a QMenu* in widget applications");
61 if (m_menu)
62 m_menu->d_func()->setOverrideMenuAction(nullptr); //we reset the default action of any previous menu
63 m_menu = theMenu;
64 if (m_menu)
65 m_menu->d_func()->setOverrideMenuAction(q);
66 sendDataChanged();
67}
