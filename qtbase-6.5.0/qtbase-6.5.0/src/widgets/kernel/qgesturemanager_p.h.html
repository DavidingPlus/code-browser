32class QGestureManager : public QObject

36 QGestureManager(QObject *parent);

40 void unregisterGestureRecognizer(Qt::GestureType type);
41
42 bool filterEvent(QWidget *receiver, QEvent *event);
43 bool filterEvent(QObject *receiver, QEvent *event);
44#if QT_CONFIG(graphicsview)
45 bool filterEvent(QGraphicsObject *receiver, QEvent *event);
46#endif // QT_CONFIG(graphicsview)
47
48 enum InstanceCreation { ForceCreation, DontForceCreation };
49
50 static QGestureManager *instance(InstanceCreation ic = ForceCreation); // declared in qapplication.cpp
51 static bool gesturePending(QObject *o);
52
53 void cleanupCachedGestures(QObject *target, Qt::GestureType type);
54
55 void recycle(QGesture *gesture);
56
57protected:

60
61private:
62 QMultiMap<Qt::GestureType, QGestureRecognizer *> m_recognizers;
63
64 QSet<QGesture *> m_activeGestures;
65 QSet<QGesture *> m_maybeGestures;
66
67 enum State {
68 Gesture,
69 NotGesture,
70 MaybeGesture // this means timers are up and waiting for some
71 // more events, and input events are handled by
72 // gesture recognizer explicitly
73 } state;
74
75 struct ObjectGesture
76 {
77 QObject* object;
78 Qt::GestureType gesture;
79
80 ObjectGesture(QObject *o, const Qt::GestureType &g) : object(o), gesture(g) { }
81 inline bool operator<(const ObjectGesture &rhs) const
82 {
83 if (std::less<QObject *>{}(object, rhs.object))
84 return true;
85 if (object == rhs.object)
86 return gesture < rhs.gesture;
87 return false;
88 }
89 };
90
91 QMap<ObjectGesture, QList<QGesture *> > m_objectGestures;
92 QHash<QGesture *, QGestureRecognizer *> m_gestureToRecognizer;
93 QHash<QGesture *, QObject *> m_gestureOwners;
94
95 QHash<QGesture *, QPointer<QWidget> > m_gestureTargets;
96
97 int m_lastCustomGestureId;
98
99 QHash<QGestureRecognizer *, QSet<QGesture *> > m_obsoleteGestures;
100 QHash<QGesture *, QGestureRecognizer *> m_deletedRecognizers;
101 QSet<QGesture *> m_gesturesToDelete;
102 void cleanupGesturesForRemovedRecognizer(QGesture *gesture);
103
104 QGesture *getState(QObject *widget, QGestureRecognizer *recognizer,
105 Qt::GestureType gesture);
106 void deliverEvents(const QSet<QGesture *> &gestures,
107 QSet<QGesture *> *undeliveredGestures);
108 void getGestureTargets(const QSet<QGesture*> &gestures,
109 QHash<QWidget *, QList<QGesture *> > *conflicts,
110 QHash<QWidget *, QList<QGesture *> > *normal);
111
112 void cancelGesturesForChildren(QGesture *originatingGesture);
113};
