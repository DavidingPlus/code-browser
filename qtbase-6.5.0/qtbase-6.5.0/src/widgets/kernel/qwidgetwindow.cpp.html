35class QWidgetWindowPrivate : public QWindowPrivate
36{
37 Q_DECLARE_PUBLIC(QWidgetWindow)
38public:
39 void setVisible(bool visible) override
40 {
41 Q_Q(QWidgetWindow);
42 if (QWidget *widget = q->widget()) {
43 // Check if the widget was already hidden, as this indicates it was done
44 // explicitly and not because the parent window in this case made it hidden.
45 // In which case do not automatically show the widget when the parent
46 // window is shown.
47 const bool wasExplicitShowHide = widget->testAttribute(attribute: Qt::WA_WState_ExplicitShowHide);
48 const bool wasHidden = widget->testAttribute(attribute: Qt::WA_WState_Hidden);
49 QWidgetPrivate::get(w: widget)->setVisible(visible);
50 if (wasExplicitShowHide) {
51 widget->setAttribute(Qt::WA_WState_ExplicitShowHide, on: wasExplicitShowHide);
52 widget->setAttribute(Qt::WA_WState_Hidden, on: wasHidden);
53 }
54 } else {
55 QWindowPrivate::setVisible(visible);
56 }
57 }
58
59 QWindow *eventReceiver() override {
60 Q_Q(QWidgetWindow);
61 QWindow *w = q;
62 while (w->parent() && qobject_cast<QWidgetWindow *>(object: w) && qobject_cast<QWidgetWindow *>(object: w->parent())) {
63 w = w->parent();
64 }
65 return w;
66 }

68 void clearFocusObject() override
69 {
70 Q_Q(QWidgetWindow);

89QRectF QWidgetWindowPrivate::closestAcceptableGeometry(const QRectF &rect) const
90{
91 Q_Q(const QWidgetWindow);
92 const QWidget *widget = q->widget();
93 if (!widget || !widget->isWindow() || !widget->hasHeightForWidth())
94 return QRect();
95 const QSize oldSize = rect.size().toSize();
96 const QSize newSize = QLayout::closestAcceptableSize(w: widget, s: oldSize);
97 if (newSize == oldSize)
98 return QRectF();
99 const int dw = newSize.width() - oldSize.width();
100 const int dh = newSize.height() - oldSize.height();
101 QRectF result = rect;
102 const QRectF currentGeometry(widget->geometry());
103 const qreal topOffset = result.top() - currentGeometry.top();
104 const qreal bottomOffset = result.bottom() - currentGeometry.bottom();
105 if (qAbs(t: topOffset) > qAbs(t: bottomOffset))
106 result.setTop(result.top() - dh); // top edge drag
107 else
108 result.setBottom(result.bottom() + dh); // bottom edge drag
109 const qreal leftOffset = result.left() - currentGeometry.left();
110 const qreal rightOffset = result.right() - currentGeometry.right();
111 if (qAbs(t: leftOffset) > qAbs(t: rightOffset))
112 result.setLeft(result.left() - dw); // left edge drag
113 else
114 result.setRight(result.right() + dw); // right edge drag
115 return result;
116}

148QObject *QWidgetWindow::focusObject() const
149{
150 QWidget *windowWidget = m_widget;
151 if (!windowWidget)
152 return nullptr;
153
154 // A window can't have a focus object if it's being destroyed.
155 if (QWidgetPrivate::get(w: windowWidget)->data.in_destructor)
156 return nullptr;
157
158 QWidget *widget = windowWidget->focusWidget();
159
160 if (!widget)
161 widget = windowWidget;
162
163 QObject *focusObj = QWidgetPrivate::get(w: widget)->focusObject();
164 if (focusObj)
165 return focusObj;
166
167 return widget;
168}

179static inline bool shouldBePropagatedToWidget(QEvent *event)
180{
181 switch (event->type()) {
182 // Handing show events to widgets would cause them to be triggered twice
183 case QEvent::Show:
184 case QEvent::Hide:
185 case QEvent::Timer:
186 case QEvent::DynamicPropertyChange:
187 case QEvent::ChildAdded:
188 case QEvent::ChildRemoved:
189 case QEvent::Paint:
190 case QEvent::Close: // Propagated manually in closeEvent
191 return false;
192 default:
193 return true;
194 }
195}

197bool QWidgetWindow::event(QEvent *event)
198{
199 if (!m_widget)
200 return QWindow::event(event);
201
202 switch (event->type()) {
203 case QEvent::Enter:
204 case QEvent::Leave:
205 handleEnterLeaveEvent(event);
206 return true;
207
208 // these should not be sent to QWidget, the corresponding events
209 // are sent by QApplicationPrivate::notifyActiveWindowChange()
210 case QEvent::FocusIn:
211 handleFocusInEvent(static_cast<QFocusEvent *>(event));
212 Q_FALLTHROUGH();
213 case QEvent::FocusOut: {
214#if QT_CONFIG(accessibility)
215 QAccessible::State state;
216 state.active = true;
217 QAccessibleStateChangeEvent ev(m_widget, state);
218 QAccessible::updateAccessibility(event: &ev);
219#endif
220 return false; }
221
222 case QEvent::FocusAboutToChange:
223 if (QApplicationPrivate::focus_widget) {
224 if (QApplicationPrivate::focus_widget->testAttribute(attribute: Qt::WA_InputMethodEnabled))
225 QGuiApplication::inputMethod()->commit();
226
227 QGuiApplication::forwardEvent(receiver: QApplicationPrivate::focus_widget, event);
228 }
229 return true;
230
231 case QEvent::KeyPress:
232 case QEvent::KeyRelease:
233 case QEvent::ShortcutOverride:
234 handleKeyEvent(static_cast<QKeyEvent *>(event));
235 return true;
236
237 case QEvent::MouseMove:
238 case QEvent::MouseButtonPress:
239 case QEvent::MouseButtonRelease:
240 case QEvent::MouseButtonDblClick:
241 handleMouseEvent(static_cast<QMouseEvent *>(event));
242 return true;
243
244 case QEvent::NonClientAreaMouseMove:
245 case QEvent::NonClientAreaMouseButtonPress:
246 case QEvent::NonClientAreaMouseButtonRelease:
247 case QEvent::NonClientAreaMouseButtonDblClick:
248 handleNonClientAreaMouseEvent(static_cast<QMouseEvent *>(event));
249 return true;
250
251 case QEvent::TouchBegin:
252 case QEvent::TouchUpdate:
253 case QEvent::TouchEnd:
254 case QEvent::TouchCancel:
255 handleTouchEvent(static_cast<QTouchEvent *>(event));
256 return true;
257
258 case QEvent::Move:
259 handleMoveEvent(static_cast<QMoveEvent *>(event));
260 return true;
261
262 case QEvent::Resize:
263 handleResizeEvent(static_cast<QResizeEvent *>(event));
264 return true;
265
266#if QT_CONFIG(wheelevent)
267 case QEvent::Wheel:
268 handleWheelEvent(static_cast<QWheelEvent *>(event));
269 return true;
270#endif
271
272#if QT_CONFIG(draganddrop)
273 case QEvent::DragEnter:
274 handleDragEnterEvent(static_cast<QDragEnterEvent *>(event));
275 return true;
276 case QEvent::DragMove:
277 handleDragMoveEvent(static_cast<QDragMoveEvent *>(event));
278 return true;
279 case QEvent::DragLeave:
280 handleDragLeaveEvent(static_cast<QDragLeaveEvent *>(event));
281 return true;
282 case QEvent::Drop:
283 handleDropEvent(static_cast<QDropEvent *>(event));
284 return true;
285#endif
286
287 case QEvent::Expose:
288 handleExposeEvent(static_cast<QExposeEvent *>(event));
289 return true;
290
291 case QEvent::WindowStateChange:
292 QWindow::event(event); // Update QWindow::Visibility and emit signals.
293 handleWindowStateChangedEvent(event: static_cast<QWindowStateChangeEvent *>(event));
294 return true;
295
296 case QEvent::ThemeChange: {
297 QEvent widgetEvent(QEvent::ThemeChange);
298 QCoreApplication::forwardEvent(receiver: m_widget, event: &widgetEvent, originatingEvent: event);
299 }
300 return true;
301
302#if QT_CONFIG(tabletevent)
303 case QEvent::TabletPress:
304 case QEvent::TabletMove:
305 case QEvent::TabletRelease:
306 handleTabletEvent(static_cast<QTabletEvent *>(event));
307 return true;
308#endif
309
310#ifndef QT_NO_GESTURES
311 case QEvent::NativeGesture:
312 handleGestureEvent(static_cast<QNativeGestureEvent *>(event));
313 return true;
314#endif
315
316#ifndef QT_NO_CONTEXTMENU
317 case QEvent::ContextMenu:
318 handleContextMenuEvent(static_cast<QContextMenuEvent *>(event));
319 return true;
320#endif // QT_NO_CONTEXTMENU
321
322 case QEvent::WindowBlocked:
323 qt_button_down = nullptr;
324 break;
325
326 case QEvent::UpdateRequest:
327 // This is not the same as an UpdateRequest for a QWidget. That just
328 // syncs the backing store while here we also must mark as dirty.
329 m_widget->repaint();
330 return true;
331
332 default:
333 break;
334 }
335
336 if (shouldBePropagatedToWidget(event) && QCoreApplication::forwardEvent(receiver: m_widget, event))
337 return true;
338
339 return QWindow::event(event);
340}

344void QWidgetWindow::handleEnterLeaveEvent(QEvent *event)
345{
346 // Ignore all enter/leave events from QPA if we are not on the first-level context menu.
347 // This prevents duplicated events on most platforms. Fake events will be delivered in
348 // QWidgetWindow::handleMouseEvent(QMouseEvent *). Make an exception whether the widget
349 // is already under mouse - let the mouse leave.
350 if (QApplicationPrivate::inPopupMode() && m_widget != QApplication::activePopupWidget() && !m_widget->underMouse())
351 return;
352
353 if (event->type() == QEvent::Leave) {
354 QWidget *enter = nullptr;
355 // Check from window system event queue if the next queued enter targets a window
356 // in the same window hierarchy (e.g. enter a child of this window). If so,
357 // remove the enter event from queue and handle both in single dispatch.
358 QWindowSystemInterfacePrivate::EnterEvent *systemEvent =
359 static_cast<QWindowSystemInterfacePrivate::EnterEvent *>
360 (QWindowSystemInterfacePrivate::peekWindowSystemEvent(t: QWindowSystemInterfacePrivate::Enter));
361 const QPointF globalPosF = systemEvent ? systemEvent->globalPos : QPointF(QGuiApplicationPrivate::lastCursorPosition);
362 if (systemEvent) {
363 if (QWidgetWindow *enterWindow = qobject_cast<QWidgetWindow *>(object: systemEvent->enter))
364 {
365 QWindow *thisParent = this;
366 QWindow *enterParent = enterWindow;
367 while (thisParent->parent())
368 thisParent = thisParent->parent();
369 while (enterParent->parent())
370 enterParent = enterParent->parent();
371 if (thisParent == enterParent) {
372 QGuiApplicationPrivate::currentMouseWindow = enterWindow;
373 enter = enterWindow->widget();
374 QWindowSystemInterfacePrivate::removeWindowSystemEvent(event: systemEvent);
375 }
376 }
377 }
378 // Enter-leave between sibling widgets is ignored when there is a mousegrabber - this makes
379 // both native and non-native widgets work similarly.
380 // When mousegrabbing, leaves are only generated if leaving the parent window.
381 if (!enter || !QWidget::mouseGrabber()) {
382 // Preferred leave target is the last mouse receiver, unless it has native window,
383 // in which case it is assumed to receive it's own leave event when relevant.
384 QWidget *leave = m_widget;
385 if (qt_last_mouse_receiver && !qt_last_mouse_receiver->internalWinId())
386 leave = qt_last_mouse_receiver.data();
387 QApplicationPrivate::dispatchEnterLeave(enter, leave, globalPosF);
388 qt_last_mouse_receiver = enter;
389 }
390 } else {
391 const QEnterEvent *ee = static_cast<QEnterEvent *>(event);
392 QWidget *child = m_widget->childAt(p: ee->position().toPoint());
393 QWidget *receiver = child ? child : m_widget.data();
394 QWidget *leave = nullptr;
395 if (QApplicationPrivate::inPopupMode() && receiver == m_widget
396 && qt_last_mouse_receiver != m_widget) {
397 // This allows to deliver the leave event to the native widget
398 // action on first-level menu.
399 leave = qt_last_mouse_receiver;
400 }
401 QApplicationPrivate::dispatchEnterLeave(enter: receiver, leave, globalPosF: ee->globalPosition());
402 qt_last_mouse_receiver = receiver;
403 }
404}

406QWidget *QWidgetWindow::getFocusWidget(FocusWidgets fw)
407{
408 QWidget *tlw = m_widget;
409 QWidget *w = tlw->nextInFocusChain();
410
411 QWidget *last = tlw;
412
413 uint focus_flag = qt_tab_all_widgets() ? Qt::TabFocus : Qt::StrongFocus;
414
415 while (w != tlw)
416 {
417 if (((w->focusPolicy() & focus_flag) == focus_flag)
418 && w->isVisibleTo(m_widget) && w->isEnabled())
419 {
420 last = w;
421 if (fw == FirstFocusWidget)
422 break;
423 }
424 w = w->nextInFocusChain();
425 }
426
427 return last;
428}

447void QWidgetWindow::handleMouseEvent(QMouseEvent *event)
448{
449 static const QEvent::Type contextMenuTrigger =
450 QGuiApplicationPrivate::platformTheme()->themeHint(hint: QPlatformTheme::ContextMenuOnMouseRelease).toBool() ?
451 QEvent::MouseButtonRelease : QEvent::MouseButtonPress;
452 if (QApplicationPrivate::inPopupMode()) {
453 QPointer<QWidget> activePopupWidget = QApplication::activePopupWidget();
454 QPoint mapped = event->position().toPoint();
455 if (activePopupWidget != m_widget)
456 mapped = activePopupWidget->mapFromGlobal(event->globalPosition().toPoint());
457 bool releaseAfter = false;
458 QWidget *popupChild = activePopupWidget->childAt(p: mapped);
459
460 if (activePopupWidget != qt_popup_down) {
461 qt_button_down = nullptr;
462 qt_popup_down = nullptr;
463 }
464
465 switch (event->type()) {
466 case QEvent::MouseButtonPress:
467 case QEvent::MouseButtonDblClick:
468 qt_button_down = popupChild;
469 qt_popup_down = activePopupWidget;
470 qt_popup_down_closed = false;
471 break;
472 case QEvent::MouseButtonRelease:
473 releaseAfter = true;
474 break;
475 default:
476 break; // nothing for mouse move
477 }
478
479 int oldOpenPopupCount = openPopupCount;
480
481 if (activePopupWidget->isEnabled()) {
482 // deliver event
483 qt_replay_popup_mouse_event = false;
484 QPointer<QWidget> receiver = activePopupWidget;
485 QPoint widgetPos = mapped;
486 if (qt_button_down)
487 receiver = qt_button_down;
488 else if (popupChild)
489 receiver = popupChild;
490 if (receiver != activePopupWidget)
491 widgetPos = receiver->mapFromGlobal(event->globalPosition().toPoint());
492
493 const bool reallyUnderMouse = activePopupWidget->rect().contains(p: mapped);
494 const bool underMouse = activePopupWidget->underMouse();
495 if (underMouse != reallyUnderMouse) {
496 if (reallyUnderMouse) {
497 const QPoint receiverMapped = receiver->mapFromGlobal(event->globalPosition().toPoint());
498 // Prevent negative mouse position on enter event - this event
499 // should be properly handled in "handleEnterLeaveEvent()".
500 if (receiverMapped.x() >= 0 && receiverMapped.y() >= 0) {
501 QApplicationPrivate::dispatchEnterLeave(enter: receiver, leave: nullptr, globalPosF: event->globalPosition());
502 qt_last_mouse_receiver = receiver;
503 }
504 } else {
505 QApplicationPrivate::dispatchEnterLeave(enter: nullptr, leave: qt_last_mouse_receiver, globalPosF: event->globalPosition());
506 qt_last_mouse_receiver = receiver;
507 receiver = activePopupWidget;
508 }
509 }
510
511 if ((event->type() != QEvent::MouseButtonPress) || !(QMutableSinglePointEvent::from(e: event)->isDoubleClick())) {
512 // if the widget that was pressed is gone, then deliver move events without buttons
513 const auto buttons = event->type() == QEvent::MouseMove && qt_popup_down_closed
514 ? Qt::NoButton : event->buttons();
515 QMouseEvent e(event->type(), widgetPos, event->scenePosition(), event->globalPosition(),
516 event->button(), buttons, event->modifiers(),
517 event->source(), event->pointingDevice());
518 e.setTimestamp(event->timestamp());
519 QApplicationPrivate::sendMouseEvent(receiver, event: &e, alienWidget: receiver, native: receiver->window(), buttonDown: &qt_button_down, lastMouseReceiver&: qt_last_mouse_receiver);
520 qt_last_mouse_receiver = receiver;
521 }
522 } else {
523 // close disabled popups when a mouse button is pressed or released
524 switch (event->type()) {
525 case QEvent::MouseButtonPress:
526 case QEvent::MouseButtonDblClick:
527 case QEvent::MouseButtonRelease:
528 activePopupWidget->close();
529 break;
530 default:
531 break;
532 }
533 }
534
535 if (QApplication::activePopupWidget() != activePopupWidget
536 && qt_replay_popup_mouse_event
537 && QGuiApplicationPrivate::platformIntegration()->styleHint(hint: QPlatformIntegration::ReplayMousePressOutsidePopup).toBool()) {
538 if (m_widget->windowType() != Qt::Popup)
539 qt_button_down = nullptr;
540 if (event->type() == QEvent::MouseButtonPress) {
541 // the popup disappeared, replay the mouse press event
542 QWidget *w = QApplication::widgetAt(p: event->globalPosition().toPoint());
543 if (w && !QApplicationPrivate::isBlockedByModal(widget: w)) {
544 // activate window of the widget under mouse pointer
545 if (!w->isActiveWindow()) {
546 w->activateWindow();
547 w->window()->raise();
548 }
549
550 if (auto win = qt_widget_private(widget: w)->windowHandle(mode: QWidgetPrivate::WindowHandleMode::Closest)) {
551 const QRect globalGeometry = win->isTopLevel()
552 ? win->geometry()
553 : QRect(win->mapToGlobal(pos: QPoint(0, 0)), win->size());
554 if (globalGeometry.contains(p: event->globalPosition().toPoint())) {
555 // Use postEvent() to ensure the local QEventLoop terminates when called from QMenu::exec()
556 const QPoint localPos = win->mapFromGlobal(pos: event->globalPosition().toPoint());
557 QMouseEvent *e = new QMouseEvent(QEvent::MouseButtonPress, localPos, localPos, event->globalPosition().toPoint(),
558 event->button(), event->buttons(), event->modifiers(), event->source());
559 QCoreApplicationPrivate::setEventSpontaneous(e, spontaneous: true);
560 e->setTimestamp(event->timestamp());
561 QCoreApplication::postEvent(receiver: win, event: e);
562 }
563 }
564 }
565 }
566 qt_replay_popup_mouse_event = false;
567#ifndef QT_NO_CONTEXTMENU
568 } else if (event->type() == contextMenuTrigger
569 && event->button() == Qt::RightButton
570 && (openPopupCount == oldOpenPopupCount)) {
571 QWidget *receiver = activePopupWidget;
572 if (qt_button_down)
573 receiver = qt_button_down;
574 else if (popupChild)
575 receiver = popupChild;
576 const QPoint localPos = receiver->mapFromGlobal(event->globalPosition().toPoint());
577 QContextMenuEvent e(QContextMenuEvent::Mouse, localPos, event->globalPosition().toPoint(), event->modifiers());
578 QApplication::forwardEvent(receiver, event: &e, originatingEvent: event);
579 }
580#else
581 Q_UNUSED(contextMenuTrigger);
582 Q_UNUSED(oldOpenPopupCount);
583 }
584#endif
585
586 if (releaseAfter) {
587 qt_button_down = nullptr;
588 qt_popup_down_closed = false;
589 qt_popup_down = nullptr;
590 }
591 return;
592 }
593
594 qt_popup_down_closed = false;
595 // modal event handling
596 if (QApplicationPrivate::instance()->modalState() && !qt_try_modal(widget: m_widget, type: event->type()))
597 return;
598
599 // which child should have it?
600 QWidget *widget = m_widget->childAt(p: event->position().toPoint());
601 QPoint mapped = event->position().toPoint();
602
603 if (!widget)
604 widget = m_widget;
605
606 const bool initialPress = event->buttons() == event->button();
607 if (event->type() == QEvent::MouseButtonPress && initialPress)
608 qt_button_down = widget;
609
610 QWidget *receiver = QApplicationPrivate::pickMouseReceiver(candidate: m_widget, windowPos: event->scenePosition().toPoint(), pos: &mapped, type: event->type(), buttons: event->buttons(),
611 buttonDown: qt_button_down, alienWidget: widget);
612 if (!receiver)
613 return;
614
615 if ((event->type() != QEvent::MouseButtonPress) || !QMutableSinglePointEvent::from(e: event)->isDoubleClick()) {
616
617 // The preceding statement excludes MouseButtonPress events which caused
618 // creation of a MouseButtonDblClick event. QTBUG-25831
619 QMouseEvent translated(event->type(), mapped, event->scenePosition(), event->globalPosition(),
620 event->button(), event->buttons(), event->modifiers(),
621 event->source(), event->pointingDevice());
622 translated.setTimestamp(event->timestamp());
623 QApplicationPrivate::sendMouseEvent(receiver, event: &translated, alienWidget: widget, native: m_widget,
624 buttonDown: &qt_button_down, lastMouseReceiver&: qt_last_mouse_receiver);
625 event->setAccepted(translated.isAccepted());
626 }
627#ifndef QT_NO_CONTEXTMENU
628 if (event->type() == contextMenuTrigger && event->button() == Qt::RightButton
629 && m_widget->rect().contains(p: event->position().toPoint())) {
630 QContextMenuEvent e(QContextMenuEvent::Mouse, mapped, event->globalPosition().toPoint(), event->modifiers());
631 QGuiApplication::forwardEvent(receiver, event: &e, originatingEvent: event);
632 if (e.isAccepted())
633 event->accept();
634 }
635#endif
636}

638void QWidgetWindow::handleTouchEvent(QTouchEvent *event)
639{
640 if (event->type() == QEvent::TouchCancel) {
641 QApplicationPrivate::translateTouchCancel(device: event->pointingDevice(), timestamp: event->timestamp());
642 event->accept();
643 } else if (QApplicationPrivate::inPopupMode()) {
644 // Ignore touch events for popups. This will cause QGuiApplication to synthesise mouse
645 // events instead, which QWidgetWindow::handleMouseEvent will forward correctly:
646 event->ignore();
647 } else {
648 event->setAccepted(QApplicationPrivate::translateRawTouchEvent(widget: m_widget, touchEvent: event));
649 }

652void QWidgetWindow::handleKeyEvent(QKeyEvent *event)
653{
654 if (QApplicationPrivate::instance()->modalState() && !qt_try_modal(widget: m_widget, type: event->type()))
655 return;
656
657 QObject *receiver = QWidget::keyboardGrabber();
658 if (!receiver && QApplicationPrivate::inPopupMode()) {
659 QWidget *popup = QApplication::activePopupWidget();
660 QWidget *popupFocusWidget = popup->focusWidget();
661 receiver = popupFocusWidget ? popupFocusWidget : popup;
662 }
663 if (!receiver)
664 receiver = focusObject();

895void QWidgetWindow::handleDragMoveEvent(QDragMoveEvent *event)
896{
897 QPointer<QWidget> widget = findDnDTarget(parent: m_widget, pos: event->position().toPoint());
898 if (!widget) {
899 event->ignore();
900 if (m_dragTarget) { // Send DragLeave to previous
901 QDragLeaveEvent leaveEvent;
902 QWidget *dragTarget = m_dragTarget;
903 m_dragTarget = nullptr;
904 QGuiApplication::forwardEvent(receiver: dragTarget, event: &leaveEvent, originatingEvent: event);
905 }
906 } else {
907 const QPoint mapped = widget->mapFromGlobal(m_widget->mapToGlobal(event->position().toPoint()));
908 QDragMoveEvent translated(mapped, event->possibleActions(), event->mimeData(),
909 event->buttons(), event->modifiers());
910
911 if (widget == m_dragTarget) { // Target widget unchanged: Send DragMove
912 translated.setDropAction(event->dropAction());
913 translated.setAccepted(event->isAccepted());
914 QGuiApplication::forwardEvent(receiver: m_dragTarget, event: &translated, originatingEvent: event);
915 } else {
916 if (m_dragTarget) { // Send DragLeave to previous
917 QDragLeaveEvent leaveEvent;
918 QWidget *dragTarget = m_dragTarget;
919 m_dragTarget = nullptr;
920 QGuiApplication::forwardEvent(receiver: dragTarget, event: &leaveEvent, originatingEvent: event);
921 }
922 // widget might have been deleted when handling the leaveEvent
923 if (widget) {
924 // Send DragEnter to new widget.
925 handleDragEnterEvent(event: static_cast<QDragEnterEvent*>(event), widget);
926 // Handling 'DragEnter' should suffice for the application.
927 translated.setDropAction(event->dropAction());
928 translated.setAccepted(event->isAccepted());
929 // The drag enter event is always immediately followed by a drag move event,
930 // see QDragEnterEvent documentation.
931 if (m_dragTarget)
932 QGuiApplication::forwardEvent(receiver: m_dragTarget, event: &translated, originatingEvent: event);
933 }
934 }
935 event->setAccepted(translated.isAccepted());
936 event->setDropAction(translated.dropAction());
937 }

967void QWidgetWindow::handleExposeEvent(QExposeEvent *event)
968{
969 if (m_widget->testAttribute(attribute: Qt::WA_DontShowOnScreen))
970 return; // Ignore for widgets that fake exposure
971
972 QWidgetPrivate *wPriv = m_widget->d_func();
973 const bool exposed = isExposed();
974
975 if (wPriv->childrenHiddenByWState) {
976 // If widgets has been previously hidden by window state change event
977 // and they aren't yet shown...
978 if (exposed) {
979 // If the window becomes exposed...
980 if (!wPriv->childrenShownByExpose) {
981 // ... and they haven't been shown by this function yet - show it.
982 wPriv->showChildren(spontaneous: true);
983 QShowEvent showEvent;
984 QCoreApplication::forwardEvent(receiver: m_widget, event: &showEvent, originatingEvent: event);
985 wPriv->childrenShownByExpose = true;
986 }
987 } else {
988 // If the window becomes not exposed...
989 if (wPriv->childrenShownByExpose) {
990 // ... and child widgets was previously shown by the expose event - hide widgets again.
991 // This is a workaround, because sometimes when window is minimized programmatically,
992 // the QPA can notify that the window is exposed after changing window state to minimized
993 // and then, the QPA can send next expose event with null exposed region (not exposed).
994 wPriv->hideChildren(spontaneous: true);
995 QHideEvent hideEvent;
996 QCoreApplication::forwardEvent(receiver: m_widget, event: &hideEvent, originatingEvent: event);
997 wPriv->childrenShownByExpose = false;
998 }
999 }
1000 }
1001
1002 if (exposed) {
1003 // QTBUG-39220, QTBUG-58575: set all (potentially fully obscured parent widgets) mapped.
1004 m_widget->setAttribute(Qt::WA_Mapped);
1005 for (QWidget *p = m_widget->parentWidget(); p && !p->testAttribute(attribute: Qt::WA_Mapped); p = p->parentWidget())
1006 p->setAttribute(Qt::WA_Mapped);
1007 if (!event->m_region.isNull())
1008 wPriv->syncBackingStore(region: event->m_region);
1009 } else {
1010 m_widget->setAttribute(Qt::WA_Mapped, on: false);
1011 }
1012}

1014void QWidgetWindow::handleWindowStateChangedEvent(QWindowStateChangeEvent *event)
1015{
1016 // QWindow does currently not know 'active'.
1017 Qt::WindowStates eventState = event->oldState();
1018 Qt::WindowStates widgetState = m_widget->windowState();
1019 Qt::WindowStates windowState = windowStates();
1020 if (widgetState & Qt::WindowActive)
1021 eventState |= Qt::WindowActive;
1022
1023 // Determine the new widget state, remember maximized/full screen
1024 // during minimized.
1025 if (windowState & Qt::WindowMinimized) {
1026 widgetState |= Qt::WindowMinimized;
1027 } else {
1028 widgetState = windowState | (widgetState & Qt::WindowActive);
1029 if (windowState) // Maximized or FullScreen
1030 updateNormalGeometry();
1031 }
1032
1033 // Sent event if the state changed (that is, it is not triggered by
1034 // QWidget::setWindowState(), which also sends an event to the widget).
1035 if (widgetState != Qt::WindowStates::Int(m_widget->data->window_state)) {
1036 m_widget->data->window_state = uint(widgetState);
1037 QWindowStateChangeEvent widgetEvent(eventState);
1038 QGuiApplication::forwardEvent(receiver: m_widget, event: &widgetEvent, originatingEvent: event);
1039 }
