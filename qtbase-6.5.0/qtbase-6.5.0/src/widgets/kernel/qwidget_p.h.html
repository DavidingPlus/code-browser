159/*!
160 \internal
161
162 Returns \c true if \a p or any of its parents enable the
163 Qt::BypassGraphicsProxyWidget window flag. Used in QWidget::show() and
164 QWidget::setParent() to determine whether it's necessary to embed the
165 widget into a QGraphicsProxyWidget or not.
166*/
167static inline bool bypassGraphicsProxyWidget(const QWidget *p)
168{
169 while (p) {
170 if (p->windowFlags() & Qt::BypassGraphicsProxyWidget)
171 return true;
172 p = p->parentWidget();
173 }
174 return false;
175}

442 // These helper functions return the (available) geometry for the screen
443 // the widget is on, and takes care if this one is embedded in a QGraphicsView.
444 static QWidget *parentGraphicsView(const QWidget *widget)
445 {
446#if QT_CONFIG(graphicsview)
447 QGraphicsProxyWidget *ancestorProxy = widget->d_func()->nearestGraphicsProxyWidget(origin: widget);
448 //It's embedded if it has an ancestor
449 if (ancestorProxy) {
450 if (!bypassGraphicsProxyWidget(p: widget) && ancestorProxy->scene() != nullptr) {
451 if (!ancestorProxy->scene()->views().empty()) {
452 return ancestorProxy->scene()->views().at(i: 0);
453 }
454 }
455 }
456#else
457 Q_UNUSED(widget);
458#endif
459 return nullptr;
460 }
