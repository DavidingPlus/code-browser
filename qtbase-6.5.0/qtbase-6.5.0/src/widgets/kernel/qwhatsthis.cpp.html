29/*!
30 \class QWhatsThis
31 \brief The QWhatsThis class provides a simple description of any
32 widget, i.e. answering the question "What's This?".
33
34 \ingroup helpsystem
35 \inmodule QtWidgets
36
37 "What's This?" help is part of an application's online help
38 system, and provides users with information about the
39 functionality and usage of a particular widget. "What's This?"
40 help texts are typically longer and more detailed than
41 \l{QToolTip}{tooltips}, but generally provide less information
42 than that supplied by separate help windows.
43
44 QWhatsThis provides a single window with an explanatory text that
45 pops up when the user asks "What's This?". The default way for
46 users to ask the question is to move the focus to the relevant
47 widget and press Shift+F1. The help text appears immediately; it
48 goes away as soon as the user does something else.
49 (Note that if there is a shortcut for Shift+F1, this mechanism
50 will not work.) Some dialogs provide a "?" button that users can
51 click to enter "What's This?" mode; they then click the relevant
52 widget to pop up the "What's This?" window. It is also possible to
53 provide a a menu option or toolbar button to switch into "What's
54 This?" mode.
55
56 To add "What's This?" text to a widget or an action, you simply
57 call QWidget::setWhatsThis() or QAction::setWhatsThis().
58
59 The text can be either rich text or plain text. If you specify a
60 rich text formatted string, it will be rendered using the default
61 stylesheet, making it possible to embed images in the displayed
62 text. To be as fast as possible, the default stylesheet uses a
63 simple method to determine whether the text can be rendered as
64 plain text. See Qt::mightBeRichText() for details.
65
66 \snippet whatsthis/whatsthis.cpp 0
67
68 An alternative way to enter "What's This?" mode is to call
69 createAction(), and add the returned QAction to either a menu or
70 a tool bar. By invoking this context help action (in the picture
71 below, the button with the arrow and question mark icon) the user
72 switches into "What's This?" mode. If they now click on a widget
73 the appropriate help text is shown. The mode is left when help is
74 given or when the user presses Esc.
75
76 \image whatsthis.png
77
78 You can enter "What's This?" mode programmatically with
79 enterWhatsThisMode(), check the mode with inWhatsThisMode(), and
80 return to normal mode with leaveWhatsThisMode().
81
82 If you want to control the "What's This?" behavior of a widget
83 manually see Qt::WA_CustomWhatsThis.
84
85 It is also possible to show different help texts for different
86 regions of a widget, by using a QHelpEvent of type
87 QEvent::WhatsThis. Intercept the help event in your widget's
88 QWidget::event() function and call QWhatsThis::showText() with the
89 text you want to display for the position specified in
90 QHelpEvent::pos(). If the text is rich text and the user clicks
91 on a link, the widget also receives a QWhatsThisClickedEvent with
92 the link's reference as QWhatsThisClickedEvent::href(). If a
93 QWhatsThisClickedEvent is handled (i.e. QWidget::event() returns
94 true), the help window remains visible. Call
95 QWhatsThis::hideText() to hide it explicitly.
96
97 \sa QToolTip
98*/

131// shadowWidth not const, for XP drop-shadow-fu turns it to 0
132static int shadowWidth = 6; // also used as '5' and '6' and even '8' below
133static const int vMargin = 8;
134static const int hMargin = 12;

136static inline bool dropShadow()
137{
138 if (const QPlatformTheme *theme = QGuiApplicationPrivate::platformTheme())
139 return theme->themeHint(hint: QPlatformTheme::DropShadow).toBool();
140 return false;
141}

143QWhatsThat::QWhatsThat(const QString& txt, QWidget* parent, QWidget *showTextFor)
144 : QWidget(parent, Qt::Popup),
145 widget(showTextFor), pressed(false), text(txt)
146{
147 delete instance;
148 instance = this;
149 setAttribute(Qt::WA_DeleteOnClose, on: true);
150 setAttribute(Qt::WA_NoSystemBackground, on: true);
151 if (parent)
152 setPalette(parent->palette());
153 setMouseTracking(true);
154 setFocusPolicy(Qt::StrongFocus);
155#ifndef QT_NO_CURSOR
156 setCursor(Qt::ArrowCursor);
157#endif
158 QRect r;
159 doc = nullptr;
160 ensurePolished(); // Ensures style sheet font before size calc
161 if (Qt::mightBeRichText(text)) {
162 doc = new QTextDocument();
163 doc->setUndoRedoEnabled(false);
164 doc->setDefaultFont(QApplication::font(this));
165#ifdef QT_NO_TEXTHTMLPARSER
166 doc->setPlainText(text);
167#else
168 doc->setHtml(text);
169#endif
170 doc->setUndoRedoEnabled(false);
171 doc->adjustSize();
172 r.setTop(0);
173 r.setLeft(0);
174 r.setSize(doc->size().toSize());
175 }
176 else
177 {
178 int sw = QGuiApplication::primaryScreen()->virtualGeometry().width() / 3;
179 if (sw < 200)
180 sw = 200;
181 else if (sw > 300)
182 sw = 300;
183
184 r = fontMetrics().boundingRect(x: 0, y: 0, w: sw, h: 1000,
185 flags: Qt::AlignLeft | Qt::AlignTop
186 | Qt::TextWordWrap | Qt::TextExpandTabs,
187 text);
188 }
189 shadowWidth = dropShadow() ? 0 : 6;
190 resize(w: r.width() + 2*hMargin + shadowWidth, h: r.height() + 2*vMargin + shadowWidth);
191}

205void QWhatsThat::mousePressEvent(QMouseEvent* e)
206{
207 pressed = true;
208 if (e->button() == Qt::LeftButton && rect().contains(p: e->position().toPoint())) {
209 if (doc)
210 anchor = doc->documentLayout()->anchorAt(pos: e->position().toPoint() - QPoint(hMargin, vMargin));
211 return;
212 }
213 close();
214}
215
216void QWhatsThat::mouseReleaseEvent(QMouseEvent* e)
217{
218 if (!pressed)
219 return;
220 if (widget && e->button() == Qt::LeftButton && doc && rect().contains(p: e->position().toPoint())) {
221 QString a = doc->documentLayout()->anchorAt(pos: e->position().toPoint() - QPoint(hMargin, vMargin));
222 QString href;
223 if (anchor == a)
224 href = a;
225 anchor.clear();
226 if (!href.isEmpty()) {
227 QWhatsThisClickedEvent e(href);
228 if (QCoreApplication::sendEvent(receiver: widget, event: &e))
229 return;
230 }
231 }
232 close();
233}

255void QWhatsThat::paintEvent(QPaintEvent*)
256{
257 const bool drawShadow = dropShadow();
258
259 QRect r = rect();
260 r.adjust(dx1: 0, dy1: 0, dx2: -1, dy2: -1);
261 if (drawShadow)
262 r.adjust(dx1: 0, dy1: 0, dx2: -shadowWidth, dy2: -shadowWidth);
263 QPainter p(this);
264 p.drawPixmap(x: 0, y: 0, pm: background);
265 p.setPen(QPen(palette().toolTipText(), 0));
266 p.setBrush(palette().toolTipBase());
267 p.drawRect(r);
268 int w = r.width();
269 int h = r.height();
270 p.setPen(palette().brush(cr: QPalette::Dark).color());
271 p.drawRect(x: 1, y: 1, w: w-2, h: h-2);
272 if (drawShadow) {
273 p.setPen(palette().shadow().color());
274 p.drawPoint(x: w + 5, y: 6);
275 p.drawLine(x1: w + 3, y1: 6, x2: w + 5, y2: 8);
276 p.drawLine(x1: w + 1, y1: 6, x2: w + 5, y2: 10);
277 int i;
278 for(i=7; i < h; i += 2)
279 p.drawLine(x1: w, y1: i, x2: w + 5, y2: i + 5);
280 for(i = w - i + h; i > 6; i -= 2)
281 p.drawLine(x1: i, y1: h, x2: i + 5, y2: h + 5);
282 for(; i > 0 ; i -= 2)
283 p.drawLine(x1: 6, y1: h + 6 - i, x2: i + 5, y2: h + 5);
284 }
285 r.adjust(dx1: 0, dy1: 0, dx2: 1, dy2: 1);
286 p.setPen(palette().toolTipText().color());
287 r.adjust(dx1: hMargin, dy1: vMargin, dx2: -hMargin, dy2: -vMargin);
288
289 if (doc) {
290 p.translate(dx: r.x(), dy: r.y());
291 QRect rect = r;
292 rect.translate(dx: -r.x(), dy: -r.y());
293 p.setClipRect(rect);
294 QAbstractTextDocumentLayout::PaintContext context;
295 context.palette.setBrush(acr: QPalette::Text, abrush: context.palette.toolTipText());
296 doc->documentLayout()->draw(painter: &p, context);
297 }
298 else
299 {
300 p.drawText(r, flags: Qt::AlignLeft | Qt::AlignTop | Qt::TextWordWrap | Qt::TextExpandTabs, text);
301 }
302}

350QWhatsThisPrivate::QWhatsThisPrivate()
351 : leaveOnMouseRelease(false)
352{
353 instance = this;
354 qApp->installEventFilter(filterObj: this);
355
356 QPoint pos = QCursor::pos();
357 if (QWidget *w = QApplication::widgetAt(p: pos)) {
358 QHelpEvent e(QEvent::QueryWhatsThis, w->mapFromGlobal(pos), pos);
359 const bool sentEvent = QCoreApplication::sendEvent(receiver: w, event: &e);
360#ifdef QT_NO_CURSOR
361 Q_UNUSED(sentEvent);
362#else
363 QGuiApplication::setOverrideCursor((!sentEvent || !e.isAccepted())?
364 Qt::ForbiddenCursor:Qt::WhatsThisCursor);
365 } else {
366 QGuiApplication::setOverrideCursor(Qt::WhatsThisCursor);
367#endif
368 }
369#if QT_CONFIG(accessibility)
370 QAccessibleEvent event(this, QAccessible::ContextHelpStart);
371 QAccessible::updateAccessibility(event: &event);
372#endif
373}

391bool QWhatsThisPrivate::eventFilter(QObject *o, QEvent *e)
392{
393 if (!o->isWidgetType())
394 return false;
395 QWidget * w = static_cast<QWidget *>(o);
396 bool customWhatsThis = w->testAttribute(attribute: Qt::WA_CustomWhatsThis);
397 switch (e->type()) {
398 case QEvent::MouseButtonPress:
399 {
400 QMouseEvent *me = static_cast<QMouseEvent*>(e);
401 if (me->button() == Qt::RightButton || customWhatsThis)
402 return false;
403 QHelpEvent e(QEvent::WhatsThis, me->position().toPoint(), me->globalPosition().toPoint());
404 if (!QCoreApplication::sendEvent(receiver: w, event: &e) || !e.isAccepted())
405 leaveOnMouseRelease = true;
406
407 } break;
408
409 case QEvent::MouseMove:
410 {
411 QMouseEvent *me = static_cast<QMouseEvent*>(e);
412 QHelpEvent e(QEvent::QueryWhatsThis, me->position().toPoint(), me->globalPosition().toPoint());
413 const bool sentEvent = QCoreApplication::sendEvent(receiver: w, event: &e);
414#ifdef QT_NO_CURSOR
415 Q_UNUSED(sentEvent);
416#else
417 QGuiApplication::changeOverrideCursor((!sentEvent || !e.isAccepted())?
418 Qt::ForbiddenCursor:Qt::WhatsThisCursor);
419#endif
420 Q_FALLTHROUGH();
421 }
422 case QEvent::MouseButtonRelease:
423 case QEvent::MouseButtonDblClick:
424 if (leaveOnMouseRelease && e->type() == QEvent::MouseButtonRelease)
425 QWhatsThis::leaveWhatsThisMode();
426 if (static_cast<QMouseEvent*>(e)->button() == Qt::RightButton || customWhatsThis)
427 return false; // ignore RMB release
428 break;
429 case QEvent::KeyPress:
430 {
431 QKeyEvent *kev = static_cast<QKeyEvent *>(e);
432#if QT_CONFIG(shortcut)
433 if (kev->matches(key: QKeySequence::Cancel)) {
434 QWhatsThis::leaveWhatsThisMode();
435 return true;
436 } else
437#endif
438 if (customWhatsThis) {
439 return false;
440 } else if (kev->key() == Qt::Key_Menu ||
441 (kev->key() == Qt::Key_F10 &&
442 kev->modifiers() == Qt::ShiftModifier)) {
443 // we don't react to these keys, they are used for context menus
444 return false;
445 } else if (kev->key() != Qt::Key_Shift && kev->key() != Qt::Key_Alt // not a modifier key
446 && kev->key() != Qt::Key_Control && kev->key() != Qt::Key_Meta) {
447 QWhatsThis::leaveWhatsThisMode();
448 }
449 } break;
450 default:
451 return false;
452 }
453 return true;
454}

537void QWhatsThisPrivate::say(QWidget * widget, const QString &text, int x, int y)
538{
539 if (text.size() == 0)
540 return;
541 // make a fresh widget, and set it up
542 QWhatsThat *whatsThat = new QWhatsThat(text, nullptr, widget);
543
544 // okay, now to find a suitable location
545 QScreen *screen = widget ? widget->screen()
546 : QGuiApplication::screenAt(point: QPoint(x, y));
547 if (!screen)
548 screen = QGuiApplication::primaryScreen();
549 QRect screenRect = screen->geometry();
550
551 int w = whatsThat->width();
552 int h = whatsThat->height();
553 int sx = screenRect.x();
554 int sy = screenRect.y();
555
556 // first try locating the widget immediately above/below,
557 // with nice alignment if possible.
558 QPoint pos;
559 if (widget)
560 pos = widget->mapToGlobal(QPoint(0,0));
561
562 if (widget && w > widget->width() + 16)
563 x = pos.x() + widget->width()/2 - w/2;
564 else
565 x = x - w/2;
566
567 // squeeze it in if that would result in part of what's this
568 // being only partially visible
569 if (x + w + shadowWidth > sx+screenRect.width()) {
570 x = (widget ? qMin(a: screenRect.width(), b: pos.x() + widget->width())
571 : screenRect.width())
572 - w;
573 }
574
575 if (x < sx)
576 x = sx;
577
578 if (widget && h > widget->height() + 16) {
579 y = pos.y() + widget->height() + 2; // below, two pixels spacing
580 // what's this is above or below, wherever there's most space
581 if (y + h + 10 > sy + screenRect.height())
582 y = pos.y() + 2 - shadowWidth - h; // above, overlap
583 }
584 y = y + 2;
585
586 // squeeze it in if that would result in part of what's this
587 // being only partially visible
588 if (y + h + shadowWidth > sy + screenRect.height()) {
589 y = (widget ? qMin(a: screenRect.height(), b: pos.y() + widget->height())
590 : screenRect.height())
591 - h;
592 }
593 if (y < sy)
594 y = sy;
595
596 whatsThat->move(ax: x, ay: y);
597 whatsThat->show();
598 whatsThat->grabKeyboard();
599}
