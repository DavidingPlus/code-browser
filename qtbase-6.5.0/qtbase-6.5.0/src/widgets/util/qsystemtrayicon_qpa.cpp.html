18QSystemTrayIconPrivate::QSystemTrayIconPrivate()

22}

24QSystemTrayIconPrivate::~QSystemTrayIconPrivate()

29void QSystemTrayIconPrivate::install_sys()

38 remove_sys_qpa();

41QRect QSystemTrayIconPrivate::geometry_sys() const
42{
43 if (!qpa_sys)
44 return QRect();
45 auto screen = QGuiApplication::primaryScreen();
46#if QT_CONFIG(menu)
47 if (menu)
48 screen = menu->screen();
49#endif
50 return QHighDpi::fromNativePixels(value: qpa_sys->geometry(), context: screen);
51}
52
53void QSystemTrayIconPrivate::updateIcon_sys()
54{
55 if (qpa_sys)
56 qpa_sys->updateIcon(icon);
57}
58
59void QSystemTrayIconPrivate::updateMenu_sys()
60{
61#if QT_CONFIG(menu)
62 if (qpa_sys && menu) {
63 addPlatformMenu(menu);
64 qpa_sys->updateMenu(menu: menu->platformMenu());
65 }
66#endif
67}

69void QSystemTrayIconPrivate::updateToolTip_sys()
70{
71 if (qpa_sys)
72 qpa_sys->updateToolTip(tooltip: toolTip);
73}

75bool QSystemTrayIconPrivate::isSystemTrayAvailable_sys()
76{
77 QScopedPointer<QPlatformSystemTrayIcon> sys(QGuiApplicationPrivate::platformTheme()->createPlatformSystemTrayIcon());
78 if (sys)
79 return sys->isSystemTrayAvailable();
80 else
81 return false;
82}

84bool QSystemTrayIconPrivate::supportsMessages_sys()
85{
86 QScopedPointer<QPlatformSystemTrayIcon> sys(QGuiApplicationPrivate::platformTheme()->createPlatformSystemTrayIcon());
87 if (sys)

89 else
90 return false;
91}
