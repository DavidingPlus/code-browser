53// returns true only if the abs. value of BOTH x and y are <= f
54inline bool operator<=(const QPointF &p, qreal f)
55{
56 return (qAbs(t: p.x()) <= f) && (qAbs(t: p.y()) <= f);
57}
58
59// returns true only if the abs. value of BOTH x and y are < f
60inline bool operator<(const QPointF &p, qreal f)
61{
62 return (qAbs(t: p.x()) < f) && (qAbs(t: p.y()) < f);
63}
64
65// returns true if the abs. value of EITHER x or y are >= f
66inline bool operator>=(const QPointF &p, qreal f)
67{
68 return (qAbs(t: p.x()) >= f) || (qAbs(t: p.y()) >= f);
69}
70
71// returns true if the abs. value of EITHER x or y are > f
72inline bool operator>(const QPointF &p, qreal f)
73{
74 return (qAbs(t: p.x()) > f) || (qAbs(t: p.y()) > f);
75}

77// returns a new point with both coordinates having the abs. value of the original one
78inline QPointF qAbs(const QPointF &p)
79{
80 return QPointF(qAbs(t: p.x()), qAbs(t: p.y()));
81}

96{
97 qreal x = qBound(min: rect.left(), val: p.x(), max: rect.right());

108// this version is not mathematically exact, but it just works for every
109// easing curve type (even custom ones)
110
111static qreal differentialForProgress(const QEasingCurve &curve, qreal pos)
112{
113 const qreal dx = 0.01;
114 qreal left = (pos < qreal(0.5)) ? pos : pos - qreal(dx);
115 qreal right = (pos >= qreal(0.5)) ? pos : pos + qreal(dx);
116 qreal d = (curve.valueForProgress(progress: right) - curve.valueForProgress(progress: left)) / qreal(dx);
117
118 qCDebug(lcScroller) << "differentialForProgress(type: " << curve.type()
119 << ", pos: " << pos << ") = " << d;
120
121 return d;
122}
123
124// this version is not mathematically exact, but it just works for every
125// easing curve type (even custom ones)
126
127static qreal progressForValue(const QEasingCurve &curve, qreal value)
128{
129 if (Q_UNLIKELY(curve.type() >= QEasingCurve::InElastic &&
130 curve.type() < QEasingCurve::Custom)) {
131 qWarning(msg: "progressForValue(): QEasingCurves of type %d do not have an "
132 "inverse, since they are not injective.", curve.type());
133 return value;
134 }
135 if (value < qreal(0) || value > qreal(1))
136 return value;
137
138 qreal progress = value, left(0), right(1);
139 for (int iterations = 6; iterations; --iterations) {
140 qreal v = curve.valueForProgress(progress);
141 if (v < value)
142 left = progress;
143 else if (v > value)
144 right = progress;
145 else
146 break;
147 progress = (left + right) / qreal(2);
148 }
149 return progress;
150}

194/*!
195 \class QScroller
196 \brief The QScroller class enables kinetic scrolling for any scrolling widget or graphics item.
197 \since 5.0
198
199 \inmodule QtWidgets
200
201 With kinetic scrolling, the user can push the widget in a given
202 direction and it will continue to scroll in this direction until it is
203 stopped either by the user or by friction. Aspects of inertia, friction
204 and other physical concepts can be changed in order to fine-tune an
205 intuitive user experience.
206
207 The QScroller object is the object that stores the current position and
208 scrolling speed and takes care of updates.
209 QScroller can be triggered by a flick gesture
210
211 \snippet code/src_widgets_util_qscroller.cpp 0
212
213 or directly like this:
214
215 \snippet code/src_widgets_util_qscroller.cpp 1
216
217 The scrolled QObjects receive a QScrollPrepareEvent whenever the scroller needs to
218 update its geometry information and a QScrollEvent whenever the content of the object should
219 actually be scrolled.
220
221 The scroller uses the global QAbstractAnimation timer to generate its QScrollEvents. This
222 can be changed with QScrollerProperties::FrameRate on a per-QScroller basis.
223
224 The \l {Dir View Example} shows one way to use a QScroller with a QTreeView.
225 An example in the \c scroller examples directory also demonstrates QScroller.
226
227 Even though this kinetic scroller has a large number of settings available via
228 QScrollerProperties, we recommend that you leave them all at their default, platform optimized
229 values. Before changing them you can experiment with the \c plot example in
230 the \c scroller examples directory.
231
232 \sa QScrollEvent, QScrollPrepareEvent, QScrollerProperties
233*/

337/*!
338 Registers a custom scroll gesture recognizer, grabs it for the \a
339 target and returns the resulting gesture type. If \a scrollGestureType is
340 set to TouchGesture the gesture triggers on touch events. If it is set to
341 one of LeftMouseButtonGesture, RightMouseButtonGesture or
342 MiddleMouseButtonGesture it triggers on mouse events of the
343 corresponding button.
344
345 Only one scroll gesture can be active on a single object at the same
346 time. If you call this function twice on the same object, it will
347 ungrab the existing gesture before grabbing the new one.
348
349 \note To avoid unwanted side-effects, mouse events are consumed while
350 the gesture is triggered. Since the initial mouse press event is
351 not consumed, the gesture sends a fake mouse release event
352 at the global position \c{(INT_MIN, INT_MIN)}. This ensures that
353 internal states of the widget that received the original mouse press
354 are consistent.
355
356 \sa ungrabGesture(), grabbedGesture()
357*/
358Qt::GestureType QScroller::grabGesture(QObject *target, ScrollerGestureType scrollGestureType)
359{
360 // ensure that a scroller for target is created
361 QScroller *s = scroller(target);
362 if (!s)
363 return Qt::GestureType(0);
364
365 QScrollerPrivate *sp = s->d_ptr;
366 if (sp->recognizer)
367 ungrabGesture(target); // ungrab the old gesture
368
369 Qt::MouseButton button;
370 switch (scrollGestureType) {
371 case LeftMouseButtonGesture : button = Qt::LeftButton; break;
372 case RightMouseButtonGesture : button = Qt::RightButton; break;
373 case MiddleMouseButtonGesture: button = Qt::MiddleButton; break;
374 default :
375 case TouchGesture : button = Qt::NoButton; break; // NoButton == Touch
376 }
377
378 sp->recognizer = new QFlickGestureRecognizer(button);
379 sp->recognizerType = QGestureRecognizer::registerRecognizer(recognizer: sp->recognizer);
380
381 if (target->isWidgetType()) {
382 QWidget *widget = static_cast<QWidget *>(target);
383 widget->grabGesture(type: sp->recognizerType);
384 if (scrollGestureType == TouchGesture)
385 widget->setAttribute(Qt::WA_AcceptTouchEvents);
386#if QT_CONFIG(graphicsview)
387 } else if (QGraphicsObject *go = qobject_cast<QGraphicsObject*>(object: target)) {
388 if (scrollGestureType == TouchGesture)
389 go->setAcceptTouchEvents(true);
390 go->grabGesture(type: sp->recognizerType);
391#endif // QT_CONFIG(graphicsview)
392 }
393 return sp->recognizerType;
394}

437 // do not delete the recognizer. The QGestureManager is doing this.

497{

499 if (d->state != Inactive) {

501 qreal snapX = d->nextSnapPos(p: here.x(), dir: 0, orientation: Qt::Horizontal);
502 qreal snapY = d->nextSnapPos(p: here.y(), dir: 0, orientation: Qt::Vertical);

508 d->contentPosition = snap;
509 d->overshootPosition = QPointF(0, 0);
510
511 d->setState(Inactive);
512 }

515/*!
516 Returns the pixel per meter metric for the scrolled widget.
517
518 The value is reported for both the x and y axis separately by using a QPointF.
519
520 \note Please note that this value should be physically correct. The actual DPI settings
521 that Qt returns for the display may be reported wrongly on purpose by the underlying
522 windowing system, for example on \macos.
523*/
524QPointF QScroller::pixelPerMeter() const
525{
526 Q_D(const QScroller);
527 QPointF ppm = d->pixelPerMeter;
528
529#if QT_CONFIG(graphicsview)
530 if (QGraphicsObject *go = qobject_cast<QGraphicsObject *>(object: d->target)) {
531 QTransform viewtr;
532 //TODO: the first view isn't really correct - maybe use an additional field in the prepare event?
533 if (const auto *scene = go->scene()) {
534 const auto views = scene->views();
535 if (!views.isEmpty())
536 viewtr = views.first()->viewportTransform();
537 }
538 QTransform tr = go->deviceTransform(viewportTransform: viewtr);
539 if (tr.isScaling()) {
540 QPointF p0 = tr.map(p: QPointF(0, 0));
541 QPointF px = tr.map(p: QPointF(1, 0));
542 QPointF py = tr.map(p: QPointF(0, 1));
543 ppm.rx() /= QLineF(p0, px).length();
544 ppm.ry() /= QLineF(p0, py).length();
545 }
546 }
547#endif // QT_CONFIG(graphicsview)
548 return ppm;
549}

551/*!
552 Returns the current scrolling velocity in meter per second when the state is Scrolling or Dragging.
553 Returns a zero velocity otherwise.
554
555 The velocity is reported for both the x and y axis separately by using a QPointF.
556
557 \sa pixelPerMeter()
558*/
559QPointF QScroller::velocity() const
560{
561 Q_D(const QScroller);
562 const QScrollerPropertiesPrivate *sp = d->properties.d.data();
563
564 switch (state()) {
565 case Dragging:
566 return d->releaseVelocity;
567 case Scrolling: {
568 QPointF vel;
569 qint64 now = d->monotonicTimer.elapsed();
570
571 if (!d->xSegments.isEmpty()) {
572 const QScrollerPrivate::ScrollSegment &s = d->xSegments.head();
573 qreal progress = qreal(now - s.startTime) / qreal(s.deltaTime);
574 qreal v = qSign(r: s.deltaPos) * qreal(s.deltaTime) / qreal(1000)
575 * sp->decelerationFactor * qreal(0.5)
576 * differentialForProgress(curve: s.curve, pos: progress);
577 vel.setX(v);
578 }
579
580 if (!d->ySegments.isEmpty()) {
581 const QScrollerPrivate::ScrollSegment &s = d->ySegments.head();
582 qreal progress = qreal(now - s.startTime) / qreal(s.deltaTime);
583 qreal v = qSign(r: s.deltaPos) * qreal(s.deltaTime) / qreal(1000)
584 * sp->decelerationFactor * qreal(0.5)
585 * differentialForProgress(curve: s.curve, pos: progress);
586 vel.setY(v);
587 }
588 return vel;
589 }
590 default:
591 return QPointF(0, 0);
592 }
593}

631/*! \overload
632
633 This version will reach its destination position in \a scrollTime milliseconds.
634*/
635void QScroller::scrollTo(const QPointF &pos, int scrollTime)
636{
637 Q_D(QScroller);
638
639 if (d->state == Pressed || d->state == Dragging )
640 return;
641
642 // no need to resend a prepare event if we are already scrolling
643 if (d->state == Inactive && !d->prepareScrolling(position: QPointF()))
644 return;
645
646 QPointF newpos = clampToRect(p: pos, rect: d->contentPosRange);
647 qreal snapX = d->nextSnapPos(p: newpos.x(), dir: 0, orientation: Qt::Horizontal);
648 qreal snapY = d->nextSnapPos(p: newpos.y(), dir: 0, orientation: Qt::Vertical);
649 if (!qIsNaN(d: snapX))
650 newpos.setX(snapX);
651 if (!qIsNaN(d: snapY))
652 newpos.setY(snapY);
653
654 qCDebug(lcScroller) << "QScroller::scrollTo(req:" << pos << " [pix] / snap:"
655 << newpos << ", " << scrollTime << " [ms])";
656
657 if (newpos == d->contentPosition + d->overshootPosition)
658 return;
659
660 QPointF vel = velocity();
661
662 if (scrollTime < 0)
663 scrollTime = 0;
664 qreal time = qreal(scrollTime) / 1000;
665
666 d->createScrollToSegments(v: vel.x(), deltaTime: time, endPos: newpos.x(), orientation: Qt::Horizontal, type: QScrollerPrivate::ScrollTypeScrollTo);
667 d->createScrollToSegments(v: vel.y(), deltaTime: time, endPos: newpos.y(), orientation: Qt::Vertical, type: QScrollerPrivate::ScrollTypeScrollTo);
668
669 if (!scrollTime)
670 d->setContentPositionHelperScrolling();
671 d->setState(scrollTime ? Scrolling : Inactive);
672}

695/*! \overload
696
697 This version will reach its destination position in \a scrollTime milliseconds.
698*/
699void QScroller::ensureVisible(const QRectF &rect, qreal xmargin, qreal ymargin, int scrollTime)
700{
701 Q_D(QScroller);
702
703 if (d->state == Pressed || d->state == Dragging )
704 return;
705
706 if (d->state == Inactive && !d->prepareScrolling(position: QPointF()))
707 return;
708
709 // -- calculate the current pos (or the position after the current scroll)
710 QPointF startPos(d->scrollingSegmentsEndPos(orientation: Qt::Horizontal),
711 d->scrollingSegmentsEndPos(orientation: Qt::Vertical));
712
713 QRectF marginRect(rect.x() - xmargin, rect.y() - ymargin,
714 rect.width() + 2 * xmargin, rect.height() + 2 * ymargin);
715
716 QSizeF visible = d->viewportSize;
717 QRectF visibleRect(startPos, visible);
718
719 qCDebug(lcScroller) << "QScroller::ensureVisible(" << rect << " [pix], " << xmargin

734

737 } else if (marginRect.left() > visibleRect.left()) {
738 newPos.setX(marginRect.left());
739 } else if (marginRect.right() < visibleRect.right()) {
740 newPos.setX(marginRect.right() - visible.width());
741 }
742
743 if (visibleRect.height() < rect.height()) {
744 // at least try to move the rect into view
745 if (rect.top() > visibleRect.top())
746 newPos.setX(rect.top());
747 else if (rect.bottom() < visibleRect.bottom())
748 newPos.setX(rect.bottom() - visible.height());
749
750 } else if (visibleRect.height() < marginRect.height()) {
751 newPos.setY(rect.center().y() - visibleRect.height() / 2);
752 } else if (marginRect.top() > visibleRect.top()) {

764}

952 State state;
953 Input input;

955 inputhandler_t handler;
956 };
957
958 statechange statechanges[] = {
959 { .state: QScroller::Inactive, .input: InputPress, .handler: &QScrollerPrivate::pressWhileInactive },
960 { .state: QScroller::Pressed, .input: InputMove, .handler: &QScrollerPrivate::moveWhilePressed },

1005/*! \internal
1006 Updates the velocity during dragging.
1007 Sets releaseVelocity.
1008*/
1009void QScrollerPrivate::updateVelocity(const QPointF &deltaPixelRaw, qint64 deltaTime)
1010{
1011 if (deltaTime <= 0)
1012 return;
1013
1014 Q_Q(QScroller);
1015 QPointF ppm = q->pixelPerMeter();
1016 const QScrollerPropertiesPrivate *sp = properties.d.data();
1017 QPointF deltaPixel = deltaPixelRaw;
1018
1019 qCDebug(lcScroller) << "QScroller::updateVelocity(" << deltaPixelRaw
1020 << " [delta pix], " << deltaTime << " [delta ms])";
1021
1022 // faster than 2.5mm/ms seems bogus (that would be a screen height in ~20 ms)
1023 if (((deltaPixelRaw / qreal(deltaTime)).manhattanLength() / ((ppm.x() + ppm.y()) / 2) * 1000) > qreal(2.5))
1024 deltaPixel = deltaPixelRaw * qreal(2.5) * ppm / 1000 / (deltaPixelRaw / qreal(deltaTime)).manhattanLength();
1025
1026 QPointF newv = -deltaPixel / qreal(deltaTime) * qreal(1000) / ppm;
1027 // around 95% of all updates are in the [1..50] ms range, so make sure
1028 // to scale the smoothing factor over that range: this way a 50ms update
1029 // will have full impact, while 5ms update will only have a 10% impact.
1030 qreal smoothing = sp->dragVelocitySmoothingFactor * qMin(a: qreal(deltaTime), b: qreal(50)) / qreal(50);
1031
1032 // only smooth if we already have a release velocity and only if the
1033 // user hasn't stopped to move his finger for more than 100ms
1034 if ((releaseVelocity != QPointF(0, 0)) && (deltaTime < 100)) {
1035 qCDebug(lcScroller) << "SMOOTHED from " << newv << " to "
1036 << newv * smoothing + releaseVelocity * (qreal(1) - smoothing);
1037 // smooth x or y only if the new velocity is either 0 or at least in
1038 // the same direction of the release velocity
1039 if (!newv.x() || (qSign(r: releaseVelocity.x()) == qSign(r: newv.x())))
1040 newv.setX(newv.x() * smoothing + releaseVelocity.x() * (qreal(1) - smoothing));
1041 if (!newv.y() || (qSign(r: releaseVelocity.y()) == qSign(r: newv.y())))
1042 newv.setY(newv.y() * smoothing + releaseVelocity.y() * (qreal(1) - smoothing));
1043 } else
1044 qCDebug(lcScroller) << "NO SMOOTHING to " << newv;
1045
1046 releaseVelocity.setX(qBound(min: -sp->maximumVelocity, val: newv.x(), max: sp->maximumVelocity));
1047 releaseVelocity.setY(qBound(min: -sp->maximumVelocity, val: newv.y(), max: sp->maximumVelocity));
1048
1049 qCDebug(lcScroller) << " --> new velocity:" << releaseVelocity;
1050}

1052void QScrollerPrivate::pushSegment(ScrollType type, qreal deltaTime, qreal stopProgress,
1053 qreal startPos, qreal deltaPos, qreal stopPos,
1054 QEasingCurve::Type curve, Qt::Orientation orientation)
1055{
1056 if (startPos == stopPos || deltaPos == 0)
1057 return;
1058
1059 ScrollSegment s;
1060 if (orientation == Qt::Horizontal && !xSegments.isEmpty()) {
1061 const auto &lastX = xSegments.constLast();

1063 } else if (orientation == Qt::Vertical && !ySegments.isEmpty()) {
1064 const auto &lastY = ySegments.constLast();

1067 s.startTime = monotonicTimer.elapsed();
1068 }
1069
1070 s.startPos = startPos;
1071 s.deltaPos = deltaPos;
1072 s.stopPos = stopPos;
1073 s.deltaTime = deltaTime * 1000;
1074 s.stopProgress = stopProgress;
1075 s.curve.setType(curve);
1076 s.type = type;
1077
1078 if (orientation == Qt::Horizontal)
1079 xSegments.enqueue(t: s);
1080 else
1081 ySegments.enqueue(t: s);
1082
1083 qCDebug(lcScroller) << "+++ Added a new ScrollSegment: " << s;
1084}

1121/*! \internal
1122 Checks if the scroller segment end in a valid position.
1123*/
1124bool QScrollerPrivate::scrollingSegmentsValid(Qt::Orientation orientation) const
1125{
1126 const QQueue<ScrollSegment> *segments;
1127 qreal minPos;
1128 qreal maxPos;
1129
1130 if (orientation == Qt::Horizontal) {
1131 segments = &xSegments;
1132 minPos = contentPosRange.left();
1133 maxPos = contentPosRange.right();
1134 } else {
1135 segments = &ySegments;
1136 minPos = contentPosRange.top();
1137 maxPos = contentPosRange.bottom();
1138 }
1139
1140 if (segments->isEmpty())
1141 return true;
1142
1143 const ScrollSegment &last = segments->last();
1144 qreal stopPos = last.stopPos;
1145
1146 if (last.type == ScrollTypeScrollTo)
1147 return true; // scrollTo is always valid
1148
1149 if (last.type == ScrollTypeOvershoot &&
1150 (stopPos != minPos && stopPos != maxPos))
1151 return false;
1152
1153 if (stopPos < minPos || stopPos > maxPos)
1154 return false;
1155
1156 if (stopPos == minPos || stopPos == maxPos) // the begin and the end of the list are always ok
1157 return true;
1158
1159 qreal nextSnap = nextSnapPos(p: stopPos, dir: 0, orientation);
1160 if (!qIsNaN(d: nextSnap) && stopPos != nextSnap)
1161 return false;
1162
1163 return true;

1194/*! \internal
1195*/
1196void QScrollerPrivate::createScrollingSegments(qreal v, qreal startPos,
1197 qreal deltaTime, qreal deltaPos,
1198 Qt::Orientation orientation)
1199{
1200 const QScrollerPropertiesPrivate *sp = properties.d.data();
1201
1202 QScrollerProperties::OvershootPolicy policy;
1203 qreal minPos;
1204 qreal maxPos;
1205 qreal viewSize;
1206
1207 if (orientation == Qt::Horizontal) {
1208 xSegments.clear();
1209 policy = sp->hOvershootPolicy;
1210 minPos = contentPosRange.left();
1211 maxPos = contentPosRange.right();
1212 viewSize = viewportSize.width();
1213 } else {
1214 ySegments.clear();
1215 policy = sp->vOvershootPolicy;
1216 minPos = contentPosRange.top();
1217 maxPos = contentPosRange.bottom();
1218 viewSize = viewportSize.height();
1219 }
1220
1221 bool alwaysOvershoot = (policy == QScrollerProperties::OvershootAlwaysOn);
1222 bool noOvershoot = (policy == QScrollerProperties::OvershootAlwaysOff) || !sp->overshootScrollDistanceFactor;
1223 bool canOvershoot = !noOvershoot && (alwaysOvershoot || maxPos);
1224
1225 qCDebug(lcScroller) << "+++ createScrollingSegments: s:" << startPos << "maxPos:" << maxPos
1226 << "o:" << int(orientation);
1227
1228 qCDebug(lcScroller) << "v = " << v << ", decelerationFactor = " << sp->decelerationFactor
1229 << ", curveType = " << sp->scrollingCurve.type();
1230
1231 qreal endPos = startPos + deltaPos;
1232
1233 qCDebug(lcScroller) << " Real Delta:" << deltaPos;
1234
1235 // -- check if are in overshoot and end in overshoot
1236 if ((startPos < minPos && endPos < minPos) ||
1237 (startPos > maxPos && endPos > maxPos)) {
1238 qreal stopPos = endPos < minPos ? minPos : maxPos;
1239 qreal oDeltaTime = sp->overshootScrollTime;
1240
1241 pushSegment(type: ScrollTypeOvershoot, deltaTime: oDeltaTime * qreal(0.7), stopProgress: qreal(1.0), startPos,
1242 deltaPos: stopPos - startPos, stopPos, curve: sp->scrollingCurve.type(), orientation);
1243 return;
1244 }
1245
1246 // -- determine snap points
1247 qreal nextSnap = nextSnapPos(p: endPos, dir: 0, orientation);
1248 qreal lowerSnapPos = nextSnapPos(p: startPos, dir: -1, orientation);
1249 qreal higherSnapPos = nextSnapPos(p: startPos, dir: 1, orientation);
1250
1251 qCDebug(lcScroller) << " Real Delta:" << lowerSnapPos << '-' << nextSnap << '-' <<higherSnapPos;
1252
1253 // - check if we can reach another snap point
1254 if (nextSnap > higherSnapPos || qIsNaN(d: higherSnapPos))
1255 higherSnapPos = nextSnap;
1256 if (nextSnap < lowerSnapPos || qIsNaN(d: lowerSnapPos))
1257 lowerSnapPos = nextSnap;
1258
1259 if (qAbs(t: v) < sp->minimumVelocity) {
1260
1261 qCDebug(lcScroller) << "### below minimum Vel" << orientation;
1262
1263 // - no snap points or already at one
1264 if (qIsNaN(d: nextSnap) || nextSnap == startPos)
1265 return; // nothing to do, no scrolling needed.
1266
1267 // - decide which point to use
1268
1269 qreal snapDistance = higherSnapPos - lowerSnapPos;
1270
1271 qreal pressDistance = (orientation == Qt::Horizontal) ?
1272 lastPosition.x() - pressPosition.x() :
1273 lastPosition.y() - pressPosition.y();
1274
1275 // if not dragged far enough, pick the next snap point.
1276 if (sp->snapPositionRatio == 0.0 || qAbs(t: pressDistance / sp->snapPositionRatio) > snapDistance)
1277 endPos = nextSnap;
1278 else if (pressDistance < 0.0)
1279 endPos = lowerSnapPos;
1280 else
1281 endPos = higherSnapPos;
1282
1283 deltaPos = endPos - startPos;
1284 qreal midPos = startPos + deltaPos * qreal(0.3);
1285 pushSegment(type: ScrollTypeFlick, deltaTime: sp->snapTime * qreal(0.3), stopProgress: qreal(1.0), startPos,
1286 deltaPos: midPos - startPos, stopPos: midPos, curve: QEasingCurve::InQuad, orientation);
1287 pushSegment(type: ScrollTypeFlick, deltaTime: sp->snapTime * qreal(0.7), stopProgress: qreal(1.0), startPos: midPos,
1288 deltaPos: endPos - midPos, stopPos: endPos, curve: sp->scrollingCurve.type(), orientation);
1289 return;
1290 }
1291
1292 // - go to the next snappoint if there is one
1293 if (v > 0 && !qIsNaN(d: higherSnapPos)) {
1294 // change the time in relation to the changed end position
1295 if (endPos - startPos)
1296 deltaTime *= qAbs(t: (higherSnapPos - startPos) / (endPos - startPos));
1297 if (deltaTime > sp->snapTime)
1298 deltaTime = sp->snapTime;
1299 endPos = higherSnapPos;
1300
1301 } else if (v < 0 && !qIsNaN(d: lowerSnapPos)) {
1302 // change the time in relation to the changed end position
1303 if (endPos - startPos)
1304 deltaTime *= qAbs(t: (lowerSnapPos - startPos) / (endPos - startPos));
1305 if (deltaTime > sp->snapTime)
1306 deltaTime = sp->snapTime;
1307 endPos = lowerSnapPos;
1308
1309 // -- check if we are overshooting
1310 } else if (endPos < minPos || endPos > maxPos) {
1311 qreal stopPos = endPos < minPos ? minPos : maxPos;
1312
1313 qCDebug(lcScroller) << "Overshoot: delta:" << (stopPos - startPos);
1314
1315 qreal stopProgress = progressForValue(curve: sp->scrollingCurve, value: qAbs(t: (stopPos - startPos) / deltaPos));
1316
1317 if (!canOvershoot) {
1318 qCDebug(lcScroller) << "Overshoot stopp:" << stopProgress;
1319
1320 pushSegment(type: ScrollTypeFlick, deltaTime, stopProgress, startPos, deltaPos: endPos, stopPos,
1321 curve: sp->scrollingCurve.type(), orientation);
1322 } else {
1323 qreal oDeltaTime = sp->overshootScrollTime;
1324 qreal oStopProgress = qMin(a: stopProgress + oDeltaTime * qreal(0.3) / deltaTime, b: qreal(1));
1325 qreal oDistance = startPos + deltaPos * sp->scrollingCurve.valueForProgress(progress: oStopProgress) - stopPos;

1327
1328 qCDebug(lcScroller) << "1 oDistance:" << oDistance << "Max:" << oMaxDistance
1329 << "stopP/oStopP" << stopProgress << oStopProgress;
1330
1331 if (qAbs(t: oDistance) > qAbs(t: oMaxDistance)) {
1332 oStopProgress = progressForValue(curve: sp->scrollingCurve,
1333 value: qAbs(t: (stopPos + oMaxDistance - startPos) / deltaPos));
1334 oDistance = oMaxDistance;
1335 qCDebug(lcScroller) << "2 oDistance:" << oDistance << "Max:" << oMaxDistance
1336 << "stopP/oStopP" << stopProgress << oStopProgress;
1337 }
1338
1339 pushSegment(type: ScrollTypeFlick, deltaTime, stopProgress: oStopProgress, startPos, deltaPos,
1340 stopPos: stopPos + oDistance, curve: sp->scrollingCurve.type(), orientation);
1341 pushSegment(type: ScrollTypeOvershoot, deltaTime: oDeltaTime * qreal(0.7), stopProgress: qreal(1.0),
1342 startPos: stopPos + oDistance, deltaPos: -oDistance, stopPos, curve: sp->scrollingCurve.type(),
1343 orientation);
1344 }
1345 return;
1346 }
1347
1348 pushSegment(type: ScrollTypeFlick, deltaTime, stopProgress: qreal(1.0), startPos, deltaPos, stopPos: endPos,
1349 curve: sp->scrollingCurve.type(), orientation);

1353void QScrollerPrivate::createScrollingSegments(const QPointF &v,
1354 const QPointF &startPos,
1355 const QPointF &ppm)
1356{
1357 const QScrollerPropertiesPrivate *sp = properties.d.data();
1358
1359 // This is only correct for QEasingCurve::OutQuad (linear velocity,
1360 // constant deceleration), but the results look and feel ok for OutExpo
1361 // and OutSine as well
1362
1363 // v(t) = deltaTime * a * 0.5 * differentialForProgress(t / deltaTime)
1364 // v(0) = vrelease
1365 // v(deltaTime) = 0
1366 // deltaTime = (2 * vrelease) / (a * differential(0))
1367
1368 // pos(t) = integrate(v(t)dt)
1369 // pos(t) = vrelease * t - 0.5 * a * t * t
1370 // pos(t) = deltaTime * a * 0.5 * progress(t / deltaTime) * deltaTime
1371 // deltaPos = pos(deltaTime)
1372
1373 QVector2D vel(v);
1374 qreal deltaTime = (qreal(2) * vel.length())
1375 / (sp->decelerationFactor * differentialForProgress(curve: sp->scrollingCurve, pos: 0));
1376 QPointF deltaPos = (vel.normalized() * QVector2D(ppm)).toPointF()
1377 * deltaTime * deltaTime * qreal(0.5) * sp->decelerationFactor;
1378
1379 createScrollingSegments(v: v.x(), startPos: startPos.x(), deltaTime, deltaPos: deltaPos.x(),
1380 orientation: Qt::Horizontal);
1381 createScrollingSegments(v: v.y(), startPos: startPos.y(), deltaTime, deltaPos: deltaPos.y(),
1382 orientation: Qt::Vertical);
1383}

1848/*! \internal
1849 Returns the next snap point in direction.
1850 If \a direction >0 it will return the next snap point that is larger than the current position.
1851 If \a direction <0 it will return the next snap point that is smaller than the current position.
1852 If \a direction ==0 it will return the nearest snap point (or the current position if we are already
1853 on a snap point.
1854 Returns the nearest snap position or NaN if no such point could be found.
1855 */
1856qreal QScrollerPrivate::nextSnapPos(qreal p, int dir, Qt::Orientation orientation) const
1857{
1858 qreal bestSnapPos = Q_QNAN;
1859 qreal bestSnapPosDist = Q_INFINITY;
1860
1861 qreal minPos;
1862 qreal maxPos;
1863
1864 if (orientation == Qt::Horizontal) {
1865 minPos = contentPosRange.left();
1866 maxPos = contentPosRange.right();
1867 } else {
1868 minPos = contentPosRange.top();
1869 maxPos = contentPosRange.bottom();
1870 }
1871
1872 if (orientation == Qt::Horizontal) {
1873 // the snap points in the list
1874 for (qreal snapPos : snapPositionsX) {
1875 qreal snapPosDist = snapPos - p;
1876 if ((dir > 0 && snapPosDist < 0) ||
1877 (dir < 0 && snapPosDist > 0))
1878 continue; // wrong direction
1879 if (snapPos < minPos || snapPos > maxPos )
1880 continue; // invalid
1881
1882 if (qIsNaN(d: bestSnapPos) ||
1883 qAbs(t: snapPosDist) < bestSnapPosDist ) {
1884 bestSnapPos = snapPos;
1885 bestSnapPosDist = qAbs(t: snapPosDist);
1886 }
1887 }
1888
1889 // the snap point interval
1890 if (snapIntervalX > 0.0) {
1891 qreal first = minPos + snapFirstX;
1892 qreal snapPos;
1893 if (dir > 0)
1894 snapPos = qCeil(v: (p - first) / snapIntervalX) * snapIntervalX + first;
1895 else if (dir < 0)
1896 snapPos = qFloor(v: (p - first) / snapIntervalX) * snapIntervalX + first;
1897 else if (p <= first)
1898 snapPos = first;
1899 else
1900 {
1901 qreal last = qFloor(v: (maxPos - first) / snapIntervalX) * snapIntervalX + first;
1902 if (p >= last)
1903 snapPos = last;
1904 else
1905 snapPos = qRound(d: (p - first) / snapIntervalX) * snapIntervalX + first;
1906 }
1907
1908 if (snapPos >= first && snapPos <= maxPos ) {
1909 qreal snapPosDist = snapPos - p;
1910
1911 if (qIsNaN(d: bestSnapPos) ||
1912 qAbs(t: snapPosDist) < bestSnapPosDist ) {
1913 bestSnapPos = snapPos;
1914 bestSnapPosDist = qAbs(t: snapPosDist);
1915 }
1916 }
1917 }
1918
1919 } else { // (orientation == Qt::Vertical)
1920 // the snap points in the list
1921 for (qreal snapPos : snapPositionsY) {
1922 qreal snapPosDist = snapPos - p;
1923 if ((dir > 0 && snapPosDist < 0) ||
1924 (dir < 0 && snapPosDist > 0))
1925 continue; // wrong direction
1926 if (snapPos < minPos || snapPos > maxPos )
1927 continue; // invalid
1928
1929 if (qIsNaN(d: bestSnapPos) ||
1930 qAbs(t: snapPosDist) < bestSnapPosDist) {
1931 bestSnapPos = snapPos;
1932 bestSnapPosDist = qAbs(t: snapPosDist);
1933 }
1934 }
1935
1936 // the snap point interval
1937 if (snapIntervalY > 0.0) {
1938 qreal first = minPos + snapFirstY;
1939 qreal snapPos;
1940 if (dir > 0)
1941 snapPos = qCeil(v: (p - first) / snapIntervalY) * snapIntervalY + first;
1942 else if (dir < 0)
1943 snapPos = qFloor(v: (p - first) / snapIntervalY) * snapIntervalY + first;
1944 else if (p <= first)
1945 snapPos = first;
1946 else
1947 {
1948 qreal last = qFloor(v: (maxPos - first) / snapIntervalY) * snapIntervalY + first;
1949 if (p >= last)
1950 snapPos = last;
1951 else
1952 snapPos = qRound(d: (p - first) / snapIntervalY) * snapIntervalY + first;
1953 }
1954
1955 if (snapPos >= first && snapPos <= maxPos ) {
1956 qreal snapPosDist = snapPos - p;
1957
1958 if (qIsNaN(d: bestSnapPos) ||
1959 qAbs(t: snapPosDist) < bestSnapPosDist) {
1960 bestSnapPos = snapPos;
1961 bestSnapPosDist = qAbs(t: snapPosDist);
1962 }
1963 }
1964 }
1965 }
1966
1967 return bestSnapPos;
1968}
