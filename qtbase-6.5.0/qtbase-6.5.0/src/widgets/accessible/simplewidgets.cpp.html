103/*! \reimp */
104QString QAccessibleButton::text(QAccessible::Text t) const
105{
106 QString str;
107 switch (t) {
108 case QAccessible::Accelerator:
109 {
110#if QT_CONFIG(shortcut) && QT_CONFIG(pushbutton)
111 QPushButton *pb = qobject_cast<QPushButton*>(object: object());
112 if (pb && pb->isDefault())
113 str = QKeySequence(Qt::Key_Enter).toString(format: QKeySequence::NativeText);
114#endif
115 if (str.isEmpty())
116 str = qt_accHotKey(text: button()->text());
117 }
118 break;
119 case QAccessible::Name:
120 str = widget()->accessibleName();
121 if (str.isEmpty())
122 str = qt_accStripAmp(text: button()->text());
123 break;
124 default:
125 break;
126 }
127 if (str.isEmpty())
128 str = QAccessibleWidget::text(t);
129 return str;
130}

132QAccessible::State QAccessibleButton::state() const
133{
134 QAccessible::State state = QAccessibleWidget::state();
135
136 QAbstractButton *b = button();
137#if QT_CONFIG(checkbox)
138 QCheckBox *cb = qobject_cast<QCheckBox *>(object: b);
139#endif
140 if (b->isCheckable())
141 state.checkable = true;
142 if (b->isChecked())
143 state.checked = true;
144#if QT_CONFIG(checkbox)
145 if (cb && cb->checkState() == Qt::PartiallyChecked)
146 state.checkStateMixed = true;
147#endif
148 if (b->isDown())
149 state.pressed = true;
150#if QT_CONFIG(pushbutton)
151 QPushButton *pb = qobject_cast<QPushButton*>(object: b);
152 if (pb) {
153 if (pb->isDefault())
154 state.defaultButton = true;
155#if QT_CONFIG(menu)
156 if (pb->menu())
157 state.hasPopup = true;
158#endif
159 }
160#endif
161
162 return state;
163}

207QStringList QAccessibleButton::actionNames() const
208{
209 QStringList names;
210 if (widget()->isEnabled()) {
211 switch (role()) {
212 case QAccessible::ButtonMenu:
213 names << showMenuAction();
214 break;
215 case QAccessible::RadioButton:
216 names << toggleAction();
217 break;
218 default:
219 if (button()->isCheckable()) {
220 names << toggleAction();
221 } else {
222 names << pressAction();
223 }
224 break;
225 }
226 }
227 names << QAccessibleWidget::actionNames();
228 return names;
229}

231void QAccessibleButton::doAction(const QString &actionName)
232{
233 if (!widget()->isEnabled())
234 return;
235 if (actionName == pressAction() ||
236 actionName == showMenuAction()) {
237#if QT_CONFIG(menu)
238 QPushButton *pb = qobject_cast<QPushButton*>(object: object());
239 if (pb && pb->menu())
240 pb->showMenu();
241 else
242#endif
243 button()->animateClick();
244 } else if (actionName == toggleAction()) {
245 button()->toggle();
246 } else {
247 QAccessibleWidget::doAction(actionName);
248 }
249}

342/*
343 The three different tool button types can have the following actions:
344| DelayedPopup | ShowMenuAction + (PressedAction || CheckedAction) |
345| MenuButtonPopup | ShowMenuAction + (PressedAction || CheckedAction) |
346| InstantPopup | ShowMenuAction |
347*/
348QStringList QAccessibleToolButton::actionNames() const
349{
350 QStringList names;
351 if (widget()->isEnabled()) {
352#if QT_CONFIG(menu)
353 if (toolButton()->menu())
354 names << showMenuAction();
355 if (toolButton()->popupMode() != QToolButton::InstantPopup)
356 names << QAccessibleButton::actionNames();
357#endif
358 }
359 return names;
360}
361
362void QAccessibleToolButton::doAction(const QString &actionName)
363{
364 if (!widget()->isEnabled())
365 return;
366
367 if (actionName == pressAction()) {
368 button()->click();
369 } else if (actionName == showMenuAction()) {
370#if QT_CONFIG(menu)
371 if (toolButton()->popupMode() != QToolButton::InstantPopup) {
372 toolButton()->setDown(true);
373 toolButton()->showMenu();
374 }
375#endif
376 } else {
377 QAccessibleButton::doAction(actionName);
378 }

436QString QAccessibleDisplay::text(QAccessible::Text t) const
437{
438 QString str;
439 switch (t) {
440 case QAccessible::Name:
441 str = widget()->accessibleName();
442 if (str.isEmpty()) {
443 if (false) {
444#if QT_CONFIG(label)
445 } else if (qobject_cast<QLabel*>(object: object())) {
446 QLabel *label = qobject_cast<QLabel*>(object: object());
447 str = label->text();
448#ifndef QT_NO_TEXTHTMLPARSER
449 if (label->textFormat() == Qt::RichText
450 || (label->textFormat() == Qt::AutoText && Qt::mightBeRichText(str))) {
451 QTextDocument doc;
452 doc.setHtml(str);
453 str = doc.toPlainText();
454 }
455#endif
456#ifndef QT_NO_SHORTCUT
457 if (label->buddy())
458 str = qt_accStripAmp(text: str);
459#endif
460#endif // QT_CONFIG(label)
461#if QT_CONFIG(lcdnumber)
462 } else if (qobject_cast<QLCDNumber*>(object: object())) {
463 QLCDNumber *l = qobject_cast<QLCDNumber*>(object: object());
464 if (l->digitCount())
465 str = QString::number(l->value());
466 else
467 str = QString::number(l->intValue());
468#endif
469#if QT_CONFIG(statusbar)
470 } else if (qobject_cast<QStatusBar*>(object: object())) {
471 return qobject_cast<QStatusBar*>(object: object())->currentMessage();
472#endif
473 }
474 }
475 break;
476 case QAccessible::Value:
477#if QT_CONFIG(progressbar)
478 if (qobject_cast<QProgressBar*>(object: object()))
479 str = QString::number(qobject_cast<QProgressBar*>(object: object())->value());
480#endif
481 break;
482 default:
483 break;
484 }
485 if (str.isEmpty())
486 str = QAccessibleWidget::text(t);
487 return str;
488}

690{

697#if QT_CONFIG(validator)

699 int pos = 0;
700 if (lineEdit()->validator()->validate(newText, pos) != QValidator::Acceptable)
701 return;
702 }
703#endif
704 lineEdit()->setText(newText);
705}

973QString QAccessibleMessageBox::text(QAccessible::Text t) const
974{
975 QString str;
976
977 switch (t) {
978 case QAccessible::Name:
979 str = QAccessibleWidget::text(t);
980 if (str.isEmpty()) // implies no title text is set
981 str = messageBox()->text();
982 break;
983 case QAccessible::Description:
984 str = widget()->accessibleDescription();
985 break;
986 case QAccessible::Value:
987 str = messageBox()->text();
988 break;
989 case QAccessible::Help:
990 str = messageBox()->informativeText();
991 break;
992 default:
993 break;
994 }
995
996 return str;
