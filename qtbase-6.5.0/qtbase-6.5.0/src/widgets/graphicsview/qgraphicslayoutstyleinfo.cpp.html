62QStyle *QGraphicsLayoutStyleInfo::style() const
63{
64 if (!m_style) {
65 Q_ASSERT(m_layout);
66 QGraphicsItem *item = m_layout->parentItem();
67 m_style = (item && item->isWidget()) ? static_cast<QGraphicsWidget*>(item)->style() : QApplication::style();
68 }
69 return m_style;
70}
