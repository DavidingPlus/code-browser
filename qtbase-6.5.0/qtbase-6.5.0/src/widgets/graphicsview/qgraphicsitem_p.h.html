294 inline qreal calcEffectiveOpacity() const
295 {
296 qreal o = opacity;
297 QGraphicsItem *p = parent;
298 int myFlags = flags;
299 while (p) {
300 int parentFlags = p->d_ptr->flags;
301
302 // If I have a parent, and I don't ignore my parent's opacity, and my
303 // parent propagates to me, then combine my local opacity with my parent's
304 // effective opacity into my effective opacity.
305 if ((myFlags & QGraphicsItem::ItemIgnoresParentOpacity)
306 || (parentFlags & QGraphicsItem::ItemDoesntPropagateOpacityToChildren)) {
307 break;
308 }
309
310 o *= p->d_ptr->opacity;
311 p = p->d_ptr->parent;
312 myFlags = parentFlags;
313 }
314 return o;
315 }

492struct QGraphicsItemPrivate::TransformData
493{
494 QTransform transform;
495 qreal scale;
496 qreal rotation;
497 qreal xOrigin;
498 qreal yOrigin;
499 QList<QGraphicsTransform *> graphicsTransforms;
500 bool onlyTransform;
501
502 TransformData() :
503 scale(1.0), rotation(0.0),
504 xOrigin(0.0), yOrigin(0.0),
505 onlyTransform(true)
506 { }
507
508 QTransform computedFullTransform(QTransform *postmultiplyTransform = nullptr) const
509 {
510 if (onlyTransform) {
511 if (!postmultiplyTransform || postmultiplyTransform->isIdentity())
512 return transform;
513 if (transform.isIdentity())
514 return *postmultiplyTransform;
515 return transform * *postmultiplyTransform;
516 }
517
518 QTransform x(transform);
519 if (!graphicsTransforms.isEmpty()) {
520 QMatrix4x4 m;
521 for (int i = 0; i < graphicsTransforms.size(); ++i)
522 graphicsTransforms.at(i)->applyTo(matrix: &m);
523 x *= m.toTransform();
524 }
525 x.translate(dx: xOrigin, dy: yOrigin);
526 x.rotate(a: rotation);
527 x.scale(sx: scale, sy: scale);
528 x.translate(dx: -xOrigin, dy: -yOrigin);
529 if (postmultiplyTransform)
530 x *= *postmultiplyTransform;
531 return x;
532 }
533};

619/*!
620 Returns \c true if \a item1 is on top of \a item2.
621 The items don't need to be siblings.
622
623 \internal
624*/
625inline bool qt_closestItemFirst(const QGraphicsItem *item1, const QGraphicsItem *item2)
626{
627 // Siblings? Just check their z-values.
628 const QGraphicsItemPrivate *d1 = item1->d_ptr.data();
629 const QGraphicsItemPrivate *d2 = item2->d_ptr.data();
630 if (d1->parent == d2->parent)
631 return qt_closestLeaf(item1, item2);
632
633 // Find common ancestor, and each item's ancestor closest to the common
634 // ancestor.
635 int item1Depth = d1->depth();
636 int item2Depth = d2->depth();
637 const QGraphicsItem *p = item1;
638 const QGraphicsItem *t1 = item1;
639 while (item1Depth > item2Depth && (p = p->d_ptr->parent)) {
640 if (p == item2) {
641 // item2 is one of item1's ancestors; item1 is on top
642 return !(t1->d_ptr->flags & QGraphicsItem::ItemStacksBehindParent);
643 }
644 t1 = p;
645 --item1Depth;
646 }
647 p = item2;
648 const QGraphicsItem *t2 = item2;
649 while (item2Depth > item1Depth && (p = p->d_ptr->parent)) {
650 if (p == item1) {
651 // item1 is one of item2's ancestors; item1 is not on top
652 return (t2->d_ptr->flags & QGraphicsItem::ItemStacksBehindParent);
653 }
654 t2 = p;
655 --item2Depth;
656 }
657
658 // item1Ancestor is now at the same level as item2Ancestor, but not the same.
659 const QGraphicsItem *p1 = t1;
660 const QGraphicsItem *p2 = t2;
661 while (t1 && t1 != t2) {
662 p1 = t1;
663 p2 = t2;
664 t1 = t1->d_ptr->parent;
665 t2 = t2->d_ptr->parent;
666 }
667
668 // in case we have a common ancestor, we compare the immediate children in the ancestor's path.
669 // otherwise we compare the respective items' topLevelItems directly.
670 return qt_closestLeaf(p1, p2);
671}

684/*!
685 \internal
686*/
687inline bool qt_closestLeaf(const QGraphicsItem *item1, const QGraphicsItem *item2)
688{
689 // Return true if sibling item1 is on top of item2.
690 const QGraphicsItemPrivate *d1 = item1->d_ptr.data();
691 const QGraphicsItemPrivate *d2 = item2->d_ptr.data();
692 bool f1 = d1->flags & QGraphicsItem::ItemStacksBehindParent;
693 bool f2 = d2->flags & QGraphicsItem::ItemStacksBehindParent;
694 if (f1 != f2)
695 return f2;
696 if (d1->z != d2->z)
697 return d1->z > d2->z;
698 return d1->siblingIndex > d2->siblingIndex;
699}

720inline void QGraphicsItemPrivate::ensureSortedChildren()
721{
722 if (needSortChildren) {
723 needSortChildren = 0;
724 sequentialOrdering = 1;
725 if (children.isEmpty())
726 return;
727 std::sort(first: children.begin(), last: children.end(), comp: qt_notclosestLeaf);
728 for (int i = 0; i < children.size(); ++i) {
729 if (children.at(i)->d_ptr->siblingIndex != i) {
730 sequentialOrdering = 0;
731 break;
732 }
733 }
734 }

755 ->source->d_func())->invalidateCache();
756 }
757#endif
758 while (parentp->parent) {
759 parentp = parentp->parent->d_ptr.data();
760 parentp->dirtyChildren = 1;
761
762 if (updateBoundingRect) {
763 parentp->dirtyChildrenBoundingRect = 1;
764 // ### Only do this if the parent's effect applies to the entire subtree.
765 parentp->notifyBoundingRectChanged = 1;
766 }
767#if QT_CONFIG(graphicseffect)
768 if (parentp->graphicsEffect) {
769 if (updateBoundingRect) {
770 static_cast<QGraphicsItemEffectSourcePrivate *>(parentp->graphicsEffect->d_func()
771 ->source->d_func())->invalidateCache();
772 parentp->notifyInvalidated = 1;
773 }
774 if (parentp->scene && parentp->graphicsEffect->isEnabled()) {
775 parentp->dirty = 1;
776 parentp->fullUpdatePending = 1;
777 }
778 }
779#endif
780 }
