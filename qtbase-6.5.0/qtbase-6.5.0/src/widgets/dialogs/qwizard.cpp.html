67static QWidget *iWantTheFocus(QWidget *ancestor)
68{
69 const int MaxIterations = 100;
70
71 QWidget *candidate = ancestor;
72 for (int i = 0; i < MaxIterations; ++i) {
73 candidate = candidate->nextInFocusChain();
74 if (!candidate)
75 break;
76
77 if (candidate->focusPolicy() & Qt::TabFocus) {
78 if (candidate != ancestor && ancestor->isAncestorOf(child: candidate))
79 return candidate;
80 }
81 }
82 return nullptr;
83}

85static bool objectInheritsXAndXIsCloserThanY(const QObject *object, const QByteArray &classX,
86 const QByteArray &classY)
87{
88 const QMetaObject *metaObject = object->metaObject();
89 while (metaObject) {
90 if (metaObject->className() == classX)
91 return true;
92 if (metaObject->className() == classY)
93 return false;
94 metaObject = metaObject->superClass();
95 }
96 return false;
97}

114static const char *changed_signal(int which)
115{
116 // since it might expand to a runtime function call (to
117 // qFlagLocations()), we cannot store the result of SIGNAL() in a
118 // character array and expect it to be statically initialized. To
119 // avoid the relocations caused by a char pointer table, use a
120 // switch statement:
121 switch (which) {
122 case 0: return SIGNAL(toggled(bool));
123 case 1: return SIGNAL(valueChanged(int));
124 case 2: return SIGNAL(currentIndexChanged(int));
125 case 3: return SIGNAL(dateTimeChanged(QDateTime));
126 case 4: return SIGNAL(textChanged(QString));
127 case 5: return SIGNAL(currentRowChanged(int));
128 case 6: return SIGNAL(valueChanged(int));
129 };
130 static_assert(7 == NFallbackDefaultProperties);
131 Q_UNREACHABLE_RETURN(nullptr);
132}

324void QWizardHeader::setup(const QWizardLayoutInfo &info, const QString &title,
325 const QString &subTitle, const QPixmap &logo, const QPixmap &banner,
326 Qt::TextFormat titleFormat, Qt::TextFormat subTitleFormat)
327{
328 bool modern = ((info.wizStyle == QWizard::ModernStyle)
329#if QT_CONFIG(style_windowsvista)
330 || ((info.wizStyle == QWizard::AeroStyle
331 && QVistaHelper::vistaState() == QVistaHelper::Classic) || vistaDisabled())
332#endif
333 );
334
335 layout->setRowMinimumHeight(row: 0, minSize: modern ? ModernHeaderTopMargin : 0);
336 layout->setRowMinimumHeight(row: 1, minSize: modern ? info.topLevelMarginTop - ModernHeaderTopMargin - 1 : 0);
337 layout->setRowMinimumHeight(row: 6, minSize: (modern ? 3 : GapBetweenLogoAndRightEdge) + 2);
338
339 int minColumnWidth0 = modern ? info.topLevelMarginLeft + info.topLevelMarginRight : 0;
340 int minColumnWidth1 = modern ? info.topLevelMarginLeft + info.topLevelMarginRight + 1
341 : info.topLevelMarginLeft + ClassicHMargin;
342 layout->setColumnMinimumWidth(column: 0, minSize: minColumnWidth0);
343 layout->setColumnMinimumWidth(column: 1, minSize: minColumnWidth1);
344
345 titleLabel->setTextFormat(titleFormat);
346 titleLabel->setText(title);
347 logoLabel->setPixmap(logo);
348
349 subTitleLabel->setTextFormat(subTitleFormat);
350 subTitleLabel->setText("Pq\nPq"_L1);
351 int desiredSubTitleHeight = subTitleLabel->sizeHint().height();
352 subTitleLabel->setText(subTitle);
353
354 if (modern) {
355 bannerPixmap = banner;
356 } else {
357 bannerPixmap = QPixmap();
358 }
359
360 if (bannerPixmap.isNull()) {
361 /*
362 There is no widthForHeight() function, so we simulate it with a loop.
363 */
364 int candidateSubTitleWidth = qMin(a: 512, b: 2 * QGuiApplication::primaryScreen()->virtualGeometry().width() / 3);
365 int delta = candidateSubTitleWidth >> 1;
366 while (delta > 0) {
367 if (subTitleLabel->heightForWidth(candidateSubTitleWidth - delta)
368 <= desiredSubTitleHeight)
369 candidateSubTitleWidth -= delta;
370 delta >>= 1;
371 }
372
373 subTitleLabel->setMinimumSize(minw: candidateSubTitleWidth, minh: desiredSubTitleHeight);
374
375 QSize size = layout->totalMinimumSize();
376 setMinimumSize(size);
377 setMaximumSize(QWIDGETSIZE_MAX, maxh: size.height());
378 } else {
379 subTitleLabel->setMinimumSize(minw: 0, minh: 0);
380 setFixedSize(banner.size() + QSize(0, 2));
381 }
382 updateGeometry();
383}

758void QWizardPrivate::switchToPage(int newId, Direction direction)
759{
760 Q_Q(QWizard);
761
762 disableUpdates();
763
764 int oldId = current;
765 if (QWizardPage *oldPage = q->currentPage()) {
766 oldPage->hide();
767
768 if (direction == Backward) {
769 if (!(opts & QWizard::IndependentPages)) {
770 q->cleanupPage(id: oldId);
771 oldPage->d_func()->initialized = false;
772 }
773 Q_ASSERT(history.constLast() == oldId);
774 history.removeLast();
775 Q_ASSERT(history.constLast() == newId);
776 }
777 }
778
779 current = newId;
780
781 QWizardPage *newPage = q->currentPage();
782 if (newPage) {
783 if (direction == Forward) {
784 if (!newPage->d_func()->initialized) {
785 newPage->d_func()->initialized = true;
786 q->initializePage(id: current);
787 }
788 history.append(t: current);
789 }
790 newPage->show();
791 }
792
793 canContinue = (q->nextId() != -1);
794 canFinish = (newPage && newPage->isFinalPage());
795
796 _q_updateButtonStates();
797 updateButtonTexts();
798
799 const QWizard::WizardButton nextOrCommit =
800 newPage && newPage->isCommitPage() ? QWizard::CommitButton : QWizard::NextButton;
801 QAbstractButton *nextOrFinishButton =
802 btns[canContinue ? nextOrCommit : QWizard::FinishButton];
803 QWidget *candidate = nullptr;
804
805 /*
806 If there is no default button and the Next or Finish button
807 is enabled, give focus directly to it as a convenience to the
808 user. This is the normal case on OS X.
809
810 Otherwise, give the focus to the new page's first child that
811 can handle it. If there is no such child, give the focus to
812 Next or Finish.
813 */
814 if ((opts & QWizard::NoDefaultButton) && nextOrFinishButton->isEnabled()) {
815 candidate = nextOrFinishButton;
816 } else if (newPage) {
817 candidate = iWantTheFocus(ancestor: newPage);
818 }
819 if (!candidate)
820 candidate = nextOrFinishButton;
821 candidate->setFocus();
822
823 if (wizStyle == QWizard::MacStyle)
824 q->updateGeometry();
825
826 enableUpdates();
827 updateLayout();
828 updatePalette();
829
830 emit q->currentIdChanged(id: current);
831}

1194{

1196

1200 if (layoutInfo != info)

1205 // than the QSpacerItem at the bottom. Otherwise, let the QSpacerItem
1206 // stretch "infinitely" more than the page. Change the bottom item's
1207 // policy accordingly. The case that the page has no layout is basically
1208 // for Designer, only.
1209 if (page) {
1210 bool expandPage = !page->layout();
1211 if (!expandPage) {
1212 const QLayoutItem *pageItem = pageVBoxLayout->itemAt(pageVBoxLayout->indexOf(page));
1213 expandPage = pageItem->expandingDirections() & Qt::Vertical;
1214 }
1215 QSpacerItem *bottomSpacer = pageVBoxLayout->itemAt(pageVBoxLayout->count() - 1)->spacerItem();
1216 Q_ASSERT(bottomSpacer);
1217 bottomSpacer->changeSize(w: 0, h: 0, hData: QSizePolicy::Ignored, vData: expandPage ? QSizePolicy::Ignored : QSizePolicy::MinimumExpanding);
1218 pageVBoxLayout->invalidate();
1219 }
1220
1221 if (info.header) {
1222 Q_ASSERT(page);
1223 headerWidget->setup(info, title: page->title(), subTitle: page->subTitle(),
1224 logo: page->pixmap(which: QWizard::LogoPixmap), banner: page->pixmap(which: QWizard::BannerPixmap),
1225 titleFormat: titleFmt, subTitleFormat: subTitleFmt);
1226 }

1229 QPixmap pix;
1230 if (info.watermark) {
1231 if (page)
1232 pix = page->pixmap(which: QWizard::WatermarkPixmap);
1233 else
1234 pix = q->pixmap(which: QWizard::WatermarkPixmap);
1235 }
1236 watermarkLabel->setPixmap(pix); // in case there is no watermark and we show the side widget we need to clear the watermark

1251 updateMinMaxSizes(info);
1252}

1775 find difficult to learn.

1880 displayed on the left side, below the header. At the bottom,
