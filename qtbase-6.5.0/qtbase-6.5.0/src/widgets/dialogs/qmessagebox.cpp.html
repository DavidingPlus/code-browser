329void QMessageBoxPrivate::updateSize()
330{
331 Q_Q(QMessageBox);
332
333 if (!q->isVisible())
334 return;
335
336 const QSize screenSize = q->screen()->availableGeometry().size();
337 int hardLimit = qMin(a: screenSize.width() - 480, b: 1000); // can never get bigger than this
338 // on small screens allows the messagebox be the same size as the screen
339 if (screenSize.width() <= 1024)
340 hardLimit = screenSize.width();
341#ifdef Q_OS_MAC
342 int softLimit = qMin(screenSize.width()/2, 420);
343#else
344 // note: ideally on windows, hard and soft limits but it breaks compat
345 int softLimit = qMin(a: screenSize.width()/2, b: 500);
346#endif
347
348 if (informativeLabel)
349 informativeLabel->setSizePolicy(hor: QSizePolicy::Ignored, ver: QSizePolicy::Ignored);
350
351 label->setWordWrap(false); // makes the label return min size
352 int width = layoutMinimumWidth();
353
354 if (width > softLimit) {
355 label->setWordWrap(true);
356 width = qMax(a: softLimit, b: layoutMinimumWidth());
357
358 if (width > hardLimit) {
359 label->d_func()->ensureTextControl();
360 if (QWidgetTextControl *control = label->d_func()->control) {
361 QTextOption opt = control->document()->defaultTextOption();
362 opt.setWrapMode(QTextOption::WrapAnywhere);
363 control->document()->setDefaultTextOption(opt);
364 }
365 width = hardLimit;
366 }
367 }
368
369 if (informativeLabel) {
370 label->setSizePolicy(hor: QSizePolicy::Ignored, ver: QSizePolicy::Ignored);
371 QSizePolicy policy(QSizePolicy::Minimum, QSizePolicy::Preferred);
372 policy.setHeightForWidth(true);
373 informativeLabel->setSizePolicy(policy);
374 width = qMax(a: width, b: layoutMinimumWidth());
375 if (width > hardLimit) { // longest word is really big, so wrap anywhere
376 informativeLabel->d_func()->ensureTextControl();
377 if (QWidgetTextControl *control = informativeLabel->d_func()->control) {
378 QTextOption opt = control->document()->defaultTextOption();
379 opt.setWrapMode(QTextOption::WrapAnywhere);
380 control->document()->setDefaultTextOption(opt);
381 }
382 width = hardLimit;
383 }
384 policy.setHeightForWidth(label->wordWrap());
385 label->setSizePolicy(policy);
386 }
387
388 QFontMetrics fm(QApplication::font(className: "QMdiSubWindowTitleBar"));
389 int windowTitleWidth = qMin(a: fm.horizontalAdvance(q->windowTitle()) + 50, b: hardLimit);
390 if (windowTitleWidth > width)
391 width = windowTitleWidth;
392
393 layout->activate();
394 int height = (layout->hasHeightForWidth())
395 ? layout->totalHeightForWidth(w: width)
396 : layout->totalMinimumSize().height();
397
398 q->setFixedSize(w: width, h: height);
399 QCoreApplication::removePostedEvents(receiver: q, eventType: QEvent::LayoutRequest);
400}

510/*!
511 \class QMessageBox
512
513 \brief The QMessageBox class provides a modal dialog for informing
514 the user or for asking the user a question and receiving an answer.
515
516 \ingroup standard-dialogs
517 \inmodule QtWidgets
518
519 A message box displays a primary \l{QMessageBox::text}{text} to
520 alert the user to a situation, an \l{QMessageBox::informativeText}
521 {informative text} to further explain the situation, and an optional
522 \l{QMessageBox::detailedText} {detailed text} to provide even more data
523 if the user requests it.
524
525 A message box can also display an \l{QMessageBox::icon} {icon}
526 and \l{QMessageBox::standardButtons} {standard buttons} for
527 accepting a user response.
528
529 Two APIs for using QMessageBox are provided, the property-based
530 API, and the static functions. Calling one of the static functions
531 is the simpler approach, but it is less flexible than using the
532 property-based API, and the result is less informative. Using the
533 property-based API is recommended.
534
535 \section1 The Property-based API
536
537 To use the property-based API, construct an instance of
538 QMessageBox, set the desired properties, and call exec() to show
539 the message. The simplest configuration is to set only the
540 \l{QMessageBox::text} {message text} property.
541
542 \snippet code/src_gui_dialogs_qmessagebox.cpp 5
543
544 The user must click the \uicontrol{OK} button to dismiss the message
545 box. The rest of the GUI is blocked until the message box is
546 dismissed.
547
548 \image msgbox1.png
549
550 A better approach than just alerting the user to an event is to
551 also ask the user what to do about it.
552
553 Set the \l{QMessageBox::standardButtons} {standard buttons}
554 property to the set of buttons you want as the set of user
555 responses. The buttons are specified by combining values from
556 StandardButtons using the bitwise OR operator. The display order
557 for the buttons is platform-dependent. For example, on Windows,
558 \uicontrol{Save} is displayed to the left of \uicontrol{Cancel}, whereas on
559 \macos, the order is reversed. Mark one of your standard buttons to be your
560 \l{QMessageBox::defaultButton()} {default button}.
561
562 The \l{QMessageBox::informativeText} {informative text} property can
563 be used to add additional context to help the user choose the appropriate action.
564
565 \snippet code/src_gui_dialogs_qmessagebox.cpp 6
566
567 \image msgbox2.png
568
569 The exec() slot returns the StandardButtons value of the button
570 that was clicked.
571
572 \snippet code/src_gui_dialogs_qmessagebox.cpp 7
573
574 To give the user more information to help them choose the appropriate,
575 action, set the \l{QMessageBox::detailedText} {detailed text} property.
576 Depending on the platform the \l{QMessageBox::detailedText} {detailed text},
577 may require the user to click a \uicontrol{Show Details...} button to be shown.
578
579 \image msgbox3.png
580
581 Clicking the \uicontrol{Show Details...} button displays the detailed text.
582
583 \image msgbox4.png
584
585 \section2 Rich Text and the Text Format Property
586
587 The \l{QMessageBox::detailedText} {detailed text} property is
588 always interpreted as plain text. The \l{QMessageBox::text} {main
589 text} and \l{QMessageBox::informativeText} {informative text}
590 properties can be either plain text or rich text. These strings
591 are interpreted according to the setting of the
592 \l{QMessageBox::textFormat} {text format} property. The default
593 setting is \l{Qt::AutoText} {auto-text}.
594
595 Note that for some plain text strings containing XML
596 meta-characters, the auto-text \l{Qt::mightBeRichText()} {rich
597 text detection test} may fail causing your plain text string to be
598 interpreted incorrectly as rich text. In these rare cases, use
599 Qt::convertFromPlainText() to convert your plain text string to a
600 visually equivalent rich text string, or set the
601 \l{QMessageBox::textFormat} {text format} property explicitly with
602 setTextFormat().
603
604 \section2 Severity Levels and the Icon and Pixmap Properties
605
606 QMessageBox supports four predefined message severity levels, or message
607 types, which really only differ in the predefined icon they each show.
608 Specify one of the four predefined message types by setting the
609 \l{QMessageBox::icon}{icon} property to one of the
610 \l{QMessageBox::Icon}{predefined icons}. The following rules are
611 guidelines:
612
613 \table
614 \row
615 \li \image qmessagebox-quest.png
616 \li \l Question
617 \li For asking a question during normal operations.
618 \row
619 \li \image qmessagebox-info.png
620 \li \l Information
621 \li For reporting information about normal operations.
622 \row
623 \li \image qmessagebox-warn.png
624 \li \l Warning
625 \li For reporting non-critical errors.
626 \row
627 \li \image qmessagebox-crit.png
628 \li \l Critical
629 \li For reporting critical errors.
630 \endtable
631
632 \l{QMessageBox::Icon}{Predefined icons} are not defined by QMessageBox, but
633 provided by the style. The default value is \l{QMessageBox::NoIcon}
634 {No Icon}. The message boxes are otherwise the same for all cases. When
635 using a standard icon, use the one recommended in the table, or use the
636 one recommended by the style guidelines for your platform. If none of the
637 standard icons is right for your message box, you can use a custom icon by
638 setting the \l{QMessageBox::iconPixmap}{icon pixmap} property instead of
639 setting the \l{QMessageBox::icon}{icon} property.
640
641 In summary, to set an icon, use \e{either} setIcon() for one of the
642 standard icons, \e{or} setIconPixmap() for a custom icon.
643
644 \section1 The Static Functions API
645
646 Building message boxes with the static functions API, although
647 convenient, is less flexible than using the property-based API,
648 because the static function signatures lack parameters for setting
649 the \l{QMessageBox::informativeText} {informative text} and
650 \l{QMessageBox::detailedText} {detailed text} properties. One
651 work-around for this has been to use the \c{title} parameter as
652 the message box main text and the \c{text} parameter as the
653 message box informative text. Because this has the obvious
654 drawback of making a less readable message box, platform
655 guidelines do not recommend it. The \e{Microsoft Windows User
656 Interface Guidelines} recommend using the
657 \l{QCoreApplication::applicationName} {application name} as the
658 \l{QMessageBox::setWindowTitle()} {window's title}, which means
659 that if you have an informative text in addition to your main
660 text, you must concatenate it to the \c{text} parameter.
661
662 Note that the static function signatures have changed with respect
663 to their button parameters, which are now used to set the
664 \l{QMessageBox::standardButtons} {standard buttons} and the
665 \l{QMessageBox::defaultButton()} {default button}.
666
667 Static functions are available for creating information(),
668 question(), warning(), and critical() message boxes.

1060void QMessageBoxPrivate::detectEscapeButton()
1061{
1062 if (escapeButton) { // escape button explicitly set
1063 detectedEscapeButton = escapeButton;
1064 return;
1065 }
1066
1067 // Cancel button automatically becomes escape button
1068 detectedEscapeButton = buttonBox->button(which: QDialogButtonBox::Cancel);
1069 if (detectedEscapeButton)
1070 return;
1071
1072 // If there is only one button, make it the escape button
1073 const QList<QAbstractButton *> buttons = buttonBox->buttons();
1074 if (buttons.size() == 1) {
1075 detectedEscapeButton = buttons.first();
1076 return;
1077 }
1078
1079 // If there are two buttons and one of them is the "Show Details..."
1080 // button, then make the other one the escape button
1081 if (buttons.size() == 2 && detailsButton) {
1082 auto idx = buttons.indexOf(t: detailsButton);
1083 if (idx != -1) {
1084 detectedEscapeButton = buttons.at(i: 1 - idx);
1085 return;
1086 }
1087 }
1088
1089 // if the message box has one RejectRole button, make it the escape button
1090 for (auto *button : buttons) {
1091 if (buttonBox->buttonRole(button) == QDialogButtonBox::RejectRole) {
1092 if (detectedEscapeButton) { // already detected!
1093 detectedEscapeButton = nullptr;
1094 break;
1095 }
1096 detectedEscapeButton = button;
1097 }
1098 }
1099 if (detectedEscapeButton)
1100 return;
1101
1102 // if the message box has one NoRole button, make it the escape button
1103 for (auto *button : buttons) {
1104 if (buttonBox->buttonRole(button) == QDialogButtonBox::NoRole) {
1105 if (detectedEscapeButton) { // already detected!
1106 detectedEscapeButton = nullptr;
1107 break;
1108 }
1109 detectedEscapeButton = button;
1110 }
1111 }

1642 QMessageBox::Icon icon,
1643 const QString& title, const QString& text,
1644 QMessageBox::StandardButtons buttons,
1645 QMessageBox::StandardButton defaultButton)
1646{
1647 // necessary for source compatibility with Qt 4.0 and 4.1
1648 // handles (Yes, No) and (Yes|Default, No)
1649 if (defaultButton && !(buttons & defaultButton))
1650 return (QMessageBox::StandardButton)
1651 QMessageBoxPrivate::showOldMessageBox(parent, icon, title,
1652 text, button0: int(buttons),

1661 uint sb = buttons & mask;
1662 mask <<= 1;
1663 if (!sb)

1670 || (defaultButton != QMessageBox::NoButton && sb == uint(defaultButton)))
