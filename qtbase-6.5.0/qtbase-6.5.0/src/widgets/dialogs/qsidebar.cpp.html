54/*!
55 \reimp
56*/
57Qt::ItemFlags QUrlModel::flags(const QModelIndex &index) const
58{
59 Qt::ItemFlags flags = QStandardItemModel::flags(index);
60 if (index.isValid()) {
61 flags &= ~Qt::ItemIsEditable;
62 // ### some future version could support "moving" urls onto a folder
63 flags &= ~Qt::ItemIsDropEnabled;
64 }
65
66 if (index.data(arole: Qt::DecorationRole).isNull())
67 flags &= ~Qt::ItemIsEnabled;
68
69 return flags;
70}

75QMimeData *QUrlModel::mimeData(const QModelIndexList &indexes) const
76{
77 QList<QUrl> list;
78 for (const auto &index : indexes) {
79 if (index.column() == 0)
80 list.append(t: index.data(arole: UrlRole).toUrl());
81 }
82 QMimeData *data = new QMimeData();
83 data->setUrls(list);
84 return data;
85}

89/*!
90 Decide based upon the data if it should be accepted or not
91
92 We only accept dirs and not files
93*/
94bool QUrlModel::canDrop(QDragEnterEvent *event)
95{
96 if (!event->mimeData()->formats().contains(str: mimeTypes().constFirst()))
97 return false;
98
99 const QList<QUrl> list = event->mimeData()->urls();
100 for (const auto &url : list) {
101 const QModelIndex idx = fileSystemModel->index(path: url.toLocalFile());
102 if (!fileSystemModel->isDir(index: idx))
103 return false;
104 }
105 return true;
106}

108/*!
109 \reimp
110*/
111bool QUrlModel::dropMimeData(const QMimeData *data, Qt::DropAction action,
112 int row, int column, const QModelIndex &parent)
113{
114 if (!data->formats().contains(str: mimeTypes().constFirst()))
115 return false;
116 Q_UNUSED(action);
117 Q_UNUSED(column);
118 Q_UNUSED(parent);
119 addUrls(urls: data->urls(), row);
120 return true;
121}

125/*!
126 \reimp
127
128 If the role is the UrlRole then handle otherwise just pass to QStandardItemModel
129*/
130bool QUrlModel::setData(const QModelIndex &index, const QVariant &value, int role)
131{
132 if (value.userType() == QMetaType::QUrl) {
133 QUrl url = value.toUrl();
134 QModelIndex dirIndex = fileSystemModel->index(path: url.toLocalFile());
135 //On windows the popup display the "C:\", convert to nativeSeparators
136 if (showFullPath)
137 QStandardItemModel::setData(index, value: QDir::toNativeSeparators(pathName: fileSystemModel->data(index: dirIndex, role: QFileSystemModel::FilePathRole).toString()));
138 else {
139 QStandardItemModel::setData(index, value: QDir::toNativeSeparators(pathName: fileSystemModel->data(index: dirIndex, role: QFileSystemModel::FilePathRole).toString()), role: Qt::ToolTipRole);
140 QStandardItemModel::setData(index, value: fileSystemModel->data(index: dirIndex).toString());
141 }
142 QStandardItemModel::setData(index, value: fileSystemModel->data(index: dirIndex, role: Qt::DecorationRole),
143 role: Qt::DecorationRole);
144 QStandardItemModel::setData(index, value: url, role: UrlRole);
145 return true;
146 }
147 return QStandardItemModel::setData(index, value, role);
148}
149
150void QUrlModel::setUrl(const QModelIndex &index, const QUrl &url, const QModelIndex &dirIndex)
151{
152 setData(index, value: url, role: UrlRole);
153 if (url.path().isEmpty()) {
154 setData(index, value: fileSystemModel->myComputer());
155 setData(index, value: fileSystemModel->myComputer(role: Qt::DecorationRole), role: Qt::DecorationRole);
156 } else {
157 QString newName;
158 if (showFullPath) {
159 //On windows the popup display the "C:\", convert to nativeSeparators
160 newName = QDir::toNativeSeparators(pathName: dirIndex.data(arole: QFileSystemModel::FilePathRole).toString());
161 } else {
162 newName = dirIndex.data().toString();
163 }
164
165 QIcon newIcon = qvariant_cast<QIcon>(v: dirIndex.data(arole: Qt::DecorationRole));
166 if (!dirIndex.isValid()) {
167 const QAbstractFileIconProvider *provider = fileSystemModel->iconProvider();
168 if (provider)
169 newIcon = provider->icon(QAbstractFileIconProvider::Folder);
170 newName = QFileInfo(url.toLocalFile()).fileName();
171 if (!invalidUrls.contains(t: url))
172 invalidUrls.append(t: url);
173 //The bookmark is invalid then we set to false the EnabledRole
174 setData(index, value: false, role: EnabledRole);
175 } else {
176 //The bookmark is valid then we set to true the EnabledRole
177 setData(index, value: true, role: EnabledRole);
178 }
179
180 // Make sure that we have at least 32x32 images
181 const QSize size = newIcon.actualSize(size: QSize(32,32));
182 if (size.width() < 32) {
183 QPixmap smallPixmap = newIcon.pixmap(size: QSize(32, 32));
184 newIcon.addPixmap(pixmap: smallPixmap.scaledToWidth(w: 32, mode: Qt::SmoothTransformation));
185 }
186
187 if (index.data().toString() != newName)
188 setData(index, value: newName);
189 QIcon oldIcon = qvariant_cast<QIcon>(v: index.data(arole: Qt::DecorationRole));
190 if (oldIcon.cacheKey() != newIcon.cacheKey())
191 setData(index, value: newIcon, role: Qt::DecorationRole);
192 }
193}

195void QUrlModel::setUrls(const QList<QUrl> &list)
196{
197 removeRows(row: 0, count: rowCount());
198 invalidUrls.clear();
199 watching.clear();
200 addUrls(urls: list, row: 0);
201}

203/*!
204 Add urls \a list into the list at \a row. If move then movie
205 existing ones to row.
206
207 \sa dropMimeData()
208*/
209void QUrlModel::addUrls(const QList<QUrl> &list, int row, bool move)
210{
211 if (row == -1)
212 row = rowCount();
213 row = qMin(a: row, b: rowCount());
214 for (int i = list.size() - 1; i >= 0; --i) {
215 QUrl url = list.at(i);
216 if (!url.isValid() || url.scheme() != "file"_L1)
217 continue;
218 //this makes sure the url is clean
219 const QString cleanUrl = QDir::cleanPath(path: url.toLocalFile());
220 if (!cleanUrl.isEmpty())
221 url = QUrl::fromLocalFile(localfile: cleanUrl);
222
223 for (int j = 0; move && j < rowCount(); ++j) {
224 QString local = index(row: j, column: 0).data(arole: UrlRole).toUrl().toLocalFile();
225#if defined(Q_OS_WIN)
226 const Qt::CaseSensitivity cs = Qt::CaseInsensitive;
227#else
228 const Qt::CaseSensitivity cs = Qt::CaseSensitive;
229#endif
230 if (!cleanUrl.compare(s: local, cs)) {
231 removeRow(arow: j);
232 if (j <= row)
233 row--;
234 break;
235 }
236 }
237 row = qMax(a: row, b: 0);
238 QModelIndex idx = fileSystemModel->index(path: cleanUrl);
239 if (!fileSystemModel->isDir(index: idx))
240 continue;
241 insertRows(row, count: 1);
242 setUrl(index: index(row, column: 0), url, dirIndex: idx);
243 watching.append(t: {.index: idx, .path: cleanUrl});
244 }
245}

247/*!
248 Return the complete list of urls in a QList.
249*/
250QList<QUrl> QUrlModel::urls() const
251{
252 QList<QUrl> list;
253 const int numRows = rowCount();
254 list.reserve(size: numRows);
255 for (int i = 0; i < numRows; ++i)
256 list.append(t: data(index: index(row: i, column: 0), role: UrlRole).toUrl());
257 return list;
258}

260/*!
261 QFileSystemModel to get index's from, clears existing rows
262*/
263void QUrlModel::setFileSystemModel(QFileSystemModel *model)
264{
265 if (model == fileSystemModel)
266 return;
267 if (fileSystemModel != nullptr) {
268 disconnect(sender: model, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
269 receiver: this, SLOT(dataChanged(QModelIndex,QModelIndex)));
270 disconnect(sender: model, SIGNAL(layoutChanged()),
271 receiver: this, SLOT(layoutChanged()));
272 disconnect(sender: model, SIGNAL(rowsRemoved(QModelIndex,int,int)),
273 receiver: this, SLOT(layoutChanged()));
274 }
275 fileSystemModel = model;
276 if (fileSystemModel != nullptr) {
277 connect(sender: model, SIGNAL(dataChanged(QModelIndex,QModelIndex)),
278 receiver: this, SLOT(dataChanged(QModelIndex,QModelIndex)));
279 connect(sender: model, SIGNAL(layoutChanged()),
280 receiver: this, SLOT(layoutChanged()));
281 connect(sender: model, SIGNAL(rowsRemoved(QModelIndex,int,int)),
282 receiver: this, SLOT(layoutChanged()));
283 }
284 clear();
285 insertColumns(column: 0, count: 1);
286}

288/*
289 If one of the index's we are watching has changed update our internal data
290*/
291void QUrlModel::dataChanged(const QModelIndex &topLeft, const QModelIndex &bottomRight)
292{
293 QModelIndex parent = topLeft.parent();
294 for (int i = 0; i < watching.size(); ++i) {
295 QModelIndex index = watching.at(i).index;
296 if (index.model() && topLeft.model()) {
297 Q_ASSERT(index.model() == topLeft.model());
298 }
299 if ( index.row() >= topLeft.row()
300 && index.row() <= bottomRight.row()
301 && index.column() >= topLeft.column()
302 && index.column() <= bottomRight.column()
303 && index.parent() == parent) {
304 changed(path: watching.at(i).path);
305 }
306 }
307}

309/*!
310 Re-get all of our data, anything could have changed!
311 */
312void QUrlModel::layoutChanged()
313{
314 QStringList paths;
315 const int numPaths = watching.size();
316 paths.reserve(size: numPaths);
317 for (int i = 0; i < numPaths; ++i)
318 paths.append(t: watching.at(i).path);
319 watching.clear();
320 for (int i = 0; i < numPaths; ++i) {
321 QString path = paths.at(i);
322 QModelIndex newIndex = fileSystemModel->index(path);
323 watching.append(t: {.index: newIndex, .path: path});
324 if (newIndex.isValid())
325 changed(path);
326 }
327}

329/*!
330 The following path changed data update our copy of that data
331
332 \sa layoutChanged(), dataChanged()
333*/
334void QUrlModel::changed(const QString &path)
335{
336 for (int i = 0; i < rowCount(); ++i) {
337 QModelIndex idx = index(row: i, column: 0);
338 if (idx.data(arole: UrlRole).toUrl().toLocalFile() == path) {
339 setData(index: idx, value: idx.data(arole: UrlRole).toUrl());
340 }
341 }

348void QSidebar::setModelAndUrls(QFileSystemModel *model, const QList<QUrl> &newUrls)
349{
350 setUniformItemSizes(true);
351 urlModel = new QUrlModel(this);
352 urlModel->setFileSystemModel(model);
353 setModel(urlModel);
354 setItemDelegate(new QSideBarDelegate(this));
355
356 connect(sender: selectionModel(), SIGNAL(currentChanged(QModelIndex,QModelIndex)),
357 receiver: this, SLOT(clicked(QModelIndex)));
358#if QT_CONFIG(draganddrop)
359 setDragDropMode(QAbstractItemView::DragDrop);
360#endif
361 setContextMenuPolicy(Qt::CustomContextMenu);
362 connect(sender: this, SIGNAL(customContextMenuRequested(QPoint)),
363 receiver: this, SLOT(showContextMenu(QPoint)));
364 urlModel->setUrls(newUrls);
365 setCurrentIndex(this->model()->index(row: 0,column: 0));
366}

387void QSidebar::selectUrl(const QUrl &url)
388{
389 disconnect(sender: selectionModel(), SIGNAL(currentChanged(QModelIndex,QModelIndex)),
390 receiver: this, SLOT(clicked(QModelIndex)));
391
392 selectionModel()->clear();
393 for (int i = 0; i < model()->rowCount(); ++i) {
394 if (model()->index(row: i, column: 0).data(arole: QUrlModel::UrlRole).toUrl() == url) {
395 selectionModel()->select(index: model()->index(row: i, column: 0), command: QItemSelectionModel::Select);
396 break;
397 }
398 }
399
400 connect(sender: selectionModel(), SIGNAL(currentChanged(QModelIndex,QModelIndex)),
401 receiver: this, SLOT(clicked(QModelIndex)));
402}

410void QSidebar::showContextMenu(const QPoint &position)
411{

425/*!
426 \internal
427
428 \sa showContextMenu()
429*/
430void QSidebar::removeEntry()
431{
432 QList<QModelIndex> idxs = selectionModel()->selectedIndexes();
433 QList<QPersistentModelIndex> indexes;
434 const int numIndexes = idxs.size();
435 indexes.reserve(size: numIndexes);
436 for (int i = 0; i < numIndexes; i++)
437 indexes.append(t: idxs.at(i));
438
439 for (int i = 0; i < numIndexes; ++i) {
440 if (!indexes.at(i).data(role: QUrlModel::UrlRole).toUrl().path().isEmpty())
441 model()->removeRow(arow: indexes.at(i).row());
442 }
443}
