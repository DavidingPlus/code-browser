77QPlatformDialogHelper *QDialogPrivate::platformHelper() const
78{
79 // Delayed creation of the platform, ensuring that
80 // that qobject_cast<> on the dialog works in the plugin.
81 if (!m_platformHelperCreated && canBeNativeDialog()) {
82 m_platformHelperCreated = true;
83 QDialogPrivate *ncThis = const_cast<QDialogPrivate *>(this);
84 QDialog *dialog = ncThis->q_func();
85 const int type = themeDialogType(dialog);
86 if (type >= 0) {
87 m_platformHelper = QGuiApplicationPrivate::platformTheme()
88 ->createPlatformDialogHelper(type: static_cast<QPlatformTheme::DialogType>(type));
89 if (m_platformHelper) {
90 QObject::connect(sender: m_platformHelper, SIGNAL(accept()), receiver: dialog, SLOT(accept()));
91 QObject::connect(sender: m_platformHelper, SIGNAL(reject()), receiver: dialog, SLOT(reject()));
92 ncThis->initHelper(m_platformHelper);
93 }
94 }
95 }
96 return m_platformHelper;
97}

99bool QDialogPrivate::canBeNativeDialog() const
100{
101 if (QCoreApplication::testAttribute(attribute: Qt::AA_DontUseNativeDialogs))
102 return false;
103
104 QDialogPrivate *ncThis = const_cast<QDialogPrivate *>(this);
105 QDialog *dialog = ncThis->q_func();
106 const int type = themeDialogType(dialog);
107 if (type >= 0)
108 return QGuiApplicationPrivate::platformTheme()
109 ->usePlatformNativeDialog(type: static_cast<QPlatformTheme::DialogType>(type));
110 return false;
111}

113/*!
114 \internal
115
116 Properly closes dialog and sets the \a resultCode.
117 */
118void QDialogPrivate::close(int resultCode)
119{
120 Q_Q(QDialog);
121
122 q->setResult(resultCode);
123
124 if (!data.is_closing) {
125 // Until Qt 6.3 we didn't close dialogs, so they didn't receive a QCloseEvent.
126 // It is likely that subclasses implement closeEvent and handle them as rejection
127 // (like QMessageBox and QProgressDialog do), so eat those events.
128 struct CloseEventEater : QObject
129 {
130 using QObject::QObject;
131 protected:
132 bool eventFilter(QObject *o, QEvent *e) override
133 {
134 if (e->type() == QEvent::Close)
135 return true;
136 return QObject::eventFilter(watched: o, event: e);
137 }
138 } closeEventEater;
139 q->installEventFilter(filterObj: &closeEventEater);
140 QWidgetPrivate::close();
141 } else {
142 // If the close was initiated outside of QDialog we will end up
143 // here via QDialog::closeEvent calling reject(), in which case
144 // we need to hide the dialog to ensure QDialog::closeEvent does
145 // not ignore the close event. FIXME: Why is QDialog doing this?
146 q->hide();
147 }
148
149 resetModalitySetByOpen();
150}

156 is equal to 0 emits rejected(), if the \a dialogCode is equal to

166 emit q->rejected();
167
168 emit q->finished(result: resultCode);

202/*!
203 \class QDialog
204 \brief The QDialog class is the base class of dialog windows.
205
206 \ingroup dialog-classes
207 \ingroup abstractwidgets
208 \inmodule QtWidgets
209
210 A dialog window is a top-level window mostly used for short-term
211 tasks and brief communications with the user. QDialogs may be
212 modal or modeless. QDialogs can
213 provide a \l{#return}{return value}, and they can have \l{#default}{default buttons}. QDialogs can also have a QSizeGrip in their
214 lower-right corner, using setSizeGripEnabled().
215
216 Note that QDialog (and any other widget that has type \c Qt::Dialog) uses
217 the parent widget slightly differently from other classes in Qt. A dialog is
218 always a top-level widget, but if it has a parent, its default location is
219 centered on top of the parent's top-level widget (if it is not top-level
220 itself). It will also share the parent's taskbar entry.
221
222 Use the overload of the QWidget::setParent() function to change
223 the ownership of a QDialog widget. This function allows you to
224 explicitly set the window flags of the reparented widget; using
225 the overloaded function will clear the window flags specifying the
226 window-system properties for the widget (in particular it will
227 reset the Qt::Dialog flag).
228
229 \note The parent relationship of the dialog does \e{not} imply
230 that the dialog will always be stacked on top of the parent
231 window. To ensure that the dialog is always on top, make the
232 dialog modal. This also applies for child windows of the dialog
233 itself. To ensure that child windows of the dialog stay on top
234 of the dialog, make the child windows modal as well.
235
236 \section1 Modal Dialogs
237
238 A \b{modal} dialog is a dialog that blocks input to other
239 visible windows in the same application. Dialogs that are used to
240 request a file name from the user or that are used to set
241 application preferences are usually modal. Dialogs can be
242 \l{Qt::ApplicationModal}{application modal} (the default) or
243 \l{Qt::WindowModal}{window modal}.
244
245 When an application modal dialog is opened, the user must finish
246 interacting with the dialog and close it before they can access
247 any other window in the application. Window modal dialogs only
248 block access to the window associated with the dialog, allowing
249 the user to continue to use other windows in an application.
250
251 The most common way to display a modal dialog is to call its
252 exec() function. When the user closes the dialog, exec() will
253 provide a useful \l{#return}{return value}. To close the dialog
254 and return the appropriate value, you must connect a default button,
255 e.g. an \uicontrol OK button to the accept() slot and a
256 \uicontrol Cancel button to the reject() slot. Alternatively, you
257 can call the done() slot with \c Accepted or \c Rejected.
258
259 An alternative is to call setModal(true) or setWindowModality(),
260 then show(). Unlike exec(), show() returns control to the caller
261 immediately. Calling setModal(true) is especially useful for
262 progress dialogs, where the user must have the ability to interact
263 with the dialog, e.g. to cancel a long running operation. If you
264 use show() and setModal(true) together to perform a long operation,
265 you must call QCoreApplication::processEvents() periodically during
266 processing to enable the user to interact with the dialog. (See
267 QProgressDialog.)
268
269 \section1 Modeless Dialogs
270
271 A \b{modeless} dialog is a dialog that operates
272 independently of other windows in the same application. Find and
273 replace dialogs in word-processors are often modeless to allow the
274 user to interact with both the application's main window and with
275 the dialog.
276
277 Modeless dialogs are displayed using show(), which returns control
278 to the caller immediately.
279
280 If you invoke the \l{QWidget::show()}{show()} function after hiding
281 a dialog, the dialog will be displayed in its original position. This is
282 because the window manager decides the position for windows that
283 have not been explicitly placed by the programmer. To preserve the
284 position of a dialog that has been moved by the user, save its position
285 in your \l{QWidget::closeEvent()}{closeEvent()} handler and then
286 move the dialog to that position, before showing it again.
287
288 \target default
289 \section1 Default Button
290
291 A dialog's \e default button is the button that's pressed when the
292 user presses Enter (Return). This button is used to signify that
293 the user accepts the dialog's settings and wants to close the
294 dialog. Use QPushButton::setDefault(), QPushButton::isDefault()
295 and QPushButton::autoDefault() to set and control the dialog's
296 default button.
297
298 \target escapekey
299 \section1 Escape Key
300
301 If the user presses the Esc key in a dialog, QDialog::reject()
302 will be called. This will cause the window to close: The \l{QCloseEvent}{close event} cannot be \l{QEvent::ignore()}{ignored}.
303
304 \section1 Extensibility
305
306 Extensibility is the ability to show the dialog in two ways: a
307 partial dialog that shows the most commonly used options, and a
308 full dialog that shows all the options. Typically an extensible
309 dialog will initially appear as a partial dialog, but with a
310 \uicontrol More toggle button. If the user presses the \uicontrol More button down,
311 the dialog is expanded. The \l{Extension Example} shows how to achieve
312 extensible dialogs using Qt.
313
314 \target return
315 \section1 Return Value (Modal Dialogs)
316
317 Modal dialogs are often used in situations where a return value is
318 required, e.g. to indicate whether the user pressed \uicontrol OK or
319 \uicontrol Cancel. A dialog can be closed by calling the accept() or the
320 reject() slots, and exec() will return \c Accepted or \c Rejected
321 as appropriate. The exec() call returns the result of the dialog.
322 The result is also available from result() if the dialog has not
323 been destroyed.
324
325 In order to modify your dialog's close behavior, you can reimplement
326 the functions accept(), reject() or done(). The
327 \l{QWidget::closeEvent()}{closeEvent()} function should only be
328 reimplemented to preserve the dialog's position or to override the
329 standard close or reject behavior.
330
331 \target examples
332 \section1 Code Examples
333
334 A modal dialog:
335
336 \snippet dialogs/dialogs.cpp 1
337
338 A modeless dialog:
339
340 \snippet dialogs/dialogs.cpp 0
341
342 \sa QDialogButtonBox, QTabWidget, QWidget, QProgressDialog, {Extension Example},
343 {Standard Dialogs Example}
344*/

533/*!
534 Shows the dialog as a \l{QDialog#Modal Dialogs}{modal dialog},
535 blocking until the user closes it. The function returns a \l
536 DialogCode result.
537
538 If the dialog is \l{Qt::ApplicationModal}{application modal}, users cannot
539 interact with any other window in the same application until they close
540 the dialog. If the dialog is \l{Qt::ApplicationModal}{window modal}, only
541 interaction with the parent window is blocked while the dialog is open.
542 By default, the dialog is application modal.
543
544 \note Avoid using this function; instead, use \c{open()}. Unlike exec(),
545 open() is asynchronous, and does not spin an additional event loop. This
546 prevents a series of dangerous bugs from happening (e.g. deleting the
547 dialog's parent while the dialog is open via exec()). When using open() you
548 can connect to the finished() signal of QDialog to be notified when the
549 dialog is closed.
550
551 \sa open(), show(), result(), setWindowModality()
552*/
553
554int QDialog::exec()
555{
556 Q_D(QDialog);
557
558 if (Q_UNLIKELY(d->eventLoop)) {
559 qWarning(msg: "QDialog::exec: Recursive call detected");
560 return -1;
561 }
562
563 bool deleteOnClose = testAttribute(attribute: Qt::WA_DeleteOnClose);
564 setAttribute(Qt::WA_DeleteOnClose, on: false);
565
566 d->resetModalitySetByOpen();
567
568 bool wasShowModal = testAttribute(attribute: Qt::WA_ShowModal);
569 setAttribute(Qt::WA_ShowModal, on: true);
570 setResult(0);
571
572 show();
573
574 QPointer<QDialog> guard = this;
575 if (d->nativeDialogInUse) {
576 d->platformHelper()->exec();
577 } else {
578 QEventLoop eventLoop;
579 d->eventLoop = &eventLoop;
580 (void) eventLoop.exec(flags: QEventLoop::DialogExec);
581 }
582 if (guard.isNull())
583 return QDialog::Rejected;
584 d->eventLoop = nullptr;
585
586 setAttribute(Qt::WA_ShowModal, on: wasShowModal);
587
588 int res = result();
589 if (d->nativeDialogInUse)
590 d->helperDone(static_cast<QDialog::DialogCode>(res), d->platformHelper());
591 if (deleteOnClose)
592 delete this;
593 return res;
594}

681/*! \reimp */
682void QDialog::keyPressEvent(QKeyEvent *e)
683{
684#ifndef QT_NO_SHORTCUT
685 // Calls reject() if Escape is pressed. Simulates a button
686 // click for the default button if Enter is pressed. Move focus
687 // for the arrow keys. Ignore the rest.
688 if (e->matches(key: QKeySequence::Cancel)) {
689 reject();
690 } else
691#endif
692 if (!e->modifiers() || (e->modifiers() & Qt::KeypadModifier && e->key() == Qt::Key_Enter)) {
693 switch (e->key()) {
694#if QT_CONFIG(pushbutton)
695 case Qt::Key_Enter:
696 case Qt::Key_Return: {
697 QList<QPushButton*> list = findChildren<QPushButton*>();
698 for (int i=0; i<list.size(); ++i) {
699 QPushButton *pb = list.at(i);
700 if (pb->isDefault() && pb->isVisible()) {
701 if (pb->isEnabled())
702 pb->click();
703 return;
704 }
705 }
706 }
707 break;
708#endif
709 default:
710 e->ignore();
711 return;
712 }
713 } else {
714 e->ignore();
715 }
716}

718/*! \reimp */
719void QDialog::closeEvent(QCloseEvent *e)
720{
721#if QT_CONFIG(whatsthis)
722 if (isModal() && QWhatsThis::inWhatsThisMode())
723 QWhatsThis::leaveWhatsThisMode();
724#endif
725 if (isVisible()) {
726 QPointer<QObject> that = this;
727 reject();
728 if (that && isVisible())
729 e->ignore();
730 } else {
731 e->accept();
732 }

747
748void QDialogPrivate::setVisible(bool visible)
749{
750 Q_Q(QDialog);
751 if (!q->testAttribute(attribute: Qt::WA_DontShowOnScreen) && canBeNativeDialog() && setNativeDialogVisible(visible))
752 return;
753
754 // We should not block windows by the invisible modal dialog
755 // if a platform-specific dialog is implemented as an in-process
756 // Qt window, because in this case it will also be blocked.
757 const bool dontBlockWindows = q->testAttribute(attribute: Qt::WA_DontShowOnScreen)
758 && styleHint(hint: QPlatformDialogHelper::DialogIsQtWindow).toBool();
759 Qt::WindowModality oldModality;
760 bool wasModalitySet;
761
762 if (dontBlockWindows) {
763 oldModality = q->windowModality();
764 wasModalitySet = q->testAttribute(attribute: Qt::WA_SetWindowModality);
765 q->setWindowModality(Qt::NonModal);
766 }
767
768 if (visible) {
769 if (q->testAttribute(attribute: Qt::WA_WState_ExplicitShowHide) && !q->testAttribute(attribute: Qt::WA_WState_Hidden))
770 return;
771
772 q->QWidget::setVisible(visible);
773
774 // Window activation might be prevented. We can't test isActiveWindow here,
775 // as the window will be activated asynchronously by the window manager.
776 if (!q->testAttribute(attribute: Qt::WA_ShowWithoutActivating)) {
777 QWidget *fw = q->window()->focusWidget();
778 if (!fw)
779 fw = q;
780
781 /*
782 The following block is to handle a special case, and does not
783 really follow proper logic in concern of autoDefault and TAB
784 order. However, it's here to ease usage for the users. If a
785 dialog has a default QPushButton, and first widget in the TAB
786 order also is a QPushButton, then we give focus to the main
787 default QPushButton. This simplifies code for the developers,
788 and actually catches most cases... If not, then they simply
789 have to use [widget*]->setFocus() themselves...
790 */
791#if QT_CONFIG(pushbutton)
792 if (mainDef && fw->focusPolicy() == Qt::NoFocus) {
793 QWidget *first = fw;
794 while ((first = first->nextInFocusChain()) != fw && first->focusPolicy() == Qt::NoFocus)
795 ;
796 if (first != mainDef && qobject_cast<QPushButton*>(object: first))
797 mainDef->setFocus();
798 }
799 if (!mainDef && q->isWindow()) {
800 QWidget *w = fw;
801 while ((w = w->nextInFocusChain()) != fw) {
802 QPushButton *pb = qobject_cast<QPushButton *>(object: w);
803 if (pb && pb->autoDefault() && pb->focusPolicy() != Qt::NoFocus) {
804 pb->setDefault(true);
805 break;
806 }
807 }
808 }
809#endif
810 if (fw && !fw->hasFocus()) {
811 QFocusEvent e(QEvent::FocusIn, Qt::TabFocusReason);
812 QCoreApplication::sendEvent(receiver: fw, event: &e);
813 }
814 }
815
816#if QT_CONFIG(accessibility)
817 QAccessibleEvent event(q, QAccessible::DialogStart);
818 QAccessible::updateAccessibility(event: &event);
819#endif
820
821 } else {
822 if (q->testAttribute(attribute: Qt::WA_WState_ExplicitShowHide) && q->testAttribute(attribute: Qt::WA_WState_Hidden))
823 return;
824
825#if QT_CONFIG(accessibility)
826 if (q->isVisible()) {
827 QAccessibleEvent event(q, QAccessible::DialogEnd);
828 QAccessible::updateAccessibility(event: &event);
829 }
830#endif
831
832 // Reimplemented to exit a modal event loop when the dialog is hidden.
833 q->QWidget::setVisible(visible);
834 if (eventLoop)
835 eventLoop->exit();
836 }
837
838 if (dontBlockWindows) {
839 q->setWindowModality(oldModality);
840 q->setAttribute(Qt::WA_SetWindowModality, on: wasModalitySet);
841 }
842
843#if QT_CONFIG(pushbutton)
844 const QPlatformTheme *theme = QGuiApplicationPrivate::platformTheme();
845 if (mainDef && q->isActiveWindow()
846 && theme->themeHint(hint: QPlatformTheme::DialogSnapToDefaultButton).toBool())
847 QCursor::setPos(mainDef->mapToGlobal(mainDef->rect().center()));
848#endif
849}

863/*! \internal */
864void QDialog::adjustPosition(QWidget* w)
865{
866 Q_D(QDialog);
867
868 if (const QPlatformTheme *theme = QGuiApplicationPrivate::platformTheme())
869 if (theme->themeHint(hint: QPlatformTheme::WindowAutoPlacement).toBool())
870 return;
871 QPoint p(0, 0);
872 int extraw = 0, extrah = 0;
873 const QWindow *parentWindow = nullptr;
874 if (w) {
875 w = w->window();
876 } else {
877 parentWindow = d->transientParentWindow();
878 }
879 QRect desk;
880 QScreen *scrn = nullptr;
881 if (w)
882 scrn = w->screen();
883 else if (parentWindow)
884 scrn = parentWindow->screen();
885 else if (QGuiApplication::primaryScreen()->virtualSiblings().size() > 1)
886 scrn = QGuiApplication::screenAt(point: QCursor::pos());
887 else
888 scrn = screen();
889 if (scrn)
890 desk = scrn->availableGeometry();
891
892 QWidgetList list = QApplication::topLevelWidgets();
893 for (int i = 0; (extraw == 0 || extrah == 0) && i < list.size(); ++i) {
894 QWidget * current = list.at(i);
895 if (current->isVisible()) {
896 int framew = current->geometry().x() - current->x();
897 int frameh = current->geometry().y() - current->y();
898
899 extraw = qMax(a: extraw, b: framew);
900 extrah = qMax(a: extrah, b: frameh);
901 }
902 }
903
904 // sanity check for decoration frames. With embedding, we
905 // might get extraordinary values
906 if (extraw == 0 || extrah == 0 || extraw >= 10 || extrah >= 40) {
907 extrah = 40;
908 extraw = 10;
909 }
910
911
912 if (w) {
913 // Use pos() if the widget is embedded into a native window
914 QPoint pp;
915 if (w->windowHandle() && qvariant_cast<WId>(v: w->windowHandle()->property(name: "_q_embedded_native_parent_handle")))
916 pp = w->pos();
917 else
918 pp = w->mapToGlobal(QPoint(0,0));
919 p = QPoint(pp.x() + w->width()/2,
920 pp.y() + w->height()/ 2);
921 } else if (parentWindow) {
922 // QTBUG-63406: Widget-based dialog in QML, which has no Widget parent
923 // but a transient parent window.
924 QPoint pp = parentWindow->mapToGlobal(pos: QPoint(0, 0));
925 p = QPoint(pp.x() + parentWindow->width() / 2, pp.y() + parentWindow->height() / 2);
926 } else {
927 // p = middle of the desktop
928 p = QPoint(desk.x() + desk.width()/2, desk.y() + desk.height()/2);
929 }
930
931 // p = origin of this
932 p = QPoint(p.x()-width()/2 - extraw,
933 p.y()-height()/2 - extrah);
934
935
936 if (p.x() + extraw + width() > desk.x() + desk.width())
937 p.setX(desk.x() + desk.width() - width() - extraw);
938 if (p.x() < desk.x())
939 p.setX(desk.x());
940
941 if (p.y() + extrah + height() > desk.y() + desk.height())
942 p.setY(desk.y() + desk.height() - height() - extrah);
943 if (p.y() < desk.y())
944 p.setY(desk.y());
945
946 // QTBUG-52735: Manually set the correct target screen since scaling in a
947 // subsequent call to QWindow::resize() may otherwise use the wrong factor
948 // if the screen changed notification is still in an event queue.
949 if (scrn) {
950 if (QWindow *window = windowHandle())
951 window->setScreen(scrn);
952 }
953
954 move(p);
955}
