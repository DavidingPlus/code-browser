251void QWellArray::paintEvent(QPaintEvent *e)
252{
253 QRect r = e->rect();
254 int cx = r.x();
255 int cy = r.y();
256 int ch = r.height();
257 int cw = r.width();
258 int colfirst = columnAt(x: cx);
259 int collast = columnAt(x: cx + cw);
260 int rowfirst = rowAt(y: cy);
261 int rowlast = rowAt(y: cy + ch);
262
263 if (isRightToLeft()) {
264 int t = colfirst;
265 colfirst = collast;
266 collast = t;
267 }
268
269 QPainter painter(this);
270 QPainter *p = &painter;
271 QRect rect(0, 0, cellWidth(), cellHeight());
272
273
274 if (collast < 0 || collast >= ncols)
275 collast = ncols-1;
276 if (rowlast < 0 || rowlast >= nrows)
277 rowlast = nrows-1;
278
279 // Go through the rows
280 for (int r = rowfirst; r <= rowlast; ++r) {
281 // get row position and height
282 int rowp = rowY(row: r);
283
284 // Go through the columns in the row r
285 // if we know from where to where, go through [colfirst, collast],
286 // else go through all of them
287 for (int c = colfirst; c <= collast; ++c) {
288 // get position and width of column c
289 int colp = columnX(column: c);
290 // Translate painter and draw the cell
291 rect.translate(dx: colp, dy: rowp);
292 paintCell(p, row: r, col: c, rect);
293 rect.translate(dx: -colp, dy: -rowp);
294 }
295 }
296}

318void QWellArray::paintCell(QPainter* p, int row, int col, const QRect &rect)
319{
320 int b = 3; //margin
321
322 const QPalette & g = palette();
323 QStyleOptionFrame opt;
324 opt.initFrom(w: this);
325 int dfw = style()->pixelMetric(metric: QStyle::PM_DefaultFrameWidth, option: &opt);
326 opt.lineWidth = dfw;
327 opt.midLineWidth = 1;
328 opt.rect = rect.adjusted(xp1: b, yp1: b, xp2: -b, yp2: -b);
329 opt.palette = g;
330 opt.state = QStyle::State_Enabled | QStyle::State_Sunken;
331 style()->drawPrimitive(pe: QStyle::PE_Frame, opt: &opt, p, w: this);
332 b += dfw;
333
334 if ((row == curRow) && (col == curCol)) {
335 if (hasFocus()) {
336 QStyleOptionFocusRect opt;
337 opt.palette = g;
338 opt.rect = rect;
339 opt.state = QStyle::State_None | QStyle::State_KeyboardFocusChange;
340 style()->drawPrimitive(pe: QStyle::PE_FrameFocusRect, opt: &opt, p, w: this);
341 }
342 }
343 paintCellContents(p, row, col, opt.rect.adjusted(xp1: dfw, yp1: dfw, xp2: -dfw, yp2: -dfw));
344}

378void QWellArray::setCurrent(int row, int col)
379{
380 if ((curRow == row) && (curCol == col))
381 return;
382
383 if (row < 0 || col < 0)
384 row = col = -1;
385
386 int oldRow = curRow;
387 int oldCol = curCol;
388
389 curRow = row;
390 curCol = col;
391
392 updateCell(row: oldRow, column: oldCol);
393 updateCell(row: curRow, column: curCol);
394
395 emit currentChanged(row: curRow, col: curCol);
396}

399 Sets the currently selected cell to \a row, \a column. If \a row or
400 \a column are less than zero, the current cell is unselected.
401
402 Does not set the position of the focus indicator.

413 selRow = row;

438void QWellArray::keyPressEvent(QKeyEvent* e)
439{
440 switch(e->key()) { // Look at the key code
441 case Qt::Key_Left: // If 'left arrow'-key,
442 if (curCol > 0) // and cr't not in leftmost col
443 setCurrent(row: curRow, col: curCol - 1); // set cr't to next left column
444 break;
445 case Qt::Key_Right: // Correspondingly...
446 if (curCol < numCols()-1)
447 setCurrent(row: curRow, col: curCol + 1);
448 break;
449 case Qt::Key_Up:
450 if (curRow > 0)
451 setCurrent(row: curRow - 1, col: curCol);
452 break;
453 case Qt::Key_Down:
454 if (curRow < numRows()-1)
455 setCurrent(row: curRow + 1, col: curCol);
456 break;
457#if 0
458 // bad idea that shouldn't have been implemented; very counterintuitive
459 case Qt::Key_Return:
460 case Qt::Key_Enter:
461 /*
462 ignore the key, so that the dialog get it, but still select
463 the current row/col
464 */
465 e->ignore();
466 // fallthrough intended
467#endif
468 case Qt::Key_Space:
469 setSelected(row: curRow, col: curCol);
470 break;
471 default: // If not an interesting key,
472 e->ignore(); // we don't accept the event
473 return;
474 }
