102template <typename T>
103class QAutoPointer {
104 QPointer<T> o;
105public:
106 explicit QAutoPointer(T *t) noexcept : o(t) {}
107 ~QAutoPointer() { delete o; }
108
109 T *operator->() const noexcept { return get(); }
110 T *get() const noexcept { return o; }
111 T &operator*() const { return *get(); }
112 explicit operator bool() const noexcept { return !o.isNull(); }
113 bool operator!() const noexcept { return !o; }
114private:
115 Q_DISABLE_COPY(QAutoPointer);
116};
