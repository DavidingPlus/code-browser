303#define QT_REQUIRE_VERSION(argc, argv, str) { QString s = QString::fromLatin1(str);\
304QString sq = QString::fromLatin1(qVersion()); \
305if ((sq.section(QChar::fromLatin1('.'),0,0).toInt()<<16)+\
306(sq.section(QChar::fromLatin1('.'),1,1).toInt()<<8)+\
307sq.section(QChar::fromLatin1('.'),2,2).toInt()<(s.section(QChar::fromLatin1('.'),0,0).toInt()<<16)+\
308(s.section(QChar::fromLatin1('.'),1,1).toInt()<<8)+\
309s.section(QChar::fromLatin1('.'),2,2).toInt()) { \
310if (!qApp){ \
311 new QApplication(argc,argv); \
312} \
313QString s = QApplication::tr("Executable '%1' requires Qt "\
314 "%2, found Qt %3.").arg(qAppName()).arg(QString::fromLatin1(\
315str)).arg(QString::fromLatin1(qVersion())); QMessageBox::critical(0, QApplication::tr(\
316"Incompatible Qt Library Error"), s, QMessageBox::Abort, 0); qFatal("%s", s.toLatin1().data()); }}
