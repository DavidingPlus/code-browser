51QString QDomBuilder::dtdInternalSubset(const QString &dtd)
52{
53 // https://www.w3.org/TR/xml/#NT-intSubset
54 // doctypedecl: '<!DOCTYPE' S Name (S ExternalID)? S? ('[' intSubset ']' S?)? '>'
55 const QString &name = doc->doctype()->name;
56 QStringView tmp = QStringView(dtd).sliced(pos: dtd.indexOf(s: name) + name.size());
57
58 const QString &publicId = doc->doctype()->publicId;
59 if (!publicId.isEmpty())
60 tmp = tmp.sliced(pos: tmp.indexOf(s: publicId) + publicId.size());
61
62 const QString &systemId = doc->doctype()->systemId;
63 if (!systemId.isEmpty())
64 tmp = tmp.sliced(pos: tmp.indexOf(s: systemId) + systemId.size());
65
66 const qsizetype obra = tmp.indexOf(c: u'[');
67 const qsizetype cbra = tmp.lastIndexOf(c: u']');
68 if (obra >= 0 && cbra >= 0)
69 return tmp.left(n: cbra).sliced(pos: obra + 1).toString();
70
71 return QString();
72}

80bool QDomBuilder::startElement(const QString &nsURI, const QString &qName,
81 const QXmlStreamAttributes &atts)
82{
83 const bool nsProcessing =
84 parseOptions.testFlag(flag: QDomDocument::ParseOption::UseNamespaceProcessing);
85 QDomNodePrivate *n =
86 nsProcessing ? doc->createElementNS(nsURI, qName) : doc->createElement(tagName: qName);
87 if (!n)
88 return false;
89
90 n->setLocation(lineNumber: int(reader->lineNumber()), columnNumber: int(reader->columnNumber()));
91
92 node->appendChild(newChild: n);
93 node = n;
94
95 // attributes
96 for (const auto &attr : atts) {
97 auto domElement = static_cast<QDomElementPrivate *>(node);
98 if (nsProcessing) {
99 domElement->setAttributeNS(nsURI: attr.namespaceUri().toString(),
100 qName: attr.qualifiedName().toString(),
101 newValue: attr.value().toString());
102 } else {
103 domElement->setAttribute(name: attr.qualifiedName().toString(),
104 value: attr.value().toString());
105 }
106 }
107
108 return true;
109}

111bool QDomBuilder::endElement()
112{
113 if (!node || node == doc)
114 return false;
115 node = node->parent();
116
117 return true;

120bool QDomBuilder::characters(const QString &characters, bool cdata)
121{
122 // No text as child of some document
123 if (node == doc)
124 return false;
125
126 std::unique_ptr<QDomNodePrivate> n;
127 if (cdata) {
128 n.reset(p: doc->createCDATASection(data: characters));
129 } else if (!entityName.isEmpty()) {
130 auto e = std::make_unique<QDomEntityPrivate>(
131 args&: doc, args: nullptr, args&: entityName, args: QString(), args: QString(), args: QString());
132 e->value = characters;
133 e->ref.deref();
134 doc->doctype()->appendChild(newChild: e.get());
135 Q_UNUSED(e.release());
136 n.reset(p: doc->createEntityReference(name: entityName));
137 } else {
138 n.reset(p: doc->createTextNode(data: characters));
139 }
140 n->setLocation(lineNumber: int(reader->lineNumber()), columnNumber: int(reader->columnNumber()));
141 node->appendChild(newChild: n.get());
142 Q_UNUSED(n.release());
143
144 return true;
145}

234bool QDomParser::parse()
235{
236 return parseProlog() && parseBody();
237}
238
239bool QDomParser::parseProlog()
240{
241 Q_ASSERT(reader);
242
243 bool foundDtd = false;
244
245 while (!reader->atEnd()) {
246 reader->readNext();
247
248 if (reader->hasError()) {
249 domBuilder.fatalError(message: reader->errorString());
250 return false;
251 }
252
253 switch (reader->tokenType()) {
254 case QXmlStreamReader::StartDocument:
255 if (!reader->documentVersion().isEmpty()) {
256 QString value(u"version='"_s);
257 value += reader->documentVersion();
258 value += u'\'';
259 if (!reader->documentEncoding().isEmpty()) {
260 value += u" encoding='"_s;
261 value += reader->documentEncoding();
262 value += u'\'';
263 }
264 if (reader->isStandaloneDocument()) {
265 value += u" standalone='yes'"_s;
266 } else {
267 // TODO: Add standalone='no', if 'standalone' is specified. With the current
268 // QXmlStreamReader there is no way to figure out if it was specified or not.
269 // QXmlStreamReader needs to be modified for handling that case correctly.
270 }
271
272 if (!domBuilder.processingInstruction(target: u"xml"_s, data: value)) {
273 domBuilder.fatalError(
274 message: QDomParser::tr(sourceText: "Error occurred while processing XML declaration"));
275 return false;
276 }
277 }
278 break;
279 case QXmlStreamReader::DTD:
280 if (foundDtd) {
281 domBuilder.fatalError(message: QDomParser::tr(sourceText: "Multiple DTD sections are not allowed"));
282 return false;
283 }
284 foundDtd = true;
285
286 if (!domBuilder.startDTD(name: reader->dtdName().toString(),
287 publicId: reader->dtdPublicId().toString(),
288 systemId: reader->dtdSystemId().toString())) {
289 domBuilder.fatalError(
290 message: QDomParser::tr(sourceText: "Error occurred while processing document type declaration"));
291 return false;
292 }
293 if (!domBuilder.parseDTD(dtd: reader->text().toString()))
294 return false;
295 if (!parseMarkupDecl())
296 return false;
297 break;
298 case QXmlStreamReader::Comment:
299 if (!domBuilder.comment(characters: reader->text().toString())) {
300 domBuilder.fatalError(message: QDomParser::tr(sourceText: "Error occurred while processing comment"));
301 return false;
302 }
303 break;
304 case QXmlStreamReader::ProcessingInstruction:
305 if (!domBuilder.processingInstruction(target: reader->processingInstructionTarget().toString(),
306 data: reader->processingInstructionData().toString())) {
307 domBuilder.fatalError(
308 message: QDomParser::tr(sourceText: "Error occurred while processing a processing instruction"));
309 return false;
310 }
311 break;
312 default:
313 // If the token is none of the above, prolog processing is done.
314 return true;
315 }
316 }
317
318 return true;
319}

321bool QDomParser::parseBody()
322{
323 Q_ASSERT(reader);
324
325 std::stack<QString> tagStack;
326 while (!reader->atEnd() && !reader->hasError()) {
327 switch (reader->tokenType()) {
328 case QXmlStreamReader::StartElement:
329 tagStack.push(x: reader->qualifiedName().toString());
330 if (!domBuilder.startElement(nsURI: reader->namespaceUri().toString(),
331 qName: reader->qualifiedName().toString(),
332 atts: reader->attributes())) {
333 domBuilder.fatalError(
334 message: QDomParser::tr(sourceText: "Error occurred while processing a start element"));
335 return false;
336 }
337 break;
338 case QXmlStreamReader::EndElement:
339 if (tagStack.empty() || reader->qualifiedName() != tagStack.top()) {
340 domBuilder.fatalError(
341 message: QDomParser::tr(sourceText: "Unexpected end element '%1'").arg(a: reader->name()));
342 return false;
343 }
344 tagStack.pop();
345 if (!domBuilder.endElement()) {
346 domBuilder.fatalError(
347 message: QDomParser::tr(sourceText: "Error occurred while processing an end element"));
348 return false;
349 }
350 break;
351 case QXmlStreamReader::Characters:
352 // Skip the content if it contains only spacing characters,
353 // unless it's CDATA or PreserveSpacingOnlyNodes was specified.
354 if (reader->isCDATA() || domBuilder.preserveSpacingOnlyNodes()
355 || !(reader->isWhitespace() || reader->text().trimmed().isEmpty())) {
356 if (!domBuilder.characters(characters: reader->text().toString(), cdata: reader->isCDATA())) {
357 domBuilder.fatalError(
358 message: QDomParser::tr(sourceText: "Error occurred while processing the element content"));
359 return false;
360 }
361 }
362 break;
363 case QXmlStreamReader::Comment:
364 if (!domBuilder.comment(characters: reader->text().toString())) {
365 domBuilder.fatalError(message: QDomParser::tr(sourceText: "Error occurred while processing comments"));
366 return false;
367 }
368 break;
369 case QXmlStreamReader::ProcessingInstruction:
370 if (!domBuilder.processingInstruction(target: reader->processingInstructionTarget().toString(),
371 data: reader->processingInstructionData().toString())) {
372 domBuilder.fatalError(
373 message: QDomParser::tr(sourceText: "Error occurred while processing a processing instruction"));
374 return false;
375 }
376 break;
377 case QXmlStreamReader::EntityReference:
378 if (!domBuilder.skippedEntity(name: reader->name().toString())) {
379 domBuilder.fatalError(
380 message: QDomParser::tr(sourceText: "Error occurred while processing an entity reference"));
381 return false;
382 }
383 break;
384 default:
385 domBuilder.fatalError(message: QDomParser::tr(sourceText: "Unexpected token"));
386 return false;
387 }
388
389 reader->readNext();
390 }
391
392 if (reader->hasError()) {
393 domBuilder.fatalError(message: reader->errorString());
394 reader->readNext();
395 return false;
396 }
397
398 if (!tagStack.empty()) {
399 domBuilder.fatalError(message: QDomParser::tr(sourceText: "Tag mismatch"));
400 return false;
401 }
402
403 return true;
404}

406bool QDomParser::parseMarkupDecl()
407{
408 Q_ASSERT(reader);
409
410 const auto entities = reader->entityDeclarations();
411 for (const auto &entityDecl : entities) {
412 // Entity declarations are created only for External Entities. Internal Entities
413 // are parsed, and QXmlStreamReader handles the parsing itself and returns the
414 // parsed result. So we don't need to do anything for the Internal Entities.
415 if (!entityDecl.publicId().isEmpty() || !entityDecl.systemId().isEmpty()) {
416 // External Entity
417 if (!domBuilder.unparsedEntityDecl(name: entityDecl.name().toString(),
418 publicId: entityDecl.publicId().toString(),
419 systemId: entityDecl.systemId().toString(),
420 notationName: entityDecl.notationName().toString())) {
421 domBuilder.fatalError(
422 message: QDomParser::tr(sourceText: "Error occurred while processing entity declaration"));
423 return false;
424 }
425 }
426 }
427
428 const auto notations = reader->notationDeclarations();
429 for (const auto &notationDecl : notations) {
430 if (!domBuilder.notationDecl(name: notationDecl.name().toString(),
431 publicId: notationDecl.publicId().toString(),
432 systemId: notationDecl.systemId().toString())) {
433 domBuilder.fatalError(
434 message: QDomParser::tr(sourceText: "Error occurred while processing notation declaration"));
435 return false;
436 }
437 }
438
439 return true;
440}
