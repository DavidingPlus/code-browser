19class Q_TESTLIB_EXPORT QTestEventLoop : public QObject

23public:
24 QTestEventLoop(QObject *parent = nullptr)
25 : QObject(parent), _timeout(false)

28 inline void enterLoopMSecs(int ms);
29 inline void enterLoop(int secs) { enterLoopMSecs(ms: secs * 1000); }
30
31 inline void changeInterval(int secs)
32 { killTimer(id: timerId); timerId = startTimer(interval: secs * 1000); }
33
34 inline bool timeout() const
35 { return _timeout; }

37 inline static QTestEventLoop &instance()
38 {
39 Q_CONSTINIT static QPointer<QTestEventLoop> testLoop;
40 if (testLoop.isNull())

42 return *static_cast<QTestEventLoop *>(testLoop);
43 }

45public Q_SLOTS:

58inline void QTestEventLoop::enterLoopMSecs(int ms)
59{
60 Q_ASSERT(!loop);
61 _timeout = false;
62
63 if (QTest::runningTest() && QTest::currentTestResolved())
64 return;
65
66 QEventLoop l;
67 // if tests want to measure sub-second precision, use a precise timer
68 timerId = startTimer(interval: ms, timerType: ms < 1000 ? Qt::PreciseTimer : Qt::CoarseTimer);
69
70 loop = &l;
71 l.exec();
72 loop = nullptr;
73}
74
75inline void QTestEventLoop::exitLoop()
76{
77 if (thread() != QThread::currentThread())
78 {
79 QMetaObject::invokeMethod(obj: this, member: "exitLoop", c: Qt::QueuedConnection);
80 return;
81 }
82
83 if (timerId != -1)
84 killTimer(id: timerId);
85 timerId = -1;
86
87 if (loop)
88 loop->exit();
89}
90
91inline void QTestEventLoop::timerEvent(QTimerEvent *e)
92{
93 if (e->timerId() != timerId)
94 return;
95 _timeout = true;
96 exitLoop();
97}
