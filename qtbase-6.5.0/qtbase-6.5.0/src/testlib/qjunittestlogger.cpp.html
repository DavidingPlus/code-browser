40QJUnitTestLogger::~QJUnitTestLogger()
41{
42 Q_ASSERT(!currentTestSuite);
43 delete logFormatter;
44}

46// We track test timing per test case, so we
47// need to maintain our own elapsed timer.
48Q_CONSTINIT static QElapsedTimer elapsedTestcaseTime;
49static qreal elapsedTestCaseSeconds()
50{
51 return elapsedTestcaseTime.nsecsElapsed() / 1e9;
52}
53
54static QByteArray toSecondsFormat(qreal ms)
55{
56 return QByteArray::number(ms / 1000, format: 'f', precision: 3);
57}

59void QJUnitTestLogger::startLogging()
60{
61 QAbstractTestLogger::startLogging();
62
63 logFormatter = new QTestJUnitStreamer(this);
64
65 Q_ASSERT(!currentTestSuite);
66 currentTestSuite = new QTestElement(QTest::LET_TestSuite);
67 currentTestSuite->addAttribute(index: QTest::AI_Name, value: QTestResult::currentTestObjectName());
68
69 auto localTime = QDateTime::currentDateTime();
70 currentTestSuite->addAttribute(index: QTest::AI_Timestamp,
71 value: localTime.toString(format: Qt::ISODate).toUtf8().constData());
72
73 currentTestSuite->addAttribute(index: QTest::AI_Hostname,
74 value: QSysInfo::machineHostName().toUtf8().constData());
75
76 QTestElement *property;
77 QTestElement *properties = new QTestElement(QTest::LET_Properties);
78
79 property = new QTestElement(QTest::LET_Property);
80 property->addAttribute(index: QTest::AI_Name, value: "QTestVersion");
81 property->addAttribute(index: QTest::AI_PropertyValue, QTEST_VERSION_STR);
82 properties->addChild(element: property);
83
84 property = new QTestElement(QTest::LET_Property);
85 property->addAttribute(index: QTest::AI_Name, value: "QtVersion");
86 property->addAttribute(index: QTest::AI_PropertyValue, value: qVersion());
87 properties->addChild(element: property);
88
89 property = new QTestElement(QTest::LET_Property);
90 property->addAttribute(index: QTest::AI_Name, value: "QtBuild");
91 property->addAttribute(index: QTest::AI_PropertyValue, value: QLibraryInfo::build());
92 properties->addChild(element: property);
93
94 currentTestSuite->addChild(element: properties);
95
96 elapsedTestcaseTime.start();
97}

99void QJUnitTestLogger::stopLogging()
100{
101 char buf[10];
102
103 qsnprintf(str: buf, n: sizeof(buf), fmt: "%i", testCounter);
104 currentTestSuite->addAttribute(index: QTest::AI_Tests, value: buf);
105
106 qsnprintf(str: buf, n: sizeof(buf), fmt: "%i", failureCounter);
107 currentTestSuite->addAttribute(index: QTest::AI_Failures, value: buf);
108
109 qsnprintf(str: buf, n: sizeof(buf), fmt: "%i", errorCounter);
110 currentTestSuite->addAttribute(index: QTest::AI_Errors, value: buf);
111
112 qsnprintf(str: buf, n: sizeof(buf), fmt: "%i", QTestLog::skipCount());
113 currentTestSuite->addAttribute(index: QTest::AI_Skipped, value: buf);
114
115 currentTestSuite->addAttribute(index: QTest::AI_Time,
116 value: toSecondsFormat(ms: QTestLog::msecsTotalTime()).constData());
117
118 for (auto *testCase : listOfTestcases)
119 currentTestSuite->addChild(element: testCase);
120 listOfTestcases.clear();
121
122 logFormatter->output(element: currentTestSuite);
123
124 delete currentTestSuite;
125 currentTestSuite = nullptr;
126
127 QAbstractTestLogger::stopLogging();
128}

135void QJUnitTestLogger::enterTestCase(const char *name)
136{
137 currentTestCase = new QTestElement(QTest::LET_TestCase);
138 currentTestCase->addAttribute(index: QTest::AI_Name, value: name);
139 currentTestCase->addAttribute(index: QTest::AI_Classname, value: QTestResult::currentTestObjectName());
140 listOfTestcases.push_back(x: currentTestCase);
141
142 Q_ASSERT(!systemOutputElement && !systemErrorElement);
143 systemOutputElement = new QTestElement(QTest::LET_SystemOutput);
144 systemErrorElement = new QTestElement(QTest::LET_SystemError);
145
146 // The element will be deleted when the suite is deleted
147
148 ++testCounter;
149
150 elapsedTestcaseTime.restart();
151}

153void QJUnitTestLogger::enterTestData(QTestData *)
154{
155 QTestCharBuffer testIdentifier;
156 QTestPrivate::generateTestIdentifier(identifier: &testIdentifier,
157 parts: QTestPrivate::TestFunction | QTestPrivate::TestDataTag);
158
159 static const char *lastTestFunction = nullptr;
160 if (QTestResult::currentTestFunction() != lastTestFunction) {
161 // Adopt existing testcase for the initial test data
162 auto *name = const_cast<QTestElementAttribute*>(
163 currentTestCase->attribute(index: QTest::AI_Name));
164 name->setPair(attributeIndex: QTest::AI_Name, value: testIdentifier.data());
165 lastTestFunction = QTestResult::currentTestFunction();
166 elapsedTestcaseTime.restart();
167 } else {
168 // Create new test cases for remaining test data
169 leaveTestCase();
170 enterTestCase(name: testIdentifier.data());
171 }
172}
173
174void QJUnitTestLogger::leaveTestFunction()
175{
176 leaveTestCase();
177}

179void QJUnitTestLogger::leaveTestCase()
180{
181 currentTestCase->addAttribute(index: QTest::AI_Time,
182 value: toSecondsFormat(ms: elapsedTestCaseSeconds() * 1000).constData());
183
184 if (!systemOutputElement->childElements().empty())
185 currentTestCase->addChild(element: systemOutputElement);
186 else
187 delete systemOutputElement;
188
189 if (!systemErrorElement->childElements().empty())
190 currentTestCase->addChild(element: systemErrorElement);
191 else
192 delete systemErrorElement;
193
194 systemOutputElement = nullptr;
195 systemErrorElement = nullptr;
196}

198void QJUnitTestLogger::addIncident(IncidentTypes type, const char *description,
199 const char *file, int line)
200{
201 if (type == Fail || type == XPass) {
202 auto failureType = [&]() {
203 switch (type) {
204 case QAbstractTestLogger::Fail: return "fail";
205 case QAbstractTestLogger::XPass: return "xpass";
206 default: Q_UNREACHABLE();
207 }
208 }();
209
210 addFailure(elementType: QTest::LET_Failure, failureType, failureDescription: QString::fromUtf8(utf8: description));
211 } else if (type == XFail) {
212 // Since XFAIL does not add a failure to the testlog in JUnit XML we add a
213 // message, so we still have some information about the expected failure.
214 addMessage(type: Info, message: QString::fromUtf8(utf8: description), file, line);
215 } else if (type == Skip) {
216 auto skippedElement = new QTestElement(QTest::LET_Skipped);
217 skippedElement->addAttribute(index: QTest::AI_Message, value: description);
218 currentTestCase->addChild(element: skippedElement);
219 }
220}

222void QJUnitTestLogger::addFailure(QTest::LogElementType elementType,
223 const char *failureType, const QString &failureDescription)
224{
225 if (elementType == QTest::LET_Failure) {
226 // Make sure we're not adding failure when we already have error,
227 // or adding additional failures when we already have a failure.
228 for (auto *childElement : currentTestCase->childElements()) {
229 if (childElement->elementType() == QTest::LET_Error ||
230 childElement->elementType() == QTest::LET_Failure)
231 return;
232 }
233 }
234
235 QTestElement *failureElement = new QTestElement(elementType);
236 failureElement->addAttribute(index: QTest::AI_Type, value: failureType);
237
238 // Assume the first line is the message, and the remainder are details
239 QString message = failureDescription.section(asep: u'\n', astart: 0, aend: 0);
240 QString details = failureDescription.section(asep: u'\n', astart: 1);
241
242 failureElement->addAttribute(index: QTest::AI_Message, value: message.toUtf8().constData());
243
244 if (!details.isEmpty()) {
245 auto textNode = new QTestElement(QTest::LET_Text);
246 textNode->addAttribute(index: QTest::AI_Value, value: details.toUtf8().constData());
247 failureElement->addChild(element: textNode);
248 }
249
250 currentTestCase->addChild(element: failureElement);
251
252 switch (elementType) {
253 case QTest::LET_Failure: ++failureCounter; break;
254 case QTest::LET_Error: ++errorCounter; break;
255 default: Q_UNREACHABLE();
256 }
257}

259void QJUnitTestLogger::addMessage(MessageTypes type, const QString &message, const char *file, int line)
260{
261 Q_UNUSED(file);
262 Q_UNUSED(line);
263
264 if (type == QFatal) {
265 addFailure(elementType: QTest::LET_Error, failureType: "qfatal", failureDescription: message);
266 return;
267 }
268
269 auto systemLogElement = [&]() {
270 switch (type) {
271 case QAbstractTestLogger::QDebug:
272 case QAbstractTestLogger::Info:
273 case QAbstractTestLogger::QInfo:
274 return systemOutputElement;
275 case QAbstractTestLogger::Warn:
276 case QAbstractTestLogger::QWarning:
277 case QAbstractTestLogger::QCritical:
278 return systemErrorElement;
279 default:
280 Q_UNREACHABLE();
281 }
282 }();
283
284 if (!systemLogElement)
285 return; // FIXME: Handle messages outside of test functions
286
287 auto textNode = new QTestElement(QTest::LET_Text);
288 textNode->addAttribute(index: QTest::AI_Value, value: message.toUtf8().constData());
289 systemLogElement->addChild(element: textNode);
290}
