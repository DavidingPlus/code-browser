51struct PerfEvent
52{
53 quint32 type;
54 quint64 config;
55};
56static perf_event_attr attr;
57Q_GLOBAL_STATIC(QList<PerfEvent>, eventTypes);
58
59static void initPerf()
60{
61 static bool done;
62 if (!done) {
63 memset(s: &attr, c: 0, n: sizeof attr);
64 attr.size = sizeof attr;
65 attr.read_format = PERF_FORMAT_TOTAL_TIME_ENABLED | PERF_FORMAT_TOTAL_TIME_RUNNING;
66 attr.disabled = true; // we'll enable later
67 attr.inherit = true; // let children processes inherit the monitoring
68 attr.pinned = true; // keep it running in the hardware
69 attr.inherit_stat = true; // aggregate all the info from child processes
70 attr.task = true; // trace fork/exits
71
72 done = true;
73 }
74}

88/*
89 \class QBenchmarkPerfEvents
90 \brief The Linux perf events benchmark backend
91
92 This benchmark backend uses the Linux Performance Counters interface,
93 introduced with the Linux kernel v2.6.31. The interface is done by one
94 system call (perf_event_open) which takes an attribute structure and
95 returns a file descriptor.
96
97 More information:
98 \li design docs: tools/perf/design.txt <http://lxr.linux.no/linux/tools/perf/design.txt>
99 \li sample tool: tools/perf/builtin-stat.c <http://lxr.linux.no/linux/tools/perf/builtin-stat.c>
100 (note: as of v3.3.1, the documentation is out-of-date with the kernel
101 interface, so reading the source code of existing tools is necessary)
102
103 This benchlib backend monitors the current process as well as child process
104 launched. We do not try to benchmark in kernel or hypervisor mode, as that
105 usually requires elevated privileges.
106 */
107
108static int perf_event_open(perf_event_attr *attr, pid_t pid, int cpu, int group_fd, unsigned long flags)
109{
110#ifdef SYS_perf_event_open
111 return syscall(SYS_perf_event_open, attr, pid, cpu, group_fd, flags);
112#else
113 Q_UNUSED(attr);
114 Q_UNUSED(pid);
115 Q_UNUSED(cpu);
116 Q_UNUSED(group_fd);
117 Q_UNUSED(flags);
118 errno = ENOSYS;
119 return -1;
120#endif
121}
122
123bool QBenchmarkPerfEventsMeasurer::isAvailable()
124{
125 // this generates an EFAULT because attr == NULL if perf_event_open is available
126 // if the kernel is too old, it generates ENOSYS
127 return perf_event_open(attr: nullptr, pid: 0, cpu: 0, group_fd: 0, flags: 0) == -1 && errno != ENOSYS;
128}

407void QBenchmarkPerfEventsMeasurer::setCounter(const char *name)
408{
409 initPerf();
410 eventTypes->clear();
411 std::string_view input = name;
412 if (qsizetype idx = input.find(c: ':'); idx >= 0)
413 input = input.substr(pos: 0, n: idx);
414
415 while (!input.empty()) {
416 std::string_view countername = input;
417 if (qsizetype idx = countername.find(c: ','); idx >= 0)
418 countername = countername.substr(pos: 0, n: idx);
419
420 for (const Events &ev : eventlist) {
421 int c = countername.compare(str: eventlist_strings + ev.offset);
422 if (c > 0)
423 continue;
424 if (c < 0) {
425 fprintf(stderr, format: "ERROR: Performance counter type '%.*s' is unknown\n",
426 int(countername.size()), countername.data());
427 exit(status: 1);
428 }
429 eventTypes->append(t: { .type: ev.type, .config: ev.event_id });
430 break;
431 }
432
433 if (countername.size() == input.size())
434 input = {};
435 else
436 input.remove_prefix(n: countername.size() + 1);
437 }
438
439 // We used to support attributes, but our code was the opposite of what
440 // perf(1) does, plus QBenchlib isn't exactly expected to be used to
441 // profile Linux kernel code or launch guest VMs as part of the workload.
442 // So we keep accepting the colon as a delimiter but ignore it.
443}

473void QBenchmarkPerfEventsMeasurer::start()
474{
475 initPerf();
476 QList<PerfEvent> &counters = *eventTypes;
477 if (counters.isEmpty())
478 counters = defaultCounters();
479 if (fds.isEmpty()) {
480 pid_t pid = 0; // attach to the current process only
481 int cpu = -1; // on any CPU
482 int group_fd = -1;
483 int flags = PERF_FLAG_FD_CLOEXEC;
484
485 fds.reserve(size: counters.size());
486 for (PerfEvent counter : std::as_const(t&: counters)) {
487 attr.type = counter.type;
488 attr.config = counter.config;
489 int fd = perf_event_open(attr: &attr, pid, cpu, group_fd, flags);
490 if (fd == -1) {
491 // probably a paranoid kernel (/proc/sys/kernel/perf_event_paranoid)
492 attr.exclude_kernel = true;
493 attr.exclude_hv = true;
494 fd = perf_event_open(attr: &attr, pid, cpu, group_fd, flags);
495 }
496 if (fd == -1) {
497 perror(s: "QBenchmarkPerfEventsMeasurer::start: perf_event_open");
498 exit(status: 1);
499 }
500
501 fds.append(t: fd);
502 }
503 }
504
505 // enable the counters
506 for (int fd : std::as_const(t&: fds))
507 ::ioctl(fd: fd, PERF_EVENT_IOC_RESET);
508 prctl(PR_TASK_PERF_EVENTS_ENABLE);
509}

539static quint64 rawReadValue(int fd)
540{
541 /* from the kernel docs:
542 * struct read_format {
543 * { u64 value;
544 * { u64 time_enabled; } && PERF_FORMAT_TOTAL_TIME_ENABLED
545 * { u64 time_running; } && PERF_FORMAT_TOTAL_TIME_RUNNING
546 * { u64 id; } && PERF_FORMAT_ID
547 * } && !PERF_FORMAT_GROUP
548 */
549
550 struct read_format {
551 quint64 value;
552 quint64 time_enabled;
553 quint64 time_running;
554 } results;
555
556 size_t nread = 0;
557 while (nread < sizeof results) {
558 char *ptr = reinterpret_cast<char *>(&results);
559 qint64 r = qt_safe_read(fd, data: ptr + nread, maxlen: sizeof results - nread);
560 if (r == -1) {
561 perror(s: "QBenchmarkPerfEventsMeasurer::readValue: reading the results");
562 exit(status: 1);
563 }
564 nread += quint64(r);
565 }
566
567 if (results.time_running == results.time_enabled)
568 return results.value;
569
570 // scale the results, though this shouldn't happen!
571 return results.value * (double(results.time_running) / double(results.time_enabled));
572}
573
574QBenchmarkMeasurerBase::Measurement QBenchmarkPerfEventsMeasurer::readValue(qsizetype idx)
575{
576 quint64 raw = rawReadValue(fd: fds.at(i: idx));
577 return { .value: qreal(qint64(raw)), .metric: metricForEvent(counter: eventTypes->at(i: idx)) };
578}
