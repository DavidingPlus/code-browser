43static const char multiplePrefixes[] = "\0kMGTPE"; // kilo, mega, giga, tera, peta, exa
44static const char submultiplePrefixes[] = "afpnum"; // atto, femto, pico, nano, micro, milli
45
46template <int N> struct FixedBufString
47{
48 static constexpr size_t MaxSize = N;
49 size_t used = 0;
50 std::array<char, N + 2> buf; // for the newline and terminating null
51 FixedBufString()
52 {
53 clear();
54 }
55 void clear()
56 {
57 used = 0;
58 buf[0] = '\0';
59 }
60
61 operator const char *() const
62 {
63 return buf.data();
64 }
65
66 void append(const char *text)
67 {
68 size_t len = qMin(a: strlen(s: text), b: MaxSize - used);
69 memcpy(buf.data() + used, text, len);
70 used += len;
71 buf[used] = '\0';
72 }
73
74 template <typename... Args> void appendf(const char *format, Args &&... args)
75 {
76 // vsnprintf includes the terminating null
77 used += qsnprintf(buf.data() + used, MaxSize - used + 1, format,
78 std::forward<Args>(args)...);
79 }
80
81 template <int Power = 1000> void appendScaled(qreal value, const char *unit)
82 {
83 char prefix[2] = {};
84 qreal v = qAbs(t: value);
85 qint64 ratio;
86 if (v < 1 && Power == 1000) {
87 const char *prefixes = submultiplePrefixes;
88 ratio = qreal(std::atto::num) / std::atto::den;
89 while (value * ratio > 1000 && *prefixes) {
90 ++prefixes;
91 ratio *= 1000;
92 }
93 prefix[0] = *prefixes;
94 } else {
95 const char *prefixes = multiplePrefixes;
96 ratio = 1;
97 while (value > 1000 * ratio) { // yes, even for binary
98 ++prefixes;
99 ratio *= Power;
100 }
101 prefix[0] = *prefixes;
102 }
103
104 // adjust the value by the ratio
105 value /= ratio;
106 appendf(", %.3g %s%s", value, prefix, unit);
107 }
108};
109} // unnamed namespace

164 template <typename T>
165 static int countSignificantDigits(T num)
166 {
167 if (num <= 0)
168 return 0;
169
170 int digits = 0;
171 qreal divisor = 1;
172
173 while (num / divisor >= 1) {
174 divisor *= 10;
175 ++digits;
176 }
177
178 return digits;
179 }
180
181 // Pretty-prints a benchmark result using the given number of digits.
182 template <typename T> QByteArray formatResult(T number, int significantDigits)
183 {
184 if (number < T(0))
185 return "NAN";
186 if (number == T(0))
187 return "0";
188
189 QByteArray beforeDecimalPoint = QByteArray::number(qint64(number), format: 'f', precision: 0);
190 QByteArray afterDecimalPoint = QByteArray::number(number, 'f', 20);
191 afterDecimalPoint.remove(index: 0, len: beforeDecimalPoint.size() + 1);
192
193 int beforeUse = qMin(a: beforeDecimalPoint.size(), b: significantDigits);
194 int beforeRemove = beforeDecimalPoint.size() - beforeUse;
195
196 // Replace insignificant digits before the decimal point with zeros.
197 beforeDecimalPoint.chop(n: beforeRemove);
198 for (int i = 0; i < beforeRemove; ++i) {
199 beforeDecimalPoint.append(c: u'0');
200 }
201
202 int afterUse = significantDigits - beforeUse;
203
204 // leading zeroes after the decimal point does not count towards the digit use.
205 if (beforeDecimalPoint == "0" && !afterDecimalPoint.isEmpty()) {
206 ++afterUse;
207
208 int i = 0;
209 while (i < afterDecimalPoint.size() && afterDecimalPoint.at(i) == '0')
210 ++i;
211
212 afterUse += i;
213 }
214
215 int afterRemove = afterDecimalPoint.size() - afterUse;
216 afterDecimalPoint.chop(n: afterRemove);
217
218 char separator = ',';
219 char decimalPoint = '.';
220
221 // insert thousands separators
222 int length = beforeDecimalPoint.size();
223 for (int i = beforeDecimalPoint.size() -1; i >= 1; --i) {
224 if ((length - i) % 3 == 0)
225 beforeDecimalPoint.insert(i, c: separator);
226 }
227
228 QByteArray print;
229 print = beforeDecimalPoint;
230 if (afterUse > 0)
231 print.append(c: decimalPoint);
232
233 print += afterDecimalPoint;
234
235
236 return print;
237 }

249void QPlainTestLogger::outputMessage(const char *str)
250{
251#if defined(Q_OS_WIN)
252 // Log to system log only if output is not redirected and stderr not preferred
253 if (stream == stdout && !QtPrivate::shouldLogToStderr()) {
254 OutputDebugStringA(str);
255 return;
256 }
257#elif defined(Q_OS_ANDROID)
258 __android_log_write(ANDROID_LOG_INFO, "QTestLib", str);
259#endif
260 outputString(msg: str);
261}

263void QPlainTestLogger::printMessage(MessageSource source, const char *type, const char *msg,
264 const char *file, int line)
265{
266 QTEST_ASSERT(type);
267 QTEST_ASSERT(msg);
268
269 QTestCharBuffer messagePrefix;
270
271 QTestCharBuffer messageLocation;
272#ifdef Q_OS_WIN
273 constexpr const char *INCIDENT_LOCATION_STR = "\n%s(%d) : failure location";
274 constexpr const char *OTHER_LOCATION_STR = "\n%s(%d) : message location";
275#else
276 constexpr const char *INCIDENT_LOCATION_STR = "\n Loc: [%s(%d)]";
277 constexpr const char *OTHER_LOCATION_STR = INCIDENT_LOCATION_STR;
278#endif
279
280 if (file) {
281 switch (source) {
282 case MessageSource::Incident:
283 QTest::qt_asprintf(buf: &messageLocation, format: INCIDENT_LOCATION_STR, file, line);
284 break;
285 case MessageSource::Other:
286 QTest::qt_asprintf(buf: &messageLocation, format: OTHER_LOCATION_STR, file, line);
287 break;
288 }
289 }
290
291 const char *msgFiller = msg[0] ? " " : "";
292 QTestCharBuffer testIdentifier;
293 QTestPrivate::generateTestIdentifier(identifier: &testIdentifier);
294 QTest::qt_asprintf(buf: &messagePrefix, format: "%s: %s%s%s%s\n",
295 type, testIdentifier.data(), msgFiller, msg, messageLocation.data());
296
297 // In colored mode, printf above stripped our nonprintable control characters.
298 // Put them back.
299 memcpy(dest: messagePrefix.data(), src: type, n: strlen(s: type));
300
301 outputMessage(str: messagePrefix.data());
302}

304void QPlainTestLogger::printBenchmarkResultsHeader(const QBenchmarkResult &result)
305{
306 FixedBufString<1022> buf;
307 buf.appendf(format: "%s: %s::%s", args: QTest::benchmarkResult2String(),
308 args: QTestResult::currentTestObjectName(), args: result.context.slotName.toLatin1().data());
309
310 if (QByteArray tag = result.context.tag.toLocal8Bit(); !tag.isEmpty())
311 buf.appendf(format: ":\"%s\":\n", args: tag.data());
312 else
313 buf.append(text: ":\n");
314 outputMessage(str: buf);
315}

317void QPlainTestLogger::printBenchmarkResults(const QList<QBenchmarkResult> &results)
318{
319 using namespace std::chrono;
320 FixedBufString<1022> buf;
321 auto findResultFor = [&results](QTest::QBenchmarkMetric metric) -> std::optional<qreal> {
322 for (const QBenchmarkResult &result : results) {
323 if (result.measurement.metric == metric)
324 return result.measurement.value;
325 }
326 return std::nullopt;
327 };
328
329 // we need the execution time quite often, so find it first
330 qreal executionTime = 0;
331 if (auto ns = findResultFor(QTest::WalltimeNanoseconds))
332 executionTime = *ns / (1000 * 1000 * 1000);
333 else if (auto ms = findResultFor(QTest::WalltimeMilliseconds))
334 executionTime = *ms / 1000;
335
336 for (const QBenchmarkResult &result : results) {
337 buf.clear();
338
339 const char * unitText = QTest::benchmarkMetricUnit(metric: result.measurement.metric);
340 int significantDigits = QTest::countSignificantDigits(num: result.measurement.value);
341 qreal valuePerIteration = qreal(result.measurement.value) / qreal(result.iterations);
342 buf.appendf(format: " %s %s%s", args: QTest::formatResult(number: valuePerIteration, significantDigits).constData(),
343 args&: unitText, args: result.setByMacro ? " per iteration" : "");
344
345 switch (result.measurement.metric) {
346 case QTest::BitsPerSecond:
347 // for bits/s, we'll use powers of 10 (1 Mbit/s = 1000 kbit/s = 1000000 bit/s)
348 buf.appendScaled<1000>(value: result.measurement.value, unit: "bit/s");
349 break;
350 case QTest::BytesPerSecond:
351 // for B/s, we'll use powers of 2 (1 MB/s = 1024 kB/s = 1048576 B/s)
352 buf.appendScaled<1024>(value: result.measurement.value, unit: "B/s");
353 break;
354
355 case QTest::CPUCycles:
356 case QTest::RefCPUCycles:
357 if (!qIsNull(d: executionTime))
358 buf.appendScaled(value: result.measurement.value / executionTime, unit: "Hz");
359 break;
360
361 case QTest::Instructions:
362 if (auto cycles = findResultFor(QTest::CPUCycles)) {
363 buf.appendf(format: ", %.3f instr/cycle", args: result.measurement.value / *cycles);
364 break;
365 }
366 Q_FALLTHROUGH();
367
368 case QTest::InstructionReads:
369 case QTest::Events:
370 case QTest::BytesAllocated:
371 case QTest::CPUMigrations:
372 case QTest::BusCycles:
373 case QTest::StalledCycles:
374 case QTest::BranchInstructions:
375 case QTest::BranchMisses:
376 case QTest::CacheReferences:
377 case QTest::CacheReads:
378 case QTest::CacheWrites:
379 case QTest::CachePrefetches:
380 case QTest::CacheMisses:
381 case QTest::CacheReadMisses:
382 case QTest::CacheWriteMisses:
383 case QTest::CachePrefetchMisses:
384 case QTest::ContextSwitches:
385 case QTest::PageFaults:
386 case QTest::MinorPageFaults:
387 case QTest::MajorPageFaults:
388 case QTest::AlignmentFaults:
389 case QTest::EmulationFaults:
390 if (!qIsNull(d: executionTime))
391 buf.appendScaled(value: result.measurement.value / executionTime, unit: "/sec");
392 break;
393
394 case QTest::FramesPerSecond:
395 case QTest::CPUTicks:
396 case QTest::WalltimeMilliseconds:
397 case QTest::WalltimeNanoseconds:
398 break; // no additional information
399 }
400
401 Q_ASSERT(result.iterations > 0);
402 buf.appendf(format: " (total: %s, iterations: %d)\n",
403 args: QTest::formatResult(number: result.measurement.value, significantDigits).constData(),
404 args: result.iterations);
405
406 outputMessage(str: buf);
407 }

417void QPlainTestLogger::startLogging()
418{
419 QAbstractTestLogger::startLogging();
420
421 char buf[1024];
422 if (QTestLog::verboseLevel() < 0) {
423 qsnprintf(str: buf, n: sizeof(buf), fmt: "Testing %s\n", QTestResult::currentTestObjectName());
424 } else {
425 qsnprintf(str: buf, n: sizeof(buf),
426 fmt: "********* Start testing of %s *********\n"
427 "Config: Using QtTest library " QTEST_VERSION_STR
428 ", %s, %s %s\n", QTestResult::currentTestObjectName(), QLibraryInfo::build(),
429 qPrintable(QSysInfo::productType()), qPrintable(QSysInfo::productVersion()));
430 }
431 outputMessage(str: buf);
432}
433
434void QPlainTestLogger::stopLogging()
435{
436 char buf[1024];
437 const int timeMs = qRound(d: QTestLog::msecsTotalTime());
438 if (QTestLog::verboseLevel() < 0) {
439 qsnprintf(str: buf, n: sizeof(buf), fmt: "Totals: %d passed, %d failed, %d skipped, %d blacklisted, %dms\n",
440 QTestLog::passCount(), QTestLog::failCount(),
441 QTestLog::skipCount(), QTestLog::blacklistCount(), timeMs);
442 } else {
443 qsnprintf(str: buf, n: sizeof(buf),
444 fmt: "Totals: %d passed, %d failed, %d skipped, %d blacklisted, %dms\n"
445 "********* Finished testing of %s *********\n",
446 QTestLog::passCount(), QTestLog::failCount(),
447 QTestLog::skipCount(), QTestLog::blacklistCount(), timeMs,
448 QTestResult::currentTestObjectName());
449 }
450 outputMessage(str: buf);
451
452 QAbstractTestLogger::stopLogging();
453}

466void QPlainTestLogger::addIncident(IncidentTypes type, const char *description,
467 const char *file, int line)
468{
469 // suppress B?PASS and B?XFAIL in silent mode
470 if ((type == Pass || type == BlacklistedPass || type == XFail || type == BlacklistedXFail)
471 && QTestLog::verboseLevel() < 0)
472 return;
473
474 printMessage(source: MessageSource::Incident, type: QTest::ptIncidentType2String(type), msg: description, file, line);
475}

477void QPlainTestLogger::addBenchmarkResults(const QList<QBenchmarkResult> &results)
478{
479 // suppress benchmark results in silent mode
480 if (QTestLog::verboseLevel() < 0 || results.isEmpty())
481 return;
482
483 printBenchmarkResultsHeader(result: results.first());
484 printBenchmarkResults(results);
485}

492void QPlainTestLogger::addMessage(MessageTypes type, const QString &message,
493 const char *file, int line)
494{
495 // suppress non-fatal messages in silent mode
496 if (type != QFatal && QTestLog::verboseLevel() < 0)
497 return;
498
499 printMessage(source: MessageSource::Other, type: QTest::ptMessageType2String(type), qPrintable(message), file, line);
500}
