18#include "qshader_p.h"
19#include <QtCore/QAtomicInt>
20#include <QtCore/QMap>
21#include <QtCore/QDebug>
22
23QT_BEGIN_NAMESPACE
24
25struct Q_GUI_EXPORT QShaderPrivate
26{
27 static const int QSB_VERSION = 8;
28 static const int QSB_VERSION_WITHOUT_EXTENDED_STORAGE_BUFFER_INFO = 7;
29 static const int QSB_VERSION_WITHOUT_NATIVE_SHADER_INFO = 6;
30 static const int QSB_VERSION_WITHOUT_SEPARATE_IMAGES_AND_SAMPLERS = 5;
31 static const int QSB_VERSION_WITHOUT_VAR_ARRAYDIMS = 4;
32 static const int QSB_VERSION_WITH_CBOR = 3;
33 static const int QSB_VERSION_WITH_BINARY_JSON = 2;
34 static const int QSB_VERSION_WITHOUT_BINDINGS = 1;
35
36 enum MslNativeShaderInfoExtraBufferBindings {
37 MslTessVertIndicesBufferBinding = 0,
38 MslTessVertTescOutputBufferBinding,
39 MslTessTescTessLevelBufferBinding,
40 MslTessTescPatchOutputBufferBinding,
41 MslTessTescParamsBufferBinding,
42 MslTessTescInputBufferBinding
43 };
44
45 QShaderPrivate()
46 : ref(1)
47 {
48 }
49
50 QShaderPrivate(const QShaderPrivate &other)
51 : ref(1),
52 qsbVersion(other.qsbVersion),
53 stage(other.stage),
54 desc(other.desc),
55 shaders(other.shaders),
56 bindings(other.bindings),
57 combinedImageMap(other.combinedImageMap),
58 nativeShaderInfoMap(other.nativeShaderInfoMap)
59 {
60 }
61
62 static QShaderPrivate *get(QShader *s) { return s->d; }
63 static const QShaderPrivate *get(const QShader *s) { return s->d; }
64 static int qtQsbVersion(QShader::SerializedFormatVersion qtVersion) {
65 switch (qtVersion) {
66 case QShader::SerializedFormatVersion::Qt_6_4:
67 return (QShaderPrivate::QSB_VERSION_WITHOUT_SEPARATE_IMAGES_AND_SAMPLERS + 1);
68 case QShader::SerializedFormatVersion::Qt_6_5:
69 return (QShaderPrivate::QSB_VERSION_WITHOUT_EXTENDED_STORAGE_BUFFER_INFO + 1);
70 default:
71 return QShaderPrivate::QSB_VERSION;
72 }
73 }
74
75 QAtomicInt ref;
76 int qsbVersion = QSB_VERSION;
77 QShader::Stage stage = QShader::VertexStage;
78 QShaderDescription desc;
79 // QMap not QHash because we need to be able to iterate based on sorted keys
80 QMap<QShaderKey, QShaderCode> shaders;
81 QMap<QShaderKey, QShader::NativeResourceBindingMap> bindings;
82 QMap<QShaderKey, QShader::SeparateToCombinedImageSamplerMappingList> combinedImageMap;
83 QMap<QShaderKey, QShader::NativeShaderInfo> nativeShaderInfoMap;
84};
85
86QT_END_NAMESPACE
