97 && a.glwrapt == b.glwrapt

601 QVarLengthArray<QByteArray, 4> dataRetainPool;
602 QVarLengthArray<QRhiBufferData, 4> bufferDataRetainPool;
603 QVarLengthArray<QImage, 4> imageRetainPool;
604
605 // relies heavily on implicit sharing (no copies of the actual data will be made)
606 const void *retainData(const QByteArray &data) {
607 dataRetainPool.append(t: data);
608 return dataRetainPool.last().constData();
609 }
610 const uchar *retainBufferData(const QRhiBufferData &data) {
611 bufferDataRetainPool.append(t: data);
612 return reinterpret_cast<const uchar *>(bufferDataRetainPool.last().constData());
613 }
614 const void *retainImage(const QImage &image) {
615 imageRetainPool.append(t: image);
616 return imageRetainPool.last().constBits();
617 }
618 void resetCommands() {
619 commands.reset();
620 dataRetainPool.clear();
621 bufferDataRetainPool.clear();
622 imageRetainPool.clear();
623
624 passResTrackers.clear();
625 currentPassResTrackerIndex = -1;
626 }
627 void resetState() {
628 recordingPass = NoPass;
629 passNeedsResourceTracking = true;
630 currentTarget = nullptr;
631 resetCommands();
632 resetCachedState();
633 }
634 void resetCachedState() {
635 currentGraphicsPipeline = nullptr;
636 currentComputePipeline = nullptr;
637 currentPipelineGeneration = 0;
638 currentGraphicsSrb = nullptr;
639 currentComputeSrb = nullptr;
640 currentSrbGeneration = 0;
641 graphicsPassState.reset();
642 computePassState.reset();
643 memset(s: textureUnitState, c: 0, n: sizeof(textureUnitState));
644 }
