254template<QRhiTargetRectBoundMode boundingMode, typename T, size_t N>
255bool qrhi_toTopLeftRenderTargetRect(const QSize &outputSize, const std::array<T, N> &r,
256 T *x, T *y, T *w, T *h)
257{
258 // x,y are bottom-left in QRhiScissor and QRhiViewport but top-left in
259 // Vulkan/Metal/D3D. Our input is an OpenGL-style scissor rect where both
260 // negative x or y, and partly or completely out of bounds rects are
261 // allowed. The only thing the input here cannot have is a negative width
262 // or height. We must handle all other input gracefully, clamping to a zero
263 // width or height rect in the worst case, and ensuring the resulting rect
264 // is inside the rendertarget's bounds because some APIs' validation/debug
265 // layers are allergic to out of bounds scissor rects.
266
267 const T outputWidth = outputSize.width();
268 const T outputHeight = outputSize.height();
269 const T inputWidth = r[2];
270 const T inputHeight = r[3];
271
272 if (inputWidth < 0 || inputHeight < 0)
273 return false;
274
275 *x = r[0];
276 *y = outputHeight - (r[1] + inputHeight);
277 *w = inputWidth;
278 *h = inputHeight;
279
280 if (boundingMode == Bounded) {
281 const T widthOffset = *x < 0 ? -*x : 0;
282 const T heightOffset = *y < 0 ? -*y : 0;
283 *w = *x < outputWidth ? qMax<T>(0, inputWidth - widthOffset) : 0;
284 *h = *y < outputHeight ? qMax<T>(0, inputHeight - heightOffset) : 0;
285
286 *x = qBound<T>(0, *x, outputWidth - 1);
287 *y = qBound<T>(0, *y, outputHeight - 1);
288
289 if (*x + *w > outputWidth)
290 *w = qMax<T>(0, outputWidth - *x);
291 if (*y + *h > outputHeight)
292 *h = qMax<T>(0, outputHeight - *y);
293 }
294 return true;
295}

310// no detach-with-contents, no atomic refcount, no shrink
311class QRhiBufferData
312{
313public:
314 QRhiBufferData() = default;
315 ~QRhiBufferData()
316 {
317 if (d && !--d->ref)
318 delete d;
319 }
320 QRhiBufferData(const QRhiBufferData &other)
321 : d(other.d)
322 {
323 if (d)
324 d->ref += 1;
325 }
326 QRhiBufferData &operator=(const QRhiBufferData &other)
327 {
328 if (d == other.d)
329 return *this;
330 if (other.d)
331 other.d->ref += 1;
332 if (d && !--d->ref)
333 delete d;
334 d = other.d;
335 return *this;
336 }
337 const char *constData() const
338 {
339 return d->size <= QRhiBufferDataPrivate::SMALL_DATA_SIZE ? d->data : d->largeData;
340 }
341 quint32 size() const
342 {
343 return d->size;
344 }
345 void assign(const char *s, quint32 size)
346 {
347 if (!d) {
348 d = new QRhiBufferDataPrivate;
349 } else if (d->ref != 1) {
350 d->ref -= 1;
351 d = new QRhiBufferDataPrivate;
352 }
353 d->size = size;
354 if (size <= QRhiBufferDataPrivate::SMALL_DATA_SIZE) {
355 memcpy(dest: d->data, src: s, n: size);
356 } else {
357 if (d->largeAlloc < size) {
358 delete[] d->largeData;
359 d->largeAlloc = size;
360 d->largeData = new char[size];
361 }
362 memcpy(dest: d->largeData, src: s, n: size);
363 }
364 }
365private:
366 QRhiBufferDataPrivate *d = nullptr;
367};

523template<typename T>
524struct QRhiBatchedBindings
525{
526 void feed(int binding, T resource) { // binding must be strictly increasing
527 if (curBinding == -1 || binding > curBinding + 1) {
528 finish();
529 curBatch.startBinding = binding;
530 curBatch.resources.clear();
531 curBatch.resources.append(resource);
532 } else {
533 Q_ASSERT(binding == curBinding + 1);
534 curBatch.resources.append(resource);
535 }
536 curBinding = binding;
537 }
538
539 bool finish() {
540 if (!curBatch.resources.isEmpty())
541 batches.append(curBatch);
542 return !batches.isEmpty();
543 }
544
545 void clear() {
546 batches.clear();
547 curBatch.resources.clear();
548 curBinding = -1;
549 }
550
551 struct Batch {
552 uint startBinding;
553 QVarLengthArray<T, 4> resources;
554
555 bool operator==(const Batch &other) const
556 {
557 return startBinding == other.startBinding && resources == other.resources;
558 }
559
560 bool operator!=(const Batch &other) const
561 {
562 return !operator==(other);
563 }
564 };
565
566 QVarLengthArray<Batch, 4> batches; // sorted by startBinding
567
568 bool operator==(const QRhiBatchedBindings<T> &other) const
569 {
570 return batches == other.batches;
571 }
572
573 bool operator!=(const QRhiBatchedBindings<T> &other) const
574 {
575 return !operator==(other);
576 }
577
578private:
579 Batch curBatch;
580 int curBinding = -1;
581};

683template<typename T, int GROW = 1024>
684class QRhiBackendCommandList
685{
686public:
687 QRhiBackendCommandList() = default;
688 ~QRhiBackendCommandList() { delete[] v; }
689 inline void reset() { p = 0; }
690 inline bool isEmpty() const { return p == 0; }
691 inline T &get() {
692 if (p == a) {
693 a += GROW;
694 T *nv = new T[a];
695 if (v) {
696 memcpy(nv, v, p * sizeof(T));
697 delete[] v;
698 }
699 v = nv;
700 }
701 return v[p++];
702 }
703 inline void unget() { --p; }
704 inline T *cbegin() const { return v; }
705 inline T *cend() const { return v + p; }
706 inline T *begin() { return v; }
707 inline T *end() { return v + p; }
708private:
709 Q_DISABLE_COPY(QRhiBackendCommandList)
710 T *v = nullptr;
711 int a = 0;
712 int p = 0;
713};

738void QRhiRenderTargetAttachmentTracker::updateResIdList(const QRhiTextureRenderTargetDescription &desc, ResIdList *dst)
739{
740 const quintptr colorAttCount = desc.cendColorAttachments() - desc.cbeginColorAttachments();
741 const bool hasDepthStencil = desc.depthStencilBuffer() || desc.depthTexture();
742 dst->resize(sz: colorAttCount * 2 + (hasDepthStencil ? 1 : 0));
743 int n = 0;
744 for (auto it = desc.cbeginColorAttachments(), itEnd = desc.cendColorAttachments(); it != itEnd; ++it, ++n) {
745 const QRhiColorAttachment &colorAtt(*it);
746 if (colorAtt.texture()) {
747 TexType *texD = QRHI_RES(TexType, colorAtt.texture());
748 (*dst)[n] = { texD->globalResourceId(), texD->generation };
749 } else if (colorAtt.renderBuffer()) {
750 RenderBufferType *rbD = QRHI_RES(RenderBufferType, colorAtt.renderBuffer());
751 (*dst)[n] = { rbD->globalResourceId(), rbD->generation };
752 } else {
753 (*dst)[n] = { .id: 0, .generation: 0 };
754 }
755 ++n;
756 if (colorAtt.resolveTexture()) {
757 TexType *texD = QRHI_RES(TexType, colorAtt.resolveTexture());
758 (*dst)[n] = { texD->globalResourceId(), texD->generation };
759 } else {
760 (*dst)[n] = { .id: 0, .generation: 0 };
761 }
762 }
763 if (hasDepthStencil) {
764 if (desc.depthTexture()) {
765 TexType *depthTexD = QRHI_RES(TexType, desc.depthTexture());
766 (*dst)[n] = { depthTexD->globalResourceId(), depthTexD->generation };
767 } else if (desc.depthStencilBuffer()) {
768 RenderBufferType *depthRbD = QRHI_RES(RenderBufferType, desc.depthStencilBuffer());
769 (*dst)[n] = { depthRbD->globalResourceId(), depthRbD->generation };
770 } else {
771 (*dst)[n] = { .id: 0, .generation: 0 };
772 }
773 }
774}

776template<typename TexType, typename RenderBufferType>
777bool QRhiRenderTargetAttachmentTracker::isUpToDate(const QRhiTextureRenderTargetDescription &desc, const ResIdList &currentResIdList)
778{
779 // Just as setShaderResources() recognizes if an srb's referenced
780 // resources have been rebuilt (got a create() since the srb's
781 // create()), we should do the same for the textures and renderbuffers
782 // referenced from the rendertarget. It is not uncommon that a texture
783 // or ds buffer gets resized due to following a window size in some
784 // form, which involves a create() on them. It is then nice if the
785 // render target auto-rebuilds in beginPass().
786
787 ResIdList resIdList;
788 updateResIdList<TexType, RenderBufferType>(desc, &resIdList);
789 return resIdList == currentResIdList;
790}
