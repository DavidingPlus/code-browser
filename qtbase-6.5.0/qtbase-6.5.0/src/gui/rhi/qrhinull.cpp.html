10/*!
11 \class QRhiNullInitParams
12 \internal
13 \inmodule QtGui
14 \brief Null backend specific initialization parameters.
15
16 A Null QRhi needs no special parameters for initialization.
17
18 \badcode
19 QRhiNullInitParams params;
20 rhi = QRhi::create(QRhi::Null, &params);
21 \endcode
22
23 The Null backend does not issue any graphics calls and creates no
24 resources. All QRhi operations will succeed as normal so applications can
25 still be run, albeit potentially at an unthrottled speed, depending on
26 their frame rendering strategy.
27 */
28

381void QRhiNull::simulateTextureUpload(const QRhiResourceUpdateBatchPrivate::TextureOp &u)
382{
383 QNullTexture *texD = QRHI_RES(QNullTexture, u.dst);
384 for (int layer = 0, maxLayer = u.subresDesc.size(); layer < maxLayer; ++layer) {
385 for (int level = 0; level < QRhi::MAX_MIP_LEVELS; ++level) {
386 for (const QRhiTextureSubresourceUploadDescription &subresDesc : std::as_const(t: u.subresDesc[layer][level])) {
387 if (!subresDesc.image().isNull()) {
388 const QImage src = subresDesc.image();
389 QPainter painter(&texD->image[layer][level]);
390 const QSize srcSize = subresDesc.sourceSize().isEmpty()
391 ? src.size() : subresDesc.sourceSize();
392 painter.setCompositionMode(QPainter::CompositionMode_Source);
393 painter.drawImage(p: subresDesc.destinationTopLeft(), image: src,
394 sr: QRect(subresDesc.sourceTopLeft(), srcSize));
395 } else if (!subresDesc.data().isEmpty()) {
396 const QSize subresSize = q->sizeForMipLevel(mipLevel: level, baseLevelSize: texD->pixelSize());
397 int w = subresSize.width();
398 int h = subresSize.height();
399 if (!subresDesc.sourceSize().isEmpty()) {
400 w = subresDesc.sourceSize().width();
401 h = subresDesc.sourceSize().height();
402 }
403 // sourceTopLeft is not supported on this path as per QRhi docs
404 const char *src = subresDesc.data().constData();
405 const int srcBpl = w * 4;
406 int srcStride = srcBpl;
407 if (subresDesc.dataStride())
408 srcStride = subresDesc.dataStride();
409 const QPoint dstOffset = subresDesc.destinationTopLeft();
410 uchar *dst = texD->image[layer][level].bits();
411 const int dstBpl = texD->image[layer][level].bytesPerLine();
412 for (int y = 0; y < h; ++y) {
413 memcpy(dest: dst + dstOffset.x() * 4 + (y + dstOffset.y()) * dstBpl,
414 src: src + y * srcStride,
415 n: size_t(srcBpl));
416 }
417 }
418 }
419 }
420 }
421}

423void QRhiNull::simulateTextureCopy(const QRhiResourceUpdateBatchPrivate::TextureOp &u)
424{
425 QNullTexture *srcD = QRHI_RES(QNullTexture, u.src);
426 QNullTexture *dstD = QRHI_RES(QNullTexture, u.dst);
427 const QImage &srcImage(srcD->image[u.desc.sourceLayer()][u.desc.sourceLevel()]);
428 QImage &dstImage(dstD->image[u.desc.destinationLayer()][u.desc.destinationLevel()]);
429 const QPoint dstPos = u.desc.destinationTopLeft();
430 const QSize size = u.desc.pixelSize().isEmpty() ? srcD->pixelSize() : u.desc.pixelSize();
431 const QPoint srcPos = u.desc.sourceTopLeft();
432
433 QPainter painter(&dstImage);
434 painter.setCompositionMode(QPainter::CompositionMode_Source);
435 painter.drawImage(targetRect: QRect(dstPos, size), image: srcImage, sourceRect: QRect(srcPos, size));
436}

438void QRhiNull::simulateTextureGenMips(const QRhiResourceUpdateBatchPrivate::TextureOp &u)
439{
440 QNullTexture *texD = QRHI_RES(QNullTexture, u.dst);
441 const QSize baseSize = texD->pixelSize();
442 const int levelCount = q->mipLevelsForSize(size: baseSize);
443 for (int level = 1; level < levelCount; ++level)
444 texD->image[0][level] = texD->image[0][0].scaled(s: q->sizeForMipLevel(mipLevel: level, baseLevelSize: baseSize));
445}

447void QRhiNull::resourceUpdate(QRhiCommandBuffer *cb, QRhiResourceUpdateBatch *resourceUpdates)
448{
449 Q_UNUSED(cb);
450 QRhiResourceUpdateBatchPrivate *ud = QRhiResourceUpdateBatchPrivate::get(b: resourceUpdates);
451 for (int opIdx = 0; opIdx < ud->activeBufferOpCount; ++opIdx) {
452 const QRhiResourceUpdateBatchPrivate::BufferOp &u(ud->bufferOps[opIdx]);
453 if (u.type == QRhiResourceUpdateBatchPrivate::BufferOp::DynamicUpdate
454 || u.type == QRhiResourceUpdateBatchPrivate::BufferOp::StaticUpload)
455 {
456 QNullBuffer *bufD = QRHI_RES(QNullBuffer, u.buf);
457 memcpy(dest: bufD->data + u.offset, src: u.data.constData(), n: size_t(u.data.size()));
458 } else if (u.type == QRhiResourceUpdateBatchPrivate::BufferOp::Read) {
459 QRhiBufferReadbackResult *result = u.result;
460 result->data.resize(size: u.readSize);
461 QNullBuffer *bufD = QRHI_RES(QNullBuffer, u.buf);
462 memcpy(dest: result->data.data(), src: bufD->data + u.offset, n: size_t(u.readSize));
463 if (result->completed)
464 result->completed();
465 }
466 }
467 for (int opIdx = 0; opIdx < ud->activeTextureOpCount; ++opIdx) {
468 const QRhiResourceUpdateBatchPrivate::TextureOp &u(ud->textureOps[opIdx]);
469 if (u.type == QRhiResourceUpdateBatchPrivate::TextureOp::Upload) {
470 if (u.dst->format() == QRhiTexture::RGBA8)
471 simulateTextureUpload(u);
472 } else if (u.type == QRhiResourceUpdateBatchPrivate::TextureOp::Copy) {
473 if (u.src->format() == QRhiTexture::RGBA8 && u.dst->format() == QRhiTexture::RGBA8)
474 simulateTextureCopy(u);
475 } else if (u.type == QRhiResourceUpdateBatchPrivate::TextureOp::Read) {
476 QRhiReadbackResult *result = u.result;
477 QNullTexture *texD = QRHI_RES(QNullTexture, u.rb.texture());
478 if (texD) {
479 result->format = texD->format();
480 result->pixelSize = q->sizeForMipLevel(mipLevel: u.rb.level(), baseLevelSize: texD->pixelSize());
481 } else {
482 Q_ASSERT(currentSwapChain);
483 result->format = QRhiTexture::RGBA8;
484 result->pixelSize = currentSwapChain->currentPixelSize();
485 }
486 quint32 bytesPerLine = 0;
487 quint32 byteSize = 0;
488 textureFormatInfo(format: result->format, size: result->pixelSize, bpl: &bytesPerLine, byteSize: &byteSize, bytesPerPixel: nullptr);
489 if (texD && texD->format() == QRhiTexture::RGBA8) {
490 result->data.resize(size: int(byteSize));
491 const QImage &src(texD->image[u.rb.layer()][u.rb.level()]);
492 char *dst = result->data.data();
493 for (int y = 0, h = src.height(); y < h; ++y) {
494 memcpy(dest: dst, src: src.constScanLine(y), n: bytesPerLine);
495 dst += bytesPerLine;
496 }
497 } else {
498 result->data.fill(c: 0, size: int(byteSize));
499 }
500 if (result->completed)
501 result->completed();
502 } else if (u.type == QRhiResourceUpdateBatchPrivate::TextureOp::GenMips) {
503 if (u.dst->format() == QRhiTexture::RGBA8)
504 simulateTextureGenMips(u);
505 }
506 }
507 ud->free();
508}

651
652bool QNullTexture::create()
653{
654 if (valid)
655 destroy();
656
657 valid = true;
658
659 QRHI_RES_RHI(QRhiNull);
660 const bool isCube = m_flags.testFlag(flag: CubeMap);
661 const bool is3D = m_flags.testFlag(flag: ThreeDimensional);
662 const bool isArray = m_flags.testFlag(flag: TextureArray);
663 const bool hasMipMaps = m_flags.testFlag(flag: MipMapped);
664 const bool is1D = m_flags.testFlags(flags: OneDimensional);
665 QSize size = is1D ? QSize(qMax(a: 1, b: m_pixelSize.width()), 1)
666 : (m_pixelSize.isEmpty() ? QSize(1, 1) : m_pixelSize);
667 m_depth = qMax(a: 1, b: m_depth);
668 const int mipLevelCount = hasMipMaps ? rhiD->q->mipLevelsForSize(size) : 1;
669 m_arraySize = qMax(a: 0, b: m_arraySize);
670 const int layerCount = is3D ? m_depth : (isCube ? 6 : (isArray ? m_arraySize : 1));
671
672 if (m_format == RGBA8) {
673 image.resize(sz: layerCount);
674 for (int layer = 0; layer < layerCount; ++layer) {
675 for (int level = 0; level < mipLevelCount; ++level) {
676 image[layer][level] = QImage(rhiD->q->sizeForMipLevel(mipLevel: level, baseLevelSize: size),
677 QImage::Format_RGBA8888_Premultiplied);
678 image[layer][level].fill(color: Qt::yellow);
679 }
680 }
681 }
682
683 generation += 1;
684
685 rhiD->registerResource(res: this);
686
687 return true;
688}

808bool QNullTextureRenderTarget::create()
809{
810 QRHI_RES_RHI(QRhiNull);
811 d.rp = QRHI_RES(QNullRenderPassDescriptor, m_renderPassDesc);
812 if (m_desc.cbeginColorAttachments() != m_desc.cendColorAttachments()) {
813 const QRhiColorAttachment *colorAtt = m_desc.cbeginColorAttachments();
814 QRhiTexture *tex = colorAtt->texture();
815 QRhiRenderBuffer *rb = colorAtt->renderBuffer();
816 d.pixelSize = tex ? rhiD->q->sizeForMipLevel(mipLevel: colorAtt->level(), baseLevelSize: tex->pixelSize()) : rb->pixelSize();
817 } else if (m_desc.depthStencilBuffer()) {
818 d.pixelSize = m_desc.depthStencilBuffer()->pixelSize();
819 } else if (m_desc.depthTexture()) {
820 d.pixelSize = m_desc.depthTexture()->pixelSize();
821 }
822 QRhiRenderTargetAttachmentTracker::updateResIdList<QNullTexture, QNullRenderBuffer>(desc: m_desc, dst: &d.currentResIdList);
823 return true;
824}

975bool QNullSwapChain::createOrResize()
976{
977 const bool needsRegistration = !window || window != m_window;
978 if (window && window != m_window)
979 destroy();
980
981 window = m_window;
982 m_currentPixelSize = surfacePixelSize();
983 rt.setRenderPassDescriptor(m_renderPassDesc); // for the public getter in QRhiRenderTarget
984 rt.d.rp = QRHI_RES(QNullRenderPassDescriptor, m_renderPassDesc);
985 rt.d.pixelSize = m_currentPixelSize;
986 frameCount = 0;
987
988 if (needsRegistration) {
989 QRHI_RES_RHI(QRhiNull);
990 rhiD->registerResource(res: this);
991 }
992
993 return true;
994}
