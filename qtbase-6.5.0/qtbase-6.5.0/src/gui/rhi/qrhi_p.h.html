399 {
400 // must write out exactly LAYOUT_DESC_ENTRIES_PER_BINDING elements here
401 *dst++ = quint32(binding);
402 *dst++ = quint32(stage);
403 *dst++ = quint32(type);
404 *dst++ = quint32(arraySize());
405 return dst;
