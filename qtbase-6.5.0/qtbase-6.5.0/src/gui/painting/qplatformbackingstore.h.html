41struct Q_GUI_EXPORT QPlatformBackingStoreRhiConfig
42{
43 enum Api {
44 OpenGL,
45 Metal,
46 Vulkan,
47 D3D11,
48 Null
49 };
50
51 QPlatformBackingStoreRhiConfig()
52 : m_enable(false)
53 { }
54
55 QPlatformBackingStoreRhiConfig(Api api)
56 : m_enable(true),
57 m_api(api)
58 { }
59
60 bool isEnabled() const { return m_enable; }
61 void setEnabled(bool enable) { m_enable = enable; }
62
63 Api api() const { return m_api; }
64 void setApi(Api api) { m_api = api; }
65
66 bool isDebugLayerEnabled() const { return m_debugLayer; }
67 void setDebugLayer(bool enable) { m_debugLayer = enable; }
68
69private:
70 bool m_enable;
71 Api m_api = Null;
72 bool m_debugLayer = false;
73 friend bool operator==(const QPlatformBackingStoreRhiConfig &a, const QPlatformBackingStoreRhiConfig &b);
74};
75
76inline bool operator==(const QPlatformBackingStoreRhiConfig &a, const QPlatformBackingStoreRhiConfig &b)
77{
78 return a.m_enable == b.m_enable
79 && a.m_api == b.m_api
80 && a.m_debugLayer == b.m_debugLayer;
81}
82
83inline bool operator!=(const QPlatformBackingStoreRhiConfig &a, const QPlatformBackingStoreRhiConfig &b)
84{
85 return !(a == b);
86}
