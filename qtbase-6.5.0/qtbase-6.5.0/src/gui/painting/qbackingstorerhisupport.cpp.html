44void QBackingStoreRhiSupport::reset()
45{
46 for (SwapchainData &d : m_swapchains)
47 d.reset();
48
49 m_swapchains.clear();
50
51 delete m_rhi;
52 m_rhi = nullptr;
53
54 delete m_openGLFallbackSurface;
55 m_openGLFallbackSurface = nullptr;
56}

58bool QBackingStoreRhiSupport::create()
59{
60 if (!QGuiApplicationPrivate::platformIntegration()->hasCapability(cap: QPlatformIntegration::RhiBasedRendering))
61 return false;
62
63 // note: m_window may be null (special case for fully offscreen rendering)
64
65 QRhi *rhi = nullptr;
66 QOffscreenSurface *surface = nullptr;
67 QRhi::Flags flags;
68
69#if QT_CONFIG(opengl)
70 if (!rhi && m_config.api() == QPlatformBackingStoreRhiConfig::OpenGL) {
71 surface = QRhiGles2InitParams::newFallbackSurface(format: m_format);
72 QRhiGles2InitParams params;
73 params.fallbackSurface = surface;
74 params.window = m_window;
75 params.format = m_format;
76 params.shareContext = qt_gl_global_share_context();
77 rhi = QRhi::create(impl: QRhi::OpenGLES2, params: &params, flags);
78 }
79#endif
80
81#ifdef Q_OS_WIN
82 if (!rhi && m_config.api() == QPlatformBackingStoreRhiConfig::D3D11) {
83 QRhiD3D11InitParams params;
84 params.enableDebugLayer = m_config.isDebugLayerEnabled();
85 rhi = QRhi::create(QRhi::D3D11, &params, flags);
86 }
87#endif
88
89#if defined(Q_OS_MACOS) || defined(Q_OS_IOS)
90 if (!rhi && m_config.api() == QPlatformBackingStoreRhiConfig::Metal) {
91 QRhiMetalInitParams params;
92 // For parity with Qt Quick, fall back to OpenGL when there is no Metal (f.ex. in macOS virtual machines).
93 if (QRhi::probe(QRhi::Metal, &params)) {
94 rhi = QRhi::create(QRhi::Metal, &params, flags);
95 } else {
96 qCDebug(lcQpaBackingStore, "Metal does not seem to be supported. Falling back to OpenGL.");
97 rhi = QRhi::create(QRhi::OpenGLES2, &params, flags);
98 }
99 }
100#endif
101
102#if QT_CONFIG(vulkan)
103 if (!rhi && m_config.api() == QPlatformBackingStoreRhiConfig::Vulkan) {
104 if (m_config.isDebugLayerEnabled())
105 QVulkanDefaultInstance::setFlag(QVulkanDefaultInstance::EnableValidation);
106 QRhiVulkanInitParams params;
107 if (m_window) {
108 if (!m_window->vulkanInstance())
109 m_window->setVulkanInstance(QVulkanDefaultInstance::instance());
110 params.inst = m_window->vulkanInstance();
111 } else {
112 params.inst = QVulkanDefaultInstance::instance();
113 }
114 if (!params.inst) {
115 qWarning("No QVulkanInstance set for the top-level window, this is wrong.");
116 return false;
117 }
118 params.window = m_window;
119 rhi = QRhi::create(QRhi::Vulkan, &params, flags);
120 }
121#endif
122
123 if (!rhi) {
124 qWarning(msg: "Failed to create QRhi for QBackingStoreRhiSupport");
125 delete surface;
126 return false;
127 }
128
129 m_rhi = rhi;
130 m_openGLFallbackSurface = surface;
131 return true;
132}

134QRhiSwapChain *QBackingStoreRhiSupport::swapChainForWindow(QWindow *window)
135{
136 auto it = m_swapchains.constFind(key: window);
137 if (it != m_swapchains.constEnd())
138 return it.value().swapchain;
139
140 QRhiSwapChain *swapchain = nullptr;
141 QRhiRenderPassDescriptor *rp = nullptr;
142 if (window && m_rhi) {
143 QRhiSwapChain::Flags flags;
144 const QSurfaceFormat format = window->requestedFormat();
145 if (format.swapInterval() == 0)
146 flags |= QRhiSwapChain::NoVSync;
147 if (format.alphaBufferSize() > 0)
148 flags |= QRhiSwapChain::SurfaceHasNonPreMulAlpha;
149#if QT_CONFIG(vulkan)
150 if (m_config.api() == QPlatformBackingStoreRhiConfig::Vulkan && !window->vulkanInstance())
151 window->setVulkanInstance(QVulkanDefaultInstance::instance());
152#endif
153 qCDebug(lcQpaBackingStore) << "Creating swapchain for window" << window;
154 swapchain = m_rhi->newSwapChain();
155 swapchain->setWindow(window);
156 swapchain->setFlags(flags);
157 rp = swapchain->newCompatibleRenderPassDescriptor();
158 swapchain->setRenderPassDescriptor(rp);
159 if (!swapchain->createOrResize()) {
160 qWarning(msg: "Failed to create swapchain for window flushed with an RHI-enabled backingstore");
161 delete rp;
162 return nullptr;
163 }
164 }
165 if (swapchain) {
166 SwapchainData d;
167 d.swapchain = swapchain;
168 d.renderPassDescriptor = rp;
169 d.windowWatcher = new QBackingStoreRhiSupportWindowWatcher(this);
170 m_swapchains.insert(key: window, value: d);
171 window->installEventFilter(filterObj: d.windowWatcher);
172 }
173 return swapchain;
174}

176bool QBackingStoreRhiSupportWindowWatcher::eventFilter(QObject *obj, QEvent *event)
177{
178 if (event->type() == QEvent::PlatformSurface
179 && static_cast<QPlatformSurfaceEvent *>(event)->surfaceEventType() == QPlatformSurfaceEvent::SurfaceAboutToBeDestroyed)
180 {
181 QWindow *window = qobject_cast<QWindow *>(o: obj);
182 auto it = m_rhiSupport->m_swapchains.find(key: window);
183 if (it != m_rhiSupport->m_swapchains.end()) {
184 qCDebug(lcQpaBackingStore) << "SurfaceAboutToBeDestroyed received for tracked window" << window << "cleaning up swapchain";
185 auto data = *it;
186 m_rhiSupport->m_swapchains.erase(it);
187 data.reset(); // deletes 'this'
188 }
189 }
190 return false;
191}

193QSurface::SurfaceType QBackingStoreRhiSupport::surfaceTypeForConfig(const QPlatformBackingStoreRhiConfig &config)
194{
195 QSurface::SurfaceType type = QSurface::RasterSurface;
196 switch (config.api()) {
197 case QPlatformBackingStoreRhiConfig::D3D11:
198 type = QSurface::Direct3DSurface;
199 break;
200 case QPlatformBackingStoreRhiConfig::Vulkan:
201 type = QSurface::VulkanSurface;
202 break;
203 case QPlatformBackingStoreRhiConfig::Metal:
204 type = QSurface::MetalSurface;
205 break;
206 case QPlatformBackingStoreRhiConfig::OpenGL:
207 type = QSurface::OpenGLSurface;
208 break;
209 default:
210 break;
211 }
212 return type;
213}

215QRhi::Implementation QBackingStoreRhiSupport::apiToRhiBackend(QPlatformBackingStoreRhiConfig::Api api)
216{
217 switch (api) {
218 case QPlatformBackingStoreRhiConfig::OpenGL:
219 return QRhi::OpenGLES2;
220 case QPlatformBackingStoreRhiConfig::Metal:
221 return QRhi::Metal;
222 case QPlatformBackingStoreRhiConfig::Vulkan:
223 return QRhi::Vulkan;
224 case QPlatformBackingStoreRhiConfig::D3D11:
225 return QRhi::D3D11;
226 case QPlatformBackingStoreRhiConfig::Null:
227 return QRhi::Null;
228 default:
229 break;
230 }
231 return QRhi::Null;
232}

234bool QBackingStoreRhiSupport::checkForceRhi(QPlatformBackingStoreRhiConfig *outConfig, QSurface::SurfaceType *outType)
235{
236 static QPlatformBackingStoreRhiConfig config;
237 static bool checked = false;
238
239 if (!checked) {
240 checked = true;
241
242 const bool alwaysRhi = qEnvironmentVariableIntValue(varName: "QT_WIDGETS_RHI");
243 if (alwaysRhi)
244 config.setEnabled(true);
245
246 // if enabled, choose an api
247 if (config.isEnabled()) {
248#if defined(Q_OS_WIN)
249 config.setApi(QPlatformBackingStoreRhiConfig::D3D11);
250#elif defined(Q_OS_MACOS) || defined(Q_OS_IOS)
251 config.setApi(QPlatformBackingStoreRhiConfig::Metal);
252#elif QT_CONFIG(opengl)
253 config.setApi(QPlatformBackingStoreRhiConfig::OpenGL);
254#elif QT_CONFIG(vulkan)
255 config.setApi(QPlatformBackingStoreRhiConfig::Vulkan);
256#else
257 qWarning("QT_WIDGETS_RHI is set but no backend is available; ignoring");
258 return false;
259#endif
260
261 // the env.var. will always override
262 if (qEnvironmentVariableIsSet(varName: "QT_WIDGETS_RHI_BACKEND")) {
263 const QString backend = qEnvironmentVariable(varName: "QT_WIDGETS_RHI_BACKEND");
264#ifdef Q_OS_WIN
265 if (backend == QStringLiteral("d3d11") || backend == QStringLiteral("d3d"))
266 config.setApi(QPlatformBackingStoreRhiConfig::D3D11);
267#endif
268#if defined(Q_OS_MACOS) || defined(Q_OS_IOS)
269 if (backend == QStringLiteral("metal"))
270 config.setApi(QPlatformBackingStoreRhiConfig::Metal);
271#endif
272#if QT_CONFIG(opengl)
273 if (backend == QStringLiteral("opengl") || backend == QStringLiteral("gl"))
274 config.setApi(QPlatformBackingStoreRhiConfig::OpenGL);
275#endif
276#if QT_CONFIG(vulkan)
277 if (backend == QStringLiteral("vulkan"))
278 config.setApi(QPlatformBackingStoreRhiConfig::Vulkan);
279#endif
280 }
281
282 if (qEnvironmentVariableIntValue(varName: "QT_WIDGETS_RHI_DEBUG_LAYER"))
283 config.setDebugLayer(true);
284 }
285
286 qCDebug(lcQpaBackingStore) << "Check for forced use of QRhi resulted in enable"
287 << config.isEnabled() << "with api" << QRhi::backendName(impl: apiToRhiBackend(api: config.api()));
288 }
289
290 if (config.isEnabled()) {
291 if (outConfig)
292 *outConfig = config;
293 if (outType)
294 *outType = surfaceTypeForConfig(config);
295 return true;
296 }
297 return false;
298}
