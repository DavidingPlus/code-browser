25struct QFixed {
26private:
27 constexpr QFixed(int val, int) : val(val) {} // 2nd int is just a dummy for disambiguation
28public:
29 constexpr QFixed() : val(0) {}
30 constexpr QFixed(int i) : val(i * 64) {}
31 constexpr QFixed(long i) : val(i * 64) {}
32 constexpr QFixed(long long i) : val(i * 64) {}
33
34 constexpr static QFixed fromReal(qreal r) { return fromFixed(fixed: (int)(r*qreal(64))); }
35 constexpr static QFixed fromFixed(int fixed) { return QFixed(fixed,0); } // uses private ctor
36
37 constexpr inline int value() const { return val; }
38 inline void setValue(int value) { val = value; }
39
40 constexpr inline int toInt() const { return (((val)+32) & -64)>>6; }
41 constexpr inline qreal toReal() const { return ((qreal)val)/(qreal)64; }
42
43 constexpr inline int truncate() const { return val>>6; }
44 constexpr inline QFixed round() const { return fromFixed(fixed: ((val)+32) & -64); }
45 constexpr inline QFixed floor() const { return fromFixed(fixed: (val) & -64); }
46 constexpr inline QFixed ceil() const { return fromFixed(fixed: (val+63) & -64); }
47
48 constexpr inline QFixed operator+(int i) const { return fromFixed(fixed: val + i * 64); }
49 constexpr inline QFixed operator+(uint i) const { return fromFixed(fixed: (val + (i<<6))); }
50 constexpr inline QFixed operator+(QFixed other) const { return fromFixed(fixed: (val + other.val)); }
51 inline QFixed &operator+=(int i) { val += i * 64; return *this; }
52 inline QFixed &operator+=(uint i) { val += (i<<6); return *this; }
53 inline QFixed &operator+=(QFixed other) { val += other.val; return *this; }
54 constexpr inline QFixed operator-(int i) const { return fromFixed(fixed: val - i * 64); }
55 constexpr inline QFixed operator-(uint i) const { return fromFixed(fixed: (val - (i<<6))); }
56 constexpr inline QFixed operator-(QFixed other) const { return fromFixed(fixed: (val - other.val)); }
57 inline QFixed &operator-=(int i) { val -= i * 64; return *this; }
58 inline QFixed &operator-=(uint i) { val -= (i<<6); return *this; }
59 inline QFixed &operator-=(QFixed other) { val -= other.val; return *this; }
60 constexpr inline QFixed operator-() const { return fromFixed(fixed: -val); }
61
62#define REL_OP(op) \
63 friend constexpr bool operator op(QFixed lhs, QFixed rhs) noexcept \
64 { return lhs.val op rhs.val; }
65 REL_OP(==)
66 REL_OP(!=)
67 REL_OP(< )

73 constexpr inline bool operator!() const { return !val; }
74
75 inline QFixed &operator/=(int x) { val /= x; return *this; }
76 inline QFixed &operator/=(QFixed o) {
77 if (o.val == 0) {
78 val = 0x7FFFFFFFL;
79 } else {
80 bool neg = false;
81 qint64 a = val;
82 qint64 b = o.val;
83 if (a < 0) { a = -a; neg = true; }
84 if (b < 0) { b = -b; neg = !neg; }
85
86 int res = (int)(((a << 6) + (b >> 1)) / b);
87
88 val = (neg ? -res : res);
89 }
90 return *this;
91 }
92 constexpr inline QFixed operator/(int d) const { return fromFixed(fixed: val/d); }
93 inline QFixed operator/(QFixed b) const { QFixed f = *this; return (f /= b); }
94 inline QFixed operator>>(int d) const { QFixed f = *this; f.val >>= d; return f; }
95 inline QFixed &operator*=(int i) { val *= i; return *this; }
96 inline QFixed &operator*=(uint i) { val *= i; return *this; }
97 inline QFixed &operator*=(QFixed o) {
98 bool neg = false;
99 qint64 a = val;
100 qint64 b = o.val;
101 if (a < 0) { a = -a; neg = true; }
102 if (b < 0) { b = -b; neg = !neg; }
103
104 int res = (int)((a * b + 0x20L) >> 6);
105 val = neg ? -res : res;
106 return *this;
107 }
108 constexpr inline QFixed operator*(int i) const { return fromFixed(fixed: val * i); }
109 constexpr inline QFixed operator*(uint i) const { return fromFixed(fixed: val * i); }
110 inline QFixed operator*(QFixed o) const { QFixed f = *this; return (f *= o); }
111
112private:
113 constexpr QFixed(qreal i) : val((int)(i*qreal(64))) {}
114 constexpr inline QFixed operator+(qreal i) const { return fromFixed(fixed: (val + (int)(i*qreal(64)))); }
115 inline QFixed &operator+=(qreal i) { val += (int)(i*64); return *this; }
116 constexpr inline QFixed operator-(qreal i) const { return fromFixed(fixed: (val - (int)(i*qreal(64)))); }
117 inline QFixed &operator-=(qreal i) { val -= (int)(i*64); return *this; }
118 inline QFixed &operator/=(qreal r) { val = (int)(val/r); return *this; }
119 constexpr inline QFixed operator/(qreal d) const { return fromFixed(fixed: (int)(val/d)); }
120 inline QFixed &operator*=(qreal d) { val = (int) (val*d); return *this; }
121 constexpr inline QFixed operator*(qreal d) const { return fromFixed(fixed: (int) (val*d)); }
122 int val;
123};

144struct QFixedPoint {
145 QFixed x;
146 QFixed y;
147 constexpr inline QFixedPoint() {}
148 constexpr inline QFixedPoint(QFixed _x, QFixed _y) : x(_x), y(_y) {}
149 constexpr QPointF toPointF() const { return QPointF(x.toReal(), y.toReal()); }
150 constexpr static QFixedPoint fromPointF(const QPointF &p) {
151 return QFixedPoint(QFixed::fromReal(r: p.x()), QFixed::fromReal(r: p.y()));
152 }
153 constexpr inline bool operator==(const QFixedPoint &other) const
154 {
155 return x == other.x && y == other.y;
156 }
157};
158Q_DECLARE_TYPEINFO(QFixedPoint, Q_PRIMITIVE_TYPE);
