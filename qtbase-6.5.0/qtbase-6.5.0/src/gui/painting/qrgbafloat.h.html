4#ifndef QRGBAFLOAT_H
5#define QRGBAFLOAT_H
6
7#include <QtGui/qtguiglobal.h>
8#include <QtCore/qfloat16.h>
9
10#include <algorithm>
11#include <cmath>
12#include <type_traits>
13
14QT_BEGIN_NAMESPACE
15
16template<typename F>
17class alignas(sizeof(F) * 4) QRgbaFloat
18{
19 static_assert(std::is_same<F, qfloat16>::value || std::is_same<F, float>::value);
20public:
21 using Type = F;
22 using FastType = float;
23 F r;
24 F g;
25 F b;
26 F a;
27
28 static constexpr
29 QRgbaFloat fromRgba64(quint16 red, quint16 green, quint16 blue, quint16 alpha)
30 {
31 constexpr FastType scale = 1.0f / 65535.0f;
32 return QRgbaFloat{
33 F(red * scale),
34 F(green * scale),
35 F(blue * scale),
36 F(alpha * scale) };
37 }
38
39 static constexpr
40 QRgbaFloat fromRgba(quint8 red, quint8 green, quint8 blue, quint8 alpha)
41 {
42 constexpr FastType scale = 1.0f / 255.0f;
43 return QRgbaFloat{
44 F(red * scale),
45 F(green * scale),
46 F(blue * scale),
47 F(alpha * scale) };
48 }
49 static constexpr
50 QRgbaFloat fromArgb32(uint rgb)
51 {
52 return fromRgba(red: quint8(rgb >> 16), green: quint8(rgb >> 8), blue: quint8(rgb), alpha: quint8(rgb >> 24));
53 }
54
55 constexpr bool isOpaque() const { return a >= 1.0f; }
56 constexpr bool isTransparent() const { return a <= 0.0f; }
57
58 constexpr FastType red() const { return r; }
59 constexpr FastType green() const { return g; }
60 constexpr FastType blue() const { return b; }
61 constexpr FastType alpha() const { return a; }
62 void setRed(FastType _red) { r = F(_red); }
63 void setGreen(FastType _green) { g = F(_green); }
64 void setBlue(FastType _blue) { b = F(_blue); }
65 void setAlpha(FastType _alpha) { a = F(_alpha); }
66
67 constexpr FastType redNormalized() const { return std::clamp(val: static_cast<FastType>(r), lo: 0.0f, hi: 1.0f); }
68 constexpr FastType greenNormalized() const { return std::clamp(val: static_cast<FastType>(g), lo: 0.0f, hi: 1.0f); }
69 constexpr FastType blueNormalized() const { return std::clamp(val: static_cast<FastType>(b), lo: 0.0f, hi: 1.0f); }
70 constexpr FastType alphaNormalized() const { return std::clamp(val: static_cast<FastType>(a), lo: 0.0f, hi: 1.0f); }
71
72 constexpr quint8 red8() const { return std::lround(redNormalized() * 255.0f); }
73 constexpr quint8 green8() const { return std::lround(greenNormalized() * 255.0f); }
74 constexpr quint8 blue8() const { return std::lround(blueNormalized() * 255.0f); }
75 constexpr quint8 alpha8() const { return std::lround(alphaNormalized() * 255.0f); }
76 constexpr uint toArgb32() const
77 {
78 return uint((alpha8() << 24) | (red8() << 16) | (green8() << 8) | blue8());
79 }
80
81 constexpr quint16 red16() const { return std::lround(redNormalized() * 65535.0f); }
82 constexpr quint16 green16() const { return std::lround(greenNormalized() * 65535.0f); }
83 constexpr quint16 blue16() const { return std::lround(blueNormalized() * 65535.0f); }
84 constexpr quint16 alpha16() const { return std::lround(alphaNormalized() * 65535.0f); }
85
86 constexpr Q_ALWAYS_INLINE QRgbaFloat premultiplied() const
87 {
88 return QRgbaFloat{r * a, g * a, b * a, a};
89 }
90 constexpr Q_ALWAYS_INLINE QRgbaFloat unpremultiplied() const
91 {
92 if (a <= F{0.0f})
93 return QRgbaFloat{}; // default-initialization: zeroes
94 if (a >= F{1.0f})
95 return *this;
96 const FastType ia = 1.0f / a;
97 return QRgbaFloat{F(r * ia), F(g * ia), F(b * ia), F(a)};
98 }
99 constexpr bool operator==(QRgbaFloat f) const
100 {
101 return r == f.r && g == f.g && b == f.b && a == f.a;
102 }
103 constexpr bool operator!=(QRgbaFloat f) const
104 {
105 return !(*this == f);
106 }
107};
108
109typedef QRgbaFloat<qfloat16> QRgbaFloat16;
110typedef QRgbaFloat<float> QRgbaFloat32;
111
112QT_END_NAMESPACE
113
114#endif // QRGBAFLOAT_H
