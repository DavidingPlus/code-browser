18void QBackingStoreDefaultCompositor::reset()
19{
20 delete m_psNoBlend;
21 m_psNoBlend = nullptr;
22 delete m_psBlend;
23 m_psBlend = nullptr;
24 delete m_psPremulBlend;
25 m_psPremulBlend = nullptr;
26 delete m_sampler;
27 m_sampler = nullptr;
28 delete m_vbuf;

30 delete m_texture;
31 m_texture = nullptr;

34 d.reset();
35}

46QRhiTexture *QBackingStoreDefaultCompositor::toTexture(const QImage &sourceImage,
47 QRhi *rhi,
48 QRhiResourceUpdateBatch *resourceUpdates,
49 const QRegion &dirtyRegion,
50 QPlatformBackingStore::TextureFlags *flags) const
51{
52 Q_ASSERT(rhi);
53 Q_ASSERT(resourceUpdates);
54 Q_ASSERT(flags);
55
56 if (!m_rhi) {
57 m_rhi = rhi;
58 } else if (m_rhi != rhi) {
59 qWarning(msg: "QBackingStoreDefaultCompositor: the QRhi has changed unexpectedly, this should not happen");
60 return nullptr;
61 }
62
63 QImage image = sourceImage;
64
65 bool needsConversion = false;
66 *flags = {};
67
68 switch (image.format()) {
69 case QImage::Format_ARGB32_Premultiplied:
70 *flags |= QPlatformBackingStore::TexturePremultiplied;
71 Q_FALLTHROUGH();
72 case QImage::Format_RGB32:
73 case QImage::Format_ARGB32:
74 *flags |= QPlatformBackingStore::TextureSwizzle;
75 break;
76 case QImage::Format_RGBA8888_Premultiplied:
77 *flags |= QPlatformBackingStore::TexturePremultiplied;
78 Q_FALLTHROUGH();
79 case QImage::Format_RGBX8888:
80 case QImage::Format_RGBA8888:
81 break;
82 case QImage::Format_BGR30:
83 case QImage::Format_A2BGR30_Premultiplied:
84 // no fast path atm
85 needsConversion = true;
86 break;
87 case QImage::Format_RGB30:
88 case QImage::Format_A2RGB30_Premultiplied:
89 // no fast path atm
90 needsConversion = true;
91 break;
92 default:
93 needsConversion = true;
94 break;
95 }
96
97 if (image.size().isEmpty())
98 return nullptr;
99
100 const bool resized = !m_texture || m_texture->pixelSize() != image.size();
101 if (dirtyRegion.isEmpty() && !resized)
102 return m_texture;
103
104 if (needsConversion)
105 image = image.convertToFormat(f: QImage::Format_RGBA8888);
106 else
107 image.detach(); // if it was just wrapping data, that's no good, we need ownership, so detach
108
109 if (resized) {
110 if (!m_texture)
111 m_texture = rhi->newTexture(format: QRhiTexture::RGBA8, pixelSize: image.size());
112 else
113 m_texture->setPixelSize(image.size());
114 m_texture->create();
115 resourceUpdates->uploadTexture(tex: m_texture, image);
116 } else {
117 QRect imageRect = image.rect();
118 QRect rect = dirtyRegion.boundingRect() & imageRect;
119 QRhiTextureSubresourceUploadDescription subresDesc(image);
120 subresDesc.setSourceTopLeft(rect.topLeft());
121 subresDesc.setSourceSize(rect.size());
122 subresDesc.setDestinationTopLeft(rect.topLeft());
123 QRhiTextureUploadDescription uploadDesc(QRhiTextureUploadEntry(0, 0, subresDesc));
124 resourceUpdates->uploadTexture(tex: m_texture, desc: uploadDesc);
125 }
126
127 return m_texture;
128}

135static inline QPoint scaledOffset(const QPoint &pt, qreal factor)

139
140static QRegion scaledRegion(const QRegion &region, qreal factor, const QPoint &offset)
141{
142 if (offset.isNull() && factor <= 1)
143 return region;
144
145 QVarLengthArray<QRect, 4> rects;
146 rects.reserve(sz: region.rectCount());
147 for (const QRect &rect : region)
148 rects.append(t: scaledRect(rect: rect.translated(p: offset), factor));
149
150 QRegion deviceRegion;
151 deviceRegion.setRects(rect: rects.constData(), num: rects.size());
152 return deviceRegion;
153}

155static QMatrix4x4 targetTransform(const QRectF &target, const QRect &viewport, bool invertY)
156{
157 qreal x_scale = target.width() / viewport.width();
158 qreal y_scale = target.height() / viewport.height();
159
160 const QPointF relative_to_viewport = target.topLeft() - viewport.topLeft();
161 qreal x_translate = x_scale - 1 + ((relative_to_viewport.x() / viewport.width()) * 2);
162 qreal y_translate;
163 if (invertY)
164 y_translate = y_scale - 1 + ((relative_to_viewport.y() / viewport.height()) * 2);
165 else
166 y_translate = -y_scale + 1 - ((relative_to_viewport.y() / viewport.height()) * 2);
167
168 QMatrix4x4 matrix;
169 matrix(0,3) = x_translate;
170 matrix(1,3) = y_translate;
171
172 matrix(0,0) = x_scale;
173 matrix(1,1) = y_scale;
174
175 return matrix;
176}

178enum class SourceTransformOrigin {
179 BottomLeft,
180 TopLeft
181};
182
183static QMatrix3x3 sourceTransform(const QRectF &subTexture,
184 const QSize &textureSize,
185 SourceTransformOrigin origin)
186{
187 qreal x_scale = subTexture.width() / textureSize.width();
188 qreal y_scale = subTexture.height() / textureSize.height();
189
190 const QPointF topLeft = subTexture.topLeft();
191 qreal x_translate = topLeft.x() / textureSize.width();
192 qreal y_translate = topLeft.y() / textureSize.height();
193
194 if (origin == SourceTransformOrigin::TopLeft) {
195 y_scale = -y_scale;
196 y_translate = 1 - y_translate;
197 }
198
199 QMatrix3x3 matrix;
200 matrix(0,2) = x_translate;
201 matrix(1,2) = y_translate;
202
203 matrix(0,0) = x_scale;
204 matrix(1,1) = y_scale;
205
206 return matrix;
207}

209static inline QRect toBottomLeftRect(const QRect &topLeftRect, int windowHeight)
210{

215static bool prepareDrawForRenderToTextureWidget(const QPlatformTextureList *textures,
216 int idx,
217 QWindow *window,
218 const QRect &deviceWindowRect,
219 const QPoint &offset,
220 bool invertTargetY,
221 bool invertSource,
222 QMatrix4x4 *target,
223 QMatrix3x3 *source)
224{
225 const QRect clipRect = textures->clipRect(index: idx);
226 if (clipRect.isEmpty())
227 return false;
228
229 QRect rectInWindow = textures->geometry(index: idx);
230 // relative to the TLW, not necessarily our window (if the flush is for a native child widget), have to adjust
231 rectInWindow.translate(p: -offset);
232
233 const QRect clippedRectInWindow = rectInWindow & clipRect.translated(p: rectInWindow.topLeft());
234 const QRect srcRect = toBottomLeftRect(topLeftRect: clipRect, windowHeight: rectInWindow.height());
235
236 *target = targetTransform(target: scaledRect(rect: clippedRectInWindow, factor: window->devicePixelRatio()),
237 viewport: deviceWindowRect,
238 invertY: invertTargetY);
239
240 *source = sourceTransform(subTexture: scaledRect(rect: srcRect, factor: window->devicePixelRatio()),
241 textureSize: scaledRect(rect: rectInWindow, factor: window->devicePixelRatio()).size(),
242 origin: invertSource ? SourceTransformOrigin::TopLeft : SourceTransformOrigin::BottomLeft);
243
244 return true;
245}

247static QShader getShader(const QString &name)
248{
249 QFile f(name);
250 if (f.open(flags: QIODevice::ReadOnly))
251 return QShader::fromSerialized(data: f.readAll());
252

259static void updateMatrix3x3(QRhiResourceUpdateBatch *resourceUpdates, QRhiBuffer *ubuf, const QMatrix3x3 &m)
260{
261 // mat3 is still 4 floats per column in the uniform buffer (but there is no
262 // 4th column), so 48 bytes altogether, not 36 or 64.
263
264 float f[12];
265 const float *src = static_cast<const float *>(m.constData());
266 float *dst = f;
267 memcpy(dest: dst, src: src, n: 3 * sizeof(float));
268 memcpy(dest: dst + 4, src: src + 3, n: 3 * sizeof(float));
269 memcpy(dest: dst + 8, src: src + 6, n: 3 * sizeof(float));
270
271 resourceUpdates->updateDynamicBuffer(buf: ubuf, offset: 64, size: 48, data: f);
272}

274enum class PipelineBlend {
275 None,
276 Alpha,
277 PremulAlpha
278};
279
280static QRhiGraphicsPipeline *createGraphicsPipeline(QRhi *rhi,
281 QRhiShaderResourceBindings *srb,
282 QRhiSwapChain *swapchain,
283 PipelineBlend blend)
284{
285 QRhiGraphicsPipeline *ps = rhi->newGraphicsPipeline();
286
287 switch (blend) {
288 case PipelineBlend::Alpha:
289 {
290 QRhiGraphicsPipeline::TargetBlend blend;
291 blend.enable = true;
292 blend.srcColor = QRhiGraphicsPipeline::SrcAlpha;
293 blend.dstColor = QRhiGraphicsPipeline::OneMinusSrcAlpha;
294 blend.srcAlpha = QRhiGraphicsPipeline::One;
295 blend.dstAlpha = QRhiGraphicsPipeline::One;
296 ps->setTargetBlends({ blend });
297 }
298 break;
299 case PipelineBlend::PremulAlpha:
300 {
301 QRhiGraphicsPipeline::TargetBlend blend;
302 blend.enable = true;
303 blend.srcColor = QRhiGraphicsPipeline::One;
304 blend.dstColor = QRhiGraphicsPipeline::OneMinusSrcAlpha;
305 blend.srcAlpha = QRhiGraphicsPipeline::One;
306 blend.dstAlpha = QRhiGraphicsPipeline::One;
307 ps->setTargetBlends({ blend });
308 }
309 break;
310 default:
311 break;
312 }
313
314 ps->setShaderStages({
315 { QRhiShaderStage::Vertex, getShader(name: ":/qt-project.org/gui/painting/shaders/backingstorecompose.vert.qsb"_L1) },
316 { QRhiShaderStage::Fragment, getShader(name: ":/qt-project.org/gui/painting/shaders/backingstorecompose.frag.qsb"_L1) }
317 });
318 QRhiVertexInputLayout inputLayout;
319 inputLayout.setBindings({ { 5 * sizeof(float) } });
320 inputLayout.setAttributes({
321 { 0, 0, QRhiVertexInputAttribute::Float3, 0 },
322 { 0, 1, QRhiVertexInputAttribute::Float2, quint32(3 * sizeof(float)) }
323 });
324 ps->setVertexInputLayout(inputLayout);
325 ps->setShaderResourceBindings(srb);
326 ps->setRenderPassDescriptor(swapchain->renderPassDescriptor());
327
328 if (!ps->create()) {
329 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to build graphics pipeline");
330 delete ps;
331 return nullptr;
332 }
333 return ps;
334}

338QBackingStoreDefaultCompositor::PerQuadData QBackingStoreDefaultCompositor::createPerQuadData(QRhiTexture *texture, QRhiTexture *textureExtra)
339{
340 PerQuadData d;
341
342 d.ubuf = m_rhi->newBuffer(type: QRhiBuffer::Dynamic, usage: QRhiBuffer::UniformBuffer, size: UBUF_SIZE);
343 if (!d.ubuf->create())
344 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to create uniform buffer");
345
346 d.srb = m_rhi->newShaderResourceBindings();
347 d.srb->setBindings({
348 QRhiShaderResourceBinding::uniformBuffer(binding: 0, stage: QRhiShaderResourceBinding::VertexStage | QRhiShaderResourceBinding::FragmentStage, buf: d.ubuf, offset: 0, size: UBUF_SIZE),
349 QRhiShaderResourceBinding::sampledTexture(binding: 1, stage: QRhiShaderResourceBinding::FragmentStage, tex: texture, sampler: m_sampler)
350 });
351 if (!d.srb->create())
352 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to create srb");
353 d.lastUsedTexture = texture;
354
355 if (textureExtra) {
356 d.srbExtra = m_rhi->newShaderResourceBindings();
357 d.srbExtra->setBindings({
358 QRhiShaderResourceBinding::uniformBuffer(binding: 0, stage: QRhiShaderResourceBinding::VertexStage | QRhiShaderResourceBinding::FragmentStage, buf: d.ubuf, offset: 0, size: UBUF_SIZE),
359 QRhiShaderResourceBinding::sampledTexture(binding: 1, stage: QRhiShaderResourceBinding::FragmentStage, tex: textureExtra, sampler: m_sampler)
360 });
361 if (!d.srbExtra->create())
362 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to create srb");
363 }
364
365 d.lastUsedTextureExtra = textureExtra;
366
367 return d;
368}

370void QBackingStoreDefaultCompositor::updatePerQuadData(PerQuadData *d, QRhiTexture *texture, QRhiTexture *textureExtra)
371{
372 // This whole check-if-texture-ptr-is-different is needed because there is
373 // nothing saying a QPlatformTextureList cannot return a different
374 // QRhiTexture* from the same index in a subsequent flush.
375
376 if (d->lastUsedTexture == texture || !d->srb)
377 return;
378
379 d->srb->setBindings({
380 QRhiShaderResourceBinding::uniformBuffer(binding: 0, stage: QRhiShaderResourceBinding::VertexStage | QRhiShaderResourceBinding::FragmentStage, buf: d->ubuf, offset: 0, size: UBUF_SIZE),
381 QRhiShaderResourceBinding::sampledTexture(binding: 1, stage: QRhiShaderResourceBinding::FragmentStage, tex: texture, sampler: m_sampler)
382 });
383
384 d->srb->updateResources(flags: QRhiShaderResourceBindings::BindingsAreSorted);
385 d->lastUsedTexture = texture;
386
387 if (textureExtra) {
388 d->srbExtra->setBindings({
389 QRhiShaderResourceBinding::uniformBuffer(binding: 0, stage: QRhiShaderResourceBinding::VertexStage | QRhiShaderResourceBinding::FragmentStage, buf: d->ubuf, offset: 0, size: UBUF_SIZE),
390 QRhiShaderResourceBinding::sampledTexture(binding: 1, stage: QRhiShaderResourceBinding::FragmentStage, tex: textureExtra, sampler: m_sampler)
391 });
392
393 d->srbExtra->updateResources(flags: QRhiShaderResourceBindings::BindingsAreSorted);
394 d->lastUsedTextureExtra = textureExtra;
395 }
396}

398void QBackingStoreDefaultCompositor::updateUniforms(PerQuadData *d, QRhiResourceUpdateBatch *resourceUpdates,
399 const QMatrix4x4 &target, const QMatrix3x3 &source, UpdateUniformOption option)
400{
401 resourceUpdates->updateDynamicBuffer(buf: d->ubuf, offset: 0, size: 64, data: target.constData());
402 updateMatrix3x3(resourceUpdates, ubuf: d->ubuf, m: source);
403 float opacity = 1.0f;
404 resourceUpdates->updateDynamicBuffer(buf: d->ubuf, offset: 112, size: 4, data: &opacity);
405 qint32 textureSwizzle = static_cast<qint32>(option);
406 resourceUpdates->updateDynamicBuffer(buf: d->ubuf, offset: 116, size: 4, data: &textureSwizzle);
407}

409void QBackingStoreDefaultCompositor::ensureResources(QRhiSwapChain *swapchain, QRhiResourceUpdateBatch *resourceUpdates)
410{
411 static const float vertexData[] = {
412 -1, -1, 0, 0, 0,
413 -1, 1, 0, 0, 1,
414 1, -1, 0, 1, 0,
415 -1, 1, 0, 0, 1,
416 1, -1, 0, 1, 0,
417 1, 1, 0, 1, 1
418 };
419
420 if (!m_vbuf) {
421 m_vbuf = m_rhi->newBuffer(type: QRhiBuffer::Immutable, usage: QRhiBuffer::VertexBuffer, size: sizeof(vertexData));
422 if (m_vbuf->create())
423 resourceUpdates->uploadStaticBuffer(buf: m_vbuf, data: vertexData);
424 else
425 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to create vertex buffer");
426 }
427
428 if (!m_sampler) {
429 m_sampler = m_rhi->newSampler(magFilter: QRhiSampler::Linear, minFilter: QRhiSampler::Linear, mipmapMode: QRhiSampler::None,
430 addressU: QRhiSampler::ClampToEdge, addressV: QRhiSampler::ClampToEdge);
431 if (!m_sampler->create())
432 qWarning(msg: "QBackingStoreDefaultCompositor: Failed to create sampler");
433 }
434
435 if (!m_widgetQuadData.isValid())
436 m_widgetQuadData = createPerQuadData(texture: m_texture);
437
438 QRhiShaderResourceBindings *srb = m_widgetQuadData.srb; // just for the layout
439 if (!m_psNoBlend)
440 m_psNoBlend = createGraphicsPipeline(rhi: m_rhi, srb, swapchain, blend: PipelineBlend::None);
441 if (!m_psBlend)
442 m_psBlend = createGraphicsPipeline(rhi: m_rhi, srb, swapchain, blend: PipelineBlend::Alpha);
443 if (!m_psPremulBlend)
444 m_psPremulBlend = createGraphicsPipeline(rhi: m_rhi, srb, swapchain, blend: PipelineBlend::PremulAlpha);
445}

447QPlatformBackingStore::FlushResult QBackingStoreDefaultCompositor::flush(QPlatformBackingStore *backingStore,
448 QRhi *rhi,
449 QRhiSwapChain *swapchain,
450 QWindow *window,
451 qreal sourceDevicePixelRatio,
452 const QRegion &region,
453 const QPoint &offset,
454 QPlatformTextureList *textures,
455 bool translucentBackground)
456{
457 Q_ASSERT(textures); // may be empty if there are no render-to-texture widgets at all, but null it cannot be
458
459 if (!m_rhi) {
460 m_rhi = rhi;
461 } else if (m_rhi != rhi) {
462 qWarning(msg: "QBackingStoreDefaultCompositor: the QRhi has changed unexpectedly, this should not happen");
463 return QPlatformBackingStore::FlushFailed;
464 }
465
466 if (!qt_window_private(window)->receivedExpose)
467 return QPlatformBackingStore::FlushSuccess;
468
469 qCDebug(lcQpaBackingStore) << "Composing and flushing" << region << "of" << window
470 << "at offset" << offset << "with" << textures->count() << "texture(s) in" << textures
471 << "via swapchain" << swapchain;
472
473 QWindowPrivate::get(window)->lastComposeTime.start();
474
475 if (swapchain->currentPixelSize() != swapchain->surfacePixelSize())
476 swapchain->createOrResize();
477
478 // Start recording a new frame.
479 QRhi::FrameOpResult frameResult = rhi->beginFrame(swapChain: swapchain);
480 if (frameResult == QRhi::FrameOpSwapChainOutOfDate) {
481 if (!swapchain->createOrResize())
482 return QPlatformBackingStore::FlushFailed;
483 frameResult = rhi->beginFrame(swapChain: swapchain);
484 }
485 if (frameResult == QRhi::FrameOpDeviceLost)
486 return QPlatformBackingStore::FlushFailedDueToLostDevice;
487 if (frameResult != QRhi::FrameOpSuccess)
488 return QPlatformBackingStore::FlushFailed;
489
490 // Prepare resource updates.
491 QRhiResourceUpdateBatch *resourceUpdates = rhi->nextResourceUpdateBatch();
492 QPlatformBackingStore::TextureFlags flags;
493
494 bool gotTextureFromGraphicsBuffer = false;
495 if (QPlatformGraphicsBuffer *graphicsBuffer = backingStore->graphicsBuffer()) {
496 if (graphicsBuffer->lock(access: QPlatformGraphicsBuffer::SWReadAccess)) {
497 const QImage::Format format = QImage::toImageFormat(format: graphicsBuffer->format());
498 const QSize size = graphicsBuffer->size();
499 QImage wrapperImage(graphicsBuffer->data(), size.width(), size.height(), graphicsBuffer->bytesPerLine(), format);
500 toTexture(sourceImage: wrapperImage, rhi, resourceUpdates, dirtyRegion: scaledRegion(region, factor: sourceDevicePixelRatio, offset), flags: &flags);
501 gotTextureFromGraphicsBuffer = true;
502 graphicsBuffer->unlock();
503 if (graphicsBuffer->origin() == QPlatformGraphicsBuffer::OriginBottomLeft)
504 flags |= QPlatformBackingStore::TextureFlip;
505 }
506 }
507 if (!gotTextureFromGraphicsBuffer)
508 toTexture(backingStore, rhi, resourceUpdates, dirtyRegion: scaledRegion(region, factor: sourceDevicePixelRatio, offset), flags: &flags);
509
510 ensureResources(swapchain, resourceUpdates);
511
512#if Q_BYTE_ORDER == Q_LITTLE_ENDIAN
513 const UpdateUniformOption uniformOption = (flags & QPlatformBackingStore::TextureSwizzle) != 0? NeedsRedBlueSwap : NoOption;
514#else
515 const UpdateUniformOption uniformOption = (flags & QPlatformBackingStore::TextureSwizzle) != 0? NeedsAlphaRotate : NoOption;
516#endif
517 const bool premultiplied = (flags & QPlatformBackingStore::TexturePremultiplied) != 0;
518 SourceTransformOrigin origin = SourceTransformOrigin::TopLeft;
519 if (flags & QPlatformBackingStore::TextureFlip)
520 origin = SourceTransformOrigin::BottomLeft;
521
522 const qreal dpr = window->devicePixelRatio();
523 const QRect deviceWindowRect = scaledRect(rect: QRect(QPoint(), window->size()), factor: dpr);
524 const QPoint deviceWindowOffset = scaledOffset(pt: offset, factor: dpr);
525
526 const bool invertTargetY = rhi->clipSpaceCorrMatrix().data()[5] < 0.0f;
527 const bool invertSource = rhi->isYUpInFramebuffer() != rhi->isYUpInNDC();
528 if (m_texture) {
529 // The backingstore is for the entire tlw.
530 // In case of native children offset tells the position relative to the tlw.
531 const QRect srcRect = toBottomLeftRect(topLeftRect: deviceWindowRect.translated(p: deviceWindowOffset), windowHeight: m_texture->pixelSize().height());
532 const QMatrix3x3 source = sourceTransform(subTexture: srcRect, textureSize: m_texture->pixelSize(), origin);
533 QMatrix4x4 target; // identity
534 if (invertTargetY)
535 target.data()[5] = -1.0f;
536 updateUniforms(d: &m_widgetQuadData, resourceUpdates, target, source, option: uniformOption);
537 }
538
539 const int textureWidgetCount = textures->count();
540 const int oldTextureQuadDataCount = m_textureQuadData.size();
541 if (oldTextureQuadDataCount != textureWidgetCount) {
542 for (int i = textureWidgetCount; i < oldTextureQuadDataCount; ++i)
543 m_textureQuadData[i].reset();
544 m_textureQuadData.resize(sz: textureWidgetCount);
545 }
546
547 for (int i = 0; i < textureWidgetCount; ++i) {
548 QMatrix4x4 target;
549 QMatrix3x3 source;
550 if (!prepareDrawForRenderToTextureWidget(textures, idx: i, window, deviceWindowRect,
551 offset, invertTargetY, invertSource, target: &target, source: &source))
552 {
553 m_textureQuadData[i].reset();
554 continue;
555 }
556 QRhiTexture *t = textures->texture(index: i);
557 QRhiTexture *tExtra = textures->textureExtra(index: i);
558 if (t) {
559 if (!m_textureQuadData[i].isValid()) {
560 m_textureQuadData[i] = createPerQuadData(texture: t, textureExtra: tExtra);
561 }
562 else {
563 updatePerQuadData(d: &m_textureQuadData[i], texture: t, textureExtra: tExtra);
564 }
565 updateUniforms(d: &m_textureQuadData[i], resourceUpdates, target, source, option: NoOption);
566 } else {
567 m_textureQuadData[i].reset();
568 }
569 }
570
571 // Record the render pass (with committing the resource updates).
572 QRhiCommandBuffer *cb = swapchain->currentFrameCommandBuffer();
573 const QSize outputSizeInPixels = swapchain->currentPixelSize();
574 QColor clearColor = translucentBackground ? Qt::transparent : Qt::black;
575
576 cb->resourceUpdate(resourceUpdates);
577
578 auto render = [&](std::optional<QRhiSwapChain::StereoTargetBuffer> buffer = std::nullopt) {
579 QRhiRenderTarget* target = nullptr;
580 if (buffer.has_value())
581 target = swapchain->currentFrameRenderTarget(targetBuffer: buffer.value());
582 else
583 target = swapchain->currentFrameRenderTarget();
584
585 cb->beginPass(rt: target, colorClearValue: clearColor, depthStencilClearValue: { 1.0f, 0 });
586
587 cb->setGraphicsPipeline(m_psNoBlend);
588 cb->setViewport({ 0, 0, float(outputSizeInPixels.width()), float(outputSizeInPixels.height()) });
589 QRhiCommandBuffer::VertexInput vbufBinding(m_vbuf, 0);
590 cb->setVertexInput(startBinding: 0, bindingCount: 1, bindings: &vbufBinding);
591
592 // Textures for renderToTexture widgets.
593 for (int i = 0; i < textureWidgetCount; ++i) {
594 if (!textures->flags(index: i).testFlag(flag: QPlatformTextureList::StacksOnTop)) {
595 if (m_textureQuadData[i].isValid()) {
596
597 QRhiShaderResourceBindings* srb = m_textureQuadData[i].srb;
598 if (buffer == QRhiSwapChain::RightBuffer && m_textureQuadData[i].srbExtra)
599 srb = m_textureQuadData[i].srbExtra;
600
601 cb->setShaderResources(srb);
602 cb->draw(vertexCount: 6);
603 }
604 }
605 }
606
607 cb->setGraphicsPipeline(premultiplied ? m_psPremulBlend : m_psBlend);
608
609 // Backingstore texture with the normal widgets.
610 if (m_texture) {
611 cb->setShaderResources(srb: m_widgetQuadData.srb);
612 cb->draw(vertexCount: 6);
613 }
614
615 // Textures for renderToTexture widgets that have WA_AlwaysStackOnTop set.
616 for (int i = 0; i < textureWidgetCount; ++i) {
617 const QPlatformTextureList::Flags flags = textures->flags(index: i);
618 if (flags.testFlag(flag: QPlatformTextureList::StacksOnTop)) {
619 if (m_textureQuadData[i].isValid()) {
620 if (flags.testFlag(flag: QPlatformTextureList::NeedsPremultipliedAlphaBlending))
621 cb->setGraphicsPipeline(m_psPremulBlend);
622 else
623 cb->setGraphicsPipeline(m_psBlend);
624
625 QRhiShaderResourceBindings* srb = m_textureQuadData[i].srb;
626 if (buffer == QRhiSwapChain::RightBuffer && m_textureQuadData[i].srbExtra)
627 srb = m_textureQuadData[i].srbExtra;
628
629 cb->setShaderResources(srb);
630 cb->draw(vertexCount: 6);
631 }
632 }
633 }
634
635 cb->endPass();
636 };
637
638 if (swapchain->window()->format().stereo()) {
639 render(QRhiSwapChain::LeftBuffer);
640 render(QRhiSwapChain::RightBuffer);
641 } else
642 render();
643
644 rhi->endFrame(swapChain: swapchain);
645
646 return QPlatformBackingStore::FlushSuccess;
647}
