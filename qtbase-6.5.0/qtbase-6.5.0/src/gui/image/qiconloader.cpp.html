175/*!
176 \internal
177 Helper class that reads and looks up into the icon-theme.cache generated with
178 gtk-update-icon-cache. If at any point we detect a corruption in the file
179 (because the offsets point at wrong locations for example), the reader
180 is marked as invalid.
181*/
182class QIconCacheGtkReader
183{
184public:
185 explicit QIconCacheGtkReader(const QString &themeDir);
186 QList<const char *> lookup(QStringView);
187 bool isValid() const { return m_isValid; }
188private:
189 QFile m_file;
190 const unsigned char *m_data;
191 quint64 m_size;
192 bool m_isValid;
193
194 quint16 read16(uint offset)
195 {
196 if (offset > m_size - 2 || (offset & 0x1)) {
197 m_isValid = false;
198 return 0;
199 }
200 return m_data[offset+1] | m_data[offset] << 8;
201 }
202 quint32 read32(uint offset)
203 {
204 if (offset > m_size - 4 || (offset & 0x3)) {
205 m_isValid = false;
206 return 0;
207 }
208 return m_data[offset+3] | m_data[offset+2] << 8
209 | m_data[offset+1] << 16 | m_data[offset] << 24;
210 }
211};

214QIconCacheGtkReader::QIconCacheGtkReader(const QString &dirName)
215 : m_isValid(false)
216{
217 QFileInfo info(dirName + "/icon-theme.cache"_L1);
218 if (!info.exists() || info.lastModified() < QFileInfo(dirName).lastModified())
219 return;
220 m_file.setFileName(info.absoluteFilePath());
221 if (!m_file.open(flags: QFile::ReadOnly))
222 return;
223 m_size = m_file.size();
224 m_data = m_file.map(offset: 0, size: m_size);
225 if (!m_data)
226 return;
227 if (read16(offset: 0) != 1) // VERSION_MAJOR
228 return;
229
230 m_isValid = true;
231
232 // Check that all the directories are older than the cache
233 auto lastModified = info.lastModified();
234 quint32 dirListOffset = read32(offset: 8);
235 quint32 dirListLen = read32(offset: dirListOffset);
236 for (uint i = 0; i < dirListLen; ++i) {
237 quint32 offset = read32(offset: dirListOffset + 4 + 4 * i);
238 if (!m_isValid || offset >= m_size || lastModified < QFileInfo(dirName + u'/'
239 + QString::fromUtf8(utf8: reinterpret_cast<const char*>(m_data + offset))).lastModified()) {
240 m_isValid = false;
241 return;
242 }
243 }
244}

246static quint32 icon_name_hash(const char *p)
247{
248 quint32 h = static_cast<signed char>(*p);
249 for (p += 1; *p != '\0'; p++)
250 h = (h << 5) - h + *p;
251 return h;
252}

254/*! \internal
255 lookup the icon name and return the list of subdirectories in which an icon
256 with this name is present. The char* are pointers to the mapped data.
257 For example, this would return { "32x32/apps", "24x24/apps" , ... }
258 */
259QList<const char *> QIconCacheGtkReader::lookup(QStringView name)
260{
261 QList<const char *> ret;
262 if (!isValid() || name.isEmpty())
263 return ret;
264
265 QByteArray nameUtf8 = name.toUtf8();
266 quint32 hash = icon_name_hash(p: nameUtf8);
267
268 quint32 hashOffset = read32(offset: 4);
269 quint32 hashBucketCount = read32(offset: hashOffset);
270
271 if (!isValid() || hashBucketCount == 0) {
272 m_isValid = false;
273 return ret;
274 }
275
276 quint32 bucketIndex = hash % hashBucketCount;
277 quint32 bucketOffset = read32(offset: hashOffset + 4 + bucketIndex * 4);
278 while (bucketOffset > 0 && bucketOffset <= m_size - 12) {
279 quint32 nameOff = read32(offset: bucketOffset + 4);
280 if (nameOff < m_size && strcmp(s1: reinterpret_cast<const char*>(m_data + nameOff), s2: nameUtf8) == 0) {
281 quint32 dirListOffset = read32(offset: 8);
282 quint32 dirListLen = read32(offset: dirListOffset);
283
284 quint32 listOffset = read32(offset: bucketOffset+8);
285 quint32 listLen = read32(offset: listOffset);
286
287 if (!m_isValid || listOffset + 4 + 8 * listLen > m_size) {
288 m_isValid = false;
289 return ret;
290 }
291
292 ret.reserve(size: listLen);
293 for (uint j = 0; j < listLen && m_isValid; ++j) {
294 quint32 dirIndex = read16(offset: listOffset + 4 + 8 * j);
295 quint32 o = read32(offset: dirListOffset + 4 + dirIndex*4);
296 if (!m_isValid || dirIndex >= dirListLen || o >= m_size) {
297 m_isValid = false;
298 return ret;
299 }
300 ret.append(t: reinterpret_cast<const char*>(m_data) + o);
301 }
302 return ret;
303 }
304 bucketOffset = read32(offset: bucketOffset);
305 }
306 return ret;
307}

309QIconTheme::QIconTheme(const QString &themeName)
310 : m_valid(false)
311{
312 QFile themeIndex;
313
314 const QStringList iconDirs = QIcon::themeSearchPaths();
315 for ( int i = 0 ; i < iconDirs.size() ; ++i) {
316 QDir iconDir(iconDirs[i]);
317 QString themeDir = iconDir.path() + u'/' + themeName;
318 QFileInfo themeDirInfo(themeDir);
319
320 if (themeDirInfo.isDir()) {
321 m_contentDirs << themeDir;
322 m_gtkCaches << QSharedPointer<QIconCacheGtkReader>::create(arguments&: themeDir);
323 }
324
325 if (!m_valid) {
326 themeIndex.setFileName(themeDir + "/index.theme"_L1);
327 if (themeIndex.exists())
328 m_valid = true;
329 }
330 }
331#if QT_CONFIG(settings)
332 if (themeIndex.exists()) {
333 const QSettings indexReader(themeIndex.fileName(), QSettings::IniFormat);
334 const QStringList keys = indexReader.allKeys();
335 for (const QString &key : keys) {
336 if (key.endsWith(s: "/Size"_L1)) {
337 // Note the QSettings ini-format does not accept
338 // slashes in key names, hence we have to cheat
339 if (int size = indexReader.value(key).toInt()) {
340 QString directoryKey = key.left(n: key.size() - 5);
341 QIconDirInfo dirInfo(directoryKey);
342 dirInfo.size = size;
343 QString type = indexReader.value(key: directoryKey + "/Type"_L1).toString();
344
345 if (type == "Fixed"_L1)
346 dirInfo.type = QIconDirInfo::Fixed;
347 else if (type == "Scalable"_L1)
348 dirInfo.type = QIconDirInfo::Scalable;
349 else
350 dirInfo.type = QIconDirInfo::Threshold;
351
352 dirInfo.threshold = indexReader.value(key: directoryKey +
353 "/Threshold"_L1,
354 defaultValue: 2).toInt();
355
356 dirInfo.minSize = indexReader.value(key: directoryKey + "/MinSize"_L1, defaultValue: size).toInt();
357
358 dirInfo.maxSize = indexReader.value(key: directoryKey + "/MaxSize"_L1, defaultValue: size).toInt();
359
360 dirInfo.scale = indexReader.value(key: directoryKey + "/Scale"_L1, defaultValue: 1).toInt();
361 m_keyList.append(t: dirInfo);
362 }
363 }
364 }
365
366 // Parent themes provide fallbacks for missing icons
367 m_parents = indexReader.value(key: "Icon Theme/Inherits"_L1).toStringList();
368 m_parents.removeAll(t: QString());
369
370 // Ensure a default platform fallback for all themes
371 if (m_parents.isEmpty()) {
372 const QString fallback = QIconLoader::instance()->fallbackThemeName();
373 if (!fallback.isEmpty())
374 m_parents.append(t: fallback);
375 }
376
377 // Ensure that all themes fall back to hicolor
378 if (!m_parents.contains(str: "hicolor"_L1))
379 m_parents.append(t: "hicolor"_L1);
380 }
381#endif // settings
382}

384QThemeIconInfo QIconLoader::findIconHelper(const QString &themeName,
385 const QString &iconName,
386 QStringList &visited) const
387{
388 QThemeIconInfo info;
389 Q_ASSERT(!themeName.isEmpty());
390
391 // Used to protect against potential recursions
392 visited << themeName;
393
394 QIconTheme &theme = themeList[themeName];
395 if (!theme.isValid()) {
396 theme = QIconTheme(themeName);
397 if (!theme.isValid())
398 theme = QIconTheme(fallbackThemeName());
399 }
400
401 const QStringList contentDirs = theme.contentDirs();
402
403 QStringView iconNameFallback(iconName);
404
405 // Iterate through all icon's fallbacks in current theme
406 while (info.entries.empty()) {
407 const QString svgIconName = iconNameFallback + ".svg"_L1;
408 const QString pngIconName = iconNameFallback + ".png"_L1;
409
410 // Add all relevant files
411 for (int i = 0; i < contentDirs.size(); ++i) {
412 QList<QIconDirInfo> subDirs = theme.keyList();
413
414 // Try to reduce the amount of subDirs by looking in the GTK+ cache in order to save
415 // a massive amount of file stat (especially if the icon is not there)
416 auto cache = theme.m_gtkCaches.at(i);
417 if (cache->isValid()) {
418 const auto result = cache->lookup(name: iconNameFallback);
419 if (cache->isValid()) {
420 const QList<QIconDirInfo> subDirsCopy = subDirs;
421 subDirs.clear();
422 subDirs.reserve(size: result.size());
423 for (const char *s : result) {
424 QString path = QString::fromUtf8(utf8: s);
425 auto it = std::find_if(first: subDirsCopy.cbegin(), last: subDirsCopy.cend(),
426 pred: [&](const QIconDirInfo &info) {
427 return info.path == path; } );
428 if (it != subDirsCopy.cend()) {
429 subDirs.append(t: *it);
430 }
431 }
432 }
433 }
434
435 QString contentDir = contentDirs.at(i) + u'/';
436 for (int j = 0; j < subDirs.size() ; ++j) {
437 const QIconDirInfo &dirInfo = subDirs.at(i: j);
438 const QString subDir = contentDir + dirInfo.path + u'/';
439 const QString pngPath = subDir + pngIconName;
440 if (QFile::exists(fileName: pngPath)) {
441 auto iconEntry = std::make_unique<PixmapEntry>();
442 iconEntry->dir = dirInfo;
443 iconEntry->filename = pngPath;
444 // Notice we ensure that pixmap entries always come before
445 // scalable to preserve search order afterwards
446 info.entries.insert(position: info.entries.begin(), x: std::move(iconEntry));
447 } else if (m_supportsSvg) {
448 const QString svgPath = subDir + svgIconName;
449 if (QFile::exists(fileName: svgPath)) {
450 auto iconEntry = std::make_unique<ScalableEntry>();
451 iconEntry->dir = dirInfo;
452 iconEntry->filename = svgPath;
453 info.entries.push_back(x: std::move(iconEntry));
454 }
455 }
456 }
457 }
458
459 if (!info.entries.empty()) {
460 info.iconName = iconNameFallback.toString();
461 break;
462 }
463
464 // If it's possible - find next fallback for the icon
465 const int indexOfDash = iconNameFallback.lastIndexOf(c: u'-');
466 if (indexOfDash == -1)
467 break;
468
469 iconNameFallback.truncate(n: indexOfDash);
470 }
471
472 if (info.entries.empty()) {
473 const QStringList parents = theme.parents();
474 // Search recursively through inherited themes
475 for (int i = 0 ; i < parents.size() ; ++i) {
476
477 const QString parentTheme = parents.at(i).trimmed();
478
479 if (!visited.contains(str: parentTheme)) // guard against recursion
480 info = findIconHelper(themeName: parentTheme, iconName, visited);
481
482 if (!info.entries.empty()) // success
483 break;
484 }
485 }
486 return info;
487}

489QThemeIconInfo QIconLoader::lookupFallbackIcon(const QString &iconName) const
490{
491 QThemeIconInfo info;
492
493 const QString pngIconName = iconName + ".png"_L1;
494 const QString xpmIconName = iconName + ".xpm"_L1;
495 const QString svgIconName = iconName + ".svg"_L1;
496
497 const auto searchPaths = QIcon::fallbackSearchPaths();
498 for (const QString &iconDir: searchPaths) {
499 QDir currentDir(iconDir);
500 std::unique_ptr<QIconLoaderEngineEntry> iconEntry;
501 if (currentDir.exists(name: pngIconName)) {
502 iconEntry = std::make_unique<PixmapEntry>();
503 iconEntry->dir.type = QIconDirInfo::Fallback;
504 iconEntry->filename = currentDir.filePath(fileName: pngIconName);
505 } else if (currentDir.exists(name: xpmIconName)) {
506 iconEntry = std::make_unique<PixmapEntry>();
507 iconEntry->dir.type = QIconDirInfo::Fallback;
508 iconEntry->filename = currentDir.filePath(fileName: xpmIconName);
509 } else if (m_supportsSvg &&
510 currentDir.exists(name: svgIconName)) {
511 iconEntry = std::make_unique<ScalableEntry>();
512 iconEntry->dir.type = QIconDirInfo::Fallback;
513 iconEntry->filename = currentDir.filePath(fileName: svgIconName);
514 }
515 if (iconEntry) {
516 info.entries.push_back(x: std::move(iconEntry));
517 break;
518 }
519 }
520
521 if (!info.entries.empty())
522 info.iconName = iconName;
523
524 return info;
525}

596/*
597 * This algorithm is defined by the freedesktop spec:
598 * http://standards.freedesktop.org/icon-theme-spec/icon-theme-spec-latest.html
599 */
600static bool directoryMatchesSize(const QIconDirInfo &dir, int iconsize, int iconscale)
601{
602 if (dir.scale != iconscale)
603 return false;
604
605 if (dir.type == QIconDirInfo::Fixed) {
606 return dir.size == iconsize;
607
608 } else if (dir.type == QIconDirInfo::Scalable) {
609 return iconsize <= dir.maxSize &&
610 iconsize >= dir.minSize;
611
612 } else if (dir.type == QIconDirInfo::Threshold) {
613 return iconsize >= dir.size - dir.threshold &&
614 iconsize <= dir.size + dir.threshold;
615 } else if (dir.type == QIconDirInfo::Fallback) {
616 return true;
617 }
618
619 Q_ASSERT(1); // Not a valid value
620 return false;
621}
622
623/*
624 * This algorithm is defined by the freedesktop spec:
625 * http://standards.freedesktop.org/icon-theme-spec/icon-theme-spec-latest.html
626 */
627static int directorySizeDistance(const QIconDirInfo &dir, int iconsize, int iconscale)
628{
629 const int scaledIconSize = iconsize * iconscale;
630 if (dir.type == QIconDirInfo::Fixed) {
631 return qAbs(t: dir.size * dir.scale - scaledIconSize);
632
633 } else if (dir.type == QIconDirInfo::Scalable) {
634 if (scaledIconSize < dir.minSize * dir.scale)
635 return dir.minSize * dir.scale - scaledIconSize;
636 else if (scaledIconSize > dir.maxSize * dir.scale)
637 return scaledIconSize - dir.maxSize * dir.scale;
638 else
639 return 0;
640
641 } else if (dir.type == QIconDirInfo::Threshold) {
642 if (scaledIconSize < (dir.size - dir.threshold) * dir.scale)
643 return dir.minSize * dir.scale - scaledIconSize;
644 else if (scaledIconSize > (dir.size + dir.threshold) * dir.scale)
645 return scaledIconSize - dir.maxSize * dir.scale;
646 else return 0;
647 } else if (dir.type == QIconDirInfo::Fallback) {
648 return 0;
649 }
650
651 Q_ASSERT(1); // Not a valid value
652 return INT_MAX;
653}

655QIconLoaderEngineEntry *QIconLoaderEngine::entryForSize(const QThemeIconInfo &info, const QSize &size, int scale)
656{
657 int iconsize = qMin(a: size.width(), b: size.height());
658
659 // Note that m_info.entries are sorted so that png-files
660 // come first
661
662 // Search for exact matches first
663 for (const auto &entry : info.entries) {
664 if (directoryMatchesSize(dir: entry->dir, iconsize, iconscale: scale)) {
665 return entry.get();
666 }
667 }
668
669 // Find the minimum distance icon
670 int minimalSize = INT_MAX;
671 QIconLoaderEngineEntry *closestMatch = nullptr;
672 for (const auto &entry : info.entries) {
673 int distance = directorySizeDistance(dir: entry->dir, iconsize, iconscale: scale);
674 if (distance < minimalSize) {
675 minimalSize = distance;
676 closestMatch = entry.get();
677 }
678 }
679 return closestMatch;
680}

682/*
683 * Returns the actual icon size. For scalable svg's this is equivalent
684 * to the requested size. Otherwise the closest match is returned but
685 * we can never return a bigger size than the requested size.
686 *
687 */
688QSize QIconLoaderEngine::actualSize(const QSize &size, QIcon::Mode mode,
689 QIcon::State state)
690{
691 Q_UNUSED(mode);
692 Q_UNUSED(state);
693
694 ensureLoaded();
695
696 QIconLoaderEngineEntry *entry = entryForSize(info: m_info, size);
697 if (entry) {
698 const QIconDirInfo &dir = entry->dir;
699 if (dir.type == QIconDirInfo::Scalable) {
700 return size;
701 } else if (dir.type == QIconDirInfo::Fallback) {
702 return QIcon(entry->filename).actualSize(size, mode, state);
703 } else {
704 int result = qMin<int>(a: dir.size, b: qMin(a: size.width(), b: size.height()));
705 return QSize(result, result);
706 }
707 }
708 return QSize(0, 0);
709}

711QPixmap PixmapEntry::pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state)
712{
713 Q_UNUSED(state);
714
715 // Ensure that basePixmap is lazily initialized before generating the
716 // key, otherwise the cache key is not unique
717 if (basePixmap.isNull())
718 basePixmap.load(fileName: filename);
719
720 QSize actualSize = basePixmap.size();
721 // If the size of the best match we have (basePixmap) is larger than the
722 // requested size, we downscale it to match.
723 if (!actualSize.isNull() && (actualSize.width() > size.width() || actualSize.height() > size.height()))
724 actualSize.scale(s: size, mode: Qt::KeepAspectRatio);
725
726 QString key = "$qt_theme_"_L1
727 % HexString<qint64>(basePixmap.cacheKey())
728 % HexString<int>(mode)
729 % HexString<qint64>(QGuiApplication::palette().cacheKey())
730 % HexString<int>(actualSize.width())
731 % HexString<int>(actualSize.height());
732
733 QPixmap cachedPixmap;
734 if (QPixmapCache::find(key, pixmap: &cachedPixmap)) {
735 return cachedPixmap;
736 } else {
737 if (basePixmap.size() != actualSize)
738 cachedPixmap = basePixmap.scaled(s: actualSize, aspectMode: Qt::IgnoreAspectRatio, mode: Qt::SmoothTransformation);
739 else
740 cachedPixmap = basePixmap;
741 if (QGuiApplication *guiApp = qobject_cast<QGuiApplication *>(qApp))
742 cachedPixmap = static_cast<QGuiApplicationPrivate*>(QObjectPrivate::get(o: guiApp))->applyQIconStyleHelper(mode, basePixmap: cachedPixmap);
743 QPixmapCache::insert(key, pixmap: cachedPixmap);
744 }
745 return cachedPixmap;
746}

798QList<QSize> QIconLoaderEngine::availableSizes(QIcon::Mode mode, QIcon::State state)
799{
800 Q_UNUSED(mode);
801 Q_UNUSED(state);
802 ensureLoaded();
803 const qsizetype N = qsizetype(m_info.entries.size());
804 QList<QSize> sizes;
805 sizes.reserve(size: N);
806
807 // Gets all sizes from the DirectoryInfo entries
808 for (const auto &entry : m_info.entries) {
809 if (entry->dir.type == QIconDirInfo::Fallback) {
810 sizes.append(l: QIcon(entry->filename).availableSizes());

813 sizes.append(t: QSize(size, size));
814 }
815 }
