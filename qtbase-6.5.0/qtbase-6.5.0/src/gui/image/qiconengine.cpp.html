9/*!
10 \class QIconEngine
11
12 \brief The QIconEngine class provides an abstract base class for QIcon renderers.
13
14 \ingroup painting
15 \inmodule QtGui
16
17 An icon engine provides the rendering functions for a QIcon. Each icon has a
18 corresponding icon engine that is responsible for drawing the icon with a
19 requested size, mode and state.
20
21 The icon is rendered by the paint() function, and the icon can additionally be
22 obtained as a pixmap with the pixmap() function (the default implementation
23 simply uses paint() to achieve this). The addPixmap() function can be used to
24 add new pixmaps to the icon engine, and is used by QIcon to add specialized
25 custom pixmaps.
26
27 The paint(), pixmap(), and addPixmap() functions are all virtual, and can
28 therefore be reimplemented in subclasses of QIconEngine.
29
30 \sa QIconEnginePlugin
31
32*/
33

74/*!
75 Returns the icon as a pixmap with the required \a size, \a mode,
76 and \a state. The default implementation creates a new pixmap and
77 calls paint() to fill it.
78*/
79QPixmap QIconEngine::pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state)
80{
81 QPixmap pm(size);
82 {
83 QPainter p(&pm);
84 paint(painter: &p, rect: QRect(QPoint(0,0),size), mode, state);
85 }
86 return pm;
87}

114/*!
115 \enum QIconEngine::IconEngineHook
116 \since 4.5
117
118 These enum values are used for virtual_hook() to allow additional
119 queries to icon engine without breaking binary compatibility.
120
121 \value IsNullHook Allow to query if this engine represents a null
122 icon. The \a data argument of the virtual_hook() is a pointer to a
123 bool that can be set to true if the icon is null. This enum value
124 was added in Qt 5.7.
125
126 \value ScaledPixmapHook Provides a way to get a pixmap that is scaled
127 according to the given scale (typically equal to the \l {High
128 DPI}{device pixel ratio}). The \a data argument of the virtual_hook()
129 function is a \l ScaledPixmapArgument pointer that contains both the input and
130 output arguments. This enum value was added in Qt 5.9.
131
132 \sa virtual_hook()
133 */
134
135/*!
136 \class QIconEngine::ScaledPixmapArgument
137 \since 5.9
138
139 \inmodule QtGui
140
141 This struct represents arguments to the virtual_hook() function when
142 the \a id parameter is QIconEngine::ScaledPixmapHook.
143
144 The struct provides a way for icons created via \l QIcon::fromTheme()
145 to return pixmaps that are designed for the current \l {High
146 DPI}{device pixel ratio}. The scale for such an icon is specified
147 using the \l {Icon Theme Specification - Directory Layout}{Scale directory key}
148 in the appropriate \c index.theme file.
149
150 Icons created via other approaches will return the same result as a call to
151 \l pixmap() would, and continue to benefit from Qt's \l {High Resolution
152 Versions of Images}{"@nx" high DPI syntax}.
153
154 \sa virtual_hook(), QIconEngine::IconEngineHook, {High DPI Icons}
155 */
156

225/*!
226 \since 4.5
227
228 Additional method to allow extending QIconEngine without
229 adding new virtual methods (and without breaking binary compatibility).
230 The actual action and format of \a data depends on \a id argument
231 which is in fact a constant from IconEngineHook enum.
232
233 \sa IconEngineHook
234*/
235void QIconEngine::virtual_hook(int id, void *data)
236{
237 switch (id) {
238 case QIconEngine::ScaledPixmapHook: {
239 // We don't have any notion of scale besides "@nx", so just call pixmap() here.
240 QIconEngine::ScaledPixmapArgument &arg =
241 *reinterpret_cast<QIconEngine::ScaledPixmapArgument*>(data);
242 arg.pixmap = pixmap(size: arg.size, mode: arg.mode, state: arg.state);
243 break;
244 }
245 default:
246 break;
247 }
248}

285/*!
286 \since 5.9
287
288 Returns a pixmap for the given \a size, \a mode, \a state and \a scale.
289
290 The \a scale argument is typically equal to the \l {High DPI}
291 {device pixel ratio} of the display.
292
293 \include qiconengine-virtualhookhelper.qdocinc
294
295 \note Some engines may cast \a scale to an integer.
296
297 \sa ScaledPixmapArgument
298*/
299QPixmap QIconEngine::scaledPixmap(const QSize &size, QIcon::Mode mode, QIcon::State state, qreal scale)
300{
301 ScaledPixmapArgument arg;
302 arg.size = size;
303 arg.mode = mode;
304 arg.state = state;
305 arg.scale = scale;
306 const_cast<QIconEngine *>(this)->virtual_hook(id: QIconEngine::ScaledPixmapHook, data: reinterpret_cast<void*>(&arg));
307 return arg.pixmap;
308}
