13/*!
14 \class QPixmapCache
15 \inmodule QtGui
16
17 \brief The QPixmapCache class provides an application-wide cache for pixmaps.
18
19 This class is a tool for optimized drawing with QPixmap. You can
20 use it to store temporary pixmaps that are expensive to generate
21 without using more storage space than cacheLimit(). Use insert()
22 to insert pixmaps, find() to find them, and clear() to empty the
23 cache.
24
25 QPixmapCache contains no member data, only static functions to
26 access the global pixmap cache. It creates an internal QCache
27 object for caching the pixmaps.
28
29 The cache associates a pixmap with a user-provided string as a key,
30 or with a QPixmapCache::Key that the cache generates.
31 Using QPixmapCache::Key for keys is faster than using strings. The string API is
32 very convenient for complex keys but the QPixmapCache::Key API will be very
33 efficient and convenient for a one-to-one object-to-pixmap mapping - in
34 this case, you can store the keys as members of an object.
35
36 If two pixmaps are inserted into the cache using equal keys then the
37 last pixmap will replace the first pixmap in the cache. This follows the
38 behavior of the QHash and QCache classes.
39
40 The cache becomes full when the total size of all pixmaps in the
41 cache exceeds cacheLimit(). The initial cache limit is 10240 KB (10 MB);
42 you can change this by calling setCacheLimit() with the required value.
43 A pixmap takes roughly (\e{width} * \e{height} * \e{depth})/8 bytes of
44 memory.
45
46 The \e{Qt Quarterly} article
47 \l{http://doc.qt.io/archives/qq/qq12-qpixmapcache.html}{Optimizing
48 with QPixmapCache} explains how to use QPixmapCache to speed up
49 applications by caching the results of painting.
50
51 \note QPixmapCache is only usable from the application's main thread.
52 Access from other threads will be ignored and return failure.
53
54 \sa QCache, QPixmap
55*/
56
57static const int cache_limit_default = 10240; // 10 MB cache limit

59static inline qsizetype cost(const QPixmap &pixmap)
60{
61 // make sure to do a 64bit calculation; qsizetype might be smaller
62 const qint64 costKb = static_cast<qint64>(pixmap.width())
63 * pixmap.height() * pixmap.depth() / (8 * 1024);
64 const qint64 costMax = std::numeric_limits<qsizetype>::max();
65 // a small pixmap should have at least a cost of 1(kb)
66 return static_cast<qsizetype>(qBound(min: 1LL, val: costKb, max: costMax));
67}

237/*
238 This is supposed to cut the cache size down by about 25% in a
239 minute once the application becomes idle, to let any inserted pixmap
240 remain in the cache for some time before it becomes a candidate for
241 cleaning-up, and to not cut down the size of the cache while the
242 cache is in active use.
243
244 When the last detached pixmap has been deleted from the cache, kill the
245 timer so Qt won't keep the CPU from going into sleep mode. Currently
246 the timer is not restarted when the pixmap becomes unused, but it does
247 restart once something else is added (i.e. the cache space is actually needed).
248
249 Returns \c true if any were removed.
250*/
251bool QPMCache::flushDetachedPixmaps(bool nt)
252{
253 auto mc = maxCost();
254 const qsizetype currentTotal = totalCost();
255 if (currentTotal)
256 setMaxCost(nt ? currentTotal * 3 / 4 : currentTotal - 1);
257 setMaxCost(mc);
258 ps = totalCost();
259
260 bool any = false;
261 QHash<QString, QPixmapCache::Key>::iterator it = cacheKeys.begin();
262 while (it != cacheKeys.end()) {
263 const auto value = it.value();
264 if (value.isValid() && !contains(key: value)) {
265 releaseKey(key: value);
266 it = cacheKeys.erase(it);
267 any = true;
268 } else {
269 ++it;
270 }
271 }
272
273 return any;
274}

276void QPMCache::timerEvent(QTimerEvent *)
277{
278 bool nt = totalCost() == ps;
279 if (!flushDetachedPixmaps(nt)) {
280 killTimer(id: theid);
281 theid = 0;
282 } else if (nt != t) {
283 killTimer(id: theid);
284 theid = startTimer(interval: nt ? soon_time : flush_time);
285 t = nt;
286 }
287}

290QPixmap *QPMCache::object(const QString &key) const
291{
292 QPixmapCache::Key cacheKey = cacheKeys.value(key);
293 if (!cacheKey.d || !cacheKey.d->isValid) {
294 const_cast<QPMCache *>(this)->cacheKeys.remove(key);
295 return nullptr;
296 }
297 QPixmap *ptr = QCache<QPixmapCache::Key, QPixmapCacheEntry>::object(key: cacheKey);
298 //We didn't find the pixmap in the cache, the key is not valid anymore
299 if (!ptr) {
300 const_cast<QPMCache *>(this)->cacheKeys.remove(key);
301 }
302 return ptr;
303}

305QPixmap *QPMCache::object(const QPixmapCache::Key &key) const
306{
307 Q_ASSERT(key.isValid());
308 QPixmap *ptr = QCache<QPixmapCache::Key, QPixmapCacheEntry>::object(key);
309 //We didn't find the pixmap in the cache, the key is not valid anymore
310 if (!ptr)
311 const_cast<QPMCache *>(this)->releaseKey(key);
312 return ptr;
313}

315bool QPMCache::insert(const QString& key, const QPixmap &pixmap, int cost)
316{
317 QPixmapCache::Key &cacheKey = cacheKeys[key];
318 //If for the same key we add already a pixmap we should delete it
319 if (cacheKey.d)
320 QCache<QPixmapCache::Key, QPixmapCacheEntry>::remove(key: cacheKey);
321
322 //we create a new key the old one has been removed
323 cacheKey = createKey();
324
325 bool success = QCache<QPixmapCache::Key, QPixmapCacheEntry>::insert(key: cacheKey, object: new QPixmapCacheEntry(cacheKey, pixmap), cost);
326 if (success) {
327 if (!theid) {
328 theid = startTimer(interval: flush_time);
329 t = false;
330 }
331 } else {
332 //Insertion failed we released the new allocated key
333 cacheKeys.remove(key);
334 }
335 return success;
336}

351bool QPMCache::replace(const QPixmapCache::Key &key, const QPixmap &pixmap, int cost)
352{
353 Q_ASSERT(key.isValid());
354 //If for the same key we had already an entry so we should delete the pixmap and use the new one
355 QCache<QPixmapCache::Key, QPixmapCacheEntry>::remove(key);
356
357 QPixmapCache::Key cacheKey = createKey();
358
359 bool success = QCache<QPixmapCache::Key, QPixmapCacheEntry>::insert(key: cacheKey, object: new QPixmapCacheEntry(cacheKey, pixmap), cost);
360 if (success) {
361 if (!theid) {
362 theid = startTimer(interval: flush_time);
363 t = false;
364 }
365 const_cast<QPixmapCache::Key&>(key) = cacheKey;
366 }
367 return success;
368}

370bool QPMCache::remove(const QString &key)
371{
372 auto cacheKey = cacheKeys.constFind(key);
373 //The key was not in the cache
374 if (cacheKey == cacheKeys.constEnd())
375 return false;
376 const bool result = QCache<QPixmapCache::Key, QPixmapCacheEntry>::remove(key: cacheKey.value());
377 cacheKeys.erase(it: cacheKey);
378 return result;
379}

386void QPMCache::resizeKeyArray(int size)
387{
388 if (size <= keyArraySize || size == 0)
389 return;
390 keyArray = q_check_ptr(p: static_cast<int *>(realloc(ptr: keyArray,
391 size: size * sizeof(int))));
392 for (int i = keyArraySize; i != size; ++i)
393 keyArray[i] = i + 1;
394 keyArraySize = size;
395}
396
397QPixmapCache::Key QPMCache::createKey()
398{
399 if (freeKey == keyArraySize)
400 resizeKeyArray(size: keyArraySize ? keyArraySize << 1 : 2);
401 int id = freeKey;
402 freeKey = keyArray[id];
403 QPixmapCache::Key key;
404 QPixmapCache::KeyData *d = QPMCache::getKeyData(key: &key);
405 d->key = ++id;
406 return key;
407}
408
409void QPMCache::releaseKey(const QPixmapCache::Key &key)
410{
411 QPixmapCache::KeyData *keyData = key.d;
412 if (!keyData || keyData->key > keyArraySize || keyData->key <= 0)
413 return;
414 keyData->key--;
415 keyArray[keyData->key] = freeKey;
416 freeKey = keyData->key;
417 keyData->isValid = false;
418 keyData->key = 0;
419}
420
421void QPMCache::clear()
422{
423 free(ptr: keyArray);
424 keyArray = nullptr;
425 freeKey = 0;
426 keyArraySize = 0;
427 //Mark all keys as invalid
428 const QList<QPixmapCache::Key> keys = QCache<QPixmapCache::Key, QPixmapCacheEntry>::keys();
429 for (const auto &key : keys) {
430 if (key.d)
431 key.d->isValid = false;
432 }
433 QCache<QPixmapCache::Key, QPixmapCacheEntry>::clear();
434 // Nothing left to flush; stop the timer
435 if (theid) {
436 killTimer(id: theid);
437 theid = 0;
438 }
439}

481/*!
482 Looks for a cached pixmap associated with the given \a key in the cache.
483 If the pixmap is found, the function sets \a pixmap to that pixmap and
484 returns \c true; otherwise it leaves \a pixmap alone and returns \c false. If
485 the pixmap is not found, it means that the \a key is no longer valid,
486 so it will be released for the next insertion.
487
488 \since 4.6
489*/
490bool QPixmapCache::find(const Key &key, QPixmap *pixmap)
491{
492 if (!qt_pixmapcache_thread_test())
493 return false;
494 //The key is not valid anymore, a flush happened before probably
495 if (!key.d || !key.d->isValid)
496 return false;
497 QPixmap *ptr = pm_cache()->object(key);
498 if (ptr && pixmap)
499 *pixmap = *ptr;
500 return ptr != nullptr;
501}

503/*!
504 Inserts a copy of the pixmap \a pixmap associated with the \a key into
505 the cache.
506
507 All pixmaps inserted by the Qt library have a key starting with
508 "$qt", so your own pixmap keys should never begin "$qt".
509
510 When a pixmap is inserted and the cache is about to exceed its
511 limit, it removes pixmaps until there is enough room for the
512 pixmap to be inserted.
513
514 The oldest pixmaps (least recently accessed in the cache) are
515 deleted when more space is needed.
516
517 The function returns \c true if the object was inserted into the
518 cache; otherwise it returns \c false.
519
520 \sa setCacheLimit()
521*/
522
523bool QPixmapCache::insert(const QString &key, const QPixmap &pixmap)
524{
525 if (!qt_pixmapcache_thread_test())
526 return false;
527 return pm_cache()->insert(key, pixmap, cost: cost(pixmap));
528}
