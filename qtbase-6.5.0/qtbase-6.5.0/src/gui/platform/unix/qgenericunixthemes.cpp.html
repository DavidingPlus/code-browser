119/*!
120 * \internal
121 * The QGenericUnixThemeDBusListener class listens to the SettingChanged DBus signal
122 * and translates it into the QDbusSettingType enum.
123 * Upon construction, it logs success/failure of the DBus connection.
124 *
125 * The signal settingChanged delivers the normalized setting type and the new value as a string.
126 * It is emitted on known setting types only.
127 */
128
129class QGenericUnixThemeDBusListener : public QObject
130{
131 Q_OBJECT
132
133public:
134 QGenericUnixThemeDBusListener(const QString &service, const QString &path, const QString &interface, const QString &signal);
135
136 enum class SettingType {
137 KdeGlobalTheme,
138 KdeApplicationStyle,
139 GtkTheme,
140 Unknown
141 };
142 Q_ENUM(SettingType)
143
144 static SettingType toSettingType(const QString &location, const QString &key);
145
146private Q_SLOTS:
147 void onSettingChanged(const QString &location, const QString &key, const QDBusVariant &value);
148
149Q_SIGNALS:
150 void settingChanged(QGenericUnixThemeDBusListener::SettingType type, const QString &value);
151
152};
153
154QGenericUnixThemeDBusListener::QGenericUnixThemeDBusListener(const QString &service,
155 const QString &path, const QString &interface, const QString &signal)
156{
157 QDBusConnection dbus = QDBusConnection::sessionBus();
158 const bool dBusRunning = dbus.isConnected();
159 bool dBusSignalConnected = false;
160#define LOG service << path << interface << signal;
161
162 if (dBusRunning) {
163 qRegisterMetaType<QDBusVariant>();
164 dBusSignalConnected = dbus.connect(service, path, interface, name: signal, receiver: this,
165 SLOT(onSettingChanged(QString,QString,QDBusVariant)));
166 }
167
168 if (dBusSignalConnected) {
169 // Connection successful
170 qCDebug(lcQpaThemeDBus) << LOG;
171 } else {
172 if (dBusRunning) {
173 // DBus running, but connection failed
174 qCWarning(lcQpaThemeDBus) << "DBus connection failed:" << LOG;
175 } else {
176 // DBus not running
177 qCWarning(lcQpaThemeDBus) << "Session DBus not running.";
178 }
179 qCWarning(lcQpaThemeDBus) << "Application will not react to setting changes.\n"
180 << "Check your DBus installation.";
181 }
182#undef LOG
183}
184
185QGenericUnixThemeDBusListener::SettingType QGenericUnixThemeDBusListener::toSettingType(
186 const QString &location, const QString &key)
187{
188 if (location == QLatin1StringView("org.kde.kdeglobals.KDE")
189 && key == QLatin1StringView("widgetStyle"))
190 return SettingType::KdeApplicationStyle;
191 if (location == QLatin1StringView("org.kde.kdeglobals.General")
192 && key == QLatin1StringView("ColorScheme"))
193 return SettingType::KdeGlobalTheme;
194 if (location == QLatin1StringView("org.gnome.desktop.interface")
195 && key == QLatin1StringView("gtk-theme"))
196 return SettingType::GtkTheme;
197 return SettingType::Unknown;
198}
199
200void QGenericUnixThemeDBusListener::onSettingChanged(const QString &location, const QString &key, const QDBusVariant &value)
201{
202 const SettingType type = toSettingType(location, key);
203 if (type != SettingType::Unknown)
204 emit settingChanged(type, value: value.variant().toString());
205}

287QVariant QGenericUnixTheme::themeHint(ThemeHint hint) const
288{
289 switch (hint) {
290 case QPlatformTheme::SystemIconFallbackThemeName:
291 return QVariant(QString(QStringLiteral("hicolor")));
292 case QPlatformTheme::IconThemeSearchPaths:
293 return xdgIconThemePaths();
294 case QPlatformTheme::IconFallbackSearchPaths:
295 return iconFallbackPaths();
296 case QPlatformTheme::DialogButtonBoxButtonsHaveIcons:
297 return QVariant(true);
298 case QPlatformTheme::StyleNames: {
299 QStringList styleNames;
300 styleNames << QStringLiteral("Fusion") << QStringLiteral("Windows");
301 return QVariant(styleNames);
302 }
303 case QPlatformTheme::KeyboardScheme:
304 return QVariant(int(X11KeyboardScheme));
305 case QPlatformTheme::UiEffects:
306 return QVariant(int(HoverEffect));
307 case QPlatformTheme::MouseCursorTheme:
308 return QVariant(mouseCursorTheme());
309 case QPlatformTheme::MouseCursorSize:
310 return QVariant(mouseCursorSize());
311 default:
312 break;
313 }
314 return QPlatformTheme::themeHint(hint);
315}

407bool QKdeThemePrivate::initDbus()
408{
409 static constexpr QLatin1StringView service("");
410 static constexpr QLatin1StringView path("/org/freedesktop/portal/desktop");
411 static constexpr QLatin1StringView interface("org.freedesktop.portal.Settings");
412 static constexpr QLatin1StringView signal("SettingChanged");
413
414 dbus.reset(p: new QGenericUnixThemeDBusListener(service, path, interface, signal));
415 Q_ASSERT(dbus);
416
417 // Wrap slot in a lambda to avoid inheriting QKdeThemePrivate from QObject
418 auto wrapper = [this](QGenericUnixThemeDBusListener::SettingType type, const QString &value) {
419 settingChangedHandler(type, value);
420 };
421
422 return QObject::connect(sender: dbus.get(), signal: &QGenericUnixThemeDBusListener::settingChanged, slot: wrapper);
423}
424#endif // QT_NO_DBUS

434void QKdeThemePrivate::refresh()
435{
436 resources.clear();
437
438 toolButtonStyle = Qt::ToolButtonTextBesideIcon;
439 toolBarIconSize = 0;
440 styleNames.clear();
441 if (kdeVersion >= 5)
442 styleNames << QStringLiteral("breeze");
443 styleNames << QStringLiteral("Oxygen") << QStringLiteral("Fusion") << QStringLiteral("windows");
444 if (kdeVersion >= 5)
445 iconFallbackThemeName = iconThemeName = QStringLiteral("breeze");
446 else
447 iconFallbackThemeName = iconThemeName = QStringLiteral("oxygen");
448
449 QHash<QString, QSettings*> kdeSettings;
450
451 QPalette systemPalette = QPalette();
452 readKdeSystemPalette(kdeDirs, kdeVersion, kdeSettings, pal: &systemPalette);
453 resources.palettes[QPlatformTheme::SystemPalette] = new QPalette(systemPalette);
454 //## TODO tooltip color
455
456 const QVariant styleValue = readKdeSetting(QStringLiteral("widgetStyle"), kdeDirs, kdeVersion, kdeSettings);
457 if (styleValue.isValid()) {
458 const QString style = styleValue.toString();
459 if (style != styleNames.front())
460 styleNames.push_front(t: style);
461 }
462
463 const QVariant colorScheme = readKdeSetting(QStringLiteral("ColorScheme"), kdeDirs,
464 kdeVersion, kdeSettings);
465
466 if (colorScheme.isValid())
467 updateColorScheme(themeName: colorScheme.toString());
468 else
469 m_colorScheme = Qt::ColorScheme::Unknown;
470
471 const QVariant singleClickValue = readKdeSetting(QStringLiteral("KDE/SingleClick"), kdeDirs, kdeVersion, kdeSettings);
472 if (singleClickValue.isValid())
473 singleClick = singleClickValue.toBool();
474
475 const QVariant showIconsOnPushButtonsValue = readKdeSetting(QStringLiteral("KDE/ShowIconsOnPushButtons"), kdeDirs, kdeVersion, kdeSettings);
476 if (showIconsOnPushButtonsValue.isValid())
477 showIconsOnPushButtons = showIconsOnPushButtonsValue.toBool();
478
479 const QVariant themeValue = readKdeSetting(QStringLiteral("Icons/Theme"), kdeDirs, kdeVersion, kdeSettings);
480 if (themeValue.isValid())
481 iconThemeName = themeValue.toString();
482
483 const QVariant toolBarIconSizeValue = readKdeSetting(QStringLiteral("ToolbarIcons/Size"), kdeDirs, kdeVersion, kdeSettings);
484 if (toolBarIconSizeValue.isValid())
485 toolBarIconSize = toolBarIconSizeValue.toInt();
486
487 const QVariant toolbarStyleValue = readKdeSetting(QStringLiteral("Toolbar style/ToolButtonStyle"), kdeDirs, kdeVersion, kdeSettings);
488 if (toolbarStyleValue.isValid()) {
489 const QString toolBarStyle = toolbarStyleValue.toString();
490 if (toolBarStyle == "TextBesideIcon"_L1)
491 toolButtonStyle = Qt::ToolButtonTextBesideIcon;
492 else if (toolBarStyle == "TextOnly"_L1)
493 toolButtonStyle = Qt::ToolButtonTextOnly;
494 else if (toolBarStyle == "TextUnderIcon"_L1)
495 toolButtonStyle = Qt::ToolButtonTextUnderIcon;
496 }
497
498 const QVariant wheelScrollLinesValue = readKdeSetting(QStringLiteral("KDE/WheelScrollLines"), kdeDirs, kdeVersion, kdeSettings);
499 if (wheelScrollLinesValue.isValid())
500 wheelScrollLines = wheelScrollLinesValue.toInt();
501
502 const QVariant doubleClickIntervalValue = readKdeSetting(QStringLiteral("KDE/DoubleClickInterval"), kdeDirs, kdeVersion, kdeSettings);
503 if (doubleClickIntervalValue.isValid())
504 doubleClickInterval = doubleClickIntervalValue.toInt();
505
506 const QVariant startDragDistValue = readKdeSetting(QStringLiteral("KDE/StartDragDist"), kdeDirs, kdeVersion, kdeSettings);
507 if (startDragDistValue.isValid())
508 startDragDist = startDragDistValue.toInt();
509
510 const QVariant startDragTimeValue = readKdeSetting(QStringLiteral("KDE/StartDragTime"), kdeDirs, kdeVersion, kdeSettings);
511 if (startDragTimeValue.isValid())
512 startDragTime = startDragTimeValue.toInt();
513
514 const QVariant cursorBlinkRateValue = readKdeSetting(QStringLiteral("KDE/CursorBlinkRate"), kdeDirs, kdeVersion, kdeSettings);
515 if (cursorBlinkRateValue.isValid()) {
516 cursorBlinkRate = cursorBlinkRateValue.toInt();
517 cursorBlinkRate = cursorBlinkRate > 0 ? qBound(min: 200, val: cursorBlinkRate, max: 2000) : 0;
518 }
519
520 // Read system font, ignore 'smallestReadableFont'
521 if (QFont *systemFont = kdeFont(fontValue: readKdeSetting(QStringLiteral("font"), kdeDirs, kdeVersion, kdeSettings)))
522 resources.fonts[QPlatformTheme::SystemFont] = systemFont;
523 else
524 resources.fonts[QPlatformTheme::SystemFont] = new QFont(QLatin1StringView(defaultSystemFontNameC), defaultSystemFontSize);
525
526 if (QFont *fixedFont = kdeFont(fontValue: readKdeSetting(QStringLiteral("fixed"), kdeDirs, kdeVersion, kdeSettings))) {
527 resources.fonts[QPlatformTheme::FixedFont] = fixedFont;
528 } else {
529 fixedFont = new QFont(QLatin1StringView(defaultFixedFontNameC), defaultSystemFontSize);
530 fixedFont->setStyleHint(QFont::TypeWriter);
531 resources.fonts[QPlatformTheme::FixedFont] = fixedFont;
532 }
533
534 if (QFont *menuFont = kdeFont(fontValue: readKdeSetting(QStringLiteral("menuFont"), kdeDirs, kdeVersion, kdeSettings))) {
535 resources.fonts[QPlatformTheme::MenuFont] = menuFont;
536 resources.fonts[QPlatformTheme::MenuBarFont] = new QFont(*menuFont);
537 }
538
539 if (QFont *toolBarFont = kdeFont(fontValue: readKdeSetting(QStringLiteral("toolBarFont"), kdeDirs, kdeVersion, kdeSettings)))
540 resources.fonts[QPlatformTheme::ToolButtonFont] = toolBarFont;
541
542 QWindowSystemInterface::handleThemeChange();
543
544 qCDebug(lcQpaFonts) << "default fonts: system" << resources.fonts[QPlatformTheme::SystemFont]
545 << "fixed" << resources.fonts[QPlatformTheme::FixedFont];
546 qDeleteAll(c: kdeSettings);
547}

549QVariant QKdeThemePrivate::readKdeSetting(const QString &key, const QStringList &kdeDirs, int kdeVersion, QHash<QString, QSettings*> &kdeSettings)
550{
551 for (const QString &kdeDir : kdeDirs) {
552 QSettings *settings = kdeSettings.value(key: kdeDir);
553 if (!settings) {
554 const QString kdeGlobalsPath = kdeGlobals(kdeDir, kdeVersion);
555 if (QFileInfo(kdeGlobalsPath).isReadable()) {
556 settings = new QSettings(kdeGlobalsPath, QSettings::IniFormat);
557 kdeSettings.insert(key: kdeDir, value: settings);
558 }
559 }
560 if (settings) {
561 const QVariant value = settings->value(key);
562 if (value.isValid())
563 return value;
564 }
565 }
566 return QVariant();
567}

569// Reads the color from the KDE configuration, and store it in the
570// palette with the given color role if found.
571static inline bool kdeColor(QPalette *pal, QPalette::ColorRole role, const QVariant &value)
572{
573 if (!value.isValid())
574 return false;
575 const QStringList values = value.toStringList();
576 if (values.size() != 3)
577 return false;
578 pal->setBrush(acr: role, abrush: QColor(values.at(i: 0).toInt(), values.at(i: 1).toInt(), values.at(i: 2).toInt()));
579 return true;
580}

582void QKdeThemePrivate::readKdeSystemPalette(const QStringList &kdeDirs, int kdeVersion, QHash<QString, QSettings*> &kdeSettings, QPalette *pal)
583{
584 if (!kdeColor(pal, role: QPalette::Button, value: readKdeSetting(QStringLiteral("Colors:Button/BackgroundNormal"), kdeDirs, kdeVersion, kdeSettings))) {
585 // kcolorscheme.cpp: SetDefaultColors
586 const QColor defaultWindowBackground(214, 210, 208);
587 const QColor defaultButtonBackground(223, 220, 217);
588 *pal = QPalette(defaultButtonBackground, defaultWindowBackground);
589 return;
590 }
591
592 kdeColor(pal, role: QPalette::Window, value: readKdeSetting(QStringLiteral("Colors:Window/BackgroundNormal"), kdeDirs, kdeVersion, kdeSettings));
593 kdeColor(pal, role: QPalette::Text, value: readKdeSetting(QStringLiteral("Colors:View/ForegroundNormal"), kdeDirs, kdeVersion, kdeSettings));
594 kdeColor(pal, role: QPalette::WindowText, value: readKdeSetting(QStringLiteral("Colors:Window/ForegroundNormal"), kdeDirs, kdeVersion, kdeSettings));
595 kdeColor(pal, role: QPalette::Base, value: readKdeSetting(QStringLiteral("Colors:View/BackgroundNormal"), kdeDirs, kdeVersion, kdeSettings));
596 kdeColor(pal, role: QPalette::Highlight, value: readKdeSetting(QStringLiteral("Colors:Selection/BackgroundNormal"), kdeDirs, kdeVersion, kdeSettings));
597 kdeColor(pal, role: QPalette::HighlightedText, value: readKdeSetting(QStringLiteral("Colors:Selection/ForegroundNormal"), kdeDirs, kdeVersion, kdeSettings));
598 kdeColor(pal, role: QPalette::AlternateBase, value: readKdeSetting(QStringLiteral("Colors:View/BackgroundAlternate"), kdeDirs, kdeVersion, kdeSettings));
599 kdeColor(pal, role: QPalette::ButtonText, value: readKdeSetting(QStringLiteral("Colors:Button/ForegroundNormal"), kdeDirs, kdeVersion, kdeSettings));
600 kdeColor(pal, role: QPalette::Link, value: readKdeSetting(QStringLiteral("Colors:View/ForegroundLink"), kdeDirs, kdeVersion, kdeSettings));
601 kdeColor(pal, role: QPalette::LinkVisited, value: readKdeSetting(QStringLiteral("Colors:View/ForegroundVisited"), kdeDirs, kdeVersion, kdeSettings));
602 kdeColor(pal, role: QPalette::ToolTipBase, value: readKdeSetting(QStringLiteral("Colors:Tooltip/BackgroundNormal"), kdeDirs, kdeVersion, kdeSettings));
603 kdeColor(pal, role: QPalette::ToolTipText, value: readKdeSetting(QStringLiteral("Colors:Tooltip/ForegroundNormal"), kdeDirs, kdeVersion, kdeSettings));
604
605 // The above code sets _all_ color roles to "normal" colors. In KDE, the disabled
606 // color roles are calculated by applying various effects described in kdeglobals.
607 // We use a bit simpler approach here, similar logic than in qt_palette_from_color().
608 const QColor button = pal->color(cr: QPalette::Button);
609 int h, s, v;
610 button.getHsv(h: &h, s: &s, v: &v);
611

757/*!
758 \internal
759 \brief QKdeTheme::setColorScheme - guess and set appearance for unix themes.
760 KDE themes do not have an appearance property.
761 The key words "dark" or "light" should be part of the theme name.
762 This is, however, not a mandatory convention.
763
764 If \param themeName contains a key word, the respective appearance is set.
765 If it doesn't, the appearance is heuristically determined by comparing text and base color
766 of the system palette.
767 */
768void QKdeThemePrivate::updateColorScheme(const QString &themeName)
769{
770 if (themeName.contains(s: QLatin1StringView("light"), cs: Qt::CaseInsensitive)) {
771 m_colorScheme = Qt::ColorScheme::Light;
772 return;
773 }
774 if (themeName.contains(s: QLatin1StringView("dark"), cs: Qt::CaseInsensitive)) {
775 m_colorScheme = Qt::ColorScheme::Dark;
776 return;
777 }
778
779 if (systemPalette) {
780 if (systemPalette->text().color().lightness() < systemPalette->base().color().lightness()) {
781 m_colorScheme = Qt::ColorScheme::Light;
782 return;
783 }
784 if (systemPalette->text().color().lightness() > systemPalette->base().color().lightness()) {
785 m_colorScheme = Qt::ColorScheme::Dark;
786 return;
787 }
788 }
789
790 m_colorScheme = Qt::ColorScheme::Unknown;
791}

806QPlatformTheme *QKdeTheme::createKdeTheme()
807{
808 const QByteArray kdeVersionBA = qgetenv(varName: "KDE_SESSION_VERSION");
809 const int kdeVersion = kdeVersionBA.toInt();
810 if (kdeVersion < 4)
811 return nullptr;
812
813 if (kdeVersion > 4)
814 // Plasma 5 follows XDG spec
815 // but uses the same config file format:
816 return new QKdeTheme(QStandardPaths::standardLocations(type: QStandardPaths::GenericConfigLocation), kdeVersion);
817
818 // Determine KDE prefixes in the following priority order:
819 // - KDEHOME and KDEDIRS environment variables
820 // - ~/.kde(<version>)
821 // - read prefixes from /etc/kde<version>rc
822 // - fallback to /etc/kde<version>
823
824 QStringList kdeDirs;
825 const QString kdeHomePathVar = QFile::decodeName(localFileName: qgetenv(varName: "KDEHOME"));
826 if (!kdeHomePathVar.isEmpty())
827 kdeDirs += kdeHomePathVar;
828
829 const QString kdeDirsVar = QFile::decodeName(localFileName: qgetenv(varName: "KDEDIRS"));
830 if (!kdeDirsVar.isEmpty())
831 kdeDirs += kdeDirsVar.split(sep: u':', behavior: Qt::SkipEmptyParts);
832
833 const QString kdeVersionHomePath = QDir::homePath() + "/.kde"_L1 + QLatin1StringView(kdeVersionBA);
834 if (QFileInfo(kdeVersionHomePath).isDir())
835 kdeDirs += kdeVersionHomePath;
836
837 const QString kdeHomePath = QDir::homePath() + "/.kde"_L1;
838 if (QFileInfo(kdeHomePath).isDir())
839 kdeDirs += kdeHomePath;
840
841 const QString kdeRcPath = "/etc/kde"_L1 + QLatin1StringView(kdeVersionBA) + "rc"_L1;
842 if (QFileInfo(kdeRcPath).isReadable()) {
843 QSettings kdeSettings(kdeRcPath, QSettings::IniFormat);
844 kdeSettings.beginGroup(QStringLiteral("Directories-default"));
845 kdeDirs += kdeSettings.value(QStringLiteral("prefixes")).toStringList();
846 }
847
848 const QString kdeVersionPrefix = "/etc/kde"_L1 + QLatin1StringView(kdeVersionBA);
849 if (QFileInfo(kdeVersionPrefix).isDir())
850 kdeDirs += kdeVersionPrefix;
851
852 kdeDirs.removeDuplicates();
853 if (kdeDirs.isEmpty()) {
854 qWarning(msg: "Unable to determine KDE dirs");
855 return nullptr;
856 }
857
858 return new QKdeTheme(kdeDirs, kdeVersion);
859}

1109QStringList QGenericUnixTheme::themeNames()
1110{
1111 QStringList result;
1112 if (QGuiApplication::desktopSettingsAware()) {
1113 const QByteArray desktopEnvironment = QGuiApplicationPrivate::platformIntegration()->services()->desktopEnvironment();
1114 QList<QByteArray> gtkBasedEnvironments;
1115 gtkBasedEnvironments << "GNOME"
1116 << "X-CINNAMON"
1117 << "UNITY"
1118 << "MATE"
1119 << "XFCE"
1120 << "LXDE";
1121 const QList<QByteArray> desktopNames = desktopEnvironment.split(sep: ':');
1122 for (const QByteArray &desktopName : desktopNames) {
1123 if (desktopEnvironment == "KDE") {
1124#if QT_CONFIG(settings)
1125 result.push_back(t: QLatin1StringView(QKdeTheme::name));
1126#endif
1127 } else if (gtkBasedEnvironments.contains(t: desktopName)) {
1128 // prefer the GTK3 theme implementation with native dialogs etc.
1129 result.push_back(QStringLiteral("gtk3"));
1130 // fallback to the generic Gnome theme if loading the GTK3 theme fails
1131 result.push_back(t: QLatin1StringView(QGnomeTheme::name));
1132 } else {
1133 // unknown, but lowercase the name (our standard practice) and
1134 // remove any "x-" prefix
1135 QString s = QString::fromLatin1(ba: desktopName.toLower());
1136 result.push_back(t: s.startsWith(s: "x-"_L1) ? s.mid(position: 2) : s);
1137 }
1138 }
1139 } // desktopSettingsAware
1140 result.append(t: QLatin1StringView(QGenericUnixTheme::name));
1141 return result;
1142}
