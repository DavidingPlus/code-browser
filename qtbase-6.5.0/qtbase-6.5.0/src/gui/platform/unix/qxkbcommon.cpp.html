26typedef struct xkb2qt
27{
28 unsigned int xkb;
29 unsigned int qt;
30
31 constexpr bool operator <=(const xkb2qt &that) const noexcept
32 {
33 return xkb <= that.xkb;
34 }
35
36 constexpr bool operator <(const xkb2qt &that) const noexcept
37 {
38 return xkb < that.xkb;
39 }
40} xkb2qt_t;
41
42template<std::size_t Xkb, std::size_t Qt>
43struct Xkb2Qt
44{
45 using Type = xkb2qt_t;
46 static constexpr Type data() noexcept { return Type{.xkb: Xkb, .qt: Qt}; }
47};

429QList<xkb_keysym_t> QXkbCommon::toKeysym(QKeyEvent *event)
430{
431 QList<xkb_keysym_t> keysyms;
432 int qtKey = event->key();
433
434 if (qtKey >= Qt::Key_F1 && qtKey <= Qt::Key_F35) {
435 keysyms.append(XKB_KEY_F1 + (qtKey - Qt::Key_F1));
436 } else if (event->modifiers() & Qt::KeypadModifier) {
437 if (qtKey >= Qt::Key_0 && qtKey <= Qt::Key_9)
438 keysyms.append(XKB_KEY_KP_0 + (qtKey - Qt::Key_0));
439 } else if (isLatin1(sym: qtKey) && event->text().isUpper()) {
440 keysyms.append(t: qtKey);
441 }
442
443 if (!keysyms.isEmpty())
444 return keysyms;
445
446 // check if we have a direct mapping
447 auto it = std::find_if(first: KeyTbl.cbegin(), last: KeyTbl.cend(), pred: [&qtKey](xkb2qt_t elem) {
448 return elem.qt == static_cast<uint>(qtKey);
449 });
450 if (it != KeyTbl.end()) {
451 keysyms.append(t: it->xkb);
452 return keysyms;
453 }
454
455 QList<uint> ucs4;
456 if (event->text().isEmpty())
457 ucs4.append(t: qtKey);
458 else
459 ucs4 = event->text().toUcs4();
460
461 // From libxkbcommon keysym-utf.c:
462 // "We allow to represent any UCS character in the range U-00000000 to
463 // U-00FFFFFF by a keysym value in the range 0x01000000 to 0x01ffffff."
464 for (uint utf32 : std::as_const(t&: ucs4))
465 keysyms.append(t: utf32 | 0x01000000);
466
467 return keysyms;
468}

475int QXkbCommon::keysymToQtKey(xkb_keysym_t keysym, Qt::KeyboardModifiers modifiers,
476 xkb_state *state, xkb_keycode_t code,
477 bool superAsMeta, bool hyperAsMeta)
478{
479 // Note 1: All standard key sequences on linux (as defined in platform theme)
480 // that use a latin character also contain a control modifier, which is why
481 // checking for Qt::ControlModifier is sufficient here. It is possible to
482 // override QPlatformTheme::keyBindings() and provide custom sequences for
483 // QKeySequence::StandardKey. Custom sequences probably should respect this
484 // convention (alternatively, we could test against other modifiers here).
485 // Note 2: The possibleKeys() shorcut mechanism is not affected by this value
486 // adjustment and does its own thing.
487 if (modifiers & Qt::ControlModifier) {
488 // With standard shortcuts we should prefer a latin character, this is
489 // for checks like "some qkeyevent == QKeySequence::Copy" to work even
490 // when using for example 'russian' keyboard layout.
491 if (!QXkbCommon::isLatin1(sym: keysym)) {
492 xkb_keysym_t latinKeysym = QXkbCommon::lookupLatinKeysym(state, keycode: code);
493 if (latinKeysym != XKB_KEY_NoSymbol)
494 keysym = latinKeysym;
495 }
496 }
497
498 return keysymToQtKey_internal(keysym, modifiers, state, code, superAsMeta, hyperAsMeta);
499}
500
501static int keysymToQtKey_internal(xkb_keysym_t keysym, Qt::KeyboardModifiers modifiers,
502 xkb_state *state, xkb_keycode_t code,
503 bool superAsMeta, bool hyperAsMeta)
504{
505 int qtKey = 0;
506
507 // lookup from direct mapping
508 if (keysym >= XKB_KEY_F1 && keysym <= XKB_KEY_F35) {
509 // function keys
510 qtKey = Qt::Key_F1 + (keysym - XKB_KEY_F1);
511 } else if (keysym >= XKB_KEY_KP_0 && keysym <= XKB_KEY_KP_9) {
512 // numeric keypad keys
513 qtKey = Qt::Key_0 + (keysym - XKB_KEY_KP_0);
514 } else if (QXkbCommon::isLatin1(sym: keysym)) {
515 // Upper-case first, since Qt::Keys are defined in terms of their
516 // upper-case versions.
517 qtKey = QXkbCommon::qxkbcommon_xkb_keysym_to_upper(ks: keysym);
518 // Upper-casing a Latin1 character might move it out of Latin1 range,
519 // for example U+00B5 MICRO SIGN, which upper-case equivalent is
520 // U+039C GREEK CAPITAL LETTER MU. If that's the case, then map the
521 // original lower-case character.
522 if (!QXkbCommon::isLatin1(sym: qtKey))
523 qtKey = keysym;
524 } else {
525 // check if we have a direct mapping
526 xkb2qt_t searchKey{.xkb: keysym, .qt: 0};
527 auto it = std::lower_bound(KeyTbl.cbegin(), KeyTbl.cend(), searchKey);
528 if (it != KeyTbl.end() && !(searchKey < *it))
529 qtKey = it->qt;
530
531 // translate Super/Hyper keys to Meta if we're using them as the MetaModifier
532 if (superAsMeta && (qtKey == Qt::Key_Super_L || qtKey == Qt::Key_Super_R))
533 qtKey = Qt::Key_Meta;
534 if (hyperAsMeta && (qtKey == Qt::Key_Hyper_L || qtKey == Qt::Key_Hyper_R))
535 qtKey = Qt::Key_Meta;
536 }
537
538 if (qtKey)
539 return qtKey;
540
541 // lookup from unicode
542 QString text;
543 if (!state || modifiers & Qt::ControlModifier) {
544 // Control modifier changes the text to ASCII control character, therefore we
545 // can't use this text to map keysym to a qt key. We can use the same keysym
546 // (it is not affectd by transformation) to obtain untransformed text. For details
547 // see "Appendix A. Default Symbol Transformations" in the XKB specification.
548 text = QXkbCommon::lookupStringNoKeysymTransformations(keysym);
549 } else {
550 text = QXkbCommon::lookupString(state, code);
551 }
552 if (!text.isEmpty()) {
553 if (text.unicode()->isDigit()) {
554 // Ensures that also non-latin digits are mapped to corresponding qt keys,
555 // e.g CTRL + (arabic two), is mapped to CTRL + Qt::Key_2.
556 qtKey = Qt::Key_0 + text.unicode()->digitValue();
557 } else {
558 text = text.toUpper();
559 QStringIterator i(text);
560 qtKey = i.next(invalidAs: 0);
561 }
562 }
563
564 return qtKey;
565}

567Qt::KeyboardModifiers QXkbCommon::modifiers(struct xkb_state *state)
568{
569 Qt::KeyboardModifiers modifiers = Qt::NoModifier;
570
571 if (xkb_state_mod_name_is_active(state, XKB_MOD_NAME_CTRL, type: XKB_STATE_MODS_EFFECTIVE) > 0)
572 modifiers |= Qt::ControlModifier;
573 if (xkb_state_mod_name_is_active(state, XKB_MOD_NAME_ALT, type: XKB_STATE_MODS_EFFECTIVE) > 0)
574 modifiers |= Qt::AltModifier;
575 if (xkb_state_mod_name_is_active(state, XKB_MOD_NAME_SHIFT, type: XKB_STATE_MODS_EFFECTIVE) > 0)
576 modifiers |= Qt::ShiftModifier;
577 if (xkb_state_mod_name_is_active(state, XKB_MOD_NAME_LOGO, type: XKB_STATE_MODS_EFFECTIVE) > 0)
578 modifiers |= Qt::MetaModifier;
579
580 return modifiers;
581}

583// Possible modifier states.
584static const Qt::KeyboardModifiers ModsTbl[] = {
585 Qt::NoModifier, // 0
586 Qt::ShiftModifier, // 1
587 Qt::ControlModifier, // 2
588 Qt::ControlModifier | Qt::ShiftModifier, // 3
589 Qt::AltModifier, // 4
590 Qt::AltModifier | Qt::ShiftModifier, // 5
591 Qt::AltModifier | Qt::ControlModifier, // 6
592 Qt::AltModifier | Qt::ShiftModifier | Qt::ControlModifier, // 7
593 Qt::NoModifier // Fall-back to raw Key_*, for non-latin1 kb layouts
594};

596QList<int> QXkbCommon::possibleKeys(xkb_state *state, const QKeyEvent *event,
597 bool superAsMeta, bool hyperAsMeta)
598{
599 QList<int> result;
600 quint32 keycode = event->nativeScanCode();
601 if (!keycode)
602 return result;
603
604 Qt::KeyboardModifiers modifiers = event->modifiers();
605 xkb_keymap *keymap = xkb_state_get_keymap(state);
606 // turn off the modifier bits which doesn't participate in shortcuts
607 Qt::KeyboardModifiers notNeeded = Qt::KeypadModifier | Qt::GroupSwitchModifier;
608 modifiers &= ~notNeeded;
609 // create a fresh kb state and test against the relevant modifier combinations
610 ScopedXKBState scopedXkbQueryState(xkb_state_new(keymap));
611 xkb_state *queryState = scopedXkbQueryState.get();
612 if (!queryState) {
613 qCWarning(lcXkbcommon) << Q_FUNC_INFO << "failed to compile xkb keymap";
614 return result;
615 }
616 // get kb state from the master state and update the temporary state
617 xkb_layout_index_t lockedLayout = xkb_state_serialize_layout(state, components: XKB_STATE_LAYOUT_LOCKED);
618 xkb_mod_mask_t latchedMods = xkb_state_serialize_mods(state, components: XKB_STATE_MODS_LATCHED);
619 xkb_mod_mask_t lockedMods = xkb_state_serialize_mods(state, components: XKB_STATE_MODS_LOCKED);
620 xkb_mod_mask_t depressedMods = xkb_state_serialize_mods(state, components: XKB_STATE_MODS_DEPRESSED);
621 xkb_state_update_mask(state: queryState, depressed_mods: depressedMods, latched_mods: latchedMods, locked_mods: lockedMods, depressed_layout: 0, latched_layout: 0, locked_layout: lockedLayout);
622 // handle shortcuts for level three and above
623 xkb_layout_index_t layoutIndex = xkb_state_key_get_layout(state: queryState, key: keycode);
624 xkb_level_index_t levelIndex = 0;
625 if (layoutIndex != XKB_LAYOUT_INVALID) {
626 levelIndex = xkb_state_key_get_level(state: queryState, key: keycode, layout: layoutIndex);
627 if (levelIndex == XKB_LEVEL_INVALID)
628 levelIndex = 0;
629 }
630 if (levelIndex <= 1)
631 xkb_state_update_mask(state: queryState, depressed_mods: 0, latched_mods: latchedMods, locked_mods: lockedMods, depressed_layout: 0, latched_layout: 0, locked_layout: lockedLayout);
632
633 xkb_keysym_t sym = xkb_state_key_get_one_sym(state: queryState, key: keycode);
634 if (sym == XKB_KEY_NoSymbol)
635 return result;
636
637 int baseQtKey = keysymToQtKey_internal(keysym: sym, modifiers, state: queryState, code: keycode, superAsMeta, hyperAsMeta);
638 if (baseQtKey)
639 result += (baseQtKey + int(modifiers));
640
641 xkb_mod_index_t shiftMod = xkb_keymap_mod_get_index(keymap, name: "Shift");
642 xkb_mod_index_t altMod = xkb_keymap_mod_get_index(keymap, name: "Alt");
643 xkb_mod_index_t controlMod = xkb_keymap_mod_get_index(keymap, name: "Control");
644 xkb_mod_index_t metaMod = xkb_keymap_mod_get_index(keymap, name: "Meta");
645
646 Q_ASSERT(shiftMod < 32);
647 Q_ASSERT(altMod < 32);
648 Q_ASSERT(controlMod < 32);
649
650 xkb_mod_mask_t depressed;
651 int qtKey = 0;
652 // obtain a list of possible shortcuts for the given key event
653 for (uint i = 1; i < sizeof(ModsTbl) / sizeof(*ModsTbl) ; ++i) {
654 Qt::KeyboardModifiers neededMods = ModsTbl[i];
655 if ((modifiers & neededMods) == neededMods) {
656 if (i == 8) {
657 if (isLatin1(sym: baseQtKey))
658 continue;
659 // add a latin key as a fall back key
660 sym = lookupLatinKeysym(state, keycode);
661 } else {
662 depressed = 0;
663 if (neededMods & Qt::AltModifier)
664 depressed |= (1 << altMod);
665 if (neededMods & Qt::ShiftModifier)
666 depressed |= (1 << shiftMod);
667 if (neededMods & Qt::ControlModifier)
668 depressed |= (1 << controlMod);
669 if (metaMod < 32 && neededMods & Qt::MetaModifier)
670 depressed |= (1 << metaMod);
671 xkb_state_update_mask(state: queryState, depressed_mods: depressed, latched_mods: latchedMods, locked_mods: lockedMods, depressed_layout: 0, latched_layout: 0, locked_layout: lockedLayout);
672 sym = xkb_state_key_get_one_sym(state: queryState, key: keycode);
673 }
674 if (sym == XKB_KEY_NoSymbol)
675 continue;
676
677 Qt::KeyboardModifiers mods = modifiers & ~neededMods;
678 qtKey = keysymToQtKey_internal(keysym: sym, modifiers: mods, state: queryState, code: keycode, superAsMeta, hyperAsMeta);
679 if (!qtKey || qtKey == baseQtKey)
680 continue;
681
682 // catch only more specific shortcuts, i.e. Ctrl+Shift+= also generates Ctrl++ and +,
683 // but Ctrl++ is more specific than +, so we should skip the last one
684 bool ambiguous = false;
685 for (int shortcut : std::as_const(t&: result)) {
686 if (int(shortcut & ~Qt::KeyboardModifierMask) == qtKey && (shortcut & mods) == mods) {
687 ambiguous = true;
688 break;
689 }
690 }
691 if (ambiguous)
692 continue;
693
694 result += (qtKey + int(mods));
695 }
696 }
697
698 return result;
699}

701void QXkbCommon::verifyHasLatinLayout(xkb_keymap *keymap)
702{
703 const xkb_layout_index_t layoutCount = xkb_keymap_num_layouts(keymap);
704 const xkb_keycode_t minKeycode = xkb_keymap_min_keycode(keymap);
705 const xkb_keycode_t maxKeycode = xkb_keymap_max_keycode(keymap);
706
707 const xkb_keysym_t *keysyms = nullptr;
708 int nrLatinKeys = 0;
709 for (xkb_layout_index_t layout = 0; layout < layoutCount; ++layout) {
710 for (xkb_keycode_t code = minKeycode; code < maxKeycode; ++code) {
711 xkb_keymap_key_get_syms_by_level(keymap, key: code, layout, level: 0, syms_out: &keysyms);
712 if (keysyms && isLatin1(sym: keysyms[0]))
713 nrLatinKeys++;
714 if (nrLatinKeys > 10) // arbitrarily chosen threshold
715 return;
716 }
717 }
718 // This means that lookupLatinKeysym() will not find anything and latin
719 // key shortcuts might not work. This is a bug in the affected desktop
720 // environment. Usually can be solved via system settings by adding e.g. 'us'
721 // layout to the list of selected layouts, or by using command line, "setxkbmap
722 // -layout rus,en". The position of latin key based layout in the list of the
723 // selected layouts is irrelevant. Properly functioning desktop environments
724 // handle this behind the scenes, even if no latin key based layout has been
725 // explicitly listed in the selected layouts.
726 qCDebug(lcXkbcommon, "no keyboard layouts with latin keys present");
727}

729xkb_keysym_t QXkbCommon::lookupLatinKeysym(xkb_state *state, xkb_keycode_t keycode)
730{
731 xkb_layout_index_t layout;
732 xkb_keysym_t sym = XKB_KEY_NoSymbol;
733 xkb_keymap *keymap = xkb_state_get_keymap(state);
734 const xkb_layout_index_t layoutCount = xkb_keymap_num_layouts_for_key(keymap, key: keycode);
735 const xkb_layout_index_t currentLayout = xkb_state_key_get_layout(state, key: keycode);
736 // Look at user layouts in the order in which they are defined in system
737 // settings to find a latin keysym.
738 for (layout = 0; layout < layoutCount; ++layout) {
739 if (layout == currentLayout)
740 continue;
741 const xkb_keysym_t *syms = nullptr;
742 xkb_level_index_t level = xkb_state_key_get_level(state, key: keycode, layout);
743 if (xkb_keymap_key_get_syms_by_level(keymap, key: keycode, layout, level, syms_out: &syms) != 1)
744 continue;
745 if (isLatin1(sym: syms[0])) {
746 sym = syms[0];
747 break;
748 }
749 }
750
751 if (sym == XKB_KEY_NoSymbol)
752 return sym;
753
754 xkb_mod_mask_t latchedMods = xkb_state_serialize_mods(state, components: XKB_STATE_MODS_LATCHED);
755 xkb_mod_mask_t lockedMods = xkb_state_serialize_mods(state, components: XKB_STATE_MODS_LOCKED);
756
757 // Check for uniqueness, consider the following setup:
758 // setxkbmap -layout us,ru,us -variant dvorak,, -option 'grp:ctrl_alt_toggle' (set 'ru' as active).
759 // In this setup, the user would expect to trigger a ctrl+q shortcut by pressing ctrl+<physical x key>,
760 // because "US dvorak" is higher up in the layout settings list. This check verifies that an obtained
761 // 'sym' can not be acquired by any other layout higher up in the user's layout list. If it can be acquired
762 // then the obtained key is not unique. This prevents ctrl+<physical q key> from generating a ctrl+q
763 // shortcut in the above described setup. We don't want ctrl+<physical x key> and ctrl+<physical q key> to
764 // generate the same shortcut event in this case.
765 const xkb_keycode_t minKeycode = xkb_keymap_min_keycode(keymap);
766 const xkb_keycode_t maxKeycode = xkb_keymap_max_keycode(keymap);
767 ScopedXKBState queryState(xkb_state_new(keymap));
768 for (xkb_layout_index_t prevLayout = 0; prevLayout < layout; ++prevLayout) {
769 xkb_state_update_mask(state: queryState.get(), depressed_mods: 0, latched_mods: latchedMods, locked_mods: lockedMods, depressed_layout: 0, latched_layout: 0, locked_layout: prevLayout);
770 for (xkb_keycode_t code = minKeycode; code < maxKeycode; ++code) {
771 xkb_keysym_t prevSym = xkb_state_key_get_one_sym(state: queryState.get(), key: code);
772 if (prevSym == sym) {
773 sym = XKB_KEY_NoSymbol;
774 break;
775 }
776 }
777 }
778
779 return sym;
780}

782void QXkbCommon::setXkbContext(QPlatformInputContext *inputContext, struct xkb_context *context)
783{
784 if (!inputContext || !context)
785 return;
786
787 const char *const inputContextClassName = "QComposeInputContext";
788 const char *const normalizedSignature = "setXkbContext(xkb_context*)";
789
790 if (inputContext->objectName() != QLatin1StringView(inputContextClassName))
791 return;
792
793 static const QMetaMethod setXkbContext = [&]() {
794 int methodIndex = inputContext->metaObject()->indexOfMethod(method: normalizedSignature);
795 QMetaMethod method = inputContext->metaObject()->method(index: methodIndex);
796 Q_ASSERT(method.isValid());
797 if (!method.isValid())
798 qCWarning(lcXkbcommon) << normalizedSignature << "not found on" << inputContextClassName;
799 return method;
800 }();

802 if (!setXkbContext.isValid())

805 setXkbContext.invoke(obj: inputContext, c: Qt::DirectConnection, Q_ARG(struct xkb_context*, context));
