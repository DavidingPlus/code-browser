49bool QFontDef::exactMatch(const QFontDef &other) const
50{
51 /*
52 QFontDef comparison is more complicated than just simple
53 per-member comparisons.
54
55 When comparing point/pixel sizes, either point or pixelsize
56 could be -1. in This case we have to compare the non negative
57 size value.
58
59 This test will fail if the point-sizes differ by 1/2 point or
60 more or they do not round to the same value. We have to do this
61 since our API still uses 'int' point-sizes in the API, but store
62 deci-point-sizes internally.
63
64 To compare the family members, we need to parse the font names
65 and compare the family/foundry strings separately. This allows
66 us to compare e.g. "Helvetica" and "Helvetica [Adobe]" with
67 positive results.
68 */
69 if (pixelSize != -1 && other.pixelSize != -1) {
70 if (pixelSize != other.pixelSize)
71 return false;
72 } else if (pointSize != -1 && other.pointSize != -1) {
73 if (pointSize != other.pointSize)
74 return false;
75 } else {
76 return false;
77 }
78
79 if (!ignorePitch && !other.ignorePitch && fixedPitch != other.fixedPitch)
80 return false;
81
82 if (stretch != 0 && other.stretch != 0 && stretch != other.stretch)
83 return false;
84
85 QString this_family, this_foundry, other_family, other_foundry;
86 for (int i = 0; i < families.size(); ++i) {
87 QFontDatabasePrivate::parseFontName(name: families.at(i), foundry&: this_foundry, family&: this_family);
88 QFontDatabasePrivate::parseFontName(name: other.families.at(i), foundry&: other_foundry, family&: other_family);
89 if (this_family != other_family || this_foundry != other_foundry)
90 return false;
91 }
92
93 return (styleHint == other.styleHint
94 && styleStrategy == other.styleStrategy
95 && weight == other.weight
96 && style == other.style
97 && this_family == other_family
98 && (styleName.isEmpty() || other.styleName.isEmpty() || styleName == other.styleName)
99 && (this_foundry.isEmpty()
100 || other_foundry.isEmpty()
101 || this_foundry == other_foundry)
102 );
103}

142/* Helper function to convert between legacy Qt and OpenType font weights. */
143static int convertWeights(int weight, bool inverted)
144{
145 static constexpr std::array<int, 2> legacyToOpenTypeMap[] = {
146 { 0, QFont::Thin }, { 12, QFont::ExtraLight }, { 25, QFont::Light },
147 { 50, QFont::Normal }, { 57, QFont::Medium }, { 63, QFont::DemiBold },
148 { 75, QFont::Bold }, { 81, QFont::ExtraBold }, { 87, QFont::Black },
149 };
150
151 int closestDist = INT_MAX;
152 int result = -1;
153
154 // Go through and find the closest mapped value
155 for (auto mapping : legacyToOpenTypeMap) {
156 const int weightOld = mapping[ inverted];
157 const int weightNew = mapping[!inverted];
158 const int dist = qAbs(t: weightOld - weight);
159 if (dist < closestDist) {
160 result = weightNew;
161 closestDist = dist;
162 } else {
163 // Break early since following values will be further away
164 break;
165 }
166 }
167
168 return result;
169}

171// Splits the family string on a comma and returns the list based on that
172static QStringList splitIntoFamilies(const QString &family)
173{
174 QStringList familyList;
175 if (family.isEmpty())
176 return familyList;
177 const auto list = QStringView{family}.split(sep: u',');
178 const int numFamilies = list.size();
179 familyList.reserve(size: numFamilies);
180 for (int i = 0; i < numFamilies; ++i) {
181 auto str = list.at(i).trimmed();
182 if ((str.startsWith(c: u'"') && str.endsWith(c: u'"'))
183 || (str.startsWith(c: u'\'') && str.endsWith(c: u'\''))) {
184 str = str.mid(pos: 1, n: str.size() - 2);
185 }
186 familyList << str.toString();
187 }
188 return familyList;
189}

285void QFontPrivate::resolve(uint mask, const QFontPrivate *other)
286{
287 Q_ASSERT(other != nullptr);
288
289 dpi = other->dpi;
290
291 if ((mask & QFont::AllPropertiesResolved) == QFont::AllPropertiesResolved) return;
292
293 // assign the unset-bits with the set-bits of the other font def
294 if (!(mask & QFont::FamiliesResolved))
295 request.families = other->request.families;
296
297 if (! (mask & QFont::StyleNameResolved))
298 request.styleName = other->request.styleName;
299
300 if (! (mask & QFont::SizeResolved)) {
301 request.pointSize = other->request.pointSize;
302 request.pixelSize = other->request.pixelSize;
303 }
304
305 if (! (mask & QFont::StyleHintResolved))
306 request.styleHint = other->request.styleHint;
307
308 if (! (mask & QFont::StyleStrategyResolved))
309 request.styleStrategy = other->request.styleStrategy;
310
311 if (! (mask & QFont::WeightResolved))
312 request.weight = other->request.weight;
313
314 if (! (mask & QFont::StyleResolved))
315 request.style = other->request.style;
316
317 if (! (mask & QFont::FixedPitchResolved))
318 request.fixedPitch = other->request.fixedPitch;
319
320 if (! (mask & QFont::StretchResolved))
321 request.stretch = other->request.stretch;
322
323 if (! (mask & QFont::HintingPreferenceResolved))
324 request.hintingPreference = other->request.hintingPreference;
325
326 if (! (mask & QFont::UnderlineResolved))
327 underline = other->underline;
328
329 if (! (mask & QFont::OverlineResolved))
330 overline = other->overline;
331
332 if (! (mask & QFont::StrikeOutResolved))
333 strikeOut = other->strikeOut;
334
335 if (! (mask & QFont::KerningResolved))
336 kerning = other->kerning;
337
338 if (! (mask & QFont::LetterSpacingResolved)) {
339 letterSpacing = other->letterSpacing;
340 letterSpacingIsAbsolute = other->letterSpacingIsAbsolute;
341 }
342 if (! (mask & QFont::WordSpacingResolved))
343 wordSpacing = other->wordSpacing;
344 if (! (mask & QFont::CapitalizationResolved))
345 capital = other->capital;
346}

372/*!
373 \class QFont
374 \reentrant
375
376 \brief The QFont class specifies a query for a font used for drawing text.
377
378 \ingroup painting
379 \ingroup appearance
380 \ingroup shared
381 \ingroup richtext-processing
382 \inmodule QtGui
383
384 QFont can be regarded as a query for one or more fonts on the system.
385
386 When you create a QFont object you specify various attributes that
387 you want the font to have. Qt will use the font with the specified
388 attributes, or if no matching font exists, Qt will use the closest
389 matching installed font. The attributes of the font that is
390 actually used are retrievable from a QFontInfo object. If the
391 window system provides an exact match exactMatch() returns \c true.
392 Use QFontMetricsF to get measurements, e.g. the pixel length of a
393 string using QFontMetrics::width().
394
395 Attributes which are not specifically set will not affect the font
396 selection algorithm, and default values will be preferred instead.
397
398 To load a specific physical font, typically represented by a single file,
399 use QRawFont instead.
400
401 Note that a QGuiApplication instance must exist before a QFont can be
402 used. You can set the application's default font with
403 QGuiApplication::setFont().
404
405 If a chosen font does not include all the characters that
406 need to be displayed, QFont will try to find the characters in the
407 nearest equivalent fonts. When a QPainter draws a character from a
408 font the QFont will report whether or not it has the character; if
409 it does not, QPainter will draw an unfilled square.
410
411 Create QFonts like this:
412
413 \snippet code/src_gui_text_qfont.cpp 0
414
415 The attributes set in the constructor can also be set later, e.g.
416 setFamily(), setPointSize(), setPointSizeF(), setWeight() and
417 setItalic(). The remaining attributes must be set after
418 construction, e.g. setBold(), setUnderline(), setOverline(),
419 setStrikeOut() and setFixedPitch(). QFontInfo objects should be
420 created \e after the font's attributes have been set. A QFontInfo
421 object will not change, even if you change the font's
422 attributes. The corresponding "get" functions, e.g. family(),
423 pointSize(), etc., return the values that were set, even though
424 the values used may differ. The actual values are available from a
425 QFontInfo object.
426
427 If the requested font family is unavailable you can influence the
428 \l{#fontmatching}{font matching algorithm} by choosing a
429 particular \l{QFont::StyleHint} and \l{QFont::StyleStrategy} with
430 setStyleHint(). The default family (corresponding to the current
431 style hint) is returned by defaultFamily().
432
433 You can provide substitutions for font family names using
434 insertSubstitution() and insertSubstitutions(). Substitutions can
435 be removed with removeSubstitutions(). Use substitute() to retrieve
436 a family's first substitute, or the family name itself if it has
437 no substitutes. Use substitutes() to retrieve a list of a family's
438 substitutes (which may be empty). After substituting a font, you must
439 trigger the updating of the font by destroying and re-creating all
440 QFont objects.
441
442 Every QFont has a key() which you can use, for example, as the key
443 in a cache or dictionary. If you want to store a user's font
444 preferences you could use QSettings, writing the font information
445 with toString() and reading it back with fromString(). The
446 operator<<() and operator>>() functions are also available, but
447 they work on a data stream.
448
449 It is possible to set the height of characters shown on the screen
450 to a specified number of pixels with setPixelSize(); however using
451 setPointSize() has a similar effect and provides device
452 independence.
453
454 Loading fonts can be expensive, especially on X11. QFont contains
455 extensive optimizations to make the copying of QFont objects fast,
456 and to cache the results of the slow window system functions it
457 depends upon.
458
459 \target fontmatching
460 The font matching algorithm works as follows:
461 \list 1
462 \li The specified font families (set by setFamilies()) are searched for.
463 \li If not found, then if set the specified font family exists and can be used to represent
464 the writing system in use, it will be selected.
465 \li If not, a replacement font that supports the writing system is
466 selected. The font matching algorithm will try to find the
467 best match for all the properties set in the QFont. How this is
468 done varies from platform to platform.
469 \li If no font exists on the system that can support the text,
470 then special "missing character" boxes will be shown in its place.
471 \endlist
472
473 \note If the selected font, though supporting the writing system in general,
474 is missing glyphs for one or more specific characters, then Qt will try to
475 find a fallback font for this or these particular characters. This feature
476 can be disabled using QFont::NoFontMerging style strategy.
477
478 In Windows a request for the "Courier" font is automatically changed to
479 "Courier New", an improved version of Courier that allows for smooth scaling.
480 The older "Courier" bitmap font can be selected by setting the PreferBitmap
481 style strategy (see setStyleStrategy()).
482
483 Once a font is found, the remaining attributes are matched in order of
484 priority:
485 \list 1
486 \li fixedPitch()
487 \li pointSize() (see below)
488 \li weight()
489 \li style()
490 \endlist
491
492 If you have a font which matches on family, even if none of the
493 other attributes match, this font will be chosen in preference to
494 a font which doesn't match on family but which does match on the
495 other attributes. This is because font family is the dominant
496 search criteria.
497
498 The point size is defined to match if it is within 20% of the
499 requested point size. When several fonts match and are only
500 distinguished by point size, the font with the closest point size
501 to the one requested will be chosen.
502
503 The actual family, font size, weight and other font attributes
504 used for drawing text will depend on what's available for the
505 chosen family under the window system. A QFontInfo object can be
506 used to determine the actual values used for drawing the text.
507
508 Examples:
509
510 \snippet code/src_gui_text_qfont.cpp 1
511 If you had both an Adobe and a Cronyx Helvetica, you might get
512 either.
513
514 \snippet code/src_gui_text_qfont.cpp 2
515
516 You can specify the foundry you want in the family name. The font f
517 in the above example will be set to "Helvetica
518 [Cronyx]".
519
520 To determine the attributes of the font actually used in the window
521 system, use a QFontInfo object, e.g.
522
523 \snippet code/src_gui_text_qfont.cpp 3
524
525 To find out font metrics use a QFontMetrics object, e.g.
526
527 \snippet code/src_gui_text_qfont.cpp 4
528
529 For more general information on fonts, see the
530 \l{comp.fonts FAQ}{comp.fonts FAQ}.
531 Information on encodings can be found from the
532 \l{UTR17} page.
533
534 \sa QFontMetrics, QFontInfo, QFontDatabase, {Character Map Example}
535*/

660/*!
661 Constructs a font object with the specified \a family, \a
662 pointSize, \a weight and \a italic settings.
663
664 If \a pointSize is zero or negative, the point size of the font
665 is set to a system-dependent default value. Generally, this is
666 12 points.
667
668 The \a family name may optionally also include a foundry name,
669 e.g. "Helvetica [Cronyx]". If the \a family is
670 available from more than one foundry and the foundry isn't
671 specified, an arbitrary foundry is chosen. If the family isn't
672 available a family will be set using the \l{QFont}{font matching}
673 algorithm.
674
675 This will split the family string on a comma and call setFamilies() with the
676 resulting list. To preserve a font that uses a comma in its name, use
677 the constructor that takes a QStringList.
678
679 \sa Weight, setFamily(), setPointSize(), setWeight(), setItalic(),
680 setStyleHint(), setFamilies(), QGuiApplication::font()
681*/
682QFont::QFont(const QString &family, int pointSize, int weight, bool italic)
683 : d(new QFontPrivate()), resolve_mask(QFont::FamiliesResolved)
684{
685 if (pointSize <= 0) {
686 pointSize = 12;
687 } else {
688 resolve_mask |= QFont::SizeResolved;
689 }
690
691 if (weight < 0) {
692 weight = Normal;
693 } else {
694 resolve_mask |= QFont::WeightResolved | QFont::StyleResolved;
695 }
696
697 if (italic)
698 resolve_mask |= QFont::StyleResolved;
699
700 d->request.families = splitIntoFamilies(family);
701 d->request.pointSize = qreal(pointSize);
702 d->request.pixelSize = -1;
703 d->request.weight = weight;
704 d->request.style = italic ? QFont::StyleItalic : QFont::StyleNormal;
705}

710

712 is set to a system-dependent default value. Generally, this is
713 12 points.
714
715 Each family name entry in \a families may optionally also include
716 a foundry name, e.g. "Helvetica [Cronyx]". If the family is
717 available from more than one foundry and the foundry isn't
718 specified, an arbitrary foundry is chosen. If the family isn't
719 available a family will be set using the \l{QFont}{font matching}
720 algorithm.
721
722 \sa Weight, setPointSize(), setWeight(), setItalic(),
723 setStyleHint(), setFamilies(), QGuiApplication::font()
724 */
725QFont::QFont(const QStringList &families, int pointSize, int weight, bool italic)
726 : d(new QFontPrivate()), resolve_mask(QFont::FamiliesResolved)
727{
728 if (pointSize <= 0)
729 pointSize = 12;
730 else
