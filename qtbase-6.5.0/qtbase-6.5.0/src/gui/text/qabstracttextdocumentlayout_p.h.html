26struct QTextObjectHandler
27{
28 QTextObjectHandler() : iface(nullptr) {}
29 QTextObjectInterface *iface;
30 QPointer<QObject> component;
31};
32typedef QHash<int, QTextObjectHandler> HandlerHash;
33
34class Q_GUI_EXPORT QAbstractTextDocumentLayoutPrivate : public QObjectPrivate
35{
36public:
37 Q_DECLARE_PUBLIC(QAbstractTextDocumentLayout)
38
39 inline QAbstractTextDocumentLayoutPrivate()
40 : paintDevice(nullptr) {}
41 ~QAbstractTextDocumentLayoutPrivate();
42
43 inline void setDocument(QTextDocument *doc) {
44 document = doc;
45 docPrivate = nullptr;
46 if (doc)
47 docPrivate = QTextDocumentPrivate::get(document: doc);
48 }
49
50 static QAbstractTextDocumentLayoutPrivate *get(QAbstractTextDocumentLayout *layout)
51 {
52 return layout->d_func();
53 }
54
55 bool hasHandlers() const
56 {
57 return !handlers.isEmpty();
58 }
59
60 inline int _q_dynamicPageCountSlot() const
61 { return q_func()->pageCount(); }
62 inline QSizeF _q_dynamicDocumentSizeSlot() const
63 { return q_func()->documentSize(); }
64
65 HandlerHash handlers;
66
67 void _q_handlerDestroyed(QObject *obj);
68 QPaintDevice *paintDevice;
69
70 QTextDocument *document;
71 QTextDocumentPrivate *docPrivate;
72};
