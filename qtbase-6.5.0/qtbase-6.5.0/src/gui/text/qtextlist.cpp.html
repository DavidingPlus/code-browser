24/*!
25 \class QTextList
26 \reentrant
27
28 \brief The QTextList class provides a decorated list of items in a QTextDocument.
29 \inmodule QtGui
30
31 \ingroup richtext-processing
32
33 A list contains a sequence of text blocks, each of which is marked with a
34 bullet point or other symbol. Multiple levels of lists can be used, and
35 the automatic numbering feature provides support for ordered numeric and
36 alphabetical lists.
37
38 Lists are created by using a text cursor to insert an empty list at the
39 current position or by moving existing text into a new list.
40 The \l{QTextCursor::insertList()} function inserts an empty block into the
41 document at the cursor position, and makes it the first item in a list.
42
43 \snippet textdocument-lists/mainwindow.cpp 0
44
45 The \l{QTextCursor::createList()} function takes the contents of the
46 cursor's current block and turns it into the first item of a new list.
47
48 The cursor's current list is found with \l{QTextCursor::currentList()}.
49
50 The number of items in a list is given by count(). Each item can be
51 obtained by its index in the list with the item() function. Similarly,
52 the index of a given item can be found with itemNumber(). The text of
53 each item can be found with the itemText() function.
54
55 Note that the items in the list may not be adjacent elements in the
56 document. For example, the top-level items in a multi-level list will
57 be separated by the items in lower levels of the list.
58
59 List items can be deleted by index with the removeItem() function.
60 remove() deletes the specified item in the list.
61
62 The list's format is set with setFormat() and read with format().
63 The format describes the decoration of the list itself, and not the
64 individual items.
65
66 \sa QTextBlock, QTextListFormat, QTextCursor
67*/

92/*!

94
95 \sa count(), itemText()

97QTextBlock QTextList::item(int i) const
98{
99 Q_D(const QTextList);
100 if (i < 0 || i >= d->blocks.size())
101 return QTextBlock();
102 return d->blocks.at(i);

129/*!
130 \fn QString QTextList::itemText(const QTextBlock &block) const
131
132 Returns the text of the list item that corresponds to the given \a block.
133*/
134QString QTextList::itemText(const QTextBlock &blockIt) const
135{
136 Q_D(const QTextList);
137 int item = d->blocks.indexOf(t: blockIt) + 1;
138 if (item <= 0)
139 return QString();
140
141 QTextBlock block = d->blocks.at(i: item-1);
142 QTextBlockFormat blockFormat = block.blockFormat();
143
144 QString result;
145
146 const int style = format().style();
147 QString numberPrefix;
148 QString numberSuffix = u"."_s;
149
150 if (format().hasProperty(propertyId: QTextFormat::ListNumberPrefix))
151 numberPrefix = format().numberPrefix();
152 if (format().hasProperty(propertyId: QTextFormat::ListNumberSuffix))
153 numberSuffix = format().numberSuffix();
154
155 switch (style) {
156 case QTextListFormat::ListDecimal:
157 result = QString::number(item);
158 break;
159 // from the old richtext
160 case QTextListFormat::ListLowerAlpha:
161 case QTextListFormat::ListUpperAlpha:
162 {
163 const char baseChar = style == QTextListFormat::ListUpperAlpha ? 'A' : 'a';
164
165 int c = item;
166 while (c > 0) {
167 c--;
168 result.prepend(c: QChar::fromUcs2(c: baseChar + (c % 26)));
169 c /= 26;
170 }
171 }
172 break;
173 case QTextListFormat::ListLowerRoman:
174 case QTextListFormat::ListUpperRoman:
175 {
176 if (item < 5000) {
177 QByteArray romanNumeral;
178
179 // works for up to 4999 items
180 static const char romanSymbolsLower[] = "iiivixxxlxcccdcmmmm";
181 static const char romanSymbolsUpper[] = "IIIVIXXXLXCCCDCMMMM";
182 QByteArray romanSymbols; // wrap to have "mid"
183 if (style == QTextListFormat::ListLowerRoman)
184 romanSymbols = QByteArray::fromRawData(data: romanSymbolsLower, size: sizeof(romanSymbolsLower));
185 else
186 romanSymbols = QByteArray::fromRawData(data: romanSymbolsUpper, size: sizeof(romanSymbolsUpper));
187
188 int c[] = { 1, 4, 5, 9, 10, 40, 50, 90, 100, 400, 500, 900, 1000 };
189 int n = item;
190 for (int i = 12; i >= 0; n %= c[i], i--) {
191 int q = n / c[i];
192 if (q > 0) {
193 int startDigit = i + (i+3)/4;
194 int numDigits;
195 if (i % 4) {
196 // c[i] == 4|5|9|40|50|90|400|500|900
197 if ((i-2) % 4) {
198 // c[i] == 4|9|40|90|400|900 => with subtraction (IV, IX, XL, XC, ...)
199 numDigits = 2;
200 }
201 else {
202 // c[i] == 5|50|500 (V, L, D)
203 numDigits = 1;
204 }
205 }
206 else {
207 // c[i] == 1|10|100|1000 (I, II, III, X, XX, ...)
208 numDigits = q;
209 }
210
211 romanNumeral.append(a: romanSymbols.mid(index: startDigit, len: numDigits));
212 }
213 }
214 result = QString::fromLatin1(ba: romanNumeral);
215 }
216 else {
217 result = u"?"_s;
218 }
219
220 }
221 break;
222 default:
223 Q_ASSERT(false);
224 }
225 if (blockIt.textDirection() == Qt::RightToLeft)
226 return numberSuffix + result + numberPrefix;
227 else
228 return numberPrefix + result + numberSuffix;
229}

231/*!
232 Removes the item at item position \a i from the list. When the last item in the
233 list is removed, the list is automatically deleted by the QTextDocument that owns
234 it.
235
236 \sa add(), remove()
237*/
238void QTextList::removeItem(int i)
239{
240 Q_D(QTextList);
241 if (i < 0 || i >= d->blocks.size())
242 return;
243
244 QTextBlock block = d->blocks.at(i);
245 remove(block);
246}

249/*!
250 Removes the given \a block from the list.
251
252 \sa add(), removeItem()
253*/
254void QTextList::remove(const QTextBlock &block)
255{
256 QTextBlockFormat fmt = block.blockFormat();
257 fmt.setIndent(fmt.indent() + format().indent());
258 fmt.setObjectIndex(-1);
259 const_cast<QTextDocumentPrivate *>(QTextDocumentPrivate::get(block))->setBlockFormat(from: block, to: block, newFormat: fmt, mode: QTextDocumentPrivate::SetFormat);
260}
261
262/*!
263 Makes the given \a block part of the list.
264
265 \sa remove(), removeItem()
266*/
267void QTextList::add(const QTextBlock &block)
268{
269 QTextBlockFormat fmt = block.blockFormat();
270 fmt.setObjectIndex(objectIndex());
271 const_cast<QTextDocumentPrivate *>(QTextDocumentPrivate::get(block))->setBlockFormat(from: block, to: block, newFormat: fmt, mode: QTextDocumentPrivate::SetFormat);
272}
