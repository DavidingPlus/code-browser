295inline bool QFont::bold() const
296{ return weight() > Medium; }

298
299inline void QFont::setBold(bool enable)
300{ setWeight(enable ? Bold : Normal); }
301
302inline bool QFont::italic() const
303{
304 return (style() != StyleNormal);
305}
306
307inline void QFont::setItalic(bool b) {
308 setStyle(b ? StyleItalic : StyleNormal);
