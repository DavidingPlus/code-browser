20static inline QString findAtNxFileOrResource(const QString &baseFileName,
21 qreal targetDevicePixelRatio,
22 qreal *sourceDevicePixelRatio)
23{
24 // qt_findAtNxFile expects a file name that can be tested with QFile::exists.
25 // so if the format.name() is a file:/ or qrc:/ URL, then we need to strip away the schema.
26 QString localFile = baseFileName;
27 if (localFile.startsWith(s: "file:/"_L1))
28 localFile = localFile.sliced(pos: 6);
29 else if (localFile.startsWith(s: "qrc:/"_L1))
30 localFile = localFile.sliced(pos: 3);
31
32 extern QString qt_findAtNxFile(const QString &baseFileName, qreal targetDevicePixelRatio,
33 qreal *sourceDevicePixelRatio);
34 return qt_findAtNxFile(baseFileName: localFile, targetDevicePixelRatio, sourceDevicePixelRatio);
35}

37static inline QUrl fromLocalfileOrResources(QString path)
38{
39 if (path.startsWith(s: ":/"_L1)) // auto-detect resources and convert them to url
40 path = path.prepend(s: "qrc"_L1);
41 return QUrl(path);
42}

44static QPixmap getPixmap(QTextDocument *doc, const QTextImageFormat &format, const qreal devicePixelRatio = 1.0)
45{
46 qreal sourcePixelRatio = 1.0;
47 const QString name = findAtNxFileOrResource(baseFileName: format.name(), targetDevicePixelRatio: devicePixelRatio, sourceDevicePixelRatio: &sourcePixelRatio);
48 const QUrl url = fromLocalfileOrResources(path: name);
49
50 QPixmap pm;
51 const QVariant data = doc->resource(type: QTextDocument::ImageResource, name: url);
52 if (data.userType() == QMetaType::QPixmap || data.userType() == QMetaType::QImage) {
53 pm = qvariant_cast<QPixmap>(v: data);
54 } else if (data.userType() == QMetaType::QByteArray) {
55 pm.loadFromData(buf: data.toByteArray());
56 }
57
58 if (pm.isNull()) {
59 QImage img;
60 if (name.isEmpty() || !img.load(fileName: name))
61 return QPixmap(":/qt-project.org/styles/commonstyle/images/file-16.png"_L1);
62
63 pm = QPixmap::fromImage(image: img);
64 doc->addResource(type: QTextDocument::ImageResource, name: url, resource: pm);
65 }
66
67 if (name.contains(s: "@2x"_L1))
68 pm.setDevicePixelRatio(sourcePixelRatio);
69
70 return pm;
71}

73static QSize getPixmapSize(QTextDocument *doc, const QTextImageFormat &format)
74{
75 QPixmap pm;
76
77 const bool hasWidth = format.hasProperty(propertyId: QTextFormat::ImageWidth);
78 const int width = qRound(d: format.width());
79 const bool hasHeight = format.hasProperty(propertyId: QTextFormat::ImageHeight);
80 const int height = qRound(d: format.height());
81
82 QSize size(width, height);
83 if (!hasWidth || !hasHeight) {
84 pm = getPixmap(doc, format);
85 const QSizeF pmSize = pm.deviceIndependentSize();
86
87 if (!hasWidth) {
88 if (!hasHeight)
89 size.setWidth(pmSize.width());
90 else
91 size.setWidth(qRound(d: height * (pmSize.width() / (qreal) pmSize.height())));
92 }
93 if (!hasHeight) {
94 if (!hasWidth)
95 size.setHeight(pmSize.height());
96 else
97 size.setHeight(qRound(d: width * (pmSize.height() / (qreal) pmSize.width())));
98 }
99 }
100
101 qreal scale = 1.0;
102 QPaintDevice *pdev = doc->documentLayout()->paintDevice();
103 if (pdev) {
104 if (pm.isNull())
105 pm = getPixmap(doc, format);
106 if (!pm.isNull())
107 scale = qreal(pdev->logicalDpiY()) / qreal(qt_defaultDpi());
108 }
109 size *= scale;
110
111 return size;
112}

114static QImage getImage(QTextDocument *doc, const QTextImageFormat &format, const qreal devicePixelRatio = 1.0)
115{
116 qreal sourcePixelRatio = 1.0;
117 const QString name = findAtNxFileOrResource(baseFileName: format.name(), targetDevicePixelRatio: devicePixelRatio, sourceDevicePixelRatio: &sourcePixelRatio);
118 const QUrl url = fromLocalfileOrResources(path: name);
119
120 QImage image;
121 const QVariant data = doc->resource(type: QTextDocument::ImageResource, name: url);
122 if (data.userType() == QMetaType::QImage) {
123 image = qvariant_cast<QImage>(v: data);
124 } else if (data.userType() == QMetaType::QByteArray) {
125 image.loadFromData(data: data.toByteArray());
126 }
127
128 if (image.isNull()) {
129 if (name.isEmpty() || !image.load(fileName: name))
130 return QImage(":/qt-project.org/styles/commonstyle/images/file-16.png"_L1);
131
132 doc->addResource(type: QTextDocument::ImageResource, name: url, resource: image);
133 }
134
135 if (sourcePixelRatio != 1.0)
136 image.setDevicePixelRatio(sourcePixelRatio);
137
138 return image;
139}

141static QSize getImageSize(QTextDocument *doc, const QTextImageFormat &format)
142{
143 QImage image;
144
145 const bool hasWidth = format.hasProperty(propertyId: QTextFormat::ImageWidth);
146 const int width = qRound(d: format.width());
147 const bool hasHeight = format.hasProperty(propertyId: QTextFormat::ImageHeight);
148 const int height = qRound(d: format.height());
149
150 QSize size(width, height);
151 if (!hasWidth || !hasHeight) {
152 image = getImage(doc, format);
153 QSizeF imageSize = image.deviceIndependentSize();
154 if (!hasWidth)
155 size.setWidth(imageSize.width());
156 if (!hasHeight)
157 size.setHeight(imageSize.height());
158 }
159
160 qreal scale = 1.0;
161 QPaintDevice *pdev = doc->documentLayout()->paintDevice();
162 if (pdev) {
163 if (image.isNull())
164 image = getImage(doc, format);
165 if (!image.isNull())
166 scale = qreal(pdev->logicalDpiY()) / qreal(qt_defaultDpi());
167 }
168 size *= scale;
169
170 return size;
171}

178QSizeF QTextImageHandler::intrinsicSize(QTextDocument *doc, int posInDocument, const QTextFormat &format)

195void QTextImageHandler::drawObject(QPainter *p, const QRectF &rect, QTextDocument *doc, int posInDocument, const QTextFormat &format)
196{
197 Q_UNUSED(posInDocument);
198 const QTextImageFormat imageFormat = format.toImageFormat();
199
200 if (QCoreApplication::instance()->thread() != QThread::currentThread()) {
201 const QImage image = getImage(doc, format: imageFormat, devicePixelRatio: p->device()->devicePixelRatio());
202 p->drawImage(targetRect: rect, image, sourceRect: image.rect());
203 } else {
204 const QPixmap pixmap = getPixmap(doc, format: imageFormat, devicePixelRatio: p->device()->devicePixelRatio());
205 p->drawPixmap(targetRect: rect, pixmap, sourceRect: pixmap.rect());
206 }
207}
