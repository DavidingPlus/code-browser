29/// Convert pixels to postscript point units
30static QString pixelToPoint(qreal pixels)
31{
32 // we hardcode 96 DPI, we do the same in the ODF importer to have a perfect roundtrip.
33 return QString::number(pixels * 72 / 96) + "pt"_L1;
34}

70class QZipStreamStrategy : public QOutputStrategy {
71public:
72 QZipStreamStrategy(QIODevice *device)
73 : zip(device),
74 manifestWriter(&manifest)
75 {
76 QByteArray mime("application/vnd.oasis.opendocument.text");
77 zip.setCompressionPolicy(QZipWriter::NeverCompress);
78 zip.addFile(fileName: QString::fromLatin1(ba: "mimetype"), data: mime); // for mime-magick
79 zip.setCompressionPolicy(QZipWriter::AutoCompress);
80 contentStream = &content;
81 content.open(openMode: QIODevice::WriteOnly);
82 manifest.open(openMode: QIODevice::WriteOnly);
83
84 manifestNS = QString::fromLatin1(ba: "urn:oasis:names:tc:opendocument:xmlns:manifest:1.0");
85 // prettyfy
86 manifestWriter.setAutoFormatting(true);
87 manifestWriter.setAutoFormattingIndent(1);
88
89 manifestWriter.writeNamespace(namespaceUri: manifestNS, prefix: QString::fromLatin1(ba: "manifest"));
90 manifestWriter.writeStartDocument();
91 manifestWriter.writeStartElement(namespaceUri: manifestNS, name: QString::fromLatin1(ba: "manifest"));
92 manifestWriter.writeAttribute(namespaceUri: manifestNS, name: QString::fromLatin1(ba: "version"), value: QString::fromLatin1(ba: "1.2"));
93 addFile(fileName: QString::fromLatin1(ba: "/"), mimeType: QString::fromLatin1(ba: "application/vnd.oasis.opendocument.text"));
94 addFile(fileName: QString::fromLatin1(ba: "content.xml"), mimeType: QString::fromLatin1(ba: "text/xml"));
95 }
96
97 ~QZipStreamStrategy()
98 {
99 manifestWriter.writeEndDocument();
100 manifest.close();
101 zip.addFile(fileName: QString::fromLatin1(ba: "META-INF/manifest.xml"), device: &manifest);
102 content.close();
103 zip.addFile(fileName: QString::fromLatin1(ba: "content.xml"), device: &content);
104 zip.close();
105 }
106
107 virtual void addFile(const QString &fileName, const QString &mimeType, const QByteArray &bytes) override
108 {
109 zip.addFile(fileName, data: bytes);
110 addFile(fileName, mimeType);
111 }
112
113private:
114 void addFile(const QString &fileName, const QString &mimeType)
115 {
116 manifestWriter.writeEmptyElement(namespaceUri: manifestNS, name: QString::fromLatin1(ba: "file-entry"));
117 manifestWriter.writeAttribute(namespaceUri: manifestNS, name: QString::fromLatin1(ba: "media-type"), value: mimeType);
118 manifestWriter.writeAttribute(namespaceUri: manifestNS, name: QString::fromLatin1(ba: "full-path"), value: fileName);
119 }
120
121 QBuffer content;
122 QBuffer manifest;
123 QZipWriter zip;
124 QXmlStreamWriter manifestWriter;
125 QString manifestNS;
126};

128static QStringView bullet_char(QTextListFormat::Style style)
129{
130 static_assert(int(QTextListFormat::ListDisc) == -1);
131 static_assert(int(QTextListFormat::ListUpperRoman) == -8);
132 static const char16_t chars[] = {
133 u'\x25cf', // bullet character
134 u'\x25cb', // white circle
135 u'\x25a1', // white square
136 u'1',
137 u'a',
138 u'A',
139 u'i',
140 u'I',
141 };
142 const auto map = [](QTextListFormat::Style s) { return -int(s) - 1; };
143 static_assert(uint(map(QTextListFormat::ListUpperRoman)) == std::size(chars) - 1);
144 const auto idx = map(style);
145 if (idx < 0)
146 return nullptr;
147 else
148 return {chars + idx, 1};
149}

185void QTextOdfWriter::writeFrame(QXmlStreamWriter &writer, const QTextFrame *frame)
186{
187 Q_ASSERT(frame);
188 const QTextTable *table = qobject_cast<const QTextTable*> (object: frame);
189
190 if (table) { // Start a table.
191 writer.writeStartElement(namespaceUri: tableNS, name: QString::fromLatin1(ba: "table"));
192 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "style-name"),
193 value: QString::fromLatin1(ba: "Table%1").arg(a: table->formatIndex()));
194 // check if column widths are set, if so add TableNS line above for all columns and link to style
195 if (m_tableFormatsWithColWidthConstraints.contains(value: table->formatIndex())) {
196 for (int colit = 0; colit < table->columns(); ++colit) {
197 writer.writeStartElement(namespaceUri: tableNS, name: QString::fromLatin1(ba: "table-column"));
198 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "style-name"),
199 value: QString::fromLatin1(ba: "Table%1.%2").arg(a: table->formatIndex()).arg(a: colit));
200 writer.writeEndElement();
201 }
202 } else {
203 writer.writeEmptyElement(namespaceUri: tableNS, name: QString::fromLatin1(ba: "table-column"));
204 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "number-columns-repeated"),
205 value: QString::number(table->columns()));
206 }
207 } else if (frame->document() && frame->document()->rootFrame() != frame) { // start a section
208 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "section"));
209 }
210
211 QTextFrame::iterator iterator = frame->begin();
212 QTextFrame *child = nullptr;
213
214 int tableRow = -1;
215 while (! iterator.atEnd()) {
216 if (iterator.currentFrame() && child != iterator.currentFrame())
217 writeFrame(writer, frame: iterator.currentFrame());
218 else { // no frame, its a block
219 QTextBlock block = iterator.currentBlock();
220 if (table) {
221 QTextTableCell cell = table->cellAt(position: block.position());
222 if (tableRow < cell.row()) {
223 if (tableRow >= 0)
224 writer.writeEndElement(); // close table row
225 tableRow = cell.row();
226 writer.writeStartElement(namespaceUri: tableNS, name: QString::fromLatin1(ba: "table-row"));
227 }
228 writer.writeStartElement(namespaceUri: tableNS, name: QString::fromLatin1(ba: "table-cell"));
229 if (cell.columnSpan() > 1)
230 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "number-columns-spanned"), value: QString::number(cell.columnSpan()));
231 if (cell.rowSpan() > 1)
232 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "number-rows-spanned"), value: QString::number(cell.rowSpan()));
233 if (cell.format().isTableCellFormat()) {
234 if (m_cellFormatsInTablesWithBorders.contains(key: cell.tableCellFormatIndex()) ) {
235 // writing table:style-name tag in <table:table-cell> element
236 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "style-name"),
237 value: QString::fromLatin1(ba: "TB%1.%2").arg(a: table->formatIndex())
238 .arg(a: cell.tableCellFormatIndex()));
239 } else {
240 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "style-name"),
241 value: QString::fromLatin1(ba: "T%1").arg(a: cell.tableCellFormatIndex()));
242 }
243 }
244 }
245 writeBlock(writer, block);
246 if (table)
247 writer.writeEndElement(); // table-cell
248 }
249 child = iterator.currentFrame();
250 ++iterator;
251 }
252 if (tableRow >= 0)
253 writer.writeEndElement(); // close table-row
254
255 if (table || (frame->document() && frame->document()->rootFrame() != frame))
256 writer.writeEndElement(); // close table or section element
257}

259void QTextOdfWriter::writeBlock(QXmlStreamWriter &writer, const QTextBlock &block)
260{
261 if (block.textList()) { // its a list-item
262 const int listLevel = block.textList()->format().indent();
263 if (m_listStack.isEmpty() || m_listStack.top() != block.textList()) {
264 // not the same list we were in.
265 while (m_listStack.size() >= listLevel && !m_listStack.isEmpty() && m_listStack.top() != block.textList() ) { // we need to close tags
266 m_listStack.pop();
267 writer.writeEndElement(); // list
268 if (m_listStack.size())
269 writer.writeEndElement(); // list-item
270 }
271 while (m_listStack.size() < listLevel) {
272 if (m_listStack.size())
273 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "list-item"));
274 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "list"));
275 if (m_listStack.size() == listLevel - 1) {
276 m_listStack.push(t: block.textList());
277 writer.writeAttribute(namespaceUri: textNS, name: QString::fromLatin1(ba: "style-name"), value: QString::fromLatin1(ba: "L%1")
278 .arg(a: block.textList()->formatIndex()));
279 }
280 else {
281 m_listStack.push(t: nullptr);
282 }
283 }
284 }
285 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "list-item"));
286 }
287 else {
288 while (! m_listStack.isEmpty()) {
289 m_listStack.pop();
290 writer.writeEndElement(); // list
291 if (m_listStack.size())
292 writer.writeEndElement(); // list-item
293 }
294 }
295
296 if (block.length() == 1) { // only a linefeed
297 writer.writeEmptyElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "p"));
298 writer.writeAttribute(namespaceUri: textNS, name: QString::fromLatin1(ba: "style-name"), value: QString::fromLatin1(ba: "p%1")
299 .arg(a: block.blockFormatIndex()));
300 if (block.textList())
301 writer.writeEndElement(); // numbered-paragraph
302 return;
303 }
304 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "p"));
305 writer.writeAttribute(namespaceUri: textNS, name: QString::fromLatin1(ba: "style-name"), value: QString::fromLatin1(ba: "p%1")
306 .arg(a: block.blockFormatIndex()));
307 for (QTextBlock::Iterator frag = block.begin(); !frag.atEnd(); ++frag) {
308 bool isHyperlink = frag.fragment().charFormat().hasProperty(propertyId: QTextFormat::AnchorHref);
309 if (isHyperlink) {
310 QString value = frag.fragment().charFormat().property(propertyId: QTextFormat::AnchorHref).toString();
311 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "a"));
312 writer.writeAttribute(namespaceUri: xlinkNS, name: QString::fromLatin1(ba: "href"), value);
313 }
314 writer.writeCharacters(text: QString()); // Trick to make sure that the span gets no linefeed in front of it.
315 writer.writeStartElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "span"));
316
317 QString fragmentText = frag.fragment().text();
318 if (fragmentText.size() == 1 && fragmentText[0] == u'\xFFFC') { // its an inline character.
319 writeInlineCharacter(writer, fragment: frag.fragment());
320 writer.writeEndElement(); // span
321 continue;
322 }
323
324 writer.writeAttribute(namespaceUri: textNS, name: QString::fromLatin1(ba: "style-name"), value: QString::fromLatin1(ba: "c%1")
325 .arg(a: frag.fragment().charFormatIndex()));
326 bool escapeNextSpace = true;
327 int precedingSpaces = 0;
328 int exportedIndex = 0;
329 for (int i=0; i <= fragmentText.size(); ++i) {
330 QChar character = (i == fragmentText.size() ? QChar() : fragmentText.at(i));
331 bool isSpace = character.unicode() == ' ';
332
333 // find more than one space. -> <text:s text:c="2" />
334 if (!isSpace && escapeNextSpace && precedingSpaces > 1) {
335 const bool startParag = exportedIndex == 0 && i == precedingSpaces;
336 if (!startParag)
337 writer.writeCharacters(text: fragmentText.mid(position: exportedIndex, n: i - precedingSpaces + 1 - exportedIndex));
338 writer.writeEmptyElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "s"));
339 const int count = precedingSpaces - (startParag?0:1);
340 if (count > 1)
341 writer.writeAttribute(namespaceUri: textNS, name: QString::fromLatin1(ba: "c"), value: QString::number(count));
342 precedingSpaces = 0;
343 exportedIndex = i;
344 }
345
346 if (i < fragmentText.size()) {
347 if (character.unicode() == 0x2028) { // soft-return
348 //if (exportedIndex < i)
349 writer.writeCharacters(text: fragmentText.mid(position: exportedIndex, n: i - exportedIndex));
350 // adding tab before line-break, so last line in justified paragraph
351 // will not stretch to the end
352 writer.writeEmptyElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "tab"));
353 writer.writeEmptyElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "line-break"));
354 exportedIndex = i+1;
355 continue;
356 } else if (character.unicode() == '\t') { // Tab
357 //if (exportedIndex < i)
358 writer.writeCharacters(text: fragmentText.mid(position: exportedIndex, n: i - exportedIndex));
359 writer.writeEmptyElement(namespaceUri: textNS, name: QString::fromLatin1(ba: "tab"));
360 exportedIndex = i+1;
361 precedingSpaces = 0;
362 } else if (isSpace) {
363 ++precedingSpaces;
364 escapeNextSpace = true;
365 } else if (!isSpace) {
366 precedingSpaces = 0;
367 }
368 }
369 }
370
371 writer.writeCharacters(text: fragmentText.mid(position: exportedIndex));
372 writer.writeEndElement(); // span
373 writer.writeCharacters(text: QString()); // Trick to make sure that the span gets no linefeed behind it.
374 if (isHyperlink)
375 writer.writeEndElement(); // a
376 }
377 writer.writeCharacters(text: QString()); // Trick to make sure that the span gets no linefeed behind it.
378 writer.writeEndElement(); // p
379 if (block.textList())
380 writer.writeEndElement(); // list-item
381}

383static bool probeImageData(QIODevice *device, QImage *image, QString *mimeType, qreal *width, qreal *height)
384{
385 QImageReader reader(device);
386 const QByteArray format = reader.format().toLower();
387 if (format == "png") {
388 *mimeType = QStringLiteral("image/png");
389 } else if (format == "jpg") {
390 *mimeType = QStringLiteral("image/jpg");
391 } else if (format == "svg") {
392 *mimeType = QStringLiteral("image/svg+xml");
393 } else {
394 *image = reader.read();
395 return false;
396 }
397
398 const QSize size = reader.size();
399
400 *width = size.width();
401 *height = size.height();
402
403 return true;
404}

406void QTextOdfWriter::writeInlineCharacter(QXmlStreamWriter &writer, const QTextFragment &fragment) const
407{
408 writer.writeStartElement(namespaceUri: drawNS, name: QString::fromLatin1(ba: "frame"));
409 if (m_strategy == nullptr) {
410 // don't do anything.
411 }
412 else if (fragment.charFormat().isImageFormat()) {
413 QTextImageFormat imageFormat = fragment.charFormat().toImageFormat();
414 writer.writeAttribute(namespaceUri: drawNS, name: QString::fromLatin1(ba: "name"), value: imageFormat.name());
415
416 QByteArray data;
417 QString mimeType;
418 qreal width = 0;
419 qreal height = 0;
420
421 QImage image;
422 QString name = imageFormat.name();
423 if (name.startsWith(s: ":/"_L1)) // auto-detect resources
424 name.prepend(s: "qrc"_L1);
425 QUrl url = QUrl(name);
426 const QVariant variant = m_document->resource(type: QTextDocument::ImageResource, name: url);
427 if (variant.userType() == QMetaType::QPixmap || variant.userType() == QMetaType::QImage) {
428 image = qvariant_cast<QImage>(v: variant);
429 } else if (variant.userType() == QMetaType::QByteArray) {
430 data = variant.toByteArray();
431
432 QBuffer buffer(&data);
433 buffer.open(openMode: QIODevice::ReadOnly);
434 probeImageData(device: &buffer, image: &image, mimeType: &mimeType, width: &width, height: &height);
435 } else {
436 // try direct loading
437 QFile file(imageFormat.name());
438 if (file.open(flags: QIODevice::ReadOnly) && !probeImageData(device: &file, image: &image, mimeType: &mimeType, width: &width, height: &height)) {
439 file.seek(offset: 0);
440 data = file.readAll();
441 }
442 }
443
444 if (! image.isNull()) {
445 QBuffer imageBytes;
446
447 int imgQuality = imageFormat.quality();
448 if (imgQuality >= 100 || imgQuality <= 0 || image.hasAlphaChannel()) {
449 QImageWriter imageWriter(&imageBytes, "png");
450 imageWriter.write(image);
451
452 data = imageBytes.data();
453 mimeType = QStringLiteral("image/png");
454 } else {
455 // Write images without alpha channel as jpg with quality set by QTextImageFormat
456 QImageWriter imageWriter(&imageBytes, "jpg");
457 imageWriter.setQuality(imgQuality);
458 imageWriter.write(image);
459
460 data = imageBytes.data();
461 mimeType = QStringLiteral("image/jpg");
462 }
463
464 width = image.width();
465 height = image.height();
466 }
467
468 if (!data.isEmpty()) {
469 if (imageFormat.hasProperty(propertyId: QTextFormat::ImageWidth)) {
470 width = imageFormat.width();
471 }
472 if (imageFormat.hasProperty(propertyId: QTextFormat::ImageHeight)) {
473 height = imageFormat.height();
474 }
475
476 QString filename = m_strategy->createUniqueImageName();
477
478 m_strategy->addFile(fileName: filename, mimeType, bytes: data);
479
480 writer.writeAttribute(namespaceUri: svgNS, name: QString::fromLatin1(ba: "width"), value: pixelToPoint(pixels: width));
481 writer.writeAttribute(namespaceUri: svgNS, name: QString::fromLatin1(ba: "height"), value: pixelToPoint(pixels: height));
482 writer.writeAttribute(namespaceUri: textNS, QStringLiteral("anchor-type"), QStringLiteral("as-char"));
483 writer.writeStartElement(namespaceUri: drawNS, name: QString::fromLatin1(ba: "image"));
484 writer.writeAttribute(namespaceUri: xlinkNS, name: QString::fromLatin1(ba: "href"), value: filename);
485 writer.writeEndElement(); // image
486 }
487 }
488 writer.writeEndElement(); // frame
489}

491void QTextOdfWriter::writeFormats(QXmlStreamWriter &writer, const QSet<int> &formats) const
492{
493 writer.writeStartElement(namespaceUri: officeNS, name: QString::fromLatin1(ba: "automatic-styles"));
494 QList<QTextFormat> allStyles = m_document->allFormats();
495 for (int formatIndex : formats) {
496 QTextFormat textFormat = allStyles.at(i: formatIndex);
497 switch (textFormat.type()) {
498 case QTextFormat::CharFormat:
499 if (textFormat.isTableCellFormat())
500 writeTableCellFormat(writer, format: textFormat.toTableCellFormat(), formatIndex, styles&: allStyles);
501 else
502 writeCharacterFormat(writer, format: textFormat.toCharFormat(), formatIndex);
503 break;
504 case QTextFormat::BlockFormat:
505 writeBlockFormat(writer, format: textFormat.toBlockFormat(), formatIndex);
506 break;
507 case QTextFormat::ListFormat:
508 writeListFormat(writer, format: textFormat.toListFormat(), formatIndex);
509 break;
510 case QTextFormat::FrameFormat:
511 if (textFormat.isTableFormat())
512 writeTableFormat(writer, format: textFormat.toTableFormat(), formatIndex);
513 else
514 writeFrameFormat(writer, format: textFormat.toFrameFormat(), formatIndex);
515 break;
516 }
517 }
518
519 writer.writeEndElement(); // automatic-styles
520}

522void QTextOdfWriter::writeBlockFormat(QXmlStreamWriter &writer, QTextBlockFormat format, int formatIndex) const
523{
524 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "style"));
525 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "name"), value: QString::fromLatin1(ba: "p%1").arg(a: formatIndex));
526 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "family"), value: QString::fromLatin1(ba: "paragraph"));
527 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "paragraph-properties"));
528
529 if (format.hasProperty(propertyId: QTextBlockFormat::LineHeightType)) {
530 const int blockLineHeightType = format.lineHeightType();
531 const qreal blockLineHeight = format.lineHeight();
532 QString type, value;
533 switch (blockLineHeightType) {
534 case QTextBlockFormat::SingleHeight:
535 type = QString::fromLatin1(ba: "line-height");
536 value = QString::fromLatin1(ba: "100%");
537 break;
538 case QTextBlockFormat::ProportionalHeight:
539 type = QString::fromLatin1(ba: "line-height");
540 value = QString::number(blockLineHeight) + QString::fromLatin1(ba: "%");
541 break;
542 case QTextBlockFormat::FixedHeight:
543 type = QString::fromLatin1(ba: "line-height");
544 value = pixelToPoint(pixels: qMax(a: qreal(0.), b: blockLineHeight));
545 break;
546 case QTextBlockFormat::MinimumHeight:
547 type = QString::fromLatin1(ba: "line-height-at-least");
548 value = pixelToPoint(pixels: qMax(a: qreal(0.), b: blockLineHeight));
549 break;
550 case QTextBlockFormat::LineDistanceHeight:
551 type = QString::fromLatin1(ba: "line-spacing");
552 value = pixelToPoint(pixels: qMax(a: qreal(0.), b: blockLineHeight));
553 }
554
555 if (!type.isNull())
556 writer.writeAttribute(namespaceUri: styleNS, name: type, value);
557 }
558
559 if (format.hasProperty(propertyId: QTextFormat::BlockAlignment)) {
560 const Qt::Alignment alignment = format.alignment() & Qt::AlignHorizontal_Mask;
561 QString value;
562 if (alignment == Qt::AlignLeading)
563 value = QString::fromLatin1(ba: "start");
564 else if (alignment == Qt::AlignTrailing)
565 value = QString::fromLatin1(ba: "end");
566 else if (alignment == (Qt::AlignLeft | Qt::AlignAbsolute))
567 value = QString::fromLatin1(ba: "left");
568 else if (alignment == (Qt::AlignRight | Qt::AlignAbsolute))
569 value = QString::fromLatin1(ba: "right");
570 else if (alignment == Qt::AlignHCenter)
571 value = QString::fromLatin1(ba: "center");
572 else if (alignment == Qt::AlignJustify)
573 value = QString::fromLatin1(ba: "justify");
574 else
575 qWarning() << "QTextOdfWriter: unsupported paragraph alignment; " << format.alignment();
576 if (! value.isNull())
577 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "text-align"), value);
578 }
579
580 if (format.hasProperty(propertyId: QTextFormat::BlockTopMargin))
581 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "margin-top"), value: pixelToPoint(pixels: qMax(a: qreal(0.), b: format.topMargin())) );
582 if (format.hasProperty(propertyId: QTextFormat::BlockBottomMargin))
583 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "margin-bottom"), value: pixelToPoint(pixels: qMax(a: qreal(0.), b: format.bottomMargin())) );
584 if (format.hasProperty(propertyId: QTextFormat::BlockLeftMargin) || format.hasProperty(propertyId: QTextFormat::BlockIndent))
585 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "margin-left"), value: pixelToPoint(pixels: qMax(a: qreal(0.),
586 b: format.leftMargin() + format.indent())));
587 if (format.hasProperty(propertyId: QTextFormat::BlockRightMargin))
588 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "margin-right"), value: pixelToPoint(pixels: qMax(a: qreal(0.), b: format.rightMargin())) );
589 if (format.hasProperty(propertyId: QTextFormat::TextIndent))
590 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "text-indent"), value: pixelToPoint(pixels: format.textIndent()));
591 if (format.hasProperty(propertyId: QTextFormat::PageBreakPolicy)) {
592 if (format.pageBreakPolicy() & QTextFormat::PageBreak_AlwaysBefore)
593 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "break-before"), value: QString::fromLatin1(ba: "page"));
594 if (format.pageBreakPolicy() & QTextFormat::PageBreak_AlwaysAfter)
595 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "break-after"), value: QString::fromLatin1(ba: "page"));
596 }
597 if (format.hasProperty(propertyId: QTextFormat::BackgroundBrush)) {
598 QBrush brush = format.background();
599 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "background-color"), value: brush.color().name());
600 }
601 if (format.hasProperty(propertyId: QTextFormat::BlockNonBreakableLines))
602 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "keep-together"),
603 value: format.nonBreakableLines() ? QString::fromLatin1(ba: "true") : QString::fromLatin1(ba: "false"));
604 if (format.hasProperty(propertyId: QTextFormat::TabPositions)) {
605 QList<QTextOption::Tab> tabs = format.tabPositions();
606 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "tab-stops"));
607 QList<QTextOption::Tab>::Iterator iterator = tabs.begin();
608 while(iterator != tabs.end()) {
609 writer.writeEmptyElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "tab-stop"));
610 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "position"), value: pixelToPoint(pixels: iterator->position) );
611 QString type;
612 switch(iterator->type) {
613 case QTextOption::DelimiterTab: type = QString::fromLatin1(ba: "char"); break;
614 case QTextOption::LeftTab: type = QString::fromLatin1(ba: "left"); break;
615 case QTextOption::RightTab: type = QString::fromLatin1(ba: "right"); break;
616 case QTextOption::CenterTab: type = QString::fromLatin1(ba: "center"); break;
617 }
618 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "type"), value: type);
619 if (!iterator->delimiter.isNull())
620 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "char"), value: iterator->delimiter);
621 ++iterator;
622 }
623
624 writer.writeEndElement(); // tab-stops
625 }
626
627 writer.writeEndElement(); // paragraph-properties
628 writer.writeEndElement(); // style
629}

803void QTextOdfWriter::writeTableFormat(QXmlStreamWriter &writer, QTextTableFormat format, int formatIndex) const
804{
805 // start writing table style element
806 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "style"));
807 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "name"),
808 value: QString::fromLatin1(ba: "Table%1").arg(a: formatIndex));
809 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "family"), value: QString::fromLatin1(ba: "table"));
810 writer.writeEmptyElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "table-properties"));
811
812 if (m_tableFormatsWithBorders.contains(value: formatIndex)) {
813 // write border format collapsing to table style
814 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "border-model"),
815 value: QString::fromLatin1(ba: "collapsing"));
816 }
817 const char* align = nullptr;
818 switch (format.alignment()) {
819 case Qt::AlignLeft:
820 align = "left";
821 break;
822 case Qt::AlignRight:
823 align = "right";
824 break;
825 case Qt::AlignHCenter:
826 align = "center";
827 break;
828 case Qt::AlignJustify:
829 align = "margins";
830 break;
831 }
832 if (align)
833 writer.writeAttribute(namespaceUri: tableNS, name: QString::fromLatin1(ba: "align"), value: QString::fromLatin1(ba: align));
834 if (format.width().rawValue()) {
835 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "width"),
836 value: QString::number(format.width().rawValue()) + "pt"_L1);
837 }
838 writer.writeEndElement();
839 // start writing table-column style element
840 if (format.columnWidthConstraints().size()) {
841 // write table-column-properties for columns with constraints
842 m_tableFormatsWithColWidthConstraints.insert(value: formatIndex); // needed for linking of columns to styles
843 for (int colit = 0; colit < format.columnWidthConstraints().size(); ++colit) {
844 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "style"));
845 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "name"),
846 value: QString::fromLatin1(ba: "Table%1.%2").arg(a: formatIndex).arg(a: colit));
847 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "family"), value: QString::fromLatin1(ba: "table-column"));
848 writer.writeEmptyElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "table-column-properties"));
849 QString columnWidth;
850 if (format.columnWidthConstraints().at(i: colit).type() == QTextLength::PercentageLength) {
851 columnWidth = QString::number(format.columnWidthConstraints().at(i: colit).rawValue())
852 + "%"_L1;
853 } else if (format.columnWidthConstraints().at(i: colit).type() == QTextLength::FixedLength) {
854 columnWidth = QString::number(format.columnWidthConstraints().at(i: colit).rawValue())
855 + "pt"_L1;
856 } else {
857 //!! HARD-CODING variableWidth Constraints to 100% / nr constraints
858 columnWidth = QString::number(100 / format.columnWidthConstraints().size())
859 + "%"_L1;
860 }
861 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "column-width"), value: columnWidth);
862 writer.writeEndElement();
863 }
864 }

886void QTextOdfWriter::tableCellStyleElement(QXmlStreamWriter &writer, const int &formatIndex,
887 const QTextTableCellFormat &format,
888 bool hasBorder, int tableId,
889 const QTextTableFormat tableFormatTmp) const {
890 writer.writeStartElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "style"));
891 if (hasBorder) {
892 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "name"),
893 value: QString::fromLatin1(ba: "TB%1.%2").arg(a: tableId).arg(a: formatIndex));

895 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "name"), value: QString::fromLatin1(ba: "T%1").arg(a: formatIndex));
896 }
897 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "family"), value: QString::fromLatin1(ba: "table-cell"));
898 writer.writeEmptyElement(namespaceUri: styleNS, name: QString::fromLatin1(ba: "table-cell-properties"));
899 if (hasBorder) {
900 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "border"),
901 value: pixelToPoint(pixels: tableFormatTmp.border()) + " "_L1
902 + borderStyleName(style: tableFormatTmp.borderStyle()) + " "_L1
903 + tableFormatTmp.borderBrush().color().name(format: QColor::HexRgb));
904 }

910 }
911 else {
912 if (padding > 0)
913 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "padding-top"), value: pixelToPoint(pixels: padding));
914 padding = format.bottomPadding() + tableFormatTmp.cellPadding();
915 if (padding > 0)

917 value: pixelToPoint(pixels: padding));

919 if (padding > 0)
920 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "padding-left"),
921 value: pixelToPoint(pixels: padding));
922 padding = format.rightPadding() + tableFormatTmp.cellPadding();
923 if (padding > 0)
924 writer.writeAttribute(namespaceUri: foNS, name: QString::fromLatin1(ba: "padding-right"),
925 value: pixelToPoint(pixels: padding));
926 }
927
928 if (format.hasProperty(propertyId: QTextFormat::TextVerticalAlignment)) {
929 QString pos;
930 switch (format.verticalAlignment()) { // TODO - review: doesn't handle all cases
931 case QTextCharFormat::AlignMiddle:
932 pos = QString::fromLatin1(ba: "middle"); break;
933 case QTextCharFormat::AlignTop:
934 pos = QString::fromLatin1(ba: "top"); break;
935 case QTextCharFormat::AlignBottom:
936 pos = QString::fromLatin1(ba: "bottom"); break;
937 default:
938 pos = QString::fromLatin1(ba: "automatic"); break;
939 }
940 writer.writeAttribute(namespaceUri: styleNS, name: QString::fromLatin1(ba: "vertical-align"), value: pos);
941 }
942
943 // TODO
944 // ODF just search for style-table-cell-properties-attlist)
945 // QTextFormat::BackgroundImageUrl
946 // format.background
947 writer.writeEndElement(); // style
948}

968bool QTextOdfWriter::writeAll()
969{
970 if (m_createArchive)
971 m_strategy = new QZipStreamStrategy(m_device);
972 else
973 m_strategy = new QXmlStreamStrategy(m_device);
974
975 if (!m_device->isWritable() && ! m_device->open(mode: QIODevice::WriteOnly)) {
976 qWarning(msg: "QTextOdfWriter::writeAll: the device cannot be opened for writing");
977 return false;
978 }
979 QXmlStreamWriter writer(m_strategy->contentStream);
980 // prettyfy
981 writer.setAutoFormatting(true);
982 writer.setAutoFormattingIndent(2);
983
984 writer.writeNamespace(namespaceUri: officeNS, prefix: QString::fromLatin1(ba: "office"));
985 writer.writeNamespace(namespaceUri: textNS, prefix: QString::fromLatin1(ba: "text"));
986 writer.writeNamespace(namespaceUri: styleNS, prefix: QString::fromLatin1(ba: "style"));
987 writer.writeNamespace(namespaceUri: foNS, prefix: QString::fromLatin1(ba: "fo"));
988 writer.writeNamespace(namespaceUri: tableNS, prefix: QString::fromLatin1(ba: "table"));
989 writer.writeNamespace(namespaceUri: drawNS, prefix: QString::fromLatin1(ba: "draw"));
990 writer.writeNamespace(namespaceUri: xlinkNS, prefix: QString::fromLatin1(ba: "xlink"));
991 writer.writeNamespace(namespaceUri: svgNS, prefix: QString::fromLatin1(ba: "svg"));
992 writer.writeStartDocument();
993 writer.writeStartElement(namespaceUri: officeNS, name: QString::fromLatin1(ba: "document-content"));
994 writer.writeAttribute(namespaceUri: officeNS, name: QString::fromLatin1(ba: "version"), value: QString::fromLatin1(ba: "1.2"));
995
996 // add fragments. (for character formats)
997 QTextDocumentPrivate::FragmentIterator fragIt = QTextDocumentPrivate::get(document: m_document)->begin();
998 QSet<int> formats;
999 while (fragIt != QTextDocumentPrivate::get(document: m_document)->end()) {
1000 const QTextFragmentData * const frag = fragIt.value();
1001 formats << frag->format;
1002 ++fragIt;
1003 }
1004
1005 // add blocks (for blockFormats)
1006 QTextDocumentPrivate::BlockMap &blocks = const_cast<QTextDocumentPrivate *>(QTextDocumentPrivate::get(document: m_document))->blockMap();
1007 QTextDocumentPrivate::BlockMap::Iterator blockIt = blocks.begin();
1008 while (blockIt != blocks.end()) {
1009 const QTextBlockData * const block = blockIt.value();
1010 formats << block->format;
1011 ++blockIt;
1012 }
1013
1014 // add objects for lists, frames and tables
1015 const QList<QTextFormat> allFormats = m_document->allFormats();
1016 const QList<int> copy = formats.values();
1017 for (auto index : copy) {
1018 QTextObject *object = m_document->objectForFormat(allFormats[index]);
1019 if (object) {
1020 formats << object->formatIndex();
1021 if (auto *tableobject = qobject_cast<QTextTable *>(object)) {
1022 if (tableobject->format().borderStyle()) {
1023 int tableID = tableobject->formatIndex();
1024 m_tableFormatsWithBorders.insert(value: tableID);
1025 // loop through all rows and cols of table and store cell IDs,
1026 // create Hash with cell ID as Key and table IDs as Vector
1027 for (int rowindex = 0; rowindex < tableobject->rows(); ++rowindex) {
1028 for (int colindex = 0; colindex < tableobject->columns(); ++colindex) {
1029 const int cellFormatID = tableobject->cellAt(row: rowindex, col: colindex).tableCellFormatIndex();
1030 QList<int> tableIdsTmp;
1031 if (m_cellFormatsInTablesWithBorders.contains(key: cellFormatID))
1032 tableIdsTmp = m_cellFormatsInTablesWithBorders.value(key: cellFormatID);
1033 if (!tableIdsTmp.contains(t: tableID))
1034 tableIdsTmp.append(t: tableID);
1035 m_cellFormatsInTablesWithBorders.insert(key: cellFormatID, value: tableIdsTmp);
1036 }
1037 }
1038 }
1039 }
1040 }
1041 }
1042
1043 writeFormats(writer, formats);
1044
1045 writer.writeStartElement(namespaceUri: officeNS, name: QString::fromLatin1(ba: "body"));
1046 writer.writeStartElement(namespaceUri: officeNS, name: QString::fromLatin1(ba: "text"));
1047 QTextFrame *rootFrame = m_document->rootFrame();
1048 writeFrame(writer, frame: rootFrame);
1049 writer.writeEndElement(); // text
1050 writer.writeEndElement(); // body
1051 writer.writeEndElement(); // document-content
1052 writer.writeEndDocument();
1053 delete m_strategy;
1054 m_strategy = nullptr;
1055
1056 return true;
1057}
