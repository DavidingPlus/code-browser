339inline QFontEngineFT::Glyph *QFontEngineFT::QGlyphSet::getGlyph(glyph_t index,
340 const QFixedPoint &subPixelPosition) const
341{
342 if (useFastGlyphData(index, subPixelPosition))
343 return fast_glyph_data[index];
344 return glyph_data.value(key: GlyphAndSubPixelPosition(index, subPixelPosition));
345}
