178 inline bool isListStart() const
179 { return id == Html_ol || id == Html_ul; }
180 inline bool isTableCell() const
181 { return id == Html_td || id == Html_th; }
182 inline bool isBlock() const
183 { return displayMode == QTextHtmlElement::DisplayBlock; }
184
185 inline bool isNotSelfNesting() const
186 { return id == Html_p || id == Html_li; }
187
188 inline bool allowedInContext(int parentId) const
189 {
190 switch (id) {
191 case Html_dd:
192 case Html_dt: return (parentId == Html_dl);
193 case Html_tr: return (parentId == Html_table
194 || parentId == Html_thead
195 || parentId == Html_tbody
196 || parentId == Html_tfoot
197 );
198 case Html_th:
199 case Html_td: return (parentId == Html_tr);
200 case Html_thead:
201 case Html_tbody:
202 case Html_tfoot: return (parentId == Html_table);
203 case Html_caption: return (parentId == Html_table);
204 case Html_body: return parentId != Html_head;
205 default: break;
206 }
207 return true;
208 }
209
210 inline bool mayNotHaveChildren() const
211 { return id == Html_img || id == Html_hr || id == Html_br || id == Html_meta; }
