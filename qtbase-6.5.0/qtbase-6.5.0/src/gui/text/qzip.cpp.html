31static inline uint readUInt(const uchar *data)
32{
33 return (data[0]) + (data[1]<<8) + (data[2]<<16) + (data[3]<<24);
34}
35
36static inline ushort readUShort(const uchar *data)
37{
38 return (data[0]) + (data[1]<<8);
39}
40
41static inline void writeUInt(uchar *data, uint i)
42{
43 data[0] = i & 0xff;
44 data[1] = (i>>8) & 0xff;
45 data[2] = (i>>16) & 0xff;
46 data[3] = (i>>24) & 0xff;
47}
48
49static inline void writeUShort(uchar *data, ushort i)
50{
51 data[0] = i & 0xff;
52 data[1] = (i>>8) & 0xff;
53}

55static inline void copyUInt(uchar *dest, const uchar *src)
56{
57 dest[0] = src[0];
58 dest[1] = src[1];
59 dest[2] = src[2];
60 dest[3] = src[3];
61}

63static inline void copyUShort(uchar *dest, const uchar *src)
64{
65 dest[0] = src[0];
66 dest[1] = src[1];
67}

69static void writeMSDosDate(uchar *dest, const QDateTime& dt)
70{
71 if (dt.isValid()) {
72 quint16 time =
73 (dt.time().hour() << 11) // 5 bit hour
74 | (dt.time().minute() << 5) // 6 bit minute
75 | (dt.time().second() >> 1); // 5 bit double seconds
76
77 dest[0] = time & 0xff;
78 dest[1] = time >> 8;
79
80 quint16 date =
81 ((dt.date().year() - 1980) << 9) // 7 bit year 1980-based
82 | (dt.date().month() << 5) // 4 bit month
83 | (dt.date().day()); // 5 bit day
84
85 dest[2] = char(date);
86 dest[3] = char(date >> 8);
87 } else {
88 dest[0] = 0;
89 dest[1] = 0;
90 dest[2] = 0;
91 dest[3] = 0;
92 }
93}

95static int inflate(Bytef *dest, ulong *destLen, const Bytef *source, ulong sourceLen)
96{
97 z_stream stream;
98 int err;
99
100 stream.next_in = const_cast<Bytef*>(source);
101 stream.avail_in = (uInt)sourceLen;
102 if ((uLong)stream.avail_in != sourceLen)
103 return Z_BUF_ERROR;
104
105 stream.next_out = dest;
106 stream.avail_out = (uInt)*destLen;
107 if ((uLong)stream.avail_out != *destLen)
108 return Z_BUF_ERROR;
109
110 stream.zalloc = (alloc_func)nullptr;
111 stream.zfree = (free_func)nullptr;
112
113 err = inflateInit2(&stream, -MAX_WBITS);
114 if (err != Z_OK)
115 return err;
116
117 err = inflate(strm: &stream, Z_FINISH);
118 if (err != Z_STREAM_END) {
119 inflateEnd(strm: &stream);
120 if (err == Z_NEED_DICT || (err == Z_BUF_ERROR && stream.avail_in == 0))
121 return Z_DATA_ERROR;
122 return err;
123 }
124 *destLen = stream.total_out;
125
126 err = inflateEnd(strm: &stream);
127 return err;
128}
129
130static int deflate (Bytef *dest, ulong *destLen, const Bytef *source, ulong sourceLen)
131{
132 z_stream stream;
133 int err;
134
135 stream.next_in = const_cast<Bytef*>(source);
136 stream.avail_in = (uInt)sourceLen;
137 stream.next_out = dest;
138 stream.avail_out = (uInt)*destLen;
139 if ((uLong)stream.avail_out != *destLen) return Z_BUF_ERROR;
140
141 stream.zalloc = (alloc_func)nullptr;
142 stream.zfree = (free_func)nullptr;
143 stream.opaque = (voidpf)nullptr;
144
145 err = deflateInit2(&stream, Z_DEFAULT_COMPRESSION, Z_DEFLATED, -MAX_WBITS, 8, Z_DEFAULT_STRATEGY);
146 if (err != Z_OK) return err;
147
148 err = deflate(strm: &stream, Z_FINISH);
149 if (err != Z_STREAM_END) {
150 deflateEnd(strm: &stream);
151 return err == Z_OK ? Z_BUF_ERROR : err;
152 }
153 *destLen = stream.total_out;
154
155 err = deflateEnd(strm: &stream);
156 return err;
157}

191static QFile::Permissions modeToPermissions(quint32 mode)
192{
193 QFile::Permissions ret;
194 if (mode & UnixFileAttributes::ReadUser)
195 ret |= QFile::ReadOwner | QFile::ReadUser;
196 if (mode & UnixFileAttributes::WriteUser)
197 ret |= QFile::WriteOwner | QFile::WriteUser;
198 if (mode & UnixFileAttributes::ExeUser)
199 ret |= QFile::ExeOwner | QFile::ExeUser;
200 if (mode & UnixFileAttributes::ReadGroup)
201 ret |= QFile::ReadGroup;
202 if (mode & UnixFileAttributes::WriteGroup)
203 ret |= QFile::WriteGroup;
204 if (mode & UnixFileAttributes::ExeGroup)
205 ret |= QFile::ExeGroup;
206 if (mode & UnixFileAttributes::ReadOther)
207 ret |= QFile::ReadOther;
208 if (mode & UnixFileAttributes::WriteOther)
209 ret |= QFile::WriteOther;
210 if (mode & UnixFileAttributes::ExeOther)
211 ret |= QFile::ExeOther;
212 return ret;
213}

239static QDateTime readMSDosDate(const uchar *src)
240{
241 uint dosDate = readUInt(data: src);
242 quint64 uDate;
243 uDate = (quint64)(dosDate >> 16);
244 uint tm_mday = (uDate & 0x1f);
245 uint tm_mon = ((uDate & 0x1E0) >> 5);
246 uint tm_year = (((uDate & 0x0FE00) >> 9) + 1980);
247 uint tm_hour = ((dosDate & 0xF800) >> 11);
248 uint tm_min = ((dosDate & 0x7E0) >> 5);
249 uint tm_sec = ((dosDate & 0x1f) << 1);
250
251 return QDateTime(QDate(tm_year, tm_mon, tm_mday), QTime(tm_hour, tm_min, tm_sec));
252}

409QZipReader::FileInfo QZipPrivate::fillFileInfo(int index) const
410{
411 QZipReader::FileInfo fileInfo;
412 FileHeader header = fileHeaders.at(i: index);
413 quint32 mode = readUInt(data: header.h.external_file_attributes);
414 const HostOS hostOS = HostOS(readUShort(data: header.h.version_made) >> 8);
415 switch (hostOS) {
416 case HostUnix:
417 mode = (mode >> 16) & 0xffff;
418 switch (mode & UnixFileAttributes::TypeMask) {
419 case UnixFileAttributes::SymLink:
420 fileInfo.isSymLink = true;
421 break;
422 case UnixFileAttributes::Dir:
423 fileInfo.isDir = true;
424 break;
425 case UnixFileAttributes::File:
426 default: // ### just for the case; should we warn?
427 fileInfo.isFile = true;
428 break;
429 }
430 fileInfo.permissions = modeToPermissions(mode);
431 break;
432 case HostFAT:
433 case HostNTFS:
434 case HostHPFS:
435 case HostVFAT:
436 switch (mode & WindowsFileAttributes::TypeMask) {
437 case WindowsFileAttributes::Dir:
438 fileInfo.isDir = true;
439 break;
440 case WindowsFileAttributes::File:
441 default:
442 fileInfo.isFile = true;
443 break;
444 }
445 fileInfo.permissions |= QFile::ReadOwner | QFile::ReadUser | QFile::ReadGroup | QFile::ReadOther;
446 if ((mode & WindowsFileAttributes::ReadOnly) == 0)
447 fileInfo.permissions |= QFile::WriteOwner | QFile::WriteUser | QFile::WriteGroup | QFile::WriteOther;
448 if (fileInfo.isDir)
449 fileInfo.permissions |= QFile::ExeOwner | QFile::ExeUser | QFile::ExeGroup | QFile::ExeOther;
450 break;
451 default:
452 qWarning(msg: "QZip: Zip entry format at %d is not supported.", index);
453 return fileInfo; // we don't support anything else
454 }
455
456 ushort general_purpose_bits = readUShort(data: header.h.general_purpose_bits);
457 // if bit 11 is set, the filename and comment fields must be encoded using UTF-8
458 const bool inUtf8 = (general_purpose_bits & Utf8Names) != 0;
459 fileInfo.filePath = inUtf8 ? QString::fromUtf8(ba: header.file_name) : QString::fromLocal8Bit(ba: header.file_name);
460 fileInfo.crc = readUInt(data: header.h.crc_32);
461 fileInfo.size = readUInt(data: header.h.uncompressed_size);
462 fileInfo.lastModified = readMSDosDate(src: header.h.last_mod_file);
463
464 // fix the file path, if broken (convert separators, eat leading and trailing ones)
465 fileInfo.filePath = QDir::fromNativeSeparators(pathName: fileInfo.filePath);
466 QStringView filePathRef(fileInfo.filePath);
467 while (filePathRef.startsWith(c: u'.') || filePathRef.startsWith(c: u'/'))
468 filePathRef = filePathRef.mid(pos: 1);
469 while (filePathRef.endsWith(c: u'/'))
470 filePathRef.chop(n: 1);
471
472 fileInfo.filePath = filePathRef.toString();
473 return fileInfo;
474}

525void QZipReaderPrivate::scanFiles()
526{
527 if (!dirtyFileTree)
528 return;
529
530 if (! (device->isOpen() || device->open(mode: QIODevice::ReadOnly))) {
531 status = QZipReader::FileOpenError;
532 return;
533 }
534
535 if ((device->openMode() & QIODevice::ReadOnly) == 0) { // only read the index from readable files.
536 status = QZipReader::FileReadError;
537 return;
538 }
539
540 dirtyFileTree = false;
541 uchar tmp[4];
542 device->read(data: (char *)tmp, maxlen: 4);
543 if (readUInt(data: tmp) != 0x04034b50) {
544 qWarning(msg: "QZip: not a zip file!");
545 return;
546 }
547
548 // find EndOfDirectory header
549 int i = 0;
550 int start_of_directory = -1;
551 int num_dir_entries = 0;
552 EndOfDirectory eod;
553 while (start_of_directory == -1) {
554 const int pos = device->size() - int(sizeof(EndOfDirectory)) - i;
555 if (pos < 0 || i > 65535) {
556 qWarning(msg: "QZip: EndOfDirectory not found");
557 return;
558 }
559
560 device->seek(pos);
561 device->read(data: (char *)&eod, maxlen: sizeof(EndOfDirectory));
562 if (readUInt(data: eod.signature) == 0x06054b50)
563 break;
564 ++i;
565 }
566
567 // have the eod
568 start_of_directory = readUInt(data: eod.dir_start_offset);
569 num_dir_entries = readUShort(data: eod.num_dir_entries);
570 ZDEBUG(msg: "start_of_directory at %d, num_dir_entries=%d", start_of_directory, num_dir_entries);
571 int comment_length = readUShort(data: eod.comment_length);
572 if (comment_length != i)
573 qWarning(msg: "QZip: failed to parse zip file.");
574 comment = device->read(maxlen: qMin(a: comment_length, b: i));
575
576
577 device->seek(pos: start_of_directory);
578 for (i = 0; i < num_dir_entries; ++i) {
579 FileHeader header;
580 int read = device->read(data: (char *) &header.h, maxlen: sizeof(CentralFileHeader));
581 if (read < (int)sizeof(CentralFileHeader)) {
582 qWarning(msg: "QZip: Failed to read complete header, index may be incomplete");
583 break;
584 }
585 if (readUInt(data: header.h.signature) != 0x02014b50) {
586 qWarning(msg: "QZip: invalid header signature, index may be incomplete");
587 break;
588 }
589
590 int l = readUShort(data: header.h.file_name_length);
591 header.file_name = device->read(maxlen: l);
592 if (header.file_name.size() != l) {
593 qWarning(msg: "QZip: Failed to read filename from zip index, index may be incomplete");
594 break;
595 }
596 l = readUShort(data: header.h.extra_field_length);
597 header.extra_field = device->read(maxlen: l);
598 if (header.extra_field.size() != l) {
599 qWarning(msg: "QZip: Failed to read extra field in zip file, skipping file, index may be incomplete");
600 break;
601 }
602 l = readUShort(data: header.h.file_comment_length);
603 header.file_comment = device->read(maxlen: l);
604 if (header.file_comment.size() != l) {
605 qWarning(msg: "QZip: Failed to read read file comment, index may be incomplete");
606 break;
607 }
608
609 ZDEBUG(msg: "found file '%s'", header.file_name.data());
610 fileHeaders.append(t: header);
611 }

614void QZipWriterPrivate::addEntry(EntryType type, const QString &fileName, const QByteArray &contents/*, QFile::Permissions permissions, QZip::Method m*/)
615{
616#ifndef NDEBUG
617 static const char *const entryTypes[] = {
618 "directory",
619 "file ",
620 "symlink " };
621 ZDEBUG() << "adding" << entryTypes[type] <<":" << fileName.toUtf8().data() << (type == 2 ? QByteArray(" -> " + contents).constData() : "");
622#endif
623
624 if (! (device->isOpen() || device->open(mode: QIODevice::WriteOnly))) {
625 status = QZipWriter::FileOpenError;
626 return;
627 }
628 device->seek(pos: start_of_directory);
629
630 // don't compress small files
631 QZipWriter::CompressionPolicy compression = compressionPolicy;
632 if (compressionPolicy == QZipWriter::AutoCompress) {
633 if (contents.size() < 64)
634 compression = QZipWriter::NeverCompress;
635 else
636 compression = QZipWriter::AlwaysCompress;
637 }
638
639 FileHeader header;
640 memset(s: &header.h, c: 0, n: sizeof(CentralFileHeader));
641 writeUInt(data: header.h.signature, i: 0x02014b50);
642
643 writeUShort(data: header.h.version_needed, ZIP_VERSION);
644 writeUInt(data: header.h.uncompressed_size, i: contents.size());
645 writeMSDosDate(dest: header.h.last_mod_file, dt: QDateTime::currentDateTime());
646 QByteArray data = contents;
647 if (compression == QZipWriter::AlwaysCompress) {
648 writeUShort(data: header.h.compression_method, i: CompressionMethodDeflated);
649
650 ulong len = contents.size();
651 // shamelessly copied form zlib
652 len += (len >> 12) + (len >> 14) + 11;
653 int res;
654 do {
655 data.resize(size: len);
656 res = deflate(dest: (uchar*)data.data(), destLen: &len, source: (const uchar*)contents.constData(), sourceLen: contents.size());
657
658 switch (res) {
659 case Z_OK:
660 data.resize(size: len);
661 break;
662 case Z_MEM_ERROR:
663 qWarning(msg: "QZip: Z_MEM_ERROR: Not enough memory to compress file, skipping");
664 data.resize(size: 0);
665 break;
666 case Z_BUF_ERROR:
667 len *= 2;
668 break;
669 }
670 } while (res == Z_BUF_ERROR);
671 }
672// TODO add a check if data.length() > contents.length(). Then try to store the original and revert the compression method to be uncompressed
673 writeUInt(data: header.h.compressed_size, i: data.size());
674 uint crc_32 = ::crc32(crc: 0, buf: nullptr, len: 0);
675 crc_32 = ::crc32(crc: crc_32, buf: (const uchar *)contents.constData(), len: contents.size());
676 writeUInt(data: header.h.crc_32, i: crc_32);
677
678 // if bit 11 is set, the filename and comment fields must be encoded using UTF-8
679 ushort general_purpose_bits = Utf8Names; // always use utf-8
680 writeUShort(data: header.h.general_purpose_bits, i: general_purpose_bits);
681
682 const bool inUtf8 = (general_purpose_bits & Utf8Names) != 0;
683 header.file_name = inUtf8 ? fileName.toUtf8() : fileName.toLocal8Bit();
684 if (header.file_name.size() > 0xffff) {
685 qWarning(msg: "QZip: Filename is too long, chopping it to 65535 bytes");
686 header.file_name = header.file_name.left(len: 0xffff); // ### don't break the utf-8 sequence, if any
687 }
688 if (header.file_comment.size() + header.file_name.size() > 0xffff) {
689 qWarning(msg: "QZip: File comment is too long, chopping it to 65535 bytes");
690 header.file_comment.truncate(pos: 0xffff - header.file_name.size()); // ### don't break the utf-8 sequence, if any
691 }
692 writeUShort(data: header.h.file_name_length, i: header.file_name.size());
693 //h.extra_field_length[2];
694
695 writeUShort(data: header.h.version_made, i: HostUnix << 8);
696 //uchar internal_file_attributes[2];
697 //uchar external_file_attributes[4];
698 quint32 mode = permissionsToMode(perms: permissions);
699 switch (type) {
700 case Symlink:
701 mode |= UnixFileAttributes::SymLink;
702 break;
703 case Directory:
704 mode |= UnixFileAttributes::Dir;
705 break;
706 case File:
707 mode |= UnixFileAttributes::File;
708 break;
709 default:
710 Q_UNREACHABLE();
711 break;
712 }
713 writeUInt(data: header.h.external_file_attributes, i: mode << 16);
714 writeUInt(data: header.h.offset_local_header, i: start_of_directory);
715
716
717 fileHeaders.append(t: header);
718
719 LocalFileHeader h = toLocalHeader(ch: header.h);
720 device->write(data: (const char *)&h, len: sizeof(LocalFileHeader));
721 device->write(data: header.file_name);
722 device->write(data);
723 start_of_directory = device->pos();
724 dirtyFileTree = true;
725}

899/*!
900 Fetch the file contents from the zip archive and return the uncompressed bytes.
901*/
902QByteArray QZipReader::fileData(const QString &fileName) const
903{
904 d->scanFiles();
905 int i;
906 for (i = 0; i < d->fileHeaders.size(); ++i) {
907 if (QString::fromLocal8Bit(ba: d->fileHeaders.at(i).file_name) == fileName)
908 break;
909 }
910 if (i == d->fileHeaders.size())
911 return QByteArray();
912
913 FileHeader header = d->fileHeaders.at(i);
914
915 ushort version_needed = readUShort(data: header.h.version_needed);
916 if (version_needed > ZIP_VERSION) {
917 qWarning(msg: "QZip: .ZIP specification version %d implementationis needed to extract the data.", version_needed);
918 return QByteArray();
919 }
920
921 ushort general_purpose_bits = readUShort(data: header.h.general_purpose_bits);
922 int compressed_size = readUInt(data: header.h.compressed_size);
923 int uncompressed_size = readUInt(data: header.h.uncompressed_size);
924 int start = readUInt(data: header.h.offset_local_header);
925 //qDebug("uncompressing file %d: local header at %d", i, start);
926
927 d->device->seek(pos: start);
928 LocalFileHeader lh;
929 d->device->read(data: (char *)&lh, maxlen: sizeof(LocalFileHeader));
930 uint skip = readUShort(data: lh.file_name_length) + readUShort(data: lh.extra_field_length);
931 d->device->seek(pos: d->device->pos() + skip);
932
933 int compression_method = readUShort(data: lh.compression_method);
934 //qDebug("file=%s: compressed_size=%d, uncompressed_size=%d", fileName.toLocal8Bit().data(), compressed_size, uncompressed_size);
935
936 if ((general_purpose_bits & Encrypted) != 0) {
937 qWarning(msg: "QZip: Unsupported encryption method is needed to extract the data.");
938 return QByteArray();
939 }
940
941 //qDebug("file at %lld", d->device->pos());
942 QByteArray compressed = d->device->read(maxlen: compressed_size);
943 if (compression_method == CompressionMethodStored) {
944 // no compression
945 compressed.truncate(pos: uncompressed_size);
946 return compressed;
947 } else if (compression_method == CompressionMethodDeflated) {
948 // Deflate
949 //qDebug("compressed=%d", compressed.size());
950 compressed.truncate(pos: compressed_size);
951 QByteArray baunzip;
952 ulong len = qMax(a: uncompressed_size, b: 1);
953 int res;
954 do {
955 baunzip.resize(size: len);
956 res = inflate(dest: (uchar*)baunzip.data(), destLen: &len,
957 source: (const uchar*)compressed.constData(), sourceLen: compressed_size);
958
959 switch (res) {
960 case Z_OK:
961 if ((int)len != baunzip.size())
962 baunzip.resize(size: len);
963 break;
964 case Z_MEM_ERROR:
965 qWarning(msg: "QZip: Z_MEM_ERROR: Not enough memory");
966 break;
967 case Z_BUF_ERROR:
968 len *= 2;
969 break;
970 case Z_DATA_ERROR:
971 qWarning(msg: "QZip: Z_DATA_ERROR: Input data is corrupted");
972 break;
973 }
974 } while (res == Z_BUF_ERROR);
975 return baunzip;
976 }
977
978 qWarning(msg: "QZip: Unsupported compression method %d is needed to extract the data.", compression_method);
979 return QByteArray();
980}

982/*!
983 Extracts the full contents of the zip file into \a destinationDir on
984 the local filesystem.
985 In case writing or linking a file fails, the extraction will be aborted.
986*/
987bool QZipReader::extractAll(const QString &destinationDir) const
988{
989 QDir baseDir(destinationDir);
990
991 // create directories first
992 const QList<FileInfo> allFiles = fileInfoList();
993 bool foundDirs = false;
994 bool hasDirs = false;
995 for (const FileInfo &fi : allFiles) {
996 const QString absPath = destinationDir + QDir::separator() + fi.filePath;
997 if (fi.isDir) {
998 foundDirs = true;
999 if (!baseDir.mkpath(dirPath: fi.filePath))
1000 return false;
1001 if (!QFile::setPermissions(filename: absPath, permissionSpec: fi.permissions))
1002 return false;
1003 } else if (!hasDirs && fi.filePath.contains(s: u"/")) {
1004 // filePath does not have leading or trailing '/', so if we find
1005 // one, than the file path contains directories.
1006 hasDirs = true;
1007 }
1008 }
1009
1010 // Some zip archives can be broken in the sense that they do not report
1011 // separate entries for directories, only for files. In this case we
1012 // need to recreate directory structure based on the file paths.
1013 if (hasDirs && !foundDirs) {
1014 for (const FileInfo &fi : allFiles) {
1015 const auto dirPath = fi.filePath.left(n: fi.filePath.lastIndexOf(s: u"/"));
1016 if (!baseDir.mkpath(dirPath))
1017 return false;
1018 // We will leave the directory permissions default in this case,
1019 // because setting dir permissions based on file is incorrect
1020 }
1021 }
1022
1023 // set up symlinks
1024 for (const FileInfo &fi : allFiles) {
1025 const QString absPath = destinationDir + QDir::separator() + fi.filePath;
1026 if (fi.isSymLink) {
1027 QString destination = QFile::decodeName(localFileName: fileData(fileName: fi.filePath));
1028 if (destination.isEmpty())
1029 return false;
1030 QFileInfo linkFi(absPath);
1031 if (!QFile::exists(fileName: linkFi.absolutePath()))
1032 QDir::root().mkpath(dirPath: linkFi.absolutePath());
1033 if (!QFile::link(fileName: destination, newName: absPath))
1034 return false;
1035 /* cannot change permission of links
1036 if (!QFile::setPermissions(absPath, fi.permissions))
1037 return false;
1038 */
1039 }
1040 }
1041
1042 for (const FileInfo &fi : allFiles) {
1043 const QString absPath = destinationDir + QDir::separator() + fi.filePath;
1044 if (fi.isFile) {
1045 QFile f(absPath);
1046 if (!f.open(flags: QIODevice::WriteOnly))
1047 return false;
1048 f.write(data: fileData(fileName: fi.filePath));
1049 f.setPermissions(fi.permissions);
1050 f.close();
1051 }
1052 }
1053
1054 return true;
1055}

1312/*!

1314*/
1315void QZipWriter::close()
1316{
1317 if (!(d->device->openMode() & QIODevice::WriteOnly)) {
1318 d->device->close();
1319 return;
1320 }
1321
1322 //qDebug("QZip::close writing directory, %d entries", d->fileHeaders.size());
1323 d->device->seek(pos: d->start_of_directory);
1324 // write new directory
1325 for (int i = 0; i < d->fileHeaders.size(); ++i) {
1326 const FileHeader &header = d->fileHeaders.at(i);
1327 d->device->write(data: (const char *)&header.h, len: sizeof(CentralFileHeader));
1328 d->device->write(data: header.file_name);
1329 d->device->write(data: header.extra_field);
1330 d->device->write(data: header.file_comment);
1331 }
1332 int dir_size = d->device->pos() - d->start_of_directory;
1333 // write end of directory
1334 EndOfDirectory eod;
1335 memset(s: &eod, c: 0, n: sizeof(EndOfDirectory));
1336 writeUInt(data: eod.signature, i: 0x06054b50);
1337 //uchar this_disk[2];
1338 //uchar start_of_directory_disk[2];
1339 writeUShort(data: eod.num_dir_entries_this_disk, i: d->fileHeaders.size());
1340 writeUShort(data: eod.num_dir_entries, i: d->fileHeaders.size());
1341 writeUInt(data: eod.directory_size, i: dir_size);
1342 writeUInt(data: eod.dir_start_offset, i: d->start_of_directory);
1343 writeUShort(data: eod.comment_length, i: d->comment.size());
1344
1345 d->device->write(data: (const char *)&eod, len: sizeof(EndOfDirectory));
1346 d->device->write(data: d->comment);
