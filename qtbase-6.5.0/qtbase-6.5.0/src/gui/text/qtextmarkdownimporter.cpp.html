118void QTextMarkdownImporter::import(QTextDocument *doc, const QString &markdown)
119{
120 MD_PARSER callbacks = {
121 .abi_version: 0, // abi_version
122 .flags: unsigned(m_features),
123 .enter_block: &CbEnterBlock,
124 .leave_block: &CbLeaveBlock,
125 .enter_span: &CbEnterSpan,
126 .leave_span: &CbLeaveSpan,
127 .text: &CbText,
128 .debug_log: &CbDebugLog,
129 .syntax: nullptr // syntax
130 };
131 m_doc = doc;
132 m_paragraphMargin = m_doc->defaultFont().pointSize() * 2 / 3;
133 m_cursor = new QTextCursor(doc);
134 doc->clear();
135 if (doc->defaultFont().pointSize() != -1)
136 m_monoFont.setPointSize(doc->defaultFont().pointSize());
137 else
138 m_monoFont.setPixelSize(doc->defaultFont().pixelSize());
139 qCDebug(lcMD) << "default font" << doc->defaultFont() << "mono font" << m_monoFont;
140 QByteArray md = markdown.toUtf8();
141 md_parse(text: md.constData(), size: MD_SIZE(md.size()), parser: &callbacks, userdata: this);
142 delete m_cursor;
143 m_cursor = nullptr;
144}

146int QTextMarkdownImporter::cbEnterBlock(int blockType, void *det)
147{
148 m_blockType = blockType;
149 switch (blockType) {
150 case MD_BLOCK_P:
151 if (!m_listStack.isEmpty())
152 qCDebug(lcMD, m_listItem ? "P of LI at level %d" : "P continuation inside LI at level %d", int(m_listStack.size()));
153 else
154 qCDebug(lcMD, "P");
155 m_needsInsertBlock = true;
156 break;
157 case MD_BLOCK_QUOTE:
158 ++m_blockQuoteDepth;
159 qCDebug(lcMD, "QUOTE level %d", m_blockQuoteDepth);
160 break;
161 case MD_BLOCK_CODE: {
162 MD_BLOCK_CODE_DETAIL *detail = static_cast<MD_BLOCK_CODE_DETAIL *>(det);
163 m_codeBlock = true;
164 m_blockCodeLanguage = QLatin1StringView(detail->lang.text, int(detail->lang.size));
165 m_blockCodeFence = detail->fence_char;
166 QString info = QLatin1StringView(detail->info.text, int(detail->info.size));
167 m_needsInsertBlock = true;
168 if (m_blockQuoteDepth)
169 qCDebug(lcMD, "CODE lang '%s' info '%s' fenced with '%c' inside QUOTE %d", qPrintable(m_blockCodeLanguage), qPrintable(info), m_blockCodeFence, m_blockQuoteDepth);
170 else
171 qCDebug(lcMD, "CODE lang '%s' info '%s' fenced with '%c'", qPrintable(m_blockCodeLanguage), qPrintable(info), m_blockCodeFence);
172 } break;
173 case MD_BLOCK_H: {
174 MD_BLOCK_H_DETAIL *detail = static_cast<MD_BLOCK_H_DETAIL *>(det);
175 QTextBlockFormat blockFmt;
176 QTextCharFormat charFmt;
177 int sizeAdjustment = 4 - int(detail->level); // H1 to H6: +3 to -2
178 charFmt.setProperty(propertyId: QTextFormat::FontSizeAdjustment, value: sizeAdjustment);
179 charFmt.setFontWeight(QFont::Bold);
180 blockFmt.setHeadingLevel(int(detail->level));
181 m_needsInsertBlock = false;
182 if (m_doc->isEmpty()) {
183 m_cursor->setBlockFormat(blockFmt);
184 m_cursor->setCharFormat(charFmt);
185 } else {
186 m_cursor->insertBlock(format: blockFmt, charFormat: charFmt);
187 }
188 qCDebug(lcMD, "H%d", detail->level);
189 } break;
190 case MD_BLOCK_LI: {
191 m_needsInsertBlock = true;
192 m_listItem = true;
193 MD_BLOCK_LI_DETAIL *detail = static_cast<MD_BLOCK_LI_DETAIL *>(det);
194 m_markerType = detail->is_task ?
195 (detail->task_mark == ' ' ? QTextBlockFormat::MarkerType::Unchecked : QTextBlockFormat::MarkerType::Checked) :
196 QTextBlockFormat::MarkerType::NoMarker;
197 qCDebug(lcMD) << "LI";
198 } break;
199 case MD_BLOCK_UL: {
200 if (m_needsInsertList) // list nested in an empty list
201 m_listStack.push(t: m_cursor->insertList(format: m_listFormat));
202 else
203 m_needsInsertList = true;
204 MD_BLOCK_UL_DETAIL *detail = static_cast<MD_BLOCK_UL_DETAIL *>(det);
205 m_listFormat = QTextListFormat();
206 m_listFormat.setIndent(m_listStack.size() + 1);
207 switch (detail->mark) {
208 case '*':
209 m_listFormat.setStyle(QTextListFormat::ListCircle);
210 break;
211 case '+':
212 m_listFormat.setStyle(QTextListFormat::ListSquare);
213 break;
214 default: // including '-'
215 m_listFormat.setStyle(QTextListFormat::ListDisc);
216 break;
217 }
218 qCDebug(lcMD, "UL %c level %d", detail->mark, int(m_listStack.size()) + 1);
219 } break;
220 case MD_BLOCK_OL: {
221 if (m_needsInsertList) // list nested in an empty list
222 m_listStack.push(t: m_cursor->insertList(format: m_listFormat));
223 else
224 m_needsInsertList = true;
225 MD_BLOCK_OL_DETAIL *detail = static_cast<MD_BLOCK_OL_DETAIL *>(det);
226 m_listFormat = QTextListFormat();
227 m_listFormat.setIndent(m_listStack.size() + 1);
228 m_listFormat.setNumberSuffix(QChar::fromLatin1(c: detail->mark_delimiter));
229 m_listFormat.setStyle(QTextListFormat::ListDecimal);
230 qCDebug(lcMD, "OL xx%d level %d", detail->mark_delimiter, int(m_listStack.size()) + 1);
231 } break;
232 case MD_BLOCK_TD: {
233 MD_BLOCK_TD_DETAIL *detail = static_cast<MD_BLOCK_TD_DETAIL *>(det);
234 ++m_tableCol;
235 // absolute movement (and storage of m_tableCol) shouldn't be necessary, but
236 // movePosition(QTextCursor::NextCell) doesn't work
237 QTextTableCell cell = m_currentTable->cellAt(row: m_tableRowCount - 1, col: m_tableCol);
238 if (!cell.isValid()) {
239 qWarning(msg: "malformed table in Markdown input");
240 return 1;
241 }
242 *m_cursor = cell.firstCursorPosition();
243 QTextBlockFormat blockFmt = m_cursor->blockFormat();
244 blockFmt.setAlignment(MdAlignment(a: detail->align));
245 m_cursor->setBlockFormat(blockFmt);
246 qCDebug(lcMD) << "TD; align" << detail->align << MdAlignment(a: detail->align) << "col" << m_tableCol;
247 } break;
248 case MD_BLOCK_TH: {
249 ++m_tableColumnCount;
250 ++m_tableCol;
251 if (m_currentTable->columns() < m_tableColumnCount)
252 m_currentTable->appendColumns(count: 1);
253 auto cell = m_currentTable->cellAt(row: m_tableRowCount - 1, col: m_tableCol);
254 if (!cell.isValid()) {
255 qWarning(msg: "malformed table in Markdown input");
256 return 1;
257 }
258 auto fmt = cell.format();
259 fmt.setFontWeight(QFont::Bold);
260 cell.setFormat(fmt);
261 } break;
262 case MD_BLOCK_TR: {
263 ++m_tableRowCount;
264 m_nonEmptyTableCells.clear();
265 if (m_currentTable->rows() < m_tableRowCount)
266 m_currentTable->appendRows(count: 1);
267 m_tableCol = -1;
268 qCDebug(lcMD) << "TR" << m_currentTable->rows();
269 } break;
270 case MD_BLOCK_TABLE:
271 m_tableColumnCount = 0;
272 m_tableRowCount = 0;
273 m_currentTable = m_cursor->insertTable(rows: 1, cols: 1); // we don't know the dimensions yet
274 break;
275 case MD_BLOCK_HR: {
276 qCDebug(lcMD, "HR");
277 QTextBlockFormat blockFmt;
278 blockFmt.setProperty(propertyId: QTextFormat::BlockTrailingHorizontalRulerWidth, value: 1);
279 m_cursor->insertBlock(format: blockFmt, charFormat: QTextCharFormat());
280 } break;
281 default:
282 break; // nothing to do for now
283 }
284 return 0; // no error
285}

287int QTextMarkdownImporter::cbLeaveBlock(int blockType, void *detail)
288{
289 Q_UNUSED(detail);
290 switch (blockType) {
291 case MD_BLOCK_P:
292 m_listItem = false;
293 break;
294 case MD_BLOCK_UL:
295 case MD_BLOCK_OL:
296 if (Q_UNLIKELY(m_needsInsertList))
297 m_listStack.push(t: m_cursor->createList(format: m_listFormat));
298 if (Q_UNLIKELY(m_listStack.isEmpty())) {
299 qCWarning(lcMD, "list ended unexpectedly");
300 } else {
301 qCDebug(lcMD, "list at level %d ended", int(m_listStack.size()));
302 m_listStack.pop();
303 }
304 break;
305 case MD_BLOCK_TR: {
306 // https://github.com/mity/md4c/issues/29
307 // MD4C doesn't tell us explicitly which cells are merged, so merge empty cells
308 // with previous non-empty ones
309 int mergeEnd = -1;
310 int mergeBegin = -1;
311 for (int col = m_tableCol; col >= 0; --col) {
312 if (m_nonEmptyTableCells.contains(t: col)) {
313 if (mergeEnd >= 0 && mergeBegin >= 0) {
314 qCDebug(lcMD) << "merging cells" << mergeBegin << "to" << mergeEnd << "inclusive, on row" << m_currentTable->rows() - 1;
315 m_currentTable->mergeCells(row: m_currentTable->rows() - 1, col: mergeBegin - 1, numRows: 1, numCols: mergeEnd - mergeBegin + 2);
316 }
317 mergeEnd = -1;
318 mergeBegin = -1;
319 } else {
320 if (mergeEnd < 0)
321 mergeEnd = col;
322 else
323 mergeBegin = col;
324 }
325 }
326 } break;
327 case MD_BLOCK_QUOTE: {
328 qCDebug(lcMD, "QUOTE level %d ended", m_blockQuoteDepth);
329 --m_blockQuoteDepth;
330 m_needsInsertBlock = true;
331 } break;
332 case MD_BLOCK_TABLE:
333 qCDebug(lcMD) << "table ended with" << m_currentTable->columns() << "cols and" << m_currentTable->rows() << "rows";
334 m_currentTable = nullptr;
335 m_cursor->movePosition(op: QTextCursor::End);
336 break;
337 case MD_BLOCK_LI:
338 qCDebug(lcMD, "LI at level %d ended", int(m_listStack.size()));
339 m_listItem = false;
340 break;
341 case MD_BLOCK_CODE: {
342 m_codeBlock = false;
343 m_blockCodeLanguage.clear();
344 m_blockCodeFence = 0;
345 if (m_blockQuoteDepth)
346 qCDebug(lcMD, "CODE ended inside QUOTE %d", m_blockQuoteDepth);
347 else
348 qCDebug(lcMD, "CODE ended");
349 m_needsInsertBlock = true;
350 } break;
351 case MD_BLOCK_H:
352 m_cursor->setCharFormat(QTextCharFormat());
353 break;
354 default:
355 break;
356 }
357 return 0; // no error
358}

360int QTextMarkdownImporter::cbEnterSpan(int spanType, void *det)
361{
362 QTextCharFormat charFmt;
363 if (!m_spanFormatStack.isEmpty())
364 charFmt = m_spanFormatStack.top();
365 switch (spanType) {
366 case MD_SPAN_EM:
367 charFmt.setFontItalic(true);
368 break;
369 case MD_SPAN_STRONG:
370 charFmt.setFontWeight(QFont::Bold);
371 break;
372 case MD_SPAN_U:
373 charFmt.setFontUnderline(true);
374 break;
375 case MD_SPAN_A: {
376 MD_SPAN_A_DETAIL *detail = static_cast<MD_SPAN_A_DETAIL *>(det);
377 QString url = QString::fromUtf8(utf8: detail->href.text, size: int(detail->href.size));
378 QString title = QString::fromUtf8(utf8: detail->title.text, size: int(detail->title.size));
379 charFmt.setAnchor(true);
380 charFmt.setAnchorHref(url);
381 if (!title.isEmpty())
382 charFmt.setToolTip(title);
383 charFmt.setForeground(m_palette.link());
384 qCDebug(lcMD) << "anchor" << url << title;
385 } break;
386 case MD_SPAN_IMG: {
387 m_imageSpan = true;
388 m_imageFormat = QTextImageFormat();
389 MD_SPAN_IMG_DETAIL *detail = static_cast<MD_SPAN_IMG_DETAIL *>(det);
390 m_imageFormat.setName(QString::fromUtf8(utf8: detail->src.text, size: int(detail->src.size)));
391 m_imageFormat.setProperty(propertyId: QTextFormat::ImageTitle, value: QString::fromUtf8(utf8: detail->title.text, size: int(detail->title.size)));
392 break;
393 }
394 case MD_SPAN_CODE:
395 charFmt.setFont(font: m_monoFont);
396 charFmt.setFontFixedPitch(true);
397 break;
398 case MD_SPAN_DEL:
399 charFmt.setFontStrikeOut(true);
400 break;
401 }
402 m_spanFormatStack.push(t: charFmt);
403 qCDebug(lcMD) << spanType << "setCharFormat" << charFmt.font().families().first()
404 << charFmt.fontWeight() << (charFmt.fontItalic() ? "italic" : "")
405 << charFmt.foreground().color().name();
406 m_cursor->setCharFormat(charFmt);
407 return 0; // no error
408}

410int QTextMarkdownImporter::cbLeaveSpan(int spanType, void *detail)
411{
412 Q_UNUSED(detail);
413 QTextCharFormat charFmt;
414 if (!m_spanFormatStack.isEmpty()) {
415 m_spanFormatStack.pop();
416 if (!m_spanFormatStack.isEmpty())
417 charFmt = m_spanFormatStack.top();
418 }
419 m_cursor->setCharFormat(charFmt);
420 qCDebug(lcMD) << spanType << "setCharFormat" << charFmt.font().families().first()
421 << charFmt.fontWeight() << (charFmt.fontItalic() ? "italic" : "")
422 << charFmt.foreground().color().name();
423 if (spanType == int(MD_SPAN_IMG))
424 m_imageSpan = false;
425 return 0; // no error
426}

428int QTextMarkdownImporter::cbText(int textType, const char *text, unsigned size)
429{
430 if (m_needsInsertBlock)
431 insertBlock();
432#if QT_CONFIG(regularexpression)
433 static const QRegularExpression openingBracket(QStringLiteral("<[a-zA-Z]"));
434 static const QRegularExpression closingBracket(QStringLiteral("(/>|</)"));
435#endif
436 QString s = QString::fromUtf8(utf8: text, size: int(size));
437
438 switch (textType) {
439 case MD_TEXT_NORMAL:
440#if QT_CONFIG(regularexpression)
441 if (m_htmlTagDepth) {
442 m_htmlAccumulator += s;
443 s = QString();
444 }
445#endif
446 break;
447 case MD_TEXT_NULLCHAR:
448 s = QString(QChar(u'\xFFFD')); // CommonMark-required replacement for null
449 break;
450 case MD_TEXT_BR:
451 s = QString(qtmi_Newline);
452 break;
453 case MD_TEXT_SOFTBR:
454 s = QString(qtmi_Space);
455 break;
456 case MD_TEXT_CODE:
457 // We'll see MD_SPAN_CODE too, which will set the char format, and that's enough.
458 break;
459#if QT_CONFIG(texthtmlparser)
460 case MD_TEXT_ENTITY:
461 if (m_htmlTagDepth)
462 m_htmlAccumulator += s;
463 else
464 m_cursor->insertHtml(html: s);
465 s = QString();
466 break;
467#endif
468 case MD_TEXT_HTML:
469 // count how many tags are opened and how many are closed
470#if QT_CONFIG(regularexpression) && QT_CONFIG(texthtmlparser)
471 {
472 int startIdx = 0;
473 while ((startIdx = s.indexOf(re: openingBracket, from: startIdx)) >= 0) {
474 ++m_htmlTagDepth;
475 startIdx += 2;
476 }
477 startIdx = 0;
478 while ((startIdx = s.indexOf(re: closingBracket, from: startIdx)) >= 0) {
479 --m_htmlTagDepth;
480 startIdx += 2;
481 }
482 }
483 m_htmlAccumulator += s;
484 if (!m_htmlTagDepth) { // all open tags are now closed
485 qCDebug(lcMD) << "HTML" << m_htmlAccumulator;
486 m_cursor->insertHtml(html: m_htmlAccumulator);
487 if (m_spanFormatStack.isEmpty())
488 m_cursor->setCharFormat(QTextCharFormat());
489 else
490 m_cursor->setCharFormat(m_spanFormatStack.top());
491 m_htmlAccumulator = QString();
492 }
493#endif
494 s = QString();
495 break;
496 }
497
498 switch (m_blockType) {
499 case MD_BLOCK_TD:
500 m_nonEmptyTableCells.append(t: m_tableCol);
501 break;
502 case MD_BLOCK_CODE:
503 if (s == qtmi_Newline) {
504 // defer a blank line until we see something else in the code block,
505 // to avoid ending every code block with a gratuitous blank line
506 m_needsInsertBlock = true;
507 s = QString();
508 }
509 break;
510 default:
511 break;
512 }
513
514 if (m_imageSpan) {
515 // TODO we don't yet support alt text with formatting, because of the cases where m_cursor
516 // already inserted the text above. Rather need to accumulate it in case we need it here.
517 m_imageFormat.setProperty(propertyId: QTextFormat::ImageAltText, value: s);
518 qCDebug(lcMD) << "image" << m_imageFormat.name()
519 << "title" << m_imageFormat.stringProperty(propertyId: QTextFormat::ImageTitle)
520 << "alt" << s << "relative to" << m_doc->baseUrl();
521 m_cursor->insertImage(format: m_imageFormat);
522 return 0; // no error
523 }
524
525 if (!s.isEmpty())
526 m_cursor->insertText(text: s);
527 if (m_cursor->currentList()) {
528 // The list item will indent the list item's text, so we don't need indentation on the block.
529 QTextBlockFormat bfmt = m_cursor->blockFormat();
530 bfmt.setIndent(0);
531 m_cursor->setBlockFormat(bfmt);
532 }
533 if (lcMD().isEnabled(type: QtDebugMsg)) {
534 QTextBlockFormat bfmt = m_cursor->blockFormat();
535 QString debugInfo;
536 if (m_cursor->currentList())
537 debugInfo = "in list at depth "_L1 + QString::number(m_cursor->currentList()->format().indent());
538 if (bfmt.hasProperty(propertyId: QTextFormat::BlockQuoteLevel))
539 debugInfo += "in blockquote at depth "_L1 +
540 QString::number(bfmt.intProperty(propertyId: QTextFormat::BlockQuoteLevel));
541 if (bfmt.hasProperty(propertyId: QTextFormat::BlockCodeLanguage))
542 debugInfo += "in a code block"_L1;
543 qCDebug(lcMD) << textType << "in block" << m_blockType << s << qPrintable(debugInfo)
544 << "bindent" << bfmt.indent() << "tindent" << bfmt.textIndent()
545 << "margins" << bfmt.leftMargin() << bfmt.topMargin() << bfmt.bottomMargin() << bfmt.rightMargin();
546 }
547 return 0; // no error
548}

550/*!
551 Insert a new block based on stored state.
552
553 m_cursor cannot store the state for the _next_ block ahead of time, because
554 m_cursor->setBlockFormat() controls the format of the block that the cursor
555 is already in; so cbLeaveBlock() cannot call setBlockFormat() without
556 altering the block that was just added. Therefore cbLeaveBlock() and the
557 following cbEnterBlock() set variables to remember what formatting should
558 come next, and insertBlock() is called just before the actual text
559 insertion, to create a new block with the right formatting.
560*/
561void QTextMarkdownImporter::insertBlock()
562{
563 QTextCharFormat charFormat;
564 if (!m_spanFormatStack.isEmpty())
565 charFormat = m_spanFormatStack.top();
566 QTextBlockFormat blockFormat;
567 if (!m_listStack.isEmpty() && !m_needsInsertList && m_listItem) {
568 QTextList *list = m_listStack.top();
569 if (list)
570 blockFormat = list->item(i: list->count() - 1).blockFormat();
571 else
572 qWarning() << "attempted to insert into a list that no longer exists";
573 }
574 if (m_blockQuoteDepth) {
575 blockFormat.setProperty(propertyId: QTextFormat::BlockQuoteLevel, value: m_blockQuoteDepth);
576 blockFormat.setLeftMargin(qtmi_BlockQuoteIndent * m_blockQuoteDepth);
577 blockFormat.setRightMargin(qtmi_BlockQuoteIndent);
578 }
579 if (m_codeBlock) {
580 blockFormat.setProperty(propertyId: QTextFormat::BlockCodeLanguage, value: m_blockCodeLanguage);
581 if (m_blockCodeFence) {
582 blockFormat.setNonBreakableLines(true);
583 blockFormat.setProperty(propertyId: QTextFormat::BlockCodeFence, value: QString(QLatin1Char(m_blockCodeFence)));
584 }
585 charFormat.setFont(font: m_monoFont);
586 } else {
587 blockFormat.setTopMargin(m_paragraphMargin);
588 blockFormat.setBottomMargin(m_paragraphMargin);
589 }
590 if (m_markerType == QTextBlockFormat::MarkerType::NoMarker)
591 blockFormat.clearProperty(propertyId: QTextFormat::BlockMarker);
592 else
593 blockFormat.setMarker(m_markerType);
594 if (!m_listStack.isEmpty())
595 blockFormat.setIndent(m_listStack.size());
596 if (m_doc->isEmpty()) {
597 m_cursor->setBlockFormat(blockFormat);
598 m_cursor->setCharFormat(charFormat);
599 } else if (m_listItem) {
600 m_cursor->insertBlock(format: blockFormat, charFormat: QTextCharFormat());
601 m_cursor->setCharFormat(charFormat);
602 } else {
603 m_cursor->insertBlock(format: blockFormat, charFormat);
604 }
605 if (m_needsInsertList) {
606 m_listStack.push(t: m_cursor->createList(format: m_listFormat));
607 } else if (!m_listStack.isEmpty() && m_listItem && m_listStack.top()) {
608 m_listStack.top()->add(block: m_cursor->block());
609 }
610 m_needsInsertList = false;
611 m_needsInsertBlock = false;
612}
