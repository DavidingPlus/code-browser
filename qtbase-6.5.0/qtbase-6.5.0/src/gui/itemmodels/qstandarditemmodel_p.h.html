31class QStandardItemData
32{
33public:
34 inline QStandardItemData() : role(-1) {}
35 inline QStandardItemData(int r, const QVariant &v) : role(r), value(v) {}
36 inline QStandardItemData(const std::pair<const int&, const QVariant&> &p) : role(p.first), value(p.second) {}

38 QVariant value;
39 inline bool operator==(const QStandardItemData &other) const { return role == other.role && value == other.value; }

70class QStandardItemPrivate
71{
72 Q_DECLARE_PUBLIC(QStandardItem)
73public:
74 inline QStandardItemPrivate()
75 : model(nullptr),
76 parent(nullptr),
77 rows(0),
78 columns(0),
79 q_ptr(nullptr),
80 lastKnownIndex(-1)
81 { }
82
83 inline int childIndex(int row, int column) const {
84 if ((row < 0) || (column < 0)
85 || (row >= rowCount()) || (column >= columnCount())) {
86 return -1;
87 }
88 return (row * columnCount()) + column;
89 }
90 inline int childIndex(const QStandardItem *child) const {
91 const int lastChild = children.size() - 1;
92 int &childsLastIndexInParent = child->d_func()->lastKnownIndex;
93 if (childsLastIndexInParent != -1 && childsLastIndexInParent <= lastChild) {
94 if (children.at(i: childsLastIndexInParent) == child)
95 return childsLastIndexInParent;
96 } else {
97 childsLastIndexInParent = lastChild / 2;
98 }
99
100 // assuming the item is in the vicinity of the previous index, iterate forwards and
101 // backwards through the children
102 int backwardIter = childsLastIndexInParent - 1;
103 int forwardIter = childsLastIndexInParent;
104 for (;;) {
105 if (forwardIter <= lastChild) {
106 if (children.at(i: forwardIter) == child) {
107 childsLastIndexInParent = forwardIter;
108 break;
109 }
110 ++forwardIter;
111 } else if (backwardIter < 0) {
112 childsLastIndexInParent = -1;
113 break;
114 }
115 if (backwardIter >= 0) {
116 if (children.at(i: backwardIter) == child) {
117 childsLastIndexInParent = backwardIter;
118 break;
119 }
120 --backwardIter;
121 }
122 }
123 return childsLastIndexInParent;
124 }
125 QPair<int, int> position() const;
126 void setChild(int row, int column, QStandardItem *item,
127 bool emitChanged = false);
128 inline int rowCount() const {
129 return rows;
130 }
131 inline int columnCount() const {
132 return columns;
133 }
134 void childDeleted(QStandardItem *child);
135
136 void setModel(QStandardItemModel *mod);
137
138 inline void setParentAndModel(
139 QStandardItem *par,
140 QStandardItemModel *mod) {
141 setModel(mod);
142 parent = par;
143 }
144
145 void changeFlags(bool enable, Qt::ItemFlags f);
146 void setItemData(const QMap<int, QVariant> &roles);
147 QMap<int, QVariant> itemData() const;
148
149 bool insertRows(int row, int count, const QList<QStandardItem*> &items);
150 bool insertRows(int row, const QList<QStandardItem*> &items);
151 bool insertColumns(int column, int count, const QList<QStandardItem*> &items);
152
153 void sortChildren(int column, Qt::SortOrder order);
154
155 QStandardItemModel *model;
156 QStandardItem *parent;

158 QList<QStandardItem *> children;
159 int rows;

165};

177 inline QStandardItem *createItem() const {
178 return itemPrototype ? itemPrototype->clone() : new QStandardItem;
179 }
180
181 inline QStandardItem *itemFromIndex(const QModelIndex &index) const {
182 Q_Q(const QStandardItemModel);
183 if (!index.isValid())
184 return root.data();
185 if (index.model() != q)
186 return nullptr;
187 QStandardItem *parent = static_cast<QStandardItem*>(index.internalPointer());
188 if (parent == nullptr)
189 return nullptr;
190 return parent->child(row: index.row(), column: index.column());
191 }
