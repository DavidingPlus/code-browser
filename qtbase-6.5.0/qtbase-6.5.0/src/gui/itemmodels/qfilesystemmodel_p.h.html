146 void updateIcon(QAbstractFileIconProvider *iconProvider, const QString &path) {
147 if (info)
148 info->icon = iconProvider->icon(QFileInfo(path));
149 for (QFileSystemNode *child : std::as_const(t&: children)) {
150 //On windows the root (My computer) has no path so we don't want to add a / for nothing (e.g. /C:/)
151 if (!path.isEmpty()) {
152 if (path.endsWith(c: u'/'))
153 child->updateIcon(iconProvider, path: path + child->fileName);
154 else
155 child->updateIcon(iconProvider, path: path + u'/' + child->fileName);
156 } else
157 child->updateIcon(iconProvider, path: child->fileName);
158 }
159 }

167 if (path.endsWith(c: u'/'))

173 }

209 inline int translateVisibleLocation(QFileSystemNode *parent, int row) const {
210 if (sortOrder != Qt::AscendingOrder) {
211 if (parent->dirtyChildrenIndex == -1)
212 return parent->visibleChildren.size() - row - 1;
213
214 if (row < parent->dirtyChildrenIndex)
215 return parent->dirtyChildrenIndex - row - 1;
216 }
217
218 return row;
219 }
