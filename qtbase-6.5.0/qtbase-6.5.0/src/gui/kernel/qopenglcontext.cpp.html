70/*!
71 \class QOpenGLContext
72 \inmodule QtGui
73 \since 5.0
74 \brief The QOpenGLContext class represents a native OpenGL context, enabling
75 OpenGL rendering on a QSurface.
76
77 QOpenGLContext represents the OpenGL state of an underlying OpenGL context.
78 To set up a context, set its screen and format such that they match those
79 of the surface or surfaces with which the context is meant to be used, if
80 necessary make it share resources with other contexts with
81 setShareContext(), and finally call create(). Use the return value or isValid()
82 to check if the context was successfully initialized.
83
84 A context can be made current against a given surface by calling
85 makeCurrent(). When OpenGL rendering is done, call swapBuffers() to swap
86 the front and back buffers of the surface, so that the newly rendered
87 content becomes visible. To be able to support certain platforms,
88 QOpenGLContext requires that you call makeCurrent() again before starting
89 rendering a new frame, after calling swapBuffers().
90
91 If the context is temporarily not needed, such as when the application is
92 not rendering, it can be useful to delete it in order to free resources.
93 You can connect to the aboutToBeDestroyed() signal to clean up any
94 resources that have been allocated with different ownership from the
95 QOpenGLContext itself.
96
97 Once a QOpenGLContext has been made current, you can render to it in a
98 platform independent way by using Qt's OpenGL enablers such as
99 QOpenGLFunctions, QOpenGLBuffer, QOpenGLShaderProgram, and
100 QOpenGLFramebufferObject. It is also possible to use the platform's OpenGL
101 API directly, without using the Qt enablers, although potentially at the
102 cost of portability. The latter is necessary when wanting to use OpenGL 1.x
103 or OpenGL ES 1.x.
104
105 For more information about the OpenGL API, refer to the official
106 \l{http://www.opengl.org}{OpenGL documentation}.
107
108 For an example of how to use QOpenGLContext see the
109 \l{OpenGL Window Example}{OpenGL Window} example.
110
111 \section1 Thread Affinity
112
113 QOpenGLContext can be moved to a different thread with moveToThread(). Do
114 not call makeCurrent() from a different thread than the one to which the
115 QOpenGLContext object belongs. A context can only be current in one thread
116 and against one surface at a time, and a thread only has one context
117 current at a time.
118
119 \section1 Context Resource Sharing
120
121 Resources such as textures and vertex buffer objects
122 can be shared between contexts. Use setShareContext() before calling
123 create() to specify that the contexts should share these resources.
124 QOpenGLContext internally keeps track of a QOpenGLContextGroup object which
125 can be accessed with shareGroup(), and which can be used to find all the
126 contexts in a given share group. A share group consists of all contexts that
127 have been successfully initialized and are sharing with an existing context in
128 the share group. A non-sharing context has a share group consisting of a
129 single context.
130
131 \section1 Default Framebuffer
132
133 On certain platforms, a framebuffer other than 0 might be the default frame
134 buffer depending on the current surface. Instead of calling
135 glBindFramebuffer(0), it is recommended that you use
136 glBindFramebuffer(ctx->defaultFramebufferObject()), to ensure that your
137 application is portable between different platforms. However, if you use
138 QOpenGLFunctions::glBindFramebuffer(), this is done automatically for you.
139
140 \sa QOpenGLFunctions, QOpenGLBuffer, QOpenGLShaderProgram, QOpenGLFramebufferObject
141*/
142
143/*!

145
146 Set the current context. Returns the previously current context.
147*/
148QOpenGLContext *QOpenGLContextPrivate::setCurrentContext(QOpenGLContext *context)
149{
150 QGuiGLThreadContext *threadContext = qwindow_context_storage()->localData();
151 if (!threadContext) {
152 if (!QThread::currentThread()) {
153 qWarning(msg: "No QTLS available. currentContext won't work");
154 return nullptr;
155 }
156 threadContext = new QGuiGLThreadContext;
157 qwindow_context_storage()->setLocalData(threadContext);
158 }
159 QOpenGLContext *previous = threadContext->context;
160 threadContext->context = context;
161 return previous;
162}

164int QOpenGLContextPrivate::maxTextureSize()
165{
166 if (max_texture_size != -1)
167 return max_texture_size;
168
169 Q_Q(QOpenGLContext);
170 QOpenGLFunctions *funcs = q->functions();
171 funcs->glGetIntegerv(GL_MAX_TEXTURE_SIZE, params: &max_texture_size);
172
173#if !QT_CONFIG(opengles2)
174 if (!q->isOpenGLES()) {
175 GLenum proxy = GL_PROXY_TEXTURE_2D;
176
177 GLint size;
178 GLint next = 64;
179 funcs->glTexImage2D(target: proxy, level: 0, GL_RGBA, width: next, height: next, border: 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels: nullptr);
180
181 QOpenGLExtraFunctions *extraFuncs = q->extraFunctions();
182 extraFuncs->glGetTexLevelParameteriv(target: proxy, level: 0, GL_TEXTURE_WIDTH, params: &size);
183
184 if (size == 0) {
185 return max_texture_size;
186 }
187 do {
188 size = next;
189 next = size * 2;
190
191 if (next > max_texture_size)
192 break;
193 funcs->glTexImage2D(target: proxy, level: 0, GL_RGBA, width: next, height: next, border: 0, GL_RGBA, GL_UNSIGNED_BYTE, pixels: nullptr);
194 extraFuncs->glGetTexLevelParameteriv(target: proxy, level: 0, GL_TEXTURE_WIDTH, params: &next);
195 } while (next > size);
196
197 max_texture_size = size;
198 }
199#endif // QT_CONFIG(opengles2)
200
201 return max_texture_size;
202}

326/*!
327 Attempts to create the OpenGL context with the current configuration.
328
329 The current configuration includes the format, the share context, and the
330 screen.
331
332 If the OpenGL implementation on your system does not support the requested
333 version of OpenGL context, then QOpenGLContext will try to create the closest
334 matching version. The actual created context properties can be queried
335 using the QSurfaceFormat returned by the format() function. For example, if
336 you request a context that supports OpenGL 4.3 Core profile but the driver
337 and/or hardware only supports version 3.2 Core profile contexts then you will
338 get a 3.2 Core profile context.
339
340 Returns \c true if the native context was successfully created and is ready to
341 be used with makeCurrent(), swapBuffers(), etc.
342
343 \note If the context already exists, this function destroys the existing
344 context first, and then creates a new one.
345
346 \sa makeCurrent(), format()
347*/
348bool QOpenGLContext::create()
349{
350 Q_D(QOpenGLContext);
351 if (d->platformGLContext)
352 destroy();
353
354 auto *platformContext = QGuiApplicationPrivate::platformIntegration()->createPlatformOpenGLContext(context: this);
355 if (!platformContext)
356 return false;
357
358 d->adopt(platformContext);
359
360 return isValid();
361}

381/*!
382 \internal
383
384 Destroy the underlying platform context associated with this context.
385
386 If any other context is directly or indirectly sharing resources with this
387 context, the shared resources, which includes vertex buffer objects, shader
388 objects, textures, and framebuffer objects, are not freed. However,
389 destroying the underlying platform context frees any state associated with
390 the context.
391
392 After \c destroy() has been called, you must call create() if you wish to
393 use the context again.
394
395 \note This implicitly calls doneCurrent() if the context is current.
396
397 \sa create()
398*/
399void QOpenGLContext::destroy()
400{
401 Q_D(QOpenGLContext);
402
403 // Notify that the native context and the QPlatformOpenGLContext are going
404 // to go away.
405 if (d->platformGLContext)
406 emit aboutToBeDestroyed();
407
408 // Invoke callbacks for helpers and invalidate.
409 if (d->textureFunctionsDestroyCallback) {
410 d->textureFunctionsDestroyCallback();
411 d->textureFunctionsDestroyCallback = nullptr;
412 }
413 d->textureFunctions = nullptr;
414
415 delete d->versionFunctions;
416 d->versionFunctions = nullptr;
417
418 if (d->vaoHelperDestroyCallback) {
419 Q_ASSERT(d->vaoHelper);
420 d->vaoHelperDestroyCallback(d->vaoHelper);
421 d->vaoHelperDestroyCallback = nullptr;
422 }
423 d->vaoHelper = nullptr;
424
425 // Tear down function wrappers.
426 delete d->versionFunctions;
427 d->versionFunctions = nullptr;
428
429 delete d->functions;
430 d->functions = nullptr;
431
432 // Clean up and destroy the native context machinery.
433 if (QOpenGLContext::currentContext() == this)
434 doneCurrent();
435
436 if (d->shareGroup)
437 d->shareGroup->d_func()->removeContext(ctx: this);
438
439 d->shareGroup = nullptr;
440
441 delete d->platformGLContext;
442 d->platformGLContext = nullptr;
443}

472
473 On some platforms the return value of \c false for a context that was

490 \sa create()
491*/

494 Q_D(const QOpenGLContext);

570/*!
571 Call this to get the default framebuffer object for the current surface.
572
573 On some platforms (for instance, iOS) the default framebuffer object depends
574 on the surface being rendered to, and might be different from 0. Thus,
575 instead of calling glBindFramebuffer(0), you should call
576 glBindFramebuffer(ctx->defaultFramebufferObject()) if you want your
577 application to work across different Qt platforms.
578
579 If you use the glBindFramebuffer() in QOpenGLFunctions you do not have to
580 worry about this, as it automatically binds the current context's
581 defaultFramebufferObject() when 0 is passed.
582
583 \note Widgets that render via framebuffer objects, like QOpenGLWidget and
584 QQuickWidget, will override the value returned from this function when
585 painting is active, because at that time the correct "default" framebuffer
586 is the widget's associated backing framebuffer, not the platform-specific
587 one belonging to the top-level window's surface. This ensures the expected
588 behavior for this function and other classes relying on it (for example,
589 QOpenGLFramebufferObject::bindDefault() or
590 QOpenGLFramebufferObject::release()).
591
592 \sa QOpenGLFramebufferObject
593*/
594GLuint QOpenGLContext::defaultFramebufferObject() const
595{
596 if (!isValid())
597 return 0;
598
599 Q_D(const QOpenGLContext);
600 if (!d->surface || !d->surface->surfaceHandle())
601 return 0;
602
603 if (d->defaultFboRedirect)
604 return d->defaultFboRedirect;
605
606 return d->platformGLContext->defaultFramebufferObject(surface: d->surface->surfaceHandle());

609/*!
610 Makes the context current in the current thread, against the given
611 \a surface. Returns \c true if successful; otherwise returns \c false.
612 The latter may happen if the surface is not exposed, or the graphics
613 hardware is not available due to e.g. the application being suspended.
614
615 If \a surface is \nullptr this is equivalent to calling doneCurrent().
616
617 Avoid calling this function from a different thread than the one the
618 QOpenGLContext instance lives in. If you wish to use QOpenGLContext from a
619 different thread you should first make sure it's not current in the
620 current thread, by calling doneCurrent() if necessary. Then call
621 moveToThread(otherThread) before using it in the other thread.
622
623 By default Qt employs a check that enforces the above condition on the
624 thread affinity. It is still possible to disable this check by setting the
625 \c{Qt::AA_DontCheckOpenGLContextThreadAffinity} application attribute. Be
626 sure to understand the consequences of using QObjects from outside
627 the thread they live in, as explained in the
628 \l{QObject#Thread Affinity}{QObject thread affinity} documentation.
629
630 \sa functions(), doneCurrent(), Qt::AA_DontCheckOpenGLContextThreadAffinity
631*/
632bool QOpenGLContext::makeCurrent(QSurface *surface)
633{
634 Q_D(QOpenGLContext);
635 if (!isValid())
636 return false;
637
638 if (Q_UNLIKELY(!qApp->testAttribute(Qt::AA_DontCheckOpenGLContextThreadAffinity)
639 && thread() != QThread::currentThread())) {
640 qFatal(msg: "Cannot make QOpenGLContext current in a different thread");
641 }
642
643 if (!surface) {
644 doneCurrent();
645 return true;
646 }
647
648 if (!surface->surfaceHandle())
649 return false;
650 if (!surface->supportsOpenGL()) {
651 qWarning() << "QOpenGLContext::makeCurrent() called with non-opengl surface" << surface;
652 return false;
653 }
654
655 if (!d->platformGLContext->makeCurrent(surface: surface->surfaceHandle()))
656 return false;
657
658 QOpenGLContextPrivate::setCurrentContext(this);
659#ifndef QT_NO_DEBUG
660 QOpenGLContextPrivate::toggleMakeCurrentTracker(this, true);
661#endif
662
663 d->surface = surface;
664
665 static bool needsWorkaroundSet = false;
666 static bool needsWorkaround = false;
667
668 if (!needsWorkaroundSet) {
669 QByteArray env;
670#ifdef Q_OS_ANDROID
671 env = qgetenv(QByteArrayLiteral("QT_ANDROID_DISABLE_GLYPH_CACHE_WORKAROUND"));
672 needsWorkaround = env.isEmpty() || env == QByteArrayLiteral("0") || env == QByteArrayLiteral("false");
673#endif
674 env = qgetenv(QByteArrayLiteral("QT_ENABLE_GLYPH_CACHE_WORKAROUND"));
675 if (env == QByteArrayLiteral("1") || env == QByteArrayLiteral("true"))
676 needsWorkaround = true;
677
678 if (!needsWorkaround) {
679 const char *rendererString = reinterpret_cast<const char *>(functions()->glGetString(GL_RENDERER));
680 if (rendererString)
681 needsWorkaround =
682 qstrncmp(str1: rendererString, str2: "Mali-4xx", len: 6) == 0 // Mali-400, Mali-450
683 || qstrcmp(str1: rendererString, str2: "Mali-T880") == 0
684 || qstrncmp(str1: rendererString, str2: "Adreno (TM) 2xx", len: 13) == 0 // Adreno 200, 203, 205
685 || qstrncmp(str1: rendererString, str2: "Adreno 2xx", len: 8) == 0 // Same as above but without the '(TM)'
686 || qstrncmp(str1: rendererString, str2: "Adreno (TM) 3xx", len: 13) == 0 // Adreno 302, 305, 320, 330
687 || qstrncmp(str1: rendererString, str2: "Adreno 3xx", len: 8) == 0 // Same as above but without the '(TM)'
688 || qstrncmp(str1: rendererString, str2: "Adreno (TM) 4xx", len: 13) == 0 // Adreno 405, 418, 420, 430
689 || qstrncmp(str1: rendererString, str2: "Adreno 4xx", len: 8) == 0 // Same as above but without the '(TM)'
690 || qstrncmp(str1: rendererString, str2: "Adreno (TM) 5xx", len: 13) == 0 // Adreno 505, 506, 510, 530, 540
691 || qstrncmp(str1: rendererString, str2: "Adreno 5xx", len: 8) == 0 // Same as above but without the '(TM)'
692 || qstrncmp(str1: rendererString, str2: "Adreno (TM) 6xx", len: 13) == 0 // Adreno 610, 620, 630
693 || qstrncmp(str1: rendererString, str2: "Adreno 6xx", len: 8) == 0 // Same as above but without the '(TM)'
694 || qstrcmp(str1: rendererString, str2: "GC800 core") == 0
695 || qstrcmp(str1: rendererString, str2: "GC1000 core") == 0
696 || strstr(haystack: rendererString, needle: "GC2000") != nullptr
697 || qstrcmp(str1: rendererString, str2: "Immersion.16") == 0
698 || qstrncmp(str1: rendererString, str2: "Apple Mx", len: 7) == 0;
699 }
700 needsWorkaroundSet = true;
701 }
702
703 if (needsWorkaround)
704 d->workaround_brokenFBOReadBack = true;
705
706 d->shareGroup->d_func()->deletePendingResources(ctx: this);
707
708 return true;
709}

711/*!
712 Convenience function for calling makeCurrent with a 0 surface.
713
714 This results in no context being current in the current thread.
715
716 \sa makeCurrent(), currentContext()
717*/
718void QOpenGLContext::doneCurrent()
719{
720 Q_D(QOpenGLContext);
721 if (!isValid())
722 return;
723

1098/*!
1099 \class QOpenGLSharedResource
1100 \internal
1101 \since 5.0
1102 \brief The QOpenGLSharedResource class is used to keep track of resources
1103 that are shared between OpenGL contexts (like textures, framebuffer
1104 objects, shader programs, etc), and clean them up in a safe way when
1105 they're no longer needed.
1106 \inmodule QtGui
1107
1108 The QOpenGLSharedResource instance should never be deleted, instead free()
1109 should be called when it's no longer needed. Thus it will be put on a queue
1110 and freed at an appropriate time (when a context in the share group becomes
1111 current).
1112
1113 The sub-class needs to implement two pure virtual functions. The first,
1114 freeResource() must be implemented to actually do the freeing, for example
1115 call glDeleteTextures() on a texture id. Qt makes sure a valid context in
1116 the resource's share group is current at the time. The other,
1117 invalidateResource(), is called by Qt in the circumstance when the last
1118 context in the share group is destroyed before free() has been called. The
1119 implementation of invalidateResource() should set any identifiers to 0 or
1120 set a flag to prevent them from being used later on.
1121*/
1122QOpenGLSharedResource::QOpenGLSharedResource(QOpenGLContextGroup *group)
1123 : m_group(group)
1124{
1125 const auto locker = qt_scoped_lock(mutex&: m_group->d_func()->m_mutex);
1126 m_group->d_func()->m_sharedResources << this;
1127}

1134void QOpenGLSharedResource::free()
1135{
1136 if (!m_group) {
1137 delete this;
1138 return;

1141 const auto locker = qt_scoped_lock(mutex&: m_group->d_func()->m_mutex);
1142 m_group->d_func()->m_sharedResources.removeOne(t: this);
