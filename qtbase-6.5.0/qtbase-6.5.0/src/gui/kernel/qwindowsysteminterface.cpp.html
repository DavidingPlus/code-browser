43/*!
44 \internal
45
46 The QWindowSystemHelper creates synchronously delivered events on the stack, unless
47 the calling thread is not the Gui thread.
48
49 Asynchronously delivered events, and events created outside the Gui thread are
50 allocated on the heap.
51*/
52
53template<typename Delivery>
54struct QWindowSystemHelper
55{
56 template<typename EventType, typename ...Args>
57 static bool handleEvent(Args ...);
58};
59
60/*
61 Handles a window system event.
62
63 By default this function posts the event on the window system event queue and
64 wakes the Gui event dispatcher. Qt Gui will then handle the event asynchronously
65 at a later point. The return value is not used in asynchronous mode and will
66 always be true.
67
68 In synchronous mode Qt Gui will process the event immediately. The return value
69 indicates if Qt accepted the event. If the event is delivered from another thread
70 than the Qt main thread the window system event queue is flushed, which may deliver
71 other events as well.
72
73 \sa flushWindowSystemEvents(), setSynchronousWindowSystemEvents()
74*/
75template<>
76template<typename EventType, typename ...Args>
77bool QWindowSystemHelper<QWindowSystemInterface::DefaultDelivery>::handleEvent(Args ...args)
78{
79 return QWindowSystemInterfacePrivate::synchronousWindowSystemEvents
80 ? QWindowSystemHelper<QWindowSystemInterface::SynchronousDelivery>::handleEvent<EventType>(args...)
81 : QWindowSystemHelper<QWindowSystemInterface::AsynchronousDelivery>::handleEvent<EventType>(args...);
82}
83
84/*
85 Handles a window system event synchronously.
86
87 Qt Gui will process the event immediately. The return value indicates if Qt
88 accepted the event.
89
90 If the event is delivered from another thread than the Qt main thread the
91 window system event queue is flushed, which may deliver other events as
92 well.
93*/
94template<>
95template<typename EventType, typename ...Args>
96bool QWindowSystemHelper<QWindowSystemInterface::SynchronousDelivery>::handleEvent(Args ...args)
97{
98 if (QThread::currentThread() == QGuiApplication::instance()->thread()) {
99 EventType event(args...);
100 // Process the event immediately on the Gui thread and return the accepted state
101 if (QWindowSystemInterfacePrivate::eventHandler) {
102 if (!QWindowSystemInterfacePrivate::eventHandler->sendEvent(event: &event))
103 return false;
104 } else {
105 QGuiApplicationPrivate::processWindowSystemEvent(e: &event);
106 }
107 return event.eventAccepted;
108 } else {
109 // Post the event on the Qt main thread queue and flush the queue.
110 // This will wake up the Gui thread which will process the event.
111 // Return the accepted state for the last event on the queue,
112 // which is the event posted by this function.
113 QWindowSystemHelper<QWindowSystemInterface::AsynchronousDelivery>::handleEvent<EventType>(args...);
114 return QWindowSystemInterface::flushWindowSystemEvents();
115 }
116}
117
118/*
119 Handles a window system event asynchronously by posting the event to Qt Gui.
120
121 This function posts the event on the window system event queue and wakes the
122 Gui event dispatcher. Qt Gui will then handle the event asynchronously at a
123 later point.
124*/
125template<>
126template<typename EventType, typename ...Args>
127bool QWindowSystemHelper<QWindowSystemInterface::AsynchronousDelivery>::handleEvent(Args ...args)
128{
129 QWindowSystemInterfacePrivate::windowSystemEventQueue.append(e: new EventType(args...));
130 if (QAbstractEventDispatcher *dispatcher = QGuiApplicationPrivate::qt_qpa_core_dispatcher())
131 dispatcher->wakeUp();
132 return true;
133}
134
135template <typename EventType, typename Delivery = QWindowSystemInterface::DefaultDelivery, typename ...Args>
136static bool handleWindowSystemEvent(Args ...args)
137{
138 return QWindowSystemHelper<Delivery>::template handleEvent<EventType>(args...);
139}

312/*! \internal
313 Handles an expose event.
314
315 The platform plugin sends expose events when an area of the window
316 is invalidated or window exposure changes. \a region is in window
317 local coordinates. An empty region indicates that the window is
318 obscured, but note that the exposed property of the QWindow will be set
319 based on what QPlatformWindow::isExposed() returns at the time of this call,
320 not based on what the region is. // FIXME: this should probably be fixed.
321
322 The platform plugin may omit sending expose events (or send obscure
323 events) for windows that are on screen but where the client area is
324 completely covered by other windows or otherwise not visible. Expose
325 event consumers can then use this to disable updates for such windows.
326 This is required behavior on platforms where OpenGL swapbuffers stops
327 blocking for obscured windows (like macOS).
328*/

443bool QWindowSystemInterface::handleShortcutEvent(QWindow *window, ulong timestamp, int keyCode, Qt::KeyboardModifiers modifiers, quint32 nativeScanCode,
444 quint32 nativeVirtualKey, quint32 nativeModifiers, const QString &text, bool autorepeat, ushort count)
445{
446#if QT_CONFIG(shortcut)
447 if (!window)
448 window = QGuiApplication::focusWindow();
449
450 QShortcutMap &shortcutMap = QGuiApplicationPrivate::instance()->shortcutMap;
451 if (shortcutMap.state() == QKeySequence::NoMatch) {
452 // Check if the shortcut is overridden by some object in the event delivery path (typically the focus object).
453 // If so, we should not look up the shortcut in the shortcut map, but instead deliver the event as a regular
454 // key event, so that the target that accepted the shortcut override event can handle it. Note that we only
455 // do this if the shortcut map hasn't found a partial shortcut match yet. If it has, the shortcut can not be
456 // overridden.
457 bool overridden = handleWindowSystemEvent<QWindowSystemInterfacePrivate::KeyEvent, SynchronousDelivery>
458 (args: window,args: timestamp, args: QEvent::ShortcutOverride, args: keyCode, args: modifiers, args: nativeScanCode,
459 args: nativeVirtualKey, args: nativeModifiers, args: text, args: autorepeat, args: count);
460 if (overridden)
461 return false;
462 }
463
464 // The shortcut event is dispatched as a QShortcutEvent, not a QKeyEvent, but we use
465 // the QKeyEvent as a container for the various properties that the shortcut map needs
466 // to inspect to determine if a shortcut matched the keys that were pressed.
467 QKeyEvent keyEvent(QEvent::ShortcutOverride, keyCode, modifiers, nativeScanCode,
468 nativeVirtualKey, nativeModifiers, text, autorepeat, count);
469
470 return shortcutMap.tryShortcut(e: &keyEvent);
471#else
472 Q_UNUSED(window);
473 Q_UNUSED(timestamp);
474 Q_UNUSED(keyCode);
475 Q_UNUSED(modifiers);
476 Q_UNUSED(nativeScanCode);
477 Q_UNUSED(nativeVirtualKey);
478 Q_UNUSED(nativeModifiers);
479 Q_UNUSED(text);
480 Q_UNUSED(autorepeat);
481 Q_UNUSED(count);
482 return false;
483#endif
484}

532bool QWindowSystemInterface::handleWheelEvent(QWindow *window, ulong timestamp, const QPointingDevice *device,
533 const QPointF &local, const QPointF &global, QPoint pixelDelta, QPoint angleDelta,
534 Qt::KeyboardModifiers mods, Qt::ScrollPhase phase,
535 Qt::MouseEventSource source, bool invertedScrolling)
536{
537 // Qt 4 sends two separate wheel events for horizontal and vertical
538 // deltas. For Qt 5 we want to send the deltas in one event, but at the
539 // same time preserve source and behavior compatibility with Qt 4.
540 //
541 // In addition high-resolution pixel-based deltas are also supported.
542 // Platforms that does not support these may pass a null point here.
543 // Angle deltas must always be sent in addition to pixel deltas.
544
545 // Pass Qt::ScrollBegin and Qt::ScrollEnd through
546 // even if the wheel delta is null.
547 if (angleDelta.isNull() && phase == Qt::ScrollUpdate)
548 return false;
549
550 // Simple case: vertical deltas only:
551 if (angleDelta.y() != 0 && angleDelta.x() == 0) {
552 return handleWindowSystemEvent<QWindowSystemInterfacePrivate::WheelEvent>(args: window,
553 args: timestamp, args: QHighDpi::fromNativeLocalPosition(value: local, context: window), args: QHighDpi::fromNativeGlobalPosition(value: global, context: window),
554 args: pixelDelta, args: angleDelta, args: angleDelta.y(), args: Qt::Vertical, args: mods, args: phase, args: source, args: invertedScrolling, args: device);
555 }
556
557 // Simple case: horizontal deltas only:
558 if (angleDelta.y() == 0 && angleDelta.x() != 0) {
559 return handleWindowSystemEvent<QWindowSystemInterfacePrivate::WheelEvent>(args: window,
560 args: timestamp, args: QHighDpi::fromNativeLocalPosition(value: local, context: window), args: QHighDpi::fromNativeGlobalPosition(value: global, context: window),
561 args: pixelDelta, args: angleDelta, args: angleDelta.x(), args: Qt::Horizontal, args: mods, args: phase, args: source, args: invertedScrolling, args: device);
562 }
563
564 bool acceptVert;
565 bool acceptHorz;
566 // Both horizontal and vertical deltas: Send two wheel events.
567 // The first event contains the Qt 5 pixel and angle delta as points,
568 // and in addition the Qt 4 compatibility vertical angle delta.
569 acceptVert = handleWindowSystemEvent<QWindowSystemInterfacePrivate::WheelEvent>(args: window,
570 args: timestamp, args: QHighDpi::fromNativeLocalPosition(value: local, context: window), args: QHighDpi::fromNativeGlobalPosition(value: global, context: window),
571 args: pixelDelta, args: angleDelta, args: angleDelta.y(), args: Qt::Vertical, args: mods, args: phase, args: source, args: invertedScrolling, args: device);
572
573 // The second event contains null pixel and angle points and the
574 // Qt 4 compatibility horizontal angle delta.
575 acceptHorz = handleWindowSystemEvent<QWindowSystemInterfacePrivate::WheelEvent>(args: window,

604/*!
605 \internal
606 Convert a list of \l QWindowSystemInterface::TouchPoint \a points to a list
607 of \e temporary QEventPoint instances, scaled (but not localized)
608 for delivery to the given \a window.
609
610 This is called from QWindowSystemInterface::handleTouchEvent():
611 that is too early to update the QEventPoint instances in QPointingDevice,
612 because we want those to hold "current" state from the applcation's
613 point of view. The QWindowSystemInterfacePrivate::TouchEvent, to which
614 the returned touchpoints will "belong", might go through the queue before
615 being processed; the application doesn't see the equivalent QTouchEvent
616 until later on. Therefore the responsibility to update the QEventPoint
617 instances in QPointingDevice is in QGuiApplication, not here.
618
619 QGuiApplicationPrivate::processMouseEvent() also calls this function
620 when it synthesizes a touch event from a mouse event. But that's outside
621 the normal use case.
622
623 It might be better if we change all the platform plugins to create
624 temporary instances of QEventPoint directly, and remove
625 QWindowSystemInterface::TouchPoint completely. Then we will no longer need
626 this function either. But that's only possible as long as QEventPoint
627 remains a Q_GADGET, not a QObject, so that it continues to be small and
628 suitable for temporary stack allocation. QEventPoint is a little bigger
629 than QWindowSystemInterface::TouchPoint, though.
630*/
631QList<QEventPoint>
632 QWindowSystemInterfacePrivate::fromNativeTouchPoints(const QList<QWindowSystemInterface::TouchPoint> &points,
633 const QWindow *window, QEvent::Type *type)
634{
635 QList<QEventPoint> touchPoints;
636 QEventPoint::States states;
637
638 touchPoints.reserve(size: points.size());
639 QList<QWindowSystemInterface::TouchPoint>::const_iterator point = points.constBegin();
640 QList<QWindowSystemInterface::TouchPoint>::const_iterator end = points.constEnd();
641 while (point != end) {
642 QPointF globalPos = QHighDpi::fromNativePixels(value: point->area.center(), context: window);
643 QEventPoint p(point->id, point->state, globalPos, globalPos);
644 states |= point->state;
645 if (point->uniqueId >= 0)
646 QMutableEventPoint::setUniqueId(p, arg: QPointingDeviceUniqueId::fromNumericId(id: point->uniqueId));
647 QMutableEventPoint::setPressure(p, arg: point->pressure);
648 QMutableEventPoint::setRotation(p, arg: point->rotation);
649 QMutableEventPoint::setEllipseDiameters(p, arg: QHighDpi::fromNativePixels(value: point->area.size(), context: window));
650 QMutableEventPoint::setVelocity(p, arg: QHighDpi::fromNativePixels(value: point->velocity, context: window));
651
652 // The local pos is not set: it will be calculated
653 // when the event gets processed by QGuiApplication.
654
655 touchPoints.append(t: p);
656 ++point;
657 }
658
659 // Determine the event type based on the combined point states.
660 if (type) {
661 *type = QEvent::TouchUpdate;
662 if (states == QEventPoint::State::Pressed)
663 *type = QEvent::TouchBegin;
664 else if (states == QEventPoint::State::Released)
665 *type = QEvent::TouchEnd;
666 }
667
668 return touchPoints;
669}

671QWindowSystemInterface::TouchPoint
672QWindowSystemInterfacePrivate::toNativeTouchPoint(const QEventPoint &pt, const QWindow *window)
673{
674 QWindowSystemInterface::TouchPoint p;
675 p.id = pt.id();
676 QRectF area(QPointF(), pt.ellipseDiameters());
677 area.moveCenter(p: pt.globalPosition());
678 // TODO store ellipseDiameters in QWindowSystemInterface::TouchPoint or just use QEventPoint
679 p.area = QHighDpi::toNativePixels(value: area, context: window);
680 p.pressure = pt.pressure();
681 p.state = pt.state();
682 p.velocity = QHighDpi::toNativePixels(value: pt.velocity(), context: window);
683 return p;
684}

1040/*!
1041 Make Qt Gui process all events on the event queue immediately. Return the
1042 accepted state for the last event on the queue.
1043*/
1044bool QWindowSystemInterface::flushWindowSystemEvents(QEventLoop::ProcessEventsFlags flags)
1045{
1046 const int count = QWindowSystemInterfacePrivate::windowSystemEventQueue.count();
1047 if (!count)
1048 return false;
1049 if (!QGuiApplication::instance()) {
1050 qWarning().nospace()
1051 << "QWindowSystemInterface::flushWindowSystemEvents() invoked after "
1052 "QGuiApplication destruction, discarding " << count << " events.";
1053 QWindowSystemInterfacePrivate::windowSystemEventQueue.clear();
1054 return false;
1055 }
1056 if (QThread::currentThread() != QGuiApplication::instance()->thread()) {
1057 // Post a FlushEvents event which will trigger a call back to
1058 // deferredFlushWindowSystemEvents from the Gui thread.
1059 QMutexLocker locker(&QWindowSystemInterfacePrivate::flushEventMutex);
1060 handleWindowSystemEvent<QWindowSystemInterfacePrivate::FlushEventsEvent, AsynchronousDelivery>(args: flags);
1061 QWindowSystemInterfacePrivate::eventsFlushed.wait(lockedMutex: &QWindowSystemInterfacePrivate::flushEventMutex);
1062 } else {
1063 sendWindowSystemEvents(flags);
1064 }
1065 return QWindowSystemInterfacePrivate::eventAccepted.loadRelaxed() > 0;
1066}
1067
1068void QWindowSystemInterface::deferredFlushWindowSystemEvents(QEventLoop::ProcessEventsFlags flags)
1069{
1070 Q_ASSERT(QThread::currentThread() == QGuiApplication::instance()->thread());
1071
1072 QMutexLocker locker(&QWindowSystemInterfacePrivate::flushEventMutex);
1073 sendWindowSystemEvents(flags);
1074 QWindowSystemInterfacePrivate::eventsFlushed.wakeOne();
1075}
1076
1077bool QWindowSystemInterface::sendWindowSystemEvents(QEventLoop::ProcessEventsFlags flags)
1078{
1079 int nevents = 0;
1080
1081 while (QWindowSystemInterfacePrivate::windowSystemEventsQueued()) {
1082 QWindowSystemInterfacePrivate::WindowSystemEvent *event =
1083 flags & QEventLoop::ExcludeUserInputEvents ?
1084 QWindowSystemInterfacePrivate::getNonUserInputWindowSystemEvent() :
1085 QWindowSystemInterfacePrivate::getWindowSystemEvent();
1086 if (!event)
1087 break;
1088
1089 if (QWindowSystemInterfacePrivate::eventHandler) {
1090 if (QWindowSystemInterfacePrivate::eventHandler->sendEvent(e: event))
1091 nevents++;
1092 } else {
1093 nevents++;
1094 QGuiApplicationPrivate::processWindowSystemEvent(e: event);
1095 }
1096
1097 // Record the accepted state for the processed event
1098 // (excluding flush events). This state can then be
1099 // returned by flushWindowSystemEvents().
1100 if (event->type != QWindowSystemInterfacePrivate::FlushEvents)
1101 QWindowSystemInterfacePrivate::eventAccepted.storeRelaxed(newValue: event->eventAccepted);
1102
1103 delete event;
1104 }
1105
1106 return (nevents > 0);
1107}
