98#if defined(Q_OS_UNIX)
99struct Q_GUI_EXPORT QWaylandWindow : public QObject
100{
101 Q_OBJECT
102public:
103 QT_DECLARE_NATIVE_INTERFACE(QWaylandWindow, 1, QWindow)
104
105 virtual wl_surface *surface() const = 0;
106 virtual void setCustomMargins(const QMargins &margins) = 0;
107 virtual void requestXdgActivationToken(uint serial) = 0;
108 template<typename T>
109 T *surfaceRole() const
110 {
111 std::any anyRole = _surfaceRole();
112 auto role = std::any_cast<T *>(&anyRole);
113 return role ? *role : nullptr;
114 }
115Q_SIGNALS:
116 void surfaceCreated();
117 void surfaceDestroyed();
118 void xdgActivationTokenCreated(const QString &token);
119
120protected:
121 virtual std::any _surfaceRole() const = 0;
122};
