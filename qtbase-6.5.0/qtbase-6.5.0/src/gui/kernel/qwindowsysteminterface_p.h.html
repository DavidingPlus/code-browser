74 class WindowSystemEvent {
75 public:
76 enum {
77 Synthetic = 0x1,
78 NullWindow = 0x2
79 };
80
81 explicit WindowSystemEvent(EventType t)
82 : type(t), flags(0), eventAccepted(true) { }
83 virtual ~WindowSystemEvent() { }
84
85 bool synthetic() const { return flags & Synthetic; }

435 class WindowSystemEventList {
436 QList<WindowSystemEvent *> impl;
437 mutable QMutex mutex;
438 public:
439 WindowSystemEventList() : impl(), mutex() {}
440 ~WindowSystemEventList() { clear(); }
441
442 void clear()
443 { const QMutexLocker locker(&mutex); qDeleteAll(c: impl); impl.clear(); }
444 void prepend(WindowSystemEvent *e)
445 { const QMutexLocker locker(&mutex); impl.prepend(t: e); }
446 WindowSystemEvent *takeFirstOrReturnNull()
447 { const QMutexLocker locker(&mutex); return impl.empty() ? nullptr : impl.takeFirst(); }
448 WindowSystemEvent *takeFirstNonUserInputOrReturnNull()
449 {
450 const QMutexLocker locker(&mutex);
451 for (int i = 0; i < impl.size(); ++i)
452 if (!(impl.at(i)->type & QWindowSystemInterfacePrivate::UserInputEvent))
453 return impl.takeAt(i);
454 return nullptr;
455 }
456 bool nonUserInputEventsQueued()
457 {
458 const QMutexLocker locker(&mutex);
459 for (int i = 0; i < impl.size(); ++i)
460 if (!(impl.at(i)->type & QWindowSystemInterfacePrivate::UserInputEvent))
461 return true;
462 return false;
463 }
464 void append(WindowSystemEvent *e)
465 { const QMutexLocker locker(&mutex); impl.append(t: e); }
466 int count() const
467 { const QMutexLocker locker(&mutex); return impl.size(); }
468 WindowSystemEvent *peekAtFirstOfType(EventType t) const
469 {
470 const QMutexLocker locker(&mutex);
471 for (int i = 0; i < impl.size(); ++i) {
472 if (impl.at(i)->type == t)
473 return impl.at(i);
474 }
475 return nullptr;
476 }
477 void remove(const WindowSystemEvent *e)
478 {
479 const QMutexLocker locker(&mutex);
480 for (int i = 0; i < impl.size(); ++i) {
481 if (impl.at(i) == e) {
482 delete impl.takeAt(i);
483 break;
484 }
485 }
486 }
487 private:
488 Q_DISABLE_COPY_MOVE(WindowSystemEventList)
489 };
