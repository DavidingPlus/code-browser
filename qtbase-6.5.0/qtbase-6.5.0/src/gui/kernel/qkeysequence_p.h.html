27struct QKeyBinding
28{
29 QKeySequence::StandardKey standardKey;
30 uchar priority;
31 QKeyCombination shortcut;
32 uint platform;
33};

35class QKeySequencePrivate
36{
37public:
38 enum { MaxKeyCount = 4 }; // also used in QKeySequenceEdit
39 constexpr QKeySequencePrivate() : ref(1), key{} {}
40 inline QKeySequencePrivate(const QKeySequencePrivate &copy) : ref(1)
41 {
42 std::copy(copy.key, copy.key + MaxKeyCount,
43 QT_MAKE_CHECKED_ARRAY_ITERATOR(key, MaxKeyCount));
44 }
45 QAtomicInt ref;
46 int key[MaxKeyCount];
47 static QString encodeString(int key, QKeySequence::SequenceFormat format);
48 // used in dbusmenu
49 Q_GUI_EXPORT static QString keyName(int key, QKeySequence::SequenceFormat format);
50 static int decodeString(QString accel, QKeySequence::SequenceFormat format);
51};
