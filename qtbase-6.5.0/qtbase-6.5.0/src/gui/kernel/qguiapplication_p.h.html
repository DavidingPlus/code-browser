199 static struct QLastCursorPosition {
200 constexpr inline QLastCursorPosition() noexcept : thePoint(qt_inf(), qt_inf()) {}
201 constexpr inline Q_IMPLICIT QLastCursorPosition(QPointF p) noexcept : thePoint(p) {}
202 constexpr inline Q_IMPLICIT operator QPointF() const noexcept { return thePoint; }
203 constexpr inline qreal x() const noexcept{ return thePoint.x(); }
204 constexpr inline qreal y() const noexcept{ return thePoint.y(); }
205 Q_GUI_EXPORT QPoint toPoint() const noexcept;
206
207 constexpr void reset() noexcept { *this = QLastCursorPosition{}; }
208
209 // QGuiApplicationPrivate::lastCursorPosition is used for mouse-move detection
210 // but even QPointF's qFuzzCompare on doubles is too precise, and causes move-noise
211 // e.g. on macOS (see QTBUG-111170). So we specialize the equality operators here
212 // to use single-point precision.
213 friend constexpr bool operator==(const QLastCursorPosition &p1, const QPointF &p2) noexcept
214 {
215 return qFuzzyCompare(p1: float(p1.x()), p2: float(p2.x()))
216 && qFuzzyCompare(p1: float(p1.y()), p2: float(p2.y()));
217 }
218 friend constexpr bool operator!=(const QLastCursorPosition &p1, const QPointF &p2) noexcept
219 {
220 return !(p1 == p2);
221 }
222 friend constexpr bool operator==(const QPointF &p1, const QLastCursorPosition &p2) noexcept
223 {
224 return p2 == p1;
225 }
226 friend constexpr bool operator!=(const QPointF &p1, const QLastCursorPosition &p2) noexcept
227 {
228 return !(p2 == p1);
229 }
230
231 private:
232 QPointF thePoint;
233 } lastCursorPosition;
