35static QWindow* topLevelAt(const QPoint &pos)
36{
37 QWindowList list = QGuiApplication::topLevelWindows();
38 for (int i = list.size()-1; i >= 0; --i) {
39 QWindow *w = list.at(i);
40 if (w->isVisible() && w->handle() && w->geometry().contains(p: pos) && !qobject_cast<QShapedPixmapWindow*>(object: w))
41 return w;
42 }
43 return nullptr;
44}

48 \brief QBasicDrag is a base class for implementing platform drag and drop.
49 \since 5.0
50 \internal
51 \ingroup qpa
52
53 QBasicDrag implements QPlatformDrag::drag() by running a local event loop in which

84bool QBasicDrag::eventFilter(QObject *o, QEvent *e)
85{
86 Q_UNUSED(o);
87
88 if (!m_drag) {
89 if (e->type() == QEvent::KeyRelease && static_cast<QKeyEvent*>(e)->key() == Qt::Key_Escape) {
90 disableEventFilter();
91 exitDndEventLoop();
92 return true; // block the key release
93 }
94 return false;
95 }
96
97 switch (e->type()) {
98 case QEvent::ShortcutOverride:
99 // prevent accelerators from firing while dragging
100 e->accept();
101 return true;
102
103 case QEvent::KeyPress:
104 case QEvent::KeyRelease:
105 {
106 QKeyEvent *ke = static_cast<QKeyEvent *>(e);
107 if (ke->key() == Qt::Key_Escape && e->type() == QEvent::KeyPress) {
108 cancel();
109 disableEventFilter();
110 exitDndEventLoop();
111
112 } else if (ke->modifiers() != QGuiApplication::keyboardModifiers()) {
113 move(globalPos: m_lastPos, b: QGuiApplication::mouseButtons(), mods: ke->modifiers());
114 }
115 return true; // Eat all key events
116 }
117
118 case QEvent::MouseMove:
119 {
120 m_lastPos = getNativeMousePos(e, window: m_drag_icon_window);
121 auto mouseMove = static_cast<QMouseEvent *>(e);
122 move(globalPos: m_lastPos, b: mouseMove->buttons(), mods: mouseMove->modifiers());
123 return true; // Eat all mouse move events
124 }
125 case QEvent::MouseButtonRelease:
126 {
127 QPointer<QObject> objGuard(o);
128 disableEventFilter();
129 if (canDrop()) {
130 QPoint nativePosition = getNativeMousePos(e, window: m_drag_icon_window);
131 auto mouseRelease = static_cast<QMouseEvent *>(e);
132 drop(globalPos: nativePosition, b: mouseRelease->buttons(), mods: mouseRelease->modifiers());
133 } else {
134 cancel();
135 }
136 exitDndEventLoop();
137 if (!objGuard)
138 return true;
139
140 // If a QShapedPixmapWindow (drag feedback) is being dragged along, the
141 // mouse event's localPos() will be relative to that, which is useless.
142 // We want a position relative to the window where the drag ends, if possible (?).
143 // If there is no such window (belonging to this Qt application),
144 // make the event relative to the window where the drag started. (QTBUG-66103)
145 const QMouseEvent *release = static_cast<QMouseEvent *>(e);
146 const QWindow *releaseWindow = topLevelAt(pos: release->globalPosition().toPoint());
147 qCDebug(lcDnd) << "mouse released over" << releaseWindow << "after drag from" << m_sourceWindow << "globalPos" << release->globalPosition().toPoint();
148 if (!releaseWindow)
149 releaseWindow = m_sourceWindow;
150 QPointF releaseWindowPos = (releaseWindow ? releaseWindow->mapFromGlobal(pos: release->globalPosition()) : release->globalPosition());
151 QMouseEvent *newRelease = new QMouseEvent(release->type(),
152 releaseWindowPos, releaseWindowPos, release->globalPosition(),
153 release->button(), release->buttons(),
154 release->modifiers(), release->source(), release->pointingDevice());
155 QCoreApplication::postEvent(receiver: o, event: newRelease);
156 return true; // defer mouse release events until drag event loop has returned
157 }
158 case QEvent::MouseButtonDblClick:
159 case QEvent::Wheel:
160 return true;
161 default:
162 break;
163 }
164 return false;
165}

167Qt::DropAction QBasicDrag::drag(QDrag *o)
168{
169 m_drag = o;
170 m_executed_drop_action = Qt::IgnoreAction;
171 m_can_drop = false;
172
173 startDrag();
174 m_eventLoop = new QEventLoop;
175 m_eventLoop->exec();
176 delete m_eventLoop;
177 m_eventLoop = nullptr;
178 m_drag = nullptr;
179 endDrag();
180
181 return m_executed_drop_action;
182}

192void QBasicDrag::startDrag()
193{
194 QPoint pos;
195#ifndef QT_NO_CURSOR
196 pos = QCursor::pos();
197 if (pos.x() == int(qInf())) {
198 // ### fixme: no mouse pos registered. Get pos from touch...
199 pos = QPoint();
200 }
201#endif
202 m_lastPos = pos;
203 recreateShapedPixmapWindow(screen: m_screen, pos);
204 enableEventFilter();
205}

211void QBasicDrag::recreateShapedPixmapWindow(QScreen *screen, const QPoint &pos)
212{
213 delete m_drag_icon_window;
214 // ### TODO Check if its really necessary to have m_drag_icon_window
215 // when QDrag is used without a pixmap - QDrag::setPixmap()
216 m_drag_icon_window = new QShapedPixmapWindow(screen);
217
218 m_drag_icon_window->setUseCompositing(m_useCompositing);
219 m_drag_icon_window->setPixmap(m_drag->pixmap());
220 m_drag_icon_window->setHotspot(m_drag->hotSpot());
221 m_drag_icon_window->updateGeometry(pos);
222 m_drag_icon_window->setVisible(true);
223}

229 m_drag_icon_window->setVisible(false);

256void QBasicDrag::updateCursor(Qt::DropAction action)
257{
258#ifndef QT_NO_CURSOR
259 Qt::CursorShape cursorShape = Qt::ForbiddenCursor;
260 if (canDrop()) {
261 switch (action) {
262 case Qt::CopyAction:
263 cursorShape = Qt::DragCopyCursor;
264 break;
265 case Qt::LinkAction:
266 cursorShape = Qt::DragLinkCursor;
267 break;
268 default:
269 cursorShape = Qt::DragMoveCursor;
270 break;
271 }
272 }
273
274 QPixmap pixmap = m_drag->dragCursor(action);
275
276 if (!m_dndHasSetOverrideCursor) {
277 QCursor newCursor = !pixmap.isNull() ? QCursor(pixmap) : QCursor(cursorShape);
278 QGuiApplication::setOverrideCursor(newCursor);
279 m_dndHasSetOverrideCursor = true;
280 } else {
281 QCursor *cursor = QGuiApplication::overrideCursor();
282 if (!cursor) {
283 QGuiApplication::changeOverrideCursor(pixmap.isNull() ? QCursor(cursorShape) : QCursor(pixmap));
284 } else {
285 if (!pixmap.isNull()) {
286 if (cursor->pixmap().cacheKey() != pixmap.cacheKey())
287 QGuiApplication::changeOverrideCursor(QCursor(pixmap));
288 } else if (cursorShape != cursor->shape()) {
289 QGuiApplication::changeOverrideCursor(QCursor(cursorShape));
290 }
291 }
292 }
293#endif
294 updateAction(action);
295}

297void QBasicDrag::restoreCursor()
298{
299#ifndef QT_NO_CURSOR
300 if (m_dndHasSetOverrideCursor) {
301 QGuiApplication::restoreOverrideCursor();
302 m_dndHasSetOverrideCursor = false;
303 }
304#endif
305}

307static inline QPoint fromNativeGlobalPixels(const QPoint &point)
308{
309#ifndef QT_NO_HIGHDPISCALING
310 QPoint res = point;
311 if (QHighDpiScaling::isActive()) {
312 for (const QScreen *s : std::as_const(t&: QGuiApplicationPrivate::screen_list)) {
313 if (s->handle()->geometry().contains(p: point)) {
314 res = QHighDpi::fromNativePixels(value: point, context: s);
315 break;
316 }
317 }
318 }
319 return res;
320#else
321 return point;
322#endif
323}

327 \brief QSimpleDrag implements QBasicDrag for Drag and Drop operations within the Qt Application itself.
328 \since 5.0
329 \internal
330 \ingroup qpa
331
332 The class checks whether the receiving window is a window of the Qt application
333 and sets the state accordingly. It does not take windows of other applications
334 into account.
335*/

337QSimpleDrag::QSimpleDrag()
338{
339}
340
341void QSimpleDrag::startDrag()
342{
343 setExecutedDropAction(Qt::IgnoreAction);
344
345 QBasicDrag::startDrag();
346 // Here we can be fairly sure that QGuiApplication::mouseButtons/keyboardModifiers() will
347 // contain sensible values as startDrag() normally is called from mouse event handlers
348 // by QDrag::exec(). A better API would be if we could pass something like "input device
349 // pointer" to QDrag::exec(). My guess is that something like that might be required for
350 // QTBUG-52430.
351 m_sourceWindow = topLevelAt(pos: QCursor::pos());
352 m_windowUnderCursor = m_sourceWindow;
353 if (m_sourceWindow) {
354 auto nativePixelPos = QHighDpi::toNativePixels(value: QCursor::pos(), context: m_sourceWindow);
355 move(globalPos: nativePixelPos, b: QGuiApplication::mouseButtons(), mods: QGuiApplication::keyboardModifiers());
356 } else {
357 setCanDrop(false);
358 updateCursor(action: Qt::IgnoreAction);
359 }
360
361 qCDebug(lcDnd) << "drag began from" << m_sourceWindow << "cursor pos" << QCursor::pos() << "can drop?" << canDrop();
362}

366 QWindowSystemInterface::handleDrag(window, dropData: nullptr, p: QPoint(), supportedActions: Qt::IgnoreAction, buttons: { }, modifiers: { });

369void QSimpleDrag::cancel()
370{
371 QBasicDrag::cancel();
372 if (drag() && m_sourceWindow) {
373 sendDragLeave(window: m_sourceWindow);
374 m_sourceWindow = nullptr;
375 }

378void QSimpleDrag::move(const QPoint &nativeGlobalPos, Qt::MouseButtons buttons,
379 Qt::KeyboardModifiers modifiers)
380{
381 QPoint globalPos = fromNativeGlobalPixels(point: nativeGlobalPos);
382 moveShapedPixmapWindow(globalPos);
383 QWindow *window = topLevelAt(pos: globalPos);
384
385 if (!window || window != m_windowUnderCursor) {
386 if (m_windowUnderCursor)
387 sendDragLeave(window: m_windowUnderCursor);
388 m_windowUnderCursor = window;
389 if (!window) {
390 // QSimpleDrag supports only in-process dnd, we can't drop anywhere else.
391 setCanDrop(false);
392 updateCursor(action: Qt::IgnoreAction);
393 return;
394 }
395 }
396
397 const QPoint pos = nativeGlobalPos - window->handle()->geometry().topLeft();
398 const QPlatformDragQtResponse qt_response = QWindowSystemInterface::handleDrag(
399 window, dropData: drag()->mimeData(), p: pos, supportedActions: drag()->supportedActions(),
400 buttons, modifiers);
401
402 setCanDrop(qt_response.isAccepted());
403 updateCursor(action: qt_response.acceptedAction());
404}
405
406void QSimpleDrag::drop(const QPoint &nativeGlobalPos, Qt::MouseButtons buttons,
407 Qt::KeyboardModifiers modifiers)
408{
409 QPoint globalPos = fromNativeGlobalPixels(point: nativeGlobalPos);
410
411 QBasicDrag::drop(nativeGlobalPos, buttons, modifiers);
412 QWindow *window = topLevelAt(pos: globalPos);
413 if (!window)
414 return;
415
416 const QPoint pos = nativeGlobalPos - window->handle()->geometry().topLeft();
417 const QPlatformDropQtResponse response = QWindowSystemInterface::handleDrop(
418 window, dropData: drag()->mimeData(), p: pos, supportedActions: drag()->supportedActions(),
419 buttons, modifiers);
420 if (response.isAccepted()) {
421 setExecutedDropAction(response.acceptedAction());
422 } else {
423 setExecutedDropAction(Qt::IgnoreAction);
424 }
425}
