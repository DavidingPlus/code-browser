13/*!
14 \class QSurface
15 \inmodule QtGui
16 \since 5.0
17 \brief The QSurface class is an abstraction of renderable surfaces in Qt.
18
19 The size of the surface is accessible with the size() function. The rendering
20 specific attributes of the surface are accessible through the format() function.
21 */
22
23
24/*!
25 \enum QSurface::SurfaceClass
26
27 The SurfaceClass enum describes the actual subclass of the surface.
28
29 \value Window The surface is an instance of QWindow.
30 \value Offscreen The surface is an instance of QOffscreenSurface.
31 */
32
33
34/*!
35 \enum QSurface::SurfaceType
36
37 The SurfaceType enum describes what type of surface this is.
38
39 \value RasterSurface The surface is is composed of pixels and can be rendered to using
40 a software rasterizer like Qt's raster paint engine.
41 \value OpenGLSurface The surface is an OpenGL compatible surface and can be used
42 in conjunction with QOpenGLContext.
43 \value RasterGLSurface The surface can be rendered to using a software rasterizer,
44 and also supports OpenGL. This surface type is intended for internal Qt use, and
45 requires the use of private API.
46 \value OpenVGSurface The surface is an OpenVG compatible surface and can be used
47 in conjunction with OpenVG contexts.
48 \value VulkanSurface The surface is a Vulkan compatible surface and can be used
49 in conjunction with the Vulkan graphics API.
50 \value MetalSurface The surface is a Metal compatible surface and can be used
51 in conjunction with Apple's Metal graphics API. This surface type is only supported
52 on \macos and iOS.
53 \value Direct3DSurface The surface is a Direct 3D 11 and 12 compatible
54 surface and can be used in conjunction with the DXGI and Direct3D APIs. This
55 surface type is only supported on Windows.
56 */
57
58

65/*!
66 Returns true if the surface is OpenGL compatible and can be used in
67 conjunction with QOpenGLContext; otherwise returns false.
68
69 \since 5.3
70*/
71
72bool QSurface::supportsOpenGL() const
73{
74 SurfaceType type = surfaceType();
75 if (type == RasterSurface) {
76 QPlatformIntegration *integ = QGuiApplicationPrivate::instance()->platformIntegration();
77 return integ->hasCapability(cap: QPlatformIntegration::OpenGLOnRasterSurface);
78 }
79 return type == OpenGLSurface || type == RasterGLSurface;
80}

108/*!
109 Destroys the surface.
110*/
111QSurface::~QSurface()
112{
113#ifndef QT_NO_OPENGL
114 QOpenGLContext *context = QOpenGLContext::currentContext();
115 if (context && context->surface() == this)
116 context->doneCurrent();
117#endif
118}

120/*!
121 Returns the surface class of this surface.
122 */
123QSurface::SurfaceClass QSurface::surfaceClass() const
124{
125 return m_type;
126}
