135 // A very-temporary QPA touchpoint which gets converted to a QEventPoint as early as possible
136 // in QWindowSystemInterfacePrivate::fromNativeTouchPoints()
137 struct TouchPoint {
138 TouchPoint() : id(0), uniqueId(-1), pressure(0), rotation(0), state(QEventPoint::State::Stationary) { }
139 int id; // for application use
140 qint64 uniqueId; // for TUIO: object/token ID; otherwise empty
141 // TODO for TUIO 2.0: add registerPointerUniqueID(QPointingDeviceUniqueId)
142 QPointF normalPosition; // touch device coordinates, (0 to 1, 0 to 1)
143 QRectF area; // dimensions of the elliptical contact patch, unrotated, and centered at position in screen coordinates
144 // width is the horizontal diameter, height is the vertical diameter
145 qreal pressure; // 0 to 1
146 qreal rotation; // rotation applied to the elliptical contact patch
147 // 0 means pointing straight up; 0 if unknown (like QTabletEvent::rotation)
148 QEventPoint::State state; // Pressed|Updated|Stationary|Released
149 QVector2D velocity; // in screen coordinate system, pixels / seconds
150 QList<QPointF> rawPositions; // in screen coordinates
151 };
