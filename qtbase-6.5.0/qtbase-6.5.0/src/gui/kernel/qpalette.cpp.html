17static constexpr QPalette::ResolveMask colorRoleOffset(QPalette::ColorGroup colorGroup)
18{
19 return qToUnderlying(e: QPalette::NColorRoles) * qToUnderlying(e: colorGroup);
20}

22static constexpr QPalette::ResolveMask bitPosition(QPalette::ColorGroup colorGroup,
23 QPalette::ColorRole colorRole)
24{
25 return colorRole + colorRoleOffset(colorGroup);
26}
27
28static_assert(bitPosition(colorGroup: QPalette::ColorGroup(QPalette::NColorGroups - 1),
29 colorRole: QPalette::ColorRole(QPalette::NColorRoles - 1))
30 < sizeof(QPalette::ResolveMask) * CHAR_BIT,
31 "The resolve mask type is not wide enough to fit the entire bit mask.");

33class QPalettePrivate
34{
35public:
36 class Data : public QSharedData {
37 public:
38 // Every instance of Data has to have a unique serial number, even
39 // if it gets created by copying another - we wouldn't create a copy
40 // in the first place if the serial number should be the same!
41 Data(const Data &other)
42 : QSharedData(other)
43 {
44 for (int grp = 0; grp < int(QPalette::NColorGroups); grp++) {
45 for (int role = 0; role < int(QPalette::NColorRoles); role++)
46 br[grp][role] = other.br[grp][role];
47 }
48 }
49 Data() = default;
50
51 QBrush br[QPalette::NColorGroups][QPalette::NColorRoles];
52 const int ser_no = qt_palette_count++;
53 };
54
55 QPalettePrivate(const QExplicitlySharedDataPointer<Data> &data)
56 : ref(1), data(data)
57 { }
58 QPalettePrivate()
59 : QPalettePrivate(QExplicitlySharedDataPointer<Data>(new Data))
60 { }

62 QAtomicInt ref;
63 QPalette::ResolveMask resolveMask = {0};
64 static inline int qt_palette_private_count = 0;
65 int detach_no = ++qt_palette_private_count;
66 QExplicitlySharedDataPointer<Data> data;

69static QColor qt_mix_colors(QColor a, QColor b)
70{
71 return QColor((a.red() + b.red()) / 2, (a.green() + b.green()) / 2,
72 (a.blue() + b.blue()) / 2, (a.alpha() + b.alpha()) / 2);
73}

75/*!
76 \internal
77
78 Derive undefined \l PlaceholderText colors from \l Text colors.
79 Unless already set, PlaceholderText colors will be derived from their Text pendents.
80 Colors of existing PlaceHolderText brushes will not be replaced.
81
82 \a alpha represents the dim factor as a percentage. By default, a PlaceHolderText color
83 becomes a 50% more transparent version of the corresponding Text color.
84*/
85static void qt_placeholder_from_text(QPalette &pal, int alpha = 50)
86{
87 if (alpha < 0 or alpha > 100)
88 return;
89
90 for (int cg = 0; cg < int(QPalette::NColorGroups); ++cg) {
91 const QPalette::ColorGroup group = QPalette::ColorGroup(cg);
92
93 // skip if the brush has been set already
94 if (!pal.isBrushSet(cg: group, cr: QPalette::PlaceholderText)) {
95 QColor c = pal.color(cg: group, cr: QPalette::Text);
96 const int a = (c.alpha() * alpha) / 100;
97 c.setAlpha(a);
98 pal.setColor(acg: group, acr: QPalette::PlaceholderText, acolor: c);
99 }
100 }
101}

103static void qt_palette_from_color(QPalette &pal, const QColor &button)
104{
105 int h, s, v;
106 button.getHsv(h: &h, s: &s, v: &v);
107 // inactive and active are the same..
108 const QBrush whiteBrush = QBrush(Qt::white);
109 const QBrush blackBrush = QBrush(Qt::black);
110 const QBrush baseBrush = v > 128 ? whiteBrush : blackBrush;
111 const QBrush foregroundBrush = v > 128 ? blackBrush : whiteBrush;
112 const QBrush buttonBrush = QBrush(button);
113 const QBrush buttonBrushDark = QBrush(button.darker());
114 const QBrush buttonBrushDark150 = QBrush(button.darker(f: 150));
115 const QBrush buttonBrushLight150 = QBrush(button.lighter(f: 150));
116 pal.setColorGroup(cr: QPalette::Active, windowText: foregroundBrush, button: buttonBrush, light: buttonBrushLight150,
117 dark: buttonBrushDark, mid: buttonBrushDark150, text: foregroundBrush, bright_text: whiteBrush,
118 base: baseBrush, window: buttonBrush);
119 pal.setColorGroup(cr: QPalette::Inactive, windowText: foregroundBrush, button: buttonBrush, light: buttonBrushLight150,
120 dark: buttonBrushDark, mid: buttonBrushDark150, text: foregroundBrush, bright_text: whiteBrush,
121 base: baseBrush, window: buttonBrush);
122 pal.setColorGroup(cr: QPalette::Disabled, windowText: buttonBrushDark, button: buttonBrush, light: buttonBrushLight150,
123 dark: buttonBrushDark, mid: buttonBrushDark150, text: buttonBrushDark,
124 bright_text: whiteBrush, base: buttonBrush, window: buttonBrush);
125
126 qt_placeholder_from_text(pal);
127}

367/*!
368 \class QPalette
369
370 \brief The QPalette class contains color groups for each widget state.
371
372 \inmodule QtGui
373 \ingroup appearance
374 \ingroup shared
375
376 A palette consists of three color groups: \e Active, \e Disabled,
377 and \e Inactive. All widgets in Qt contain a palette and
378 use their palette to draw themselves. This makes the user
379 interface easily configurable and easier to keep consistent.
380
381
382 If you create a new widget we strongly recommend that you use the
383 colors in the palette rather than hard-coding specific colors.
384
385 The color groups:
386 \list
387 \li The Active group is used for the window that has keyboard focus.
388 \li The Inactive group is used for other windows.
389 \li The Disabled group is used for widgets (not windows) that are
390 disabled for some reason.
391 \endlist
392
393 Both active and inactive windows can contain disabled widgets.
394 (Disabled widgets are often called \e inaccessible or \e{grayed
395 out}.)
396
397 In most styles, Active and Inactive look the same.
398
399 Colors and brushes can be set for particular roles in any of a palette's
400 color groups with setColor() and setBrush(). A color group contains a
401 group of colors used by widgets for drawing themselves. We recommend that
402 widgets use color group roles from the palette such as "foreground" and
403 "base" rather than literal colors like "red" or "turquoise". The color
404 roles are enumerated and defined in the \l ColorRole documentation.
405
406 We strongly recommend that you use the default palette of the
407 current style (returned by QGuiApplication::palette()) and
408 modify that as necessary. This is done by Qt's widgets when they
409 are drawn.
410
411 To modify a color group you call the functions
412 setColor() and setBrush(), depending on whether you want a pure
413 color or a pixmap pattern.
414
415 There are also corresponding color() and brush() getters, and a
416 commonly used convenience function to get the ColorRole for the current ColorGroup:
417 window(), windowText(), base(), etc.
418
419
420 You can copy a palette using the copy constructor and test to see
421 if two palettes are \e identical using isCopyOf().
422
423 QPalette is optimized by the use of \l{implicit sharing},
424 so it is very efficient to pass QPalette objects as arguments.
425
426 \warning Some styles do not use the palette for all drawing, for
427 instance, if they make use of native theme engines. This is the
428 case for both the Windows Vista and the \macos
429 styles.
430
431 \sa QApplication::setPalette(), QWidget::setPalette(), QColor
432*/
433
434/*!
435 \enum QPalette::ColorGroup
436
437 \value Disabled
438 \value Active
439 \value Inactive
440 \value Normal synonym for Active
441
442 \omitvalue All
443 \omitvalue NColorGroups
444 \omitvalue Current
445*/

447/*!
448 \enum QPalette::ColorRole
449
450 \image palette.png Color Roles
451
452 The ColorRole enum defines the different symbolic color roles used
453 in current GUIs.
454
455 The central roles are:
456
457 \value Window A general background color.
458
459 \value WindowText A general foreground color.
460
461 \value Base Used mostly as the background color for text entry widgets,
462 but can also be used for other painting - such as the
463 background of combobox drop down lists and toolbar handles.
464 It is usually white or another light color.
465
466 \value AlternateBase Used as the alternate background color in views with
467 alternating row colors (see
468 QAbstractItemView::setAlternatingRowColors()).
469
470 \value ToolTipBase Used as the background color for QToolTip and
471 QWhatsThis. Tool tips use the Inactive color group
472 of QPalette, because tool tips are not active
473 windows.
474
475 \value ToolTipText Used as the foreground color for QToolTip and
476 QWhatsThis. Tool tips use the Inactive color group
477 of QPalette, because tool tips are not active
478 windows.
479
480 \value PlaceholderText Used as the placeholder color for various text input widgets.
481 This enum value has been introduced in Qt 5.12
482
483 \value Text The foreground color used with \c Base. This is usually
484 the same as the \c WindowText, in which case it must provide
485 good contrast with \c Window and \c Base.
486
487 \value Button The general button background color. This background can be different from
488 \c Window as some styles require a different background color for buttons.
489
490 \value ButtonText A foreground color used with the \c Button color.
491
492 \value BrightText A text color that is very different from
493 \c WindowText, and contrasts well with e.g. \c
494 Dark. Typically used for text that needs to be
495 drawn where \c Text or \c WindowText would give
496 poor contrast, such as on pressed push buttons.
497 Note that text colors can be used for things
498 other than just words; text colors are \e
499 usually used for text, but it's quite common to
500 use the text color roles for lines, icons, etc.
501
502
503 There are some color roles used mostly for 3D bevel and shadow effects.
504 All of these are normally derived from \c Window, and used in ways that
505 depend on that relationship. For example, buttons depend on it to make the
506 bevels look attractive, and Motif scroll bars depend on \c Mid to be
507 slightly different from \c Window.
508
509 \value Light Lighter than \c Button color.
510
511 \value Midlight Between \c Button and \c Light.
512
513 \value Dark Darker than \c Button.
514
515 \value Mid Between \c Button and \c Dark.
516
517 \value Shadow A very dark color. By default, the shadow color is
518 Qt::black.
519
520
521 Selected (marked) items have two roles:
522
523 \value Highlight A color to indicate a selected item or the current
524 item. By default, the highlight color is
525 Qt::darkBlue.
526
527 \value HighlightedText A text color that contrasts with \c Highlight.
528 By default, the highlighted text color is Qt::white.
529
530 There are two color roles related to hyperlinks:
531
532 \value Link A text color used for unvisited hyperlinks.
533 By default, the link color is Qt::blue.
534
535 \value LinkVisited A text color used for already visited hyperlinks.
536 By default, the linkvisited color is Qt::magenta.
537
538 Note that we do not use the \c Link and \c LinkVisited roles when
539 rendering rich text in Qt, and that we recommend that you use CSS
540 and the QTextDocument::setDefaultStyleSheet() function to alter
541 the appearance of links. For example:
542
543 \snippet textdocument-css/main.cpp 0
544
545 \value NoRole No role; this special role is often used to indicate that a
546 role has not been assigned.
547
548 \omitvalue NColorRoles
549*/
550
551/*!
552 Constructs an empty palette object with no color roles set.
553
554 When used as the palette of a QWidget the colors are resolved
555 as described by QWidget::setPalette().
556
557 \sa QApplication::setPalette(), QApplication::palette()
558*/
559QPalette::QPalette()
560 : d(nullptr)
561{
562 // Initialize to application palette if present, else default to black.
563 // This makes it possible to instantiate QPalette outside QGuiApplication,
564 // for example in the platform plugins.
565 if (QGuiApplicationPrivate::app_pal) {
566 d = QGuiApplicationPrivate::app_pal->d;
567 d->ref.ref();
568 setResolveMask(0);
569 } else {
570 init();
571 qt_palette_from_color(pal&: *this, button: Qt::black);
572 d->resolveMask = 0;
573 }
574}

761const QBrush &QPalette::brush(ColorGroup gr, ColorRole cr) const

763 Q_ASSERT(cr < NColorRoles);
764 if (gr >= (int)NColorGroups) {
765 if (gr == Current) {
766 gr = currentGroup;
767 } else {
768 qWarning(msg: "QPalette::brush: Unknown ColorGroup: %d", (int)gr);

770 }
771 }

789 color \a role, to \a brush.
790
791 \sa brush(), setColor(), ColorRole
792*/
793void QPalette::setBrush(ColorGroup cg, ColorRole cr, const QBrush &b)
794{
795 Q_ASSERT(cr < NColorRoles);
796
797 if (cg == All) {
798 for (uint i = 0; i < NColorGroups; i++)
799 setBrush(cg: ColorGroup(i), cr, b);
800 return;
801 }
802
803 if (cg == Current) {
804 cg = currentGroup;
805 } else if (cg >= NColorGroups) {
806 qWarning(msg: "QPalette::setBrush: Unknown ColorGroup: %d", cg);
807 cg = Active;
808 }
809
810 const auto newResolveMask = d->resolveMask | ResolveMask(1) << bitPosition(colorGroup: cg, colorRole: cr);
811 const auto valueChanged = d->data->br[cg][cr] != b;
812
813 if (valueChanged) {
814 detach();
815 d->data.detach();
816 d->data->br[cg][cr] = b;
817 } else if (d->resolveMask != newResolveMask) {
818 detach();
819 }
820
821 d->resolveMask = newResolveMask;
822}

828 set previously on this palette; otherwise returns \c false.
829
830 The ColorGroup \a cg should be less than QPalette::NColorGroups,
831 but you can use QPalette::Current. In this case, the previously
832 set current color group will be used.
833
834 The ColorRole \a cr should be less than QPalette::NColorRoles.
835

838bool QPalette::isBrushSet(ColorGroup cg, ColorRole cr) const
839{
840 if (cg == Current)
841 cg = currentGroup;

845 return false;

851 }

962/*!
963 Returns a new QPalette that is a union of this instance and \a other.
964 Color roles set in this instance take precedence.
965*/
966QPalette QPalette::resolve(const QPalette &other) const
967{
968 if ((*this == other && d->resolveMask == other.d->resolveMask)
969 || d->resolveMask == 0) {
970 QPalette o = other;
971 o.setResolveMask(d->resolveMask);
972 return o;
973 }
974
975 if (d->resolveMask == allResolveMask())
976 return *this;
977
978 QPalette palette(*this);
979 palette.detach();
980
981 for (int role = 0; role < int(NColorRoles); ++role) {
982 for (int grp = 0; grp < int(NColorGroups); ++grp) {
983 if (!(d->resolveMask & (ResolveMask(1) << bitPosition(colorGroup: ColorGroup(grp), colorRole: ColorRole(role))))) {
984 palette.d->data.detach();
985 palette.d->data->br[grp][role] = other.d->data->br[grp][role];
986 }
987 }
988 }
989
990 palette.d->resolveMask |= other.d->resolveMask;
991
992 return palette;
993}
