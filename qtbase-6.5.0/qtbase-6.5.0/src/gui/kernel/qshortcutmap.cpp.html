22/* \internal
23 Entry data for QShortcutMap
24 Contains:
25 Keysequence for entry
26 Pointer to parent owning the sequence
27*/
28
29struct QShortcutEntry
30{
31 QShortcutEntry()
32 : keyseq(0), context(Qt::WindowShortcut), enabled(false), autorepeat(1), id(0), owner(nullptr), contextMatcher(nullptr)
33 {}
34
35 QShortcutEntry(const QKeySequence &k)
36 : keyseq(k), context(Qt::WindowShortcut), enabled(false), autorepeat(1), id(0), owner(nullptr), contextMatcher(nullptr)
37 {}
38
39 QShortcutEntry(QObject *o, const QKeySequence &k, Qt::ShortcutContext c, int i, bool a, QShortcutMap::ContextMatcher m)
40 : keyseq(k), context(c), enabled(true), autorepeat(a), id(i), owner(o), contextMatcher(m)
41 {}
42
43 bool correctContext() const { return contextMatcher(owner, context); }
44
45 bool operator<(const QShortcutEntry &f) const
46 { return keyseq < f.keyseq; }
47
48 QKeySequence keyseq;
49 Qt::ShortcutContext context;
50 bool enabled : 1;
51 bool autorepeat : 1;
52 signed int id;
53 QObject *owner;
54 QShortcutMap::ContextMatcher contextMatcher;
55};
56Q_DECLARE_TYPEINFO(QShortcutEntry, Q_RELOCATABLE_TYPE);

138/*! \internal
139 Removes a shortcut from the global map.
140 If \a owner is \nullptr, all entries in the map with the key sequence specified
141 is removed. If \a key is null, all sequences for \a owner is removed from
142 the map. If \a id is 0, any identical \a key sequences owned by \a owner
143 are removed.
144 Returns the number of sequences removed from the map.
145*/
146
147int QShortcutMap::removeShortcut(int id, QObject *owner, const QKeySequence &key)
148{
149 Q_D(QShortcutMap);
150 int itemsRemoved = 0;
151 bool allOwners = (owner == nullptr);
152 bool allKeys = key.isEmpty();
153 bool allIds = id == 0;
154
155 auto debug = qScopeGuard(f: [&](){
156 qCDebug(lcShortcutMap).nospace()
157 << "QShortcutMap::removeShortcut(" << id << ", " << owner << ", "
158 << key << ") removed " << itemsRemoved << " shortcuts(s)";
159 });
160
161 // Special case, remove everything
162 if (allOwners && allKeys && allIds) {
163 itemsRemoved = d->sequences.size();
164 d->sequences.clear();
165 return itemsRemoved;
166 }
167
168 int i = d->sequences.size()-1;
169 while (i>=0)
170 {
171 const QShortcutEntry &entry = d->sequences.at(i);
172 int entryId = entry.id;
173 if ((allOwners || entry.owner == owner)
174 && (allIds || entry.id == id)
175 && (allKeys || entry.keyseq == key)) {
176 d->sequences.removeAt(i);
177 ++itemsRemoved;
178 }
179 if (id == entryId)
180 return itemsRemoved;
181 --i;
182 }
183 return itemsRemoved;
184}

277/*! \internal
278 Uses nextState(QKeyEvent) to check for a grabbed shortcut.
279
280 If so, it is dispatched using dispatchEvent().
281
282 Returns true if a shortcut handled the event.
283
284 \sa nextState, dispatchEvent
285*/
286bool QShortcutMap::tryShortcut(QKeyEvent *e)
287{
288 Q_D(QShortcutMap);
289
290 if (e->key() == Qt::Key_unknown)
291 return false;
292
293 QKeySequence::SequenceMatch previousState = state();
294
295 switch (nextState(e)) {
296 case QKeySequence::NoMatch:
297 // In the case of going from a partial match to no match we handled the
298 // event, since we already stated that we did for the partial match. But
299 // in the normal case of directly going to no match we say we didn't.
300 return previousState == QKeySequence::PartialMatch;
301 case QKeySequence::PartialMatch:
302 // For a partial match we don't know yet if we will handle the shortcut
303 // but we need to say we did, so that we get the follow-up key-presses.
304 return true;
305 case QKeySequence::ExactMatch: {
306 // Save number of identical matches before dispatching
307 // to keep QShortcutMap and tryShortcut reentrant.
308 const int identicalMatches = d->identicals.size();
309 resetState();
310 dispatchEvent(e);
311 // If there are no identicals we've only found disabled shortcuts, and
312 // shouldn't say that we handled the event.
313 return identicalMatches > 0;
314 }
315 }
316 Q_UNREACHABLE_RETURN(false);
317}

319/*! \internal
320 Returns the next state of the statemachine
321 If return value is SequenceMatch::ExactMatch, then a call to matches()
322 will return a QObjects* list of all matching objects for the last matching
323 sequence.
324*/
325QKeySequence::SequenceMatch QShortcutMap::nextState(QKeyEvent *e)
326{
327 Q_D(QShortcutMap);
328 // Modifiers can NOT be shortcuts...
329 if (e->key() >= Qt::Key_Shift &&
330 e->key() <= Qt::Key_ScrollLock)
331 return d->currentState;
332
333 QKeySequence::SequenceMatch result = QKeySequence::NoMatch;
334
335 // We start fresh each time..
336 d->identicals.clear();
337
338 result = find(e);
339 if (result == QKeySequence::NoMatch && (e->modifiers() & Qt::KeypadModifier)) {
340 // Try to find a match without keypad modifier
341 result = find(e, ignoredModifiers: Qt::KeypadModifier);
342 }
343 if (result == QKeySequence::NoMatch && e->modifiers() & Qt::ShiftModifier) {
344 // If Shift + Key_Backtab, also try Shift + Qt::Key_Tab
345 if (e->key() == Qt::Key_Backtab) {
346 QKeyEvent pe = QKeyEvent(e->type(), Qt::Key_Tab, e->modifiers(), e->text());
347 result = find(e: &pe);
348 }
349 }
350
351 // Does the new state require us to clean up?
352 if (result == QKeySequence::NoMatch)
353 clearSequence(ksl&: d->currentSequences);
354 d->currentState = result;
355
356 qCDebug(lcShortcutMap).nospace() << "QShortcutMap::nextState(" << e << ") = " << result;
357 return result;
358}

361/*! \internal
362 Determines if an enabled shortcut has a matching key sequence.
363*/
364bool QShortcutMap::hasShortcutForKeySequence(const QKeySequence &seq) const
365{
366 Q_D(const QShortcutMap);
367 QShortcutEntry entry(seq); // needed for searching
368 const auto itEnd = d->sequences.cend();
369 auto it = std::lower_bound(d->sequences.cbegin(), itEnd, entry);
370
371 for (;it != itEnd; ++it) {
372 if (matches(seq1: entry.keyseq, seq2: (*it).keyseq) == QKeySequence::ExactMatch && (*it).correctContext() && (*it).enabled) {
373 return true;
374 }
375 }
376
377 //end of the loop: we didn't find anything
378 return false;
379}

381/*! \internal
382 Returns the next state of the statemachine, based
383 on the new key event \a e.
384 Matches are appended to the list of identicals,
385 which can be access through matches().
386 \sa matches
387*/
388QKeySequence::SequenceMatch QShortcutMap::find(QKeyEvent *e, int ignoredModifiers)
389{
390 Q_D(QShortcutMap);
391 if (!d->sequences.size())
392 return QKeySequence::NoMatch;
393
394 createNewSequences(e, ksl&: d->newEntries, ignoredModifiers);
395 qCDebug(lcShortcutMap) << "Possible shortcut key sequences:" << d->newEntries;
396
397 // Should never happen
398 if (d->newEntries == d->currentSequences) {
399 Q_ASSERT_X(e->key() != Qt::Key_unknown || e->text().size(),
400 "QShortcutMap::find", "New sequence to find identical to previous");
401 return QKeySequence::NoMatch;
402 }
403
404 // Looking for new identicals, scrap old
405 d->identicals.clear();
406
407 bool partialFound = false;
408 bool identicalDisabledFound = false;
409 QList<QKeySequence> okEntries;
410 int result = QKeySequence::NoMatch;
411 for (int i = d->newEntries.size()-1; i >= 0 ; --i) {
412 QShortcutEntry entry(d->newEntries.at(i)); // needed for searching
413 const auto itEnd = d->sequences.constEnd();
414 auto it = std::lower_bound(d->sequences.constBegin(), itEnd, entry);
415
416 int oneKSResult = QKeySequence::NoMatch;
417 int tempRes = QKeySequence::NoMatch;
418 do {
419 if (it == itEnd)
420 break;
421 tempRes = matches(seq1: entry.keyseq, seq2: (*it).keyseq);
422 oneKSResult = qMax(a: oneKSResult, b: tempRes);
423 if (tempRes != QKeySequence::NoMatch && (*it).correctContext()) {
424 if (tempRes == QKeySequence::ExactMatch) {
425 if ((*it).enabled)
426 d->identicals.append(t: &*it);
427 else
428 identicalDisabledFound = true;
429 } else if (tempRes == QKeySequence::PartialMatch) {
430 // We don't need partials, if we have identicals
431 if (d->identicals.size())
432 break;
433 // We only care about enabled partials, so we don't consume
434 // key events when all partials are disabled!
435 partialFound |= (*it).enabled;
436 }
437 }
438 ++it;
439 // If we got a valid match on this run, there might still be more keys to check against,
440 // so we'll loop once more. If we get NoMatch, there's guaranteed no more possible
441 // matches in the shortcutmap.
442 } while (tempRes != QKeySequence::NoMatch);
443
444 // If the type of match improves (ergo, NoMatch->Partial, or Partial->Exact), clear the
445 // previous list. If this match is equal or better than the last match, append to the list
446 if (oneKSResult > result) {
447 okEntries.clear();
448 qCDebug(lcShortcutMap) << "Found better match (" << d->newEntries << "), clearing key sequence list";
449 }
450 if (oneKSResult && oneKSResult >= result) {
451 okEntries << d->newEntries.at(i);
452 qCDebug(lcShortcutMap) << "Added ok key sequence" << d->newEntries;
453 }
454 }
455
456 if (d->identicals.size()) {
457 result = QKeySequence::ExactMatch;
458 } else if (partialFound) {
459 result = QKeySequence::PartialMatch;
460 } else if (identicalDisabledFound) {
461 result = QKeySequence::ExactMatch;
462 } else {
463 clearSequence(ksl&: d->currentSequences);
464 result = QKeySequence::NoMatch;
465 }
466 if (result != QKeySequence::NoMatch)
467 d->currentSequences = okEntries;
468 qCDebug(lcShortcutMap) << "Returning shortcut match == " << result;
469 return QKeySequence::SequenceMatch(result);
470}

483/*! \internal
484 Alters \a seq to the new sequence state, based on the
485 current sequence state, and the new key event \a e.
486*/
487void QShortcutMap::createNewSequences(QKeyEvent *e, QList<QKeySequence> &ksl, int ignoredModifiers)
488{
489 Q_D(QShortcutMap);
490 QList<int> possibleKeys = QKeyMapper::possibleKeys(e);
491 qCDebug(lcShortcutMap) << "Creating new sequences for" << e
492 << "with ignoredModifiers=" << Qt::KeyboardModifiers(ignoredModifiers);
493 int pkTotal = possibleKeys.size();
494 if (!pkTotal)
495 return;
496
497 int ssActual = d->currentSequences.size();
498 int ssTotal = qMax(a: 1, b: ssActual);
499 // Resize to possible permutations of the current sequence(s).
500 ksl.resize(size: pkTotal * ssTotal);
501
502 int index = ssActual ? d->currentSequences.at(i: 0).count() : 0;
503 for (int pkNum = 0; pkNum < pkTotal; ++pkNum) {
504 for (int ssNum = 0; ssNum < ssTotal; ++ssNum) {
505 int i = (pkNum * ssTotal) + ssNum;
506 QKeySequence &curKsl = ksl[i];
507 if (ssActual) {
508 const QKeySequence &curSeq = d->currentSequences.at(i: ssNum);
509 curKsl.setKey(key: curSeq[0], index: 0);
510 curKsl.setKey(key: curSeq[1], index: 1);
511 curKsl.setKey(key: curSeq[2], index: 2);
512 curKsl.setKey(key: curSeq[3], index: 3);
513 } else {
514 curKsl.setKey(key: QKeyCombination::fromCombined(combined: 0), index: 0);
515 curKsl.setKey(key: QKeyCombination::fromCombined(combined: 0), index: 1);
516 curKsl.setKey(key: QKeyCombination::fromCombined(combined: 0), index: 2);
517 curKsl.setKey(key: QKeyCombination::fromCombined(combined: 0), index: 3);
518 }
519 curKsl.setKey(key: QKeyCombination::fromCombined(combined: possibleKeys.at(i: pkNum) & ~ignoredModifiers), index);
520 }
521 }
522}

524/*! \internal
525 Basically the same function as QKeySequence::matches(const QKeySequence &seq) const
526 only that is specially handles Key_hyphen as Key_Minus, as people mix these up all the time and
527 they conceptually the same.
528*/
529QKeySequence::SequenceMatch QShortcutMap::matches(const QKeySequence &seq1,
530 const QKeySequence &seq2) const
531{
532 uint userN = seq1.count(),
533 seqN = seq2.count();
534
535 if (userN > seqN)
536 return QKeySequence::NoMatch;
537
538 // If equal in length, we have a potential ExactMatch sequence,
539 // else we already know it can only be partial.
540 QKeySequence::SequenceMatch match = (userN == seqN
541 ? QKeySequence::ExactMatch
542 : QKeySequence::PartialMatch);
543
544 for (uint i = 0; i < userN; ++i) {
545 int userKey = seq1[i].toCombined(),
546 sequenceKey = seq2[i].toCombined();
547 if ((userKey & Qt::Key_unknown) == Qt::Key_hyphen)
548 userKey = (userKey & Qt::KeyboardModifierMask) | Qt::Key_Minus;
549 if ((sequenceKey & Qt::Key_unknown) == Qt::Key_hyphen)
550 sequenceKey = (sequenceKey & Qt::KeyboardModifierMask) | Qt::Key_Minus;
551 if (userKey != sequenceKey)
552 return QKeySequence::NoMatch;
553 }
554 return match;
555}

584/*! \internal
585 Dispatches QShortcutEvents to widgets who grabbed the matched key sequence.
586*/
587void QShortcutMap::dispatchEvent(QKeyEvent *e)
588{
589 Q_D(QShortcutMap);
590 if (!d->identicals.size())
591 return;
592
593 const QKeySequence &curKey = d->identicals.at(i: 0)->keyseq;
594 if (d->prevSequence != curKey) {
595 d->ambigCount = 0;
596 d->prevSequence = curKey;
597 }
598 // Find next
599 const QShortcutEntry *current = nullptr, *next = nullptr;
600 int i = 0, enabledShortcuts = 0;
601 QList<const QShortcutEntry*> ambiguousShortcuts;
602 while(i < d->identicals.size()) {
603 current = d->identicals.at(i);
604 if (current->enabled || !next){
605 ++enabledShortcuts;
606 if (lcShortcutMap().isDebugEnabled())
607 ambiguousShortcuts.append(t: current);
608 if (enabledShortcuts > d->ambigCount + 1)
609 break;
610 next = current;
611 }
612 ++i;
613 }
614 d->ambigCount = (d->identicals.size() == i ? 0 : d->ambigCount + 1);
615 // Don't trigger shortcut if we're autorepeating and the shortcut is
616 // grabbed with not accepting autorepeats.
617 if (!next || (e->isAutoRepeat() && !next->autorepeat))
618 return;
619 // Dispatch next enabled
620 if (lcShortcutMap().isDebugEnabled()) {
621 if (ambiguousShortcuts.size() > 1) {
622 qCDebug(lcShortcutMap) << "The following shortcuts are about to be activated ambiguously:";
623 for (const QShortcutEntry *entry : std::as_const(t&: ambiguousShortcuts))
624 qCDebug(lcShortcutMap).nospace() << "- " << entry->keyseq << " (belonging to " << entry->owner << ")";
625 }
626
627 qCDebug(lcShortcutMap).nospace()
628 << "QShortcutMap::dispatchEvent(): Sending QShortcutEvent(\""
629 << next->keyseq.toString() << "\", " << next->id << ", "
630 << static_cast<bool>(enabledShortcuts>1) << ") to object(" << next->owner << ')';
631 }
632 QShortcutEvent se(next->keyseq, next->id, enabledShortcuts>1);
633 QCoreApplication::sendEvent(receiver: const_cast<QObject *>(next->owner), event: &se);
634}

636QList<QKeySequence> QShortcutMap::keySequences(bool getAll) const
637{
638 Q_D(const QShortcutMap);
639 QList<QKeySequence> keys;
640 for (auto sequence : d->sequences) {
641 bool addSequence = false;
642 if (sequence.enabled) {
643 if (getAll || sequence.context == Qt::ApplicationShortcut ||
644 sequence.owner == QGuiApplication::focusObject()) {
645 addSequence = true;
646 } else {
647 QObject *possibleWindow = sequence.owner;
648 while (possibleWindow) {
649 if (qobject_cast<QWindow *>(o: possibleWindow))
650 break;
651 possibleWindow = possibleWindow->parent();
652 }
653 if (possibleWindow == QGuiApplication::focusWindow()) {
654 if (sequence.context == Qt::WindowShortcut) {
655 addSequence = true;
656 } else if (sequence.context == Qt::WidgetWithChildrenShortcut) {
657 QObject *possibleWidget = QGuiApplication::focusObject();
658 while (possibleWidget->parent()) {
659 possibleWidget = possibleWidget->parent();
660 if (possibleWidget == sequence.owner) {
661 addSequence = true;
662 break;
663 }
664 }
665 }
666 }
667 }
668 if (addSequence)
669 keys << sequence.keyseq;
670 }
671 }
672 return keys;
673
674}

676/* \internal
677 QShortcutMap dump function, only available when DEBUG_QSHORTCUTMAP is
678 defined.
679*/
680#if defined(Dump_QShortcutMap)
681void QShortcutMap::dumpMap() const
682{
