17/*!
18 \class QScreen
19 \since 5.0
20 \brief The QScreen class is used to query screen properties.
21 \inmodule QtGui
22
23 A note on logical vs physical dots per inch: physical DPI is based on the
24 actual physical pixel sizes when available, and is useful for print preview
25 and other cases where it's desirable to know the exact physical dimensions
26 of screen displayed contents.
27
28 Logical dots per inch are used to convert font and user interface elements
29 from point sizes to pixel sizes, and might be different from the physical
30 dots per inch. The logical dots per inch are sometimes user-settable in the
31 desktop environment's settings panel, to let the user globally control UI
32 and font sizes in different applications.
33
34 \note Both physical and logical DPI are expressed in device-independent dots.
35 Multiply by QScreen::devicePixelRatio() to get device-dependent density.
36
37 \inmodule QtGui
38*/
39
40QScreen::QScreen(QPlatformScreen *screen)
41 : QObject(*new QScreenPrivate(), nullptr)
42{
43 Q_D(QScreen);
44 d->setPlatformScreen(screen);
45}
46
47void QScreenPrivate::setPlatformScreen(QPlatformScreen *screen)
48{
49 Q_Q(QScreen);
50 platformScreen = screen;
51 platformScreen->d_func()->screen = q;
52 orientation = platformScreen->orientation();
53
54 logicalDpi = QPlatformScreen::overrideDpi(in: platformScreen->logicalDpi());
55
56 refreshRate = platformScreen->refreshRate();
57 // safeguard ourselves against buggy platform behavior...
58 if (refreshRate < 1.0)
59 refreshRate = 60.0;
60
61 updateGeometry();
62 updatePrimaryOrientation(); // derived from the geometry
63}

65void QScreenPrivate::updateGeometry()
66{
67 qreal scaleFactor = QHighDpiScaling::factor(context: platformScreen);
68 QRect nativeGeometry = platformScreen->geometry();
69 geometry = QRect(nativeGeometry.topLeft(), QHighDpi::fromNative(value: nativeGeometry.size(), scaleFactor));
70 availableGeometry = QHighDpi::fromNative(value: platformScreen->availableGeometry(), scaleFactor, origin: geometry.topLeft());
71}

73/*!
74 Destroys the screen.
75 */
76QScreen::~QScreen()
77{
78 // Remove screen
79 const bool wasPrimary = QGuiApplication::primaryScreen() == this;
80 QGuiApplicationPrivate::screen_list.removeOne(t: this);
81 QGuiApplicationPrivate::resetCachedDevicePixelRatio();
82
83 if (!qGuiApp)
84 return;
85
86 QScreen *newPrimaryScreen = QGuiApplication::primaryScreen();
87 if (wasPrimary && newPrimaryScreen)
88 emit qGuiApp->primaryScreenChanged(screen: newPrimaryScreen);
89
90 // Allow clients to manage windows that are affected by the screen going
91 // away, before we fall back to moving them to the primary screen.
92 emit qApp->screenRemoved(screen: this);
93
94 if (QGuiApplication::closingDown())
95 return;
96
97 bool movingFromVirtualSibling = newPrimaryScreen
98 && newPrimaryScreen->handle()->virtualSiblings().contains(t: handle());
99
100 // Move any leftover windows to the primary screen
101 const auto allWindows = QGuiApplication::allWindows();
102 for (QWindow *window : allWindows) {
103 if (!window->isTopLevel() || window->screen() != this)
104 continue;
105
106 const bool wasVisible = window->isVisible();
107 window->setScreen(newPrimaryScreen);
108
109 // Re-show window if moved from a virtual sibling screen. Otherwise
110 // leave it up to the application developer to show the window.
111 if (movingFromVirtualSibling)
112 window->setVisible(wasVisible);
113 }
114}

200/*!
201 \property QScreen::physicalDotsPerInchX
202 \brief the number of physical dots or pixels per inch in the horizontal direction
203
204 This value represents the actual horizontal pixel density on the screen's display.
205 Depending on what information the underlying system provides the value might not be
206 entirely accurate.
207
208 \note Physical DPI is expressed in device-independent dots. Multiply by QScreen::devicePixelRatio()
209 to get device-dependent density.
210
211 \sa physicalDotsPerInchY()
212*/
213qreal QScreen::physicalDotsPerInchX() const
214{
215 return size().width() / physicalSize().width() * qreal(25.4);
216}

218/*!
219 \property QScreen::physicalDotsPerInchY
220 \brief the number of physical dots or pixels per inch in the vertical direction
221
222 This value represents the actual vertical pixel density on the screen's display.
223 Depending on what information the underlying system provides the value might not be
224 entirely accurate.
225
226 \note Physical DPI is expressed in device-independent dots. Multiply by QScreen::devicePixelRatio()
227 to get device-dependent density.
228
229 \sa physicalDotsPerInchX()
230*/
231qreal QScreen::physicalDotsPerInchY() const
232{
233 return size().height() / physicalSize().height() * qreal(25.4);
234}
235
236/*!
237 \property QScreen::physicalDotsPerInch
238 \brief the number of physical dots or pixels per inch
239
240 This value represents the pixel density on the screen's display.
241 Depending on what information the underlying system provides the value might not be
242 entirely accurate.
243
244 This is a convenience property that's simply the average of the physicalDotsPerInchX
245 and physicalDotsPerInchY properties.
246
247 \note Physical DPI is expressed in device-independent dots. Multiply by QScreen::devicePixelRatio()
248 to get device-dependent density.
249
250 \sa physicalDotsPerInchX()
251 \sa physicalDotsPerInchY()
252*/
253qreal QScreen::physicalDotsPerInch() const
254{
255 QSize sz = size();
256 QSizeF psz = physicalSize();
257 return ((sz.height() / psz.height()) + (sz.width() / psz.width())) * qreal(25.4 * 0.5);
258}

388/*!
389 Get the screen's virtual siblings.
390
391 The virtual siblings are the screen instances sharing the same virtual desktop.
392 They share a common coordinate system, and windows can freely be moved or
393 positioned across them without having to be re-created.
394*/
395QList<QScreen *> QScreen::virtualSiblings() const
396{
397 Q_D(const QScreen);
398 const QList<QPlatformScreen *> platformScreens = d->platformScreen->virtualSiblings();
399 QList<QScreen *> screens;
400 screens.reserve(size: platformScreens.size());
401 for (QPlatformScreen *platformScreen : platformScreens)
402 screens << platformScreen->screen();
403 return screens;
404}

565/*!
566 Convenience function to compute a transform that maps from the coordinate system
567 defined by orientation \a a into the coordinate system defined by orientation
568 \a b and target dimensions \a target.
569
570 Example, \a a is Qt::Landscape, \a b is Qt::Portrait, and \a target is QRect(0, 0, w, h)
571 the resulting transform will be such that the point QPoint(0, 0) is mapped to QPoint(0, w),
572 and QPoint(h, w) is mapped to QPoint(0, h). Thus, the landscape coordinate system QRect(0, 0, h, w)
573 is mapped (with a 90 degree rotation) into the portrait coordinate system QRect(0, 0, w, h).
574
575 Qt::PrimaryOrientation is interpreted as the screen's primaryOrientation().
576*/
577QTransform QScreen::transformBetween(Qt::ScreenOrientation a, Qt::ScreenOrientation b, const QRect &target) const
578{
579 if (a == Qt::PrimaryOrientation)
580 a = primaryOrientation();
581
582 if (b == Qt::PrimaryOrientation)
583 b = primaryOrientation();
584
585 return QPlatformScreen::transformBetween(a, b, target);
586}

674/*!
675 Creates and returns a pixmap constructed by grabbing the contents
676 of the given \a window restricted by QRect(\a x, \a y, \a width,
677 \a height). If \a window is 0, then the entire screen will be
678 grabbed.
679
680 The arguments (\a{x}, \a{y}) specify the offset in the window,
681 whereas (\a{width}, \a{height}) specify the area to be copied. If
682 \a width is negative, the function copies everything to the right
683 border of the window. If \a height is negative, the function
684 copies everything to the bottom of the window.
685
686 The offset and size arguments are specified in device independent
687 pixels. The returned pixmap may be larger than the requested size
688 when grabbing from a high-DPI screen. Call QPixmap::devicePixelRatio()
689 to determine if this is the case.
690
691 The window system identifier (\c WId) can be retrieved using the
692 QWidget::winId() function. The rationale for using a window
693 identifier and not a QWidget, is to enable grabbing of windows
694 that are not part of the application, window system frames, and so
695 on.
696
697 \warning Grabbing windows that are not part of the application is
698 not supported on systems such as iOS, where sandboxing/security
699 prevents reading pixels of windows not owned by the application.
700
701 The grabWindow() function grabs pixels from the screen, not from
702 the window, i.e. if there is another window partially or entirely
703 over the one you grab, you get pixels from the overlying window,
704 too. The mouse cursor is generally not grabbed.
705
706 Note on X11 that if the given \a window doesn't have the same depth
707 as the root window, and another window partially or entirely
708 obscures the one you grab, you will \e not get pixels from the
709 overlying window. The contents of the obscured areas in the
710 pixmap will be undefined and uninitialized.
711
712 On Windows Vista and above grabbing a layered window, which is
713 created by setting the Qt::WA_TranslucentBackground attribute, will
714 not work. Instead grabbing the desktop widget should work.
715
716 \warning In general, grabbing an area outside the screen is not
717 safe. This depends on the underlying window system.
718*/
719
720QPixmap QScreen::grabWindow(WId window, int x, int y, int width, int height)
721{
722 const QPlatformScreen *platformScreen = handle();
723 if (!platformScreen) {
724 qWarning(msg: "invoked with handle==0");
725 return QPixmap();
726 }
727 const qreal factor = QHighDpiScaling::factor(context: this);
728 if (qFuzzyCompare(p1: factor, p2: 1))
729 return platformScreen->grabWindow(window, x, y, width, height);
730
731 const QPoint nativePos = QHighDpi::toNative(value: QPoint(x, y), scaleFactor: factor);
732 QSize nativeSize(width, height);
733 if (nativeSize.isValid())
734 nativeSize = QHighDpi::toNative(value: nativeSize, scaleFactor: factor);
735 QPixmap result =
736 platformScreen->grabWindow(window, x: nativePos.x(), y: nativePos.y(),
737 width: nativeSize.width(), height: nativeSize.height());
738 result.setDevicePixelRatio(result.devicePixelRatio() * factor);
739 return result;
740}
741void *QScreen::resolveInterface(const char *name, int revision) const
742{
743 using namespace QNativeInterface::Private;
744
745 auto *platformScreen = handle();
746 Q_UNUSED(platformScreen);
747 Q_UNUSED(name);
748 Q_UNUSED(revision);
749
750#if QT_CONFIG(xcb)
751 QT_NATIVE_INTERFACE_RETURN_IF(QXcbScreen, platformScreen);
752#endif
753
754#if QT_CONFIG(vsp2)
755 QT_NATIVE_INTERFACE_RETURN_IF(QVsp2Screen, platformScreen);
756#endif
757
758#if defined(Q_OS_WEBOS)
759 QT_NATIVE_INTERFACE_RETURN_IF(QWebOSScreen, platformScreen);
760#endif
761
762#if defined(Q_OS_WIN32)
763 QT_NATIVE_INTERFACE_RETURN_IF(QWindowsScreen, platformScreen);
764#endif
765
766#if defined(Q_OS_ANDROID)
767 QT_NATIVE_INTERFACE_RETURN_IF(QAndroidScreen, platformScreen);
768#endif
769
770#if defined(Q_OS_UNIX)
771 QT_NATIVE_INTERFACE_RETURN_IF(QWaylandScreen, platformScreen);
772#endif
773
774 return nullptr;
775}

803#endif // !QT_NO_DEBUG_STREAM

805QScreenPrivate::UpdateEmitter::UpdateEmitter(QScreen *screen)
806{
807 initialState.platformScreen = screen->handle();
808
809 // Use public APIs to read out current state, rather
810 // than accessing the QScreenPrivate members, so that
811 // we detect any changes to the high-DPI scale factors
812 // that may be applied in the getters.
813
814 initialState.logicalDpi = QDpi{
815 screen->logicalDotsPerInchX(),
816 screen->logicalDotsPerInchY()
817 };
818 initialState.geometry = screen->geometry();
819 initialState.availableGeometry = screen->availableGeometry();
820 initialState.primaryOrientation = screen->primaryOrientation();
821}
822
823QScreenPrivate::UpdateEmitter::~UpdateEmitter()
824{
825 QScreen *screen = initialState.platformScreen->screen();
826
827 const auto logicalDotsPerInch = QDpi{
828 screen->logicalDotsPerInchX(),
829 screen->logicalDotsPerInchY()
830 };
831 if (logicalDotsPerInch != initialState.logicalDpi)
832 emit screen->logicalDotsPerInchChanged(dpi: screen->logicalDotsPerInch());
833
834 const auto geometry = screen->geometry();
835 const auto geometryChanged = geometry != initialState.geometry;
836 if (geometryChanged)
837 emit screen->geometryChanged(geometry);
838
839 const auto availableGeometry = screen->availableGeometry();
840 const auto availableGeometryChanged = availableGeometry != initialState.availableGeometry;
841 if (availableGeometryChanged)
842 emit screen->availableGeometryChanged(geometry: availableGeometry);
843
844 if (geometryChanged || availableGeometryChanged) {
845 const auto siblings = screen->virtualSiblings();
846 for (QScreen* sibling : siblings)
847 emit sibling->virtualGeometryChanged(rect: sibling->virtualGeometry());
848 }
849
850 if (geometryChanged) {
851 emit screen->physicalDotsPerInchChanged(dpi: screen->physicalDotsPerInch());
852
853 const auto primaryOrientation = screen->primaryOrientation();
854 if (primaryOrientation != initialState.primaryOrientation)
855 emit screen->primaryOrientationChanged(orientation: primaryOrientation);
856 }
857}
