25class Q_GUI_EXPORT QOpenGLExtensions : public QOpenGLExtraFunctions

28public:
29 QOpenGLExtensions();
30 QOpenGLExtensions(QOpenGLContext *context);
31 ~QOpenGLExtensions() {}

33 enum OpenGLExtension {
34 TextureRectangle = 0x00000001,
35 GenerateMipmap = 0x00000002,
36 TextureCompression = 0x00000004,
37 MirroredRepeat = 0x00000008,
38 FramebufferMultisample = 0x00000010,
39 StencilTwoSide = 0x00000020,
40 StencilWrap = 0x00000040,
41 PackedDepthStencil = 0x00000080,
42 NVFloatBuffer = 0x00000100,
43 PixelBufferObject = 0x00000200,
44 FramebufferBlit = 0x00000400,
45 BGRATextureFormat = 0x00000800,
46 DDSTextureCompression = 0x00001000,
47 ETC1TextureCompression = 0x00002000,
48 PVRTCTextureCompression = 0x00004000,
49 ElementIndexUint = 0x00008000,
50 Depth24 = 0x00010000,
51 SRGBFrameBuffer = 0x00020000,
52 MapBuffer = 0x00040000,
53 GeometryShaders = 0x00080000,
54 MapBufferRange = 0x00100000,
55 Sized8Formats = 0x00200000,
56 DiscardFramebuffer = 0x00400000,
57 Sized16Formats = 0x00800000,
58 TextureSwizzle = 0x01000000,
59 StandardDerivatives = 0x02000000,
60 ASTCTextureCompression = 0x04000000,
61 ETC2TextureCompression = 0x08000000
62 };
63 Q_DECLARE_FLAGS(OpenGLExtensions, OpenGLExtension)
64
65 OpenGLExtensions openGLExtensions();
66 bool hasOpenGLExtension(QOpenGLExtensions::OpenGLExtension extension) const;
67
68 GLvoid *glMapBuffer(GLenum target, GLenum access);
69 void glGetBufferSubData(GLenum target, qopengl_GLintptr offset, qopengl_GLsizeiptr size, GLvoid *data);

72 void flushShared();

76private:

78};

80Q_DECLARE_OPERATORS_FOR_FLAGS(QOpenGLExtensions::OpenGLExtensions)

82class QOpenGLExtensionsPrivate : public QOpenGLExtraFunctionsPrivate
83{
84public:
85 explicit QOpenGLExtensionsPrivate(QOpenGLContext *ctx);
86
87 GLvoid* (QOPENGLF_APIENTRYP MapBuffer)(GLenum target, GLenum access);
88 void (QOPENGLF_APIENTRYP GetBufferSubData)(GLenum target, qopengl_GLintptr offset, qopengl_GLsizeiptr size, GLvoid *data);
89 void (QOPENGLF_APIENTRYP DiscardFramebuffer)(GLenum target, GLsizei numAttachments, const GLenum *attachments);
90
91 bool flushVendorChecked;
92 bool flushIsSufficientToSyncContexts;
93};
94
95inline QOpenGLExtensionsPrivate *QOpenGLExtensions::d() const
96{
97 return static_cast<QOpenGLExtensionsPrivate *>(d_ptr);
98}
99
100inline GLvoid *QOpenGLExtensions::glMapBuffer(GLenum target, GLenum access)
101{
102 Q_D(QOpenGLExtensions);
103 Q_ASSERT(QOpenGLExtensions::isInitialized(d));
104 GLvoid *result = d->MapBuffer(target, access);
105 Q_OPENGL_FUNCTIONS_DEBUG
106 return result;
107}
108
109inline void QOpenGLExtensions::glGetBufferSubData(GLenum target, qopengl_GLintptr offset, qopengl_GLsizeiptr size, GLvoid *data)
110{
111 Q_D(QOpenGLExtensions);
112 Q_ASSERT(QOpenGLExtensions::isInitialized(d));
113 d->GetBufferSubData(target, offset, size, data);
114 Q_OPENGL_FUNCTIONS_DEBUG
115}
116
117
118inline void QOpenGLExtensions::glDiscardFramebufferEXT (GLenum target, GLsizei numAttachments, const GLenum *attachments)
119{
120 Q_D(QOpenGLExtensions);
121 Q_ASSERT(QOpenGLExtensions::isInitialized(d));
122 d->DiscardFramebuffer(target,numAttachments, attachments);
123 Q_OPENGL_FUNCTIONS_DEBUG
124}
