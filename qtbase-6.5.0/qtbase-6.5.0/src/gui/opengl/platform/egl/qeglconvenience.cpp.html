22QList<EGLint> q_createConfigAttributesFromFormat(const QSurfaceFormat &format)
23{
24 int redSize = format.redBufferSize();
25 int greenSize = format.greenBufferSize();
26 int blueSize = format.blueBufferSize();
27 int alphaSize = format.alphaBufferSize();
28 int depthSize = format.depthBufferSize();
29 int stencilSize = format.stencilBufferSize();
30 int sampleCount = format.samples();
31
32 QList<EGLint> configAttributes;
33
34 // Map default, unspecified values (-1) to 0. This is important due to sorting rule #3
35 // in section 3.4.1 of the spec and allows picking a potentially faster 16-bit config
36 // over 32-bit ones when there is no explicit request for the color channel sizes:
37 //
38 // The red/green/blue sizes have a sort priority of 3, so they are sorted by
39 // first. (unless a caveat like SLOW or NON_CONFORMANT is present) The sort order is
40 // Special and described as "by larger _total_ number of color bits.". So EGL will put
41 // 32-bit configs in the list before the 16-bit configs. However, the spec also goes
42 // on to say "If the requested number of bits in attrib_list for a particular
43 // component is 0, then the number of bits for that component is not considered". This
44 // part of the spec also seems to imply that setting the red/green/blue bits to zero
45 // means none of the components are considered and EGL disregards the entire sorting
46 // rule. It then looks to the next highest priority rule, which is
47 // EGL_BUFFER_SIZE. Despite the selection criteria being "AtLeast" for
48 // EGL_BUFFER_SIZE, it's sort order is "smaller" meaning 16-bit configs are put in the
49 // list before 32-bit configs.
50 //
51 // This also means that explicitly specifying a size like 565 will still result in
52 // having larger (888) configs first in the returned list. We need to handle this
53 // ourselves later by manually filtering the list, instead of just blindly taking the
54 // first config from it.
55
56 configAttributes.append(EGL_RED_SIZE);
57 configAttributes.append(t: redSize > 0 ? redSize : 0);
58
59 configAttributes.append(EGL_GREEN_SIZE);
60 configAttributes.append(t: greenSize > 0 ? greenSize : 0);
61
62 configAttributes.append(EGL_BLUE_SIZE);
63 configAttributes.append(t: blueSize > 0 ? blueSize : 0);
64
65 configAttributes.append(EGL_ALPHA_SIZE);
66 configAttributes.append(t: alphaSize > 0 ? alphaSize : 0);
67
68 configAttributes.append(EGL_SAMPLES);
69 configAttributes.append(t: sampleCount > 0 ? sampleCount : 0);
70
71 configAttributes.append(EGL_SAMPLE_BUFFERS);
72 configAttributes.append(t: sampleCount > 0);
73
74 if (format.renderableType() != QSurfaceFormat::OpenVG) {
75 configAttributes.append(EGL_DEPTH_SIZE);
76 configAttributes.append(t: depthSize > 0 ? depthSize : 0);
77
78 configAttributes.append(EGL_STENCIL_SIZE);
79 configAttributes.append(t: stencilSize > 0 ? stencilSize : 0);
80 } else {
81 // OpenVG needs alpha mask for clipping
82 configAttributes.append(EGL_ALPHA_MASK_SIZE);
83 configAttributes.append(t: 8);
84 }
85
86 return configAttributes;
87}

89bool q_reduceConfigAttributes(QList<EGLint> *configAttributes)
90{
91 int i = -1;
92 // Reduce the complexity of a configuration request to ask for less
93 // because the previous request did not result in success. Returns
94 // true if the complexity was reduced, or false if no further
95 // reductions in complexity are possible.
96
97 i = configAttributes->indexOf(EGL_SWAP_BEHAVIOR);
98 if (i >= 0) {
99 configAttributes->remove(i,n: 2);
100 }
101
102#ifdef EGL_VG_ALPHA_FORMAT_PRE_BIT
103 // For OpenVG, we sometimes try to create a surface using a pre-multiplied format. If we can't
104 // find a config which supports pre-multiplied formats, remove the flag on the surface type:
105
106 i = configAttributes->indexOf(EGL_SURFACE_TYPE);
107 if (i >= 0) {
108 EGLint surfaceType = configAttributes->at(i: i +1);
109 if (surfaceType & EGL_VG_ALPHA_FORMAT_PRE_BIT) {
110 surfaceType ^= EGL_VG_ALPHA_FORMAT_PRE_BIT;
111 configAttributes->replace(i: i+1,t: surfaceType);
112 return true;
113 }
114 }
115#endif
116
117 // EGL chooses configs with the highest color depth over
118 // those with smaller (but faster) lower color depths. One
119 // way around this is to set EGL_BUFFER_SIZE to 16, which
120 // trumps the others. Of course, there may not be a 16-bit
121 // config available, so it's the first restraint we remove.
122 i = configAttributes->indexOf(EGL_BUFFER_SIZE);
123 if (i >= 0) {
124 if (configAttributes->at(i: i+1) == 16) {
125 configAttributes->remove(i,n: 2);
126 return true;
127 }
128 }
129
130 i = configAttributes->indexOf(EGL_SAMPLES);
131 if (i >= 0) {
132 EGLint value = configAttributes->value(i: i+1, defaultValue: 0);
133 if (value > 1)
134 configAttributes->replace(i: i+1, t: qMin(a: EGLint(16), b: value / 2));
135 else
136 configAttributes->remove(i, n: 2);
137 return true;
138 }
139
140 i = configAttributes->indexOf(EGL_SAMPLE_BUFFERS);
141 if (i >= 0) {
142 configAttributes->remove(i,n: 2);
143 return true;
144 }
145
146 i = configAttributes->indexOf(EGL_DEPTH_SIZE);
147 if (i >= 0) {
148 if (configAttributes->at(i: i + 1) >= 32)
149 configAttributes->replace(i: i + 1, t: 24);
150 else if (configAttributes->at(i: i + 1) > 1)
151 configAttributes->replace(i: i + 1, t: 1);
152 else
153 configAttributes->remove(i, n: 2);
154 return true;
155 }
156
157 i = configAttributes->indexOf(EGL_ALPHA_SIZE);
158 if (i >= 0) {
159 configAttributes->remove(i,n: 2);
160#if defined(EGL_BIND_TO_TEXTURE_RGBA) && defined(EGL_BIND_TO_TEXTURE_RGB)
161 i = configAttributes->indexOf(EGL_BIND_TO_TEXTURE_RGBA);
162 if (i >= 0) {
163 configAttributes->replace(i,EGL_BIND_TO_TEXTURE_RGB);
164 configAttributes->replace(i: i+1,t: true);
165
166 }
167#endif
168 return true;
169 }
170
171 i = configAttributes->indexOf(EGL_STENCIL_SIZE);
172 if (i >= 0) {
173 if (configAttributes->at(i: i + 1) > 1)
174 configAttributes->replace(i: i + 1, t: 1);
175 else
176 configAttributes->remove(i, n: 2);
177 return true;
178 }
179
180#ifdef EGL_BIND_TO_TEXTURE_RGB
181 i = configAttributes->indexOf(EGL_BIND_TO_TEXTURE_RGB);
182 if (i >= 0) {
183 configAttributes->remove(i,n: 2);
184 return true;
185 }
186#endif
187
188 return false;
189}

206EGLConfig QEglConfigChooser::chooseConfig()
207{
208 QList<EGLint> configureAttributes = q_createConfigAttributesFromFormat(format: m_format);
209 configureAttributes.append(EGL_SURFACE_TYPE);
210 configureAttributes.append(t: surfaceType());
211
212 configureAttributes.append(EGL_RENDERABLE_TYPE);
213 bool needsES2Plus = false;
214 switch (m_format.renderableType()) {
215 case QSurfaceFormat::OpenVG:
216 configureAttributes.append(EGL_OPENVG_BIT);
217 break;
218#ifdef EGL_VERSION_1_4
219 case QSurfaceFormat::DefaultRenderableType:
220#ifndef QT_NO_OPENGL
221 if (QOpenGLContext::openGLModuleType() == QOpenGLContext::LibGL)
222 configureAttributes.append(EGL_OPENGL_BIT);
223 else
224#endif // QT_NO_OPENGL
225 needsES2Plus = true;
226 break;
227 case QSurfaceFormat::OpenGL:
228 configureAttributes.append(EGL_OPENGL_BIT);
229 break;
230#endif
231 case QSurfaceFormat::OpenGLES:
232 if (m_format.majorVersion() == 1) {
233 configureAttributes.append(EGL_OPENGL_ES_BIT);
234 break;
235 }
236 Q_FALLTHROUGH();
237 default:
238 needsES2Plus = true;
239 break;
240 }
241 if (needsES2Plus) {
242 if (m_format.majorVersion() >= 3 && q_hasEglExtension(display: display(), extensionName: "EGL_KHR_create_context"))
243 configureAttributes.append(EGL_OPENGL_ES3_BIT_KHR);
244 else
245 configureAttributes.append(EGL_OPENGL_ES2_BIT);
246 }
247 configureAttributes.append(EGL_NONE);
248
249 EGLConfig cfg = nullptr;
250 do {
251 // Get the number of matching configurations for this set of properties.
252 EGLint matching = 0;
253 if (!eglChooseConfig(dpy: display(), attrib_list: configureAttributes.constData(), configs: nullptr, config_size: 0, num_config: &matching) || !matching)
254 continue;
255
256 // Fetch all of the matching configurations and find the
257 // first that matches the pixel format we wanted.
258 int i = configureAttributes.indexOf(EGL_RED_SIZE);
259 m_confAttrRed = configureAttributes.at(i: i+1);
260 i = configureAttributes.indexOf(EGL_GREEN_SIZE);
261 m_confAttrGreen = configureAttributes.at(i: i+1);
262 i = configureAttributes.indexOf(EGL_BLUE_SIZE);
263 m_confAttrBlue = configureAttributes.at(i: i+1);
264 i = configureAttributes.indexOf(EGL_ALPHA_SIZE);
265 m_confAttrAlpha = i == -1 ? 0 : configureAttributes.at(i: i+1);
266
267 QList<EGLConfig> configs(matching);
268 eglChooseConfig(dpy: display(), attrib_list: configureAttributes.constData(), configs: configs.data(), config_size: configs.size(), num_config: &matching);
269 if (!cfg && matching > 0)
270 cfg = configs.first();
271
272 // Filter the list. Due to the EGL sorting rules configs with higher depth are
273 // placed first when the minimum color channel sizes have been specified (i.e. the
274 // QSurfaceFormat contains color sizes > 0). To prevent returning a 888 config
275 // when the QSurfaceFormat explicitly asked for 565, go through the returned
276 // configs and look for one that exactly matches the requested sizes. When no
277 // sizes have been given, take the first, which will be a config with the smaller
278 // (e.g. 16-bit) depth.
279 for (int i = 0; i < configs.size(); ++i) {
280 if (filterConfig(config: configs[i]))
281 return configs.at(i);
282 }
283 } while (q_reduceConfigAttributes(configAttributes: &configureAttributes));
284
285 if (!cfg)
286 qWarning(msg: "Cannot find EGLConfig, returning null config");
287 return cfg;
288}

290bool QEglConfigChooser::filterConfig(EGLConfig config) const
291{
292 // If we are fine with the highest depth (e.g. RGB888 configs) even when something
293 // smaller (565) was explicitly requested, do nothing.
294 if (m_ignore)
295 return true;
296
297 EGLint red = 0;
298 EGLint green = 0;
299 EGLint blue = 0;
300 EGLint alpha = 0;
301
302 // Compare only if a size was given. Otherwise just accept.
303 if (m_confAttrRed)
304 eglGetConfigAttrib(dpy: display(), config, EGL_RED_SIZE, value: &red);
305 if (m_confAttrGreen)
306 eglGetConfigAttrib(dpy: display(), config, EGL_GREEN_SIZE, value: &green);
307 if (m_confAttrBlue)
308 eglGetConfigAttrib(dpy: display(), config, EGL_BLUE_SIZE, value: &blue);
309 if (m_confAttrAlpha)
310 eglGetConfigAttrib(dpy: display(), config, EGL_ALPHA_SIZE, value: &alpha);
311
312 return red == m_confAttrRed && green == m_confAttrGreen
313 && blue == m_confAttrBlue && alpha == m_confAttrAlpha;
314}

326QSurfaceFormat q_glFormatFromConfig(EGLDisplay display, const EGLConfig config, const QSurfaceFormat &referenceFormat)
327{
328 QSurfaceFormat format;
329 EGLint redSize = 0;
330 EGLint greenSize = 0;
331 EGLint blueSize = 0;
332 EGLint alphaSize = 0;
333 EGLint depthSize = 0;
334 EGLint stencilSize = 0;
335 EGLint sampleCount = 0;
336 EGLint renderableType = 0;
337
338 eglGetConfigAttrib(dpy: display, config, EGL_RED_SIZE, value: &redSize);
339 eglGetConfigAttrib(dpy: display, config, EGL_GREEN_SIZE, value: &greenSize);
340 eglGetConfigAttrib(dpy: display, config, EGL_BLUE_SIZE, value: &blueSize);
341 eglGetConfigAttrib(dpy: display, config, EGL_ALPHA_SIZE, value: &alphaSize);
342 eglGetConfigAttrib(dpy: display, config, EGL_DEPTH_SIZE, value: &depthSize);
343 eglGetConfigAttrib(dpy: display, config, EGL_STENCIL_SIZE, value: &stencilSize);
344 eglGetConfigAttrib(dpy: display, config, EGL_SAMPLES, value: &sampleCount);
345 eglGetConfigAttrib(dpy: display, config, EGL_RENDERABLE_TYPE, value: &renderableType);
346
347 if (referenceFormat.renderableType() == QSurfaceFormat::OpenVG && (renderableType & EGL_OPENVG_BIT))
348 format.setRenderableType(QSurfaceFormat::OpenVG);
349#ifdef EGL_VERSION_1_4
350 else if (referenceFormat.renderableType() == QSurfaceFormat::OpenGL
351 && (renderableType & EGL_OPENGL_BIT))
352 format.setRenderableType(QSurfaceFormat::OpenGL);
353 else if (referenceFormat.renderableType() == QSurfaceFormat::DefaultRenderableType
354#ifndef QT_NO_OPENGL
355 && QOpenGLContext::openGLModuleType() == QOpenGLContext::LibGL
356#endif
357 && (renderableType & EGL_OPENGL_BIT))
358 format.setRenderableType(QSurfaceFormat::OpenGL);
359#endif
360 else
361 format.setRenderableType(QSurfaceFormat::OpenGLES);
362
363 format.setRedBufferSize(redSize);
364 format.setGreenBufferSize(greenSize);
365 format.setBlueBufferSize(blueSize);
366 format.setAlphaBufferSize(alphaSize);
367 format.setDepthBufferSize(depthSize);
368 format.setStencilBufferSize(stencilSize);
369 format.setSamples(sampleCount);
370 format.setStereo(false); // EGL doesn't support stereo buffers
371 format.setSwapInterval(referenceFormat.swapInterval());
372
373 // Clear the EGL error state because some of the above may
374 // have errored out because the attribute is not applicable
375 // to the surface type. Such errors don't matter.
376 eglGetError();
377
378 return format;
379}

433QSizeF q_physicalScreenSizeFromFb(int framebufferDevice, const QSize &screenSize)
434{
435#ifndef Q_OS_LINUX
436 Q_UNUSED(framebufferDevice);
437#endif
438 const int defaultPhysicalDpi = 100;
439 static QSizeF size;
440
441 if (size.isEmpty()) {
442 // Note: in millimeters
443 int width = qEnvironmentVariableIntValue(varName: "QT_QPA_EGLFS_PHYSICAL_WIDTH");
444 int height = qEnvironmentVariableIntValue(varName: "QT_QPA_EGLFS_PHYSICAL_HEIGHT");
445
446 if (width && height) {
447 size.setWidth(width);
448 size.setHeight(height);
449 return size;
450 }
451
452 int w = -1;
453 int h = -1;
454 QSize screenResolution;
455#ifdef Q_OS_LINUX
456 struct fb_var_screeninfo vinfo;
457
458 if (framebufferDevice != -1) {
459 if (ioctl(fd: framebufferDevice, FBIOGET_VSCREENINFO, &vinfo) == -1) {
460 qWarning(msg: "eglconvenience: Could not query screen info");
461 } else {
462 w = vinfo.width;
463 h = vinfo.height;
464 screenResolution = QSize(vinfo.xres, vinfo.yres);
465 }
466 } else
467#endif
468 {
469 // Use the provided screen size, when available, since some platforms may have their own
470 // specific way to query it. Otherwise try querying it from the framebuffer.
471 screenResolution = screenSize.isEmpty() ? q_screenSizeFromFb(framebufferDevice) : screenSize;
472 }
473
474 size.setWidth(w <= 0 ? screenResolution.width() * Q_MM_PER_INCH / defaultPhysicalDpi : qreal(w));
475 size.setHeight(h <= 0 ? screenResolution.height() * Q_MM_PER_INCH / defaultPhysicalDpi : qreal(h));
476
477 if (w <= 0 || h <= 0)
478 qWarning(msg: "Unable to query physical screen size, defaulting to %d dpi.\n"
479 "To override, set QT_QPA_EGLFS_PHYSICAL_WIDTH "
480 "and QT_QPA_EGLFS_PHYSICAL_HEIGHT (in millimeters).", defaultPhysicalDpi);
481 }
482
483 return size;
484}

486QSize q_screenSizeFromFb(int framebufferDevice)
487{
488#ifndef Q_OS_LINUX
489 Q_UNUSED(framebufferDevice);
490#endif
491 const int defaultWidth = 800;
492 const int defaultHeight = 600;
493 static QSize size;
494
495 if (size.isEmpty()) {
496 int width = qEnvironmentVariableIntValue(varName: "QT_QPA_EGLFS_WIDTH");
497 int height = qEnvironmentVariableIntValue(varName: "QT_QPA_EGLFS_HEIGHT");
498
499 if (width && height) {
500 size.setWidth(width);
501 size.setHeight(height);
502 return size;
503 }
504
505#ifdef Q_OS_LINUX
506 struct fb_var_screeninfo vinfo;
507 int xres = -1;
508 int yres = -1;
509
510 if (framebufferDevice != -1) {
511 if (ioctl(fd: framebufferDevice, FBIOGET_VSCREENINFO, &vinfo) == -1) {
512 qWarning(msg: "eglconvenience: Could not read screen info");
513 } else {
514 xres = vinfo.xres;
515 yres = vinfo.yres;
516 }
517 }
518
519 size.setWidth(xres <= 0 ? defaultWidth : xres);
520 size.setHeight(yres <= 0 ? defaultHeight : yres);
521#else
522 size.setWidth(defaultWidth);
523 size.setHeight(defaultHeight);
524#endif
525 }
526
527 return size;
528}

559qreal q_refreshRateFromFb(int framebufferDevice)
560{
561#ifndef Q_OS_LINUX
562 Q_UNUSED(framebufferDevice);
563#endif
564
565 static qreal rate = 0;
566
567#ifdef Q_OS_LINUX
568 if (rate == 0) {
569 if (framebufferDevice != -1) {
570 struct fb_var_screeninfo vinfo;
571 if (ioctl(fd: framebufferDevice, FBIOGET_VSCREENINFO, &vinfo) != -1) {
572 const quint64 quot = quint64(vinfo.left_margin + vinfo.right_margin + vinfo.xres + vinfo.hsync_len)
573 * quint64(vinfo.upper_margin + vinfo.lower_margin + vinfo.yres + vinfo.vsync_len)
574 * vinfo.pixclock;
575 if (quot)
576 rate = 1000000000000LLU / quot;
577 } else {
578 qWarning(msg: "eglconvenience: Could not query screen info");
579 }
580 }
581 }
582#endif
583
584 if (rate == 0)
585 rate = 60;
586
587 return rate;
588}
