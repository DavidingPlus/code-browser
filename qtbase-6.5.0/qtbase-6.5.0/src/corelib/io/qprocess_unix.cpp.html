54QProcessEnvironment QProcessEnvironment::systemEnvironment()
55{
56 QProcessEnvironment env;
57 const char *entry;
58 for (int count = 0; (entry = environ[count]); ++count) {
59 const char *equal = strchr(s: entry, c: '=');
60 if (!equal)
61 continue;
62
63 QByteArray name(entry, equal - entry);
64 QByteArray value(equal + 1);
65 env.d->vars.insert(key: QProcessEnvironmentPrivate::Key(name),
66 value: QProcessEnvironmentPrivate::Value(value));
67 }
68 return env;
69}

76struct AutoPipe
77{
78 int pipe[2] = { -1, -1 };
79 AutoPipe(int flags = 0)
80 {
81 qt_safe_pipe(pipefd: pipe, flags);
82 }
83 ~AutoPipe()
84 {
85 for (int fd : pipe) {
86 if (fd >= 0)
87 qt_safe_close(fd);
88 }
89 }
90
91 explicit operator bool() const { return pipe[0] >= 0; }
92 int &operator[](int idx) { return pipe[idx]; }
93 int operator[](int idx) const { return pipe[idx]; }

123 pollfd &stderrPipe() { return pfds[2]; }

130QProcessPoller::QProcessPoller(const QProcessPrivate &proc)
131{
132 for (int i = 0; i < n_pfds; i++)
133 pfds[i] = qt_make_pollfd(fd: -1, POLLIN);
134
135 stdoutPipe().fd = proc.stdoutChannel.pipe[0];
136 stderrPipe().fd = proc.stderrChannel.pipe[0];
137
138 if (!proc.writeBuffer.isEmpty()) {
139 stdinPipe().fd = proc.stdinChannel.pipe[1];
140 stdinPipe().events = POLLOUT;
141 }
142
143 forkfd().fd = proc.forkfd;
144}

151CharPointerList::CharPointerList(const QString &program, const QStringList &args)
152{
153 qsizetype count = 1 + args.size();
154 pointers.reset(p: new char *[count + 1]);
155 pointers[count] = nullptr;
156
157 // we abuse the pointer array to store offsets first (QByteArray will
158 // reallocate, after all)
159 pointers[0] = reinterpret_cast<char *>(0);
160 data = QFile::encodeName(fileName: program);
161 data += '\0';
162
163 const auto end = args.end();
164 auto it = args.begin();
165 for (qsizetype i = 1; it != end; ++it, ++i) {
166 pointers[i] = reinterpret_cast<char *>(data.size());
167 data += QFile::encodeName(fileName: *it);
168 data += '\0';
169 }
170
171 updatePointers(count);
172}
173
174CharPointerList::CharPointerList(const QProcessEnvironmentPrivate *environment)
175{
176 if (!environment)
177 return;
178
179 const QProcessEnvironmentPrivate::Map &env = environment->vars;
180 qsizetype count = env.size();
181 pointers.reset(p: new char *[count + 1]);
182 pointers[count] = nullptr;
183
184 const auto end = env.end();
185 auto it = env.begin();
186 for (qsizetype i = 0; it != end; ++it, ++i) {
187 // we abuse the pointer array to store offsets first (QByteArray will
188 // reallocate, after all)
189 pointers[i] = reinterpret_cast<char *>(data.size());
190
191 data += it.key();
192 data += '=';
193 data += it->bytes();
194 data += '\0';
195 }
196
197 updatePointers(count);
198}

200void CharPointerList::updatePointers(qsizetype count)
201{
202 char *const base = const_cast<char *>(data.constBegin());
203 for (qsizetype i = 0; i < count; ++i)
204 pointers[i] = base + qptrdiff(pointers[i]);
205}

208static bool qt_pollfd_check(const pollfd &pfd, short revents)
209{
210 return pfd.fd >= 0 && (pfd.revents & (revents | POLLHUP | POLLERR | POLLNVAL)) != 0;
211}

214{

261/*
262 Create the pipes to a QProcessPrivate::Channel.
263*/
264bool QProcessPrivate::openChannel(Channel &channel)
265{
266 Q_Q(QProcess);
267
268 if (channel.type == Channel::Normal) {
269 // we're piping this channel to our own process
270 if (qt_create_pipe(pipe: channel.pipe) != 0)
271 return false;
272
273 // create the socket notifiers
274 if (threadData.loadRelaxed()->hasEventDispatcher()) {
275 if (&channel == &stdinChannel) {
276 channel.notifier = new QSocketNotifier(QSocketNotifier::Write, q);
277 channel.notifier->setSocket(channel.pipe[1]);
278 QObject::connect(sender: channel.notifier, SIGNAL(activated(QSocketDescriptor)),
279 receiver: q, SLOT(_q_canWrite()));
280 } else {
281 channel.notifier = new QSocketNotifier(QSocketNotifier::Read, q);
282 channel.notifier->setSocket(channel.pipe[0]);
283 const char *receiver;
284 if (&channel == &stdoutChannel)
285 receiver = SLOT(_q_canReadStandardOutput());
286 else
287 receiver = SLOT(_q_canReadStandardError());
288 QObject::connect(sender: channel.notifier, SIGNAL(activated(QSocketDescriptor)),
289 receiver: q, member: receiver);
290 }
291 }
292
293 return true;
294 } else if (channel.type == Channel::Redirect) {
295 // we're redirecting the channel to/from a file
296 QByteArray fname = QFile::encodeName(fileName: channel.file);
297
298 if (&channel == &stdinChannel) {
299 // try to open in read-only mode
300 channel.pipe[1] = -1;
301 if ( (channel.pipe[0] = qt_safe_open(pathname: fname, O_RDONLY)) != -1)
302 return true; // success
303 setErrorAndEmit(error: QProcess::FailedToStart,
304 description: QProcess::tr(s: "Could not open input redirection for reading"));
305 } else {
306 int mode = O_WRONLY | O_CREAT;
307 if (channel.append)
308 mode |= O_APPEND;
309 else
310 mode |= O_TRUNC;
311
312 channel.pipe[0] = -1;
313 if ( (channel.pipe[1] = qt_safe_open(pathname: fname, flags: mode, mode: 0666)) != -1)
314 return true; // success
315
316 setErrorAndEmit(error: QProcess::FailedToStart,
317 description: QProcess::tr(s: "Could not open input redirection for reading"));
318 }
319 cleanup();
320 return false;
321 } else {
322 Q_ASSERT_X(channel.process, "QProcess::start", "Internal error");
323
324 Channel *source;
325 Channel *sink;
326
327 if (channel.type == Channel::PipeSource) {
328 // we are the source
329 source = &channel;
330 sink = &channel.process->stdinChannel;
331
332 Q_ASSERT(source == &stdoutChannel);
333 Q_ASSERT(sink->process == this && sink->type == Channel::PipeSink);
334 } else {
335 // we are the sink;
336 source = &channel.process->stdoutChannel;
337 sink = &channel;
338
339 Q_ASSERT(sink == &stdinChannel);
340 Q_ASSERT(source->process == this && source->type == Channel::PipeSource);
341 }
342
343 if (source->pipe[1] != INVALID_Q_PIPE || sink->pipe[0] != INVALID_Q_PIPE) {
344 // already created, do nothing
345 return true;
346 } else {
347 Q_ASSERT(source->pipe[0] == INVALID_Q_PIPE && source->pipe[1] == INVALID_Q_PIPE);
348 Q_ASSERT(sink->pipe[0] == INVALID_Q_PIPE && sink->pipe[1] == INVALID_Q_PIPE);
349
350 Q_PIPE pipe[2] = { -1, -1 };
351 if (qt_create_pipe(pipe) != 0)
352 return false;
353 sink->pipe[0] = pipe[0];
354 source->pipe[1] = pipe[1];
355
356 return true;
357 }
358 }
359}

361void QProcessPrivate::commitChannels()
362{
363 // copy the stdin socket if asked to (without closing on exec)
364 if (stdinChannel.pipe[0] != INVALID_Q_PIPE)
365 qt_safe_dup2(oldfd: stdinChannel.pipe[0], STDIN_FILENO, flags: 0);
366
367 // copy the stdout and stderr if asked to
368 if (stdoutChannel.pipe[1] != INVALID_Q_PIPE)
369 qt_safe_dup2(oldfd: stdoutChannel.pipe[1], STDOUT_FILENO, flags: 0);
370 if (stderrChannel.pipe[1] != INVALID_Q_PIPE) {
371 qt_safe_dup2(oldfd: stderrChannel.pipe[1], STDERR_FILENO, flags: 0);
372 } else {
373 // merge stdout and stderr if asked to
374 if (processChannelMode == QProcess::MergedChannels)
375 qt_safe_dup2(STDOUT_FILENO, STDERR_FILENO, flags: 0);
376 }
377}

379static QString resolveExecutable(const QString &program)
380{
381#ifdef Q_OS_DARWIN
382 // allow invoking of .app bundles on the Mac.
383 QFileInfo fileInfo(program);
384 if (program.endsWith(".app"_L1) && fileInfo.isDir()) {
385 QCFType<CFURLRef> url = CFURLCreateWithFileSystemPath(0,
386 QCFString(fileInfo.absoluteFilePath()),
387 kCFURLPOSIXPathStyle, true);
388 {
389 // CFBundle is not reentrant, since CFBundleCreate might return a reference
390 // to a cached bundle object. Protect the bundle calls with a mutex lock.
391 Q_CONSTINIT static QBasicMutex cfbundleMutex;
392 const auto locker = qt_scoped_lock(cfbundleMutex);
393 QCFType<CFBundleRef> bundle = CFBundleCreate(0, url);
394 // 'executableURL' can be either relative or absolute ...
395 QCFType<CFURLRef> executableURL = CFBundleCopyExecutableURL(bundle);
396 // not to depend on caching - make sure it's always absolute.
397 url = CFURLCopyAbsoluteURL(executableURL);
398 }
399 if (url) {
400 const QCFString str = CFURLCopyFileSystemPath(url, kCFURLPOSIXPathStyle);
401 return QString::fromCFString(str);
402 }
403 }
404#endif
405
406 if (!program.contains(c: u'/')) {
407 // findExecutable() returns its argument if it's an absolute path,
408 // otherwise it searches $PATH; returns empty if not found (we handle
409 // that case much later)
410 return QStandardPaths::findExecutable(executableName: program);
411 }
412 return program;
413}

415void QProcessPrivate::startProcess()
416{
417 Q_Q(QProcess);
418
419#if defined (QPROCESS_DEBUG)
420 qDebug("QProcessPrivate::startProcess()");
421#endif
422
423 // Initialize pipes
424 if (!openChannels() || qt_create_pipe(pipe: childStartedPipe) != 0) {
425 setErrorAndEmit(error: QProcess::FailedToStart, description: qt_error_string(errno));
426 cleanup();
427 return;
428 }
429
430 if (threadData.loadRelaxed()->hasEventDispatcher()) {
431 // Set up to notify about startup completion (and premature death).
432 // Once the process has started successfully, we reconfigure the
433 // notifier to watch the fork_fd for expected death.
434 stateNotifier = new QSocketNotifier(childStartedPipe[0],
435 QSocketNotifier::Read, q);
436 QObject::connect(sender: stateNotifier, SIGNAL(activated(QSocketDescriptor)),
437 receiver: q, SLOT(_q_startupNotification()));
438 }
439
440 // Start the process (platform dependent)
441 q->setProcessState(QProcess::Starting);
442
443 // Prepare the arguments and the environment
444 const CharPointerList argv(resolveExecutable(program), arguments);
445 const CharPointerList envp(environment.d.constData());
446
447 // Encode the working directory if it's non-empty, otherwise just pass 0.
448 const char *workingDirPtr = nullptr;
449 QByteArray encodedWorkingDirectory;
450 if (!workingDirectory.isEmpty()) {
451 encodedWorkingDirectory = QFile::encodeName(fileName: workingDirectory);
452 workingDirPtr = encodedWorkingDirectory.constData();
453 }
454
455 // Start the child.
456 auto execChild1 = [this, workingDirPtr, &argv, &envp]() {
457 execChild(workingDirectory: workingDirPtr, argv: argv.pointers.get(), envp: envp.pointers.get());
458 };
459 auto execChild2 = [](void *lambda) {
460 static_cast<decltype(execChild1) *>(lambda)->operator()();
461 return -1;
462 };
463
464 int ffdflags = FFD_CLOEXEC;
465
466 // QTBUG-86285
467#if defined(Q_OS_LINUX) && !QT_CONFIG(forkfd_pidfd)
468 ffdflags |= FFD_USE_FORK;
469#endif
470
471 pid_t childPid;
472 forkfd = ::vforkfd(flags: ffdflags , ppid: &childPid, childFn: execChild2, token: &execChild1);
473 int lastForkErrno = errno;
474
475 if (forkfd == -1) {
476 // Cleanup, report error and return
477#if defined (QPROCESS_DEBUG)
478 qDebug("fork failed: %ls", qUtf16Printable(qt_error_string(lastForkErrno)));
479#endif
480 q->setProcessState(QProcess::NotRunning);
481 setErrorAndEmit(error: QProcess::FailedToStart,
482 description: QProcess::tr(s: "Resource error (fork failure): %1").arg(a: qt_error_string(errorCode: lastForkErrno)));
483 cleanup();
484 return;
485 }
486
487 pid = qint64(childPid);
488 Q_ASSERT(pid > 0);
489
490 // parent
491 // close the ends we don't use and make all pipes non-blocking
492 qt_safe_close(fd: childStartedPipe[1]);
493 childStartedPipe[1] = -1;
494
495 if (stdinChannel.pipe[0] != -1) {
496 qt_safe_close(fd: stdinChannel.pipe[0]);
497 stdinChannel.pipe[0] = -1;
498 }
499
500 if (stdinChannel.pipe[1] != -1)
501 ::fcntl(fd: stdinChannel.pipe[1], F_SETFL, ::fcntl(fd: stdinChannel.pipe[1], F_GETFL) | O_NONBLOCK);
502
503 if (stdoutChannel.pipe[1] != -1) {
504 qt_safe_close(fd: stdoutChannel.pipe[1]);
505 stdoutChannel.pipe[1] = -1;
506 }
507
508 if (stdoutChannel.pipe[0] != -1)
509 ::fcntl(fd: stdoutChannel.pipe[0], F_SETFL, ::fcntl(fd: stdoutChannel.pipe[0], F_GETFL) | O_NONBLOCK);
510
511 if (stderrChannel.pipe[1] != -1) {
512 qt_safe_close(fd: stderrChannel.pipe[1]);
513 stderrChannel.pipe[1] = -1;
514 }
515 if (stderrChannel.pipe[0] != -1)
516 ::fcntl(fd: stderrChannel.pipe[0], F_SETFL, ::fcntl(fd: stderrChannel.pipe[0], F_GETFL) | O_NONBLOCK);
517}

519void QProcessPrivate::execChild(const char *workingDir, char **argv, char **envp)
520{
521 ::signal(SIGPIPE, SIG_DFL); // reset the signal that we ignored
522
523 ChildError error = { .code: 0, .function: {} }; // force zeroing of function[8]
524
525 // Render channels configuration.
526 commitChannels();
527
528 // make sure this fd is closed if execv() succeeds
529 qt_safe_close(fd: childStartedPipe[0]);
530
531 // enter the working directory
532 if (workingDir && QT_CHDIR(path: workingDir) == -1) {
533 // failed, stop the process
534 strcpy(dest: error.function, src: "chdir");
535 goto report_errno;
536 }
537
538 if (childProcessModifier)
539 childProcessModifier();
540
541 // execute the process
542 if (!envp) {
543 qt_safe_execv(path: argv[0], argv);
544 strcpy(dest: error.function, src: "execvp");
545 } else {
546#if defined (QPROCESS_DEBUG)
547 fprintf(stderr, "QProcessPrivate::execChild() starting %s\n", argv[0]);
548#endif
549 qt_safe_execve(filename: argv[0], argv, envp);
550 strcpy(dest: error.function, src: "execve");
551 }
552
553 // notify failure
554 // don't use strerror or any other routines that may allocate memory, since
555 // some buggy libc versions can deadlock on locked mutexes.
556report_errno:
557 error.code = errno;
558 qt_safe_write(fd: childStartedPipe[1], data: &error, len: sizeof(error));
559 childStartedPipe[1] = -1;
560}

562bool QProcessPrivate::processStarted(QString *errorMessage)
563{
564 Q_Q(QProcess);
565
566 ChildError buf;
567 int ret = qt_safe_read(fd: childStartedPipe[0], data: &buf, maxlen: sizeof(buf));
568
569 if (stateNotifier) {
570 stateNotifier->setEnabled(false);
571 stateNotifier->disconnect(receiver: q);
572 }
573 qt_safe_close(fd: childStartedPipe[0]);
574 childStartedPipe[0] = -1;
575
576#if defined (QPROCESS_DEBUG)
577 qDebug("QProcessPrivate::processStarted() == %s", ret <= 0 ? "true" : "false");
578#endif
579
580 if (ret <= 0) { // process successfully started
581 if (stateNotifier) {
582 QObject::connect(sender: stateNotifier, SIGNAL(activated(QSocketDescriptor)),
583 receiver: q, SLOT(_q_processDied()));
584 stateNotifier->setSocket(forkfd);
585 stateNotifier->setEnabled(true);
586 }
587 if (stdoutChannel.notifier)
588 stdoutChannel.notifier->setEnabled(true);
589 if (stderrChannel.notifier)
590 stderrChannel.notifier->setEnabled(true);
591
592 return true;
593 }
594
595 // did we read an error message?
596 if (errorMessage)
597 *errorMessage = QLatin1StringView(buf.function) + ": "_L1 + qt_error_string(errorCode: buf.code);
598
599 return false;
600}

652#endif
653 return len;

656bool QProcessPrivate::_q_canWrite()
657{
658 if (writeBuffer.isEmpty()) {
659 if (stdinChannel.notifier)
660 stdinChannel.notifier->setEnabled(false);
661#if defined QPROCESS_DEBUG
662 qDebug("QProcessPrivate::canWrite(), not writing anything (empty write buffer).");
663#endif
664 return false;
665 }
666
667 const bool writeSucceeded = writeToStdin();
668
669 if (writeBuffer.isEmpty() && stdinChannel.closed)
670 closeWriteChannel();
671 else if (stdinChannel.notifier)
672 stdinChannel.notifier->setEnabled(!writeBuffer.isEmpty());
673
674 return writeSucceeded;
675}
676
677bool QProcessPrivate::writeToStdin()
678{
679 const char *data = writeBuffer.readPointer();
680 const qint64 bytesToWrite = writeBuffer.nextDataBlockSize();
681
682 qint64 written = qt_safe_write_nosignal(fd: stdinChannel.pipe[1], data, len: bytesToWrite);
683#if defined QPROCESS_DEBUG
684 qDebug("QProcessPrivate::writeToStdin(), write(%p \"%s\", %lld) == %lld", data,
685 QtDebugUtils::toPrintable(data, bytesToWrite, 16).constData(), bytesToWrite, written);
686 if (written == -1)
687 qDebug("QProcessPrivate::writeToStdin(), failed to write (%ls)", qUtf16Printable(qt_error_string(errno)));
688#endif
689 if (written == -1) {
690 // If the O_NONBLOCK flag is set and If some data can be written without blocking
691 // the process, write() will transfer what it can and return the number of bytes written.
692 // Otherwise, it will return -1 and set errno to EAGAIN
693 if (errno == EAGAIN)
694 return true;
695
696 closeChannel(channel: &stdinChannel);
697 setErrorAndEmit(error: QProcess::WriteError);
698 return false;
699 }
700 writeBuffer.free(bytes: written);
701 if (!emittedBytesWritten && written != 0) {
702 emittedBytesWritten = true;
703 emit q_func()->bytesWritten(bytes: written);
704 emittedBytesWritten = false;
705 }
706 return true;
707}

752bool QProcessPrivate::waitForReadyRead(const QDeadlineTimer &deadline)
753{
754#if defined (QPROCESS_DEBUG)
755 qDebug("QProcessPrivate::waitForReadyRead(%lld)", deadline.remainingTime());
756#endif
757
758 forever {
759 QProcessPoller poller(*this);
760
761 int ret = poller.poll(deadline);
762
763 if (ret < 0) {
764 break;
765 }
766 if (ret == 0) {
767 setError(error: QProcess::Timedout);
768 return false;
769 }
770
771 // This calls QProcessPrivate::tryReadFromChannel(), which returns true
772 // if we emitted readyRead() signal on the current read channel.
773 bool readyReadEmitted = false;
774 if (qt_pollfd_check(pfd: poller.stdoutPipe(), POLLIN) && _q_canReadStandardOutput())
775 readyReadEmitted = true;
776 if (qt_pollfd_check(pfd: poller.stderrPipe(), POLLIN) && _q_canReadStandardError())
777 readyReadEmitted = true;
778
779 if (readyReadEmitted)
780 return true;
781
782 if (qt_pollfd_check(pfd: poller.stdinPipe(), POLLOUT))
783 _q_canWrite();
784
785 // Signals triggered by I/O may have stopped this process:
786 if (processState == QProcess::NotRunning)
787 return false;
788
789 // We do this after checking the pipes, so we cannot reach it as long
790 // as there is any data left to be read from an already dead process.
791 if (qt_pollfd_check(pfd: poller.forkfd(), POLLIN)) {
792 processFinished();
793 return false;
794 }
795 }
796 return false;
797}

800{

803#endif
804
805 while (!writeBuffer.isEmpty()) {

834 return false;
835 }
836 }

838 return false;

881void QProcessPrivate::waitForDeadChild()
882{
883 Q_ASSERT(forkfd != -1);
884
885 // read the process information from our fd
886 forkfd_info info;
887 int ret;
888 EINTR_LOOP(ret, forkfd_wait(forkfd, &info, nullptr));
889
890 exitCode = info.status;
891 crashed = info.code != CLD_EXITED;
892
893 delete stateNotifier;
894 stateNotifier = nullptr;
895
896 EINTR_LOOP(ret, forkfd_close(forkfd));
897 forkfd = -1; // Child is dead, don't try to kill it anymore
898
899#if defined QPROCESS_DEBUG
900 qDebug() << "QProcessPrivate::waitForDeadChild() dead with exitCode"
901 << exitCode << ", crashed?" << crashed;
902#endif
903}

905bool QProcessPrivate::startDetached(qint64 *pid)
906{
907 QByteArray encodedWorkingDirectory = QFile::encodeName(fileName: workingDirectory);
908
909#ifdef PIPE_BUF
910 static_assert(PIPE_BUF >= sizeof(ChildError));
911#else
912 static_assert(_POSIX_PIPE_BUF >= sizeof(ChildError));
913#endif
914 ChildError childStatus = { .code: 0, .function: {} };
915
916 AutoPipe startedPipe, pidPipe;
917 if (!startedPipe || !pidPipe) {
918 setErrorAndEmit(error: QProcess::FailedToStart, description: "pipe: "_L1 + qt_error_string(errno));
919 return false;
920 }
921
922 if (!openChannelsForDetached()) {
923 // openChannel sets the error string
924 closeChannels();
925 return false;
926 }
927
928 const CharPointerList argv(resolveExecutable(program), arguments);
929 const CharPointerList envp(environment.d.constData());
930
931 pid_t childPid = fork();
932 if (childPid == 0) {
933 ::signal(SIGPIPE, SIG_DFL); // reset the signal that we ignored
934 ::setsid();
935
936 qt_safe_close(fd: startedPipe[0]);
937 qt_safe_close(fd: pidPipe[0]);
938
939 auto reportFailed = [&](const char *function) {
940 childStatus.code = errno;
941 strcpy(dest: childStatus.function, src: function);
942 qt_safe_write(fd: startedPipe[1], data: &childStatus, len: sizeof(childStatus));
943 ::_exit(status: 1);
944 };
945
946 if (!encodedWorkingDirectory.isEmpty()) {
947 if (QT_CHDIR(path: encodedWorkingDirectory.constData()) < 0)
948 reportFailed("chdir: ");
949 }
950
951 pid_t doubleForkPid = fork();
952 if (doubleForkPid == 0) {
953 // Render channels configuration.
954 commitChannels();
955
956 if (envp.pointers)
957 qt_safe_execve(filename: argv.pointers[0], argv: argv.pointers.get(), envp: envp.pointers.get());
958 else
959 qt_safe_execv(path: argv.pointers[0], argv: argv.pointers.get());
960
961 reportFailed("execv: ");
962 } else if (doubleForkPid == -1) {
963 reportFailed("fork: ");
964 }
965
966 // success
967 qt_safe_write(fd: pidPipe[1], data: &doubleForkPid, len: sizeof(pid_t));
968 ::_exit(status: 1);
969 }
970
971 int savedErrno = errno;
972 closeChannels();
973
974 if (childPid == -1) {
975 setErrorAndEmit(error: QProcess::FailedToStart, description: "fork: "_L1 + qt_error_string(errorCode: savedErrno));
976 return false;
977 }
978
979 // close the writing ends of the pipes so we can properly get EOFs
980 qt_safe_close(fd: pidPipe[1]);
981 qt_safe_close(fd: startedPipe[1]);
982 pidPipe[1] = startedPipe[1] = -1;
983
984 // This read() will block until we're cleared to proceed. If it returns 0
985 // (EOF), it means the direct child has exited and the grandchild
986 // successfully execve()'d the target process. If it returns any positive
987 // result, it means one of the two children wrote an error result. Negative
988 // values should not happen.
989 ssize_t startResult = qt_safe_read(fd: startedPipe[0], data: &childStatus, maxlen: sizeof(childStatus));
990
991 // reap the intermediate child
992 int result;
993 qt_safe_waitpid(pid: childPid, status: &result, options: 0);
994
995 bool success = (startResult == 0); // nothing written -> no error
996 if (success && pid) {
997 pid_t actualPid;
998 if (qt_safe_read(fd: pidPipe[0], data: &actualPid, maxlen: sizeof(pid_t)) != sizeof(pid_t))
999 actualPid = 0; // this shouldn't happen!
1000 *pid = actualPid;
1001 } else if (!success) {
1002 if (pid)
1003 *pid = -1;
1004 QString msg;
1005 if (startResult == sizeof(childStatus))
1006 msg = QLatin1StringView(childStatus.function) + qt_error_string(errorCode: childStatus.code);
1007 setErrorAndEmit(error: QProcess::FailedToStart, description: msg);
1008 }
1009 return success;
1010}
