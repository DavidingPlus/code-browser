27static void appendOrganizationAndApp(QString &path)
28{
29#ifndef QT_BOOTSTRAPPED
30 const QString org = QCoreApplication::organizationName();
31 if (!org.isEmpty())
32 path += u'/' + org;
33 const QString appName = QCoreApplication::applicationName();
34 if (!appName.isEmpty())
35 path += u'/' + appName;
36#else
37 Q_UNUSED(path);
38#endif
39}

67static QByteArray unixPermissionsText(QFile::Permissions permissions)
68{
69 mode_t perms = 0;
70 if (permissions & QFile::ReadOwner)
71 perms |= S_IRUSR;
72 if (permissions & QFile::WriteOwner)
73 perms |= S_IWUSR;
74 if (permissions & QFile::ExeOwner)
75 perms |= S_IXUSR;
76 if (permissions & QFile::ReadGroup)
77 perms |= S_IRGRP;
78 if (permissions & QFile::WriteGroup)
79 perms |= S_IWGRP;
80 if (permissions & QFile::ExeGroup)
81 perms |= S_IXGRP;
82 if (permissions & QFile::ReadOther)
83 perms |= S_IROTH;
84 if (permissions & QFile::WriteOther)
85 perms |= S_IWOTH;
86 if (permissions & QFile::ExeOther)
87 perms |= S_IXOTH;
88 return '0' + QByteArray::number(perms, base: 8);
89}

91static bool checkXdgRuntimeDir(const QString &xdgRuntimeDir)
92{
93 auto describeMetaData = [](const QFileSystemMetaData &metaData) -> QByteArray {
94 if (!metaData.exists())
95 return "a broken symlink";
96
97 QByteArray description;
98 if (metaData.isLink())
99 description = "a symbolic link to ";
100
101 if (metaData.isFile())
102 description += "a regular file";
103 else if (metaData.isDirectory())
104 description += "a directory";
105 else if (metaData.isSequential())
106 description += "a character device, socket or FIFO";
107 else
108 description += "a block device";
109
110 description += " permissions " + unixPermissionsText(permissions: metaData.permissions());
111
112 return description
113 + " owned by UID " + QByteArray::number(metaData.userId())
114 + " GID " + QByteArray::number(metaData.groupId());
115 };
116
117 // http://standards.freedesktop.org/basedir-spec/latest/
118 const uint myUid = uint(geteuid());
119 const QFile::Permissions wantedPerms = QFile::ReadOwner | QFile::WriteOwner | QFile::ExeOwner;
120 const QFileSystemMetaData::MetaDataFlags statFlags = QFileSystemMetaData::PosixStatFlags
121 | QFileSystemMetaData::LinkType;
122 QFileSystemMetaData metaData;
123 QFileSystemEntry entry(xdgRuntimeDir);
124
125 // Check that the xdgRuntimeDir is a directory by attempting to create it.
126 // A stat() before mkdir() that concluded it doesn't exist is a meaningless
127 // result: we'd race against someone else attempting to create it.
128 // ### QFileSystemEngine::createDirectory cannot take the extra mode argument.
129 if (QT_MKDIR(path: entry.nativeFilePath(), mode: 0700) == 0)
130 return true;
131 if (errno != EEXIST) {
132 qErrnoWarning(msg: "QStandardPaths: error creating runtime directory '%ls'",
133 qUtf16Printable(xdgRuntimeDir));
134 return false;
135 }
136
137 // We use LinkType to force an lstat(), but fillMetaData() still returns error
138 // on broken symlinks.
139 if (!QFileSystemEngine::fillMetaData(entry, data&: metaData, what: statFlags) && !metaData.isLink()) {
140 qErrnoWarning(msg: "QStandardPaths: error obtaining permissions of runtime directory '%ls'",
141 qUtf16Printable(xdgRuntimeDir));
142 return false;
143 }
144
145 // Checks:
146 // - is a directory
147 // - is not a symlink (even is pointing to a directory)
148 if (metaData.isLink() || !metaData.isDirectory()) {
149 qWarning(msg: "QStandardPaths: runtime directory '%ls' is not a directory, but %s",
150 qUtf16Printable(xdgRuntimeDir), describeMetaData(metaData).constData());
151 return false;
152 }
153
154 // - "The directory MUST be owned by the user"
155 if (metaData.userId() != myUid) {
156 qWarning(msg: "QStandardPaths: runtime directory '%ls' is not owned by UID %d, but %s",
157 qUtf16Printable(xdgRuntimeDir), myUid, describeMetaData(metaData).constData());
158 return false;
159 }
160
161 // "and he MUST be the only one having read and write access to it. Its Unix access mode MUST be 0700."
162 if (metaData.permissions() != wantedPerms) {
163 qWarning(msg: "QStandardPaths: wrong permissions on runtime directory %ls, %s instead of %s",
164 qUtf16Printable(xdgRuntimeDir),
165 unixPermissionsText(permissions: metaData.permissions()).constData(),
166 unixPermissionsText(permissions: wantedPerms).constData());
167 return false;
168 }
169
170 return true;
171}

173QString QStandardPaths::writableLocation(StandardLocation type)
174{
175 switch (type) {
176 case HomeLocation:
177 return QDir::homePath();
178 case TempLocation:
179 return QDir::tempPath();
180 case CacheLocation:
181 case GenericCacheLocation:
182 {
183 if (isTestModeEnabled())
184 return QDir::homePath() + "/.qttest/cache"_L1;
185
186 // http://standards.freedesktop.org/basedir-spec/basedir-spec-0.6.html
187 QString xdgCacheHome = QFile::decodeName(localFileName: qgetenv(varName: "XDG_CACHE_HOME"));
188 if (!xdgCacheHome.startsWith(c: u'/'))
189 xdgCacheHome.clear(); // spec says relative paths should be ignored
190
191 if (xdgCacheHome.isEmpty())
192 xdgCacheHome = QDir::homePath() + "/.cache"_L1;
193 if (type == QStandardPaths::CacheLocation)
194 appendOrganizationAndApp(path&: xdgCacheHome);
195 return xdgCacheHome;
196 }
197 case AppDataLocation:
198 case AppLocalDataLocation:
199 case GenericDataLocation:
200 {
201 if (isTestModeEnabled())
202 return QDir::homePath() + "/.qttest/share"_L1;
203
204 QString xdgDataHome = QFile::decodeName(localFileName: qgetenv(varName: "XDG_DATA_HOME"));
205 if (!xdgDataHome.startsWith(c: u'/'))
206 xdgDataHome.clear(); // spec says relative paths should be ignored
207
208 if (xdgDataHome.isEmpty())
209 xdgDataHome = QDir::homePath() + "/.local/share"_L1;
210 if (type == AppDataLocation || type == AppLocalDataLocation)
211 appendOrganizationAndApp(path&: xdgDataHome);
212 return xdgDataHome;
213 }
214 case ConfigLocation:
215 case GenericConfigLocation:
216 case AppConfigLocation:
217 {
218 if (isTestModeEnabled())
219 return QDir::homePath() + "/.qttest/config"_L1;
220
221 // http://standards.freedesktop.org/basedir-spec/latest/
222 QString xdgConfigHome = QFile::decodeName(localFileName: qgetenv(varName: "XDG_CONFIG_HOME"));
223 if (!xdgConfigHome.startsWith(c: u'/'))
224 xdgConfigHome.clear(); // spec says relative paths should be ignored
225
226 if (xdgConfigHome.isEmpty())
227 xdgConfigHome = QDir::homePath() + "/.config"_L1;
228 if (type == AppConfigLocation)
229 appendOrganizationAndApp(path&: xdgConfigHome);
230 return xdgConfigHome;
231 }
232 case RuntimeLocation:
233 {
234 QString xdgRuntimeDir = QFile::decodeName(localFileName: qgetenv(varName: "XDG_RUNTIME_DIR"));
235 if (!xdgRuntimeDir.startsWith(c: u'/'))
236 xdgRuntimeDir.clear(); // spec says relative paths should be ignored
237
238 bool fromEnv = !xdgRuntimeDir.isEmpty();
239 if (xdgRuntimeDir.isEmpty() || !checkXdgRuntimeDir(xdgRuntimeDir)) {
240 // environment variable not set or is set to something unsuitable
241 const uint myUid = uint(geteuid());
242 const QString userName = QFileSystemEngine::resolveUserName(userId: myUid);
243 xdgRuntimeDir = QDir::tempPath() + "/runtime-"_L1 + userName;
244
245 if (!fromEnv) {
246#ifndef Q_OS_WASM
247 qWarning(msg: "QStandardPaths: XDG_RUNTIME_DIR not set, defaulting to '%ls'", qUtf16Printable(xdgRuntimeDir));
248#endif
249 }
250
251 if (!checkXdgRuntimeDir(xdgRuntimeDir))
252 xdgRuntimeDir.clear();
253 }
254
255 return xdgRuntimeDir;
256 }
257 default:
258 break;
259 }
260
261#if QT_CONFIG(regularexpression)
262 // http://www.freedesktop.org/wiki/Software/xdg-user-dirs
263 QString xdgConfigHome = QFile::decodeName(localFileName: qgetenv(varName: "XDG_CONFIG_HOME"));
264 if (!xdgConfigHome.startsWith(c: u'/'))
265 xdgConfigHome.clear(); // spec says relative paths should be ignored
266
267 if (xdgConfigHome.isEmpty())
268 xdgConfigHome = QDir::homePath() + "/.config"_L1;
269 QFile file(xdgConfigHome + "/user-dirs.dirs"_L1);
270 const QLatin1StringView key = xdg_key_name(type);
271 if (!key.isEmpty() && !isTestModeEnabled() && file.open(flags: QIODevice::ReadOnly)) {
272 QTextStream stream(&file);
273 // Only look for lines like: XDG_DESKTOP_DIR="$HOME/Desktop"
274 static const QRegularExpression exp(u"^XDG_(.*)_DIR=(.*)$"_s);
275 QString result;
276 while (!stream.atEnd()) {
277 const QString &line = stream.readLine();
278 QRegularExpressionMatch match = exp.match(subject: line);
279 if (match.hasMatch() && match.capturedView(nth: 1) == key) {
280 QStringView value = match.capturedView(nth: 2);
281 if (value.size() > 2
282 && value.startsWith(c: u'\"')
283 && value.endsWith(c: u'\"'))
284 value = value.mid(pos: 1, n: value.size() - 2);
285 // value can start with $HOME
286 if (value.startsWith(s: "$HOME"_L1))
287 result = QDir::homePath() + value.mid(pos: 5);
288 else
289 result = value.toString();
290 if (result.size() > 1 && result.endsWith(c: u'/'))
291 result.chop(n: 1);
292 }
293 }
294 if (!result.isNull())
295 return result;
296 }
297#endif // QT_CONFIG(regularexpression)

344static QStringList dirsList(const QString &xdgEnvVar)
345{
346 QStringList dirs;
347 // http://standards.freedesktop.org/basedir-spec/latest/
348 // Normalize paths, skip relative paths (the spec says relative paths
349 // should be ignored)
350 for (const auto dir : qTokenize(h: xdgEnvVar, n: u':'))
351 if (dir.startsWith(c: u'/'))
352 dirs.push_back(t: QDir::cleanPath(path: dir.toString()));
353
354 // Remove duplicates from the list, there's no use for duplicated
355 // paths in XDG_DATA_DIRS - if it's not found in the given
356 // directory the first time, it won't be there the second time.
357 // Plus duplicate paths causes problems for example for mimetypes,
358 // where duplicate paths here lead to duplicated mime types returned
359 // for a file, eg "text/plain,text/plain" instead of "text/plain"
360 dirs.removeDuplicates();
361
362 return dirs;
363}

365static QStringList xdgDataDirs()
366{
367 // http://standards.freedesktop.org/basedir-spec/latest/
368 QString xdgDataDirsEnv = QFile::decodeName(localFileName: qgetenv(varName: "XDG_DATA_DIRS"));
369
370 QStringList dirs = dirsList(xdgEnvVar: xdgDataDirsEnv);
371 if (dirs.isEmpty())
372 dirs = QStringList{u"/usr/local/share"_s, u"/usr/share"_s};
373
374 return dirs;
375}

377static QStringList xdgConfigDirs()
378{
379 // http://standards.freedesktop.org/basedir-spec/latest/
380 const QString xdgConfigDirs = QFile::decodeName(localFileName: qgetenv(varName: "XDG_CONFIG_DIRS"));
381
382 QStringList dirs = dirsList(xdgEnvVar: xdgConfigDirs);
383 if (dirs.isEmpty())
384 dirs.push_back(t: u"/etc/xdg"_s);
385
386 return dirs;

389QStringList QStandardPaths::standardLocations(StandardLocation type)
390{
391 QStringList dirs;
392 switch (type) {
393 case ConfigLocation:
394 case GenericConfigLocation:
395 dirs = xdgConfigDirs();
396 break;
397 case AppConfigLocation:
398 dirs = xdgConfigDirs();
399 for (int i = 0; i < dirs.size(); ++i)
400 appendOrganizationAndApp(path&: dirs[i]);
401 break;
402 case GenericDataLocation:
403 dirs = xdgDataDirs();
404 break;
405 case ApplicationsLocation:
406 dirs = xdgDataDirs();
407 for (int i = 0; i < dirs.size(); ++i)
408 dirs[i].append(s: "/applications"_L1);
409 break;
410 case AppDataLocation:
411 case AppLocalDataLocation:
412 dirs = xdgDataDirs();
413 for (int i = 0; i < dirs.size(); ++i)
414 appendOrganizationAndApp(path&: dirs[i]);
415 break;
416 case FontsLocation:
417 dirs += QDir::homePath() + "/.fonts"_L1;
418 dirs += xdgDataDirs();
419 for (int i = 1; i < dirs.size(); ++i)
420 dirs[i].append(s: "/fonts"_L1);
421 break;
422 default:
423 break;
424 }
425 const QString localDir = writableLocation(type);
426 dirs.prepend(t: localDir);
427 return dirs;
428}
