12class Q_CORE_EXPORT QLoggingCategory
13{
14 Q_DISABLE_COPY(QLoggingCategory)
15public:
16 explicit QLoggingCategory(const char *category, QtMsgType severityLevel = QtDebugMsg);
17 ~QLoggingCategory();
18
19 bool isEnabled(QtMsgType type) const;
20 void setEnabled(QtMsgType type, bool enable);
21
22 bool isDebugEnabled() const { return bools.enabledDebug.loadRelaxed(); }
23 bool isInfoEnabled() const { return bools.enabledInfo.loadRelaxed(); }
24 bool isWarningEnabled() const { return bools.enabledWarning.loadRelaxed(); }
25 bool isCriticalEnabled() const { return bools.enabledCritical.loadRelaxed(); }
26
27 const char *categoryName() const { return name; }
28
29 // allows usage of both factory method and variable in qCX macros
30 QLoggingCategory &operator()() { return *this; }
31 const QLoggingCategory &operator()() const { return *this; }

33 static QLoggingCategory *defaultCategory();

35 typedef void (*CategoryFilter)(QLoggingCategory*);

38 static void setFilterRules(const QString &rules);

40private:
41 void init(const char *category, QtMsgType severityLevel);

46 struct AtomicBools {
47 QBasicAtomicInteger<bool> enabledDebug;
48 QBasicAtomicInteger<bool> enabledWarning;

59namespace { // allow different TUs to have different QT_NO_xxx_OUTPUT
60template <QtMsgType Which> struct QLoggingCategoryMacroHolder
61{
62 static const bool IsOutputEnabled;
63 const QLoggingCategory *category = nullptr;
64 bool control = false;
65 explicit QLoggingCategoryMacroHolder(const QLoggingCategory &cat)
66 {
67 if (IsOutputEnabled)
68 init(cat);
69 }
70 explicit QLoggingCategoryMacroHolder(QMessageLogger::CategoryFunction catfunc)
71 {
72 if (IsOutputEnabled)
73 init(cat: catfunc());
74 }
75 void init(const QLoggingCategory &cat) noexcept
76 {
77 category = &cat;
78 // same as:
79 // control = cat.isEnabled(Which);
80 // but without an out-of-line call
81 if constexpr (Which == QtDebugMsg) {
82 control = cat.isDebugEnabled();
83 } else if constexpr (Which == QtInfoMsg) {
84 control = cat.isInfoEnabled();
85 } else if constexpr (Which == QtWarningMsg) {
86 control = cat.isWarningEnabled();
87 } else if constexpr (Which == QtCriticalMsg) {
88 control = cat.isCriticalEnabled();
89 } else if constexpr (Which == QtFatalMsg) {
90 control = true;
91 } else {
92 static_assert(QtPrivate::value_dependent_false<Which>(), "Unknown Qt message type");
93 }
94 }
95 const char *name() const { return category->categoryName(); }
96 explicit operator bool() const { return Q_UNLIKELY(control); }
97};
98
99template <QtMsgType Which> const bool QLoggingCategoryMacroHolder<Which>::IsOutputEnabled = true;
100#if defined(QT_NO_DEBUG_OUTPUT)
101template <> const bool QLoggingCategoryMacroHolder<QtDebugMsg>::IsOutputEnabled = false;
102#endif
103#if defined(QT_NO_INFO_OUTPUT)
104template <> const bool QLoggingCategoryMacroHolder<QtInfoMsg>::IsOutputEnabled = false;
105#endif
106#if defined(QT_NO_WARNING_OUTPUT)
107template <> const bool QLoggingCategoryMacroHolder<QtWarningMsg>::IsOutputEnabled = false;
108#endif
109} // unnamed namespace
