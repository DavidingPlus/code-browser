94static inline bool hasResourcePropertyFlag(const QFileSystemMetaData &data,
95 const QFileSystemEntry &entry,
96 CFStringRef key)
97{
98 QCFString path = CFStringCreateWithFileSystemRepresentation(0,
99 entry.nativeFilePath().constData());
100 if (!path)
101 return false;
102
103 QCFType<CFURLRef> url = CFURLCreateWithFileSystemPath(0, path, kCFURLPOSIXPathStyle,
104 data.hasFlags(QFileSystemMetaData::DirectoryType));
105 if (!url)
106 return false;
107
108 CFBooleanRef value;
109 if (CFURLCopyResourcePropertyForKey(url, key, &value, NULL)) {
110 if (value == kCFBooleanTrue)
111 return true;
112 }
113
114 return false;
115}

117static bool isPackage(const QFileSystemMetaData &data, const QFileSystemEntry &entry)
118{
119 if (!data.isDirectory())
120 return false;
121
122 QFileInfo info(entry.filePath());
123 QString suffix = info.suffix();
124
125 if (suffix.length() > 0) {
126 // First step: is the extension known ?
127 QCFType<CFStringRef> extensionRef = suffix.toCFString();
128 QCFType<CFStringRef> uniformTypeIdentifier = UTTypeCreatePreferredIdentifierForTag(kUTTagClassFilenameExtension, extensionRef, NULL);
129 if (UTTypeConformsTo(uniformTypeIdentifier, kUTTypeBundle))
130 return true;
131
132 // Second step: check if an application knows the package type
133 QCFType<CFStringRef> path = entry.filePath().toCFString();
134 QCFType<CFURLRef> url = CFURLCreateWithFileSystemPath(0, path, kCFURLPOSIXPathStyle, true);
135
136 UInt32 type, creator;
137 // Well created packages have the PkgInfo file
138 if (CFBundleGetPackageInfoInDirectory(url, &type, &creator))
139 return true;
140
141#ifdef Q_OS_MACOS
142 // Find if an application other than Finder claims to know how to handle the package
143 QCFType<CFURLRef> application = LSCopyDefaultApplicationURLForURL(url,
144 kLSRolesEditor | kLSRolesViewer, nullptr);
145
146 if (application) {
147 QCFType<CFBundleRef> bundle = CFBundleCreate(kCFAllocatorDefault, application);
148 CFStringRef identifier = CFBundleGetIdentifier(bundle);
149 QString applicationId = QString::fromCFString(identifier);
150 if (applicationId != "com.apple.finder"_L1)
151 return true;
152 }
153#endif
154 }
155
156 // Third step: check if the directory has the package bit set
157 return hasResourcePropertyFlag(data, entry, kCFURLIsPackageKey);
158}

266inline void QFileSystemMetaData::fillFromStatxBuf(const struct statx &statxBuffer)
267{
268 // Permissions
269 if (statxBuffer.stx_mode & S_IRUSR)
270 entryFlags |= QFileSystemMetaData::OwnerReadPermission;
271 if (statxBuffer.stx_mode & S_IWUSR)
272 entryFlags |= QFileSystemMetaData::OwnerWritePermission;
273 if (statxBuffer.stx_mode & S_IXUSR)
274 entryFlags |= QFileSystemMetaData::OwnerExecutePermission;
275
276 if (statxBuffer.stx_mode & S_IRGRP)
277 entryFlags |= QFileSystemMetaData::GroupReadPermission;
278 if (statxBuffer.stx_mode & S_IWGRP)
279 entryFlags |= QFileSystemMetaData::GroupWritePermission;
280 if (statxBuffer.stx_mode & S_IXGRP)
281 entryFlags |= QFileSystemMetaData::GroupExecutePermission;
282
283 if (statxBuffer.stx_mode & S_IROTH)
284 entryFlags |= QFileSystemMetaData::OtherReadPermission;
285 if (statxBuffer.stx_mode & S_IWOTH)
286 entryFlags |= QFileSystemMetaData::OtherWritePermission;
287 if (statxBuffer.stx_mode & S_IXOTH)
288 entryFlags |= QFileSystemMetaData::OtherExecutePermission;
289
290 // Type
291 if (S_ISLNK(statxBuffer.stx_mode))
292 entryFlags |= QFileSystemMetaData::LinkType;
293 if ((statxBuffer.stx_mode & S_IFMT) == S_IFREG)
294 entryFlags |= QFileSystemMetaData::FileType;
295 else if ((statxBuffer.stx_mode & S_IFMT) == S_IFDIR)
296 entryFlags |= QFileSystemMetaData::DirectoryType;
297 else if ((statxBuffer.stx_mode & S_IFMT) != S_IFBLK)
298 entryFlags |= QFileSystemMetaData::SequentialType;
299
300 // Attributes
301 entryFlags |= QFileSystemMetaData::ExistsAttribute; // inode exists
302 if (statxBuffer.stx_nlink == 0)
303 entryFlags |= QFileSystemMetaData::WasDeletedAttribute;
304 size_ = qint64(statxBuffer.stx_size);
305
306 // Times
307 auto toMSecs = [](struct statx_timestamp ts)
308 {
309 return qint64(ts.tv_sec) * 1000 + (ts.tv_nsec / 1000000);
310 };
311 accessTime_ = toMSecs(statxBuffer.stx_atime);
312 metadataChangeTime_ = toMSecs(statxBuffer.stx_ctime);
313 modificationTime_ = toMSecs(statxBuffer.stx_mtime);
314 if (statxBuffer.stx_mask & STATX_BTIME)
315 birthTime_ = toMSecs(statxBuffer.stx_btime);
316 else
317 birthTime_ = 0;
318
319 userId_ = statxBuffer.stx_uid;
320 groupId_ = statxBuffer.stx_gid;
321}

336//static
337bool QFileSystemEngine::fillMetaData(int fd, QFileSystemMetaData &data)
338{
339 data.entryFlags &= ~QFileSystemMetaData::PosixStatFlags;
340 data.knownFlagsMask |= QFileSystemMetaData::PosixStatFlags;
341
342 struct statx statxBuffer;
343
344 int ret = qt_fstatx(fd, statxBuffer: &statxBuffer);
345 if (ret != -ENOSYS) {
346 if (ret == 0) {
347 data.fillFromStatxBuf(statxBuffer);
348 return true;
349 }
350 return false;
351 }
352
353 QT_STATBUF statBuffer;
354
355 if (QT_FSTAT(fd: fd, statbuf: &statBuffer) == 0) {
356 data.fillFromStatBuf(statBuffer);
357 return true;
358 }
359
360 return false;
361}

382void QFileSystemMetaData::fillFromStatBuf(const QT_STATBUF &statBuffer)
383{
384 // Permissions
385 if (statBuffer.st_mode & S_IRUSR)
386 entryFlags |= QFileSystemMetaData::OwnerReadPermission;
387 if (statBuffer.st_mode & S_IWUSR)
388 entryFlags |= QFileSystemMetaData::OwnerWritePermission;
389 if (statBuffer.st_mode & S_IXUSR)
390 entryFlags |= QFileSystemMetaData::OwnerExecutePermission;
391
392 if (statBuffer.st_mode & S_IRGRP)
393 entryFlags |= QFileSystemMetaData::GroupReadPermission;
394 if (statBuffer.st_mode & S_IWGRP)
395 entryFlags |= QFileSystemMetaData::GroupWritePermission;
396 if (statBuffer.st_mode & S_IXGRP)
397 entryFlags |= QFileSystemMetaData::GroupExecutePermission;
398
399 if (statBuffer.st_mode & S_IROTH)
400 entryFlags |= QFileSystemMetaData::OtherReadPermission;
401 if (statBuffer.st_mode & S_IWOTH)
402 entryFlags |= QFileSystemMetaData::OtherWritePermission;
403 if (statBuffer.st_mode & S_IXOTH)
404 entryFlags |= QFileSystemMetaData::OtherExecutePermission;
405
406 // Type
407 if ((statBuffer.st_mode & S_IFMT) == S_IFREG)
408 entryFlags |= QFileSystemMetaData::FileType;
409 else if ((statBuffer.st_mode & S_IFMT) == S_IFDIR)
410 entryFlags |= QFileSystemMetaData::DirectoryType;
411 else if ((statBuffer.st_mode & S_IFMT) != S_IFBLK)
412 entryFlags |= QFileSystemMetaData::SequentialType;
413
414 // Attributes
415 entryFlags |= QFileSystemMetaData::ExistsAttribute; // inode exists
416 if (statBuffer.st_nlink == 0)
417 entryFlags |= QFileSystemMetaData::WasDeletedAttribute;
418 size_ = statBuffer.st_size;
419#ifdef UF_HIDDEN
420 if (statBuffer.st_flags & UF_HIDDEN) {
421 entryFlags |= QFileSystemMetaData::HiddenAttribute;
422 knownFlagsMask |= QFileSystemMetaData::HiddenAttribute;
423 }
424#endif
425
426 // Times
427 accessTime_ = GetFileTimes::atime(statBuffer, 0);
428 birthTime_ = GetFileTimes::birthtime(statBuffer, 0);
429 metadataChangeTime_ = GetFileTimes::ctime(statBuffer, 0);
430 modificationTime_ = GetFileTimes::mtime(statBuffer, 0);
431
432 userId_ = statBuffer.st_uid;
433 groupId_ = statBuffer.st_gid;
434}

436void QFileSystemMetaData::fillFromDirEnt(const QT_DIRENT &entry)
437{
438#if defined(_DEXTRA_FIRST)
439 knownFlagsMask = {};
440 entryFlags = {};
441 for (dirent_extra *extra = _DEXTRA_FIRST(&entry); _DEXTRA_VALID(extra, &entry);
442 extra = _DEXTRA_NEXT(extra)) {
443 if (extra->d_type == _DTYPE_STAT || extra->d_type == _DTYPE_LSTAT) {
444
445 const struct dirent_extra_stat * const extra_stat =
446 reinterpret_cast<struct dirent_extra_stat *>(extra);
447
448 // Remember whether this was a link or not, this saves an lstat() call later.
449 if (extra->d_type == _DTYPE_LSTAT) {
450 knownFlagsMask |= QFileSystemMetaData::LinkType;
451 if (S_ISLNK(extra_stat->d_stat.st_mode))
452 entryFlags |= QFileSystemMetaData::LinkType;
453 }
454
455 // For symlinks, the extra type _DTYPE_LSTAT doesn't work for filling out the meta data,
456 // as we need the stat() information there, not the lstat() information.
457 // In this case, don't use the extra information.
458 // Unfortunately, readdir() never seems to return extra info of type _DTYPE_STAT, so for
459 // symlinks, we always incur the cost of an extra stat() call later.
460 if (S_ISLNK(extra_stat->d_stat.st_mode) && extra->d_type == _DTYPE_LSTAT)
461 continue;
462
463#if defined(QT_USE_XOPEN_LFS_EXTENSIONS) && defined(QT_LARGEFILE_SUPPORT)
464 // Even with large file support, d_stat is always of type struct stat, not struct stat64,
465 // so it needs to be converted
466 struct stat64 statBuf;
467 fillStat64fromStat32(&statBuf, extra_stat->d_stat);
468 fillFromStatBuf(statBuf);
469#else
470 fillFromStatBuf(extra_stat->d_stat);
471#endif
472 knownFlagsMask |= QFileSystemMetaData::PosixStatFlags;
473 if (!S_ISLNK(extra_stat->d_stat.st_mode)) {
474 knownFlagsMask |= QFileSystemMetaData::ExistsAttribute;
475 entryFlags |= QFileSystemMetaData::ExistsAttribute;
476 }
477 }
478 }
479#elif defined(_DIRENT_HAVE_D_TYPE) || defined(Q_OS_BSD4)
480 // BSD4 includes OS X and iOS
481
482 // ### This will clear all entry flags and knownFlagsMask
483 switch (entry.d_type)
484 {
485 case DT_DIR:
486 knownFlagsMask = QFileSystemMetaData::LinkType
487 | QFileSystemMetaData::FileType
488 | QFileSystemMetaData::DirectoryType
489 | QFileSystemMetaData::SequentialType
490 | QFileSystemMetaData::ExistsAttribute;
491
492 entryFlags = QFileSystemMetaData::DirectoryType
493 | QFileSystemMetaData::ExistsAttribute;
494
495 break;
496
497 case DT_BLK:
498 knownFlagsMask = QFileSystemMetaData::LinkType
499 | QFileSystemMetaData::FileType
500 | QFileSystemMetaData::DirectoryType
501 | QFileSystemMetaData::BundleType
502 | QFileSystemMetaData::AliasType
503 | QFileSystemMetaData::SequentialType
504 | QFileSystemMetaData::ExistsAttribute;
505
506 entryFlags = QFileSystemMetaData::ExistsAttribute;
507
508 break;
509
510 case DT_CHR:
511 case DT_FIFO:
512 case DT_SOCK:
513 // ### System attribute
514 knownFlagsMask = QFileSystemMetaData::LinkType
515 | QFileSystemMetaData::FileType
516 | QFileSystemMetaData::DirectoryType
517 | QFileSystemMetaData::BundleType
518 | QFileSystemMetaData::AliasType
519 | QFileSystemMetaData::SequentialType
520 | QFileSystemMetaData::ExistsAttribute;
521
522 entryFlags = QFileSystemMetaData::SequentialType
523 | QFileSystemMetaData::ExistsAttribute;
524
525 break;
526
527 case DT_LNK:
528 knownFlagsMask = QFileSystemMetaData::LinkType;
529 entryFlags = QFileSystemMetaData::LinkType;
530 break;
531
532 case DT_REG:
533 knownFlagsMask = QFileSystemMetaData::LinkType
534 | QFileSystemMetaData::FileType
535 | QFileSystemMetaData::DirectoryType
536 | QFileSystemMetaData::BundleType
537 | QFileSystemMetaData::SequentialType
538 | QFileSystemMetaData::ExistsAttribute;
539
540 entryFlags = QFileSystemMetaData::FileType
541 | QFileSystemMetaData::ExistsAttribute;
542
543 break;
544
545 case DT_UNKNOWN:
546 default:
547 clear();
548 }
549#else
550 Q_UNUSED(entry);
551#endif
552}

554//static
555QFileSystemEntry QFileSystemEngine::getLinkTarget(const QFileSystemEntry &link, QFileSystemMetaData &data)
556{
557 Q_CHECK_FILE_NAME(link, link);
558
559 QByteArray s = qt_readlink(path: link.nativeFilePath().constData());
560 if (s.size() > 0) {
561 QString ret;
562 if (!data.hasFlags(flags: QFileSystemMetaData::DirectoryType))
563 fillMetaData(entry: link, data, what: QFileSystemMetaData::DirectoryType);
564 if (data.isDirectory() && s[0] != '/') {
565 QDir parent(link.filePath());
566 parent.cdUp();
567 ret = parent.path();
568 if (!ret.isEmpty() && !ret.endsWith(c: u'/'))
569 ret += u'/';
570 }
571 ret += QFile::decodeName(localFileName: s);
572
573 if (!ret.startsWith(c: u'/'))
574 ret.prepend(s: absoluteName(entry: link).path() + u'/');
575 ret = QDir::cleanPath(path: ret);
576 if (ret.size() > 1 && ret.endsWith(c: u'/'))
577 ret.chop(n: 1);
578 return QFileSystemEntry(ret);
579 }
580#if defined(Q_OS_DARWIN)
581 {
582 QCFString path = CFStringCreateWithFileSystemRepresentation(0,
583 QFile::encodeName(QDir::cleanPath(link.filePath())).data());
584 if (!path)
585 return QFileSystemEntry();
586
587 QCFType<CFURLRef> url = CFURLCreateWithFileSystemPath(0, path, kCFURLPOSIXPathStyle,
588 data.hasFlags(QFileSystemMetaData::DirectoryType));
589 if (!url)
590 return QFileSystemEntry();
591
592 QCFType<CFDataRef> bookmarkData = CFURLCreateBookmarkDataFromFile(0, url, NULL);
593 if (!bookmarkData)
594 return QFileSystemEntry();
595
596 QCFType<CFURLRef> resolvedUrl = CFURLCreateByResolvingBookmarkData(0,
597 bookmarkData,
598 (CFURLBookmarkResolutionOptions)(kCFBookmarkResolutionWithoutUIMask
599 | kCFBookmarkResolutionWithoutMountingMask), NULL, NULL, NULL, NULL);
600 if (!resolvedUrl)
601 return QFileSystemEntry();
602
603 QCFString cfstr(CFURLCopyFileSystemPath(resolvedUrl, kCFURLPOSIXPathStyle));
604 if (!cfstr)
605 return QFileSystemEntry();
606
607 return QFileSystemEntry(QString::fromCFString(cfstr));
608 }
609#endif
610 return QFileSystemEntry();
611}

614QFileSystemEntry QFileSystemEngine::canonicalName(const QFileSystemEntry &entry, QFileSystemMetaData &data)
615{
616 Q_CHECK_FILE_NAME(entry, entry);
617
618#if !defined(Q_OS_MAC) && !defined(Q_OS_QNX) && !defined(Q_OS_ANDROID) && !defined(Q_OS_HAIKU) && _POSIX_VERSION < 200809L
619 // realpath(X,0) is not supported
620 Q_UNUSED(data);
621 return QFileSystemEntry(slowCanonicalized(absoluteName(entry).filePath()));
622#else
623# if defined(Q_OS_DARWIN) || defined(Q_OS_ANDROID) || _POSIX_VERSION < 200801L
624 // used to store the result of realpath in case where realpath cannot allocate itself
625 char stack_result[PATH_MAX + 1];
626#else
627 // enables unconditionally passing stack_result below
628 std::nullptr_t stack_result = nullptr;
629# endif
630 auto resolved_path_deleter = [&](char *ptr) {
631 // frees resolved_name if it was allocated by realpath
632# if defined(Q_OS_DARWIN) || defined(Q_OS_ANDROID) || _POSIX_VERSION < 200801L
633 // ptr is either null, or points to stack_result
634 Q_ASSERT(!ptr || ptr == stack_result);
635 return;
636#else
637 free(ptr: ptr);
638# endif
639 };
640 std::unique_ptr<char, decltype (resolved_path_deleter)> resolved_name {nullptr, resolved_path_deleter};
641# if defined(Q_OS_DARWIN) || defined(Q_OS_ANDROID)
642 // On some Android and macOS versions, realpath() will return a path even if
643 // it does not exist. To work around this, we check existence in advance.
644 if (!data.hasFlags(QFileSystemMetaData::ExistsAttribute))
645 fillMetaData(entry, data, QFileSystemMetaData::ExistsAttribute);
646
647 if (!data.exists())
648 errno = ENOENT;
649 else
650 resolved_name.reset(realpath(entry.nativeFilePath().constData(), stack_result));
651# else
652 resolved_name.reset(p: realpath(name: entry.nativeFilePath().constData(), resolved: stack_result));
653# endif
654 if (resolved_name) {
655 data.knownFlagsMask |= QFileSystemMetaData::ExistsAttribute;
656 data.entryFlags |= QFileSystemMetaData::ExistsAttribute;
657 QString canonicalPath = QDir::cleanPath(path: QFile::decodeName(localFileName: resolved_name.get()));
658 return QFileSystemEntry(canonicalPath);
659 } else if (errno == ENOENT || errno == ENOTDIR) { // file doesn't exist
660 data.knownFlagsMask |= QFileSystemMetaData::ExistsAttribute;
661 data.entryFlags &= ~(QFileSystemMetaData::ExistsAttribute);
662 return QFileSystemEntry();
663 }
664 return entry;
665#endif

815//static
816bool QFileSystemEngine::fillMetaData(const QFileSystemEntry &entry, QFileSystemMetaData &data,
817 QFileSystemMetaData::MetaDataFlags what)
818{
819 Q_CHECK_FILE_NAME(entry, false);
820
821#if defined(Q_OS_DARWIN)
822 if (what & QFileSystemMetaData::BundleType) {
823 if (!data.hasFlags(QFileSystemMetaData::DirectoryType))
824 what |= QFileSystemMetaData::DirectoryType;
825 }
826 if (what & QFileSystemMetaData::AliasType)
827 what |= QFileSystemMetaData::LinkType;
828#endif
829#ifdef UF_HIDDEN
830 if (what & QFileSystemMetaData::HiddenAttribute) {
831 // OS X >= 10.5: st_flags & UF_HIDDEN
832 what |= QFileSystemMetaData::PosixStatFlags;
833 }
834#endif // defined(Q_OS_DARWIN)
835
836 // if we're asking for any of the stat(2) flags, then we're getting them all
837 if (what & QFileSystemMetaData::PosixStatFlags)
838 what |= QFileSystemMetaData::PosixStatFlags;
839
840 data.entryFlags &= ~what;
841
842 const QByteArray nativeFilePath = entry.nativeFilePath();
843 int entryErrno = 0; // innocent until proven otherwise
844
845 // first, we may try lstat(2). Possible outcomes:
846 // - success and is a symlink: filesystem entry exists, but we need stat(2)
847 // -> statResult = -1;
848 // - success and is not a symlink: filesystem entry exists and we're done
849 // -> statResult = 0
850 // - failure: really non-existent filesystem entry
851 // -> entryExists = false; statResult = 0;
852 // both stat(2) and lstat(2) may generate a number of different errno
853 // conditions, but of those, the only ones that could happen and the
854 // entry still exist are EACCES, EFAULT, ENOMEM and EOVERFLOW. If we get
855 // EACCES or ENOMEM, then we have no choice on how to proceed, so we may
856 // as well conclude it doesn't exist; EFAULT can't happen and EOVERFLOW
857 // shouldn't happen because we build in _LARGEFIE64.
858 union {
859 QT_STATBUF statBuffer;
860 struct statx statxBuffer;
861 };
862 int statResult = -1;
863 if (what & QFileSystemMetaData::LinkType) {
864 mode_t mode = 0;
865 statResult = qt_lstatx(pathname: nativeFilePath, statxBuffer: &statxBuffer);
866 if (statResult == -ENOSYS) {
867 // use lstst(2)
868 statResult = QT_LSTAT(path: nativeFilePath, statbuf: &statBuffer);
869 if (statResult == 0)
870 mode = statBuffer.st_mode;
871 } else if (statResult == 0) {
872 statResult = 1; // record it was statx(2) that succeeded
873 mode = statxBuffer.stx_mode;
874 }
875
876 if (statResult >= 0) {
877 if (S_ISLNK(mode)) {
878 // it's a symlink, we don't know if the file "exists"
879 data.entryFlags |= QFileSystemMetaData::LinkType;
880 statResult = -1; // force stat(2) below
881 } else {
882 // it's a reagular file and it exists
883 if (statResult)
884 data.fillFromStatxBuf(statxBuffer: statxBuffer);
885 else
886 data.fillFromStatBuf(statBuffer: statBuffer);
887 data.knownFlagsMask |= QFileSystemMetaData::PosixStatFlags
888 | QFileSystemMetaData::ExistsAttribute;
889 data.entryFlags |= QFileSystemMetaData::ExistsAttribute;
890 }
891 } else {
892 // it doesn't exist
893 entryErrno = errno;
894 data.knownFlagsMask |= QFileSystemMetaData::ExistsAttribute;
895 }
896
897 data.knownFlagsMask |= QFileSystemMetaData::LinkType;
898 }
899
900 // second, we try a regular stat(2)
901 if (statResult == -1 && (what & QFileSystemMetaData::PosixStatFlags)) {
902 if (entryErrno == 0 && statResult == -1) {
903 data.entryFlags &= ~QFileSystemMetaData::PosixStatFlags;
904 statResult = qt_statx(pathname: nativeFilePath, statxBuffer: &statxBuffer);
905 if (statResult == -ENOSYS) {
906 // use stat(2)
907 statResult = QT_STAT(path: nativeFilePath, statbuf: &statBuffer);
908 if (statResult == 0)
909 data.fillFromStatBuf(statBuffer: statBuffer);
910 } else if (statResult == 0) {
911 data.fillFromStatxBuf(statxBuffer: statxBuffer);
912 }
913 }
914
915 if (statResult != 0) {
916 entryErrno = errno;
917 data.birthTime_ = 0;
918 data.metadataChangeTime_ = 0;
919 data.modificationTime_ = 0;
920 data.accessTime_ = 0;
921 data.size_ = 0;
922 data.userId_ = (uint) -2;
923 data.groupId_ = (uint) -2;
924 }
925
926 // reset the mask
927 data.knownFlagsMask |= QFileSystemMetaData::PosixStatFlags
928 | QFileSystemMetaData::ExistsAttribute;
929 }
930
931 // third, we try access(2)
932 if (what & (QFileSystemMetaData::UserPermissions | QFileSystemMetaData::ExistsAttribute)) {
933 // calculate user permissions
934 auto checkAccess = [&](QFileSystemMetaData::MetaDataFlag flag, int mode) {
935 if (entryErrno != 0 || (what & flag) == 0)
936 return;
937 if (QT_ACCESS(name: nativeFilePath, type: mode) == 0) {
938 // access ok (and file exists)
939 data.entryFlags |= flag | QFileSystemMetaData::ExistsAttribute;
940 } else if (errno != EACCES && errno != EROFS) {
941 entryErrno = errno;
942 }
943 };
944
945 checkAccess(QFileSystemMetaData::UserReadPermission, R_OK);
946 checkAccess(QFileSystemMetaData::UserWritePermission, W_OK);
947 checkAccess(QFileSystemMetaData::UserExecutePermission, X_OK);
948
949 // if we still haven't found out if the file exists, try F_OK
950 if (entryErrno == 0 && (data.entryFlags & QFileSystemMetaData::ExistsAttribute) == 0) {
951 if (QT_ACCESS(name: nativeFilePath, F_OK) == -1)
952 entryErrno = errno;
953 else
954 data.entryFlags |= QFileSystemMetaData::ExistsAttribute;
955 }
956
957 data.knownFlagsMask |= (what & QFileSystemMetaData::UserPermissions) |
958 QFileSystemMetaData::ExistsAttribute;
959 }
960
961#if defined(Q_OS_DARWIN)
962 if (what & QFileSystemMetaData::AliasType) {
963 if (entryErrno == 0 && hasResourcePropertyFlag(data, entry, kCFURLIsAliasFileKey)) {
964 // kCFURLIsAliasFileKey includes symbolic links, so filter those out
965 if (!(data.entryFlags & QFileSystemMetaData::LinkType))
966 data.entryFlags |= QFileSystemMetaData::AliasType;
967 }
968 data.knownFlagsMask |= QFileSystemMetaData::AliasType;
969 }
970
971 if (what & QFileSystemMetaData::BundleType) {
972 if (entryErrno == 0 && isPackage(data, entry))
973 data.entryFlags |= QFileSystemMetaData::BundleType;
974
975 data.knownFlagsMask |= QFileSystemMetaData::BundleType;
976 }
977#endif
978
979 if (what & QFileSystemMetaData::HiddenAttribute
980 && !data.isHidden()) {
981 QString fileName = entry.fileName();
982 if (fileName.startsWith(c: u'.')
983#if defined(Q_OS_DARWIN)
984 || (entryErrno == 0 && hasResourcePropertyFlag(data, entry, kCFURLIsHiddenKey))
985#endif
986 )
987 data.entryFlags |= QFileSystemMetaData::HiddenAttribute;
988 data.knownFlagsMask |= QFileSystemMetaData::HiddenAttribute;
989 }
990
991 if (entryErrno != 0) {
992 what &= ~QFileSystemMetaData::LinkType; // don't clear link: could be broken symlink
993 data.clearFlags(flags: what);
994 return false;
995 }
996 return true;
997}

999// static
1000bool QFileSystemEngine::cloneFile(int srcfd, int dstfd, const QFileSystemMetaData &knownData)
1001{
1002 QT_STATBUF statBuffer;
1003 if (knownData.hasFlags(flags: QFileSystemMetaData::PosixStatFlags) &&
1004 knownData.isFile()) {
1005 statBuffer.st_mode = S_IFREG;
1006 } else if (knownData.hasFlags(flags: QFileSystemMetaData::PosixStatFlags) &&
1007 knownData.isDirectory()) {
1008 return false; // fcopyfile(3) returns success on directories
1009 } else if (QT_FSTAT(fd: srcfd, statbuf: &statBuffer) == -1) {
1010 return false;
1011 } else if (!S_ISREG((statBuffer.st_mode))) {
1012 // not a regular file, let QFile do the copy
1013 return false;
1014 }
1015
1016#if defined(Q_OS_LINUX)
1017 // first, try FICLONE (only works on regular files and only on certain fs)
1018 if (::ioctl(fd: dstfd, FICLONE, srcfd) == 0)
1019 return true;
1020
1021 // Second, try sendfile (it can send to some special types too).
1022 // sendfile(2) is limited in the kernel to 2G - 4k
1023 const size_t SendfileSize = 0x7ffff000;
1024
1025 ssize_t n = ::sendfile(out_fd: dstfd, in_fd: srcfd, offset: nullptr, count: SendfileSize);
1026 if (n == -1) {
1027 // if we got an error here, give up and try at an upper layer
1028 return false;
1029 }
1030
1031 while (n) {
1032 n = ::sendfile(out_fd: dstfd, in_fd: srcfd, offset: nullptr, count: SendfileSize);
1033 if (n == -1) {
1034 // uh oh, this is probably a real error (like ENOSPC), but we have
1035 // no way to notify QFile of partial success, so just erase any work
1036 // done (hopefully we won't get any errors, because there's nothing
1037 // we can do about them)
1038 n = ftruncate(fd: dstfd, length: 0);
1039 n = lseek(fd: srcfd, offset: 0, SEEK_SET);
1040 n = lseek(fd: dstfd, offset: 0, SEEK_SET);
1041 return false;
1042 }
1043 }
1044
1045 return true;
1046#elif defined(Q_OS_DARWIN)
1047 // try fcopyfile
1048 return fcopyfile(srcfd, dstfd, nullptr, COPYFILE_DATA | COPYFILE_STAT) == 0;
1049#else
1050 Q_UNUSED(dstfd);
1051 return false;
1052#endif
1053}

1055// Note: if \a shouldMkdirFirst is false, we assume the caller did try to mkdir
1056// before calling this function.
1057static bool createDirectoryWithParents(const QByteArray &nativeName, mode_t mode,
1058 bool shouldMkdirFirst = true)
1059{
1060 // helper function to check if a given path is a directory, since mkdir can
1061 // fail if the dir already exists (it may have been created by another
1062 // thread or another process)
1063 const auto isDir = [](const QByteArray &nativeName) {
1064 QT_STATBUF st;
1065 return QT_STAT(path: nativeName.constData(), statbuf: &st) == 0 && (st.st_mode & S_IFMT) == S_IFDIR;
1066 };
1067
1068 if (shouldMkdirFirst && QT_MKDIR(path: nativeName, mode: mode) == 0)
1069 return true;
1070 if (errno == EISDIR)
1071 return true;
1072 if (errno == EEXIST)
1073 return isDir(nativeName);
1074 if (errno != ENOENT)
1075 return false;
1076
1077 // mkdir failed because the parent dir doesn't exist, so try to create it
1078 int slash = nativeName.lastIndexOf(c: '/');
1079 if (slash < 1)
1080 return false;
1081
1082 QByteArray parentNativeName = nativeName.left(len: slash);
1083 if (!createDirectoryWithParents(nativeName: parentNativeName, mode))
1084 return false;
1085
1086 // try again
1087 if (QT_MKDIR(path: nativeName, mode: mode) == 0)
1088 return true;
1089 return errno == EEXIST && isDir(nativeName);
1090}

1156// bootstrapped tools don't need this, and we don't want QStorageInfo
1157#ifndef QT_BOOTSTRAPPED
1158static QString freeDesktopTrashLocation(const QString &sourcePath)
1159{
1160 auto makeTrashDir = [](const QDir &topDir, const QString &trashDir) -> QString {
1161 auto ownerPerms = QFileDevice::ReadOwner
1162 | QFileDevice::WriteOwner
1163 | QFileDevice::ExeOwner;
1164 QString targetDir = topDir.filePath(trashDir);
1165 // deliberately not using mkpath, since we want to fail if topDir doesn't exist
1166 if (topDir.mkdir(trashDir))
1167 QFile::setPermissions(targetDir, ownerPerms);
1168 if (QFileInfo(targetDir).isDir())
1169 return targetDir;
1170 return QString();
1171 };
1172 auto isSticky = [](const QFileInfo &fileInfo) -> bool {
1173 struct stat st;
1174 if (stat(QFile::encodeName(fileInfo.absoluteFilePath()).constData(), &st) == 0)
1175 return st.st_mode & S_ISVTX;
1176
1177 return false;
1178 };
1179
1180 QString trash;
1181 const QStorageInfo sourceStorage(sourcePath);
1182 const QStorageInfo homeStorage(QDir::home());
1183 // We support trashing of files outside the users home partition
1184 if (sourceStorage != homeStorage) {
1185 const auto dotTrash = ".Trash"_L1;
1186 QDir topDir(sourceStorage.rootPath());
1187 /*
1188 Method 1:
1189 "An administrator can create an $topdir/.Trash directory. The permissions on this
1190 directories should permit all users who can trash files at all to write in it;
1191 and the “sticky bit” in the permissions must be set, if the file system supports
1192 it.
1193 When trashing a file from a non-home partition/device, an implementation
1194 (if it supports trashing in top directories) MUST check for the presence
1195 of $topdir/.Trash."
1196 */
1197 const QString userID = QString::number(::getuid());
1198 if (topDir.cd(dotTrash)) {
1199 const QFileInfo trashInfo(topDir.path());
1200
1201 // we MUST check that the sticky bit is set, and that it is not a symlink
1202 if (trashInfo.isSymLink()) {
1203 // we SHOULD report the failed check to the administrator
1204 qCritical("Warning: '%s' is a symlink to '%s'",
1205 trashInfo.absoluteFilePath().toLocal8Bit().constData(),
1206 trashInfo.symLinkTarget().toLatin1().constData());
1207 } else if (!isSticky(trashInfo)) {
1208 // we SHOULD report the failed check to the administrator
1209 qCritical("Warning: '%s' doesn't have sticky bit set!",
1210 trashInfo.absoluteFilePath().toLocal8Bit().constData());
1211 } else if (trashInfo.isDir()) {
1212 /*
1213 "If the directory exists and passes the checks, a subdirectory of the
1214 $topdir/.Trash directory is to be used as the user's trash directory
1215 for this partition/device. The name of this subdirectory is the numeric
1216 identifier of the current user ($topdir/.Trash/$uid).
1217 When trashing a file, if this directory does not exist for the current user,
1218 the implementation MUST immediately create it, without any warnings or
1219 delays for the user."
1220 */
1221 trash = makeTrashDir(topDir, userID);
1222 }
1223 }
1224 /*
1225 Method 2:
1226 "If an $topdir/.Trash directory is absent, an $topdir/.Trash-$uid directory is to be
1227 used as the user's trash directory for this device/partition. [...] When trashing a
1228 file, if an $topdir/.Trash-$uid directory does not exist, the implementation MUST
1229 immediately create it, without any warnings or delays for the user."
1230 */
1231 if (trash.isEmpty()) {
1232 topDir = QDir(sourceStorage.rootPath());
1233 const QString userTrashDir = dotTrash + u'-' + userID;
1234 trash = makeTrashDir(topDir, userTrashDir);
1235 }
1236 }
1237 /*
1238 "If both (1) and (2) fail [...], the implementation MUST either trash the
1239 file into the user's “home trash” or refuse to trash it."
1240
1241 We trash the file into the user's home trash.
1242
1243 "Its name and location are $XDG_DATA_HOME/Trash"; $XDG_DATA_HOME is what
1244 QStandardPaths returns for GenericDataLocation. If that doesn't exist, then
1245 we are not running on a freedesktop.org-compliant environment, and give up.
1246 */
1247 if (trash.isEmpty()) {
1248 QDir topDir = QStandardPaths::writableLocation(QStandardPaths::GenericDataLocation);
1249 trash = makeTrashDir(topDir, "Trash"_L1);
1250 if (!QFileInfo(trash).isDir()) {
1251 qWarning("Unable to establish trash directory in %s",
1252 topDir.path().toLocal8Bit().constData());
1253 }
1254 }
1255
1256 return trash;
1257}

1272 error = QSystemError(ENOENT, QSystemError::StandardLibraryError);

1274 }
1275 const QString sourcePath = sourceInfo.absoluteFilePath();

1277 QDir trashDir(freeDesktopTrashLocation(sourcePath));
1278 if (!trashDir.exists())
1279 return false;
1280 /*
1281 "A trash directory contains two subdirectories, named info and files."
1282 */
1283 const auto filesDir = "files"_L1;
1284 const auto infoDir = "info"_L1;
1285 trashDir.mkdir(filesDir);
1286 int savedErrno = errno;
1287 trashDir.mkdir(infoDir);
1288 if (!savedErrno)
1289 savedErrno = errno;
1290 if (!trashDir.exists(filesDir) || !trashDir.exists(infoDir)) {
1291 error = QSystemError(savedErrno, QSystemError::StandardLibraryError);
1292 return false;
1293 }
1294 /*
1295 "The $trash/files directory contains the files and directories that were trashed.
1296 The names of files in this directory are to be determined by the implementation;
1297 the only limitation is that they must be unique within the directory. Even if a
1298 file with the same name and location gets trashed many times, each subsequent
1299 trashing must not overwrite a previous copy."
1300 */
1301 const QString trashedName = sourceInfo.isDir()
1302 ? QDir(sourcePath).dirName()
1303 : sourceInfo.fileName();
1304 QString uniqueTrashedName = u'/' + trashedName;
1305 QString infoFileName;
1306 int counter = 0;
1307 QFile infoFile;
1308 auto makeUniqueTrashedName = [trashedName, &counter]() -> QString {
1309 return QString::asprintf("/%ls-%04d", qUtf16Printable(trashedName), ++counter);
1310 };
1311 do {
1312 while (QFile::exists(trashDir.filePath(filesDir) + uniqueTrashedName))
1313 uniqueTrashedName = makeUniqueTrashedName();
1314 /*
1315 "The $trash/info directory contains an "information file" for every file and directory
1316 in $trash/files. This file MUST have exactly the same name as the file or directory in
1317 $trash/files, plus the extension ".trashinfo"
1318 [...]
1319 When trashing a file or directory, the implementation MUST create the corresponding
1320 file in $trash/info first. Moreover, it MUST try to do this in an atomic fashion,
1321 so that if two processes try to trash files with the same filename this will result
1322 in two different trash files. On Unix-like systems this is done by generating a
1323 filename, and then opening with O_EXCL. If that succeeds the creation was atomic
1324 (at least on the same machine), if it fails you need to pick another filename."
1325 */
1326 infoFileName = trashDir.filePath(infoDir)
1327 + uniqueTrashedName + ".trashinfo"_L1;
1328 infoFile.setFileName(infoFileName);
1329 if (!infoFile.open(QIODevice::NewOnly | QIODevice::WriteOnly | QIODevice::Text))
1330 uniqueTrashedName = makeUniqueTrashedName();
1331 } while (!infoFile.isOpen());
1332
1333 const QString targetPath = trashDir.filePath(filesDir) + uniqueTrashedName;
1334 const QFileSystemEntry target(targetPath);
1335
1336 QString infoPath;
1337 const QStorageInfo storageInfo(sourcePath);
1338 if (storageInfo.isValid() && storageInfo.rootPath() != rootPath() && storageInfo != QStorageInfo(QDir::home())) {
1339 infoPath = sourcePath.mid(storageInfo.rootPath().length());
1340 if (infoPath.front() == u'/')
1341 infoPath = infoPath.mid(1);
1342 } else {
1343 infoPath = sourcePath;
1344 }

1346 /*
1347 We might fail to rename if source and target are on different file systems.
1348 In that case, we don't try further, i.e. copying and removing the original
1349 is usually not what the user would expect to happen.
1350 */
1351 if (!renameFile(source, target, error)) {
1352 infoFile.close();
1353 infoFile.remove();
1354 return false;
1355 }
1356
1357 QByteArray info =
1358 "[Trash Info]\n"
1359 "Path=" + QUrl::toPercentEncoding(infoPath, "/") + "\n"
1360 "DeletionDate=" + QDateTime::currentDateTime().toString("yyyy-MM-ddThh:mm:ss"_L1).toUtf8()
1361 + "\n";
1362 infoFile.write(info);
1363 infoFile.close();
1364
1365 newLocation = QFileSystemEntry(targetPath);
1366 return true;
1367#endif // QT_BOOTSTRAPPED
1368}

1388//static
1389bool QFileSystemEngine::renameFile(const QFileSystemEntry &source, const QFileSystemEntry &target, QSystemError &error)
1390{
1391 QFileSystemEntry::NativePath srcPath = source.nativeFilePath();
1392 QFileSystemEntry::NativePath tgtPath = target.nativeFilePath();
1393
1394 Q_CHECK_FILE_NAME(srcPath, false);
1395 Q_CHECK_FILE_NAME(tgtPath, false);
1396
1397#if defined(RENAME_NOREPLACE) && QT_CONFIG(renameat2)
1398 if (renameat2(AT_FDCWD, old: srcPath, AT_FDCWD, new: tgtPath, RENAME_NOREPLACE) == 0)
1399 return true;
1400
1401 // We can also get EINVAL for some non-local filesystems.
1402 if (errno != EINVAL) {
1403 error = QSystemError(errno, QSystemError::StandardLibraryError);
1404 return false;
1405 }
1406#endif
1407#if defined(Q_OS_DARWIN) && defined(RENAME_EXCL)
1408 if (renameatx_np(AT_FDCWD, srcPath, AT_FDCWD, tgtPath, RENAME_EXCL) == 0)
1409 return true;
1410 if (errno != ENOTSUP) {
1411 error = QSystemError(errno, QSystemError::StandardLibraryError);
1412 return false;
1413 }
1414#endif
1415
1416 if (SupportsHardlinking && ::link(from: srcPath, to: tgtPath) == 0) {
1417 if (::unlink(name: srcPath) == 0)
1418 return true;
1419
1420 // if we managed to link but can't unlink the source, it's likely
1421 // it's in a directory we don't have write access to; fail the
1422 // renaming instead
1423 int savedErrno = errno;
1424
1425 // this could fail too, but there's nothing we can do about it now
1426 ::unlink(name: tgtPath);
1427
1428 error = QSystemError(savedErrno, QSystemError::StandardLibraryError);
1429 return false;
1430 } else if (!SupportsHardlinking) {
1431 // man 2 link on Linux has:
1432 // EPERM The filesystem containing oldpath and newpath does not
1433 // support the creation of hard links.
1434 errno = EPERM;
1435 }
1436
1437 switch (errno) {
1438 case EACCES:
1439 case EEXIST:
1440 case ENAMETOOLONG:
1441 case ENOENT:
1442 case ENOTDIR:
1443 case EROFS:
1444 case EXDEV:
1445 // accept the error from link(2) (especially EEXIST) and don't retry
1446 break;
1447
1448 default:
1449 // fall back to rename()
1450 // ### Race condition. If a file is moved in after this, it /will/ be
1451 // overwritten.
1452 if (::rename(old: srcPath, new: tgtPath) == 0)
1453 return true;
1454 }
1455
1456 error = QSystemError(errno, QSystemError::StandardLibraryError);
1457 return false;
1458}
