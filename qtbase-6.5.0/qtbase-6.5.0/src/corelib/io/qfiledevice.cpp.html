242/*!
243 Flushes any buffered data to the file. Returns \c true if successful;
244 otherwise returns \c false.
245*/
246bool QFileDevice::flush()
247{
248 Q_D(QFileDevice);
249 if (!d->fileEngine) {
250 qWarning(msg: "QFileDevice::flush: No file engine. Is IODevice open?");
251 return false;
252 }
253
254 if (!d->writeBuffer.isEmpty()) {
255 qint64 size = d->writeBuffer.nextDataBlockSize();
256 qint64 written = d->fileEngine->write(data: d->writeBuffer.readPointer(), len: size);
257 if (written > 0)
258 d->writeBuffer.free(bytes: written);
259 if (written != size) {
260 QFileDevice::FileError err = d->fileEngine->error();
261 if (err == QFileDevice::UnspecifiedError)
262 err = QFileDevice::WriteError;
263 d->setError(err, errStr: d->fileEngine->errorString());
264 return false;
265 }
266 }
267
268 if (!d->fileEngine->flush()) {
269 QFileDevice::FileError err = d->fileEngine->error();
270 if (err == QFileDevice::UnspecifiedError)
271 err = QFileDevice::WriteError;
272 d->setError(err, errStr: d->fileEngine->errorString());
273 return false;
274 }
275 return true;
276}

278/*!
279 Calls QFileDevice::flush() and closes the file. Errors from flush are ignored.
280
281 \sa QIODevice::close()
282*/
283void QFileDevice::close()
284{
285 Q_D(QFileDevice);
286 if (!isOpen())
287 return;
288 bool flushed = flush();
289 QIODevice::close();
290
291 // reset write buffer
292 d->lastWasWrite = false;
293 d->writeBuffer.clear();
294
295 // reset cached size
296 d->cachedSize = 0;
297
298 // keep earlier error from flush
299 if (d->fileEngine->close() && flushed)
300 unsetError();
301 else if (flushed)
302 d->setError(err: d->fileEngine->error(), errStr: d->fileEngine->errorString());
303}

313/*!
314 Returns \c true if the end of the file has been reached; otherwise returns
315 false.
316
317 For regular empty files on Unix (e.g. those in \c /proc), this function
318 returns \c true, since the file system reports that the size of such a file is
319 0. Therefore, you should not depend on atEnd() when reading data from such a
320 file, but rather call read() until no more data can be read.
321*/
322bool QFileDevice::atEnd() const
323{
324 Q_D(const QFileDevice);
325
326 // If there's buffered data left, we're not at the end.
327 if (!d->isBufferEmpty())
328 return false;
329
330 if (!isOpen())
331 return true;
332
333 if (!d->ensureFlushed())
334 return false;
335
336 // If the file engine knows best, say what it says.
337 if (d->fileEngine->supportsExtension(extension: QAbstractFileEngine::AtEndExtension)) {
338 // Check if the file engine supports AtEndExtension, and if it does,
339 // check if the file engine claims to be at the end.
340 return d->fileEngine->atEnd();
341 }
342
343 // if it looks like we are at the end, or if size is not cached,
344 // fall through to bytesAvailable() to make sure.
345 if (pos() < d->cachedSize)
346 return false;
347
348 // Fall back to checking how much is available (will stat files).
349 return bytesAvailable() == 0;
350}

352/*!
353 \fn bool QFileDevice::seek(qint64 pos)
354
355 For random-access devices, this function sets the current position
356 to \a pos, returning true on success, or false if an error occurred.
357 For sequential devices, the default behavior is to do nothing and
358 return false.
359
360 Seeking beyond the end of a file:
361 If the position is beyond the end of a file, then seek() will not
362 immediately extend the file. If a write is performed at this position,
363 then the file will be extended. The content of the file between the
364 previous end of file and the newly written data is UNDEFINED and
365 varies between platforms and file systems.
366*/
367bool QFileDevice::seek(qint64 off)
368{
369 Q_D(QFileDevice);
370 if (!isOpen()) {
371 qWarning(msg: "QFileDevice::seek: IODevice is not open");
372 return false;
373 }
374
375 if (!d->ensureFlushed())
376 return false;
377
378 if (!d->fileEngine->seek(pos: off) || !QIODevice::seek(pos: off)) {
379 QFileDevice::FileError err = d->fileEngine->error();
380 if (err == QFileDevice::UnspecifiedError)
381 err = QFileDevice::PositionError;
382 d->setError(err, errStr: d->fileEngine->errorString());
383 return false;
384 }
385 unsetError();
386 return true;
387}

389/*!
390 \reimp
391*/
392qint64 QFileDevice::readLineData(char *data, qint64 maxlen)
393{
394 Q_D(QFileDevice);
395 if (!d->ensureFlushed())
396 return -1;
397
398 qint64 read;
399 if (d->fileEngine->supportsExtension(extension: QAbstractFileEngine::FastReadLineExtension)) {
400 read = d->fileEngine->readLine(data, maxlen);
401 } else {
402 // Fall back to QIODevice's readLine implementation if the engine
403 // cannot do it faster.
404 read = QIODevice::readLineData(data, maxlen);
405 }
406
407 if (read < maxlen) {
408 // failed to read all requested, may be at the end of file, stop caching size so that it's rechecked
409 d->cachedSize = 0;
410 }
411
412 return read;
413}

415/*!
416 \reimp
417*/
418qint64 QFileDevice::readData(char *data, qint64 len)
419{
420 Q_D(QFileDevice);
421 if (!len)
422 return 0;
423 unsetError();
424 if (!d->ensureFlushed())
425 return -1;
426
427 const qint64 read = d->fileEngine->read(data, maxlen: len);
428 if (read < 0) {
429 QFileDevice::FileError err = d->fileEngine->error();
430 if (err == QFileDevice::UnspecifiedError)
431 err = QFileDevice::ReadError;
432 d->setError(err, errStr: d->fileEngine->errorString());
433 }
434
435 if (read < len) {
436 // failed to read all requested, may be at the end of file, stop caching size so that it's rechecked
437 d->cachedSize = 0;
438 }
439
440 return read;
441}

443/*!
444 \internal
445*/
446bool QFileDevicePrivate::putCharHelper(char c)
447{
448#ifdef QT_NO_QOBJECT
449 return QIODevicePrivate::putCharHelper(c);
450#else
451
452 // Cutoff for code that doesn't only touch the buffer.
453 qint64 writeBufferSize = writeBuffer.size();
454 if ((openMode & QIODevice::Unbuffered) || writeBufferSize + 1 >= writeBufferChunkSize
455#ifdef Q_OS_WIN
456 || ((openMode & QIODevice::Text) && c == '\n'
457 && writeBufferSize + 2 >= writeBufferChunkSize)
458#endif
459 ) {
460 return QIODevicePrivate::putCharHelper(c);
461 }
462
463 if (!(openMode & QIODevice::WriteOnly)) {
464 if (openMode == QIODevice::NotOpen)
465 qWarning("QIODevice::putChar: Closed device");
466 else
467 qWarning("QIODevice::putChar: ReadOnly device");
468 return false;
469 }
470
471 // Make sure the device is positioned correctly.
472 const bool sequential = isSequential();
473 if (pos != devicePos && !sequential && !q_func()->seek(pos))
474 return false;
475
476 lastWasWrite = true;
477
478 int len = 1;
479#ifdef Q_OS_WIN
480 if ((openMode & QIODevice::Text) && c == '\n') {
481 ++len;
482 *writeBuffer.reserve(1) = '\r';
483 }
484#endif
485
486 // Write to buffer.
487 *writeBuffer.reserve(1) = c;
488
489 if (!sequential) {
490 pos += len;
491 devicePos += len;
492 if (!buffer.isEmpty())
493 buffer.skip(len);
494 }
495
496 return true;
497#endif
498}

500/*!
501 \reimp
502*/
503qint64 QFileDevice::writeData(const char *data, qint64 len)
504{
505 Q_D(QFileDevice);
506 unsetError();
507 d->lastWasWrite = true;
508 bool buffered = !(d->openMode & Unbuffered);
509
510 // Flush buffered data if this read will overflow.
511 if (buffered && (d->writeBuffer.size() + len) > d->writeBufferChunkSize) {
512 if (!flush())
513 return -1;
514 }
515
516 // Write directly to the engine if the block size is larger than
517 // the write buffer size.
518 if (!buffered || len > d->writeBufferChunkSize) {
519 const qint64 ret = d->fileEngine->write(data, len);
520 if (ret < 0) {
521 QFileDevice::FileError err = d->fileEngine->error();
522 if (err == QFileDevice::UnspecifiedError)
523 err = QFileDevice::WriteError;
524 d->setError(err, errStr: d->fileEngine->errorString());
525 }
526 return ret;
527 }
528
529 // Write to the buffer.
530 d->writeBuffer.append(data, size: len);
531 return len;
532}

576/*!
577 Sets the file size (in bytes) \a sz. Returns \c true if the
578 resize succeeds; false otherwise. If \a sz is larger than the file
579 currently is, the new bytes will be set to 0; if \a sz is smaller, the
580 file is simply truncated.
581
582 \warning This function can fail if the file doesn't exist.
583
584 \sa size()
585*/
586bool QFileDevice::resize(qint64 sz)
587{
588 Q_D(QFileDevice);
589 if (!d->ensureFlushed())
590 return false;
591 d->engine();
592 if (isOpen() && d->fileEngine->pos() > sz)
593 seek(off: sz);
594 if (d->fileEngine->setSize(sz)) {
595 unsetError();
596 d->cachedSize = sz;
597 return true;
598 }
599 d->cachedSize = 0;
600 d->setError(err: QFile::ResizeError, errStr: d->fileEngine->errorString());
601 return false;

641
642 This enum describes special options that may be used by the map()

644
645 \value NoOptions No options.
646 \value MapPrivateOption The mapped memory will be private, so any
647 modifications will not be visible to other processes and will not

654/*!
655 Maps \a size bytes of the file into memory starting at \a offset. A file
656 should be open for a map to succeed but the file does not need to stay
657 open after the memory has been mapped. When the QFile is destroyed
658 or a new file is opened with this object, any maps that have not been
659 unmapped will automatically be unmapped.
660
661 The mapping will have the same open mode as the file (read and/or write),
662 except when using MapPrivateOption, in which case it is always possible
663 to write to the mapped memory.
664
665 Any mapping options can be passed through \a flags.
666
667 Returns a pointer to the memory or \nullptr if there is an error.
668
669 \sa unmap()
670 */
671uchar *QFileDevice::map(qint64 offset, qint64 size, MemoryMapFlags flags)
672{
673 Q_D(QFileDevice);
674 if (d->engine()
675 && d->fileEngine->supportsExtension(extension: QAbstractFileEngine::MapExtension)) {
676 unsetError();
677 uchar *address = d->fileEngine->map(offset, size, flags);
678 if (address == nullptr)
679 d->setError(err: d->fileEngine->error(), errStr: d->fileEngine->errorString());
680 return address;
681 }
682 return nullptr;
683}
