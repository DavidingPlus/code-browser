42/*!
43 \class QLockFile
44 \inmodule QtCore
45 \brief The QLockFile class provides locking between processes using a file.
46 \since 5.1
47
48 A lock file can be used to prevent multiple processes from accessing concurrently
49 the same resource. For instance, a configuration file on disk, or a socket, a port,
50 a region of shared memory...
51
52 Serialization is only guaranteed if all processes that access the shared resource
53 use QLockFile, with the same file path.
54
55 QLockFile supports two use cases:
56 to protect a resource for a short-term operation (e.g. verifying if a configuration
57 file has changed before saving new settings), and for long-lived protection of a
58 resource (e.g. a document opened by a user in an editor) for an indefinite amount of time.
59
60 When protecting for a short-term operation, it is acceptable to call lock() and wait
61 until any running operation finishes.
62 When protecting a resource over a long time, however, the application should always
63 call setStaleLockTime(0) and then tryLock() with a short timeout, in order to
64 warn the user that the resource is locked.
65
66 If the process holding the lock crashes, the lock file stays on disk and can prevent
67 any other process from accessing the shared resource, ever. For this reason, QLockFile
68 tries to detect such a "stale" lock file, based on the process ID written into the file.
69 To cover the situation that the process ID got reused meanwhile, the current process name is
70 compared to the name of the process that corresponds to the process ID from the lock file.
71 If the process names differ, the lock file is considered stale.
72 Additionally, the last modification time of the lock file (30s by default, for the use case of a
73 short-lived operation) is taken into account.
74 If the lock file is found to be stale, it will be deleted.
75
76 For the use case of protecting a resource over a long time, you should therefore call
77 setStaleLockTime(0), and when tryLock() returns LockFailedError, inform the user
78 that the document is locked, possibly using getLockInfo() for more details.
79
80 \note On Windows, this class has problems detecting a stale lock if the
81 machine's hostname contains characters outside the US-ASCII character set.
82*/

127/*!
128 Sets \a staleLockTime to be the time in milliseconds after which
129 a lock file is considered stale.
130 The default value is 30000, i.e. 30 seconds.
131 If your application typically keeps the file locked for more than 30 seconds
132 (for instance while saving megabytes of data for 2 minutes), you should set
133 a bigger value using setStaleLockTime().
134
135 The value of \a staleLockTime is used by lock() and tryLock() in order
136 to determine when an existing lock file is considered stale, i.e. left over
137 by a crashed process. This is useful for the case where the PID got reused
138 meanwhile, so one way to detect a stale lock file is by the fact that
139 it has been around for a long time.
140
141 \sa staleLockTime()
142*/
143void QLockFile::setStaleLockTime(int staleLockTime)
144{
145 Q_D(QLockFile);
146 d->staleLockTime = staleLockTime;

202/*!
203 Creates the lock file.
204
205 If another process (or another thread) has created the lock file already,
206 this function will block until that process (or thread) releases it.
207
208 Calling this function multiple times on the same lock from the same
209 thread without unlocking first is not allowed. This function will
210 \e dead-lock when the file is locked recursively.
211
212 Returns \c true if the lock was acquired, false if it could not be acquired
213 due to an unrecoverable error, such as no permissions in the parent directory.
214
215 \sa unlock(), tryLock()
216*/
217bool QLockFile::lock()
218{
219 return tryLock(timeout: -1);
220}

222/*!
223 Attempts to create the lock file. This function returns \c true if the
224 lock was obtained; otherwise it returns \c false. If another process (or
225 another thread) has created the lock file already, this function will
226 wait for at most \a timeout milliseconds for the lock file to become
227 available.
228
229 Note: Passing a negative number as the \a timeout is equivalent to
230 calling lock(), i.e. this function will wait forever until the lock
231 file can be locked if \a timeout is negative.
232
233 If the lock was obtained, it must be released with unlock()
234 before another process (or thread) can successfully lock it.
235
236 Calling this function multiple times on the same lock from the same
237 thread without unlocking first is not allowed, this function will
238 \e always return false when attempting to lock the file recursively.
239
240 \sa lock(), unlock()
241*/
242bool QLockFile::tryLock(int timeout)
243{
244 Q_D(QLockFile);
245 QDeadlineTimer timer(qMax(a: timeout, b: -1)); // QDT only takes -1 as "forever"
246 int sleepTime = 100;
247 forever {
248 d->lockError = d->tryLock_sys();
249 switch (d->lockError) {
250 case NoError:
251 d->isLocked = true;
252 return true;
253 case PermissionError:
254 case UnknownError:
255 return false;
256 case LockFailedError:
257 if (!d->isLocked && d->isApparentlyStale()) {
258 if (Q_UNLIKELY(QFileInfo(d->fileName).lastModified() > QDateTime::currentDateTimeUtc()))
259 qInfo(msg: "QLockFile: Lock file '%ls' has a modification time in the future", qUtf16Printable(d->fileName));
260 // Stale lock from another thread/process
261 // Ensure two processes don't remove it at the same time
262 QLockFile rmlock(d->fileName + ".rmlock"_L1);
263 if (rmlock.tryLock()) {
264 if (d->isApparentlyStale() && d->removeStaleLock())
265 continue;
266 }
267 }
268 break;
269 }
270
271 int remainingTime = timer.remainingTime();
272 if (remainingTime == 0)
273 return false;
274 else if (uint(sleepTime) > uint(remainingTime))
275 sleepTime = remainingTime;
276
277 QThread::msleep(sleepTime);
278 if (sleepTime < 5 * 1000)
279 sleepTime *= 2;
280 }
281 // not reached
282 return false;
283}

313/*!
314 Retrieves information about the current owner of the lock file.
315
316 If tryLock() returns \c false, and error() returns LockFailedError,
317 this function can be called to find out more information about the existing
318 lock file:
319 \list
320 \li the PID of the application (returned in \a pid)
321 \li the \a hostname it's running on (useful in case of networked filesystems),
322 \li the name of the application which created it (returned in \a appname),
323 \endlist
324
325 Note that tryLock() automatically deleted the file if there is no
326 running application with this PID, so LockFailedError can only happen if there is
327 an application with this PID (it could be unrelated though).
328
329 This can be used to inform users about the existing lock file and give them
330 the choice to delete it. After removing the file using removeStaleLockFile(),
331 the application can call tryLock() again.
332
333 This function returns \c true if the information could be successfully retrieved, false
334 if the lock file doesn't exist or doesn't contain the expected data.
335 This can happen if the lock file was deleted between the time where tryLock() failed
336 and the call to this function. Simply call tryLock() again if this happens.
337*/
338bool QLockFile::getLockInfo(qint64 *pid, QString *hostname, QString *appname) const
339{
340 Q_D(const QLockFile);
341 LockFileInfo info;
342 if (!getLockInfo_helper(fileName: d->fileName, info: &info))
343 return false;
344 if (pid)
345 *pid = info.pid;
346 if (hostname)
347 *hostname = info.hostname;
348 if (appname)
349 *appname = info.appname;
350 return true;
351}

353QByteArray QLockFilePrivate::lockFileContents() const
354{
355 // Use operator% from the fast builder to avoid multiple memory allocations.
356 return QByteArray::number(QCoreApplication::applicationPid()) % '\n'
357 % processNameByPid(pid: QCoreApplication::applicationPid()).toUtf8() % '\n'
358 % machineName().toUtf8() % '\n'
359 % QSysInfo::machineUniqueId() % '\n'
360 % QSysInfo::bootUniqueId() % '\n';
361}

363static bool getLockInfo_helper(const QString &fileName, LockFileInfo *info)
364{
365 QFile reader(fileName);
366 if (!reader.open(flags: QIODevice::ReadOnly | QIODevice::Text))
367 return false;
368
369 QByteArray pidLine = reader.readLine();
370 pidLine.chop(n: 1);
371 if (pidLine.isEmpty())
372 return false;
373 QByteArray appNameLine = reader.readLine();
374 appNameLine.chop(n: 1);
375 QByteArray hostNameLine = reader.readLine();
376 hostNameLine.chop(n: 1);
377
378 // prior to Qt 5.10, only the lines above were recorded
379 QByteArray hostId = reader.readLine();
380 hostId.chop(n: 1);
381 QByteArray bootId = reader.readLine();
382 bootId.chop(n: 1);
383
384 bool ok;
385 info->appname = QString::fromUtf8(ba: appNameLine);
386 info->hostname = QString::fromUtf8(ba: hostNameLine);
387 info->hostid = hostId;
388 info->bootid = bootId;
389 info->pid = pidLine.toLongLong(ok: &ok);
390 return ok && info->pid > 0;
391}

393bool QLockFilePrivate::isApparentlyStale() const
394{
395 LockFileInfo info;
396 if (getLockInfo_helper(fileName, info: &info)) {
397 bool sameHost = info.hostname.isEmpty() || info.hostname == machineName();
398 if (!info.hostid.isEmpty()) {
399 // Override with the host ID, if we know it.
400 QByteArray ourHostId = QSysInfo::machineUniqueId();
401 if (!ourHostId.isEmpty())
402 sameHost = (ourHostId == info.hostid);
403 }
404
405 if (sameHost) {
406 if (!info.bootid.isEmpty()) {
407 // If we've rebooted, then the lock is definitely stale.
408 if (info.bootid != QSysInfo::bootUniqueId())
409 return true;
410 }
411 if (!isProcessRunning(pid: info.pid, appname: info.appname))
412 return true;
413 }
414 }
415
416 const qint64 age = QFileInfo(fileName).lastModified().msecsTo(QDateTime::currentDateTimeUtc());
417 return staleLockTime > 0 && qAbs(t: age) > staleLockTime;
418}

420/*!
421 Attempts to forcefully remove an existing lock file.
422
423 Calling this is not recommended when protecting a short-lived operation: QLockFile
424 already takes care of removing lock files after they are older than staleLockTime().
425
426 This method should only be called when protecting a resource for a long time, i.e.
427 with staleLockTime(0), and after tryLock() returned LockFailedError, and the user
428 agreed on removing the lock file.
429
430 Returns \c true on success, false if the lock file couldn't be removed. This happens
431 on Windows, when the application owning the lock is still running.
432*/
433bool QLockFile::removeStaleLockFile()
434{
435 Q_D(QLockFile);
436 if (d->isLocked) {
437 qWarning(msg: "removeStaleLockFile can only be called when not holding the lock");
438 return false;
439 }
440 return d->removeStaleLock();
441}
