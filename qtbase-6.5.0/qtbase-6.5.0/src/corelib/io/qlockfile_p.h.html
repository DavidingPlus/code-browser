67 static int getLockFileHandle(QLockFile *f)
68 {
69 int fd;
70#ifdef Q_OS_WIN
71 // Use of this function on Windows WILL leak a file descriptor.
72 fd = _open_osfhandle(intptr_t(f->d_func()->fileHandle), 0);
73#else
74 fd = f->d_func()->fileHandle;
75#endif
76 QT_LSEEK(fd: fd, offset: 0, SEEK_SET);
77 return fd;
78 }
79};
