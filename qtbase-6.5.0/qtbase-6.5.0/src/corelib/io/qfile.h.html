31inline QString fromFilesystemPath(const std::filesystem::path &path)
32{
33#ifdef Q_OS_WIN
34 return QString::fromStdWString(path.native());
35#else
36 return QString::fromStdString(s: path.native());
37#endif
38}
39
40inline std::filesystem::path toFilesystemPath(const QString &path)
41{
42 return std::filesystem::path(reinterpret_cast<const char16_t *>(path.cbegin()),
43 reinterpret_cast<const char16_t *>(path.cend()));
44}
45
46// Both std::filesystem::path and QString (without QT_NO_CAST_FROM_ASCII) can be implicitly
47// constructed from string literals so we force the std::fs::path parameter to only
48// accept std::fs::path with no implicit conversions.
49template<typename T>
50using ForceFilesystemPath = typename std::enable_if_t<std::is_same_v<std::filesystem::path, T>, int>;
51}
52#endif // QT_CONFIG(cxx17_filesystem)
