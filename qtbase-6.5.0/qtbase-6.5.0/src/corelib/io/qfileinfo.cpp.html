17QString QFileInfoPrivate::getFileName(QAbstractFileEngine::FileName name) const
18{
19 if (cache_enabled && !fileNames[(int)name].isNull())
20 return fileNames[(int)name];
21
22 QString ret;
23 if (fileEngine == nullptr) { // local file; use the QFileSystemEngine directly
24 switch (name) {
25 case QAbstractFileEngine::CanonicalName:
26 case QAbstractFileEngine::CanonicalPathName: {
27 QFileSystemEntry entry = QFileSystemEngine::canonicalName(entry: fileEntry, data&: metaData);
28 if (cache_enabled) { // be smart and store both
29 fileNames[QAbstractFileEngine::CanonicalName] = entry.filePath();
30 fileNames[QAbstractFileEngine::CanonicalPathName] = entry.path();
31 }
32 if (name == QAbstractFileEngine::CanonicalName)
33 ret = entry.filePath();
34 else
35 ret = entry.path();
36 break;
37 }
38 case QAbstractFileEngine::AbsoluteLinkTarget:
39 ret = QFileSystemEngine::getLinkTarget(link: fileEntry, data&: metaData).filePath();
40 break;
41 case QAbstractFileEngine::JunctionName:
42 ret = QFileSystemEngine::getJunctionTarget(link: fileEntry, data&: metaData).filePath();
43 break;
44 case QAbstractFileEngine::BundleName:
45 ret = QFileSystemEngine::bundleName(fileEntry);
46 break;
47 case QAbstractFileEngine::AbsoluteName:
48 case QAbstractFileEngine::AbsolutePathName: {
49 QFileSystemEntry entry = QFileSystemEngine::absoluteName(entry: fileEntry);
50 if (cache_enabled) { // be smart and store both
51 fileNames[QAbstractFileEngine::AbsoluteName] = entry.filePath();
52 fileNames[QAbstractFileEngine::AbsolutePathName] = entry.path();
53 }
54 if (name == QAbstractFileEngine::AbsoluteName)
55 ret = entry.filePath();
56 else
57 ret = entry.path();
58 break;
59 }
60 default: break;
61 }
62 } else {
63 ret = fileEngine->fileName(file: name);
64 }
65 if (ret.isNull())
66 ret = ""_L1;
67 if (cache_enabled)
68 fileNames[(int)name] = ret;
69 return ret;
70}

96uint QFileInfoPrivate::getFileFlags(QAbstractFileEngine::FileFlags request) const
97{
98 Q_ASSERT(fileEngine); // should never be called when using the native FS
99 // We split the testing into tests for for LinkType, BundleType, PermsMask
100 // and the rest.
101 // Tests for file permissions on Windows can be slow, especially on network
102 // paths and NTFS drives.
103 // In order to determine if a file is a symlink or not, we have to lstat().
104 // If we're not interested in that information, we might as well avoid one
105 // extra syscall. Bundle detecton on Mac can be slow, especially on network
106 // paths, so we separate out that as well.
107
108 QAbstractFileEngine::FileFlags req;
109 uint cachedFlags = 0;
110
111 if (request & (QAbstractFileEngine::FlagsMask | QAbstractFileEngine::TypesMask)) {
112 if (!getCachedFlag(c: CachedFileFlags)) {
113 req |= QAbstractFileEngine::FlagsMask;
114 req |= QAbstractFileEngine::TypesMask;
115 req &= (~QAbstractFileEngine::LinkType);
116 req &= (~QAbstractFileEngine::BundleType);
117
118 cachedFlags |= CachedFileFlags;
119 }
120
121 if (request & QAbstractFileEngine::LinkType) {
122 if (!getCachedFlag(c: CachedLinkTypeFlag)) {
123 req |= QAbstractFileEngine::LinkType;
124 cachedFlags |= CachedLinkTypeFlag;
125 }
126 }
127
128 if (request & QAbstractFileEngine::BundleType) {
129 if (!getCachedFlag(c: CachedBundleTypeFlag)) {
130 req |= QAbstractFileEngine::BundleType;
131 cachedFlags |= CachedBundleTypeFlag;
132 }
133 }
134 }
135
136 if (request & QAbstractFileEngine::PermsMask) {
137 if (!getCachedFlag(c: CachedPerms)) {
138 req |= QAbstractFileEngine::PermsMask;
139 cachedFlags |= CachedPerms;
140 }
141 }
142
143 if (req) {
144 if (cache_enabled)
145 req &= (~QAbstractFileEngine::Refresh);
146 else
147 req |= QAbstractFileEngine::Refresh;
148
149 QAbstractFileEngine::FileFlags flags = fileEngine->fileFlags(type: req);
150 fileFlags |= uint(flags.toInt());
151 setCachedFlag(cachedFlags);
152 }
153
154 return fileFlags & request.toInt();
155}

157QDateTime &QFileInfoPrivate::getFileTime(QAbstractFileEngine::FileTime request) const
158{
159 Q_ASSERT(fileEngine); // should never be called when using the native FS
160 if (!cache_enabled)
161 clearFlags();
162
163 uint cf = 0;
164 switch (request) {
165 case QAbstractFileEngine::AccessTime:
166 cf = CachedATime;
167 break;
168 case QAbstractFileEngine::BirthTime:
169 cf = CachedBTime;
170 break;
171 case QAbstractFileEngine::MetadataChangeTime:
172 cf = CachedMCTime;
173 break;
174 case QAbstractFileEngine::ModificationTime:
175 cf = CachedMTime;
176 break;
177 }
178
179 if (!getCachedFlag(c: cf)) {
180 fileTimes[request] = fileEngine->fileTime(time: request);
181 setCachedFlag(cf);
182 }
183 return fileTimes[request];
184}

188/*!
189 \class QFileInfo
190 \inmodule QtCore
191 \reentrant
192 \brief The QFileInfo class provides system-independent file information.
193
194 \ingroup io
195 \ingroup shared
196
197 QFileInfo provides information about a file's name and position
198 (path) in the file system, its access rights and whether it is a
199 directory or symbolic link, etc. The file's size and last
200 modified/read times are also available. QFileInfo can also be
201 used to obtain information about a Qt \l{resource
202 system}{resource}.
203
204 A QFileInfo can point to a file with either a relative or an
205 absolute file path. Absolute file paths begin with the directory
206 separator "/" (or with a drive specification on Windows). Relative
207 file names begin with a directory name or a file name and specify
208 a path relative to the current working directory. An example of an
209 absolute path is the string "/tmp/quartz". A relative path might
210 look like "src/fatlib". You can use the function isRelative() to
211 check whether a QFileInfo is using a relative or an absolute file
212 path. You can call the function makeAbsolute() to convert a
213 relative QFileInfo's path to an absolute path.
214
215 \note Paths starting with a colon (\e{:}) are always considered
216 absolute, as they denote a QResource.
217
218 The file that the QFileInfo works on is set in the constructor or
219 later with setFile(). Use exists() to see if the file exists and
220 size() to get its size.
221
222 The file's type is obtained with isFile(), isDir() and
223 isSymLink(). The symLinkTarget() function provides the name of the file
224 the symlink points to.
225
226 On Unix (including \macos and iOS), the property getter functions in this
227 class return the properties such as times and size of the target file, not
228 the symlink, because Unix handles symlinks transparently. Opening a symlink
229 using QFile effectively opens the link's target. For example:
230
231 \snippet code/src_corelib_io_qfileinfo.cpp 0
232
233 On Windows, shortcuts (\c .lnk files) are currently treated as symlinks. As
234 on Unix systems, the property getters return the size of the targeted file,
235 not the \c .lnk file itself. This behavior is deprecated and will likely be
236 removed in a future version of Qt, after which \c .lnk files will be treated
237 as regular files.
238
239 \snippet code/src_corelib_io_qfileinfo.cpp 1
240
241 Elements of the file's name can be extracted with path() and
242 fileName(). The fileName()'s parts can be extracted with
243 baseName(), suffix() or completeSuffix(). QFileInfo objects to
244 directories created by Qt classes will not have a trailing file
245 separator. If you wish to use trailing separators in your own file
246 info objects, just append one to the file name given to the constructors
247 or setFile().
248
249 The file's dates are returned by birthTime(), lastModified(), lastRead() and
250 fileTime(). Information about the file's access permissions is
251 obtained with isReadable(), isWritable() and isExecutable(). The
252 file's ownership is available from owner(), ownerId(), group() and
253 groupId(). You can examine a file's permissions and ownership in a
254 single statement using the permission() function.
255
256 \target NTFS permissions
257 \note On NTFS file systems, ownership and permissions checking is
258 disabled by default for performance reasons. To enable it,
259 include the following line:
260
261 \snippet ntfsp.cpp 0
262
263 Permission checking is then turned on and off by incrementing and
264 decrementing \c qt_ntfs_permission_lookup by 1.
265
266 \snippet ntfsp.cpp 1
267
268 \note Since this is a non-atomic global variable, it is only safe
269 to increment or decrement \c qt_ntfs_permission_lookup before any
270 threads other than the main thread have started or after every thread
271 other than the main thread has ended.
272
273 \section1 Performance Issues
274
275 Some of QFileInfo's functions query the file system, but for
276 performance reasons, some functions only operate on the
277 file name itself. For example: To return the absolute path of
278 a relative file name, absolutePath() has to query the file system.
279 The path() function, however, can work on the file name directly,
280 and so it is faster.
281
282 \note To speed up performance, QFileInfo caches information about
283 the file.
284
285 Because files can be changed by other users or programs, or
286 even by other parts of the same program, there is a function that
287 refreshes the file information: refresh(). If you want to switch
288 off a QFileInfo's caching and force it to access the file system
289 every time you request information from it call setCaching(false).
290 If you want to make sure that all information is read from the
291 file system, use stat().
292
293 \section1 Platform Specific Issues
294
295 \include android-content-uri-limitations.qdocinc
296
297 \sa QDir, QFile
298*/

383/*!
384 \fn bool QFileInfo::operator!=(const QFileInfo &fileinfo) const
385
386 Returns \c true if this QFileInfo object refers to a different file
387 than the one specified by \a fileinfo; otherwise returns \c false.
388
389 \sa operator==()
390*/

392/*!
393 Returns \c true if this QFileInfo object refers to a file in the same
394 location as \a fileinfo; otherwise returns \c false.
395
396 Note that the result of comparing two empty QFileInfo objects,
397 containing no file references (file paths that do not exist or
398 are empty), is undefined.
399
400 \warning This will not compare two different symbolic links
401 pointing to the same file.
402
403 \warning Long and short file names that refer to the same file on Windows
404 are treated as if they referred to different files.
405
406 \sa operator!=()
407*/
408bool QFileInfo::operator==(const QFileInfo &fileinfo) const
409{
410 Q_D(const QFileInfo);
411 // ### Qt 5: understand long and short file names on Windows
412 // ### (GetFullPathName()).
413 if (fileinfo.d_ptr == d_ptr)
414 return true;
415 if (d->isDefaultConstructed || fileinfo.d_ptr->isDefaultConstructed)
416 return false;
417
418 // Assume files are the same if path is the same
419 if (d->fileEntry.filePath() == fileinfo.d_ptr->fileEntry.filePath())
420 return true;
421
422 Qt::CaseSensitivity sensitive;
423 if (d->fileEngine == nullptr || fileinfo.d_ptr->fileEngine == nullptr) {
424 if (d->fileEngine != fileinfo.d_ptr->fileEngine) // one is native, the other is a custom file-engine
425 return false;
426
427 sensitive = QFileSystemEngine::isCaseSensitive() ? Qt::CaseSensitive : Qt::CaseInsensitive;
428 } else {
429 if (d->fileEngine->caseSensitive() != fileinfo.d_ptr->fileEngine->caseSensitive())
430 return false;
431 sensitive = d->fileEngine->caseSensitive() ? Qt::CaseSensitive : Qt::CaseInsensitive;
432 }
433
434 // Fallback to expensive canonical path computation
435 return canonicalFilePath().compare(s: fileinfo.canonicalFilePath(), cs: sensitive) == 0;
436}

665/*!
666 Returns \c true if the file exists; otherwise returns \c false.
667
668 \note If the file is a symlink that points to a non-existing
669 file, false is returned.
670*/
671bool QFileInfo::exists() const
672{
673 Q_D(const QFileInfo);
674 if (d->isDefaultConstructed)
675 return false;
676 if (d->fileEngine == nullptr) {
677 if (!d->cache_enabled || !d->metaData.hasFlags(flags: QFileSystemMetaData::ExistsAttribute))
678 QFileSystemEngine::fillMetaData(entry: d->fileEntry, data&: d->metaData, what: QFileSystemMetaData::ExistsAttribute);
679 return d->metaData.exists();
680 }
681 return d->getFileFlags(request: QAbstractFileEngine::ExistsFlag);

684/*!
685 \since 5.2
686
687 Returns \c true if the \a file exists; otherwise returns \c false.
688
689 \note If \a file is a symlink that points to a non-existing
690 file, false is returned.
691
692 \note Using this function is faster than using
693 \c QFileInfo(file).exists() for file system access.
694*/
695bool QFileInfo::exists(const QString &file)
696{
697 if (file.isEmpty())
698 return false;
699 QFileSystemEntry entry(file);
700 QFileSystemMetaData data;
701 std::unique_ptr<QAbstractFileEngine> engine
702 {QFileSystemEngine::resolveEntryAndCreateLegacyEngine(entry, data)};
703 // Expensive fallback to non-QFileSystemEngine implementation
704 if (engine)
705 return QFileInfo(new QFileInfoPrivate(entry, data, std::move(engine))).exists();
706
707 QFileSystemEngine::fillMetaData(entry, data, what: QFileSystemMetaData::ExistsAttribute);
708 return data.exists();
709}
