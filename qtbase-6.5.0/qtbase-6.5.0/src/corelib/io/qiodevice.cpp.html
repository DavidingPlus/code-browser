24[[maybe_unused]]
25static void debugBinaryString(const char *input, qint64 maxlen)
26{
27 QByteArray tmp;
28 qlonglong startOffset = 0;
29 for (qint64 i = 0; i < maxlen; ++i) {
30 tmp += input[i];
31
32 if ((i % 16) == 15 || i == (maxlen - 1)) {
33 printf(format: "\n%15lld:", startOffset);
34 startOffset += tmp.size();
35
36 for (qsizetype j = 0; j < tmp.size(); ++j)
37 printf(format: " %02x", int(uchar(tmp[j])));
38 for (qsizetype j = tmp.size(); j < 16 + 1; ++j)
39 printf(format: " ");
40 for (qsizetype j = 0; j < tmp.size(); ++j)
41 printf(format: "%c", isprint(int(uchar(tmp[j]))) ? tmp[j] : '.');
42 tmp.clear();
43 }
44 }
45 printf(format: "\n\n");
46}

137/*!
138 \class QIODevice
139 \inmodule QtCore
140 \reentrant
141
142 \brief The QIODevice class is the base interface class of all I/O
143 devices in Qt.
144
145 \ingroup io
146
147 QIODevice provides both a common implementation and an abstract
148 interface for devices that support reading and writing of blocks
149 of data, such as QFile, QBuffer and QTcpSocket. QIODevice is
150 abstract and cannot be instantiated, but it is common to use the
151 interface it defines to provide device-independent I/O features.
152 For example, Qt's XML classes operate on a QIODevice pointer,
153 allowing them to be used with various devices (such as files and
154 buffers).
155
156 Before accessing the device, open() must be called to set the
157 correct OpenMode (such as ReadOnly or ReadWrite). You can then
158 write to the device with write() or putChar(), and read by calling
159 either read(), readLine(), or readAll(). Call close() when you are
160 done with the device.
161
162 QIODevice distinguishes between two types of devices:
163 random-access devices and sequential devices.
164
165 \list
166 \li Random-access devices support seeking to arbitrary
167 positions using seek(). The current position in the file is
168 available by calling pos(). QFile and QBuffer are examples of
169 random-access devices.
170
171 \li Sequential devices don't support seeking to arbitrary
172 positions. The data must be read in one pass. The functions
173 pos() and size() don't work for sequential devices.
174 QTcpSocket and QProcess are examples of sequential devices.
175 \endlist
176
177 You can use isSequential() to determine the type of device.
178
179 QIODevice emits readyRead() when new data is available for
180 reading; for example, if new data has arrived on the network or if
181 additional data is appended to a file that you are reading
182 from. You can call bytesAvailable() to determine the number of
183 bytes that are currently available for reading. It's common to use
184 bytesAvailable() together with the readyRead() signal when
185 programming with asynchronous devices such as QTcpSocket, where
186 fragments of data can arrive at arbitrary points in
187 time. QIODevice emits the bytesWritten() signal every time a
188 payload of data has been written to the device. Use bytesToWrite()
189 to determine the current amount of data waiting to be written.
190
191 Certain subclasses of QIODevice, such as QTcpSocket and QProcess,
192 are asynchronous. This means that I/O functions such as write()
193 or read() always return immediately, while communication with the
194 device itself may happen when control goes back to the event loop.
195 QIODevice provides functions that allow you to force these
196 operations to be performed immediately, while blocking the
197 calling thread and without entering the event loop. This allows
198 QIODevice subclasses to be used without an event loop, or in
199 a separate thread:
200
201 \list
202 \li waitForReadyRead() - This function suspends operation in the
203 calling thread until new data is available for reading.
204
205 \li waitForBytesWritten() - This function suspends operation in the
206 calling thread until one payload of data has been written to the
207 device.
208
209 \li waitFor....() - Subclasses of QIODevice implement blocking
210 functions for device-specific operations. For example, QProcess
211 has a function called \l {QProcess::}{waitForStarted()} which suspends operation in
212 the calling thread until the process has started.
213 \endlist
214
215 Calling these functions from the main, GUI thread, may cause your
216 user interface to freeze. Example:
217
218 \snippet code/src_corelib_io_qiodevice.cpp 0
219
220 By subclassing QIODevice, you can provide the same interface to
221 your own I/O devices. Subclasses of QIODevice are only required to
222 implement the protected readData() and writeData() functions.
223 QIODevice uses these functions to implement all its convenience
224 functions, such as getChar(), readLine() and write(). QIODevice
225 also handles access control for you, so you can safely assume that
226 the device is opened in write mode if writeData() is called.
227
228 Some subclasses, such as QFile and QTcpSocket, are implemented
229 using a memory buffer for intermediate storing of data. This
230 reduces the number of required device accessing calls, which are
231 often very slow. Buffering makes functions like getChar() and
232 putChar() fast, as they can operate on the memory buffer instead
233 of directly on the device itself. Certain I/O operations, however,
234 don't work well with a buffer. For example, if several users open
235 the same device and read it character by character, they may end
236 up reading the same data when they meant to read a separate chunk
237 each. For this reason, QIODevice allows you to bypass any
238 buffering by passing the Unbuffered flag to open(). When
239 subclassing QIODevice, remember to bypass any buffer you may use
240 when the device is open in Unbuffered mode.
241
242 Usually, the incoming data stream from an asynchronous device is
243 fragmented, and chunks of data can arrive at arbitrary points in time.
244 To handle incomplete reads of data structures, use the transaction
245 mechanism implemented by QIODevice. See startTransaction() and related
246 functions for more details.
247
248 Some sequential devices support communicating via multiple channels. These
249 channels represent separate streams of data that have the property of
250 independently sequenced delivery. Once the device is opened, you can
251 determine the number of channels by calling the readChannelCount() and
252 writeChannelCount() functions. To switch between channels, call
253 setCurrentReadChannel() and setCurrentWriteChannel(), respectively.
254 QIODevice also provides additional signals to handle asynchronous
255 communication on a per-channel basis.
256
257 \sa QBuffer, QFile, QTcpSocket
258*/

268/*!
269 \enum QIODeviceBase::OpenModeFlag
270
271 This enum is used with QIODevice::open() to describe the mode in which a
272 device is opened. It is also returned by QIODevice::openMode().
273
274 \value NotOpen The device is not open.
275 \value ReadOnly The device is open for reading.
276 \value WriteOnly The device is open for writing. Note that, for file-system
277 subclasses (e.g. QFile), this mode implies Truncate unless
278 combined with ReadOnly, Append or NewOnly.
279 \value ReadWrite The device is open for reading and writing.
280 \value Append The device is opened in append mode so that all data is
281 written to the end of the file.
282 \value Truncate If possible, the device is truncated before it is opened.
283 All earlier contents of the device are lost.
284 \value Text When reading, the end-of-line terminators are
285 translated to '\\n'. When writing, the end-of-line
286 terminators are translated to the local encoding, for
287 example '\\r\\n' for Win32.
288 \value Unbuffered Any buffer in the device is bypassed.
289 \value NewOnly Fail if the file to be opened already exists. Create and
290 open the file only if it does not exist. There is a
291 guarantee from the operating system that you are the only
292 one creating and opening the file. Note that this mode
293 implies WriteOnly, and combining it with ReadWrite is
294 allowed. This flag currently only affects QFile. Other
295 classes might use this flag in the future, but until then
296 using this flag with any classes other than QFile may
297 result in undefined behavior. (since Qt 5.11)
298 \value ExistingOnly Fail if the file to be opened does not exist. This flag
299 must be specified alongside ReadOnly, WriteOnly, or
300 ReadWrite. Note that using this flag with ReadOnly alone
301 is redundant, as ReadOnly already fails when the file does
302 not exist. This flag currently only affects QFile. Other
303 classes might use this flag in the future, but until then
304 using this flag with any classes other than QFile may
305 result in undefined behavior. (since Qt 5.11)
306
307 Certain flags, such as \c Unbuffered and \c Truncate, are
308 meaningless when used with some subclasses. Some of these
309 restrictions are implied by the type of device that is represented
310 by a subclass. In other cases, the restriction may be due to the
311 implementation, or may be imposed by the underlying platform; for
312 example, QTcpSocket does not support \c Unbuffered mode, and
313 limitations in the native API prevent QFile from supporting \c
314 Unbuffered on Windows.
315*/

317/*! \fn QIODevice::bytesWritten(qint64 bytes)
318
319 This signal is emitted every time a payload of data has been
320 written to the device's current write channel. The \a bytes argument is
321 set to the number of bytes that were written in this payload.
322
323 bytesWritten() is not emitted recursively; if you reenter the event loop
324 or call waitForBytesWritten() inside a slot connected to the
325 bytesWritten() signal, the signal will not be reemitted (although
326 waitForBytesWritten() may still return true).
327
328 \sa readyRead()
329*/

331/*!
332 \fn QIODevice::channelBytesWritten(int channel, qint64 bytes)
333 \since 5.7
334
335 This signal is emitted every time a payload of data has been written to
336 the device. The \a bytes argument is set to the number of bytes that were
337 written in this payload, while \a channel is the channel they were written
338 to. Unlike bytesWritten(), it is emitted regardless of the
339 \l{currentWriteChannel()}{current write channel}.
340

343

347/*!
348 \fn QIODevice::readyRead()
349
350 This signal is emitted once every time new data is available for
351 reading from the device's current read channel. It will only be emitted
352 again once new data is available, such as when a new payload of network
353 data has arrived on your network socket, or when a new block of data has
354 been appended to your device.
355
356 readyRead() is not emitted recursively; if you reenter the event loop or
357 call waitForReadyRead() inside a slot connected to the readyRead() signal,
358 the signal will not be reemitted (although waitForReadyRead() may still
359 return true).
360
361 Note for developers implementing classes derived from QIODevice:
362 you should always emit readyRead() when new data has arrived (do not
363 emit it only because there's data still to be read in your
364 buffers). Do not emit readyRead() in other conditions.
365
366 \sa bytesWritten()
367*/

372

374 device. The \a channel argument is set to the index of the read channel on
375 which the data has arrived. Unlike readyRead(), it is emitted regardless of

660/*!
661 \internal
662*/
663void QIODevicePrivate::setReadChannelCount(int count)
664{
665 if (count > readBuffers.size()) {
666 readBuffers.reserve(sz: count);
667
668 // If readBufferChunkSize is zero, we should bypass QIODevice's
669 // read buffers, even if the QIODeviceBase::Unbuffered flag is not
670 // set when opened. However, if a read transaction is started or
671 // ungetChar() is called, we still have to use the internal buffer.
672 // To support these cases, pass a default value to the QRingBuffer
673 // constructor.
674
675 while (readBuffers.size() < count)
676 readBuffers.emplace_back(args: readBufferChunkSize != 0 ? readBufferChunkSize
677 : QIODEVICE_BUFFERSIZE);
678 } else {
679 readBuffers.resize(sz: count);
680 }
681 readChannelCount = count;
682 setCurrentReadChannel(currentReadChannel);
683}

734 }

1035/*!
1036 \internal
1037*/
1038qint64 QIODevicePrivate::read(char *data, qint64 maxSize, bool peeking)
1039{
1040 Q_Q(QIODevice);
1041
1042 const bool buffered = (readBufferChunkSize != 0 && (openMode & QIODevice::Unbuffered) == 0);
1043 const bool sequential = isSequential();
1044 const bool keepDataInBuffer = sequential
1045 ? peeking || transactionStarted
1046 : peeking && buffered;
1047 const qint64 savedPos = pos;
1048 qint64 readSoFar = 0;
1049 bool madeBufferReadsOnly = true;
1050 bool deviceAtEof = false;
1051 char *readPtr = data;
1052 qint64 bufferPos = (sequential && transactionStarted) ? transactionPos : Q_INT64_C(0);
1053 forever {
1054 // Try reading from the buffer.
1055 qint64 bufferReadChunkSize = keepDataInBuffer
1056 ? buffer.peek(data, maxLength: maxSize, pos: bufferPos)
1057 : buffer.read(data, maxLength: maxSize);
1058 if (bufferReadChunkSize > 0) {
1059 bufferPos += bufferReadChunkSize;
1060 if (!sequential)
1061 pos += bufferReadChunkSize;
1062#if defined QIODEVICE_DEBUG
1063 printf("%p \treading %lld bytes from buffer into position %lld\n", q,
1064 bufferReadChunkSize, readSoFar);
1065#endif
1066 readSoFar += bufferReadChunkSize;
1067 data += bufferReadChunkSize;
1068 maxSize -= bufferReadChunkSize;
1069 }
1070
1071 if (maxSize > 0 && !deviceAtEof) {
1072 qint64 readFromDevice = 0;
1073 // Make sure the device is positioned correctly.
1074 if (sequential || pos == devicePos || q->seek(pos)) {
1075 madeBufferReadsOnly = false; // fix readData attempt
1076 if ((!buffered || maxSize >= readBufferChunkSize) && !keepDataInBuffer) {
1077 // Read big chunk directly to output buffer
1078 readFromDevice = q->readData(data, maxlen: maxSize);
1079 deviceAtEof = (readFromDevice != maxSize);
1080#if defined QIODEVICE_DEBUG
1081 printf("%p \treading %lld bytes from device (total %lld)\n", q,
1082 readFromDevice, readSoFar);
1083#endif
1084 if (readFromDevice > 0) {
1085 readSoFar += readFromDevice;
1086 data += readFromDevice;
1087 maxSize -= readFromDevice;
1088 if (!sequential) {
1089 pos += readFromDevice;
1090 devicePos += readFromDevice;
1091 }
1092 }
1093 } else {
1094 // Do not read more than maxSize on unbuffered devices
1095 const qint64 bytesToBuffer = (!buffered && maxSize < buffer.chunkSize())
1096 ? maxSize
1097 : qint64(buffer.chunkSize());
1098 // Try to fill QIODevice buffer by single read
1099 readFromDevice = q->readData(data: buffer.reserve(bytes: bytesToBuffer), maxlen: bytesToBuffer);
1100 deviceAtEof = (readFromDevice != bytesToBuffer);
1101 buffer.chop(bytes: bytesToBuffer - qMax(Q_INT64_C(0), b: readFromDevice));
1102 if (readFromDevice > 0) {
1103 if (!sequential)
1104 devicePos += readFromDevice;
1105#if defined QIODEVICE_DEBUG
1106 printf("%p \treading %lld from device into buffer\n", q,
1107 readFromDevice);
1108#endif
1109 continue;
1110 }
1111 }
1112 } else {
1113 readFromDevice = -1;
1114 }
1115
1116 if (readFromDevice < 0 && readSoFar == 0) {
1117 // error and we haven't read anything: return immediately
1118 return qint64(-1);
1119 }
1120 }
1121
1122 if ((openMode & QIODevice::Text) && readPtr < data) {
1123 const char *endPtr = data;
1124
1125 // optimization to avoid initial self-assignment
1126 while (*readPtr != '\r') {
1127 if (++readPtr == endPtr)
1128 break;
1129 }
1130
1131 char *writePtr = readPtr;
1132
1133 while (readPtr < endPtr) {
1134 char ch = *readPtr++;
1135 if (ch != '\r')
1136 *writePtr++ = ch;
1137 else {
1138 --readSoFar;
1139 --data;
1140 ++maxSize;
1141 }
1142 }
1143
1144 // Make sure we get more data if there is room for more. This
1145 // is very important for when someone seeks to the start of a
1146 // '\r\n' and reads one character - they should get the '\n'.
1147 readPtr = data;
1148 continue;
1149 }
1150
1151 break;
1152 }
1153
1154 // Restore positions after reading
1155 if (keepDataInBuffer) {
1156 if (peeking)
1157 pos = savedPos; // does nothing on sequential devices
1158 else
1159 transactionPos = bufferPos;
1160 } else if (peeking) {
1161 seekBuffer(newPos: savedPos); // unbuffered random-access device
1162 }
1163
1164 if (madeBufferReadsOnly && isBufferEmpty())
1165 q->readData(data, maxlen: 0);
1166
1167 return readSoFar;
1168}

1220 byte array.
1221
1222 This function has no way of reporting errors; returning an empty
1223 QByteArray can mean either that no data was currently available
1224 for reading, or that an error occurred. This function also has no
1225 way of indicating that more data may have been available and
1226 couldn't be read.

1229{
1230 Q_D(QIODevice);
1231#if defined QIODEVICE_DEBUG

1233 this, d->pos, d->buffer.size());
1234#endif
1235
1236 QByteArray result;
1237 CHECK_READABLE(read, result);
1238
1239 qint64 readBytes = (d->isSequential() ? Q_INT64_C(0) : size());
1240 if (readBytes == 0) {
1241 // Size is unknown, read incrementally.
1242 qint64 readChunkSize = qMax(a: qint64(d->buffer.chunkSize()),

1340qint64 QIODevicePrivate::readLine(char *data, qint64 maxSize)
1341{
1342 Q_Q(QIODevice);
1343 Q_ASSERT(maxSize >= 2);
1344
1345 // Leave room for a '\0'
1346 --maxSize;
1347
1348 const bool sequential = isSequential();
1349 const bool keepDataInBuffer = sequential && transactionStarted;
1350
1351 qint64 readSoFar = 0;
1352 if (keepDataInBuffer) {
1353 if (transactionPos < buffer.size()) {
1354 // Peek line from the specified position
1355 const qint64 i = buffer.indexOf(c: '\n', maxLength: maxSize, pos: transactionPos);
1356 readSoFar = buffer.peek(data, maxLength: i >= 0 ? (i - transactionPos + 1) : maxSize,
1357 pos: transactionPos);
1358 transactionPos += readSoFar;
1359 if (transactionPos == buffer.size())
1360 q->readData(data, maxlen: 0);
1361 }
1362 } else if (!buffer.isEmpty()) {
1363 // QRingBuffer::readLine() terminates the line with '\0'
1364 readSoFar = buffer.readLine(data, maxLength: maxSize + 1);
1365 if (buffer.isEmpty())
1366 q->readData(data, maxlen: 0);
1367 if (!sequential)
1368 pos += readSoFar;
1369 }
1370
1371 if (readSoFar) {
1372#if defined QIODEVICE_DEBUG
1373 printf("%p \tread from buffer: %lld bytes, last character read: %hhx\n", q,
1374 readSoFar, data[readSoFar - 1]);
1375 debugBinaryString(data, readSoFar);
1376#endif
1377 if (data[readSoFar - 1] == '\n') {
1378 if (openMode & QIODevice::Text) {
1379 // QRingBuffer::readLine() isn't Text aware.
1380 if (readSoFar > 1 && data[readSoFar - 2] == '\r') {
1381 --readSoFar;
1382 data[readSoFar - 1] = '\n';
1383 }
1384 }
1385 data[readSoFar] = '\0';
1386 return readSoFar;
1387 }
1388 }
1389
1390 if (pos != devicePos && !sequential && !q->seek(pos))
1391 return qint64(-1);
1392 baseReadLineDataCalled = false;
1393 // Force base implementation for transaction on sequential device
1394 // as it stores the data in internal buffer automatically.
1395 qint64 readBytes = keepDataInBuffer
1396 ? q->QIODevice::readLineData(data: data + readSoFar, maxlen: maxSize - readSoFar)
1397 : q->readLineData(data: data + readSoFar, maxlen: maxSize - readSoFar);
1398#if defined QIODEVICE_DEBUG
1399 printf("%p \tread from readLineData: %lld bytes, readSoFar = %lld bytes\n", q,
1400 readBytes, readSoFar);
1401 if (readBytes > 0) {
1402 debugBinaryString(data, readSoFar + readBytes);
1403 }
1404#endif
1405 if (readBytes < 0) {
1406 data[readSoFar] = '\0';
1407 return readSoFar ? readSoFar : -1;
1408 }
1409 readSoFar += readBytes;
1410 if (!baseReadLineDataCalled && !sequential) {
1411 pos += readBytes;
1412 // If the base implementation was not called, then we must
1413 // assume the device position is invalid and force a seek.
1414 devicePos = qint64(-1);
1415 }
1416 data[readSoFar] = '\0';
1417
1418 if (openMode & QIODevice::Text) {
1419 if (readSoFar > 1 && data[readSoFar - 1] == '\n' && data[readSoFar - 2] == '\r') {
1420 data[readSoFar - 2] = '\n';
1421 data[readSoFar - 1] = '\0';
1422 --readSoFar;
1423 }
1424 }
1425
1426 return readSoFar;

1920/*!
1921 \since 5.10
1922
1923 Skips up to \a maxSize bytes from the device. Returns the number of bytes
1924 actually skipped, or -1 on error.
1925
1926 This function does not wait and only discards the data that is already
1927 available for reading.
1928
1929 If the device is opened in text mode, end-of-line terminators are
1930 translated to '\n' symbols and count as a single byte identically to the
1931 read() and peek() behavior.
1932
1933 This function works for all devices, including sequential ones that cannot
1934 seek(). It is optimized to skip unwanted data after a peek() call.
1935
1936 For random-access devices, skip() can be used to seek forward from the
1937 current position. Negative \a maxSize values are not allowed.
1938
1939 \sa skipData(), peek(), seek(), read()
1940*/
1941qint64 QIODevice::skip(qint64 maxSize)
1942{
1943 Q_D(QIODevice);
1944 CHECK_MAXLEN(skip, qint64(-1));
1945 CHECK_READABLE(skip, qint64(-1));
1946
1947 const bool sequential = d->isSequential();
1948
1949#if defined QIODEVICE_DEBUG
1950 printf("%p QIODevice::skip(%lld), d->pos = %lld, d->buffer.size() = %lld\n",
1951 this, maxSize, d->pos, d->buffer.size());
1952#endif
1953
1954 if ((sequential && d->transactionStarted) || (d->openMode & QIODevice::Text) != 0)
1955 return d->skipByReading(maxSize);
1956
1957 // First, skip over any data in the internal buffer.
1958 qint64 skippedSoFar = 0;
1959 if (!d->buffer.isEmpty()) {
1960 skippedSoFar = d->buffer.skip(length: maxSize);
1961#if defined QIODEVICE_DEBUG
1962 printf("%p \tskipping %lld bytes in buffer\n", this, skippedSoFar);
1963#endif
1964 if (!sequential)
1965 d->pos += skippedSoFar;
1966 if (d->buffer.isEmpty())
1967 readData(data: nullptr, maxlen: 0);
1968 if (skippedSoFar == maxSize)
1969 return skippedSoFar;
1970
1971 maxSize -= skippedSoFar;
1972 }
1973
1974 // Try to seek on random-access device. At this point,
1975 // the internal read buffer is empty.
1976 if (!sequential) {
1977 const qint64 bytesToSkip = qMin(a: size() - d->pos, b: maxSize);
1978
1979 // If the size is unknown or file position is at the end,
1980 // fall back to reading below.
1981 if (bytesToSkip > 0) {
1982 if (!seek(pos: d->pos + bytesToSkip))
1983 return skippedSoFar ? skippedSoFar : Q_INT64_C(-1);
1984 if (bytesToSkip == maxSize)
1985 return skippedSoFar + bytesToSkip;
1986
1987 skippedSoFar += bytesToSkip;
1988 maxSize -= bytesToSkip;
1989 }
1990 }
1991
1992 const qint64 skipResult = skipData(maxSize);
1993 if (skippedSoFar == 0)
1994 return skipResult;
1995
1996 if (skipResult == -1)
1997 return skippedSoFar;
1998
1999 return skippedSoFar + skipResult;
2000}

2006{

2008 do {
