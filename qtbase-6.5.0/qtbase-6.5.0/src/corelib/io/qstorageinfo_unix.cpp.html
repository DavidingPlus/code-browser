155template <typename String>
156static bool isParentOf(const String &parent, const QString &dirName)
157{
158 return dirName.startsWith(parent) &&
159 (dirName.size() == parent.size() || dirName.at(i: parent.size()) == u'/' ||
160 parent.size() == 1);
161}

163static bool shouldIncludeFs(const QStorageIterator &it)
164{
165 /*
166 * This function implements a heuristic algorithm to determine whether a
167 * given mount should be reported to the user. Our objective is to list
168 * only entries that the end-user would find useful.
169 *
170 * We therefore ignore:
171 * - mounted in /dev, /proc, /sys: special mounts
172 * (this will catch /sys/fs/cgroup, /proc/sys/fs/binfmt_misc, /dev/pts,
173 * some of which are tmpfs on Linux)
174 * - mounted in /var/run or /var/lock: most likely pseudofs
175 * (on earlier systemd versions, /var/run was a bind-mount of /run, so
176 * everything would be unnecessarily duplicated)
177 * - filesystem type is "rootfs": artifact of the root-pivot on some Linux
178 * initrd
179 * - if the filesystem total size is zero, it's a pseudo-fs (not checked here).
180 */
181
182 QString mountDir = it.rootPath();
183 if (isParentOf(parent: "/dev"_L1, dirName: mountDir)
184 || isParentOf(parent: "/proc"_L1, dirName: mountDir)
185 || isParentOf(parent: "/sys"_L1, dirName: mountDir)
186 || isParentOf(parent: "/var/run"_L1, dirName: mountDir)
187 || isParentOf(parent: "/var/lock"_L1, dirName: mountDir)) {
188 return false;
189 }
190
191#if defined(Q_OS_LINUX) && !defined(Q_OS_ANDROID)
192 if (it.fileSystemType() == "rootfs")
193 return false;
194#endif
195
196 // size checking in mountedVolumes()
197 return true;
198}

311{

313 // If file is virtual, file.readLine() may succeed even when file.atEnd().
314 do {
315 const QByteArray line = file.readLine();
316 if (line.isEmpty() && file.atEnd())

320
321 m_device = data.at(0);
322 m_rootPath = data.at(1);
323 m_fileSystemType = data.at(2);
324 m_options = data.at(3);
325
326 return true;
327}

390inline bool QStorageIterator::next()
391{
392 mnt.subvolume = nullptr;
393 mnt.superopts = nullptr;
394 if (!usingMountinfo)
395 return ::getmntent_r(stream: fp, result: &mnt, buffer: buffer.data(), bufsize: buffer.size()) != nullptr;
396
397 // Helper function to parse paths that the kernel inserts escape sequences
398 // for. The unescaped string is left at \a src and is properly
399 // NUL-terminated. Returns a pointer to the delimiter that terminated the
400 // path, or nullptr if it failed.
401 auto parseMangledPath = [](char *src) {
402 // The kernel escapes with octal the following characters:
403 // space ' ', tab '\t', backslask '\\', and newline '\n'
404 char *dst = src;
405 while (*src) {
406 switch (*src) {
407 case ' ':
408 // Unescaped space: end of the field.
409 *dst = '\0';
410 return src;
411
412 default:
413 *dst++ = *src++;
414 break;
415
416 case '\\':
417 // It always uses exactly three octal characters.
418 ++src;
419 char c = (*src++ - '0') << 6;
420 c |= (*src++ - '0') << 3;
421 c |= (*src++ - '0');
422 *dst++ = c;
423 break;
424 }
425 }
426
427 // Found a NUL before the end of the field.
428 src = nullptr;
429 return src;
430 };
431
432 char *ptr = buffer.data();
433 if (fgets(s: ptr, n: buffer.size(), stream: fp) == nullptr)
434 return false;
435
436 size_t len = strlen(s: ptr);
437 if (len == 0)
438 return false;
439 while (Q_UNLIKELY(ptr[len - 1] != '\n' && !feof(fp))) {
440 // buffer wasn't large enough. Enlarge and try again.
441 // (we're readidng from the kernel, so OOM is unlikely)
442 buffer.resize(size: (buffer.size() + 4096) & ~4095);
443 ptr = buffer.data();
444 if (fgets(s: ptr + len, n: buffer.size() - len, stream: fp) == nullptr)
445 return false;
446
447 len += strlen(s: ptr + len);
448 Q_ASSERT(len < size_t(buffer.size()));
449 }
450 ptr[len - 1] = '\0';
451 const char *const stop = ptr + len - 1;
452
453 // parse the line
454 mnt.mnt_freq = 0;
455 mnt.mnt_passno = 0;
456
457 auto r = qstrntoll(nptr: ptr, size: stop - ptr, base: 10);
458 if (!r.ok())
459 return false;
460 mnt.mount_id = r.result;
461
462 ptr += r.used;
463 r = qstrntoll(nptr: ptr, size: stop - ptr, base: 10);
464 if (!r.ok())
465 return false;
466 int parent_id = r.result;
467 Q_UNUSED(parent_id);
468
469 ptr += r.used;
470 r = qstrntoll(nptr: ptr, size: stop - ptr, base: 10);
471 if (!r.ok())
472 return false;
473 ptr += r.used;
474 if (*ptr != ':')
475 return false;
476 int rdevmajor = r.result;
477 ++ptr; // Skip over the ':'
478 r = qstrntoll(nptr: ptr, size: stop - ptr, base: 10);
479 if (!r.ok())
480 return false;
481 mnt.rdev = makedev(rdevmajor, r.result);
482
483 ptr += r.used;
484 if (*ptr != ' ')
485 return false;
486
487 mnt.subvolume = ++ptr;
488 ptr = parseMangledPath(ptr);
489 if (!ptr)
490 return false;
491
492 // unset a subvolume of "/" -- it's not a *sub* volume
493 if (mnt.subvolume + 1 == ptr)
494 *mnt.subvolume = '\0';
495
496 mnt.mnt_dir = ++ptr;
497 ptr = parseMangledPath(ptr);
498 if (!ptr)
499 return false;
500
501 mnt.mnt_opts = ++ptr;
502 ptr = strchr(s: ptr, c: ' ');
503 if (!ptr)
504 return false;
505
506 // we don't parse the flags, so just find the separator
507 if (char *const dashed = strstr(haystack: ptr, needle: " - ")) {
508 *ptr = '\0';
509 ptr = dashed + strlen(s: " - ") - 1;
510 } else {
511 return false;
512 }
513
514 mnt.mnt_type = ++ptr;
515 ptr = strchr(s: ptr, c: ' ');
516 if (!ptr)
517 return false;
518 *ptr = '\0';
519
520 mnt.mnt_fsname = ++ptr;
521 ptr = parseMangledPath(ptr);
522 if (!ptr)
523 return false;
524
525 mnt.superopts = ++ptr;
526 ptr += strcspn(s: ptr, reject: " \n");
527 *ptr = '\0';
528
529 return true;
530}

542inline QByteArray QStorageIterator::device() const
543{
544#ifdef Q_OS_LINUX
545 // check that the device exists
546 if (mnt.mnt_fsname[0] == '/' && access(name: mnt.mnt_fsname, F_OK) != 0) {
547 // It doesn't, so let's try to resolve the dev_t from /dev/block.
548 // Note how strlen("4294967295") == digits10 + 1, so we need to add 1
549 // for each number, plus the ':'.
550 char buf[sizeof("/dev/block/") + 2 * std::numeric_limits<unsigned>::digits10 + 3];
551 QByteArray dev(PATH_MAX, Qt::Uninitialized);
552 char *devdata = dev.data();
553
554 snprintf(s: buf, maxlen: sizeof(buf), format: "/dev/block/%u:%u", major(mnt.rdev), minor(mnt.rdev));
555 if (realpath(name: buf, resolved: devdata)) {
556 dev.truncate(pos: strlen(s: devdata));
557 return dev;
558 }
559 }
560#endif
561 return QByteArray(mnt.mnt_fsname);
562}

564inline QByteArray QStorageIterator::options() const
565{
566 // Merge the two options, starting with the superblock options and letting
567 // the per-mount options override.
568 const char *superopts = mnt.superopts;
569
570 // Both mnt_opts and superopts start with "ro" or "rw", so we can skip the
571 // superblock's field (see show_mountinfo() in fs/proc_namespace.c).
572 if (superopts && superopts[0] == 'r') {
573 if (superopts[2] == '\0') // no other superopts besides "ro" / "rw"?
574 superopts = nullptr;
575 else if (superopts[2] == ',')
576 superopts += 3;
577 }
578
579 if (superopts)
580 return QByteArray(superopts) + ',' + mnt.mnt_opts;
581 return QByteArray(mnt.mnt_opts);
582}

602inline bool QStorageIterator::next()
603{
604 BVolume volume;
605
606 if (m_volumeRoster.GetNextVolume(&volume) != B_OK)
607 return false;
608
609 BDirectory directory;
610 if (volume.GetRootDirectory(&directory) != B_OK)
611 return false;
612
613 const BPath path(&directory);
614
615 fs_info fsInfo;
616 memset(&fsInfo, 0, sizeof(fsInfo));
617
618 if (fs_stat_dev(volume.Device(), &fsInfo) != 0)
619 return false;
620
621 m_rootPath = path.Path();
622 m_fileSystemType = QByteArray(fsInfo.fsh_name);
623
624 const QByteArray deviceName(fsInfo.device_name);
625 m_device = (deviceName.isEmpty() ? QByteArray::number(qint32(volume.Device())) : deviceName);
626
627 return true;
628}

700void QStorageInfoPrivate::initRootPath()
701{
702 rootPath = QFileInfo(rootPath).canonicalFilePath();
703
704 if (rootPath.isEmpty())
705 return;
706
707 QStorageIterator it;
708 if (!it.isValid()) {
709 rootPath = QStringLiteral("/");
710 return;
711 }
712
713 int maxLength = 0;
714 const QString oldRootPath = rootPath;
715 rootPath.clear();
716
717 while (it.next()) {
718 const QString mountDir = it.rootPath();
719 const QByteArray fsName = it.fileSystemType();
720 // we try to find most suitable entry
721 if (isParentOf(parent: mountDir, dirName: oldRootPath) && maxLength < mountDir.size()) {
722 maxLength = mountDir.size();
723 rootPath = mountDir;
724 device = it.device();
725 fileSystemType = fsName;
726 subvolume = it.subvolume();
727 }
728 }
729}

731#ifdef Q_OS_LINUX
732// udev encodes the labels with ID_LABEL_FS_ENC which is done with
733// blkid_encode_string(). Within this function some 1-byte utf-8
734// characters not considered safe (e.g. '\' or ' ') are encoded as hex
735static QString decodeFsEncString(const QString &str)
736{
737 QString decoded;
738 decoded.reserve(asize: str.size());
739
740 int i = 0;
741 while (i < str.size()) {
742 if (i <= str.size() - 4) { // we need at least four characters \xAB
743 if (QStringView{str}.sliced(pos: i).startsWith(s: "\\x"_L1)) {
744 bool bOk;
745 const int code = QStringView{str}.mid(pos: i+2, n: 2).toInt(ok: &bOk, base: 16);
746 // only decode characters between 0x20 and 0x7f but not
747 // the backslash to prevent collisions
748 if (bOk && code >= 0x20 && code < 0x80 && code != '\\') {
749 decoded += QChar(code);
750 i += 4;
751 continue;
752 }
753 }
754 }
755 decoded += str.at(i);
756 ++i;
757 }
758 return decoded;
759}
760#endif

762static inline QString retrieveLabel(const QByteArray &device)
763{
764#ifdef Q_OS_LINUX
765 static const char pathDiskByLabel[] = "/dev/disk/by-label";
766
767 QFileInfo devinfo(QFile::decodeName(localFileName: device));
768 QString devicePath = devinfo.canonicalFilePath();
769
770 QDirIterator it(QLatin1StringView(pathDiskByLabel), QDir::NoDotAndDotDot);
771 while (it.hasNext()) {
772 QFileInfo fileInfo = it.nextFileInfo();
773 if (fileInfo.isSymLink() && fileInfo.symLinkTarget() == devicePath)
774 return decodeFsEncString(str: fileInfo.fileName());
775 }
776#elif defined Q_OS_HAIKU
777 fs_info fsInfo;
778 memset(&fsInfo, 0, sizeof(fsInfo));
779
780 int32 pos = 0;
781 dev_t dev;
782 while ((dev = next_dev(&pos)) >= 0) {
783 if (fs_stat_dev(dev, &fsInfo) != 0)
784 continue;
785
786 if (qstrcmp(fsInfo.device_name, device.constData()) == 0)
787 return QString::fromLocal8Bit(fsInfo.volume_name);
788 }
789#else
790 Q_UNUSED(device);
791#endif
792
793 return QString();
794}

796void QStorageInfoPrivate::doStat()
797{
798 initRootPath();

806void QStorageInfoPrivate::retrieveVolumeInfo()
807{
808 QT_STATFSBUF statfs_buf;
809 int result;
810 EINTR_LOOP(result, QT_STATFS(QFile::encodeName(rootPath).constData(), &statfs_buf));
811 if (result == 0) {
812 valid = true;
813 ready = true;
814
815#if defined(Q_OS_INTEGRITY) || (defined(Q_OS_BSD4) && !defined(Q_OS_NETBSD)) || defined(Q_OS_RTEMS)
816 bytesTotal = statfs_buf.f_blocks * statfs_buf.f_bsize;
817 bytesFree = statfs_buf.f_bfree * statfs_buf.f_bsize;
818 bytesAvailable = statfs_buf.f_bavail * statfs_buf.f_bsize;
819#else
820 bytesTotal = statfs_buf.f_blocks * statfs_buf.f_frsize;
821 bytesFree = statfs_buf.f_bfree * statfs_buf.f_frsize;
822 bytesAvailable = statfs_buf.f_bavail * statfs_buf.f_frsize;
823#endif
824 blockSize = statfs_buf.f_bsize;
825#if defined(Q_OS_ANDROID) || defined(Q_OS_BSD4) || defined(Q_OS_INTEGRITY) || defined(Q_OS_RTEMS)
826#if defined(_STATFS_F_FLAGS)
827 readOnly = (statfs_buf.f_flags & ST_RDONLY) != 0;
828#endif
829#else
830 readOnly = (statfs_buf.f_flag & ST_RDONLY) != 0;
831#endif
832 }
833}

835QList<QStorageInfo> QStorageInfoPrivate::mountedVolumes()
836{
837 QStorageIterator it;
838 if (!it.isValid())
839 return QList<QStorageInfo>() << root();
840
841 QList<QStorageInfo> volumes;
842
843 while (it.next()) {
844 if (!shouldIncludeFs(it))
845 continue;
846
847 const QString mountDir = it.rootPath();
848 QStorageInfo info(mountDir);
849 info.d->device = it.device();
850 info.d->fileSystemType = it.fileSystemType();
851 info.d->subvolume = it.subvolume();
852 if (info.bytesTotal() == 0 && info != root())
853 continue;
854 volumes.append(t: info);
855 }
856
857 return volumes;
858}
