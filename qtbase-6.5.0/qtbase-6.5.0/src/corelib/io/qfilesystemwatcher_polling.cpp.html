10QPollingFileSystemWatcherEngine::QPollingFileSystemWatcherEngine(QObject *parent)
11 : QFileSystemWatcherEngine(parent),
12 timer(this)
13{
14 connect(asender: &timer, SIGNAL(timeout()), SLOT(timeout()));
15}

17QStringList QPollingFileSystemWatcherEngine::addPaths(const QStringList &paths,
18 QStringList *files,
19 QStringList *directories)
20{
21 QStringList unhandled;
22 for (const QString &path : paths) {
23 auto sg = qScopeGuard(f: [&]{ unhandled.push_back(t: path); });
24 QFileInfo fi(path);
25 if (!fi.exists())
26 continue;
27 if (fi.isDir()) {
28 if (directories->contains(str: path))
29 continue;
30 directories->append(t: path);
31 if (!path.endsWith(c: u'/'))
32 fi = QFileInfo(path + u'/');
33 this->directories.insert(key: path, value: fi);
34 } else {
35 if (files->contains(str: path))
36 continue;
37 files->append(t: path);
38 this->files.insert(key: path, value: fi);
39 }
40 sg.dismiss();
41 }
42
43 if ((!this->files.isEmpty() ||
44 !this->directories.isEmpty()) &&
45 !timer.isActive()) {
46 timer.start(msec: PollingInterval);
47 }
48
49 return unhandled;
50}

52QStringList QPollingFileSystemWatcherEngine::removePaths(const QStringList &paths,
53 QStringList *files,
54 QStringList *directories)
55{
56 QStringList unhandled;
57 for (const QString &path : paths) {
58 if (this->directories.remove(key: path)) {
59 directories->removeAll(t: path);
60 } else if (this->files.remove(key: path)) {
61 files->removeAll(t: path);
62 } else {
63 unhandled.push_back(t: path);
64 }
65 }
66
67 if (this->files.isEmpty() &&
68 this->directories.isEmpty()) {
69 timer.stop();
70 }
71
72 return unhandled;
73}

75void QPollingFileSystemWatcherEngine::timeout()
76{
77 for (auto it = files.begin(), end = files.end(); it != end; /*erasing*/) {
78 QString path = it.key();
79 QFileInfo fi(path);
80 if (!fi.exists()) {
81 it = files.erase(it);
82 emit fileChanged(path, removed: true);
83 continue;
84 } else if (it.value() != fi) {
85 it.value() = fi;
86 emit fileChanged(path, removed: false);
87 }
88 ++it;
89 }
90
91 for (auto it = directories.begin(), end = directories.end(); it != end; /*erasing*/) {
92 QString path = it.key();
93 QFileInfo fi(path);
94 if (!path.endsWith(c: u'/'))
95 fi = QFileInfo(path + u'/');
96 if (!fi.exists()) {
97 it = directories.erase(it);
98 emit directoryChanged(path, removed: true);
99 continue;
100 } else if (it.value() != fi) {
101 fi.refresh();
102 if (!fi.exists()) {
103 it = directories.erase(it);
104 emit directoryChanged(path, removed: true);
105 continue;
106 } else {
107 it.value() = fi;
108 emit directoryChanged(path, removed: false);
109 }
110 }
111 ++it;
112 }
113}
