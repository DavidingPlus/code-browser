25/*
26 Returns a human readable representation of the first \a maxSize
27 characters in \a data. The size, \a len, is a 64-bit quantity to
28 avoid truncation due to implicit conversions in callers.
29*/
30QByteArray QtDebugUtils::toPrintable(const char *data, qint64 len, qsizetype maxSize)
31{
32 if (!data)
33 return "(null)";
34
35 QByteArray out;
36 for (qsizetype i = 0; i < qMin(a: len, b: maxSize); ++i) {
37 char c = data[i];
38 if (isprint(c)) {
39 out += c;
40 } else {
41 switch (c) {
42 case '\n':
43 out += "\\n";
44 break;
45 case '\r':
46 out += "\\r";
47 break;
48 case '\t':
49 out += "\\t";
50 break;
51 default: {
52 const char buf[] = {
53 '\\',
54 'x',
55 toHexLower(value: uchar(c) / 16),
56 toHexLower(value: uchar(c) % 16),
57 0
58 };
59 out += buf;
60 }
61 }
62 }
63 }
64
65 if (maxSize < len)
66 out += "...";
67
68 return out;
69}

73/*!
74 \class QDebug
75 \inmodule QtCore
76 \ingroup shared
77
78 \brief The QDebug class provides an output stream for debugging information.
79
80 QDebug is used whenever the developer needs to write out debugging or tracing
81 information to a device, file, string or console.
82
83 \section1 Basic Use
84
85 In the common case, it is useful to call the qDebug() function to obtain a
86 default QDebug object to use for writing debugging information.
87
88 \snippet qdebug/qdebugsnippet.cpp 1
89
90 This constructs a QDebug object using the constructor that accepts a QtMsgType
91 value of QtDebugMsg. Similarly, the qWarning(), qCritical() and qFatal()
92 functions also return QDebug objects for the corresponding message types.
93
94 The class also provides several constructors for other situations, including
95 a constructor that accepts a QFile or any other QIODevice subclass that is
96 used to write debugging information to files and other devices. The constructor
97 that accepts a QString is used to write to a string for display or serialization.
98
99 \section1 Formatting Options
100
101 QDebug formats output so that it's easily readable. It automatically adds spaces
102 between arguments, and adds quotes around QString, QByteArray, QChar arguments.
103
104 You can tweak these options through the space(), nospace() and quote(), noquote()
105 methods. Furthermore, \l{QTextStream manipulators} can be piped into a QDebug
106 stream.
107
108 QDebugStateSaver limits changes to the formatting to the current scope.
109 resetFormat() resets the options to the default ones.
110
111 \section1 Writing Custom Types to a Stream
112
113 Many standard types can be written to QDebug objects, and Qt provides support for
114 most Qt value types. To add support for custom types, you need to implement a
115 streaming operator, as in the following example:
116
117 \snippet qdebug/qdebugsnippet.cpp 0
118
119 This is described in the \l{Debugging Techniques} and
120 \l{Creating Custom Qt Types#Making the Type Printable}{Creating Custom Qt Types}
121 documents.
122*/

175/*!
176 \internal
177*/
178void QDebug::putUcs4(uint ucs4)
179{
180 maybeQuote(c: '\'');
181 if (ucs4 < 0x20) {
182 stream->ts << "\\x" << Qt::hex << ucs4 << Qt::reset;
183 } else if (ucs4 < 0x80) {
184 stream->ts << char(ucs4);
185 } else {
186 if (ucs4 < 0x10000)
187 stream->ts << "\\u" << qSetFieldWidth(width: 4);
188 else
189 stream->ts << "\\U" << qSetFieldWidth(width: 8);
190 stream->ts << Qt::hex << qSetPadChar(ch: u'0') << ucs4 << Qt::reset;
191 }
192 maybeQuote(c: '\'');
193}

195// These two functions return true if the character should be printed by QDebug.
196// For QByteArray, this is technically identical to US-ASCII isprint();
197// for QString, we use QChar::isPrint, which requires a full UCS-4 decode.
198static inline bool isPrintable(uint ucs4)
199{ return QChar::isPrint(ucs4); }
200static inline bool isPrintable(ushort uc)
201{ return QChar::isPrint(ucs4: uc); }
202static inline bool isPrintable(uchar c)
203{ return c >= ' ' && c < 0x7f; }
204
205template <typename Char>
206static inline void putEscapedString(QTextStreamPrivate *d, const Char *begin, size_t length, bool isUnicode = true)
207{
208 QChar quote(u'"');
209 d->write(data: &quote, len: 1);
210
211 bool lastWasHexEscape = false;
212 const Char *end = begin + length;
213 for (const Char *p = begin; p != end; ++p) {
214 // check if we need to insert "" to break an hex escape sequence
215 if (Q_UNLIKELY(lastWasHexEscape)) {
216 if (fromHex(*p) != -1) {
217 // yes, insert it
218 QChar quotes[] = { quote, quote };
219 d->write(data: quotes, len: 2);
220 }
221 lastWasHexEscape = false;
222 }
223
224 if (sizeof(Char) == sizeof(QChar)) {
225 // Surrogate characters are category Cs (Other_Surrogate), so isPrintable = false for them
226 qsizetype runLength = 0;
227 while (p + runLength != end &&
228 isPrintable(p[runLength]) && p[runLength] != '\\' && p[runLength] != '"')
229 ++runLength;
230 if (runLength) {
231 d->write(data: reinterpret_cast<const QChar *>(p), len: runLength);
232 p += runLength - 1;
233 continue;
234 }
235 } else if (isPrintable(*p) && *p != '\\' && *p != '"') {
236 QChar c = QLatin1Char(*p);
237 d->write(data: &c, len: 1);
238 continue;
239 }
240
241 // print as an escape sequence (maybe, see below for surrogate pairs)
242 qsizetype buflen = 2;
243 ushort buf[sizeof "\\U12345678" - 1];
244 buf[0] = '\\';
245
246 switch (*p) {
247 case '"':
248 case '\\':
249 buf[1] = *p;
250 break;
251 case '\b':
252 buf[1] = 'b';
253 break;
254 case '\f':
255 buf[1] = 'f';
256 break;
257 case '\n':
258 buf[1] = 'n';
259 break;
260 case '\r':
261 buf[1] = 'r';
262 break;
263 case '\t':
264 buf[1] = 't';
265 break;
266 default:
267 if (!isUnicode) {
268 // print as hex escape
269 buf[1] = 'x';
270 buf[2] = toHexUpper(value: uchar(*p) >> 4);
271 buf[3] = toHexUpper(value: uchar(*p));
272 buflen = 4;
273 lastWasHexEscape = true;
274 break;
275 }
276 if (QChar::isHighSurrogate(*p)) {
277 if ((p + 1) != end && QChar::isLowSurrogate(p[1])) {
278 // properly-paired surrogates
279 uint ucs4 = QChar::surrogateToUcs4(*p, p[1]);
280 if (isPrintable(ucs4)) {
281 buf[0] = *p;
282 buf[1] = p[1];
283 buflen = 2;
284 } else {
285 buf[1] = 'U';
286 buf[2] = '0'; // toHexUpper(ucs4 >> 32);
287 buf[3] = '0'; // toHexUpper(ucs4 >> 28);
288 buf[4] = toHexUpper(value: ucs4 >> 20);
289 buf[5] = toHexUpper(value: ucs4 >> 16);
290 buf[6] = toHexUpper(value: ucs4 >> 12);
291 buf[7] = toHexUpper(value: ucs4 >> 8);
292 buf[8] = toHexUpper(value: ucs4 >> 4);
293 buf[9] = toHexUpper(value: ucs4);
294 buflen = 10;
295 }
296 ++p;
297 break;
298 }
299 // improperly-paired surrogates, fall through
300 }
301 buf[1] = 'u';
302 buf[2] = toHexUpper(value: ushort(*p) >> 12);
303 buf[3] = toHexUpper(value: ushort(*p) >> 8);
304 buf[4] = toHexUpper(*p >> 4);
305 buf[5] = toHexUpper(*p);
306 buflen = 6;
307 }
308 d->write(data: reinterpret_cast<QChar *>(buf), len: buflen);
309 }
310
311 d->write(data: &quote, len: 1);
312}

314/*!
315 \internal
316 Duplicated from QtTest::toPrettyUnicode().
317*/
318void QDebug::putString(const QChar *begin, size_t length)
319{
320 if (stream->noQuotes) {
321 // no quotes, write the string directly too (no pretty-printing)
322 // this respects the QTextStream state, though
323 stream->ts.d_ptr->putString(data: begin, len: qsizetype(length));
324 } else {
325 // we'll reset the QTextStream formatting mechanisms, so save the state
326 QDebugStateSaver saver(*this);
327 stream->ts.d_ptr->params.reset();
328 putEscapedString(d: stream->ts.d_ptr.data(), begin: reinterpret_cast<const ushort *>(begin), length);
329 }
330}

332/*!
333 \internal
334 Duplicated from QtTest::toPrettyCString().
335*/
336void QDebug::putByteArray(const char *begin, size_t length, Latin1Content content)
337{
338 if (stream->noQuotes) {
339 // no quotes, write the string directly too (no pretty-printing)
340 // this respects the QTextStream state, though
341 QString string = content == ContainsLatin1 ? QString::fromLatin1(str: begin, size: qsizetype(length))
342 : QString::fromUtf8(utf8: begin, size: qsizetype(length));
343 stream->ts.d_ptr->putString(string);
344 } else {
345 // we'll reset the QTextStream formatting mechanisms, so save the state
346 QDebugStateSaver saver(*this);
347 stream->ts.d_ptr->params.reset();
348 putEscapedString(d: stream->ts.d_ptr.data(), begin: reinterpret_cast<const uchar *>(begin),
349 length, isUnicode: content == ContainsLatin1);
350 }
351}

932/*!
933 \class QDebugStateSaver
934 \inmodule QtCore
935 \brief Convenience class for custom QDebug operators.
936
937 Saves the settings used by QDebug, and restores them upon destruction,
938 then calls \l {QDebug::maybeSpace()}{maybeSpace()}, to separate arguments with a space if
939 \l {QDebug::autoInsertSpaces()}{autoInsertSpaces()} was true at the time of constructing the QDebugStateSaver.
940
941 The automatic insertion of spaces between writes is one of the settings
942 that QDebugStateSaver stores for the duration of the current block.
943
944 The settings of the internal QTextStream are also saved and restored,
945 so that using << Qt::hex in a QDebug operator doesn't affect other QDebug
946 operators.
947
948 QDebugStateSaver is typically used in the implementation of an operator<<() for debugging:
949
950 \snippet tools/customtype/message.cpp custom type streaming operator
951
952 \since 5.1
953*/
954
955class QDebugStateSaverPrivate
956{
957public:
958 QDebugStateSaverPrivate(QDebug::Stream *stream)
959 : m_stream(stream),
960 m_spaces(stream->space),
961 m_noQuotes(stream->noQuotes),
962 m_verbosity(stream->verbosity),
963 m_streamParams(stream->ts.d_ptr->params)
964 {
965 }
966 void restoreState()
967 {
968 const bool currentSpaces = m_stream->space;
969 if (currentSpaces && !m_spaces)
970 if (m_stream->buffer.endsWith(c: u' '))
971 m_stream->buffer.chop(n: 1);
972
973 m_stream->space = m_spaces;
974 m_stream->noQuotes = m_noQuotes;
975 m_stream->ts.d_ptr->params = m_streamParams;
976 m_stream->verbosity = m_verbosity;
977
978 if (!currentSpaces && m_spaces)
979 m_stream->ts << ' ';
980 }
981
982 QDebug::Stream *m_stream;
983
984 // QDebug state
985 const bool m_spaces;
986 const bool m_noQuotes;
987 const int m_verbosity;
988
989 // QTextStream state
990 const QTextStreamPrivate::Params m_streamParams;
991};

1031 \fn QDebug qt_QMetaEnum_debugOperator(QDebug &, int value, const QMetaObject *, const char *name)
1032 \internal
1033
1034 Formats the given enum \a value for debug output.
1035
1036 The supported verbosity are:
1037
1038 0: Just the key, or value with enum name if no key is found:
1039
1040 MyEnum2
1041 MyEnum(123)
1042 MyScopedEnum::Enum3
1043 MyScopedEnum(456)
1044
1045 1: Same as 0, but treating all enums as scoped:
1046
1047 MyEnum::MyEnum2
1048 MyEnum(123)
1049 MyScopedEnum::Enum3
1050 MyScopedEnum(456)
1051
1052 2: The QDebug default. Same as 0, and includes class/namespace scope:
1053
1054 MyNamespace::MyClass::MyEnum2
1055 MyNamespace::MyClass::MyEnum(123)
1056 MyNamespace::MyClass::MyScopedEnum::Enum3
1057 MyNamespace::MyClass::MyScopedEnum(456)
1058
1059 3: Same as 2, but treating all enums as scoped:
1060
1061 MyNamespace::MyClass::MyEnum::MyEnum2
1062 MyNamespace::MyClass::MyEnum(123)
1063 MyNamespace::MyClass::MyScopedEnum::Enum3
1064 MyNamespace::MyClass::MyScopedEnum(456)
1065 */
1066QDebug qt_QMetaEnum_debugOperator(QDebug &dbg, qint64 value, const QMetaObject *meta, const char *name)
1067{
1068 QDebugStateSaver saver(dbg);
1069 dbg.nospace();
1070 QMetaEnum me = meta->enumerator(meta->indexOfEnumerator(name));
1071
1072 const int verbosity = dbg.verbosity();
1073 if (verbosity >= QDebug::DefaultVerbosity) {
1074 if (const char *scope = me.scope())
1075 dbg << scope << u"::";
1076 }
1077
1078 const char *key = me.valueToKey(value);
1079 const bool scoped = me.isScoped() || verbosity & 1;
1080 if (scoped || !key)
1081 dbg << me.enumName() << (!key ? u"(" : u"::");
1082
1083 if (key)
1084 dbg << key;
1085 else
1086 dbg << value << ')';
1087
1088 return dbg;
1089}

1091/*!
1092 \fn QDebug qt_QMetaEnum_flagDebugOperator(QDebug &, quint64 value, const QMetaObject *, const char *name)
1093 \internal
1094
1095 Formats the given flag \a value for debug output.
1096
1097 The supported verbosity are:
1098
1099 0: Just the key(s):
1100
1101 MyFlag1
1102 MyFlag2|MyFlag3
1103 MyScopedFlag(MyFlag2)
1104 MyScopedFlag(MyFlag2|MyFlag3)
1105
1106 1: Same as 0, but treating all flags as scoped:
1107
1108 MyFlag(MyFlag1)
1109 MyFlag(MyFlag2|MyFlag3)
1110 MyScopedFlag(MyFlag2)
1111 MyScopedFlag(MyFlag2|MyFlag3)
1112
1113 2: The QDebug default. Same as 1, and includes class/namespace scope:
1114
1115 QFlags<MyNamespace::MyClass::MyFlag>(MyFlag1)
1116 QFlags<MyNamespace::MyClass::MyFlag>(MyFlag2|MyFlag3)
1117 QFlags<MyNamespace::MyClass::MyScopedFlag>(MyFlag2)
1118 QFlags<MyNamespace::MyClass::MyScopedFlag>(MyFlag2|MyFlag3)
1119 */
1120QDebug qt_QMetaEnum_flagDebugOperator(QDebug &debug, quint64 value, const QMetaObject *meta, const char *name)
1121{
1122 const int verbosity = debug.verbosity();
1123
1124 QDebugStateSaver saver(debug);
1125 debug.resetFormat();
1126 debug.noquote();
1127 debug.nospace();
1128
1129 const QMetaEnum me = meta->enumerator(meta->indexOfEnumerator(name));
1130
1131 const bool classScope = verbosity >= QDebug::DefaultVerbosity;
1132 if (classScope) {
1133 debug << u"QFlags<";
1134
1135 if (const char *scope = me.scope())
1136 debug << scope << u"::";
1137 }
1138
1139 const bool enumScope = me.isScoped() || verbosity > QDebug::MinimumVerbosity;
1140 if (enumScope) {
1141 debug << me.enumName();
1142 if (classScope)
1143 debug << '>';
1144 debug << '(';
1145 }
1146
1147 debug << me.valueToKeys(value);
1148
1149 if (enumScope)
1150 debug << ')';
1151
1152 return debug;
1153}
