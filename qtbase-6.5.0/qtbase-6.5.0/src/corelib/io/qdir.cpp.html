58// Return the length of the root part of an absolute path, for use by cleanPath(), cd().
59static qsizetype rootLength(QStringView name, bool allowUncPaths)
60{
61 const qsizetype len = name.size();
62 // starts with double slash
63 if (allowUncPaths && name.startsWith(s: "//"_L1)) {
64 // Server name '//server/path' is part of the prefix.
65 const qsizetype nextSlash = name.indexOf(c: u'/', from: 2);
66 return nextSlash >= 0 ? nextSlash + 1 : len;
67 }
68#if defined(Q_OS_WIN)
69 if (len >= 2 && name.at(1) == u':') {
70 // Handle a possible drive letter
71 return len > 2 && name.at(2) == u'/' ? 3 : 2;
72 }
73#endif
74 if (name.at(n: 0) == u'/')
75 return 1;
76 return 0;
77}

199{
200 QDir::SortFlags qt_cmp_si_sort_flags;
201public:
202 QDirSortItemComparator(QDir::SortFlags flags) : qt_cmp_si_sort_flags(flags) {}
203 bool operator()(const QDirSortItem &, const QDirSortItem &) const;
204};

206bool QDirSortItemComparator::operator()(const QDirSortItem &n1, const QDirSortItem &n2) const
207{
208 const QDirSortItem* f1 = &n1;
209 const QDirSortItem* f2 = &n2;
210
211 if ((qt_cmp_si_sort_flags & QDir::DirsFirst) && (f1->item.isDir() != f2->item.isDir()))
212 return f1->item.isDir();
213 if ((qt_cmp_si_sort_flags & QDir::DirsLast) && (f1->item.isDir() != f2->item.isDir()))
214 return !f1->item.isDir();
215
216 qint64 r = 0;
217 int sortBy = ((qt_cmp_si_sort_flags & QDir::SortByMask)
218 | (qt_cmp_si_sort_flags & QDir::Type)).toInt();
219
220 switch (sortBy) {
221 case QDir::Time: {
222 QDateTime firstModified = f1->item.lastModified();
223 QDateTime secondModified = f2->item.lastModified();
224
225 // QDateTime by default will do all sorts of conversions on these to
226 // find timezones, which is incredibly expensive. As we aren't
227 // presenting these to the user, we don't care (at all) about the
228 // local timezone, so force them to UTC to avoid that conversion.
229 firstModified.setTimeZone(QTimeZone::UTC);
230 secondModified.setTimeZone(QTimeZone::UTC);
231
232 r = firstModified.msecsTo(secondModified);
233 break;
234 }
235 case QDir::Size:
236 r = f2->item.size() - f1->item.size();
237 break;
238 case QDir::Type:
239 {
240 bool ic = qt_cmp_si_sort_flags.testAnyFlag(flag: QDir::IgnoreCase);
241
242 if (f1->suffix_cache.isNull())
243 f1->suffix_cache = ic ? f1->item.suffix().toLower()
244 : f1->item.suffix();
245 if (f2->suffix_cache.isNull())
246 f2->suffix_cache = ic ? f2->item.suffix().toLower()
247 : f2->item.suffix();
248
249 r = qt_cmp_si_sort_flags & QDir::LocaleAware
250 ? f1->suffix_cache.localeAwareCompare(s: f2->suffix_cache)
251 : f1->suffix_cache.compare(s: f2->suffix_cache);
252 }
253 break;
254 default:
255 ;
256 }
257
258 if (r == 0 && sortBy != QDir::Unsorted) {
259 // Still not sorted - sort by name
260 bool ic = qt_cmp_si_sort_flags.testAnyFlag(flag: QDir::IgnoreCase);
261
262 if (f1->filename_cache.isNull())
263 f1->filename_cache = ic ? f1->item.fileName().toLower()
264 : f1->item.fileName();
265 if (f2->filename_cache.isNull())
266 f2->filename_cache = ic ? f2->item.fileName().toLower()
267 : f2->item.fileName();
268
269 r = qt_cmp_si_sort_flags & QDir::LocaleAware
270 ? f1->filename_cache.localeAwareCompare(s: f2->filename_cache)
271 : f1->filename_cache.compare(s: f2->filename_cache);
272 }
273 if (qt_cmp_si_sort_flags & QDir::Reversed)
274 return r > 0;
275 return r < 0;
276}

278inline void QDirPrivate::sortFileList(QDir::SortFlags sort, const QFileInfoList &l,
279 QStringList *names, QFileInfoList *infos)
280{
281 // names and infos are always empty lists or 0 here
282 qsizetype n = l.size();
283 if (n > 0) {
284 if (n == 1 || (sort & QDir::SortByMask) == QDir::Unsorted) {
285 if (infos)
286 *infos = l;
287 if (names) {
288 for (const QFileInfo &fi : l)
289 names->append(t: fi.fileName());
290 }
291 } else {
292 QScopedArrayPointer<QDirSortItem> si(new QDirSortItem[n]);
293 for (qsizetype i = 0; i < n; ++i)
294 si[i].item = l.at(i);
295 std::sort(first: si.data(), last: si.data() + n, comp: QDirSortItemComparator(sort));
296 // put them back in the list(s)
297 if (infos) {
298 for (qsizetype i = 0; i < n; ++i)
299 infos->append(t: si[i].item);
300 }
301 if (names) {
302 for (qsizetype i = 0; i < n; ++i)
303 names->append(t: si[i].item.fileName());
304 }
305 }
306 }

325/*!
326 \class QDir
327 \inmodule QtCore
328 \brief The QDir class provides access to directory structures and their contents.
329
330 \ingroup io
331 \ingroup shared
332 \reentrant
333
334
335 A QDir is used to manipulate path names, access information
336 regarding paths and files, and manipulate the underlying file
337 system. It can also be used to access Qt's \l{resource system}.
338
339 Qt uses "/" as a universal directory separator in the same way
340 that "/" is used as a path separator in URLs. If you always use
341 "/" as a directory separator, Qt will translate your paths to
342 conform to the underlying operating system.
343
344 A QDir can point to a file using either a relative or an absolute
345 path. Absolute paths begin with the directory separator
346 (optionally preceded by a drive specification under Windows).
347 Relative file names begin with a directory name or a file name and
348 specify a path relative to the current directory.
349
350 Examples of absolute paths:
351
352 \snippet code/src_corelib_io_qdir.cpp 0
353
354 On Windows, the second example above will be translated to
355 \c{C:\Users} when used to access files.
356
357 Examples of relative paths:
358
359 \snippet code/src_corelib_io_qdir.cpp 1
360
361 You can use the isRelative() or isAbsolute() functions to check if
362 a QDir is using a relative or an absolute file path. Call
363 makeAbsolute() to convert a relative QDir to an absolute one.
364
365 \note Paths starting with a colon (\e{:}) are always considered
366 absolute, as they denote a QResource.
367
368 \section1 Navigation and Directory Operations
369
370 A directory's path can be obtained with the path() function, and
371 a new path set with the setPath() function. The absolute path to
372 a directory is found by calling absolutePath().
373
374 The name of a directory is found using the dirName() function. This
375 typically returns the last element in the absolute path that specifies
376 the location of the directory. However, it can also return "." if
377 the QDir represents the current directory.
378
379 \snippet code/src_corelib_io_qdir.cpp 2
380
381 The path for a directory can also be changed with the cd() and cdUp()
382 functions, both of which operate like familiar shell commands.
383 When cd() is called with the name of an existing directory, the QDir
384 object changes directory so that it represents that directory instead.
385 The cdUp() function changes the directory of the QDir object so that
386 it refers to its parent directory; i.e. cd("..") is equivalent to
387 cdUp().
388
389 Directories can be created with mkdir(), renamed with rename(), and
390 removed with rmdir().
391
392 You can test for the presence of a directory with a given name by
393 using exists(), and the properties of a directory can be tested with
394 isReadable(), isAbsolute(), isRelative(), and isRoot().
395
396 The refresh() function re-reads the directory's data from disk.
397
398 \section1 Files and Directory Contents
399
400 Directories contain a number of entries, representing files,
401 directories, and symbolic links. The number of entries in a
402 directory is returned by count().
403 A string list of the names of all the entries in a directory can be
404 obtained with entryList(). If you need information about each
405 entry, use entryInfoList() to obtain a list of QFileInfo objects.
406
407 Paths to files and directories within a directory can be
408 constructed using filePath() and absoluteFilePath().
409 The filePath() function returns a path to the specified file
410 or directory relative to the path of the QDir object;
411 absoluteFilePath() returns an absolute path to the specified
412 file or directory. Neither of these functions checks for the
413 existence of files or directory; they only construct paths.
414
415 \snippet code/src_corelib_io_qdir.cpp 3
416
417 Files can be removed by using the remove() function. Directories
418 cannot be removed in the same way as files; use rmdir() to remove
419 them instead.
420
421 It is possible to reduce the number of entries returned by
422 entryList() and entryInfoList() by applying filters to a QDir object.
423 You can apply a name filter to specify a pattern with wildcards that
424 file names need to match, an attribute filter that selects properties
425 of entries and can distinguish between files and directories, and a
426 sort order.
427
428 Name filters are lists of strings that are passed to setNameFilters().
429 Attribute filters consist of a bitwise OR combination of Filters, and
430 these are specified when calling setFilter().
431 The sort order is specified using setSorting() with a bitwise OR
432 combination of SortFlags.
433
434 You can test to see if a filename matches a filter using the match()
435 function.
436
437 Filter and sort order flags may also be specified when calling
438 entryList() and entryInfoList() in order to override previously defined
439 behavior.
440
441 \section1 The Current Directory and Other Special Paths
442
443 Access to some common directories is provided with a number of static
444 functions that return QDir objects. There are also corresponding functions
445 for these that return strings:
446
447 \table
448 \header \li QDir \li QString \li Return Value
449 \row \li current() \li currentPath() \li The application's working directory
450 \row \li home() \li homePath() \li The user's home directory
451 \row \li root() \li rootPath() \li The root directory
452 \row \li temp() \li tempPath() \li The system's temporary directory
453 \endtable
454
455 The setCurrent() static function can also be used to set the application's
456 working directory.
457
458 If you want to find the directory containing the application's executable,
459 see \l{QCoreApplication::applicationDirPath()}.
460
461 The drives() static function provides a list of root directories for each
462 device that contains a filing system. On Unix systems this returns a list
463 containing a single root directory "/"; on Windows the list will usually
464 contain \c{C:/}, and possibly other drive letters such as \c{D:/}, depending
465 on the configuration of the user's system.
466
467 \section1 Path Manipulation and Strings
468
469 Paths containing "." elements that reference the current directory at that
470 point in the path, ".." elements that reference the parent directory, and
471 symbolic links can be reduced to a canonical form using the canonicalPath()
472 function.
473
474 Paths can also be simplified by using cleanPath() to remove redundant "/"
475 and ".." elements.
476
477 It is sometimes necessary to be able to show a path in the native
478 representation for the user's platform. The static toNativeSeparators()
479 function returns a copy of the specified path in which each directory
480 separator is replaced by the appropriate separator for the underlying
481 operating system.
482

505 \sa QFileInfo, QFile, QFileDialog, QCoreApplication::applicationDirPath(),
506 {Fetch More Example}
507*/

679#ifdef Q_OS_WIN
680static qsizetype drivePrefixLength(QStringView path)
681{
682 // Used to extract path's drive for use as prefix for an "absolute except for drive" path
683 const qsizetype size = path.size();
684 qsizetype drive = 2; // length of drive prefix
685 if (size > 1 && path.at(1).unicode() == ':') {
686 if (Q_UNLIKELY(!path.at(0).isLetter()))
687 return 0;
688 } else if (path.startsWith("//"_L1)) {
689 // UNC path; use its //server/share part as "drive" - it's as sane a
690 // thing as we can do.
691 for (int i = 0 ; i < 2 ; ++i) { // Scan two "path fragments":
692 while (drive < size && path.at(drive).unicode() == '/')
693 drive++;
694 if (drive >= size) {
695 qWarning("Base directory starts with neither a drive nor a UNC share: %s",
696 qUtf8Printable(QDir::toNativeSeparators(path.toString())));
697 return 0;
698 }
699 while (drive < size && path.at(drive).unicode() != '/')
700 drive++;
701 }
702 } else {
703 return 0;
704 }
705 return drive;
706}

709static bool treatAsAbsolute(const QString &path)
710{
711 // ### Qt 6: be consistent about absolute paths
712
713 // QFileInfo will use the right FS-engine for virtual file-systems
714 // (e.g. resource paths). Unfortunately, for real file-systems, it relies
715 // on QFileSystemEntry's isRelative(), which is flawed on MS-Win, ignoring
716 // its (correct) isAbsolute(). So only use that isAbsolute() unless there's
717 // a colon in the path.
718 // FIXME: relies on virtual file-systems having colons in their prefixes.
