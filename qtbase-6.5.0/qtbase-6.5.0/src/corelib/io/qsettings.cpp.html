226/*
227 Returns a string that never starts nor ends with a slash (or an
228 empty string). Examples:
229
230 "foo" becomes "foo"
231 "/foo//bar///" becomes "foo/bar"
232 "///" becomes ""
233*/
234QString QSettingsPrivate::normalizedKey(QAnyStringView key)
235{
236 QString result(key.size(), Qt::Uninitialized);
237 auto out = const_cast<QChar*>(result.constData()); // don't detach
238
239 const bool maybeEndsInSlash = key.visit(v: [&out](auto key) {
240 using View = decltype(key);
241
242 auto it = key.begin();
243 const auto end = key.end();
244
245 while (it != end) {
246 while (*it == u'/') {
247 ++it;
248 if (it == end)
249 return true;
250 }
251 auto mark = it;
252 while (*it != u'/') {
253 ++it;
254 if (it == end)
255 break;
256 }
257 out = write(out, View{mark, it});
258 if (it == end)
259 return false;
260 Q_ASSERT(*it == u'/');
261 *out++ = u'/';
262 ++it;
263 }
264 return true;
265 });
266
267 if (maybeEndsInSlash && out != result.constData())
268 --out; // remove the trailing slash
269 result.truncate(pos: out - result.constData());
270 return result;
271}

373QString QSettingsPrivate::variantToString(const QVariant &v)
374{
375 QString result;
376
377 switch (v.metaType().id()) {
378 case QMetaType::UnknownType:
379 result = "@Invalid()"_L1;
380 break;
381
382 case QMetaType::QByteArray: {
383 QByteArray a = v.toByteArray();
384 result = "@ByteArray("_L1 + QLatin1StringView(a) + u')';
385 break;
386 }
387
388#if QT_CONFIG(shortcut)
389 case QMetaType::QKeySequence:
390#endif
391 case QMetaType::QString:
392 case QMetaType::LongLong:
393 case QMetaType::ULongLong:
394 case QMetaType::Int:
395 case QMetaType::UInt:
396 case QMetaType::Bool:
397 case QMetaType::Float:
398 case QMetaType::Double: {
399 result = v.toString();
400 if (result.contains(c: QChar::Null))
401 result = "@String("_L1 + result + u')';
402 else if (result.startsWith(c: u'@'))
403 result.prepend(c: u'@');
404 break;
405 }
406#ifndef QT_NO_GEOM_VARIANT
407 case QMetaType::QRect: {
408 QRect r = qvariant_cast<QRect>(v);
409 result = QString::asprintf(format: "@Rect(%d %d %d %d)", r.x(), r.y(), r.width(), r.height());
410 break;
411 }
412 case QMetaType::QSize: {
413 QSize s = qvariant_cast<QSize>(v);
414 result = QString::asprintf(format: "@Size(%d %d)", s.width(), s.height());
415 break;
416 }
417 case QMetaType::QPoint: {
418 QPoint p = qvariant_cast<QPoint>(v);
419 result = QString::asprintf(format: "@Point(%d %d)", p.x(), p.y());
420 break;
421 }
422#endif // !QT_NO_GEOM_VARIANT
423
424 default: {
425#ifndef QT_NO_DATASTREAM
426 QDataStream::Version version;
427 const char *typeSpec;
428 if (v.userType() == QMetaType::QDateTime) {
429 version = QDataStream::Qt_5_6;
430 typeSpec = "@DateTime(";
431 } else {
432 version = QDataStream::Qt_4_0;
433 typeSpec = "@Variant(";
434 }
435 QByteArray a;
436 {
437 QDataStream s(&a, QIODevice::WriteOnly);
438 s.setVersion(version);
439 s << v;
440 }
441
442 result = QLatin1StringView(typeSpec)
443 + QLatin1StringView(a.constData(), a.size())
444 + u')';
445#else
446 Q_ASSERT(!"QSettings: Cannot save custom types without QDataStream support");
447#endif
448 break;
449 }
450 }
451
452 return result;
453}

456QVariant QSettingsPrivate::stringToVariant(const QString &s)
457{
458 if (s.startsWith(c: u'@')) {
459 if (s.endsWith(c: u')')) {
460 if (s.startsWith(s: "@ByteArray("_L1)) {
461 return QVariant(QStringView{s}.sliced(pos: 11).chopped(n: 1).toLatin1());
462 } else if (s.startsWith(s: "@String("_L1)) {
463 return QVariant(QStringView{s}.sliced(pos: 8).chopped(n: 1).toString());
464 } else if (s.startsWith(s: "@Variant("_L1)
465 || s.startsWith(s: "@DateTime("_L1)) {
466#ifndef QT_NO_DATASTREAM
467 QDataStream::Version version;
468 int offset;
469 if (s.at(i: 1) == u'D') {
470 version = QDataStream::Qt_5_6;
471 offset = 10;
472 } else {
473 version = QDataStream::Qt_4_0;
474 offset = 9;
475 }
476 QByteArray a = QStringView{s}.sliced(pos: offset).toLatin1();
477 QDataStream stream(&a, QIODevice::ReadOnly);
478 stream.setVersion(version);
479 QVariant result;
480 stream >> result;
481 return result;
482#else
483 Q_ASSERT(!"QSettings: Cannot load custom types without QDataStream support");
484#endif
485#ifndef QT_NO_GEOM_VARIANT
486 } else if (s.startsWith(s: "@Rect("_L1)) {
487 QStringList args = QSettingsPrivate::splitArgs(s, idx: 5);
488 if (args.size() == 4)
489 return QVariant(QRect(args[0].toInt(), args[1].toInt(), args[2].toInt(), args[3].toInt()));
490 } else if (s.startsWith(s: "@Size("_L1)) {
491 QStringList args = QSettingsPrivate::splitArgs(s, idx: 5);
492 if (args.size() == 2)
493 return QVariant(QSize(args[0].toInt(), args[1].toInt()));
494 } else if (s.startsWith(s: "@Point("_L1)) {
495 QStringList args = QSettingsPrivate::splitArgs(s, idx: 6);
496 if (args.size() == 2)
497 return QVariant(QPoint(args[0].toInt(), args[1].toInt()));
498#endif
499 } else if (s == "@Invalid()"_L1) {
500 return QVariant();
501 }
502
503 }
504 if (s.startsWith(s: "@@"_L1))
505 return QVariant(s.sliced(pos: 1));
506 }
507
508 return QVariant(s);
509}

511void QSettingsPrivate::iniEscapedKey(const QString &key, QByteArray &result)
512{
513 result.reserve(asize: result.size() + key.size() * 3 / 2);
514 for (qsizetype i = 0; i < key.size(); ++i) {
515 uint ch = key.at(i).unicode();
516
517 if (ch == '/') {
518 result += '\\';
519 } else if (isAsciiLetterOrNumber(c: ch) || ch == '_' || ch == '-' || ch == '.') {
520 result += (char)ch;
521 } else if (ch <= 0xFF) {
522 result += '%';
523 result += QtMiscUtils::toHexUpper(value: ch / 16);
524 result += QtMiscUtils::toHexUpper(value: ch % 16);
525 } else {
526 result += "%U";
527 QByteArray hexCode;
528 for (int j = 0; j < 4; ++j) {
529 hexCode.prepend(c: QtMiscUtils::toHexUpper(value: ch % 16));
530 ch >>= 4;
531 }
532 result += hexCode;
533 }
534 }

537bool QSettingsPrivate::iniUnescapedKey(QByteArrayView key, QString &result)
538{
539 const QString decoded = QString::fromUtf8(utf8: key);
540 const qsizetype size = decoded.size();
541 result.reserve(asize: result.size() + size);
542 qsizetype i = 0;
543 bool lowercaseOnly = true;
544 while (i < size) {
545 char16_t ch = decoded.at(i).unicode();
546
547 if (ch == '\\') {
548 result += u'/';
549 ++i;
550 continue;
551 }
552
553 if (ch != '%' || i == size - 1) {
554 QChar qch(ch);
555 if (qch.isUpper())
556 lowercaseOnly = false;
557 result += qch;
558 ++i;
559 continue;
560 }
561
562 int numDigits = 2;
563 int firstDigitPos = i + 1;
564
565 ch = decoded.at(i: i + 1).unicode();
566 if (ch == 'U') {
567 ++firstDigitPos;
568 numDigits = 4;
569 }
570
571 if (firstDigitPos + numDigits > size) {
572 result += u'%';
573 ++i;
574 continue;
575 }
576
577 bool ok;
578 ch = QStringView(decoded).sliced(pos: firstDigitPos, n: numDigits).toUShort(ok: &ok, base: 16);
579 if (!ok) {
580 result += u'%';
581 ++i;
582 continue;
583 }
584
585 QChar qch(ch);
586 if (qch.isUpper())
587 lowercaseOnly = false;
588 result += qch;
589 i = firstDigitPos + numDigits;
590 }
591 return lowercaseOnly;
592}

594void QSettingsPrivate::iniEscapedString(const QString &str, QByteArray &result)
595{
596 bool needsQuotes = false;
597 bool escapeNextIfDigit = false;
598 const bool useCodec = !(str.startsWith(s: "@ByteArray("_L1)
599 || str.startsWith(s: "@Variant("_L1)
600 || str.startsWith(s: "@DateTime("_L1));
601 const qsizetype startPos = result.size();
602
603 QStringEncoder toUtf8(QStringEncoder::Utf8);
604
605 result.reserve(asize: startPos + str.size() * 3 / 2);
606 for (QChar qch : str) {
607 uint ch = qch.unicode();
608 if (ch == ';' || ch == ',' || ch == '=')
609 needsQuotes = true;
610
611 if (escapeNextIfDigit && isHexDigit(c: ch)) {
612 result += "\\x" + QByteArray::number(ch, base: 16);
613 continue;
614 }
615
616 escapeNextIfDigit = false;
617
618 switch (ch) {
619 case '\0':
620 result += "\\0";
621 escapeNextIfDigit = true;
622 break;
623 case '\a':
624 result += "\\a";
625 break;
626 case '\b':
627 result += "\\b";
628 break;
629 case '\f':
630 result += "\\f";
631 break;
632 case '\n':
633 result += "\\n";
634 break;
635 case '\r':
636 result += "\\r";
637 break;
638 case '\t':
639 result += "\\t";
640 break;
641 case '\v':
642 result += "\\v";
643 break;
644 case '"':
645 case '\\':
646 result += '\\';
647 result += (char)ch;
648 break;
649 default:
650 if (ch <= 0x1F || (ch >= 0x7F && !useCodec)) {
651 result += "\\x" + QByteArray::number(ch, base: 16);
652 escapeNextIfDigit = true;
653 } else if (useCodec) {
654 // slow
655 result += toUtf8(qch);
656 } else {
657 result += (char)ch;
658 }
659 }
660 }
661
662 if (needsQuotes
663 || (startPos < result.size() && (result.at(i: startPos) == ' '
664 || result.at(i: result.size() - 1) == ' '))) {
665 result.insert(i: startPos, c: '"');
666 result += '"';
667 }
668}

699bool QSettingsPrivate::iniUnescapedStringList(QByteArrayView str,
700 QString &stringResult, QStringList &stringListResult)
701{
702 static const char escapeCodes[][2] =
703 {
704 { 'a', '\a' },
705 { 'b', '\b' },
706 { 'f', '\f' },
707 { 'n', '\n' },
708 { 'r', '\r' },
709 { 't', '\t' },
710 { 'v', '\v' },
711 { '"', '"' },
712 { '?', '?' },
713 { '\'', '\'' },
714 { '\\', '\\' }
715 };
716
717 bool isStringList = false;
718 bool inQuotedString = false;
719 bool currentValueIsQuoted = false;
720 char16_t escapeVal = 0;
721 qsizetype i = 0;
722 char ch;
723 QStringDecoder fromUtf8(QStringDecoder::Utf8);
724
725StSkipSpaces:
726 while (i < str.size() && ((ch = str.at(n: i)) == ' ' || ch == '\t'))
727 ++i;
728 // fallthrough
729
730StNormal:
731 qsizetype chopLimit = stringResult.size();
732 while (i < str.size()) {
733 switch (str.at(n: i)) {
734 case '\\':
735 ++i;
736 if (i >= str.size())
737 goto end;
738
739 ch = str.at(n: i++);
740 for (const auto &escapeCode : escapeCodes) {
741 if (ch == escapeCode[0]) {
742 stringResult += QLatin1Char(escapeCode[1]);
743 goto StNormal;
744 }
745 }
746
747 if (ch == 'x') {
748 escapeVal = 0;
749
750 if (i >= str.size())
751 goto end;
752
753 ch = str.at(n: i);
754 if (isHexDigit(c: ch))
755 goto StHexEscape;
756 } else if (isOctalDigit(c: ch)) {
757 escapeVal = ch - '0';
758 goto StOctEscape;
759 } else if (ch == '\n' || ch == '\r') {
760 if (i < str.size()) {
761 char ch2 = str.at(n: i);
762 // \n, \r, \r\n, and \n\r are legitimate line terminators in INI files
763 if ((ch2 == '\n' || ch2 == '\r') && ch2 != ch)
764 ++i;
765 }
766 } else {
767 // the character is skipped
768 }
769 chopLimit = stringResult.size();
770 break;
771 case '"':
772 ++i;
773 currentValueIsQuoted = true;
774 inQuotedString = !inQuotedString;
775 if (!inQuotedString)
776 goto StSkipSpaces;
777 break;
778 case ',':
779 if (!inQuotedString) {
780 if (!currentValueIsQuoted)
781 iniChopTrailingSpaces(str&: stringResult, limit: chopLimit);
782 if (!isStringList) {
783 isStringList = true;
784 stringListResult.clear();
785 stringResult.squeeze();
786 }
787 stringListResult.append(t: stringResult);
788 stringResult.clear();
789 currentValueIsQuoted = false;
790 ++i;
791 goto StSkipSpaces;
792 }
793 Q_FALLTHROUGH();
794 default: {
795 qsizetype j = i + 1;
796 while (j < str.size()) {
797 ch = str.at(n: j);
798 if (ch == '\\' || ch == '"' || ch == ',')
799 break;
800 ++j;
801 }
802
803 stringResult += fromUtf8(str.first(n: j).sliced(pos: i));
804 i = j;
805 }
806 }
807 }
808 if (!currentValueIsQuoted)
809 iniChopTrailingSpaces(str&: stringResult, limit: chopLimit);
810 goto end;
811
812StHexEscape:
813 if (i >= str.size()) {
814 stringResult += escapeVal;
815 goto end;
816 }
817
818 ch = str.at(n: i);
819 if (ch >= 'a')
820 ch -= 'a' - 'A';
821 if ((ch >= '0' && ch <= '9') || (ch >= 'A' && ch <= 'F')) {
822 escapeVal <<= 4;
823 escapeVal += QtMiscUtils::fromHex(c: ch);
824 ++i;
825 goto StHexEscape;
826 } else {
827 stringResult += escapeVal;
828 goto StNormal;
829 }
830
831StOctEscape:
832 if (i >= str.size()) {
833 stringResult += escapeVal;
834 goto end;
835 }
836
837 ch = str.at(n: i);
838 if (ch >= '0' && ch <= '7') {
839 escapeVal <<= 3;
840 escapeVal += ch - '0';
841 ++i;
842 goto StOctEscape;
843 } else {
844 stringResult += escapeVal;
845 goto StNormal;
846 }
847
848end:
849 if (isStringList)
850 stringListResult.append(t: stringResult);
851 return isStringList;
852}

1308void QConfFileSettingsPrivate::syncConfFile(QConfFile *confFile)
1309{
1310 bool readOnly = confFile->addedKeys.isEmpty() && confFile->removedKeys.isEmpty();
1311
1312 /*
1313 We can often optimize the read-only case, if the file on disk
1314 hasn't changed.
1315 */
1316 if (readOnly && confFile->size > 0) {
1317 QFileInfo fileInfo(confFile->name);
1318 if (confFile->size == fileInfo.size() && confFile->timeStamp == fileInfo.lastModified())
1319 return;
1320 }
1321
1322 if (!readOnly && !confFile->isWritable()) {
1323 setStatus(QSettings::AccessError);
1324 return;
1325 }
1326
1327#ifndef QT_BOOTSTRAPPED
1328 QString lockFileName = confFile->name + ".lock"_L1;
1329
1330# if defined(Q_OS_ANDROID) && defined(QSETTINGS_USE_QSTANDARDPATHS)
1331 // On android and if it is a content URL put the lock file in a
1332 // writable location to prevent permissions issues and invalid paths.
1333 if (confFile->name.startsWith("content:"_L1))
1334 lockFileName = QStandardPaths::writableLocation(QStandardPaths::CacheLocation)
1335 + QFileInfo(lockFileName).fileName();
1336# endif
1337 /*
1338 Use a lockfile in order to protect us against other QSettings instances
1339 trying to write the same settings at the same time.
1340
1341 We only need to lock if we are actually writing as only concurrent writes are a problem.
1342 Concurrent read and write are not a problem because the writing operation is atomic.
1343 */
1344 QLockFile lockFile(lockFileName);
1345 if (!readOnly && !lockFile.lock() && atomicSyncOnly) {
1346 setStatus(QSettings::AccessError);
1347 return;
1348 }
1349#endif
1350
1351 /*
1352 We hold the lock. Let's reread the file if it has changed
1353 since last time we read it.
1354 */
1355 QFileInfo fileInfo(confFile->name);
1356 bool mustReadFile = true;
1357 bool createFile = !fileInfo.exists();
1358
1359 if (!readOnly)
1360 mustReadFile = (confFile->size != fileInfo.size()
1361 || (confFile->size != 0 && confFile->timeStamp != fileInfo.lastModified()));
1362
1363 if (mustReadFile) {
1364 confFile->unparsedIniSections.clear();
1365 confFile->originalKeys.clear();
1366
1367 QFile file(confFile->name);
1368 if (!createFile && !file.open(flags: QFile::ReadOnly)) {
1369 setStatus(QSettings::AccessError);
1370 return;
1371 }
1372
1373 /*
1374 Files that we can't read (because of permissions or
1375 because they don't exist) are treated as empty files.
1376 */
1377 if (file.isReadable() && file.size() != 0) {
1378 bool ok = false;
1379#ifdef Q_OS_MAC
1380 if (format == QSettings::NativeFormat) {
1381 QByteArray data = file.readAll();
1382 ok = readPlistFile(data, &confFile->originalKeys);
1383 } else
1384#endif
1385 if (format <= QSettings::IniFormat) {
1386 QByteArray data = file.readAll();
1387 ok = readIniFile(data, unparsedIniSections: &confFile->unparsedIniSections);
1388 } else if (readFunc) {
1389 QSettings::SettingsMap tempNewKeys;
1390 ok = readFunc(file, tempNewKeys);
1391
1392 if (ok) {
1393 auto i = tempNewKeys.constBegin();
1394 while (i != tempNewKeys.constEnd()) {
1395 confFile->originalKeys.insert(key: QSettingsKey(i.key(), caseSensitivity),
1396 value: i.value());
1397 ++i;
1398 }
1399 }
1400 }
1401
1402 if (!ok)
1403 setStatus(QSettings::FormatError);
1404 }
1405
1406 confFile->size = fileInfo.size();
1407 confFile->timeStamp = fileInfo.lastModified();
1408 }
1409
1410 /*
1411 We also need to save the file. We still hold the file lock,
1412 so everything is under control.
1413 */
1414 if (!readOnly) {
1415 bool ok = false;
1416 ensureAllSectionsParsed(confFile);
1417 ParsedSettingsMap mergedKeys = confFile->mergedKeyMap();
1418
1419#if !defined(QT_BOOTSTRAPPED) && QT_CONFIG(temporaryfile)
1420 QSaveFile sf(confFile->name);
1421 sf.setDirectWriteFallback(!atomicSyncOnly);
1422# ifdef Q_OS_ANDROID
1423 // QSaveFile requires direct write when using content scheme URL in Android
1424 if (confFile->name.startsWith("content:"_L1))
1425 sf.setDirectWriteFallback(true);
1426# endif
1427#else
1428 QFile sf(confFile->name);
1429#endif
1430 if (!sf.open(flags: QIODevice::WriteOnly)) {
1431 setStatus(QSettings::AccessError);
1432 return;
1433 }
1434
1435#ifdef Q_OS_MAC
1436 if (format == QSettings::NativeFormat) {
1437 ok = writePlistFile(sf, mergedKeys);
1438 } else
1439#endif
1440 if (format <= QSettings::IniFormat) {

1508 qsizetype &lineStart, qsizetype &lineLen,
1509 qsizetype &equalsPos)
1510{
1511 qsizetype dataLen = data.size();
1512 bool inQuotes = false;
1513
1514 equalsPos = -1;
1515
1516 lineStart = dataPos;
1517 while (lineStart < dataLen && (charTraits[uint(uchar(data.at(n: lineStart)))] & Space))
1518 ++lineStart;
1519
1520 qsizetype i = lineStart;
1521 while (i < dataLen) {
1522 char ch = data.at(n: i);
1523 while (!(charTraits[uchar(ch)] & Special)) {
1524 if (++i == dataLen)
1525 goto break_out_of_outer_loop;
1526 ch = data.at(n: i);
1527 }
1528
1529 ++i;
1530 if (ch == '=') {
1531 if (!inQuotes && equalsPos == -1)
1532 equalsPos = i - 1;
1533 } else if (ch == '\n' || ch == '\r') {
1534 if (i == lineStart + 1) {
1535 ++lineStart;
1536 } else if (!inQuotes) {
1537 --i;
1538 goto break_out_of_outer_loop;
1539 }

1542 char ch = data.at(n: i++);

1546 if ((ch == '\n' && ch2 == '\r') || (ch == '\r' && ch2 == '\n'))
1547 ++i;
1548 }
1549 }
1550 } else if (ch == '"') {

1552 } else {
1553 Q_ASSERT(ch == ';');
1554

1556 while (i < dataLen && (((ch = data.at(n: i)) != '\n') && ch != '\r'))
1557 ++i;

1565 }

2116 \section1 Accessing Settings from Multiple Threads or Processes Simultaneously
2117
2118 QSettings is \l{reentrant}. This means that you can use
2119 distinct QSettings object in different threads
2120 simultaneously. This guarantee stands even when the QSettings
2121 objects refer to the same files on disk (or to the same entries
2122 in the system registry). If a setting is modified through one
2123 QSettings object, the change will immediately be visible in
2124 any other QSettings objects that operate on the same location
2125 and that live in the same process.
2126
2127 QSettings can safely be used from different processes (which can
2128 be different instances of your application running at the same
2129 time or different applications altogether) to read and write to
2130 the same system locations, provided certain conditions are met. For
2131 QSettings::IniFormat, it uses advisory file locking and a smart merging
2132 algorithm to ensure data integrity. The condition for that to work is that
2133 the writeable configuration file must be a regular file and must reside in
2134 a directory that the current user can create new, temporary files in. If
2135 that is not the case, then one must use setAtomicSyncRequired() to turn the
2136 safety off.
2137
2138 Note that sync() imports changes made by other processes (in addition to
2139 writing the changes from this QSettings).
