30enum { PollingInterval = 1000 };
31
32class QPollingFileSystemWatcherEngine : public QFileSystemWatcherEngine
33{
34 Q_OBJECT
35
36 class FileInfo
37 {
38 uint ownerId;
39 uint groupId;
40 QFile::Permissions permissions;
41 QDateTime lastModified;
42 QStringList entries;
43
44 public:
45 FileInfo(const QFileInfo &fileInfo)
46 : ownerId(fileInfo.ownerId()),
47 groupId(fileInfo.groupId()),
48 permissions(fileInfo.permissions()),
49 lastModified(fileInfo.lastModified())
50 {
51 if (fileInfo.isDir()) {
52 entries = fileInfo.absoluteDir().entryList(filters: QDir::AllEntries);
53 }
54 }
55 FileInfo &operator=(const QFileInfo &fileInfo)
56 {
57 *this = FileInfo(fileInfo);
58 return *this;
59 }
60
61 bool operator!=(const QFileInfo &fileInfo) const
62 {
63 if (fileInfo.isDir() && entries != fileInfo.absoluteDir().entryList(filters: QDir::AllEntries))
64 return true;
65 return (ownerId != fileInfo.ownerId()
66 || groupId != fileInfo.groupId()
67 || permissions != fileInfo.permissions()
68 || lastModified != fileInfo.lastModified());
69 }
70 };
71
72 QHash<QString, FileInfo> files, directories;
73
74public:
75 QPollingFileSystemWatcherEngine(QObject *parent);
76
77 QStringList addPaths(const QStringList &paths, QStringList *files, QStringList *directories) override;
78 QStringList removePaths(const QStringList &paths, QStringList *files, QStringList *directories) override;
79
80private Q_SLOTS:
81 void timeout();
82
83private:
84 QTimer timer;
85};
