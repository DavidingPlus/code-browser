18#include <QtCore/private/qglobal_p.h>
19#include <QtCore/qstring.h>
20#include <QtCore/qbytearray.h>
21
22QT_BEGIN_NAMESPACE
23
24class QFileSystemEntry
25{
26public:
27
28#ifndef Q_OS_WIN
29 typedef QByteArray NativePath;
30#else
31 typedef QString NativePath;
32#endif
33 struct FromNativePath{};
34 struct FromInternalPath{};
35
36 QFileSystemEntry();
37 explicit QFileSystemEntry(const QString &filePath);
38
39 QFileSystemEntry(const QString &filePath, FromInternalPath dummy);
40 QFileSystemEntry(const NativePath &nativeFilePath, FromNativePath dummy);
41 QFileSystemEntry(const QString &filePath, const NativePath &nativeFilePath);
42
43 QString filePath() const;
44 QString fileName() const;
45 QString path() const;
46 NativePath nativeFilePath() const;
47 QString baseName() const;
48 QString completeBaseName() const;
49 QString suffix() const;
50 QString completeSuffix() const;
51 bool isAbsolute() const;
52 bool isRelative() const;
53 bool isClean() const;
54
55#if defined(Q_OS_WIN)
56 bool isDriveRoot() const;
57 static bool isDriveRootPath(const QString &path);
58 static QString removeUncOrLongPathPrefix(QString path);
59#endif
60 bool isRoot() const;
61
62 bool isEmpty() const
63 {
64 return m_filePath.isEmpty() && m_nativeFilePath.isEmpty();
65 }
66 void clear()
67 {
68 *this = QFileSystemEntry();
69 }
70
71 Q_CORE_EXPORT static bool isRootPath(const QString &path);
72
73private:
74 // creates the QString version out of the bytearray version
75 void resolveFilePath() const;
76 // creates the bytearray version out of the QString version
77 void resolveNativeFilePath() const;
78 // resolves the separator
79 void findLastSeparator() const;
80 // resolves the dots and the separator
81 void findFileNameSeparators() const;
82
83 mutable QString m_filePath; // always has slashes as separator
84 mutable NativePath m_nativeFilePath; // native encoding and separators
85
86 mutable qint16 m_lastSeparator; // index in m_filePath of last separator
87 mutable qint16 m_firstDotInFileName; // index after m_filePath for first dot (.)
88 mutable qint16 m_lastDotInFileName; // index after m_firstDotInFileName for last dot (.)
89};
90
91QT_END_NAMESPACE
