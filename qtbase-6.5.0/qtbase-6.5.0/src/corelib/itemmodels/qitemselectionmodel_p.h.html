46 inline void remove(QList<QItemSelectionRange> &r)

48 QList<QItemSelectionRange>::const_iterator it = r.constBegin();
49 for (; it != r.constEnd(); ++it)
50 ranges.removeAll(t: *it);

56 if (!currentSelection.isEmpty()) // ### perhaps this should be in QList
57 currentSelection.clear();
58 }
