14/*!
15 \since 4.1
16 \class QAbstractProxyModel
17 \brief The QAbstractProxyModel class provides a base class for proxy item
18 models that can do sorting, filtering or other data processing tasks.
19 \ingroup model-view
20 \inmodule QtCore
21
22 This class defines the standard interface that proxy models must use to be
23 able to interoperate correctly with other model/view components. It is not
24 supposed to be instantiated directly.
25
26 All standard proxy models are derived from the QAbstractProxyModel class.
27 If you need to create a new proxy model class, it is usually better to
28 subclass an existing class that provides the closest behavior to the one
29 you want to provide.
30
31 Proxy models that filter or sort items of data from a source model should
32 be created by using or subclassing QSortFilterProxyModel.
33
34 To subclass QAbstractProxyModel, you need to implement mapFromSource() and
35 mapToSource(). The mapSelectionFromSource() and mapSelectionToSource()
36 functions only need to be reimplemented if you need a behavior different
37 from the default behavior.
38
39 \note If the source model is deleted or no source model is specified, the
40 proxy model operates on a empty placeholder model.
41
42 \sa QSortFilterProxyModel, QAbstractItemModel, {Model/View Programming}
43*/
44

65void QAbstractProxyModelPrivate::_q_sourceModelRowsInserted(const QModelIndex &parent, int, int)

70 Q_Q(QAbstractProxyModel);

149/*!
150 Sets the given \a sourceModel to be processed by the proxy model.
151
152 Subclasses should call beginResetModel() at the beginning of the method,
153 disconnect from the old model, call this method, connect to the new model,
154 and call endResetModel().
155*/
156void QAbstractProxyModel::setSourceModel(QAbstractItemModel *sourceModel)
157{
158 Q_D(QAbstractProxyModel);
159 d->model.removeBindingUnlessInWrapper();
160 // Special case to handle nullptr models. Otherwise we will have unwanted
161 // notifications.
162 if (!sourceModel && d->model == QAbstractItemModelPrivate::staticEmptyModel())
163 return;
164 static const struct {
165 const char *signalName;
166 const char *slotName;
167 } connectionTable[] = {
168 { SIGNAL(destroyed()), SLOT(_q_sourceModelDestroyed()) },
169 { SIGNAL(rowsAboutToBeInserted(QModelIndex, int, int)), SLOT(_q_sourceModelRowsAboutToBeInserted(QModelIndex, int, int)) },
170 { SIGNAL(rowsInserted(QModelIndex, int, int)), SLOT(_q_sourceModelRowsInserted(QModelIndex, int, int)) },
171 { SIGNAL(rowsRemoved(QModelIndex, int, int)), SLOT(_q_sourceModelRowsRemoved(QModelIndex, int, int)) },
172 { SIGNAL(columnsAboutToBeInserted(QModelIndex, int, int)), SLOT(_q_sourceModelColumnsAboutToBeInserted(QModelIndex, int, int)) },
173 { SIGNAL(columnsInserted(QModelIndex, int, int)), SLOT(_q_sourceModelColumnsInserted(QModelIndex, int, int)) },
174 { SIGNAL(columnsRemoved(QModelIndex, int, int)), SLOT(_q_sourceModelColumnsRemoved(QModelIndex, int, int)) }
175 };
176
177 if (sourceModel != d->model) {
178 if (d->model) {
179 for (const auto &c : connectionTable)
180 disconnect(sender: d->model, signal: c.signalName, receiver: this, member: c.slotName);
181 }
182
183 if (sourceModel) {
184 d->model.setValueBypassingBindings(sourceModel);
185 for (const auto &c : connectionTable)
186 connect(sender: d->model, signal: c.signalName, receiver: this, member: c.slotName);
187 } else {
188 d->model.setValueBypassingBindings(QAbstractItemModelPrivate::staticEmptyModel());
189 }
190 d->model.notify();
191 }
192}

248/*!
249 Returns a source selection mapped from the specified \a proxySelection.
250
251 Reimplement this method to map proxy selections to source selections.
252 */
253QItemSelection QAbstractProxyModel::mapSelectionToSource(const QItemSelection &proxySelection) const
254{
255 QModelIndexList proxyIndexes = proxySelection.indexes();
256 QItemSelection sourceSelection;
257 for (int i = 0; i < proxyIndexes.size(); ++i) {
258 const QModelIndex proxyIdx = mapToSource(proxyIndex: proxyIndexes.at(i));
259 if (!proxyIdx.isValid())
260 continue;
261 sourceSelection << QItemSelectionRange(proxyIdx);
262 }
263 return sourceSelection;
264}
265
266/*!
267 Returns a proxy selection mapped from the specified \a sourceSelection.
268
269 Reimplement this method to map source selections to proxy selections.
270*/
271QItemSelection QAbstractProxyModel::mapSelectionFromSource(const QItemSelection &sourceSelection) const
272{
273 QModelIndexList sourceIndexes = sourceSelection.indexes();
274 QItemSelection proxySelection;
275 for (int i = 0; i < sourceIndexes.size(); ++i) {
276 const QModelIndex srcIdx = mapFromSource(sourceIndex: sourceIndexes.at(i));
277 if (!srcIdx.isValid())
278 continue;
279 proxySelection << QItemSelectionRange(srcIdx);
280 }
281 return proxySelection;
282}

298 Q_D(const QAbstractProxyModel);
299 int sourceSection = section;

301 if (rowCount() > 0) {
302 const QModelIndex proxyIndex = index(row: 0, column: section);
303 sourceSection = mapToSource(proxyIndex).column();
304 }
305 } else {
306 if (columnCount() > 0) {
307 const QModelIndex proxyIndex = index(row: section, column: 0);
308 sourceSection = mapToSource(proxyIndex).row();
309 }
310 }
311 return d->model->headerData(section: sourceSection, orientation, role);
312}

452void QAbstractProxyModelPrivate::mapDropCoordinatesToSource(int row, int column, const QModelIndex &parent,
453 int *sourceRow, int *sourceColumn, QModelIndex *sourceParent) const
454{
455 Q_Q(const QAbstractProxyModel);
456 *sourceRow = -1;
457 *sourceColumn = -1;
458 if (row == -1 && column == -1) {
459 *sourceParent = q->mapToSource(proxyIndex: parent);
460 } else if (row == q->rowCount(parent)) {
461 *sourceParent = q->mapToSource(proxyIndex: parent);
462 *sourceRow = model->rowCount(parent: *sourceParent);
463 } else {
464 QModelIndex proxyIndex = q->index(row, column, parent);
465 QModelIndex sourceIndex = q->mapToSource(proxyIndex);
466 *sourceRow = sourceIndex.row();
467 *sourceColumn = sourceIndex.column();
468 *sourceParent = sourceIndex.parent();
469 }
470}

538/*!
539 Equivalent to calling createIndex on the source model.
540
541 This method is useful if your proxy model wants to maintain the
542 parent-child relationship of items in the source model.
543 When reimplementing mapToSource(), you can call this method to
544 create an index for row \a row and column \a col of the source model.
545
546 A typical use would be to save the internal pointer coming from the source model
547 in the proxy index when reimplementing mapFromSource() and use the same internal
548 pointer as \a internalPtr to recover the original source index when
549 reimplementing mapToSource().
550 \since 6.2
551 */
552QModelIndex QAbstractProxyModel::createSourceIndex(int row, int col, void *internalPtr) const
553{
554 if (sourceModel())
555 return sourceModel()->createIndex(arow: row, acolumn: col, adata: internalPtr);
556 return QModelIndex();
557}
