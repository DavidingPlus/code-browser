51/*!
52 \internal
53*/
54void QTimeLinePrivate::setCurrentTime(int msecs)
55{
56 Q_Q(QTimeLine);
57 currentTime.removeBindingUnlessInWrapper();
58 auto previousCurrentTime = currentTime.value();
59
60 qreal lastValue = q->currentValue();
61 int lastFrame = q->currentFrame();
62
63 // Determine if we are looping.
64 int elapsed = (direction == QTimeLine::Backward) ? (-msecs + duration) : msecs;
65 int loopCountNow = elapsed / duration;
66
67 bool looping = (loopCountNow != currentLoopCount);
68#ifdef QTIMELINE_DEBUG
69 qDebug() << "QTimeLinePrivate::setCurrentTime:" << msecs << duration << "with loopCountNow"
70 << loopCountNow << "currentLoopCount" << currentLoopCount << "looping" << looping;
71#endif
72 if (looping)
73 currentLoopCount = loopCountNow;
74
75 // Normalize msecs to be between 0 and duration, inclusive.
76 currentTime.setValueBypassingBindings(elapsed % duration);
77 if (direction.value() == QTimeLine::Backward)
78 currentTime.setValueBypassingBindings(duration - currentTime);
79
80 // Check if we have reached the end of loopcount.
81 bool finished = false;
82 if (loopCount && currentLoopCount >= loopCount) {
83 finished = true;
84 currentTime.setValueBypassingBindings((direction == QTimeLine::Backward) ? 0 : duration);
85 currentLoopCount = loopCount - 1;
86 }
87
88 int currentFrame = q->frameForTime(msec: currentTime);
89#ifdef QTIMELINE_DEBUG
90 qDebug() << "QTimeLinePrivate::setCurrentTime: frameForTime" << currentTime << currentFrame;
91#endif
92 if (!qFuzzyCompare(p1: lastValue, p2: q->currentValue()))
93 emit q->valueChanged(x: q->currentValue(), QTimeLine::QPrivateSignal());
94 if (lastFrame != currentFrame) {
95 const int transitionframe = (direction == QTimeLine::Forward ? endFrame : startFrame);
96 if (looping && !finished && transitionframe != currentFrame) {
97#ifdef QTIMELINE_DEBUG
98 qDebug("QTimeLinePrivate::setCurrentTime: transitionframe");
99#endif
100 emit q->frameChanged(transitionframe, QTimeLine::QPrivateSignal());
101 }
102#ifdef QTIMELINE_DEBUG
103 else {
104 QByteArray reason;
105 if (!looping)
106 reason += " not looping";
107 if (finished) {
108 if (!reason.isEmpty())
109 reason += " and";
110 reason += " finished";
111 }
112 if (transitionframe == currentFrame) {
113 if (!reason.isEmpty())
114 reason += " and";
115 reason += " transitionframe is equal to currentFrame: " + QByteArray::number(currentFrame);
116 }
117 qDebug("QTimeLinePrivate::setCurrentTime: not transitionframe because %s", reason.constData());
118 }
119#endif
120 emit q->frameChanged(currentFrame, QTimeLine::QPrivateSignal());
121 }
122 if (finished && state == QTimeLine::Running) {
123 q->stop();
124 emit q->finished(QTimeLine::QPrivateSignal());
125 }
126 if (currentTime.value() != previousCurrentTime)
127 currentTime.notify();
128}

135/*!
136 \class QTimeLine
137 \inmodule QtCore
138 \brief The QTimeLine class provides a timeline for controlling animations.
139 \since 4.2
140 \ingroup animation
141
142 It's most commonly used to animate a GUI control by calling a slot
143 periodically. You can construct a timeline by passing its duration in
144 milliseconds to QTimeLine's constructor. The timeline's duration describes
145 for how long the animation will run. Then you set a suitable frame range
146 by calling setFrameRange(). Finally connect the frameChanged() signal to a
147 suitable slot in the widget you wish to animate (for example, \l {QProgressBar::}{setValue()}
148 in QProgressBar). When you proceed to calling start(), QTimeLine will enter
149 Running state, and start emitting frameChanged() at regular intervals,
150 causing your widget's connected property's value to grow from the lower
151 end to the upper and of your frame range, at a steady rate. You can
152 specify the update interval by calling setUpdateInterval(). When done,
153 QTimeLine enters NotRunning state, and emits finished().
154
155 Example:
156
157 \snippet code/src_corelib_tools_qtimeline.cpp 0
158
159 By default the timeline runs once, from its beginning to its end,
160 upon which you must call start() again to restart from the beginning. To
161 make the timeline loop, you can call setLoopCount(), passing the number of
162 times the timeline should run before finishing. The direction can also be
163 changed, causing the timeline to run backward, by calling
164 setDirection(). You can also pause and unpause the timeline while it's
165 running by calling setPaused(). For interactive control, the
166 setCurrentTime() function is provided, which sets the time position of the
167 time line directly. Although most useful in NotRunning state (e.g.,
168 connected to a valueChanged() signal in a QSlider), this function can be
169 called at any time.
170
171 The frame interface is useful for standard widgets, but QTimeLine can be
172 used to control any type of animation. The heart of QTimeLine lies in the
173 valueForTime() function, which generates a \e value between 0 and 1 for a
174 given time. This value is typically used to describe the steps of an
175 animation, where 0 is the first step of an animation, and 1 is the last
176 step. When running, QTimeLine generates values between 0 and 1 by calling
177 valueForTime() and emitting valueChanged(). By default, valueForTime()
178 applies an interpolation algorithm to generate these value. You can choose
179 from a set of predefined timeline algorithms by calling setEasingCurve().
180
181 Note that, by default, QTimeLine uses QEasingCurve::InOutSine, which
182 provides a value that grows slowly, then grows steadily, and finally grows
183 slowly. For a custom timeline, you can reimplement valueForTime(), in which
184 case QTimeLine's easingCurve property is ignored.
185
186 \sa QProgressBar, QProgressDialog
187*/
188
189/*!
190 \enum QTimeLine::State
191
192 This enum describes the state of the timeline.
193
194 \value NotRunning The timeline is not running. This is the initial state
195 of QTimeLine, and the state QTimeLine reenters when finished. The current
196 time, frame and value remain unchanged until either setCurrentTime() is
197 called, or the timeline is started by calling start().
198
199 \value Paused The timeline is paused (i.e., temporarily
200 suspended). Calling setPaused(false) will resume timeline activity.
201
202 \value Running The timeline is running. While control is in the event
203 loop, QTimeLine will update its current time at regular intervals,
204 emitting valueChanged() and frameChanged() when appropriate.
205
206 \sa state(), stateChanged()
207*/
208
209/*!
210 \enum QTimeLine::Direction
211
212 This enum describes the direction of the timeline when in \l Running state.
213
214 \value Forward The current time of the timeline increases with time (i.e.,
215 moves from 0 and towards the end / duration).
216
217 \value Backward The current time of the timeline decreases with time (i.e.,
218 moves from the end / duration and towards 0).
219
220 \sa setDirection()
221*/
222

234 \fn void QTimeLine::frameChanged(int frame)

238
239 \sa QTimeLine::setFrameRange(), QTimeLine::updateInterval
240*/
241

243 \fn void QTimeLine::stateChanged(QTimeLine::State newState)

562/*!
563 Returns the frame corresponding to the time \a msec. This value is
564 calculated using a linear interpolation of the start and end frame, based
565 on the value returned by valueForTime().
566
567 \sa valueForTime(), setFrameRange()
568*/
569int QTimeLine::frameForTime(int msec) const
570{
571 Q_D(const QTimeLine);
572 if (d->direction == Forward)
573 return d->startFrame + int((d->endFrame - d->startFrame) * valueForTime(msec));
574 return d->startFrame + qCeil(v: (d->endFrame - d->startFrame) * valueForTime(msec));
575}

577/*!
578 Returns the timeline value for the time \a msec. The returned value, which
579 varies depending on the curve shape, is always between 0 and 1. If \a msec
580 is 0, the default implementation always returns 0.
581
582 Reimplement this function to provide a custom curve shape for your
583 timeline.
584
585 \sa easingCurve, frameForTime()
586*/
587qreal QTimeLine::valueForTime(int msec) const
588{
589 Q_D(const QTimeLine);
590 msec = qBound(min: 0, val: msec, max: d->duration.value());
591
592 qreal value = msec / qreal(d->duration.value());
593 return d->easingCurve.value().valueForProgress(progress: value);
594}

596/*!
597 Starts the timeline. QTimeLine will enter Running state, and once it
598 enters the event loop, it will update its current time, frame and value at
599 regular intervals. The default interval is 40 ms (i.e., 25 times per
600 second). You can change the update interval by calling
601 setUpdateInterval().
602
603 The timeline will start from position 0, or the end if going backward.
604 If you want to resume a stopped timeline without restarting, you can call
605 resume() instead.
606
607 \sa resume(), updateInterval(), frameChanged(), valueChanged()
608*/
609void QTimeLine::start()
610{
611 Q_D(QTimeLine);
612 if (d->timerId) {
613 qWarning(msg: "QTimeLine::start: already running");
614 return;
615 }
616 int curTime = 0;
617 if (d->direction == Backward)
618 curTime = d->duration;
619 d->timerId = startTimer(interval: d->updateInterval);
620 d->startTime = curTime;
621 d->currentLoopCount = 0;
622 d->timer.start();
623 d->setState(Running);
624 d->setCurrentTime(curTime);
625}

627/*!
628 Resumes the timeline from the current time. QTimeLine will reenter Running
629 state, and once it enters the event loop, it will update its current time,
630 frame and value at regular intervals.
631
632 In contrast to start(), this function does not restart the timeline before
633 it resumes.
634
635 \sa start(), updateInterval(), frameChanged(), valueChanged()
636*/
637void QTimeLine::resume()
638{
639 Q_D(QTimeLine);
640 if (d->timerId) {
641 qWarning(msg: "QTimeLine::resume: already running");
642 return;
643 }
644 d->timerId = startTimer(interval: d->updateInterval);
645 d->startTime = d->currentTime;
646 d->timer.start();
647 d->setState(Running);
648}

664/*!
665 If \a paused is true, the timeline is paused, causing QTimeLine to enter
666 Paused state. No updates will be signaled until either start() or
667 setPaused(false) is called. If \a paused is false, the timeline is resumed
668 and continues where it left.
669
670 \sa state(), start()
671*/
672void QTimeLine::setPaused(bool paused)
673{
674 Q_D(QTimeLine);
675 if (d->state == NotRunning) {
676 qWarning(msg: "QTimeLine::setPaused: Not running");
677 return;
678 }
679 if (paused && d->state != Paused) {
680 d->startTime = d->currentTime;
681 killTimer(id: d->timerId);
682 d->timerId = 0;
683 d->setState(Paused);
684 } else if (!paused && d->state == Paused) {
685 // Same as resume()
686 d->timerId = startTimer(interval: d->updateInterval);
687 d->startTime = d->currentTime;
688 d->timer.start();
689 d->setState(Running);
690 }
691}

707/*!
708 \reimp
709*/
710void QTimeLine::timerEvent(QTimerEvent *event)
711{
712 Q_D(QTimeLine);
713 if (event->timerId() != d->timerId) {
714 event->ignore();
715 return;
716 }
717 event->accept();
718
719 if (d->direction == Forward) {
720 d->setCurrentTime(d->startTime + d->timer.elapsed());
721 } else {
722 d->setCurrentTime(d->startTime - d->timer.elapsed());
723 }
724}
