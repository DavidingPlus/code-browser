290 template <typename InputIterator, QtPrivate::IfIsInputIterator<InputIterator> = true>
291 QList(InputIterator i1, InputIterator i2)
292 {
293 if constexpr (!std::is_convertible_v<typename std::iterator_traits<InputIterator>::iterator_category, std::forward_iterator_tag>) {
294 std::copy(i1, i2, std::back_inserter(*this));
295 } else {
296 const auto distance = std::distance(i1, i2);
297 if (distance) {
298 d = DataPointer(Data::allocate(distance));
299 // appendIteratorRange can deal with contiguous iterators on its own,
300 // this is an optimization for C++17 code.
301 if constexpr (std::is_same_v<std::decay_t<InputIterator>, iterator> ||
302 std::is_same_v<std::decay_t<InputIterator>, const_iterator>) {
303 d->copyAppend(i1.i, i2.i);
304 } else {
305 d->appendIteratorRange(i1, i2);
306 }
307 }
308 }

572 void move(qsizetype from, qsizetype to)
573 {
574 Q_ASSERT_X(from >= 0 && from < size(), "QList::move(qsizetype, qsizetype)", "'from' is out-of-range");
575 Q_ASSERT_X(to >= 0 && to < size(), "QList::move(qsizetype, qsizetype)", "'to' is out-of-range");
576 if (from == to) // don't detach when no-op
577 return;
578 detach();
579 T * const b = d->begin();
580 if (from < to)
581 std::rotate(b + from, b + from + 1, b + to + 1);
582 else
583 std::rotate(b + to, b + from, b + from + 1);
584 }

713inline void QList<T>::resize_internal(qsizetype newSize)

716
717 if (d->needsDetach() || newSize > capacity() - d.freeSpaceAtBegin()) {
718 d.detachAndGrow(QArrayData::GrowsAtEnd, newSize - d.size, nullptr, nullptr);

724template <typename T>
725void QList<T>::reserve(qsizetype asize)
726{
727 // capacity() == 0 for immutable data, so this will force a detaching below
728 if (asize <= capacity() - d.freeSpaceAtBegin()) {
729 if (d->flags() & Data::CapacityReserved)
730 return; // already reserved, don't shrink
731 if (!d->isShared()) {
732 // accept current allocation, don't shrink
733 d->setFlag(Data::CapacityReserved);
734 return;
735 }
736 }
737
738 DataPointer detached(Data::allocate(qMax(asize, size())));
739 detached->copyAppend(d->begin(), d->end());
740 if (detached.d_ptr())
741 detached->setFlag(Data::CapacityReserved);
742 d.swap(detached);
743}

745template <typename T>
746inline void QList<T>::squeeze()
747{
748 if (!d.isMutable())
749 return;
750 if (d->needsDetach() || size() < capacity()) {
751 // must allocate memory
752 DataPointer detached(Data::allocate(size()));
753 if (size()) {
754 if (d.needsDetach())
755 detached->copyAppend(d.data(), d.data() + d.size);
756 else
757 detached->moveAppend(d.data(), d.data() + d.size);
758 }
759 d.swap(detached);
760 }
761 // We're detached so this is fine
762 d->clearFlag(Data::CapacityReserved);
763}

807template <typename T>
808inline void QList<T>::append(QList<T> &&other)
809{
810 Q_ASSERT(&other != this);
811 if (other.isEmpty())
812 return;
813 if (other.d->needsDetach() || !std::is_nothrow_move_constructible_v<T>)
814 return append(other);
815
816 // due to precondition &other != this, we can unconditionally modify 'this'
817 d.detachAndGrow(QArrayData::GrowsAtEnd, other.size(), nullptr, nullptr);
818 Q_ASSERT(d.freeSpaceAtEnd() >= other.size());
819 d->moveAppend(other.d->begin(), other.d->end());
820}

874template <typename T>
875inline QList<T> &QList<T>::fill(parameter_type t, qsizetype newSize)
876{
877 if (newSize == -1)
878 newSize = size();
879 if (d->needsDetach() || newSize > capacity()) {
880 // must allocate memory
881 DataPointer detached(Data::allocate(d->detachCapacity(newSize)));
882 detached->copyAppend(newSize, t);
883 d.swap(detached);
884 } else {
885 // we're detached
886 const T copy(t);
887 d->assign(d.begin(), d.begin() + qMin(size(), newSize), t);
888 if (newSize > size()) {
889 d->copyAppend(newSize - size(), copy);
890 } else if (newSize < size()) {
891 d->truncate(newSize);
892 }
893 }
894 return *this;
895}

898template <typename T, typename U>
899qsizetype indexOf(const QList<T> &vector, const U &u, qsizetype from) noexcept
900{
901 if (from < 0)
902 from = qMax(from + vector.size(), qsizetype(0));
903 if (from < vector.size()) {
904 auto n = vector.begin() + from - 1;
905 auto e = vector.end();
906 while (++n != e)
907 if (*n == u)
908 return qsizetype(n - vector.begin());
909 }
910 return -1;
911}

913template <typename T, typename U>
914qsizetype lastIndexOf(const QList<T> &vector, const U &u, qsizetype from) noexcept
915{
916 if (from < 0)
917 from += vector.d->size;
918 else if (from >= vector.size())
919 from = vector.size() - 1;
920 if (from >= 0) {
921 auto b = vector.begin();
922 auto n = vector.begin() + from + 1;
923 while (n != b) {
924 if (*--n == u)
925 return qsizetype(n - b);
926 }
927 }
928 return -1;
