13template <typename T>
14struct QScopedPointerDeleter
15{
16 static inline void cleanup(T *pointer) noexcept
17 {
18 // Enforce a complete type.
19 // If you get a compile error here, read the section on forward declared
20 // classes in the QScopedPointer documentation.
21 typedef char IsIncompleteType[ sizeof(T) ? 1 : -1 ];
22 (void) sizeof(IsIncompleteType);
23
24 delete pointer;
25 }
26 void operator()(T *pointer) const noexcept
27 {
28 cleanup(pointer);
29 }
30};
31
32template <typename T>
33struct QScopedPointerArrayDeleter
34{
35 static inline void cleanup(T *pointer) noexcept
36 {
37 // Enforce a complete type.
38 // If you get a compile error here, read the section on forward declared
39 // classes in the QScopedPointer documentation.
40 typedef char IsIncompleteType[ sizeof(T) ? 1 : -1 ];
41 (void) sizeof(IsIncompleteType);
42
43 delete[] pointer;
44 }
45 void operator()(T *pointer) const noexcept
46 {
47 cleanup(pointer);
48 }
49};

51struct QScopedPointerPodDeleter
52{
53 static inline void cleanup(void *pointer) noexcept { free(ptr: pointer); }
54 void operator()(void *pointer) const noexcept { cleanup(pointer); }

58template <typename T>
59struct QScopedPointerObjectDeleteLater
60{
61 static inline void cleanup(T *pointer) { if (pointer) pointer->deleteLater(); }
62 void operator()(T *pointer) const { cleanup(pointer); }

69template <typename T, typename Cleanup = QScopedPointerDeleter<T> >
70class [[nodiscard]] QScopedPointer
71{
72public:
73 explicit QScopedPointer(T *p = nullptr) noexcept : d(p)
74 {
75 }
76
77 inline ~QScopedPointer()
78 {
79 T *oldD = this->d;
80 Cleanup::cleanup(oldD);
81 }
82
83 inline T &operator*() const
84 {
85 Q_ASSERT(d);
86 return *d;
87 }
88
89 T *operator->() const noexcept
90 {
91 return d;
92 }
93
94 bool operator!() const noexcept
95 {
96 return !d;
97 }
98
99 explicit operator bool() const
100 {
101 return !isNull();
102 }
103
104 T *data() const noexcept
105 {
106 return d;
107 }
108
109 T *get() const noexcept
110 {
111 return d;
112 }
113
114 bool isNull() const noexcept
115 {
116 return !d;
117 }
118
119 void reset(T *other = nullptr) noexcept(noexcept(Cleanup::cleanup(std::declval<T *>())))
120 {
121 if (d == other)
122 return;
123 T *oldD = std::exchange(d, other);
124 Cleanup::cleanup(oldD);
125 }
126
127#if QT_DEPRECATED_SINCE(6, 1)
128 QT_DEPRECATED_VERSION_X_6_1("Use std::unique_ptr instead, and call release().")
129 T *take() noexcept
130 {
131 T *oldD = std::exchange(d, nullptr);
132 return oldD;
133 }
134#endif
135
136#if QT_DEPRECATED_SINCE(6, 2)
137 QT_DEPRECATED_VERSION_X_6_2("Use std::unique_ptr instead of QScopedPointer.")
138 void swap(QScopedPointer<T, Cleanup> &other) noexcept
139 {
140 qt_ptr_swap(d, other.d);
141 }
142#endif
143
144 typedef T *pointer;
145
146 friend bool operator==(const QScopedPointer<T, Cleanup> &lhs, const QScopedPointer<T, Cleanup> &rhs) noexcept
147 {
148 return lhs.data() == rhs.data();
149 }
150
151 friend bool operator!=(const QScopedPointer<T, Cleanup> &lhs, const QScopedPointer<T, Cleanup> &rhs) noexcept
152 {
153 return lhs.data() != rhs.data();
154 }
155
156 friend bool operator==(const QScopedPointer<T, Cleanup> &lhs, std::nullptr_t) noexcept
157 {
158 return lhs.isNull();
159 }
160
161 friend bool operator==(std::nullptr_t, const QScopedPointer<T, Cleanup> &rhs) noexcept
162 {
163 return rhs.isNull();
164 }
165
166 friend bool operator!=(const QScopedPointer<T, Cleanup> &lhs, std::nullptr_t) noexcept
167 {
168 return !lhs.isNull();
169 }
170
171 friend bool operator!=(std::nullptr_t, const QScopedPointer<T, Cleanup> &rhs) noexcept
172 {
173 return !rhs.isNull();
174 }
175
176#if QT_DEPRECATED_SINCE(6, 2)
177 QT_DEPRECATED_VERSION_X_6_2("Use std::unique_ptr instead of QScopedPointer.")
178 friend void swap(QScopedPointer<T, Cleanup> &p1, QScopedPointer<T, Cleanup> &p2) noexcept
179 { p1.swap(p2); }
180#endif
181
182protected:
183 T *d;
184
185private:
186 Q_DISABLE_COPY_MOVE(QScopedPointer)
187};

189template <typename T, typename Cleanup = QScopedPointerArrayDeleter<T> >
190class [[nodiscard]] QScopedArrayPointer : public QScopedPointer<T, Cleanup>
191{
192 template <typename Ptr>
193 using if_same_type = typename std::enable_if<std::is_same<typename std::remove_cv<T>::type, Ptr>::value, bool>::type;
194public:
195 inline QScopedArrayPointer() : QScopedPointer<T, Cleanup>(nullptr) {}
196 inline ~QScopedArrayPointer() = default;
197
198 template <typename D, if_same_type<D> = true>
199 explicit QScopedArrayPointer(D *p)
200 : QScopedPointer<T, Cleanup>(p)
201 {
202 }
203
204 T &operator[](qsizetype i)
205 {
206 return this->d[i];
207 }
208
209 const T &operator[](qsizetype i) const
210 {
211 return this->d[i];
212 }
213
214#if QT_DEPRECATED_SINCE(6, 2)
215 QT_DEPRECATED_VERSION_X_6_2("Use std::unique_ptr instead of QScopedArrayPointer.")
216 void swap(QScopedArrayPointer &other) noexcept // prevent QScopedPointer <->QScopedArrayPointer swaps
217 { QScopedPointer<T, Cleanup>::swap(other); }
218#endif
219
220private:
221 explicit inline QScopedArrayPointer(void *)
222 {
223 // Enforce the same type.
224
225 // If you get a compile error here, make sure you declare
226 // QScopedArrayPointer with the same template type as you pass to the
227 // constructor. See also the QScopedPointer documentation.
228
229 // Storing a scalar array as a pointer to a different type is not
230 // allowed and results in undefined behavior.
231 }
232
233 Q_DISABLE_COPY_MOVE(QScopedArrayPointer)
234};
