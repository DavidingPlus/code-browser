13#define Q_DECLARE_SEQUENTIAL_ITERATOR(C) \
14\
15template <class T> \
16class Q##C##Iterator \
17{ \
18 typedef typename Q##C<T>::const_iterator const_iterator; \
19 Q##C<T> c; \
20 const_iterator i; \
21public: \
22 inline Q##C##Iterator(const Q##C<T> &container) \
23 : c(container), i(c.constBegin()) {} \
24 inline Q##C##Iterator &operator=(const Q##C<T> &container) \
25 { c = container; i = c.constBegin(); return *this; } \
26 inline void toFront() { i = c.constBegin(); } \
27 inline void toBack() { i = c.constEnd(); } \
28 inline bool hasNext() const { return i != c.constEnd(); } \
29 inline const T &next() { return *i++; } \
30 inline const T &peekNext() const { return *i; } \
31 inline bool hasPrevious() const { return i != c.constBegin(); } \
32 inline const T &previous() { return *--i; } \
33 inline const T &peekPrevious() const { const_iterator p = i; return *--p; } \
34 inline bool findNext(const T &t) \
35 { while (i != c.constEnd()) if (*i++ == t) return true; return false; } \
36 inline bool findPrevious(const T &t) \
37 { while (i != c.constBegin()) if (*(--i) == t) return true; \
38 return false; } \
39};

41#define Q_DECLARE_MUTABLE_SEQUENTIAL_ITERATOR(C) \
42\
43template <class T> \
44class QMutable##C##Iterator \
45{ \
46 typedef typename Q##C<T>::iterator iterator; \
47 typedef typename Q##C<T>::const_iterator const_iterator; \
48 Q##C<T> *c; \
49 iterator i, n; \
50 inline bool item_exists() const { return const_iterator(n) != c->constEnd(); } \
51public: \
52 inline QMutable##C##Iterator(Q##C<T> &container) \
53 : c(&container) \
54 { i = c->begin(); n = c->end(); } \
55 inline QMutable##C##Iterator &operator=(Q##C<T> &container) \
56 { c = &container; i = c->begin(); n = c->end(); return *this; } \
57 inline void toFront() { i = c->begin(); n = c->end(); } \
58 inline void toBack() { i = c->end(); n = i; } \
59 inline bool hasNext() const { return c->constEnd() != const_iterator(i); } \
60 inline T &next() { n = i++; return *n; } \
61 inline T &peekNext() const { return *i; } \
62 inline bool hasPrevious() const { return c->constBegin() != const_iterator(i); } \
63 inline T &previous() { n = --i; return *n; } \
64 inline T &peekPrevious() const { iterator p = i; return *--p; } \
65 inline void remove() \
66 { if (c->constEnd() != const_iterator(n)) { i = c->erase(n); n = c->end(); } } \
67 inline void setValue(const T &t) const { if (c->constEnd() != const_iterator(n)) *n = t; } \
68 inline T &value() { Q_ASSERT(item_exists()); return *n; } \
69 inline const T &value() const { Q_ASSERT(item_exists()); return *n; } \
70 inline void insert(const T &t) { n = i = c->insert(i, t); ++i; } \
71 inline bool findNext(const T &t) \
72 { while (c->constEnd() != const_iterator(n = i)) if (*i++ == t) return true; return false; } \
73 inline bool findPrevious(const T &t) \
74 { while (c->constBegin() != const_iterator(i)) if (*(n = --i) == t) return true; \
75 n = c->end(); return false; } \
76};

216template<typename Key, typename T, class Iterator>
217class QKeyValueIterator
218{
219public:
220 typedef typename Iterator::iterator_category iterator_category;
221 typedef typename Iterator::difference_type difference_type;
222 typedef std::pair<Key, T> value_type;
223 typedef const value_type &reference;
224
225 QKeyValueIterator() = default;
226 constexpr explicit QKeyValueIterator(Iterator o) noexcept(std::is_nothrow_move_constructible<Iterator>::value)
227 : i(std::move(o)) {}
228
229 std::pair<Key, T> operator*() const {
230 return std::pair<Key, T>(i.key(), i.value());
231 }
232
233 struct pointer
234 {
235 pointer(value_type &&r_) : r(std::move(r_)) { }
236
237 pointer() = default;
238 pointer(const pointer &other) = default;
239 pointer(pointer &&other) = default;
240 pointer &operator=(const pointer &other) = default;
241 pointer &operator=(pointer &&other) = default;
242
243 value_type &operator*() const { return r; }
244
245 value_type r;
246 const value_type *operator->() const {
247 return &r;
248 }
249 };
250
251 pointer operator->() const {
252 return pointer(std::pair<Key, T>(i.key(), i.value()));
253 }
254
255 friend bool operator==(QKeyValueIterator lhs, QKeyValueIterator rhs) noexcept { return lhs.i == rhs.i; }
256 friend bool operator!=(QKeyValueIterator lhs, QKeyValueIterator rhs) noexcept { return lhs.i != rhs.i; }
257
258 inline QKeyValueIterator &operator++() { ++i; return *this; }
259 inline QKeyValueIterator operator++(int) { return QKeyValueIterator(i++);}
260 inline QKeyValueIterator &operator--() { --i; return *this; }
261 inline QKeyValueIterator operator--(int) { return QKeyValueIterator(i--); }
262 Iterator base() const { return i; }
263
264private:
265 Iterator i;
266};
