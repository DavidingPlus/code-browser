5#ifndef QSCOPEGUARD_H
6#define QSCOPEGUARD_H
7
8#include <QtCore/qglobal.h>
9
10#include <type_traits>
11#include <utility>
12
13QT_BEGIN_NAMESPACE
14
15template <typename F>
16class [[nodiscard]] QScopeGuard
17{
18public:
19 explicit QScopeGuard(F &&f) noexcept
20 : m_func(std::move(f))
21 {
22 }
23
24 explicit QScopeGuard(const F &f) noexcept
25 : m_func(f)
26 {
27 }
28
29 QScopeGuard(QScopeGuard &&other) noexcept
30 : m_func(std::move(other.m_func))
31 , m_invoke(std::exchange(other.m_invoke, false))
32 {
33 }
34
35 ~QScopeGuard() noexcept
36 {
37 if (m_invoke)
38 m_func();
39 }
40
41 void dismiss() noexcept
42 {
43 m_invoke = false;
44 }
45
46private:
47 Q_DISABLE_COPY(QScopeGuard)
48
49 F m_func;
50 bool m_invoke = true;
51};
52
53template <typename F> QScopeGuard(F(&)()) -> QScopeGuard<F(*)()>;
54
55//! [qScopeGuard]
56template <typename F>
57[[nodiscard]] QScopeGuard<typename std::decay<F>::type> qScopeGuard(F &&f)
58{
59 return QScopeGuard<typename std::decay<F>::type>(std::forward<F>(f));
60}
61
62QT_END_NAMESPACE
63
64#endif // QSCOPEGUARD_H
