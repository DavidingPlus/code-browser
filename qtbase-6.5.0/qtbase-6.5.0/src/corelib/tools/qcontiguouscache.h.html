143template <typename T>
144void QContiguousCache<T>::detach_helper()
145{
146 Data *x = allocateData(aalloc: d->alloc);
147 x->ref.storeRelaxed(1);
148 x->count = d->count;
149 x->start = d->start;
150 x->offset = d->offset;
151 x->alloc = d->alloc;
152
153 T *dest = x->array + x->start;
154 T *src = d->array + d->start;
155 qsizetype oldcount = x->count;
156 while (oldcount--) {
157 new (dest) T(*src);
158 dest++;
159 if (dest == x->array + x->alloc)
160 dest = x->array;
161 src++;
162 if (src == d->array + d->alloc)
163 src = d->array;
164 }
165
166 if (!d->ref.deref())
167 freeData(x: d);
168 d = x;
169}

171template <typename T>
172void QContiguousCache<T>::setCapacity(qsizetype asize)
173{
174 Q_ASSERT(asize >= 0);
175 if (asize == d->alloc)
176 return;
177 detach();
178 Data *x = allocateData(aalloc: asize);
179 x->ref.storeRelaxed(1);
180 x->alloc = asize;
181 x->count = qMin(d->count, asize);
182 x->offset = d->offset + d->count - x->count;
183 if (asize)
184 x->start = x->offset % x->alloc;
185 else
186 x->start = 0;
187
188 qsizetype oldcount = x->count;
189 if (oldcount)
190 {
191 T *dest = x->array + (x->start + x->count-1) % x->alloc;
192 T *src = d->array + (d->start + d->count-1) % d->alloc;
193 while (oldcount--) {
194 new (dest) T(*src);
195 if (dest == x->array)
196 dest = x->array + x->alloc;
197 dest--;
198 if (src == d->array)
199 src = d->array + d->alloc;
200 src--;
201 }
202 }
203 /* free old */
204 freeData(x: d);
205 d = x;
206}

208template <typename T>
209void QContiguousCache<T>::clear()
210{
211 if (d->ref.loadRelaxed() == 1) {
212 if (QTypeInfo<T>::isComplex) {
213 qsizetype oldcount = d->count;
214 T * i = d->array + d->start;
215 T * e = d->array + d->alloc;
216 while (oldcount--) {
217 i->~T();
218 i++;
219 if (i == e)
220 i = d->array;
221 }
222 }
223 d->count = d->start = d->offset = 0;
224 } else {
225 Data *x = allocateData(aalloc: d->alloc);
226 x->ref.storeRelaxed(1);
227 x->alloc = d->alloc;
228 x->count = x->start = x->offset = 0;
229 if (!d->ref.deref())
230 freeData(x: d);
231 d = x;
232 }
233}

261template <typename T>
262void QContiguousCache<T>::freeData(Data *x)
263{
264 if (QTypeInfo<T>::isComplex) {
265 qsizetype oldcount = d->count;
266 T * i = d->array + d->start;
267 T * e = d->array + d->alloc;
268 while (oldcount--) {
269 i->~T();
270 i++;
271 if (i == e)
272 i = d->array;
273 }
274 }
275 Data::freeData(x);
276}
277template <typename T>
278void QContiguousCache<T>::append(T &&value)
279{
280 if (!d->alloc)
281 return; // zero capacity
282 detach();
283 if (d->count == d->alloc)
284 (d->array + (d->start+d->count) % d->alloc)->~T();
285 new (d->array + (d->start+d->count) % d->alloc) T(std::move(value));
286
287 if (d->count == d->alloc) {
288 d->start++;
289 d->start %= d->alloc;
290 d->offset++;
291 } else {
292 d->count++;
293 }
294}

315template<typename T>
316void QContiguousCache<T>::prepend(T &&value)
317{
318 if (!d->alloc)
319 return; // zero capacity
320 detach();
321 if (d->start)
322 d->start--;
323 else
324 d->start = d->alloc-1;
325 d->offset--;
326
327 if (d->count != d->alloc)
328 d->count++;
329 else
330 (d->array + d->start)->~T();
331
332 new (d->array + d->start) T(std::move(value));
333}

355template<typename T>
356void QContiguousCache<T>::insert(qsizetype pos, T &&value)
357{
358 Q_ASSERT_X(pos >= 0, "QContiguousCache<T>::insert", "index out of range");
359 if (!d->alloc)
360 return; // zero capacity
361 detach();
362 if (containsIndex(pos)) {
363 d->array[pos % d->alloc] = std::move(value);
364 } else if (pos == d->offset-1)
365 prepend(value);
366 else if (pos == d->offset+d->count)
367 append(value);
368 else {
369 // we don't leave gaps.
370 clear();
371 d->offset = pos;
372 d->start = pos % d->alloc;
373 d->count = 1;
374 new (d->array + d->start) T(std::move(value));
375 }
376}

399template <typename T>
400inline void QContiguousCache<T>::removeFirst()
401{
402 Q_ASSERT(d->count > 0);
403 detach();
404 d->count--;
405 if (QTypeInfo<T>::isComplex)
406 (d->array + d->start)->~T();
407 d->start = (d->start + 1) % d->alloc;
408 d->offset++;
409}

411template <typename T>
412inline void QContiguousCache<T>::removeLast()
413{
414 Q_ASSERT(d->count > 0);
415 detach();
416 d->count--;
417 if (QTypeInfo<T>::isComplex)
418 (d->array + (d->start + d->count) % d->alloc)->~T();
419}
