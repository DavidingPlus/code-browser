4#ifndef QSCOPEDVALUEROLLBACK_H
5#define QSCOPEDVALUEROLLBACK_H
6
7#include <QtCore/qglobal.h>
8
9QT_BEGIN_NAMESPACE
10
11template <typename T>
12class [[nodiscard]] QScopedValueRollback
13{
14public:
15 explicit constexpr QScopedValueRollback(T &var)
16 : varRef(var), oldValue(var)
17 {
18 }
19
20 explicit constexpr QScopedValueRollback(T &var, T value)
21 : varRef(var), oldValue(qExchange(var, std::move(value)))
22 {
23 }
24
25#if __cpp_constexpr >= 201907L
26 constexpr
27#endif
28 ~QScopedValueRollback()
29 {
30 varRef = std::move(oldValue);
31 }
32
33 constexpr void commit()
34 {
35 oldValue = varRef;
36 }
37
38private:
39 T &varRef;
40 T oldValue;
41
42 Q_DISABLE_COPY_MOVE(QScopedValueRollback)
43};
44
45QT_END_NAMESPACE
46
47#endif // QSCOPEDVALUEROLLBACK_H
