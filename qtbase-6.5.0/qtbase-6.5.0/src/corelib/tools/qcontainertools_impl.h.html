29/*!
30 \internal
31
32 Returns whether \a p is within a range [b, e). In simplest form equivalent to:
33 b <= p < e.
34*/
35template<typename T, typename Cmp = std::less<>>
36static constexpr bool q_points_into_range(const T *p, const T *b, const T *e,
37 Cmp less = {}) noexcept
38{
39 return !less(p, b) && less(p, e);
40}
41
42/*!
43 \internal
44
45 Returns whether \a p is within container \a c. In its simplest form equivalent to:
46 c.data() <= p < c.data() + c.size()
47*/
48template <typename C, typename T>
49static constexpr bool q_points_into_range(const T &p, const C &c) noexcept
50{
51 static_assert(std::is_same_v<decltype(std::data(c)), T>);
52
53 // std::distance because QArrayDataPointer has a "qsizetype size"
54 // member but no size() function
55 return q_points_into_range(p, std::data(c),
56 std::data(c) + std::distance(std::begin(c), std::end(c)));
57}

62template <typename T, typename N>
63void q_uninitialized_move_if_noexcept_n(T* first, N n, T* out)
64{
65 if constexpr (std::is_nothrow_move_constructible_v<T> || !std::is_copy_constructible_v<T>)
66 std::uninitialized_move_n(first, n, out);
67 else
68 std::uninitialized_copy_n(first, n, out);
69}

71template <typename T, typename N>
72void q_uninitialized_relocate_n(T* first, N n, T* out)
73{
74 if constexpr (QTypeInfo<T>::isRelocatable) {
75 if (n != N(0)) { // even if N == 0, out == nullptr or first == nullptr are UB for memmove()
76 std::memmove(dest: static_cast<void*>(out),
77 src: static_cast<const void*>(first),
78 n: n * sizeof(T));
79 }
80 } else {
81 q_uninitialized_move_if_noexcept_n(first, n, out);
82 if constexpr (QTypeInfo<T>::isComplex)
83 std::destroy_n(first, n);
84 }
85}

89/*!
90 \internal
91
92 A wrapper around std::rotate(), with an optimization for
93 Q_RELOCATABLE_TYPEs. We omit the return value, as it would be more work to
94 compute in the Q_RELOCATABLE_TYPE case and, unlike std::rotate on
95 ForwardIterators, callers can compute the result in constant time
96 themselves.
97*/
98template <typename T>
99void q_rotate(T *first, T *mid, T *last)
100{
101 if constexpr (QTypeInfo<T>::isRelocatable) {
102 const auto cast = [](T *p) { return reinterpret_cast<uchar*>(p); };
103 std::rotate(cast(first), cast(mid), cast(last));
104 } else {
105 std::rotate(first, mid, last);
106 }
107}

109template<typename iterator, typename N>
110void q_relocate_overlap_n_left_move(iterator first, N n, iterator d_first)
111{
112 // requires: [first, n) is a valid range
113 // requires: d_first + n is reachable from d_first
114 // requires: iterator is at least a random access iterator
115 // requires: value_type(iterator) has a non-throwing destructor
116
117 Q_ASSERT(n);
118 Q_ASSERT(d_first < first); // only allow moves to the "left"
119 using T = typename std::iterator_traits<iterator>::value_type;
120
121 // Watches passed iterator. Unless commit() is called, all the elements that
122 // the watched iterator passes through are deleted at the end of object
123 // lifetime. freeze() could be used to stop watching the passed iterator and
124 // remain at current place.
125 //
126 // requires: the iterator is expected to always point to an invalid object
127 // (to uninitialized memory)
128 struct Destructor
129 {
130 iterator *iter;
131 iterator end;
132 iterator intermediate;
133
134 Destructor(iterator &it) noexcept : iter(std::addressof(it)), end(it) { }
135 void commit() noexcept { iter = std::addressof(end); }
136 void freeze() noexcept
137 {
138 intermediate = *iter;
139 iter = std::addressof(intermediate);
140 }
141 ~Destructor() noexcept
142 {
143 for (const int step = *iter < end ? 1 : -1; *iter != end;) {
144 std::advance(*iter, step);
145 (*iter)->~T();
146 }
147 }
148 } destroyer(d_first);
149
150 const iterator d_last = d_first + n;
151 // Note: use pair and explicitly copy iterators from it to prevent
152 // accidental reference semantics instead of copy. equivalent to:
153 //
154 // auto [overlapBegin, overlapEnd] = std::minmax(d_last, first);
155 auto pair = std::minmax(d_last, first);
156
157 // overlap area between [d_first, d_first + n) and [first, first + n) or an
158 // uninitialized memory area between the two ranges
159 iterator overlapBegin = pair.first;
160 iterator overlapEnd = pair.second;
161
162 // move construct elements in uninitialized region
163 while (d_first != overlapBegin) {
164 // account for std::reverse_iterator, cannot use new(d_first) directly
165 new (std::addressof(*d_first)) T(std::move_if_noexcept(*first));
166 ++d_first;
167 ++first;
168 }
169
170 // cannot commit but have to stop - there might be an overlap region
171 // which we don't want to delete (because it's part of existing data)
172 destroyer.freeze();
173
174 // move assign elements in overlap region
175 while (d_first != d_last) {
176 *d_first = std::move_if_noexcept(*first);
177 ++d_first;
178 ++first;
179 }
180
181 Q_ASSERT(d_first == destroyer.end + n);
182 destroyer.commit(); // can commit here as ~T() below does not throw
183
184 while (first != overlapEnd)
185 (--first)->~T();
186}

188/*!
189 \internal
190
191 Relocates a range [first, n) to [d_first, n) taking care of potential memory
192 overlaps. This is a generic equivalent of memmove.
193
194 If an exception is thrown during the relocation, all the relocated elements
195 are destroyed and [first, n) may contain valid but unspecified values,
196 including moved-from values (basic exception safety).
197*/
198template<typename T, typename N>
199void q_relocate_overlap_n(T *first, N n, T *d_first)
200{
201 static_assert(std::is_nothrow_destructible_v<T>,
202 "This algorithm requires that T has a non-throwing destructor");
203
204 if (n == N(0) || first == d_first || first == nullptr || d_first == nullptr)
205 return;
206
207 if constexpr (QTypeInfo<T>::isRelocatable) {
208 std::memmove(dest: static_cast<void *>(d_first), src: static_cast<const void *>(first), n: n * sizeof(T));
209 } else { // generic version has to be used
210 if (d_first < first) {
211 q_relocate_overlap_n_left_move(first, n, d_first);
212 } else { // first < d_first
213 auto rfirst = std::make_reverse_iterator(first + n);
214 auto rd_first = std::make_reverse_iterator(d_first + n);
215 q_relocate_overlap_n_left_move(rfirst, n, rd_first);
216 }
217 }
218}

277template <typename Container, typename Predicate>
278auto sequential_erase_if(Container &c, Predicate &pred)
279{
280 // This is remove_if() modified to perform the find_if step on
281 // const_iterators to avoid shared container detaches if nothing needs to
282 // be removed. We cannot run remove_if after find_if: doing so would apply
283 // the predicate to the first matching element twice!
284
285 const auto cbegin = c.cbegin();
286 const auto cend = c.cend();
287 const auto t_it = std::find_if(cbegin, cend, pred);
288 auto result = std::distance(cbegin, t_it);
289 if (result == c.size())
290 return result - result; // `0` of the right type
291
292 // now detach:
293 const auto e = c.end();
294
295 auto it = std::next(c.begin(), result);
296 auto dest = it;
297
298 // Loop Invariants:
299 // - it != e
300 // - [next(it), e[ still to be checked
301 // - [c.begin(), dest[ are result
302 while (++it != e) {
303 if (!pred(*it)) {
304 *dest = std::move(*it);
305 ++dest;
306 }
307 }
308
309 result = std::distance(dest, e);
310 c.erase(dest, e);
311 return result;
312}

314template <typename Container, typename T>
315auto sequential_erase(Container &c, const T &t)
316{
317 // use the equivalence relation from http://eel.is/c++draft/list.erasure#1
318 auto cmp = [&](auto &e) { return e == t; };
319 return sequential_erase_if(c, cmp); // can't pass rvalues!
320}

322template <typename Container, typename T>
323auto sequential_erase_with_copy(Container &c, const T &t)
324{
325 using CopyProxy = std::conditional_t<std::is_copy_constructible_v<T>, T, const T &>;
326 const T &tCopy = CopyProxy(t);
327 return sequential_erase(c, tCopy);
328}

330template <typename Container, typename T>
331auto sequential_erase_one(Container &c, const T &t)
332{
333 const auto cend = c.cend();
334 const auto it = std::find(c.cbegin(), cend, t);
335 if (it == cend)
336 return false;
337 c.erase(it);
338 return true;
339}

341template <typename T, typename Predicate>
342qsizetype qset_erase_if(QSet<T> &set, Predicate &pred)
343{
344 qsizetype result = 0;
345 auto it = set.begin();
346 const auto e = set.end();
347 while (it != e) {
348 if (pred(*it)) {
349 ++result;
350 it = set.erase(it);
351 } else {
352 ++it;
353 }
354 }
355 return result;
356}

359// Prerequisite: F is invocable on ArgTypes
360template <typename R, typename F, typename ... ArgTypes>
361struct is_invoke_result_explicitly_convertible : std::is_constructible<R, std::invoke_result_t<F, ArgTypes...>>
362{};
363
364// is_invocable_r checks for implicit conversions, but we need to check
365// for explicit conversions in remove_if. So, roll our own trait.
366template <typename R, typename F, typename ... ArgTypes>
367constexpr bool is_invocable_explicit_r_v = std::conjunction_v<
368 std::is_invocable<F, ArgTypes...>,
369 is_invoke_result_explicitly_convertible<R, F, ArgTypes...>
370>;
371
372template <typename Container, typename Predicate>
373auto associative_erase_if(Container &c, Predicate &pred)
374{
375 // we support predicates callable with either Container::iterator
376 // or with std::pair<const Key &, Value &>
377 using Iterator = typename Container::iterator;
378 using Key = typename Container::key_type;
379 using Value = typename Container::mapped_type;
380 using KeyValuePair = std::pair<const Key &, Value &>;
381
382 typename Container::size_type result = 0;
383
384 auto it = c.begin();
385 const auto e = c.end();
386 while (it != e) {
387 if constexpr (is_invocable_explicit_r_v<bool, Predicate &, Iterator &>) {
388 if (pred(it)) {
389 it = c.erase(it);
390 ++result;
391 } else {
392 ++it;
393 }
394 } else if constexpr (is_invocable_explicit_r_v<bool, Predicate &, KeyValuePair &&>) {
395 KeyValuePair p(it.key(), it.value());
396 if (pred(std::move(p))) {
397 it = c.erase(it);
398 ++result;
399 } else {
400 ++it;
401 }
402 } else {
403 static_assert(sizeof(Container) == 0, "Predicate has an incompatible signature");
404 }
405 }
406
407 return result;
408}
