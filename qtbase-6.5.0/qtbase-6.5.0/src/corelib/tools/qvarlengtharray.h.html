29template <size_t Size, size_t Align, qsizetype Prealloc>
30class QVLAStorage
31{
32 template <size_t> class print;
33protected:
34 ~QVLAStorage() = default;
35
36 alignas(Align) char array[Prealloc * (Align > Size ? Align : Size)];
37 QT_WARNING_PUSH
38 QT_WARNING_DISABLE_DEPRECATED
39 // ensure we maintain BC: std::aligned_storage_t was only specified by a
40 // minimum size, but for BC we need the substitution to be exact in size:
41 static_assert(std::is_same_v<print<sizeof(std::aligned_storage_t<Size, Align>[Prealloc])>,
42 print<sizeof(array)>>);
43 QT_WARNING_POP
44};

55 Q_ALWAYS_INLINE constexpr void verify(qsizetype pos = 0, qsizetype n = 1) const
56 {
57 Q_ASSERT(pos >= 0);
58 Q_ASSERT(pos <= size());
59 Q_ASSERT(n >= 0);
60 Q_ASSERT(n <= size() - pos);

238 ++s;
239 }

241 s = sz;
242 }

306 QVarLengthArray(QVarLengthArray &&other)

308 : Base(other)
309 {
310 const auto otherInlineStorage = reinterpret_cast<T*>(other.array);
311 if (data() == otherInlineStorage) {
312 // inline buffer - move into our inline buffer:
313 this->ptr = this->array;
314 QtPrivate::q_uninitialized_relocate_n(otherInlineStorage, size(), data());

344 inline QVarLengthArray<T, Prealloc> &operator=(const QVarLengthArray<T, Prealloc> &other)

352
353 QVarLengthArray &operator=(QVarLengthArray &&other)
354 noexcept(std::is_nothrow_move_constructible_v<T>)
355 {
356 // we're only required to be self-move-assignment-safe
357 // when we're in the moved-from state (Hinnant criterion)
358 // the moved-from state is the empty state, so we're good with the clear() here:
359 clear();
360 Q_ASSERT(capacity() >= Prealloc);
361 const auto otherInlineStorage = other.array;
362 if (other.ptr != otherInlineStorage) {
363 // heap storage: steal the external buffer, reset other to otherInlineStorage
364 this->a = std::exchange(other.a, Prealloc);
365 this->ptr = std::exchange(other.ptr, otherInlineStorage);
366 } else {
367 // inline storage: move into our storage (doesn't matter whether inline or external)
368 QtPrivate::q_uninitialized_relocate_n(other.data(), other.size(), data());
369 }
370 this->s = std::exchange(other.s, 0);
371 return *this;
372 }

655template <class T, qsizetype Prealloc>
656Q_INLINE_TEMPLATE QVarLengthArray<T, Prealloc>::QVarLengthArray(qsizetype asize)
657{
658 this->s = asize;
659 Q_ASSERT_X(size() >= 0, "QVarLengthArray::QVarLengthArray()", "Size must be greater than or equal to 0.");
660 if (size() > Prealloc) {
661 this->ptr = malloc(size() * sizeof(T));
662 Q_CHECK_PTR(data());
663 this->a = size();
664 } else {
665 this->ptr = this->array;
666 this->a = Prealloc;
667 }
668 if constexpr (QTypeInfo<T>::isComplex) {
669 T *i = end();
670 while (i != begin())
671 q20::construct_at(--i);
672 }
673}

724Q_OUTOFLINE_TEMPLATE void QVLABase<T>::append_impl(qsizetype prealloc, void *array, const T *abuf, qsizetype increment)
725{

727 if (increment <= 0)
728 return;
729

732 if (asize >= capacity())
733 growBy(prealloc, array, increment);

736 std::uninitialized_copy_n(abuf, increment, end());

740 this->s = asize;
741}

743template <class T>
744Q_OUTOFLINE_TEMPLATE void QVLABase<T>::reallocate_impl(qsizetype prealloc, void *array, qsizetype asize, qsizetype aalloc)
745{
746 Q_ASSERT(aalloc >= asize);
747 Q_ASSERT(data());
748 T *oldPtr = data();
749 qsizetype osize = size();
750
751 const qsizetype copySize = qMin(a: asize, b: osize);
752 Q_ASSUME(copySize >= 0);
753
754 if (aalloc != capacity()) {
755 QVLABaseBase::malloced_ptr guard;
756 void *newPtr;
757 qsizetype newA;
758 if (aalloc > prealloc) {
759 newPtr = malloc(size: aalloc * sizeof(T));
760 guard.reset(p: newPtr);
761 Q_CHECK_PTR(newPtr); // could throw
762 // by design: in case of QT_NO_EXCEPTIONS malloc must not fail or it crashes here
763 newA = aalloc;
764 } else {
765 newPtr = array;
766 newA = prealloc;
767 }
768 QtPrivate::q_uninitialized_relocate_n(oldPtr, copySize,
769 reinterpret_cast<T *>(newPtr));
770 // commit:
771 ptr = newPtr;
772 guard.release();
773 a = newA;
774 }
775 s = copySize;
776
777 // destroy remaining old objects
778 if constexpr (QTypeInfo<T>::isComplex) {
779 if (osize > asize)
780 std::destroy(oldPtr + asize, oldPtr + osize);
781 }
782
783 if (oldPtr != reinterpret_cast<T *>(array) && oldPtr != data())
784 free(oldPtr);
785}

873template <class T>
874Q_OUTOFLINE_TEMPLATE auto QVLABase<T>::erase(const_iterator abegin, const_iterator aend) -> iterator
875{
876 Q_ASSERT_X(isValidIterator(abegin), "QVarLengthArray::insert", "The specified const_iterator argument 'abegin' is invalid");
877 Q_ASSERT_X(isValidIterator(aend), "QVarLengthArray::insert", "The specified const_iterator argument 'aend' is invalid");
878
879 qsizetype f = qsizetype(abegin - cbegin());
880 qsizetype l = qsizetype(aend - cbegin());
881 qsizetype n = l - f;
882
883 if (n == 0) // avoid UB in std::move() below
884 return data() + f;
885
886 Q_ASSERT(n > 0); // aend must be reachable from abegin
887
888 if constexpr (QTypeInfo<T>::isComplex) {
889 std::move(begin() + l, end(), QT_MAKE_CHECKED_ARRAY_ITERATOR(begin() + f, size() - f));
890 std::destroy(end() - n, end());
891 } else {
892 memmove(static_cast<void *>(data() + f), static_cast<const void *>(data() + l), (size() - l) * sizeof(T));
893 }
894 this->s -= n;
895 return data() + f;
896}
