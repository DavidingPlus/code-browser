17class QMargins

25 constexpr int left() const noexcept;
26 constexpr int top() const noexcept;
27 constexpr int right() const noexcept;
28 constexpr int bottom() const noexcept;
29

33 constexpr void setBottom(int bottom) noexcept;
34
35 constexpr QMargins &operator+=(const QMargins &margins) noexcept;
36 constexpr QMargins &operator-=(const QMargins &margins) noexcept;
37 constexpr QMargins &operator+=(int) noexcept;
38 constexpr QMargins &operator-=(int) noexcept;
39 constexpr QMargins &operator*=(int) noexcept;

46private:
47 int m_left;
48 int m_top;
49 int m_right;
50 int m_bottom;
51
52 friend constexpr inline bool operator==(const QMargins &m1, const QMargins &m2) noexcept
53 {
54 return
55 m1.m_left == m2.m_left &&
56 m1.m_top == m2.m_top &&
57 m1.m_right == m2.m_right &&
58 m1.m_bottom == m2.m_bottom;
59 }
60
61 friend constexpr inline bool operator!=(const QMargins &m1, const QMargins &m2) noexcept
62 {
63 return !(m1 == m2);
64 }
65
66 template <std::size_t I,
67 typename M,
68 std::enable_if_t<(I < 4), bool> = true,
69 std::enable_if_t<std::is_same_v<std::decay_t<M>, QMargins>, bool> = true>
70 friend constexpr decltype(auto) get(M &&m) noexcept
71 {
72 if constexpr (I == 0)
73 return (std::forward<M>(m).m_left);
74 else if constexpr (I == 1)
75 return (std::forward<M>(m).m_top);
76 else if constexpr (I == 2)
77 return (std::forward<M>(m).m_right);
78 else if constexpr (I == 3)
79 return (std::forward<M>(m).m_bottom);
80 }
81};
