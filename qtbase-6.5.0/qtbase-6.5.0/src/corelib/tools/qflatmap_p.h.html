32/*
33 QFlatMap provides an associative container backed by sorted sequential
34 containers. By default, QList is used.
35
36 Keys and values are stored in two separate containers. This provides improved
37 cache locality for key iteration and makes keys() and values() fast
38 operations.
39
40 One can customize the underlying container type by passing the KeyContainer
41 and MappedContainer template arguments:
42 QFlatMap<float, int, std::less<float>, std::vector<float>, std::vector<int>>
43*/
44
45// Qt 6.4:
46// - removed QFlatMap API which was incompatible with STL semantics
47// - will be released with said API disabled, to catch any out-of-tree users
48// - also allows opting in to the new API using QFLATMAP_ENABLE_STL_COMPATIBLE_INSERT
49// Qt 6.5

65template <class Key, class T, class Compare>
66class QFlatMapValueCompare : protected Compare
67{
68public:
69 QFlatMapValueCompare() = default;
70 QFlatMapValueCompare(const Compare &key_compare)
71 : Compare(key_compare)
72 {
73 }
74
75 using value_type = std::pair<const Key, T>;
76 static constexpr bool is_comparator_noexcept = noexcept(
77 std::declval<Compare>()(std::declval<const Key &>(), std::declval<const Key &>()));
78
79 bool operator()(const value_type &lhs, const value_type &rhs) const
80 noexcept(is_comparator_noexcept)
81 {
82 return Compare::operator()(lhs.first, rhs.first);
83 }
84};

696 template <typename...Args>
697 std::pair<iterator, bool> try_emplace(const Key &key, Args&&...args)
698 {
699 auto it = lower_bound(key);
700 if (it == end() || key_compare::operator()(key, it.key())) {
701 c.values.emplace(toValuesIterator(it), std::forward<Args>(args)...);
702 return { fromKeysIterator(c.keys.insert(toKeysIterator(it), key)), true };
703 } else {
704 return {it, false};
705 }
706 }
707
708 template <typename...Args>
709 std::pair<iterator, bool> try_emplace(Key &&key, Args&&...args)
710 {
711 auto it = lower_bound(key);
712 if (it == end() || key_compare::operator()(key, it.key())) {
713 c.values.emplace(toValuesIterator(it), std::forward<Args>(args)...);
714 return { fromKeysIterator(c.keys.insert(toKeysIterator(it), std::move(key))), true };
715 } else {
716 return {it, false};
717 }

824 const_iterator find(const Key &key) const
825 {
826 auto it = lower_bound(key);
827 if (it != end()) {
828 if (!key_compare::operator()(key, it.key()))
829 return it;
830 it = end();
831 }
832 return it;

835 template <class X, class Y = Compare, is_marked_transparent<Y> = nullptr>
836 const_iterator find(const X &key) const

838 auto it = lower_bound(key);
839 if (it != end()) {
840 if (!key_compare::operator()(key, it.key()))
841 return it;
842 it = end();
843 }
844 return it;

847 template <typename Predicate>
848 size_type remove_if(Predicate pred)
849 {
850 const auto indirect_call_to_pred = [pred = std::move(pred)](iterator it) {
851 using Pair = decltype(*it);
852 using K = decltype(it.key());
853 using V = decltype(it.value());
854 using P = Predicate;
855 if constexpr (std::is_invocable_v<P, K, V>) {
856 return pred(it.key(), it.value());
857 } else if constexpr (std::is_invocable_v<P, Pair> && !std::is_invocable_v<P, K>) {
858 return pred(*it);
859 } else if constexpr (std::is_invocable_v<P, K> && !std::is_invocable_v<P, Pair>) {
860 return pred(it.key());
861 } else {
862 static_assert(QtPrivate::type_dependent_false<Predicate>(),
863 "Don't know how to call the predicate.\n"
864 "Options:\n"
865 "- pred(*it)\n"
866 "- pred(it.key(), it.value())\n"
867 "- pred(it.key())");
868 }
869 };
870
871 auto first = begin();
872 const auto last = end();
873
874 // find_if prefix loop
875 while (first != last && !indirect_call_to_pred(first))
876 ++first;
877
878 if (first == last)
879 return 0; // nothing to do
880
881 // we know that we need to remove *first
882
883 auto kdest = toKeysIterator(it: first);
884 auto vdest = toValuesIterator(it: first);
885
886 ++first;
887
888 auto k = std::next(kdest);
889 auto v = std::next(vdest);
890
891 // Main Loop
892 // - first is used only for indirect_call_to_pred
893 // - operations are done on k, v
894 // Loop invariants:
895 // - first, k, v are pointing to the same element
896 // - [begin(), first[, [c.keys.begin(), k[, [c.values.begin(), v[: already processed
897 // - [first, end()[, [k, c.keys.end()[, [v, c.values.end()[: still to be processed
898 // - [c.keys.begin(), kdest[ and [c.values.begin(), vdest[ are keepers
899 // - [kdest, k[, [vdest, v[ are considered removed
900 // - kdest is not c.keys.end()
901 // - vdest is not v.values.end()
902 while (first != last) {
903 if (!indirect_call_to_pred(first)) {
904 // keep *first, aka {*k, *v}
905 *kdest = std::move(*k);
906 *vdest = std::move(*v);
907 ++kdest;
908 ++vdest;
909 }
910 ++k;
911 ++v;
912 ++first;
913 }
914
915 const size_type r = std::distance(kdest, c.keys.end());
916 c.keys.erase(kdest, c.keys.end());
917 c.values.erase(vdest, c.values.end());
918 return r;
919 }

1012 template <class InputIt>
1013 void insertOrderedUniqueRange(InputIt first, InputIt last)
1014 {
1015 const size_type s = c.keys.size();
1016 c.keys.resize(s + std::distance(first, last));
1017 c.values.resize(c.keys.size());
1018 for (size_type i = s; first != last; ++first, ++i) {
1019 c.keys[i] = first->first;
1020 c.values[i] = first->second;
1021 }
1022
1023 std::vector<size_type> p(size_t(c.keys.size()));
1024 std::iota(p.begin(), p.end(), 0);
1025 std::inplace_merge(p.begin(), p.begin() + s, p.end(), IndexedKeyComparator(this));
1026 applyPermutation(p);
1027 makeUnique();
1028 }
1029
1030 void ensureOrderedUnique()
1031 {
1032 std::vector<size_type> p(size_t(c.keys.size()));
1033 std::iota(p.begin(), p.end(), 0);
1034 std::stable_sort(p.begin(), p.end(), IndexedKeyComparator(this));
1035 applyPermutation(p);
1036 makeUnique();
1037 }
1038
1039 void applyPermutation(const std::vector<size_type> &p)
1040 {
1041 const size_type s = c.keys.size();
1042 std::vector<bool> done(s);
1043 for (size_type i = 0; i < s; ++i) {
1044 if (done[i])
1045 continue;
1046 done[i] = true;
1047 size_type j = i;
1048 size_type k = p[i];
1049 while (i != k) {
1050 qSwap(c.keys[j], c.keys[k]);
1051 qSwap(c.values[j], c.values[k]);
1052 done[k] = true;
1053 j = k;
1054 k = p[j];
1055 }
1056 }
1057 }
1058
1059 void makeUnique()
1060 {
1061 // std::unique, but over two ranges
1062 auto equivalent = [this](const auto &lhs, const auto &rhs) {
1063 return !key_compare::operator()(lhs, rhs) && !key_compare::operator()(rhs, lhs);
1064 };
1065 const auto kb = c.keys.begin();
1066 const auto ke = c.keys.end();
1067 auto k = std::adjacent_find(kb, ke, equivalent);
1068 if (k == ke)
1069 return;
1070
1071 // equivalent keys found, we need to do actual work:
1072 auto v = std::next(c.values.begin(), std::distance(kb, k));
1073
1074 auto kdest = k;
1075 auto vdest = v;
1076
1077 ++k;
1078 ++v;
1079
1080 // Loop Invariants:
1081 //
1082 // - [keys.begin(), kdest] and [values.begin(), vdest] are unique
1083 // - k is not keys.end(), v is not values.end()
1084 // - [next(k), keys.end()[ and [next(v), values.end()[ still need to be checked
1085 while ((++v, ++k) != ke) {
1086 if (!equivalent(*kdest, *k)) {
1087 *++kdest = std::move(*k);
1088 *++vdest = std::move(*v);
1089 }
1090 }
1091
1092 c.keys.erase(std::next(kdest), ke);
1093 c.values.erase(std::next(vdest), c.values.end());
1094 }

1096 containers c;
1097};
