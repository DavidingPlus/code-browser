72/*
73 These 2 functions replace macros of the same name in sha224-256.c and
74 sha384-512.c. Originally, these macros relied on a global static 'addTemp'
75 variable. We do not want this for 2 reasons:
76
77 1. since we are including the sources directly, the declaration of the 2 conflict
78
79 2. static variables are not thread-safe, we do not want multiple threads
80 computing a hash to corrupt one another
81*/
82static int SHA224_256AddLength(SHA256Context *context, unsigned int length);
83static int SHA384_512AddLength(SHA512Context *context, unsigned int length);
84
85// Sources from rfc6234, with 4 modifications:
86// sha224-256.c - commented out 'static uint32_t addTemp;' on line 68
87// sha224-256.c - appended 'M' to the SHA224_256AddLength macro on line 70
88#include "../../3rdparty/rfc6234/sha224-256.c"
89// sha384-512.c - commented out 'static uint64_t addTemp;' on line 302
90// sha384-512.c - appended 'M' to the SHA224_256AddLength macro on line 304
91#include "../../3rdparty/rfc6234/sha384-512.c"
92
93static inline int SHA224_256AddLength(SHA256Context *context, unsigned int length)
94{
95 uint32_t addTemp;
96 return SHA224_256AddLengthM(context, length);
97}
98static inline int SHA384_512AddLength(SHA512Context *context, unsigned int length)
99{
100 uint64_t addTemp;
101 return SHA384_512AddLengthM(context, length);

145static constexpr qsizetype MaxHashLength = 64;

147static constexpr int hashLengthInternal(QCryptographicHash::Algorithm method) noexcept
148{
149 switch (method) {
150#define CASE(Enum, Size) \
151 case QCryptographicHash:: Enum : \
152 /* if this triggers, then increase MaxHashLength accordingly */ \
153 static_assert(MaxHashLength >= qsizetype(Size) ); \
154 return Size \
155 /*end*/
156 CASE(Sha1, 20);
157#ifndef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
158 CASE(Md4, 16);
159 CASE(Md5, 16);
160 CASE(Sha224, SHA224HashSize);
161 CASE(Sha256, SHA256HashSize);
162 CASE(Sha384, SHA384HashSize);
163 CASE(Sha512, SHA512HashSize);
164 CASE(Blake2s_128, 128 / 8);
165 case QCryptographicHash::Blake2b_160:
166 case QCryptographicHash::Blake2s_160:
167 static_assert(160 / 8 <= MaxHashLength);
168 return 160 / 8;
169 case QCryptographicHash::RealSha3_224:
170 case QCryptographicHash::Keccak_224:
171 case QCryptographicHash::Blake2s_224:
172 static_assert(224 / 8 <= MaxHashLength);
173 return 224 / 8;
174 case QCryptographicHash::RealSha3_256:
175 case QCryptographicHash::Keccak_256:
176 case QCryptographicHash::Blake2b_256:
177 case QCryptographicHash::Blake2s_256:
178 static_assert(256 / 8 <= MaxHashLength);
179 return 256 / 8;
180 case QCryptographicHash::RealSha3_384:
181 case QCryptographicHash::Keccak_384:
182 case QCryptographicHash::Blake2b_384:
183 static_assert(384 / 8 <= MaxHashLength);
184 return 384 / 8;
185 case QCryptographicHash::RealSha3_512:
186 case QCryptographicHash::Keccak_512:
187 case QCryptographicHash::Blake2b_512:
188 static_assert(512 / 8 <= MaxHashLength);
189 return 512 / 8;
190#endif
191#undef CASE
192 }
193 return 0;
194}

311#ifndef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
312#ifndef USING_OPENSSL30
313void QCryptographicHashPrivate::sha3Finish(int bitCount, Sha3Variant sha3Variant)
314{
315 /*
316 FIPS 202 ยง6.1 defines SHA-3 in terms of calculating the Keccak function
317 over the original message with the two-bit suffix "01" appended to it.
318 This variable stores that suffix (and it's fed into the calculations
319 when the hash is returned to users).
320
321 Only 2 bits of this variable are actually used (see the call to sha3Update
322 below). The Keccak implementation we're using will actually use the
323 *leftmost* 2 bits, and interpret them right-to-left. In other words, the
324 bits must appear in order of *increasing* significance; and as the two most
325 significant bits of the byte -- the rightmost 6 are ignored. (Yes, this
326 seems self-contradictory, but it's the way it is...)
327
328 Overall, this means:
329 * the leftmost two bits must be "10" (not "01"!);
330 * we don't care what the other six bits are set to (they can be set to
331 any value), but we arbitrarily set them to 0;
332
333 and for an unsigned char this gives us 0b10'00'00'00, or 0x80.
334 */
335 static const unsigned char sha3FinalSuffix = 0x80;
336
337 result.resizeForOverwrite(bitCount / 8);
338
339 SHA3Context copy = sha3Context;
340
341 switch (sha3Variant) {
342 case Sha3Variant::Sha3:
343 sha3Update(&copy, reinterpret_cast<const BitSequence *>(&sha3FinalSuffix), 2);
344 break;
345 case Sha3Variant::Keccak:
346 break;
347 }
348
349 sha3Final(&copy, result.data());
350}
351#endif // !QT_CONFIG(opensslv30)
352#endif

476void QCryptographicHashPrivate::reset() noexcept
477{
478 result.clear();
479#ifdef USING_OPENSSL30
480 if (method == QCryptographicHash::Blake2b_160 ||
481 method == QCryptographicHash::Blake2b_256 ||
482 method == QCryptographicHash::Blake2b_384) {
483 new (&blake2bContext) blake2b_state;
484 blake2b_init(&blake2bContext, hashLengthInternal(method));
485 return;
486 } else if (method == QCryptographicHash::Blake2s_128 ||
487 method == QCryptographicHash::Blake2s_160 ||
488 method == QCryptographicHash::Blake2s_224) {
489 new (&blake2sContext) blake2s_state;
490 blake2s_init(&blake2sContext, hashLengthInternal(method));
491 return;
492 }
493
494 initializationFailed = true;
495
496 if (method == QCryptographicHash::Md4) {
497 /*
498 * We need to load the legacy provider in order to have the MD4
499 * algorithm available.
500 */
501 if (!OSSL_PROVIDER_load(nullptr, "legacy"))
502 return;
503 if (!OSSL_PROVIDER_load(nullptr, "default"))
504 return;
505 }
506
507 context = EVP_MD_CTX_ptr(EVP_MD_CTX_new());
508
509 if (!context) {
510 return;
511 }
512
513 /*
514 * Using the "-fips" option will disable the global "fips=yes" for
515 * this one lookup and the algorithm can be fetched from any provider
516 * that implements the algorithm (including the FIPS provider).
517 */
518 algorithm = EVP_MD_ptr(EVP_MD_fetch(nullptr, methodToName(method), "-fips"));
519 if (!algorithm) {
520 return;
521 }
522
523 initializationFailed = !EVP_DigestInit_ex(context.get(), algorithm.get(), nullptr);
524#else
525 switch (method) {
526 case QCryptographicHash::Sha1:
527 new (&sha1Context) Sha1State;
528 sha1InitState(state: &sha1Context);
529 break;
530#ifdef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
531 default:
532 Q_ASSERT_X(false, "QCryptographicHash", "Method not compiled in");
533 Q_UNREACHABLE();
534 break;
535#else
536 case QCryptographicHash::Md4:
537 new (&md4Context) md4_context;
538 md4_init(&md4Context);
539 break;
540 case QCryptographicHash::Md5:
541 new (&md5Context) MD5Context;
542 MD5Init(&md5Context);
543 break;
544 case QCryptographicHash::Sha224:
545 new (&sha224Context) SHA224Context;
546 SHA224Reset(&sha224Context);
547 break;
548 case QCryptographicHash::Sha256:
549 new (&sha256Context) SHA256Context;
550 SHA256Reset(&sha256Context);
551 break;
552 case QCryptographicHash::Sha384:
553 new (&sha384Context) SHA384Context;
554 SHA384Reset(&sha384Context);
555 break;
556 case QCryptographicHash::Sha512:
557 new (&sha512Context) SHA512Context;
558 SHA512Reset(&sha512Context);
559 break;
560 case QCryptographicHash::RealSha3_224:
561 case QCryptographicHash::Keccak_224:
562 case QCryptographicHash::RealSha3_256:
563 case QCryptographicHash::Keccak_256:
564 case QCryptographicHash::RealSha3_384:
565 case QCryptographicHash::Keccak_384:
566 case QCryptographicHash::RealSha3_512:
567 case QCryptographicHash::Keccak_512:
568 new (&sha3Context) SHA3Context;
569 sha3Init(&sha3Context, hashLengthInternal(method) * 8);
570 break;
571 case QCryptographicHash::Blake2b_160:
572 case QCryptographicHash::Blake2b_256:
573 case QCryptographicHash::Blake2b_384:
574 case QCryptographicHash::Blake2b_512:
575 new (&blake2bContext) blake2b_state;
576 blake2b_init(&blake2bContext, hashLengthInternal(method));
577 break;
578 case QCryptographicHash::Blake2s_128:
579 case QCryptographicHash::Blake2s_160:
580 case QCryptographicHash::Blake2s_224:
581 case QCryptographicHash::Blake2s_256:
582 new (&blake2sContext) blake2s_state;
583 blake2s_init(&blake2sContext, hashLengthInternal(method));
584 break;
585#endif
586 }
587#endif // !QT_CONFIG(opensslv30)

616void QCryptographicHashPrivate::addData(QByteArrayView bytes) noexcept
617{
618 const char *data = bytes.data();
619 auto length = bytes.size();
620
621#if QT_POINTER_SIZE == 8
622 // feed the data UINT_MAX bytes at a time, as some of the methods below
623 // take a uint (of course, feeding more than 4G of data into the hashing
624 // functions will be pretty slow anyway)
625 for (auto remaining = length; remaining; remaining -= length, data += length) {
626 length = qMin(a: qsizetype(std::numeric_limits<uint>::max()), b: remaining);
627#else
628 {
629#endif
630
631#ifdef USING_OPENSSL30
632 if (method == QCryptographicHash::Blake2b_160 ||
633 method == QCryptographicHash::Blake2b_256 ||
634 method == QCryptographicHash::Blake2b_384) {
635 blake2b_update(&blake2bContext, reinterpret_cast<const uint8_t *>(data), length);
636 } else if (method == QCryptographicHash::Blake2s_128 ||
637 method == QCryptographicHash::Blake2s_160 ||
638 method == QCryptographicHash::Blake2s_224) {
639 blake2s_update(&blake2sContext, reinterpret_cast<const uint8_t *>(data), length);
640 } else if (!initializationFailed) {
641 EVP_DigestUpdate(context.get(), (const unsigned char *)data, length);
642 }
643#else
644 switch (method) {
645 case QCryptographicHash::Sha1:
646 sha1Update(state: &sha1Context, data: (const unsigned char *)data, len: length);
647 break;
648#ifdef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
649 default:
650 Q_ASSERT_X(false, "QCryptographicHash", "Method not compiled in");
651 Q_UNREACHABLE();
652 break;
653#else
654 case QCryptographicHash::Md4:
655 md4_update(&md4Context, (const unsigned char *)data, length);
656 break;
657 case QCryptographicHash::Md5:
658 MD5Update(&md5Context, (const unsigned char *)data, length);
659 break;
660 case QCryptographicHash::Sha224:
661 SHA224Input(&sha224Context, reinterpret_cast<const unsigned char *>(data), length);
662 break;
663 case QCryptographicHash::Sha256:
664 SHA256Input(&sha256Context, reinterpret_cast<const unsigned char *>(data), length);
665 break;
666 case QCryptographicHash::Sha384:
667 SHA384Input(&sha384Context, reinterpret_cast<const unsigned char *>(data), length);
668 break;
669 case QCryptographicHash::Sha512:

671 break;
672 case QCryptographicHash::RealSha3_224:
673 case QCryptographicHash::Keccak_224:
674 case QCryptographicHash::RealSha3_256:
675 case QCryptographicHash::Keccak_256:
676 case QCryptographicHash::RealSha3_384:
677 case QCryptographicHash::Keccak_384:
678 case QCryptographicHash::RealSha3_512:
679 case QCryptographicHash::Keccak_512:
680 sha3Update(&sha3Context, reinterpret_cast<const BitSequence *>(data), uint64_t(length) * 8);
681 break;
682 case QCryptographicHash::Blake2b_160:
683 case QCryptographicHash::Blake2b_256:
684 case QCryptographicHash::Blake2b_384:
685 case QCryptographicHash::Blake2b_512:
686 blake2b_update(&blake2bContext, reinterpret_cast<const uint8_t *>(data), length);
687 break;
688 case QCryptographicHash::Blake2s_128:
689 case QCryptographicHash::Blake2s_160:
690 case QCryptographicHash::Blake2s_224:
691 case QCryptographicHash::Blake2s_256:
692 blake2s_update(&blake2sContext, reinterpret_cast<const uint8_t *>(data), length);
693 break;
694#endif
695 }
696#endif // !QT_CONFIG(opensslv30)
697 }
698 result.clear();

766/*!
767 \internal
768
769 Must be called with finalizeMutex held (except from static hash() function,
770 where no sharing can take place).
771*/
772void QCryptographicHashPrivate::finalizeUnchecked() noexcept
773{
774#ifdef USING_OPENSSL30
775 if (method == QCryptographicHash::Blake2b_160 ||
776 method == QCryptographicHash::Blake2b_256 ||
777 method == QCryptographicHash::Blake2b_384) {
778 const auto length = hashLengthInternal(method);
779 blake2b_state copy = blake2bContext;
780 result.resizeForOverwrite(length);
781 blake2b_final(&copy, result.data(), length);
782 } else if (method == QCryptographicHash::Blake2s_128 ||
783 method == QCryptographicHash::Blake2s_160 ||
784 method == QCryptographicHash::Blake2s_224) {
785 const auto length = hashLengthInternal(method);
786 blake2s_state copy = blake2sContext;
787 result.resizeForOverwrite(length);
788 blake2s_final(&copy, result.data(), length);
789 } else if (!initializationFailed) {
790 EVP_MD_CTX_ptr copy = EVP_MD_CTX_ptr(EVP_MD_CTX_new());
791 EVP_MD_CTX_copy_ex(copy.get(), context.get());
792 result.resizeForOverwrite(EVP_MD_get_size(algorithm.get()));
793 EVP_DigestFinal_ex(copy.get(), result.data(), nullptr);
794 }
795#else
796 switch (method) {
797 case QCryptographicHash::Sha1: {
798 Sha1State copy = sha1Context;
799 result.resizeForOverwrite(s: 20);
800 sha1FinalizeState(state: &copy);
801 sha1ToHash(state: &copy, buffer: result.data());
802 break;
803 }
804#ifdef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
805 default:
806 Q_ASSERT_X(false, "QCryptographicHash", "Method not compiled in");
807 Q_UNREACHABLE();
808 break;
809#else
810 case QCryptographicHash::Md4: {
811 md4_context copy = md4Context;
812 result.resizeForOverwrite(MD4_RESULTLEN);
813 md4_final(&copy, result.data());
814 break;
815 }
816 case QCryptographicHash::Md5: {
817 MD5Context copy = md5Context;
818 result.resizeForOverwrite(16);
819 MD5Final(&copy, result.data());
820 break;
821 }
822 case QCryptographicHash::Sha224: {
823 SHA224Context copy = sha224Context;
824 result.resizeForOverwrite(SHA224HashSize);
825 SHA224Result(&copy, result.data());
826 break;
827 }
828 case QCryptographicHash::Sha256: {
829 SHA256Context copy = sha256Context;
830 result.resizeForOverwrite(SHA256HashSize);
831 SHA256Result(&copy, result.data());
832 break;
833 }
834 case QCryptographicHash::Sha384: {
835 SHA384Context copy = sha384Context;
836 result.resizeForOverwrite(SHA384HashSize);
837 SHA384Result(&copy, result.data());
838 break;
839 }
840 case QCryptographicHash::Sha512: {
841 SHA512Context copy = sha512Context;
842 result.resizeForOverwrite(SHA512HashSize);
843 SHA512Result(&copy, result.data());
844 break;
845 }
846 case QCryptographicHash::RealSha3_224:
847 case QCryptographicHash::RealSha3_256:
848 case QCryptographicHash::RealSha3_384:
849 case QCryptographicHash::RealSha3_512: {
850 sha3Finish(8 * hashLengthInternal(method), Sha3Variant::Sha3);
851 break;
852 }
853 case QCryptographicHash::Keccak_224:
854 case QCryptographicHash::Keccak_256:
855 case QCryptographicHash::Keccak_384:
856 case QCryptographicHash::Keccak_512: {
857 sha3Finish(8 * hashLengthInternal(method), Sha3Variant::Keccak);
858 break;
859 }
860 case QCryptographicHash::Blake2b_160:
861 case QCryptographicHash::Blake2b_256:
862 case QCryptographicHash::Blake2b_384:
863 case QCryptographicHash::Blake2b_512: {
864 const auto length = hashLengthInternal(method);
865 blake2b_state copy = blake2bContext;
866 result.resizeForOverwrite(length);
867 blake2b_final(&copy, result.data(), length);

870 case QCryptographicHash::Blake2s_128:
871 case QCryptographicHash::Blake2s_160:
872 case QCryptographicHash::Blake2s_224:
873 case QCryptographicHash::Blake2s_256: {
874 const auto length = hashLengthInternal(method);

879 }
880#endif
881 }
882#endif // !QT_CONFIG(opensslv30)

1018void QMessageAuthenticationCodePrivate::initMessageHash()
1019{
1020 if (messageHashInited)
1021 return;
1022 messageHashInited = true;
1023
1024 const int blockSize = qt_hash_block_size(method);
1025
1026 if (key.size() > blockSize) {
1027 key = QCryptographicHash::hash(data: key, method);
1028 }
1029
1030 if (key.size() < blockSize) {
1031 const int size = key.size();
1032 key.resize(size: blockSize);
1033 memset(s: key.data() + size, c: 0, n: blockSize - size);
1034 }
1035
1036 QVarLengthArray<char> iKeyPad(blockSize);
1037 const char * const keyData = key.constData();
1038
1039 for (int i = 0; i < blockSize; ++i)
1040 iKeyPad[i] = keyData[i] ^ 0x36;
1041
1042 messageHash.addData(bytes: iKeyPad);
1043}

1162void QMessageAuthenticationCodePrivate::finalizeUnchecked()
1163{
1164 const int blockSize = qt_hash_block_size(method);
1165
1166 QByteArrayView hashedMessage = messageHash.resultView();
1167
1168 QVarLengthArray<char> oKeyPad(blockSize);
1169 const char * const keyData = key.constData();
1170
1171 for (int i = 0; i < blockSize; ++i)
1172 oKeyPad[i] = keyData[i] ^ 0x5c;
1173
1174 QCryptographicHashPrivate hash(method);
1175 hash.addData(bytes: oKeyPad);
1176 hash.addData(bytes: hashedMessage);
1177 hash.finalizeUnchecked();

1180}
