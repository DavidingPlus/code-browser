24constexpr inline char toHexUpper(uint value) noexcept
25{
26 return "0123456789ABCDEF"[value & 0xF];
27}
28
29constexpr inline char toHexLower(uint value) noexcept
30{
31 return "0123456789abcdef"[value & 0xF];
32}
33
34[[nodiscard]] constexpr inline bool isHexDigit(char32_t c) noexcept
35{
36 return (c >= '0' && c <= '9')
37 || (c >= 'A' && c <= 'F')
38 || (c >= 'a' && c <= 'f');
39}
40
41constexpr inline int fromHex(uint c) noexcept
42{
43 return ((c >= '0') && (c <= '9')) ? int(c - '0') :
44 ((c >= 'A') && (c <= 'F')) ? int(c - 'A' + 10) :
45 ((c >= 'a') && (c <= 'f')) ? int(c - 'a' + 10) :
46 /* otherwise */ -1;
47}
48
49constexpr inline char toOct(uint value) noexcept
50{
51 return char('0' + (value & 0x7));
52}
53
54[[nodiscard]] constexpr inline bool isOctalDigit(char32_t c) noexcept
55{
56 return c >= '0' && c <= '7';
57}
58
59constexpr inline int fromOct(uint c) noexcept
60{
61 return isOctalDigit(c) ? int(c - '0') : -1;
62}
63
64[[nodiscard]] constexpr inline bool isAsciiDigit(char32_t c) noexcept
65{
66 return c >= '0' && c <= '9';
67}
68
69constexpr inline bool isAsciiUpper(char32_t c) noexcept
70{
71 return c >= 'A' && c <= 'Z';
72}
73
74constexpr inline bool isAsciiLower(char32_t c) noexcept
75{
76 return c >= 'a' && c <= 'z';
77}
78
79constexpr inline bool isAsciiLetterOrNumber(char32_t c) noexcept
80{
81 return isAsciiDigit(c) || isAsciiLower(c) || isAsciiUpper(c);
82}
83
84constexpr inline char toAsciiLower(char ch) noexcept
85{
86 return isAsciiUpper(c: ch) ? ch - 'A' + 'a' : ch;
87}
88
89constexpr inline char toAsciiUpper(char ch) noexcept
90{
91 return isAsciiLower(c: ch) ? ch - 'a' + 'A' : ch;
92}
93
94constexpr inline int caseCompareAscii(char lhs, char rhs) noexcept
95{
96 const char lhsLower = QtMiscUtils::toAsciiLower(ch: lhs);
97 const char rhsLower = QtMiscUtils::toAsciiLower(ch: rhs);
98 return int(uchar(lhsLower)) - int(uchar(rhsLower));
99}

101constexpr inline int qt_lencmp(qsizetype lhs, qsizetype rhs) noexcept
102{
103 return lhs == rhs ? 0 :
104 lhs > rhs ? 1 :
105 /* else */ -1 ;
106}
