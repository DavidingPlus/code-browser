48class Q_CORE_EXPORT QReadLocker
49{
50public:
51 inline QReadLocker(QReadWriteLock *readWriteLock);
52
53 inline ~QReadLocker()
54 { unlock(); }
55
56 inline void unlock()
57 {
58 if (q_val) {
59 if ((q_val & quintptr(1u)) == quintptr(1u)) {
60 q_val &= ~quintptr(1u);
61 readWriteLock()->unlock();
62 }
63 }
64 }
65
66 inline void relock()
67 {
68 if (q_val) {
69 if ((q_val & quintptr(1u)) == quintptr(0u)) {
70 readWriteLock()->lockForRead();
71 q_val |= quintptr(1u);
72 }
73 }
74 }
75
76 inline QReadWriteLock *readWriteLock() const
77 { return reinterpret_cast<QReadWriteLock *>(q_val & ~quintptr(1u)); }
78
79private:
80 Q_DISABLE_COPY(QReadLocker)
81 quintptr q_val;
82};
83
84inline QReadLocker::QReadLocker(QReadWriteLock *areadWriteLock)
85 : q_val(reinterpret_cast<quintptr>(areadWriteLock))
86{
87 Q_ASSERT_X((q_val & quintptr(1u)) == quintptr(0),
88 "QReadLocker", "QReadWriteLock pointer is misaligned");
89 relock();
90}
