21/*
22 * Implementation details of QReadWriteLock:
23 *
24 * Depending on the valued of d_ptr, the lock is in the following state:
25 * - when d_ptr == 0x0: Unlocked (no readers, no writers) and non-recursive.
26 * - when d_ptr & 0x1: If the least significant bit is set, we are locked for read.
27 * In that case, d_ptr>>4 represents the number of reading threads minus 1. No writers
28 * are waiting, and the lock is not recursive.
29 * - when d_ptr == 0x2: We are locked for write and nobody is waiting. (no contention)
30 * - In any other case, d_ptr points to an actual QReadWriteLockPrivate.
31 */
32
33namespace {

35using ms = std::chrono::milliseconds;

37enum {
38 StateMask = 0x3,
39 StateLockedForRead = 0x1,
40 StateLockedForWrite = 0x2,
41};
42const auto dummyLockedForRead = reinterpret_cast<QReadWriteLockPrivate *>(quintptr(StateLockedForRead));
43const auto dummyLockedForWrite = reinterpret_cast<QReadWriteLockPrivate *>(quintptr(StateLockedForWrite));
44inline bool isUncontendedLocked(const QReadWriteLockPrivate *d)
45{ return quintptr(d) & StateMask; }
46}

48/*! \class QReadWriteLock
49 \inmodule QtCore
50 \brief The QReadWriteLock class provides read-write locking.
51
52 \threadsafe
53
54 \ingroup thread
55
56 A read-write lock is a synchronization tool for protecting
57 resources that can be accessed for reading and writing. This type
58 of lock is useful if you want to allow multiple threads to have
59 simultaneous read-only access, but as soon as one thread wants to
60 write to the resource, all other threads must be blocked until
61 the writing is complete.
62
63 In many cases, QReadWriteLock is a direct competitor to QMutex.
64 QReadWriteLock is a good choice if there are many concurrent
65 reads and writing occurs infrequently.
66
67 Example:
68
69 \snippet code/src_corelib_thread_qreadwritelock.cpp 0
70
71 To ensure that writers aren't blocked forever by readers, readers
72 attempting to obtain a lock will not succeed if there is a blocked
73 writer waiting for access, even if the lock is currently only
74 accessed by other readers. Also, if the lock is accessed by a
75 writer and another writer comes in, that writer will have
76 priority over any readers that might also be waiting.
77
78 Like QMutex, a QReadWriteLock can be recursively locked by the
79 same thread when constructed with \l{QReadWriteLock::Recursive} as
80 \l{QReadWriteLock::RecursionMode}. In such cases,
81 unlock() must be called the same number of times lockForWrite() or
82 lockForRead() was called. Note that the lock type cannot be
83 changed when trying to lock recursively, i.e. it is not possible
84 to lock for reading in a thread that already has locked for
85 writing (and vice versa).
86
87 \sa QReadLocker, QWriteLocker, QMutex, QSemaphore
88*/
89
90/*!
91 \enum QReadWriteLock::RecursionMode
92 \since 4.4
93
94 \value Recursive In this mode, a thread can lock the same
95 QReadWriteLock multiple times. The QReadWriteLock won't be unlocked
96 until a corresponding number of unlock() calls have been made.
97
98 \value NonRecursive In this mode, a thread may only lock a
99 QReadWriteLock once.
100
101 \sa QReadWriteLock()
102*/
103
104/*!
105 \since 4.4
106
107 Constructs a QReadWriteLock object in the given \a recursionMode.
108
109 The default recursion mode is NonRecursive.
110
111 \sa lockForRead(), lockForWrite(), RecursionMode
112*/
113QReadWriteLock::QReadWriteLock(RecursionMode recursionMode)

116 Q_ASSERT_X(!(quintptr(d_ptr.loadRelaxed()) & StateMask), "QReadWriteLock::QReadWriteLock", "bad d_ptr alignment");
117}

172/*! \overload
173
174 Attempts to lock for reading. This function returns \c true if the
175 lock was obtained; otherwise it returns \c false. If another thread
176 has locked for writing, this function will wait for at most \a
177 timeout milliseconds for the lock to become available.
178
179 Note: Passing a negative number as the \a timeout is equivalent to
180 calling lockForRead(), i.e. this function will wait forever until
181 lock can be locked for reading when \a timeout is negative.
182
183 If the lock was obtained, the lock must be unlocked with unlock()
184 before another thread can successfully lock it for writing.
185
186 It is not possible to lock for read if the thread already has
187 locked for write.
188
189 \sa unlock(), lockForRead()
190*/
191bool QReadWriteLock::tryLockForRead(int timeout)
192{
193 // Fast case: non contended:
194 QReadWriteLockPrivate *d;
195 if (d_ptr.testAndSetAcquire(expectedValue: nullptr, newValue: dummyLockedForRead, currentValue&: d))
196 return true;
197
198 while (true) {
199 if (d == nullptr) {
200 if (!d_ptr.testAndSetAcquire(expectedValue: nullptr, newValue: dummyLockedForRead, currentValue&: d))
201 continue;
202 return true;
203 }
204
205 if ((quintptr(d) & StateMask) == StateLockedForRead) {
206 // locked for read, increase the counter
207 const auto val = reinterpret_cast<QReadWriteLockPrivate *>(quintptr(d) + (1U<<4));
208 Q_ASSERT_X(quintptr(val) > (1U<<4), "QReadWriteLock::tryLockForRead()",
209 "Overflow in lock counter");
210 if (!d_ptr.testAndSetAcquire(expectedValue: d, newValue: val, currentValue&: d))
211 continue;
212 return true;
213 }
214
215 if (d == dummyLockedForWrite) {
216 if (!timeout)
217 return false;
218
219 // locked for write, assign a d_ptr and wait.
220 auto val = QReadWriteLockPrivate::allocate();
221 val->writerCount = 1;
222 if (!d_ptr.testAndSetOrdered(expectedValue: d, newValue: val, currentValue&: d)) {
223 val->writerCount = 0;
224 val->release();
225 continue;
226 }
227 d = val;
228 }
229 Q_ASSERT(!isUncontendedLocked(d));
230 // d is an actual pointer;
231
232 if (d->recursive)
233 return d->recursiveLockForRead(timeout);
234
235 auto lock = qt_unique_lock(mutex&: d->mutex);
236 if (d != d_ptr.loadRelaxed()) {
237 // d_ptr has changed: this QReadWriteLock was unlocked before we had
238 // time to lock d->mutex.
239 // We are holding a lock to a mutex within a QReadWriteLockPrivate
240 // that is already released (or even is already re-used). That's ok
241 // because the QFreeList never frees them.
242 // Just unlock d->mutex (at the end of the scope) and retry.
243 d = d_ptr.loadAcquire();
244 continue;
245 }
246 return d->lockForRead(lock, timeout);
247 }
248}

286/*! \overload
287
288 Attempts to lock for writing. This function returns \c true if the
289 lock was obtained; otherwise it returns \c false. If another thread
290 has locked for reading or writing, this function will wait for at
291 most \a timeout milliseconds for the lock to become available.
292
293 Note: Passing a negative number as the \a timeout is equivalent to
294 calling lockForWrite(), i.e. this function will wait forever until
295 lock can be locked for writing when \a timeout is negative.
296
297 If the lock was obtained, the lock must be unlocked with unlock()
298 before another thread can successfully lock it.
299
300 It is not possible to lock for write if the thread already has
301 locked for read.
302
303 \sa unlock(), lockForWrite()
304*/
305bool QReadWriteLock::tryLockForWrite(int timeout)
306{
307 // Fast case: non contended:
308 QReadWriteLockPrivate *d;
309 if (d_ptr.testAndSetAcquire(expectedValue: nullptr, newValue: dummyLockedForWrite, currentValue&: d))
310 return true;
311
312 while (true) {
313 if (d == nullptr) {
314 if (!d_ptr.testAndSetAcquire(expectedValue: d, newValue: dummyLockedForWrite, currentValue&: d))
315 continue;
316 return true;
317 }
318
319 if (isUncontendedLocked(d)) {
320 if (!timeout)
321 return false;
322
323 // locked for either read or write, assign a d_ptr and wait.
324 auto val = QReadWriteLockPrivate::allocate();
325 if (d == dummyLockedForWrite)
326 val->writerCount = 1;
327 else
328 val->readerCount = (quintptr(d) >> 4) + 1;
329 if (!d_ptr.testAndSetOrdered(expectedValue: d, newValue: val, currentValue&: d)) {
330 val->writerCount = val->readerCount = 0;
331 val->release();
332 continue;
333 }
334 d = val;
335 }
336 Q_ASSERT(!isUncontendedLocked(d));
337 // d is an actual pointer;
338
339 if (d->recursive)
340 return d->recursiveLockForWrite(timeout);
341
342 auto lock = qt_unique_lock(mutex&: d->mutex);
343 if (d != d_ptr.loadRelaxed()) {
344 // The mutex was unlocked before we had time to lock the mutex.
345 // We are holding to a mutex within a QReadWriteLockPrivate that is already released
346 // (or even is already re-used) but that's ok because the QFreeList never frees them.
347 d = d_ptr.loadAcquire();
348 continue;
349 }
350 return d->lockForWrite(lock, timeout);
351 }
352}

354/*!
355 Unlocks the lock.
356
357 Attempting to unlock a lock that is not locked is an error, and will result
358 in program termination.
359
360 \sa lockForRead(), lockForWrite(), tryLockForRead(), tryLockForWrite()
361*/
362void QReadWriteLock::unlock()
363{
364 QReadWriteLockPrivate *d = d_ptr.loadAcquire();
365 while (true) {
366 Q_ASSERT_X(d, "QReadWriteLock::unlock()", "Cannot unlock an unlocked lock");
367
368 // Fast case: no contention: (no waiters, no other readers)
369 if (quintptr(d) <= 2) { // 1 or 2 (StateLockedForRead or StateLockedForWrite)
370 if (!d_ptr.testAndSetOrdered(expectedValue: d, newValue: nullptr, currentValue&: d))
371 continue;
372 return;
373 }
374
375 if ((quintptr(d) & StateMask) == StateLockedForRead) {
376 Q_ASSERT(quintptr(d) > (1U<<4)); //otherwise that would be the fast case
377 // Just decrease the reader's count.
378 auto val = reinterpret_cast<QReadWriteLockPrivate *>(quintptr(d) - (1U<<4));
379 if (!d_ptr.testAndSetOrdered(expectedValue: d, newValue: val, currentValue&: d))
380 continue;
381 return;
382 }
383
384 Q_ASSERT(!isUncontendedLocked(d));
385
386 if (d->recursive) {
387 d->recursiveUnlock();
388 return;
389 }
390
391 const auto lock = qt_scoped_lock(mutex&: d->mutex);
392 if (d->writerCount) {
393 Q_ASSERT(d->writerCount == 1);
394 Q_ASSERT(d->readerCount == 0);
395 d->writerCount = 0;
396 } else {
397 Q_ASSERT(d->readerCount > 0);
398 d->readerCount--;
399 if (d->readerCount > 0)
400 return;
401 }
402
403 if (d->waitingReaders || d->waitingWriters) {
404 d->unlock();
405 } else {
406 Q_ASSERT(d_ptr.loadRelaxed() == d); // should not change when we still hold the mutex
407 d_ptr.storeRelease(newValue: nullptr);
408 d->release();
409 }
410 return;
411 }
412}

414/*! \internal Helper for QWaitCondition::wait */
415QReadWriteLock::StateForWaitCondition QReadWriteLock::stateForWaitCondition() const
416{
417 QReadWriteLockPrivate *d = d_ptr.loadAcquire();
418 switch (quintptr(d) & StateMask) {
419 case StateLockedForRead: return LockedForRead;
420 case StateLockedForWrite: return LockedForWrite;
421 }
422
423 if (!d)
424 return Unlocked;
425 const auto lock = qt_scoped_lock(mutex&: d->mutex);
426 if (d->writerCount > 1)
427 return RecursivelyLocked;
428 else if (d->writerCount == 1)
429 return LockedForWrite;
430 return LockedForRead;
431
432}

434bool QReadWriteLockPrivate::lockForRead(std::unique_lock<QtPrivate::mutex> &lock, int timeout)
435{
436 Q_ASSERT(!mutex.try_lock()); // mutex must be locked when entering this function
437
438 QElapsedTimer t;
439 if (timeout > 0)
440 t.start();
441
442 while (waitingWriters || writerCount) {
443 if (timeout == 0)
444 return false;
445 if (timeout > 0) {
446 auto elapsed = t.elapsed();
447 if (elapsed > timeout)
448 return false;
449 waitingReaders++;
450 readerCond.wait_for(lock&: lock, rtime: ms{timeout - elapsed});
451 } else {
452 waitingReaders++;
453 readerCond.wait(lock&: lock);
454 }
455 waitingReaders--;
456 }
457 readerCount++;
458 Q_ASSERT(writerCount == 0);
459 return true;
460}
461
462bool QReadWriteLockPrivate::lockForWrite(std::unique_lock<QtPrivate::mutex> &lock, int timeout)
463{
464 Q_ASSERT(!mutex.try_lock()); // mutex must be locked when entering this function
465
466 QElapsedTimer t;
467 if (timeout > 0)
468 t.start();
469
470 while (readerCount || writerCount) {
471 if (timeout == 0)
472 return false;
473 if (timeout > 0) {
474 auto elapsed = t.elapsed();
475 if (elapsed > timeout) {
476 if (waitingReaders && !waitingWriters && !writerCount) {
477 // We timed out and now there is no more writers or waiting writers, but some
478 // readers were queued (probably because of us). Wake the waiting readers.
479 readerCond.notify_all();
480 }
481 return false;
482 }
483 waitingWriters++;
484 writerCond.wait_for(lock&: lock, rtime: ms{timeout - elapsed});
485 } else {
486 waitingWriters++;
487 writerCond.wait(lock&: lock);
488 }
489 waitingWriters--;
490 }
491
492 Q_ASSERT(writerCount == 0);
493 Q_ASSERT(readerCount == 0);
494 writerCount = 1;
495 return true;
496}

498void QReadWriteLockPrivate::unlock()
499{
500 Q_ASSERT(!mutex.try_lock()); // mutex must be locked when entering this function
501 if (waitingWriters)
502 writerCond.notify_one();
503 else if (waitingReaders)
504 readerCond.notify_all();
505}

507static auto handleEquals(Qt::HANDLE handle)
508{
509 return [handle](QReadWriteLockPrivate::Reader reader) { return reader.handle == handle; };
510}
511
512bool QReadWriteLockPrivate::recursiveLockForRead(int timeout)
513{
514 Q_ASSERT(recursive);
515 auto lock = qt_unique_lock(mutex);
516
517 Qt::HANDLE self = QThread::currentThreadId();
518
519 auto it = std::find_if(first: currentReaders.begin(), last: currentReaders.end(),
520 pred: handleEquals(handle: self));
521 if (it != currentReaders.end()) {
522 ++it->recursionLevel;
523 return true;
524 }
525
526 if (!lockForRead(lock, timeout))
527 return false;
528
529 Reader r = {.handle: self, .recursionLevel: 1};
530 currentReaders.append(t: std::move(r));
531 return true;
532}
533
534bool QReadWriteLockPrivate::recursiveLockForWrite(int timeout)
535{
536 Q_ASSERT(recursive);
537 auto lock = qt_unique_lock(mutex);
538
539 Qt::HANDLE self = QThread::currentThreadId();
540 if (currentWriter == self) {
541 writerCount++;
542 return true;
543 }
544
545 if (!lockForWrite(lock, timeout))
546 return false;
547
548 currentWriter = self;
549 return true;
550}
551
552void QReadWriteLockPrivate::recursiveUnlock()
553{
554 Q_ASSERT(recursive);
555 auto lock = qt_unique_lock(mutex);
556
557 Qt::HANDLE self = QThread::currentThreadId();
558 if (self == currentWriter) {
559 if (--writerCount > 0)
560 return;
561 currentWriter = nullptr;
562 } else {
563 auto it = std::find_if(first: currentReaders.begin(), last: currentReaders.end(),
564 pred: handleEquals(handle: self));
565 if (it == currentReaders.end()) {
566 qWarning(msg: "QReadWriteLock::unlock: unlocking from a thread that did not lock");
567 return;
568 } else {
569 if (--it->recursionLevel <= 0) {
570 currentReaders.erase(pos: it);
571 readerCount--;
572 }
573 if (readerCount)
574 return;
575 }
576 }
577
578 unlock();
579}

597QReadWriteLockPrivate *QReadWriteLockPrivate::allocate()
598{
599 int i = qrwl_freelist->next();
600 QReadWriteLockPrivate *d = &(*qrwl_freelist)[i];
601 d->id = i;
602 Q_ASSERT(!d->recursive);
603 Q_ASSERT(!d->waitingReaders && !d->waitingWriters && !d->readerCount && !d->writerCount);
604 return d;
605}
606
607void QReadWriteLockPrivate::release()
608{
609 Q_ASSERT(!recursive);
610 Q_ASSERT(!waitingReaders && !waitingWriters && !readerCount && !writerCount);
611 qrwl_freelist->release(id);
612}
