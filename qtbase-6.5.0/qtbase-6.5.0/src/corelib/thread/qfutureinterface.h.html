281template <typename T>
282inline bool QFutureInterface<T>::reportResult(const T *result, int index)
283{
284 QMutexLocker<QMutex> locker{&mutex()};
285 if (this->queryState(Canceled) || this->queryState(Finished))
286 return false;
287
288 Q_ASSERT(!hasException());
289 QtPrivate::ResultStoreBase &store = resultStoreBase();
290
291 const int resultCountBefore = store.count();
292 const int insertIndex = store.addResult<T>(index, result);
293 if (insertIndex == -1)
294 return false;
295 if (store.filterMode()) {
296 this->reportResultsReady(resultCountBefore, store.count());
297 } else {
298 this->reportResultsReady(insertIndex, insertIndex + 1);
299 }
300 return true;
301}
302
303template<typename T>
304bool QFutureInterface<T>::reportAndMoveResult(T &&result, int index)
305{
306 QMutexLocker<QMutex> locker{&mutex()};
307 if (queryState(state: Canceled) || queryState(state: Finished))
308 return false;
309
310 Q_ASSERT(!hasException());
311 QtPrivate::ResultStoreBase &store = resultStoreBase();
312
313 const int oldResultCount = store.count();
314 const int insertIndex = store.moveResult(index, std::forward<T>(result));
315 // Let's make sure it's not in pending results.
316 if (insertIndex != -1 && (!store.filterMode() || oldResultCount < store.count()))
317 reportResultsReady(beginIndex: insertIndex, endIndex: store.count());
318 return insertIndex != -1;
319}

324 return reportAndMoveResult(result: std::move(result), index);

333template<typename T>
334inline bool QFutureInterface<T>::reportResults(const QList<T> &_results, int beginIndex, int count)
335{
336 QMutexLocker<QMutex> locker{&mutex()};
337 if (this->queryState(Canceled) || this->queryState(Finished))
338 return false;
339
340 Q_ASSERT(!hasException());
341 auto &store = resultStoreBase();
342
343 const int resultCountBefore = store.count();
344 const int insertIndex = store.addResults(beginIndex, &_results, count);
345 if (insertIndex == -1)
346 return false;
347 if (store.filterMode()) {
348 this->reportResultsReady(resultCountBefore, store.count());
349 } else {
350 this->reportResultsReady(insertIndex, insertIndex + _results.size());
351 }
352 return true;
353}

383template <typename T>
384inline QList<T> QFutureInterface<T>::results()
385{
386 if (this->isCanceled()) {
387 rethrowPossibleException();
388 return QList<T>();
389 }
390
391 QFutureInterfaceBase::waitForResult(resultIndex: -1);
392
393 QList<T> res;
394 QMutexLocker<QMutex> locker{&mutex()};
395
396 QtPrivate::ResultIteratorBase it = resultStoreBase().begin();
397 while (it != resultStoreBase().end()) {
398 res.append(it.value<T>());
399 ++it;
400 }
401
402 return res;
403}

405template<typename T>
406T QFutureInterface<T>::takeResult()
407{
408 Q_ASSERT(isValid());
409
410 // Note: we wait for all, this is intentional,
411 // not to mess with other unready results.
412 waitForResult(resultIndex: -1);
413
414 Q_ASSERT(!hasException());
415
416 const QMutexLocker<QMutex> locker{&mutex()};
417 QtPrivate::ResultIteratorBase position = resultStoreBase().resultAt(0);
418 T ret(std::move_if_noexcept(position.value<T>()));
419 reset();
420 resultStoreBase().template clear<T>();
421
422 return ret;
423}
