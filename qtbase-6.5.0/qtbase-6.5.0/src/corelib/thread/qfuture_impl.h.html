100template<class T>
101inline constexpr bool IsPrivateSignalArg<T, typename std::enable_if_t<

106template<class Tuple, std::size_t... I>
107auto cutTuple(Tuple &&t, std::index_sequence<I...>)
108{
109 return std::make_tuple(std::get<I>(t)...);
110}
111
112template<class Arg, class... Args>
113auto createTuple(Arg &&arg, Args &&... args)
114{
115 using TupleType = std::tuple<std::decay_t<Arg>, std::decay_t<Args>...>;
116 constexpr auto Size = sizeof...(Args); // One less than the size of all arguments
117 if constexpr (QtPrivate::IsPrivateSignalArg<std::tuple_element_t<Size, TupleType>>) {
118 if constexpr (Size == 1) {
119 return std::forward<Arg>(arg);
120 } else {
121 return cutTuple(std::make_tuple(std::forward<Arg>(arg), std::forward<Args>(args)...),
122 std::make_index_sequence<Size>());
123 }
124 } else {
125 return std::make_tuple(std::forward<Arg>(arg), std::forward<Args>(args)...);
126 }
127}

411template<typename Function, typename ResultType, typename ParentResultType>
412void Continuation<Function, ResultType, ParentResultType>::runFunction()
413{
414 promise.start();
415
416 Q_ASSERT(parentFuture.isFinished());
417
418#ifndef QT_NO_EXCEPTIONS
419 try {
420#endif
421 if constexpr (!std::is_void_v<ResultType>) {
422 if constexpr (std::is_void_v<ParentResultType>) {
423 fulfillPromiseWithVoidResult();
424 } else if constexpr (std::is_invocable_v<Function, ParentResultType>) {
425 fulfillPromiseWithResult();
426 } else {
427 // This assert normally should never fail, this is to make sure
428 // that nothing unexpected happened.
429 static_assert(std::is_invocable_v<Function, QFuture<ParentResultType>>,
430 "The continuation is not invocable with the provided arguments");
431 fulfillPromise(parentFuture);
432 }
433 } else {
434 if constexpr (std::is_void_v<ParentResultType>) {
435 if constexpr (std::is_invocable_v<Function, QFuture<void>>)
436 function(parentFuture);
437 else
438 function();
439 } else if constexpr (std::is_invocable_v<Function, ParentResultType>) {
440 fulfillVoidPromise();
441 } else {
442 // This assert normally should never fail, this is to make sure
443 // that nothing unexpected happened.
444 static_assert(std::is_invocable_v<Function, QFuture<ParentResultType>>,
445 "The continuation is not invocable with the provided arguments");
446 function(parentFuture);
447 }
448 }
449#ifndef QT_NO_EXCEPTIONS
450 } catch (...) {
451 promise.setException(std::current_exception());
452 }
453#endif
454 promise.finish();
455}

457template<typename Function, typename ResultType, typename ParentResultType>
458bool Continuation<Function, ResultType, ParentResultType>::execute()
459{
460 Q_ASSERT(parentFuture.isFinished());
461
462 if (parentFuture.d.isChainCanceled()) {
463#ifndef QT_NO_EXCEPTIONS
464 if (parentFuture.d.hasException()) {
465 // If the continuation doesn't take a QFuture argument, propagate the exception
466 // to the caller, by reporting it. If the continuation takes a QFuture argument,
467 // the user may want to catch the exception inside the continuation, to not
468 // interrupt the continuation chain, so don't report anything yet.
469 if constexpr (!std::is_invocable_v<std::decay_t<Function>, QFuture<ParentResultType>>) {
470 promise.start();
471 promise.setException(parentFuture.d.exceptionStore().exception());
472 promise.finish();
473 return false;
474 }
475 } else
476#endif
477 {
478 promise.start();
479 promise.future().cancel();
480 promise.finish();
481 return false;
482 }
483 }
484
485 runImpl();
486 return true;
487}

489// Workaround for keeping move-only lambdas inside std::function
490template<class Function>
491struct ContinuationWrapper
492{
493 ContinuationWrapper(Function &&f) : function(std::move(f)) { }
494 ContinuationWrapper(const ContinuationWrapper &other)
495 : function(std::move(const_cast<ContinuationWrapper &>(other).function))
496 {
497 Q_ASSERT_X(false, "QFuture", "Continuation shouldn't be copied");
498 }
499 ContinuationWrapper(ContinuationWrapper &&other) = default;
500 ContinuationWrapper &operator=(ContinuationWrapper &&) = default;
501
502 void operator()(const QFutureInterfaceBase &parentData) { function(parentData); }
503
504private:
505 Function function;
506};

508template<typename Function, typename ResultType, typename ParentResultType>
509template<typename F>
510void Continuation<Function, ResultType, ParentResultType>::create(F &&func,
511 QFuture<ParentResultType> *f,
512 QFutureInterface<ResultType> &fi,
513 QtFuture::Launch policy)
514{
515 Q_ASSERT(f);
516
517 QThreadPool *pool = nullptr;
518
519 bool launchAsync = (policy == QtFuture::Launch::Async);
520 if (policy == QtFuture::Launch::Inherit) {
521 launchAsync = f->d.launchAsync();
522
523 // If the parent future was using a custom thread pool, inherit it as well.
524 if (launchAsync && f->d.threadPool()) {
525 pool = f->d.threadPool();
526 fi.setThreadPool(pool);
527 }
528 }
529
530 fi.setLaunchAsync(launchAsync);
531
532 auto continuation = [func = std::forward<F>(func), fi, promise = QPromise(fi), pool,
533 launchAsync](const QFutureInterfaceBase &parentData) mutable {
534 const auto parent = QFutureInterface<ParentResultType>(parentData).future();
535 Continuation<Function, ResultType, ParentResultType> *continuationJob = nullptr;
536 if (launchAsync) {
537 auto asyncJob = new AsyncContinuation<Function, ResultType, ParentResultType>(
538 std::forward<Function>(func), parent, std::move(promise), pool);
539 fi.setRunnable(asyncJob);
540 continuationJob = asyncJob;
541 } else {
542 continuationJob = new SyncContinuation<Function, ResultType, ParentResultType>(
543 std::forward<Function>(func), parent, std::move(promise));
544 }
545
546 bool isLaunched = continuationJob->execute();
547 // If continuation is successfully launched, AsyncContinuation will be deleted
548 // by the QThreadPool which has started it. Synchronous continuation will be
549 // executed immediately, so it's safe to always delete it here.
550 if (!(launchAsync && isLaunched)) {
551 delete continuationJob;
552 continuationJob = nullptr;
553 }
554 };
555 f->d.setContinuation(ContinuationWrapper(std::move(continuation)), fi.d);
556}

558template<typename Function, typename ResultType, typename ParentResultType>
559template<typename F>
560void Continuation<Function, ResultType, ParentResultType>::create(F &&func,
561 QFuture<ParentResultType> *f,
562 QFutureInterface<ResultType> &fi,
563 QThreadPool *pool)
564{
565 Q_ASSERT(f);
566
567 fi.setLaunchAsync(true);
568 fi.setThreadPool(pool);
569
570 auto continuation = [func = std::forward<F>(func), promise = QPromise(fi),
571 pool](const QFutureInterfaceBase &parentData) mutable {
572 const auto parent = QFutureInterface<ParentResultType>(parentData).future();
573 auto continuationJob = new AsyncContinuation<Function, ResultType, ParentResultType>(
574 std::forward<Function>(func), parent, std::move(promise), pool);
575 bool isLaunched = continuationJob->execute();
576 // If continuation is successfully launched, AsyncContinuation will be deleted
577 // by the QThreadPool which has started it.
578 if (!isLaunched) {
579 delete continuationJob;
580 continuationJob = nullptr;
581 }
582 };
583 f->d.setContinuation(ContinuationWrapper(std::move(continuation)), fi.d);
584}

645template<class T>
646void fulfillPromise(QPromise<T> &promise, QFuture<T> &future)
647{
648 if constexpr (!std::is_void_v<T>) {
649 if constexpr (std::is_copy_constructible_v<T>)
650 promise.addResult(future.result());
651 else
652 promise.addResult(future.takeResult());
653 }

656template<class T, class Function>
657void fulfillPromise(QPromise<T> &promise, Function &&handler)
658{
659 if constexpr (std::is_void_v<T>)
660 handler();
661 else
662 promise.addResult(handler());

710template<class Function, class ResultType>
711void FailureHandler<Function, ResultType>::run()
712{
713 Q_ASSERT(parentFuture.isFinished());
714
715 promise.start();
716
717 if (parentFuture.d.hasException()) {
718 using ArgType = typename QtPrivate::ArgResolver<Function>::First;
719 if constexpr (std::is_void_v<ArgType>) {
720 handleAllExceptions();
721 } else {
722 handleException<ArgType>();
723 }
724 } else {
725 QtPrivate::fulfillPromise(promise, parentFuture);
726 }
727 promise.finish();
728}
729
730template<class Function, class ResultType>
731template<class ArgType>
732void FailureHandler<Function, ResultType>::handleException()
733{
734 try {
735 Q_ASSERT(parentFuture.d.hasException());
736 parentFuture.d.exceptionStore().rethrowException();
737 } catch (const ArgType &e) {
738 try {
739 // Handle exceptions matching with the handler's argument type
740 if constexpr (std::is_void_v<ResultType>)
741 handler(e);
742 else
743 promise.addResult(handler(e));
744 } catch (...) {
745 promise.setException(std::current_exception());
746 }
747 } catch (...) {
748 // Exception doesn't match with handler's argument type, propagate
749 // the exception to be handled later.
750 promise.setException(std::current_exception());
751 }
752}

754template<class Function, class ResultType>
755void FailureHandler<Function, ResultType>::handleAllExceptions()
756{
757 try {
758 Q_ASSERT(parentFuture.d.hasException());
759 parentFuture.d.exceptionStore().rethrowException();
760 } catch (...) {
761 try {

810 {
811 promise.start();
812
813 if (parentFuture.isCanceled()) {
814#ifndef QT_NO_EXCEPTIONS
815 if (parentFuture.d.hasException()) {
816 // Propagate the exception to the result future
817 promise.setException(parentFuture.d.exceptionStore().exception());
818 } else {
819 try {
820#endif
821 QtPrivate::fulfillPromise(promise, std::forward<F>(handler));
822#ifndef QT_NO_EXCEPTIONS
823 } catch (...) {
824 promise.setException(std::current_exception());
825 }
826 }
827#endif
828 } else {
829 QtPrivate::fulfillPromise(promise, parentFuture);
830 }
831
832 promise.finish();
833 }

836struct UnwrapHandler
837{
838 template<class T>
839 static auto unwrapImpl(T *outer)
840 {
841 Q_ASSERT(outer);
842
843 using ResultType = typename QtPrivate::Future<std::decay_t<T>>::type;
844 using NestedType = typename QtPrivate::Future<ResultType>::type;
845 QFutureInterface<NestedType> promise(QFutureInterfaceBase::State::Pending);
846
847 outer->then([promise](const QFuture<ResultType> &outerFuture) mutable {
848 // We use the .then([](QFuture<ResultType> outerFuture) {...}) version
849 // (where outerFuture == *outer), to propagate the exception if the
850 // outer future has failed.
851 Q_ASSERT(outerFuture.isFinished());
852#ifndef QT_NO_EXCEPTIONS
853 if (outerFuture.d.hasException()) {
854 promise.reportStarted();
855 promise.reportException(outerFuture.d.exceptionStore().exception());
856 promise.reportFinished();
857 return;
858 }
859#endif
860
861 promise.reportStarted();
862 ResultType nestedFuture = outerFuture.result();
863
864 nestedFuture.then([promise] (const QFuture<NestedType> &nested) mutable {
865#ifndef QT_NO_EXCEPTIONS
866 if (nested.d.hasException()) {
867 promise.reportException(nested.d.exceptionStore().exception());
868 } else
869#endif
870 {
871 if constexpr (!std::is_void_v<NestedType>)
872 promise.reportResults(nested.results());
873 }
874 promise.reportFinished();
875 }).onCanceled([promise] () mutable {
876 promise.reportCanceled();
877 promise.reportFinished();
878 });
879 }).onCanceled([promise]() mutable {
880 // propagate the cancellation of the outer future
881 promise.reportStarted();
882 promise.reportCanceled();
883 promise.reportFinished();
884 });
885 return promise.future();
886 }
887};

896template<class Sender, class Signal, typename = QtPrivate::EnableIfInvocable<Sender, Signal>>
897static QFuture<ArgsType<Signal>> connect(Sender *sender, Signal signal)
898{
899 using ArgsType = ArgsType<Signal>;
900 QFutureInterface<ArgsType> promise;
901 promise.reportStarted();
902 if (!sender) {
903 promise.reportCanceled();
904 promise.reportFinished();
905 return promise.future();
906 }
907
908 using Connections = std::pair<QMetaObject::Connection, QMetaObject::Connection>;
909 auto connections = std::make_shared<Connections>();
910
911 if constexpr (std::is_void_v<ArgsType>) {
912 connections->first =
913 QObject::connect(sender, signal, sender, [promise, connections]() mutable {
914 QObject::disconnect(connections->first);
915 QObject::disconnect(connections->second);
916 promise.reportFinished();
917 });
918 } else if constexpr (QtPrivate::ArgResolver<Signal>::HasExtraArgs) {
919 connections->first = QObject::connect(sender, signal, sender,
920 [promise, connections](auto... values) mutable {
921 QObject::disconnect(connections->first);
922 QObject::disconnect(connections->second);
923 promise.reportResult(QtPrivate::createTuple(
924 std::move(values)...));
925 promise.reportFinished();
926 });
927 } else {
928 connections->first = QObject::connect(sender, signal, sender,
929 [promise, connections](ArgsType value) mutable {
930 QObject::disconnect(connections->first);
931 QObject::disconnect(connections->second);
932 promise.reportResult(value);
933 promise.reportFinished();
934 });
935 }
936
937 if (!connections->first) {
938 promise.reportCanceled();
939 promise.reportFinished();
940 return promise.future();
941 }
942
943 connections->second =
944 QObject::connect(sender, &QObject::destroyed, sender, [promise, connections]() mutable {
945 QObject::disconnect(connections->first);
946 QObject::disconnect(connections->second);
947 promise.reportCanceled();
948 promise.reportFinished();
949 });
950
951 return promise.future();
952}

1020template<typename ResultFutures>
1021struct WhenAllContext
1022{
1023 using ValueType = typename ResultFutures::value_type;
1024
1025 explicit WhenAllContext(qsizetype size) : remaining(size) {}
1026
1027 template<typename T = ValueType>
1028 void checkForCompletion(qsizetype index, T &&future)
1029 {
1030 futures[index] = std::forward<T>(future);
1031 const auto oldRemaining = remaining.fetchAndSubRelaxed(valueToAdd: 1);
1032 Q_ASSERT(oldRemaining > 0);
1033 if (oldRemaining <= 1) { // that was the last one
1034 promise.addResult(futures);
1035 promise.finish();
1036 }
1037 }
1038
1039 QAtomicInteger<qsizetype> remaining;
1040 QPromise<ResultFutures> promise;
1041 ResultFutures futures;
1042};

1044template<typename ResultType>
1045struct WhenAnyContext
1046{

1053 promise.addResult(std::forward<T>(result));

1062template<qsizetype Index, typename ContextType, typename... Ts>
1063void addCompletionHandlersImpl(const std::shared_ptr<ContextType> &context,
1064 const std::tuple<Ts...> &t)
1065{
1066 auto future = std::get<Index>(t);
1067 using ResultType = typename ContextType::ValueType;
1068 future.then([context](const std::tuple_element_t<Index, std::tuple<Ts...>> &f) {
1069 context->checkForCompletion(Index, ResultType { std::in_place_index<Index>, f });
1070 }).onCanceled([context, future]() {
1071 context->checkForCompletion(Index, ResultType { std::in_place_index<Index>, future });
1072 });
1073
1074 if constexpr (Index != 0)
1075 addCompletionHandlersImpl<Index - 1, ContextType, Ts...>(context, t);
1076}

1080{

1082 addCompletionHandlersImpl<size - 1, ContextType, Ts...>(context, t);

1086QFuture<OutputSequence> whenAllImpl(InputIt first, InputIt last)
1087{
1088 const qsizetype size = std::distance(first, last);
1089 if (size == 0)
1090 return QtFuture::makeReadyFuture(OutputSequence());
1091
1092 const auto context = std::make_shared<QtPrivate::WhenAllContext<OutputSequence>>(size);
1093 context->futures.resize(size);
1094 context->promise.start();
1095
1096 qsizetype idx = 0;
1097 for (auto it = first; it != last; ++it, ++idx) {
1098 it->then([context, idx](const ValueType &f) {
1099 context->checkForCompletion(idx, f);
1100 }).onCanceled([context, idx, f = *it] {
1101 context->checkForCompletion(idx, f);
1102 });
1103 }
1104 return context->promise.future();
