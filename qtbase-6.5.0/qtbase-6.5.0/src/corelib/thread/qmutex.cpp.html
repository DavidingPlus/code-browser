41/*!
42 \class QMutex
43 \inmodule QtCore
44 \brief The QMutex class provides access serialization between threads.
45
46 \threadsafe
47
48 \ingroup thread
49
50 The purpose of a QMutex is to protect an object, data structure or
51 section of code so that only one thread can access it at a time
52 (this is similar to the Java \c synchronized keyword). It is
53 usually best to use a mutex with a QMutexLocker since this makes
54 it easy to ensure that locking and unlocking are performed
55 consistently.
56
57 For example, say there is a method that prints a message to the
58 user on two lines:
59
60 \snippet code/src_corelib_thread_qmutex.cpp 0
61
62 If these two methods are called in succession, the following happens:
63
64 \snippet code/src_corelib_thread_qmutex.cpp 1
65
66 If these two methods are called simultaneously from two threads then the
67 following sequence could result:
68
69 \snippet code/src_corelib_thread_qmutex.cpp 2
70
71 If we add a mutex, we should get the result we want:
72
73 \snippet code/src_corelib_thread_qmutex.cpp 3
74
75 Then only one thread can modify \c number at any given time and
76 the result is correct. This is a trivial example, of course, but
77 applies to any other case where things need to happen in a
78 particular sequence.
79
80 When you call lock() in a thread, other threads that try to call
81 lock() in the same place will block until the thread that got the
82 lock calls unlock(). A non-blocking alternative to lock() is
83 tryLock().
84
85 QMutex is optimized to be fast in the non-contended case. It
86 will not allocate memory if there is no contention on that mutex.
87 It is constructed and destroyed with almost no overhead,
88 which means it is fine to have many mutexes as part of other classes.
89
90 \sa QRecursiveMutex, QMutexLocker, QReadWriteLock, QSemaphore, QWaitCondition
91*/

105void QBasicMutex::destroyInternal(QMutexPrivate *d)
106{
107 if (!d)
108 return;
109 if (!futexAvailable()) {
110 if (d != dummyLocked() && d->possiblyUnlocked.loadRelaxed() && tryLock()) {
111 unlock();
112 return;
113 }
114 }
115 qWarning(msg: "QMutex: destroying locked mutex");
116}

223/*!
224 \class QRecursiveMutex
225 \inmodule QtCore
226 \since 5.14
227 \brief The QRecursiveMutex class provides access serialization between threads.
228
229 \threadsafe
230
231 \ingroup thread
232
233 The QRecursiveMutex class is a mutex, like QMutex, with which it is
234 API-compatible. It differs from QMutex by accepting lock() calls from
235 the same thread any number of times. QMutex would deadlock in this situation.
236
237 QRecursiveMutex is much more expensive to construct and operate on, so
238 use a plain QMutex whenever you can. Sometimes, one public function,
239 however, calls another public function, and they both need to lock the
240 same mutex. In this case, you have two options:
241
242 \list
243 \li Factor the code that needs mutex protection into private functions,
244 which assume that the mutex is held when they are called, and lock a
245 plain QMutex in the public functions before you call the private
246 implementation ones.
247 \li Or use a recursive mutex, so it doesn't matter that the first public
248 function has already locked the mutex when the second one wishes to do so.
249 \endlist
250
251 \sa QMutex, QMutexLocker, QReadWriteLock, QSemaphore, QWaitCondition
252*/

281/*!
282 Attempts to lock the mutex. This function returns \c true if the lock
283 was obtained; otherwise it returns \c false. If another thread has
284 locked the mutex, this function will wait for at most \a timeout
285 milliseconds for the mutex to become available.
286
287 Note: Passing a negative number as the \a timeout is equivalent to
288 calling lock(), i.e. this function will wait forever until mutex
289 can be locked if \a timeout is negative.
290
291 If the lock was obtained, the mutex must be unlocked with unlock()
292 before another thread can successfully lock it.
293
294 Calling this function multiple times on the same mutex from the
295 same thread is allowed.
296
297 \sa lock(), unlock()
298*/
299bool QRecursiveMutex::tryLock(int timeout) QT_MUTEX_LOCK_NOEXCEPT
300{
301 unsigned tsanFlags = QtTsan::MutexWriteReentrant | QtTsan::TryLock;
302 QtTsan::mutexPreLock(this, tsanFlags);
303
304 Qt::HANDLE self = QThread::currentThreadId();
305 if (owner.loadRelaxed() == self) {
306 ++count;
307 Q_ASSERT_X(count != 0, "QMutex::lock", "Overflow in recursion counter");
308 QtTsan::mutexPostLock(this, tsanFlags, 0);
309 return true;
310 }
311 bool success = true;
312 if (timeout == -1) {
313 mutex.lock();
314 } else {
315 success = mutex.tryLock(timeout);
316 }
317
318 if (success)
319 owner.storeRelaxed(newValue: self);
320 else
321 tsanFlags |= QtTsan::TryLockFailed;
322
323 QtTsan::mutexPostLock(this, tsanFlags, 0);
324
325 return success;
326}

385void QRecursiveMutex::unlock() noexcept
386{
387 Q_ASSERT(owner.loadRelaxed() == QThread::currentThreadId());
388 QtTsan::mutexPreUnlock(this, 0u);
389
390 if (count > 0) {
391 count--;
392 } else {
393 owner.storeRelaxed(newValue: nullptr);
394 mutex.unlock();
395 }
396
397 QtTsan::mutexPostUnlock(this, 0u);
398}

401/*!
402 \class QMutexLocker
403 \inmodule QtCore
404 \brief The QMutexLocker class is a convenience class that simplifies
405 locking and unlocking mutexes.
406
407 \threadsafe
408
409 \ingroup thread
410
411 Locking and unlocking a QMutex or QRecursiveMutex in complex functions and
412 statements or in exception handling code is error-prone and
413 difficult to debug. QMutexLocker can be used in such situations
414 to ensure that the state of the mutex is always well-defined.
415
416 QMutexLocker should be created within a function where a
417 QMutex needs to be locked. The mutex is locked when QMutexLocker
418 is created. You can unlock and relock the mutex with \c unlock()
419 and \c relock(). If locked, the mutex will be unlocked when the
420 QMutexLocker is destroyed.
421
422 For example, this complex function locks a QMutex upon entering
423 the function and unlocks the mutex at all the exit points:
424
425 \snippet code/src_corelib_thread_qmutex.cpp 4
426
427 This example function will get more complicated as it is
428 developed, which increases the likelihood that errors will occur.
429
430 Using QMutexLocker greatly simplifies the code, and makes it more
431 readable:
432
433 \snippet code/src_corelib_thread_qmutex.cpp 5
434
435 Now, the mutex will always be unlocked when the QMutexLocker
436 object is destroyed (when the function returns since \c locker is
437 an auto variable).
438
439 The same principle applies to code that throws and catches
440 exceptions. An exception that is not caught in the function that
441 has locked the mutex has no way of unlocking the mutex before the
442 exception is passed up the stack to the calling function.
443
444 QMutexLocker also provides a \c mutex() member function that returns
445 the mutex on which the QMutexLocker is operating. This is useful
446 for code that needs access to the mutex, such as
447 QWaitCondition::wait(). For example:

449 \snippet code/src_corelib_thread_qmutex.cpp 6

451 \sa QReadLocker, QWriteLocker, QMutex

539/*
540 For a rough introduction on how this works, refer to
541 http://woboq.com/blog/internals-of-qmutex-in-qt5.html
542 which explains a slightly simplified version of it.
543 The differences are that here we try to work with timeout (requires the
544 possiblyUnlocked flag) and that we only wake one thread when unlocking
545 (requires maintaining the waiters count)
546 We also support recursive mutexes which always have a valid d_ptr.
547
548 The waiters flag represents the number of threads that are waiting or about
549 to wait on the mutex. There are two tricks to keep in mind:
550 We don't want to increment waiters after we checked no threads are waiting
551 (waiters == 0). That's why we atomically set the BigNumber flag on waiters when
552 we check waiters. Similarly, if waiters is decremented right after we checked,
553 the mutex would be unlocked (d->wakeUp() has (or will) be called), but there is
554 no thread waiting. This is only happening if there was a timeout in tryLock at the
555 same time as the mutex is unlocked. So when there was a timeout, we set the
556 possiblyUnlocked flag.
557*/
558
559/*
560 * QBasicMutex implementation with futexes (Linux, Windows 10)
561 *
562 * QBasicMutex contains one pointer value, which can contain one of four
563 * different values:
564 * 0x0 unlocked
565 * 0x1 locked, no waiters
566 * 0x3 locked, at least one waiter
567 *
568 * LOCKING:
569 *
570 * A starts in the 0x0 state, indicating that it's unlocked. When the first
571 * thread attempts to lock it, it will perform a testAndSetAcquire
572 * from 0x0 to 0x1. If that succeeds, the caller concludes that it
573 * successfully locked the mutex. That happens in fastTryLock().
574 *
575 * If that testAndSetAcquire fails, QBasicMutex::lockInternal is called.
576 *
577 * lockInternal will examine the value of the pointer. Otherwise, it will use
578 * futexes to sleep and wait for another thread to unlock. To do that, it needs
579 * to set a pointer value of 0x3, which indicates that thread is waiting. It
580 * does that by a simple fetchAndStoreAcquire operation.
581 *
582 * If the pointer value was 0x0, it means we succeeded in acquiring the mutex.
583 * For other values, it will then call FUTEX_WAIT and with an expected value of
584 * 0x3.
585 *
586 * If the pointer value changed before futex(2) managed to sleep, it will
587 * return -1 / EWOULDBLOCK, in which case we have to start over. And even if we
588 * are woken up directly by a FUTEX_WAKE, we need to acquire the mutex, so we
589 * start over again.
590 *
591 * UNLOCKING:
592 *
593 * To unlock, we need to set a value of 0x0 to indicate it's unlocked. The
594 * first attempt is a testAndSetRelease operation from 0x1 to 0x0. If that
595 * succeeds, we're done.
596 *
597 * If it fails, unlockInternal() is called. The only possibility is that the
598 * mutex value was 0x3, which indicates some other thread is waiting or was
599 * waiting in the past. We then set the mutex to 0x0 and perform a FUTEX_WAKE.
600 */
601
602/*!
603 \internal helper for lock()
604 */
605void QBasicMutex::lockInternal() QT_MUTEX_LOCK_NOEXCEPT
606{
607 if (futexAvailable()) {
608 // note we must set to dummyFutexValue because there could be other threads
609 // also waiting
610 while (d_ptr.fetchAndStoreAcquire(newValue: dummyFutexValue()) != nullptr) {
611 // successfully set the waiting bit, now sleep
612 futexWait(futex&: d_ptr, expectedValue: dummyFutexValue());
613
614 // we got woken up, so try to acquire the mutex
615 }
616 Q_ASSERT(d_ptr.loadRelaxed());
617 } else {
618 lockInternal(timeout: -1);
619 }
620}

622/*!
623 \internal helper for lock(int)
624 */
625bool QBasicMutex::lockInternal(int timeout) QT_MUTEX_LOCK_NOEXCEPT
626{
627 if (timeout == 0)
628 return false;
629
630 if (futexAvailable()) {
631 if (Q_UNLIKELY(timeout < 0)) {
632 lockInternal();
633 return true;
634 }
635
636 QDeadlineTimer deadlineTimer(timeout);
637 // The mutex is already locked, set a bit indicating we're waiting.
638 // Note we must set to dummyFutexValue because there could be other threads
639 // also waiting.
640 if (d_ptr.fetchAndStoreAcquire(newValue: dummyFutexValue()) == nullptr)
641 return true;
642
643 qint64 remainingTime = deadlineTimer.remainingTimeNSecs();
644 Q_FOREVER {
645 if (!futexWait(futex&: d_ptr, expectedValue: dummyFutexValue(), nstimeout: remainingTime))
646 return false;
647
648 // We got woken up, so must try to acquire the mutex. We must set
649 // to dummyFutexValue() again because there could be other threads
650 // waiting.
651 if (d_ptr.fetchAndStoreAcquire(newValue: dummyFutexValue()) == nullptr)
652 return true;
653
654 // calculate the remaining time
655 remainingTime = deadlineTimer.remainingTimeNSecs();
656 if (remainingTime <= 0)
657 return false;
658 }
659 }
660
661#if !defined(QT_ALWAYS_USE_FUTEX)
662 while (!fastTryLock()) {
663 QMutexPrivate *copy = d_ptr.loadAcquire();
664 if (!copy) // if d is 0, the mutex is unlocked
665 continue;
666
667 if (copy == dummyLocked()) {
668 if (timeout == 0)
669 return false;
670 // The mutex is locked but does not have a QMutexPrivate yet.
671 // we need to allocate a QMutexPrivate
672 QMutexPrivate *newD = QMutexPrivate::allocate();
673 if (!d_ptr.testAndSetOrdered(dummyLocked(), newD)) {
674 //Either the mutex is already unlocked, or another thread already set it.
675 newD->deref();
676 continue;
677 }
678 copy = newD;
679 //the d->refCount is already 1 the deref will occurs when we unlock
680 }
681
682 QMutexPrivate *d = static_cast<QMutexPrivate *>(copy);
683 if (timeout == 0 && !d->possiblyUnlocked.loadRelaxed())
684 return false;
685
686 // At this point we have a pointer to a QMutexPrivate. But the other thread
687 // may unlock the mutex at any moment and release the QMutexPrivate to the pool.
688 // We will try to reference it to avoid unlock to release it to the pool to make
689 // sure it won't be released. But if the refcount is already 0 it has been released.
690 if (!d->ref())
691 continue; //that QMutexPrivate was already released
692
693 // We now hold a reference to the QMutexPrivate. It won't be released and re-used.
694 // But it is still possible that it was already re-used by another QMutex right before
695 // we did the ref(). So check if we still hold a pointer to the right mutex.
696 if (d != d_ptr.loadAcquire()) {
697 //Either the mutex is already unlocked, or relocked with another mutex
698 d->deref();
699 continue;
700 }
701
702 // In this part, we will try to increment the waiters count.
703 // We just need to take care of the case in which the old_waiters
704 // is set to the BigNumber magic value set in unlockInternal()
705 int old_waiters;
706 do {
707 old_waiters = d->waiters.loadAcquire();
708 if (old_waiters == -QMutexPrivate::BigNumber) {
709 // we are unlocking, and the thread that unlocks is about to change d to 0
710 // we try to acquire the mutex by changing to dummyLocked()
711 if (d_ptr.testAndSetAcquire(d, dummyLocked())) {
712 // Mutex acquired
713 d->deref();
714 return true;
715 } else {
716 Q_ASSERT(d != d_ptr.loadRelaxed()); //else testAndSetAcquire should have succeeded
717 // Mutex is likely to bo 0, we should continue the outer-loop,
718 // set old_waiters to the magic value of BigNumber
719 old_waiters = QMutexPrivate::BigNumber;
720 break;
721 }
722 }
723 } while (!d->waiters.testAndSetRelaxed(old_waiters, old_waiters + 1));
724
725 if (d != d_ptr.loadAcquire()) {
726 // The mutex was unlocked before we incremented waiters.
727 if (old_waiters != QMutexPrivate::BigNumber) {
728 //we did not break the previous loop
729 Q_ASSERT(d->waiters.loadRelaxed() >= 1);
730 d->waiters.deref();
731 }
732 d->deref();
733 continue;
734 }
735
736 if (d->wait(timeout)) {
737 // reset the possiblyUnlocked flag if needed (and deref its corresponding reference)
738 if (d->possiblyUnlocked.loadRelaxed() && d->possiblyUnlocked.testAndSetRelaxed(true, false))
739 d->deref();
740 d->derefWaiters(1);
741 //we got the lock. (do not deref)
742 Q_ASSERT(d == d_ptr.loadRelaxed());
743 return true;
744 } else {
745 Q_ASSERT(timeout >= 0);
746 //timeout
747 d->derefWaiters(1);
748 //There may be a race in which the mutex is unlocked right after we timed out,
749 // and before we deref the waiters, so maybe the mutex is actually unlocked.
750 // Set the possiblyUnlocked flag to indicate this possibility.
751 if (!d->possiblyUnlocked.testAndSetRelaxed(false, true)) {
752 // We keep a reference when possiblyUnlocked is true.
753 // but if possiblyUnlocked was already true, we don't need to keep the reference.
754 d->deref();
755 }
756 return false;
757 }
758 }
759 Q_ASSERT(d_ptr.loadRelaxed() != 0);
760 return true;
761#else
762 Q_UNREACHABLE();
763#endif
764}
765
766/*!
767 \internal
768*/
769void QBasicMutex::unlockInternal() noexcept
770{
771 QMutexPrivate *copy = d_ptr.loadAcquire();
772 Q_ASSERT(copy); //we must be locked
773 Q_ASSERT(copy != dummyLocked()); // testAndSetRelease(dummyLocked(), 0) failed
774
775 if (futexAvailable()) {
776 d_ptr.storeRelease(newValue: nullptr);
777 return futexWakeOne(futex&: d_ptr);
778 }
779
780#if !defined(QT_ALWAYS_USE_FUTEX)
781 QMutexPrivate *d = reinterpret_cast<QMutexPrivate *>(copy);
782
783 // If no one is waiting for the lock anymore, we should reset d to 0x0.
784 // Using fetchAndAdd, we atomically check that waiters was equal to 0, and add a flag
785 // to the waiters variable (BigNumber). That way, we avoid the race in which waiters is
786 // incremented right after we checked, because we won't increment waiters if is
787 // equal to -BigNumber
788 if (d->waiters.fetchAndAddRelease(-QMutexPrivate::BigNumber) == 0) {
789 //there is no one waiting on this mutex anymore, set the mutex as unlocked (d = 0)
790 if (d_ptr.testAndSetRelease(d, 0)) {
791 // reset the possiblyUnlocked flag if needed (and deref its corresponding reference)
792 if (d->possiblyUnlocked.loadRelaxed() && d->possiblyUnlocked.testAndSetRelaxed(true, false))
793 d->deref();
794 }
795 d->derefWaiters(0);
796 } else {
797 d->derefWaiters(0);
798 //there are thread waiting, transfer the lock.
799 d->wakeUp();
800 }
801 d->deref();
802#else
803 Q_UNUSED(copy);
804#endif
805}

830QMutexPrivate *QMutexPrivate::allocate()
831{
832 int i = freelist()->next();
833 QMutexPrivate *d = &(*freelist())[i];
834 d->id = i;
835 Q_ASSERT(d->refCount.loadRelaxed() == 0);
836 Q_ASSERT(!d->possiblyUnlocked.loadRelaxed());
837 Q_ASSERT(d->waiters.loadRelaxed() == 0);
838 d->refCount.storeRelaxed(1);
839 return d;
840}

842void QMutexPrivate::release()
843{
844 Q_ASSERT(refCount.loadRelaxed() == 0);
845 Q_ASSERT(!possiblyUnlocked.loadRelaxed());
846 Q_ASSERT(waiters.loadRelaxed() == 0);
847 freelist()->release(id);
848}

850// atomically subtract "value" to the waiters, and remove the QMutexPrivate::BigNumber flag
851void QMutexPrivate::derefWaiters(int value) noexcept
852{
853 int old_waiters;
854 int new_waiters;
855 do {
856 old_waiters = waiters.loadRelaxed();
857 new_waiters = old_waiters;
858 if (new_waiters < 0) {
859 new_waiters += QMutexPrivate::BigNumber;
860 }
861 new_waiters -= value;
862 } while (!waiters.testAndSetRelaxed(old_waiters, new_waiters));
863}
