27/*
28 Locks 2 mutexes in a defined order, avoiding a recursive lock if
29 we're trying to lock the same mutex twice.
30*/
31class QOrderedMutexLocker
32{
33public:
34 QOrderedMutexLocker(QBasicMutex *m1, QBasicMutex *m2)
35 : mtx1((m1 == m2) ? m1 : (std::less<QBasicMutex *>()(m1, m2) ? m1 : m2)),
36 mtx2((m1 == m2) ? nullptr : (std::less<QBasicMutex *>()(m1, m2) ? m2 : m1)),
37 locked(false)
38 {
39 relock();
40 }
41
42 Q_DISABLE_COPY(QOrderedMutexLocker)
43
44 void swap(QOrderedMutexLocker &other) noexcept
45 {
46 qSwap(value1&: this->mtx1, value2&: other.mtx1);
47 qSwap(value1&: this->mtx2, value2&: other.mtx2);
48 qSwap(value1&: this->locked, value2&: other.locked);
49 }
50
51 QT_MOVE_ASSIGNMENT_OPERATOR_IMPL_VIA_MOVE_AND_SWAP(QOrderedMutexLocker)
52
53 QOrderedMutexLocker(QOrderedMutexLocker &&other) noexcept
54 : mtx1(std::exchange(obj&: other.mtx1, new_val: nullptr))
55 , mtx2(std::exchange(obj&: other.mtx2, new_val: nullptr))
56 , locked(std::exchange(obj&: other.locked, new_val: false))
57 {}
58
59 ~QOrderedMutexLocker()
60 {
61 unlock();
62 }
63
64 void relock()
65 {
66 if (!locked) {
67 if (mtx1) mtx1->lock();
68 if (mtx2) mtx2->lock();
69 locked = true;
70 }
71 }
72
73 /*!
74 \internal
75 Can be called if the mutexes have been unlocked manually, and sets the
76 state of the QOrderedMutexLocker to unlocked.
77 The caller is expected to have unlocked both of them if they
78 are not the same. Calling this method when the QOrderedMutexLocker is
79 unlocked or when the provided mutexes have not actually been unlocked is
80 UB.
81 */
82 void dismiss()
83 {
84 Q_ASSERT(locked);
85 locked = false;
86 }
87
88 void unlock()
89 {
90 if (locked) {
91 if (mtx2) mtx2->unlock();
92 if (mtx1) mtx1->unlock();
93 locked = false;
94 }
95 }
96
97 static bool relock(QBasicMutex *mtx1, QBasicMutex *mtx2)
98 {
99 // mtx1 is already locked, mtx2 not... do we need to unlock and relock?
100 if (mtx1 == mtx2)
101 return false;
102 if (std::less<QBasicMutex *>()(mtx1, mtx2)) {
103 mtx2->lock();
104 return true;
105 }
106 if (!mtx2->tryLock()) {
107 mtx1->unlock();
108 mtx2->lock();
109 mtx1->lock();
110 }
111 return true;
112 }
113
114private:
115 QBasicMutex *mtx1, *mtx2;
116 bool locked;
117};
118
119class QBasicMutexLocker
120{
121public:
122 inline explicit QBasicMutexLocker(QBasicMutex *m) QT_MUTEX_LOCK_NOEXCEPT
123 : m(m), isLocked(true)
124 {
125 m->lock();
126 }
127 inline ~QBasicMutexLocker() { if (isLocked) unlock(); }
128
129 inline void unlock() noexcept
130 {
131 isLocked = false;
132 m->unlock();
133 }
134
135 inline void relock() QT_MUTEX_LOCK_NOEXCEPT
136 {
137 isLocked = true;
138 m->lock();
139 }
140
141private:
142 Q_DISABLE_COPY(QBasicMutexLocker)
143
144 QBasicMutex *m;
145 bool isLocked;
146};

154 QOrderedMutexLocker(QBasicMutex *, QBasicMutex *) {}

157 ~QOrderedMutexLocker() {}

159 void relock() {}
160 void unlock() {}
161 void dismiss() {}
162
163 static bool relock(QBasicMutex *, QBasicMutex *) { return false; }
164};
