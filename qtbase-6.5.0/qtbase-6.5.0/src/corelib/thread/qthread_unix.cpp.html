85static void destroy_current_thread_data(void *p)
86{
87 QThreadData *data = static_cast<QThreadData *>(p);
88 // thread_local variables are set to zero before calling this destructor function,
89 // if they are internally using pthread-specific data management,
90 // so we need to set it back to the right value...
91 currentThreadData = data;
92 if (data->isAdopted) {
93 QThread *thread = data->thread.loadAcquire();
94 Q_ASSERT(thread);
95 QThreadPrivate *thread_p = static_cast<QThreadPrivate *>(QObjectPrivate::get(o: thread));
96 Q_ASSERT(!thread_p->finished);
97 thread_p->finish(thread);
98 }
99 data->deref();
100
101 // ... but we must reset it to zero before returning so we aren't
102 // leaving a dangling pointer.
103 currentThreadData = nullptr;
104}

172QThreadData *QThreadData::current(bool createIfNecessary)
173{
174 QThreadData *data = get_thread_data();
175 if (!data && createIfNecessary) {
176 data = new QThreadData;
177 QT_TRY {
178 set_thread_data(data);
179 data->thread = new QAdoptedThread(data);
180 } QT_CATCH(...) {
181 clear_thread_data();
182 data->deref();
183 data = nullptr;
184 QT_RETHROW;
185 }
186 data->deref();
187 data->isAdopted = true;
188 data->threadId.storeRelaxed(newValue: to_HANDLE(id: pthread_self()));
189 if (!QCoreApplicationPrivate::theMainThread.loadAcquire())
190 QCoreApplicationPrivate::theMainThread.storeRelease(newValue: data->thread.loadRelaxed());
191 }
192 return data;
193}

251template <typename T>
252void terminate_on_exception(T &&t)
253{
254#ifndef QT_NO_EXCEPTIONS
255 try {
256#endif
257 std::forward<T>(t)();
258#ifndef QT_NO_EXCEPTIONS
259#ifdef __GLIBCXX__
260 // POSIX thread cancellation under glibc is implemented by throwing an exception
261 // of this type. Do what libstdc++ is doing and handle it specially in order not to
262 // abort the application if user's code calls a cancellation function.
263 } catch (abi::__forced_unwind &) {
264 throw;
265#endif // __GLIBCXX__
266 } catch (...) {
267 qTerminate();
268 }
269#endif // QT_NO_EXCEPTIONS
270}
271} // unnamed namespace

273void *QThreadPrivate::start(void *arg)
274{
275#if !defined(Q_OS_ANDROID)
276 pthread_setcancelstate(PTHREAD_CANCEL_DISABLE, oldstate: nullptr);
277#endif
278 pthread_cleanup_push(QThreadPrivate::finish, arg);
279
280 terminate_on_exception(t: [&] {
281 QThread *thr = reinterpret_cast<QThread *>(arg);
282 QThreadData *data = QThreadData::get2(thread: thr);
283
284 {
285 QMutexLocker locker(&thr->d_func()->mutex);
286
287 // do we need to reset the thread priority?
288 if (thr->d_func()->priority & ThreadPriorityResetFlag) {
289 thr->d_func()->setPriority(QThread::Priority(thr->d_func()->priority & ~ThreadPriorityResetFlag));
290 }
291
292 // threadId is set in QThread::start()
293 Q_ASSERT(pthread_equal(from_HANDLE<pthread_t>(data->threadId.loadRelaxed()),
294 pthread_self()));
295 set_thread_data(data);
296
297 data->ref();
298 data->quitNow = thr->d_func()->exited;
299 }
300
301 data->ensureEventDispatcher();
302 data->eventDispatcher.loadRelaxed()->startingUp();
303
304#if (defined(Q_OS_LINUX) || defined(Q_OS_MAC) || defined(Q_OS_QNX))
305 {
306 // Sets the name of the current thread. We can only do this
307 // when the thread is starting, as we don't have a cross
308 // platform way of setting the name of an arbitrary thread.
309 if (Q_LIKELY(thr->d_func()->objectName.isEmpty()))
310 setCurrentThreadName(thr->metaObject()->className());
311 else
312 setCurrentThreadName(std::exchange(obj&: thr->d_func()->objectName, new_val: {}).toLocal8Bit());
313 }
314#endif
315
316 emit thr->started(QThread::QPrivateSignal());
317#if !defined(Q_OS_ANDROID)
318 pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, oldstate: nullptr);
319 pthread_testcancel();
320#endif
321 thr->run();
322 });
323
324 // This pop runs finish() below. It's outside the try/catch (and has its
325 // own try/catch) to prevent finish() to be run in case an exception is
326 // thrown.
327 pthread_cleanup_pop(1);
328
329 return nullptr;
330}

332void QThreadPrivate::finish(void *arg)
333{
334 terminate_on_exception(t: [&] {
335 QThread *thr = reinterpret_cast<QThread *>(arg);
336 QThreadPrivate *d = thr->d_func();
337
338 QMutexLocker locker(&d->mutex);
339
340 d->isInFinish = true;
341 d->priority = QThread::InheritPriority;
342 void *data = &d->data->tls;
343 locker.unlock();
344 emit thr->finished(QThread::QPrivateSignal());
345 QCoreApplication::sendPostedEvents(receiver: nullptr, event_type: QEvent::DeferredDelete);
346 QThreadStorageData::finish((void **)data);
347 locker.relock();
348
349 QAbstractEventDispatcher *eventDispatcher = d->data->eventDispatcher.loadRelaxed();
350 if (eventDispatcher) {
351 d->data->eventDispatcher = nullptr;
352 locker.unlock();
353 eventDispatcher->closingDown();
354 delete eventDispatcher;
355 locker.relock();
356 }
357
358 d->running = false;
359 d->finished = true;
360 d->interruptionRequested = false;
361
362 d->isInFinish = false;
363 d->data->threadId.storeRelaxed(newValue: nullptr);
364
365 d->thread_done.wakeAll();
366 });
367}

402int QThread::idealThreadCount() noexcept
403{
404 int cores = 1;
405
406#if defined(Q_OS_HPUX)
407 // HP-UX
408 struct pst_dynamic psd;
409 if (pstat_getdynamic(&psd, sizeof(psd), 1, 0) == -1) {
410 perror("pstat_getdynamic");
411 } else {
412 cores = (int)psd.psd_proc_cnt;
413 }
414#elif (defined(Q_OS_LINUX) && !defined(Q_OS_ANDROID)) || defined(Q_OS_FREEBSD)
415# if defined(Q_OS_FREEBSD) && !defined(CPU_COUNT_S)
416# define CPU_COUNT_S(setsize, cpusetp) ((int)BIT_COUNT(setsize, cpusetp))
417 // match the Linux API for simplicity
418 using cpu_set_t = cpuset_t;
419 auto sched_getaffinity = [](pid_t, size_t cpusetsize, cpu_set_t *mask) {
420 return cpuset_getaffinity(CPU_LEVEL_WHICH, CPU_WHICH_PID, -1, cpusetsize, mask);
421 };
422# endif
423
424 // get the number of threads we're assigned, not the total in the system
425 QVarLengthArray<cpu_set_t, 1> cpuset(1);
426 int size = 1;
427 if (Q_UNLIKELY(sched_getaffinity(0, sizeof(cpu_set_t), cpuset.data()) < 0)) {
428 for (size = 2; size <= 4; size *= 2) {
429 cpuset.resize(sz: size);
430 if (sched_getaffinity(pid: 0, cpusetsize: sizeof(cpu_set_t) * size, cpuset: cpuset.data()) == 0)
431 break;
432 }
433 if (size > 4)
434 return 1;
435 }
436 cores = CPU_COUNT_S(sizeof(cpu_set_t) * size, cpuset.data());
437#elif defined(Q_OS_BSD4)
438 // OpenBSD, NetBSD, BSD/OS, Darwin (macOS, iOS, etc.)
439 size_t len = sizeof(cores);
440 int mib[2];
441 mib[0] = CTL_HW;
442 mib[1] = HW_NCPU;
443 if (sysctl(mib, 2, &cores, &len, NULL, 0) != 0) {
444 perror("sysctl");
445 }
446#elif defined(Q_OS_INTEGRITY)
447#if (__INTEGRITY_MAJOR_VERSION >= 10)
448 // Integrity V10+ does support multicore CPUs
449 Value processorCount;
450 if (GetProcessorCount(CurrentTask(), &processorCount) == 0)
451 cores = processorCount;
452 else
453#endif
454 // as of aug 2008 Integrity only supports one single core CPU
455 cores = 1;
456#elif defined(Q_OS_VXWORKS)
457 // VxWorks
458# if defined(QT_VXWORKS_HAS_CPUSET)
459 cpuset_t cpus = vxCpuEnabledGet();
460 cores = 0;
461
462 // 128 cores should be enough for everyone ;)
463 for (int i = 0; i < 128 && !CPUSET_ISZERO(cpus); ++i) {
464 if (CPUSET_ISSET(cpus, i)) {
465 CPUSET_CLR(cpus, i);
466 cores++;
467 }
468 }
469# else
470 // as of aug 2008 VxWorks < 6.6 only supports one single core CPU
471 cores = 1;
472# endif
473#elif defined(Q_OS_WASM)
474 cores = QThreadPrivate::idealThreadCount;
475#else
476 // the rest: Solaris, AIX, Tru64
477 cores = (int)sysconf(_SC_NPROCESSORS_ONLN);
478 if (cores == -1)
479 return 1;
480#endif
481 return cores;
482}

491static timespec makeTimespec(std::chrono::nanoseconds nsecs)
492{
493 using namespace std::chrono;
494 const seconds secs = duration_cast<seconds>(d: nsecs);
495 const nanoseconds frac = nsecs - secs;
496 struct timespec ts;
497 ts.tv_sec = secs.count();
498 ts.tv_nsec = frac.count();
499 return ts;
500}

502static void qt_nanosleep(timespec amount)
503{
504 // We'd like to use clock_nanosleep.
505 //
506 // But clock_nanosleep is from POSIX.1-2001 and both are *not*
507 // affected by clock changes when using relative sleeps, even for
508 // CLOCK_REALTIME.
509 //
510 // nanosleep is POSIX.1-1993
511
512 int r;
513 EINTR_LOOP(r, nanosleep(&amount, &amount));
514}

533#ifdef QT_HAS_THREAD_PRIORITY_SCHEDULING
534#if defined(Q_OS_QNX)
535static bool calculateUnixPriority(int priority, int *sched_policy, int *sched_priority)
536{
537 // On QNX, NormalPriority is mapped to 10. A QNX system could use a value different
538 // than 10 for the "normal" priority but it's difficult to achieve this so we'll
539 // assume that no one has ever created such a system. This makes the mapping from
540 // Qt priorities to QNX priorities lopsided. There's usually more space available
541 // to map into above the "normal" priority than below it. QNX also has a privileged
542 // priority range (for threads that assist the kernel). We'll assume that no Qt
543 // thread needs to use priorities in that range.
544 int priority_norm = 10;
545 // _sched_info::priority_priv isn't documented. You'd think that it's the start of the
546 // privileged priority range but it's actually the end of the unpriviledged range.
547 struct _sched_info info;
548 if (SchedInfo_r(0, *sched_policy, &info) != EOK)
549 return false;
550
551 if (priority == QThread::IdlePriority) {
552 *sched_priority = info.priority_min;
553 return true;
554 }
555
556 if (priority_norm < info.priority_min)
557 priority_norm = info.priority_min;
558 if (priority_norm > info.priority_priv)
559 priority_norm = info.priority_priv;
560
561 int to_min, to_max;
562 int from_min, from_max;
563 int prio;
564 if (priority < QThread::NormalPriority) {
565 to_min = info.priority_min;
566 to_max = priority_norm;
567 from_min = QThread::LowestPriority;
568 from_max = QThread::NormalPriority;
569 } else {
570 to_min = priority_norm;
571 to_max = info.priority_priv;
572 from_min = QThread::NormalPriority;
573 from_max = QThread::TimeCriticalPriority;
574 }
575
576 prio = ((priority - from_min) * (to_max - to_min)) / (from_max - from_min) + to_min;
577 prio = qBound(to_min, prio, to_max);
578
579 *sched_priority = prio;
580 return true;
581}
582#else
583// Does some magic and calculate the Unix scheduler priorities
584// sched_policy is IN/OUT: it must be set to a valid policy before calling this function
585// sched_priority is OUT only
586static bool calculateUnixPriority(int priority, int *sched_policy, int *sched_priority)
587{
588#ifdef SCHED_IDLE
589 if (priority == QThread::IdlePriority) {
590 *sched_policy = SCHED_IDLE;
591 *sched_priority = 0;
592 return true;
593 }
594 const int lowestPriority = QThread::LowestPriority;
595#else
596 const int lowestPriority = QThread::IdlePriority;
597#endif
598 const int highestPriority = QThread::TimeCriticalPriority;
599
600 int prio_min;
601 int prio_max;
602#if defined(Q_OS_VXWORKS) && defined(VXWORKS_DKM)
603 // for other scheduling policies than SCHED_RR or SCHED_FIFO
604 prio_min = SCHED_FIFO_LOW_PRI;
605 prio_max = SCHED_FIFO_HIGH_PRI;
606
607 if ((*sched_policy == SCHED_RR) || (*sched_policy == SCHED_FIFO))
608#endif
609 {
610 prio_min = sched_get_priority_min(algorithm: *sched_policy);
611 prio_max = sched_get_priority_max(algorithm: *sched_policy);
612 }
613
614 if (prio_min == -1 || prio_max == -1)
615 return false;
616
617 int prio;
618 // crudely scale our priority enum values to the prio_min/prio_max
619 prio = ((priority - lowestPriority) * (prio_max - prio_min) / highestPriority) + prio_min;
620 prio = qMax(a: prio_min, b: qMin(a: prio_max, b: prio));
621
622 *sched_priority = prio;
623 return true;
624}
625#endif
626#endif

628void QThread::start(Priority priority)
629{
630 Q_D(QThread);
631 QMutexLocker locker(&d->mutex);
632
633 if (d->isInFinish)
634 d->thread_done.wait(lockedMutex: locker.mutex());
635
636 if (d->running)
637 return;
638
639 d->running = true;
640 d->finished = false;
641 d->returnCode = 0;
642 d->exited = false;
643 d->interruptionRequested = false;
644
645 pthread_attr_t attr;
646 pthread_attr_init(attr: &attr);
647 pthread_attr_setdetachstate(attr: &attr, PTHREAD_CREATE_DETACHED);
648
649 d->priority = priority;
650
651#if defined(QT_HAS_THREAD_PRIORITY_SCHEDULING)
652 switch (priority) {
653 case InheritPriority:
654 {
655 pthread_attr_setinheritsched(attr: &attr, PTHREAD_INHERIT_SCHED);
656 break;
657 }
658
659 default:
660 {
661 int sched_policy;
662 if (pthread_attr_getschedpolicy(attr: &attr, policy: &sched_policy) != 0) {
663 // failed to get the scheduling policy, don't bother
664 // setting the priority
665 qWarning(msg: "QThread::start: Cannot determine default scheduler policy");
666 break;
667 }
668
669 int prio;
670 if (!calculateUnixPriority(priority, sched_policy: &sched_policy, sched_priority: &prio)) {
671 // failed to get the scheduling parameters, don't
672 // bother setting the priority
673 qWarning(msg: "QThread::start: Cannot determine scheduler priority range");
674 break;
675 }
676
677 sched_param sp;
678 sp.sched_priority = prio;
679
680 if (pthread_attr_setinheritsched(attr: &attr, PTHREAD_EXPLICIT_SCHED) != 0
681 || pthread_attr_setschedpolicy(attr: &attr, policy: sched_policy) != 0
682 || pthread_attr_setschedparam(attr: &attr, param: &sp) != 0) {
683 // could not set scheduling hints, fallback to inheriting them
684 // we'll try again from inside the thread
685 pthread_attr_setinheritsched(attr: &attr, PTHREAD_INHERIT_SCHED);
686 d->priority = qToUnderlying(e: priority) | ThreadPriorityResetFlag;
687 }
688 break;
689 }
690 }
691#endif // QT_HAS_THREAD_PRIORITY_SCHEDULING
692
693
694 if (d->stackSize > 0) {
695#if defined(_POSIX_THREAD_ATTR_STACKSIZE) && (_POSIX_THREAD_ATTR_STACKSIZE-0 > 0)
696 int code = pthread_attr_setstacksize(attr: &attr, stacksize: d->stackSize);
697#else
698 int code = ENOSYS; // stack size not supported, automatically fail
699#endif // _POSIX_THREAD_ATTR_STACKSIZE
700
701 if (code) {
702 qErrnoWarning(code, msg: "QThread::start: Thread stack size error");
703
704 // we failed to set the stacksize, and as the documentation states,
705 // the thread will fail to run...
706 d->running = false;
707 d->finished = false;
708 return;
709 }
710 }
711
712#ifdef Q_OS_INTEGRITY
713 if (Q_LIKELY(objectName().isEmpty()))
714 pthread_attr_setthreadname(&attr, metaObject()->className());
715 else
716 pthread_attr_setthreadname(&attr, objectName().toLocal8Bit());
717#else
718 // avoid interacting with the binding system
719 d->objectName = d->extraData ? d->extraData->objectName.valueBypassingBindings()
720 : QString();
721#endif
722
723 pthread_t threadId;
724 int code = pthread_create(newthread: &threadId, attr: &attr, start_routine: QThreadPrivate::start, arg: this);
725 if (code == EPERM) {
726 // caller does not have permission to set the scheduling
727 // parameters/policy
728#if defined(QT_HAS_THREAD_PRIORITY_SCHEDULING)
729 pthread_attr_setinheritsched(attr: &attr, PTHREAD_INHERIT_SCHED);
730#endif
731 code = pthread_create(newthread: &threadId, attr: &attr, start_routine: QThreadPrivate::start, arg: this);
732 }
733 d->data->threadId.storeRelaxed(newValue: to_HANDLE(id: threadId));
734
735 pthread_attr_destroy(attr: &attr);
736
737 if (code) {
738 qErrnoWarning(code, msg: "QThread::start: Thread creation error");
739
740 d->running = false;
741 d->finished = false;
742 d->data->threadId.storeRelaxed(newValue: nullptr);
743 }
744}

762bool QThread::wait(QDeadlineTimer deadline)
763{
764 Q_D(QThread);
765 QMutexLocker locker(&d->mutex);
766
767 if (from_HANDLE<pthread_t>(id: d->data->threadId.loadRelaxed()) == pthread_self()) {
768 qWarning(msg: "QThread::wait: Thread tried to wait on itself");
769 return false;
770 }
771
772 if (d->finished || !d->running)
773 return true;
774
775 while (d->running) {
776 if (!d->thread_done.wait(lockedMutex: locker.mutex(), deadline))
777 return false;
778 }
779 Q_ASSERT(d->data->threadId.loadRelaxed() == nullptr);
780
781 return true;
782}

801void QThreadPrivate::setPriority(QThread::Priority threadPriority)
802{
803 priority = threadPriority;
804
805 // copied from start() with a few modifications:
806
807#ifdef QT_HAS_THREAD_PRIORITY_SCHEDULING
808 int sched_policy;
809 sched_param param;
810
811 if (pthread_getschedparam(target_thread: from_HANDLE<pthread_t>(id: data->threadId.loadRelaxed()), policy: &sched_policy, param: &param) != 0) {
812 // failed to get the scheduling policy, don't bother setting
813 // the priority
814 qWarning(msg: "QThread::setPriority: Cannot get scheduler parameters");
815 return;
816 }
817
818 int prio;
819 if (!calculateUnixPriority(priority, sched_policy: &sched_policy, sched_priority: &prio)) {
820 // failed to get the scheduling parameters, don't
821 // bother setting the priority
822 qWarning(msg: "QThread::setPriority: Cannot determine scheduler priority range");
823 return;
824 }
825
826 param.sched_priority = prio;
827 int status = pthread_setschedparam(target_thread: from_HANDLE<pthread_t>(id: data->threadId.loadRelaxed()), policy: sched_policy, param: &param);
828
829# ifdef SCHED_IDLE
830 // were we trying to set to idle priority and failed?
831 if (status == -1 && sched_policy == SCHED_IDLE && errno == EINVAL) {
832 // reset to lowest priority possible
833 pthread_getschedparam(target_thread: from_HANDLE<pthread_t>(id: data->threadId.loadRelaxed()), policy: &sched_policy, param: &param);
834 param.sched_priority = sched_get_priority_min(algorithm: sched_policy);
835 pthread_setschedparam(target_thread: from_HANDLE<pthread_t>(id: data->threadId.loadRelaxed()), policy: sched_policy, param: &param);
836 }
837# else
838 Q_UNUSED(status);
839# endif // SCHED_IDLE
840#endif
841}
