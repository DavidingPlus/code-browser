25template <typename T> struct QAtomicAdditiveType
26{
27 typedef T AdditiveT;
28 static const int AddScale = 1;
29};

32 typedef qptrdiff AdditiveT;
33 static const int AddScale = sizeof(T);

41 template <typename T> static void acquireMemoryFence(const T &_q_value) noexcept
42 {
43 BaseClass::orderedMemoryFence(_q_value);
44 }
45 template <typename T> static void releaseMemoryFence(const T &_q_value) noexcept
46 {
47 BaseClass::orderedMemoryFence(_q_value);
48 }
49 template <typename T> static void orderedMemoryFence(const T &) noexcept
50 {
51 }

53 template <typename T> static Q_ALWAYS_INLINE
54 T load(const T &_q_value) noexcept
55 {
56 return _q_value;
57 }

59 template <typename T, typename X> static Q_ALWAYS_INLINE
60 void store(T &_q_value, X newValue) noexcept

62 _q_value = newValue;
63 }

65 template <typename T> static Q_ALWAYS_INLINE
66 T loadRelaxed(const T &_q_value) noexcept

68 return _q_value;
69 }
70
71 template <typename T, typename X> static Q_ALWAYS_INLINE
72 void storeRelaxed(T &_q_value, X newValue) noexcept
73 {
74 _q_value = newValue;

77 template <typename T> static Q_ALWAYS_INLINE
78 T loadAcquire(const T &_q_value) noexcept
79 {
80 T tmp = *static_cast<const volatile T *>(&_q_value);
81 BaseClass::acquireMemoryFence(_q_value);
82 return tmp;
83 }
84
85 template <typename T, typename X> static Q_ALWAYS_INLINE
86 void storeRelease(T &_q_value, X newValue) noexcept
87 {
88 BaseClass::releaseMemoryFence(_q_value);
89 *static_cast<volatile T *>(&_q_value) = newValue;
90 }

92 static inline constexpr bool isReferenceCountingNative() noexcept
93 { return BaseClass::isFetchAndAddNative(); }
94 static inline constexpr bool isReferenceCountingWaitFree() noexcept
95 { return BaseClass::isFetchAndAddWaitFree(); }
96 template <typename T> static Q_ALWAYS_INLINE
97 bool ref(T &_q_value) noexcept
98 {
99 return BaseClass::fetchAndAddRelaxed(_q_value, 1) != T(-1);
100 }
101
102 template <typename T> static Q_ALWAYS_INLINE
103 bool deref(T &_q_value) noexcept
104 {
105 return BaseClass::fetchAndAddRelaxed(_q_value, -1) != 1;
106 }

166 template <typename T, typename X> static Q_ALWAYS_INLINE
167 T fetchAndStoreRelaxed(T &_q_value, X newValue) noexcept
168 {
169 // implement fetchAndStore on top of testAndSet
170 for (;;) {
171 T tmp = loadRelaxed(_q_value);
172 if (BaseClass::testAndSetRelaxed(_q_value, tmp, newValue))
173 return tmp;
174 }
175 }

201 template <typename T> static Q_ALWAYS_INLINE
202 T fetchAndAddRelaxed(T &_q_value, typename QAtomicAdditiveType<T>::AdditiveT valueToAdd) noexcept
203 {
204 // implement fetchAndAdd on top of testAndSet
205 for (;;) {
206 T tmp = BaseClass::loadRelaxed(_q_value);
207 if (BaseClass::testAndSetRelaxed(_q_value, tmp, T(tmp + valueToAdd)))
208 return tmp;
209 }
