4#ifndef QPROMISE_H
5#define QPROMISE_H
6
7#include <QtCore/qglobal.h>
8#include <QtCore/qfutureinterface.h>
9
10#include <utility>
11
12QT_REQUIRE_CONFIG(future);
13
14QT_BEGIN_NAMESPACE
15
16namespace QtPrivate {
17
18template<class T, class U>
19using EnableIfSameOrConvertible = std::enable_if_t<std::is_convertible_v<T, U>>;
20
21} // namespace QtPrivate
22
23template<typename T>
24class QPromise
25{
26 static_assert (std::is_move_constructible_v<T>
27 || std::is_same_v<T, void>,
28 "A move-constructible type or type void is required");
29public:
30 QPromise() = default;
31 Q_DISABLE_COPY(QPromise)
32 QPromise(QPromise<T> &&other) = default;
33 QPromise(const QFutureInterface<T> &other) : d(other) {}
34 QPromise(QFutureInterface<T> &&other) noexcept : d(std::move(other)) {}
35 QT_MOVE_ASSIGNMENT_OPERATOR_IMPL_VIA_MOVE_AND_SWAP(QPromise)
36 ~QPromise()
37 {
38 // If computation is not finished at this point, cancel
39 // potential waits
40 if (d.d && !(d.loadState() & QFutureInterfaceBase::State::Finished)) {
41 d.cancelAndFinish(); // cancel and finalize the state
42 d.runContinuation();
43 }
44 d.cleanContinuation();
45 }
46
47 // Core QPromise APIs
48 QFuture<T> future() const { return d.future(); }
49 template<typename U, typename = QtPrivate::EnableIfSameOrConvertible<U, T>>
50 bool addResult(U &&result, int index = -1)
51 {
52 return d.reportResult(std::forward<U>(result), index);
53 }
54#ifndef QT_NO_EXCEPTIONS
55 void setException(const QException &e) { d.reportException(e); }
56#if QT_VERSION < QT_VERSION_CHECK(7, 0, 0)
57 void setException(std::exception_ptr e) { d.reportException(e); }
58#else
59 void setException(const std::exception_ptr &e) { d.reportException(e); }
60#endif
61#endif
62 void start() { d.reportStarted(); }
63 void finish() { d.reportFinished(); }
64
65 void suspendIfRequested() { d.suspendIfRequested(); }
66
67 bool isCanceled() const { return d.isCanceled(); }
68
69 // Progress methods
70 void setProgressRange(int minimum, int maximum) { d.setProgressRange(minimum, maximum); }
71 void setProgressValue(int progressValue) { d.setProgressValue(progressValue); }
72 void setProgressValueAndText(int progressValue, const QString &progressText)
73 {
74 d.setProgressValueAndText(progressValue, progressText);
75 }
76
77 void swap(QPromise<T> &other) noexcept
78 {
79 d.swap(other.d);
80 }
81
82#if defined(Q_QDOC) // documentation-only simplified signatures
83 bool addResult(const T &result, int index = -1) { }
84 bool addResult(T &&result, int index = -1) { }
85#endif
86private:
87 mutable QFutureInterface<T> d;
88};
89
90template<typename T>
91inline void swap(QPromise<T> &a, QPromise<T> &b) noexcept
92{
93 a.swap(b);
94}
95
96QT_END_NAMESPACE
97
98#endif // QPROMISE_H
