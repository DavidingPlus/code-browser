34 template<class Rep, class Period>
35 static int convertToMilliseconds(std::chrono::duration<Rep, Period> duration)
36 {
37 // N4606 ยง 30.4.1.3.5 [thread.timedmutex.requirements] specifies that a
38 // duration less than or equal to duration.zero() shall result in a
39 // try_lock, unlike QMutex's tryLock with a negative duration which
40 // results in a lock.
41
42 if (duration <= duration.zero())
43 return 0;
44
45 // when converting from 'duration' to milliseconds, make sure that
46 // the result is not shorter than 'duration':
47 std::chrono::milliseconds wait = std::chrono::duration_cast<std::chrono::milliseconds>(duration);
48 if (wait < duration)
49 wait += std::chrono::milliseconds(1);
50 Q_ASSERT(wait >= duration);
51 const auto ms = wait.count();
52 const auto maxInt = (std::numeric_limits<int>::max)();
53
54 return ms < maxInt ? int(ms) : maxInt;
55 }

58class Q_CORE_EXPORT QBasicMutex
59{
60 Q_DISABLE_COPY_MOVE(QBasicMutex)
61public:
62 constexpr QBasicMutex()
63 : d_ptr(nullptr)
64 {}
65
66 // BasicLockable concept
67 inline void lock() QT_MUTEX_LOCK_NOEXCEPT {
68 QtTsan::mutexPreLock(this, 0u);
69
70 if (!fastTryLock())
71 lockInternal();
72
73 QtTsan::mutexPostLock(this, 0u, 0);
74 }
75
76 // BasicLockable concept
77 inline void unlock() noexcept {
78 Q_ASSERT(d_ptr.loadRelaxed()); //mutex must be locked
79
80 QtTsan::mutexPreUnlock(this, 0u);
81
82 if (!fastTryUnlock())
83 unlockInternal();
84
85 QtTsan::mutexPostUnlock(this, 0u);
86 }
87
88 bool tryLock() noexcept {
89 unsigned tsanFlags = QtTsan::TryLock;
90 QtTsan::mutexPreLock(this, tsanFlags);
91
92 const bool success = fastTryLock();
93
94 if (!success)
95 tsanFlags |= QtTsan::TryLockFailed;
96 QtTsan::mutexPostLock(this, tsanFlags, 0);
97
98 return success;
99 }
100
101 // Lockable concept
102 bool try_lock() noexcept { return tryLock(); }
103
104private:
105 inline bool fastTryLock() noexcept {
106 return d_ptr.testAndSetAcquire(expectedValue: nullptr, newValue: dummyLocked());
107 }
108 inline bool fastTryUnlock() noexcept {
109 return d_ptr.testAndSetRelease(expectedValue: dummyLocked(), newValue: nullptr);
110 }
111
112 void lockInternal() QT_MUTEX_LOCK_NOEXCEPT;
113 bool lockInternal(int timeout) QT_MUTEX_LOCK_NOEXCEPT;
114 void unlockInternal() noexcept;
115 void destroyInternal(QMutexPrivate *d);
116
117 QBasicAtomicPointer<QMutexPrivate> d_ptr;
118 static inline QMutexPrivate *dummyLocked() {
119 return reinterpret_cast<QMutexPrivate *>(quintptr(1));
120 }
121
122 friend class QMutex;
123 friend class QMutexPrivate;
124};

126class Q_CORE_EXPORT QMutex : public QBasicMutex
127{
128public:
129 constexpr QMutex() = default;
130 ~QMutex()
131 {
132 QMutexPrivate *d = d_ptr.loadRelaxed();
133 if (d)
134 destroyInternal(d);
135 }
136
137#ifdef Q_QDOC
138 inline void lock() QT_MUTEX_LOCK_NOEXCEPT;
139 inline void unlock() noexcept;
140 bool tryLock() noexcept;
141#endif
142
143 // Lockable concept
144 bool try_lock() noexcept { return tryLock(); }
145
146
147 using QBasicMutex::tryLock;
148 bool tryLock(int timeout) QT_MUTEX_LOCK_NOEXCEPT
149 {
150 unsigned tsanFlags = QtTsan::TryLock;
151 QtTsan::mutexPreLock(this, tsanFlags);
152
153 bool success = fastTryLock();
154
155 if (success) {
156 QtTsan::mutexPostLock(this, tsanFlags, 0);
157 return success;
158 }
159
160 success = lockInternal(timeout);
161
162 if (!success)
163 tsanFlags |= QtTsan::TryLockFailed;
164 QtTsan::mutexPostLock(this, tsanFlags, 0);
165
166 return success;
167 }
168
169 // TimedLockable concept
170 template <class Rep, class Period>
171 bool try_lock_for(std::chrono::duration<Rep, Period> duration)
172 {
173 return tryLock(QtPrivate::convertToMilliseconds(duration));
174 }
175
176 // TimedLockable concept
177 template<class Clock, class Duration>
178 bool try_lock_until(std::chrono::time_point<Clock, Duration> timePoint)
179 {
180 // Implemented in terms of try_lock_for to honor the similar
181 // requirement in N4606 ยง 30.4.1.3 [thread.timedmutex.requirements]/12.
182
183 return try_lock_for(timePoint - Clock::now());
184 }
185};

187class Q_CORE_EXPORT QRecursiveMutex
188{
189 Q_DISABLE_COPY_MOVE(QRecursiveMutex)
190 // written to by the thread that first owns 'mutex';
191 // read during attempts to acquire ownership of 'mutex' from any other thread:
192 QAtomicPointer<void> owner = nullptr;
193 // only ever accessed from the thread that owns 'mutex':
194 uint count = 0;
195 QMutex mutex;
196
197public:
198 constexpr QRecursiveMutex() = default;
199 ~QRecursiveMutex();
200
201
202 // BasicLockable concept
203 void lock() QT_MUTEX_LOCK_NOEXCEPT
204 { tryLock(timeout: -1); }
205 bool tryLock(int timeout = 0) QT_MUTEX_LOCK_NOEXCEPT;
206 // BasicLockable concept
207 void unlock() noexcept;
208
209 // Lockable concept
210 bool try_lock() QT_MUTEX_LOCK_NOEXCEPT { return tryLock(); }
211
212 // TimedLockable concept
213 template <class Rep, class Period>
214 bool try_lock_for(std::chrono::duration<Rep, Period> duration)
215 {
216 return tryLock(timeout: QtPrivate::convertToMilliseconds(duration));
217 }
218
219 // TimedLockable concept
220 template<class Clock, class Duration>
221 bool try_lock_until(std::chrono::time_point<Clock, Duration> timePoint)
222 {
223 // Implemented in terms of try_lock_for to honor the similar
224 // requirement in N4606 ยง 30.4.1.3 [thread.timedmutex.requirements]/12.

226 return try_lock_for(timePoint - Clock::now());
227 }

230template <typename Mutex>
231class [[nodiscard]] QMutexLocker
232{
233public:
234 inline explicit QMutexLocker(Mutex *mutex) QT_MUTEX_LOCK_NOEXCEPT
235 {
236 m_mutex = mutex;
237 if (Q_LIKELY(mutex)) {
238 mutex->lock();
239 m_isLocked = true;
240 }
241 }
242
243 inline QMutexLocker(QMutexLocker &&other) noexcept
244 : m_mutex(std::exchange(other.m_mutex, nullptr)),
245 m_isLocked(std::exchange(other.m_isLocked, false))
246 {}
247
248 QT_MOVE_ASSIGNMENT_OPERATOR_IMPL_VIA_MOVE_AND_SWAP(QMutexLocker)
249
250 inline ~QMutexLocker()
251 {
252 if (m_isLocked)
253 unlock();
254 }
255
256 inline bool isLocked() const noexcept
257 {
258 return m_isLocked;
259 }
260
261 inline void unlock() noexcept
262 {
263 Q_ASSERT(m_isLocked);
264 m_mutex->unlock();
265 m_isLocked = false;
266 }
267
268 inline void relock() QT_MUTEX_LOCK_NOEXCEPT
269 {
270 Q_ASSERT(!m_isLocked);
271 m_mutex->lock();
272 m_isLocked = true;
273 }
274
275 inline void swap(QMutexLocker &other) noexcept
276 {
277 qt_ptr_swap(m_mutex, other.m_mutex);
278 std::swap(m_isLocked, other.m_isLocked);
279 }
280
281 Mutex *mutex() const
282 {
283 return m_mutex;
284 }
285private:
286 Q_DISABLE_COPY(QMutexLocker)
287
288 Mutex *m_mutex;
289 bool m_isLocked = false;
290};
