4#ifndef QFUTURESYNCHRONIZER_H
5#define QFUTURESYNCHRONIZER_H
6
7#include <QtCore/qfuture.h>
8
9QT_REQUIRE_CONFIG(future);
10
11QT_BEGIN_NAMESPACE
12
13
14template <typename T>
15class QFutureSynchronizer
16{
17 Q_DISABLE_COPY(QFutureSynchronizer)
18
19public:
20 QFutureSynchronizer() : m_cancelOnWait(false) { }
21 explicit QFutureSynchronizer(const QFuture<T> &future)
22 : m_cancelOnWait(false)
23 { addFuture(future); }
24 ~QFutureSynchronizer() { waitForFinished(); }
25
26 void setFuture(const QFuture<T> &future)
27 {
28 waitForFinished();
29 m_futures.clear();
30 addFuture(future);
31 }
32
33 void addFuture(const QFuture<T> &future)
34 {
35 m_futures.append(future);
36 }
37
38 void waitForFinished()
39 {
40 if (m_cancelOnWait) {
41 for (int i = 0; i < m_futures.size(); ++i) {
42 m_futures[i].cancel();
43 }
44 }
45
46 for (int i = 0; i < m_futures.size(); ++i) {
47 m_futures[i].waitForFinished();
48 }
49 }
50
51 void clearFutures()
52 {
53 m_futures.clear();
54 }
55
56 QList<QFuture<T> > futures() const
57 {
58 return m_futures;
59 }
60
61 void setCancelOnWait(bool enabled)
62 {
63 m_cancelOnWait = enabled;
64 }
65
66 bool cancelOnWait() const
67 {
68 return m_cancelOnWait;
69 }
70
71protected:
72 QList<QFuture<T>> m_futures;
73 bool m_cancelOnWait;
74};
75
76QT_END_NAMESPACE
77
78#endif // QFUTURESYNCHRONIZER_H
