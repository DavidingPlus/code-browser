12/*!
13 \class QRunnable
14 \inmodule QtCore
15 \since 4.4
16 \brief The QRunnable class is the base class for all runnable objects.
17
18 \ingroup thread
19
20 The QRunnable class is an interface for representing a task or
21 piece of code that needs to be executed, represented by your
22 reimplementation of the run() function.
23
24 You can use QThreadPool to execute your code in a separate
25 thread. QThreadPool deletes the QRunnable automatically if
26 autoDelete() returns \c true (the default). Use setAutoDelete() to
27 change the auto-deletion flag.
28
29 QThreadPool supports executing the same QRunnable more than once
30 by calling QThreadPool::tryStart(this) from within the run() function.
31 If autoDelete is enabled the QRunnable will be deleted when
32 the last thread exits the run function. Calling QThreadPool::start()
33 multiple times with the same QRunnable when autoDelete is enabled
34 creates a race condition and is not recommended.
35
36 \sa QThreadPool
37*/
38
39/*! \fn QRunnable::run()
40 Implement this pure virtual function in your subclass.
41*/

53/*! \fn bool QRunnable::autoDelete() const
54
55 Returns \c true is auto-deletion is enabled; false otherwise.
56
57 If auto-deletion is enabled, QThreadPool will automatically delete
58 this runnable after calling run(); otherwise, ownership remains
59 with the application programmer.
60
61 \sa setAutoDelete(), QThreadPool
62*/
63
64/*! \fn bool QRunnable::setAutoDelete(bool autoDelete)
65
66 Enables auto-deletion if \a autoDelete is true; otherwise
67 auto-deletion is disabled.
68
69 If auto-deletion is enabled, QThreadPool will automatically delete
70 this runnable after calling run(); otherwise, ownership remains
71 with the application programmer.
72
73 Note that this flag must be set before calling
74 QThreadPool::start(). Calling this function after
75 QThreadPool::start() results in undefined behavior.
76
77 \sa autoDelete(), QThreadPool
78*/
79
80class FunctionRunnable : public QRunnable
81{
82 std::function<void()> m_functionToRun;
83public:
84 FunctionRunnable(std::function<void()> functionToRun) : m_functionToRun(std::move(functionToRun))
85 {
86 }
87 void run() override
88 {
89 m_functionToRun();
90 }
91};
92
93/*!
94 \since 5.15
95
96 Creates a QRunnable that calls \a functionToRun in run().
97
98 Auto-deletion is enabled by default.
99
100 \sa run(), autoDelete()
101*/
102QRunnable *QRunnable::create(std::function<void()> functionToRun)
103{
104 return new FunctionRunnable(std::move(functionToRun));
105}
