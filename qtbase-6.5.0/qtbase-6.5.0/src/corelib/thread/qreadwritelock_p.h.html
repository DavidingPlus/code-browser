19#include <QtCore/private/qglobal_p.h>
20#include <QtCore/private/qwaitcondition_p.h>
21#include <QtCore/qvarlengtharray.h>

23QT_REQUIRE_CONFIG(thread);

25QT_BEGIN_NAMESPACE
26
27class QReadWriteLockPrivate
28{
29public:
30 explicit QReadWriteLockPrivate(bool isRecursive = false)
31 : recursive(isRecursive) {}
32
33 QtPrivate::mutex mutex;
34 QtPrivate::condition_variable writerCond;
35 QtPrivate::condition_variable readerCond;
36 int readerCount = 0;
37 int writerCount = 0;
38 int waitingReaders = 0;
39 int waitingWriters = 0;
40 const bool recursive;
41
42 //Called with the mutex locked
43 bool lockForWrite(std::unique_lock<QtPrivate::mutex> &lock, int timeout);
44 bool lockForRead(std::unique_lock<QtPrivate::mutex> &lock, int timeout);
45 void unlock();
46
47 //memory management
48 int id = 0;
49 void release();
50 static QReadWriteLockPrivate *allocate();
51
52 // Recursive mutex handling
53 Qt::HANDLE currentWriter = {};
54
55 struct Reader {
56 Qt::HANDLE handle;
57 int recursionLevel;
58 };
59
60 QVarLengthArray<Reader, 16> currentReaders;
61
62 // called with the mutex unlocked
63 bool recursiveLockForWrite(int timeout);
64 bool recursiveLockForRead(int timeout);
65 void recursiveUnlock();
66};
67Q_DECLARE_TYPEINFO(QReadWriteLockPrivate::Reader, Q_PRIMITIVE_TYPE);
