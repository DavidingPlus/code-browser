121#if QT_CONFIG(cxx11_future)
122template <typename Function, typename... Args>
123QThread *QThread::create(Function &&f, Args &&... args)
124{
125 using DecayedFunction = typename std::decay<Function>::type;
126 auto threadFunction =
127 [f = static_cast<DecayedFunction>(std::forward<Function>(f))](auto &&... largs) mutable -> void
128 {
129 (void)std::invoke(std::move(f), std::forward<decltype(largs)>(largs)...);
130 };
131
132 return createThreadImpl(future: std::async(std::launch::deferred,
133 std::move(threadFunction),
134 std::forward<Args>(args)...));
135}

138/*
139 On architectures and platforms we know, interpret the thread control
140 block (TCB) as a unique identifier for a thread within a process. Otherwise,
141 fall back to a slower but safe implementation.
142
143 As per the documentation of currentThreadId, we return an opaque handle
144 as a thread identifier, and application code is not supposed to use that
145 value for anything. In Qt we use the handle to check if threads are identical,
146 for which the TCB is sufficient.
147
148 So we use the fastest possible way, rather than spend time on returning
149 some pseudo-interoperable value.
150*/
151inline Qt::HANDLE QThread::currentThreadId() noexcept
152{
153 // define is undefed if we have to fall back to currentThreadIdImpl
154#define QT_HAS_FAST_CURRENT_THREAD_ID
155 Qt::HANDLE tid; // typedef to void*
156 static_assert(sizeof(tid) == sizeof(void*));
157 // See https://akkadia.org/drepper/tls.pdf for x86 ABI
158#if defined(Q_PROCESSOR_X86_32) && ((defined(Q_OS_LINUX) && defined(__GLIBC__)) || defined(Q_OS_FREEBSD)) // x86 32-bit always uses GS
159 __asm__("movl %%gs:%c1, %0" : "=r" (tid) : "i" (2 * sizeof(void*)) : );
160#elif defined(Q_PROCESSOR_X86_64) && defined(Q_OS_DARWIN64)
161 // 64bit macOS uses GS, see https://github.com/apple/darwin-xnu/blob/master/libsyscall/os/tsd.h
162 __asm__("movq %%gs:0, %0" : "=r" (tid) : : );
163#elif defined(Q_PROCESSOR_X86_64) && ((defined(Q_OS_LINUX) && defined(__GLIBC__)) || defined(Q_OS_FREEBSD))
164 // x86_64 Linux, BSD uses FS
165 __asm__("movq %%fs:%c1, %0" : "=r" (tid) : "i" (2 * sizeof(void*)) : );
166#elif defined(Q_PROCESSOR_X86_64) && defined(Q_OS_WIN)
167 // See https://en.wikipedia.org/wiki/Win32_Thread_Information_Block
168 // First get the pointer to the TIB
169 quint8 *tib;
170# if defined(Q_CC_MINGW) // internal compiler error when using the intrinsics
171 __asm__("movq %%gs:0x30, %0" : "=r" (tib) : :);
172# else
173 tib = reinterpret_cast<quint8 *>(__readgsqword(0x30));
174# endif
175 // Then read the thread ID
176 tid = *reinterpret_cast<Qt::HANDLE *>(tib + 0x48);
177#elif defined(Q_PROCESSOR_X86_32) && defined(Q_OS_WIN)
178 // First get the pointer to the TIB
179 quint8 *tib;
180# if defined(Q_CC_MINGW) // internal compiler error when using the intrinsics
181 __asm__("movl %%fs:0x18, %0" : "=r" (tib) : :);
182# else
183 tib = reinterpret_cast<quint8 *>(__readfsdword(0x18));
184# endif
185 // Then read the thread ID
186 tid = *reinterpret_cast<Qt::HANDLE *>(tib + 0x24);
187#else
188#undef QT_HAS_FAST_CURRENT_THREAD_ID
189 tid = currentThreadIdImpl();
190#endif
191 return tid;
192}
