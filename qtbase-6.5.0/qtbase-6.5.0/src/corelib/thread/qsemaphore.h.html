16class QSemaphorePrivate;
17
18class Q_CORE_EXPORT QSemaphore
19{
20public:
21 explicit QSemaphore(int n = 0);
22 ~QSemaphore();
23
24 void acquire(int n = 1);
25 bool tryAcquire(int n = 1);
26 bool tryAcquire(int n, int timeout);
27 template <typename Rep, typename Period>
28 bool tryAcquire(int n, std::chrono::duration<Rep, Period> timeout)
29 { return tryAcquire(n, QtPrivate::convertToMilliseconds(timeout)); }
30
31 void release(int n = 1);
32
33 int available() const;
34
35 // std::counting_semaphore compatibility:
36 bool try_acquire() noexcept { return tryAcquire(); }
37 template <typename Rep, typename Period>
38 bool try_acquire_for(const std::chrono::duration<Rep, Period> &timeout)
39 { return tryAcquire(1, timeout); }
40 template <typename Clock, typename Duration>
41 bool try_acquire_until(const std::chrono::time_point<Clock, Duration> &tp)
42 {
43 return try_acquire_for(tp - Clock::now());
44 }
45private:
46 Q_DISABLE_COPY(QSemaphore)
47
48 union {
49 QSemaphorePrivate *d;
50 QBasicAtomicInteger<quintptr> u;
51 QBasicAtomicInteger<quint32> u32[2];
52 QBasicAtomicInteger<quint64> u64;
53 };
54};
55
56class QSemaphoreReleaser
57{
58public:
59 QSemaphoreReleaser() = default;
60 explicit QSemaphoreReleaser(QSemaphore &sem, int n = 1) noexcept
61 : m_sem(&sem), m_n(n) {}
62 explicit QSemaphoreReleaser(QSemaphore *sem, int n = 1) noexcept
63 : m_sem(sem), m_n(n) {}
64 QSemaphoreReleaser(QSemaphoreReleaser &&other) noexcept
65 : m_sem(other.cancel()), m_n(other.m_n) {}
66 QT_MOVE_ASSIGNMENT_OPERATOR_IMPL_VIA_MOVE_AND_SWAP(QSemaphoreReleaser)
67
68 ~QSemaphoreReleaser()
69 {
70 if (m_sem)
71 m_sem->release(n: m_n);
72 }
73
74 void swap(QSemaphoreReleaser &other) noexcept
75 {
76 qt_ptr_swap(lhs&: m_sem, rhs&: other.m_sem);
77 std::swap(a&: m_n, b&: other.m_n);
78 }
79
80 QSemaphore *semaphore() const noexcept
81 { return m_sem; }
82
83 QSemaphore *cancel() noexcept
84 {
85 return std::exchange(obj&: m_sem, new_val: nullptr);
86 }
87
88private:
89 QSemaphore *m_sem = nullptr;
90 int m_n;
91};
