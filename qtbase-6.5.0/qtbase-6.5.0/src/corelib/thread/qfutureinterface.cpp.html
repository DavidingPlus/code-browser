65static inline int switch_on(QAtomicInt &a, int which)
66{
67 return a.fetchAndOrRelaxed(valueToAdd: which) | which;
68}
69
70static inline int switch_off(QAtomicInt &a, int which)
71{
72 return a.fetchAndAndRelaxed(valueToAdd: ~which) & ~which;
73}
74
75static inline int switch_from_to(QAtomicInt &a, int from, int to)
76{
77 const auto adjusted = [&](int old) { return (old & ~from) | to; };
78 int value = a.loadRelaxed();
79 while (!a.testAndSetRelaxed(expectedValue: value, newValue: adjusted(value), currentValue&: value))
80 qYieldCpu();
81 return value;
82}

89void QFutureInterfaceBase::cancel(QFutureInterfaceBase::CancelMode mode)
90{
91 QMutexLocker locker(&d->m_mutex);
92
93 const auto oldState = d->state.loadRelaxed();
94
95 switch (mode) {
96 case CancelMode::CancelAndFinish:
97 if ((oldState & Finished) && (oldState & Canceled))
98 return;
99 switch_from_to(a&: d->state, from: suspendingOrSuspended | Running, to: Canceled | Finished);
100 break;
101 case CancelMode::CancelOnly:
102 if (oldState & Canceled)
103 return;
104 switch_from_to(a&: d->state, from: suspendingOrSuspended, to: Canceled);
105 break;
106 }
107
108 // Cancel the continuations chain
109 QFutureInterfaceBasePrivate *next = d->continuationData;
110 while (next) {
111 next->continuationState = QFutureInterfaceBasePrivate::Canceled;
112 next = next->continuationData;
113 }
114
115 d->waitCondition.wakeAll();
116 d->pausedWaitCondition.wakeAll();
117
118 if (!(oldState & Canceled))
119 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Canceled));
120 if (mode == CancelMode::CancelAndFinish && !(oldState & Finished))
121 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Finished));
122
123 d->isValid = false;
124}

126void QFutureInterfaceBase::setSuspended(bool suspend)
127{
128 QMutexLocker locker(&d->m_mutex);
129 if (suspend) {
130 switch_on(a&: d->state, which: Suspending);
131 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Suspending));
132 } else {
133 switch_off(a&: d->state, which: suspendingOrSuspended);
134 d->pausedWaitCondition.wakeAll();
135 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Resumed));
136 }

139void QFutureInterfaceBase::toggleSuspended()
140{
141 QMutexLocker locker(&d->m_mutex);
142 if (d->state.loadRelaxed() & suspendingOrSuspended) {
143 switch_off(a&: d->state, which: suspendingOrSuspended);
144 d->pausedWaitCondition.wakeAll();
145 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Resumed));
146 } else {
147 switch_on(a&: d->state, which: Suspending);
148 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Suspending));
149 }

152void QFutureInterfaceBase::reportSuspended() const
153{
154 // Needs to be called when pause is in effect,
155 // i.e. no more events will be reported.
156
157 QMutexLocker locker(&d->m_mutex);
158 const int state = d->state.loadRelaxed();
159 if (!(state & Suspending) || (state & Suspended))
160 return;
161
162 switch_from_to(a&: d->state, from: Suspending, to: Suspended);
163 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Suspended));
164}

244void QFutureInterfaceBase::waitForResume()
245{
246 // return early if possible to avoid taking the mutex lock.
247 {
248 const int state = d->state.loadRelaxed();
249 if (!(state & suspendingOrSuspended) || (state & Canceled))
250 return;
251 }
252
253 QMutexLocker lock(&d->m_mutex);
254 const int state = d->state.loadRelaxed();
255 if (!(state & suspendingOrSuspended) || (state & Canceled))
256 return;
257
258 // decrease active thread count since this thread will wait.
259 const ThreadPoolThreadReleaser releaser(d->pool());
260
261 d->pausedWaitCondition.wait(lockedMutex: &d->m_mutex);
262}

264void QFutureInterfaceBase::suspendIfRequested()
265{
266 const auto canSuspend = [] (int state) {
267 // can suspend only if 1) in any suspend-related state; 2) not canceled
268 return (state & suspendingOrSuspended) && !(state & Canceled);
269 };
270
271 // return early if possible to avoid taking the mutex lock.
272 {
273 const int state = d->state.loadRelaxed();
274 if (!canSuspend(state))
275 return;
276 }
277
278 QMutexLocker lock(&d->m_mutex);
279 const int state = d->state.loadRelaxed();
280 if (!canSuspend(state))
281 return;
282
283 // Note: expecting that Suspending and Suspended are mutually exclusive
284 if (!(state & Suspended)) {
285 // switch state in case this is the first invocation
286 switch_from_to(a&: d->state, from: Suspending, to: Suspended);
287 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Suspended));
288 }
289
290 // decrease active thread count since this thread will wait.
291 const ThreadPoolThreadReleaser releaser(d->pool());
292 d->pausedWaitCondition.wait(lockedMutex: &d->m_mutex);
293}

410void QFutureInterfaceBase::waitForResult(int resultIndex)
411{
412 if (d->hasException)
413 d->data.m_exceptionStore.rethrowException();
414
415 QMutexLocker lock(&d->m_mutex);
416 if (!isRunningOrPending())
417 return;
418 lock.unlock();
419
420 // To avoid deadlocks and reduce the number of threads used, try to
421 // run the runnable in the current thread.
422 d->pool()->d_func()->stealAndRunRunnable(runnable: d->runnable);
423
424 lock.relock();
425
426 const int waitIndex = (resultIndex == -1) ? INT_MAX : resultIndex;
427 while (isRunningOrPending() && !d->internal_isResultReadyAt(index: waitIndex))
428 d->waitCondition.wait(lockedMutex: &d->m_mutex);
429
430 if (d->hasException)
431 d->data.m_exceptionStore.rethrowException();
432}
433
434void QFutureInterfaceBase::waitForFinished()
435{
436 QMutexLocker lock(&d->m_mutex);
437 const bool alreadyFinished = isFinished();
438 lock.unlock();
439
440 if (!alreadyFinished) {
441 d->pool()->d_func()->stealAndRunRunnable(runnable: d->runnable);
442
443 lock.relock();
444
445 while (!isFinished())
446 d->waitCondition.wait(lockedMutex: &d->m_mutex);
447 }
448
449 if (d->hasException)
450 d->data.m_exceptionStore.rethrowException();
451}

453void QFutureInterfaceBase::reportResultsReady(int beginIndex, int endIndex)
454{
455 if (beginIndex == endIndex || (d->state.loadRelaxed() & (Canceled|Finished)))
456 return;
457
458 d->waitCondition.wakeAll();
459
460 if (!d->m_progress) {
461 if (d->internal_updateProgressValue(progress: d->m_progressValue + endIndex - beginIndex) == false) {
462 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::ResultsReady,
463 beginIndex,
464 endIndex));
465 return;
466 }
467
468 d->sendCallOuts(callOut1: QFutureCallOutEvent(QFutureCallOutEvent::Progress,
469 d->m_progressValue,
470 QString()),
471 callOut2: QFutureCallOutEvent(QFutureCallOutEvent::ResultsReady,
472 beginIndex,
473 endIndex));
474 return;
475 }
476 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::ResultsReady, beginIndex, endIndex));
477}

501/*!
502 \internal
503 Sets the progress range's minimum and maximum values to \a minimum and
504 \a maximum respectively.
505
506 If \a maximum is smaller than \a minimum, \a minimum becomes the only
507 legal value.
508
509 The progress value is reset to be \a minimum.
510
511 The progress range usage can be disabled by using setProgressRange(0, 0).
512 In this case progress value is also reset to 0.
513
514 The behavior of this method is mostly inspired by
515 \l QProgressBar::setRange.
516*/
517void QFutureInterfaceBase::setProgressRange(int minimum, int maximum)
518{
519 QMutexLocker locker(&d->m_mutex);
520 if (!d->m_progress)
521 d->m_progress.reset(other: new QFutureInterfaceBasePrivate::ProgressData());
522 d->m_progress->minimum = minimum;
523 d->m_progress->maximum = qMax(a: minimum, b: maximum);
524 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::ProgressRange, minimum, maximum));
525 d->m_progressValue = minimum;
526}

533/*!
534 \internal
535 In case of the \a progressValue falling out of the progress range,
536 this method has no effect.
537 Such behavior is inspired by \l QProgressBar::setValue.
538*/
539void QFutureInterfaceBase::setProgressValueAndText(int progressValue,
540 const QString &progressText)
541{
542 QMutexLocker locker(&d->m_mutex);
543 if (!d->m_progress)
544 d->m_progress.reset(other: new QFutureInterfaceBasePrivate::ProgressData());
545
546 const bool useProgressRange = (d->m_progress->maximum != 0) || (d->m_progress->minimum != 0);
547 if (useProgressRange
548 && ((progressValue < d->m_progress->minimum) || (progressValue > d->m_progress->maximum))) {
549 return;
550 }
551
552 if (d->m_progressValue >= progressValue)
553 return;
554
555 if (d->state.loadRelaxed() & (Canceled|Finished))
556 return;
557
558 if (d->internal_updateProgress(progress: progressValue, progressText)) {
559 d->sendCallOut(callOut: QFutureCallOutEvent(QFutureCallOutEvent::Progress,
560 d->m_progressValue,
561 d->m_progress->text));
562 }

656bool QFutureInterfaceBasePrivate::internal_waitForNextResult()
657{
658 if (hasException)
659 return false;
660
661 if (data.m_results.hasNextResult())
662 return true;
663
664 while ((state.loadRelaxed() & QFutureInterfaceBase::Running)
665 && data.m_results.hasNextResult() == false)
666 waitCondition.wait(lockedMutex: &m_mutex);
667
668 return !(state.loadRelaxed() & QFutureInterfaceBase::Canceled)
669 && data.m_results.hasNextResult();
670}

672bool QFutureInterfaceBasePrivate::internal_updateProgressValue(int progress)
673{
674 if (m_progressValue >= progress)
675 return false;
676
677 m_progressValue = progress;
678
679 if (progressTime.isValid() && m_progressValue != 0) // make sure the first and last steps are emitted.
680 if (progressTime.elapsed() < (1000 / MaxProgressEmitsPerSecond))
681 return false;
682
683 progressTime.start();
684 return true;
685
686}
687
688bool QFutureInterfaceBasePrivate::internal_updateProgress(int progress,
689 const QString &progressText)
690{
691 if (m_progressValue >= progress)
692 return false;
693
694 Q_ASSERT(m_progress);
695
696 m_progressValue = progress;
697 m_progress->text = progressText;
698
699 if (progressTime.isValid() && m_progressValue != m_progress->maximum) // make sure the first and last steps are emitted.
700 if (progressTime.elapsed() < (1000 / MaxProgressEmitsPerSecond))
701 return false;
702
703 progressTime.start();
704 return true;
705}

707void QFutureInterfaceBasePrivate::internal_setThrottled(bool enable)
708{
709 // bail out if we are not changing the state
710 if ((enable && (state.loadRelaxed() & QFutureInterfaceBase::Throttled))
711 || (!enable && !(state.loadRelaxed() & QFutureInterfaceBase::Throttled)))
712 return;
713
714 // change the state
715 if (enable) {
716 switch_on(a&: state, which: QFutureInterfaceBase::Throttled);
717 } else {
718 switch_off(a&: state, which: QFutureInterfaceBase::Throttled);
719 if (!(state.loadRelaxed() & suspendingOrSuspended))
720 pausedWaitCondition.wakeAll();
721 }
722}

733void QFutureInterfaceBasePrivate::sendCallOuts(const QFutureCallOutEvent &callOutEvent1,
734 const QFutureCallOutEvent &callOutEvent2)
735{
736 if (outputConnections.isEmpty())
737 return;
738
739 for (int i = 0; i < outputConnections.size(); ++i) {
740 QFutureCallOutInterface *interface = outputConnections.at(i);
741 interface->postCallOutEvent(callOutEvent1);
742 interface->postCallOutEvent(callOutEvent2);
743 }
744}
745
746// This function connects an output interface (for example a QFutureWatcher)
747// to this future. While holding the lock we check the state and ready results
748// and add the appropriate callouts to the queue. In order to avoid deadlocks,
749// the actual callouts are made at the end while not holding the lock.
750void QFutureInterfaceBasePrivate::connectOutputInterface(QFutureCallOutInterface *interface)
751{
752 QMutexLocker locker(&m_mutex);
753
754 const auto currentState = state.loadRelaxed();
755 if (currentState & QFutureInterfaceBase::Started) {
756 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Started));
757 if (m_progress) {
758 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::ProgressRange,
759 m_progress->minimum,
760 m_progress->maximum));
761 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Progress,
762 m_progressValue,
763 m_progress->text));
764 } else {
765 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::ProgressRange,
766 0,
767 0));
768 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Progress,
769 m_progressValue,
770 QString()));
771 }
772 }
773
774 if (!hasException) {
775 QtPrivate::ResultIteratorBase it = data.m_results.begin();
776 while (it != data.m_results.end()) {
777 const int begin = it.resultIndex();
778 const int end = begin + it.batchSize();
779 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::ResultsReady,
780 begin,
781 end));
782 it.batchedAdvance();
783 }
784 }
785
786 if (currentState & QFutureInterfaceBase::Suspended)
787 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Suspended));
788 else if (currentState & QFutureInterfaceBase::Suspending)
789 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Suspending));
790
791 if (currentState & QFutureInterfaceBase::Canceled)
792 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Canceled));
793
794 if (currentState & QFutureInterfaceBase::Finished)
795 interface->postCallOutEvent(QFutureCallOutEvent(QFutureCallOutEvent::Finished));
796
797 outputConnections.append(t: interface);
798}

804 if (index == -1)

806 outputConnections.removeAt(i: index);

821void QFutureInterfaceBase::setContinuation(std::function<void(const QFutureInterfaceBase &)> func,
822 QFutureInterfaceBasePrivate *continuationFutureData)
823{
824 QMutexLocker lock(&d->continuationMutex);
825
826 // If the state is ready, run continuation immediately,
827 // otherwise save it for later.
828 if (isFinished()) {
829 lock.unlock();
830 func(*this);
831 lock.relock();
832 }
833 // Unless the continuation has been cleaned earlier, we have to
834 // store the move-only continuation, to guarantee that the associated
835 // future's data stays alive.
836 if (d->continuationState != QFutureInterfaceBasePrivate::Cleaned) {
837 d->continuation = std::move(func);
838 d->continuationData = continuationFutureData;
839 }
840}

842void QFutureInterfaceBase::cleanContinuation()
843{
844 if (!d)
845 return;
846
847 QMutexLocker lock(&d->continuationMutex);
848 d->continuation = nullptr;
849 d->continuationState = QFutureInterfaceBasePrivate::Cleaned;
850 d->continuationData = nullptr;
851}
852
853void QFutureInterfaceBase::runContinuation() const
854{
855 QMutexLocker lock(&d->continuationMutex);
856 if (d->continuation) {
857 // Save the continuation in a local function, to avoid calling
858 // a null std::function below, in case cleanContinuation() is
859 // called from some other thread right after unlock() below.
860 auto fn = std::move(d->continuation);
861 lock.unlock();
862 fn(*this);
863
864 lock.relock();
865 // Unless the continuation has been cleaned earlier, we have to
866 // store the move-only continuation, to guarantee that the associated
867 // future's data stays alive.
868 if (d->continuationState != QFutureInterfaceBasePrivate::Cleaned)
869 d->continuation = std::move(fn);
870 }
871}
