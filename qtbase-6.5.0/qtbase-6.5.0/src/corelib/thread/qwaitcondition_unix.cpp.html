25#ifdef Q_OS_ANDROID
26// pthread_condattr_setclock is available only since Android 5.0. On older versions, there's
27// a private function for relative waits (hidden in 5.0).
28// Use weakref so we can determine at runtime whether each of them is present.
29static int local_condattr_setclock(pthread_condattr_t*, clockid_t)
30__attribute__((weakref("pthread_condattr_setclock")));
31
32static int local_cond_timedwait_relative(pthread_cond_t*, pthread_mutex_t *, const timespec *)
33__attribute__((weakref("__pthread_cond_timedwait_relative")));
34#endif

36static void report_error(int code, const char *where, const char *what)
37{
38 if (code != 0)
39 qErrnoWarning(code, msg: "%s: %s failure", where, what);
40}

42void qt_initialize_pthread_cond(pthread_cond_t *cond, const char *where)
43{
44 pthread_condattr_t condattr;
45
46 pthread_condattr_init(attr: &condattr);
47#if (_POSIX_MONOTONIC_CLOCK-0 >= 0)
48#if defined(Q_OS_ANDROID)
49 if (local_condattr_setclock && QElapsedTimer::clockType() == QElapsedTimer::MonotonicClock)
50 local_condattr_setclock(&condattr, CLOCK_MONOTONIC);
51#elif !defined(Q_OS_MAC)
52 if (QElapsedTimer::clockType() == QElapsedTimer::MonotonicClock)
53 pthread_condattr_setclock(attr: &condattr, CLOCK_MONOTONIC);
54#endif
55#endif
56 report_error(code: pthread_cond_init(cond: cond, cond_attr: &condattr), where, what: "cv init");
57 pthread_condattr_destroy(attr: &condattr);
58}

60void qt_abstime_for_timeout(timespec *ts, QDeadlineTimer deadline)
61{
62#ifdef Q_OS_MAC
63 // on Mac, qt_gettime() (on qelapsedtimer_mac.cpp) returns ticks related to the Mach absolute time
64 // that doesn't work with pthread
65 // Mac also doesn't have clock_gettime
66 struct timeval tv;
67 qint64 nsec = deadline.remainingTimeNSecs();
68 gettimeofday(&tv, 0);
69 ts->tv_sec = tv.tv_sec + nsec / (1000 * 1000 * 1000);
70 ts->tv_nsec = tv.tv_usec * 1000 + nsec % (1000 * 1000 * 1000);
71
72 normalizedTimespec(*ts);
73#else
74 // depends on QDeadlineTimer's internals!!
75 static_assert(QDeadlineTimerNanosecondsInT2);
76 ts->tv_sec = deadline._q_data().first;
77 ts->tv_nsec = deadline._q_data().second;
78#endif
79}

81class QWaitConditionPrivate
82{
83public:
84 pthread_mutex_t mutex;
85 pthread_cond_t cond;
86 int waiters;
87 int wakeups;
88
89 int wait_relative(QDeadlineTimer deadline)
90 {
91 timespec ti;
92#ifdef Q_OS_ANDROID
93 if (!local_condattr_setclock && local_cond_timedwait_relative) {
94 qint64 nsec = deadline.remainingTimeNSecs();
95 ti.tv_sec = nsec / (1000 * 1000 * 1000);
96 ti.tv_nsec = nsec - ti.tv_sec * 1000 * 1000 * 1000;
97 return local_cond_timedwait_relative(&cond, &mutex, &ti);
98 }
99#endif
100 qt_abstime_for_timeout(ts: &ti, deadline);
101 return pthread_cond_timedwait(cond: &cond, mutex: &mutex, abstime: &ti);
102 }
103
104 bool wait(QDeadlineTimer deadline)
105 {
106 int code;
107 forever {
108 if (!deadline.isForever()) {
109 code = wait_relative(deadline);
110 } else {
111 code = pthread_cond_wait(cond: &cond, mutex: &mutex);
112 }
113 if (code == 0 && wakeups == 0) {
114 // many vendors warn of spurious wakeups from
115 // pthread_cond_wait(), especially after signal delivery,
116 // even though POSIX doesn't allow for it... sigh
117 continue;
118 }
119 break;
120 }
121
122 Q_ASSERT_X(waiters > 0, "QWaitCondition::wait", "internal error (waiters)");
123 --waiters;
124 if (code == 0) {
125 Q_ASSERT_X(wakeups > 0, "QWaitCondition::wait", "internal error (wakeups)");
126 --wakeups;
127 }
128 report_error(code: pthread_mutex_unlock(mutex: &mutex), where: "QWaitCondition::wait()", what: "mutex unlock");
129
130 if (code && code != ETIMEDOUT)
131 report_error(code, where: "QWaitCondition::wait()", what: "cv wait");
132
133 return (code == 0);
134 }
135};

137QWaitCondition::QWaitCondition()
138{
139 d = new QWaitConditionPrivate;
140 report_error(code: pthread_mutex_init(mutex: &d->mutex, mutexattr: nullptr), where: "QWaitCondition", what: "mutex init");
141 qt_initialize_pthread_cond(cond: &d->cond, where: "QWaitCondition");
142 d->waiters = d->wakeups = 0;
143}

145QWaitCondition::~QWaitCondition()
146{
147 report_error(code: pthread_cond_destroy(cond: &d->cond), where: "QWaitCondition", what: "cv destroy");
148 report_error(code: pthread_mutex_destroy(mutex: &d->mutex), where: "QWaitCondition", what: "mutex destroy");
149 delete d;
150}

152void QWaitCondition::wakeOne()
153{
154 report_error(code: pthread_mutex_lock(mutex: &d->mutex), where: "QWaitCondition::wakeOne()", what: "mutex lock");
155 d->wakeups = qMin(a: d->wakeups + 1, b: d->waiters);
156 report_error(code: pthread_cond_signal(cond: &d->cond), where: "QWaitCondition::wakeOne()", what: "cv signal");
157 report_error(code: pthread_mutex_unlock(mutex: &d->mutex), where: "QWaitCondition::wakeOne()", what: "mutex unlock");
158}

160void QWaitCondition::wakeAll()
161{
162 report_error(code: pthread_mutex_lock(mutex: &d->mutex), where: "QWaitCondition::wakeAll()", what: "mutex lock");
163 d->wakeups = d->waiters;
164 report_error(code: pthread_cond_broadcast(cond: &d->cond), where: "QWaitCondition::wakeAll()", what: "cv broadcast");
165 report_error(code: pthread_mutex_unlock(mutex: &d->mutex), where: "QWaitCondition::wakeAll()", what: "mutex unlock");
166}

168bool QWaitCondition::wait(QMutex *mutex, unsigned long time)
169{
170 if (time == std::numeric_limits<unsigned long>::max())
171 return wait(lockedMutex: mutex, deadline: QDeadlineTimer(QDeadlineTimer::Forever));
172 return wait(lockedMutex: mutex, deadline: QDeadlineTimer(time));
173}
174
175bool QWaitCondition::wait(QMutex *mutex, QDeadlineTimer deadline)
176{
177 if (!mutex)
178 return false;
179
180 report_error(code: pthread_mutex_lock(mutex: &d->mutex), where: "QWaitCondition::wait()", what: "mutex lock");
181 ++d->waiters;
182 mutex->unlock();
183
184 bool returnValue = d->wait(deadline);
185
186 mutex->lock();
187
188 return returnValue;
189}

191bool QWaitCondition::wait(QReadWriteLock *readWriteLock, unsigned long time)
192{
193 if (time == std::numeric_limits<unsigned long>::max())
194 return wait(lockedReadWriteLock: readWriteLock, deadline: QDeadlineTimer(QDeadlineTimer::Forever));
195 return wait(lockedReadWriteLock: readWriteLock, deadline: QDeadlineTimer(time));
196}

198bool QWaitCondition::wait(QReadWriteLock *readWriteLock, QDeadlineTimer deadline)
199{
200 if (!readWriteLock)
201 return false;
202 auto previousState = readWriteLock->stateForWaitCondition();
203 if (previousState == QReadWriteLock::Unlocked)
204 return false;
205 if (previousState == QReadWriteLock::RecursivelyLocked) {
206 qWarning(msg: "QWaitCondition: cannot wait on QReadWriteLocks with recursive lockForWrite()");
207 return false;
208 }
209
210 report_error(code: pthread_mutex_lock(mutex: &d->mutex), where: "QWaitCondition::wait()", what: "mutex lock");
211 ++d->waiters;
212
213 readWriteLock->unlock();
214
215 bool returnValue = d->wait(deadline);
216
217 if (previousState == QReadWriteLock::LockedForWrite)
218 readWriteLock->lockForWrite();
219 else
220 readWriteLock->lockForRead();
221
222 return returnValue;
223}
