117struct QLocaleId
118{
119 [[nodiscard]] Q_AUTOTEST_EXPORT static QLocaleId fromName(QStringView name);
120 [[nodiscard]] inline bool operator==(QLocaleId other) const
121 { return language_id == other.language_id && script_id == other.script_id && territory_id == other.territory_id; }
122 [[nodiscard]] inline bool operator!=(QLocaleId other) const
123 { return !operator==(other); }
124 [[nodiscard]] inline bool isValid() const
125 {
126 return language_id <= QLocale::LastLanguage && script_id <= QLocale::LastScript
127 && territory_id <= QLocale::LastTerritory;
128 }
129 [[nodiscard]] inline bool matchesAll() const
130 {
131 return !language_id && !script_id && !territory_id;
132 }
133 // Use as: filter.accept...(candidate)
134 [[nodiscard]] inline bool acceptLanguage(quint16 lang) const
135 {
136 // Always reject AnyLanguage (only used for last entry in locale_data array).
137 // So, when searching for AnyLanguage, accept everything *but* AnyLanguage.
138 return language_id ? lang == language_id : lang;
139 }
140 [[nodiscard]] inline bool acceptScriptTerritory(QLocaleId other) const
141 {
142 return (!territory_id || other.territory_id == territory_id)
143 && (!script_id || other.script_id == script_id);
144 }
145
146 [[nodiscard]] QLocaleId withLikelySubtagsAdded() const;
147 [[nodiscard]] QLocaleId withLikelySubtagsRemoved() const;
148
149 [[nodiscard]] QByteArray name(char separator = '-') const;
150
151 ushort language_id = 0, script_id = 0, territory_id = 0;
152};

222 // this function is meant to be called with the result of stringToDouble or bytearrayToDouble
223 [[nodiscard]] static float convertDoubleToFloat(double d, bool *ok)
224 {
225 if (qIsInf(d))
226 return float(d);
227 if (std::fabs(x: d) > (std::numeric_limits<float>::max)()) {
228 if (ok)
229 *ok = false;
230 const float huge = std::numeric_limits<float>::infinity();
231 return d < 0 ? -huge : huge;
232 }
233 if (d != 0 && float(d) == 0) {
234 // Values that underflow double already failed. Match them:

236 *ok = false;
237 return 0;
238 }

277 struct DataRange
278 {
279 quint16 offset;
280 quint16 size;
281 [[nodiscard]] QString getData(const char16_t *table) const
282 {
283 return size > 0
284 ? QString::fromRawData(reinterpret_cast<const QChar *>(table + offset), size)
285 : QString();
286 }
287 [[nodiscard]] QStringView viewData(const char16_t *table) const
288 {
289 return { reinterpret_cast<const QChar *>(table + offset), size };
290 }
291 [[nodiscard]] QString getListEntry(const char16_t *table, qsizetype index) const
292 {
293 return listEntry(table, index).getData(table);
294 }
295 [[nodiscard]] QStringView viewListEntry(const char16_t *table, qsizetype index) const
296 {
297 return listEntry(table, index).viewData(table);
298 }
299 [[nodiscard]] char32_t ucsFirst(const char16_t *table) const
300 {
301 if (size && !QChar::isSurrogate(ucs4: table[offset]))
302 return table[offset];
303 if (size > 1 && QChar::isHighSurrogate(ucs4: table[offset]))
304 return QChar::surrogateToUcs4(high: table[offset], low: table[offset + 1]);
305 return 0;
306 }
307 private:
308 [[nodiscard]] DataRange listEntry(const char16_t *table, qsizetype index) const
309 {
310 const char16_t separator = ';';
311 quint16 i = 0;
312 while (index > 0 && i < size) {
313 if (table[offset + i] == separator)
314 index--;
315 i++;
316 }
317 quint16 end = i;
318 while (end < size && table[offset + end] != separator)
319 end++;
320 return { .offset: quint16(offset + i), .size: quint16(end - i) };
321 }
322 };

427inline char QLocaleData::numericToCLocale(QStringView in) const
428{
429 Q_ASSERT(in.size() == 1 || (in.size() == 2 && in.at(0).isHighSurrogate()));
430
431 if (in == positiveSign() || in == u"+")
432 return '+';
433
434 if (in == negativeSign() || in == u"-" || in == u"\x2212")
435 return '-';
436
437 if (in == decimalPoint())
438 return '.';
439
440 if (in.compare(other: exponentSeparator(), cs: Qt::CaseInsensitive) == 0)
441 return 'e';
442
443 const QString group = groupSeparator();
444 if (in == group)
445 return ',';
446
447 // In several languages group() is a non-breaking space (U+00A0) or its thin
448 // version (U+202f), which look like spaces. People (and thus some of our
449 // tests) use a regular space instead and complain if it doesn't work.
450 // Should this be extended generally to any case where group is a space ?
451 if ((group == u"\xa0" || group == u"\x202f") && in == u" ")
452 return ',';
453
454 const char32_t inUcs4 = in.size() == 2
455 ? QChar::surrogateToUcs4(high: in.at(n: 0), low: in.at(n: 1)) : in.at(n: 0).unicode();
456 const char32_t zeroUcs4 = zeroUcs();
457 // Must match qlocale_tools.h's unicodeForDigit()
458 if (zeroUcs4 == u'\u3007') {
459 // QTBUG-85409: Suzhou's digits aren't contiguous !
460 if (inUcs4 == zeroUcs4)
461 return '0';
462 if (inUcs4 > u'\u3020' && inUcs4 <= u'\u3029')
463 return inUcs4 - u'\u3020';
464 } else if (zeroUcs4 <= inUcs4 && inUcs4 < zeroUcs4 + 10) {
465 return '0' + inUcs4 - zeroUcs4;
466 }
467 if ('0' <= inUcs4 && inUcs4 <= '9')
468 return inUcs4;
469
470 return 0;
471}

480enum { AsciiSpaceMask = (1u << (' ' - 1)) |
481 (1u << ('\t' - 1)) | // 9: HT - horizontal tab
482 (1u << ('\n' - 1)) | // 10: LF - line feed
483 (1u << ('\v' - 1)) | // 11: VT - vertical tab
484 (1u << ('\f' - 1)) | // 12: FF - form feed
485 (1u << ('\r' - 1)) }; // 13: CR - carriage return
486[[nodiscard]] constexpr inline bool ascii_isspace(uchar c)
487{
488 return c >= 1u && c <= 32u && (AsciiSpaceMask >> uint(c - 1)) & 1u;
489}
