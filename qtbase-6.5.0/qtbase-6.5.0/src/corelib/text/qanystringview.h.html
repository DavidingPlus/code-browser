36 static constexpr size_t SizeMask = (std::numeric_limits<size_t>::max)() / 4;
37#if QT_VERSION >= QT_VERSION_CHECK(7, 0, 0) || defined(QT_BOOTSTRAPPED)
38 static constexpr int SizeShift = 2;
39 static constexpr size_t Latin1Flag = 1;
40#else
41 static constexpr int SizeShift = 0;
42 static constexpr size_t Latin1Flag = SizeMask + 1;
43#endif
44 static constexpr size_t TwoByteCodePointFlag = Latin1Flag << 1;
45 static constexpr size_t TypeMask = ~(SizeMask << SizeShift);
46 static_assert(TypeMask == (Latin1Flag|TwoByteCodePointFlag));
47
48 // Tag bits
49 // 0 0 Utf8
50 // 0 1 Latin1
51 // 1 0 Utf16
52 // 1 1 Unused
53 // ^ ^ latin1
54 // | sizeof code-point == 2
55 enum Tag : size_t {
56 Utf8 = 0,
57 Latin1 = Latin1Flag,
58 Utf16 = TwoByteCodePointFlag,
59 Unused = TypeMask,
60 };

99 template<typename Char>
100 static constexpr bool isAsciiOnlyCharsAtCompileTime(Char *str, qsizetype sz) noexcept
101 {
102 // do not perform check if not at compile time
103#if !(defined(__cpp_lib_is_constant_evaluated) || defined(Q_CC_GNU))
104 Q_UNUSED(str);
105 Q_UNUSED(sz);
106 return false;
107#else
108# if defined(__cpp_lib_is_constant_evaluated)
109 if (!std::is_constant_evaluated())
110 return false;
111# elif defined(Q_CC_GNU) && !defined(Q_CC_CLANG)
112 if (!str || !__builtin_constant_p(*str))
113 return false;
114# endif
115 if constexpr (sizeof(Char) != sizeof(char)) {
116 Q_UNUSED(str);
117 Q_UNUSED(sz);
118 return false;
119 } else {
120 for (qsizetype i = 0; i < sz; ++i) {
121 if (uchar(str[i]) > 0x7f)
122 return false;
123 }
124 return true;
125 }
126#endif
127 }

129 template<typename Char>
130 static constexpr std::size_t encodeType(const Char *str, qsizetype sz) noexcept
131 {
132 // Utf16 if 16 bit, Latin1 if ASCII, else Utf8
133 Q_ASSERT(sz >= 0);
134 Q_ASSERT(sz <= qsizetype(SizeMask));
135 Q_ASSERT(str || !sz);
136 return (std::size_t(sz) << SizeShift)
137 | uint(sizeof(Char) == sizeof(char16_t)) * Tag::Utf16
138 | uint(isAsciiOnlyCharsAtCompileTime(str, sz)) * Tag::Latin1;
139 }

141 template <typename Char>
142 static qsizetype lengthHelperPointer(const Char *str) noexcept
143 {
144#if defined(Q_CC_GNU) && !defined(Q_CC_CLANG)
145 if (__builtin_constant_p(*str)) {
146 qsizetype result = 0;
147 while (*str++ != u'\0')
148 ++result;
149 return result;
150 }
151#endif
152 if constexpr (sizeof(Char) == sizeof(char16_t))
153 return QtPrivate::qustrlen(str: reinterpret_cast<const char16_t*>(str));
154 else
155 return qsizetype(strlen(s: reinterpret_cast<const char*>(str)));
156 }

164 template <typename Char, size_t N>
165 static constexpr qsizetype lengthHelperContainer(const Char (&str)[N]) noexcept
166 {
167 const auto it = std::char_traits<Char>::find(str, N, Char(0));
168 const auto end = it ? it : std::next(str, N);
169 return qsizetype(std::distance(str, end));
170 }

344 [[nodiscard]] constexpr Tag tag() const noexcept { return Tag{m_size & TypeMask}; }
345 [[nodiscard]] constexpr bool isUtf16() const noexcept { return tag() == Tag::Utf16; }
346 [[nodiscard]] constexpr bool isUtf8() const noexcept { return tag() == Tag::Utf8; }
347 [[nodiscard]] constexpr bool isLatin1() const noexcept { return tag() == Tag::Latin1; }
348 [[nodiscard]] constexpr QStringView asStringView() const
349 { return Q_ASSERT(isUtf16()), QStringView{m_data_utf16, size()}; }
350 [[nodiscard]] constexpr q_no_char8_t::QUtf8StringView asUtf8StringView() const
351 { return Q_ASSERT(isUtf8()), q_no_char8_t::QUtf8StringView{m_data_utf8, size()}; }
352 [[nodiscard]] inline constexpr QLatin1StringView asLatin1StringView() const;
353 [[nodiscard]] constexpr size_t charSize() const noexcept { return isUtf16() ? 2 : 1; }
354 constexpr void setSize(qsizetype sz) noexcept { m_size = size_t(sz) | tag(); }
355 constexpr void advanceData(qsizetype delta) noexcept
356 { m_data_utf8 += delta * charSize(); }
357 Q_ALWAYS_INLINE constexpr void verify(qsizetype pos, qsizetype n = 0) const
358 {
359 Q_ASSERT(pos >= 0);
360 Q_ASSERT(pos <= size());
361 Q_ASSERT(n >= 0);
362 Q_ASSERT(n <= size() - pos);
363 }
364 union {
365 const void *m_data;
366 const char *m_data_utf8;
367 const char16_t *m_data_utf16;
368 };
