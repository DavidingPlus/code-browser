300// support for range-based for loop
301class QRegularExpressionMatchIteratorRangeBasedForIterator
302{
303public:
304 using value_type = QRegularExpressionMatch;
305 using difference_type = int;
306 using reference_type = const QRegularExpressionMatch &;
307 using pointer_type = const QRegularExpressionMatch *;
308 using iterator_category = std::forward_iterator_tag;
309
310 QRegularExpressionMatchIteratorRangeBasedForIterator()
311 : m_atEnd(true)
312 {
313 }
314
315 explicit QRegularExpressionMatchIteratorRangeBasedForIterator(const QRegularExpressionMatchIterator &iterator)
316 : m_matchIterator(iterator),
317 m_currentMatch(),
318 m_atEnd(false)
319 {
320 ++*this;
321 }
322
323 const QRegularExpressionMatch &operator*() const
324 {
325 Q_ASSERT_X(!m_atEnd, Q_FUNC_INFO, "operator* called on an iterator already at the end");
326 return m_currentMatch;
327 }
328
329 QRegularExpressionMatchIteratorRangeBasedForIterator &operator++()
330 {
331 Q_ASSERT_X(!m_atEnd, Q_FUNC_INFO, "operator++ called on an iterator already at the end");
332 if (m_matchIterator.hasNext()) {
333 m_currentMatch = m_matchIterator.next();
334 } else {
335 m_currentMatch = QRegularExpressionMatch();
336 m_atEnd = true;
337 }
338
339 return *this;
340 }
341
342 QRegularExpressionMatchIteratorRangeBasedForIterator operator++(int)
343 {
344 QRegularExpressionMatchIteratorRangeBasedForIterator i = *this;
345 ++*this;
346 return i;
347 }
348
349private:
350 // [input.iterators] imposes operator== on us. Unfortunately, it's not
351 // trivial to implement, so just do the bare minimum to satifisfy
352 // Cpp17EqualityComparable.
353 friend bool operator==(const QRegularExpressionMatchIteratorRangeBasedForIterator &lhs,
354 const QRegularExpressionMatchIteratorRangeBasedForIterator &rhs) noexcept
355 {
356 return (&lhs == &rhs);
357 }
358
359 friend bool operator!=(const QRegularExpressionMatchIteratorRangeBasedForIterator &lhs,
360 const QRegularExpressionMatchIteratorRangeBasedForIterator &rhs) noexcept
361 {
362 return !(lhs == rhs);
363 }
364
365 // This is what we really use in a range-based for.
366 friend bool operator==(const QRegularExpressionMatchIteratorRangeBasedForIterator &lhs,
367 QRegularExpressionMatchIteratorRangeBasedForIteratorSentinel) noexcept
368 {
369 return lhs.m_atEnd;
370 }
371
372 friend bool operator!=(const QRegularExpressionMatchIteratorRangeBasedForIterator &lhs,
373 QRegularExpressionMatchIteratorRangeBasedForIteratorSentinel) noexcept
374 {
375 return !lhs.m_atEnd;
376 }
377
378 QRegularExpressionMatchIterator m_matchIterator;
379 QRegularExpressionMatch m_currentMatch;
380 bool m_atEnd;
381};
