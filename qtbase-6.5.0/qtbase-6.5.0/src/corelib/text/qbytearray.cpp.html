112/*! \relates QByteArray
113
114 A safe \c strncpy() function.
115
116 Copies at most \a len bytes from \a src (stopping at \a len or the
117 terminating '\\0' whichever comes first) into \a dst. Guarantees that \a
118 dst is '\\0'-terminated, except when \a dst is \nullptr or \a len is 0. If
119 \a src is \nullptr, returns \nullptr, otherwise returns \a dst.
120
121 This function assumes that \a dst is at least \a len characters
122 long.
123
124 \note If \a dst and \a src overlap, the behavior is undefined.
125
126 \sa qstrcpy()
127*/
128
129char *qstrncpy(char *dst, const char *src, size_t len)
130{
131 if (dst && len > 0) {
132 if (!src) {
133 *dst = '\0';
134 return nullptr;
135 }
136#ifdef Q_CC_MSVC
137 strncpy_s(dst, len, src, len - 1);
138#else
139 strncpy(dest: dst, src: src, n: len);
140#endif
141 dst[len-1] = '\0';
142 }
143 return src ? dst : nullptr;
144}

211/*! \relates QByteArray
212
213 A safe \c stricmp() function.
214
215 Compares \a str1 and \a str2, ignoring differences in the case of any ASCII
216 characters.
217
218 Returns a negative value if \a str1 is less than \a str2, 0 if \a
219 str1 is equal to \a str2 or a positive value if \a str1 is greater
220 than \a str2.
221
222 If both strings are \nullptr, they are deemed equal; otherwise, if either is
223 \nullptr, it is treated as less than the other (even if the other is an
224 empty string).
225
226 \sa qstrcmp(), qstrncmp(), qstrnicmp(), {Character Case},
227 QByteArray::compare()
228*/
229
230int qstricmp(const char *str1, const char *str2)
231{
232 const uchar *s1 = reinterpret_cast<const uchar *>(str1);
233 const uchar *s2 = reinterpret_cast<const uchar *>(str2);
234 if (!s1)
235 return s2 ? -1 : 0;
236 if (!s2)
237 return 1;
238
239 enum { Incomplete = 256 };
240 qptrdiff offset = 0;
241 auto innerCompare = [=, &offset](qptrdiff max, bool unlimited) {
242 max += offset;
243 do {
244 uchar c = s1[offset];
245 if (int res = QtMiscUtils::caseCompareAscii(lhs: c, rhs: s2[offset]))
246 return res;
247 if (!c)
248 return 0;
249 ++offset;
250 } while (unlimited || offset < max);
251 return int(Incomplete);
252 };
253
254#if defined(__SSE4_1__) && !(defined(__SANITIZE_ADDRESS__) || __has_feature(address_sanitizer))
255 enum { PageSize = 4096, PageMask = PageSize - 1 };
256 const __m128i zero = _mm_setzero_si128();
257 forever {
258 // Calculate how many bytes we can load until we cross a page boundary
259 // for either source. This isn't an exact calculation, just something
260 // very quick.
261 quintptr u1 = quintptr(s1 + offset);
262 quintptr u2 = quintptr(s2 + offset);
263 size_t n = PageSize - ((u1 | u2) & PageMask);
264
265 qptrdiff maxoffset = offset + n;
266 for ( ; offset + 16 <= maxoffset; offset += sizeof(__m128i)) {
267 // load 16 bytes from either source
268 __m128i a = _mm_loadu_si128(reinterpret_cast<const __m128i *>(s1 + offset));
269 __m128i b = _mm_loadu_si128(reinterpret_cast<const __m128i *>(s2 + offset));
270
271 // compare the two against each other
272 __m128i cmp = _mm_cmpeq_epi8(a, b);
273
274 // find NUL terminators too
275 cmp = _mm_min_epu8(cmp, a);
276 cmp = _mm_cmpeq_epi8(cmp, zero);
277
278 // was there any difference or a NUL?
279 uint mask = _mm_movemask_epi8(cmp);
280 if (mask) {
281 // yes, find out where
282 uint start = qCountTrailingZeroBits(mask);
283 uint end = sizeof(mask) * 8 - qCountLeadingZeroBits(mask);
284 Q_ASSUME(end >= start);
285 offset += start;
286 n = end - start;
287 break;
288 }
289 }
290
291 // using SIMD could cause a page fault, so iterate byte by byte
292 int res = innerCompare(n, false);
293 if (res != Incomplete)
294 return res;
295 }
296#endif
297
298 return innerCompare(-1, true);
299}

301/*! \relates QByteArray
302
303 A safe \c strnicmp() function.
304
305 Compares at most \a len bytes of \a str1 and \a str2, ignoring differences
306 in the case of any ASCII characters.
307
308 Returns a negative value if \a str1 is less than \a str2, 0 if \a str1
309 is equal to \a str2 or a positive value if \a str1 is greater than \a
310 str2.
311
312 If both strings are \nullptr, they are deemed equal; otherwise, if either is
313 \nullptr, it is treated as less than the other (even if the other is an
314 empty string or \a len is 0).
315
316 \sa qstrcmp(), qstrncmp(), qstricmp(), {Character Case},
317 QByteArray::compare()
318*/
319
320int qstrnicmp(const char *str1, const char *str2, size_t len)
321{
322 const uchar *s1 = reinterpret_cast<const uchar *>(str1);
323 const uchar *s2 = reinterpret_cast<const uchar *>(str2);
324 if (!s1 || !s2)
325 return s1 ? 1 : (s2 ? -1 : 0);
326 for (; len--; ++s1, ++s2) {
327 const uchar c = *s1;
328 if (int res = QtMiscUtils::caseCompareAscii(lhs: c, rhs: *s2))
329 return res;
330 if (!c) // strings are equal
331 break;
332 }
333 return 0;
334}

336/*!
337 \internal
338 \since 5.12
339
340 A helper for QByteArray::compare. Compares \a len1 bytes from \a str1 to \a
341 len2 bytes from \a str2. If \a len2 is -1, then \a str2 is expected to be
342 '\\0'-terminated.
343 */
344int qstrnicmp(const char *str1, qsizetype len1, const char *str2, qsizetype len2)
345{
346 Q_ASSERT(len1 >= 0);
347 Q_ASSERT(len2 >= -1);
348 const uchar *s1 = reinterpret_cast<const uchar *>(str1);
349 const uchar *s2 = reinterpret_cast<const uchar *>(str2);
350 if (!s1 || !len1) {
351 if (len2 == 0)
352 return 0;
353 if (len2 == -1)
354 return (!s2 || !*s2) ? 0 : -1;
355 Q_ASSERT(s2);
356 return -1;
357 }
358 if (!s2)
359 return len1 == 0 ? 0 : 1;
360
361 if (len2 == -1) {
362 // null-terminated str2
363 qsizetype i;
364 for (i = 0; i < len1; ++i) {
365 const uchar c = s2[i];
366 if (!c)
367 return 1;
368
369 if (int res = QtMiscUtils::caseCompareAscii(lhs: s1[i], rhs: c))
370 return res;
371 }
372 return s2[i] ? -1 : 0;
373 } else {
374 // not null-terminated
375 const qsizetype len = qMin(a: len1, b: len2);
376 for (qsizetype i = 0; i < len; ++i) {
377 if (int res = QtMiscUtils::caseCompareAscii(lhs: s1[i], rhs: s2[i]))
378 return res;
379 }
380 if (len1 == len2)
381 return 0;
382 return len1 < len2 ? -1 : 1;
383 }
384}

454/*!
455 \relates QByteArray
456 \since 5.9
457
458 Returns the CRC-16 checksum of \a data.
459
460 The checksum is independent of the byte order (endianness) and will
461 be calculated accorded to the algorithm published in \a standard.
462 By default the algorithm published in ISO 3309 (Qt::ChecksumIso3309) is used.
463
464 \note This function is a 16-bit cache conserving (16 entry table)
465 implementation of the CRC-16-CCITT algorithm.
466*/
467quint16 qChecksum(QByteArrayView data, Qt::ChecksumType standard)
468{
469 quint16 crc = 0x0000;
470 switch (standard) {
471 case Qt::ChecksumIso3309:
472 crc = 0xffff;
473 break;
474 case Qt::ChecksumItuV41:
475 crc = 0x6363;
476 break;
477 }
478 uchar c;
479 const uchar *p = reinterpret_cast<const uchar *>(data.data());
480 qsizetype len = data.size();
481 while (len--) {
482 c = *p++;
483 crc = ((crc >> 4) & 0x0fff) ^ crc_tbl[((crc ^ c) & 15)];
484 c >>= 4;
485 crc = ((crc >> 4) & 0x0fff) ^ crc_tbl[((crc ^ c) & 15)];
486 }
487 switch (standard) {
488 case Qt::ChecksumIso3309:
489 crc = ~crc;
490 break;
491 case Qt::ChecksumItuV41:
492 break;
493 }
494 return crc & 0xffff;
495}

583static QByteArray xxflate(ZLibOp op, QArrayDataPointer<char> out, QByteArrayView input,
584 qxp::function_ref<int(z_stream *) const> init,
585 qxp::function_ref<int(z_stream *, size_t) const> processChunk,
586 qxp::function_ref<void(z_stream *) const> deinit)
587{
588 if (out.data() == nullptr) // allocation failed
589 return tooMuchData(op);
590 qsizetype capacity = out.allocatedCapacity();
591
592 const auto initalSize = out.size;
593
594 z_stream zs = {};
595 zs.next_in = reinterpret_cast<uchar *>(const_cast<char *>(input.data())); // 1980s C API...
596 if (const int err = init(&zs); err != Z_OK)
597 return unexpectedZlibError(op, err, zs.msg);
598 const auto sg = qScopeGuard([&] { deinit(&zs); });
599
600 using ZlibChunkSize_t = decltype(zs.avail_in);
601 static_assert(!std::is_signed_v<ZlibChunkSize_t>);
602 static_assert(std::is_same_v<ZlibChunkSize_t, decltype(zs.avail_out)>);
603 constexpr auto MaxChunkSize = std::numeric_limits<ZlibChunkSize_t>::max();
604 [[maybe_unused]]
605 constexpr auto MaxStatisticsSize = std::numeric_limits<decltype(zs.total_out)>::max();
606
607 size_t inputLeft = size_t(input.size());
608
609 int res;
610 do {
611 Q_ASSERT(out.freeSpaceAtBegin() == 0); // ensure prepend optimization stays out of the way
612 Q_ASSERT(capacity == out.allocatedCapacity());
613
614 if (zs.avail_out == 0) {
615 Q_ASSERT(size_t(out.size) - initalSize > MaxStatisticsSize || // total_out overflow
616 size_t(out.size) - initalSize == zs.total_out);
617 Q_ASSERT(out.size <= capacity);
618
619 qsizetype avail_out = capacity - out.size;
620 if (avail_out == 0) {
621 out->reallocateAndGrow(QArrayData::GrowsAtEnd, 1); // grow to next natural capacity
622 if (out.data() == nullptr) // reallocation failed
623 return tooMuchData(op);
624 capacity = out.allocatedCapacity();
625 avail_out = capacity - out.size;
626 }
627 zs.next_out = reinterpret_cast<uchar *>(out.data()) + out.size;
628 zs.avail_out = size_t(avail_out) > size_t(MaxChunkSize) ? MaxChunkSize
629 : ZlibChunkSize_t(avail_out);
630 out.size += zs.avail_out;
631
632 Q_ASSERT(zs.avail_out > 0);
633 }
634
635 if (zs.avail_in == 0) {
636 // zs.next_in is kept up-to-date by processChunk(), so nothing to do
637 zs.avail_in = inputLeft > MaxChunkSize ? MaxChunkSize : ZlibChunkSize_t(inputLeft);
638 inputLeft -= zs.avail_in;
639 }
640
641 res = processChunk(&zs, inputLeft);
642 } while (res == Z_OK);
643
644 switch (res) {
645 case Z_STREAM_END:
646 out.size -= zs.avail_out;
647 Q_ASSERT(size_t(out.size) - initalSize > MaxStatisticsSize || // total_out overflow
648 size_t(out.size) - initalSize == zs.total_out);
649 Q_ASSERT(out.size <= out.allocatedCapacity());
650 out.data()[out.size] = '\0';
651 return QByteArray(std::move(out));
652
653 case Z_MEM_ERROR:
654 return tooMuchData(op);
655
656 case Z_BUF_ERROR:
657 Q_UNREACHABLE(); // cannot happen - we supply a buffer that can hold the result,
658 // or else error out early
659
660 case Z_DATA_ERROR: // can only happen on decompression
661 Q_ASSERT(op == ZLibOp::Decompression);
662 return invalidCompressedData();
663
664 default:
665 return unexpectedZlibError(op, res, zs.msg);
666 }
667}

669QByteArray qCompress(const uchar* data, qsizetype nbytes, int compressionLevel)
670{
671 constexpr qsizetype HeaderSize = sizeof(CompressSizeHint_t);
672 if (nbytes == 0) {
673 return QByteArray(HeaderSize, '\0');
674 }
675 if (!data)
676 return dataIsNull(ZLibOp::Compression);
677
678 if (nbytes < 0)
679 return lengthIsNegative(ZLibOp::Compression);
680
681 if (compressionLevel < -1 || compressionLevel > 9)
682 compressionLevel = -1;
683
684 QArrayDataPointer out = [&] {
685 constexpr qsizetype SingleAllocLimit = 256 * 1024; // the maximum size for which we use
686 // zlib's compressBound() to guarantee
687 // the output buffer size is sufficient
688 // to hold result
689 qsizetype capacity = HeaderSize;
690 if (nbytes < SingleAllocLimit) {
691 // use maximum size
692 capacity += compressBound(uLong(nbytes)); // cannot overflow (both times)!
693 return QArrayDataPointer{QTypedArrayData<char>::allocate(capacity)};
694 }
695
696 // for larger buffers, assume it compresses optimally, and
697 // grow geometrically from there:
698 constexpr qsizetype MaxCompressionFactor = 1024; // max theoretical factor is 1032
699 // cf. http://www.zlib.org/zlib_tech.html,
700 // but use a nearby power-of-two (faster)
701 capacity += std::max(qsizetype(compressBound(uLong(SingleAllocLimit))),
702 nbytes / MaxCompressionFactor);
703 return QArrayDataPointer{QTypedArrayData<char>::allocate(capacity, QArrayData::Grow)};
704 }();
705
706 if (out.data() == nullptr) // allocation failed
707 return tooMuchData(ZLibOp::Compression);
708
709 qToBigEndian(qt_saturate<CompressSizeHint_t>(nbytes), out.data());
710 out.size = HeaderSize;
711
712 return xxflate(ZLibOp::Compression, std::move(out), {data, nbytes},
713 [=] (z_stream *zs) { return deflateInit(zs, compressionLevel); },
714 [] (z_stream *zs, size_t inputLeft) {
715 return deflate(zs, inputLeft ? Z_NO_FLUSH : Z_FINISH);
716 },
717 [] (z_stream *zs) { deflateEnd(zs); });
