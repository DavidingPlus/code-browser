15namespace {
16struct GenerationalCollator
17{
18 QCollator theCollator;
19 int generation = QLocalePrivate::s_generation.loadRelaxed();
20public:
21 GenerationalCollator() = default;
22 GenerationalCollator(const QCollator &copy) : theCollator(copy) {}
23 QCollator &collator()
24 {
25 int currentGeneration = QLocalePrivate::s_generation.loadRelaxed();
26 if (Q_UNLIKELY(generation != currentGeneration)) {
27 // reinitialize the collator
28 generation = currentGeneration;
29 theCollator = QCollator();
30 }
31 return theCollator;
32 }
33};
34}

37/*!
38 \class QCollator
39 \inmodule QtCore
40 \brief The QCollator class compares strings according to a localized collation algorithm.
41
42 \since 5.2
43
44 \reentrant
45 \ingroup i18n
46 \ingroup string-processing
47 \ingroup shared
48
49 QCollator is initialized with a QLocale. It can then be used to compare and
50 sort strings by using the ordering appropriate for that locale.
51
52 A QCollator object can be used together with template-based sorting
53 algorithms, such as std::sort(), to sort a list with QString entries.
54
55 \snippet code/src_corelib_text_qcollator.cpp 0
56
57 In addition to the locale, several optional flags can be set that influence
58 the result of the collation.
59
60 \note On Linux, Qt is normally compiled to use ICU. When it isn't, all
61 options are ignored and the only supported locales are the system default
62 (that \c{setlocale(LC_COLLATE, nullptr)} would report) and the "C" locale.
63*/
64
65/*!
66 \since 5.13

70 The system locale, when used as default locale, may have a collation locale
71 other than itself (e.g. on Unix, if LC_COLLATE is set differently to LANG in
72 the environment). All other locales are their own collation locales.
73
74 \sa setLocale(), QLocale::collation(), QLocale::setDefault()
75*/

114/*!
115 Assigns \a other to this collator.
116*/
117QCollator &QCollator::operator=(const QCollator &other)
118{
119 if (this != &other) {
120 if (d && !d->ref.deref())
121 delete d;
122 d = other.d;
123 if (d) {
124 // Ensure clean, lest both copies try to init() at the same time:
125 d->ensureInitialized();
126 d->ref.ref();
127 }
128 }
129 return *this;
130}

159/*!
160 \internal
161*/
162void QCollator::detach()
163{
164 if (d->ref.loadRelaxed() != 1) {
165 QCollatorPrivate *x = new QCollatorPrivate(d->locale);
166 if (!d->ref.deref())
167 delete d;
168 d = x;
169 }
170 // All callers need this, because about to modify the object:
171 d->dirty = true;
172}

215/*!
216 Returns case sensitivity of the collator.
217
218 This defaults to case-sensitive until set.
219
220 \note In the C locale, when case-sensitive, all lower-case letters sort
221 after all upper-case letters, where most locales sort each lower-case letter
222 either immediately before or immediately after its upper-case partner. Thus
223 "Zap" sorts before "ape" in the C locale but after in most others.
224
225 \sa setCaseSensitivity()
226*/
227Qt::CaseSensitivity QCollator::caseSensitivity() const
228{
229 return d->caseSensitivity;

246/*!
247 Returns \c true if numeric sorting is enabled, \c false otherwise.
248
249 When \c true, numerals are recognized as numbers and sorted in arithmetic
250 order; for example, 100 sortes after 99. When \c false, numbers are sorted
251 in lexical order, so that 100 sorts before 99 (because 1 is before 9). By
252 default, this option is disabled.
253
254 \sa setNumericMode()
255*/

258 return d->numericMode;
