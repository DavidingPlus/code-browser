94 /// returns 0 on success; errors can only happen if \a u is a surrogate:
95 /// Error if \a u is a low surrogate;
96 /// if \a u is a high surrogate, Error if the next isn't a low one,
97 /// EndOfString if we run into the end of the string.
98 template <typename Traits, typename OutputPtr, typename InputPtr> inline
99 int toUtf8(char16_t u, OutputPtr &dst, InputPtr &src, InputPtr end)
100 {
101 if (!Traits::skipAsciiHandling && u < 0x80) {
102 // U+0000 to U+007F (US-ASCII) - one byte
103 Traits::appendByte(dst, uchar(u));
104 return 0;
105 } else if (u < 0x0800) {
106 // U+0080 to U+07FF - two bytes
107 // first of two bytes
108 Traits::appendByte(dst, 0xc0 | uchar(u >> 6));
109 } else {
110 if (!QChar::isSurrogate(ucs4: u)) {
111 // U+0800 to U+FFFF (except U+D800-U+DFFF) - three bytes
112 if (!Traits::allowNonCharacters && QChar::isNonCharacter(ucs4: u))
113 return Traits::Error;
114
115 // first of three bytes
116 Traits::appendByte(dst, 0xe0 | uchar(u >> 12));
117 } else {
118 // U+10000 to U+10FFFF - four bytes
119 // need to get one extra codepoint
120 if (Traits::availableUtf16(src, end) == 0)
121 return Traits::EndOfString;
122
123 char16_t low = Traits::peekUtf16(src);
124 if (!QChar::isHighSurrogate(ucs4: u))
125 return Traits::Error;
126 if (!QChar::isLowSurrogate(ucs4: low))
127 return Traits::Error;
128
129 Traits::advanceUtf16(src);
130 char32_t ucs4 = QChar::surrogateToUcs4(high: u, low);
131
132 if (!Traits::allowNonCharacters && QChar::isNonCharacter(ucs4))
133 return Traits::Error;
134
135 // first byte
136 Traits::appendByte(dst, 0xf0 | (uchar(ucs4 >> 18) & 0xf));
137
138 // second of four bytes
139 Traits::appendByte(dst, 0x80 | (uchar(ucs4 >> 12) & 0x3f));
140
141 // for the rest of the bytes
142 u = char16_t(ucs4);
143 }
144
145 // second to last byte
146 Traits::appendByte(dst, 0x80 | (uchar(u >> 6) & 0x3f));
147 }
148
149 // last byte
150 Traits::appendByte(dst, 0x80 | (u & 0x3f));
151 return 0;
152 }

154 inline bool isContinuationByte(uchar b)
155 {
156 return (b & 0xc0) == 0x80;
157 }
158
159 /// returns the number of characters consumed (including \a b) in case of success;
160 /// returns negative in case of error: Traits::Error or Traits::EndOfString
161 template <typename Traits, typename OutputPtr, typename InputPtr> inline
162 qsizetype fromUtf8(uchar b, OutputPtr &dst, InputPtr &src, InputPtr end)
163 {
164 qsizetype charsNeeded;
165 char32_t min_uc;
166 char32_t uc;
167
168 if (!Traits::skipAsciiHandling && b < 0x80) {
169 // US-ASCII
170 Traits::appendUtf16(dst, b);
171 return 1;
172 }
173
174 if (!Traits::isTrusted && Q_UNLIKELY(b <= 0xC1)) {
175 // an UTF-8 first character must be at least 0xC0
176 // however, all 0xC0 and 0xC1 first bytes can only produce overlong sequences
177 return Traits::Error;
178 } else if (b < 0xe0) {
179 charsNeeded = 2;
180 min_uc = 0x80;
181 uc = b & 0x1f;
182 } else if (b < 0xf0) {
183 charsNeeded = 3;
184 min_uc = 0x800;
185 uc = b & 0x0f;
186 } else if (b < 0xf5) {
187 charsNeeded = 4;
188 min_uc = 0x10000;
189 uc = b & 0x07;
190 } else {
191 // the last Unicode character is U+10FFFF
192 // it's encoded in UTF-8 as "\xF4\x8F\xBF\xBF"
193 // therefore, a byte higher than 0xF4 is not the UTF-8 first byte
194 return Traits::Error;
195 }
196
197 qptrdiff bytesAvailable = Traits::availableBytes(src, end);
198 if (Q_UNLIKELY(bytesAvailable < charsNeeded - 1)) {
199 // it's possible that we have an error instead of just unfinished bytes
200 if (bytesAvailable > 0 && !isContinuationByte(Traits::peekByte(src, 0)))
201 return Traits::Error;
202 if (bytesAvailable > 1 && !isContinuationByte(Traits::peekByte(src, 1)))
203 return Traits::Error;
204 return Traits::EndOfString;
205 }
206
207 // first continuation character
208 b = Traits::peekByte(src, 0);
209 if (!isContinuationByte(b))
210 return Traits::Error;
211 uc <<= 6;
212 uc |= b & 0x3f;
213
214 if (charsNeeded > 2) {
215 // second continuation character
216 b = Traits::peekByte(src, 1);
217 if (!isContinuationByte(b))
218 return Traits::Error;
219 uc <<= 6;
220 uc |= b & 0x3f;
221
222 if (charsNeeded > 3) {
223 // third continuation character
224 b = Traits::peekByte(src, 2);
225 if (!isContinuationByte(b))
226 return Traits::Error;
227 uc <<= 6;
228 uc |= b & 0x3f;
229 }
230 }
231
232 // we've decoded something; safety-check it
233 if (!Traits::isTrusted) {
234 if (uc < min_uc)
235 return Traits::Error;
236 if (QChar::isSurrogate(ucs4: uc) || uc > QChar::LastValidCodePoint)
237 return Traits::Error;
238 if (!Traits::allowNonCharacters && QChar::isNonCharacter(ucs4: uc))
239 return Traits::Error;
240 }
241
242 // write the UTF-16 sequence
243 if (!QChar::requiresSurrogates(ucs4: uc)) {
244 // UTF-8 decoded and no surrogates are required
245 // detach if necessary
246 Traits::appendUtf16(dst, char16_t(uc));
247 } else {
248 // UTF-8 decoded to something that requires a surrogate pair
249 Traits::appendUcs4(dst, uc);
250 }
251
252 Traits::advanceByte(src, charsNeeded - 1);
253 return charsNeeded;
254 }
255}

311 static int checkUtf8();
312 static bool isUtf8()
313 {
314 Q_CONSTINIT
315 static QBasicAtomicInteger<qint8> result = { 0 };
316 int r = result.loadRelaxed();
317 if (r == 0) {
318 r = checkUtf8();
319 result.storeRelaxed(r);
320 }
321 return r > 0;
322 }
323 static QString convertToUnicode_sys(QByteArrayView, QStringConverter::State *);
324 static QString convertToUnicode(QByteArrayView in, QStringConverter::State *state)
325 {
326 if (isUtf8())
327 return QUtf8::convertToUnicode(in, state);
328 return convertToUnicode_sys(in, state);
329 }
