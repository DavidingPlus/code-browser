65template <typename A, typename B>
66class QStringBuilder : public QStringBuilderBase<QStringBuilder<A, B>, typename QtStringBuilder::ConvertToTypeHelper<typename QConcatenable<A>::ConvertTo, typename QConcatenable<B>::ConvertTo>::ConvertTo>
67{
68public:
69 QStringBuilder(const A &a_, const B &b_) : a(a_), b(b_) {}
70private:
71 friend class QByteArray;
72 friend class QString;
73 template <typename T> T convertTo() const
74 {
75 const qsizetype len = QConcatenable< QStringBuilder<A, B> >::size(*this);
76 T s(len, Qt::Uninitialized);
77
78 // we abuse const_cast / constData here because we know we've just
79 // allocated the data and we're the only reference count
80 typename T::iterator d = const_cast<typename T::iterator>(s.constData());
81 typename T::const_iterator const start = d;
82 QConcatenable< QStringBuilder<A, B> >::appendTo(*this, d);
83
84 if (!QConcatenable< QStringBuilder<A, B> >::ExactSize && len != d - start) {
85 // this resize is necessary since we allocate a bit too much
86 // when dealing with variable sized 8-bit encodings
87 s.resize(d - start);
88 }
89 return s;
90 }
91
92 typedef QConcatenable<QStringBuilder<A, B> > Concatenable;
93public:
94 typedef typename Concatenable::ConvertTo ConvertTo;
95 operator ConvertTo() const { return convertTo<ConvertTo>(); }
96
97 qsizetype size() const { return Concatenable::size(*this); }
98
99 const A &a;
100 const B &b;
101};

267template <qsizetype N> struct QConcatenable<const char[N]> : private QAbstractConcatenable

353template <> struct QConcatenable<QByteArray> : private QAbstractConcatenable
354{
355 typedef QByteArray type;
356 typedef QByteArray ConvertTo;
357 enum { ExactSize = false };
358 static qsizetype size(const QByteArray &ba) { return ba.size(); }
359#ifndef QT_NO_CAST_FROM_ASCII

363 }
364#endif
365 static inline void appendTo(const QByteArray &ba, char *&out)
366 {
367 const char *a = ba.constData();
368 const char * const end = ba.end();

375template <typename A, typename B>
376struct QConcatenable< QStringBuilder<A, B> >
377{
378 typedef QStringBuilder<A, B> type;
379 typedef typename QtStringBuilder::ConvertToTypeHelper<typename QConcatenable<A>::ConvertTo, typename QConcatenable<B>::ConvertTo>::ConvertTo ConvertTo;
380 enum { ExactSize = QConcatenable<A>::ExactSize && QConcatenable<B>::ExactSize };
381 static qsizetype size(const type &p)
382 {
383 return QConcatenable<A>::size(p.a) + QConcatenable<B>::size(p.b);
384 }
385 template<typename T> static inline void appendTo(const type &p, T *&out)
386 {
387 QConcatenable<A>::appendTo(p.a, out);
388 QConcatenable<B>::appendTo(p.b, out);
389 }
390};

411template <typename A, typename B>
412QByteArray &appendToByteArray(QByteArray &a, const QStringBuilder<A, B> &b, char)
413{
414 // append 8-bit data to a byte array
415 qsizetype len = a.size() + QConcatenable< QStringBuilder<A, B> >::size(b);
416 a.detach(); // a detach() in a.data() could reset a.capacity() to a.size()
417 if (len > a.data_ptr().freeSpaceAtEnd()) // capacity() was broken when prepend()-optimization landed
418 a.reserve(asize: qMax(a: len, b: 2 * a.capacity()));
419 char *it = a.data() + a.size();
420 QConcatenable< QStringBuilder<A, B> >::appendTo(b, it);
421 a.resize(size: len); //we need to resize after the appendTo for the case str+=foo+str
422 return a;
423}

428{
429 return a += QString(b).toUtf8();

441template <typename A, typename B>
442QString &operator+=(QString &a, const QStringBuilder<A, B> &b)
443{
444 qsizetype len = a.size() + QConcatenable< QStringBuilder<A, B> >::size(b);
445 a.detach(); // a detach() in a.data() could reset a.capacity() to a.size()
446 if (len > a.data_ptr().freeSpaceAtEnd()) // capacity() was broken when prepend()-optimization landed
447 a.reserve(asize: qMax(a: len, b: 2 * a.capacity()));
448 QChar *it = a.data() + a.size();
449 QConcatenable< QStringBuilder<A, B> >::appendTo(b, it);
450 // we need to resize after the appendTo for the case str+=foo+str
451 a.resize(size: it - a.constData()); //may be smaller than len if there was conversion from utf8
452 return a;
453}
