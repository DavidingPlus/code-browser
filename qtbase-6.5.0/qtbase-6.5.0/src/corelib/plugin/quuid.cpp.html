16// 16 bytes (a uint, two shorts and a uchar[8]), each represented by two hex
17// digits; plus four dashes and a pair of enclosing brace: 16*2 + 4 + 2 = 38.
18enum { MaxStringUuidLength = 38 };
19
20template <class Integral>
21void _q_toHex(char *&dst, Integral value)
22{
23 value = qToBigEndian(value);
24
25 const char *p = reinterpret_cast<const char *>(&value);
26
27 for (uint i = 0; i < sizeof(Integral); ++i, dst += 2) {
28 dst[0] = QtMiscUtils::toHexLower(value: (p[i] >> 4) & 0xf);
29 dst[1] = QtMiscUtils::toHexLower(value: p[i] & 0xf);
30 }
31}
32
33template <class Integral>
34bool _q_fromHex(const char *&src, Integral &value)
35{
36 value = 0;
37
38 for (uint i = 0; i < sizeof(Integral) * 2; ++i) {
39 uint ch = *src++;
40 int tmp = QtMiscUtils::fromHex(c: ch);
41 if (tmp == -1)
42 return false;
43
44 value = value * 16 + tmp;
45 }
46
47 return true;
48}

50static char *_q_uuidToHex(const QUuid &uuid, char *dst, QUuid::StringFormat mode = QUuid::WithBraces)
51{
52 if ((mode & QUuid::WithoutBraces) == 0)
53 *dst++ = '{';
54 _q_toHex(dst, value: uuid.data1);
55 if ((mode & QUuid::Id128) != QUuid::Id128)
56 *dst++ = '-';
57 _q_toHex(dst, value: uuid.data2);
58 if ((mode & QUuid::Id128) != QUuid::Id128)
59 *dst++ = '-';
60 _q_toHex(dst, value: uuid.data3);
61 if ((mode & QUuid::Id128) != QUuid::Id128)
62 *dst++ = '-';
63 for (int i = 0; i < 2; i++)
64 _q_toHex(dst, value: uuid.data4[i]);
65 if ((mode & QUuid::Id128) != QUuid::Id128)
66 *dst++ = '-';
67 for (int i = 2; i < 8; i++)
68 _q_toHex(dst, value: uuid.data4[i]);
69 if ((mode & QUuid::WithoutBraces) == 0)
70 *dst++ = '}';
71 return dst;
72}

74/*!
75 \internal
76
77 Parses the string representation of a UUID (with optional surrounding "{}")
78 by reading at most MaxStringUuidLength (38) characters from \a src, which
79 may be \nullptr. Stops at the first invalid character (which includes a
80 premature NUL).
81
82 Returns the successfully parsed QUuid, or a null QUuid in case of failure.
83*/
84Q_NEVER_INLINE
85static QUuid _q_uuidFromHex(const char *src)
86{
87 uint d1;
88 ushort d2, d3;
89 uchar d4[8];
90
91 if (src) {
92 if (*src == '{')
93 src++;
94 if (Q_LIKELY( _q_fromHex(src, d1)
95 && *src++ == '-'
96 && _q_fromHex(src, d2)
97 && *src++ == '-'
98 && _q_fromHex(src, d3)
99 && *src++ == '-'
100 && _q_fromHex(src, d4[0])
101 && _q_fromHex(src, d4[1])
102 && *src++ == '-'
103 && _q_fromHex(src, d4[2])
104 && _q_fromHex(src, d4[3])
105 && _q_fromHex(src, d4[4])
106 && _q_fromHex(src, d4[5])
107 && _q_fromHex(src, d4[6])
108 && _q_fromHex(src, d4[7]))) {
109 return QUuid(d1, d2, d3, d4[0], d4[1], d4[2], d4[3], d4[4], d4[5], d4[6], d4[7]);
110 }
111 }
112
113 return QUuid();
114}

116static QUuid createFromName(const QUuid &ns, const QByteArray &baseData, QCryptographicHash::Algorithm algorithm, int version)
117{
118 QCryptographicHash hash(algorithm);
119 hash.addData(data: ns.toRfc4122());
120 hash.addData(data: baseData);
121 QByteArrayView hashResult = hash.resultView();
122 Q_ASSERT(hashResult.size() >= 16);
123 hashResult.truncate(n: 16); // Sha1 will be too long
124
125 QUuid result = QUuid::fromRfc4122(hashResult);
126
127 result.data3 &= 0x0FFF;
128 result.data3 |= (version << 12);
129 result.data4[0] &= 0x3F;
130 result.data4[0] |= 0x80;
131
132 return result;
133}

135/*!
136 \class QUuid
137 \inmodule QtCore
138 \brief The QUuid class stores a Universally Unique Identifier (UUID).
139
140 \reentrant
141
142 Using \e{U}niversally \e{U}nique \e{ID}entifiers (UUID) is a
143 standard way to uniquely identify entities in a distributed
144 computing environment. A UUID is a 16-byte (128-bit) number
145 generated by some algorithm that is meant to guarantee that the
146 UUID will be unique in the distributed computing environment where
147 it is used. The acronym GUID is often used instead, \e{G}lobally
148 \e{U}nique \e{ID}entifiers, but it refers to the same thing.
149
150 \target Variant field
151 Actually, the GUID is one \e{variant} of UUID. Multiple variants
152 are in use. Each UUID contains a bit field that specifies which
153 type (variant) of UUID it is. Call variant() to discover which
154 type of UUID an instance of QUuid contains. It extracts the three
155 most significant bits of byte 8 of the 16 bytes. In QUuid, byte 8
156 is \c{QUuid::data4[0]}. If you create instances of QUuid using the
157 constructor that accepts all the numeric values as parameters, use
158 the following table to set the three most significant bits of
159 parameter \c{b1}, which becomes \c{QUuid::data4[0]} and contains
160 the variant field in its three most significant bits. In the
161 table, 'x' means \e {don't care}.
162
163 \table
164 \header
165 \li msb0
166 \li msb1
167 \li msb2
168 \li Variant
169
170 \row
171 \li 0
172 \li x
173 \li x
174 \li NCS (Network Computing System)
175
176 \row
177 \li 1
178 \li 0
179 \li x
180 \li DCE (Distributed Computing Environment)
181
182 \row
183 \li 1
184 \li 1
185 \li 0
186 \li Microsoft (GUID)
187
188 \row
189 \li 1
190 \li 1
191 \li 1
192 \li Reserved for future expansion
193
194 \endtable
195
196 \target Version field
197 If variant() returns QUuid::DCE, the UUID also contains a
198 \e{version} field in the four most significant bits of
199 \c{QUuid::data3}, and you can call version() to discover which
200 version your QUuid contains. If you create instances of QUuid
201 using the constructor that accepts all the numeric values as
202 parameters, use the following table to set the four most
203 significant bits of parameter \c{w2}, which becomes
204 \c{QUuid::data3} and contains the version field in its four most
205 significant bits.
206
207 \table
208 \header
209 \li msb0
210 \li msb1
211 \li msb2
212 \li msb3
213 \li Version
214
215 \row
216 \li 0
217 \li 0
218 \li 0
219 \li 1
220 \li Time
221
222 \row
223 \li 0
224 \li 0
225 \li 1
226 \li 0
227 \li Embedded POSIX
228
229 \row
230 \li 0
231 \li 0
232 \li 1
233 \li 1
234 \li Md5(Name)
235
236 \row
237 \li 0
238 \li 1
239 \li 0
240 \li 0
241 \li Random
242
243 \row
244 \li 0
245 \li 1
246 \li 0
247 \li 1
248 \li Sha1
249
250 \endtable
251
252 The field layouts for the DCE versions listed in the table above
253 are specified in the \l{RFC 4122}
254 {Network Working Group UUID Specification}.
255
256 Most platforms provide a tool for generating new UUIDs, e.g. \c
257 uuidgen and \c guidgen. You can also use createUuid(). UUIDs
258 generated by createUuid() are of the random type. Their
259 QUuid::Version bits are set to QUuid::Random, and their
260 QUuid::Variant bits are set to QUuid::DCE. The rest of the UUID is
261 composed of random numbers. Theoretically, this means there is a
262 small chance that a UUID generated by createUuid() will not be
263 unique. But it is
264 \l{http://en.wikipedia.org/wiki/Universally_Unique_Identifier#Random_UUID_probability_of_duplicates}
265 {a \e{very} small chance}.
266
267 UUIDs can be constructed from numeric values or from strings, or
268 using the static createUuid() function. They can be converted to a
269 string with toString(). UUIDs have a variant() and a version(),
270 and null UUIDs return true from isNull().
271*/

349/*!
350 \fn static QUuid::fromString(QAnyStringView string)
351 \since 5.10
352
353 Creates a QUuid object from the string \a string, which must be
354 formatted as five hex fields separated by '-', e.g.,
355 "{xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx}" where each 'x' is a hex
356 digit. The curly braces shown here are optional, but it is normal to
357 include them. If the conversion fails, a null UUID is returned. See
358 toString() for an explanation of how the five hex fields map to the
359 public data members in QUuid.
360
361 \note In Qt versions prior to 6.3, this function was an overload
362 set consisting of QStringView and QLatin1StringView instead of
363 one function taking QAnyStringView.
364
365 \sa toString(), QUuid()
366*/
367static QUuid uuidFromString(QStringView text) noexcept
368{
369 if (text.size() > MaxStringUuidLength)
370 text.truncate(n: MaxStringUuidLength);
371
372 char latin1[MaxStringUuidLength + 1];
373 char *dst = latin1;
374
375 for (QChar ch : text)
376 *dst++ = ch.toLatin1();
377
378 *dst++ = '\0'; // don't read garbage as potentially valid data
379
380 return _q_uuidFromHex(src: latin1);
381}

383static QUuid uuidFromString(QLatin1StringView text) noexcept
384{
385 if (Q_UNLIKELY(text.size() < MaxStringUuidLength - 2
386 || (text.front() == '{' && text.size() < MaxStringUuidLength - 1))) {
387 // Too short. Don't call _q_uuidFromHex(); QL1Ss need not be NUL-terminated,
388 // and we don't want to read trailing garbage as potentially valid data.
389 text = QLatin1StringView();
390 }
391 return _q_uuidFromHex(src: text.data());
392}

457/*!
458 Creates a QUuid object from the binary representation of the UUID, as
459 specified by RFC 4122 section 4.1.2. See toRfc4122() for a further
460 explanation of the order of \a bytes required.
461
462 The byte array accepted is NOT a human readable format.
463
464 If the conversion fails, a null UUID is created.
465
466 \note In Qt versions prior to 6.3, this function took QByteArray, not
467 QByteArrayView.
468
469 \since 4.8
470
471 \sa toRfc4122(), QUuid()
472*/
473QUuid QUuid::fromRfc4122(QByteArrayView bytes) noexcept
474{
475 if (bytes.isEmpty() || bytes.size() != 16)
476 return QUuid();
477
478 uint d1;
479 ushort d2, d3;
480 uchar d4[8];
481
482 const uchar *data = reinterpret_cast<const uchar *>(bytes.data());
483
484 d1 = qFromBigEndian<quint32>(src: data);
485 data += sizeof(quint32);
486 d2 = qFromBigEndian<quint16>(src: data);
487 data += sizeof(quint16);
488 d3 = qFromBigEndian<quint16>(src: data);
489 data += sizeof(quint16);
490
491 for (int i = 0; i < 8; ++i) {
492 d4[i] = *(data);
493 data++;
494 }
495
496 return QUuid(d1, d2, d3, d4[0], d4[1], d4[2], d4[3], d4[4], d4[5], d4[6], d4[7]);
497}

597 Returns the binary representation of this QUuid. The byte array is in big
598 endian format, and formatted according to RFC 4122, section 4.1.2 -
599 "Layout and byte order".
600
601 The order is as follows:
602
603 \table
604 \header
605 \li Field #
606 \li Source
607
608 \row
609 \li 1
610 \li data1
611
612 \row
613 \li 2
614 \li data2
615
616 \row
617 \li 3
618 \li data3
619
620 \row
621 \li 4
622 \li data4[0] .. data4[7]
623
624 \endtable
625
626 \since 4.8
627*/
628QByteArray QUuid::toRfc4122() const
629{
630 // we know how many bytes a UUID has, I hope :)
631 QByteArray bytes(16, Qt::Uninitialized);
632 uchar *data = reinterpret_cast<uchar *>(bytes.data());
633
634 qToBigEndian(src: data1, dest: data);
635 data += sizeof(quint32);
636 qToBigEndian(src: data2, dest: data);
637 data += sizeof(quint16);
638 qToBigEndian(src: data3, dest: data);
639 data += sizeof(quint16);
640
641 for (int i = 0; i < 8; ++i) {
642 *(data) = data4[i];
643 data++;
644 }
645
646 return bytes;
647}

649#ifndef QT_NO_DATASTREAM
650/*!
651 \relates QUuid
652 Writes the UUID \a id to the data stream \a s.
653*/
654QDataStream &operator<<(QDataStream &s, const QUuid &id)
655{
656 QByteArray bytes;
657 if (s.byteOrder() == QDataStream::BigEndian) {
658 bytes = id.toRfc4122();
659 } else {
660 // we know how many bytes a UUID has, I hope :)
661 bytes = QByteArray(16, Qt::Uninitialized);
662 uchar *data = reinterpret_cast<uchar *>(bytes.data());
663
664 qToLittleEndian(id.data1, data);
665 data += sizeof(quint32);
666 qToLittleEndian(id.data2, data);
667 data += sizeof(quint16);
668 qToLittleEndian(id.data3, data);
669 data += sizeof(quint16);
670
671 for (int i = 0; i < 8; ++i) {
672 *(data) = id.data4[i];
673 data++;
674 }
675 }
676
677 if (s.writeRawData(bytes.data(), 16) != 16) {
678 s.setStatus(QDataStream::WriteFailed);
679 }
680 return s;
681}

683/*!
684 \relates QUuid
685 Reads a UUID from the stream \a s into \a id.
686*/
687QDataStream &operator>>(QDataStream &s, QUuid &id)
688{
689 std::array<char, 16> bytes;
690 if (s.readRawData(bytes.data(), 16) != 16) {
691 s.setStatus(QDataStream::ReadPastEnd);
692 return s;
693 }
694
695 if (s.byteOrder() == QDataStream::BigEndian) {
696 id = QUuid::fromRfc4122(bytes);
697 } else {
698 const uchar *data = reinterpret_cast<const uchar *>(bytes.data());
699
700 id.data1 = qFromLittleEndian<quint32>(data);
701 data += sizeof(quint32);
702 id.data2 = qFromLittleEndian<quint16>(data);
703 data += sizeof(quint16);
704 id.data3 = qFromLittleEndian<quint16>(data);
705 data += sizeof(quint16);
706
707 for (int i = 0; i < 8; ++i) {
708 id.data4[i] = *(data);
709 data++;
710 }
711 }
712
713 return s;
714}

728/*!
729 \enum QUuid::Variant
730
731 This enum defines the values used in the \l{Variant field}
732 {variant field} of the UUID. The value in the variant field
733 determines the layout of the 128-bit value.
734
735 \value VarUnknown Variant is unknown
736 \value NCS Reserved for NCS (Network Computing System) backward compatibility
737 \value DCE Distributed Computing Environment, the scheme used by QUuid
738 \value Microsoft Reserved for Microsoft backward compatibility (GUID)
739 \value Reserved Reserved for future definition
740*/
741
742/*!
743 \enum QUuid::Version
744
745 This enum defines the values used in the \l{Version field}
746 {version field} of the UUID. The version field is meaningful
747 only if the value in the \l{Variant field} {variant field}
748 is QUuid::DCE.
749
750 \value VerUnknown Version is unknown
751 \value Time Time-based, by using timestamp, clock sequence, and
752 MAC network card address (if available) for the node sections
753 \value EmbeddedPOSIX DCE Security version, with embedded POSIX UUIDs
754 \value Name Name-based, by using values from a name for all sections
755 \value Md5 Alias for Name
756 \value Random Random-based, by using random numbers for all sections
757 \value Sha1
758*/

760/*!
761 \fn QUuid::Variant QUuid::variant() const
762
763 Returns the value in the \l{Variant field} {variant field} of the
764 UUID. If the return value is QUuid::DCE, call version() to see
765 which layout it uses. The null UUID is considered to be of an
766 unknown variant.
767
768 \sa version()
769*/
770QUuid::Variant QUuid::variant() const noexcept
771{
772 if (isNull())
773 return VarUnknown;
774 // Check the 3 MSB of data4[0]
775 if ((data4[0] & 0x80) == 0x00) return NCS;
776 else if ((data4[0] & 0xC0) == 0x80) return DCE;
777 else if ((data4[0] & 0xE0) == 0xC0) return Microsoft;
778 else if ((data4[0] & 0xE0) == 0xE0) return Reserved;
779 return VarUnknown;
780}
781
782/*!
783 \fn QUuid::Version QUuid::version() const
784
785 Returns the \l{Version field} {version field} of the UUID, if the
786 UUID's \l{Variant field} {variant field} is QUuid::DCE. Otherwise
787 it returns QUuid::VerUnknown.
788
789 \sa variant()
790*/
791QUuid::Version QUuid::version() const noexcept
792{
793 // Check the 4 MSB of data3
794 Version ver = (Version)(data3>>12);
795 if (isNull()
796 || (variant() != DCE)
797 || ver < Time
798 || ver > Sha1)
799 return VerUnknown;
800 return ver;
801}

803/*!
804 \fn bool QUuid::operator<(const QUuid &other) const
805
806 Returns \c true if this QUuid has the same \l{Variant field}
807 {variant field} as the \a other QUuid and is lexicographically
808 \e{before} the \a other QUuid. If the \a other QUuid has a
809 different variant field, the return value is determined by
810 comparing the two \l{QUuid::Variant} {variants}.
811
812 \sa variant()
813*/
814bool QUuid::operator<(const QUuid &other) const noexcept
815{
816 if (variant() != other.variant())
817 return variant() < other.variant();
818
819#define ISLESS(f1, f2) if (f1!=f2) return (f1<f2);
820 ISLESS(data1, other.data1);
821 ISLESS(data2, other.data2);
822 ISLESS(data3, other.data3);
823 for (int n = 0; n < 8; n++) {
824 ISLESS(data4[n], other.data4[n]);
825 }
826#undef ISLESS
827 return false;
828}

830/*!
831 \fn bool QUuid::operator>(const QUuid &other) const
832
833 Returns \c true if this QUuid has the same \l{Variant field}
834 {variant field} as the \a other QUuid and is lexicographically
835 \e{after} the \a other QUuid. If the \a other QUuid has a
836 different variant field, the return value is determined by
837 comparing the two \l{QUuid::Variant} {variants}.
838
839 \sa variant()
840*/
841bool QUuid::operator>(const QUuid &other) const noexcept
842{
843 return other < *this;
844}

941/*!
942 \since 5.0
943 \relates QUuid
944 Returns a hash of the UUID \a uuid, using \a seed to seed the calculation.
945*/
946size_t qHash(const QUuid &uuid, size_t seed) noexcept
947{
948 return uuid.data1 ^ uuid.data2 ^ (uuid.data3 << 16)
949 ^ ((uuid.data4[0] << 24) | (uuid.data4[1] << 16) | (uuid.data4[2] << 8) | uuid.data4[3])
950 ^ ((uuid.data4[4] << 24) | (uuid.data4[5] << 16) | (uuid.data4[6] << 8) | uuid.data4[7])
951 ^ seed;
952}
