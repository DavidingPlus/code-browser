161struct ElfHeaderCommonCheck
162{
163 static_assert(std::is_same_v<decltype(Elf32_Ehdr::e_ident), decltype(Elf64_Ehdr::e_ident)>,
164 "e_ident field is not the same in both Elf32_Ehdr and Elf64_Ehdr");
165
166 // bytes 0-3
167 static bool checkElfMagic(const uchar *ident)
168 {
169 return memcmp(s1: ident, ELFMAG, SELFMAG) == 0;
170 }
171
172 // byte 6
173 static bool checkElfVersion(const uchar *ident)
174 {
175 uchar elfversion = ident[EI_VERSION];
176 return elfversion == EV_CURRENT;
177 }
178
179 struct CommonHeader {
180 Elf32_Half type;
181 Elf32_Half machine;
182 Elf32_Word version;
183 };
184};

186template <typename EquivalentPointerType = quintptr, QSysInfo::Endian Order = QSysInfo::ByteOrder>
187struct ElfHeaderCheck : public ElfHeaderCommonCheck
188{
189 using TypeTraits = ElfTypeTraits<EquivalentPointerType>;
190 using EndianTraits = ElfEndianTraits<Order>;
191 using Ehdr = typename TypeTraits::Ehdr;
192
193 // byte 4
194 static bool checkClass(const uchar *ident)
195 {
196 uchar klass = ident[EI_CLASS];
197 return klass == TypeTraits::Class;
198 }
199
200 // byte 5
201 static bool checkDataOrder(const uchar *ident)
202 {
203 uchar data = ident[EI_DATA];
204 return data == EndianTraits::DataOrder;
205 }
206
207 // byte 7
208 static bool checkOsAbi(const uchar *ident)
209 {
210 uchar osabi = ident[EI_OSABI];
211 // we don't check
212 Q_UNUSED(osabi);
213 return true;
214 }
215
216 // byte 8
217 static bool checkAbiVersion(const uchar *ident)
218 {
219 uchar abiversion = ident[EI_ABIVERSION];
220 // we don't check (and I don't know anyone who uses this)
221 Q_UNUSED(abiversion);
222 return true;
223 }
224
225 // bytes 9-16
226 static bool checkPadding(const uchar *ident)
227 {
228 // why would we check this?
229 Q_UNUSED(ident);
230 return true;
231 }
232
233 static bool checkIdent(const Ehdr &header)
234 {
235 return checkElfMagic(ident: header.e_ident)
236 && checkClass(ident: header.e_ident)
237 && checkDataOrder(ident: header.e_ident)
238 && checkElfVersion(ident: header.e_ident)
239 && checkOsAbi(ident: header.e_ident)
240 && checkAbiVersion(ident: header.e_ident)
241 && checkPadding(ident: header.e_ident);
242 }
243
244 static bool checkType(const Ehdr &header)
245 {
246 return header.e_type == ET_DYN;
247 }
248
249 static bool checkMachine(const Ehdr &header)
250 {
251 return header.e_machine == ElfMachineCheck::ExpectedMachine;
252 }
253
254 static bool checkFileVersion(const Ehdr &header)
255 {
256 return header.e_version == EV_CURRENT;
257 }
258
259 static bool checkHeader(const Ehdr &header)
260 {
261 if (!checkIdent(header))
262 return false;
263 if (!IncludeValidityChecks)
264 return true;
265 return checkType(header)
266 && checkMachine(header)
267 && checkFileVersion(header);
268 }
269
270 Q_DECL_COLD_FUNCTION static QString explainCheckFailure(const Ehdr &header)
271 {
272 if (!checkElfMagic(ident: header.e_ident))
273 return QLibrary::tr(s: "invalid signature");
274 if (!checkClass(ident: header.e_ident))
275 return QLibrary::tr(s: "file is for a different word size");
276 if (!checkDataOrder(ident: header.e_ident))
277 return QLibrary::tr(s: "file is for the wrong endianness");
278 if (!checkElfVersion(ident: header.e_ident) || !checkFileVersion(header))
279 return QLibrary::tr(s: "file has an unknown ELF version");
280 if (!checkOsAbi(ident: header.e_ident) || !checkAbiVersion(ident: header.e_ident))
281 return QLibrary::tr(s: "file has an unexpected ABI");
282 if (!checkType(header))
283 return QLibrary::tr(s: "file is not a shared object");
284 if (!checkMachine(header))
285 return QLibrary::tr(s: "file is for a different processor");
286 return QString();
287 }

289 static CommonHeader extractCommonHeader(const uchar *data)
290 {
291 auto header = reinterpret_cast<const Ehdr *>(data);
292 CommonHeader r;
293 r.type = EndianTraits::fromEndian(header->e_type);
294 r.machine = EndianTraits::fromEndian(header->e_machine);
295 r.version = EndianTraits::fromEndian(header->e_version);
296 return r;
297 }

300struct ElfHeaderDebug { const uchar *e_ident; };
301Q_DECL_UNUSED Q_DECL_COLD_FUNCTION static QDebug &operator<<(QDebug &d, ElfHeaderDebug h)
302{
303 const uchar *e_ident = h.e_ident;
304 if (!ElfHeaderCommonCheck::checkElfMagic(ident: e_ident)) {
305 d << "Not an ELF file (invalid signature)";
306 return d;
307 }
308
309 QDebugStateSaver saver(d);
310 d.nospace();
311 quint8 elfclass = e_ident[EI_CLASS];
312 switch (elfclass) {
313 case ELFCLASSNONE:
314 default:
315 d << "Invalid ELF file (class " << e_ident[EI_CLASS] << "), ";
316 break;
317 case ELFCLASS32:
318 d << "ELF 32-bit ";
319 break;
320 case ELFCLASS64:
321 d << "ELF 64-bit ";
322 break;
323 }
324
325 quint8 dataorder = e_ident[EI_DATA];
326 switch (dataorder) {
327 case ELFDATANONE:
328 default:
329 d << "invalid endianness (" << e_ident[EI_DATA] << ')';
330 break;
331 case ELFDATA2LSB:
332 d << "LSB";
333 break;
334 case ELFDATA2MSB:
335 d << "MSB";
336 break;
337 }
338
339 switch (e_ident[EI_OSABI]) {
340 case ELFOSABI_SYSV: d << " (SYSV"; break;
341 case ELFOSABI_HPUX: d << " (HP-UX"; break;
342 case ELFOSABI_NETBSD: d << " (NetBSD"; break;
343 case ELFOSABI_GNU: d << " (GNU/Linux"; break;
344 case ELFOSABI_SOLARIS: d << " (Solaris"; break;
345 case ELFOSABI_AIX: d << " (AIX"; break;
346 case ELFOSABI_IRIX: d << " (IRIX"; break;
347 case ELFOSABI_FREEBSD: d << " (FreeBSD"; break;
348 case ELFOSABI_OPENBSD: d << " (OpenBSD"; break;
349 default: d << " (OS ABI " << e_ident[EI_VERSION]; break;
350 }
351
352 if (e_ident[EI_ABIVERSION])
353 d << " v" << e_ident[EI_ABIVERSION];
354 d << ')';
355
356 if (e_ident[EI_VERSION] != 1) {
357 d << ", file version " << e_ident[EI_VERSION];
358 return d;
359 }
360
361 ElfHeaderCommonCheck::CommonHeader r;
362 if (elfclass == ELFCLASS64 && dataorder == ELFDATA2LSB)
363 r = ElfHeaderCheck<quint64, QSysInfo::LittleEndian>::extractCommonHeader(data: e_ident);
364 else if (elfclass == ELFCLASS32 && dataorder == ELFDATA2LSB)
365 r = ElfHeaderCheck<quint32, QSysInfo::LittleEndian>::extractCommonHeader(data: e_ident);
366 else if (elfclass == ELFCLASS64 && dataorder == ELFDATA2MSB)
367 r = ElfHeaderCheck<quint64, QSysInfo::BigEndian>::extractCommonHeader(data: e_ident);
368 else if (elfclass == ELFCLASS32 && dataorder == ELFDATA2MSB)
369 r = ElfHeaderCheck<quint32, QSysInfo::BigEndian>::extractCommonHeader(data: e_ident);
370 else
371 return d;
372
373 d << ", version " << r.version;
374
375 switch (r.type) {
376 case ET_NONE: d << ", no type"; break;
377 case ET_REL: d << ", relocatable"; break;
378 case ET_EXEC: d << ", executable"; break;
379 case ET_DYN: d << ", shared library or PIC executable"; break;
380 case ET_CORE: d << ", core dump"; break;
381 default: d << ", unknown type " << r.type; break;
382 }
383
384 switch (r.machine) {
385 // list definitely not exhaustive!
386 case EM_NONE: d << ", no machine"; break;
387 case EM_ALPHA: d << ", Alpha"; break;
388 case EM_68K: d << ", MC68000"; break;
389 case EM_ARM: d << ", ARM"; break;
390 case EM_AARCH64: d << ", AArch64"; break;
391#ifdef EM_BLACKFIN
392 case EM_BLACKFIN: d << ", Blackfin"; break;
393#endif
394 case EM_IA_64: d << ", IA-64"; break;
395#ifdef EM_LOONGARCH
396 case EM_LOONGARCH: d << ", LoongArch"; break;
397#endif
398 case EM_MIPS: d << ", MIPS"; break;
399 case EM_PARISC: d << ", HPPA"; break;
400 case EM_PPC: d << ", PowerPC"; break;
401 case EM_PPC64: d << ", PowerPC 64-bit"; break;
402#ifdef EM_RISCV
403 case EM_RISCV: d << ", RISC-V"; break;
404#endif
405 case EM_S390: d << ", S/390"; break;
406 case EM_SH: d << ", SuperH"; break;
407 case EM_SPARC: d << ", SPARC"; break;
408 case EM_SPARCV9: d << ", SPARCv9"; break;
409 case EM_386: d << ", i386"; break;
410 case EM_X86_64: d << ", x86-64"; break;
411 default: d << ", other machine type " << r.machine; break;
412 }
413
414 return d;
415}

524template <typename F>
525static bool scanProgramHeaders(QByteArrayView data, const ErrorMaker &error, F f)
526{
527 auto header = reinterpret_cast<const T::Ehdr *>(data.data());
528 Q_UNUSED(error);
529
530 auto phdr = reinterpret_cast<const T::Phdr *>(data.data() + header->e_phoff);
531 auto phdr_end = phdr + header->e_phnum;
532 for ( ; phdr != phdr_end; ++phdr) {
533 if (!f(phdr))
534 return false;
535 }
536 return true;
537}
538
539static bool preScanProgramHeaders(QByteArrayView data, const ErrorMaker &error)
540{
541 auto header = reinterpret_cast<const T::Ehdr *>(data.data());
542
543 // first, validate the extent of the full program header table
544 T::Word e_phnum = header->e_phnum;
545 T::Off offset = e_phnum * sizeof(T::Phdr); // can't overflow due to size of T::Half
546 if (qAddOverflow(v1: offset, v2: header->e_phoff, r: &offset) || offset > size_t(data.size()))
547 return error(QLibrary::tr(s: "program header table extends past the end of the file")), false;
548
549 // confirm validity
550 bool hasCode = false;
551 auto checker = [&](const T::Phdr *phdr) {
552 qEDebug << ElfProgramDebug{.phdr: phdr};
553
554 if (T::Off end; qAddOverflow(v1: phdr->p_offset, v2: phdr->p_filesz, r: &end)
555 || end > size_t(data.size()))
556 return error(QLibrary::tr(s: "a program header entry extends past the end of the file")), false;
557
558 // this is not a validity check, it's to exclude debug symbol files
559 if (phdr->p_type == PT_LOAD && phdr->p_filesz != 0 && (phdr->p_flags & PF_X))
560 hasCode = true;
561
562 // this probably applies to all segments, but we'll only apply it to notes
563 if (phdr->p_type == PT_NOTE && qPopulationCount(v: phdr->p_align) == 1
564 && phdr->p_offset & (phdr->p_align - 1)) {
565 return error(QLibrary::tr(s: "a note segment start is not properly aligned "
566 "(offset 0x%1, alignment %2)")
567 .arg(a: phdr->p_offset, fieldWidth: 6, base: 16, fillChar: QChar(u'0'))
568 .arg(a: phdr->p_align)), false;
569 }
570
571 return true;
572 };
573 if (!scanProgramHeaders(data, error, f: checker))
574 return false;
575 if (!hasCode)
576 return error.notplugin(explanation: QLibrary::tr(s: "file has no code")), false;
577 return true;
578}

580static QLibraryScanResult scanProgramHeadersForNotes(QByteArrayView data, const ErrorMaker &error)
581{
582 // minimum metadata payload is 2 bytes
583 constexpr size_t MinPayloadSize = sizeof(QPluginMetaData::Header) + 2;
584 constexpr qptrdiff MinNoteSize = sizeof(QPluginMetaData::ElfNoteHeader) + 2;
585 constexpr size_t NoteNameSize = sizeof(QPluginMetaData::ElfNoteHeader::name);
586 constexpr size_t NoteAlignment = alignof(QPluginMetaData::ElfNoteHeader);
587 constexpr qptrdiff PayloadStartDelta = offsetof(QPluginMetaData::ElfNoteHeader, header);
588 static_assert(MinNoteSize > PayloadStartDelta);
589 static_assert((PayloadStartDelta & (NoteAlignment - 1)) == 0);
590
591 QLibraryScanResult r = {};
592 auto noteFinder = [&](const T::Phdr *phdr) {
593 if (phdr->p_type != PT_NOTE || phdr->p_align != NoteAlignment)
594 return true;
595
596 // check for signed integer overflows, to avoid issues with the
597 // arithmetic below
598 if (qptrdiff(phdr->p_filesz) < 0) {
599 auto h = reinterpret_cast<const T::Ehdr *>(data.data());
600 auto segments = reinterpret_cast<const T::Phdr *>(data.data() + h->e_phoff);
601 qEDebug << "segment" << (phdr - segments) << "contains a note with size"
602 << Qt::hex << Qt::showbase << phdr->p_filesz
603 << "which is larger than half the virtual memory space";
604 return true;
605 }
606
607 // iterate over the notes in this segment
608 T::Off offset = phdr->p_offset;
609 const T::Off end_offset = offset + phdr->p_filesz;
610 while (qptrdiff(end_offset - offset) >= MinNoteSize) {
611 auto nhdr = reinterpret_cast<const T::Nhdr *>(data.data() + offset);
612 T::Word n_namesz = nhdr->n_namesz;
613 T::Word n_descsz = nhdr->n_descsz;
614 T::Word n_type = nhdr->n_type;
615
616 // overflow check: calculate where the next note will be, if it exists
617 T::Off next_offset = offset;
618 next_offset += sizeof(T::Nhdr); // can't overflow (we checked above)
619 next_offset += NoteAlignment - 3; // offset is aligned, this can't overflow
620 if (qAddOverflow<T::Off>(v1: next_offset, v2: n_namesz, r: &next_offset))
621 break;
622 next_offset &= -NoteAlignment;
623
624 next_offset += NoteAlignment - 3; // offset is aligned, this can't overflow
625 if (qAddOverflow<T::Off>(v1: next_offset, v2: n_descsz, r: &next_offset))
626 break;
627 next_offset &= -NoteAlignment;
628 if (next_offset > end_offset)
629 break;
630
631 if (n_namesz == NoteNameSize && n_descsz >= MinPayloadSize
632 && n_type == QPluginMetaData::ElfNoteHeader::NoteType
633 && memcmp(s1: nhdr + 1, s2: QPluginMetaData::ElfNoteHeader::NoteName, n: NoteNameSize) == 0) {
634 // yes, it's our note
635 r.pos = offset + PayloadStartDelta;
636 r.length = nhdr->n_descsz;
637 return false;
638 }
639 offset = next_offset;
640 }
641 return true;
642 };
643 scanProgramHeaders(data, error, f: noteFinder);
644
645 if (!r.length)
646 return r;
647
648 qEDebug << "found Qt metadata in ELF note at"
649 << Qt::hex << Qt::showbase << r.pos << "size" << Qt::reset << r.length;
650 return r;
651}

653static QLibraryScanResult scanSections(QByteArrayView data, const ErrorMaker &error)
654{
655 auto header = reinterpret_cast<const T::Ehdr *>(data.data());
656
657 // in order to find the .qtmetadata section, we need to:
658 // a) find the section table
659 // it's located at offset header->e_shoff
660 // validate it
661 T::Word e_shnum = header->e_shnum;
662 T::Off offset = e_shnum * sizeof(T::Shdr); // can't overflow due to size of T::Half
663 if (qAddOverflow(v1: offset, v2: header->e_shoff, r: &offset) || offset > size_t(data.size()))
664 return error(QLibrary::tr(s: "section table extends past the end of the file"));
665
666 // b) find the section entry for the section header string table (shstrab)
667 // it's a section whose entry is pointed by e_shstrndx
668 auto sections = reinterpret_cast<const T::Shdr *>(data.data() + header->e_shoff);
669 auto sections_end = sections + e_shnum;
670 auto shdr = sections + header->e_shstrndx;
671
672 // validate the shstrtab
673 offset = shdr->sh_offset;
674 T::Off shstrtab_size = shdr->sh_size;
675 qEDebug << "shstrtab section is located at offset" << offset << "size" << shstrtab_size;
676 if (T::Off end; qAddOverflow<T::Off>(v1: offset, v2: shstrtab_size, r: &end)
677 || end > size_t(data.size()))
678 return error(QLibrary::tr(s: "section header string table extends past the end of the file"));
679
680 // c) iterate over the sections to find .qtmetadata
681 const char *shstrtab_start = data.data() + offset;
682 shdr = sections;
683 for (int section = 0; shdr != sections_end; ++section, ++shdr) {
684 QLatin1StringView name;
685 if (shdr->sh_name < shstrtab_size) {
686 const char *namestart = shstrtab_start + shdr->sh_name;
687 size_t len = qstrnlen(str: namestart, maxlen: shstrtab_size - shdr->sh_name);
688 name = QLatin1StringView(namestart, len);
689 }
690 qEDebug << "section" << section << "name" << name << ElfSectionDebug{.shdr: shdr};
691
692 // sanity check the section
693 if (name.isNull())
694 return error(QLibrary::tr(s: "a section name extends past the end of the file"));
695
696 // sections aren't allowed to extend past the end of the file, unless
697 // they are NOBITS sections
698 if (shdr->sh_type == SHT_NOBITS)
699 continue;;
700 if (T::Off end; qAddOverflow(v1: shdr->sh_offset, v2: shdr->sh_size, r: &end)
701 || end > size_t(data.size())) {
702 return error(QLibrary::tr(s: "section contents extend past the end of the file"));
703 }
704
705 if (name != ".qtmetadata"_L1)
706 continue;
707 qEDebug << "found .qtmetadata section";
708 if (shdr->sh_size < sizeof(QPluginMetaData::MagicHeader))
709 return error(QLibrary::tr(s: ".qtmetadata section is too small"));
710
711 if (IncludeValidityChecks) {
712 QByteArrayView expectedMagic = QByteArrayView::fromArray(data: QPluginMetaData::MagicString);
713 QByteArrayView actualMagic = data.sliced(pos: shdr->sh_offset, n: expectedMagic.size());
714 if (expectedMagic != actualMagic)
715 return error(QLibrary::tr(s: ".qtmetadata section has incorrect magic"));
716
717 if (shdr->sh_flags & SHF_WRITE)
718 return error(QLibrary::tr(s: ".qtmetadata section is writable"));
719 if (shdr->sh_flags & SHF_EXECINSTR)
720 return error(QLibrary::tr(s: ".qtmetadata section is executable"));
721 }
722
723 return { .pos: qsizetype(shdr->sh_offset + sizeof(QPluginMetaData::MagicString)),
724 .length: qsizetype(shdr->sh_size - sizeof(QPluginMetaData::MagicString)) };
725 }
726
727 // section .qtmetadata not found
728 return error.notfound();
729}

731QLibraryScanResult QElfParser::parse(QByteArrayView data, QString *errMsg)
732{
733 ErrorMaker error(errMsg);
734 if (size_t(data.size()) < sizeof(T::Ehdr)) {
735 qEDebug << "file too small:" << size_t(data.size());
736 return error(QLibrary::tr(s: "file too small"));
737 }
738
739 qEDebug << ElfHeaderDebug{ .e_ident: reinterpret_cast<const uchar *>(data.data()) };
740
741 auto header = reinterpret_cast<const T::Ehdr *>(data.data());
742 if (!ElfHeaderCheck<>::checkHeader(header: *header))
743 return error(ElfHeaderCheck<>::explainCheckFailure(header: *header));
744
745 qEDebug << "contains" << header->e_phnum << "program headers of"
746 << header->e_phentsize << "bytes at offset" << header->e_phoff;
747 qEDebug << "contains" << header->e_shnum << "sections of" << header->e_shentsize
748 << "bytes at offset" << header->e_shoff
749 << "; section header string table (shstrtab) is entry" << header->e_shstrndx;
750
751 // some sanity checks
752 if constexpr (IncludeValidityChecks) {
753 if (header->e_phentsize != sizeof(T::Phdr))
754 return error(QLibrary::tr(s: "unexpected program header entry size (%1)")
755 .arg(a: header->e_phentsize));
756 }
757
758 if (!preScanProgramHeaders(data, error))
759 return {};
760
761 if (QLibraryScanResult r = scanProgramHeadersForNotes(data, error); r.length)
762 return r;
763
764 if (!ElfNotesAreMandatory) {
765 if constexpr (IncludeValidityChecks) {
766 if (header->e_shentsize != sizeof(T::Shdr))
767 return error(QLibrary::tr(s: "unexpected section entry size (%1)")
768 .arg(a: header->e_shentsize));
769 }
770 if (header->e_shoff == 0 || header->e_shnum == 0) {
771 // this is still a valid ELF file but we don't have a section table
772 qEDebug << "no section table present, not able to find Qt metadata";
773 return error.notfound();
774 }
775
776 if (header->e_shnum && header->e_shstrndx >= header->e_shnum)
777 return error(QLibrary::tr(s: "e_shstrndx greater than the number of sections e_shnum (%1 >= %2)")
778 .arg(a: header->e_shstrndx).arg(a: header->e_shnum));
779 return scanSections(data, error);
780 }
781 return error.notfound();
782}
