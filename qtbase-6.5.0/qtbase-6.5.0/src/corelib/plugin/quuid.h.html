30class Q_CORE_EXPORT QUuid
31{
32 QUuid(Qt::Initialization) {}
33public:
34 enum Variant {
35 VarUnknown =-1,
36 NCS = 0, // 0 - -
37 DCE = 2, // 1 0 -
38 Microsoft = 6, // 1 1 0
39 Reserved = 7 // 1 1 1
40 };
41
42 enum Version {
43 VerUnknown =-1,
44 Time = 1, // 0 0 0 1
45 EmbeddedPOSIX = 2, // 0 0 1 0
46 Md5 = 3, // 0 0 1 1
47 Name = Md5,
48 Random = 4, // 0 1 0 0
49 Sha1 = 5 // 0 1 0 1
50 };
51
52 enum StringFormat {
53 WithBraces = 0,
54 WithoutBraces = 1,
55 Id128 = 3
56 };
57
58 constexpr QUuid() noexcept : data1(0), data2(0), data3(0), data4{0,0,0,0,0,0,0,0} {}
59
60 constexpr QUuid(uint l, ushort w1, ushort w2, uchar b1, uchar b2, uchar b3,
61 uchar b4, uchar b5, uchar b6, uchar b7, uchar b8) noexcept
62 : data1(l), data2(w1), data3(w2), data4{b1, b2, b3, b4, b5, b6, b7, b8} {}
63
64 explicit QUuid(QAnyStringView string) noexcept
65 : QUuid{fromString(string)} {}
66 static QUuid fromString(QAnyStringView string) noexcept;
67#if QT_CORE_REMOVED_SINCE(6, 3)
68 explicit QUuid(const QString &);
69 static QUuid fromString(QStringView string) noexcept;
70 static QUuid fromString(QLatin1StringView string) noexcept;
71 explicit QUuid(const char *);
72 explicit QUuid(const QByteArray &);
73#endif
74 QString toString(StringFormat mode = WithBraces) const;
75 QByteArray toByteArray(StringFormat mode = WithBraces) const;
76 QByteArray toRfc4122() const;
77#if QT_CORE_REMOVED_SINCE(6, 3)
78 static QUuid fromRfc4122(const QByteArray &);
79#endif
80 static QUuid fromRfc4122(QByteArrayView) noexcept;
81 bool isNull() const noexcept;
82
83 constexpr bool operator==(const QUuid &orig) const noexcept
84 {
85 if (data1 != orig.data1 || data2 != orig.data2 ||
86 data3 != orig.data3)
87 return false;
88
89 for (uint i = 0; i < 8; i++)
90 if (data4[i] != orig.data4[i])
91 return false;
92
93 return true;
94 }
95
96 constexpr bool operator!=(const QUuid &orig) const noexcept
97 {
98 return !(*this == orig);
99 }
100
101 bool operator<(const QUuid &other) const noexcept;
102 bool operator>(const QUuid &other) const noexcept;
103
104#if defined(Q_OS_WIN) || defined(Q_QDOC)
105 // On Windows we have a type GUID that is used by the platform API, so we
106 // provide convenience operators to cast from and to this type.
107 constexpr QUuid(const GUID &guid) noexcept
108 : data1(guid.Data1), data2(guid.Data2), data3(guid.Data3),
109 data4{guid.Data4[0], guid.Data4[1], guid.Data4[2], guid.Data4[3],
110 guid.Data4[4], guid.Data4[5], guid.Data4[6], guid.Data4[7]} {}
111
112 constexpr QUuid &operator=(const GUID &guid) noexcept
113 {
114 *this = QUuid(guid);
115 return *this;
116 }
117
118 constexpr operator GUID() const noexcept
119 {
120 GUID guid = { data1, data2, data3, { data4[0], data4[1], data4[2], data4[3], data4[4], data4[5], data4[6], data4[7] } };
121 return guid;
122 }
123
124 constexpr bool operator==(const GUID &guid) const noexcept
125 {
126 return *this == QUuid(guid);
127 }
128
129 constexpr bool operator!=(const GUID &guid) const noexcept
130 {
131 return !(*this == guid);
132 }
133#endif
