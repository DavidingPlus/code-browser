63// Used by both providers
64bool QMimeMagicRule::matchSubstring(const char *dataPtr, int dataSize, int rangeStart, int rangeLength,
65 int valueLength, const char *valueData, const char *mask)
66{
67 // Size of searched data.
68 // Example: value="ABC", rangeLength=3 -> we need 3+3-1=5 bytes (ABCxx,xABCx,xxABC would match)
69 const int dataNeeded = qMin(a: rangeLength + valueLength - 1, b: dataSize - rangeStart);
70
71 if (!mask) {
72 // callgrind says QByteArray::indexOf is much slower, since our strings are typically too
73 // short for be worth Boyer-Moore matching (1 to 71 bytes, 11 bytes on average).
74 bool found = false;
75 for (int i = rangeStart; i < rangeStart + rangeLength; ++i) {
76 if (i + valueLength > dataSize)
77 break;
78
79 if (memcmp(s1: valueData, s2: dataPtr + i, n: valueLength) == 0) {
80 found = true;
81 break;
82 }
83 }
84 if (!found)
85 return false;
86 } else {
87 bool found = false;
88 const char *readDataBase = dataPtr + rangeStart;
89 // Example (continued from above):
90 // deviceSize is 4, so dataNeeded was max'ed to 4.
91 // maxStartPos = 4 - 3 + 1 = 2, and indeed
92 // we need to check for a match a positions 0 and 1 (ABCx and xABC).
93 const int maxStartPos = dataNeeded - valueLength + 1;
94 for (int i = 0; i < maxStartPos; ++i) {
95 const char *d = readDataBase + i;
96 bool valid = true;
97 for (int idx = 0; idx < valueLength; ++idx) {
98 if (((*d++) & mask[idx]) != (valueData[idx] & mask[idx])) {
99 valid = false;
100 break;
101 }
102 }
103 if (valid)
104 found = true;
105 }
106 if (!found)
107 return false;
108 }
109 //qDebug() << "Found" << value << "in" << searchedData;
110 return true;
111}

119template <typename T>
120bool QMimeMagicRule::matchNumber(const QByteArray &data) const
121{
122 const T value(m_number);
123 const T mask(m_numberMask);
124
125 //qDebug() << "matchNumber" << "0x" << QString::number(m_number, 16) << "size" << sizeof(T);
126 //qDebug() << "mask" << QString::number(m_numberMask, 16);
127
128 const char *p = data.constData() + m_startPos;
129 const char *e = data.constData() + qMin(a: data.size() - int(sizeof(T)), b: m_endPos);
130 for ( ; p <= e; ++p) {
131 if ((qFromUnaligned<T>(p) & mask) == (value & mask))
132 return true;
133 }
134
135 return false;
136}

138static inline QByteArray makePattern(const QByteArray &value)
139{
140 QByteArray pattern(value.size(), Qt::Uninitialized);
141 char *data = pattern.data();
142
143 const char *p = value.constData();
144 const char *e = p + value.size();
145 for ( ; p < e; ++p) {
146 if (*p == '\\' && ++p < e) {
147 if (*p == 'x') { // hex (\\xff)
148 char c = 0;
149 for (int i = 0; i < 2 && p + 1 < e; ++i) {
150 ++p;
151 if (*p >= '0' && *p <= '9')
152 c = (c << 4) + *p - '0';
153 else if (*p >= 'a' && *p <= 'f')
154 c = (c << 4) + *p - 'a' + 10;
155 else if (*p >= 'A' && *p <= 'F')
156 c = (c << 4) + *p - 'A' + 10;
157 else
158 continue;
159 }
160 *data++ = c;
161 } else if (isOctalDigit(c: *p)) { // oct (\\7, or \\77, or \\377)
162 char c = *p - '0';
163 if (p + 1 < e && isOctalDigit(c: p[1])) {
164 c = (c << 3) + *(++p) - '0';
165 if (p + 1 < e && isOctalDigit(c: p[1]) && p[-1] <= '3')
166 c = (c << 3) + *(++p) - '0';
167 }
168 *data++ = c;
169 } else if (*p == 'n') {
170 *data++ = '\n';
171 } else if (*p == 'r') {
172 *data++ = '\r';
173 } else if (*p == 't') {
174 *data++ = '\t';
175 } else { // escaped
176 *data++ = *p;
177 }
178 } else {
179 *data++ = *p;
180 }
181 }
182 pattern.truncate(pos: data - pattern.data());
183
184 return pattern;
185}

187// Evaluate a magic match rule like
188// <match value="must be converted with BinHex" type="string" offset="11"/>
189// <match value="0x9501" type="big16" offset="0:64"/>

191QMimeMagicRule::QMimeMagicRule(const QString &type,
192 const QByteArray &value,
193 const QString &offsets,
194 const QByteArray &mask,
195 QString *errorString)
196 : m_type(QMimeMagicRule::type(theTypeName: type.toLatin1())),
197 m_value(value),
198 m_mask(mask),
199 m_matchFunction(nullptr)
200{
201 if (Q_UNLIKELY(m_type == Invalid)) {
202 if (errorString)
203 *errorString = "Type "_L1 + type + " is not supported"_L1;
204 return;
205 }
206
207 // Parse for offset as "1" or "1:10"
208 const int colonIndex = offsets.indexOf(c: u':');
209 const QStringView startPosStr = QStringView{offsets}.mid(pos: 0, n: colonIndex); // \ These decay to returning 'offsets'
210 const QStringView endPosStr = QStringView{offsets}.mid(pos: colonIndex + 1);// / unchanged when colonIndex == -1
211 if (Q_UNLIKELY(!QMimeTypeParserBase::parseNumber(startPosStr, &m_startPos, errorString)) ||
212 Q_UNLIKELY(!QMimeTypeParserBase::parseNumber(endPosStr, &m_endPos, errorString))) {
213 m_type = Invalid;
214 return;
215 }
216
217 if (Q_UNLIKELY(m_value.isEmpty())) {
218 m_type = Invalid;
219 if (errorString)
220 *errorString = QStringLiteral("Invalid empty magic rule value");
221 return;
222 }
223
224 if (m_type >= Host16 && m_type <= Byte) {
225 bool ok;
226 m_number = m_value.toUInt(ok: &ok, base: 0); // autodetect base
227 if (Q_UNLIKELY(!ok)) {
228 m_type = Invalid;
229 if (errorString)
230 *errorString = "Invalid magic rule value \""_L1 + QLatin1StringView(m_value) + u'"';
231 return;
232 }
233 m_numberMask = !m_mask.isEmpty() ? m_mask.toUInt(ok: &ok, base: 0) : 0; // autodetect base
234 }
235
236 switch (m_type) {
237 case String:
238 m_pattern = makePattern(value: m_value);
239 m_pattern.squeeze();
240 if (!m_mask.isEmpty()) {
241 if (Q_UNLIKELY(m_mask.size() < 4 || !m_mask.startsWith("0x"))) {
242 m_type = Invalid;
243 if (errorString)
244 *errorString = "Invalid magic rule mask \""_L1 + QLatin1StringView(m_mask) + u'"';
245 return;
246 }
247 const QByteArray &tempMask = QByteArray::fromHex(hexEncoded: QByteArray::fromRawData(
248 data: m_mask.constData() + 2, size: m_mask.size() - 2));
249 if (Q_UNLIKELY(tempMask.size() != m_pattern.size())) {
250 m_type = Invalid;
251 if (errorString)
252 *errorString = "Invalid magic rule mask size \""_L1 + QLatin1StringView(m_mask) + u'"';
253 return;
254 }
255 m_mask = tempMask;
256 } else {
257 m_mask.fill(c: char(-1), size: m_pattern.size());
258 }
259 m_mask.squeeze();
260 m_matchFunction = &QMimeMagicRule::matchString;
261 break;
262 case Byte:
263 if (m_number <= quint8(-1)) {
264 if (m_numberMask == 0)
265 m_numberMask = quint8(-1);
266 m_matchFunction = &QMimeMagicRule::matchNumber<quint8>;
267 }
268 break;
269 case Big16:
270 case Little16:
271 if (m_number <= quint16(-1)) {
272 m_number = m_type == Little16 ? qFromLittleEndian<quint16>(source: m_number) : qFromBigEndian<quint16>(source: m_number);
273 if (m_numberMask != 0)
274 m_numberMask = m_type == Little16 ? qFromLittleEndian<quint16>(source: m_numberMask) : qFromBigEndian<quint16>(source: m_numberMask);
275 }
276 Q_FALLTHROUGH();
277 case Host16:
278 if (m_number <= quint16(-1)) {
279 if (m_numberMask == 0)
280 m_numberMask = quint16(-1);
281 m_matchFunction = &QMimeMagicRule::matchNumber<quint16>;
282 }
283 break;
284 case Big32:
285 case Little32:
286 m_number = m_type == Little32 ? qFromLittleEndian<quint32>(source: m_number) : qFromBigEndian<quint32>(source: m_number);
287 if (m_numberMask != 0)
288 m_numberMask = m_type == Little32 ? qFromLittleEndian<quint32>(source: m_numberMask) : qFromBigEndian<quint32>(source: m_numberMask);
289 Q_FALLTHROUGH();
290 case Host32:
291 if (m_numberMask == 0)
292 m_numberMask = quint32(-1);
293 m_matchFunction = &QMimeMagicRule::matchNumber<quint32>;
294 break;
295 default:
296 break;
297 }
298}

300QByteArray QMimeMagicRule::mask() const
301{
302 QByteArray result = m_mask;
303 if (m_type == String) {
304 // restore '0x'
305 result = "0x" + result.toHex();
306 }
307 return result;
308}

310bool QMimeMagicRule::matches(const QByteArray &data) const
311{
312 const bool ok = m_matchFunction && (this->*m_matchFunction)(data);
313 if (!ok)
314 return false;
315
316 // No submatch? Then we are done.
317 if (m_subMatches.isEmpty())
318 return true;
319
320 //qDebug() << "Checking" << m_subMatches.count() << "sub-rules";
321 // Check that one of the submatches matches too
322 for ( QList<QMimeMagicRule>::const_iterator it = m_subMatches.begin(), end = m_subMatches.end() ;
323 it != end ; ++it ) {
324 if ((*it).matches(data)) {
325 // One of the hierarchies matched -> mimetype recognized.
326 return true;
327 }
328 }
329 return false;
