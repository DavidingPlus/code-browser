32class QMimeProviderBase
33{
34public:
35 QMimeProviderBase(QMimeDatabasePrivate *db, const QString &directory);
36 virtual ~QMimeProviderBase() {}
37
38 virtual bool isValid() = 0;
39 virtual bool isInternalDatabase() const = 0;
40 virtual QMimeType mimeTypeForName(const QString &name) = 0;
41 virtual void addFileNameMatches(const QString &fileName, QMimeGlobMatchResult &result) = 0;
42 virtual void addParents(const QString &mime, QStringList &result) = 0;
43 virtual QString resolveAlias(const QString &name) = 0;
44 virtual void addAliases(const QString &name, QStringList &result) = 0;
45 virtual void findByMagic(const QByteArray &data, int *accuracyPtr, QMimeType &candidate) = 0;
46 virtual void addAllMimeTypes(QList<QMimeType> &result) = 0;
47 virtual bool loadMimeTypePrivate(QMimeTypePrivate &) { return false; }
48 virtual void loadIcon(QMimeTypePrivate &) {}
49 virtual void loadGenericIcon(QMimeTypePrivate &) {}
50 virtual void ensureLoaded() {}
51
52 QString directory() const { return m_directory; }
53
54 QMimeDatabasePrivate *m_db;
55 QString m_directory;
56};

78 void loadGenericIcon(QMimeTypePrivate &) override;
