53// Check for a match on contents of a file
54bool QMimeMagicRuleMatcher::matches(const QByteArray &data) const
55{
56 for (const QMimeMagicRule &magicRule : m_list) {
57 if (magicRule.matches(data))
58 return true;
59 }
60
61 return false;
62}
