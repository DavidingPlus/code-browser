73 {
74 return qFromBigEndian(source: *reinterpret_cast<quint16 *>(data + offset));

103bool QMimeBinaryProvider::CacheFile::load()
104{
105 if (!file.open(flags: QIODevice::ReadOnly))
106 return false;
107 data = file.map(offset: 0, size: file.size());
108 if (data) {
109 const int major = getUint16(offset: 0);
110 const int minor = getUint16(offset: 2);
111 m_valid = (major == 1 && minor >= 1 && minor <= 2);
112 }
113 m_mtime = QFileInfo(file).lastModified();
114 return m_valid;
115}
116
117bool QMimeBinaryProvider::CacheFile::reload()
118{
119 m_valid = false;
120 if (file.isOpen()) {
121 file.close();
122 }
123 data = nullptr;
124 return load();
125}

155bool QMimeBinaryProvider::checkCacheChanged()
156{
157 QFileInfo fileInfo(m_cacheFile->file);
158 if (fileInfo.lastModified() > m_cacheFile->m_mtime) {
159 // Deletion can't happen by just running update-mime-database.
160 // But the user could use rm -rf :-)
161 m_cacheFile->reload(); // will mark itself as invalid on failure
162 return true;
163 }
164 return false;
165}

167void QMimeBinaryProvider::ensureLoaded()
168{
169 if (!m_cacheFile) {
170 const QString cacheFileName = m_directory + "/mime.cache"_L1;
171 m_cacheFile = new CacheFile(cacheFileName);
172 m_mimetypeListLoaded = false;
173 m_mimetypeExtra.clear();
174 } else {
175 if (checkCacheChanged()) {
176 m_mimetypeListLoaded = false;
177 m_mimetypeExtra.clear();
178 } else {
179 return; // nothing to do
180 }
181 }
182 if (!m_cacheFile->isValid()) { // verify existence and version
183 delete m_cacheFile;
184 m_cacheFile = nullptr;
185 }

209void QMimeBinaryProvider::addFileNameMatches(const QString &fileName, QMimeGlobMatchResult &result)
210{
211 if (fileName.isEmpty())
212 return;
213 Q_ASSERT(m_cacheFile);
214 const QString lowerFileName = fileName.toLower();
215 // Check literals (e.g. "Makefile")
216 matchGlobList(result, cacheFile: m_cacheFile, offset: m_cacheFile->getUint32(offset: PosLiteralListOffset), fileName);
217 // Check the very common *.txt cases with the suffix tree
218 if (result.m_matchingMimeTypes.isEmpty()) {
219 const int reverseSuffixTreeOffset = m_cacheFile->getUint32(offset: PosReverseSuffixTreeOffset);
220 const int numRoots = m_cacheFile->getUint32(offset: reverseSuffixTreeOffset);
221 const int firstRootOffset = m_cacheFile->getUint32(offset: reverseSuffixTreeOffset + 4);
222 matchSuffixTree(result, cacheFile: m_cacheFile, numEntries: numRoots, firstOffset: firstRootOffset, fileName: lowerFileName, charPos: lowerFileName.size() - 1, caseSensitiveCheck: false);
223 if (result.m_matchingMimeTypes.isEmpty())
224 matchSuffixTree(result, cacheFile: m_cacheFile, numEntries: numRoots, firstOffset: firstRootOffset, fileName, charPos: fileName.size() - 1, caseSensitiveCheck: true);
225 }
226 // Check complex globs (e.g. "callgrind.out[0-9]*" or "README*")
227 if (result.m_matchingMimeTypes.isEmpty())
228 matchGlobList(result, cacheFile: m_cacheFile, offset: m_cacheFile->getUint32(offset: PosGlobListOffset), fileName);
229}

231void QMimeBinaryProvider::matchGlobList(QMimeGlobMatchResult &result, CacheFile *cacheFile, int off, const QString &fileName)
232{
233 const int numGlobs = cacheFile->getUint32(offset: off);
234 //qDebug() << "Loading" << numGlobs << "globs from" << cacheFile->file.fileName() << "at offset" << cacheFile->globListOffset;
235 for (int i = 0; i < numGlobs; ++i) {
236 const int globOffset = cacheFile->getUint32(offset: off + 4 + 12 * i);
237 const int mimeTypeOffset = cacheFile->getUint32(offset: off + 4 + 12 * i + 4);
238 const int flagsAndWeight = cacheFile->getUint32(offset: off + 4 + 12 * i + 8);
239 const int weight = flagsAndWeight & 0xff;
240 const bool caseSensitive = flagsAndWeight & 0x100;
241 const Qt::CaseSensitivity qtCaseSensitive = caseSensitive ? Qt::CaseSensitive : Qt::CaseInsensitive;
242 const QString pattern = QLatin1StringView(cacheFile->getCharStar(offset: globOffset));
243
244 const char *mimeType = cacheFile->getCharStar(offset: mimeTypeOffset);
245 //qDebug() << pattern << mimeType << weight << caseSensitive;
246 QMimeGlobPattern glob(pattern, QString() /*unused*/, weight, qtCaseSensitive);
247
248 if (glob.matchFileName(inputFileName: fileName))
249 result.addMatch(mimeType: QLatin1StringView(mimeType), weight, pattern);
250 }

253bool QMimeBinaryProvider::matchSuffixTree(QMimeGlobMatchResult &result, QMimeBinaryProvider::CacheFile *cacheFile, int numEntries, int firstOffset, const QString &fileName, int charPos, bool caseSensitiveCheck)
254{
255 QChar fileChar = fileName[charPos];
256 int min = 0;
257 int max = numEntries - 1;
258 while (min <= max) {
259 const int mid = (min + max) / 2;
260 const int off = firstOffset + 12 * mid;
261 const QChar ch = char16_t(cacheFile->getUint32(offset: off));
262 if (ch < fileChar)
263 min = mid + 1;
264 else if (ch > fileChar)
265 max = mid - 1;
266 else {
267 --charPos;
268 int numChildren = cacheFile->getUint32(offset: off + 4);
269 int childrenOffset = cacheFile->getUint32(offset: off + 8);
270 bool success = false;
271 if (charPos > 0)
272 success = matchSuffixTree(result, cacheFile, numEntries: numChildren, firstOffset: childrenOffset, fileName, charPos, caseSensitiveCheck);
273 if (!success) {
274 for (int i = 0; i < numChildren; ++i) {
275 const int childOff = childrenOffset + 12 * i;
276 const int mch = cacheFile->getUint32(offset: childOff);
277 if (mch != 0)
278 break;
279 const int mimeTypeOffset = cacheFile->getUint32(offset: childOff + 4);
280 const char *mimeType = cacheFile->getCharStar(offset: mimeTypeOffset);
281 const int flagsAndWeight = cacheFile->getUint32(offset: childOff + 8);
282 const int weight = flagsAndWeight & 0xff;
283 const bool caseSensitive = flagsAndWeight & 0x100;
284 if (caseSensitiveCheck || !caseSensitive) {
285 result.addMatch(mimeType: QLatin1StringView(mimeType), weight,
286 pattern: u'*' + QStringView{fileName}.mid(pos: charPos + 1),
287 knownSuffixLength: fileName.size() - charPos - 2);
288 success = true;
289 }
290 }
291 }
292 return success;
293 }
294 }
295 return false;
296}

298bool QMimeBinaryProvider::matchMagicRule(QMimeBinaryProvider::CacheFile *cacheFile, int numMatchlets, int firstOffset, const QByteArray &data)
299{
300 const char *dataPtr = data.constData();
301 const int dataSize = data.size();
302 for (int matchlet = 0; matchlet < numMatchlets; ++matchlet) {
303 const int off = firstOffset + matchlet * 32;
304 const int rangeStart = cacheFile->getUint32(offset: off);
305 const int rangeLength = cacheFile->getUint32(offset: off + 4);
306 //const int wordSize = cacheFile->getUint32(off + 8);
307 const int valueLength = cacheFile->getUint32(offset: off + 12);
308 const int valueOffset = cacheFile->getUint32(offset: off + 16);
309 const int maskOffset = cacheFile->getUint32(offset: off + 20);
310 const char *mask = maskOffset ? cacheFile->getCharStar(offset: maskOffset) : nullptr;
311
312 if (!QMimeMagicRule::matchSubstring(dataPtr, dataSize, rangeStart, rangeLength, valueLength, valueData: cacheFile->getCharStar(offset: valueOffset), mask))
313 continue;
314
315 const int numChildren = cacheFile->getUint32(offset: off + 24);
316 const int firstChildOffset = cacheFile->getUint32(offset: off + 28);
317 if (numChildren == 0) // No submatch? Then we are done.
318 return true;
319 // Check that one of the submatches matches too
320 if (matchMagicRule(cacheFile, numMatchlets: numChildren, firstOffset: firstChildOffset, data))
321 return true;
322 }
323 return false;
324}

326void QMimeBinaryProvider::findByMagic(const QByteArray &data, int *accuracyPtr, QMimeType &candidate)
327{
328 const int magicListOffset = m_cacheFile->getUint32(offset: PosMagicListOffset);
329 const int numMatches = m_cacheFile->getUint32(offset: magicListOffset);
330 //const int maxExtent = cacheFile->getUint32(magicListOffset + 4);
331 const int firstMatchOffset = m_cacheFile->getUint32(offset: magicListOffset + 8);
332
333 for (int i = 0; i < numMatches; ++i) {
334 const int off = firstMatchOffset + i * 16;
335 const int numMatchlets = m_cacheFile->getUint32(offset: off + 8);
336 const int firstMatchletOffset = m_cacheFile->getUint32(offset: off + 12);
337 if (matchMagicRule(cacheFile: m_cacheFile, numMatchlets, firstOffset: firstMatchletOffset, data)) {
338 const int mimeTypeOffset = m_cacheFile->getUint32(offset: off + 4);
339 const char *mimeType = m_cacheFile->getCharStar(offset: mimeTypeOffset);
340 *accuracyPtr = m_cacheFile->getUint32(offset: off);
341 // Return the first match. We have no rules for conflicting magic data...
342 // (mime.cache itself is sorted, but what about local overrides with a lower prio?)
343 candidate = mimeTypeForNameUnchecked(name: QLatin1StringView(mimeType));
344 return;
345 }
346 }

349void QMimeBinaryProvider::addParents(const QString &mime, QStringList &result)
350{
351 const QByteArray mimeStr = mime.toLatin1();
352 const int parentListOffset = m_cacheFile->getUint32(offset: PosParentListOffset);
353 const int numEntries = m_cacheFile->getUint32(offset: parentListOffset);
354
355 int begin = 0;
356 int end = numEntries - 1;
357 while (begin <= end) {
358 const int medium = (begin + end) / 2;
359 const int off = parentListOffset + 4 + 8 * medium;
360 const int mimeOffset = m_cacheFile->getUint32(offset: off);
361 const char *aMime = m_cacheFile->getCharStar(offset: mimeOffset);
362 const int cmp = qstrcmp(str1: aMime, str2: mimeStr);
363 if (cmp < 0) {
364 begin = medium + 1;
365 } else if (cmp > 0) {
366 end = medium - 1;
367 } else {
368 const int parentsOffset = m_cacheFile->getUint32(offset: off + 4);
369 const int numParents = m_cacheFile->getUint32(offset: parentsOffset);
370 for (int i = 0; i < numParents; ++i) {
371 const int parentOffset = m_cacheFile->getUint32(offset: parentsOffset + 4 + 4 * i);
372 const char *aParent = m_cacheFile->getCharStar(offset: parentOffset);
373 const QString strParent = QString::fromLatin1(ba: aParent);
374 if (!result.contains(str: strParent))
375 result.append(t: strParent);
376 }
377 break;
378 }
379 }

428void QMimeBinaryProvider::loadMimeTypeList()
429{
430 if (!m_mimetypeListLoaded) {
431 m_mimetypeListLoaded = true;
432 m_mimetypeNames.clear();
433 // Unfortunately mime.cache doesn't have a full list of all mimetypes.
434 // So we have to parse the plain-text files called "types".
435 QFile file(m_directory + QStringLiteral("/types"));
436 if (file.open(flags: QIODevice::ReadOnly)) {
437 while (!file.atEnd()) {
438 QByteArray line = file.readLine();
439 if (line.endsWith(c: '\n'))
440 line.chop(n: 1);
441 m_mimetypeNames.insert(value: QString::fromLatin1(ba: line));
442 }
443 }
444 }

447void QMimeBinaryProvider::addAllMimeTypes(QList<QMimeType> &result)
448{
449 loadMimeTypeList();
450 if (result.isEmpty()) {
451 result.reserve(size: m_mimetypeNames.size());
452 for (const QString &name : std::as_const(t&: m_mimetypeNames))
453 result.append(t: mimeTypeForNameUnchecked(name));
454 } else {
455 for (const QString &name : std::as_const(t&: m_mimetypeNames))
456 if (std::find_if(first: result.constBegin(), last: result.constEnd(), pred: [name](const QMimeType &mime) -> bool { return mime.name() == name; })
457 == result.constEnd())
458 result.append(t: mimeTypeForNameUnchecked(name));
459 }

462bool QMimeBinaryProvider::loadMimeTypePrivate(QMimeTypePrivate &data)
463{
464#ifdef QT_NO_XMLSTREAMREADER
465 Q_UNUSED(data);
466 qWarning("Cannot load mime type since QXmlStreamReader is not available.");
467 return false;
468#else
469 if (data.loaded)
470 return true;
471
472 auto it = m_mimetypeExtra.constFind(key: data.name);
473 if (it == m_mimetypeExtra.constEnd()) {
474 // load comment and globPatterns
475
476 // shared-mime-info since 1.3 lowercases the xml files
477 QString mimeFile = m_directory + u'/' + data.name.toLower() + ".xml"_L1;
478 if (!QFile::exists(mimeFile))
479 mimeFile = m_directory + u'/' + data.name + ".xml"_L1; // pre-1.3
480
481 QFile qfile(mimeFile);
482 if (!qfile.open(flags: QFile::ReadOnly))
483 return false;
484
485 auto insertIt = m_mimetypeExtra.insert(key: data.name, value: MimeTypeExtra{});
486 it = insertIt;
487 MimeTypeExtra &extra = insertIt.value();
488 QString mainPattern;
489
490 QXmlStreamReader xml(&qfile);
491 if (xml.readNextStartElement()) {
492 if (xml.name() != "mime-type"_L1) {
493 return false;
494 }
495 const auto name = xml.attributes().value("type"_L1);
496 if (name.isEmpty())
497 return false;
498 if (name.compare(data.name, Qt::CaseInsensitive))
499 qWarning() << "Got name" << name << "in file" << mimeFile << "expected" << data.name;
500
501 while (xml.readNextStartElement()) {
502 const auto tag = xml.name();
503 if (tag == "comment"_L1) {
504 QString lang = xml.attributes().value("xml:lang"_L1).toString();
505 const QString text = xml.readElementText();
506 if (lang.isEmpty()) {
507 lang = "default"_L1; // no locale attribute provided, treat it as default.
508 }
509 extra.localeComments.insert(key: lang, value: text);
510 continue; // we called readElementText, so we're at the EndElement already.
511 } else if (tag == "glob-deleteall"_L1) { // as written out by shared-mime-info >= 0.70
512 extra.globPatterns.clear();
513 mainPattern.clear();
514 } else if (tag == "glob"_L1) { // as written out by shared-mime-info >= 0.70
515 const QString pattern = xml.attributes().value("pattern"_L1).toString();
516 if (mainPattern.isEmpty() && pattern.startsWith(c: u'*')) {
517 mainPattern = pattern;
518 }
519 if (!extra.globPatterns.contains(str: pattern))
520 extra.globPatterns.append(t: pattern);
521 }
522 xml.skipCurrentElement();
523 }
524 Q_ASSERT(xml.name() == "mime-type"_L1);
525 }
526
527 // Let's assume that shared-mime-info is at least version 0.70
528 // Otherwise we would need 1) a version check, and 2) code for parsing patterns from the globs file.
529 if (!mainPattern.isEmpty() &&
530 (extra.globPatterns.isEmpty() || extra.globPatterns.constFirst() != mainPattern)) {
531 // ensure it's first in the list of patterns
532 extra.globPatterns.removeAll(t: mainPattern);
533 extra.globPatterns.prepend(t: mainPattern);
534 }
535 }
536 const MimeTypeExtra &e = it.value();
537 data.localeComments = e.localeComments;
538 data.globPatterns = e.globPatterns;
539 return true;
540#endif //QT_NO_XMLSTREAMREADER
541}

543// Binary search in the icons or generic-icons list
544QLatin1StringView QMimeBinaryProvider::iconForMime(CacheFile *cacheFile, int posListOffset,
545 const QByteArray &inputMime)
546{
547 const int iconsListOffset = cacheFile->getUint32(offset: posListOffset);
548 const int numIcons = cacheFile->getUint32(offset: iconsListOffset);
549 int begin = 0;
550 int end = numIcons - 1;
551 while (begin <= end) {
552 const int medium = (begin + end) / 2;
553 const int off = iconsListOffset + 4 + 8 * medium;
554 const int mimeOffset = cacheFile->getUint32(offset: off);
555 const char *mime = cacheFile->getCharStar(offset: mimeOffset);
556 const int cmp = qstrcmp(str1: mime, str2: inputMime);
557 if (cmp < 0)
558 begin = medium + 1;
559 else if (cmp > 0)
560 end = medium - 1;
561 else {
562 const int iconOffset = cacheFile->getUint32(offset: off + 4);
563 return QLatin1StringView(cacheFile->getCharStar(offset: iconOffset));
564 }
565 }
566 return QLatin1StringView();
567}

595QMimeXMLProvider::QMimeXMLProvider(QMimeDatabasePrivate *db, InternalDatabaseEnum)
596 : QMimeProviderBase(db, internalMimeFileName())
597{
598 static_assert(sizeof(mimetype_database), "Bundled MIME database is empty");
599 static_assert(sizeof(mimetype_database) <= MimeTypeDatabaseOriginalSize,
600 "Compressed MIME database is larger than the original size");
601 static_assert(MimeTypeDatabaseOriginalSize <= 16*1024*1024,
602 "Bundled MIME database is too big");
603 const char *data = reinterpret_cast<const char *>(mimetype_database);
604 qsizetype size = MimeTypeDatabaseOriginalSize;
605
606#ifdef MIME_DATABASE_IS_ZSTD
607 // uncompress with libzstd
608 std::unique_ptr<char []> uncompressed(new char[size]);
609 size = ZSTD_decompress(uncompressed.get(), size, mimetype_database, sizeof(mimetype_database));
610 Q_ASSERT(!ZSTD_isError(size));
611 data = uncompressed.get();
612#elif defined(MIME_DATABASE_IS_GZIP)
613 std::unique_ptr<char []> uncompressed(new char[size]);
614 z_stream zs = {};
615 zs.next_in = const_cast<Bytef *>(mimetype_database);
616 zs.avail_in = sizeof(mimetype_database);
617 zs.next_out = reinterpret_cast<Bytef *>(uncompressed.get());
618 zs.avail_out = size;
619
620 int res = inflateInit2(&zs, MAX_WBITS | 32);
621 Q_ASSERT(res == Z_OK);
622 res = inflate(&zs, Z_FINISH);
623 Q_ASSERT(res == Z_STREAM_END);
624 res = inflateEnd(&zs);
625 Q_ASSERT(res == Z_OK);
626
627 data = uncompressed.get();
628 size = zs.total_out;
629#endif
630
631 load(data, len: size);
632}

680{
681 QString candidateName;
682 bool foundOne = false;

687 *accuracyPtr = priority;

689 foundOne = true;

691 }

693 if (foundOne)
694 candidate = mimeTypeForName(name: candidateName);
