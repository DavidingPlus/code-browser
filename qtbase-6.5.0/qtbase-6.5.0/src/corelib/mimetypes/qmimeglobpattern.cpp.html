18 \class QMimeGlobMatchResult

20 \brief The QMimeGlobMatchResult class accumulates results from glob matching.
21
22 Handles glob weights, and preferring longer matches over shorter matches.

25void QMimeGlobMatchResult::addMatch(const QString &mimeType, int weight, const QString &pattern, int knownSuffixLength)
26{
27 if (m_allMatchingMimeTypes.contains(str: mimeType))
28 return;
29 // Is this a lower-weight pattern than the last match? Skip this match then.
30 if (weight < m_weight) {
31 m_allMatchingMimeTypes.append(t: mimeType);
32 return;
33 }
34 bool replace = weight > m_weight;
35 if (!replace) {
36 // Compare the length of the match
37 if (pattern.size() < m_matchingPatternLength)
38 return; // too short, ignore
39 else if (pattern.size() > m_matchingPatternLength) {
40 // longer: clear any previous match (like *.bz2, when pattern is *.tar.bz2)
41 replace = true;
42 }
43 }
44 if (replace) {
45 m_matchingMimeTypes.clear();
46 // remember the new "longer" length
47 m_matchingPatternLength = pattern.size();
48 m_weight = weight;
49 }
50 if (!m_matchingMimeTypes.contains(str: mimeType)) {
51 m_matchingMimeTypes.append(t: mimeType);
52 if (replace)
53 m_allMatchingMimeTypes.prepend(t: mimeType); // highest-weight first
54 else
55 m_allMatchingMimeTypes.append(t: mimeType);
56 m_knownSuffixLength = knownSuffixLength;
57 }
58}

60QMimeGlobPattern::PatternType QMimeGlobPattern::detectPatternType(const QString &pattern) const
61{
62 const int patternLength = pattern.size();
63 if (!patternLength)
64 return OtherPattern;
65
66 const int starCount = pattern.count(c: u'*');
67 const bool hasSquareBracket = pattern.indexOf(c: u'[') != -1;
68 const bool hasQuestionMark = pattern.indexOf(c: u'?') != -1;
69
70 if (!hasSquareBracket && !hasQuestionMark) {
71 if (starCount == 1) {
72 // Patterns like "*~", "*.extension"
73 if (pattern.at(i: 0) == u'*')
74 return SuffixPattern;
75 // Patterns like "README*" (well this is currently the only one like that...)
76 if (pattern.at(i: patternLength - 1) == u'*')
77 return PrefixPattern;
78 } else if (starCount == 0) {
79 // Names without any wildcards like "README"
80 return LiteralPattern;
81 }
82 }
83
84 if (pattern == "[0-9][0-9][0-9].vdr"_L1)
85 return VdrPattern;
86
87 if (pattern == "*.anim[1-9j]"_L1)
88 return AnimPattern;
89
90 return OtherPattern;
91}

95 \internal
96 \class QMimeGlobPattern
97 \inmodule QtCore
98 \brief The QMimeGlobPattern class contains the glob pattern for file names for MIME type matching.
99
100 \sa QMimeType, QMimeDatabase, QMimeMagicRuleMatcher, QMimeMagicRule
101*/

103bool QMimeGlobPattern::matchFileName(const QString &inputFileName) const
104{
105 // "Applications MUST match globs case-insensitively, except when the case-sensitive
106 // attribute is set to true."
107 // The constructor takes care of putting case-insensitive patterns in lowercase.
108 const QString fileName = m_caseSensitivity == Qt::CaseInsensitive
109 ? inputFileName.toLower() : inputFileName;
110
111 const int patternLength = m_pattern.size();
112 if (!patternLength)
113 return false;
114 const int fileNameLength = fileName.size();
115
116 switch (m_patternType) {
117 case SuffixPattern: {
118 if (fileNameLength + 1 < patternLength)
119 return false;
120
121 const QChar *c1 = m_pattern.unicode() + patternLength - 1;
122 const QChar *c2 = fileName.unicode() + fileNameLength - 1;
123 int cnt = 1;
124 while (cnt < patternLength && *c1-- == *c2--)
125 ++cnt;
126 return cnt == patternLength;
127 }
128 case PrefixPattern: {
129 if (fileNameLength + 1 < patternLength)
130 return false;
131
132 const QChar *c1 = m_pattern.unicode();
133 const QChar *c2 = fileName.unicode();
134 int cnt = 1;
135 while (cnt < patternLength && *c1++ == *c2++)
136 ++cnt;
137 return cnt == patternLength;
138 }
139 case LiteralPattern:
140 return (m_pattern == fileName);
141 case VdrPattern: // "[0-9][0-9][0-9].vdr" case
142 return fileNameLength == 7
143 && fileName.at(i: 0).isDigit() && fileName.at(i: 1).isDigit() && fileName.at(i: 2).isDigit()
144 && QStringView{fileName}.mid(pos: 3, n: 4) == ".vdr"_L1;
145 case AnimPattern: { // "*.anim[1-9j]" case
146 if (fileNameLength < 6)
147 return false;
148 const QChar lastChar = fileName.at(i: fileNameLength - 1);
149 const bool lastCharOK = (lastChar.isDigit() && lastChar != u'0')
150 || lastChar == u'j';
151 return lastCharOK && QStringView{fileName}.mid(pos: fileNameLength - 6, n: 5) == ".anim"_L1;
152 }
153 case OtherPattern:
154 // Other fallback patterns: slow but correct method
155#if QT_CONFIG(regularexpression)
156 auto rx = QRegularExpression::fromWildcard(pattern: m_pattern);
157 return rx.match(subject: fileName).hasMatch();
158#else
159 return false;
160#endif
161 }
162 return false;
163}

165static bool isSimplePattern(const QString &pattern)
166{
167 // starts with "*.", has no other '*'
168 return pattern.lastIndexOf(c: u'*') == 0
169 && pattern.size() > 1
170 && pattern.at(i: 1) == u'.' // (other dots are OK, like *.tar.bz2)
171 // and contains no other special character
172 && !pattern.contains(c: u'?')
173 && !pattern.contains(c: u'[')
174 ;
175}
176
177static bool isFastPattern(const QString &pattern)
178{
179 // starts with "*.", has no other '*' and no other '.'
180 return pattern.lastIndexOf(c: u'*') == 0
181 && pattern.lastIndexOf(c: u'.') == 1
182 // and contains no other special character
183 && !pattern.contains(c: u'?')
184 && !pattern.contains(c: u'[')
185 ;
186}

188void QMimeAllGlobPatterns::addGlob(const QMimeGlobPattern &glob)
189{
190 const QString &pattern = glob.pattern();
191 Q_ASSERT(!pattern.isEmpty());
192
193 // Store each patterns into either m_fastPatternDict (*.txt, *.html etc. with default weight 50)
194 // or for the rest, like core.*, *.tar.bz2, *~, into highWeightPatternOffset (>50)
195 // or lowWeightPatternOffset (<=50)
196
197 if (glob.weight() == 50 && isFastPattern(pattern) && !glob.isCaseSensitive()) {
198 // The bulk of the patterns is *.foo with weight 50 --> those go into the fast patterns hash.
199 const QString extension = pattern.mid(position: 2).toLower();
200 QStringList &patterns = m_fastPatterns[extension]; // find or create
201 if (!patterns.contains(str: glob.mimeType()))
202 patterns.append(t: glob.mimeType());
203 } else {
204 if (glob.weight() > 50) {
205 if (!m_highWeightGlobs.hasPattern(mimeType: glob.mimeType(), pattern: glob.pattern()))
206 m_highWeightGlobs.append(t: glob);
207 } else {
208 if (!m_lowWeightGlobs.hasPattern(mimeType: glob.mimeType(), pattern: glob.pattern()))
209 m_lowWeightGlobs.append(t: glob);
210 }
211 }
212}

214void QMimeAllGlobPatterns::removeMimeType(const QString &mimeType)
215{
216 for (auto &x : m_fastPatterns)
217 x.removeAll(t: mimeType);
218 m_highWeightGlobs.removeMimeType(mimeType);
219 m_lowWeightGlobs.removeMimeType(mimeType);
220}

222void QMimeGlobPatternList::match(QMimeGlobMatchResult &result,
223 const QString &fileName) const
224{
225
226 QMimeGlobPatternList::const_iterator it = this->constBegin();
227 const QMimeGlobPatternList::const_iterator endIt = this->constEnd();
228 for (; it != endIt; ++it) {
229 const QMimeGlobPattern &glob = *it;
230 if (glob.matchFileName(inputFileName: fileName)) {
231 const QString pattern = glob.pattern();
232 const int suffixLen = isSimplePattern(pattern) ? pattern.size() - 2 : 0;
233 result.addMatch(mimeType: glob.mimeType(), weight: glob.weight(), pattern, knownSuffixLength: suffixLen);
234 }
235 }
236}

238void QMimeAllGlobPatterns::matchingGlobs(const QString &fileName, QMimeGlobMatchResult &result) const
239{
240 // First try the high weight matches (>50), if any.
241 m_highWeightGlobs.match(result, fileName);
242
243 // Now use the "fast patterns" dict, for simple *.foo patterns with weight 50
244 // (which is most of them, so this optimization is definitely worth it)
245 const int lastDot = fileName.lastIndexOf(c: u'.');
246 if (lastDot != -1) { // if no '.', skip the extension lookup
247 const int ext_len = fileName.size() - lastDot - 1;
248 const QString simpleExtension = fileName.right(n: ext_len).toLower();
249 // (toLower because fast patterns are always case-insensitive and saved as lowercase)
250
251 const QStringList matchingMimeTypes = m_fastPatterns.value(key: simpleExtension);
252 const QString simplePattern = "*."_L1 + simpleExtension;
253 for (const QString &mime : matchingMimeTypes)
254 result.addMatch(mimeType: mime, weight: 50, pattern: simplePattern, knownSuffixLength: simpleExtension.size());
255 // Can't return yet; *.tar.bz2 has to win over *.bz2, so we need the low-weight mimetypes anyway,
256 // at least those with weight 50.
257 }
258
259 // Finally, try the low weight matches (<=50)
260 m_lowWeightGlobs.match(result, fileName);
261}
