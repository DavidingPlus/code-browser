51#define QMIMETYPE_BUILDER_FROM_RVALUE_REFS \
52 QT_BEGIN_NAMESPACE \
53 static QMimeType buildQMimeType ( \
54 QString &&name, \
55 QString &&genericIconName, \
56 QString &&iconName, \
57 QStringList &&globPatterns \
58 ) \
59 { \
60 QMimeTypePrivate qMimeTypeData; \
61 qMimeTypeData.loaded = true; \
62 qMimeTypeData.name = std::move(name); \
63 qMimeTypeData.genericIconName = std::move(genericIconName); \
64 qMimeTypeData.iconName = std::move(iconName); \
65 qMimeTypeData.globPatterns = std::move(globPatterns); \
66 return QMimeType(qMimeTypeData); \
67 } \
