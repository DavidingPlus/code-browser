81void QMimeDatabasePrivate::loadProviders()
82{
83 // We use QStandardPaths every time to check if new files appeared
84 const QStringList mimeDirs = locateMimeDirectories();
85 const auto fdoIterator = std::find_if(first: mimeDirs.constBegin(), last: mimeDirs.constEnd(), pred: [](const QString &mimeDir) -> bool {
86 return QFileInfo::exists(file: mimeDir + "/packages/freedesktop.org.xml"_L1); }
87 );
88 const bool needInternalDB = QMimeXMLProvider::InternalDatabaseAvailable && fdoIterator == mimeDirs.constEnd();
89 //qDebug() << "mime dirs:" << mimeDirs;
90
91 Providers currentProviders;
92 std::swap(x&: m_providers, y&: currentProviders);
93
94 m_providers.reserve(n: mimeDirs.size() + (needInternalDB ? 1 : 0));
95
96 for (const QString &mimeDir : mimeDirs) {
97 const QString cacheFile = mimeDir + "/mime.cache"_L1;
98 // Check if we already have a provider for this dir
99 const auto predicate = [mimeDir](const std::unique_ptr<QMimeProviderBase> &prov)
100 {
101 return prov && prov->directory() == mimeDir;
102 };
103 const auto it = std::find_if(first: currentProviders.begin(), last: currentProviders.end(), pred: predicate);
104 if (it == currentProviders.end()) {
105 std::unique_ptr<QMimeProviderBase> provider;
106#if defined(QT_USE_MMAP)
107 if (qEnvironmentVariableIsEmpty(varName: "QT_NO_MIME_CACHE") && QFileInfo::exists(file: cacheFile)) {
108 provider.reset(p: new QMimeBinaryProvider(this, mimeDir));
109 //qDebug() << "Created binary provider for" << mimeDir;
110 if (!provider->isValid()) {
111 provider.reset();
112 }
113 }
114#endif
115 if (!provider) {
116 provider.reset(p: new QMimeXMLProvider(this, mimeDir));
117 //qDebug() << "Created XML provider for" << mimeDir;
118 }
119 m_providers.push_back(x: std::move(provider));
120 } else {
121 auto provider = std::move(*it); // take provider out of the vector
122 provider->ensureLoaded();
123 if (!provider->isValid()) {
124 provider.reset(p: new QMimeXMLProvider(this, mimeDir));
125 //qDebug() << "Created XML provider to replace binary provider for" << mimeDir;
126 }
127 m_providers.push_back(x: std::move(provider));
128 }
129 }
130 // mimeDirs is sorted "most local first, most global last"
131 // so the internal XML DB goes at the end
132 if (needInternalDB) {
133 // Check if we already have a provider for the InternalDatabase
134 const auto isInternal = [](const std::unique_ptr<QMimeProviderBase> &prov)
135 {
136 return prov && prov->isInternalDatabase();
137 };
138 const auto it = std::find_if(first: currentProviders.begin(), last: currentProviders.end(), pred: isInternal);
139 if (it == currentProviders.end()) {
140 m_providers.push_back(x: Providers::value_type(new QMimeXMLProvider(this, QMimeXMLProvider::InternalDatabase)));
141 } else {
142 m_providers.push_back(x: std::move(*it));
143 }
144 }
145}

147const QMimeDatabasePrivate::Providers &QMimeDatabasePrivate::providers()
148{
149#ifndef Q_OS_WASM // stub implementation always returns true
150 Q_ASSERT(!mutex.tryLock()); // caller should have locked mutex
151#endif

207void QMimeDatabasePrivate::loadMimeTypePrivate(QMimeTypePrivate &mimePrivate)
208{
209 QMutexLocker locker(&mutex);
210 if (mimePrivate.name.isEmpty())
211 return; // invalid mimetype
212 if (!mimePrivate.loaded) { // XML provider sets loaded=true, binary provider does this on demand
213 Q_ASSERT(mimePrivate.fromCache);
214 bool found = false;
215 for (const auto &provider : providers()) {
216 if (provider->loadMimeTypePrivate(mimePrivate)) {
217 found = true;
218 break;
219 }
220 }
221 if (!found) {
222 const QString file = mimePrivate.name + ".xml"_L1;
223 qWarning() << "No file found for" << file << ", even though update-mime-info said it would exist.\n"
224 "Either it was just removed, or the directory doesn't have executable permission..."
225 << locateMimeDirectories();
226 }
227 mimePrivate.loaded = true;
228 }
229}

257QString QMimeDatabasePrivate::fallbackParent(const QString &mimeTypeName) const
258{
259 const QStringView myGroup = QStringView{mimeTypeName}.left(n: mimeTypeName.indexOf(c: u'/'));
260 // All text/* types are subclasses of text/plain.
261 if (myGroup == "text"_L1 && mimeTypeName != plainTextMimeType())
262 return plainTextMimeType();
263 // All real-file mimetypes implicitly derive from application/octet-stream
264 if (myGroup != "inode"_L1 &&
265 // ignore non-file extensions
266 myGroup != "all"_L1 && myGroup != "fonts"_L1 && myGroup != "print"_L1 && myGroup != "uri"_L1
267 && mimeTypeName != defaultMimeType()) {
268 return defaultMimeType();
269 }
270 return QString();
271}

308static inline bool isTextFile(const QByteArray &data)
309{
310 // UTF16 byte order marks
311 static const char bigEndianBOM[] = "\xFE\xFF";
312 static const char littleEndianBOM[] = "\xFF\xFE";
313 if (data.startsWith(bv: bigEndianBOM) || data.startsWith(bv: littleEndianBOM))
314 return true;
315
316 // Check the first 128 bytes (see shared-mime spec)
317 const char *p = data.constData();
318 const char *e = p + qMin(a: 128, b: data.size());
319 for ( ; p < e; ++p) {
320 if (static_cast<unsigned char>(*p) < 32 && *p != 9 && *p !=10 && *p != 13)
321 return false;
322 }
323
324 return true;
325}

327QMimeType QMimeDatabasePrivate::findByData(const QByteArray &data, int *accuracyPtr)
328{
329 if (data.isEmpty()) {
330 *accuracyPtr = 100;
331 return mimeTypeForName(QStringLiteral("application/x-zerosize"));
332 }
333
334 *accuracyPtr = 0;
335 QMimeType candidate;
336 for (const auto &provider : providers())
337 provider->findByMagic(data, accuracyPtr, candidate);
338
339 if (candidate.isValid())
340 return candidate;
341
342 if (isTextFile(data)) {
343 *accuracyPtr = 5;
344 return mimeTypeForName(nameOrAlias: plainTextMimeType());
345 }
346
347 return mimeTypeForName(nameOrAlias: defaultMimeType());
348}

350QMimeType QMimeDatabasePrivate::mimeTypeForFileNameAndData(const QString &fileName, QIODevice *device)
351{
352 // First, glob patterns are evaluated. If there is a match with max weight,
353 // this one is selected and we are done. Otherwise, the file contents are
354 // evaluated and the match with the highest value (either a magic priority or
355 // a glob pattern weight) is selected. Matching starts from max level (most
356 // specific) in both cases, even when there is already a suffix matching candidate.
357
358 // Pass 1) Try to match on the file name
359 QMimeGlobMatchResult candidatesByName = findByFileName(fileName);
360 if (candidatesByName.m_allMatchingMimeTypes.size() == 1) {
361 const QMimeType mime = mimeTypeForName(nameOrAlias: candidatesByName.m_matchingMimeTypes.at(i: 0));
362 if (mime.isValid())
363 return mime;
364 candidatesByName = {};
365 }
366
367 // Extension is unknown, or matches multiple mimetypes.
368 // Pass 2) Match on content, if we can read the data
369 const auto matchOnContent = [this, &candidatesByName](QIODevice *device) {
370 const bool openedByUs = !device->isOpen() && device->open(mode: QIODevice::ReadOnly);
371 if (device->isOpen()) {
372 // Read 16K in one go (QIODEVICE_BUFFERSIZE in qiodevice_p.h).
373 // This is much faster than seeking back and forth into QIODevice.
374 const QByteArray data = device->peek(maxlen: 16384);
375
376 if (openedByUs)
377 device->close();
378
379 int magicAccuracy = 0;
380 QMimeType candidateByData(findByData(data, accuracyPtr: &magicAccuracy));
381
382 // Disambiguate conflicting extensions (if magic matching found something)
383 if (candidateByData.isValid() && magicAccuracy > 0) {
384 const QString sniffedMime = candidateByData.name();
385 // If the sniffedMime matches a highest-weight glob match, use it
386 if (candidatesByName.m_matchingMimeTypes.contains(str: sniffedMime)) {
387 return candidateByData;
388 }
389 for (const QString &m : std::as_const(t&: candidatesByName.m_allMatchingMimeTypes)) {
390 if (inherits(mime: m, parent: sniffedMime)) {
391 // We have magic + pattern pointing to this, so it's a pretty good match
392 return mimeTypeForName(nameOrAlias: m);
393 }
394 }
395 if (candidatesByName.m_allMatchingMimeTypes.isEmpty()) {
396 // No glob, use magic
397 return candidateByData;
398 }
399 }
400 }
401
402 if (candidatesByName.m_allMatchingMimeTypes.size() > 1) {
403 candidatesByName.m_matchingMimeTypes.sort(); // make it deterministic
404 const QMimeType mime = mimeTypeForName(nameOrAlias: candidatesByName.m_matchingMimeTypes.at(i: 0));
405 if (mime.isValid())
406 return mime;
407 }
408
409 return mimeTypeForName(nameOrAlias: defaultMimeType());
410 };
411
412 if (device)
413 return matchOnContent(device);
414
415 QFile fallbackFile(fileName);
416 return matchOnContent(&fallbackFile);
417}

430QMimeType QMimeDatabasePrivate::mimeTypeForData(QIODevice *device)
431{
432 int accuracy = 0;
433 const bool openedByUs = !device->isOpen() && device->open(mode: QIODevice::ReadOnly);
434 if (device->isOpen()) {
435 // Read 16K in one go (QIODEVICE_BUFFERSIZE in qiodevice_p.h).
436 // This is much faster than seeking back and forth into QIODevice.
437 const QByteArray data = device->peek(maxlen: 16384);
438 QMimeType result = findByData(data, accuracyPtr: &accuracy);
439 if (openedByUs)
440 device->close();
441 return result;
442 }
443 return mimeTypeForName(nameOrAlias: defaultMimeType());
444}

446QMimeType QMimeDatabasePrivate::mimeTypeForFile(const QString &fileName,
447 const QFileInfo &fileInfo,
448 QMimeDatabase::MatchMode mode)
449{
450 if (false) {
451#ifdef Q_OS_UNIX
452 } else if (fileInfo.isNativePath()) {
453 // If this is a local file, we'll want to do a stat() ourselves so we can
454 // detect additional inode types. In addition we want to follow symlinks.
455 const QByteArray nativeFilePath = QFile::encodeName(fileName);
456 QT_STATBUF statBuffer;
457 if (QT_STAT(path: nativeFilePath.constData(), statbuf: &statBuffer) == 0) {
458 if (S_ISDIR(statBuffer.st_mode))
459 return mimeTypeForName(nameOrAlias: directoryMimeType());
460 if (S_ISCHR(statBuffer.st_mode))
461 return mimeTypeForName(QStringLiteral("inode/chardevice"));
462 if (S_ISBLK(statBuffer.st_mode))
463 return mimeTypeForName(QStringLiteral("inode/blockdevice"));
464 if (S_ISFIFO(statBuffer.st_mode))
465 return mimeTypeForName(QStringLiteral("inode/fifo"));
466 if (S_ISSOCK(statBuffer.st_mode))
467 return mimeTypeForName(QStringLiteral("inode/socket"));
468 }
469#endif
470 } else if (fileInfo.isDir()) {
471 return mimeTypeForName(nameOrAlias: directoryMimeType());
472 }
473
474 switch (mode) {
475 case QMimeDatabase::MatchDefault:
476 break;
477 case QMimeDatabase::MatchExtension:
478 return mimeTypeForFileExtension(fileName);
479 case QMimeDatabase::MatchContent: {
480 QFile file(fileName);
481 return mimeTypeForData(device: &file);
482 }
483 }
484 // MatchDefault:
485 return mimeTypeForFileNameAndData(fileName, device: nullptr);
486}

496bool QMimeDatabasePrivate::inherits(const QString &mime, const QString &parent)
497{
498 const QString resolvedParent = resolveAlias(nameOrAlias: parent);
499 std::stack<QString, QStringList> toCheck;
500 toCheck.push(x: mime);
501 while (!toCheck.empty()) {
502 if (toCheck.top() == resolvedParent)
503 return true;
504 const QString mimeName = toCheck.top();
505 toCheck.pop();
506 const auto parentList = parents(mimeName);
507 for (const QString &par : parentList)
508 toCheck.push(x: resolveAlias(nameOrAlias: par));
509 }
510 return false;
511}

513/*!
514 \class QMimeDatabase
515 \inmodule QtCore
516 \brief The QMimeDatabase class maintains a database of MIME types.
517
518 \since 5.0
519
520 The MIME type database is provided by the freedesktop.org shared-mime-info
521 project. If the MIME type database cannot be found on the system, as is the case
522 on most Windows, \macos, and iOS systems, Qt will use its own copy of it.
523
524 Applications which want to define custom MIME types need to install an
525 XML file into the locations searched for MIME definitions.
526 These locations can be queried with
527 \snippet code/src_corelib_mimetype_qmimedatabase.cpp 1
528 On a typical Unix system, this will be /usr/share/mime/packages/, but it is also
529 possible to extend the list of directories by setting the environment variable
530 \c XDG_DATA_DIRS. For instance adding /opt/myapp/share to \c XDG_DATA_DIRS will result
531 in /opt/myapp/share/mime/packages/ being searched for MIME definitions.
532
533 Here is an example of MIME XML:
534 \snippet code/src_corelib_mimetype_qmimedatabase.cpp 2
535
536 For more details about the syntax of XML MIME definitions, including defining
537 "magic" in order to detect MIME types based on data as well, read the
538 Shared Mime Info specification at
539 http://standards.freedesktop.org/shared-mime-info-spec/shared-mime-info-spec-latest.html
540
541 On Unix systems, a binary cache is used for more performance. This cache is generated
542 by the command "update-mime-database path", where path would be /opt/myapp/share/mime
543 in the above example. Make sure to run this command when installing the MIME type
544 definition file.
545
546 \threadsafe
547
548 \snippet code/src_corelib_mimetype_qmimedatabase.cpp 0
549
550 \sa QMimeType, {MIME Type Browser Example}
551 */
552

587/*!
588 Returns a MIME type for \a fileInfo.
589
590 A valid MIME type is always returned.
591
592 The default matching algorithm looks at both the file name and the file
593 contents, if necessary. The file extension has priority over the contents,
594 but the contents will be used if the file extension is unknown, or
595 matches multiple MIME types.
596 If \a fileInfo is a Unix symbolic link, the file that it refers to
597 will be used instead.
598 If the file doesn't match any known pattern or data, the default MIME type
599 (application/octet-stream) is returned.
600
601 When \a mode is set to MatchExtension, only the file name is used, not
602 the file contents. The file doesn't even have to exist. If the file name
603 doesn't match any known pattern, the default MIME type (application/octet-stream)
604 is returned.
605 If multiple MIME types match this file, the first one (alphabetically) is returned.
606
607 When \a mode is set to MatchContent, and the file is readable, only the
608 file contents are used to determine the MIME type. This is equivalent to
609 calling mimeTypeForData with a QFile as input device.
610
611 \a fileInfo may refer to an absolute or relative path.
612
613 \sa QMimeType::isDefault(), mimeTypeForData()
614*/
615QMimeType QMimeDatabase::mimeTypeForFile(const QFileInfo &fileInfo, MatchMode mode) const
616{
617 QMutexLocker locker(&d->mutex);
618
619 return d->mimeTypeForFile(fileName: fileInfo.filePath(), fileInfo, mode);
620}

639/*!
640 Returns the MIME types for the file name \a fileName.
641
642 If the file name doesn't match any known pattern, an empty list is returned.
643 If multiple MIME types match this file, they are all returned.
644
645 This function does not try to open the file. To also use the content
646 when determining the MIME type, use mimeTypeForFile() or
647 mimeTypeForFileNameAndData() instead.
648
649 \sa mimeTypeForFile()
650*/
651QList<QMimeType> QMimeDatabase::mimeTypesForFileName(const QString &fileName) const
652{
653 QMutexLocker locker(&d->mutex);
654
655 const QStringList matches = d->mimeTypeForFileName(fileName);
656 QList<QMimeType> mimes;
657 mimes.reserve(size: matches.size());
658 for (const QString &mime : matches)
659 mimes.append(t: d->mimeTypeForName(nameOrAlias: mime));
660 return mimes;
661}
662/*!

668QString QMimeDatabase::suffixForFileName(const QString &fileName) const
669{

675/*!
676 Returns a MIME type for \a data.
677
678 A valid MIME type is always returned. If \a data doesn't match any
679 known MIME type data, the default MIME type (application/octet-stream)
680 is returned.
681*/

704/*!
705 Returns a MIME type for \a url.
706
707 If the URL is a local file, this calls mimeTypeForFile.
708
709 Otherwise the matching is done based on the file name only,
710 except for schemes where file names don't mean much, like HTTP.
711 This method always returns the default mimetype for HTTP URLs,
712 use QNetworkAccessManager to handle HTTP URLs properly.
713
714 A valid MIME type is always returned. If \a url doesn't match any
715 known MIME type data, the default MIME type (application/octet-stream)
716 is returned.
717*/
718QMimeType QMimeDatabase::mimeTypeForUrl(const QUrl &url) const
719{
720 if (url.isLocalFile())
721 return mimeTypeForFile(fileName: url.toLocalFile());
722
723 const QString scheme = url.scheme();
724 if (scheme.startsWith(s: "http"_L1) || scheme == "mailto"_L1)
725 return mimeTypeForName(nameOrAlias: d->defaultMimeType());
726
727 return mimeTypeForFile(fileName: url.path(), mode: MatchExtension);
728}
