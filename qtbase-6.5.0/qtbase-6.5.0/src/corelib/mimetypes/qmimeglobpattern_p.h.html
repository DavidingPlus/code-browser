27struct QMimeGlobMatchResult
28{
29 void addMatch(const QString &mimeType, int weight, const QString &pattern, int knownSuffixLength = 0);

31 QStringList m_matchingMimeTypes; // only those with highest weight

38class QMimeGlobPattern
39{
40public:
41 static const unsigned MaxWeight = 100;
42 static const unsigned DefaultWeight = 50;
43 static const unsigned MinWeight = 1;
44
45 explicit QMimeGlobPattern(const QString &thePattern, const QString &theMimeType, unsigned theWeight = DefaultWeight, Qt::CaseSensitivity s = Qt::CaseInsensitive) :
46 m_pattern(s == Qt::CaseInsensitive ? thePattern.toLower() : thePattern),
47 m_mimeType(theMimeType),
48 m_weight(theWeight),
49 m_caseSensitivity(s),
50 m_patternType(detectPatternType(pattern: m_pattern))
51 {
52 }
53
54 void swap(QMimeGlobPattern &other) noexcept
55 {
56 qSwap(value1&: m_pattern, value2&: other.m_pattern);
57 qSwap(value1&: m_mimeType, value2&: other.m_mimeType);
58 qSwap(value1&: m_weight, value2&: other.m_weight);
59 qSwap(value1&: m_caseSensitivity, value2&: other.m_caseSensitivity);
60 qSwap(value1&: m_patternType, value2&: other.m_patternType);
61 }
62
63 bool matchFileName(const QString &inputFileName) const;
64
65 inline const QString &pattern() const { return m_pattern; }
66 inline unsigned weight() const { return m_weight; }
67 inline const QString &mimeType() const { return m_mimeType; }
68 inline bool isCaseSensitive() const { return m_caseSensitivity == Qt::CaseSensitive; }
69
70private:
71 enum PatternType {
72 SuffixPattern,
73 PrefixPattern,
74 LiteralPattern,
75 VdrPattern, // special handling for "[0-9][0-9][0-9].vdr" pattern
76 AnimPattern, // special handling for "*.anim[1-9j]" pattern
77 OtherPattern
78 };
79 PatternType detectPatternType(const QString &pattern) const;
80
81 QString m_pattern;
82 QString m_mimeType;
83 int m_weight;
84 Qt::CaseSensitivity m_caseSensitivity;
85 PatternType m_patternType;
86};

89class QMimeGlobPatternList : public QList<QMimeGlobPattern>
90{
91public:
92 bool hasPattern(const QString &mimeType, const QString &pattern) const
93 {
94 const_iterator it = begin();
95 const const_iterator myend = end();
96 for (; it != myend; ++it)
97 if ((*it).pattern() == pattern && (*it).mimeType() == mimeType)
98 return true;
99 return false;
100 }
101
102 /*!
103 "noglobs" is very rare occurrence, so it's ok if it's slow
104 */
105 void removeMimeType(const QString &mimeType)
106 {
107 auto isMimeTypeEqual = [&mimeType](const QMimeGlobPattern &pattern) {
108 return pattern.mimeType() == mimeType;
109 };
110 removeIf(pred: isMimeTypeEqual);
111 }
112
113 void match(QMimeGlobMatchResult &result, const QString &fileName) const;
114};

116/*!
117 Result of the globs parsing, as data structures ready for efficient MIME type matching.
118 This contains:
119 1) a map of fast regular patterns (e.g. *.txt is stored as "txt" in a qhash's key)
120 2) a linear list of high-weight globs
121 3) a linear list of low-weight globs
122 */
123class QMimeAllGlobPatterns
124{
125public:
126 typedef QHash<QString, QStringList> PatternsMap; // MIME type -> patterns
127
128 void addGlob(const QMimeGlobPattern &glob);
129 void removeMimeType(const QString &mimeType);
130 void matchingGlobs(const QString &fileName, QMimeGlobMatchResult &result) const;
131 void clear();
132
133 PatternsMap m_fastPatterns; // example: "doc" -> "application/msword", "text/plain"
134 QMimeGlobPatternList m_highWeightGlobs;
135 QMimeGlobPatternList m_lowWeightGlobs; // <= 50, including the non-fast 50 patterns
136};
