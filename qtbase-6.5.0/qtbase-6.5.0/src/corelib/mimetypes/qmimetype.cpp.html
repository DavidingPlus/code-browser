50/*!
51 \class QMimeType
52 \inmodule QtCore
53 \ingroup shared
54 \brief The QMimeType class describes types of file or data, represented by a MIME type string.
55
56 \since 5.0
57
58 For instance a file named "readme.txt" has the MIME type "text/plain".
59 The MIME type can be determined from the file name, or from the file
60 contents, or from both. MIME type determination can also be done on
61 buffers of data not coming from files.
62
63 Determining the MIME type of a file can be useful to make sure your
64 application supports it. It is also useful in file-manager-like applications
65 or widgets, in order to display an appropriate \l {QMimeType::iconName}{icon} for the file, or even
66 the descriptive \l {QMimeType::comment()}{comment} in detailed views.
67
68 To check if a file has the expected MIME type, you should use inherits()
69 rather than a simple string comparison based on the name(). This is because
70 MIME types can inherit from each other: for instance a C source file is
71 a specific type of plain text file, so text/x-csrc inherits text/plain.
72
73 \sa QMimeDatabase, {MIME Type Browser Example}
74 */

211/*!
212 \property QMimeType::comment
213 \brief the description of the MIME type to be displayed on user interfaces
214
215 The default language (QLocale().name()) is used to select the appropriate translation.
216
217 While this property was introduced in 5.10, the
218 corresponding accessor method has always been there.
219 */
220QString QMimeType::comment() const
221{
222 QMimeDatabasePrivate::instance()->loadMimeTypePrivate(mimePrivate&: const_cast<QMimeTypePrivate&>(*d));
223
224 QStringList languageList;
225 languageList << QLocale().name();
226 languageList << QLocale().uiLanguages();
227 languageList << u"default"_s; // use the default locale if possible.
228 for (const QString &language : std::as_const(t&: languageList)) {
229 const QString lang = language == "C"_L1 ? u"en_US"_s : language;
230 const QString comm = d->localeComments.value(key: lang);
231 if (!comm.isEmpty())
232 return comm;
233 const int pos = lang.indexOf(c: u'_');
234 if (pos != -1) {
235 // "pt_BR" not found? try just "pt"
236 const QString shortLang = lang.left(n: pos);
237 const QString commShort = d->localeComments.value(key: shortLang);
238 if (!commShort.isEmpty())
239 return commShort;
240 }
241 }
242
243 // Use the mimetype name as fallback
244 return d->name;
245}

247/*!
248 \property QMimeType::genericIconName
249 \brief the file name of a generic icon that represents the MIME type
250
251 This should be used if the icon returned by iconName() cannot be found on
252 the system. It is used for categories of similar types (like spreadsheets
253 or archives) that can use a common icon.
254 The freedesktop.org Icon Naming Specification lists a set of such icon names.
255
256 The icon name can be given to QIcon::fromTheme() in order to load the icon.
257
258 While this property was introduced in 5.10, the
259 corresponding accessor method has always been there.
260 */
261QString QMimeType::genericIconName() const
262{
263 QMimeDatabasePrivate::instance()->loadGenericIcon(mimePrivate&: const_cast<QMimeTypePrivate&>(*d));
264 if (d->genericIconName.isEmpty()) {
265 // From the spec:
266 // If the generic icon name is empty (not specified by the mimetype definition)
267 // then the mimetype is used to generate the generic icon by using the top-level
268 // media type (e.g. "video" in "video/ogg") and appending "-x-generic"
269 // (i.e. "video-x-generic" in the previous example).
270 const QString group = name();
271 QStringView groupRef(group);
272 const int slashindex = groupRef.indexOf(c: u'/');
273 if (slashindex != -1)
274 groupRef = groupRef.left(n: slashindex);
275 return groupRef + "-x-generic"_L1;
276 }
277 return d->genericIconName;
278}

280static QString make_default_icon_name_from_mimetype_name(QString iconName)
281{
282 const int slashindex = iconName.indexOf(c: u'/');
283 if (slashindex != -1)
284 iconName[slashindex] = u'-';
285 return iconName;
286}

319/*!
320 \property QMimeType::parentMimeTypes
321 \brief the names of parent MIME types
322
323 A type is a subclass of another type if any instance of the first type is
324 also an instance of the second. For example, all image/svg+xml files are also
325 text/xml, text/plain and application/octet-stream files. Subclassing is about
326 the format, rather than the category of the data (for example, there is no
327 'generic spreadsheet' class that all spreadsheets inherit from).
328 Conversely, the parent mimetype of image/svg+xml is text/xml.
329
330 A mimetype can have multiple parents. For instance application/x-perl
331 has two parents: application/x-executable and text/plain. This makes
332 it possible to both execute perl scripts, and to open them in text editors.
333
334 While this property was introduced in 5.10, the
335 corresponding accessor method has always been there.
336*/
337QStringList QMimeType::parentMimeTypes() const
338{
339 return QMimeDatabasePrivate::instance()->mimeParents(mimeName: d->name);
340}

342static void collectParentMimeTypes(const QString &mime, QStringList &allParents)
343{
344 const QStringList parents = QMimeDatabasePrivate::instance()->mimeParents(mimeName: mime);
345 for (const QString &parent : parents) {
346 // I would use QSet, but since order matters I better not
347 if (!allParents.contains(str: parent))
348 allParents.append(t: parent);
349 }
350 // We want a breadth-first search, so that the least-specific parent (octet-stream) is last
351 // This means iterating twice, unfortunately.
352 for (const QString &parent : parents)
353 collectParentMimeTypes(mime: parent, allParents);
354}

356/*!
357 \property QMimeType::allAncestors
358 \brief the names of direct and indirect parent MIME types
359
360 Return all the parent mimetypes of this mimetype, direct and indirect.
361 This includes the parent(s) of its parent(s), etc.
362
363 For instance, for image/svg+xml the list would be:
364 application/xml, text/plain, application/octet-stream.
365
366 Note that application/octet-stream is the ultimate parent for all types
367 of files (but not directories).
368
369 While this property was introduced in 5.10, the
370 corresponding accessor method has always been there.
371*/
372QStringList QMimeType::allAncestors() const
373{
374 QStringList allParents;
375 collectParentMimeTypes(mime: d->name, allParents);
376 return allParents;
377}

399/*!
400 \property QMimeType::suffixes
401 \brief the known suffixes for the MIME type
402
403 No leading dot is included, so for instance this would return "jpg", "jpeg" for image/jpeg.
404
405 While this property was introduced in 5.10, the
406 corresponding accessor method has always been there.
407 */
408QStringList QMimeType::suffixes() const
409{
410 QMimeDatabasePrivate::instance()->loadMimeTypePrivate(mimePrivate&: const_cast<QMimeTypePrivate&>(*d));
411
412 QStringList result;
413 for (const QString &pattern : std::as_const(t&: d->globPatterns)) {
414 // Not a simple suffix if it looks like: README or *. or *.* or *.JP*G or *.JP?
415 if (pattern.startsWith(s: "*."_L1) &&
416 pattern.size() > 2 &&
417 pattern.indexOf(c: u'*', from: 2) < 0 && pattern.indexOf(c: u'?', from: 2) < 0) {
418 const QString suffix = pattern.mid(position: 2);
419 result.append(t: suffix);
420 }
421 }
422
423 return result;
424}

426/*!
427 \property QMimeType::preferredSuffix
428 \brief the preferred suffix for the MIME type
429
430 No leading dot is included, so for instance this would return "pdf" for application/pdf.
431 The return value can be empty, for mime types which do not have any suffixes associated.
432
433 While this property was introduced in 5.10, the
434 corresponding accessor method has always been there.
435 */
436QString QMimeType::preferredSuffix() const
437{
438 if (isDefault()) // workaround for unwanted *.bin suffix for octet-stream, https://bugs.freedesktop.org/show_bug.cgi?id=101667, fixed upstream in 1.10
439 return QString();
440 const QStringList suffixList = suffixes();
441 return suffixList.isEmpty() ? QString() : suffixList.at(i: 0);
442}

444/*!
445 \property QMimeType::filterString
446 \brief a filter string usable for a file dialog
447
448 While this property was introduced in 5.10, the
449 corresponding accessor method has always been there.
450*/
451QString QMimeType::filterString() const
452{
453 QMimeDatabasePrivate::instance()->loadMimeTypePrivate(mimePrivate&: const_cast<QMimeTypePrivate&>(*d));
454 QString filter;
455
456 if (!d->globPatterns.empty()) {
457 filter += comment() + " ("_L1;
458 for (int i = 0; i < d->globPatterns.size(); ++i) {
459 if (i != 0)
460 filter += u' ';
461 filter += d->globPatterns.at(i);
462 }
463 filter += u')';
464 }
465
466 return filter;
467}
