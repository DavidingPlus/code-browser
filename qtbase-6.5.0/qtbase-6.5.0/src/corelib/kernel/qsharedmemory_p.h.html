49/*!
50 Helper class
51 */
52class QSharedMemoryLocker
53{
54
55public:
56 inline QSharedMemoryLocker(QSharedMemory *sharedMemory) : q_sm(sharedMemory)
57 {
58 Q_ASSERT(q_sm);
59 }
60
61 inline ~QSharedMemoryLocker()
62 {
63 if (q_sm)
64 q_sm->unlock();
65 }
66
67 inline bool lock()
68 {
69 if (q_sm && q_sm->lock())
70 return true;
71 q_sm = nullptr;
72 return false;
73 }
74
75private:
76 QSharedMemory *q_sm;
77};
78#endif // QT_CONFIG(systemsemaphore)
