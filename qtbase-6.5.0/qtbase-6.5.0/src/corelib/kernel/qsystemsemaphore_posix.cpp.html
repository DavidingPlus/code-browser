33bool QSystemSemaphorePrivate::handle(QSystemSemaphore::AccessMode mode)
34{
35 if (semaphore != SEM_FAILED)
36 return true; // we already have a semaphore
37
38 if (fileName.isEmpty()) {
39 errorString = QSystemSemaphore::tr("%1: key is empty").arg("QSystemSemaphore::handle"_L1);
40 error = QSystemSemaphore::KeyError;
41 return false;
42 }
43
44 const QByteArray semName = QFile::encodeName(fileName);
45
46 // Always try with O_EXCL so we know whether we created the semaphore.
47 int oflag = O_CREAT | O_EXCL;
48 for (int tryNum = 0, maxTries = 1; tryNum < maxTries; ++tryNum) {
49 do {
50 semaphore = ::sem_open(semName.constData(), oflag, 0600, initialValue);
51 } while (semaphore == SEM_FAILED && errno == EINTR);
52 if (semaphore == SEM_FAILED && errno == EEXIST) {
53 if (mode == QSystemSemaphore::Create) {
54 if (::sem_unlink(semName.constData()) == -1 && errno != ENOENT) {
55 setErrorString("QSystemSemaphore::handle (sem_unlink)"_L1);
56 return false;
57 }
58 // Race condition: the semaphore might be recreated before
59 // we call sem_open again, so we'll retry several times.
60 maxTries = 3;
61 } else {
62 // Race condition: if it no longer exists at the next sem_open
63 // call, we won't realize we created it, so we'll leak it later.
64 oflag &= ~O_EXCL;
65 maxTries = 2;
66 }
67 } else {
68 break;
69 }
70 }
71
72 if (semaphore == SEM_FAILED) {
73 setErrorString("QSystemSemaphore::handle"_L1);
74 return false;
75 }
76
77 createdSemaphore = (oflag & O_EXCL) != 0;
78
79 return true;
80}

82void QSystemSemaphorePrivate::cleanHandle()
83{
84 if (semaphore != SEM_FAILED) {
85 if (::sem_close(semaphore) == -1) {
86 setErrorString("QSystemSemaphore::cleanHandle (sem_close)"_L1);
87#if defined QSYSTEMSEMAPHORE_DEBUG
88 qDebug("QSystemSemaphore::cleanHandle sem_close failed.");
89#endif
90 }
91 semaphore = SEM_FAILED;
92 }
93
94 if (createdSemaphore) {
95 if (::sem_unlink(QFile::encodeName(fileName).constData()) == -1 && errno != ENOENT) {
96 setErrorString("QSystemSemaphore::cleanHandle (sem_unlink)"_L1);
97#if defined QSYSTEMSEMAPHORE_DEBUG
98 qDebug("QSystemSemaphore::cleanHandle sem_unlink failed.");
99#endif
100 }
101 createdSemaphore = false;
102 }
103}

105bool QSystemSemaphorePrivate::modifySemaphore(int count)
106{
107 if (!handle())
108 return false;
109
110 if (count > 0) {
111 int cnt = count;
112 do {
113 if (::sem_post(semaphore) == -1) {
114 setErrorString("QSystemSemaphore::modifySemaphore (sem_post)"_L1);
115#if defined QSYSTEMSEMAPHORE_DEBUG
116 qDebug("QSystemSemaphore::modify sem_post failed %d %d", count, errno);
117#endif
118 // rollback changes to preserve the SysV semaphore behavior
119 for ( ; cnt < count; ++cnt) {
120 int res;
121 EINTR_LOOP(res, ::sem_wait(semaphore));
122 }
123 return false;
124 }
125 --cnt;
126 } while (cnt > 0);
127 } else {
128 int res;
129 EINTR_LOOP(res, ::sem_wait(semaphore));
130 if (res == -1) {
131 // If the semaphore was removed be nice and create it and then modifySemaphore again
132 if (errno == EINVAL || errno == EIDRM) {
133 semaphore = SEM_FAILED;
134 return modifySemaphore(count);
135 }
136 setErrorString("QSystemSemaphore::modifySemaphore (sem_wait)"_L1);
137#if defined QSYSTEMSEMAPHORE_DEBUG
138 qDebug("QSystemSemaphore::modify sem_wait failed %d %d", count, errno);
139#endif
140 return false;
141 }
142 }
143
144 clearError();
145 return true;
146}
