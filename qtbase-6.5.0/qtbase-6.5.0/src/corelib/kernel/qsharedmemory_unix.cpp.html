31void QSharedMemoryPrivate::setErrorString(QLatin1StringView function)
32{
33 // EINVAL is handled in functions so they can give better error strings
34 switch (errno) {
35 case EACCES:
36 errorString = QSharedMemory::tr(s: "%1: permission denied").arg(a: function);
37 error = QSharedMemory::PermissionDenied;
38 break;
39 case EEXIST:
40 errorString = QSharedMemory::tr(s: "%1: already exists").arg(a: function);
41 error = QSharedMemory::AlreadyExists;
42 break;
43 case ENOENT:
44 errorString = QSharedMemory::tr(s: "%1: doesn't exist").arg(a: function);
45 error = QSharedMemory::NotFound;
46 break;
47 case EMFILE:
48 case ENOMEM:
49 case ENOSPC:
50 errorString = QSharedMemory::tr(s: "%1: out of resources").arg(a: function);
51 error = QSharedMemory::OutOfResources;
52 break;
53 default:
54 errorString = QSharedMemory::tr(s: "%1: unknown error %2").arg(a: function).arg(errno);
55 error = QSharedMemory::UnknownError;
56#if defined QSHAREDMEMORY_DEBUG
57 qDebug() << errorString << "key" << key << "errno" << errno << EINVAL;
58#endif
59 }
60}
