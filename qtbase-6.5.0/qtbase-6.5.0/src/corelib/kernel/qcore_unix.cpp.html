23QByteArray qt_readlink(const char *path)
24{
25#ifndef PATH_MAX
26 // suitably large value that won't consume too much memory
27# define PATH_MAX 1024*1024
28#endif
29
30 QByteArray buf(256, Qt::Uninitialized);
31
32 ssize_t len = ::readlink(path: path, buf: buf.data(), len: buf.size());
33 while (len == buf.size()) {
34 // readlink(2) will fill our buffer and not necessarily terminate with NUL;
35 if (buf.size() >= PATH_MAX) {
36 errno = ENAMETOOLONG;
37 return QByteArray();
38 }
39
40 // double the size and try again
41 buf.resize(size: buf.size() * 2);
42 len = ::readlink(path: path, buf: buf.data(), len: buf.size());
43 }
44
45 if (len == -1)
46 return QByteArray();
47
48 buf.resize(size: len);
49 return buf;
50}

72static inline bool time_update(struct timespec *tv, const struct timespec &start,
73 const struct timespec &timeout)
74{
75 // clock source is (hopefully) monotonic, so we can recalculate how much timeout is left;
76 // if it isn't monotonic, we'll simply hope that it hasn't jumped, because we have no alternative
77 struct timespec now = qt_gettime();
78 *tv = timeout + start - now;
79 return tv->tv_sec >= 0;
80}

82#if QT_CONFIG(poll_poll)
83static inline int timespecToMillisecs(const struct timespec *ts)
84{
85 return (ts == NULL) ? -1 :
86 (ts->tv_sec * 1000) + (ts->tv_nsec / 1000000);
87}

93static inline int qt_ppoll(struct pollfd *fds, nfds_t nfds, const struct timespec *timeout_ts)
94{
95#if QT_CONFIG(poll_ppoll) || QT_CONFIG(poll_pollts)
96 return ::ppoll(fds, nfds, timeout_ts, nullptr);
97#elif QT_CONFIG(poll_poll)
98 return ::poll(fds, nfds, timespecToMillisecs(timeout_ts));
99#elif QT_CONFIG(poll_select)
100 return qt_poll(fds, nfds, timeout_ts);
101#else
102 // configure.json reports an error when everything is not available
103#endif
104}

107/*!
108 \internal
109
110 Behaves as close to POSIX poll(2) as practical but may be implemented
111 using select(2) where necessary. In that case, returns -1 and sets errno
112 to EINVAL if passed any descriptor greater than or equal to FD_SETSIZE.
113*/
114int qt_safe_poll(struct pollfd *fds, nfds_t nfds, const struct timespec *timeout_ts)
115{
116 if (!timeout_ts) {
117 // no timeout -> block forever
118 int ret;
119 EINTR_LOOP(ret, qt_ppoll(fds, nfds, nullptr));
120 return ret;
121 }
122
123 timespec start = qt_gettime();
124 timespec timeout = *timeout_ts;
125
126 // loop and recalculate the timeout as needed
127 forever {
128 const int ret = qt_ppoll(fds, nfds, &timeout);
129 if (ret != -1 || errno != EINTR)
130 return ret;
131
132 // recalculate the timeout
133 if (!time_update(&timeout, start, *timeout_ts)) {
134 // timeout during update
135 // or clock reset, fake timeout error
136 return 0;
137 }
138 }
139}
