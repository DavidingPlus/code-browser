32/*!
33 \internal
34
35 If not already made create the handle used for accessing the shared memory.
36*/
37key_t QSharedMemoryPrivate::handle()
38{
39 // already made
40 if (unix_key)
41 return unix_key;
42
43 // don't allow making handles on empty keys
44 if (nativeKey.isEmpty()) {
45 errorString = QSharedMemory::tr(s: "%1: key is empty").arg(a: "QSharedMemory::handle:"_L1);
46 error = QSharedMemory::KeyError;
47 return 0;
48 }
49
50 // ftok requires that an actual file exists somewhere
51 if (!QFile::exists(fileName: nativeKey)) {
52 errorString = QSharedMemory::tr(s: "%1: UNIX key file doesn't exist").arg(a: "QSharedMemory::handle:"_L1);
53 error = QSharedMemory::NotFound;
54 return 0;
55 }
56
57 unix_key = ftok(pathname: QFile::encodeName(fileName: nativeKey).constData(), proj_id: 'Q');
58 if (-1 == unix_key) {
59 errorString = QSharedMemory::tr(s: "%1: ftok failed").arg(a: "QSharedMemory::handle:"_L1);
60 error = QSharedMemory::KeyError;
61 unix_key = 0;
62 }
63 return unix_key;
64}

66#endif // QT_CONFIG(sharedmemory)

69/*!
70 \internal
71 Creates the unix file if needed.
72 returns \c true if the unix file was created.
73
74 -1 error
75 0 already existed
76 1 created
77 */
78int QT_PREPEND_NAMESPACE(QSharedMemoryPrivate)::createUnixKeyFile(const QString &fileName)
79{
80 int fd = qt_safe_open(pathname: QFile::encodeName(fileName).constData(),
81 O_EXCL | O_CREAT | O_RDWR, mode: 0640);
82 if (-1 == fd) {
83 if (errno == EEXIST)
84 return 0;
85 return -1;
86 } else {
87 close(fd: fd);
88 }
89 return 1;
90}

101bool QSharedMemoryPrivate::create(qsizetype size)
102{
103 // build file if needed
104 bool createdFile = false;
105 int built = createUnixKeyFile(fileName: nativeKey);
106 if (built == -1) {
107 errorString = QSharedMemory::tr(s: "%1: unable to make key").arg(a: "QSharedMemory::handle:"_L1);
108 error = QSharedMemory::KeyError;
109 return false;
110 }
111 if (built == 1) {
112 createdFile = true;
113 }
114
115 // get handle
116 if (!handle()) {
117 if (createdFile)
118 QFile::remove(fileName: nativeKey);
119 return false;
120 }
121
122 // create
123 if (-1 == shmget(key: unix_key, size: size_t(size), shmflg: 0600 | IPC_CREAT | IPC_EXCL)) {
124 const auto function = "QSharedMemory::create"_L1;
125 switch (errno) {
126 case EINVAL:
127 errorString = QSharedMemory::tr(s: "%1: system-imposed size restrictions").arg(a: "QSharedMemory::handle"_L1);
128 error = QSharedMemory::InvalidSize;
129 break;
130 default:
131 setErrorString(function);
132 }
133 if (createdFile && error != QSharedMemory::AlreadyExists)
134 QFile::remove(fileName: nativeKey);
135 return false;
136 }
137
138 return true;
139}

141bool QSharedMemoryPrivate::attach(QSharedMemory::AccessMode mode)
142{
143 // grab the shared memory segment id
144 int id = shmget(key: unix_key, size: 0, shmflg: (mode == QSharedMemory::ReadOnly ? 0400 : 0600));
145 if (-1 == id) {
146 setErrorString("QSharedMemory::attach (shmget)"_L1);
147 return false;
148 }
149
150 // grab the memory
151 memory = shmat(shmid: id, shmaddr: nullptr, shmflg: (mode == QSharedMemory::ReadOnly ? SHM_RDONLY : 0));
152 if ((void *)-1 == memory) {
153 memory = nullptr;
154 setErrorString("QSharedMemory::attach (shmat)"_L1);
155 return false;
156 }
157
158 // grab the size
159 shmid_ds shmid_ds;
160 if (!shmctl(shmid: id, IPC_STAT, buf: &shmid_ds)) {
161 size = (qsizetype)shmid_ds.shm_segsz;
162 } else {
163 setErrorString("QSharedMemory::attach (shmctl)"_L1);
164 return false;
165 }
166
167 return true;
168}

170bool QSharedMemoryPrivate::detach()
171{
172 // detach from the memory segment
173 if (-1 == shmdt(shmaddr: memory)) {
174 const auto function = "QSharedMemory::detach"_L1;
175 switch (errno) {
176 case EINVAL:
177 errorString = QSharedMemory::tr(s: "%1: not attached").arg(a: function);
178 error = QSharedMemory::NotFound;
179 break;
180 default:
181 setErrorString(function);
182 }
183 return false;
184 }
185 memory = nullptr;
186 size = 0;
187
188 // Get the number of current attachments
189 int id = shmget(key: unix_key, size: 0, shmflg: 0400);
190 cleanHandle();
191
192 struct shmid_ds shmid_ds;
193 if (0 != shmctl(shmid: id, IPC_STAT, buf: &shmid_ds)) {
194 switch (errno) {
195 case EINVAL:
196 return true;
197 default:
198 return false;
199 }
200 }
201 // If there are no attachments then remove it.
202 if (shmid_ds.shm_nattch == 0) {
203 // mark for removal
204 if (-1 == shmctl(shmid: id, IPC_RMID, buf: &shmid_ds)) {
205 setErrorString("QSharedMemory::remove"_L1);
206 switch (errno) {
207 case EINVAL:
208 return true;
209 default:
210 return false;
211 }
212 }
213
214 // remove file
215 if (!QFile::remove(fileName: nativeKey))
216 return false;
217 }
218 return true;
219}
