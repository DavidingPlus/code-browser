36/*!
37 \class QMetaObject
38 \inmodule QtCore
39
40 \brief The QMetaObject class contains meta-information about Qt
41 objects.
42
43 \ingroup objectmodel
44
45 The Qt \l{Meta-Object System} in Qt is responsible for the
46 signals and slots inter-object communication mechanism, runtime
47 type information, and the Qt property system. A single
48 QMetaObject instance is created for each QObject subclass that is
49 used in an application, and this instance stores all the
50 meta-information for the QObject subclass. This object is
51 available as QObject::metaObject().
52
53 This class is not normally required for application programming,
54 but it is useful if you write meta-applications, such as scripting
55 engines or GUI builders.
56
57 The functions you are most likely to find useful are these:
58 \list
59 \li className() returns the name of a class.
60 \li superClass() returns the superclass's meta-object.
61 \li method() and methodCount() provide information
62 about a class's meta-methods (signals, slots and other
63 \l{Q_INVOKABLE}{invokable} member functions).
64 \li enumerator() and enumeratorCount() and provide information about
65 a class's enumerators.
66 \li propertyCount() and property() provide information about a
67 class's properties.
68 \li constructor() and constructorCount() provide information
69 about a class's meta-constructors.
70 \endlist
71
72 The index functions indexOfConstructor(), indexOfMethod(),
73 indexOfEnumerator(), and indexOfProperty() map names of constructors,
74 member functions, enumerators, or properties to indexes in the
75 meta-object. For example, Qt uses indexOfMethod() internally when you
76 connect a signal to a slot.
77
78 Classes can also have a list of \e{name}--\e{value} pairs of
79 additional class information, stored in QMetaClassInfo objects.
80 The number of pairs is returned by classInfoCount(), single pairs
81 are returned by classInfo(), and you can search for pairs with
82 indexOfClassInfo().
83
84 \note Operations that use the meta object system are generally thread-
85 safe, as QMetaObjects are typically static read-only instances
86 generated at compile time. However, if meta objects are dynamically
87 modified by the application (for instance, when using QQmlPropertyMap),
88 then the application has to explicitly synchronize access to the
89 respective meta object.
90
91 \sa QMetaClassInfo, QMetaEnum, QMetaMethod, QMetaProperty, QMetaType,
92 {Meta-Object System}
93*/

250/*!
251 \fn template <typename... Args> QObject *QMetaObject::newInstance(Args &&... arguments) const
252 \since 6.5
253
254 Constructs a new instance of this class and returns the new object, or
255 \nullptr if no suitable constructor is available. The types of the
256 arguments \a arguments will be used to find a matching constructor, and then
257 forwarded to it the same way signal-slot connections do.
258
259 Note that only constructors that are declared with the Q_INVOKABLE
260 modifier are made available through the meta-object system.
261
262 \sa constructor()
263*/
264
265QObject *QMetaObject::newInstanceImpl(const QMetaObject *mobj, qsizetype paramCount,
266 const void **parameters, const char **typeNames,
267 const QtPrivate::QMetaTypeInterface **metaTypes)
268{
269 if (!mobj->inherits(metaObject: &QObject::staticMetaObject)) {
270 qWarning(msg: "QMetaObject::newInstance: type %s does not inherit QObject", mobj->className());
271 return nullptr;
272 }
273
274QT_WARNING_PUSH
275#if Q_CC_GNU >= 1200
276QT_WARNING_DISABLE_GCC("-Wdangling-pointer")
277#endif
278
279 // set the return type
280 QObject *returnValue = nullptr;
281 QMetaType returnValueMetaType = QMetaType::fromType<decltype(returnValue)>();
282 parameters[0] = &returnValue;
283 typeNames[0] = returnValueMetaType.name();
284 if (metaTypes)
285 metaTypes[0] = returnValueMetaType.iface();
286
287QT_WARNING_POP
288
289 // find the constructor
290 auto priv = QMetaObjectPrivate::get(metaobject: mobj);
291 for (int i = 0; i < priv->constructorCount; ++i) {
292 QMetaMethod m = QMetaMethod::fromRelativeConstructorIndex(mobj, index: i);
293 if (m.parameterCount() != (paramCount - 1))
294 continue;
295
296 // attempt to call
297 QMetaMethodPrivate::InvokeFailReason r =
298 QMetaMethodPrivate::invokeImpl(self: m, target: nullptr, Qt::DirectConnection, paramCount,
299 parameters, typeNames, metaTypes);
300 if (r == QMetaMethodPrivate::InvokeFailReason::None)
301 return returnValue;
302 if (int(r) < 0)
303 return nullptr;
304 }
305
306 return returnValue;
307}

403/*!
404 \since 6.2
405 Returns the metatype corresponding to this metaobject.
406 If the metaobject originates from a namespace, an invalid metatype is returned.
407 */
408QMetaType QMetaObject::metaType() const
409{
410
411 const QMetaObjectPrivate *d = priv(data: this->d.data);
412 if (d->revision < 10) {
413 // before revision 10, we did not store the metatype in the metatype array
414 return QMetaType::fromName(name: className());
415 } else {
416 /* in the metatype array, we store
417 idx: 0 propertyCount - 1 propertyCount
418 data:QMetaType(prop0), ..., QMetaType(propPropCount-1), QMetaType(class),...
419 */
420 auto iface = this->d.metaTypes[d->propertyCount];
421 if (iface && QtMetaTypePrivate::isInterfaceFor<void>(iface))
422 return QMetaType(); // return invalid meta-type for namespaces
423 if (iface)
424 return QMetaType(iface);
425 else // in case of a dynamic metaobject, we might have no metatype stored
426 return QMetaType::fromName(name: className()); // try lookup by name in that case
427 }
428}

605// Returns \c true if the method defined by the given meta-object&meta-method
606// matches the given name, argument count and argument types, otherwise
607// returns \c false.
608bool QMetaObjectPrivate::methodMatch(const QMetaObject *m, const QMetaMethod &method,
609 const QByteArray &name, int argc,
610 const QArgumentType *types)
611{
612 const QMetaMethod::Data &data = method.data;
613 auto priv = QMetaMethodPrivate::get(q: &method);
614 if (priv->parameterCount() != argc)
615 return false;
616
617 if (stringData(mo: m, index: data.name()) != name)
618 return false;
619
620 const QtPrivate::QMetaTypeInterface * const *ifaces = priv->parameterMetaTypeInterfaces();
621 int paramsIndex = data.parameters() + 1;
622 for (int i = 0; i < argc; ++i) {
623 uint typeInfo = m->d.data[paramsIndex + i];
624 if (int id = types[i].type()) {
625 if (id == QMetaType(ifaces[i]).id())
626 continue;
627 if (id != typeFromTypeInfo(mo: m, typeInfo))
628 return false;
629 } else {
630 if (types[i].name() == QMetaType(ifaces[i]).name())
631 continue;
632 if (types[i].name() != typeNameFromTypeInfo(mo: m, typeInfo))
633 return false;
634 }
635 }
636
637 return true;
638}

658/**
659* \internal
660* helper function for indexOf{Method,Slot,Signal}, returns the relative index of the method within
661* the baseObject
662* \a MethodType might be MethodSignal or MethodSlot, or \nullptr to match everything.
663*/
664template<int MethodType>
665inline int QMetaObjectPrivate::indexOfMethodRelative(const QMetaObject **baseObject,
666 const QByteArray &name, int argc,
667 const QArgumentType *types)
668{
669 for (const QMetaObject *m = *baseObject; m; m = m->d.superdata) {
670 Q_ASSERT(priv(m->d.data)->revision >= 7);
671 int i = (MethodType == MethodSignal)
672 ? (priv(data: m->d.data)->signalCount - 1) : (priv(data: m->d.data)->methodCount - 1);
673 const int end = (MethodType == MethodSlot)
674 ? (priv(data: m->d.data)->signalCount) : 0;
675
676 for (; i >= end; --i) {
677 auto data = QMetaMethod::fromRelativeMethodIndex(mobj: m, index: i);
678 if (methodMatch(m, method: data, name, argc, types)) {
679 *baseObject = m;
680 return i;
681 }
682 }
683 }
684 return -1;
685}

727// Parses a string of comma-separated types into QArgumentTypes.
728// No normalization of the type names is performed.
729static void argumentTypesFromString(const char *str, const char *end,
730 QArgumentTypeArray &types)
731{
732 Q_ASSERT(str <= end);
733 while (str != end) {
734 if (!types.isEmpty())
735 ++str; // Skip comma
736 const char *begin = str;
737 int level = 0;
738 while (str != end && (level > 0 || *str != ',')) {
739 if (*str == '<')
740 ++level;
741 else if (*str == '>')
742 --level;
743 ++str;
744 }
745 QByteArray argType(begin, str - begin);
746 argType.replace(before: "QVector<", after: "QList<");
747 types += QArgumentType(std::move(argType));
748 }
749}

751// Given a method \a signature (e.g. "foo(int,double)"), this function
752// populates the argument \a types array and returns the method name.
753QByteArray QMetaObjectPrivate::decodeMethodSignature(
754 const char *signature, QArgumentTypeArray &types)
755{
756 Q_ASSERT(signature != nullptr);
757 const char *lparens = strchr(s: signature, c: '(');

1329
1330 Qt uses normalized signatures to decide whether two given signals
1331 and slots are compatible. Normalization reduces whitespace to a
1332 minimum, moves 'const' to the front where appropriate, removes
1333 'const' from value types and replaces const references with
1334 values.

1363 }
