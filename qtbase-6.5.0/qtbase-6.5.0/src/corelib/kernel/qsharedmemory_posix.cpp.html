30int QSharedMemoryPrivate::handle()
31{
32 // don't allow making handles on empty keys
33 const QString safeKey = makePlatformSafeKey(key);
34 if (safeKey.isEmpty()) {
35 errorString = QSharedMemory::tr("%1: key is empty").arg("QSharedMemory::handle"_L1);
36 error = QSharedMemory::KeyError;
37 return 0;
38 }
39
40 return 1;
41}
42
43bool QSharedMemoryPrivate::cleanHandle()
44{
45 qt_safe_close(hand);
46 hand = -1;
47
48 return true;
49}

51bool QSharedMemoryPrivate::create(qsizetype size)
52{
53 if (!handle())
54 return false;
55
56 const QByteArray shmName = QFile::encodeName(makePlatformSafeKey(key));
57
58 int fd;
59#ifdef O_CLOEXEC
60 // First try with O_CLOEXEC flag, if that fails, fall back to normal flags
61 EINTR_LOOP(fd, ::shm_open(shmName.constData(), O_RDWR | O_CREAT | O_EXCL | O_CLOEXEC, 0600));
62 if (fd == -1)
63 EINTR_LOOP(fd, ::shm_open(shmName.constData(), O_RDWR | O_CREAT | O_EXCL, 0600));
64#else
65 EINTR_LOOP(fd, ::shm_open(shmName.constData(), O_RDWR | O_CREAT | O_EXCL, 0600));
66#endif
67 if (fd == -1) {
68 const int errorNumber = errno;
69 const auto function = "QSharedMemory::attach (shm_open)"_L1;
70 switch (errorNumber) {
71 case EINVAL:
72 errorString = QSharedMemory::tr("%1: bad name").arg(function);
73 error = QSharedMemory::KeyError;
74 break;
75 default:
76 setErrorString(function);
77 }
78 return false;
79 }
80
81 // the size may only be set once
82 int ret;
83 EINTR_LOOP(ret, QT_FTRUNCATE(fd, size));
84 if (ret == -1) {
85 setErrorString("QSharedMemory::create (ftruncate)"_L1);
86 qt_safe_close(fd);
87 return false;
88 }
89
90 qt_safe_close(fd);
91
92 return true;
93}

95bool QSharedMemoryPrivate::attach(QSharedMemory::AccessMode mode)
96{
97 const QByteArray shmName = QFile::encodeName(makePlatformSafeKey(key));
98
99 const int oflag = (mode == QSharedMemory::ReadOnly ? O_RDONLY : O_RDWR);
100 const mode_t omode = (mode == QSharedMemory::ReadOnly ? 0400 : 0600);
101
102#ifdef O_CLOEXEC
103 // First try with O_CLOEXEC flag, if that fails, fall back to normal flags
104 EINTR_LOOP(hand, ::shm_open(shmName.constData(), oflag | O_CLOEXEC, omode));
105 if (hand == -1)
106 EINTR_LOOP(hand, ::shm_open(shmName.constData(), oflag, omode));
107#else
108 EINTR_LOOP(hand, ::shm_open(shmName.constData(), oflag, omode));
109#endif
110 if (hand == -1) {
111 const int errorNumber = errno;
112 const auto function = "QSharedMemory::attach (shm_open)"_L1;
113 switch (errorNumber) {
114 case EINVAL:
115 errorString = QSharedMemory::tr("%1: bad name").arg(function);
116 error = QSharedMemory::KeyError;
117 break;
118 default:
119 setErrorString(function);
120 }
121 hand = -1;
122 return false;
123 }
124
125 // grab the size
126 QT_STATBUF st;
127 if (QT_FSTAT(hand, &st) == -1) {
128 setErrorString("QSharedMemory::attach (fstat)"_L1);
129 cleanHandle();
130 return false;
131 }
132 size = qsizetype(st.st_size);
133
134 // grab the memory
135 const int mprot = (mode == QSharedMemory::ReadOnly ? PROT_READ : PROT_READ | PROT_WRITE);
136 memory = QT_MMAP(0, size_t(size), mprot, MAP_SHARED, hand, 0);
137 if (memory == MAP_FAILED || !memory) {
138 setErrorString("QSharedMemory::attach (mmap)"_L1);
139 cleanHandle();
140 memory = 0;
141 size = 0;
142 return false;
143 }
144
145#ifdef F_ADD_SEALS
146 // Make sure the shared memory region will not shrink
147 // otherwise someone could cause SIGBUS on us.
148 // (see http://lwn.net/Articles/594919/)
149 fcntl(hand, F_ADD_SEALS, F_SEAL_SHRINK);
150#endif
151
152 return true;
153}

155bool QSharedMemoryPrivate::detach()
156{
157 // detach from the memory segment
158 if (::munmap(memory, size_t(size)) == -1) {
159 setErrorString("QSharedMemory::detach (munmap)"_L1);
160 return false;
161 }
162 memory = 0;
163 size = 0;
164
165#ifdef Q_OS_QNX
166 // On QNX the st_nlink field of struct stat contains the number of
167 // active shm_open() connections to the shared memory file, so we
168 // can use it to automatically clean up the file once the last
169 // user has detached from it.
170
171 // get the number of current attachments
172 int shm_nattch = 0;
173 QT_STATBUF st;
174 if (QT_FSTAT(hand, &st) == 0) {
175 // subtract 2 from linkcount: one for our own open and one for the dir entry
176 shm_nattch = st.st_nlink - 2;
177 }
178
179 cleanHandle();
180
181 // if there are no attachments then unlink the shared memory
182 if (shm_nattch == 0) {
183 const QByteArray shmName = QFile::encodeName(makePlatformSafeKey(key));
184 if (::shm_unlink(shmName.constData()) == -1 && errno != ENOENT)
185 setErrorString("QSharedMemory::detach (shm_unlink)"_L1);
186 }
187#else
188 // On non-QNX systems (tested Linux and Haiku), the st_nlink field is always 1,
189 // so we'll simply leak the shared memory files.
190 cleanHandle();
191#endif
192
193 return true;
194}
