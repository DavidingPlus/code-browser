71// Internal operator functions for timespecs
72constexpr inline timespec &normalizedTimespec(timespec &t)
73{
74 while (t.tv_nsec >= 1000000000) {
75 ++t.tv_sec;
76 t.tv_nsec -= 1000000000;
77 }
78 while (t.tv_nsec < 0) {
79 --t.tv_sec;
80 t.tv_nsec += 1000000000;
81 }
82 return t;
83}
84constexpr inline bool operator<(const timespec &t1, const timespec &t2)
85{ return t1.tv_sec < t2.tv_sec || (t1.tv_sec == t2.tv_sec && t1.tv_nsec < t2.tv_nsec); }
86constexpr inline bool operator==(const timespec &t1, const timespec &t2)
87{ return t1.tv_sec == t2.tv_sec && t1.tv_nsec == t2.tv_nsec; }
88constexpr inline bool operator!=(const timespec &t1, const timespec &t2)
89{ return !(t1 == t2); }
90constexpr inline timespec &operator+=(timespec &t1, const timespec &t2)
91{
92 t1.tv_sec += t2.tv_sec;
93 t1.tv_nsec += t2.tv_nsec;
94 return normalizedTimespec(t&: t1);
95}
96constexpr inline timespec operator+(const timespec &t1, const timespec &t2)
97{
98 timespec tmp = {};
99 tmp.tv_sec = t1.tv_sec + t2.tv_sec;
100 tmp.tv_nsec = t1.tv_nsec + t2.tv_nsec;
101 return normalizedTimespec(t&: tmp);
102}
103constexpr inline timespec operator-(const timespec &t1, const timespec &t2)
104{
105 timespec tmp = {};
106 tmp.tv_sec = t1.tv_sec - (t2.tv_sec - 1);
107 tmp.tv_nsec = t1.tv_nsec - (t2.tv_nsec + 1000000000);
108 return normalizedTimespec(t&: tmp);
109}
110constexpr inline timespec operator*(const timespec &t1, int mul)
111{
112 timespec tmp = {};
113 tmp.tv_sec = t1.tv_sec * mul;
114 tmp.tv_nsec = t1.tv_nsec * mul;
115 return normalizedTimespec(t&: tmp);
116}
117inline timeval timespecToTimeval(const timespec &ts)
118{
119 timeval tv;
120 tv.tv_sec = ts.tv_sec;
121 tv.tv_usec = ts.tv_nsec / 1000;
122 return tv;
123}

126inline void qt_ignore_sigpipe()
127{
128 // Set to ignore SIGPIPE once only.
129 Q_CONSTINIT static QBasicAtomicInt atom = Q_BASIC_ATOMIC_INITIALIZER(0);
130 if (!atom.loadRelaxed()) {
131 // More than one thread could turn off SIGPIPE at the same time
132 // But that's acceptable because they all would be doing the same
133 // action
134 struct sigaction noaction;
135 memset(s: &noaction, c: 0, n: sizeof(noaction));
136 noaction.sa_handler = SIG_IGN;
137 ::sigaction(SIGPIPE, act: &noaction, oact: nullptr);
138 atom.storeRelaxed(newValue: 1);
139 }
140}

150// don't call QT_OPEN or ::open
151// call qt_safe_open
152static inline int qt_safe_open(const char *pathname, int flags, mode_t mode = 0777)
153{
154#ifdef O_CLOEXEC
155 flags |= O_CLOEXEC;
156#endif
157 int fd;
158 EINTR_LOOP(fd, QT_OPEN(pathname, flags, mode));
159
160#ifndef O_CLOEXEC
161 if (fd != -1)
162 ::fcntl(fd, F_SETFD, FD_CLOEXEC);
163#endif
164
165 return fd;
166}
167#undef QT_OPEN
168#define QT_OPEN qt_safe_open

170#ifndef Q_OS_VXWORKS // no POSIX pipes in VxWorks
171// don't call ::pipe
172// call qt_safe_pipe
173static inline int qt_safe_pipe(int pipefd[2], int flags = 0)
174{
175 Q_ASSERT((flags & ~O_NONBLOCK) == 0);
176
177#ifdef QT_THREADSAFE_CLOEXEC
178 // use pipe2
179 flags |= O_CLOEXEC;
180 return ::pipe2(pipefd, flags); // pipe2 is documented not to return EINTR
181#else
182 int ret = ::pipe(pipedes: pipefd);
183 if (ret == -1)
184 return -1;
185
186 ::fcntl(fd: pipefd[0], F_SETFD, FD_CLOEXEC);
187 ::fcntl(fd: pipefd[1], F_SETFD, FD_CLOEXEC);
188
189 // set non-block too?
190 if (flags & O_NONBLOCK) {
191 ::fcntl(fd: pipefd[0], F_SETFL, ::fcntl(fd: pipefd[0], F_GETFL) | O_NONBLOCK);
192 ::fcntl(fd: pipefd[1], F_SETFL, ::fcntl(fd: pipefd[1], F_GETFL) | O_NONBLOCK);
193 }
194
195 return 0;
196#endif
197}
198
199#endif // Q_OS_VXWORKS

201// don't call dup or fcntl(F_DUPFD)
202static inline int qt_safe_dup(int oldfd, int atleast = 0, int flags = FD_CLOEXEC)
203{
204 Q_ASSERT(flags == FD_CLOEXEC || flags == 0);
205
206#ifdef F_DUPFD_CLOEXEC
207 int cmd = F_DUPFD;
208 if (flags & FD_CLOEXEC)
209 cmd = F_DUPFD_CLOEXEC;
210 return ::fcntl(fd: oldfd, cmd: cmd, atleast);
211#else
212 // use F_DUPFD
213 int ret = ::fcntl(oldfd, F_DUPFD, atleast);
214
215 if (flags && ret != -1)
216 ::fcntl(ret, F_SETFD, flags);
217 return ret;
218#endif
219}
220
221// don't call dup2
222// call qt_safe_dup2
223static inline int qt_safe_dup2(int oldfd, int newfd, int flags = FD_CLOEXEC)
224{
225 Q_ASSERT(flags == FD_CLOEXEC || flags == 0);
226
227 int ret;
228#ifdef QT_THREADSAFE_CLOEXEC
229 // use dup3
230 EINTR_LOOP(ret, ::dup3(oldfd, newfd, flags ? O_CLOEXEC : 0));
231 return ret;
232#else
233 EINTR_LOOP(ret, ::dup2(oldfd, newfd));
234 if (ret == -1)
235 return -1;
236
237 if (flags)
238 ::fcntl(fd: newfd, F_SETFD, flags);
239 return 0;
240#endif
241}

333static inline int qt_poll_msecs(struct pollfd *fds, nfds_t nfds, int timeout)
334{
335 timespec ts, *pts = nullptr;
336
337 if (timeout >= 0) {
338 ts.tv_sec = timeout / 1000;
339 ts.tv_nsec = (timeout % 1000) * 1000 * 1000;
340 pts = &ts;
341 }
342
343 return qt_safe_poll(fds, nfds, timeout_ts: pts);
344}
