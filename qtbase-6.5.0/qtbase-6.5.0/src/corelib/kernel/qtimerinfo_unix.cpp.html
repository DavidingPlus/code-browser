41 } else {

43 previousTime.tv_sec = previousTime.tv_nsec = 0;
44 previousTicks = 0;
45 ticksPerSecond = 0;
46 msPerTick = 0;
47 }
48#endif
49
50 firstTimerInfo = nullptr;
51}

60timespec qAbsTimespec(const timespec &t)
61{
62 timespec tmp = t;
63 if (tmp.tv_sec < 0) {
64 tmp.tv_sec = -tmp.tv_sec - 1;
65 tmp.tv_nsec -= 1000000000;
66 }
67 if (tmp.tv_sec == 0 && tmp.tv_nsec < 0) {
68 tmp.tv_nsec = -tmp.tv_nsec;
69 }
70 return normalizedTimespec(t&: tmp);
71}

73/*
74 Returns \c true if the real time clock has changed by more than 10%
75 relative to the processor time since the last time this function was
76 called. This presumably means that the system time has been changed.
77
78 If /a delta is nonzero, delta is set to our best guess at how much the system clock was changed.
79*/
80bool QTimerInfoList::timeChanged(timespec *delta)
81{
82 struct tms unused;
83 clock_t currentTicks = times(buffer: &unused);
84
85 clock_t elapsedTicks = currentTicks - previousTicks;
86 timespec elapsedTime = currentTime - previousTime;
87
88 timespec elapsedTimeTicks;
89 elapsedTimeTicks.tv_sec = elapsedTicks / ticksPerSecond;
90 elapsedTimeTicks.tv_nsec = (((elapsedTicks * 1000) / ticksPerSecond) % 1000) * 1000 * 1000;
91
92 timespec dummy;
93 if (!delta)
94 delta = &dummy;
95 *delta = elapsedTime - elapsedTimeTicks;
96
97 previousTicks = currentTicks;
98 previousTime = currentTime;
99
100 // If tick drift is more than 10% off compared to realtime, we assume that the clock has
101 // been set. Of course, we have to allow for the tick granularity as well.
102 timespec tickGranularity;
103 tickGranularity.tv_sec = 0;
104 tickGranularity.tv_nsec = msPerTick * 1000 * 1000;
105 return elapsedTimeTicks < ((qAbsTimespec(t: *delta) - tickGranularity) * 10);
106}

111void QTimerInfoList::timerRepair(const timespec &diff)
112{
113 // repair all timers
114 for (int i = 0; i < size(); ++i) {
115 QTimerInfo *t = at(i);
116 t->timeout = t->timeout + diff;
117 }
118}

140void QTimerInfoList::timerInsert(QTimerInfo *ti)

164static constexpr timespec roundToMillisecond(timespec val)
165{
166 // always round up
167 // worst case scenario is that the first trigger of a 1-ms timer is 0.999 ms late
168
169 int ns = val.tv_nsec % (1000 * 1000);
170 if (ns)
171 val.tv_nsec += 1000 * 1000 - ns;
172 return normalizedTimespec(t&: val);
173}
174static_assert(roundToMillisecond(val: {.tv_sec: 0, .tv_nsec: 0}) == timespec{.tv_sec: 0, .tv_nsec: 0});
175static_assert(roundToMillisecond(val: {.tv_sec: 0, .tv_nsec: 1}) == timespec{.tv_sec: 0, .tv_nsec: 1'000'000});
176static_assert(roundToMillisecond(val: {.tv_sec: 0, .tv_nsec: 999'999}) == timespec{.tv_sec: 0, .tv_nsec: 1'000'000});
177static_assert(roundToMillisecond(val: {.tv_sec: 0, .tv_nsec: 1'000'000}) == timespec{.tv_sec: 0, .tv_nsec: 1'000'000});
178static_assert(roundToMillisecond(val: {.tv_sec: 0, .tv_nsec: 999'999'999}) == timespec{.tv_sec: 1, .tv_nsec: 0});
179static_assert(roundToMillisecond(val: {.tv_sec: 1, .tv_nsec: 0}) == timespec{.tv_sec: 1, .tv_nsec: 0});

197static void calculateCoarseTimerTimeout(QTimerInfo *t, timespec currentTime)
198{
199 // The coarse timer works like this:
200 // - interval under 40 ms: round to even
201 // - between 40 and 99 ms: round to multiple of 4
202 // - otherwise: try to wake up at a multiple of 25 ms, with a maximum error of 5%
203 //
204 // We try to wake up at the following second-fraction, in order of preference:
205 // 0 ms
206 // 500 ms
207 // 250 ms or 750 ms
208 // 200, 400, 600, 800 ms
209 // other multiples of 100
210 // other multiples of 50
211 // other multiples of 25
212 //
213 // The objective is to make most timers wake up at the same time, thereby reducing CPU wakeups.
214
215 uint interval = uint(t->interval);
216 Q_ASSERT(interval >= 20);
217 // Calculate how much we can round and still keep within 5% error
218 uint absMaxRounding = interval / 20;
219
220 using namespace std::chrono;
221 uint msec = duration_cast<milliseconds>(d: nanoseconds{t->timeout.tv_nsec}).count();
222
223 if (interval < 100 && interval != 25 && interval != 50 && interval != 75) {
224 // special mode for timers of less than 100 ms
225 if (interval < 50) {
226 // round to even
227 // round towards multiples of 50 ms
228 bool roundUp = (msec % 50) >= 25;
229 msec >>= 1;
230 msec |= uint(roundUp);
231 msec <<= 1;
232 } else {
233 // round to multiple of 4
234 // round towards multiples of 100 ms
235 bool roundUp = (msec % 100) >= 50;
236 msec >>= 2;
237 msec |= uint(roundUp);
238 msec <<= 2;
239 }
240 } else {
241 uint min = qMax<int>(a: 0, b: msec - absMaxRounding);
242 uint max = qMin(a: 1000u, b: msec + absMaxRounding);
243
244 // find the boundary that we want, according to the rules above
245 // extra rules:
246 // 1) whatever the interval, we'll take any round-to-the-second timeout
247 if (min == 0) {
248 msec = 0;
249 goto recalculate;
250 } else if (max == 1000) {
251 msec = 1000;
252 goto recalculate;
253 }
254
255 uint wantedBoundaryMultiple;
256
257 // 2) if the interval is a multiple of 500 ms and > 5000 ms, we'll always round
258 // towards a round-to-the-second
259 // 3) if the interval is a multiple of 500 ms, we'll round towards the nearest
260 // multiple of 500 ms
261 if ((interval % 500) == 0) {
262 if (interval >= 5000) {
263 msec = msec >= 500 ? max : min;
264 goto recalculate;
265 } else {
266 wantedBoundaryMultiple = 500;
267 }
268 } else if ((interval % 50) == 0) {
269 // 4) same for multiples of 250, 200, 100, 50
270 uint mult50 = interval / 50;
271 if ((mult50 % 4) == 0) {
272 // multiple of 200
273 wantedBoundaryMultiple = 200;
274 } else if ((mult50 % 2) == 0) {
275 // multiple of 100
276 wantedBoundaryMultiple = 100;
277 } else if ((mult50 % 5) == 0) {
278 // multiple of 250
279 wantedBoundaryMultiple = 250;
280 } else {
281 // multiple of 50
282 wantedBoundaryMultiple = 50;
283 }
284 } else {
285 wantedBoundaryMultiple = 25;
286 }
287
288 uint base = msec / wantedBoundaryMultiple * wantedBoundaryMultiple;
289 uint middlepoint = base + wantedBoundaryMultiple / 2;
290 if (msec < middlepoint)
291 msec = qMax(a: base, b: min);
292 else
293 msec = qMin(a: base + wantedBoundaryMultiple, b: max);
294 }
295
296recalculate:
297 if (msec == 1000u) {
298 ++t->timeout.tv_sec;
299 t->timeout.tv_nsec = 0;
300 } else {
301 t->timeout.tv_nsec = nanoseconds{milliseconds{msec}}.count();
302 }
303
304 if (t->timeout < currentTime)
305 t->timeout += interval;
306}

308static void calculateNextTimeout(QTimerInfo *t, timespec currentTime)
309{
310 switch (t->timerType) {
311 case Qt::PreciseTimer:
312 case Qt::CoarseTimer:
313 t->timeout += t->interval;
314 if (t->timeout < currentTime) {
315 t->timeout = currentTime;
316 t->timeout += t->interval;
317 }
318#ifdef QTIMERINFO_DEBUG
319 t->expected += t->interval;
320 if (t->expected < currentTime) {
321 t->expected = currentTime;
322 t->expected += t->interval;
323 }
324#endif
325 if (t->timerType == Qt::CoarseTimer)
326 calculateCoarseTimerTimeout(t, currentTime);
327 return;
328
329 case Qt::VeryCoarseTimer:
330 // we don't need to take care of the microsecond component of t->interval
331 t->timeout.tv_sec += t->interval;
332 if (t->timeout.tv_sec <= currentTime.tv_sec)
333 t->timeout.tv_sec = currentTime.tv_sec + t->interval;
334#ifdef QTIMERINFO_DEBUG
335 t->expected.tv_sec += t->interval;
336 if (t->expected.tv_sec <= currentTime.tv_sec)
337 t->expected.tv_sec = currentTime.tv_sec + t->interval;
338#endif
339 return;
340 }
341
342#ifdef QTIMERINFO_DEBUG
343 if (t->timerType != Qt::PreciseTimer)
344 qDebug() << "timer" << t->timerType << Qt::hex << t->id << Qt::dec << "interval" << t->interval
345 << "originally expected at" << t->expected << "will fire at" << t->timeout
346 << "or" << (t->timeout - t->expected) << "s late";
347#endif
348}

350/*
351 Returns the time to wait for the next timer, or null if no timers
352 are waiting.
353*/
354bool QTimerInfoList::timerWait(timespec &tm)
355{
356 timespec currentTime = updateCurrentTime();
357 repairTimersIfNeeded();
358
359 // Find first waiting timer not already active
360 QTimerInfo *t = nullptr;
361 for (QTimerInfoList::const_iterator it = constBegin(); it != constEnd(); ++it) {
362 if (!(*it)->activateRef) {
363 t = *it;
364 break;
365 }
366 }
367
368 if (!t)
369 return false;
370
371 if (currentTime < t->timeout) {
372 // time to wait
373 tm = roundToMillisecond(val: t->timeout - currentTime);
374 } else {
375 // no time to wait
376 tm.tv_sec = 0;
377 tm.tv_nsec = 0;
378 }
379
380 return true;
381}

383/*
384 Returns the timer's remaining time in milliseconds with the given timerId.
385 If the timer id is not found in the list, the returned value will be -1.
386 If the timer is overdue, the returned value will be 0.
387*/
388qint64 QTimerInfoList::timerRemainingTime(int timerId)
389{
390 timespec currentTime = updateCurrentTime();
391 repairTimersIfNeeded();
392 timespec tm = {.tv_sec: 0, .tv_nsec: 0};
393
394 for (const auto *t : std::as_const(t&: *this)) {
395 if (t->id == timerId) {
396 if (currentTime < t->timeout) {
397 // time to wait
398 tm = roundToMillisecond(val: t->timeout - currentTime);
399 using namespace std::chrono;
400 const auto dur = duration_cast<milliseconds>(d: seconds{tm.tv_sec} + nanoseconds{tm.tv_nsec});
401 return dur.count();
402 } else {
403 return 0;
404 }
405 }
406 }
407
408#ifndef QT_NO_DEBUG
409 qWarning("QTimerInfoList::timerRemainingTime: timer id %i not found", timerId);
410#endif
411
412 return -1;
413}

415void QTimerInfoList::registerTimer(int timerId, qint64 interval, Qt::TimerType timerType, QObject *object)
416{
417 QTimerInfo *t = new QTimerInfo;
418 t->id = timerId;
419 t->interval = interval;
420 t->timerType = timerType;
421 t->obj = object;
422 t->activateRef = nullptr;
423
424 timespec expected = updateCurrentTime() + interval;
425
426 switch (timerType) {
427 case Qt::PreciseTimer:
428 // high precision timer is based on millisecond precision
429 // so no adjustment is necessary
430 t->timeout = expected;
431 break;
432
433 case Qt::CoarseTimer:
434 // this timer has up to 5% coarseness
435 // so our boundaries are 20 ms and 20 s
436 // below 20 ms, 5% inaccuracy is below 1 ms, so we convert to high precision
437 // above 20 s, 5% inaccuracy is above 1 s, so we convert to VeryCoarseTimer
438 if (interval >= 20000) {
439 t->timerType = Qt::VeryCoarseTimer;
440 } else {
441 t->timeout = expected;
442 if (interval <= 20) {
443 t->timerType = Qt::PreciseTimer;
444 // no adjustment is necessary
445 } else if (interval <= 20000) {
446 calculateCoarseTimerTimeout(t, currentTime);
447 }
448 break;
449 }
450 Q_FALLTHROUGH();
451 case Qt::VeryCoarseTimer:
452 // the very coarse timer is based on full second precision,
453 // so we keep the interval in seconds (round to closest second)
454 t->interval /= 500;
455 t->interval += 1;
456 t->interval >>= 1;
457 t->timeout.tv_sec = currentTime.tv_sec + t->interval;
458 t->timeout.tv_nsec = 0;
459
460 // if we're past the half-second mark, increase the timeout again
461 using namespace std::chrono;
462 if (currentTime.tv_nsec > nanoseconds{500ms}.count())
463 ++t->timeout.tv_sec;
464 }
465
466 timerInsert(ti: t);
467
468#ifdef QTIMERINFO_DEBUG
469 t->expected = expected;
470 t->cumulativeError = 0;
471 t->count = 0;
472 if (t->timerType != Qt::PreciseTimer)
473 qDebug() << "timer" << t->timerType << Qt::hex <<t->id << Qt::dec << "interval" << t->interval << "expected at"
474 << t->expected << "will fire first at" << t->timeout;
475#endif
476}

478bool QTimerInfoList::unregisterTimer(int timerId)
479{
480 // set timer inactive
481 for (int i = 0; i < size(); ++i) {
482 QTimerInfo *t = at(i);
483 if (t->id == timerId) {
484 // found it
485 removeAt(i);
486 if (t == firstTimerInfo)
487 firstTimerInfo = nullptr;
488 if (t->activateRef)
489 *(t->activateRef) = nullptr;
490 delete t;
491 return true;
492 }
493 }
494 // id not found
495 return false;
496}

498bool QTimerInfoList::unregisterTimers(QObject *object)
499{
500 if (isEmpty())
501 return false;
502 for (int i = 0; i < size(); ++i) {
503 QTimerInfo *t = at(i);
504 if (t->obj == object) {
505 // object found
506 removeAt(i);
507 if (t == firstTimerInfo)
508 firstTimerInfo = nullptr;
509 if (t->activateRef)
510 *(t->activateRef) = nullptr;
511 delete t;
512 // move back one so that we don't skip the new current item
513 --i;
514 }
515 }
516 return true;
517}

535/*
536 Activate pending timers, returning how many where activated.
537*/
538int QTimerInfoList::activateTimers()
539{
540 if (qt_disable_lowpriority_timers || isEmpty())
541 return 0; // nothing to do
542
543 int n_act = 0, maxCount = 0;
544 firstTimerInfo = nullptr;
545
546 timespec currentTime = updateCurrentTime();
547 // qDebug() << "Thread" << QThread::currentThreadId() << "woken up at" << currentTime;
548 repairTimersIfNeeded();
549
550
551 // Find out how many timer have expired
552 for (QTimerInfoList::const_iterator it = constBegin(); it != constEnd(); ++it) {
553 if (currentTime < (*it)->timeout)
554 break;
555 maxCount++;
556 }
557
558 //fire the timers.
559 while (maxCount--) {
560 if (isEmpty())
561 break;
562
563 QTimerInfo *currentTimerInfo = constFirst();
564 if (currentTime < currentTimerInfo->timeout)
565 break; // no timer has expired
566
567 if (!firstTimerInfo) {
568 firstTimerInfo = currentTimerInfo;
569 } else if (firstTimerInfo == currentTimerInfo) {
570 // avoid sending the same timer multiple times
571 break;
572 } else if (currentTimerInfo->interval < firstTimerInfo->interval
573 || currentTimerInfo->interval == firstTimerInfo->interval) {
574 firstTimerInfo = currentTimerInfo;
575 }
576
577 // remove from list
578 removeFirst();
579
580#ifdef QTIMERINFO_DEBUG
581 float diff;
582 if (currentTime < currentTimerInfo->expected) {
583 // early
584 timeval early = currentTimerInfo->expected - currentTime;
585 diff = -(early.tv_sec + early.tv_usec / 1000000.0);
586 } else {
587 timeval late = currentTime - currentTimerInfo->expected;
588 diff = late.tv_sec + late.tv_usec / 1000000.0;
589 }
590 currentTimerInfo->cumulativeError += diff;
591 ++currentTimerInfo->count;
592 if (currentTimerInfo->timerType != Qt::PreciseTimer)
593 qDebug() << "timer" << currentTimerInfo->timerType << Qt::hex << currentTimerInfo->id << Qt::dec << "interval"
594 << currentTimerInfo->interval << "firing at" << currentTime
595 << "(orig" << currentTimerInfo->expected << "scheduled at" << currentTimerInfo->timeout
596 << ") off by" << diff << "activation" << currentTimerInfo->count
597 << "avg error" << (currentTimerInfo->cumulativeError / currentTimerInfo->count);
598#endif
599
600 // determine next timeout time
601 calculateNextTimeout(t: currentTimerInfo, currentTime);
602
603 // reinsert timer
604 timerInsert(ti: currentTimerInfo);
605 if (currentTimerInfo->interval > 0)
606 n_act++;
607
608 // Send event, but don't allow it to recurse:
609 if (!currentTimerInfo->activateRef) {
610 currentTimerInfo->activateRef = &currentTimerInfo;
611
612 QTimerEvent e(currentTimerInfo->id);
613 QCoreApplication::sendEvent(receiver: currentTimerInfo->obj, event: &e);
614
615 // Storing currentTimerInfo's address in its activateRef allows the
616 // handling of that event to clear this local variable on deletion
617 // of the object it points to - if it didn't, clear activateRef:
618 if (currentTimerInfo)
619 currentTimerInfo->activateRef = nullptr;
620 }
621 }
622
623 firstTimerInfo = nullptr;
624 // qDebug() << "Thread" << QThread::currentThreadId() << "activated" << n_act << "timers";
625 return n_act;
626}
