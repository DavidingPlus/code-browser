205void QCoreApplicationPrivate::processCommandLineArguments()
206{
207 int j = argc ? 1 : 0;
208 for (int i = 1; i < argc; ++i) {
209 if (!argv[i])
210 continue;
211 if (*argv[i] != '-') {
212 argv[j++] = argv[i];
213 continue;
214 }
215 const char *arg = argv[i];
216 if (arg[1] == '-') // startsWith("--")
217 ++arg;
218 if (strncmp(s1: arg, s2: "-qmljsdebugger=", n: 15) == 0) {
219 qmljs_debug_arguments = QString::fromLocal8Bit(ba: arg + 15);
220 } else if (strcmp(s1: arg, s2: "-qmljsdebugger") == 0 && i < argc - 1) {
221 ++i;
222 qmljs_debug_arguments = QString::fromLocal8Bit(ba: argv[i]);
223 } else {
224 argv[j++] = argv[i];
225 }
226 }
227
228 if (j < argc) {
229 argv[j] = nullptr;
230 argc = j;
231 }
232}

247/*!
248 \internal
249
250 Adds a global routine that will be called from the QCoreApplication
251 constructor. The public API is Q_COREAPP_STARTUP_FUNCTION.
252*/
253void qAddPreRoutine(QtStartUpFunction p)
254{
255 QStartUpFuncList *list = preRList();
256 if (!list)
257 return;
258
259 if (preRoutinesCalled) {
260 Q_ASSERT(QCoreApplication::instance());
261 p();
262 }
263
264 // Due to C++11 parallel dynamic initialization, this can be called
265 // from multiple threads.
266 const auto locker = qt_scoped_lock(mutex&: globalRoutinesMutex);
267 list->prepend(t: p); // in case QCoreApplication is re-created, see qt_call_pre_routines
268}
269
270void qAddPostRoutine(QtCleanUpFunction p)
271{
272 QVFuncList *list = postRList();
273 if (!list)
274 return;
275 const auto locker = qt_scoped_lock(mutex&: globalRoutinesMutex);
276 list->prepend(t: p);
277}
278
279void qRemovePostRoutine(QtCleanUpFunction p)
280{
281 QVFuncList *list = postRList();
282 if (!list)
283 return;
284 const auto locker = qt_scoped_lock(mutex&: globalRoutinesMutex);
285 list->removeAll(t: p);
286}

288static void qt_call_pre_routines()
289{
290 // After will be allowed invoke QtStartUpFunction when calling qAddPreRoutine
291 preRoutinesCalled = true;
292
293 if (!preRList.exists())
294 return;
295
296 QVFuncList list;
297 {
298 const auto locker = qt_scoped_lock(mutex&: globalRoutinesMutex);
299 // Unlike qt_call_post_routines, we don't empty the list, because
300 // Q_COREAPP_STARTUP_FUNCTION is a macro, so the user expects
301 // the function to be executed every time QCoreApplication is created.
302 list = *preRList;
303 }
304 for (QtCleanUpFunction f : std::as_const(t&: list))
305 f();
306}
307
308void Q_CORE_EXPORT qt_call_post_routines()
309{
310 if (!postRList.exists())
311 return;
312
313 forever {
314 QVFuncList list;
315 {
316 // extract the current list and make the stored list empty
317 const auto locker = qt_scoped_lock(mutex&: globalRoutinesMutex);
318 qSwap(value1&: *postRList, value2&: list);
319 }
320
321 if (list.isEmpty())
322 break;
323 for (QtCleanUpFunction f : std::as_const(t&: list))
324 f();
325 }
326}

386// Check whether the command line arguments passed to QCoreApplication
387// match those passed into main(), to see if the user has modified them
388// before passing them on to us. We do this by comparing to the global
389// __argv/__argc (MS extension). Deep comparison is required since

391static inline bool isArgvModified(int argc, char **argv)
392{
393 if (__argc != argc || !__argv /* wmain() */)
394 return true;
395 if (__argv == argv)
396 return false;
397 for (int a = 0; a < argc; ++a) {
398 if (argv[a] != __argv[a] && strcmp(argv[a], __argv[a]))
399 return true;
400 }
401 return false;
402}

406 for (int a = 0; a < argc; ++a) {
407 if (!strcmp(argv[a], needle))
408 return true;
409 }
410 return false;

476void QCoreApplicationPrivate::cleanupThreadData()
477{
478 auto thisThreadData = threadData.loadRelaxed();
479
480 if (thisThreadData && !threadData_clean) {
481#if QT_CONFIG(thread)
482 void *data = &thisThreadData->tls;
483 QThreadStorageData::finish((void **)data);
484#endif
485
486 // need to clear the state of the mainData, just in case a new QCoreApplication comes along.
487 const auto locker = qt_scoped_lock(thisThreadData->postEventList.mutex);
488 for (const QPostEvent &pe : std::as_const(thisThreadData->postEventList)) {
489 if (pe.event) {
490 --pe.receiver->d_func()->postedEvents;
491 pe.event->m_posted = false;
492 delete pe.event;
493 }
494 }
495 thisThreadData->postEventList.clear();
496 thisThreadData->postEventList.recursion = 0;
497 thisThreadData->quitNow = false;
498 threadData_clean = true;
499 }

568void QCoreApplicationPrivate::initConsole()
569{
570#ifdef Q_OS_WINDOWS
571 const QString env = qEnvironmentVariable("QT_WIN_DEBUG_CONSOLE");
572 if (env.isEmpty())
573 return;
574 if (env.compare(u"new"_s, Qt::CaseInsensitive) == 0) {
575 if (AllocConsole() == FALSE)
576 return;
577 consoleAllocated = true;
578 } else if (env.compare(u"attach"_s, Qt::CaseInsensitive) == 0) {
579 if (AttachConsole(ATTACH_PARENT_PROCESS) == FALSE)
580 return;
581 } else {
582 // Unknown input, don't make any decision for the user.
583 return;
584 }
585 // The std{in,out,err} handles are read-only, so we need to pass in dummies.
586 FILE *in = nullptr;
587 FILE *out = nullptr;
588 FILE *err = nullptr;
589 freopen_s(&in, "CONIN$", "r", stdin);
590 freopen_s(&out, "CONOUT$", "w", stdout);
591 freopen_s(&err, "CONOUT$", "w", stderr);
592 // However, things wouldn't work if the runtime did not preserve the pointers.
593 Q_ASSERT(in == stdin);
594 Q_ASSERT(out == stdout);
595 Q_ASSERT(err == stderr);
596#endif
597}

599void QCoreApplicationPrivate::initLocale()
600{
601#if defined(QT_BOOTSTRAPPED)
602 // Don't try to control bootstrap library locale or encoding.
603#elif defined(Q_OS_UNIX)
604 Q_CONSTINIT static bool qt_locale_initialized = false;
605 if (qt_locale_initialized)
606 return;
607 qt_locale_initialized = true;
608
609 // By default the portable "C"/POSIX locale is selected and active.
610 // Apply the locale from the environment, via setlocale(), which will
611 // read LC_ALL, LC_<category>, and LANG, in order (for each category).
612 setlocale(LC_ALL, "");
613
614 // Next, let's ensure that LC_CTYPE is UTF-8, since QStringConverter's
615 // QLocal8Bit hard-codes this, and we need to be consistent.
616# if defined(Q_OS_INTEGRITY)
617 setlocale(LC_CTYPE, "UTF-8");
618# elif defined(Q_OS_QNX)
619 // QNX has no nl_langinfo, so we can't check.
620 // FIXME: Shouldn't we still setlocale("UTF-8")?
621# elif defined(Q_OS_ANDROID) && __ANDROID_API__ < __ANDROID_API_O__
622 // Android 6 still lacks nl_langinfo(), so we can't check.
623 // FIXME: Shouldn't we still setlocale("UTF-8")?
624# else
625 const char *charEncoding = nl_langinfo(CODESET);
626 if (Q_UNLIKELY(qstricmp(charEncoding, "UTF-8") != 0 && qstricmp(charEncoding, "utf8") != 0)) {
627 const QByteArray oldLocale = setlocale(LC_ALL, nullptr);
628 QByteArray newLocale;
629 bool warnOnOverride = true;
630# if defined(Q_OS_DARWIN)
631 // Don't warn unless the char encoding has been changed from the
632 // default "C" encoding, or the user touched any of the locale
633 // environment variables to force the "C" char encoding.
634 warnOnOverride = qstrcmp(setlocale(LC_CTYPE, nullptr), "C") != 0
635 || getenv("LC_ALL") || getenv("LC_CTYPE") || getenv("LANG");
636
637 // No need to try language or region specific CTYPEs, as they
638 // all point back to the same generic UTF-8 CTYPE.
639 newLocale = setlocale(LC_CTYPE, "UTF-8");
640# else
641 newLocale = setlocale(LC_CTYPE, nullptr);
642 if (qsizetype dot = newLocale.indexOf('.'); dot != -1)
643 newLocale.truncate(dot); // remove encoding, if any
644 if (qsizetype at = newLocale.indexOf('@'); at != -1)
645 newLocale.truncate(at); // remove variant, as the old de_DE@euro
646 newLocale += ".UTF-8";
647 newLocale = setlocale(LC_CTYPE, newLocale);
648
649 // If that locale doesn't exist, try some fallbacks:
650 if (newLocale.isEmpty())
651 newLocale = setlocale(LC_CTYPE, "C.UTF-8");
652 if (newLocale.isEmpty())
653 newLocale = setlocale(LC_CTYPE, "C.utf8");
654# endif
655
656 if (newLocale.isEmpty()) {
657 // Failed to set a UTF-8 locale.
658 qWarning("Detected locale \"%s\" with character encoding \"%s\", which is not UTF-8.\n"
659 "Qt depends on a UTF-8 locale, but has failed to switch to one.\n"
660 "If this causes problems, reconfigure your locale. See the locale(1) manual\n"
661 "for more information.", oldLocale.constData(), charEncoding);
662 } else if (warnOnOverride) {
663 // Let the user know we over-rode their configuration.
664 qWarning("Detected locale \"%s\" with character encoding \"%s\", which is not UTF-8.\n"
665 "Qt depends on a UTF-8 locale, and has switched to \"%s\" instead.\n"
666 "If this causes problems, reconfigure your locale. See the locale(1) manual\n"
667 "for more information.",
668 oldLocale.constData(), charEncoding, newLocale.constData());
669 }
670 }
671# endif // Platform choice
672#endif // Unix
673}

676/*!
677 \class QCoreApplication
678 \inmodule QtCore
679 \brief The QCoreApplication class provides an event loop for Qt
680 applications without UI.
681
682 This class is used by non-GUI applications to provide their event
683 loop. For non-GUI application that uses Qt, there should be exactly
684 one QCoreApplication object. For GUI applications, see
685 QGuiApplication. For applications that use the Qt Widgets module,
686 see QApplication.
687
688 QCoreApplication contains the main event loop, where all events
689 from the operating system (e.g., timer and network events) and
690 other sources are processed and dispatched. It also handles the
691 application's initialization and finalization, as well as
692 system-wide and application-wide settings.
693
694 \section1 The Event Loop and Event Handling
695
696 The event loop is started with a call to exec(). Long-running
697 operations can call processEvents() to keep the application
698 responsive.
699
700 In general, we recommend that you create a QCoreApplication,
701 QGuiApplication or a QApplication object in your \c main()
702 function as early as possible. exec() will not return until
703 the event loop exits; e.g., when quit() is called.
704
705 Several static convenience functions are also provided. The
706 QCoreApplication object is available from instance(). Events can
707 be sent with sendEvent() or posted to an event queue with postEvent().
708 Pending events can be removed with removePostedEvents() or dispatched
709 with sendPostedEvents().
710
711 The class provides a quit() slot and an aboutToQuit() signal.
712
713 \section1 Application and Library Paths
714
715 An application has an applicationDirPath() and an
716 applicationFilePath(). Library paths (see QLibrary) can be retrieved
717 with libraryPaths() and manipulated by setLibraryPaths(), addLibraryPath(),
718 and removeLibraryPath().

720 \section1 Internationalization and Translations
721
722 Translation files can be added or removed
723 using installTranslator() and removeTranslator(). Application
724 strings can be translated using translate(). The QObject::tr()
725 function is implemented in terms of translate().

730 constructor should be accessed using the arguments() function.

732 \note QCoreApplication removes option \c -qmljsdebugger="...". It parses the

1135 Returns the value that is returned from the receiver's event

1137 any object in any thread.
1138

1146 approaches are listed below:

1391/*!
1392 Enters the main event loop and waits until exit() is called. Returns
1393 the value that was passed to exit() (which is 0 if exit() is called via
1394 quit()).
1395
1396 It is necessary to call this function to start event handling. The
1397 main event loop receives events from the window system and
1398 dispatches these to the application widgets.
1399
1400 To make your application perform idle processing (by executing a
1401 special function whenever there are no pending events), use a
1402 QTimer with 0 timeout. More advanced idle processing schemes can
1403 be achieved using processEvents().
1404
1405 We recommend that you connect clean-up code to the
1406 \l{QCoreApplication::}{aboutToQuit()} signal, instead of putting it in
1407 your application's \c{main()} function because on some platforms the
1408 exec() call may not return. For example, on Windows
1409 when the user logs off, the system terminates the process after Qt
1410 closes all top-level windows. Hence, there is no guarantee that the
1411 application will have time to exit its event loop and execute code at
1412 the end of the \c{main()} function after the exec()
1413 call.
1414
1415 \sa quit(), exit(), processEvents(), QApplication::exec()
1416*/
1417int QCoreApplication::exec()
1418{
1419 if (!QCoreApplicationPrivate::checkInstance("exec"))
1420 return -1;
1421
1422 QThreadData *threadData = self->d_func()->threadData.loadAcquire();
1423 if (threadData != QThreadData::current()) {
1424 qWarning("%s::exec: Must be called from the main thread", self->metaObject()->className());
1425 return -1;
1426 }
1427 if (!threadData->eventLoops.isEmpty()) {
1428 qWarning("QCoreApplication::exec: The event loop is already running");
1429 return -1;

1437 threadData->quitNow = false;
1438
1439 if (self)
1440 self->d_func()->execCleanup();
1441
1442 return returnCode;
1443}

1463 main event loop and returns from the call to exec(). The exec()
1464 function returns \a returnCode. If the event loop is not running,
1465 this function does nothing.
1466

1468 value indicates an error.
1469

1479 stops.

1482 from the main thread (the thread that the QCoreApplication object is
1483 processing events on). To ask the application to exit from another thread,
1484 either use QCoreApplication::quit() or instead call this function from the
1485 main thread with QMetaMethod::invokeMethod().

1488*/

1501 QEventLoop *eventLoop = data->eventLoops.at(i);

1503 }

1584/*!
1585 \since 4.3
1586
1587 Adds the event \a event, with the object \a receiver as the
1588 receiver of the event, to an event queue and returns immediately.
1589
1590 The event must be allocated on the heap since the post event queue
1591 will take ownership of the event and delete it once it has been
1592 posted. It is \e {not safe} to access the event after
1593 it has been posted.
1594
1595 When control returns to the main event loop, all events that are
1596 stored in the queue will be sent using the notify() function.
1597
1598 Events are sorted in descending \a priority order, i.e. events
1599 with a high \a priority are queued before events with a lower \a
1600 priority. The \a priority can be any integer value, i.e. between
1601 INT_MAX and INT_MIN, inclusive; see Qt::EventPriority for more
1602 details. Events with equal \a priority will be processed in the
1603 order posted.
1604
1605 \threadsafe
1606
1607 \sa sendEvent(), notify(), sendPostedEvents(), Qt::EventPriority
1608*/
1609void QCoreApplication::postEvent(QObject *receiver, QEvent *event, int priority)
1610{
1611 Q_ASSERT_X(event, "QCoreApplication::postEvent", "Unexpected null event");
1612
1613 Q_TRACE_SCOPE(QCoreApplication_postEvent, receiver, event, event->type());
1614
1615 // ### Qt 7: turn into an assert
1616 if (receiver == nullptr) {
1617 qWarning("QCoreApplication::postEvent: Unexpected null receiver");
1618 delete event;
1619 return;
1620 }
1621
1622 auto locker = QCoreApplicationPrivate::lockThreadPostEventList(receiver);
1623 if (!locker.threadData) {
1624 // posting during destruction? just delete the event to prevent a leak
1625 delete event;
1626 return;
1627 }
1628
1629 QThreadData *data = locker.threadData;
1630
1631 // if this is one of the compressible events, do compression
1632 if (receiver->d_func()->postedEvents
1633 && self && self->compressEvent(event, receiver, &data->postEventList)) {
1634 Q_TRACE(QCoreApplication_postEvent_event_compressed, receiver, event);
1635 return;
1636 }
1637
1638 if (event->type() == QEvent::DeferredDelete)
1639 receiver->d_ptr->deleteLaterCalled = true;
1640
1641 if (event->type() == QEvent::DeferredDelete && data == QThreadData::current()) {
1642 // remember the current running eventloop for DeferredDelete
1643 // events posted in the receiver's thread.
1644
1645 // Events sent by non-Qt event handlers (such as glib) may not
1646 // have the scopeLevel set correctly. The scope level makes sure that
1647 // code like this:
1648 // foo->deleteLater();
1649 // qApp->processEvents(); // without passing QEvent::DeferredDelete
1650 // will not cause "foo" to be deleted before returning to the event loop.
1651
1652 // If the scope level is 0 while loopLevel != 0, we are called from a
1653 // non-conformant code path, and our best guess is that the scope level
1654 // should be 1. (Loop level 0 is special: it means that no event loops
1655 // are running.)
1656 int loopLevel = data->loopLevel;
1657 int scopeLevel = data->scopeLevel;
1658 if (scopeLevel == 0 && loopLevel != 0)
1659 scopeLevel = 1;
1660 static_cast<QDeferredDeleteEvent *>(event)->level = loopLevel + scopeLevel;
1661 }
1662
1663 // delete the event on exceptions to protect against memory leaks till the event is
1664 // properly owned in the postEventList
1665 std::unique_ptr<QEvent> eventDeleter(event);
1666 Q_TRACE(QCoreApplication_postEvent_event_posted, receiver, event, event->type());
1667 data->postEventList.addEvent(QPostEvent(receiver, event, priority));
1668 Q_UNUSED(eventDeleter.release());
1669 event->m_posted = true;
1670 ++receiver->d_func()->postedEvents;
1671 data->canWait = false;
1672 locker.unlock();
1673
1674 QAbstractEventDispatcher* dispatcher = data->eventDispatcher.loadAcquire();
1675 if (dispatcher)
1676 dispatcher->wakeUp();
1677}

1679/*!
1680 \internal
1681 Returns \c true if \a event was compressed away (possibly deleted) and should not be added to the list.
1682*/
1683bool QCoreApplication::compressEvent(QEvent *event, QObject *receiver, QPostEventList *postedEvents)
1684{
1685 Q_ASSERT(event);
1686 Q_ASSERT(receiver);
1687 Q_ASSERT(postedEvents);
1688
1689#ifdef Q_OS_WIN
1690 // compress posted timers to this object.
1691 if (event->type() == QEvent::Timer && receiver->d_func()->postedEvents > 0) {
1692 int timerId = ((QTimerEvent *) event)->timerId();
1693 for (const QPostEvent &e : std::as_const(*postedEvents)) {
1694 if (e.receiver == receiver && e.event && e.event->type() == QEvent::Timer
1695 && ((QTimerEvent *) e.event)->timerId() == timerId) {
1696 delete event;
1697 return true;
1698 }
1699 }
1700 return false;
1701 }
1702#endif
1703
1704 if (event->type() == QEvent::DeferredDelete) {
1705 if (receiver->d_ptr->deleteLaterCalled) {
1706 // there was a previous DeferredDelete event, so we can drop the new one
1707 delete event;
1708 return true;
1709 }
1710 // deleteLaterCalled is set to true in postedEvents when queueing the very first
1711 // deferred deletion event.
1712 return false;
1713 }
1714
1715 if (event->type() == QEvent::Quit && receiver->d_func()->postedEvents > 0) {
1716 for (const QPostEvent &cur : std::as_const(*postedEvents)) {
1717 if (cur.receiver != receiver
1718 || cur.event == nullptr
1719 || cur.event->type() != event->type())
1720 continue;
1721 // found an event for this receiver
1722 delete event;
1723 return true;
1724 }
1725 }
1726
1727 return false;
1728}

1743 parameter, \a receiver, lives.
1744
1745 \sa postEvent()
1746*/
1747void QCoreApplication::sendPostedEvents(QObject *receiver, int event_type)
1748{

1750 // one, so that a user-invoked sendPostedEvents can be detected

1756
1757void QCoreApplicationPrivate::sendPostedEvents(QObject *receiver, int event_type,
1758 QThreadData *data)
1759{
1760 if (event_type == -1) {
1761 // we were called by an obsolete event dispatcher.
1762 event_type = 0;
1763 }
1764
1765 if (receiver && receiver->d_func()->threadData.loadRelaxed() != data) {
1766 qWarning("QCoreApplication::sendPostedEvents: Cannot send "
1767 "posted events for objects in another thread");
1768 return;
1769 }
1770
1771 ++data->postEventList.recursion;
1772
1773 auto locker = qt_unique_lock(data->postEventList.mutex);
1774
1775 // by default, we assume that the event dispatcher can go to sleep after
1776 // processing all events. if any new events are posted while we send
1777 // events, canWait will be set to false.
1778 data->canWait = (data->postEventList.size() == 0);
1779
1780 if (data->postEventList.size() == 0 || (receiver && !receiver->d_func()->postedEvents)) {
1781 --data->postEventList.recursion;
1782 return;
1783 }
1784
1785 data->canWait = true;
1786
1787 // okay. here is the tricky loop. be careful about optimizing
1788 // this, it looks the way it does for good reasons.
1789 qsizetype startOffset = data->postEventList.startOffset;
1790 qsizetype &i = (!event_type && !receiver) ? data->postEventList.startOffset : startOffset;
1791 data->postEventList.insertionOffset = data->postEventList.size();
1792
1793 // Exception-safe cleaning up without the need for a try/catch block
1794 struct CleanUp {
1795 QObject *receiver;
1796 int event_type;
1797 QThreadData *data;
1798 bool exceptionCaught;
1799
1800 inline CleanUp(QObject *receiver, int event_type, QThreadData *data) :
1801 receiver(receiver), event_type(event_type), data(data), exceptionCaught(true)
1802 {}
1803 inline ~CleanUp()
1804 {
1805 if (exceptionCaught) {
1806 // since we were interrupted, we need another pass to make sure we clean everything up
1807 data->canWait = false;
1808 }
1809
1810 --data->postEventList.recursion;
1811 if (!data->postEventList.recursion && !data->canWait && data->hasEventDispatcher())
1812 data->eventDispatcher.loadRelaxed()->wakeUp();
1813
1814 // clear the global list, i.e. remove everything that was
1815 // delivered.
1816 if (!event_type && !receiver && data->postEventList.startOffset >= 0) {
1817 const QPostEventList::iterator it = data->postEventList.begin();
1818 data->postEventList.erase(it, it + data->postEventList.startOffset);
1819 data->postEventList.insertionOffset -= data->postEventList.startOffset;
1820 Q_ASSERT(data->postEventList.insertionOffset >= 0);
1821 data->postEventList.startOffset = 0;
1822 }
1823 }
1824 };
1825 CleanUp cleanup(receiver, event_type, data);
1826
1827 while (i < data->postEventList.size()) {
1828 // avoid live-lock
1829 if (i >= data->postEventList.insertionOffset)
1830 break;
1831
1832 const QPostEvent &pe = data->postEventList.at(i);
1833 ++i;
1834
1835 if (!pe.event)
1836 continue;
1837 if ((receiver && receiver != pe.receiver) || (event_type && event_type != pe.event->type())) {
1838 data->canWait = false;
1839 continue;
1840 }
1841
1842 if (pe.event->type() == QEvent::DeferredDelete) {
1843 // DeferredDelete events are sent either
1844 // 1) when the event loop that posted the event has returned; or
1845 // 2) if explicitly requested (with QEvent::DeferredDelete) for
1846 // events posted by the current event loop; or
1847 // 3) if the event was posted before the outermost event loop.
1848
1849 int eventLevel = static_cast<QDeferredDeleteEvent *>(pe.event)->loopLevel();
1850 int loopLevel = data->loopLevel + data->scopeLevel;
1851 const bool allowDeferredDelete =
1852 (eventLevel > loopLevel
1853 || (!eventLevel && loopLevel > 0)
1854 || (event_type == QEvent::DeferredDelete
1855 && eventLevel == loopLevel));
1856 if (!allowDeferredDelete) {
1857 // cannot send deferred delete
1858 if (!event_type && !receiver) {
1859 // we must copy it first; we want to re-post the event
1860 // with the event pointer intact, but we can't delay
1861 // nulling the event ptr until after re-posting, as
1862 // addEvent may invalidate pe.
1863 QPostEvent pe_copy = pe;
1864
1865 // null out the event so if sendPostedEvents recurses, it
1866 // will ignore this one, as it's been re-posted.
1867 const_cast<QPostEvent &>(pe).event = nullptr;
1868
1869 // re-post the copied event so it isn't lost
1870 data->postEventList.addEvent(pe_copy);
1871 }
1872 continue;
1873 }
1874 }
1875
1876 // first, we diddle the event so that we can deliver
1877 // it, and that no one will try to touch it later.
1878 pe.event->m_posted = false;
1879 QEvent *e = pe.event;
1880 QObject * r = pe.receiver;
1881
1882 --r->d_func()->postedEvents;
1883 Q_ASSERT(r->d_func()->postedEvents >= 0);
1884
1885 // next, update the data structure so that we're ready
1886 // for the next event.
1887 const_cast<QPostEvent &>(pe).event = nullptr;
1888
1889 locker.unlock();
1890 const auto relocker = qScopeGuard([&locker] { locker.lock(); });
1891
1892 QScopedPointer<QEvent> event_deleter(e); // will delete the event (with the mutex unlocked)
1893
1894 // after all that work, it's time to deliver the event.
1895 QCoreApplication::sendEvent(r, e);
1896
1897 // careful when adding anything below this point - the
1898 // sendEvent() call might invalidate any invariants this
1899 // function depends on.
1900 }
1901
1902 cleanup.exceptionCaught = false;
1903}

1905/*!
1906 \since 4.3
1907
1908 Removes all events of the given \a eventType that were posted
1909 using postEvent() for \a receiver.
1910
1911 The events are \e not dispatched, instead they are removed from
1912 the queue. You should never need to call this function. If you do
1913 call it, be aware that killing events may cause \a receiver to
1914 break one or more invariants.
1915
1916 If \a receiver is \nullptr, the events of \a eventType are removed
1917 for all objects. If \a eventType is 0, all the events are removed
1918 for \a receiver. You should never call this function with \a
1919 eventType of 0.
1920
1921 \threadsafe
1922*/
1923
1924void QCoreApplication::removePostedEvents(QObject *receiver, int eventType)
1925{
1926 auto locker = QCoreApplicationPrivate::lockThreadPostEventList(receiver);
1927 QThreadData *data = locker.threadData;
1928
1929 // the QObject destructor calls this function directly. this can
1930 // happen while the event loop is in the middle of posting events,
1931 // and when we get here, we may not have any more posted events
1932 // for this object.
1933 if (receiver && !receiver->d_func()->postedEvents)
1934 return;
1935
1936 //we will collect all the posted events for the QObject
1937 //and we'll delete after the mutex was unlocked
1938 QVarLengthArray<QEvent*> events;
1939 qsizetype n = data->postEventList.size();
1940 qsizetype j = 0;
1941
1942 for (qsizetype i = 0; i < n; ++i) {
1943 const QPostEvent &pe = data->postEventList.at(i);
1944
1945 if ((!receiver || pe.receiver == receiver)
1946 && (pe.event && (eventType == 0 || pe.event->type() == eventType))) {
1947 --pe.receiver->d_func()->postedEvents;
1948 pe.event->m_posted = false;
1949 events.append(pe.event);
1950 const_cast<QPostEvent &>(pe).event = nullptr;
1951 } else if (!data->postEventList.recursion) {
1952 if (i != j)
1953 qSwap(data->postEventList[i], data->postEventList[j]);
1954 ++j;
1955 }
1956 }
1957
1958#ifdef QT_DEBUG
1959 if (receiver && eventType == 0) {
1960 Q_ASSERT(!receiver->d_func()->postedEvents);
1961 }
1962#endif
1963
1964 if (!data->postEventList.recursion) {
1965 // truncate list
1966 data->postEventList.erase(data->postEventList.begin() + j, data->postEventList.end());
1967 }
1968
1969 locker.unlock();
1970 qDeleteAll(events);
1971}
