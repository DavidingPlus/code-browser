24 ExcludeUserInputEvents = 0x01,

27 X11ExcludeTimers = 0x08,

30 ApplicationExec = 0x80,
31 };

34
35 bool processEvents(ProcessEventsFlags flags = AllEvents);

41 void wakeUp();

43 bool event(QEvent *event) override;
44
45public Q_SLOTS:
46 void exit(int returnCode = 0);

48};
