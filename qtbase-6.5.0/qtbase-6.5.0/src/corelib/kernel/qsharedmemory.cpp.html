30/*!
31 \internal
32
33 Generate a string from the key which can be any unicode string into
34 the subset that the win/unix kernel allows.
35
36 On Unix this will be a file name
37 */
38QString
39QSharedMemoryPrivate::makePlatformSafeKey(const QString &key,
40 const QString &prefix)
41{
42 if (key.isEmpty())
43 return QString();
44
45 QByteArray hex = QCryptographicHash::hash(data: key.toUtf8(), method: QCryptographicHash::Sha1).toHex();
46
47#if defined(Q_OS_DARWIN) && defined(QT_POSIX_IPC)
48 if (qt_apple_isSandboxed()) {
49 // Sandboxed applications on Apple platforms require the shared memory name
50 // to be in the form <application group identifier>/<custom identifier>.
51 // Since we don't know which application group identifier the user wants
52 // to apply, we instead document that requirement, and use the key directly.
53 return key;
54 } else {
55 // The shared memory name limit on Apple platforms is very low (30 characters),
56 // so we can't use the logic below of combining the prefix, key, and a hash,
57 // to ensure a unique and valid name. Instead we use the first part of the
58 // hash, which should still long enough to avoid collisions in practice.
59 return u'/' + hex.left(SHM_NAME_MAX - 1);
60 }
61#endif
62
63 QString result = prefix;
64 for (QChar ch : key) {
65 if ((ch >= u'a' && ch <= u'z') ||
66 (ch >= u'A' && ch <= u'Z'))
67 result += ch;
68 }
69 result.append(s: QLatin1StringView(hex));
70
71#ifdef Q_OS_WIN
72 return result;
73#elif defined(QT_POSIX_IPC)
74 return u'/' + result;
75#else
76 return QDir::tempPath() + u'/' + result;
77#endif
78}
79#endif // QT_CONFIG(sharedmemory) || QT_CONFIG(systemsemaphore)

83/*!
84 \class QSharedMemory
85 \inmodule QtCore
86 \since 4.4
87
88 \brief The QSharedMemory class provides access to a shared memory segment.
89
90 QSharedMemory provides access to a shared memory segment by multiple
91 threads and processes. It also provides a way for a single thread or
92 process to lock the memory for exclusive access.
93
94 When using this class, be aware of the following platform
95 differences:
96
97 \list
98
99 \li Windows: QSharedMemory does not "own" the shared memory segment.
100 When all threads or processes that have an instance of QSharedMemory
101 attached to a particular shared memory segment have either destroyed
102 their instance of QSharedMemory or exited, the Windows kernel
103 releases the shared memory segment automatically.
104
105 \li Unix: QSharedMemory "owns" the shared memory segment. When the
106 last thread or process that has an instance of QSharedMemory
107 attached to a particular shared memory segment detaches from the
108 segment by destroying its instance of QSharedMemory, the destructor
109 releases the shared memory segment. But if that last thread or
110 process crashes without running the QSharedMemory destructor, the
111 shared memory segment survives the crash.
112
113 \li Unix: QSharedMemory can be implemented by one of two different
114 backends, selected at Qt build time: System V or POSIX. Qt defaults to
115 using the System V API if it is available, and POSIX if not. These two
116 backends do not interoperate, so two applications must ensure they use the
117 same one, even if the native key (see setNativeKey()) is the same.
118
119 The POSIX backend can be explicitly selected using the
120 \c{-feature-ipc_posix} option to the Qt configure script. If it is enabled,
121 the \c{QT_POSIX_IPC} macro will be defined.
122
123 \li Sandboxed applications on Apple platforms (including apps
124 shipped through the Apple App Store): This environment requires
125 the use of POSIX shared memory (instead of System V shared memory).
126
127 Qt for iOS is built with support for POSIX shared memory out of the box.
128 However, Qt for \macos builds (including those from the Qt installer) default
129 to System V, making them unsuitable for App Store submission if QSharedMemory
130 is needed. See above for instructions to explicitly select the POSIX backend
131 when building Qt.
132
133 In addition, in a sandboxed environment, the following caveats apply:
134
135 \list
136 \li The key must be in the form \c {<application group identifier>/<custom identifier>},
137 as documented \l {https://developer.apple.com/library/archive/documentation/Security/Conceptual/AppSandboxDesignGuide/AppSandboxInDepth/AppSandboxInDepth.html#//apple_ref/doc/uid/TP40011183-CH3-SW24}
138 {here} and \l {https://developer.apple.com/documentation/bundleresources/entitlements/com_apple_security_application-groups}
139 {here}.
140
141 \li The key length is limited to 30 characters.
142
143 \li On process exit, the named shared memory entries are not
144 cleaned up, so restarting the application and re-creating the
145 shared memory under the same name will fail. To work around this,
146 fall back to attaching to the existing shared memory entry:
147
148 \code
149
150 QSharedMemory shm("DEVTEAMID.app-group/shared");
151 if (!shm.create(42) && shm.error() == QSharedMemory::AlreadyExists)
152 shm.attach();
153
154 \endcode
155
156 \endlist
157
158 \li Android: QSharedMemory is not supported.
159
160 \endlist
161
162 Remember to lock the shared memory with lock() before reading from
163 or writing to the shared memory, and remember to release the lock
164 with unlock() after you are done.
165
166 QSharedMemory automatically destroys the shared memory segment when
167 the last instance of QSharedMemory is detached from the segment, and
168 no references to the segment remain.
169
170 \warning QSharedMemory changes the key in a Qt-specific way, unless otherwise
171 specified. Interoperation with non-Qt applications is achieved by first creating
172 a default shared memory with QSharedMemory() and then setting a native key with
173 setNativeKey(), after ensuring they use the same low-level API (System V or
174 POSIX). When using native keys, shared memory is not protected against multiple
175 accesses on it (for example, unable to lock()) and a user-defined mechanism
176 should be used to achieve such protection.
177
178 \section2 Alternative: Memory-Mapped File
179
180 Another way to share memory between processes is by opening the same file
181 using \l QFile and mapping it into memory using QFile::map() (without
182 specifying the QFileDevice::MapPrivateOption option). Any writes to the
183 mapped segment will be observed by all other processes that have mapped the
184 same file. This solution has the major advantages of being independent of the
185 backend API and of being simpler to interoperate with from non-Qt
186 applications. And since \l{QTemporaryFile} is a \l{QFile}, applications can
187 use that class to achieve clean-up semantics and to create unique shared
188 memory segments too.
189
190 To achieve locking of the shared memory segment, applications will need to
191 deploy their own mechanisms. This can be achieved by using \l
192 QBasicAtomicInteger or \c{std::atomic} in a pre-determined offset in the
193 segment itself. Higher-level locking primitives may be available on some
194 operating systems; for example, on Linux, \c{pthread_mutex_create()} can be
195 passed a flag to indicate that the mutex resides in a shared memory segment.
196
197 A major drawback of using file-backed shared memory is that the operating
198 system will attempt to write the data to permanent storage, possibly causing
199 noticeable performance penalties. To avoid this, applications should locate a
200 RAM-backed filesystem, such as \c{tmpfs} on Linux (see
201 QStorageInfo::fileSystemType()), or pass a flag to the native file-opening
202 function to inform the OS to avoid committing the contents to storage.
203
204 File-backed shared memory must be used with care if another process
205 participating is untrusted. The files may be truncated/shrunk and cause
206 applications accessing memory beyond the file's size to crash.
207
208 \section3 Linux hints on memory-mapped files
209
210 On modern Linux systems, while the \c{/tmp} directory is often a \c{tmpfs}
211 mount point, that is not a requirement. However, the \c{/dev/shm} directory
212 is required to be a \c{tmpfs} and exists for this very purpose. Do note that
213 it is world-readable and writable (like \c{/tmp} and \c{/var/tmp}), so one
214 must be careful of the contents revealed there. Another alternative is to use
215 the XDG Runtime Directory (see QStandardPaths::writableLocation() and
216 \l{QStandardPaths::RuntimeLocation}), which on Linux systems using systemd is
217 a user-specific \c{tmpfs}.
218
219 An even more secure solution is to create a "memfd" using \c{memfd_create(2)}
220 and use interprocess communication to pass the file descriptor, like
221 \l{QDBusUnixFileDescriptor} or by letting the child process of a \l{QProcess}
222 inherit it. "memfds" can also be sealed against being shrunk, so they are
223 safe to be used when communicating with processes with a different privilege
224 level.
225
226 \section3 FreeBSD hints on memory-mapped files
227
228 FreeBSD also has \c{memfd_create(2)} and can pass file descriptors to other
229 processes using the same techniques as Linux. It does not have temporary
230 filesystems mounted by default.
231
232 \section3 Windows hints on memory-mapped files
233
234 On Windows, the application can request the operating system avoid committing
235 the file's contents to permanent storage. This request is performed by
236 passing the \c{FILE_ATTRIBUTE_TEMPORARY} flag in the \c{dwFlagsAndAttributes}
237 \c{CreateFile} Win32 function, the \c{_O_SHORT_LIVED} flag to \c{_open()}
238 low-level function, or by including the modifier "T" to the \c{fopen()} C
239 runtime function.
240
241 There's also a flag to inform the operating system to delete the file when
242 the last handle to it is closed (\c{FILE_FLAG_DELETE_ON_CLOSE},
243 \c{_O_TEMPORARY}, and the "D" modifier), but do note that all processes
244 attempting to open the file must agree on using this flag or not using it. A
245 mismatch will likely cause a sharing violation and failure to open the file.
246 */

292/*!
293 Sets the platform independent \a key for this shared memory object. If \a key
294 is the same as the current key, the function returns without doing anything.
295
296 You can call key() to retrieve the platform independent key. Internally,
297 QSharedMemory converts this key into a platform specific key. If you instead
298 call nativeKey(), you will get the platform specific, converted key.
299
300 If the shared memory object is attached to an underlying shared memory
301 segment, it will \l {detach()} {detach} from it before setting the new key.
302 This function does not do an attach().
303
304 \sa key(), nativeKey(), isAttached()
305*/
306void QSharedMemory::setKey(const QString &key)
307{
308 Q_D(QSharedMemory);
309 if (key == d->key && d->makePlatformSafeKey(key) == d->nativeKey)
310 return;
311
312 if (isAttached())
313 detach();
314 d->cleanHandle();
315 d->key = key;
316 d->nativeKey = d->makePlatformSafeKey(key);
317}

319/*!
320 \since 4.8
321
322 Sets the native, platform specific, \a key for this shared memory object. If
323 \a key is the same as the current native key, the function returns without
324 doing anything. If all you want is to assign a key to a segment, you should
325 call setKey() instead.
326
327 You can call nativeKey() to retrieve the native key. If a native key has been
328 assigned, calling key() will return a null string.
329
330 If the shared memory object is attached to an underlying shared memory
331 segment, it will \l {detach()} {detach} from it before setting the new key.
332 This function does not do an attach().
333
334 The application will not be portable if you set a native key.
335
336 \sa nativeKey(), key(), isAttached()
337*/
338void QSharedMemory::setNativeKey(const QString &key)
339{
340 Q_D(QSharedMemory);
341 if (key == d->nativeKey && d->key.isNull())
342 return;
343
344 if (isAttached())
345 detach();
346 d->cleanHandle();
347 d->key = QString();
348 d->nativeKey = key;
349}

351bool QSharedMemoryPrivate::initKey()
352{
353 if (!cleanHandle())
354 return false;
355#if QT_CONFIG(systemsemaphore)
356 systemSemaphore.setKey(key: QString(), initialValue: 1);
357 systemSemaphore.setKey(key, initialValue: 1);
358 if (systemSemaphore.error() != QSystemSemaphore::NoError) {
359 QString function = "QSharedMemoryPrivate::initKey"_L1;
360 errorString = QSharedMemory::tr(s: "%1: unable to set key on lock").arg(a: function);
361 switch(systemSemaphore.error()) {
362 case QSystemSemaphore::PermissionDenied:
363 error = QSharedMemory::PermissionDenied;
364 break;
365 case QSystemSemaphore::KeyError:
366 error = QSharedMemory::KeyError;
367 break;
368 case QSystemSemaphore::AlreadyExists:
369 error = QSharedMemory::AlreadyExists;
370 break;
371 case QSystemSemaphore::NotFound:
372 error = QSharedMemory::NotFound;
373 break;
374 case QSystemSemaphore::OutOfResources:
375 error = QSharedMemory::OutOfResources;
376 break;
377 case QSystemSemaphore::UnknownError:
378 default:
379 error = QSharedMemory::UnknownError;
380 break;
381 }
382 return false;
383 }
384#endif
385 errorString = QString();
386 error = QSharedMemory::NoError;
387 return true;
388}

425/*!
426 Creates a shared memory segment of \a size bytes with the key passed to the
427 constructor, set with setKey() or set with setNativeKey(), then attaches to
428 the new shared memory segment with the given access \a mode and returns
429 \tt true. If a shared memory segment identified by the key already exists,
430 the attach operation is not performed and \tt false is returned. When the
431 return value is \tt false, call error() to determine which error occurred.
432
433 \sa error()
434 */
435bool QSharedMemory::create(qsizetype size, AccessMode mode)
436{
437 Q_D(QSharedMemory);
438
439 if (!d->initKey())
440 return false;
441
442#if QT_CONFIG(systemsemaphore)
443#ifndef Q_OS_WIN
444 // Take ownership and force set initialValue because the semaphore
445 // might have already existed from a previous crash.
446 d->systemSemaphore.setKey(key: d->key, initialValue: 1, mode: QSystemSemaphore::Create);
447#endif
448#endif
449
450 QString function = "QSharedMemory::create"_L1;
451#if QT_CONFIG(systemsemaphore)
452 QSharedMemoryLocker lock(this);
453 if (!d->key.isNull() && !d->tryLocker(locker: &lock, function))
454 return false;
455#endif
456
457 if (size <= 0) {
458 d->error = QSharedMemory::InvalidSize;
459 d->errorString =
460 QSharedMemory::tr(s: "%1: create size is less then 0").arg(a: function);
461 return false;
462 }
463
464 if (!d->create(size))
465 return false;
466
467 return d->attach(mode);
468}

497/*!
498 Attempts to attach the process to the shared memory segment
499 identified by the key that was passed to the constructor or to a
500 call to setKey() or setNativeKey(). The access \a mode is \l {QSharedMemory::}
501 {ReadWrite} by default. It can also be \l {QSharedMemory::}
502 {ReadOnly}. Returns \c true if the attach operation is successful. If
503 false is returned, call error() to determine which error occurred.
504 After attaching the shared memory segment, a pointer to the shared
505 memory can be obtained by calling data().
506
507 \sa isAttached(), detach(), create()
508 */
509bool QSharedMemory::attach(AccessMode mode)
510{
511 Q_D(QSharedMemory);
512
513 if (isAttached() || !d->initKey())
514 return false;
515#if QT_CONFIG(systemsemaphore)
516 QSharedMemoryLocker lock(this);
517 if (!d->key.isNull() && !d->tryLocker(locker: &lock, function: "QSharedMemory::attach"_L1))
518 return false;
519#endif
520
521 if (isAttached() || !d->handle())
522 return false;
523
524 return d->attach(mode);
525}

603#if QT_CONFIG(systemsemaphore)
604/*!
605 This is a semaphore that locks the shared memory segment for access
606 by this process and returns \c true. If another process has locked the
607 segment, this function blocks until the lock is released. Then it
608 acquires the lock and returns \c true. If this function returns \c false,
609 it means that you have ignored a false return from create() or attach(),
610 that you have set the key with setNativeKey() or that
611 QSystemSemaphore::acquire() failed due to an unknown system error.
612
613 \sa unlock(), data(), QSystemSemaphore::acquire()
614 */
615bool QSharedMemory::lock()
616{
617 Q_D(QSharedMemory);
618 if (d->lockedByMe) {
619 qWarning(msg: "QSharedMemory::lock: already locked");
620 return true;
621 }
622 if (d->systemSemaphore.acquire()) {
623 d->lockedByMe = true;
624 return true;
625 }
626 const auto function = "QSharedMemory::lock"_L1;
627 d->errorString = QSharedMemory::tr(s: "%1: unable to lock").arg(a: function);
628 d->error = QSharedMemory::LockError;
629 return false;
630}

632/*!
633 Releases the lock on the shared memory segment and returns \c true, if
634 the lock is currently held by this process. If the segment is not
635 locked, or if the lock is held by another process, nothing happens
636 and false is returned.
637
638 \sa lock()
639 */
640bool QSharedMemory::unlock()
641{
642 Q_D(QSharedMemory);
643 if (!d->lockedByMe)
644 return false;
645 d->lockedByMe = false;
646 if (d->systemSemaphore.release())
647 return true;
648 const auto function = "QSharedMemory::unlock"_L1;
649 d->errorString = QSharedMemory::tr(s: "%1: unable to unlock").arg(a: function);
650 d->error = QSharedMemory::LockError;
651 return false;
652}
