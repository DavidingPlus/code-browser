72 // singleShot to a QObject slot
73 template <typename Duration, typename Func1>
74 static inline void singleShot(Duration interval, const typename QtPrivate::FunctionPointer<Func1>::Object *receiver, Func1 slot)
75 {
76 singleShot(interval, defaultTypeFor(interval), receiver, slot);
77 }
78 template <typename Duration, typename Func1>
79 static inline void singleShot(Duration interval, Qt::TimerType timerType, const typename QtPrivate::FunctionPointer<Func1>::Object *receiver,
80 Func1 slot)
81 {
82 typedef QtPrivate::FunctionPointer<Func1> SlotType;
83
84 //compilation error if the slot has arguments.
85 static_assert(int(SlotType::ArgumentCount) == 0,
86 "The slot must not have any arguments.");
87
88 singleShotImpl(interval, timerType, receiver,
89 new QtPrivate::QSlotObject<Func1, typename SlotType::Arguments, void>(slot));
90 }
