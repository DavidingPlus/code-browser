29/*
30 * Design:
31 *
32 * POSIX offers a facility to select the system's monotonic clock when getting
33 * the current timestamp. Whereas the functions are mandatory in POSIX.1-2008,
34 * the presence of a monotonic clock is a POSIX Option (see the document
35 * http://pubs.opengroup.org/onlinepubs/9699919799/basedefs/V1_chap02.html#tag_02_01_06 )
36 *
37 * The macro _POSIX_MONOTONIC_CLOCK can therefore assume the following values:
38 * -1 monotonic clock is never supported on this system
39 * 0 monotonic clock might be supported, runtime check is needed
40 * >1 (such as 200809L) monotonic clock is always supported
41 *
42 * The unixCheckClockType() function will return the clock to use: either
43 * CLOCK_MONOTONIC or CLOCK_REALTIME. In the case the POSIX option has a value
44 * of zero, then this function stores a static that contains the clock to be
45 * used.
46 *
47 * There's one extra case, which is when CLOCK_REALTIME isn't defined. When
48 * that's the case, we'll emulate the clock_gettime function with gettimeofday.
49 *
50 * Conforming to:
51 * POSIX.1b-1993 section "Clocks and Timers"
52 * included in UNIX98 (Single Unix Specification v2)
53 * included in POSIX.1-2001
54 * see http://pubs.opengroup.org/onlinepubs/9699919799/functions/clock_getres.html
55 */
56
57#if !defined(CLOCK_REALTIME)
58# define CLOCK_REALTIME 0
59static inline void qt_clock_gettime(int, struct timespec *ts)
60{
61 // support clock_gettime with gettimeofday
62 struct timeval tv;
63 gettimeofday(&tv, 0);
64 ts->tv_sec = tv.tv_sec;
65 ts->tv_nsec = tv.tv_usec * 1000;
66}

68static inline int regularClock()
69{
70 return 0;
71}
72#else
73static inline void qt_clock_gettime(clockid_t clock, struct timespec *ts)
74{
75 clock_gettime(clock_id: clock, tp: ts);
76}

78static inline clock_t regularClockCheck()
79{
80 struct timespec regular_clock_resolution;
81 int r = -1;
82
83# ifdef CLOCK_MONOTONIC
84 // try the monotonic clock
85 r = clock_getres(CLOCK_MONOTONIC, res: &regular_clock_resolution);
86
87# ifdef Q_OS_LINUX
88 // Despite glibc claiming that we should check at runtime, the Linux kernel
89 // always supports the monotonic clock
90 Q_ASSERT(r == 0);
91 return CLOCK_MONOTONIC;
92# endif
93
94 if (r == 0)
95 return CLOCK_MONOTONIC;
96# endif
97
98 // no monotonic, try the realtime clock
99 r = clock_getres(CLOCK_REALTIME, res: &regular_clock_resolution);
100 Q_ASSERT(r == 0);
101 return CLOCK_REALTIME;
102}
103
104static inline clock_t regularClock()
105{
106 static const clock_t clock = regularClockCheck();
107 return clock;
108}
109#endif

111bool QElapsedTimer::isMonotonic() noexcept
112{
113 return clockType() == MonotonicClock;
114}

116QElapsedTimer::ClockType QElapsedTimer::clockType() noexcept
117{
118 return regularClock() == CLOCK_REALTIME ? SystemTime : MonotonicClock;
119}

121static inline void do_gettime(qint64 *sec, qint64 *frac)
122{
123 timespec ts;
124 qt_clock_gettime(clock: regularClock(), ts: &ts);
125 *sec = ts.tv_sec;
126 *frac = ts.tv_nsec;
127}

129// used in qcore_unix.cpp and qeventdispatcher_unix.cpp
130struct timespec qt_gettime() noexcept
131{
132 qint64 sec, frac;
133 do_gettime(sec: &sec, frac: &frac);
134
135 timespec tv;
136 tv.tv_sec = sec;
137 tv.tv_nsec = frac;
138
139 return tv;
140}

142static qint64 elapsedAndRestart(qint64 sec, qint64 frac,
143 qint64 *nowsec, qint64 *nowfrac)
144{
145 do_gettime(sec: nowsec, frac: nowfrac);
146 sec = *nowsec - sec;
147 frac = *nowfrac - frac;
148 return (sec * Q_INT64_C(1000000000) + frac) / Q_INT64_C(1000000);
149}

161qint64 QElapsedTimer::nsecsElapsed() const noexcept
162{
163 qint64 sec, frac;
164 do_gettime(sec: &sec, frac: &frac);
165 sec = sec - t1;
166 frac = frac - t2;
167 return sec * Q_INT64_C(1000000000) + frac;
168}
169
170qint64 QElapsedTimer::elapsed() const noexcept
171{
172 return nsecsElapsed() / Q_INT64_C(1000000);
173}

175qint64 QElapsedTimer::msecsSinceReference() const noexcept
176{

178}

180qint64 QElapsedTimer::msecsTo(const QElapsedTimer &other) const noexcept
181{
182 qint64 secs = other.t1 - t1;
183 qint64 fraction = other.t2 - t2;
184 return (secs * Q_INT64_C(1000000000) + fraction) / Q_INT64_C(1000000);
185}
186
187qint64 QElapsedTimer::secsTo(const QElapsedTimer &other) const noexcept
188{
189 return other.t1 - t1;
190}

192bool operator<(const QElapsedTimer &v1, const QElapsedTimer &v2) noexcept
193{
194 return v1.t1 < v2.t1 || (v1.t1 == v2.t1 && v1.t2 < v2.t2);
195}

197QDeadlineTimer QDeadlineTimer::current(Qt::TimerType timerType) noexcept
198{
199 static_assert(QDeadlineTimerNanosecondsInT2);
200 QDeadlineTimer result;
201 qint64 cursec, curnsec;
202 do_gettime(sec: &cursec, frac: &curnsec);
203 result.t1 = cursec;
204 result.t2 = curnsec;
205 result.type = timerType;
206 return result;
207}
