212 template <typename Iterator>
213 static constexpr IteratorCapabilities capabilitiesForIterator()
214 {
215 using Tag = typename std::iterator_traits<Iterator>::iterator_category;
216 IteratorCapabilities caps {};
217 if constexpr (std::is_base_of_v<std::input_iterator_tag, Tag>)
218 caps |= InputCapability;
219 if constexpr (std::is_base_of_v<std::forward_iterator_tag, Tag>)
220 caps |= ForwardCapability;
221 if constexpr (std::is_base_of_v<std::bidirectional_iterator_tag, Tag>)
222 caps |= BiDirectionalCapability;
223 if constexpr (std::is_base_of_v<std::random_access_iterator_tag, Tag>)
224 caps |= RandomAccessCapability;
225 return caps;
226 }
227
228 static constexpr IteratorCapabilities getIteratorCapabilities()
229 {
230 if constexpr (QContainerInfo::has_iterator_v<C> && !std::is_const_v<C>)
231 return capabilitiesForIterator<QContainerInfo::iterator<C>>();
232 else if constexpr (QContainerInfo::has_const_iterator_v<C>)
233 return capabilitiesForIterator<QContainerInfo::const_iterator<C>>();
234 else
235 return {};
236 }
237
238 static constexpr QMetaContainerInterface::SizeFn getSizeFn()
239 {
240 if constexpr (QContainerInfo::has_size_v<C>) {
241 return [](const void *c) -> qsizetype { return static_cast<const C *>(c)->size(); };
242 } else {
243 return nullptr;
244 }
245 }

247 static constexpr QMetaContainerInterface::ClearFn getClearFn()
248 {
249 if constexpr (QContainerInfo::has_clear_v<C>) {
250 return [](void *c) { return static_cast<C *>(c)->clear(); };
251 } else {
252 return nullptr;
253 }
254 }

256 static constexpr QMetaContainerInterface::CreateIteratorFn getCreateIteratorFn()
257 {

259 return [](void *c, QMetaContainerInterface::Position p) -> void* {
260 using Iterator = QContainerInfo::iterator<C>;
261 switch (p) {
262 case QMetaContainerInterface::Unspecified:

430template<typename C>
431class QMetaSequenceForContainer : public QMetaContainerForContainer<C>

433 friend QMetaSequenceInterface;

435 static constexpr const QtPrivate::QMetaTypeInterface *getValueMetaType()
436 {
437 if constexpr (QContainerInfo::has_value_type_v<C>)
438 return QtPrivate::qMetaTypeInterfaceForType<typename C::value_type>();
439 else
440 return nullptr;
441 }
442
443 static constexpr AddRemoveCapabilities getAddRemoveCapabilities()
444 {
445 AddRemoveCapabilities caps;
446 if constexpr (QContainerInfo::has_push_back_v<C>)
447 caps |= CanAddAtEnd;
448 if constexpr (QContainerInfo::has_pop_back_v<C>)
449 caps |= CanRemoveAtEnd;
450 if constexpr (QContainerInfo::has_push_front_v<C>)
451 caps |= CanAddAtBegin;
452 if constexpr (QContainerInfo::has_pop_front_v<C>)
453 caps |= CanRemoveAtBegin;
454 return caps;
455 }
456
457 static constexpr QMetaSequenceInterface::ValueAtIndexFn getValueAtIndexFn()
458 {
459 if constexpr (QContainerInfo::has_at_index_v<C>) {
460 return [](const void *c, qsizetype i, void *r) {
461 *static_cast<QContainerInfo::value_type<C> *>(r)
462 = static_cast<const C *>(c)->at(i);
463 };
464 } else if constexpr (QContainerInfo::can_get_at_index_v<C>) {
465 return [](const void *c, qsizetype i, void *r) {
466 *static_cast<QContainerInfo::value_type<C> *>(r)
467 = (*static_cast<const C *>(c))[i];
468 };
469 } else {
470 return nullptr;
471 }
472 }
473
474 static constexpr QMetaSequenceInterface::SetValueAtIndexFn getSetValueAtIndexFn()
475 {
476 if constexpr (QContainerInfo::can_set_at_index_v<C>) {
477 return [](void *c, qsizetype i, const void *e) {
478 (*static_cast<C *>(c))[i]
479 = *static_cast<const QContainerInfo::value_type<C> *>(e);
480 };
481 } else {
482 return nullptr;
483 }
484 }
485
486 static constexpr QMetaSequenceInterface::AddValueFn getAddValueFn()
487 {
488 if constexpr (QContainerInfo::has_push_back_v<C>) {
489 if constexpr (QContainerInfo::has_push_front_v<C>) {
490 return [](void *c, const void *v, QMetaSequenceInterface::Position position) {
491 const auto &value = *static_cast<const QContainerInfo::value_type<C> *>(v);
492 switch (position) {
493 case QMetaSequenceInterface::AtBegin:
494 static_cast<C *>(c)->push_front(value);
495 break;
496 case QMetaSequenceInterface::AtEnd:
497 case QMetaSequenceInterface::Unspecified:
498 static_cast<C *>(c)->push_back(value);
499 break;
500 }
501 };
502 } else {
503 return [](void *c, const void *v, QMetaSequenceInterface::Position position) {
504 const auto &value = *static_cast<const QContainerInfo::value_type<C> *>(v);
505 switch (position) {
506 case QMetaSequenceInterface::AtBegin:
507 break;
508 case QMetaSequenceInterface::AtEnd:
509 case QMetaSequenceInterface::Unspecified:
510 static_cast<C *>(c)->push_back(value);
511 break;
512 }
513 };
514 }
515 } else if constexpr (QContainerInfo::has_push_front_v<C>) {
516 return [](void *c, const void *v, QMetaSequenceInterface::Position position) {
517 const auto &value = *static_cast<const QContainerInfo::value_type<C> *>(v);
518 switch (position) {
519 case QMetaSequenceInterface::Unspecified:
520 case QMetaSequenceInterface::AtBegin:
521 static_cast<C *>(c)->push_front(value);
522 case QMetaSequenceInterface::AtEnd:
523 break;
524 }
525 };
526 } else if constexpr (QContainerInfo::has_insert_v<C>) {
527 return [](void *c, const void *v, QMetaSequenceInterface::Position position) {
528 if (position == QMetaSequenceInterface::Unspecified) {
529 static_cast<C *>(c)->insert(
530 *static_cast<const QContainerInfo::value_type<C> *>(v));
531 }
532 };
533 } else {
534 return nullptr;
535 }
536 }
537
538 static constexpr QMetaSequenceInterface::RemoveValueFn getRemoveValueFn()
539 {
540 if constexpr (QContainerInfo::has_pop_back_v<C>) {
541 if constexpr (QContainerInfo::has_pop_front_v<C>) {
542 return [](void *c, QMetaSequenceInterface::Position position) {
543 switch (position) {
544 case QMetaSequenceInterface::AtBegin:
545 static_cast<C *>(c)->pop_front();
546 break;
547 case QMetaSequenceInterface::AtEnd:
548 case QMetaSequenceInterface::Unspecified:
549 static_cast<C *>(c)->pop_back();
550 break;
551 }
552 };
553 } else {
554 return [](void *c, QMetaSequenceInterface::Position position) {
555 switch (position) {
556 case QMetaSequenceInterface::AtBegin:
557 break;
558 case QMetaSequenceInterface::Unspecified:
559 case QMetaSequenceInterface::AtEnd:
560 static_cast<C *>(c)->pop_back();
561 break;
562 }
563 };
564 }
565 } else if constexpr (QContainerInfo::has_pop_front_v<C>) {
566 return [](void *c, QMetaSequenceInterface::Position position) {
567 switch (position) {
568 case QMetaSequenceInterface::Unspecified:
569 case QMetaSequenceInterface::AtBegin:
570 static_cast<C *>(c)->pop_front();
571 break;
572 case QMetaSequenceInterface::AtEnd:
573 break;
574 }
575 };
576 } else {
577 return nullptr;
578 }
579 }

583 if constexpr (QContainerInfo::has_iterator_v<C>

610 && QContainerInfo::can_insert_value_at_iterator_v<C> && !std::is_const_v<C>) {
611 return [](void *c, const void *i, const void *e) {
612 static_cast<C *>(c)->insert(
613 *static_cast<const QContainerInfo::iterator<C> *>(i),
614 *static_cast<const QContainerInfo::value_type<C> *>(e));

616 } else {
617 return nullptr;
618 }

675 static constexpr QMetaAssociationInterface::InsertKeyFn getInsertKeyFn()
676 {
677 if constexpr (QContainerInfo::can_insert_key_v<C>) {
678 return [](void *c, const void *k) {
679 static_cast<C *>(c)->insert(
680 *static_cast<const QContainerInfo::key_type<C> *>(k));
681 };
682 } else if constexpr (QContainerInfo::can_insert_pair_v<C>) {

687 } else if constexpr (QContainerInfo::can_insert_key_mapped_v<C>) {
688 return [](void *c, const void *k) {

690 *static_cast<const QContainerInfo::key_type<C> *>(k), {});
691 };
692 } else {
693 return nullptr;
