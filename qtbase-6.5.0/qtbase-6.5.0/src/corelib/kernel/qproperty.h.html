56template <typename T>
57class QPropertyData : public QUntypedPropertyData
58{
59protected:
60 mutable T val = T();
61private:
62 class DisableRValueRefs {};
63protected:
64 static constexpr bool UseReferences = !(std::is_arithmetic_v<T> || std::is_enum_v<T> || std::is_pointer_v<T>);
65public:
66 using value_type = T;
67 using parameter_type = std::conditional_t<UseReferences, const T &, T>;
68 using rvalue_ref = typename std::conditional_t<UseReferences, T &&, DisableRValueRefs>;
69 using arrow_operator_result = std::conditional_t<std::is_pointer_v<T>, const T &,
70 std::conditional_t<QTypeTraits::is_dereferenceable_v<T>, const T &, void>>;
71
72 QPropertyData() = default;
73 QPropertyData(parameter_type t) : val(t) {}
74 QPropertyData(rvalue_ref t) : val(std::move(t)) {}
75 ~QPropertyData() = default;
76
77 parameter_type valueBypassingBindings() const { return val; }
78 void setValueBypassingBindings(parameter_type v) { val = v; }
79 void setValueBypassingBindings(rvalue_ref v) { val = std::move(v); }
80};

327template <typename T>
328class QProperty : public QPropertyData<T>
329{
330 QtPrivate::QPropertyBindingData d;
331 bool is_equal(const T &v)
332 {
333 if constexpr (QTypeTraits::has_operator_equal_v<T>) {
334 if (v == this->val)
335 return true;
336 }
337 return false;
338 }
339
340public:
341 using value_type = typename QPropertyData<T>::value_type;
342 using parameter_type = typename QPropertyData<T>::parameter_type;
343 using rvalue_ref = typename QPropertyData<T>::rvalue_ref;
344 using arrow_operator_result = typename QPropertyData<T>::arrow_operator_result;
345
346 QProperty() = default;
347 explicit QProperty(parameter_type initialValue) : QPropertyData<T>(initialValue) {}
348 explicit QProperty(rvalue_ref initialValue) : QPropertyData<T>(std::move(initialValue)) {}
349 explicit QProperty(const QPropertyBinding<T> &binding)
350 : QProperty()
351 { setBinding(binding); }
352#ifndef Q_QDOC
353 template <typename Functor>
354 explicit QProperty(Functor &&f, const QPropertyBindingSourceLocation &location = QT_PROPERTY_DEFAULT_BINDING_LOCATION,
355 typename std::enable_if_t<std::is_invocable_r_v<T, Functor&>> * = nullptr)
356 : QProperty(QPropertyBinding<T>(std::forward<Functor>(f), location))
357 {}
358#else
359 template <typename Functor>
360 explicit QProperty(Functor &&f);
361#endif
362 ~QProperty() = default;
363
364 parameter_type value() const
365 {
366 d.registerWithCurrentlyEvaluatingBinding();
367 return this->val;
368 }
369
370 arrow_operator_result operator->() const
371 {
372 if constexpr (QTypeTraits::is_dereferenceable_v<T>) {
373 return value();
374 } else if constexpr (std::is_pointer_v<T>) {
375 value();
376 return this->val;
377 } else {
378 return;
379 }
380 }
381
382 parameter_type operator*() const
383 {
384 return value();
385 }
386
387 operator parameter_type() const
388 {
389 return value();
390 }
391
392 void setValue(rvalue_ref newValue)
393 {
394 d.removeBinding();
395 if (is_equal(v: newValue))
396 return;
397 this->val = std::move(newValue);
398 notify();
399 }
400
401 void setValue(parameter_type newValue)
402 {
403 d.removeBinding();
404 if (is_equal(v: newValue))
405 return;
406 this->val = newValue;
407 notify();
408 }
409
410 QProperty<T> &operator=(rvalue_ref newValue)
411 {
412 setValue(std::move(newValue));
413 return *this;
414 }
415
416 QProperty<T> &operator=(parameter_type newValue)
417 {
418 setValue(newValue);
419 return *this;
420 }
421
422 QPropertyBinding<T> setBinding(const QPropertyBinding<T> &newBinding)
423 {
424 return QPropertyBinding<T>(d.setBinding(newBinding, propertyDataPtr: this));
425 }
426
427 bool setBinding(const QUntypedPropertyBinding &newBinding)
428 {
429 if (!newBinding.isNull() && newBinding.valueMetaType().id() != qMetaTypeId<T>())
430 return false;
431 setBinding(static_cast<const QPropertyBinding<T> &>(newBinding));
432 return true;
433 }
434
435#ifndef Q_QDOC
436 template <typename Functor>
437 QPropertyBinding<T> setBinding(Functor &&f,
438 const QPropertyBindingSourceLocation &location = QT_PROPERTY_DEFAULT_BINDING_LOCATION,
439 std::enable_if_t<std::is_invocable_v<Functor>> * = nullptr)
440 {
441 return setBinding(Qt::makePropertyBinding(std::forward<Functor>(f), location));
442 }
443#else
444 template <typename Functor>
445 QPropertyBinding<T> setBinding(Functor f);
446#endif
447
448 bool hasBinding() const { return d.hasBinding(); }
449
450 QPropertyBinding<T> binding() const
451 {
452 return QPropertyBinding<T>(QUntypedPropertyBinding(d.binding()));
453 }
454
455 QPropertyBinding<T> takeBinding()
456 {
457 return QPropertyBinding<T>(d.setBinding(newBinding: QUntypedPropertyBinding(), propertyDataPtr: this));
458 }
459
460 template<typename Functor>
461 QPropertyChangeHandler<Functor> onValueChanged(Functor f)
462 {
463 static_assert(std::is_invocable_v<Functor>, "Functor callback must be callable without any parameters");
464 return QPropertyChangeHandler<Functor>(*this, f);
465 }
466
467 template<typename Functor>
468 QPropertyChangeHandler<Functor> subscribe(Functor f)
469 {
470 static_assert(std::is_invocable_v<Functor>, "Functor callback must be callable without any parameters");
471 f();
472 return onValueChanged(f);
473 }
474
475 template<typename Functor>
476 QPropertyNotifier addNotifier(Functor f)
477 {
478 static_assert(std::is_invocable_v<Functor>, "Functor callback must be callable without any parameters");
479 return QPropertyNotifier(*this, f);
480 }
481
482 const QtPrivate::QPropertyBindingData &bindingData() const { return d; }
483private:
484 void notify()
485 {
486 d.notifyObservers(this);
487 }

535 { *static_cast<T*>(value) = static_cast<const Property *>(d)->value(); },
536 nullptr,
537 nullptr,
538 nullptr,
539 [](const QUntypedPropertyData *d, const QPropertyBindingSourceLocation &location) -> QUntypedPropertyBinding
540 { return Qt::makePropertyBinding([d]() -> T { return static_cast<const Property *>(d)->value(); }, location); },
541 [](const QUntypedPropertyData *d, QPropertyObserver *observer) -> void
542 { observer->setSource(static_cast<const Property *>(d)->bindingData()); },

656class QUntypedBindable
657{
658 friend struct QUntypedBindablePrivate; // allows access to internal data
659protected:
660 QUntypedPropertyData *data = nullptr;
661 const QtPrivate::QBindableInterface *iface = nullptr;
662 constexpr QUntypedBindable(QUntypedPropertyData *d, const QtPrivate::QBindableInterface *i)
663 : data(d), iface(i)
664 {}
665
666 Q_CORE_EXPORT explicit QUntypedBindable(QObject* obj, const QMetaProperty &property, const QtPrivate::QBindableInterface *i);
667 Q_CORE_EXPORT explicit QUntypedBindable(QObject* obj, const char* property, const QtPrivate::QBindableInterface *i);
668
669public:
670 constexpr QUntypedBindable() = default;
671 template<typename Property>
672 QUntypedBindable(Property *p)
673 : data(const_cast<std::remove_cv_t<Property> *>(p)),
674 iface(&QtPrivate::QBindableInterfaceForProperty<Property>::iface)
675 { Q_ASSERT(data && iface); }
676
677 bool isValid() const { return data != nullptr; }
678 bool isBindable() const { return iface && iface->getBinding; }
679 bool isReadOnly() const { return !(iface && iface->setBinding && iface->setObserver); }
680
681 QUntypedPropertyBinding makeBinding(const QPropertyBindingSourceLocation &location = QT_PROPERTY_DEFAULT_BINDING_LOCATION) const
682 {
683 return iface ? iface->makeBinding(data, location) : QUntypedPropertyBinding();
684 }
685
686 QUntypedPropertyBinding takeBinding()
687 {
688 if (!iface)
689 return QUntypedPropertyBinding {};
690 // We do not have a dedicated takeBinding function pointer in the interface
691 // therefore we synthesize takeBinding by retrieving the binding with binding
692 // and calling setBinding with a default constructed QUntypedPropertyBinding
693 // afterwards.
694 if (!(iface->getBinding && iface->setBinding))
695 return QUntypedPropertyBinding {};
696 QUntypedPropertyBinding binding = iface->getBinding(data);
697 iface->setBinding(data, QUntypedPropertyBinding{});
698 return binding;
699 }
700
701 void observe(QPropertyObserver *observer) const
702 {
703 if (iface)
704 iface->setObserver(data, observer);
705#ifndef QT_NO_DEBUG
