18#include "qcoreapplication.h"
19#include "qobject_p.h"

21QT_BEGIN_NAMESPACE
22
23class QEventLoopPrivate : public QObjectPrivate
24{
25 Q_DECLARE_PUBLIC(QEventLoop)
26public:
27 inline QEventLoopPrivate()
28 : inExec(false)
29 {
30 returnCode.storeRelaxed(newValue: -1);
31 exit.storeRelaxed(newValue: true);
32 }
33
34 QAtomicInt quitLockRef;
35
36 QBasicAtomicInt exit; // bool
37 QBasicAtomicInt returnCode;
38 bool inExec;
39
40 void ref()
41 {
42 quitLockRef.ref();
43 }
44
45 void deref()
46 {
47 if (!quitLockRef.deref() && inExec) {
48 qApp->postEvent(receiver: q_ptr, event: new QEvent(QEvent::Quit));
49 }
50 }
51};
