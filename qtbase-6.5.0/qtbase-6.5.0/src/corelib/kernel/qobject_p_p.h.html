37struct QObjectPrivate::ConnectionOrSignalVector
38{
39 union {
40 // linked list of orphaned connections that need cleaning up
41 ConnectionOrSignalVector *nextInOrphanList;
42 // linked list of connections connected to slots in this object
43 Connection *next;
44 };
45
46 static SignalVector *asSignalVector(ConnectionOrSignalVector *c)
47 {
48 if (reinterpret_cast<quintptr>(c) & 1)
49 return reinterpret_cast<SignalVector *>(reinterpret_cast<quintptr>(c) & ~quintptr(1u));
50 return nullptr;
51 }
52 static Connection *fromSignalVector(SignalVector *v)
53 {
54 return reinterpret_cast<Connection *>(reinterpret_cast<quintptr>(v) | quintptr(1u));
55 }
56};
57static_assert(std::is_trivial_v<QObjectPrivate::ConnectionOrSignalVector>);

59struct QObjectPrivate::Connection : public ConnectionOrSignalVector
60{
61 // linked list of connections connected to slots in this object, next is in base class
62 Connection **prev;
63 // linked list of connections connected to signals in this object
64 QAtomicPointer<Connection> nextConnectionList;
65 Connection *prevConnectionList;
66
67 QObject *sender;
68 QAtomicPointer<QObject> receiver;
69 QAtomicPointer<QThreadData> receiverThreadData;
70 union {
71 StaticMetaCallFunction callFunction;
72 QtPrivate::QSlotObjectBase *slotObj;
73 };
74 QAtomicPointer<const int> argumentTypes;
75 QAtomicInt ref_{
76 2
77 }; // ref_ is 2 for the use in the internal lists, and for the use in QMetaObject::Connection
78 uint id = 0;
79 ushort method_offset;
80 ushort method_relative;
81 signed int signal_index : 27; // In signal range (see QObjectPrivate::signalIndex())
82 ushort connectionType : 2; // 0 == auto, 1 == direct, 2 == queued, 3 == blocking
83 ushort isSlotObject : 1;
84 ushort ownArgumentTypes : 1;
85 ushort isSingleShot : 1;
86 Connection() : ownArgumentTypes(true) { }
87 ~Connection();
88 int method() const
89 {
90 Q_ASSERT(!isSlotObject);
91 return method_offset + method_relative;
92 }
93 void ref() { ref_.ref(); }
94 void freeSlotObject()
95 {
96 if (isSlotObject) {
97 slotObj->destroyIfLastRef();
98 isSlotObject = false;
99 }
100 }
101 void deref()
102 {
103 if (!ref_.deref()) {
104 Q_ASSERT(!receiver.loadRelaxed());
105 Q_ASSERT(!isSlotObject);
106 delete this;
107 }
108 }
109};

112struct QObjectPrivate::SignalVector : public ConnectionOrSignalVector

126struct QObjectPrivate::ConnectionData
127{
128 // the id below is used to avoid activating new connections. When the object gets
129 // deleted it's set to 0, so that signal emission stops
130 QAtomicInteger<uint> currentConnectionId;
131 QAtomicInt ref;
132 QAtomicPointer<SignalVector> signalVector;
133 Connection *senders = nullptr;
134 Sender *currentSender = nullptr; // object currently activating the object
135 QAtomicPointer<Connection> orphaned;
136
137 ~ConnectionData()
138 {
139 Q_ASSERT(ref.loadRelaxed() == 0);
140 auto *c = orphaned.fetchAndStoreRelaxed(newValue: nullptr);
141 if (c)
142 deleteOrphaned(c);
143 SignalVector *v = signalVector.loadRelaxed();
144 if (v) {
145 v->~SignalVector();
146 free(ptr: v);
147 }
148 }
149
150 // must be called on the senders connection data
151 // assumes the senders and receivers lock are held
152 void removeConnection(Connection *c);
153 enum LockPolicy {
154 NeedToLock,
155 // Beware that we need to temporarily release the lock
156 // and thus calling code must carefully consider whether
157 // invariants still hold.
158 AlreadyLockedAndTemporarilyReleasingLock
159 };
160 void cleanOrphanedConnections(QObject *sender, LockPolicy lockPolicy = NeedToLock)
161 {
162 if (orphaned.loadRelaxed() && ref.loadAcquire() == 1)
163 cleanOrphanedConnectionsImpl(sender, lockPolicy);
164 }
165 void cleanOrphanedConnectionsImpl(QObject *sender, LockPolicy lockPolicy);

172 void resizeSignalVector(uint size)
173 {
174 SignalVector *vector = this->signalVector.loadRelaxed();
175 if (vector && vector->allocated > size)
176 return;
177 size = (size + 7) & ~7;
178 void *ptr = malloc(size: sizeof(SignalVector) + (size + 1) * sizeof(ConnectionList));
179 auto newVector = new (ptr) SignalVector;
180
181 int start = -1;
182 if (vector) {
183 // not (yet) existing trait:
184 // static_assert(std::is_relocatable_v<SignalVector>);
185 // static_assert(std::is_relocatable_v<ConnectionList>);
186 memcpy(dest: newVector, src: vector,
187 n: sizeof(SignalVector) + (vector->allocated + 1) * sizeof(ConnectionList));
188 start = vector->count();
189 }
190 for (int i = start; i < int(size); ++i)
191 new (&newVector->at(i)) ConnectionList();
192 newVector->next = nullptr;
193 newVector->allocated = size;
194
195 signalVector.storeRelaxed(newValue: newVector);
196 if (vector) {
197 Connection *o = nullptr;
198 /* No ABA issue here: When adding a node, we only care about the list head, it doesn't
199 * matter if the tail changes.
200 */
201 do {
202 o = orphaned.loadRelaxed();
203 vector->nextInOrphanList = o;
204 } while (!orphaned.testAndSetRelease(
205 expectedValue: o, newValue: ConnectionOrSignalVector::fromSignalVector(v: vector)));
206 }
207 }
208 int signalVectorCount() const
209 {
210 return signalVector.loadAcquire() ? signalVector.loadRelaxed()->count() : -1;
211 }
212
213 static void deleteOrphaned(ConnectionOrSignalVector *c);
214};

216struct QObjectPrivate::Sender
217{
218 Sender(QObject *receiver, QObject *sender, int signal)
219 : receiver(receiver), sender(sender), signal(signal)
220 {
221 if (receiver) {
222 ConnectionData *cd = receiver->d_func()->connections.loadRelaxed();
223 previous = cd->currentSender;
224 cd->currentSender = this;
225 }
226 }
227 ~Sender()
228 {
229 if (receiver)
230 receiver->d_func()->connections.loadRelaxed()->currentSender = previous;
231 }
232 void receiverDeleted()
233 {
234 Sender *s = this;
235 while (s) {
236 s->receiver = nullptr;
237 s = s->previous;
238 }
239 }
240 Sender *previous;
241 QObject *receiver;
242 QObject *sender;
243 int signal;
244};
