12/*!
13 Sleeps for \a ms milliseconds, blocking execution of the
14 test. qSleep() will not do any event processing and leave your test
15 unresponsive. Network communication might time out while
16 sleeping. Use \l {QTest::qWait()} to do non-blocking sleeping.
17
18 \a ms must be greater than 0.
19
20 \b {Note:} The qSleep() function calls either \c nanosleep() on
21 unix or \c Sleep() on windows, so the accuracy of time spent in
22 qSleep() depends on the operating system.
23
24 Example:
25 \snippet code/src_qtestlib_qtestcase.cpp 23
26
27 \sa {QTest::qWait()}
28*/
29Q_CORE_EXPORT void QTest::qSleep(int ms)
30{
31 Q_ASSERT(ms > 0);
32
33#if defined(Q_OS_WIN)
34 Sleep(uint(ms));
35#else
36 struct timespec ts = { .tv_sec: time_t(ms / 1000), .tv_nsec: (ms % 1000) * 1000 * 1000 };
37 nanosleep(requested_time: &ts, remaining: nullptr);
38#endif
39}

58/*! \fn void QTest::qWait(int ms)
59
60 Waits for \a ms milliseconds. While waiting, events will be processed and
61 your test will stay responsive to user interface events or network communication.
62
63 Example:
64
65 \snippet code/src_corelib_kernel_qtestsupport_core.cpp 1
66
67 The code above will wait until the network server is responding for a
68 maximum of about 12.5 seconds.
69
70 \sa QTest::qSleep(), QSignalSpy::wait()
71*/
72Q_CORE_EXPORT void QTest::qWait(int ms)
73{
74 // Ideally this method would be implemented in terms of qWaitFor(), with a
75 // predicate that always returns false, but qWaitFor() uses the 1-arg overload
76 // of processEvents(), which doesn't handle events posted in this round of event
77 // processing, which, together with the 10ms qSleep() after every processEvents(),
78 // lead to a 10x slow-down in some webengine tests.
79
80 Q_ASSERT(QCoreApplication::instance());
81
82 QDeadlineTimer timer(ms, Qt::PreciseTimer);
83 int remaining = ms;
84 do {
85 QCoreApplication::processEvents(flags: QEventLoop::AllEvents, maxtime: remaining);
86 QCoreApplication::sendPostedEvents(receiver: nullptr, event_type: QEvent::DeferredDelete);
87 remaining = timer.remainingTime();
88 if (remaining <= 0)
89 break;
90 QTest::qSleep(ms: qMin(a: 10, b: remaining));
91 remaining = timer.remainingTime();
92 } while (remaining > 0);
93}
