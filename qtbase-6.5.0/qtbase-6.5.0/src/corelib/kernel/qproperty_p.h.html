97struct [[nodiscard]] QPropertyObserverNodeProtector
98{
99 Q_DISABLE_COPY_MOVE(QPropertyObserverNodeProtector)
100
101 QPropertyObserverBase m_placeHolder;
102 QPropertyObserverNodeProtector(QPropertyObserver *observer)
103 {
104 // insert m_placeholder after observer into the linked list
105 QPropertyObserver *next = observer->next.data();
106 m_placeHolder.next = next;
107 observer->next = static_cast<QPropertyObserver *>(&m_placeHolder);
108 if (next)
109 next->prev = &m_placeHolder.next;
110 m_placeHolder.prev = &observer->next;
111 m_placeHolder.next.setTag(QPropertyObserver::ObserverIsPlaceholder);
112 }
113
114 QPropertyObserver *next() const { return m_placeHolder.next.data(); }
115
116 ~QPropertyObserverNodeProtector();
117};

418inline void QPropertyBindingDataPointer::fixupAfterMove(QtPrivate::QPropertyBindingData *ptr)
419{
420 auto &d = ptr->d_ref();
421 if (ptr->isNotificationDelayed()) {
422 QPropertyProxyBindingData *proxy = ptr->proxyData();
423 Q_ASSERT(proxy);
424 proxy->originalBindingData = ptr;
425 }
426 // If QPropertyBindingData has been moved, and it has an observer
427 // we have to adjust the firstObserver's prev pointer to point to
428 // the moved to QPropertyBindingData's d_ptr
429 if (d & QtPrivate::QPropertyBindingData::BindingBit)
430 return; // nothing to do if the observer is stored in the binding
431 if (auto observer = reinterpret_cast<QPropertyObserver *>(d))
432 observer->prev = reinterpret_cast<QPropertyObserver **>(&d);
433}

435inline QPropertyObserverPointer QPropertyBindingDataPointer::firstObserver() const
436{
437 if (auto *b = binding())
438 return b->firstObserver;
439 return { .ptr: reinterpret_cast<QPropertyObserver *>(ptr->d()) };
440}

466template<typename Class, typename T, auto Offset, auto Setter, auto Signal = nullptr,
467 auto Getter = nullptr>
468class QObjectCompatProperty : public QPropertyData<T>
469{
470 template<typename Property, typename>
471 friend class QtPrivate::QBindableInterfaceForProperty;
472
473 using ThisType = QObjectCompatProperty<Class, T, Offset, Setter, Signal, Getter>;
474 using SignalTakesValue = std::is_invocable<decltype(Signal), Class, T>;
475 Class *owner()
476 {
477 char *that = reinterpret_cast<char *>(this);
478 return reinterpret_cast<Class *>(that - QtPrivate::detail::getOffset(Offset));
479 }
480 const Class *owner() const
481 {
482 char *that = const_cast<char *>(reinterpret_cast<const char *>(this));
483 return reinterpret_cast<Class *>(that - QtPrivate::detail::getOffset(Offset));
484 }
485
486 static bool bindingWrapper(QMetaType type, QUntypedPropertyData *dataPtr, QtPrivate::QPropertyBindingFunction binding)
487 {
488 auto *thisData = static_cast<ThisType *>(dataPtr);
489 QPropertyData<T> copy;
490 binding.vtable->call(type, &copy, binding.functor);
491 if constexpr (QTypeTraits::has_operator_equal_v<T>)
492 if (copy.valueBypassingBindings() == thisData->valueBypassingBindings())
493 return false;
494 // ensure value and setValue know we're currently evaluating our binding
495 QBindingStorage *storage = qGetBindingStorage(thisData->owner());
496 QtPrivate::CompatPropertySafePoint guardThis(storage->bindingStatus, thisData);
497 (thisData->owner()->*Setter)(copy.valueBypassingBindings());
498 return true;
499 }
500 bool inBindingWrapper(const QBindingStorage *storage) const
501 {
502 return storage->bindingStatus && storage->bindingStatus->currentCompatProperty
503 && QtPrivate::isPropertyInBindingWrapper(property: this);
504 }
505
506 inline static T getPropertyValue(const QUntypedPropertyData *d) {
507 auto prop = static_cast<const ThisType *>(d);
508 if constexpr (std::is_null_pointer_v<decltype(Getter)>)
509 return prop->value();
510 else
511 return (prop->owner()->*Getter)();
512 }
513
514public:
515 using value_type = typename QPropertyData<T>::value_type;
516 using parameter_type = typename QPropertyData<T>::parameter_type;
517 using arrow_operator_result = typename QPropertyData<T>::arrow_operator_result;
518
519 QObjectCompatProperty() = default;
520 explicit QObjectCompatProperty(const T &initialValue) : QPropertyData<T>(initialValue) {}
521 explicit QObjectCompatProperty(T &&initialValue) : QPropertyData<T>(std::move(initialValue)) {}
522
523 parameter_type value() const
524 {
525 const QBindingStorage *storage = qGetBindingStorage(owner());
526 // make sure we don't register this binding as a dependency to itself
527 if (storage->bindingStatus && storage->bindingStatus->currentlyEvaluatingBinding && !inBindingWrapper(storage))
528 storage->registerDependency_helper(data: this);
529 return this->val;
530 }
531
532 arrow_operator_result operator->() const
533 {
534 if constexpr (QTypeTraits::is_dereferenceable_v<T>) {
535 return value();
536 } else if constexpr (std::is_pointer_v<T>) {
537 value();
538 return this->val;
539 } else {
540 return;
541 }
542 }
543
544 parameter_type operator*() const
545 {
546 return value();
547 }
548
549 operator parameter_type() const
550 {
551 return value();
552 }
553
554 void setValue(parameter_type t)
555 {
556 QBindingStorage *storage = qGetBindingStorage(owner());
557 if (auto *bd = storage->bindingData(this)) {
558 // make sure we don't remove the binding if called from the bindingWrapper
559 if (bd->hasBinding() && !inBindingWrapper(storage))
560 bd->removeBinding_helper();
561 }
562 this->val = t;
563 }
564
565 QObjectCompatProperty &operator=(parameter_type newValue)
566 {
567 setValue(newValue);
568 return *this;
569 }

571 QPropertyBinding<T> setBinding(const QPropertyBinding<T> &newBinding)
572 {
573 QtPrivate::QPropertyBindingData *bd = qGetBindingStorage(owner())->bindingData(this, true);
574 QUntypedPropertyBinding oldBinding(bd->setBinding(newBinding, propertyDataPtr: this, staticObserverCallback: nullptr, bindingWrapper));
575 // notification is already handled in QPropertyBindingData::setBinding

600 bool hasBinding() const {
601 auto *bd = qGetBindingStorage(owner())->bindingData(this);
602 return bd && bd->binding() != nullptr;
603 }
604
605 void removeBindingUnlessInWrapper()
606 {
607 QBindingStorage *storage = qGetBindingStorage(owner());
608 if (auto *bd = storage->bindingData(this)) {
609 // make sure we don't remove the binding if called from the bindingWrapper
610 if (bd->hasBinding() && !inBindingWrapper(storage))
611 bd->removeBinding_helper();
612 }
613 }
614
615 void notify()
616 {
617 QBindingStorage *storage = qGetBindingStorage(owner());
618 if (auto bd = storage->bindingData(this, false)) {
619 // This partly duplicates QPropertyBindingData::notifyObservers because we want to
620 // check for inBindingWrapper() after checking for isNotificationDelayed() and
621 // firstObserver. This is because inBindingWrapper() is the most expensive check.
622 if (!bd->isNotificationDelayed()) {
623 QPropertyBindingDataPointer d{bd};
624 if (QPropertyObserverPointer observer = d.firstObserver()) {
625 if (!inBindingWrapper(storage)) {
626 PendingBindingObserverList bindingObservers;
627 if (bd->notifyObserver_helper(this, storage, observer, bindingObservers)
628 == QtPrivate::QPropertyBindingData::Evaluated) {
629 // evaluateBindings() can trash the observers. We need to re-fetch here.
630 if (QPropertyObserverPointer observer = d.firstObserver())
631 observer.notifyOnlyChangeHandler(propertyDataPtr: this);
632 for (auto&& bindingObserver: bindingObservers)
633 bindingObserver.binding()->notifyNonRecursive();
634 }
635 }
636 }
637 }
638 }
639 if constexpr (!std::is_null_pointer_v<decltype(Signal)>) {
640 if constexpr (SignalTakesValue::value)
641 (owner()->*Signal)(getPropertyValue(d: this));
642 else
643 (owner()->*Signal)();
644 }
645 }

648 {
649 auto *bd = qGetBindingStorage(owner())->bindingData(this);
650 return static_cast<QPropertyBinding<T> &&>(QUntypedPropertyBinding(bd ? bd->binding() : nullptr));
651 }

790inline bool QPropertyBindingPrivate::evaluateRecursive_inline(PendingBindingObserverList &bindingObservers, QBindingStatus *status)
791{
792 if (updating) {
793 error = QPropertyBindingError(QPropertyBindingError::BindingLoop);
794 if (isQQmlPropertyBinding)
795 errorCallBack(this);
796 return false;
797 }
798
799 /*
800 * Evaluating the binding might lead to the binding being broken. This can
801 * cause ref to reach zero at the end of the function. However, the
802 * updateGuard's destructor will then still trigger, trying to set the
803 * updating bool to its old value
804 * To prevent this, we create a QPropertyBindingPrivatePtr which ensures
805 * that the object is still alive when updateGuard's dtor runs.
806 */
807 QPropertyBindingPrivatePtr keepAlive {this};
808
809 QScopedValueRollback<bool> updateGuard(updating, true);
810
811 QtPrivate::BindingEvaluationState evaluationFrame(this, status);
812
813 auto bindingFunctor = reinterpret_cast<std::byte *>(this) +
814 QPropertyBindingPrivate::getSizeEnsuringAlignment();
815 bool changed = false;
816 if (hasBindingWrapper) {
817 changed = staticBindingWrapper(metaType, propertyDataPtr,
818 {.vtable: vtable, .functor: bindingFunctor});
819 } else {
820 changed = vtable->call(metaType, propertyDataPtr, bindingFunctor);
821 }
822 // If there was a change, we must set pendingNotify.
823 // If there was not, we must not clear it, as that only should happen in notifyRecursive
824 pendingNotify = pendingNotify || changed;
825 if (!changed || !firstObserver)
826 return changed;
827
828 firstObserver.noSelfDependencies(this);
829 firstObserver.evaluateBindings(bindingObservers, status);
830 return true;
831}

833template<QPropertyObserverPointer::Notify notifyPolicy>
834inline void QPropertyObserverPointer::notify(QUntypedPropertyData *propertyDataPtr)
835{
836 auto observer = const_cast<QPropertyObserver*>(ptr);
837 /*
838 * The basic idea of the loop is as follows: We iterate over all observers in the linked list,
839 * and execute the functionality corresponding to their tag.
840 * However, complication arise due to the fact that the triggered operations might modify the list,
841 * which includes deletion and move of the current and next nodes.
842 * Therefore, we take a few safety precautions:
843 * 1. Before executing any action which might modify the list, we insert a placeholder node after the current node.
844 * As that one is stack allocated and owned by us, we can rest assured that it is
845 * still there after the action has executed, and placeHolder->next points to the actual next node in the list.
846 * Note that taking next at the beginning of the loop does not work, as the executed action might either move
847 * or delete that node.
848 * 2. After the triggered action has finished, we can use the next pointer in the placeholder node as a safe way to
849 * retrieve the next node.
850 * 3. Some care needs to be taken to avoid infinite recursion with change handlers, so we add an extra test there, that
851 * checks whether we're already have the same change handler in our call stack. This can be done by checking whether
852 * the node after the current one is a placeholder node.
853 */
854 while (observer) {
855 QPropertyObserver *next = observer->next.data();
856 switch (QPropertyObserver::ObserverTag(observer->next.tag())) {
857 case QPropertyObserver::ObserverNotifiesChangeHandler:
858 {
859 auto handlerToCall = observer->changeHandler;
860 // prevent recursion
861 if (next && next->next.tag() == QPropertyObserver::ObserverIsPlaceholder) {
862 observer = next->next.data();
863 continue;
864 }
865 // handlerToCall might modify the list
866 QPropertyObserverNodeProtector protector(observer);
867 handlerToCall(observer, propertyDataPtr);
868 next = protector.next();
869 break;
870 }
871 case QPropertyObserver::ObserverNotifiesBinding:
872 {
873 if constexpr (notifyPolicy == Notify::Everything) {
874 auto bindingToNotify = observer->binding;
875 QPropertyObserverNodeProtector protector(observer);
876 bindingToNotify->notifyRecursive();
877 next = protector.next();
878 }
879 break;
880 }
881 case QPropertyObserver::ObserverIsPlaceholder:
882 // recursion is already properly handled somewhere else
883 break;
884 case QPropertyObserver::ObserverIsAlias:
885 break;
886 default: Q_UNREACHABLE();
887 }
888 observer = next;
889 }
890}
