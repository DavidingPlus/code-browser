116inline QSocketNotifierSetUNIX::QSocketNotifierSetUNIX() noexcept
117{
118 notifiers[0] = nullptr;
119 notifiers[1] = nullptr;
120 notifiers[2] = nullptr;
121}
122
123inline bool QSocketNotifierSetUNIX::isEmpty() const noexcept
124{
125 return !notifiers[0] && !notifiers[1] && !notifiers[2];
126}
127
128inline short QSocketNotifierSetUNIX::events() const noexcept
129{
130 short result = 0;
131
132 if (notifiers[0])
133 result |= POLLIN;
134
135 if (notifiers[1])
136 result |= POLLOUT;
137
138 if (notifiers[2])
139 result |= POLLPRI;
140
141 return result;
142}
