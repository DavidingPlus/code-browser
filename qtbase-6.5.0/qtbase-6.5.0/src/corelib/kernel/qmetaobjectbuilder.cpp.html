66 QMetaMethod::Access _access = QMetaMethod::Public,
67 int _revision = 0)
68 : signature(QMetaObject::normalizedSignature(method: _signature.constData())),
69 returnType(QMetaObject::normalizedType(type: _returnType)),

107

642/*!
643 Adds the contents of \a prototype to this meta object builder.
644 This function is useful for cloning the contents of an existing QMetaObject.
645
646 The \a members parameter indicates which members of \a prototype
647 should be added. The default is AllMembers.
648*/
649void QMetaObjectBuilder::addMetaObject(const QMetaObject *prototype,
650 QMetaObjectBuilder::AddMembers members)
651{
652 Q_ASSERT(prototype);
653 int index;
654
655 if ((members & ClassName) != 0)
656 d->className = prototype->className();
657
658 if ((members & SuperClass) != 0)
659 d->superClass = prototype->superClass();
660
661 if ((members & (Methods | Signals | Slots)) != 0) {
662 for (index = prototype->methodOffset(); index < prototype->methodCount(); ++index) {
663 QMetaMethod method = prototype->method(index);
664 if (method.methodType() != QMetaMethod::Signal) {
665 if (method.access() == QMetaMethod::Public && (members & PublicMethods) == 0)
666 continue;
667 if (method.access() == QMetaMethod::Private && (members & PrivateMethods) == 0)
668 continue;
669 if (method.access() == QMetaMethod::Protected && (members & ProtectedMethods) == 0)
670 continue;
671 }
672 if (method.methodType() == QMetaMethod::Method && (members & Methods) != 0) {
673 addMethod(prototype: method);
674 } else if (method.methodType() == QMetaMethod::Signal &&
675 (members & Signals) != 0) {
676 addMethod(prototype: method);
677 } else if (method.methodType() == QMetaMethod::Slot &&
678 (members & Slots) != 0) {
679 addMethod(prototype: method);
680 }
681 }
682 }
683
684 if ((members & Constructors) != 0) {
685 for (index = 0; index < prototype->constructorCount(); ++index)
686 addConstructor(prototype: prototype->constructor(index));
687 }
688
689 if ((members & Properties) != 0) {
690 for (index = prototype->propertyOffset(); index < prototype->propertyCount(); ++index)
691 addProperty(prototype: prototype->property(index));
692 }
693
694 if ((members & Enumerators) != 0) {
695 for (index = prototype->enumeratorOffset(); index < prototype->enumeratorCount(); ++index)
696 addEnumerator(prototype: prototype->enumerator(index));
697 }
698
699 if ((members & ClassInfos) != 0) {
700 for (index = prototype->classInfoOffset(); index < prototype->classInfoCount(); ++index) {
701 QMetaClassInfo ci = prototype->classInfo(index);
702 addClassInfo(name: ci.name(), value: ci.value());
703 }
704 }
705
706 if ((members & RelatedMetaObjects) != 0) {
707 Q_ASSERT(qmobPriv(prototype->d.data)->revision >= 2);
708 const auto *objects = prototype->d.relatedMetaObjects;
709 if (objects) {
710 while (*objects != nullptr) {
711 addRelatedMetaObject(meta: *objects);
712 ++objects;
713 }
714 }
715 }
716
717 if ((members & StaticMetacall) != 0) {
718 Q_ASSERT(qmobPriv(prototype->d.data)->revision >= 6);
719 if (prototype->d.static_metacall)
720 setStaticMetacallFunction(prototype->d.static_metacall);
721 }
722}

1079static void writeString(char *out, int i, const QByteArray &str,
1080 const int offsetOfStringdataMember, int &stringdataOffset)
1081{
1082 int size = str.size();
1083 int offset = offsetOfStringdataMember + stringdataOffset;
1084 uint offsetLen[2] = { uint(offset), uint(size) };
1085
1086 memcpy(dest: out + 2 * i * sizeof(uint), src: &offsetLen, n: 2 * sizeof(uint));
1087

1089 out[offset + size] = '\0';
1090
1091 stringdataOffset += size + 1;

1095// The struct consists of an array of QByteArrayData, followed by a char array

1098void QMetaStringTable::writeBlob(char *out) const
1099{
1100 Q_ASSERT(!(reinterpret_cast<quintptr>(out) & (preferredAlignment() - 1)));

1103 int stringdataOffset = 0;

1109 it != end; ++it) {

1119// Returns the sum of all parameters (including return type) for the given
1120// \a methods. This is needed for calculating the size of the methods'
1121// parameter type/name meta-data.
1122static int aggregateParameterCount(const std::vector<QMetaMethodBuilderPrivate> &methods)
1123{

1125 for (const auto &method : methods)
1126 sum += method.parameterCount() + 1; // +1 for return type
1127 return sum;

1130enum Mode {
1131 Prepare, // compute the size of the metaobject
1132 Construct // construct metaobject in pre-allocated buffer
1133};
1134// Build a QMetaObject in "buf" based on the information in "d".
1135// If the mode is prepare, then return the number of bytes needed to
1136// build the QMetaObject.
1137template<Mode mode>
1138static int buildMetaObject(QMetaObjectBuilderPrivate *d, char *buf,
1139 int expectedSize)
1140{
1141 Q_UNUSED(expectedSize); // Avoid warning in release mode
1142 Q_UNUSED(buf);
1143 qsizetype size = 0;
1144 int dataIndex;
1145 int paramsIndex;
1146 int enumIndex;
1147 int index;
1148 bool hasRevisionedMethods = d->hasRevisionedMethods();
1149
1150 // Create the main QMetaObject structure at the start of the buffer.
1151 QMetaObject *meta = reinterpret_cast<QMetaObject *>(buf);
1152 size += sizeof(QMetaObject);
1153 ALIGN(size, int);
1154 if constexpr (mode == Construct) {
1155 meta->d.superdata = d->superClass;
1156 meta->d.relatedMetaObjects = nullptr;
1157 meta->d.extradata = nullptr;
1158 meta->d.metaTypes = nullptr;
1159 meta->d.static_metacall = d->staticMetacallFunction;
1160 }
1161
1162 // Populate the QMetaObjectPrivate structure.
1163 QMetaObjectPrivate *pmeta
1164 = reinterpret_cast<QMetaObjectPrivate *>(buf + size);
1165 //int pmetaSize = size;
1166 dataIndex = MetaObjectPrivateFieldCount;
1167 int methodParametersDataSize =
1168 ((aggregateParameterCount(methods: d->methods)
1169 + aggregateParameterCount(methods: d->constructors)) * 2) // types and parameter names
1170 - int(d->methods.size()) // return "parameters" don't have names
1171 - int(d->constructors.size()); // "this" parameters don't have names
1172 if constexpr (mode == Construct) {
1173 static_assert(QMetaObjectPrivate::OutputRevision == 11, "QMetaObjectBuilder should generate the same version as moc");
1174 pmeta->revision = QMetaObjectPrivate::OutputRevision;
1175 pmeta->flags = d->flags.toInt();
1176 pmeta->className = 0; // Class name is always the first string.
1177 //pmeta->signalCount is handled in the "output method loop" as an optimization.
1178
1179 pmeta->classInfoCount = d->classInfoNames.size();
1180 pmeta->classInfoData = dataIndex;
1181 dataIndex += 2 * d->classInfoNames.size();
1182
1183 pmeta->methodCount = int(d->methods.size());
1184 pmeta->methodData = dataIndex;
1185 dataIndex += QMetaObjectPrivate::IntsPerMethod * int(d->methods.size());
1186 if (hasRevisionedMethods)
1187 dataIndex += int(d->methods.size());
1188 paramsIndex = dataIndex;
1189 dataIndex += methodParametersDataSize;
1190
1191 pmeta->propertyCount = int(d->properties.size());
1192 pmeta->propertyData = dataIndex;
1193 dataIndex += QMetaObjectPrivate::IntsPerProperty * int(d->properties.size());
1194
1195 pmeta->enumeratorCount = int(d->enumerators.size());
1196 pmeta->enumeratorData = dataIndex;
1197 dataIndex += QMetaObjectPrivate::IntsPerEnum * int(d->enumerators.size());
1198
1199 pmeta->constructorCount = int(d->constructors.size());
1200 pmeta->constructorData = dataIndex;
1201 dataIndex += QMetaObjectPrivate::IntsPerMethod * int(d->constructors.size());
1202 } else {
1203 dataIndex += 2 * int(d->classInfoNames.size());
1204 dataIndex += QMetaObjectPrivate::IntsPerMethod * int(d->methods.size());
1205 if (hasRevisionedMethods)
1206 dataIndex += int(d->methods.size());
1207 paramsIndex = dataIndex;
1208 dataIndex += methodParametersDataSize;
1209 dataIndex += QMetaObjectPrivate::IntsPerProperty * int(d->properties.size());
1210 dataIndex += QMetaObjectPrivate::IntsPerEnum * int(d->enumerators.size());
1211 dataIndex += QMetaObjectPrivate::IntsPerMethod * int(d->constructors.size());
1212 }
1213
1214 // Allocate space for the enumerator key names and values.
1215 enumIndex = dataIndex;
1216 for (const auto &enumerator : d->enumerators)
1217 dataIndex += 2 * enumerator.keys.size();
1218
1219 // Zero terminator at the end of the data offset table.
1220 ++dataIndex;
1221
1222 // Find the start of the data and string tables.
1223 int *data = reinterpret_cast<int *>(pmeta);
1224 size += dataIndex * sizeof(int);
1225 ALIGN(size, void *);
1226 [[maybe_unused]] char *str = reinterpret_cast<char *>(buf + size);
1227 if constexpr (mode == Construct) {
1228 meta->d.stringdata = reinterpret_cast<const uint *>(str);
1229 meta->d.data = reinterpret_cast<uint *>(data);
1230 }
1231
1232 // Reset the current data position to just past the QMetaObjectPrivate.
1233 dataIndex = MetaObjectPrivateFieldCount;
1234
1235 QMetaStringTable strings(d->className);
1236
1237 // Output the class infos,
1238 Q_ASSERT(!buf || dataIndex == pmeta->classInfoData);
1239 for (index = 0; index < d->classInfoNames.size(); ++index) {
1240 [[maybe_unused]] int name = strings.enter(value: d->classInfoNames[index]);
1241 [[maybe_unused]] int value = strings.enter(value: d->classInfoValues[index]);
1242 if constexpr (mode == Construct) {
1243 data[dataIndex] = name;
1244 data[dataIndex + 1] = value;
1245 }
1246 dataIndex += 2;
1247 }
1248
1249 // Output the methods in the class.
1250 Q_ASSERT(!buf || dataIndex == pmeta->methodData);
1251 // + 1 for metatype of this metaobject
1252 int parameterMetaTypesIndex = int(d->properties.size()) + 1;
1253 for (const auto &method : d->methods) {
1254 [[maybe_unused]] int name = strings.enter(value: method.name());
1255 int argc = method.parameterCount();
1256 [[maybe_unused]] int tag = strings.enter(value: method.tag);
1257 [[maybe_unused]] int attrs = method.attributes;
1258 if constexpr (mode == Construct) {
1259 data[dataIndex] = name;
1260 data[dataIndex + 1] = argc;
1261 data[dataIndex + 2] = paramsIndex;
1262 data[dataIndex + 3] = tag;
1263 data[dataIndex + 4] = attrs;
1264 data[dataIndex + 5] = parameterMetaTypesIndex;
1265 if (method.methodType() == QMetaMethod::Signal)
1266 pmeta->signalCount++;
1267 }
1268 dataIndex += QMetaObjectPrivate::IntsPerMethod;
1269 paramsIndex += 1 + argc * 2;
1270 parameterMetaTypesIndex += 1 + argc;
1271 }
1272 if (hasRevisionedMethods) {
1273 for (const auto &method : d->methods) {
1274 if constexpr (mode == Construct)
1275 data[dataIndex] = method.revision;
1276 ++dataIndex;
1277 }
1278 }
1279
1280 // Output the method parameters in the class.
1281 Q_ASSERT(!buf || dataIndex == pmeta->methodData + int(d->methods.size()) * QMetaObjectPrivate::IntsPerMethod
1282 + (hasRevisionedMethods ? int(d->methods.size()) : 0));
1283 for (int x = 0; x < 2; ++x) {
1284 const std::vector<QMetaMethodBuilderPrivate> &methods = (x == 0) ? d->methods : d->constructors;
1285 for (const auto &method : methods) {
1286 const QList<QByteArray> paramTypeNames = method.parameterTypes();
1287 int paramCount = paramTypeNames.size();
1288 for (int i = -1; i < paramCount; ++i) {
1289 const QByteArray &typeName = (i < 0) ? method.returnType : paramTypeNames.at(i);
1290 [[maybe_unused]] int typeInfo;
1291 if (QtPrivate::isBuiltinType(type: typeName))
1292 typeInfo = QMetaType::fromName(name: typeName).id();
1293 else
1294 typeInfo = IsUnresolvedType | strings.enter(value: typeName);
1295 if constexpr (mode == Construct)
1296 data[dataIndex] = typeInfo;
1297 ++dataIndex;
1298 }
1299
1300 QList<QByteArray> paramNames = method.parameterNames;
1301 while (paramNames.size() < paramCount)
1302 paramNames.append(t: QByteArray());
1303 for (int i = 0; i < paramCount; ++i) {
1304 [[maybe_unused]] int stringIndex = strings.enter(value: paramNames.at(i));
1305 if constexpr (mode == Construct)
1306 data[dataIndex] = stringIndex;
1307 ++dataIndex;
1308 }
1309 }
1310 }
1311
1312 // Output the properties in the class.
1313 Q_ASSERT(!buf || dataIndex == pmeta->propertyData);
1314 for (QMetaPropertyBuilderPrivate &prop : d->properties) {
1315 [[maybe_unused]] int name = strings.enter(value: prop.name);
1316
1317 // try to resolve the metatype again if it was unknown
1318 if (!prop.metaType.isValid())
1319 prop.metaType = QMetaType::fromName(name: prop.type);
1320 [[maybe_unused]] const int typeInfo = prop.metaType.isValid()
1321 ? prop.metaType.id()
1322 : IsUnresolvedType | strings.enter(value: prop.type);
1323
1324 [[maybe_unused]] int flags = prop.flags;
1325
1326 if (!QtPrivate::isBuiltinType(type: prop.type))
1327 flags |= EnumOrFlag;
1328
1329 if constexpr (mode == Construct) {
1330 data[dataIndex] = name;
1331 data[dataIndex + 1] = typeInfo;
1332 data[dataIndex + 2] = flags;
1333 data[dataIndex + 3] = prop.notifySignal;
1334 data[dataIndex + 4] = prop.revision;
1335 }
1336 dataIndex += QMetaObjectPrivate::IntsPerProperty;
1337 }
1338
1339 // Output the enumerators in the class.
1340 Q_ASSERT(!buf || dataIndex == pmeta->enumeratorData);
1341 for (const auto &enumerator : d->enumerators) {
1342 [[maybe_unused]] int name = strings.enter(value: enumerator.name);
1343 [[maybe_unused]] int enumName = strings.enter(value: enumerator.enumName);
1344 [[maybe_unused]] int isFlag = enumerator.isFlag ? EnumIsFlag : 0;
1345 [[maybe_unused]] int isScoped = enumerator.isScoped ? EnumIsScoped : 0;
1346 int count = enumerator.keys.size();
1347 int enumOffset = enumIndex;
1348 if constexpr (mode == Construct) {
1349 data[dataIndex] = name;
1350 data[dataIndex + 1] = enumName;
1351 data[dataIndex + 2] = isFlag | isScoped;
1352 data[dataIndex + 3] = count;
1353 data[dataIndex + 4] = enumOffset;
1354 }
1355 for (int key = 0; key < count; ++key) {
1356 [[maybe_unused]] int keyIndex = strings.enter(value: enumerator.keys[key]);
1357 if constexpr (mode == Construct) {
1358 data[enumOffset++] = keyIndex;
1359 data[enumOffset++] = enumerator.values[key];
1360 }
1361 }
1362 dataIndex += QMetaObjectPrivate::IntsPerEnum;
1363 enumIndex += 2 * count;
1364 }
1365
1366 // Output the constructors in the class.
1367 Q_ASSERT(!buf || dataIndex == pmeta->constructorData);
1368 for (const auto &ctor : d->constructors) {
1369 [[maybe_unused]] int name = strings.enter(value: ctor.name());
1370 int argc = ctor.parameterCount();
1371 [[maybe_unused]] int tag = strings.enter(value: ctor.tag);
1372 [[maybe_unused]] int attrs = ctor.attributes;
1373 if constexpr (mode == Construct) {
1374 data[dataIndex] = name;
1375 data[dataIndex + 1] = argc;
1376 data[dataIndex + 2] = paramsIndex;
1377 data[dataIndex + 3] = tag;
1378 data[dataIndex + 4] = attrs;
1379 data[dataIndex + 5] = parameterMetaTypesIndex;
1380 }
1381 dataIndex += QMetaObjectPrivate::IntsPerMethod;
1382 paramsIndex += 1 + argc * 2;
1383 parameterMetaTypesIndex += argc;
1384 }
1385
1386 size += strings.blobSize();
1387
1388 if constexpr (mode == Construct)
1389 strings.writeBlob(out: str);
1390
1391 // Output the zero terminator in the data array.
1392 if constexpr (mode == Construct)
1393 data[enumIndex] = 0;
1394
1395 // Create the relatedMetaObjects block if we need one.
1396 if (d->relatedMetaObjects.size() > 0) {
1397 using SuperData = QMetaObject::SuperData;
1398 ALIGN(size, SuperData);
1399 auto objects = reinterpret_cast<SuperData *>(buf + size);
1400 if constexpr (mode == Construct) {
1401 meta->d.relatedMetaObjects = objects;
1402 for (index = 0; index < d->relatedMetaObjects.size(); ++index)
1403 objects[index] = d->relatedMetaObjects[index];
1404 objects[index] = nullptr;
1405 }
1406 size += sizeof(SuperData) * (d->relatedMetaObjects.size() + 1);
1407 }
1408
1409 ALIGN(size, QtPrivate::QMetaTypeInterface *);
1410 auto types = reinterpret_cast<QtPrivate::QMetaTypeInterface **>(buf + size);
1411 if constexpr (mode == Construct) {
1412 meta->d.metaTypes = types;
1413 for (const auto &prop : d->properties) {
1414 QMetaType mt = prop.metaType;
1415 *types = reinterpret_cast<QtPrivate::QMetaTypeInterface *&>(mt);
1416 types++;
1417 }
1418 // add metatype interface for this metaobject - must be null
1419 // as we can't know our metatype
1420 *types = nullptr;
1421 types++;
1422 for (const auto &method: d->methods) {
1423 QMetaType mt(QMetaType::fromName(name: method.returnType).id());
1424 *types = reinterpret_cast<QtPrivate::QMetaTypeInterface *&>(mt);
1425 types++;
1426 for (const auto &parameterType: method.parameterTypes()) {
1427 QMetaType mt = QMetaType::fromName(name: parameterType);
1428 *types = reinterpret_cast<QtPrivate::QMetaTypeInterface *&>(mt);
1429 types++;
1430 }
1431 }
1432 for (const auto &constructor : d->constructors) {
1433 for (const auto &parameterType : constructor.parameterTypes()) {
1434 QMetaType mt = QMetaType::fromName(name: parameterType);
1435 *types = reinterpret_cast<QtPrivate::QMetaTypeInterface *&>(mt);
1436 types++;
1437 }
1438 }
1439 }
1440 // parameterMetaTypesIndex is equal to the total number of metatypes
1441 size += sizeof(QMetaType) * parameterMetaTypesIndex;
1442
1443 // Align the final size and return it.
1444 ALIGN(size, void *);
