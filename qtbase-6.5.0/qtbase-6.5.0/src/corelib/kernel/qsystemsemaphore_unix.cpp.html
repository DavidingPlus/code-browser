24void QSystemSemaphorePrivate::setErrorString(const QString &function)
25{
26 // EINVAL is handled in functions so they can give better error strings
27 switch (errno) {
28 case EPERM:
29 case EACCES:
30 errorString = QSystemSemaphore::tr(sourceText: "%1: permission denied").arg(a: function);
31 error = QSystemSemaphore::PermissionDenied;
32 break;
33 case EEXIST:
34 errorString = QSystemSemaphore::tr(sourceText: "%1: already exists").arg(a: function);
35 error = QSystemSemaphore::AlreadyExists;
36 break;
37 case ENOENT:
38 errorString = QSystemSemaphore::tr(sourceText: "%1: does not exist").arg(a: function);
39 error = QSystemSemaphore::NotFound;
40 break;
41 case ERANGE:
42 case ENOSPC:
43 errorString = QSystemSemaphore::tr(sourceText: "%1: out of resources").arg(a: function);
44 error = QSystemSemaphore::OutOfResources;
45 break;
46#if defined(QT_POSIX_IPC)
47 case ENAMETOOLONG:
48 errorString = QSystemSemaphore::tr("%1: key too long").arg(function);
49 error = QSystemSemaphore::KeyError;
50 break;
51#endif
52 default:
53 errorString = QSystemSemaphore::tr(sourceText: "%1: unknown error %2").arg(a: function).arg(errno);
54 error = QSystemSemaphore::UnknownError;
55#if defined QSYSTEMSEMAPHORE_DEBUG
56 qDebug() << errorString << "key" << key << "errno" << errno << EINVAL;
57#endif
58 }
59}
