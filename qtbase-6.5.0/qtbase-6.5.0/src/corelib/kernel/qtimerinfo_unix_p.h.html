28// internal timer info
29struct QTimerInfo {
30 int id; // - timer identifier
31 Qt::TimerType timerType; // - timer type
32 qint64 interval; // - timer interval in milliseconds
33 timespec timeout; // - when to actually fire
34 QObject *obj; // - object to receive event
35 QTimerInfo **activateRef; // - ref from activateTimers
36
37#ifdef QTIMERINFO_DEBUG
38 timeval expected; // when timer is expected to fire
39 float cumulativeError;
40 uint count;
41#endif
42};

44class Q_CORE_EXPORT QTimerInfoList : public QList<QTimerInfo*>
45{
46#if ((_POSIX_MONOTONIC_CLOCK-0 <= 0) && !defined(Q_OS_MAC)) || defined(QT_BOOTSTRAPPED)
47 timespec previousTime;
48 clock_t previousTicks;
49 int ticksPerSecond;
50 int msPerTick;
51
52 bool timeChanged(timespec *delta);
53 void timerRepair(const timespec &);
54#endif
55
56 // state variables used by activateTimers()
57 QTimerInfo *firstTimerInfo;
58
59public:
60 QTimerInfoList();
61
62 timespec currentTime;
63 timespec updateCurrentTime();
64
65 // must call updateCurrentTime() first!
66 void repairTimersIfNeeded();
67
68 bool timerWait(timespec &);
69 void timerInsert(QTimerInfo *);
70
71 qint64 timerRemainingTime(int timerId);
72
73 void registerTimer(int timerId, qint64 interval, Qt::TimerType timerType, QObject *object);
74 bool unregisterTimer(int timerId);
75 bool unregisterTimers(QObject *object);
76 QList<QAbstractEventDispatcher::TimerInfo> registeredTimers(QObject *object) const;
77
78 int activateTimers();
79};
