60static qlonglong qMetaTypeNumberBySize(const QVariant::Private *d)
61{
62 switch (d->typeInterface()->size) {
63 case 1:
64 return d->get<signed char>();
65 case 2:
66 return d->get<short>();
67 case 4:
68 return d->get<int>();
69 case 8:
70 return d->get<qlonglong>();
71 }
72 Q_UNREACHABLE_RETURN(0);
73}

75static qlonglong qMetaTypeNumber(const QVariant::Private *d)
76{
77 switch (d->typeInterface()->typeId) {
78 case QMetaType::Int:
79 case QMetaType::LongLong:
80 case QMetaType::Char:
81 case QMetaType::SChar:
82 case QMetaType::Short:
83 case QMetaType::Long:
84 return qMetaTypeNumberBySize(d);
85 case QMetaType::Float:
86 return qRound64(f: d->get<float>());
87 case QMetaType::Double:
88 return qRound64(d: d->get<double>());
89#ifndef QT_BOOTSTRAPPED
90 case QMetaType::QJsonValue:
91 return d->get<QJsonValue>().toDouble();
92 case QMetaType::QCborValue:
93 return d->get<QCborValue>().toInteger();
94#endif
95 }
96 Q_UNREACHABLE_RETURN(0);
97}

114static std::optional<qlonglong> qConvertToNumber(const QVariant::Private *d, bool allowStringToBool = false)
115{
116 bool ok;
117 switch (d->typeInterface()->typeId) {
118 case QMetaType::QString: {
119 const QString &s = d->get<QString>();
120 if (qlonglong l = s.toLongLong(ok: &ok); ok)
121 return l;
122 if (allowStringToBool) {
123 if (s == "false"_L1 || s == "0"_L1)
124 return 0;
125 if (s == "true"_L1 || s == "1"_L1)
126 return 1;
127 }
128 return std::nullopt;
129 }
130 case QMetaType::QChar:
131 return d->get<QChar>().unicode();
132 case QMetaType::QByteArray:
133 if (qlonglong l = d->get<QByteArray>().toLongLong(ok: &ok); ok)
134 return l;
135 return std::nullopt;
136 case QMetaType::Bool:
137 return qlonglong(d->get<bool>());
138#ifndef QT_BOOTSTRAPPED
139 case QMetaType::QCborValue:
140 if (!d->get<QCborValue>().isInteger() && !d->get<QCborValue>().isDouble())
141 break;
142 return qMetaTypeNumber(d);
143 case QMetaType::QJsonValue:
144 if (!d->get<QJsonValue>().isDouble())
145 break;
146 Q_FALLTHROUGH();
147#endif
148 case QMetaType::Double:
149 case QMetaType::Int:
150 case QMetaType::Char:
151 case QMetaType::SChar:
152 case QMetaType::Short:
153 case QMetaType::Long:
154 case QMetaType::Float:
155 case QMetaType::LongLong:
156 return qMetaTypeNumber(d);
157 case QMetaType::ULongLong:
158 case QMetaType::UInt:
159 case QMetaType::UChar:
160 case QMetaType::Char16:
161 case QMetaType::Char32:
162 case QMetaType::UShort:
163 case QMetaType::ULong:
164 return qlonglong(qMetaTypeUNumber(d));
165 }
166
167 if (d->typeInterface()->flags & QMetaType::IsEnumeration
168 || d->typeInterface()->typeId == QMetaType::QCborSimpleType)
169 return qMetaTypeNumberBySize(d);
170
171 return std::nullopt;
172}

174static std::optional<qreal> qConvertToRealNumber(const QVariant::Private *d)
175{
176 bool ok;
177 switch (d->typeInterface()->typeId) {
178 case QMetaType::QString:
179 if (double r = d->get<QString>().toDouble(ok: &ok); ok)
180 return qreal(r);
181 return std::nullopt;
182 case QMetaType::Double:
183 return qreal(d->get<double>());
184 case QMetaType::Float:
185 return qreal(d->get<float>());
186 case QMetaType::Float16:
187 return qreal(d->get<qfloat16>());
188 case QMetaType::ULongLong:
189 case QMetaType::UInt:
190 case QMetaType::UChar:
191 case QMetaType::Char16:
192 case QMetaType::Char32:
193 case QMetaType::UShort:
194 case QMetaType::ULong:
195 return qreal(qMetaTypeUNumber(d));
196#ifndef QT_BOOTSTRAPPED
197 case QMetaType::QCborValue:
198 return d->get<QCborValue>().toDouble();
199 case QMetaType::QJsonValue:
200 return d->get<QJsonValue>().toDouble();
201#endif
202 default:
203 // includes enum conversion as well as invalid types
204 if (std::optional<qlonglong> l = qConvertToNumber(d))
205 return qreal(*l);
206 return std::nullopt;
207 }

210static bool isValidMetaTypeForVariant(const QtPrivate::QMetaTypeInterface *iface, const void *copy)
211{
212 using namespace QtMetaTypePrivate;
213 if (!iface || iface->size == 0)
214 return false;
215
216 Q_ASSERT(!isInterfaceFor<void>(iface)); // only void should have size 0
217 if (!isCopyConstructible(iface) || !isDestructible(iface)) {
218 // all meta types must be copyable (because QVariant is) and
219 // destructible (because QVariant owns it)
220 qWarning(msg: "QVariant: Provided metatype for '%s' does not support destruction and "
221 "copy construction", iface->name);
222 return false;
223 }
224 if (!copy && !isDefaultConstructible(iface)) {
225 // non-default-constructible types are acceptable, but not if you're
226 // asking us to construct from nothing
227 qWarning(msg: "QVariant: Cannot create type '%s' without a default constructor", iface->name);
228 return false;
229 }
230
231 return true;
232}

234template <typename F> static QVariant::PrivateShared *
235customConstructShared(size_t size, size_t align, F &&construct)
236{
237 struct Deleter {
238 void operator()(QVariant::PrivateShared *p) const
239 { QVariant::PrivateShared::free(p); }
240 };
241
242 // this is exception-safe
243 std::unique_ptr<QVariant::PrivateShared, Deleter> ptr;
244 ptr.reset(QVariant::PrivateShared::create(size, align));
245 construct(ptr->data());
246 return ptr.release();
247}

249// the type of d has already been set, but other field are not set
250static void customConstruct(const QtPrivate::QMetaTypeInterface *iface, QVariant::Private *d,
251 const void *copy)
252{
253 using namespace QtMetaTypePrivate;
254 Q_ASSERT(iface);
255 Q_ASSERT(iface->size);
256 Q_ASSERT(!isInterfaceFor<void>(iface));
257 Q_ASSERT(isCopyConstructible(iface));
258 Q_ASSERT(isDestructible(iface));
259 Q_ASSERT(copy || isDefaultConstructible(iface));
260
261 // need to check for nullptr_t here, as this can get called by fromValue(nullptr). fromValue() uses
262 // std::addressof(value) which in this case returns the address of the nullptr object.
263 // ### Qt 7: remove nullptr_t special casing
264 d->is_null = !copy QT6_ONLY(|| isInterfaceFor<std::nullptr_t>(iface));
265
266 if (QVariant::Private::canUseInternalSpace(type: iface)) {
267 d->is_shared = false;
268 if (!copy && !iface->defaultCtr)
269 return; // default constructor and it's OK to build in 0-filled storage, which we've already done
270 construct(iface, where: d->data.data, copy);
271 } else {
272 d->data.shared = customConstructShared(size: iface->size, align: iface->alignment, construct: [=](void *where) {
273 construct(iface, where, copy);
274 });
275 d->is_shared = true;
276 }
277}

279static void customClear(QVariant::Private *d)
280{
281 const QtPrivate::QMetaTypeInterface *iface = d->typeInterface();
282 if (!iface)
283 return;
284 if (!d->is_shared) {
285 QtMetaTypePrivate::destruct(iface, where: d->data.data);
286 } else {
287 QtMetaTypePrivate::destruct(iface, where: d->data.shared->data());
288 QVariant::PrivateShared::free(p: d->data.shared);
289 }

292static QVariant::Private clonePrivate(const QVariant::Private &other)
293{
294 QVariant::Private d = other;
295 if (d.is_shared) {
296 d.data.shared->ref.ref();
297 } else if (const QtPrivate::QMetaTypeInterface *iface = d.typeInterface()) {
298 Q_ASSERT(d.canUseInternalSpace(iface));
299
300 // if not trivially copyable, ask to copy
301 if (iface->copyCtr)
302 QtMetaTypePrivate::copyConstruct(iface, where: d.data.data, copy: other.data.data);
303 }
304 return d;
305}

309inline QVariant::PrivateShared *QVariant::PrivateShared::create(size_t size, size_t align)
310{
311 size += sizeof(PrivateShared);
312 if (align > sizeof(PrivateShared)) {
313 // The alignment is larger than the alignment we can guarantee for the pointer
314 // directly following PrivateShared, so we need to allocate some additional
315 // memory to be able to fit the object into the available memory with suitable
316 // alignment.
317 size += align - sizeof(PrivateShared);
318 }
319 void *data = operator new(size);
320 auto *ps = new (data) QVariant::PrivateShared();
321 ps->offset = int(((quintptr(ps) + sizeof(PrivateShared) + align - 1) & ~(align - 1)) - quintptr(ps));
322 return ps;
323}

327 p->~PrivateShared();
328 operator delete(p);
329}

337template <typename T> inline
338QVariant::Private::Private(std::piecewise_construct_t, const T &t)
339 : is_shared(!CanUseInternalSpace<T>), is_null(std::is_same_v<T, std::nullptr_t>)
340{
341 // confirm noexceptness
342 static constexpr bool isNothrowQVariantConstructible = noexcept(QVariant(t));
343 static constexpr bool isNothrowCopyConstructible = std::is_nothrow_copy_constructible_v<T>;
344 static constexpr bool isNothrowCopyAssignable = std::is_nothrow_copy_assignable_v<T>;
345
346 const QtPrivate::QMetaTypeInterface *iface = QtPrivate::qMetaTypeInterfaceForType<T>();
347 Q_ASSERT((quintptr(iface) & 0x3) == 0);
348 packedType = quintptr(iface) >> 2;
349
350 if constexpr (CanUseInternalSpace<T>) {
351 static_assert(isNothrowQVariantConstructible == isNothrowCopyConstructible);
352 static_assert(isNothrowQVariantConstructible == isNothrowCopyAssignable);
353 new (data.data) T(t);
354 } else {
355 static_assert(!isNothrowQVariantConstructible); // we allocate memory, even if T doesn't
356 data.shared = customConstructShared(sizeof(T), alignof(T), [=](void *where) {
357 new (where) T(t);
358 });
359 }

362/*!
363 \class QVariant
364 \inmodule QtCore
365 \brief The QVariant class acts like a union for the most common Qt data types.
366
367 \ingroup objectmodel
368 \ingroup shared
369
370
371 Because C++ forbids unions from including types that have
372 non-default constructors or destructors, most interesting Qt
373 classes cannot be used in unions. Without QVariant, this would be
374 a problem for QObject::property() and for database work, etc.
375
376 A QVariant object holds a single value of a single typeId() at a
377 time. (Some types are multi-valued, for example a string list.)
378 You can find out what type, T, the variant holds, convert it to a
379 different type using convert(), get its value using one of the
380 toT() functions (e.g., toSize()), and check whether the type can
381 be converted to a particular type using canConvert().
382
383 The methods named toT() (e.g., toInt(), toString()) are const. If
384 you ask for the stored type, they return a copy of the stored
385 object. If you ask for a type that can be generated from the
386 stored type, toT() copies and converts and leaves the object
387 itself unchanged. If you ask for a type that cannot be generated
388 from the stored type, the result depends on the type; see the
389 function documentation for details.
390
391 Here is some example code to demonstrate the use of QVariant:
392
393 \snippet code/src_corelib_kernel_qvariant.cpp 0
394
395 You can even store QList<QVariant> and QMap<QString, QVariant>
396 values in a variant, so you can easily construct arbitrarily
397 complex data structures of arbitrary types. This is very powerful
398 and versatile, but may prove less memory and speed efficient than
399 storing specific types in standard data structures.
400
401 QVariant also supports the notion of null values. A variant is null
402 if the variant contains no initialized value, or contains a null pointer.
403
404 \snippet code/src_corelib_kernel_qvariant.cpp 1
405
406 QVariant can be extended to support other types than those
407 mentioned in the \l QMetaType::Type enum.
408 See \l{Creating Custom Qt Types}{Creating Custom Qt Types} for details.
409
410 \section1 A Note on GUI Types
411
412 Because QVariant is part of the Qt Core module, it cannot provide
413 conversion functions to data types defined in Qt GUI, such as
414 QColor, QImage, and QPixmap. In other words, there is no \c
415 toColor() function. Instead, you can use the QVariant::value() or
416 the qvariant_cast() template function. For example:
417
418 \snippet code/src_corelib_kernel_qvariant.cpp 2
419
420 The inverse conversion (e.g., from QColor to QVariant) is
421 automatic for all data types supported by QVariant, including
422 GUI-related types:
423
424 \snippet code/src_corelib_kernel_qvariant.cpp 3
425
426 \section1 Using canConvert() and convert() Consecutively
427
428 When using canConvert() and convert() consecutively, it is possible for
429 canConvert() to return true, but convert() to return false. This
430 is typically because canConvert() only reports the general ability of
431 QVariant to convert between types given suitable data; it is still
432 possible to supply data which cannot actually be converted.
433
434 For example, \c{canConvert(QMetaType::fromType<int>())} would return true
435 when called on a variant containing a string because, in principle,
436 QVariant is able to convert strings of numbers to integers.
437 However, if the string contains non-numeric characters, it cannot be
438 converted to an integer, and any attempt to convert it will fail.
439 Hence, it is important to have both functions return true for a
440 successful conversion.
441
442 \sa QMetaType
443*/

1155/*!
1156 Internal function for loading a variant from stream \a s. Use the
1157 stream operators instead.
1158
1159 \internal
1160*/
1161void QVariant::load(QDataStream &s)
1162{
1163 clear();
1164
1165 quint32 typeId;
1166 s >> typeId;
1167 if (s.version() < QDataStream::Qt_4_0) {
1168 // map to Qt 5 ids
1169 if (typeId >= MapFromThreeCount)
1170 return;
1171 typeId = mapIdFromQt3ToCurrent[typeId];
1172 } else if (s.version() < QDataStream::Qt_5_0) {
1173 // map to Qt 5 type ids
1174 if (typeId == 127 /* QVariant::UserType */) {
1175 typeId = Qt5UserType;
1176 } else if (typeId >= 128 && typeId != Qt5UserType) {
1177 // In Qt4 id == 128 was FirstExtCoreType. In Qt5 ExtCoreTypes set was merged to CoreTypes
1178 // by moving all ids down by 97.
1179 typeId -= 97;
1180 } else if (typeId == 75 /* QSizePolicy */) {
1181 typeId = Qt5SizePolicy;
1182 } else if (typeId > 75 && typeId <= 86) {
1183 // and as a result these types received lower ids too
1184 // QKeySequence QPen QTextLength QTextFormat QTransform QMatrix4x4 QVector2D QVector3D QVector4D QQuaternion
1185 typeId -=1;
1186 }
1187 }
1188 if (s.version() < QDataStream::Qt_6_0) {
1189 // map from Qt 5 to Qt 6 values
1190 if (typeId == Qt5UserType) {
1191 typeId = QMetaType::User;
1192 } else if (typeId >= Qt5FirstGuiType && typeId <= Qt5LastGuiType) {
1193 typeId += Qt6ToQt5GuiTypeDelta;
1194 } else if (typeId == Qt5SizePolicy) {
1195 typeId = QMetaType::QSizePolicy;
1196 } else if (typeId == Qt5RegExp) {
1197 typeId = QMetaType::fromName("QRegExp").id();
1198 }
1199 }
1200
1201 qint8 is_null = false;
1202 if (s.version() >= QDataStream::Qt_4_2)
1203 s >> is_null;
1204 if (typeId == QMetaType::User) {
1205 QByteArray name;
1206 s >> name;
1207 typeId = QMetaType::fromName(name).id();
1208 if (typeId == QMetaType::UnknownType) {
1209 s.setStatus(QDataStream::ReadCorruptData);
1210 qWarning("QVariant::load: unknown user type with name %s.", name.constData());
1211 return;
1212 }
1213 }
1214 create(typeId, nullptr);
1215 d.is_null = is_null;
1216
1217 if (!isValid()) {
1218 if (s.version() < QDataStream::Qt_5_0) {
1219 // Since we wrote something, we should read something
1220 QString x;
1221 s >> x;
1222 }
1223 d.is_null = true;
1224 return;
1225 }
1226
1227 // const cast is safe since we operate on a newly constructed variant
1228 void *data = const_cast<void *>(constData());
1229 if (!d.type().load(s, data)) {
1230 s.setStatus(QDataStream::ReadCorruptData);
1231 qWarning("QVariant::load: unable to load type %d.", d.type().id());
1232 }
1233}

1235/*!
1236 Internal function for saving a variant to the stream \a s. Use the
1237 stream operators instead.
1238
1239 \internal
1240*/
1241void QVariant::save(QDataStream &s) const
1242{
1243 quint32 typeId = d.type().id();
1244 bool saveAsUserType = false;
1245 if (typeId >= QMetaType::User) {
1246 typeId = QMetaType::User;
1247 saveAsUserType = true;
1248 }
1249 if (s.version() < QDataStream::Qt_6_0) {
1250 // map to Qt 5 values
1251 if (typeId == QMetaType::User) {
1252 typeId = Qt5UserType;
1253 if (!strcmp(d.type().name(), "QRegExp")) {
1254 typeId = 27; // QRegExp in Qt 4/5
1255 }
1256 } else if (typeId > Qt5LastCoreType && typeId <= QMetaType::LastCoreType) {
1257 // the type didn't exist in Qt 5
1258 typeId = Qt5UserType;
1259 saveAsUserType = true;
1260 } else if (typeId >= QMetaType::FirstGuiType && typeId <= QMetaType::LastGuiType) {
1261 typeId -= Qt6ToQt5GuiTypeDelta;
1262 if (typeId > Qt5LastGuiType) {
1263 typeId = Qt5UserType;
1264 saveAsUserType = true;
1265 }
1266 } else if (typeId == QMetaType::QSizePolicy) {
1267 typeId = Qt5SizePolicy;
1268 }
1269 }
1270 if (s.version() < QDataStream::Qt_4_0) {
1271 int i;
1272 for (i = 0; i <= MapFromThreeCount - 1; ++i) {
1273 if (mapIdFromQt3ToCurrent[i] == typeId) {
1274 typeId = i;
1275 break;
1276 }
1277 }
1278 if (i >= MapFromThreeCount) {
1279 s << QVariant();
1280 return;
1281 }
1282 } else if (s.version() < QDataStream::Qt_5_0) {
1283 if (typeId == Qt5UserType) {
1284 typeId = 127; // QVariant::UserType had this value in Qt4
1285 saveAsUserType = true;
1286 } else if (typeId >= 128 - 97 && typeId <= Qt5LastCoreType) {
1287 // In Qt4 id == 128 was FirstExtCoreType. In Qt5 ExtCoreTypes set was merged to CoreTypes
1288 // by moving all ids down by 97.
1289 typeId += 97;
1290 } else if (typeId == Qt5SizePolicy) {
1291 typeId = 75;
1292 } else if (typeId >= Qt5KeySequence && typeId <= Qt5QQuaternion) {
1293 // and as a result these types received lower ids too
1294 typeId += 1;
1295 } else if (typeId > Qt5QQuaternion || typeId == QMetaType::QUuid) {
1296 // These existed in Qt 4 only as a custom type
1297 typeId = 127;
1298 saveAsUserType = true;
1299 }
1300 }
1301 const char *typeName = nullptr;
1302 if (saveAsUserType) {
1303 if (s.version() < QDataStream::Qt_6_0)
1304 typeName = QtMetaTypePrivate::typedefNameForType(d.type().d_ptr);
1305 if (!typeName)
1306 typeName = d.type().name();
1307 }
1308 s << typeId;
1309 if (s.version() >= QDataStream::Qt_4_2)
1310 s << qint8(d.is_null);
1311 if (typeName)
1312 s << typeName;
1313
1314 if (!isValid()) {
1315 if (s.version() < QDataStream::Qt_5_0)
1316 s << QString();
1317 return;
1318 }
1319
1320 if (!d.type().save(s, constData())) {
1321 qWarning("QVariant::save: unable to save type '%s' (type id: %d).\n",
1322 d.type().name(), d.type().id());
1323 Q_ASSERT_X(false, "QVariant::save", "Invalid type to save");
1324 }
1325}

2205static int numericTypePromotion(const QtPrivate::QMetaTypeInterface *iface1,
2206 const QtPrivate::QMetaTypeInterface *iface2)
2207{
2208 Q_ASSERT(canBeNumericallyCompared(iface1, iface2));
2209
2210 // We don't need QMetaType::id() here because the type Id is always stored
2211 // directly for the types we're comparing against below.
2212 uint t1 = iface1->typeId;
2213 uint t2 = iface2->typeId;
2214
2215 if ((t1 == QMetaType::Bool && t2 == QMetaType::QString) ||
2216 (t2 == QMetaType::Bool && t1 == QMetaType::QString))
2217 return QMetaType::Bool;
2218
2219 // C++ integral ranks: (4.13 Integer conversion rank [conv.rank])
2220 // bool < signed char < short < int < long < long long
2221 // unsigneds have the same rank as their signed counterparts
2222 // C++ integral promotion rules (4.5 Integral Promotions [conv.prom])
2223 // - any type with rank less than int can be converted to int or unsigned int
2224 // 5 Expressions [expr] paragraph 9:
2225 // - if either operand is double, the other shall be converted to double
2226 // - " " float, " " " float
2227 // - if both operands have the same type, no further conversion is needed.
2228 // - if both are signed or if both are unsigned, convert to the one with highest rank
2229 // - if the unsigned has higher or same rank, convert the signed to the unsigned one
2230 // - if the signed can represent all values of the unsigned, convert to the signed
2231 // - otherwise, convert to the unsigned corresponding to the rank of the signed
2232
2233 // floating point: we deviate from the C++ standard by always using qreal
2234 if (qIsFloatingPoint(tp: t1) || qIsFloatingPoint(tp: t2))
2235 return QMetaType::QReal;
2236
2237 auto isUnsigned = [](uint tp) {
2238 // only types for which sizeof(T) >= sizeof(int); lesser ones promote to int
2239 return tp == QMetaType::ULongLong || tp == QMetaType::ULong ||
2240 tp == QMetaType::UInt || tp == QMetaType::Char32;
2241 };
2242 bool isUnsigned1 = isUnsigned(t1);
2243 bool isUnsigned2 = isUnsigned(t2);
2244
2245 // integral rules:
2246 // 1) if either type is a 64-bit unsigned, compare as 64-bit unsigned
2247 if (isUnsigned1 && iface1->size > sizeof(int))
2248 return QMetaType::ULongLong;
2249 if (isUnsigned2 && iface2->size > sizeof(int))
2250 return QMetaType::ULongLong;
2251
2252 // 2) if either type is 64-bit, compare as 64-bit signed
2253 if (iface1->size > sizeof(int) || iface2->size > sizeof(int))
2254 return QMetaType::LongLong;
2255
2256 // 3) if either type is 32-bit unsigned, compare as 32-bit unsigned
2257 if (isUnsigned1 || isUnsigned2)
2258 return QMetaType::UInt;
2259
2260 // 4) otherwise, just do int promotion
2261 return QMetaType::Int;
2262}

2264template <typename Numeric> static QPartialOrdering spaceShip(Numeric lhs, Numeric rhs)
2265{
2266 if (lhs == rhs)
2267 return QPartialOrdering::Equivalent;
2268 if constexpr (std::numeric_limits<Numeric>::has_quiet_NaN) {
2269 if (std::isnan(lhs) || std::isnan(rhs))
2270 return QPartialOrdering::Unordered;
2271 }
2272
2273 bool smaller;
2274 if constexpr (std::is_same_v<Numeric, QObject *>)
2275 smaller = std::less<QObject *>()(lhs, rhs); // can't use less all the time because of bool
2276 else
2277 smaller = lhs < rhs;
2278 return smaller ? QPartialOrdering::Less : QPartialOrdering::Greater;
2279}
