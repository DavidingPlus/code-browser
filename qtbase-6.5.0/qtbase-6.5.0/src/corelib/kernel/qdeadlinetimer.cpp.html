76/*!
77 * \internal
78 *
79 * Determines the sign of a (seconds, nanoseconds) pair
80 * for differentiating overflow from underflow. It doesn't
81 * deal with equality as it shouldn't ever be called in that case.
82 *
83 * Returns true if the pair represents a positive time offset
84 * false otherwise.
85 */
86bool TimeReference::sign(qint64 secs, qint64 nsecs)
87{
88 if (secs > 0) {
89 if (nsecs > 0)
90 return true;
91 } else {
92 if (nsecs < 0)
93 return false;
94 }
95
96 // They are different in sign
97 secs += nsecs / giga;
98 if (secs > 0)
99 return true;
100 else if (secs < 0)
101 return false;
102
103 // We should never get over|underflow out of
104 // the case: secs * giga == -nsecs
105 // So the sign of nsecs is the deciding factor
106 Q_ASSERT(nsecs % giga != 0);
107 return nsecs > 0;
108}

121/*!
122 * \internal
123 *
124 * Adds \a t1 addSecs seconds and \a addNSecs nanoseconds to the
125 * time reference. The arguments are normalized to seconds (qint64)
126 * and nanoseconds (unsigned) before the actual calculation is
127 * delegated to adjust(). If the nanoseconds are negative the
128 * owed second used for the normalization is passed on to adjust()
129 * as third argument.
130 *
131 * Returns true if operation was successful, false on over|underflow
132 */
133bool TimeReference::addSecsAndNSecs(qint64 addSecs, qint64 addNSecs)
134{
135 // Normalize the arguments
136 if (qAbs(t: addNSecs) >= giga) {
137 if (add_overflow<qint64>(v1: addSecs, v2: addNSecs / giga, r: &addSecs))
138 return false;
139
140 addNSecs %= giga;
141 }
142
143 if (addNSecs < 0)
144 return adjust(addSecs, ugiga - unsigned(-addNSecs), -1);
145
146 return adjust(addSecs, unsigned(addNSecs));
147}
148
149/*!
150 * \internal
151 *
152 * Adds \a t1 seconds and \a t2 nanoseconds to the internal members.
153 * Takes into account the additional \a carrySeconds we may owe or need to carry over.
154 *
155 * Returns true if operation was successful, false on over|underflow
156 */
157bool TimeReference::adjust(const qint64 t1, const unsigned t2, qint64 carrySeconds)
158{
159 static_assert(QDeadlineTimerNanosecondsInT2);
160 nsecs += t2;
161 if (nsecs >= ugiga) {
162 nsecs -= ugiga;
163 carrySeconds++;
164 }
165
166 // We don't worry about the order of addition, because the result returned by
167 // callers of this function is unchanged regardless of us over|underflowing.
168 // If we do, we do so by no more than a second, thus saturating the timer to
169 // Forever has the same effect as if we did the arithmetic exactly and salvaged
170 // the overflow.
171 return !add_overflow<qint64>(v1: secs, v2: t1, r: &secs) && !add_overflow<qint64>(v1: secs, v2: carrySeconds, r: &secs);
172}

174/*!
175 * \internal
176 *
177 * Subtracts \a t1 seconds and \a t2 nanoseconds from the time reference.
178 * When normalizing the nanoseconds to a positive number the owed seconds is
179 * passed as third argument to adjust() as the seconds may over|underflow
180 * if we do the calculation directly. There is little sense to check the
181 * seconds for over|underflow here in case we are going to need to carry
182 * over a second _after_ we add the nanoseconds.
183 *
184 * Returns true if operation was successful, false on over|underflow
185 */
186inline bool TimeReference::subtract(const qint64 t1, const unsigned t2)
187{
188 Q_ASSERT(t2 < ugiga);
189 return adjust(t1: -t1, t2: ugiga - t2, carrySeconds: -1);
190}

192/*!
193 * \internal
194 *
195 * Converts the time reference to milliseconds.
196 *
197 * Checks are done without making use of mul_overflow because it may
198 * not be implemented on some 32bit platforms.
199 *
200 * Returns true if operation was successful, false on over|underflow
201 */
202inline bool TimeReference::toMilliseconds(qint64 *result, RoundingStrategy rounding) const
203{
204 static constexpr qint64 maxSeconds = Max / kilo;
205 static constexpr qint64 minSeconds = Min / kilo;
206 if (secs > maxSeconds || secs < minSeconds)
207 return false;
208
209 unsigned ns = rounding == RoundDown ? nsecs : nsecs + umega - 1;
210
211 return !add_overflow<qint64>(v1: secs * kilo, v2: ns / umega, r: result);
212}
213
214/*!
215 * \internal
216 *
217 * Converts the time reference to nanoseconds.
218 *
219 * Checks are done without making use of mul_overflow because it may
220 * not be implemented on some 32bit platforms.
221 *
222 * Returns true if operation was successful, false on over|underflow
223 */
224inline bool TimeReference::toNanoseconds(qint64 *result) const
225{
226 static constexpr qint64 maxSeconds = Max / giga;
227 static constexpr qint64 minSeconds = Min / giga;
228 if (secs > maxSeconds || secs < minSeconds)
229 return false;
230
231 return !add_overflow<qint64>(v1: secs * giga, v2: nsecs, r: result);
232}

248inline bool TimeReference::addSecsAndNSecs(qint64 addSecs, qint64 addNSecs)
249{
250 static constexpr qint64 maxSeconds = Max / giga;
251 static constexpr qint64 minSeconds = Min / giga;
252 if (addSecs > maxSeconds || addSecs < minSeconds || add_overflow<qint64>(addSecs * giga, addNSecs, &addNSecs))
253 return false;
254
255 return addNanoseconds(addNSecs);
256}

258inline bool TimeReference::adjust(const qint64 t1, const unsigned t2, qint64 carrySeconds)
259{
260 static_assert(!QDeadlineTimerNanosecondsInT2);
261 Q_UNUSED(t2);
262 Q_UNUSED(carrySeconds);
263
264 return !add_overflow<qint64>(secs, t1, &secs);
265}

267inline bool TimeReference::subtract(const qint64 t1, const unsigned t2)
268{
269 Q_UNUSED(t2);
270
271 return addNanoseconds(-t1);
272}

274inline bool TimeReference::toMilliseconds(qint64 *result, RoundingStrategy rounding) const
275{
276 // Force QDeadlineTimer to treat the border cases as
277 // over|underflow and saturate the results returned to the user.
278 // We don't want to get valid milliseconds out of saturated timers.
279 if (secs == Max || secs == Min)
280 return false;
281
282 *result = secs / mega;
283 if (rounding == RoundUp && secs > *result * mega)
284 (*result)++;
285
286 return true;
287}

289inline bool TimeReference::toNanoseconds(qint64 *result) const
290{
291 *result = secs;
292 return true;
293}

296/*!
297 \class QDeadlineTimer
298 \inmodule QtCore
299 \brief The QDeadlineTimer class marks a deadline in the future.
300 \since 5.8
301
302 \reentrant
303 \ingroup tools
304
305 The QDeadlineTimer class is usually used to calculate future deadlines and
306 verify whether the deadline has expired. QDeadlineTimer can also be used
307 for deadlines without expiration ("forever"). It forms a counterpart to
308 QElapsedTimer, which calculates how much time has elapsed since
309 QElapsedTimer::start() was called.
310
311 QDeadlineTimer provides a more convenient API compared to
312 QElapsedTimer::hasExpired().
313
314 The typical use-case for the class is to create a QDeadlineTimer before the
315 operation in question is started, and then use remainingTime() or
316 hasExpired() to determine whether to continue trying the operation.
317 QDeadlineTimer objects can be passed to functions being called to execute
318 this operation so they know how long to still operate.
319
320 \snippet code/src_corelib_kernel_qdeadlinetimer.cpp 0
321
322 Many QDeadlineTimer functions deal with time out values, which all are
323 measured in milliseconds. There are two special values, the same as many
324 other Qt functions named \c{waitFor} or similar:
325
326 \list
327 \li 0: no time left, expired
328 \li -1: infinite time left, timer never expires
329 \endlist
330
331 \section1 Reference Clocks
332
333 QDeadlineTimer will use the same clock as QElapsedTimer (see
334 QElapsedTimer::clockType() and QElapsedTimer::isMonotonic()).
335
336 \section1 Timer types
337
338 Like QTimer, QDeadlineTimer can select among different levels of coarseness
339 on the timers. You can select precise timing by passing Qt::PreciseTimer to
340 the functions that set of change the timer, or you can select coarse timing
341 by passing Qt::CoarseTimer. Qt::VeryCoarseTimer is currently interpreted
342 the same way as Qt::CoarseTimer.
343
344 This feature is dependent on support from the operating system: if the OS
345 does not support a coarse timer functionality, then QDeadlineTimer will
346 behave like Qt::PreciseTimer was passed.
347
348 QDeadlineTimer defaults to Qt::CoarseTimer because on operating systems
349 that do support coarse timing, making timing calls to that clock source is
350 often much more efficient. The level of coarseness depends on the
351 operating system, but should be in the order of a couple of milliseconds.
352
353 \section1 \c{std::chrono} Compatibility
354
355 QDeadlineTimer is compatible with the \c{std::chrono} API from C++11 and
356 can be constructed from or compared to both \c{std::chrono::duration} and
357 \c{std::chrono::time_point} objects. In addition, it is fully compatible
358 with the time literals from C++14, which allow one to write code as:
359
360 \snippet code/src_corelib_kernel_qdeadlinetimer.cpp 1
361
362 As can be seen in the example above, QDeadlineTimer offers a templated
363 version of remainingTime() and deadline() that can be used to return
364 \c{std::chrono} objects.
365
366 Note that comparing to \c{time_point} is not as efficient as comparing to
367 \c{duration}, since QDeadlineTimer may need to convert from its own
368 internal clock source to the clock source used by the \c{time_point} object.
369 Also note that, due to this conversion, the deadlines will not be precise,
370 so the following code is not expected to compare equally:
371
372 \snippet code/src_corelib_kernel_qdeadlinetimer.cpp 2
373
374 \sa QTime, QTimer, QDeadlineTimer, Qt::TimerType
375*/
376

494/*!
495 Sets the remaining time for this QDeadlineTimer object to \a msecs
496 milliseconds from now, if \a msecs has a positive value. If \a msecs is
497 zero, this QDeadlineTimer object will be marked as expired, whereas a value
498 of -1 will set it to never expire.
499
500 The timer type for this QDeadlineTimer object will be set to the specified \a timerType.
501
502 \sa setPreciseRemainingTime(), hasExpired(), isForever(), remainingTime()
503*/
504void QDeadlineTimer::setRemainingTime(qint64 msecs, Qt::TimerType timerType) noexcept
505{
506 if (msecs == -1) {
507 *this = QDeadlineTimer(Forever, timerType);
508 return;
509 }
510
511 *this = current(timerType);
512
513 TimeReference ref(t1, t2);
514 if (!ref.addMilliseconds(arg: msecs))
515 ref.saturate(toMax: msecs > 0);
516 ref.updateTimer(t1, t2);
517}

523 parameters are zero, this QDeadlineTimer will be marked as expired.

618/*!
619 Returns the remaining time in this QDeadlineTimer object in milliseconds.
620 If the timer has already expired, this function will return zero and it is
621 not possible to obtain the amount of time overdue with this function (to do
622 that, see deadline()). If the timer was set to never expire, this function
623 returns -1.
624
625 This function is suitable for use in Qt APIs that take a millisecond
626 timeout, such as the many \l QIODevice \c waitFor functions or the timed
627 lock functions in \l QMutex, \l QWaitCondition, \l QSemaphore, or
628 \l QReadWriteLock. For example:
629
630 \snippet code/src_corelib_kernel_qdeadlinetimer.cpp 5
631
632 \sa remainingTimeNSecs(), isForever(), hasExpired()
633*/
634qint64 QDeadlineTimer::remainingTime() const noexcept
635{
636 if (isForever())
637 return -1;
638
639 QDeadlineTimer now = current(timerType: timerType());
640 TimeReference ref(t1, t2);
641
642 qint64 msecs;
643 if (!ref.subtract(t1: now.t1, t2: now.t2))
644 return 0; // We can only underflow here
645
646 // If we fail the conversion, t1 < now.t1 means we underflowed,
647 // thus the deadline had long expired
648 if (!ref.toMilliseconds(result: &msecs, rounding: TimeReference::RoundUp))
649 return t1 < now.t1 ? 0 : -1;
650
651 return msecs < 0 ? 0 : msecs;
652}

654/*!
655 Returns the remaining time in this QDeadlineTimer object in nanoseconds. If
656 the timer has already expired, this function will return zero and it is not
657 possible to obtain the amount of time overdue with this function. If the
658 timer was set to never expire, this function returns -1.
659
660 \sa remainingTime(), isForever(), hasExpired()
661*/
662qint64 QDeadlineTimer::remainingTimeNSecs() const noexcept
663{
664 if (isForever())
665 return -1;
666 qint64 raw = rawRemainingTimeNSecs();
667 return raw < 0 ? 0 : raw;
668}

670/*!
671 \internal
672 Same as remainingTimeNSecs, but may return negative remaining times. Does
673 not deal with Forever. In case of underflow the result is saturated to
674 the minimum possible value, on overflow - the maximum possible value.
675*/
676qint64 QDeadlineTimer::rawRemainingTimeNSecs() const noexcept
677{
678 QDeadlineTimer now = current(timerType: timerType());
679 TimeReference ref(t1, t2);
680
681 qint64 nsecs;
682 if (!ref.subtract(t1: now.t1, t2: now.t2))
683 return TimeReference::Min; // We can only underflow here
684
685 // If we fail the conversion, t1 < now.t1 means we underflowed,
686 // thus the deadline had long expired
687 if (!ref.toNanoseconds(result: &nsecs))
688 return t1 < now.t1 ? TimeReference::Min : TimeReference::Max;
689 return nsecs;
690}

692/*!
693 Returns the absolute time point for the deadline stored in QDeadlineTimer
694 object, calculated in milliseconds relative to the reference clock, the
695 same as QElapsedTimer::msecsSinceReference(). The value will be in the past
696 if this QDeadlineTimer has expired.
697
698 If this QDeadlineTimer never expires, this function returns
699 \c{std::numeric_limits<qint64>::max()}.
700
701 This function can be used to calculate the amount of time a timer is

759 absolute time point, counted in milliseconds since the reference clock (the

778
779 TimeReference ref;

782 ref.updateTimer(t1, t2);

785/*!
786 Sets the deadline for this QDeadlineTimer object to be \a secs seconds and
787 \a nsecs nanoseconds since the reference clock epoch (the same as
788 QElapsedTimer::msecsSinceReference()), and the timer type to \a timerType.
789 If the value is in the past, this QDeadlineTimer will be marked as expired.

794
795 \sa setDeadline(), deadline(), deadlineNSecs(), setRemainingTime()

797void QDeadlineTimer::setPreciseDeadline(qint64 secs, qint64 nsecs, Qt::TimerType timerType) noexcept
798{
799 type = timerType;
800
801 // We don't pass the seconds to the constructor, because we don't know
802 // at this point if t1 holds the seconds or nanoseconds; it's platform specific.
803 TimeReference ref;
804 if (!ref.addSecsAndNSecs(addSecs: secs, addNSecs: nsecs))
805 ref.saturate(toMax: TimeReference::sign(secs, nsecs));
806 ref.updateTimer(t1, t2);
807}
