37/*!
38 \internal
39
40 Setup unix_key
41 */
42key_t QSystemSemaphorePrivate::handle(QSystemSemaphore::AccessMode mode)
43{
44#if defined(Q_OS_DARWIN)
45 if (qt_apple_isSandboxed()) {
46 errorString = QSystemSemaphore::tr("%1: System V semaphores are not available " \
47 "for sandboxed applications. Please build Qt with -feature-ipc_posix")
48 .arg("QSystemSemaphore::handle:"_L1);
49 error = QSystemSemaphore::PermissionDenied;
50 return -1;
51 }
52#endif
53
54 if (key.isEmpty()){
55 errorString = QSystemSemaphore::tr(sourceText: "%1: key is empty")
56 .arg(a: "QSystemSemaphore::handle:"_L1);
57 error = QSystemSemaphore::KeyError;
58 return -1;
59 }
60
61 // ftok requires that an actual file exists somewhere
62 if (-1 != unix_key)
63 return unix_key;
64
65 // Create the file needed for ftok
66 int built = QSharedMemoryPrivate::createUnixKeyFile(fileName);
67 if (-1 == built) {
68 errorString = QSystemSemaphore::tr(sourceText: "%1: unable to make key")
69 .arg(a: "QSystemSemaphore::handle:"_L1);
70 error = QSystemSemaphore::KeyError;
71 return -1;
72 }
73 createdFile = (1 == built);
74
75#if QT_CONFIG(sharedmemory) && !defined(QT_POSIX_IPC) && !defined(Q_OS_ANDROID)
76 // Get the unix key for the created file
77 unix_key = ftok(pathname: QFile::encodeName(fileName).constData(), proj_id: 'Q');
78#endif
79 if (-1 == unix_key) {
80 errorString = QSystemSemaphore::tr(sourceText: "%1: ftok failed")
81 .arg(a: "QSystemSemaphore::handle:"_L1);
82 error = QSystemSemaphore::KeyError;
83 return -1;
84 }
85
86 // Get semaphore
87 semaphore = semget(key: unix_key, nsems: 1, semflg: 0600 | IPC_CREAT | IPC_EXCL);
88 if (-1 == semaphore) {
89 if (errno == EEXIST)
90 semaphore = semget(key: unix_key, nsems: 1, semflg: 0600 | IPC_CREAT);
91 if (-1 == semaphore) {
92 setErrorString("QSystemSemaphore::handle"_L1);
93 cleanHandle();
94 return -1;
95 }
96 } else {
97 createdSemaphore = true;
98 // Force cleanup of file, it is possible that it can be left over from a crash
99 createdFile = true;
100 }
101
102 if (mode == QSystemSemaphore::Create) {
103 createdSemaphore = true;
104 createdFile = true;
105 }
106
107 // Created semaphore so initialize its value.
108 if (createdSemaphore && initialValue >= 0) {
109 qt_semun init_op;
110 init_op.val = initialValue;
111 if (-1 == semctl(semid: semaphore, semnum: 0, SETVAL, init_op)) {
112 setErrorString("QSystemSemaphore::handle"_L1);
113 cleanHandle();
114 return -1;
115 }
116 }
117
118 return unix_key;
119}

121/*!
122 \internal
123
124 Cleanup the unix_key
125 */
126void QSystemSemaphorePrivate::cleanHandle()
127{
128 unix_key = -1;
129
130 // remove the file if we made it
131 if (createdFile) {
132 QFile::remove(fileName);
133 createdFile = false;
134 }
135
136 if (createdSemaphore) {
137 if (-1 != semaphore) {
138 if (-1 == semctl(semid: semaphore, semnum: 0, IPC_RMID, 0)) {
139 setErrorString("QSystemSemaphore::cleanHandle"_L1);
140#if defined QSYSTEMSEMAPHORE_DEBUG
141 qDebug("QSystemSemaphore::cleanHandle semctl failed.");
142#endif
143 }
144 semaphore = -1;
145 }
146 createdSemaphore = false;
147 }
148}

150/*!
151 \internal
152 */
153bool QSystemSemaphorePrivate::modifySemaphore(int count)
154{
155 if (-1 == handle())
156 return false;
157
158 struct sembuf operation;
159 operation.sem_num = 0;
160 operation.sem_op = count;
161 operation.sem_flg = SEM_UNDO;
162
163 int res;
164 EINTR_LOOP(res, semop(semaphore, &operation, 1));
165 if (-1 == res) {
166 // If the semaphore was removed be nice and create it and then modifySemaphore again
167 if (errno == EINVAL || errno == EIDRM) {
168 semaphore = -1;
169 cleanHandle();
170 handle();
171 return modifySemaphore(count);
172 }
173 setErrorString("QSystemSemaphore::modifySemaphore"_L1);
174#if defined QSYSTEMSEMAPHORE_DEBUG
175 qDebug("QSystemSemaphore::modify failed %d %d %d %d %d",
176 count, int(semctl(semaphore, 0, GETVAL)), int(errno), int(EIDRM), int(EINVAL);
177#endif
178 return false;
179 }
180
181 clearError();
182 return true;
183}
