107 /*
108 This contains the all connections from and to an object.
109
110 The signalVector contains the lists of connections for a given signal. The index in the vector correspond
111 to the signal index. The signal index is the one returned by QObjectPrivate::signalIndex (not
112 QMetaObject::indexOfSignal). allsignals contains a list of special connections that will get invoked on
113 any signal emission. This is done by connecting to signal index -1.
114
115 This vector is protected by the object mutex (signalSlotLock())
116
117 Each Connection is also part of a 'senders' linked list. This one contains all connections connected
118 to a slot in this object. The mutex of the receiver must be locked when touching the pointers of this
119 linked list.
120 */

207/*
208 Catch mixing of incompatible library versions.
209
210 Should be called from the constructor of every non-final subclass
211 of QObjectPrivate, to ensure we catch incompatibilities between
212 the intermediate base and subclasses thereof.
213*/
214inline void QObjectPrivate::checkForIncompatibleLibraryVersion(int version) const
215{
216#if defined(QT_BUILD_INTERNAL)
217 // Don't check the version parameter in internal builds.
218 // This allows incompatible versions to be loaded, possibly for testing.
219 Q_UNUSED(version);
220#else
221 if (Q_UNLIKELY(version != QObjectPrivateVersion)) {
222 qFatal(msg: "Cannot mix incompatible Qt library (%d.%d.%d) with this library (%d.%d.%d)",
223 (version >> 16) & 0xff, (version >> 8) & 0xff, version & 0xff,
224 (QObjectPrivateVersion >> 16) & 0xff, (QObjectPrivateVersion >> 8) & 0xff, QObjectPrivateVersion & 0xff);
225 }
226#endif
227}

278template<typename Func, typename Args, typename R>
279class QPrivateSlotObject : public QSlotObjectBase, private FunctionStorage<Func>
280{
281 typedef QtPrivate::FunctionPointer<Func> FuncType;
282 static void impl(int which, QSlotObjectBase *this_, QObject *r, void **a, bool *ret)
283 {
284 switch (which) {
285 case Destroy:
286 delete static_cast<QPrivateSlotObject*>(this_);
287 break;
288 case Call:
289 FuncType::template call<Args, R>(static_cast<QPrivateSlotObject*>(this_)->func(),
290 static_cast<typename FuncType::Object *>(QObjectPrivate::get(o: r)), a);
291 break;
292 case Compare:
293 *ret = *reinterpret_cast<Func *>(a) == static_cast<QPrivateSlotObject*>(this_)->func();
294 break;
295 case NumOperations: ;
296 }
297 }
298public:
299 explicit QPrivateSlotObject(Func f) : QSlotObjectBase(&impl), FunctionStorage<Func>{std::move(f)} {}
300};

303template <typename Func1, typename Func2>
304inline QMetaObject::Connection QObjectPrivate::connect(const typename QtPrivate::FunctionPointer<Func1>::Object *sender, Func1 signal,
305 const typename QtPrivate::FunctionPointer<Func2>::Object *receiverPrivate, Func2 slot,
306 Qt::ConnectionType type)
307{
308 typedef QtPrivate::FunctionPointer<Func1> SignalType;
309 typedef QtPrivate::FunctionPointer<Func2> SlotType;
310 static_assert(QtPrivate::HasQ_OBJECT_Macro<typename SignalType::Object>::Value,
311 "No Q_OBJECT in the class with the signal");
312
313 //compilation error if the arguments does not match.
314 static_assert(int(SignalType::ArgumentCount) >= int(SlotType::ArgumentCount),
315 "The slot requires more arguments than the signal provides.");
316 static_assert((QtPrivate::CheckCompatibleArguments<typename SignalType::Arguments, typename SlotType::Arguments>::value),
317 "Signal and slot arguments are not compatible.");
318 static_assert((QtPrivate::AreArgumentsCompatible<typename SlotType::ReturnType, typename SignalType::ReturnType>::value),
319 "Return type of the slot is not compatible with the return type of the signal.");
320
321 const int *types = nullptr;
322 if (type == Qt::QueuedConnection || type == Qt::BlockingQueuedConnection)
323 types = QtPrivate::ConnectionTypes<typename SignalType::Arguments>::types();
324
325 return QObject::connectImpl(sender, signal: reinterpret_cast<void **>(&signal),
326 receiver: QtPrivate::getQObject(d: receiverPrivate), slotPtr: reinterpret_cast<void **>(&slot),
327 slot: new QtPrivate::QPrivateSlotObject<Func2, typename QtPrivate::List_Left<typename SignalType::Arguments, SlotType::ArgumentCount>::Value,
328 typename SignalType::ReturnType>(slot),
329 type, types, senderMetaObject: &SignalType::Object::staticMetaObject);
330}

332template <typename Func1, typename Func2>
333bool QObjectPrivate::disconnect(const typename QtPrivate::FunctionPointer< Func1 >::Object* sender, Func1 signal,
334 const typename QtPrivate::FunctionPointer< Func2 >::Object* receiverPrivate, Func2 slot)
335{
336 typedef QtPrivate::FunctionPointer<Func1> SignalType;
337 typedef QtPrivate::FunctionPointer<Func2> SlotType;
338 static_assert(QtPrivate::HasQ_OBJECT_Macro<typename SignalType::Object>::Value,
339 "No Q_OBJECT in the class with the signal");
340 //compilation error if the arguments does not match.
341 static_assert((QtPrivate::CheckCompatibleArguments<typename SignalType::Arguments, typename SlotType::Arguments>::value),
342 "Signal and slot arguments are not compatible.");
343 return QObject::disconnectImpl(sender, signal: reinterpret_cast<void **>(&signal),
344 receiver: receiverPrivate->q_ptr, slot: reinterpret_cast<void **>(&slot),
345 senderMetaObject: &SignalType::Object::staticMetaObject);
346}
