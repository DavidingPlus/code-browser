23class Q_CORE_EXPORT QDeadlineTimer
24{
25public:
26 enum ForeverConstant { Forever };
27
28 constexpr QDeadlineTimer(Qt::TimerType type_ = Qt::CoarseTimer) noexcept
29 : type(type_) {}
30 constexpr QDeadlineTimer(ForeverConstant, Qt::TimerType type_ = Qt::CoarseTimer) noexcept
31 : t1((std::numeric_limits<qint64>::max)()), type(type_) {}
32 explicit QDeadlineTimer(qint64 msecs, Qt::TimerType type = Qt::CoarseTimer) noexcept;
33
34 void swap(QDeadlineTimer &other) noexcept
35 { std::swap(a&: t1, b&: other.t1); std::swap(a&: t2, b&: other.t2); std::swap(a&: type, b&: other.type); }
36
37 constexpr bool isForever() const noexcept
38 { return t1 == (std::numeric_limits<qint64>::max)(); }
39 bool hasExpired() const noexcept;
40
41 Qt::TimerType timerType() const noexcept
42 { return Qt::TimerType(type & 0xff); }
43 void setTimerType(Qt::TimerType type);
44
45 qint64 remainingTime() const noexcept;
46 qint64 remainingTimeNSecs() const noexcept;
47 void setRemainingTime(qint64 msecs, Qt::TimerType type = Qt::CoarseTimer) noexcept;
48 void setPreciseRemainingTime(qint64 secs, qint64 nsecs = 0,
49 Qt::TimerType type = Qt::CoarseTimer) noexcept;
50
51 qint64 deadline() const noexcept Q_DECL_PURE_FUNCTION;
52 qint64 deadlineNSecs() const noexcept Q_DECL_PURE_FUNCTION;
53 void setDeadline(qint64 msecs, Qt::TimerType timerType = Qt::CoarseTimer) noexcept;
54 void setPreciseDeadline(qint64 secs, qint64 nsecs = 0,
55 Qt::TimerType type = Qt::CoarseTimer) noexcept;
56
57 static QDeadlineTimer addNSecs(QDeadlineTimer dt, qint64 nsecs) noexcept Q_DECL_PURE_FUNCTION;
58 static QDeadlineTimer current(Qt::TimerType timerType = Qt::CoarseTimer) noexcept;
59
60 friend bool operator==(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
61 { return d1.t1 == d2.t1 && d1.t2 == d2.t2; }
62 friend bool operator!=(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
63 { return !(d1 == d2); }
64 friend bool operator<(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
65 { return d1.t1 < d2.t1 || (d1.t1 == d2.t1 && d1.t2 < d2.t2); }
66 friend bool operator<=(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
67 { return d1 == d2 || d1 < d2; }
68 friend bool operator>(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
69 { return d2 < d1; }
70 friend bool operator>=(QDeadlineTimer d1, QDeadlineTimer d2) noexcept
71 { return !(d1 < d2); }
72
73 friend Q_CORE_EXPORT QDeadlineTimer operator+(QDeadlineTimer dt, qint64 msecs);
74 friend QDeadlineTimer operator+(qint64 msecs, QDeadlineTimer dt)
75 { return dt + msecs; }
76 friend QDeadlineTimer operator-(QDeadlineTimer dt, qint64 msecs)
77 { return dt + (-msecs); }
78 friend qint64 operator-(QDeadlineTimer dt1, QDeadlineTimer dt2)
79 { return (dt1.deadlineNSecs() - dt2.deadlineNSecs()) / (1000 * 1000); }
80 QDeadlineTimer &operator+=(qint64 msecs)
81 { *this = *this + msecs; return *this; }
82 QDeadlineTimer &operator-=(qint64 msecs)
83 { *this = *this + (-msecs); return *this; }
84
85 template <class Clock, class Duration = typename Clock::duration>
86 QDeadlineTimer(std::chrono::time_point<Clock, Duration> deadline_,
87 Qt::TimerType type_ = Qt::CoarseTimer) : t2(0)
88 { setDeadline(deadline_, type_); }
89 template <class Clock, class Duration = typename Clock::duration>
90 QDeadlineTimer &operator=(std::chrono::time_point<Clock, Duration> deadline_)
91 { setDeadline(deadline_); return *this; }
92
93 template <class Clock, class Duration = typename Clock::duration>
94 void setDeadline(std::chrono::time_point<Clock, Duration> deadline_,
95 Qt::TimerType type_ = Qt::CoarseTimer)
96 { setRemainingTime(deadline_ == deadline_.max() ? Duration::max() : deadline_ - Clock::now(), type_); }
97
98 template <class Clock, class Duration = typename Clock::duration>
99 std::chrono::time_point<Clock, Duration> deadline() const
100 {
101 auto val = std::chrono::nanoseconds(rawRemainingTimeNSecs()) + Clock::now();
102 return std::chrono::time_point_cast<Duration>(val);
103 }
104
105 template <class Rep, class Period>
106 QDeadlineTimer(std::chrono::duration<Rep, Period> remaining, Qt::TimerType type_ = Qt::CoarseTimer)
107 : t2(0)
108 { setRemainingTime(remaining, type_); }
109
110 template <class Rep, class Period>
111 QDeadlineTimer &operator=(std::chrono::duration<Rep, Period> remaining)
112 { setRemainingTime(remaining); return *this; }
113
114 template <class Rep, class Period>
115 void setRemainingTime(std::chrono::duration<Rep, Period> remaining, Qt::TimerType type_ = Qt::CoarseTimer)
116 {
117 if (remaining == remaining.max())
118 *this = QDeadlineTimer(Forever, type_);
119 else
120 setPreciseRemainingTime(secs: 0, nsecs: std::chrono::nanoseconds(remaining).count(), type: type_);
121 }
122
123 std::chrono::nanoseconds remainingTimeAsDuration() const noexcept
124 {
125 if (isForever())
126 return std::chrono::nanoseconds::max();
127 qint64 nsecs = rawRemainingTimeNSecs();
128 if (nsecs <= 0)
129 return std::chrono::nanoseconds::zero();
130 return std::chrono::nanoseconds(nsecs);
131 }
132
133 template <class Rep, class Period>
134 friend QDeadlineTimer operator+(QDeadlineTimer dt, std::chrono::duration<Rep, Period> value)
135 { return QDeadlineTimer::addNSecs(dt, nsecs: std::chrono::duration_cast<std::chrono::nanoseconds>(value).count()); }
136 template <class Rep, class Period>
137 friend QDeadlineTimer operator+(std::chrono::duration<Rep, Period> value, QDeadlineTimer dt)
138 { return dt + value; }
139 template <class Rep, class Period>
140 friend QDeadlineTimer operator+=(QDeadlineTimer &dt, std::chrono::duration<Rep, Period> value)
141 { return dt = dt + value; }
142
143private:
144 qint64 t1 = 0;
145 unsigned t2 = 0;
146 unsigned type;
147
148 qint64 rawRemainingTimeNSecs() const noexcept;
149
150public:
151 // This is not a public function, it's here only for Qt's internal convenience...
152 QPair<qint64, unsigned> _q_data() const { return qMakePair(value1: t1, value2: t2); }
153};

155#if defined(Q_OS_DARWIN) || defined(Q_OS_LINUX) || (defined(Q_CC_MSVC) && Q_CC_MSVC >= 1900)
156// We know for these OS/compilers that the std::chrono::steady_clock uses the same
157// reference time as QDeadlineTimer
158
159template <> inline std::chrono::steady_clock::time_point
160QDeadlineTimer::deadline<std::chrono::steady_clock, std::chrono::steady_clock::duration>() const
161{
162 return std::chrono::steady_clock::time_point(std::chrono::nanoseconds(deadlineNSecs()));
163}
164
165template <> inline void
166QDeadlineTimer::setDeadline<std::chrono::steady_clock, std::chrono::steady_clock::duration>(std::chrono::steady_clock::time_point tp, Qt::TimerType type_)
167{
168 using namespace std::chrono;
169 if (tp == tp.max()) {
170 *this = Forever;
171 type = type_;
172 } else if (type_ != Qt::PreciseTimer) {
173 // if we aren't using PreciseTimer, then we need to convert
174 setPreciseRemainingTime(secs: 0, nsecs: duration_cast<nanoseconds>(d: tp - steady_clock::now()).count(), type: type_);
175 } else {
176 setPreciseDeadline(secs: 0,
177 nsecs: duration_cast<nanoseconds>(d: tp.time_since_epoch()).count(),
178 type: type_);
179 }
180}
181#endif
