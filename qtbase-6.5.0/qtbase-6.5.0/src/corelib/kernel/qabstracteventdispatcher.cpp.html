58int QAbstractEventDispatcherPrivate::allocateTimerId()
59{
60 // This function may be called after timerIdFreeList() has been destructed
61 // for example in case when application exits without waiting for
62 // running threads to exit and running thread finished() has been connected
63 // to a slot which triggers a sequence that registers new timer.
64 // See https://bugreports.qt-project.org/browse/QTBUG-38957.
65 if (QtTimerIdFreeList *fl = timerIdFreeList())
66 return fl->next();
67 return 0; // Note! returning 0 generates a warning
68}
69
70void QAbstractEventDispatcherPrivate::releaseTimerId(int timerId)
71{
72 // this function may be called by a global destructor after
73 // timerIdFreeList() has been destructed
74 if (QtTimerIdFreeList *fl = timerIdFreeList())
75 fl->release(id: timerId);
76}

78/*!
79 \class QAbstractEventDispatcher
80 \inmodule QtCore
81 \brief The QAbstractEventDispatcher class provides an interface to manage Qt's event queue.
82
83 \ingroup events
84
85 An event dispatcher receives events from the window system and other
86 sources. It then sends them to the QCoreApplication or QApplication
87 instance for processing and delivery. QAbstractEventDispatcher provides
88 fine-grained control over event delivery.
89
90 For simple control of event processing use
91 QCoreApplication::processEvents().
92
93 For finer control of the application's event loop, call
94 instance() and call functions on the QAbstractEventDispatcher
95 object that is returned. If you want to use your own instance of
96 QAbstractEventDispatcher or of a QAbstractEventDispatcher
97 subclass, you must install it with QCoreApplication::setEventDispatcher()
98 or QThread::setEventDispatcher() \e before a default event dispatcher has
99 been installed.
100
101 The main event loop is started by calling
102 QCoreApplication::exec(), and stopped by calling
103 QCoreApplication::exit(). Local event loops can be created using
104 QEventLoop.
105
106 Programs that perform long operations can call processEvents()
107 with a bitwise OR combination of various QEventLoop::ProcessEventsFlag
108 values to control which events should be delivered.
109
110 QAbstractEventDispatcher also allows the integration of an
111 external event loop with the Qt event loop.
112
113 \sa QEventLoop, QCoreApplication, QThread
114*/
115

135/*!

144QAbstractEventDispatcher *QAbstractEventDispatcher::instance(QThread *thread)

150/*!
151 \fn bool QAbstractEventDispatcher::processEvents(QEventLoop::ProcessEventsFlags flags)
152
153 Processes pending events that match \a flags until there are no
154 more events to process. Returns \c true if an event was processed;
155 otherwise returns \c false.
156
157 This function is especially useful if you have a long running
158 operation, and want to show its progress without allowing user
159 input by using the QEventLoop::ExcludeUserInputEvents flag.
160
161 If the QEventLoop::WaitForMoreEvents flag is set in \a flags, the
162 behavior of this function is as follows:
163
164 \list
165
166 \li If events are available, this function returns after processing
167 them.
168
169 \li If no events are available, this function will wait until more
170 are available and return after processing newly available events.
171
172 \endlist
173
174 If the QEventLoop::WaitForMoreEvents flag is not set in \a flags,
175 and no events are available, this function will return
176 immediately.
177
178 \b{Note:} This function does not process events continuously; it
179 returns after all available events are processed.
180*/
181
182/*!
183 \internal
184
185 \note processEvents() only processes events queued before the function
186 is called. Events that are posted while the function runs will be queued
187 until a later round of event processing. This only applies to posted Qt
188 events. For timers and system level events, the situation is unknown.
189*/

334/*!
335 Installs an event filter \a filterObj for all native events received by the application.
336
337 The event filter \a filterObj receives events via its \l {QAbstractNativeEventFilter::}{nativeEventFilter()}
338 function, which is called for all events received by all threads.
339
340 The \l {QAbstractNativeEventFilter::}{nativeEventFilter()} function should return true
341 if the event should be filtered, (in this case, stopped). It should return false to allow
342 normal Qt processing to continue: the native event can then be translated
343 into a QEvent and handled by the standard Qt \l{QEvent} {event} filtering,
344 e.g. QObject::installEventFilter().
345
346 If multiple event filters are installed, the filter that was installed last
347 is activated first.
348
349 \note The filter function set here receives native messages,
350 that is, MSG or XEvent structs.
351
352 For maximum portability, you should always try to use QEvent objects
353 and QObject::installEventFilter() whenever possible.
354
355 \sa QObject::installEventFilter()
356
357 \since 5.0
358*/
359void QAbstractEventDispatcher::installNativeEventFilter(QAbstractNativeEventFilter *filterObj)
360{
361 Q_D(QAbstractEventDispatcher);
362
363 // clean up unused items in the list
364 d->eventFilters.removeAll(t: nullptr);
365 d->eventFilters.removeAll(t: filterObj);
366 d->eventFilters.prepend(t: filterObj);
367}

369/*!
370 Removes the event filter \a filter from this object. The
371 request is ignored if such an event filter has not been installed.
372
373 All event filters for this object are automatically removed when
374 this object is destroyed.
375
376 It is always safe to remove an event filter, even during event filter
377 filter activation (that is, even from within the \l {QAbstractNativeEventFilter::}{nativeEventFilter()} function).
378
379 \sa installNativeEventFilter(), QAbstractNativeEventFilter
380 \since 5.0
381*/
382void QAbstractEventDispatcher::removeNativeEventFilter(QAbstractNativeEventFilter *filter)
383{
384 Q_D(QAbstractEventDispatcher);
385 for (int i = 0; i < d->eventFilters.size(); ++i) {
386 if (d->eventFilters.at(i) == filter) {
387 d->eventFilters[i] = nullptr;
388 break;
389 }
390 }
391}

393/*!
394 Sends \a message through the event filters that were set by
395 installNativeEventFilter(). This function returns \c true as soon as an
396 event filter returns \c true, and false otherwise to indicate that
397 the processing of the event should continue.
398
399 Subclasses of QAbstractEventDispatcher \e must call this function
400 for \e all messages received from the system to ensure
401 compatibility with any extensions that may be used in the
402 application. The type of event \a eventType is specific to the platform
403 plugin chosen at run-time, and can be used to cast message to the right type.
404 The \a result pointer is only used on Windows, and corresponds to the LRESULT pointer.
405
406 Note that the type of \a message is platform dependent. See
407 QAbstractNativeEventFilter for details.
408
409 \sa installNativeEventFilter(), QAbstractNativeEventFilter::nativeEventFilter()
410 \since 5.0
411*/
412bool QAbstractEventDispatcher::filterNativeEvent(const QByteArray &eventType, void *message, qintptr *result)
413{
414 Q_D(QAbstractEventDispatcher);
415 if (!d->eventFilters.isEmpty()) {
416 // Raise the loopLevel so that deleteLater() calls in or triggered
417 // by event_filter() will be processed from the main event loop.
418 QScopedScopeLevelCounter scopeLevelCounter(d->threadData.loadAcquire());
419 for (int i = 0; i < d->eventFilters.size(); ++i) {
420 QAbstractNativeEventFilter *filter = d->eventFilters.at(i);
421 if (!filter)
422 continue;
423 if (filter->nativeEventFilter(eventType, message, result))
424 return true;
425 }
426 }
427 return false;
428}
