4#ifndef QELAPSEDTIMER_H
5#define QELAPSEDTIMER_H
6
7#include <QtCore/qglobal.h>
8
9QT_BEGIN_NAMESPACE
10
11
12class Q_CORE_EXPORT QElapsedTimer
13{
14public:
15 enum ClockType {
16 SystemTime,
17 MonotonicClock,
18 TickCounter Q_DECL_ENUMERATOR_DEPRECATED_X(
19 "Not supported anymore. Use PerformanceCounter instead."),
20 MachAbsoluteTime,
21 PerformanceCounter
22 };
23
24 constexpr QElapsedTimer() = default;
25
26 static ClockType clockType() noexcept;
27 static bool isMonotonic() noexcept;
28
29 void start() noexcept;
30 qint64 restart() noexcept;
31 void invalidate() noexcept;
32 bool isValid() const noexcept;
33
34 qint64 nsecsElapsed() const noexcept;
35 qint64 elapsed() const noexcept;
36 bool hasExpired(qint64 timeout) const noexcept;
37
38 qint64 msecsSinceReference() const noexcept;
39 qint64 msecsTo(const QElapsedTimer &other) const noexcept;
40 qint64 secsTo(const QElapsedTimer &other) const noexcept;
41
42 friend bool operator==(const QElapsedTimer &lhs, const QElapsedTimer &rhs) noexcept
43 { return lhs.t1 == rhs.t1 && lhs.t2 == rhs.t2; }
44 friend bool operator!=(const QElapsedTimer &lhs, const QElapsedTimer &rhs) noexcept
45 { return !(lhs == rhs); }
46
47 friend bool Q_CORE_EXPORT operator<(const QElapsedTimer &lhs, const QElapsedTimer &rhs) noexcept;
48
49private:
50 qint64 t1 = Q_INT64_C(0x8000000000000000);
51 qint64 t2 = Q_INT64_C(0x8000000000000000);
52};
53
54QT_END_NAMESPACE
55
56#endif // QELAPSEDTIMER_H
