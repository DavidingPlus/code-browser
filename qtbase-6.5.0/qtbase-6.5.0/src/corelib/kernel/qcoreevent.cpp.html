20/*!
21 \class QEvent
22 \inmodule QtCore
23 \brief The QEvent class is the base class of all
24 event classes. Event objects contain event parameters.
25
26 \ingroup events
27
28 Qt's main event loop (QCoreApplication::exec()) fetches native
29 window system events from the event queue, translates them into
30 QEvents, and sends the translated events to \l{QObject}s.
31
32 In general, events come from the underlying window system
33 (spontaneous() returns \c true), but it is also possible to manually
34 send events using QCoreApplication::sendEvent() and
35 QCoreApplication::postEvent() (spontaneous() returns \c false).
36
37 \l {QObject}{QObjects} receive events by having their QObject::event() function
38 called. The function can be reimplemented in subclasses to
39 customize event handling and add additional event types;
40 QWidget::event() is a notable example. By default, events are
41 dispatched to event handlers like QObject::timerEvent() and
42 QWidget::mouseMoveEvent(). QObject::installEventFilter() allows an
43 object to intercept events destined for another object.
44
45 The basic QEvent contains only an event type parameter and an
46 "accept" flag. The accept flag set with accept(), and cleared
47 with ignore(). It is set by default, but don't rely on this as
48 subclasses may choose to clear it in their constructor.
49
50 Subclasses of QEvent contain additional parameters that describe
51 the particular event.
52
53 \sa QObject::event(), QObject::installEventFilter(),
54 QCoreApplication::sendEvent(),
55 QCoreApplication::postEvent(), QCoreApplication::processEvents()
56*/
57

413namespace {
414template <size_t N>
415struct QBasicAtomicBitField {
416 enum {
417 BitsPerInt = std::numeric_limits<uint>::digits,
418 NumInts = (N + BitsPerInt - 1) / BitsPerInt,
419 NumBits = N
420 };
421
422 // This atomic int points to the next (possibly) free ID saving
423 // the otherwise necessary scan through 'data':
424 QBasicAtomicInteger<uint> next;
425 QBasicAtomicInteger<uint> data[NumInts];
426
427 constexpr QBasicAtomicBitField() = default;
428
429 bool allocateSpecific(int which) noexcept
430 {
431 QBasicAtomicInteger<uint> &entry = data[which / BitsPerInt];
432 const uint old = entry.loadRelaxed();
433 const uint bit = 1U << (which % BitsPerInt);
434 return !(old & bit) // wasn't taken
435 && entry.testAndSetRelaxed(expectedValue: old, newValue: old | bit); // still wasn't taken
436
437 // don't update 'next' here - it's unlikely that it will need
438 // to be updated, in the general case, and having 'next'
439 // trailing a bit is not a problem, as it is just a starting
440 // hint for allocateNext(), which, when wrong, will just
441 // result in a few more rounds through the allocateNext()
442 // loop.
443 }
444
445 int allocateNext() noexcept
446 {
447 // Unroll loop to iterate over ints, then bits? Would save
448 // potentially a lot of cmpxchgs, because we can scan the
449 // whole int before having to load it again.
450
451 // Then again, this should never execute many iterations, so
452 // leave like this for now:
453 for (uint i = next.loadRelaxed(); i < NumBits; ++i) {
454 if (allocateSpecific(which: i)) {
455 // remember next (possibly) free id:
456 const uint oldNext = next.loadRelaxed();
457 next.testAndSetRelaxed(expectedValue: oldNext, newValue: qMax(a: i + 1, b: oldNext));
458 return i;
459 }
460 }
461 return -1;
462 }
463};
464
465} // unnamed namespace

467typedef QBasicAtomicBitField<QEvent::MaxUser - QEvent::User + 1> UserEventTypeRegistry;
468
469Q_CONSTINIT static UserEventTypeRegistry userEventTypeRegistry {};
470
471static inline int registerEventTypeZeroBased(int id) noexcept
472{
473 // if the type hint hasn't been registered yet, take it:
474 if (id < UserEventTypeRegistry::NumBits && id >= 0 && userEventTypeRegistry.allocateSpecific(which: id))
475 return id;
476
477 // otherwise, ignore hint:
478 return userEventTypeRegistry.allocateNext();
479}

481/*!
482 \since 4.4
483 \threadsafe
484
485 Registers and returns a custom event type. The \a hint provided
486 will be used if it is available, otherwise it will return a value
487 between QEvent::User and QEvent::MaxUser that has not yet been
488 registered. The \a hint is ignored if its value is not between
489 QEvent::User and QEvent::MaxUser.
490
491 Returns -1 if all available values are already taken or the
492 program is shutting down.
493*/
494int QEvent::registerEventType(int hint) noexcept
495{
496 const int result = registerEventTypeZeroBased(id: QEvent::MaxUser - hint);
497 return result < 0 ? -1 : QEvent::MaxUser - result ;
498}
