115 Qt::PermissionStatus checkPermission(const QPermission &permission);

117# ifdef Q_QDOC
118 template <typename Functor>
119 void requestPermission(const QPermission &permission, Functor functor);
120 template <typename Functor>
121 void requestPermission(const QPermission &permission, const QObject *context, Functor functor);
122# else
123 template <typename Slot> // requestPermission to a QObject slot
124 void requestPermission(const QPermission &permission,
125 const typename QtPrivate::FunctionPointer<Slot>::Object *receiver, Slot slot)
126 {
127 using CallbackSignature = QtPrivate::FunctionPointer<void (*)(QPermission)>;
128 using SlotSignature = QtPrivate::FunctionPointer<Slot>;
129
130 static_assert(int(SlotSignature::ArgumentCount) <= int(CallbackSignature::ArgumentCount),
131 "Slot requires more arguments than what can be provided.");
132 static_assert((QtPrivate::CheckCompatibleArguments<typename CallbackSignature::Arguments, typename SlotSignature::Arguments>::value),
133 "Slot arguments are not compatible (must be QPermission)");
134
135 auto slotObj = new QtPrivate::QSlotObject<Slot, typename SlotSignature::Arguments, void>(slot);
136 requestPermission(permission, slotObj, receiver);
137 }
138
139 // requestPermission to a functor or function pointer (with context)
140 template <typename Func, std::enable_if_t<
141 !QtPrivate::FunctionPointer<Func>::IsPointerToMemberFunction
142 && !std::is_same<const char *, Func>::value, bool> = true>
143 void requestPermission(const QPermission &permission, const QObject *context, Func func)
144 {
145 using CallbackSignature = QtPrivate::FunctionPointer<void (*)(QPermission)>;
146 constexpr int MatchingArgumentCount = QtPrivate::ComputeFunctorArgumentCount<
147 Func, CallbackSignature::Arguments>::Value;
148
149 static_assert(MatchingArgumentCount == 0
150 || MatchingArgumentCount == CallbackSignature::ArgumentCount,
151 "Functor arguments are not compatible (must be QPermission)");
152
153 QtPrivate::QSlotObjectBase *slotObj = nullptr;
154 if constexpr (MatchingArgumentCount == CallbackSignature::ArgumentCount) {
155 slotObj = new QtPrivate::QFunctorSlotObject<Func, 1,
156 typename CallbackSignature::Arguments, void>(std::move(func));
157 } else {
158 slotObj = new QtPrivate::QFunctorSlotObject<Func, 0,
159 typename QtPrivate::List_Left<void, 0>::Value, void>(std::move(func));
160 }
161
162 requestPermission(permission, slotObj, context);
163 }
164
165 // requestPermission to a functor or function pointer (without context)
166 template <typename Func, std::enable_if_t<
167 !QtPrivate::FunctionPointer<Func>::IsPointerToMemberFunction
168 && !std::is_same<const char *, Func>::value, bool> = true>
169 void requestPermission(const QPermission &permission, Func func)
170 {
171 requestPermission(permission, nullptr, std::move(func));
172 }
