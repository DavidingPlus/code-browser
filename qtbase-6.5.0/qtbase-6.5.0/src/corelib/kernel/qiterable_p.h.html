21QT_BEGIN_NAMESPACE
22
23namespace QIterablePrivate {
24
25template<typename Callback>
26static QVariant retrieveElement(QMetaType type, Callback callback)
27{
28 QVariant v(type);
29 void *dataPtr;
30 if (type == QMetaType::fromType<QVariant>())
31 dataPtr = &v;
32 else
33 dataPtr = v.data();
34 callback(dataPtr);
35 return v;
36}
