110template <typename T> inline typename std::enable_if_t<std::is_unsigned_v<T>, bool>
111qAddOverflow(T v1, T v2, T *r)
112{
113 // unsigned additions are well-defined
114 *r = v1 + v2;
115 return v1 > T(v1 + v2);
116}
117
118template <typename T> inline typename std::enable_if_t<std::is_signed_v<T>, bool>
119qAddOverflow(T v1, T v2, T *r)
120{
121 // Here's how we calculate the overflow:
122 // 1) unsigned addition is well-defined, so we can always execute it
123 // 2) conversion from unsigned back to signed is implementation-
124 // defined and in the implementations we use, it's a no-op.
125 // 3) signed integer overflow happens if the sign of the two input operands
126 // is the same but the sign of the result is different. In other words,
127 // the sign of the result must be the same as the sign of either
128 // operand.
129
130 using U = typename std::make_unsigned_t<T>;
131 *r = T(U(v1) + U(v2));
132
133 // If int is two's complement, assume all integer types are too.
134 if (std::is_same_v<int32_t, int>) {
135 // Two's complement equivalent (generates slightly shorter code):
136 // x ^ y is negative if x and y have different signs
137 // x & y is negative if x and y are negative
138 // (x ^ z) & (y ^ z) is negative if x and z have different signs
139 // AND y and z have different signs
140 return ((v1 ^ *r) & (v2 ^ *r)) < 0;
141 }
142
143 bool s1 = (v1 < 0);
144 bool s2 = (v2 < 0);
145 bool sr = (*r < 0);
146 return s1 != sr && s2 != sr;
147 // also: return s1 == s2 && s1 != sr;
148}
149
150template <typename T> inline typename std::enable_if_t<std::is_unsigned_v<T>, bool>
151qSubOverflow(T v1, T v2, T *r)
152{
153 // unsigned subtractions are well-defined
154 *r = v1 - v2;
155 return v1 < v2;
156}
157
158template <typename T> inline typename std::enable_if_t<std::is_signed_v<T>, bool>
159qSubOverflow(T v1, T v2, T *r)
160{
161 // See above for explanation. This is the same with some signs reversed.
162 // We can't use qAddOverflow(v1, -v2, r) because it would be UB if
163 // v2 == std::numeric_limits<T>::min().
164
165 using U = typename std::make_unsigned_t<T>;
166 *r = T(U(v1) - U(v2));
167
168 if (std::is_same_v<int32_t, int>)
169 return ((v1 ^ *r) & (~v2 ^ *r)) < 0;
170
171 bool s1 = (v1 < 0);
172 bool s2 = !(v2 < 0);
173 bool sr = (*r < 0);
174 return s1 != sr && s2 != sr;
175 // also: return s1 == s2 && s1 != sr;
176}

178template <typename T> inline
179typename std::enable_if_t<std::is_unsigned_v<T> || std::is_signed_v<T>, bool>
180qMulOverflow(T v1, T v2, T *r)
181{
182 // use the next biggest type
183 // Note: for 64-bit systems where __int128 isn't supported, this will cause an error.
184 using LargerInt = QIntegerForSize<sizeof(T) * 2>;
185 using Larger = typename std::conditional_t<std::is_signed_v<T>,
186 typename LargerInt::Signed, typename LargerInt::Unsigned>;
187 Larger lr = Larger(v1) * Larger(v2);
188 *r = T(lr);
189 return lr > (std::numeric_limits<T>::max)() || lr < (std::numeric_limits<T>::min)();
190}

192# if defined(Q_INTRINSIC_MUL_OVERFLOW64)
193template <> inline bool qMulOverflow(quint64 v1, quint64 v2, quint64 *r)
194{
195 *r = v1 * v2;
196 return Q_UMULH(v1, v2);
197}
198template <> inline bool qMulOverflow(qint64 v1, qint64 v2, qint64 *r)
199{
200 // This is slightly more complex than the unsigned case above: the sign bit
201 // of 'low' must be replicated as the entire 'high', so the only valid
202 // values for 'high' are 0 and -1. Use unsigned multiply since it's the same
203 // as signed for the low bits and use a signed right shift to verify that
204 // 'high' is nothing but sign bits that match the sign of 'low'.
205
206 qint64 high = Q_SMULH(v1, v2);
207 *r = qint64(quint64(v1) * quint64(v2));
208 return (*r >> 63) != high;
209}

229// 32-bit qMulOverflow is fine with the generic code above
230
231template <> inline bool qAddOverflow(quint64 v1, quint64 v2, quint64 *r)
232{
233# if defined(Q_PROCESSOR_X86_64)
234 return _addcarry_u64(0, v1, v2, reinterpret_cast<unsigned __int64 *>(r));
235# else
236 uint low, high;
237 uchar carry = _addcarry_u32(0, unsigned(v1), unsigned(v2), &low);
238 carry = _addcarry_u32(carry, v1 >> 32, v2 >> 32, &high);
239 *r = (quint64(high) << 32) | low;
240 return carry;
241# endif // !x86-64
242}
243# endif // MSVC X86
244#endif // !GCC

246// Implementations for addition, subtraction or multiplication by a
247// compile-time constant. For addition and subtraction, we simply call the code
248// that detects overflow at runtime. For multiplication, we compare to the
249// maximum possible values before multiplying to ensure no overflow happens.

251template <typename T, T V2> bool qAddOverflow(T v1, std::integral_constant<T, V2>, T *r)
252{
253 return qAddOverflow(v1, V2, r);

271template <typename T, T V2> bool qMulOverflow(T v1, std::integral_constant<T, V2>, T *r)
272{
273 // Runtime detection for anything smaller than or equal to a register
274 // width, as most architectures' multiplication instructions actually
275 // produce a result twice as wide as the input registers, allowing us to
276 // efficiently detect the overflow.
277 if constexpr (sizeof(T) <= sizeof(qregisteruint)) {
278 return qMulOverflow(v1, V2, r);
279
280#ifdef Q_INTRINSIC_MUL_OVERFLOW64
281 } else if constexpr (sizeof(T) <= sizeof(quint64)) {
282 // If we have intrinsics detecting overflow of 64-bit multiplications,
283 // then detect overflows through them up to 64 bits.
284 return qMulOverflow(v1, V2, r);
285#endif
286
287 } else if constexpr (V2 == 0 || V2 == 1) {
288 // trivial cases (and simplify logic below due to division by zero)
289 *r = v1 * V2;
290 return false;
291 } else if constexpr (V2 == -1) {
292 // multiplication by -1 is valid *except* for signed minimum values
293 // (necessary to avoid diving min() by -1, which is an overflow)
294 if (v1 < 0 && v1 == (std::numeric_limits<T>::min)())
295 return true;
296 *r = -v1;
297 return false;
298 } else {
299 // For 64-bit multiplications on 32-bit platforms, let's instead compare v1
300 // against the bounds that would overflow.
301 constexpr T Highest = (std::numeric_limits<T>::max)() / V2;
302 constexpr T Lowest = (std::numeric_limits<T>::min)() / V2;
303 if constexpr (Highest > Lowest) {
304 if (v1 > Highest || v1 < Lowest)
305 return true;
306 } else {
307 // this can only happen if V2 < 0
308 static_assert(V2 < 0);
309 if (v1 > Lowest || v1 < Highest)
310 return true;
311 }
312
313 *r = v1 * V2;
314 return false;
315 }
316}

318template <auto V2, typename T> bool qMulOverflow(T v1, T *r)
319{
320 if constexpr (V2 == 2)
321 return qAddOverflow(v1, v1, r);
322 return qMulOverflow(v1, std::integral_constant<T, V2>{}, r);
323}

329#if defined(Q_PROCESSOR_ARM_64) && (__has_builtin(__builtin_round) || defined(Q_CC_GNU)) && !defined(Q_CC_CLANG)
330// ARM64 has a single instruction that can do C++ rounding with conversion to integer.
331// Note current clang versions have non-constexpr __builtin_round, ### allow clang this path when they fix it.
332constexpr inline int qRound(double d)
333{ return int(__builtin_round(d)); }
334constexpr inline int qRound(float f)
335{ return int(__builtin_roundf(f)); }
336constexpr inline qint64 qRound64(double d)
337{ return qint64(__builtin_round(d)); }
338constexpr inline qint64 qRound64(float f)
339{ return qint64(__builtin_roundf(f)); }
340#elif defined(__SSE2__) && (__has_builtin(__builtin_copysign) || defined(Q_CC_GNU))
341// SSE has binary operations directly on floating point making copysign fast
342constexpr inline int qRound(double d)
343{ return int(d + __builtin_copysign(0.5, d)); }
344constexpr inline int qRound(float f)
345{ return int(f + __builtin_copysignf(0.5f, f)); }
346constexpr inline qint64 qRound64(double d)
347{ return qint64(d + __builtin_copysign(0.5, d)); }
348constexpr inline qint64 qRound64(float f)
349{ return qint64(f + __builtin_copysignf(0.5f, f)); }
350#else
351constexpr inline int qRound(double d)
352{ return d >= 0.0 ? int(d + 0.5) : int(d - 0.5); }
353constexpr inline int qRound(float d)
354{ return d >= 0.0f ? int(d + 0.5f) : int(d - 0.5f); }
355
356constexpr inline qint64 qRound64(double d)

358constexpr inline qint64 qRound64(float d)
