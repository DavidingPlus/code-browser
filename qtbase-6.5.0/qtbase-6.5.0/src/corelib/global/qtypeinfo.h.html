24template <typename T>
25inline constexpr bool qIsRelocatable = std::is_trivially_copyable_v<T> && std::is_trivially_destructible_v<T>;
26
27// Denotes types that are trivially default constructible, and for which
28// value-initialization can be achieved by filling their storage with 0 bits.
29// There is no type trait we can use for this, so we hardcode a list of
30// possibilities that we know are OK on the architectures that we support.
31// The most notable exception are pointers to data members, which for instance
32// on the Itanium ABI are initialized to -1.
33template <typename T>
34inline constexpr bool qIsValueInitializationBitwiseZero =
35 std::is_scalar_v<T> && !std::is_member_object_pointer_v<T>;

43template <typename T>
44class QTypeInfo
45{
46public:
47 enum {
48 isPointer = std::is_pointer_v<T>,
49 isIntegral = std::is_integral_v<T>,
50 isComplex = !std::is_trivial_v<T>,
51 isRelocatable = QtPrivate::qIsRelocatable<T>,
52 isValueInitializationBitwiseZero = QtPrivate::qIsValueInitializationBitwiseZero<T>,
53 };
54};

69/*!
70 \class QTypeInfoMerger
71 \inmodule QtCore
72 \internal
73
74 \brief QTypeInfoMerger merges the QTypeInfo flags of T1, T2... and presents them
75 as a QTypeInfo<T> would do.
76
77 Let's assume that we have a simple set of structs:
78
79 \snippet code/src_corelib_global_qglobal.cpp 50
80
81 To create a proper QTypeInfo specialization for A struct, we have to check
82 all sub-components; B, C and D, then take the lowest common denominator and call
83 Q_DECLARE_TYPEINFO with the resulting flags. An easier and less fragile approach is to
84 use QTypeInfoMerger, which does that automatically. So struct A would have
85 the following QTypeInfo definition:
86
87 \snippet code/src_corelib_global_qglobal.cpp 51
88*/
89template <class T, class...Ts>
90class QTypeInfoMerger
91{
92 static_assert(sizeof...(Ts) > 0);
93public:
94 static constexpr bool isComplex = ((QTypeInfo<Ts>::isComplex) || ...);
95 static constexpr bool isRelocatable = ((QTypeInfo<Ts>::isRelocatable) && ...);
96 static constexpr bool isPointer = false;
97 static constexpr bool isIntegral = false;
98 static constexpr bool isValueInitializationBitwiseZero = false;
99};

101// QTypeInfo for std::pair:
102// std::pair is spec'ed to be struct { T1 first; T2 second; }, so, unlike tuple<>,
103// we _can_ specialize QTypeInfo for pair<>:
104template <class T1, class T2>
105class QTypeInfo<std::pair<T1, T2>> : public QTypeInfoMerger<std::pair<T1, T2>, T1, T2> {};

133/*
134 Specialize a specific type with:
135
136 Q_DECLARE_TYPEINFO(type, flags);
137
138 where 'type' is the name of the type to specialize and 'flags' is
139 logically-OR'ed combination of the flags below.
140*/
141enum { /* TYPEINFO flags */
142 Q_COMPLEX_TYPE = 0,
143 Q_PRIMITIVE_TYPE = 0x1,
144 Q_RELOCATABLE_TYPE = 0x2,
145 Q_MOVABLE_TYPE = 0x2,
146 Q_DUMMY_TYPE = 0x4,
147};
148
149#define Q_DECLARE_TYPEINFO_BODY(TYPE, FLAGS) \
150class QTypeInfo<TYPE > \
151{ \
152public: \
153 enum { \
154 isComplex = (((FLAGS) & Q_PRIMITIVE_TYPE) == 0) && !std::is_trivial_v<TYPE>, \
155 isRelocatable = !isComplex || ((FLAGS) & Q_RELOCATABLE_TYPE) || QtPrivate::qIsRelocatable<TYPE>, \
156 isPointer = std::is_pointer_v< TYPE >, \
157 isIntegral = std::is_integral< TYPE >::value, \
158 isValueInitializationBitwiseZero = QtPrivate::qIsValueInitializationBitwiseZero<TYPE>, \
159 }; \
160}
161
162#define Q_DECLARE_TYPEINFO(TYPE, FLAGS) \
163template<> \
164Q_DECLARE_TYPEINFO_BODY(TYPE, FLAGS)

168template<typename T>
169Q_DECLARE_TYPEINFO_BODY(QFlags<T>, Q_PRIMITIVE_TYPE);

174/*
175 The templates below aim to find out whether one can safely instantiate an operator==() or
176 operator<() for a type.
177
178 This is tricky for containers, as most containers have unconstrained comparison operators, even though they
179 rely on the corresponding operators for its content.
180 This is especially true for all of the STL template classes that have a comparison operator defined, and
181 leads to the situation, that the compiler would try to instantiate the operator, and fail if any
182 of its template arguments does not have the operator implemented.
183
184 The code tries to cover the relevant cases for Qt and the STL, by checking (recusrsively) the value_type
185 of a container (if it exists), and checking the template arguments of pair, tuple and variant.
186*/
187namespace detail {
188
189// find out whether T is a conteiner
190// this is required to check the value type of containers for the existence of the comparison operator
191template <typename, typename = void>
192struct is_container : std::false_type {};
193template <typename T>
194struct is_container<T, std::void_t<
195 typename T::value_type,
196 std::is_convertible<decltype(std::declval<T>().begin() != std::declval<T>().end()), bool>
197>> : std::true_type {};
198
199
200// Checks the existence of the comparison operator for the class itself
201QT_WARNING_PUSH
202QT_WARNING_DISABLE_FLOAT_COMPARE
203template <typename, typename = void>
204struct has_operator_equal : std::false_type {};
205template <typename T>
206struct has_operator_equal<T, std::void_t<decltype(bool(std::declval<const T&>() == std::declval<const T&>()))>>
207 : std::true_type {};
208QT_WARNING_POP
209
210// Two forward declarations
211template<typename T, bool = is_container<T>::value>
212struct expand_operator_equal_container;
213template<typename T>
214struct expand_operator_equal_tuple;
215
216// the entry point for the public method
217template<typename T>
218using expand_operator_equal = expand_operator_equal_container<T>;
219
220// if T isn't a container check if it's a tuple like object
221template<typename T, bool>
222struct expand_operator_equal_container : expand_operator_equal_tuple<T> {};
223// if T::value_type exists, check first T::value_type, then T itself
224template<typename T>
225struct expand_operator_equal_container<T, true> :
226 std::conjunction<
227 std::disjunction<
228 std::is_same<T, typename T::value_type>, // avoid endless recursion
229 expand_operator_equal<typename T::value_type>
230 >, expand_operator_equal_tuple<T>> {};
231
232// recursively check the template arguments of a tuple like object
233template<typename ...T>
234using expand_operator_equal_recursive = std::conjunction<expand_operator_equal<T>...>;
235
236template<typename T>
237struct expand_operator_equal_tuple : has_operator_equal<T> {};
238template<typename T>
239struct expand_operator_equal_tuple<std::optional<T>> : has_operator_equal<T> {};
240template<typename T1, typename T2>
241struct expand_operator_equal_tuple<std::pair<T1, T2>> : expand_operator_equal_recursive<T1, T2> {};
242template<typename ...T>
243struct expand_operator_equal_tuple<std::tuple<T...>> : expand_operator_equal_recursive<T...> {};
244template<typename ...T>
245struct expand_operator_equal_tuple<std::variant<T...>> : expand_operator_equal_recursive<T...> {};
246
247// the same for operator<(), see above for explanations
248template <typename, typename = void>
249struct has_operator_less_than : std::false_type{};
250template <typename T>
251struct has_operator_less_than<T, std::void_t<decltype(bool(std::declval<const T&>() < std::declval<const T&>()))>>
252 : std::true_type{};
253
254template<typename T, bool = is_container<T>::value>
255struct expand_operator_less_than_container;
256template<typename T>
257struct expand_operator_less_than_tuple;
258
259template<typename T>
260using expand_operator_less_than = expand_operator_less_than_container<T>;
261
262template<typename T, bool>
263struct expand_operator_less_than_container : expand_operator_less_than_tuple<T> {};
264template<typename T>
265struct expand_operator_less_than_container<T, true> :
266 std::conjunction<
267 std::disjunction<
268 std::is_same<T, typename T::value_type>,
269 expand_operator_less_than<typename T::value_type>
270 >, expand_operator_less_than_tuple<T>
271 > {};
272
273template<typename ...T>
274using expand_operator_less_than_recursive = std::conjunction<expand_operator_less_than<T>...>;
275
276template<typename T>
277struct expand_operator_less_than_tuple : has_operator_less_than<T> {};
278template<typename T>
279struct expand_operator_less_than_tuple<std::optional<T>> : has_operator_less_than<T> {};
280template<typename T1, typename T2>
281struct expand_operator_less_than_tuple<std::pair<T1, T2>> : expand_operator_less_than_recursive<T1, T2> {};
282template<typename ...T>
283struct expand_operator_less_than_tuple<std::tuple<T...>> : expand_operator_less_than_recursive<T...> {};
284template<typename ...T>
285struct expand_operator_less_than_tuple<std::variant<T...>> : expand_operator_less_than_recursive<T...> {};

336template <typename Stream, typename, typename = void>
337struct has_ostream_operator : std::false_type {};

344template <typename Stream, typename Container, typename T>

348struct has_istream_operator : std::false_type {};
349template <typename Stream, typename T>
350struct has_istream_operator<Stream, T, std::void_t<decltype(detail::reference<Stream>() >> detail::reference<T>())>>
351 : std::true_type {};
352template <typename Stream, typename T>
353inline constexpr bool has_istream_operator_v = has_istream_operator<Stream, T>::value;
354template <typename Stream, typename Container, typename T>
355using has_istream_operator_container = std::disjunction<std::is_base_of<Container, T>, QTypeTraits::has_istream_operator<Stream, T>>;

358inline constexpr bool has_stream_operator_v = has_ostream_operator_v<Stream, T> && has_istream_operator_v<Stream, T>;
