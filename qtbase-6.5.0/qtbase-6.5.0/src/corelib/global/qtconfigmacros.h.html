20/*
21 The Qt modules' export macros.
22 The options are:
23 - defined(QT_STATIC): Qt was built or is being built in static mode
24 - defined(QT_SHARED): Qt was built or is being built in shared/dynamic mode
25 If neither was defined, then QT_SHARED is implied. If Qt was compiled in static
26 mode, QT_STATIC is defined in qconfig.h. In shared mode, QT_STATIC is implied
27 for the bootstrapped tools.
28*/
29
30#ifdef QT_BOOTSTRAPPED
31# ifdef QT_SHARED
32# error "QT_SHARED and QT_BOOTSTRAPPED together don't make sense. Please fix the build"
33# elif !defined(QT_STATIC)
34# define QT_STATIC
35# endif
36#endif
37
38#if defined(QT_SHARED) || !defined(QT_STATIC)
39# ifdef QT_STATIC
40# error "Both QT_SHARED and QT_STATIC defined, please make up your mind"
41# endif
42# ifndef QT_SHARED
43# define QT_SHARED
44# endif
45#endif

47/*
48 No, this is not an evil backdoor. QT_BUILD_INTERNAL just exports more symbols
49 for Qt's internal unit tests. If you want slower loading times and more
50 symbols that can vanish from version to version, feel free to define QT_BUILD_INTERNAL.
51*/
52#if defined(QT_BUILD_INTERNAL) && defined(QT_BUILDING_QT) && defined(QT_SHARED)
53# define Q_AUTOTEST_EXPORT Q_DECL_EXPORT
54#elif defined(QT_BUILD_INTERNAL) && defined(QT_SHARED)
55# define Q_AUTOTEST_EXPORT Q_DECL_IMPORT
56#else
57# define Q_AUTOTEST_EXPORT
58#endif

60/*
61 The QT_CONFIG macro implements a safe compile time check for features of Qt.
62 Features can be in three states:
63 0 or undefined: This will lead to a compile error when testing for it
64 -1: The feature is not available
65 1: The feature is available
66*/
67#define QT_CONFIG(feature) (1/QT_FEATURE_##feature == 1)
68#define QT_REQUIRE_CONFIG(feature) Q_STATIC_ASSERT_X(QT_FEATURE_##feature == 1, "Required feature " #feature " for file " __FILE__ " not available.")
