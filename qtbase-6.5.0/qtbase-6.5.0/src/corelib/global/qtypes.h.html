46// Unlike LL / ULL in C++, for historical reasons, we force the
47// result to be of the requested type.
48#ifdef __cplusplus
49# define Q_INT64_C(c) static_cast<long long>(c ## LL) /* signed 64 bit constant */
50# define Q_UINT64_C(c) static_cast<unsigned long long>(c ## ULL) /* unsigned 64 bit constant */
51#else
52# define Q_INT64_C(c) ((long long)(c ## LL)) /* signed 64 bit constant */
53# define Q_UINT64_C(c) ((unsigned long long)(c ## ULL)) /* unsigned 64 bit constant */

90#if defined(__cplusplus)
91/*
92 quintptr are qptrdiff is guaranteed to be the same size as a pointer, i.e.
93
94 sizeof(void *) == sizeof(quintptr)
95 && sizeof(void *) == sizeof(qptrdiff)
96
97 While size_t and qsizetype are not guaranteed to be the same size as a pointer,
98 they usually are and we do check for that in qtypes.cpp, just to be sure.
99*/
100template <int> struct QIntegerForSize;
101template <> struct QIntegerForSize<1> { typedef quint8 Unsigned; typedef qint8 Signed; };
102template <> struct QIntegerForSize<2> { typedef quint16 Unsigned; typedef qint16 Signed; };
103template <> struct QIntegerForSize<4> { typedef quint32 Unsigned; typedef qint32 Signed; };
104template <> struct QIntegerForSize<8> { typedef quint64 Unsigned; typedef qint64 Signed; };
105#if defined(Q_CC_GNU) && defined(__SIZEOF_INT128__)
106template <> struct QIntegerForSize<16> { __extension__ typedef unsigned __int128 Unsigned; __extension__ typedef __int128 Signed; };
107#endif
108template <class T> struct QIntegerForSizeof: QIntegerForSize<sizeof(T)> { };
109typedef QIntegerForSize<Q_PROCESSOR_WORDSIZE>::Signed qregisterint;
110typedef QIntegerForSize<Q_PROCESSOR_WORDSIZE>::Unsigned qregisteruint;
111typedef QIntegerForSizeof<void *>::Unsigned quintptr;
112typedef QIntegerForSizeof<void *>::Signed qptrdiff;
113typedef qptrdiff qintptr;
114using qsizetype = QIntegerForSizeof<std::size_t>::Signed;

116// These custom definitions are necessary as we're not defining our
117// datatypes in terms of the language ones, but in terms of integer
118// types that have the sime size. For instance, on a 32-bit platform,
119// qptrdiff is int, while ptrdiff_t may be aliased to long; therefore
120// using %td to print a qptrdiff would be wrong (and raise -Wformat
121// warnings), although both int and long have same bit size on that
122// platform.
123//
124// We know that sizeof(size_t) == sizeof(void *) == sizeof(qptrdiff).
125#if SIZE_MAX == 0xffffffffULL
126#define PRIuQUINTPTR "u"
127#define PRIoQUINTPTR "o"
128#define PRIxQUINTPTR "x"
129#define PRIXQUINTPTR "X"
130
131#define PRIdQPTRDIFF "d"
132#define PRIiQPTRDIFF "i"
133
134#define PRIdQINTPTR "d"
135#define PRIiQINTPTR "i"
136
137#define PRIdQSIZETYPE "d"
138#define PRIiQSIZETYPE "i"
139#elif SIZE_MAX == 0xffffffffffffffffULL
140#define PRIuQUINTPTR "llu"
141#define PRIoQUINTPTR "llo"
142#define PRIxQUINTPTR "llx"
143#define PRIXQUINTPTR "llX"
144
145#define PRIdQPTRDIFF "lld"
146#define PRIiQPTRDIFF "lli"
147
148#define PRIdQINTPTR "lld"
149#define PRIiQINTPTR "lli"
150
151#define PRIdQSIZETYPE "lld"
152#define PRIiQSIZETYPE "lli"
153#else
154#error Unsupported platform (unknown value for SIZE_MAX)
155#endif

157#endif // __cplusplus
