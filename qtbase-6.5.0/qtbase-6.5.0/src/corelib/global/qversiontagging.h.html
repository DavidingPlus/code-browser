12/*
13 * Explanation
14 *
15 * This file causes all libraries, plugins, and applications that #include this
16 * file to automatically pull in a symbol found in QtCore that encodes the
17 * current Qt version number at the time of compilation. The relocation is
18 * designed so that it's impossible for the dynamic linker to perform lazy
19 * binding. Instead, it must resolve at load time or fail. That way, attempting
20 * to load such a library or plugin while an older QtCore is loaded will fail.
21 * Similarly, if an older QtCore is found when launching an application, the
22 * application will fail to launch.
23 *
24 * It's also possible to inspect which version is required by decoding the
25 * .qtversion section. The second pointer-sized variable is the required
26 * version, for example, for Qt 6.4.1:
27 *
28 * Hex dump of section [18] '.qtversion', 16 bytes at offset 0x1ee48:
29 * 0x00000000 b0ffffff ffffffff 01040600 00000000 ................
30 * ^^^^^^^^ ^^^^^^^^
31 *
32 * There will only be one copy of the section in the output library or application.
33 *
34 * This functionality can be disabled by defining QT_NO_VERSION_TAGGING. It's
35 * disabled if Qt was built statically.
36 *
37 * Windows notes:
38 *
39 * On Windows, the address of a __declspec(dllimport) variable is not a
40 * constant expression, unlike Unix systems. So we instead use the address of
41 * the import variable, which is created by prefixing the external name with
42 * "__imp_". Using that variable causes an import of the corresponding symbol
43 * from QtCore DLL.
44 *
45 * With MinGW (GCC and Clang), we use a C++17 inline variable, so the compiler
46 * and linker automatically merge the variables. The "used" __attribute__
47 * tells the compiler to always emit that variable, whether it's used or not.
48 *
49 * MSVC has no equivalent to that attribute, so instead we create an extern
50 * const variable and tell the linker to merge them all via
51 * __declspec(selectany).
52 *
53 * Unix notes:
54 *
55 * On Unix, we use the same C++17 inline variable solution as MinGW, but we
56 * don't need the "__imp_" trick.
57 *
58 * Additionally, on ELF systems like Linux and FreeBSD, the symbol in question
59 * is simply "qt_version_tag" in both QtCore and in this ELF module, but it
60 * has an ELF version attached to it (see qversiontagging.cpp and
61 * QtFlagHandlingHelpers.cmake). That way, the error message from the dynamic
62 * linker will say it can't find version "Qt_6.x".
63 */
64
65namespace QtPrivate {
66struct QVersionTag
67{
68 const void *symbol;
69 quintptr version;
70 constexpr QVersionTag(const void *sym, int currentVersion = QT_VERSION)
71 : symbol(sym), version(currentVersion)
72 {}
73};
74}

76#if !defined(QT_NO_VERSION_TAGGING) && (defined(QT_BUILD_CORE_LIB) || defined(QT_BOOTSTRAPPED) || defined(QT_STATIC))
77// don't make tags in QtCore, bootstrapped systems or if the user asked not to
78# define QT_NO_VERSION_TAGGING
79#endif

81#if defined(Q_OS_WIN)
82# ifdef Q_PROCESSOR_X86_32
83// 32-bit x86 convention does prepend a _
84# define QT_MANGLE_IMPORT_PREFIX _imp__
85# else
86// Calling convention on other architectures does not prepend a _
87# define QT_MANGLE_IMPORT_PREFIX __imp_
88# endif
89# ifdef Q_CC_MSVC
90# pragma section(".qtversion",read,shared)
91# define QT_VERSION_TAG_SECTION __declspec(allocate(".qtversion"))
92# define QT_VERSION_TAG_ATTRIBUTE __declspec(selectany) extern const
93# else
94# define QT_VERSION_TAG_ATTRIBUTE __attribute__((used)) constexpr inline
95# endif
96# define QT_VERSION_TAG2(sym, imp) \
97 extern "C" const char * const imp; \
98 QT_VERSION_TAG_ATTRIBUTE QT_VERSION_TAG_SECTION QtPrivate::QVersionTag sym ## _used(&imp)
99# define QT_VERSION_TAG(sym, imp) QT_VERSION_TAG2(sym, imp)
100#elif defined(Q_CC_GNU) && __has_attribute(used)
101# ifdef Q_OS_DARWIN
102# define QT_VERSION_TAG_SECTION __attribute__((section("__DATA,.qtversion")))
103# endif
104# define QT_VERSION_TAG_ATTRIBUTE __attribute__((visibility("hidden"), used))
105# define QT_VERSION_TAG2(sym, imp) \
106 extern "C" Q_DECL_IMPORT const char sym; \
107 QT_VERSION_TAG_ATTRIBUTE QT_VERSION_TAG_SECTION constexpr inline QtPrivate::QVersionTag sym ## _use(&sym)
108# define QT_VERSION_TAG(sym, imp) QT_VERSION_TAG2(sym, imp)
109#endif
110
111#ifdef Q_OF_ELF
112# define QT_VERSION_TAG_SYMBOL(prefix, sym, m, n) sym
113#else
114# define QT_VERSION_TAG_SYMBOL2(prefix, sym, m, n) prefix ## sym ## _ ## m ## _ ## n
115# define QT_VERSION_TAG_SYMBOL(prefix, sym, m, n) QT_VERSION_TAG_SYMBOL2(prefix, sym, m, n)
116#endif

118#if defined(QT_VERSION_TAG) && !defined(QT_NO_VERSION_TAGGING)
119# ifndef QT_VERSION_TAG_SECTION
120# define QT_VERSION_TAG_SECTION __attribute__((section(".qtversion")))
121# endif
122# define QT_MANGLED_VERSION_TAG_IMPORT QT_VERSION_TAG_SYMBOL(QT_MANGLE_IMPORT_PREFIX, QT_MANGLE_NAMESPACE(qt_version_tag), QT_VERSION_MAJOR, QT_VERSION_MINOR)
123# define QT_MANGLED_VERSION_TAG QT_VERSION_TAG_SYMBOL(, QT_MANGLE_NAMESPACE(qt_version_tag), QT_VERSION_MAJOR, QT_VERSION_MINOR)
124
125QT_VERSION_TAG(QT_MANGLED_VERSION_TAG, QT_MANGLED_VERSION_TAG_IMPORT);

127# undef QT_MANGLED_VERSION_TAG
128# undef QT_MANGLED_VERSION_TAG_IMPORT
129# undef QT_VERSION_TAG_SECTION
130#endif
