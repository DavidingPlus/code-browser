22/*
23 QT_VERSION is (major << 16) | (minor << 8) | patch.
24*/
25#define QT_VERSION QT_VERSION_CHECK(QT_VERSION_MAJOR, QT_VERSION_MINOR, QT_VERSION_PATCH)
26/*
27 can be used like #if (QT_VERSION >= QT_VERSION_CHECK(6, 4, 0))
28*/
29#define QT_VERSION_CHECK(major, minor, patch) ((major<<16)|(minor<<8)|(patch))

31/*
32 Helper macros to make some simple code active in Qt 6 or Qt 7 only,
33 like:
34 struct QT6_ONLY(Q_CORE_EXPORT) QTrivialClass
35 {
36 void QT7_ONLY(Q_CORE_EXPORT) void operate();
37 }
38*/
39#if QT_VERSION_MAJOR == 7
40# define QT7_ONLY(...) __VA_ARGS__
41# define QT6_ONLY(...)
42#elif QT_VERSION_MAJOR == 6
43# define QT7_ONLY(...)
44# define QT6_ONLY(...) __VA_ARGS__
45#else
46# error Qt major version not 6 or 7
47#endif
48
49/* Macro and tag type to help overload resolution on functions
50 that are, e.g., QT_REMOVED_SINCE'ed. Example use:
51
52 #if QT_CORE_REMOVED_SINCE(6, 4)
53 int size() const;
54 #endif
55 qsizetype size(QT6_DECL_NEW_OVERLOAD) const;
56
57 in the normal cpp file:
58
59 qsizetype size(QT6_IMPL_NEW_OVERLOAD) const {
60 ~~~
61 }
62
63 in removed_api.cpp:
64
65 int size() const { return int(size(QT6_CALL_NEW_OVERLOAD)); }
66*/
67#ifdef Q_QDOC
68# define QT6_DECL_NEW_OVERLOAD
69# define QT6_DECL_NEW_OVERLOAD_TAIL
70# define QT6_IMPL_NEW_OVERLOAD
71# define QT6_IMPL_NEW_OVERLOAD_TAIL
72# define QT6_CALL_NEW_OVERLOAD
73# define QT6_CALL_NEW_OVERLOAD_TAIL
74#else
75# define QT6_DECL_NEW_OVERLOAD QT6_ONLY(Qt::Disambiguated_t = Qt::Disambiguated)
76# define QT6_DECL_NEW_OVERLOAD_TAIL QT6_ONLY(, QT6_DECL_NEW_OVERLOAD)
77# define QT6_IMPL_NEW_OVERLOAD QT6_ONLY(Qt::Disambiguated_t)
78# define QT6_IMPL_NEW_OVERLOAD_TAIL QT6_ONLY(, QT6_IMPL_NEW_OVERLOAD)
79# define QT6_CALL_NEW_OVERLOAD QT6_ONLY(Qt::Disambiguated)
80# define QT6_CALL_NEW_OVERLOAD_TAIL QT6_ONLY(, QT6_CALL_NEW_OVERLOAD)
81#endif
