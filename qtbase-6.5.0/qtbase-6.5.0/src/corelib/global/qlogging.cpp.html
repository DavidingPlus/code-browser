173static int checked_var_value(const char *varname)
174{
175 // qEnvironmentVariableIntValue returns 0 on both parsing failure and on
176 // empty, but we need to distinguish between the two for backwards
177 // compatibility reasons.
178 QByteArray str = qgetenv(varName: varname);
179 if (str.isEmpty())
180 return 0;
181
182 bool ok;
183 int value = str.toInt(ok: &ok, base: 0);
184 return ok ? value : 1;
185}

187static bool isFatal(QtMsgType msgType)
188{
189 if (msgType == QtFatalMsg)
190 return true;
191
192 if (msgType == QtCriticalMsg) {
193 static QAtomicInt fatalCriticals = checked_var_value(varname: "QT_FATAL_CRITICALS");
194
195 // it's fatal if the current value is exactly 1,
196 // otherwise decrement if it's non-zero
197 return fatalCriticals.loadRelaxed() && fatalCriticals.fetchAndAddRelaxed(valueToAdd: -1) == 1;
198 }
199
200 if (msgType == QtWarningMsg || msgType == QtCriticalMsg) {
201 static QAtomicInt fatalWarnings = checked_var_value(varname: "QT_FATAL_WARNINGS");
202
203 // it's fatal if the current value is exactly 1,
204 // otherwise decrement if it's non-zero
205 return fatalWarnings.loadRelaxed() && fatalWarnings.fetchAndAddRelaxed(valueToAdd: -1) == 1;
206 }
207
208 return false;
209}

227/*!
228 Returns true if writing to \c stderr will end up in a console/terminal visible to the user.
229
230 This is typically the case if the application was started from the command line.
231
232 If the application is started without a controlling console/terminal, but the parent
233 process reads \c stderr and presents it to the user in some other way, the parent process
234 may override the detection in this function by setting the QT_ASSUME_STDERR_HAS_CONSOLE
235 environment variable to \c 1.
236
237 \note Qt Creator does not implement a pseudo TTY, nor does it launch apps with
238 the override environment variable set, but it will read stderr and print it to
239 the user, so in effect this function cannot be used to conclude that stderr
240 output will _not_ be visible to the user, as even if this function returns false,
241 the output might still end up visible to the user. For this reason, we don't guard
242 the stderr output in the default message handler with stderrHasConsoleAttached().
243
244 \internal
245 \sa systemHasStderr()
246*/
247static bool stderrHasConsoleAttached()
248{
249 static const bool stderrHasConsoleAttached = []() -> bool {
250 if (!systemHasStderr())
251 return false;
252
253 if (qEnvironmentVariableIntValue(varName: "QT_LOGGING_TO_CONSOLE")) {
254 fprintf(stderr, format: "warning: Environment variable QT_LOGGING_TO_CONSOLE is deprecated, use\n"
255 "QT_ASSUME_STDERR_HAS_CONSOLE and/or QT_FORCE_STDERR_LOGGING instead.\n");
256 return true;
257 }
258
259 if (qEnvironmentVariableIntValue(varName: "QT_ASSUME_STDERR_HAS_CONSOLE"))
260 return true;
261
262#if defined(Q_OS_WIN)
263 return GetConsoleWindow();
264#elif defined(Q_OS_UNIX)
265# ifndef _PATH_TTY
266# define _PATH_TTY "/dev/tty"
267# endif
268
269 // If we can open /dev/tty, we have a controlling TTY
270 int ttyDevice = -1;
271 if ((ttyDevice = qt_safe_open(_PATH_TTY, O_RDONLY)) >= 0) {
272 qt_safe_close(fd: ttyDevice);
273 return true;
274 } else if (errno == ENOENT || errno == EPERM || errno == ENXIO) {
275 // Fall back to isatty for some non-critical errors
276 return isatty(STDERR_FILENO);
277 } else {
278 return false;
279 }
280#else
281 return false; // No way to detect if stderr has a console attached
282#endif
283 }();
284
285 return stderrHasConsoleAttached;
286}

346static inline void convert_to_wchar_t_elided(wchar_t *d, size_t space, const char *s) noexcept
347{
348 size_t len = qstrlen(s);
349 if (len + 1 > space) {
350 const size_t skip = len - space + 4; // 4 for "..." + '\0'
351 s += skip;
352 len -= skip;
353 for (int i = 0; i < 3; ++i)
354 *d++ = L'.';
355 }
356 while (len--)
357 *d++ = *s++;
358 *d++ = 0;
359}

983 \internal
984*/
985Q_AUTOTEST_EXPORT QByteArray qCleanupFuncinfo(QByteArray info)
986{
987 // Strip the function info down to the base function name
988 // note that this throws away the template definitions,
989 // the parameter types (overloads) and any const/volatile qualifiers.
990
991 if (info.isEmpty())
992 return info;
993
994 qsizetype pos;
995
996 // Skip trailing [with XXX] for templates (gcc), but make
997 // sure to not affect Objective-C message names.
998 pos = info.size() - 1;
999 if (info.endsWith(c: ']') && !(info.startsWith(c: '+') || info.startsWith(c: '-'))) {
1000 while (--pos) {
1001 if (info.at(i: pos) == '[')
1002 info.truncate(pos);
1003 }
1004 }
1005
1006 // operator names with '(', ')', '<', '>' in it
1007 static const char operator_call[] = "operator()";
1008 static const char operator_lessThan[] = "operator<";
1009 static const char operator_greaterThan[] = "operator>";
1010 static const char operator_lessThanEqual[] = "operator<=";
1011 static const char operator_greaterThanEqual[] = "operator>=";
1012
1013 // canonize operator names
1014 info.replace(before: "operator ", after: "operator");
1015
1016 // remove argument list
1017 forever {
1018 int parencount = 0;
1019 pos = info.lastIndexOf(c: ')');
1020 if (pos == -1) {
1021 // Don't know how to parse this function name
1022 return info;
1023 }
1024 if (info.indexOf(c: '>', from: pos) != -1
1025 || info.indexOf(c: ':', from: pos) != -1) {
1026 // that wasn't the function argument list.
1027 pos = info.size();
1028 break;
1029 }
1030
1031 // find the beginning of the argument list
1032 --pos;
1033 ++parencount;
1034 while (pos && parencount) {
1035 if (info.at(i: pos) == ')')
1036 ++parencount;
1037 else if (info.at(i: pos) == '(')
1038 --parencount;
1039 --pos;
1040 }
1041 if (parencount != 0)
1042 return info;
1043
1044 info.truncate(pos: ++pos);
1045
1046 if (info.at(i: pos - 1) == ')') {
1047 if (info.indexOf(bv: operator_call) == pos - qsizetype(strlen(s: operator_call)))
1048 break;
1049
1050 // this function returns a pointer to a function
1051 // and we matched the arguments of the return type's parameter list
1052 // try again
1053 info.remove(index: 0, len: info.indexOf(c: '('));
1054 info.chop(n: 1);
1055 continue;
1056 } else {
1057 break;
1058 }
1059 }
1060
1061 // find the beginning of the function name
1062 int parencount = 0;
1063 int templatecount = 0;
1064 --pos;
1065
1066 // make sure special characters in operator names are kept
1067 if (pos > -1) {
1068 switch (info.at(i: pos)) {
1069 case ')':
1070 if (info.indexOf(bv: operator_call) == pos - qsizetype(strlen(s: operator_call)) + 1)
1071 pos -= 2;
1072 break;
1073 case '<':
1074 if (info.indexOf(bv: operator_lessThan) == pos - qsizetype(strlen(s: operator_lessThan)) + 1)
1075 --pos;
1076 break;
1077 case '>':
1078 if (info.indexOf(bv: operator_greaterThan) == pos - qsizetype(strlen(s: operator_greaterThan)) + 1)
1079 --pos;
1080 break;
1081 case '=': {
1082 auto operatorLength = qsizetype(strlen(s: operator_lessThanEqual));
1083 if (info.indexOf(bv: operator_lessThanEqual) == pos - operatorLength + 1)
1084 pos -= 2;
1085 else if (info.indexOf(bv: operator_greaterThanEqual) == pos - operatorLength + 1)
1086 pos -= 2;
1087 break;
1088 }
1089 default:
1090 break;
1091 }
1092 }
1093
1094 while (pos > -1) {
1095 if (parencount < 0 || templatecount < 0)
1096 return info;
1097
1098 char c = info.at(i: pos);
1099 if (c == ')')
1100 ++parencount;
1101 else if (c == '(')
1102 --parencount;
1103 else if (c == '>')
1104 ++templatecount;
1105 else if (c == '<')
1106 --templatecount;
1107 else if (c == ' ' && templatecount == 0 && parencount == 0)
1108 break;
1109
1110 --pos;
1111 }
1112 info = info.mid(index: pos + 1);
1113
1114 // remove trailing '*', '&' that are part of the return argument
1115 while ((info.at(i: 0) == '*')
1116 || (info.at(i: 0) == '&'))
1117 info = info.mid(index: 1);
1118
1119 // we have the full function name now.
1120 // clean up the templates
1121 while ((pos = info.lastIndexOf(c: '>')) != -1) {
1122 if (!info.contains(c: '<'))
1123 break;
1124
1125 // find the matching close
1126 qsizetype end = pos;
1127 templatecount = 1;
1128 --pos;
1129 while (pos && templatecount) {
1130 char c = info.at(i: pos);
1131 if (c == '>')
1132 ++templatecount;
1133 else if (c == '<')
1134 --templatecount;
1135 --pos;
1136 }
1137 ++pos;
1138 info.remove(index: pos, len: end - pos + 1);
1139 }
1140
1141 return info;
1142}

1454 if (lib.data() != cachedFname) {
1455 cachedFname = lib.data();
1456 cachedLibrary = QString::fromUtf8(cachedFname, lib.size());
1457 }
1458 return { cachedLibrary, function };
1459 };
1460# else
1461 // The results of backtrace_symbols looks like this:
1462 // /lib/libc.so.6(__libc_start_main+0xf3) [0x4a937413]
1463 // The offset and function name are optional.
1464 // This regexp tries to extract the library name (without the path) and the function name.
1465 // This code is protected by QMessagePattern::mutex so it is thread safe on all compilers
1466 static const QRegularExpression rx(QStringLiteral("^(?:[^(]*/)?([^(/]+)\\(([^+]*)(?:[\\+[a-f0-9x]*)?\\) \\[[a-f0-9x]*\\]$"));
1467
1468 auto decodeFrame = [&](void *&addr) -> DecodedFrame {
1469 QScopedPointer<char*, QScopedPointerPodDeleter> strings(backtrace_symbols(&addr, 1));
1470 QString trace = QString::fromUtf8(strings.data()[0]);
1471 QRegularExpressionMatch m = rx.match(trace);
1472 if (!m.hasMatch())
1473 return {};
1474
1475 QString library = m.captured(1);
1476 QString function = m.captured(2);
1477
1478 // skip the trace from QtCore that are because of the qDebug itself
1479 if (shouldSkipFrame(library, function))
1480 return {};
1481
1482 function = demangled(function);
1483 return { library, function };
1484 };
1485# endif
1486
1487 for (void *&addr : buffer) {
1488 DecodedFrame frame = decodeFrame(addr);
1489 if (!frame.library.isEmpty()) {
1490 if (frame.function.isEmpty())
1491 result.append(u'?' + frame.library + u'?');
1492 else
1493 result.append(frame.function);
1494 } else {
1495 // innermost, unknown frames are usually the logging framework itself
1496 if (!result.isEmpty())
1497 result.append(QStringLiteral("???"));
1498 }
1499
1500 if (result.size() == frameCount)
1501 break;
1502 }
1503 return result;
1504}

1530/*!
1531 \relates <QtLogging>
1532 \since 5.4
1533
1534 Generates a formatted string out of the \a type, \a context, \a str arguments.
1535
1536 qFormatLogMessage returns a QString that is formatted according to the current message pattern.
1537 It can be used by custom message handlers to format output similar to Qt's default message
1538 handler.
1539
1540 The function is thread-safe.
1541
1542 \sa qInstallMessageHandler(), qSetMessagePattern()
1543 */
1544QString qFormatLogMessage(QtMsgType type, const QMessageLogContext &context, const QString &str)
1545{
1546 QString message;
1547
1548 const auto locker = qt_scoped_lock(mutex&: QMessagePattern::mutex);
1549
1550 QMessagePattern *pattern = qMessagePattern();
1551 if (!pattern) {
1552 // after destruction of static QMessagePattern instance
1553 message.append(s: str);
1554 return message;
1555 }
1556
1557 bool skip = false;
1558
1559#ifndef QT_BOOTSTRAPPED
1560 int timeArgsIdx = 0;
1561#ifdef QLOGGING_HAVE_BACKTRACE
1562 int backtraceArgsIdx = 0;
1563#endif
1564#endif
1565
1566 // we do not convert file, function, line literals to local encoding due to overhead
1567 for (int i = 0; pattern->tokens[i]; ++i) {
1568 const char *token = pattern->tokens[i];
1569 if (token == endifTokenC) {
1570 skip = false;
1571 } else if (skip) {
1572 // we skip adding messages, but we have to iterate over
1573 // timeArgsIdx and backtraceArgsIdx anyway
1574#ifndef QT_BOOTSTRAPPED
1575 if (token == timeTokenC)
1576 timeArgsIdx++;
1577#ifdef QLOGGING_HAVE_BACKTRACE
1578 else if (token == backtraceTokenC)
1579 backtraceArgsIdx++;
1580#endif
1581#endif
1582 } else if (token == messageTokenC) {
1583 message.append(s: str);
1584 } else if (token == categoryTokenC) {
1585#ifndef Q_OS_ANDROID
1586 // Don't add the category to the message on Android
1587 message.append(s: QLatin1StringView(context.category));
1588#endif
1589 } else if (token == typeTokenC) {
1590 switch (type) {
1591 case QtDebugMsg: message.append(s: "debug"_L1); break;
1592 case QtInfoMsg: message.append(s: "info"_L1); break;
1593 case QtWarningMsg: message.append(s: "warning"_L1); break;
1594 case QtCriticalMsg:message.append(s: "critical"_L1); break;
1595 case QtFatalMsg: message.append(s: "fatal"_L1); break;
1596 }
1597 } else if (token == fileTokenC) {
1598 if (context.file)
1599 message.append(s: QLatin1StringView(context.file));
1600 else
1601 message.append(s: "unknown"_L1);
1602 } else if (token == lineTokenC) {
1603 message.append(s: QString::number(context.line));
1604 } else if (token == functionTokenC) {
1605 if (context.function)
1606 message.append(s: QString::fromLatin1(ba: qCleanupFuncinfo(info: context.function)));
1607 else
1608 message.append(s: "unknown"_L1);
1609#ifndef QT_BOOTSTRAPPED
1610 } else if (token == pidTokenC) {
1611 message.append(QString::number(QCoreApplication::applicationPid()));
1612 } else if (token == appnameTokenC) {
1613 message.append(QCoreApplication::applicationName());
1614 } else if (token == threadidTokenC) {
1615 // print the TID as decimal
1616 message.append(QString::number(qt_gettid()));
1617 } else if (token == qthreadptrTokenC) {
1618 message.append("0x"_L1);
1619 message.append(QString::number(qlonglong(QThread::currentThread()->currentThread()), 16));
1620#ifdef QLOGGING_HAVE_BACKTRACE
1621 } else if (token == backtraceTokenC) {
1622 QMessagePattern::BacktraceParams backtraceParams = pattern->backtraceArgs.at(backtraceArgsIdx);
1623 backtraceArgsIdx++;
1624 message.append(formatBacktraceForLogMessage(backtraceParams, context.function));
1625#endif
1626 } else if (token == timeTokenC) {
1627 QString timeFormat = pattern->timeArgs.at(timeArgsIdx);
1628 timeArgsIdx++;
1629 if (timeFormat == "process"_L1) {
1630 quint64 ms = pattern->timer.elapsed();
1631 message.append(QString::asprintf("%6d.%03d", uint(ms / 1000), uint(ms % 1000)));
1632 } else if (timeFormat == "boot"_L1) {
1633 // just print the milliseconds since the elapsed timer reference
1634 // like the Linux kernel does
1635 uint ms = QDeadlineTimer::current().deadline();
1636 message.append(QString::asprintf("%6d.%03d", uint(ms / 1000), uint(ms % 1000)));
1637#if QT_CONFIG(datestring)
1638 } else if (timeFormat.isEmpty()) {
1639 message.append(QDateTime::currentDateTime().toString(Qt::ISODate));
1640 } else {
1641 message.append(QDateTime::currentDateTime().toString(timeFormat));
1642#endif // QT_CONFIG(datestring)
1643 }
1644#endif // !QT_BOOTSTRAPPED
1645 } else if (token == ifCategoryTokenC) {
1646 if (isDefaultCategory(category: context.category))
1647 skip = true;
1648#define HANDLE_IF_TOKEN(LEVEL) \
1649 } else if (token == if##LEVEL##TokenC) { \
1650 skip = type != Qt##LEVEL##Msg;
1651 HANDLE_IF_TOKEN(Debug)
1652 HANDLE_IF_TOKEN(Info)
1653 HANDLE_IF_TOKEN(Warning)
1654 HANDLE_IF_TOKEN(Critical)
1655 HANDLE_IF_TOKEN(Fatal)
1656#undef HANDLE_IF_TOKEN
1657 } else {
1658 message.append(s: QLatin1StringView(token));
1659 }
1660 }
1661 return message;
1662}
