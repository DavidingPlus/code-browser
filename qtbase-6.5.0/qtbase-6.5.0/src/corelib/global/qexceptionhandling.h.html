17/* These wrap try/catch so we can switch off exceptions later.
18
19 Beware - do not use more than one QT_CATCH per QT_TRY, and do not use
20 the exception instance in the catch block.
21 If you can't live with those constraints, don't use these macros.
22 Use the QT_NO_EXCEPTIONS macro to protect your code instead.
23*/
24Q_NORETURN Q_DECL_COLD_FUNCTION Q_CORE_EXPORT void qTerminate() noexcept;
25#ifdef QT_NO_EXCEPTIONS
26# define QT_TRY if (true)
27# define QT_CATCH(A) else
28# define QT_THROW(A) qt_noop()
29# define QT_RETHROW qt_noop()
30# define QT_TERMINATE_ON_EXCEPTION(expr) do { expr; } while (false)
31#else
32# define QT_TRY try
33# define QT_CATCH(A) catch (A)
34# define QT_THROW(A) throw A
35# define QT_RETHROW throw
36# ifdef Q_COMPILER_NOEXCEPT
37# define QT_TERMINATE_ON_EXCEPTION(expr) do { expr; } while (false)
38# else
39# define QT_TERMINATE_ON_EXCEPTION(expr) do { try { expr; } catch (...) { qTerminate(); } } while (false)
40# endif
41#endif
