32// Used to implement a type-safe and alignment-safe copy operation
33// If you want to avoid the memcpy, you must write specializations for these functions
34template <typename T> Q_ALWAYS_INLINE void qToUnaligned(const T src, void *dest)
35{
36 // Using sizeof(T) inside memcpy function produces internal compiler error with
37 // MSVC2008/ARM in tst_endian -> use extra indirection to resolve size of T.
38 const size_t size = sizeof(T);
39#if __has_builtin(__builtin_memcpy)
40 __builtin_memcpy
41#else
42 memcpy
43#endif
44 (dest, &src, size);
45}
46
47template <typename T> Q_ALWAYS_INLINE T qFromUnaligned(const void *src)
48{
49 T dest;
50 const size_t size = sizeof(T);
51#if __has_builtin(__builtin_memcpy)
52 __builtin_memcpy
53#else
54 memcpy
55#endif
56 (&dest, src, size);
57 return dest;
58}

60// These definitions are written so that they are recognized by most compilers
61// as bswap and replaced with single instruction builtins if available.
62inline constexpr quint64 qbswap_helper(quint64 source)
63{
64 return 0
65 | ((source & Q_UINT64_C(0x00000000000000ff)) << 56)
66 | ((source & Q_UINT64_C(0x000000000000ff00)) << 40)
67 | ((source & Q_UINT64_C(0x0000000000ff0000)) << 24)
68 | ((source & Q_UINT64_C(0x00000000ff000000)) << 8)
69 | ((source & Q_UINT64_C(0x000000ff00000000)) >> 8)
70 | ((source & Q_UINT64_C(0x0000ff0000000000)) >> 24)
71 | ((source & Q_UINT64_C(0x00ff000000000000)) >> 40)
72 | ((source & Q_UINT64_C(0xff00000000000000)) >> 56);
73}
74
75inline constexpr quint32 qbswap_helper(quint32 source)
76{
77 return 0
78 | ((source & 0x000000ff) << 24)
79 | ((source & 0x0000ff00) << 8)
80 | ((source & 0x00ff0000) >> 8)
81 | ((source & 0xff000000) >> 24);
82}
83
84inline constexpr quint16 qbswap_helper(quint16 source)
85{
86 return quint16( 0
87 | ((source & 0x00ff) << 8)
88 | ((source & 0xff00) >> 8) );
89}
90
91inline constexpr quint8 qbswap_helper(quint8 source)
92{
93 return source;
94}

96/*
97 * T qbswap(T source).
98 * Changes the byte order of a value from big-endian to little-endian or vice versa.
99 * This function can be used if you are not concerned about alignment issues,
100 * and it is therefore a bit more convenient and in most cases more efficient.
101*/
102template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
103inline constexpr T qbswap(T source)
104{
105 return T(qbswap_helper(typename QIntegerForSizeof<T>::Unsigned(source)));
106}
107
108// floating specializations
109template<typename Float>
110Float qbswapFloatHelper(Float source)
111{
112 // memcpy call in qFromUnaligned is recognized by optimizer as a correct way of type prunning
113 auto temp = qFromUnaligned<typename QIntegerForSizeof<Float>::Unsigned>(&source);
114 temp = qbswap(temp);
115 return qFromUnaligned<Float>(&temp);
116}
117
118inline qfloat16 qbswap(qfloat16 source)
119{
120 return qbswapFloatHelper(source);
121}
122
123inline float qbswap(float source)
124{
125 return qbswapFloatHelper(source);
126}
127
128inline double qbswap(double source)
129{
130 return qbswapFloatHelper(source);
131}

133/*
134 * qbswap(const T src, const void *dest);
135 * Changes the byte order of \a src from big-endian to little-endian or vice versa
136 * and stores the result in \a dest.
137 * There is no alignment requirements for \a dest.
138*/
139template <typename T> inline void qbswap(const T src, void *dest)
140{
141 qToUnaligned<T>(qbswap(src), dest);
142}

202/* T qFromLittleEndian(const void *src)
203 * This function will read a little-endian encoded value from \a src
204 * and return the value in host-endian encoding.
205 * There is no requirement that \a src must be aligned.
206*/
207template <typename T> inline T qFromLittleEndian(const void *src)
208{
209 return qFromLittleEndian(qFromUnaligned<T>(src));
210}
211
212template <> inline quint8 qFromLittleEndian<quint8>(const void *src)
213{ return static_cast<const quint8 *>(src)[0]; }
214template <> inline qint8 qFromLittleEndian<qint8>(const void *src)
215{ return static_cast<const qint8 *>(src)[0]; }

217/* This function will read a big-endian (also known as network order) encoded value from \a src

231template<class S>
232class QSpecialInteger
233{
234 typedef typename S::StorageType T;
235 T val;
236public:
237 QSpecialInteger() = default;
238 explicit constexpr QSpecialInteger(T i) : val(S::toSpecial(i)) {}
239
240 QSpecialInteger &operator =(T i) { val = S::toSpecial(i); return *this; }
241 operator T() const { return S::fromSpecial(val); }
242
243 bool operator ==(QSpecialInteger<S> i) const { return val == i.val; }
244 bool operator !=(QSpecialInteger<S> i) const { return val != i.val; }
245
246 QSpecialInteger &operator +=(T i)
247 { return (*this = S::fromSpecial(val) + i); }
248 QSpecialInteger &operator -=(T i)
249 { return (*this = S::fromSpecial(val) - i); }
250 QSpecialInteger &operator *=(T i)
251 { return (*this = S::fromSpecial(val) * i); }
252 QSpecialInteger &operator >>=(T i)
253 { return (*this = S::fromSpecial(val) >> i); }
254 QSpecialInteger &operator <<=(T i)
255 { return (*this = S::fromSpecial(val) << i); }
256 QSpecialInteger &operator /=(T i)
257 { return (*this = S::fromSpecial(val) / i); }
258 QSpecialInteger &operator %=(T i)
259 { return (*this = S::fromSpecial(val) % i); }
260 QSpecialInteger &operator |=(T i)
261 { return (*this = S::fromSpecial(val) | i); }
262 QSpecialInteger &operator &=(T i)
263 { return (*this = S::fromSpecial(val) & i); }
264 QSpecialInteger &operator ^=(T i)
265 { return (*this = S::fromSpecial(val) ^ i); }
266 QSpecialInteger &operator ++()
267 { return (*this = S::fromSpecial(val) + 1); }
268 QSpecialInteger &operator --()
269 { return (*this = S::fromSpecial(val) - 1); }
270 QSpecialInteger operator ++(int)
271 {
272 QSpecialInteger<S> pre = *this;
273 *this += 1;
274 return pre;
275 }
276 QSpecialInteger operator --(int)
277 {
278 QSpecialInteger<S> pre = *this;
279 *this -= 1;
280 return pre;
281 }
282
283 static constexpr QSpecialInteger max()
284 { return QSpecialInteger((std::numeric_limits<T>::max)()); }
285 static constexpr QSpecialInteger min()
286 { return QSpecialInteger((std::numeric_limits<T>::min)()); }
287};
