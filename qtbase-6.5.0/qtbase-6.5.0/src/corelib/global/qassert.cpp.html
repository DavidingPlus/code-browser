62/*
63 The Q_ASSERT macro calls this function when the test fails.
64*/
65void qt_assert(const char *assertion, const char *file, int line) noexcept
66{
67 QMessageLogger(file, line, nullptr)
68 .fatal(msg: "ASSERT: \"%s\" in file %s, line %d", assertion, file, line);
69}

78}

80/*!
81 \macro void Q_CHECK_PTR(void *pointer)
82 \relates <QtAssert>
83
84 If \a pointer is \nullptr, prints a message containing the source
85 code's file name and line number, saying that the program ran out
86 of memory and aborts program execution. It throws \c std::bad_alloc instead
87 if exceptions are enabled.
88
89 Q_CHECK_PTR does nothing if \c QT_NO_DEBUG and \c QT_NO_EXCEPTIONS were
90 defined during compilation. Therefore you must not use Q_CHECK_PTR to check
91 for successful memory allocations because the check will be disabled in
92 some cases.
93
94 Example:
95
96 \snippet code/src_corelib_global_qglobal.cpp 21
97
98 \sa qWarning(), {Debugging Techniques}
99*/
100
101/*!
102 \fn template <typename T> T *q_check_ptr(T *p)
103 \relates <QtAssert>
104
105 Uses Q_CHECK_PTR on \a p, then returns \a p.
106
107 This can be used as an inline version of Q_CHECK_PTR.
108*/
109
110/*!
111 \internal
112 The Q_CHECK_PTR macro calls this function if an allocation check
113 fails.
114*/
115void qt_check_pointer(const char *n, int l) noexcept
116{
117 // make separate printing calls so that the first one may flush;
118 // the second one could want to allocate memory (fputs prints a
119 // newline and stderr auto-flushes).
120 fputs(s: "Out of memory", stderr);
121 fprintf(stderr, format: " in %s, line %d\n", n, l);
122
123 std::terminate();
124}

126/*
127 \internal
128 Allows you to throw an exception without including <new>
129 Called internally from Q_CHECK_PTR on certain OS combinations
130*/
131void qBadAlloc()
132{
133#ifndef QT_NO_EXCEPTIONS
134 throw std::bad_alloc();
135#else
136 std::terminate();
137#endif

140/*!
141 \macro void Q_ASSUME(bool expr)
142 \relates <QtAssert>
143 \since 5.0
144
145 Causes the compiler to assume that \a expr is \c true. This macro is useful
146 for improving code generation, by providing the compiler with hints about
147 conditions that it would not otherwise know about. However, there is no
148 guarantee that the compiler will actually use those hints.
149
150 This macro could be considered a "lighter" version of \l{Q_ASSERT()}. While
151 Q_ASSERT will abort the program's execution if the condition is \c false,
152 Q_ASSUME will tell the compiler not to generate code for those conditions.
153 Therefore, it is important that the assumptions always hold, otherwise
154 undefined behavior may occur.
155
156 If \a expr is a constantly \c false condition, Q_ASSUME will tell the compiler
157 that the current code execution cannot be reached. That is, Q_ASSUME(false)
158 is equivalent to Q_UNREACHABLE().
159
160 In debug builds the condition is enforced by an assert to facilitate debugging.
161
162 \note Q_LIKELY() tells the compiler that the expression is likely, but not
163 the only possibility. Q_ASSUME tells the compiler that it is the only
164 possibility.
165
166 \sa Q_ASSERT(), Q_UNREACHABLE(), Q_LIKELY()
167*/
168
169/*!
170 \macro void Q_UNREACHABLE()
171 \relates <QtAssert>
172 \since 5.0
173
174 Tells the compiler that the current point cannot be reached by any
175 execution, so it may optimize any code paths leading here as dead code, as
176 well as code continuing from here.
177
178 This macro is useful to mark impossible conditions. For example, given the
179 following enum:
180
181 \snippet code/src_corelib_global_qglobal.cpp qunreachable-enum
182
183 One can write a switch table like so:
184
185 \snippet code/src_corelib_global_qglobal.cpp qunreachable-switch
186
187 The advantage of inserting Q_UNREACHABLE() at that point is that the
188 compiler is told not to generate code for a shape variable containing that
189 value. If the macro is missing, the compiler will still generate the
190 necessary comparisons for that value. If the case label were removed, some
191 compilers could produce a warning that some enum values were not checked.
192
193 By using this macro in impossible conditions, code coverage may be improved
194 as dead code paths may be eliminated.
195
196 In debug builds the condition is enforced by an assert to facilitate debugging.
197
198 \note Use the macro Q_UNREACHABLE_RETURN() to insert return statements for
199 compilers that need them, without causing warnings for compilers that
200 complain about its presence.
201
202 \sa Q_ASSERT(), Q_ASSUME(), qFatal(), Q_UNREACHABLE_RETURN()
203*/
204
205/*!
206 \macro void Q_UNREACHABLE_RETURN(...)
207 \relates <QtAssert>
208 \since 6.5
209
210 This is equivalent to
211 \code
212 Q_UNREACHABLE();
213 return __VA_ARGS__;
214 \endcode
215 except it omits the return on compilers that would warn about it.
216
217 \sa Q_UNREACHABLE()
