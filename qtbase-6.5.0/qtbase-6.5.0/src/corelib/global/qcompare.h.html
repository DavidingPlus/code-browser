16namespace QtPrivate {
17using CompareUnderlyingType = qint8;
18
19// [cmp.categories.pre] / 1
20enum class Ordering : CompareUnderlyingType
21{
22 Equal = 0,
23 Equivalent = Equal,
24 Less = -1,
25 Greater = 1
26};
27
28enum class Uncomparable : CompareUnderlyingType
29{
30 Unordered = -127
31};
32
33} // namespace QtPrivate
34
35// [cmp.partialord]
36class QPartialOrdering
37{
38public:
39 static const QPartialOrdering Less;
40 static const QPartialOrdering Equivalent;
41 static const QPartialOrdering Greater;
42 static const QPartialOrdering Unordered;
43
44 friend constexpr bool operator==(QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
45 { return p.isOrdered() && p.m_order == 0; }
46 friend constexpr bool operator!=(QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
47 { return p.isOrdered() && p.m_order != 0; }
48 friend constexpr bool operator< (QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
49 { return p.isOrdered() && p.m_order < 0; }
50 friend constexpr bool operator<=(QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
51 { return p.isOrdered() && p.m_order <= 0; }
52 friend constexpr bool operator> (QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
53 { return p.isOrdered() && p.m_order > 0; }
54 friend constexpr bool operator>=(QPartialOrdering p, QtPrivate::CompareAgainstLiteralZero) noexcept
55 { return p.isOrdered() && p.m_order >= 0; }
56
57 friend constexpr bool operator==(QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
58 { return p.isOrdered() && 0 == p.m_order; }
59 friend constexpr bool operator!=(QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
60 { return p.isOrdered() && 0 != p.m_order; }
61 friend constexpr bool operator< (QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
62 { return p.isOrdered() && 0 < p.m_order; }
63 friend constexpr bool operator<=(QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
64 { return p.isOrdered() && 0 <= p.m_order; }
65 friend constexpr bool operator> (QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
66 { return p.isOrdered() && 0 > p.m_order; }
67 friend constexpr bool operator>=(QtPrivate::CompareAgainstLiteralZero, QPartialOrdering p) noexcept
68 { return p.isOrdered() && 0 >= p.m_order; }
69
70 friend constexpr bool operator==(QPartialOrdering p1, QPartialOrdering p2) noexcept
71 { return p1.m_order == p2.m_order; }
72 friend constexpr bool operator!=(QPartialOrdering p1, QPartialOrdering p2) noexcept
73 { return p1.m_order != p2.m_order; }
74
75private:
76 constexpr explicit QPartialOrdering(QtPrivate::Ordering order) noexcept
77 : m_order(static_cast<QtPrivate::CompareUnderlyingType>(order))
78 {}
79 constexpr explicit QPartialOrdering(QtPrivate::Uncomparable order) noexcept
80 : m_order(static_cast<QtPrivate::CompareUnderlyingType>(order))
81 {}
82
83 // instead of the exposition only is_ordered member in [cmp.partialord],
84 // use a private function
85 constexpr bool isOrdered() noexcept
86 { return m_order != static_cast<QtPrivate::CompareUnderlyingType>(QtPrivate::Uncomparable::Unordered); }
87
88 QtPrivate::CompareUnderlyingType m_order;
89};
90
91inline constexpr QPartialOrdering QPartialOrdering::Less(QtPrivate::Ordering::Less);
92inline constexpr QPartialOrdering QPartialOrdering::Equivalent(QtPrivate::Ordering::Equivalent);
93inline constexpr QPartialOrdering QPartialOrdering::Greater(QtPrivate::Ordering::Greater);
94inline constexpr QPartialOrdering QPartialOrdering::Unordered(QtPrivate::Uncomparable::Unordered);
