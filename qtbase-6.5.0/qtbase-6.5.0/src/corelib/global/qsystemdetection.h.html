18/*
19 The operating system, must be one of: (Q_OS_x)
20
21 DARWIN - Any Darwin system (macOS, iOS, watchOS, tvOS)
22 MACOS - macOS
23 IOS - iOS
24 WATCHOS - watchOS
25 TVOS - tvOS
26 WIN32 - Win32 (Windows 2000/XP/Vista/7 and Windows Server 2003/2008)
27 CYGWIN - Cygwin
28 SOLARIS - Sun Solaris
29 HPUX - HP-UX
30 LINUX - Linux [has variants]
31 FREEBSD - FreeBSD [has variants]
32 NETBSD - NetBSD
33 OPENBSD - OpenBSD
34 INTERIX - Interix
35 AIX - AIX
36 HURD - GNU Hurd
37 QNX - QNX [has variants]
38 QNX6 - QNX RTP 6.1
39 LYNX - LynxOS
40 BSD4 - Any BSD 4.4 system
41 UNIX - Any UNIX BSD/SYSV system
42 ANDROID - Android platform
43 HAIKU - Haiku
44 WEBOS - LG WebOS
45
46 The following operating systems have variants:
47 LINUX - both Q_OS_LINUX and Q_OS_ANDROID are defined when building for Android
48 - only Q_OS_LINUX is defined if building for other Linux systems
49 MACOS - both Q_OS_BSD4 and Q_OS_IOS are defined when building for iOS
50 - both Q_OS_BSD4 and Q_OS_MACOS are defined when building for macOS
51 FREEBSD - Q_OS_FREEBSD is defined only when building for FreeBSD with a BSD userland
52 - Q_OS_FREEBSD_KERNEL is always defined on FreeBSD, even if the userland is from GNU
53*/

247# define QT_DARWIN_PLATFORM_SDK_EQUAL_OR_ABOVE(macos, ios, tvos, watchos) \
248 ((defined(__MAC_OS_X_VERSION_MAX_ALLOWED) && macos != __MAC_NA && __MAC_OS_X_VERSION_MAX_ALLOWED >= macos) || \
249 (defined(__IPHONE_OS_VERSION_MAX_ALLOWED) && ios != __IPHONE_NA && __IPHONE_OS_VERSION_MAX_ALLOWED >= ios) || \
250 (defined(__TV_OS_VERSION_MAX_ALLOWED) && tvos != __TVOS_NA && __TV_OS_VERSION_MAX_ALLOWED >= tvos) || \
251 (defined(__WATCH_OS_VERSION_MAX_ALLOWED) && watchos != __WATCHOS_NA && __WATCH_OS_VERSION_MAX_ALLOWED >= watchos))
252
253# define QT_DARWIN_DEPLOYMENT_TARGET_BELOW(macos, ios, tvos, watchos) \
254 ((defined(__MAC_OS_X_VERSION_MIN_REQUIRED) && macos != __MAC_NA && __MAC_OS_X_VERSION_MIN_REQUIRED < macos) || \
255 (defined(__IPHONE_OS_VERSION_MIN_REQUIRED) && ios != __IPHONE_NA && __IPHONE_OS_VERSION_MIN_REQUIRED < ios) || \
256 (defined(__TV_OS_VERSION_MIN_REQUIRED) && tvos != __TVOS_NA && __TV_OS_VERSION_MIN_REQUIRED < tvos) || \
257 (defined(__WATCH_OS_VERSION_MIN_REQUIRED) && watchos != __WATCHOS_NA && __WATCH_OS_VERSION_MIN_REQUIRED < watchos))
