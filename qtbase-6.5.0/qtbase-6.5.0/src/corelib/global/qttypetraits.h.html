19// like std::to_underlying
20template <typename Enum>
21constexpr std::underlying_type_t<Enum> qToUnderlying(Enum e) noexcept
22{
23 return static_cast<std::underlying_type_t<Enum>>(e);
24}

26#ifndef QT_NO_AS_CONST
27
28// this adds const to non-const objects (like std::as_const)
29template <typename T>
30constexpr typename std::add_const<T>::type &qAsConst(T &t) noexcept { return t; }
31// prevent rvalue arguments:
32template <typename T>
33void qAsConst(const T &&) = delete;
34
35#endif // QT_NO_AS_CONST

37// like std::exchange
38template <typename T, typename U = T>
39constexpr T qExchange(T &t, U &&newValue)
40noexcept(std::conjunction_v<std::is_nothrow_move_constructible<T>,
41 std::is_nothrow_assignable<T &, U>>)
42{
43 T old = std::move(t);
44 t = std::forward<U>(newValue);
45 return old;
46}
