67/*
68 QT_DEPRECATED_SINCE(major, minor) evaluates as true if the Qt version is greater than
69 the deprecation point specified.
70
71 Use it to specify from which version of Qt a function or class has been deprecated
72
73 Example:
74 #if QT_DEPRECATED_SINCE(5,1)
75 QT_DEPRECATED void deprecatedFunction(); //function deprecated since Qt 5.1
76 #endif
77
78*/
79#ifdef QT_DEPRECATED
80#define QT_DEPRECATED_SINCE(major, minor) (QT_VERSION_CHECK(major, minor, 0) > QT_DISABLE_DEPRECATED_UP_TO)
81#else
82#define QT_DEPRECATED_SINCE(major, minor) 0
83#endif

210/*
211 QT_IF_DEPRECATED_SINCE(major, minor, whenTrue, whenFalse) expands to
212 \a whenTrue if the specified (\a major, \a minor) version is less than or
213 equal to the deprecation version defined by QT_DISABLE_DEPRECATED_UP_TO,
214 and to \a whenFalse otherwise.
215
216 Currently used for QT_INLINE_SINCE(maj, min), but can also be helpful for
217 other macros of that kind.
218
219 The implementation uses QT_DEPRECATED_SINCE(maj, min) to define a bunch of
220 helper QT_IF_DEPRECATED_SINCE_X_Y macros, which expand to \a whenTrue or
221 \a whenFalse depending on the value of QT_DEPRECATED_SINCE.
222
223 If you need to use QT_IF_DEPRECATED_SINCE() for a (major, minor) version,
224 that is not yet covered by the list below, you need to copy the definition
225 and change the major and minor versions accordingly. For example, for
226 version (X, Y), you will need to add
227
228 \code
229 #if QT_DEPRECATED_SINCE(X, Y)
230 # define QT_IF_DEPRECATED_SINCE_X_Y(whenTrue, whenFalse) whenFalse
231 #else
232 # define QT_IF_DEPRECATED_SINCE_X_Y(whenTrue, whenFalse) whenTrue
233 #endif
234 \endcode
235*/
236
237#define QT_IF_DEPRECATED_SINCE(major, minor, whenTrue, whenFalse) \
238 QT_IF_DEPRECATED_SINCE_ ## major ## _ ## minor(whenTrue, whenFalse)
