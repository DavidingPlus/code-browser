166/*****************************************************************************
167 QTextStream manipulators

169
170typedef QTextStream & (*QTextStreamFunction)(QTextStream &);// manipulator function
171typedef void (QTextStream::*QTSMFI)(int); // manipulator w/int argument
172typedef void (QTextStream::*QTSMFC)(QChar); // manipulator w/QChar argument
173
174
175class Q_CORE_EXPORT QTextStreamManipulator
176{
177public:
178 constexpr QTextStreamManipulator(QTSMFI m, int a) noexcept : mf(m), mc(nullptr), arg(a), ch() {}
179 constexpr QTextStreamManipulator(QTSMFC m, QChar c) noexcept : mf(nullptr), mc(m), arg(-1), ch(c) {}
180 void exec(QTextStream &s) { if (mf) { (s.*mf)(arg); } else { (s.*mc)(ch); } }
181
182private:
183 QTSMFI mf; // QTextStream member function
184 QTSMFC mc; // QTextStream member function
185 int arg; // member function argument
186 QChar ch;
187};
188
189inline QTextStream &operator>>(QTextStream &s, QTextStreamFunction f)
190{ return (*f)(s); }
191
192inline QTextStream &operator<<(QTextStream &s, QTextStreamFunction f)
193{ return (*f)(s); }
194
195inline QTextStream &operator<<(QTextStream &s, QTextStreamManipulator m)
196{ m.exec(s); return s; }
