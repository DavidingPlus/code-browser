192class StreamStateSaver
193{
194public:
195 inline StreamStateSaver(QDataStream *s) : stream(s), oldStatus(s->status())
196 {
197 if (!stream->isDeviceTransactionStarted())
198 stream->resetStatus();
199 }
200 inline ~StreamStateSaver()
201 {
202 if (oldStatus != QDataStream::Ok) {
203 stream->resetStatus();
204 stream->setStatus(oldStatus);
205 }
206 }
207
208private:
209 QDataStream *stream;
210 QDataStream::Status oldStatus;
211};

213template <typename Container>
214QDataStream &readArrayBasedContainer(QDataStream &s, Container &c)
215{
216 StreamStateSaver stateSaver(&s);
217
218 c.clear();
219 quint32 n;
220 s >> n;
221 c.reserve(n);
222 for (quint32 i = 0; i < n; ++i) {
223 typename Container::value_type t;
224 s >> t;
225 if (s.status() != QDataStream::Ok) {
226 c.clear();
227 break;
228 }
229 c.append(t);
230 }
231
232 return s;
233}
234
235template <typename Container>
236QDataStream &readListBasedContainer(QDataStream &s, Container &c)
237{
238 StreamStateSaver stateSaver(&s);
239
240 c.clear();
241 quint32 n;
242 s >> n;
243 for (quint32 i = 0; i < n; ++i) {
244 typename Container::value_type t;
245 s >> t;
246 if (s.status() != QDataStream::Ok) {
247 c.clear();
248 break;
249 }
250 c << t;
251 }
252
253 return s;
254}

256template <typename Container>
257QDataStream &readAssociativeContainer(QDataStream &s, Container &c)
258{
259 StreamStateSaver stateSaver(&s);
260
261 c.clear();
262 quint32 n;
263 s >> n;
264 for (quint32 i = 0; i < n; ++i) {
265 typename Container::key_type k;
266 typename Container::mapped_type t;
267 s >> k >> t;
268 if (s.status() != QDataStream::Ok) {
269 c.clear();
270 break;
271 }
272 c.insert(k, t);
273 }
274
275 return s;
276}

278template <typename Container>
279QDataStream &writeSequentialContainer(QDataStream &s, const Container &c)
280{
281 s << quint32(c.size());
282 for (const typename Container::value_type &t : c)
283 s << t;
284
285 return s;
286}
287
288template <typename Container>
289QDataStream &writeAssociativeContainer(QDataStream &s, const Container &c)
290{
291 s << quint32(c.size());
292 auto it = c.constBegin();
293 auto end = c.constEnd();
294 while (it != end) {
295 s << it.key() << it.value();
296 ++it;
297 }
298
299 return s;
300}
301
302template <typename Container>
303QDataStream &writeAssociativeMultiContainer(QDataStream &s, const Container &c)
304{
305 s << quint32(c.size());
306 auto it = c.constBegin();
307 auto end = c.constEnd();
308 while (it != end) {
309 const auto rangeStart = it++;
310 while (it != end && rangeStart.key() == it.key())
311 ++it;
312 const qint64 last = std::distance(rangeStart, it) - 1;
313 for (qint64 i = last; i >= 0; --i) {
314 auto next = std::next(rangeStart, i);
315 s << next.key() << next.value();
316 }
317 }
318
319 return s;
320}
