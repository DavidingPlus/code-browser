32class XmlStringRef
33{
34public:
35 const QString *m_string = nullptr;
36 qsizetype m_pos = 0;
37 qsizetype m_size = 0;
38
39 constexpr XmlStringRef() = default;
40 constexpr inline XmlStringRef(const QString *string, qsizetype pos, qsizetype length)
41 : m_string(string), m_pos(pos), m_size(length)
42 {
43 }
44 XmlStringRef(const QString *string)
45 : XmlStringRef(string, 0, string->size())
46 {
47 }
48
49 operator QXmlString() const {
50 if (!m_string)
51 return QXmlString();
52 QStringPrivate d = m_string->data_ptr();
53 d.setBegin(d.data() + m_pos);
54 d.size = m_size;
55 return QXmlString(std::move(d));
56 }
57
58 void clear() { m_string = nullptr; m_pos = 0; m_size= 0; }
59 QStringView view() const { return m_string ? QStringView(m_string->data() + m_pos, m_size) : QStringView(); }
60 bool isEmpty() const { return m_size == 0; }
61 bool isNull() const { return !m_string; }
62 QString toString() const { return view().toString(); }

74 using const_reverse_iterator = QStringView::const_reverse_iterator;

118template <typename T> class QXmlStreamSimpleStack
119{
120 Q_DISABLE_COPY_MOVE(QXmlStreamSimpleStack)
121
122 T *data;
123 qsizetype tos, cap;
124public:
125 inline QXmlStreamSimpleStack()
126 : data(nullptr), tos(-1), cap(0)
127 {}
128 inline ~QXmlStreamSimpleStack()
129 {
130 if (data) {
131 std::destroy_n(data, size());
132 free(data);
133 }
134 }
135
136 inline void reserve(qsizetype extraCapacity)
137 {
138 if (tos + extraCapacity + 1 > cap) {
139 cap = qMax(a: tos + extraCapacity + 1, b: cap << 1 );
140 void *ptr = realloc(ptr: static_cast<void *>(data), size: cap * sizeof(T));
141 data = reinterpret_cast<T *>(ptr);
142 Q_CHECK_PTR(data);
143 }
144 }
145
146 inline T &push() { reserve(extraCapacity: 1); return rawPush(); }
147 inline T &rawPush() { return *new (data + (++tos)) T; }
148 inline const T &top() const { return data[tos]; }
149 inline T &top() { return data[tos]; }
150 inline T pop() { T t = std::move(data[tos]); std::destroy_at(data + tos); --tos; return t; }
151 inline T &operator[](qsizetype index) { return data[index]; }
152 inline const T &at(qsizetype index) const { return data[index]; }
153 inline qsizetype size() const { return tos + 1; }
154 inline void resize(qsizetype s) { tos = s - 1; }
155 inline bool isEmpty() const { return tos < 0; }
156 inline void clear() { tos = -1; }
157
158 using const_iterator = const T*;
159 using iterator = T*;
160 T *begin() { return data; }
161 const T *begin() const { return data; }
162 const T *cbegin() const { return begin(); }
163 T *end() { return data + size(); }
164 const T *end() const { return data + size(); }
165 const T *cend() const { return end(); }
166};

198 XmlStringRef addToStringStorage(QAnyStringView s)
199 {
200 qsizetype pos = tagStackStringStorageSize;
201 if (pos != tagStackStringStorage.size())
202 tagStackStringStorage.resize(size: pos);
203 s.visit(v: [&](auto s) { tagStackStringStorage.append(s); });
204 qsizetype sz = (tagStackStringStorage.size() - pos);
205 tagStackStringStorageSize += sz;
206 return XmlStringRef(&tagStackStringStorage, pos, sz);
207 }
208
209 QXmlStreamSimpleStack<Tag> tagStack;

211
212 inline Tag tagStack_pop() {
213 Tag tag = tagStack.pop();
214 tagStackStringStorageSize = tag.tagStackStringStorageSize;
215 namespaceDeclarations.resize(s: tag.namespaceDeclarationsSize);
216 tagsDone = tagStack.isEmpty();
217 return tag;
218 }
219 inline Tag &tagStack_push() {
220 Tag &tag = tagStack.push();
221 tag.tagStackStringStorageSize = tagStackStringStorageSize;
222 tag.namespaceDeclarationsSize = namespaceDeclarations.size();
223 return tag;
224 }

259 // these hash tables use a QStringView as a key to avoid creating QStrings
260 // just for lookup. The keys are usually views into Entity::name and thus
261 // are guaranteed to have the same lifetime as the referenced data:
262 QHash<QStringView, Entity> entityHash;
263 QHash<QStringView, Entity> parameterEntityHash;
264 struct QEntityReference
265 {
266 QHash<QStringView, Entity> *hash;
267 QStringView name;
268 };
269 QXmlStreamSimpleStack<QEntityReference> entityReferenceStack;
270 int entityExpansionLimit = 4096;
271 int entityLength = 0;
272 inline bool referenceEntity(QHash<QStringView, Entity> *hash, Entity &entity)
273 {
274 Q_ASSERT(hash);
275 if (entity.isCurrentlyReferenced) {
276 raiseWellFormedError(message: QXmlStream::tr(sourceText: "Self-referencing entity detected."));
277 return false;
278 }
279 // entityLength represents the amount of additional characters the
280 // entity expands into (can be negative for e.g. &amp;). It's used to
281 // avoid DoS attacks through recursive entity expansions
282 entityLength += entity.value.size() - entity.name.size() - 2;
283 if (entityLength > entityExpansionLimit) {
284 raiseWellFormedError(message: QXmlStream::tr(sourceText: "Entity expands to more characters than the entity expansion limit."));
285 return false;
286 }
287 entity.isCurrentlyReferenced = true;
288 entityReferenceStack.push() = { .hash: hash, .name: entity.name };
289 injectToken(tokenToInject: ENTITY_DONE);
290 return true;
291 }
