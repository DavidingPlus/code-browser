20static inline uchar hexdig(uint u)
21{
22 return (u < 0xa ? '0' + u : 'a' + u - 0xa);
23}

25static QByteArray escapedString(const QString &s)
26{
27 // give it a minimum size to ensure the resize() below always adds enough space
28 QByteArray ba(qMax(a: s.size(), b: 16), Qt::Uninitialized);
29
30 uchar *cursor = reinterpret_cast<uchar *>(const_cast<char *>(ba.constData()));
31 const uchar *ba_end = cursor + ba.size();
32 const char16_t *src = reinterpret_cast<const char16_t *>(s.constBegin());
33 const char16_t *const end = reinterpret_cast<const char16_t *>(s.constEnd());
34
35 while (src != end) {
36 if (cursor >= ba_end - 6) {
37 // ensure we have enough space
38 int pos = cursor - (const uchar *)ba.constData();
39 ba.resize(size: ba.size()*2);
40 cursor = (uchar *)ba.data() + pos;
41 ba_end = (const uchar *)ba.constData() + ba.size();
42 }
43
44 char16_t u = *src++;
45 if (u < 0x80) {
46 if (u < 0x20 || u == 0x22 || u == 0x5c) {
47 *cursor++ = '\\';
48 switch (u) {
49 case 0x22:
50 *cursor++ = '"';
51 break;
52 case 0x5c:
53 *cursor++ = '\\';
54 break;
55 case 0x8:
56 *cursor++ = 'b';
57 break;
58 case 0xc:
59 *cursor++ = 'f';
60 break;
61 case 0xa:
62 *cursor++ = 'n';
63 break;
64 case 0xd:
65 *cursor++ = 'r';
66 break;
67 case 0x9:
68 *cursor++ = 't';
69 break;
70 default:
71 *cursor++ = 'u';
72 *cursor++ = '0';
73 *cursor++ = '0';
74 *cursor++ = hexdig(u: u>>4);
75 *cursor++ = hexdig(u: u & 0xf);
76 }
77 } else {
78 *cursor++ = (uchar)u;
79 }
80 } else if (QUtf8Functions::toUtf8<QUtf8BaseTraits>(u, dst&: cursor, src, end) < 0) {
81 // failed to get valid utf8 use JSON escape sequence
82 *cursor++ = '\\';
83 *cursor++ = 'u';
84 *cursor++ = hexdig(u: u>>12 & 0x0f);
85 *cursor++ = hexdig(u: u>>8 & 0x0f);
86 *cursor++ = hexdig(u: u>>4 & 0x0f);
87 *cursor++ = hexdig(u: u & 0x0f);
88 }
89 }
90
91 ba.resize(size: cursor - (const uchar *)ba.constData());
92 return ba;
93}

95static void valueToJson(const QCborValue &v, QByteArray &json, int indent, bool compact)
96{
97 QCborValue::Type type = v.type();
98 switch (type) {
99 case QCborValue::True:
100 json += "true";
101 break;
102 case QCborValue::False:
103 json += "false";
104 break;
105 case QCborValue::Integer:
106 json += QByteArray::number(v.toInteger());
107 break;
108 case QCborValue::Double: {
109 const double d = v.toDouble();
110 if (qIsFinite(d))
111 json += QByteArray::number(d, format: 'g', precision: QLocale::FloatingPointShortest);
112 else
113 json += "null"; // +INF || -INF || NaN (see RFC4627#section2.4)
114 break;
115 }
116 case QCborValue::String:
117 json += '"';
118 json += escapedString(s: v.toString());
119 json += '"';
120 break;
121 case QCborValue::Array:
122 json += compact ? "[" : "[\n";
123 arrayContentToJson(
124 a: QJsonPrivate::Value::container(v), json, indent: indent + (compact ? 0 : 1), compact);
125 json += QByteArray(4*indent, ' ');
126 json += ']';
127 break;
128 case QCborValue::Map:
129 json += compact ? "{" : "{\n";
130 objectContentToJson(
131 o: QJsonPrivate::Value::container(v), json, indent: indent + (compact ? 0 : 1), compact);
132 json += QByteArray(4*indent, ' ');
133 json += '}';
134 break;
135 case QCborValue::Null:
136 default:
137 json += "null";
138 }
139}

141static void arrayContentToJson(const QCborContainerPrivate *a, QByteArray &json, int indent, bool compact)
142{
143 if (!a || a->elements.empty())
144 return;
145
146 QByteArray indentString(4*indent, ' ');
147
148 qsizetype i = 0;
149 while (true) {
150 json += indentString;
151 valueToJson(v: a->valueAt(idx: i), json, indent, compact);
152
153 if (++i == a->elements.size()) {
154 if (!compact)
155 json += '\n';
156 break;
157 }
158
159 json += compact ? "," : ",\n";
160 }
161}

164static void objectContentToJson(const QCborContainerPrivate *o, QByteArray &json, int indent, bool compact)
165{
166 if (!o || o->elements.empty())
167 return;
168
169 QByteArray indentString(4*indent, ' ');
170
171 qsizetype i = 0;
172 while (true) {
173 QCborValue e = o->valueAt(idx: i);
174 json += indentString;
175 json += '"';
176 json += escapedString(s: o->valueAt(idx: i).toString());
177 json += compact ? "\":" : "\": ";
178 valueToJson(v: o->valueAt(idx: i + 1), json, indent, compact);
179
180 if ((i += 2) == o->elements.size()) {
181 if (!compact)
182 json += '\n';
183 break;
184 }
185
186 json += compact ? "," : ",\n";
187 }
188}

190void Writer::objectToJson(const QCborContainerPrivate *o, QByteArray &json, int indent, bool compact)
191{
192 json.reserve(asize: json.size() + (o ? (int)o->elements.size() : 16));
193 json += compact ? "{" : "{\n";
194 objectContentToJson(o, json, indent: indent + (compact ? 0 : 1), compact);
195 json += QByteArray(4*indent, ' ');
196 json += compact ? "}" : "}\n";
197}
