51// Parse zone.tab table, assume lists all installed zones, if not will need to read directories
52static QTzTimeZoneHash loadTzTimeZones()
53{
54 QString path = QStringLiteral("/usr/share/zoneinfo/zone.tab");
55 if (!QFile::exists(fileName: path))
56 path = QStringLiteral("/usr/lib/zoneinfo/zone.tab");
57
58 QFile tzif(path);
59 if (!tzif.open(flags: QIODevice::ReadOnly))
60 return QTzTimeZoneHash();
61
62 QTzTimeZoneHash zonesHash;
63 while (!tzif.atEnd()) {
64 const QByteArray line = tzif.readLine().trimmed();
65 if (line.isEmpty() || line.at(i: 0) == '#') // Ignore empty or comment
66 continue;
67 // Data rows are tab-separated columns Region, Coordinates, ID, Optional Comments
68 QByteArrayView text(line);
69 int cut = text.indexOf(ch: '\t');
70 if (Q_LIKELY(cut > 0)) {
71 QTzTimeZone zone;
72 // TODO: QLocale & friends could do this look-up without UTF8-conversion:
73 zone.territory = QLocalePrivate::codeToTerritory(code: QString::fromUtf8(utf8: text.first(n: cut)));
74 text = text.sliced(pos: cut + 1);
75 cut = text.indexOf(ch: '\t');
76 if (Q_LIKELY(cut >= 0)) { // Skip over Coordinates, read ID and comment
77 text = text.sliced(pos: cut + 1);
78 cut = text.indexOf(ch: '\t'); // < 0 if line has no comment
79 if (Q_LIKELY(cut)) {
80 const QByteArray id = (cut > 0 ? text.first(n: cut) : text).toByteArray();
81 if (cut > 0)
82 zone.comment = text.sliced(pos: cut + 1).toByteArray();
83 zonesHash.insert(key: id, value: zone);
84 }
85 }
86 }
87 }
88 return zonesHash;
89}

106struct QTzHeader {

108 char tzh_version; // '\0' or '2' as of 2005
109 char tzh_reserved[15]; // reserved--must be zero
110 quint32 tzh_ttisgmtcnt; // number of trans. time flags

112 quint32 tzh_leapcnt; // number of leap seconds
113 quint32 tzh_timecnt; // number of transition times
114 quint32 tzh_typecnt; // number of local time types
115 quint32 tzh_charcnt; // number of abbr. chars
116};

134static QTzHeader parseTzHeader(QDataStream &ds, bool *ok)
135{
136 QTzHeader hdr;
137 quint8 ch;
138 *ok = false;
139
140 // Parse Magic, 4 bytes
141 ds.readRawData(hdr.tzh_magic, len: 4);
142
143 if (memcmp(s1: hdr.tzh_magic, TZ_MAGIC, n: 4) != 0 || ds.status() != QDataStream::Ok)
144 return hdr;
145
146 // Parse Version, 1 byte, before 2005 was '\0', since 2005 a '2', since 2013 a '3'
147 ds >> ch;
148 hdr.tzh_version = ch;
149 if (ds.status() != QDataStream::Ok
150 || (hdr.tzh_version != '2' && hdr.tzh_version != '\0' && hdr.tzh_version != '3')) {
151 return hdr;
152 }
153
154 // Parse reserved space, 15 bytes
155 ds.readRawData(hdr.tzh_reserved, len: 15);
156 if (ds.status() != QDataStream::Ok)
157 return hdr;
158
159 // Parse rest of header, 6 x 4-byte transition counts
160 ds >> hdr.tzh_ttisgmtcnt >> hdr.tzh_ttisstdcnt >> hdr.tzh_leapcnt >> hdr.tzh_timecnt
161 >> hdr.tzh_typecnt >> hdr.tzh_charcnt;
162
163 // Check defined maximums
164 if (ds.status() != QDataStream::Ok
165 || hdr.tzh_timecnt > TZ_MAX_TIMES
166 || hdr.tzh_typecnt > TZ_MAX_TYPES
167 || hdr.tzh_charcnt > TZ_MAX_CHARS
168 || hdr.tzh_leapcnt > TZ_MAX_LEAPS
169 || hdr.tzh_ttisgmtcnt > hdr.tzh_typecnt
170 || hdr.tzh_ttisstdcnt > hdr.tzh_typecnt) {
171 return hdr;
172 }
173
174 *ok = true;
175 return hdr;
176}

178static QList<QTzTransition> parseTzTransitions(QDataStream &ds, int tzh_timecnt, bool longTran)
179{
180 QList<QTzTransition> transitions(tzh_timecnt);
181
182 if (longTran) {
183 // Parse tzh_timecnt x 8-byte transition times
184 for (int i = 0; i < tzh_timecnt && ds.status() == QDataStream::Ok; ++i) {
185 ds >> transitions[i].tz_time;
186 if (ds.status() != QDataStream::Ok)
187 transitions.resize(size: i);
188 }
189 } else {
190 // Parse tzh_timecnt x 4-byte transition times
191 qint32 val;
192 for (int i = 0; i < tzh_timecnt && ds.status() == QDataStream::Ok; ++i) {
193 ds >> val;
194 transitions[i].tz_time = val;
195 if (ds.status() != QDataStream::Ok)
196 transitions.resize(size: i);
197 }
198 }
199
200 // Parse tzh_timecnt x 1-byte transition type index
201 for (int i = 0; i < tzh_timecnt && ds.status() == QDataStream::Ok; ++i) {
202 quint8 typeind;
203 ds >> typeind;
204 if (ds.status() == QDataStream::Ok)
205 transitions[i].tz_typeind = typeind;
206 }
207
208 return transitions;
209}

211static QList<QTzType> parseTzTypes(QDataStream &ds, int tzh_typecnt)
212{
213 QList<QTzType> types(tzh_typecnt);
214
215 // Parse tzh_typecnt x transition types
216 for (int i = 0; i < tzh_typecnt && ds.status() == QDataStream::Ok; ++i) {
217 QTzType &type = types[i];
218 // Parse UTC Offset, 4 bytes
219 ds >> type.tz_gmtoff;
220 // Parse Is DST flag, 1 byte
221 if (ds.status() == QDataStream::Ok)
222 ds >> type.tz_isdst;
223 // Parse Abbreviation Array Index, 1 byte
224 if (ds.status() == QDataStream::Ok)
225 ds >> type.tz_abbrind;
226 if (ds.status() != QDataStream::Ok)
227 types.resize(size: i);
228 }
229
230 return types;
231}

233static QMap<int, QByteArray> parseTzAbbreviations(QDataStream &ds, int tzh_charcnt, const QList<QTzType> &types)
234{
235 // Parse the abbreviation list which is tzh_charcnt long with '\0' separated strings. The
236 // QTzType.tz_abbrind index points to the first char of the abbreviation in the array, not the
237 // occurrence in the list. It can also point to a partial string so we need to use the actual typeList
238 // index values when parsing. By using a map with tz_abbrind as ordered key we get both index
239 // methods in one data structure and can convert the types afterwards.
240 QMap<int, QByteArray> map;
241 quint8 ch;
242 QByteArray input;
243 // First parse the full abbrev string
244 for (int i = 0; i < tzh_charcnt && ds.status() == QDataStream::Ok; ++i) {
245 ds >> ch;
246 if (ds.status() == QDataStream::Ok)
247 input.append(c: char(ch));
248 else
249 return map;
250 }
251 // Then extract all the substrings pointed to by types
252 for (const QTzType &type : types) {
253 QByteArray abbrev;
254 for (int i = type.tz_abbrind; input.at(i) != '\0'; ++i)
255 abbrev.append(c: input.at(i));
256 // Have reached end of an abbreviation, so add to map
257 map[type.tz_abbrind] = abbrev;
258 }
259 return map;
260}

262static void parseTzLeapSeconds(QDataStream &ds, int tzh_leapcnt, bool longTran)
263{
264 // Parse tzh_leapcnt x pairs of leap seconds
265 // We don't use leap seconds, so only read and don't store
266 qint32 val;
267 if (longTran) {
268 // v2 file format, each entry is 12 bytes long
269 qint64 time;
270 for (int i = 0; i < tzh_leapcnt && ds.status() == QDataStream::Ok; ++i) {
271 // Parse Leap Occurrence Time, 8 bytes
272 ds >> time;
273 // Parse Leap Seconds To Apply, 4 bytes
274 if (ds.status() == QDataStream::Ok)
275 ds >> val;
276 }
277 } else {
278 // v0 file format, each entry is 8 bytes long
279 for (int i = 0; i < tzh_leapcnt && ds.status() == QDataStream::Ok; ++i) {
280 // Parse Leap Occurrence Time, 4 bytes
281 ds >> val;
282 // Parse Leap Seconds To Apply, 4 bytes
283 if (ds.status() == QDataStream::Ok)
284 ds >> val;
285 }
286 }

289static QList<QTzType> parseTzIndicators(QDataStream &ds, const QList<QTzType> &types, int tzh_ttisstdcnt,
290 int tzh_ttisgmtcnt)
291{
292 QList<QTzType> result = types;
293 bool temp;
294 /*
295 Scan and discard indicators.
296
297 These indicators are only of use (by the date program) when "handling
298 POSIX-style time zone environment variables". The flags here say whether
299 the *specification* of the zone gave the time in UTC, local standard time
300 or local wall time; but whatever was specified has been digested for us,
301 already, by the zone-info compiler (zic), so that the tz_time values read
302 from the file (by parseTzTransitions) are all in UTC.
303 */
304
305 // Scan tzh_ttisstdcnt x 1-byte standard/wall indicators
306 for (int i = 0; i < tzh_ttisstdcnt && ds.status() == QDataStream::Ok; ++i)
307 ds >> temp;
308
309 // Scan tzh_ttisgmtcnt x 1-byte UTC/local indicators
310 for (int i = 0; i < tzh_ttisgmtcnt && ds.status() == QDataStream::Ok; ++i)
311 ds >> temp;
312
313 return result;
314}

316static QByteArray parseTzPosixRule(QDataStream &ds)
317{
318 // Parse POSIX rule, variable length '\n' enclosed
319 QByteArray rule;
320
321 quint8 ch;
322 ds >> ch;
323 if (ch != '\n' || ds.status() != QDataStream::Ok)
324 return rule;
325 ds >> ch;
326 while (ch != '\n' && ds.status() == QDataStream::Ok) {
327 rule.append(c: (char)ch);
328 ds >> ch;
329 }
330
331 return rule;
332}

334static QDate calculateDowDate(int year, int month, int dayOfWeek, int week)
335{
336 if (dayOfWeek == 0) // Sunday; we represent it as 7, POSIX uses 0
337 dayOfWeek = 7;
338 else if (dayOfWeek & ~7 || month < 1 || month > 12 || week < 1 || week > 5)
339 return QDate();
340
341 QDate date(year, month, 1);
342 int startDow = date.dayOfWeek();
343 if (startDow <= dayOfWeek)
344 date = date.addDays(days: dayOfWeek - startDow - 7);
345 else
346 date = date.addDays(days: dayOfWeek - startDow);
347 date = date.addDays(days: week * 7);
348 while (date.month() != month)
349 date = date.addDays(days: -7);
350 return date;
351}

353static QDate calculatePosixDate(const QByteArray &dateRule, int year)
354{
355 bool ok;
356 // Can start with M, J, or a digit
357 if (dateRule.at(i: 0) == 'M') {
358 // nth week in month format "Mmonth.week.dow"
359 QList<QByteArray> dateParts = dateRule.split(sep: '.');
360 if (dateParts.size() > 2) {
361 int month = dateParts.at(i: 0).mid(index: 1).toInt(ok: &ok);
362 int week = ok ? dateParts.at(i: 1).toInt(ok: &ok) : 0;
363 int dow = ok ? dateParts.at(i: 2).toInt(ok: &ok) : 0;
364 if (ok)
365 return calculateDowDate(year, month, dayOfWeek: dow, week);
366 }
367 } else if (dateRule.at(i: 0) == 'J') {
368 // Day of Year 1...365, ignores Feb 29.
369 // So March always starts on day 60.
370 int doy = dateRule.mid(index: 1).toInt(ok: &ok);
371 if (ok && doy > 0 && doy < 366) {
372 // Subtract 1 because we're adding days *after* the first of
373 // January, unless it's after February in a leap year, when the leap
374 // day cancels that out:
375 if (!QDate::isLeapYear(year) || doy < 60)
376 --doy;
377 return QDate(year, 1, 1).addDays(days: doy);
378 }
379 } else {
380 // Day of Year 0...365, includes Feb 29
381 int doy = dateRule.toInt(ok: &ok);
382 if (ok && doy >= 0 && doy < 366)
383 return QDate(year, 1, 1).addDays(days: doy);
384 }
385 return QDate();
386}

388// returns the time in seconds, INT_MIN if we failed to parse
389static int parsePosixTime(const char *begin, const char *end)
390{
391 // Format "hh[:mm[:ss]]"
392 int hour, min = 0, sec = 0;
393
394 const int maxHour = 137; // POSIX's extended range.
395 auto r = qstrntoll(nptr: begin, size: end - begin, base: 10);
396 hour = r.result;
397 if (!r.ok() || hour < -maxHour || hour > maxHour || r.used > 2)
398 return INT_MIN;
399 begin += r.used;
400 if (begin < end && *begin == ':') {
401 // minutes
402 ++begin;
403 r = qstrntoll(nptr: begin, size: end - begin, base: 10);
404 min = r.result;
405 if (!r.ok() || min < 0 || min > 59 || r.used > 2)
406 return INT_MIN;
407
408 begin += r.used;
409 if (begin < end && *begin == ':') {
410 // seconds
411 ++begin;
412 r = qstrntoll(nptr: begin, size: end - begin, base: 10);
413 sec = r.result;
414 if (!r.ok() || sec < 0 || sec > 59 || r.used > 2)
415 return INT_MIN;
416 begin += r.used;
417 }
418 }
419
420 // we must have consumed everything
421 if (begin != end)
422 return INT_MIN;
423
424 return (hour * 60 + min) * 60 + sec;
425}

432static int parsePosixOffset(const char *begin, const char *end)
433{
434 // Format "[+|-]hh[:mm[:ss]]"
435 // note that the sign is inverted because POSIX counts in hours West of GMT
436 bool negate = true;
437 if (*begin == '+') {
438 ++begin;
439 } else if (*begin == '-') {
440 negate = false;
441 ++begin;
442 }
443
444 int value = parsePosixTime(begin, end);
445 if (value == INT_MIN)
446 return value;
447 return negate ? -value : value;
448}

475// Returns the zone name, the offset (in seconds) and advances \a begin to
476// where the parsing ended. Returns a zone of INT_MIN in case an offset
477// couldn't be read.
478PosixZone PosixZone::parse(const char *&pos, const char *end)
479{
480 static const char offsetChars[] = "0123456789:";
481
482 const char *nameBegin = pos;
483 const char *nameEnd;
484 Q_ASSERT(pos < end);
485
486 if (*pos == '<') {
487 ++nameBegin; // skip the '<'
488 nameEnd = nameBegin;
489 while (nameEnd < end && *nameEnd != '>') {
490 // POSIX says only alphanumeric, but we allow anything
491 ++nameEnd;
492 }
493 pos = nameEnd + 1; // skip the '>'
494 } else {
495 nameEnd = nameBegin;
496 while (nameEnd < end && asciiIsLetter(ch: *nameEnd))
497 ++nameEnd;
498 pos = nameEnd;
499 }
500 if (nameEnd - nameBegin < 3)
501 return invalid(); // name must be at least 3 characters long
502
503 // zone offset, form [+-]hh:mm:ss
504 const char *zoneBegin = pos;
505 const char *zoneEnd = pos;
506 if (zoneEnd < end && (zoneEnd[0] == '+' || zoneEnd[0] == '-'))
507 ++zoneEnd;
508 while (zoneEnd < end) {
509 if (strchr(s: offsetChars, c: char(*zoneEnd)) == nullptr)
510 break;
511 ++zoneEnd;
512 }
513
514 QString name = QString::fromUtf8(utf8: nameBegin, size: nameEnd - nameBegin);
515 const int offset = zoneEnd > zoneBegin ? parsePosixOffset(begin: zoneBegin, end: zoneEnd) : InvalidOffset;
516 pos = zoneEnd;
517 // UTC+hh:mm:ss or GMT+hh:mm:ss should be read as offsets from UTC, not as a
518 // POSIX rule naming a zone as UTC or GMT and specifying a non-zero offset.
519 if (offset != 0 && (name =="UTC"_L1 || name == "GMT"_L1))
520 return invalid();
521 return {.name: std::move(name), .offset: offset};
522}

524static auto validatePosixRule(const QByteArray &posixRule)
525{
526 // Format is described here:
527 // http://www.gnu.org/software/libc/manual/html_node/TZ-Variable.html
528 // See also calculatePosixTransition()'s reference.
529 const auto parts = posixRule.split(sep: ',');
530 const struct { bool isValid, hasDst; } fail{.isValid: false, .hasDst: false}, good{.isValid: true, .hasDst: parts.size() > 1};
531 const QByteArray &zoneinfo = parts.at(i: 0);
532 if (zoneinfo.isEmpty())
533 return fail;
534
535 const char *begin = zoneinfo.begin();
536
537 // Updates begin to point after the name and offset it parses:
538 if (PosixZone::parse(pos&: begin, end: zoneinfo.end()).name.isEmpty())
539 return fail;
540
541 if (good.hasDst) {
542 if (begin >= zoneinfo.end())
543 return fail;
544 // Expect a second name and offset after the first:
545 if (PosixZone::parse(pos&: begin, end: zoneinfo.end()).name.isEmpty())
546 return fail;
547 }
548 if (begin < zoneinfo.end())
549 return fail;
550
551 if (good.hasDst) {
552 if (parts.size() != 3 || parts.at(i: 1).isEmpty() || parts.at(i: 2).isEmpty())
553 return fail;
554 for (int i = 1; i < 3; ++i) {
555 const auto tran = parts.at(i).split(sep: '/');
556 if (!calculatePosixDate(dateRule: tran.at(i: 0), year: 1972).isValid())
557 return fail;
558 if (tran.size() > 1) {
559 const auto time = tran.at(i: 1);
560 if (parsePosixTime(begin: time.begin(), end: time.end()) == INT_MIN)
561 return fail;
562 }
563 }
564 }
565 return good;
566}

568static QList<QTimeZonePrivate::Data> calculatePosixTransitions(const QByteArray &posixRule,
569 int startYear, int endYear,
570 qint64 lastTranMSecs)
571{
572 QList<QTimeZonePrivate::Data> result;
573
574 // POSIX Format is like "TZ=CST6CDT,M3.2.0/2:00:00,M11.1.0/2:00:00"
575 // i.e. "std offset dst [offset],start[/time],end[/time]"
576 // See the section about TZ at
577 // http://pubs.opengroup.org/onlinepubs/9699919799/basedefs/V1_chap08.html
578 // and the link in validatePosixRule(), above.
579 QList<QByteArray> parts = posixRule.split(sep: ',');
580
581 PosixZone stdZone, dstZone = PosixZone::invalid();
582 {
583 const QByteArray &zoneinfo = parts.at(i: 0);
584 const char *begin = zoneinfo.constBegin();
585
586 stdZone = PosixZone::parse(pos&: begin, end: zoneinfo.constEnd());
587 if (!stdZone.hasValidOffset()) {
588 stdZone.offset = 0; // reset to UTC if we failed to parse
589 } else if (begin < zoneinfo.constEnd()) {
590 dstZone = PosixZone::parse(pos&: begin, end: zoneinfo.constEnd());
591 if (!dstZone.hasValidOffset()) {
592 // if the dst offset isn't provided, it is 1 hour ahead of the standard offset
593 dstZone.offset = stdZone.offset + (60 * 60);
594 }
595 }
596 }
597
598 // If only the name part, or no DST specified, then no transitions
599 if (parts.size() == 1 || !dstZone.hasValidOffset()) {
600 QTimeZonePrivate::Data data;
601 data.atMSecsSinceEpoch = lastTranMSecs;
602 data.offsetFromUtc = stdZone.offset;
603 data.standardTimeOffset = stdZone.offset;
604 data.daylightTimeOffset = 0;
605 data.abbreviation = stdZone.name.isEmpty() ? QString::fromUtf8(ba: parts.at(i: 0)) : stdZone.name;
606 result << data;
607 return result;
608 }
609 if (parts.size() < 3 || parts.at(i: 1).isEmpty() || parts.at(i: 2).isEmpty())
610 return result; // Malformed.
611
612 // Get the std to dst transition details
613 const int twoOClock = 7200; // Default transition time, when none specified
614 const auto dstParts = parts.at(i: 1).split(sep: '/');
615 const QByteArray dstDateRule = dstParts.at(i: 0);
616 const int dstTime = dstParts.size() < 2 ? twoOClock : parsePosixTransitionTime(timeRule: dstParts.at(i: 1));
617
618 // Get the dst to std transition details
619 const auto stdParts = parts.at(i: 2).split(sep: '/');
620 const QByteArray stdDateRule = stdParts.at(i: 0);
621 const int stdTime = stdParts.size() < 2 ? twoOClock : parsePosixTransitionTime(timeRule: stdParts.at(i: 1));
622
623 if (dstDateRule.isEmpty() || stdDateRule.isEmpty() || dstTime == INT_MIN || stdTime == INT_MIN)
624 return result; // Malformed.
625
626 // Limit year to the range QDateTime can represent:
627 const int minYear = int(QDateTime::YearRange::First);
628 const int maxYear = int(QDateTime::YearRange::Last);
629 startYear = qBound(min: minYear, val: startYear, max: maxYear);
630 endYear = qBound(min: minYear, val: endYear, max: maxYear);
631 Q_ASSERT(startYear <= endYear);
632
633 for (int year = startYear; year <= endYear; ++year) {
634 // Note: std and dst, despite being QDateTime(,, UTC), have the
635 // date() and time() of the *zone*'s description of the transition
636 // moments; the atMSecsSinceEpoch values computed from them are
637 // correctly offse to be UTC-based.
638
639 QTimeZonePrivate::Data dstData; // Transition to DST
640 QDateTime dst(calculatePosixDate(dateRule: dstDateRule, year)
641 .startOfDay(zone: QTimeZone::UTC).addSecs(secs: dstTime));
642 dstData.atMSecsSinceEpoch = dst.toMSecsSinceEpoch() - stdZone.offset * 1000;
643 dstData.offsetFromUtc = dstZone.offset;
644 dstData.standardTimeOffset = stdZone.offset;
645 dstData.daylightTimeOffset = dstZone.offset - stdZone.offset;
646 dstData.abbreviation = dstZone.name;
647 QTimeZonePrivate::Data stdData; // Transition to standard time
648 QDateTime std(calculatePosixDate(dateRule: stdDateRule, year)
649 .startOfDay(zone: QTimeZone::UTC).addSecs(secs: stdTime));
650 stdData.atMSecsSinceEpoch = std.toMSecsSinceEpoch() - dstZone.offset * 1000;
651 stdData.offsetFromUtc = stdZone.offset;
652 stdData.standardTimeOffset = stdZone.offset;
653 stdData.daylightTimeOffset = 0;
654 stdData.abbreviation = stdZone.name;
655
656 if (year == startYear) {
657 // Handle the special case of fixed state, which may be represented
658 // by fake transitions at start and end of each year:
659 if (dstData.atMSecsSinceEpoch < stdData.atMSecsSinceEpoch) {
660 if (dst <= QDate(year, 1, 1).startOfDay(zone: QTimeZone::UTC)
661 && std >= QDate(year, 12, 31).endOfDay(zone: QTimeZone::UTC)) {
662 // Permanent DST:
663 dstData.atMSecsSinceEpoch = lastTranMSecs;
664 result << dstData;
665 return result;
666 }
667 } else {
668 if (std <= QDate(year, 1, 1).startOfDay(zone: QTimeZone::UTC)
669 && dst >= QDate(year, 12, 31).endOfDay(zone: QTimeZone::UTC)) {
670 // Permanent Standard time, perversely described:
671 stdData.atMSecsSinceEpoch = lastTranMSecs;
672 result << stdData;
673 return result;
674 }
675 }
676 }
677
678 const bool useStd = std.isValid() && std.date().year() == year && !stdZone.name.isEmpty();
679 const bool useDst = dst.isValid() && dst.date().year() == year && !dstZone.name.isEmpty();
680 if (useStd && useDst) {
681 if (dst < std)
682 result << dstData << stdData;
683 else
684 result << stdData << dstData;
685 } else if (useStd) {
686 result << stdData;
687 } else if (useDst) {
688 result << dstData;
689 }
690 }
691 return result;
692}

723QTzTimeZoneCacheEntry QTzTimeZoneCache::findEntry(const QByteArray &ianaId)
724{
725 QTzTimeZoneCacheEntry ret;
726 QFile tzif;
727 if (ianaId.isEmpty()) {
728 // Open system tz
729 tzif.setFileName(QStringLiteral("/etc/localtime"));
730 if (!tzif.open(flags: QIODevice::ReadOnly))
731 return ret;
732 } else {
733 // Open named tz, try modern path first, if fails try legacy path
734 tzif.setFileName("/usr/share/zoneinfo/"_L1 + QString::fromLocal8Bit(ba: ianaId));
735 if (!tzif.open(flags: QIODevice::ReadOnly)) {
736 tzif.setFileName("/usr/lib/zoneinfo/"_L1 + QString::fromLocal8Bit(ba: ianaId));
737 if (!tzif.open(flags: QIODevice::ReadOnly)) {
738 // ianaId may be a POSIX rule, taken from $TZ or /etc/TZ
739 auto check = validatePosixRule(posixRule: ianaId);
740 if (check.isValid) {
741 ret.m_hasDst = check.hasDst;
742 ret.m_posixRule = ianaId;
743 }
744 return ret;
745 }
746 }
747 }
748
749 QDataStream ds(&tzif);
750
751 // Parse the old version block of data
752 bool ok = false;
753 QByteArray posixRule;
754 QTzHeader hdr = parseTzHeader(ds, ok: &ok);
755 if (!ok || ds.status() != QDataStream::Ok)
756 return ret;
757 QList<QTzTransition> tranList = parseTzTransitions(ds, tzh_timecnt: hdr.tzh_timecnt, longTran: false);
758 if (ds.status() != QDataStream::Ok)
759 return ret;
760 QList<QTzType> typeList = parseTzTypes(ds, tzh_typecnt: hdr.tzh_typecnt);
761 if (ds.status() != QDataStream::Ok)
762 return ret;
763 QMap<int, QByteArray> abbrevMap = parseTzAbbreviations(ds, tzh_charcnt: hdr.tzh_charcnt, types: typeList);
764 if (ds.status() != QDataStream::Ok)
765 return ret;
766 parseTzLeapSeconds(ds, tzh_leapcnt: hdr.tzh_leapcnt, longTran: false);
767 if (ds.status() != QDataStream::Ok)
768 return ret;
769 typeList = parseTzIndicators(ds, types: typeList, tzh_ttisstdcnt: hdr.tzh_ttisstdcnt, tzh_ttisgmtcnt: hdr.tzh_ttisgmtcnt);
770 if (ds.status() != QDataStream::Ok)
771 return ret;
772
773 // If version 2 then parse the second block of data
774 if (hdr.tzh_version == '2' || hdr.tzh_version == '3') {
775 ok = false;
776 QTzHeader hdr2 = parseTzHeader(ds, ok: &ok);
777 if (!ok || ds.status() != QDataStream::Ok)
778 return ret;
779 tranList = parseTzTransitions(ds, tzh_timecnt: hdr2.tzh_timecnt, longTran: true);
780 if (ds.status() != QDataStream::Ok)
781 return ret;
782 typeList = parseTzTypes(ds, tzh_typecnt: hdr2.tzh_typecnt);
783 if (ds.status() != QDataStream::Ok)
784 return ret;
785 abbrevMap = parseTzAbbreviations(ds, tzh_charcnt: hdr2.tzh_charcnt, types: typeList);
786 if (ds.status() != QDataStream::Ok)
787 return ret;
788 parseTzLeapSeconds(ds, tzh_leapcnt: hdr2.tzh_leapcnt, longTran: true);
789 if (ds.status() != QDataStream::Ok)
790 return ret;
791 typeList = parseTzIndicators(ds, types: typeList, tzh_ttisstdcnt: hdr2.tzh_ttisstdcnt, tzh_ttisgmtcnt: hdr2.tzh_ttisgmtcnt);
792 if (ds.status() != QDataStream::Ok)
793 return ret;
794 posixRule = parseTzPosixRule(ds);
795 if (ds.status() != QDataStream::Ok)
796 return ret;
797 }
798 // Translate the TZ file's raw data into our internal form:
799
800 if (!posixRule.isEmpty()) {
801 auto check = validatePosixRule(posixRule);
802 if (!check.isValid) // We got a POSIX rule, but it was malformed:
803 return ret;
804 ret.m_posixRule = posixRule;
805 ret.m_hasDst = check.hasDst;
806 }
807
808 // Translate the array-index-based tz_abbrind into list index
809 const int size = abbrevMap.size();
810 ret.m_abbreviations.clear();
811 ret.m_abbreviations.reserve(size);
812 QList<int> abbrindList;
813 abbrindList.reserve(size);
814 for (auto it = abbrevMap.cbegin(), end = abbrevMap.cend(); it != end; ++it) {
815 ret.m_abbreviations.append(t: it.value());
816 abbrindList.append(t: it.key());
817 }
818 // Map tz_abbrind from map's keys (as initially read) to abbrindList's
819 // indices (used hereafter):
820 for (int i = 0; i < typeList.size(); ++i)
821 typeList[i].tz_abbrind = abbrindList.indexOf(t: typeList.at(i).tz_abbrind);
822
823 // TODO: is typeList[0] always the "before zones" data ? It seems to be ...
824 if (typeList.size())
825 ret.m_preZoneRule = { .stdOffset: typeList.at(i: 0).tz_gmtoff, .dstOffset: 0, .abbreviationIndex: typeList.at(i: 0).tz_abbrind };
826 else
827 ret.m_preZoneRule = { .stdOffset: 0, .dstOffset: 0, .abbreviationIndex: 0 };
828
829 // Offsets are stored as total offset, want to know separate UTC and DST offsets
830 // so find the first non-dst transition to use as base UTC Offset
831 int utcOffset = ret.m_preZoneRule.stdOffset;
832 for (const QTzTransition &tran : std::as_const(t&: tranList)) {
833 if (!typeList.at(i: tran.tz_typeind).tz_isdst) {
834 utcOffset = typeList.at(i: tran.tz_typeind).tz_gmtoff;
835 break;
836 }
837 }
838
839 // Now for each transition time calculate and store our rule:
840 const int tranCount = tranList.size();;
841 ret.m_tranTimes.reserve(size: tranCount);
842 // The DST offset when in effect: usually stable, usually an hour:
843 int lastDstOff = 3600;
844 for (int i = 0; i < tranCount; i++) {
845 const QTzTransition &tz_tran = tranList.at(i);
846 QTzTransitionTime tran;
847 QTzTransitionRule rule;
848 const QTzType tz_type = typeList.at(i: tz_tran.tz_typeind);
849
850 // Calculate the associated Rule
851 if (!tz_type.tz_isdst) {
852 utcOffset = tz_type.tz_gmtoff;
853 } else if (Q_UNLIKELY(tz_type.tz_gmtoff != utcOffset + lastDstOff)) {
854 /*
855 This might be a genuine change in DST offset, but could also be
856 DST starting at the same time as the standard offset changed. See
857 if DST's end gives a more plausible utcOffset (i.e. one closer to
858 the last we saw, or a simple whole hour):
859 */
860 // Standard offset inferred from net offset and expected DST offset:
861 const int inferStd = tz_type.tz_gmtoff - lastDstOff; // != utcOffset
862 for (int j = i + 1; j < tranCount; j++) {
863 const QTzType new_type = typeList.at(i: tranList.at(i: j).tz_typeind);
864 if (!new_type.tz_isdst) {
865 const int newUtc = new_type.tz_gmtoff;
866 if (newUtc == utcOffset) {
867 // DST-end can't help us, avoid lots of messy checks.
868 // else: See if the end matches the familiar DST offset:
869 } else if (newUtc == inferStd) {
870 utcOffset = newUtc;
871 // else: let either end shift us to one hour as DST offset:
872 } else if (tz_type.tz_gmtoff - 3600 == utcOffset) {
873 // Start does it
874 } else if (tz_type.tz_gmtoff - 3600 == newUtc) {
875 utcOffset = newUtc; // End does it
876 // else: prefer whichever end gives DST offset closer to
877 // last, but consider any offset > 0 "closer" than any <= 0:
878 } else if (newUtc < tz_type.tz_gmtoff
879 ? (utcOffset >= tz_type.tz_gmtoff
880 || qAbs(t: newUtc - inferStd) < qAbs(t: utcOffset - inferStd))
881 : (utcOffset >= tz_type.tz_gmtoff
882 && qAbs(t: newUtc - inferStd) < qAbs(t: utcOffset - inferStd))) {
883 utcOffset = newUtc;
884 }
885 break;
886 }
887 }
888 lastDstOff = tz_type.tz_gmtoff - utcOffset;
889 }
890 rule.stdOffset = utcOffset;
891 rule.dstOffset = tz_type.tz_gmtoff - utcOffset;
892 rule.abbreviationIndex = tz_type.tz_abbrind;
893
894 // If the rule already exist then use that, otherwise add it
895 int ruleIndex = ret.m_tranRules.indexOf(t: rule);
896 if (ruleIndex == -1) {
897 if (rule.dstOffset != 0)
898 ret.m_hasDst = true;
899 ret.m_tranRules.append(t: rule);
900 tran.ruleIndex = ret.m_tranRules.size() - 1;
901 } else {
902 tran.ruleIndex = ruleIndex;
903 }
904
905 tran.atMSecsSinceEpoch = tz_tran.tz_time * 1000;
906 ret.m_tranTimes.append(t: tran);
907 }
908
909 return ret;
910}

985QString QTzTimeZonePrivate::displayName(QTimeZone::TimeType timeType,
986 QTimeZone::NameType nameType,
987 const QLocale &locale) const
988{
989#if QT_CONFIG(icu)
990 auto lock = qt_unique_lock(mutex&: s_icu_mutex);
991 if (!m_icu)
992 m_icu = new QIcuTimeZonePrivate(m_id);
993 // TODO small risk may not match if tran times differ due to outdated files
994 // TODO Some valid TZ names are not valid ICU names, use translation table?
995 if (m_icu->isValid())
996 return m_icu->displayName(timeType, nameType, locale);
997 lock.unlock();
998#else
999 Q_UNUSED(timeType);
1000 Q_UNUSED(nameType);
1001 Q_UNUSED(locale);
1002#endif
1003 // If no ICU available then have to use abbreviations instead
1004 // Abbreviations don't have GenericTime
1005 if (timeType == QTimeZone::GenericTime)
1006 timeType = QTimeZone::StandardTime;
1007
1008 // Get current tran, if valid and is what we want, then use it
1009 const qint64 currentMSecs = QDateTime::currentMSecsSinceEpoch();
1010 QTimeZonePrivate::Data tran = data(forMSecsSinceEpoch: currentMSecs);
1011 if (tran.atMSecsSinceEpoch != invalidMSecs()
1012 && ((timeType == QTimeZone::DaylightTime && tran.daylightTimeOffset != 0)
1013 || (timeType == QTimeZone::StandardTime && tran.daylightTimeOffset == 0))) {
1014 return tran.abbreviation;
1015 }
1016
1017 // Otherwise get next tran and if valid and is what we want, then use it
1018 tran = nextTransition(afterMSecsSinceEpoch: currentMSecs);
1019 if (tran.atMSecsSinceEpoch != invalidMSecs()
1020 && ((timeType == QTimeZone::DaylightTime && tran.daylightTimeOffset != 0)
1021 || (timeType == QTimeZone::StandardTime && tran.daylightTimeOffset == 0))) {
1022 return tran.abbreviation;
1023 }
1024
1025 // Otherwise get prev tran and if valid and is what we want, then use it
1026 tran = previousTransition(beforeMSecsSinceEpoch: currentMSecs);
1027 if (tran.atMSecsSinceEpoch != invalidMSecs())
1028 tran = previousTransition(beforeMSecsSinceEpoch: tran.atMSecsSinceEpoch);
1029 if (tran.atMSecsSinceEpoch != invalidMSecs()
1030 && ((timeType == QTimeZone::DaylightTime && tran.daylightTimeOffset != 0)
1031 || (timeType == QTimeZone::StandardTime && tran.daylightTimeOffset == 0))) {
1032 return tran.abbreviation;
1033 }
1034
1035 // Otherwise is strange sequence, so work backwards through trans looking for first match, if any
1036 auto it = std::partition_point(first: tranCache().cbegin(), last: tranCache().cend(),
1037 pred: [currentMSecs](const QTzTransitionTime &at) {
1038 return at.atMSecsSinceEpoch <= currentMSecs;
1039 });
1040
1041 while (it != tranCache().cbegin()) {
1042 --it;
1043 tran = dataForTzTransition(tran: *it);
1044 int offset = tran.daylightTimeOffset;
1045 if ((timeType == QTimeZone::DaylightTime) != (offset == 0))
1046 return tran.abbreviation;
1047 }
1048
1049 // Otherwise if no match use current data
1050 return data(forMSecsSinceEpoch: currentMSecs).abbreviation;
1051}

1104QTimeZonePrivate::Data QTzTimeZonePrivate::data(qint64 forMSecsSinceEpoch) const
1105{
1106 // If the required time is after the last transition (or there were none)
1107 // and we have a POSIX rule, then use it:
1108 if (!cached_data.m_posixRule.isEmpty()
1109 && (tranCache().isEmpty() || tranCache().last().atMSecsSinceEpoch < forMSecsSinceEpoch)) {
1110 QList<QTimeZonePrivate::Data> posixTrans = getPosixTransitions(msNear: forMSecsSinceEpoch);
1111 auto it = std::partition_point(first: posixTrans.cbegin(), last: posixTrans.cend(),
1112 pred: [forMSecsSinceEpoch] (const QTimeZonePrivate::Data &at) {
1113 return at.atMSecsSinceEpoch <= forMSecsSinceEpoch;
1114 });
1115 // Use most recent, if any in the past; or the first if we have no other rules:
1116 if (it > posixTrans.cbegin() || (tranCache().isEmpty() && it < posixTrans.cend())) {
1117 QTimeZonePrivate::Data data = *(it > posixTrans.cbegin() ? it - 1 : it);
1118 data.atMSecsSinceEpoch = forMSecsSinceEpoch;
1119 return data;
1120 }
1121 }
1122 if (tranCache().isEmpty()) // Only possible if !isValid()
1123 return invalidData();
1124
1125 // Otherwise, use the rule for the most recent or first transition:
1126 auto last = std::partition_point(first: tranCache().cbegin(), last: tranCache().cend(),
1127 pred: [forMSecsSinceEpoch] (const QTzTransitionTime &at) {
1128 return at.atMSecsSinceEpoch <= forMSecsSinceEpoch;
1129 });
1130 if (last == tranCache().cbegin())
1131 return dataFromRule(rule: cached_data.m_preZoneRule, msecsSinceEpoch: forMSecsSinceEpoch);
1132
1133 --last;
1134 return dataFromRule(rule: cached_data.m_tranRules.at(i: last->ruleIndex), msecsSinceEpoch: forMSecsSinceEpoch);
1135}

1143{
1144 // If the required time is after the last transition (or there were none)
1145 // and we have a POSIX rule, then use it:
1146 if (!cached_data.m_posixRule.isEmpty()
1147 && (tranCache().isEmpty() || tranCache().last().atMSecsSinceEpoch < afterMSecsSinceEpoch)) {
1148 QList<QTimeZonePrivate::Data> posixTrans = getPosixTransitions(msNear: afterMSecsSinceEpoch);
1149 auto it = std::partition_point(first: posixTrans.cbegin(), last: posixTrans.cend(),
1150 pred: [afterMSecsSinceEpoch] (const QTimeZonePrivate::Data &at) {

1152 });
1153
1154 return it == posixTrans.cend() ? invalidData() : *it;
1155 }
1156
1157 // Otherwise, if we can find a valid tran, use its rule:
1158 auto last = std::partition_point(first: tranCache().cbegin(), last: tranCache().cend(),
1159 pred: [afterMSecsSinceEpoch] (const QTzTransitionTime &at) {
1160 return at.atMSecsSinceEpoch <= afterMSecsSinceEpoch;
1161 });
1162 return last != tranCache().cend() ? dataForTzTransition(tran: *last) : invalidData();

1165QTimeZonePrivate::Data QTzTimeZonePrivate::previousTransition(qint64 beforeMSecsSinceEpoch) const
1166{
1167 // If the required time is after the last transition (or there were none)
1168 // and we have a POSIX rule, then use it:
1169 if (!cached_data.m_posixRule.isEmpty()
1170 && (tranCache().isEmpty() || tranCache().last().atMSecsSinceEpoch < beforeMSecsSinceEpoch)) {
1171 QList<QTimeZonePrivate::Data> posixTrans = getPosixTransitions(msNear: beforeMSecsSinceEpoch);
1172 auto it = std::partition_point(first: posixTrans.cbegin(), last: posixTrans.cend(),
1173 pred: [beforeMSecsSinceEpoch] (const QTimeZonePrivate::Data &at) {
1174 return at.atMSecsSinceEpoch < beforeMSecsSinceEpoch;
1175 });
1176 if (it > posixTrans.cbegin())
1177 return *--it;
1178 // It fell between the last transition (if any) and the first of the POSIX rule:
1179 return tranCache().isEmpty() ? invalidData() : dataForTzTransition(tran: tranCache().last());
1180 }
1181
1182 // Otherwise if we can find a valid tran then use its rule
1183 auto last = std::partition_point(first: tranCache().cbegin(), last: tranCache().cend(),
1184 pred: [beforeMSecsSinceEpoch] (const QTzTransitionTime &at) {
1185 return at.atMSecsSinceEpoch < beforeMSecsSinceEpoch;
1186 });
1187 return last > tranCache().cbegin() ? dataForTzTransition(tran: *--last) : invalidData();
1188}

1214// Getting the system zone's ID:
1215
1216namespace {
1217class ZoneNameReader
1218{
1219public:
1220 QByteArray name()
1221 {
1222 /* Assumptions:
1223 a) Systems don't change which of localtime and TZ they use without a
1224 reboot.
1225 b) When they change, they use atomic renames, hence a new device and
1226 inode for the new file.
1227 c) If we change which *name* is used for a zone, while referencing
1228 the same final zoneinfo file, we don't care about the change of
1229 name (e.g. if Europe/Oslo and Europe/Berlin are both symlinks to
1230 the same CET file, continuing to use the old name, after
1231 /etc/localtime changes which of the two it points to, is
1232 harmless).
1233
1234 The alternative would be to use a file-system watcher, but they are a
1235 scarce resource.
1236 */
1237 const StatIdent local = identify(path: "/etc/localtime");
1238 const StatIdent tz = identify(path: "/etc/TZ");
1239 const StatIdent timezone = identify(path: "/etc/timezone");
1240 if (!m_name.isEmpty() && m_last.isValid()
1241 && (m_last == local || m_last == tz || m_last == timezone)) {
1242 return m_name;
1243 }
1244
1245 m_name = etcLocalTime();
1246 if (!m_name.isEmpty()) {
1247 m_last = local;
1248 return m_name;
1249 }
1250
1251 // Some systems (e.g. uClibc) have a default value for $TZ in /etc/TZ:
1252 m_name = etcContent(QStringLiteral("/etc/TZ"));
1253 if (!m_name.isEmpty()) {
1254 m_last = tz;
1255 return m_name;
1256 }
1257
1258 // Gentoo still (2020, QTBUG-87326) uses this:
1259 m_name = etcContent(QStringLiteral("/etc/timezone"));
1260 m_last = m_name.isEmpty() ? StatIdent() : timezone;
1261 return m_name;
1262 }
1263
1264private:
1265 QByteArray m_name;
1266 struct StatIdent
1267 {
1268 static constexpr unsigned long bad = ~0ul;
1269 unsigned long m_dev, m_ino;
1270 constexpr StatIdent() : m_dev(bad), m_ino(bad) {}
1271 StatIdent(const QT_STATBUF &data) : m_dev(data.st_dev), m_ino(data.st_ino) {}
1272 bool isValid() { return m_dev != bad || m_ino != bad; }
1273 bool operator==(const StatIdent &other)
1274 { return other.m_dev == m_dev && other.m_ino == m_ino; }
1275 };
1276 StatIdent m_last;

1278 static StatIdent identify(const char *path)
1279 {
1280 QT_STATBUF data;
1281 return QT_STAT(path: path, statbuf: &data) == -1 ? StatIdent() : StatIdent(data);
1282 }

1284 static QByteArray etcLocalTime()
1285 {
1286 // On most distros /etc/localtime is a symlink to a real file so extract
1287 // name from the path
1288 const auto zoneinfo = "/zoneinfo/"_L1;
1289 QString path = QStringLiteral("/etc/localtime");
1290 long iteration = getSymloopMax();
1291 // Symlink may point to another symlink etc. before being under zoneinfo/
1292 // We stop on the first path under /zoneinfo/, even if it is itself a
1293 // symlink, like America/Montreal pointing to America/Toronto
1294 do {
1295 path = QFile::symLinkTarget(fileName: path);
1296 int index = path.indexOf(s: zoneinfo);
1297 if (index >= 0) // Found zoneinfo file; extract zone name from path:
1298 return QStringView{ path }.mid(pos: index + zoneinfo.size()).toUtf8();
1299 } while (!path.isEmpty() && --iteration > 0);
1300
1301 return QByteArray();
1302 }

1304 static QByteArray etcContent(const QString &path)
1305 {
1306 QFile zone(path);
1307 if (zone.open(flags: QIODevice::ReadOnly))
1308 return zone.readAll().trimmed();
1309
1310 return QByteArray();
1311 }
1312
1313 // Any chain of symlinks longer than this is assumed to be a loop:
1314 static long getSymloopMax()
1315 {
1316#ifdef SYMLOOP_MAX
1317 // If defined, at runtime it can only be greater than this, so this is a safe bet:
1318 return SYMLOOP_MAX;
1319#else
1320 errno = 0;
1321 long result = sysconf(_SC_SYMLOOP_MAX);
1322 if (result >= 0)
1323 return result;
1324 // result is -1, meaning either error or no limit
1325 Q_ASSERT(!errno); // ... but it can't be an error, POSIX mandates _SC_SYMLOOP_MAX
1326
1327 // therefore we can make up our own limit
1328# ifdef MAXSYMLINKS
1329 return MAXSYMLINKS;
1330# else
1331 return 8;
1332# endif
1333#endif
1334 }
1335};

1343QByteArray QTzTimeZonePrivate::staticSystemTimeZoneId()
1344{
1345 // Check TZ env var first, if not populated try find it
1346 QByteArray ianaId = qgetenv(varName: "TZ");
1347
1348 // The TZ value can be ":/etc/localtime" which libc considers
1349 // to be a "default timezone", in which case it will be read
1350 // by one of the blocks below, so unset it here so it is not
1351 // considered as a valid/found ianaId
1352 if (ianaId == ":/etc/localtime")
1353 ianaId.clear();
1354 else if (ianaId.startsWith(c: ':'))
1355 ianaId = ianaId.mid(index: 1);
1356
1357 if (ianaId.isEmpty()) {
1358 Q_CONSTINIT thread_local static ZoneNameReader reader;
1359 ianaId = reader.name();
1360 }
1361
1362 return ianaId;
