36/*!
37 \internal
38 Gets the digit from a datetime. E.g.
39
40 QDateTime var(QDate(2004, 02, 02));
41 int digit = getDigit(var, Year);
42 // digit = 2004
43*/
44
45int QDateTimeParser::getDigit(const QDateTime &t, int index) const
46{
47 if (index < 0 || index >= sectionNodes.size()) {
48 qWarning(msg: "QDateTimeParser::getDigit() Internal error (%ls %d)",
49 qUtf16Printable(t.toString()), index);
50 return -1;
51 }
52 const SectionNode &node = sectionNodes.at(i: index);
53 switch (node.type) {
54 case TimeZoneSection: return t.offsetFromUtc();
55 case Hour24Section: case Hour12Section: return t.time().hour();
56 case MinuteSection: return t.time().minute();
57 case SecondSection: return t.time().second();
58 case MSecSection: return t.time().msec();
59 case YearSection2Digits:
60 case YearSection: return t.date().year(cal: calendar);
61 case MonthSection: return t.date().month(cal: calendar);
62 case DaySection: return t.date().day(cal: calendar);
63 case DayOfWeekSectionShort:
64 case DayOfWeekSectionLong: return calendar.dayOfWeek(date: t.date());
65 case AmPmSection: return t.time().hour() > 11 ? 1 : 0;
66
67 default: break;
68 }
69
70 qWarning(msg: "QDateTimeParser::getDigit() Internal error 2 (%ls %d)",
71 qUtf16Printable(t.toString()), index);
72 return -1;
73}

75/*!
76 \internal
77 Difference between two days of the week.
78
79 Returns a difference in the range from -3 through +3, so that steps by small
80 numbers of days move us through the month in the same direction as through
81 the week.
82*/
83
84static int dayOfWeekDiff(int sought, int held)
85{
86 const int diff = sought - held;
87 return diff < -3 ? diff + 7 : diff > 3 ? diff - 7 : diff;
88}
89
90static bool preferDayOfWeek(const QList<QDateTimeParser::SectionNode> &nodes)
91{
92 // True precisely if there is a day-of-week field but no day-of-month field.
93 bool result = false;
94 for (const auto &node : nodes) {
95 if (node.type & QDateTimeParser::DaySection)
96 return false;
97 if (node.type & QDateTimeParser::DayOfWeekSectionMask)
98 result = true;
99 }
100 return result;
101}

103/*!
104 \internal
105 Sets a digit in a datetime. E.g.
106
107 QDateTime var(QDate(2004, 02, 02));
108 int digit = getDigit(var, Year);
109 // digit = 2004
110 setDigit(&var, Year, 2005);
111 digit = getDigit(var, Year);
112 // digit = 2005
113*/
114
115bool QDateTimeParser::setDigit(QDateTime &v, int index, int newVal) const
116{
117 if (index < 0 || index >= sectionNodes.size()) {
118 qWarning(msg: "QDateTimeParser::setDigit() Internal error (%ls %d %d)",
119 qUtf16Printable(v.toString()), index, newVal);
120 return false;
121 }
122
123 const QDate oldDate = v.date();
124 QCalendar::YearMonthDay date = calendar.partsFromDate(date: oldDate);
125 if (!date.isValid())
126 return false;
127 int weekDay = calendar.dayOfWeek(date: oldDate);
128 enum { NoFix, MonthDay, WeekDay } fixDay = NoFix;
129
130 const QTime time = v.time();
131 int hour = time.hour();
132 int minute = time.minute();
133 int second = time.second();
134 int msec = time.msec();
135 QTimeZone timeZone = v.timeRepresentation();
136
137 const SectionNode &node = sectionNodes.at(i: index);
138 switch (node.type) {
139 case Hour24Section: case Hour12Section: hour = newVal; break;
140 case MinuteSection: minute = newVal; break;
141 case SecondSection: second = newVal; break;
142 case MSecSection: msec = newVal; break;
143 case YearSection2Digits:
144 case YearSection: date.year = newVal; break;
145 case MonthSection: date.month = newVal; break;
146 case DaySection:
147 if (newVal > 31) {
148 // have to keep legacy behavior. setting the
149 // date to 32 should return false. Setting it
150 // to 31 for february should return true
151 return false;
152 }
153 date.day = newVal;
154 fixDay = MonthDay;
155 break;
156 case DayOfWeekSectionShort:
157 case DayOfWeekSectionLong:
158 if (newVal > 7 || newVal <= 0)
159 return false;
160 date.day += dayOfWeekDiff(sought: newVal, held: weekDay);
161 weekDay = newVal;
162 fixDay = WeekDay;
163 break;
164 case TimeZoneSection:
165 if (newVal < absoluteMin(index) || newVal > absoluteMax(index))
166 return false;
167 // Only offset from UTC is amenable to setting an int value:
168 timeZone = QTimeZone::fromSecondsAheadOfUtc(offset: newVal);
169 break;
170 case AmPmSection: hour = (newVal == 0 ? hour % 12 : (hour % 12) + 12); break;
171 default:
172 qWarning(msg: "QDateTimeParser::setDigit() Internal error (%ls)",
173 qUtf16Printable(node.name()));
174 break;
175 }
176
177 if (!(node.type & DaySectionMask)) {
178 if (date.day < cachedDay)
179 date.day = cachedDay;
180 fixDay = MonthDay;
181 if (weekDay > 0 && weekDay <= 7 && preferDayOfWeek(nodes: sectionNodes)) {
182 const int max = calendar.daysInMonth(month: date.month, year: date.year);
183 if (max > 0 && date.day > max)
184 date.day = max;
185 const int newDoW = calendar.dayOfWeek(date: calendar.dateFromParts(parts: date));
186 if (newDoW > 0 && newDoW <= 7)
187 date.day += dayOfWeekDiff(sought: weekDay, held: newDoW);
188 fixDay = WeekDay;
189 }
190 }
191
192 if (fixDay != NoFix) {
193 const int max = calendar.daysInMonth(month: date.month, year: date.year);
194 // max > 0 precisely if the year does have such a month
195 if (max > 0 && date.day > max)
196 date.day = fixDay == WeekDay ? date.day - 7 : max;
197 else if (date.day < 1)
198 date.day = fixDay == WeekDay ? date.day + 7 : 1;
199 Q_ASSERT(fixDay != WeekDay
200 || calendar.dayOfWeek(calendar.dateFromParts(date)) == weekDay);
201 }
202
203 const QDate newDate = calendar.dateFromParts(parts: date);
204 const QTime newTime(hour, minute, second, msec);
205 if (!newDate.isValid() || !newTime.isValid())
206 return false;
207
208 v = QDateTime(newDate, newTime, timeZone);
209 return true;
210}

214/*!
215 \internal
216
217 Returns the absolute maximum for a section
218*/
219
220int QDateTimeParser::absoluteMax(int s, const QDateTime &cur) const
221{
222 const SectionNode &sn = sectionNode(index: s);
223 switch (sn.type) {
224 case TimeZoneSection:
225 return QTimeZone::MaxUtcOffsetSecs;
226 case Hour24Section:
227 case Hour12Section:
228 // This is special-cased in parseSection.
229 // We want it to be 23 for the stepBy case.
230 return 23;
231 case MinuteSection:
232 case SecondSection:
233 return 59;
234 case MSecSection:
235 return 999;
236 case YearSection2Digits:
237 case YearSection:
238 // sectionMaxSize will prevent people from typing in a larger number in
239 // count == 2 sections; stepBy() will work on real years anyway.
240 return 9999;
241 case MonthSection:
242 return calendar.maximumMonthsInYear();
243 case DaySection:
244 return cur.isValid() ? cur.date().daysInMonth(cal: calendar) : calendar.maximumDaysInMonth();
245 case DayOfWeekSectionShort:
246 case DayOfWeekSectionLong:
247 return 7;
248 case AmPmSection:
249 return int(UpperCase);
250 default:
251 break;
252 }
253 qWarning(msg: "QDateTimeParser::absoluteMax() Internal error (%ls)",
254 qUtf16Printable(sn.name()));
255 return -1;
256}

365/*!
366 \internal
367
368 helper function for parseFormat. removes quotes that are
369 not escaped and removes the escaping on those that are escaped
370
371*/
372
373static QString unquote(QStringView str)
374{
375 const QLatin1Char quote('\'');
376 const QLatin1Char slash('\\');
377 const QLatin1Char zero('0');
378 QString ret;
379 QChar status(zero);
380 const int max = str.size();
381 for (int i=0; i<max; ++i) {
382 if (str.at(n: i) == quote) {
383 if (status != quote)
384 status = quote;
385 else if (!ret.isEmpty() && str.at(n: i - 1) == slash)
386 ret[ret.size() - 1] = quote;
387 else
388 status = zero;
389 } else {
390 ret += str.at(n: i);
391 }
392 }
393 return ret;
394}

396static inline int countRepeat(QStringView str, int index, int maxCount)
397{
398 str = str.sliced(pos: index);
399 if (maxCount > str.size())
400 maxCount = str.size();
401
402 const QChar ch(str[0]);
403 int count = 1;
404 while (count < maxCount && str[count] == ch)
405 ++count;
406 return count;
407}

409static inline void appendSeparator(QStringList *list, QStringView string,
410 int from, int size, int lastQuote)
411{
412 Q_ASSERT(size >= 0 && from + size <= string.size());
413 const QStringView separator = string.sliced(pos: from, n: size);
414 list->append(t: lastQuote >= from ? unquote(str: separator) : separator.toString());

417/*!
418 \internal
419
420 Parses the format \a newFormat. If successful, returns \c true and sets up
421 the format. Else keeps the old format and returns \c false.
422*/
423bool QDateTimeParser::parseFormat(QStringView newFormat)
424{
425 const QLatin1Char quote('\'');
426 const QLatin1Char slash('\\');
427 const QLatin1Char zero('0');
428 if (newFormat == displayFormat && !newFormat.isEmpty())
429 return true;
430
431 QDTPDEBUGN(msg: "parseFormat: %s", newFormat.toLatin1().constData());
432
433 QList<SectionNode> newSectionNodes;
434 Sections newDisplay;
435 QStringList newSeparators;
436 int i, index = 0;
437 int add = 0;
438 QLatin1Char status(zero);
439 const int max = newFormat.size();
440 int lastQuote = -1;
441 for (i = 0; i<max; ++i) {
442 if (newFormat.at(n: i) == quote) {
443 lastQuote = i;
444 ++add;
445 if (status != quote)
446 status = quote;
447 else if (i > 0 && newFormat.at(n: i - 1) != slash)
448 status = zero;
449 } else if (status != quote) {
450 const char sect = newFormat.at(n: i).toLatin1();
451 switch (sect) {
452 case 'H':
453 case 'h':
454 if (parserType != QMetaType::QDate) {
455 const Section hour = (sect == 'h') ? Hour12Section : Hour24Section;
456 const SectionNode sn = { .type: hour, .pos: i - add, .count: countRepeat(str: newFormat, index: i, maxCount: 2), .zeroesAdded: 0 };
457 newSectionNodes.append(t: sn);
458 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
459 i += sn.count - 1;
460 index = i + 1;
461 newDisplay |= hour;
462 }
463 break;
464 case 'm':
465 if (parserType != QMetaType::QDate) {
466 const SectionNode sn = { .type: MinuteSection, .pos: i - add, .count: countRepeat(str: newFormat, index: i, maxCount: 2), .zeroesAdded: 0 };
467 newSectionNodes.append(t: sn);
468 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
469 i += sn.count - 1;
470 index = i + 1;
471 newDisplay |= MinuteSection;
472 }
473 break;
474 case 's':
475 if (parserType != QMetaType::QDate) {
476 const SectionNode sn = { .type: SecondSection, .pos: i - add, .count: countRepeat(str: newFormat, index: i, maxCount: 2), .zeroesAdded: 0 };
477 newSectionNodes.append(t: sn);
478 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
479 i += sn.count - 1;
480 index = i + 1;
481 newDisplay |= SecondSection;
482 }
483 break;
484
485 case 'z':
486 if (parserType != QMetaType::QDate) {
487 const int repeat = countRepeat(str: newFormat, index: i, maxCount: 3);
488 const SectionNode sn = { .type: MSecSection, .pos: i - add, .count: repeat < 3 ? 1 : 3, .zeroesAdded: 0 };
489 newSectionNodes.append(t: sn);
490 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
491 i += repeat - 1;
492 index = i + 1;
493 newDisplay |= MSecSection;
494 }
495 break;
496 case 'A':
497 case 'a':
498 if (parserType != QMetaType::QDate) {
499 const int pos = i - add;
500 Case caseOpt = sect == 'A' ? UpperCase : LowerCase;
501 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
502 newDisplay |= AmPmSection;
503 if (i + 1 < newFormat.size()
504 && newFormat.sliced(pos: i + 1).startsWith(c: u'p', cs: Qt::CaseInsensitive)) {
505 ++i;
506 if (newFormat.at(n: i) != QLatin1Char(caseOpt == UpperCase ? 'P' : 'p'))
507 caseOpt = NativeCase;
508 }
509 const SectionNode sn = { .type: AmPmSection, .pos: pos, .count: int(caseOpt), .zeroesAdded: 0 };
510 newSectionNodes.append(t: sn);
511 index = i + 1;
512 }
513 break;
514 case 'y':
515 if (parserType != QMetaType::QTime) {
516 const int repeat = countRepeat(str: newFormat, index: i, maxCount: 4);
517 if (repeat >= 2) {
518 const SectionNode sn = { .type: repeat == 4 ? YearSection : YearSection2Digits,
519 .pos: i - add, .count: repeat == 4 ? 4 : 2, .zeroesAdded: 0 };
520 newSectionNodes.append(t: sn);
521 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
522 i += sn.count - 1;
523 index = i + 1;
524 newDisplay |= sn.type;
525 }
526 }
527 break;
528 case 'M':
529 if (parserType != QMetaType::QTime) {
530 const SectionNode sn = { .type: MonthSection, .pos: i - add, .count: countRepeat(str: newFormat, index: i, maxCount: 4), .zeroesAdded: 0 };
531 newSectionNodes.append(t: sn);
532 newSeparators.append(t: unquote(str: newFormat.first(n: i).sliced(pos: index)));
533 i += sn.count - 1;
534 index = i + 1;
535 newDisplay |= MonthSection;
536 }
537 break;
538 case 'd':
539 if (parserType != QMetaType::QTime) {
540 const int repeat = countRepeat(str: newFormat, index: i, maxCount: 4);
541 const Section sectionType = (repeat == 4 ? DayOfWeekSectionLong
542 : (repeat == 3 ? DayOfWeekSectionShort : DaySection));
543 const SectionNode sn = { .type: sectionType, .pos: i - add, .count: repeat, .zeroesAdded: 0 };
544 newSectionNodes.append(t: sn);
545 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
546 i += sn.count - 1;
547 index = i + 1;
548 newDisplay |= sn.type;
549 }
550 break;
551 case 't':
552 if (parserType == QMetaType::QDateTime) {
553 const SectionNode sn
554 = { .type: TimeZoneSection, .pos: i - add, .count: countRepeat(str: newFormat, index: i, maxCount: 4), .zeroesAdded: 0 };
555 newSectionNodes.append(t: sn);
556 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: i - index, lastQuote);
557 i += sn.count - 1;
558 index = i + 1;
559 newDisplay |= TimeZoneSection;
560 }
561 break;
562 default:
563 break;
564 }
565 }
566 }
567 if (newSectionNodes.isEmpty() && context == DateTimeEdit)
568 return false;
569
570 if ((newDisplay & (AmPmSection|Hour12Section)) == Hour12Section) {
571 const int count = newSectionNodes.size();
572 for (int i = 0; i < count; ++i) {
573 SectionNode &node = newSectionNodes[i];
574 if (node.type == Hour12Section)
575 node.type = Hour24Section;
576 }
577 }
578
579 if (index < max)
580 appendSeparator(list: &newSeparators, string: newFormat, from: index, size: max - index, lastQuote);
581 else
582 newSeparators.append(t: QString());
583
584 displayFormat = newFormat.toString();
585 separators = newSeparators;
586 sectionNodes = newSectionNodes;
587 display = newDisplay;
588 last.pos = -1;
589
590// for (int i=0; i<sectionNodes.size(); ++i) {
591// QDTPDEBUG << sectionNodes.at(i).name() << sectionNodes.at(i).count;
592// }
593
594 QDTPDEBUG << newFormat << displayFormat;
595 QDTPDEBUGN(msg: "separators:\n'%s'", separators.join(sep: "\n"_L1).toLatin1().constData());
596
597 return true;
598}

600/*!
601 \internal
602
603 Returns the size of section \a s.
604*/
605
606int QDateTimeParser::sectionSize(int sectionIndex) const
607{
608 if (sectionIndex < 0)
609 return 0;
610
611 if (sectionIndex >= sectionNodes.size()) {
612 qWarning(msg: "QDateTimeParser::sectionSize Internal error (%d)", sectionIndex);
613 return -1;
614 }
615
616 if (sectionIndex == sectionNodes.size() - 1) {
617 // In some cases there is a difference between displayText() and text.
618 // e.g. when text is 2000/01/31 and displayText() is "2000/2/31" - text
619 // is the previous value and displayText() is the new value.
620 // The size difference is always due to leading zeroes.
621 int sizeAdjustment = 0;
622 const int displayTextSize = displayText().size();
623 if (displayTextSize != m_text.size()) {
624 // Any zeroes added before this section will affect our size.
625 int preceedingZeroesAdded = 0;
626 if (sectionNodes.size() > 1 && context == DateTimeEdit) {
627 const auto begin = sectionNodes.cbegin();
628 const auto end = begin + sectionIndex;
629 for (auto sectionIt = begin; sectionIt != end; ++sectionIt)
630 preceedingZeroesAdded += sectionIt->zeroesAdded;
631 }
632 sizeAdjustment = preceedingZeroesAdded;
633 }
634
635 return displayTextSize + sizeAdjustment - sectionPos(sectionIndex) - separators.last().size();
636 } else {
637 return sectionPos(sectionIndex: sectionIndex + 1) - sectionPos(sectionIndex)
638 - separators.at(i: sectionIndex + 1).size();
639 }
640}

643int QDateTimeParser::sectionMaxSize(Section s, int count) const
644{
645#if QT_CONFIG(textdate)
646 int mcount = calendar.maximumMonthsInYear();
647#endif
648
649 switch (s) {
650 case FirstSection:
651 case NoSection:
652 case LastSection:
653 return 0;
654
655 case AmPmSection:
656 // Special: "count" here is a case flag, not field width !
657 return qMax(a: getAmPmText(ap: AmText, cs: Case(count)).size(),
658 b: getAmPmText(ap: PmText, cs: Case(count)).size());
659
660 case Hour24Section:
661 case Hour12Section:
662 case MinuteSection:
663 case SecondSection:
664 case DaySection:
665 return 2;
666
667 case DayOfWeekSectionShort:
668 case DayOfWeekSectionLong:
669#if QT_CONFIG(textdate)
670 mcount = 7;
671 Q_FALLTHROUGH();
672#endif
673 case MonthSection:
674#if QT_CONFIG(textdate)
675 if (count <= 2)
676 return 2;
677
678 {
679 int ret = 0;
680 const QLocale l = locale();
681 const QLocale::FormatType format = count == 4 ? QLocale::LongFormat : QLocale::ShortFormat;
682 for (int i=1; i<=mcount; ++i) {
683 const QString str = (s == MonthSection
684 ? calendar.monthName(locale: l, month: i, year: QCalendar::Unspecified, format)
685 : l.dayName(i, format));
686 ret = qMax(a: str.size(), b: ret);
687 }
688 return ret;
689 }
690#else
691 return 2;
692#endif // textdate
693 case MSecSection:
694 return 3;
695 case YearSection:
696 return 4;
697 case YearSection2Digits:
698 return 2;
699 case TimeZoneSection:
700 // Arbitrarily many tokens (each up to 14 bytes) joined with / separators:
701 return std::numeric_limits<int>::max();
702
703 case CalendarPopupSection:
704 case Internal:
705 case TimeSectionMask:
706 case DateSectionMask:
707 case HourSectionMask:
708 case YearSectionMask:
709 case DayOfWeekSectionMask:
710 case DaySectionMask:
711 qWarning(msg: "QDateTimeParser::sectionMaxSize: Invalid section %s",
712 SectionNode::name(s).toLatin1().constData());
713
714 case NoSectionIndex:
715 case FirstSectionIndex:
716 case LastSectionIndex:
717 case CalendarPopupIndex:
718 // these cases can't happen
719 break;
720 }
721 return -1;
722}

759QDateTimeParser::ParsedSection
760QDateTimeParser::parseSection(const QDateTime &currentValue, int sectionIndex, int offset) const
761{
762 ParsedSection result; // initially Invalid
763 const SectionNode &sn = sectionNode(sectionIndex);
764 Q_ASSERT_X(!(sn.type & Internal),
765 "QDateTimeParser::parseSection", "Internal error");
766
767 const int sectionmaxsize = sectionMaxSize(index: sectionIndex);
768 const bool negate = (sn.type == YearSection && m_text.size() > offset
769 && m_text.at(i: offset) == u'-');
770 const int negativeYearOffset = negate ? 1 : 0;
771
772 // If the fields we've read thus far imply a time in a spring-forward,
773 // coerce to a nearby valid time:
774 const QDateTime defaultValue = currentValue.isValid() ? currentValue
775 : QDateTime::fromMSecsSinceEpoch(msecs: currentValue.toMSecsSinceEpoch());
776
777 QStringView sectionTextRef =
778 QStringView { m_text }.mid(pos: offset + negativeYearOffset, n: sectionmaxsize);
779
780 QDTPDEBUG << "sectionValue for" << sn.name()
781 << "with text" << m_text << "and (at" << offset
782 << ") st:" << sectionTextRef;
783
784 switch (sn.type) {
785 case AmPmSection: {
786 QString sectiontext = sectionTextRef.toString();
787 int used;
788 const int ampm = findAmPm(str&: sectiontext, index: sectionIndex, used: &used);
789 switch (ampm) {
790 case AM: // sectiontext == AM
791 case PM: // sectiontext == PM
792 result = ParsedSection(Acceptable, ampm, used);
793 break;
794 case PossibleAM: // sectiontext => AM
795 case PossiblePM: // sectiontext => PM
796 result = ParsedSection(Intermediate, ampm - 2, used);
797 break;
798 case PossibleBoth: // sectiontext => AM|PM
799 result = ParsedSection(Intermediate, 0, used);
800 break;
801 case Neither:
802 QDTPDEBUG << "invalid because findAmPm(" << sectiontext << ") returned -1";
803 break;
804 default:
805 QDTPDEBUGN(msg: "This should never happen (findAmPm returned %d)", ampm);
806 break;
807 }
808 if (result.state != Invalid)
809 m_text.replace(i: offset, len: used, s: sectiontext.constData(), slen: used);
810 break; }
811 case TimeZoneSection:
812 result = findTimeZone(str: sectionTextRef, when: defaultValue,
813 maxVal: absoluteMax(s: sectionIndex),
814 minVal: absoluteMin(s: sectionIndex), mode: sn.count);
815 break;
816 case MonthSection:
817 case DayOfWeekSectionShort:
818 case DayOfWeekSectionLong:
819 if (sn.count >= 3) {
820 QString sectiontext = sectionTextRef.toString();
821 int num = 0, used = 0;
822 if (sn.type == MonthSection) {
823 const QDate minDate = getMinimum().date();
824 const int year = defaultValue.date().year(cal: calendar);
825 const int min = (year == minDate.year(cal: calendar)) ? minDate.month(cal: calendar) : 1;
826 num = findMonth(str1: sectiontext.toLower(), monthstart: min, sectionIndex, year, monthName: &sectiontext, used: &used);
827 } else {
828 num = findDay(str1: sectiontext.toLower(), intDaystart: 1, sectionIndex, dayName: &sectiontext, used: &used);
829 }
830
831 result = ParsedSection(Intermediate, num, used);
832 if (num != -1) {
833 m_text.replace(i: offset, len: used, s: sectiontext.constData(), slen: used);
834 if (used == sectiontext.size())
835 result = ParsedSection(Acceptable, num, used);
836 }
837 break;
838 }
839 Q_FALLTHROUGH();
840 // All numeric:
841 case DaySection:
842 case YearSection:
843 case YearSection2Digits:
844 case Hour12Section:
845 case Hour24Section:
846 case MinuteSection:
847 case SecondSection:
848 case MSecSection: {
849 int used = negativeYearOffset;
850 // We already sliced off the - sign if it was legitimately present.
851 if (sectionTextRef.startsWith(c: u'-')
852 || sectionTextRef.startsWith(c: u'+')) {
853 if (separators.at(i: sectionIndex + 1).startsWith(c: sectionTextRef[0]))
854 result = ParsedSection(Intermediate, 0, used);
855 break;
856 }
857 QStringView digitsStr = sectionTextRef.left(n: digitCount(str: sectionTextRef));
858
859 if (digitsStr.isEmpty()) {
860 result = ParsedSection(Intermediate, 0, used);
861 } else {
862 const QLocale loc = locale();
863 const int absMax = absoluteMax(s: sectionIndex);
864 const int absMin = absoluteMin(s: sectionIndex);
865
866 int last = -1;
867
868 for (; digitsStr.size(); digitsStr.chop(n: 1)) {
869 bool ok = false;
870 int value = int(loc.toUInt(s: digitsStr, ok: &ok));
871 if (!ok || (negate ? -value < absMin : value > absMax))
872 continue;
873
874 if (sn.type == Hour12Section) {
875 if (value > 12)
876 continue;
877 if (value == 12)
878 value = 0;
879 }
880
881 QDTPDEBUG << digitsStr << value << digitsStr.size();
882 last = value;
883 used += digitsStr.size();
884 break;
885 }
886
887 if (last == -1) {
888 const auto &sep = separators.at(i: sectionIndex + 1);
889 if (sep.startsWith(c: sectionTextRef[0])
890 || (negate && sep.startsWith(c: m_text.at(i: offset))))
891 result = ParsedSection(Intermediate, 0, 0);
892 else
893 QDTPDEBUG << "invalid because" << sectionTextRef << "can't become a uint"
894 << last;
895 } else {
896 if (negate)
897 last = -last;
898 const FieldInfo fi = fieldInfo(index: sectionIndex);
899 const bool unfilled = used - negativeYearOffset < sectionmaxsize;
900 if (unfilled && fi & Fraction) { // typing 2 in a zzz field should be .200, not .002
901 for (int i = used; i < sectionmaxsize; ++i)
902 last *= 10;
903 }
904 // Even those *= 10s can't take last above absMax:
905 Q_ASSERT(negate ? last >= absMin : last <= absMax);
906 if (negate ? last > absMax : last < absMin) {
907 if (unfilled) {
908 result = ParsedSection(Intermediate, last, used);
909 } else if (negate) {
910 QDTPDEBUG << "invalid because" << last << "is greater than absoluteMax"
911 << absMax;
912 } else {
913 QDTPDEBUG << "invalid because" << last << "is less than absoluteMin"
914 << absMin;
915 }
916
917 } else if (unfilled && (fi & (FixedWidth | Numeric)) == (FixedWidth | Numeric)) {
918 if (skipToNextSection(section: sectionIndex, current: defaultValue, sectionText: digitsStr)) {
919 const int missingZeroes = sectionmaxsize - digitsStr.size();
920 result = ParsedSection(Acceptable, last, sectionmaxsize, missingZeroes);
921 m_text.insert(i: offset, s: QString(missingZeroes, u'0'));
922 ++(const_cast<QDateTimeParser*>(this)->sectionNodes[sectionIndex].zeroesAdded);
923 } else {
924 result = ParsedSection(Intermediate, last, used);;
925 }
926 } else {
927 result = ParsedSection(Acceptable, last, used);
928 }
929 }
930 }
931 break; }
932 default:
933 qWarning(msg: "QDateTimeParser::parseSection Internal error (%ls %d)",
934 qUtf16Printable(sn.name()), sectionIndex);
935 return result;
936 }
937 Q_ASSERT(result.state != Invalid || result.value == -1);
938
939 return result;
940}

942/*!
943 \internal
944
945 Returns the day-number of a day, as close as possible to the given \a day, in
946 the specified \a month of \a year for the given \a calendar, that falls on the
947 day of the week indicated by \a weekDay.
948*/
949
950static int weekDayWithinMonth(QCalendar calendar, int year, int month, int day, int weekDay)
951{
952 // TODO: can we adapt this to cope gracefully with intercallary days (day of
953 // week > 7) without making it slower for more widely-used calendars ?
954 const int maxDay = calendar.daysInMonth(month, year); // 0 if no such month
955 day = maxDay > 1 ? qBound(min: 1, val: day, max: maxDay) : qMax(a: 1, b: day);
956 day += dayOfWeekDiff(sought: weekDay, held: calendar.dayOfWeek(date: QDate(year, month, day, calendar)));
957 return day <= 0 ? day + 7 : maxDay > 0 && day > maxDay ? day - 7 : day;
958}

960/*!
961 \internal
962
963 Returns a date consistent with the given data on parts specified by known,
964 while staying as close to the given data as it can. Returns an invalid date
965 when on valid date is consistent with the data.
966*/
967
968static QDate actualDate(QDateTimeParser::Sections known, const QCalendar &calendar,
969 int year, int year2digits, int month, int day, int dayofweek)
970{
971 QDate actual(year, month, day, calendar);
972 if (actual.isValid() && year % 100 == year2digits && calendar.dayOfWeek(date: actual) == dayofweek)
973 return actual; // The obvious candidate is fine :-)
974
975 if (dayofweek < 1 || dayofweek > 7) // Intercallary (or invalid): ignore
976 known &= ~QDateTimeParser::DayOfWeekSectionMask;
977
978 // Assuming year > 0 ...
979 if (year % 100 != year2digits) {
980 if (known & QDateTimeParser::YearSection2Digits) {
981 // Over-ride year, even if specified:
982 year += year2digits - year % 100;
983 known &= ~QDateTimeParser::YearSection;
984 } else {
985 year2digits = year % 100;
986 }
987 }
988 Q_ASSERT(year % 100 == year2digits);
989
990 if (month < 1) { // If invalid, clip to nearest valid and ignore in known.
991 month = 1;
992 known &= ~QDateTimeParser::MonthSection;
993 } else if (month > 12) {
994 month = 12;
995 known &= ~QDateTimeParser::MonthSection;
996 }
997
998 QDate first(year, month, 1, calendar);
999 int last = known & QDateTimeParser::MonthSection
1000 ? (known & QDateTimeParser::YearSection
1001 ? calendar.daysInMonth(month, year) : calendar.daysInMonth(month))
1002 : 0;
1003 // We can only fix DOW if we know year as well as month (hence last):
1004 const bool fixDayOfWeek = last && known & QDateTimeParser::YearSection
1005 && known & QDateTimeParser::DayOfWeekSectionMask;
1006 // If we also know day-of-week, tweak last to the last in the month that matches it:
1007 if (fixDayOfWeek) {
1008 const int diff = (dayofweek - calendar.dayOfWeek(date: first) - last) % 7;
1009 Q_ASSERT(diff <= 0); // C++11 specifies (-ve) % (+ve) to be <= 0.
1010 last += diff;
1011 }
1012 if (day < 1) {
1013 if (fixDayOfWeek) {
1014 day = 1 + dayofweek - calendar.dayOfWeek(date: first);
1015 if (day < 1)
1016 day += 7;
1017 } else {
1018 day = 1;
1019 }
1020 known &= ~QDateTimeParser::DaySection;
1021 } else if (day > calendar.maximumDaysInMonth()) {
1022 day = last;
1023 known &= ~QDateTimeParser::DaySection;
1024 } else if (last && day > last && (known & QDateTimeParser::DaySection) == 0) {
1025 day = last;
1026 }
1027
1028 actual = QDate(year, month, day, calendar);
1029 if (!actual.isValid() // We can't do better than we have, in this case
1030 || (known & QDateTimeParser::DaySection
1031 && known & QDateTimeParser::MonthSection
1032 && known & QDateTimeParser::YearSection) // ditto
1033 || calendar.dayOfWeek(date: actual) == dayofweek // Good enough, use it.
1034 || (known & QDateTimeParser::DayOfWeekSectionMask) == 0) { // No contradiction, use it.
1035 return actual;
1036 }
1037
1038 /*
1039 Now it gets trickier.
1040
1041 We have some inconsistency in our data; we've been told day of week, but
1042 it doesn't fit with our year, month and day. At least one of these is
1043 unknown, though: so we can fix day of week by tweaking it.
1044 */
1045
1046 if ((known & QDateTimeParser::DaySection) == 0) {
1047 // Relatively easy to fix.
1048 day = weekDayWithinMonth(calendar, year, month, day, weekDay: dayofweek);
1049 actual = QDate(year, month, day, calendar);
1050 return actual;
1051 }
1052
1053 if ((known & QDateTimeParser::MonthSection) == 0) {
1054 /*
1055 Try possible month-offsets, m, preferring small; at least one (present
1056 month doesn't work) and at most 11 (max month, 12, minus min, 1); try
1057 in both directions, ignoring any offset that takes us out of range.
1058 */
1059 for (int m = 1; m < 12; m++) {
1060 if (m < month) {
1061 actual = QDate(year, month - m, day, calendar);
1062 if (calendar.dayOfWeek(date: actual) == dayofweek)
1063 return actual;
1064 }
1065 if (m + month <= 12) {
1066 actual = QDate(year, month + m, day, calendar);
1067 if (calendar.dayOfWeek(date: actual) == dayofweek)
1068 return actual;
1069 }
1070 }
1071 // Should only get here in corner cases; e.g. day == 31
1072 actual = QDate(year, month, day, calendar); // Restore from trial values.
1073 }
1074
1075 if ((known & QDateTimeParser::YearSection) == 0) {
1076 if (known & QDateTimeParser::YearSection2Digits) {
1077 /*
1078 Two-digit year and month are specified; choice of century can only
1079 fix this if diff is in one of {1, 2, 5} or {2, 4, 6}; but not if
1080 diff is in the other. It's also only reasonable to consider
1081 adjacent century, e.g. if year thinks it's 2012 and two-digit year
1082 is '97, it makes sense to consider 1997. If either adjacent
1083 century does work, the other won't.
1084 */
1085 actual = QDate(year + 100, month, day, calendar);
1086 if (calendar.dayOfWeek(date: actual) == dayofweek)
1087 return actual;
1088 actual = QDate(year - 100, month, day, calendar);
1089 if (calendar.dayOfWeek(date: actual) == dayofweek)
1090 return actual;
1091 } else {
1092 // Offset by 7 is usually enough, but rare cases may need more:
1093 for (int y = 1; y < 12; y++) {
1094 actual = QDate(year - y, month, day, calendar);
1095 if (calendar.dayOfWeek(date: actual) == dayofweek)
1096 return actual;
1097 actual = QDate(year + y, month, day, calendar);
1098 if (calendar.dayOfWeek(date: actual) == dayofweek)
1099 return actual;
1100 }
1101 }
1102 actual = QDate(year, month, day, calendar); // Restore from trial values.
1103 }
1104
1105 return actual; // It'll just have to do :-(
1106}

1113 int hour, int hour12, int ampm,

1119 known &= ~QDateTimeParser::Hour12Section;
1120 hour12 = hour % 12;
1121 }
1122

1230 timeZone = namedZone;
1231 } else {

1234 }
1235#else

1237#endif

1242 case Hour12Section: current = &hour12; break;

1250 case DayOfWeekSectionLong: current = &dayofweek; break;

1259 Q_ASSERT(sect.state != Invalid);

1270 continue;

1275 isSet |= sn.type;

1279 QDTPDEBUG << "invalid because" << QStringView{m_text}.sliced(pos)
1280 << "!=" << separators.last() << pos;

1286 if (!(isSet & YearSection)) {
1287 year = (year / 100) * 100;
1288 year += year2digits;
1289 } else {
1290 conflicts = true;
1291 const SectionNode &sn = sectionNode(sectionIndex: currentSectionIndex);
1292 if (sn.type == YearSection2Digits) {
1293 year = (year / 100) * 100;
1294 year += year2digits;
1295 }
1296 }
1297 }

1299 const auto fieldType = sectionType(sectionIndex: currentSectionIndex);
1300 const QDate date(year, month, day, calendar);
1301 if ((!date.isValid() || dayofweek != calendar.dayOfWeek(date))
1302 && state == Acceptable && isSet & DayOfWeekSectionMask) {
1303 if (isSet & DaySection)
1304 conflicts = true;
1305 // Change to day of week should adjust day of month;
1306 // when day of month isn't set, so should change to year or month.
1307 if (currentSectionIndex == -1 || fieldType & DayOfWeekSectionMask
1308 || (!conflicts && (fieldType & (YearSectionMask | MonthSection)))) {
1309 day = weekDayWithinMonth(calendar, year, month, day, weekDay: dayofweek);

1311 << calendar.dayOfWeek(date: QDate(year, month, day, calendar));
1312 }
1313 }
1314

1316 if (fieldType & DaySectionMask) {
1317 cachedDay = day;
1318 } else if (cachedDay > day && !(isSet & DayOfWeekSectionMask && state == Acceptable)) {
1319 day = cachedDay;
1320 needfixday = true;
1321 }
1322
1323 if (!calendar.isDateValid(year, month, day)) {
1324 if (day <= calendar.maximumDaysInMonth())
1325 cachedDay = day;
1326 if (day > calendar.minimumDaysInMonth() && calendar.isDateValid(year, month, day: 1))
1327 needfixday = true;
1328 }

1330 if (context == FromString)
1331 return StateNode();
1332 if (state == Acceptable && fixday) {

1337 const SectionNode sn = sectionNode(sectionIndex: i);
1338 if (sn.type & DaySection) {
1339 m_text.replace(i: sectionPos(sn), len: sectionSize(sectionIndex: i), after: loc.toString(i: day));
1340 } else if (sn.type & DayOfWeekSectionMask) {
1341 const int dayOfWeek = calendar.dayOfWeek(date: QDate(year, month, day, calendar));
1342 const QLocale::FormatType dayFormat =
1343 (sn.type == DayOfWeekSectionShort
1344 ? QLocale::ShortFormat : QLocale::LongFormat);
1345 const QString dayName(loc.dayName(dayOfWeek, format: dayFormat));
1346 m_text.replace(i: sectionPos(sn), len: sectionSize(sectionIndex: i), after: dayName);
1347 }
1348 }
1349 } else if (state > Intermediate) {
1350 state = Intermediate;
1351 }
1352 }
1353 }
1354
1355 if (parserType != QMetaType::QDate) {
1356 if (isSet & Hour12Section) {
1357 const bool hasHour = isSet.testAnyFlag(flag: Hour24Section);
1358 if (ampm == -1) // If we don't know from hour, assume am:
1359 ampm = !hasHour || hour < 12 ? 0 : 1;
1360 hour12 = hour12 % 12 + ampm * 12;
1361 if (!hasHour)
1362 hour = hour12;
1363 else if (hour != hour12)
1364 conflicts = true;
1365 } else if (ampm != -1) {
1366 if (!(isSet & (Hour24Section)))
1367 hour = 12 * ampm; // Special case: only ap section
1368 else if ((ampm == 0) != (hour < 12))
1369 conflicts = true;
1370 }
1371 }
1372
1373 QDTPDEBUG << year << month << day << hour << minute << second << msec;
1374 Q_ASSERT(state != Invalid);
1375
1376 const QDate date(year, month, day, calendar);
1377 const QTime time(hour, minute, second, msec);
1378 const QDateTime when = QDateTime(date, time, timeZone);
1379
1380 // If hour wasn't specified, check the default we're using exists on the
1381 // given date (which might be a spring-forward, skipping an hour).
1382 if (!(isSet & HourSectionMask) && !when.isValid()) {
1383 switch (parserType) {
1384 case QMetaType::QDateTime: {
1385 qint64 msecs = when.toMSecsSinceEpoch();
1386 // Fortunately, that gets a useful answer, even though when is invalid ...
1387 const QDateTime replace = QDateTime::fromMSecsSinceEpoch(msecs, timeZone);
1388 const QTime tick = replace.time();
1389 if (replace.date() == date
1390 && (!(isSet & MinuteSection) || tick.minute() == minute)
1391 && (!(isSet & SecondSection) || tick.second() == second)
1392 && (!(isSet & MSecSection) || tick.msec() == msec)) {
1393 return StateNode(replace, state, padding, conflicts);
1394 }
1395 } break;
1396 case QMetaType::QDate:
1397 // Don't care about time, so just use start of day (and ignore spec):
1398 return StateNode(date.startOfDay(zone: QTimeZone::UTC), state, padding, conflicts);
1399 break;
1400 case QMetaType::QTime:
1401 // Don't care about date or representation, so pick a safe representation:
1402 return StateNode(QDateTime(date, time, QTimeZone::UTC), state, padding, conflicts);
1403 default:
1404 Q_UNREACHABLE_RETURN(StateNode());
1405 }
1406 }
1407
1408 return StateNode(when, state, padding, conflicts);
1409}

1532 done = true;

1535 scan.state = Intermediate;
1536 done = true;
1537 break; }

1558 */

1565/*
1566 \internal
1567 \brief Returns the index in \a entries with the best prefix match to \a text
1568
1569 Scans \a entries looking for an entry overlapping \a text as much as possible
1570 (an exact match beats any prefix match; a match of the full entry as prefix of
1571 text beats any entry but one matching a longer prefix; otherwise, the match of
1572 longest prefix wins, earlier entries beating later on a draw). Records the
1573 length of overlap in *used (if \a used is non-NULL) and the first entry that
1574 overlapped this much in *usedText (if \a usedText is non-NULL).
1575 */
1576static int findTextEntry(const QString &text, const ShortVector<QString> &entries, QString *usedText, int *used)
1577{
1578 if (text.isEmpty())
1579 return -1;
1580
1581 int bestMatch = -1;
1582 int bestCount = 0;
1583 for (int n = 0; n < entries.size(); ++n)
1584 {
1585 const QString &name = entries.at(idx: n);
1586
1587 const int limit = qMin(a: text.size(), b: name.size());
1588 int i = 0;
1589 while (i < limit && text.at(i) == name.at(i).toLower())
1590 ++i;
1591 // Full match beats an equal prefix match:
1592 if (i > bestCount || (i == bestCount && i == name.size())) {
1593 bestCount = i;
1594 bestMatch = n;
1595 if (i == name.size() && i == text.size())
1596 break; // Exact match, name == text, wins.
1597 }
1598 }
1599 if (usedText && bestMatch != -1)
1600 *usedText = entries.at(idx: bestMatch);
1601 if (used)
1602 *used = bestCount;
1603
1604 return bestMatch;
1605}

1652/*!
1653 \internal
1654
1655 Return's .value is UTC offset in seconds.
1656 The caller must verify that the offset is within a valid range.
1657 The mode is 1 for permissive parsing, 2 and 3 for strict offset-only format
1658 (no UTC prefix) with no colon for 2 and a colon for 3.
1659 */
1660QDateTimeParser::ParsedSection QDateTimeParser::findUtcOffset(QStringView str, int mode) const
1661{
1662 Q_ASSERT(mode > 0 && mode < 4);
1663 const bool startsWithUtc = str.startsWith(s: "UTC"_L1);
1664 // Deal with UTC prefix if present:
1665 if (startsWithUtc) {
1666 if (mode != 1)
1667 return ParsedSection();
1668 str = str.sliced(pos: 3);
1669 if (str.isEmpty())
1670 return ParsedSection(Acceptable, 0, 3);
1671 }
1672
1673 const bool negativeSign = str.startsWith(c: u'-');
1674 // Must start with a sign:
1675 if (!negativeSign && !str.startsWith(c: u'+'))
1676 return ParsedSection();
1677 str = str.sliced(pos: 1); // drop sign
1678
1679 const int colonPosition = str.indexOf(c: u':');
1680 // Colon that belongs to offset is at most at position 2 (hh:mm)
1681 bool hasColon = (colonPosition >= 0 && colonPosition < 3);
1682
1683 // We deal only with digits at this point (except ':'), so collect them
1684 const int digits = hasColon ? colonPosition + 3 : 4;
1685 int i = 0;
1686 for (const int offsetLength = qMin(a: qsizetype(digits), b: str.size()); i < offsetLength; ++i) {
1687 if (i != colonPosition && !str.at(n: i).isDigit())
1688 break;
1689 }
1690 const int hoursLength = qMin(a: i, b: hasColon ? colonPosition : 2);
1691 if (hoursLength < 1)
1692 return ParsedSection();
1693 // Field either ends with hours or also has two digits of minutes
1694 if (i < digits) {
1695 // Only allow single-digit hours with UTC prefix or :mm suffix
1696 if (!startsWithUtc && hoursLength != 2)
1697 return ParsedSection();
1698 i = hoursLength;
1699 hasColon = false;
1700 }
1701 if (mode == (hasColon ? 2 : 3))
1702 return ParsedSection();
1703 str.truncate(n: i); // The rest of the string is not part of the UTC offset
1704
1705 bool isInt = false;
1706 const int hours = str.first(n: hoursLength).toInt(ok: &isInt);
1707 if (!isInt)
1708 return ParsedSection();
1709 const QStringView minutesStr = str.mid(pos: hasColon ? colonPosition + 1 : 2, n: 2);
1710 const int minutes = minutesStr.isEmpty() ? 0 : minutesStr.toInt(ok: &isInt);
1711 if (!isInt)
1712 return ParsedSection();
1713
1714 // Keep in sync with QTimeZone::maxUtcOffset hours (14 at most). Also, user
1715 // could be in the middle of updating the offset (e.g. UTC+14:23) which is
1716 // an intermediate state
1717 const State status = (hours > 14 || minutes >= 60) ? Invalid
1718 : (hours == 14 && minutes > 0) ? Intermediate : Acceptable;
1719
1720 int offset = 3600 * hours + 60 * minutes;
1721 if (negativeSign)
1722 offset = -offset;
1723
1724 // Used: UTC, sign, hours, colon, minutes
1725 const int usedSymbols = (startsWithUtc ? 3 : 0) + 1 + hoursLength + (hasColon ? 1 : 0)
1726 + minutesStr.size();
1727
1728 return ParsedSection(status, offset, usedSymbols);
1729}

1731/*!
1732 \internal
1733
1734 Return's .value is zone's offset, zone time - UTC time, in seconds.
1735 The caller must verify that the offset is within a valid range.
1736 See QTimeZonePrivate::isValidId() for the format of zone names.
1737 */
1738QDateTimeParser::ParsedSection
1739QDateTimeParser::findTimeZoneName(QStringView str, const QDateTime &when) const
1740{
1741 const int systemLength = startsWithLocalTimeZone(name: str);
1742#if QT_CONFIG(timezone)
1743 // Collect up plausibly-valid characters; let QTimeZone work out what's
1744 // truly valid.
1745 const auto invalidZoneNameCharacter = [] (const QChar &c) {
1746 const auto cu = c.unicode();
1747 return cu >= 127u || !(memchr(s: "+-./:_", c: char(cu), n: 6) || c.isLetterOrNumber());
1748 };
1749 int index = std::distance(first: str.cbegin(),
1750 last: std::find_if(first: str.cbegin(), last: str.cend(), pred: invalidZoneNameCharacter));
1751
1752 // Limit name fragments (between slashes) to 20 characters.
1753 // (Valid time-zone IDs are allowed up to 14 and Android has quirks up to 17.)
1754 // Limit number of fragments to six; no known zone name has more than four.
1755 int lastSlash = -1;
1756 int count = 0;
1757 Q_ASSERT(index <= str.size());
1758 while (lastSlash < index) {
1759 int slash = str.indexOf(c: u'/', from: lastSlash + 1);
1760 if (slash < 0 || slash > index)
1761 slash = index; // i.e. the end of the candidate text
1762 else if (++count > 5)
1763 index = slash; // Truncate
1764 if (slash - lastSlash > 20)
1765 index = lastSlash + 20; // Truncate
1766 // If any of those conditions was met, index <= slash, so this exits the loop:
1767 lastSlash = slash;
1768 }
1769
1770 for (; index > systemLength; --index) { // Find longest match
1771 str.truncate(n: index);
1772 QTimeZone zone(str.toLatin1());
1773 if (zone.isValid())
1774 return ParsedSection(Acceptable, zone.offsetFromUtc(atDateTime: when), index);
1775 }
1776#endif
1777 if (systemLength > 0) // won't actually use the offset, but need it to be valid
1778 return ParsedSection(Acceptable, when.toLocalTime().offsetFromUtc(), systemLength);
1779 return ParsedSection();
1780}

2029 \internal
2030
2031 Returns \c true if str can be modified to represent a
2032 number that is within min and max.
2033*/
2034
2035bool QDateTimeParser::potentialValue(QStringView str, int min, int max, int index,
2036 const QDateTime &currentValue, int insert) const
2037{
2038 if (str.isEmpty())
2039 return true;
2040
2041 const int size = sectionMaxSize(index);
2042 int val = (int)locale().toUInt(s: str);
2043 const SectionNode &sn = sectionNode(sectionIndex: index);
2044 if (sn.type == YearSection2Digits) {
2045 const int year = currentValue.date().year(cal: calendar);
2046 val += year - (year % 100);
2047 }
2048 if (val >= min && val <= max && str.size() == size)
2049 return true;
2050 if (val > max || (str.size() == size && val < min))
2051 return false;
2052
2053 const int len = size - str.size();
2054 for (int i=0; i<len; ++i) {
2055 for (int j=0; j<10; ++j) {
2056 if (potentialValue(str: str + QLatin1Char('0' + j), min, max, index, currentValue, insert)) {
2057 return true;
2058 } else if (insert >= 0) {
2059 const QString tmp = str.left(n: insert) + QLatin1Char('0' + j) + str.mid(pos: insert);
2060 if (potentialValue(str: tmp, min, max, index, currentValue, insert))
2061 return true;
2062 }
2063 }
2064 }
2065
2066 return false;

2070 \internal
2071*/
2072bool QDateTimeParser::skipToNextSection(int index, const QDateTime &current, QStringView text) const
2073{
2074 Q_ASSERT(text.size() < sectionMaxSize(index));
2075 const SectionNode &node = sectionNode(sectionIndex: index);
2076 int min = absoluteMin(s: index);
2077 int max = absoluteMax(s: index, cur: current);
2078 // Time-zone field is only numeric if given as offset from UTC:
2079 if (node.type != TimeZoneSection || current.timeSpec() == Qt::OffsetFromUTC) {
2080 const QDateTime maximum = getMaximum();
2081 const QDateTime minimum = getMinimum();
2082 Q_ASSERT(current >= minimum && current <= maximum);
2083
2084 QDateTime tmp = current;
2085 if (!setDigit(v&: tmp, index, newVal: min) || tmp < minimum)
2086 min = getDigit(t: minimum, index);
2087
2088 if (!setDigit(v&: tmp, index, newVal: max) || tmp > maximum)
2089 max = getDigit(t: maximum, index);
2090 }
2091 int pos = cursorPosition() - node.pos;
2092 if (pos < 0 || pos >= text.size())
2093 pos = -1;
2094
2095 /*
2096 If the value potentially can become another valid entry we don't want to
2097 skip to the next. E.g. In a M field (month without leading 0) if you type
2098 1 we don't want to autoskip (there might be [012] following) but if you
2099 type 3 we do.
2100 */
2101 return !potentialValue(str: text, min, max, index, currentValue: current, insert: pos);
