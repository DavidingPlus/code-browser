22namespace QRoundingDown {
23/*
24 Division, rounding down (rather than towards zero).
25
26 From C++11 onwards, integer division is defined to round towards zero, so we
27 can rely on that when implementing this. This is only used with denominator b
28 > 0, so we only have to treat negative numerator, a, specially.
29
30 If a is a multiple of b, adding 1 before and subtracting it after dividing by
31 b gets us to where we should be (albeit by an eccentric path), since the
32 adding caused rounding up, undone by the subtracting. Otherwise, adding 1
33 doesn't change the result of dividing by b; and we want one less than that
34 result. This is equivalent to subtracting b - 1 and simply dividing, except
35 when that subtraction would underflow.
36*/
37
38template<typename Int> constexpr Int qDiv(Int a, unsigned b)
39{ return a < 0 ? (a + 1) / int(b) - 1 : a / int(b); }
40
41template<typename Int> constexpr Int qMod(Int a, unsigned b)
42{ return a - qDiv(a, b) * b; }
43
44} // QRoundingDown
