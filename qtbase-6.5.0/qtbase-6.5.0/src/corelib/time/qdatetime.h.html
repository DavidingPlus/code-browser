258 struct ShortData {
259#if QT_VERSION >= QT_VERSION_CHECK(7,0,0) || defined(QT_BOOTSTRAPPED)
260# if Q_BYTE_ORDER == Q_LITTLE_ENDIAN
261 qint64 status : 8;
262# endif
263 qint64 msecs : 56;
264
265# if Q_BYTE_ORDER == Q_BIG_ENDIAN
266 qint64 status : 8;
267# endif
268#else
269# if Q_BYTE_ORDER == Q_LITTLE_ENDIAN
270 quintptr status : 8;
271# endif
272 // note: this is only 24 bits on 32-bit systems...
273 qintptr msecs : sizeof(void *) * 8 - 8;
274
275# if Q_BYTE_ORDER == Q_BIG_ENDIAN
276 quintptr status : 8;
277# endif
278#endif
279 friend constexpr bool operator==(const ShortData &lhs, const ShortData &rhs)
280 { return lhs.status == rhs.status && lhs.msecs == rhs.msecs; }
281 };

283 union Data {
284 enum {
285 // To be of any use, we need at least 60 years around 1970, which
286 // is 1,893,456,000,000 ms. That requires 41 bits to store, plus
287 // the sign bit. With the status byte, the minimum size is 50 bits.
288 CanBeSmall = sizeof(ShortData) * 8 > 50
289 };

429 // Generic clock, as long as it's compatible with us (= system_clock)
430 template <typename Clock, typename Duration>
431 static QDateTime fromStdTimePoint(const std::chrono::time_point<Clock, Duration> &time)
432 requires
433 requires(const std::chrono::time_point<Clock, Duration> &t) {
434 // the clock can be converted to system_clock
435 std::chrono::clock_cast<std::chrono::system_clock>(t);
436 // the duration can be converted to milliseconds
437 requires std::is_convertible_v<Duration, std::chrono::milliseconds>;
438 }
439 {
440 const auto sysTime = std::chrono::clock_cast<std::chrono::system_clock>(time);
441 // clock_cast can change the duration, so convert it again to milliseconds
442 const auto timeInMSec = std::chrono::time_point_cast<std::chrono::milliseconds>(sysTime);
443 return fromMSecsSinceEpoch(timeInMSec.time_since_epoch().count(), Qt::UTC);
444 }
