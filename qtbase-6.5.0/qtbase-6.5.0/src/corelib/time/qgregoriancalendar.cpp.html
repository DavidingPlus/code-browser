45bool QGregorianCalendar::leapTest(int year)
46{
47 if (year == QCalendar::Unspecified)
48 return false;
49
50 // No year 0 in Gregorian calendar, so -1, -5, -9 etc are leap years
51 if (year < 1)
52 ++year;
53
54 return year % 4 == 0 && (year % 100 != 0 || year % 400 == 0);
55}

57// Duplicating code from QRomanCalendar, but inlining isLeapYear() as leapTest():
58int QGregorianCalendar::monthLength(int month, int year)
59{
60 if (month < 1 || month > 12)
61 return 0;
62
63 if (month == 2)
64 return leapTest(year) ? 29 : 28;
65
66 return 30 | ((month & 1) ^ (month >> 3));
67}

84bool QGregorianCalendar::julianFromParts(int year, int month, int day, qint64 *jd)
85{
86 Q_ASSERT(jd);
87 if (!validParts(year, month, day))
88 return false;
89
90 if (year < 0)
91 ++year;
92
93 /*
94 * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
95 * This formula is correct for all julian days, when using mathematical integer
96 * division (round to negative infinity), not c++11 integer division (round to zero)
97 */
98 int a = month < 3 ? 1 : 0;
99 qint64 y = qint64(year) + 4800 - a;
100 int m = month + 12 * a - 3;
101 *jd = day + qDiv(a: 153 * m + 2, b: 5) - 32045
102 + 365 * y + qDiv(a: y, b: 4) - qDiv(a: y, b: 100) + qDiv(a: y, b: 400);
103 return true;
104}

106int QGregorianCalendar::yearStartWeekDay(int year)
107{
108 // Equivalent to weekDayOfJulian(julianForParts({year, 1, 1})
109 const int y = year - (year < 0 ? 800 : 801);
110 return qMod(a: y + qDiv(a: y, b: 4) - qDiv(a: y, b: 100) + qDiv(a: y, b: 400), b: 7) + 1;
111}

113int QGregorianCalendar::yearSharingWeekDays(QDate date)
114{
115 // Returns a post-epoch year, no later than 2400, that has the same pattern
116 // of week-days (in the proleptic Gregorian calendar) as the year in which
117 // the given date falls. This will be the year in question if it's in the
118 // given range. Otherwise, the returned year's last two (decimal) digits
119 // won't coincide with the month number or day-of-month of the given date.
120 // For positive years, except when necessary to avoid such a clash, the
121 // returned year's last two digits shall coincide with those of the original
122 // year.
123
124 // Needed when formatting dates using system APIs with limited year ranges
125 // and possibly only a two-digit year. (The need to be able to safely
126 // replace the two-digit form of the returned year with a suitable form of
127 // the true year, when they don't coincide, is why the last two digits are
128 // treated specially.)
129
130 static_assert((400 * 365 + 97) % 7 == 0);
131 // A full 400-year cycle of the Gregorian calendar has 97 + 400 * 365 days;
132 // as 365 is one more than a multiple of seven and 497 is a multiple of
133 // seven, that full cycle is a whole number of weeks. So adding a multiple
134 // of four hundred years should get us a result that meets our needs.
135
136 const int year = date.year();
137 int res = (year < 1970
138 ? 2400 - (2000 - (year < 0 ? year + 1 : year)) % 400
139 : year > 2399 ? 2000 + (year - 2000) % 400 : year);
140 Q_ASSERT(res > 0);
141 if (res != year) {
142 const int lastTwo = res % 100;
143 if (lastTwo == date.month() || lastTwo == date.day()) {
144 Q_ASSERT(lastTwo && !(lastTwo & ~31));
145 // Last two digits of these years are all > 31:
146 static constexpr int usual[] = { 2198, 2199, 2098, 2099, 2399, 2298, 2299 };
147 static constexpr int leaps[] = { 2396, 2284, 2296, 2184, 2196, 2084, 2096 };
148 // Indexing is: first day of year's day-of-week, Monday = 0, one less
149 // than Qt's, as it's simpler to subtract one than to s/7/0/.
150 res = (leapTest(year) ? leaps : usual)[yearStartWeekDay(year) - 1];
151 }
152 Q_ASSERT(QDate(res, 1, 1).dayOfWeek() == QDate(year, 1, 1).dayOfWeek());
153 Q_ASSERT(QDate(res, 12, 31).dayOfWeek() == QDate(year, 12, 31).dayOfWeek());
154 }
155 Q_ASSERT(res >= 1970 && res <= 2400);
156 return res;
157}

164QCalendar::YearMonthDay QGregorianCalendar::partsFromJulian(qint64 jd)
165{
166 /*
167 * Math from The Calendar FAQ at http://www.tondering.dk/claus/cal/julperiod.php
168 * This formula is correct for all julian days, when using mathematical integer
169 * division (round to negative infinity), not c++11 integer division (round to zero)
170 */
171 qint64 a = jd + 32044;
172 qint64 b = qDiv(a: 4 * a + 3, b: 146097);
173 int c = a - qDiv(a: 146097 * b, b: 4);
174
175 int d = qDiv(a: 4 * c + 3, b: 1461);
176 int e = c - qDiv(a: 1461 * d, b: 4);
177 int m = qDiv(a: 5 * e + 2, b: 153);
178
179 int y = 100 * b + d - 4800 + qDiv(a: m, b: 10);
180
181 // Adjust for no year 0
182 int year = y > 0 ? y : y - 1;
183 int month = m + 3 - 12 * qDiv(a: m, b: 10);
184 int day = e - qDiv(a: 153 * m + 2, b: 5) + 1;
185
186 return QCalendar::YearMonthDay(year, month, day);
187}
