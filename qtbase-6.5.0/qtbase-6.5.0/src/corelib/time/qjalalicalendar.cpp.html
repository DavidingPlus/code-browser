14// Constants
15
16static const qint64 cycleDays = 1029983;
17static const int cycleYears = 2820;
18static const double yearLength = 365.24219858156028368; // 365 + leapRatio;
19static const qint64 jalaliEpoch = 2121446; // 475/01/01 AP, start of 2820 cycle
20
21// Calendar implementation
22
23static inline int cycle(qint64 jdn)
24{
25 return qDiv(a: jdn - jalaliEpoch, b: cycleDays);
26}
27
28qint64 cycleStart(int cycleNo)
29{
30 return jalaliEpoch + cycleNo * cycleDays;
31}
32
33qint64 firstDayOfYear(int year, int cycleNo)
34{
35 qint64 firstDOYinEra = static_cast<qint64>(qFloor(v: year * yearLength));
36 return jalaliEpoch + cycleNo * cycleDays + firstDOYinEra;
37}

39/*!
40 \since 5.14
41
42 \class QJalaliCalendar
43 \inmodule QtCore
44 \brief The QJalaliCalendar class provides Jalali (Hijri Shamsi) calendar
45 system implementation.
46
47 \section1 Solar Hijri Calendar System
48
49 The Solar Hijri calendar, also called the Solar Hejri calendar, Shamsi
50 Hijri calendar or Jalali calendar, is the official calendar of Iran and
51 Afghanistan. It begins on the vernal equinox (Nowruz) as determined by
52 astronomical calculation for the Iran Standard Time meridian
53 (52.5Â°E or GMT+3.5h). This determination of starting moment is more accurate
54 than the Gregorian calendar for predicting the date of the vernal equinox,
55 because it uses astronomical observations rather than mathematical rules.
56
57 \section2 Calendar Organization
58
59 Each of the twelve months corresponds with a zodiac sign. The first six
60 months have 31 days, the next five have 30 days, and the last month has 29
61 days in usual years but 30 days in leap years. The New Year's Day always
62 falls on the March equinox.
63
64 \section2 Leap Year Rules
65
66 The Solar Hijri calendar produces a five-year leap year interval after about
67 every seven four-year leap year intervals. It usually follows a 33-year
68 cycle with occasional interruptions by single 29-year or 37-year subcycles.
69 The reason for this behavior is that it tracks the observed vernal equinox.
70 By contrast, some less accurate predictive algorithms are in use based
71 on confusion between the average tropical year (365.2422 days, approximated
72 with near 128-year cycles or 2820-year great cycles) and the mean interval
73 between spring equinoxes (365.2424 days, approximated with a near 33-year
74 cycle).
75
76 Source: \l {https://en.wikipedia.org/wiki/Solar_Hijri_calendar}{Wikipedia
77 page on Solar Hijri Calendar}
78 */

93bool QJalaliCalendar::isLeapYear(int year) const
94{
95 if (year == QCalendar::Unspecified)
96 return false;
97 if (year < 0)
98 year++;
99 return qMod(a: (year + 2346) * 683, b: 2820) < 683;
100}

117bool QJalaliCalendar::dateToJulianDay(int year, int month, int day, qint64 *jd) const
118{
119 Q_ASSERT(jd);
120 if (!isDateValid(year, month, day))
121 return false;
122
123 const int y = year - (year < 0 ? 474 : 475);
124 const int c = qDiv(a: y, b: cycleYears);
125 const int yearInCycle = y - c * cycleYears;
126 int dayInYear = day;
127 for (int i = 1; i < month; ++i)
128 dayInYear += daysInMonth(month: i, year);
129 *jd = firstDayOfYear(year: yearInCycle, cycleNo: c) + dayInYear - 1;
130 return true;
131}

133QCalendar::YearMonthDay QJalaliCalendar::julianDayToDate(qint64 jd) const
134{
135 const int c = cycle(jdn: jd);
136 int yearInCycle = qFloor(v: (jd - cycleStart(cycleNo: c)) / yearLength);
137 int year = yearInCycle + 475 + c * cycleYears;
138 int day = jd - firstDayOfYear(year: yearInCycle, cycleNo: c) + 1;
139 if (day > daysInYear(year: year <= 0 ? year - 1 : year)) {
140 year++;
141 day = 1;
142 }
143 if (year <= 0)
144 year--;
145 int month;
146 for (month = 1; month < 12; ++month) {
147 const int last = daysInMonth(month, year);
148 if (day <= last)
149 break;
150 day -= last;
151 }
152 return QCalendar::YearMonthDay(year, month, day);
153}

155int QJalaliCalendar::daysInMonth(int month, int year) const
156{
157 if (year && month > 0 && month <= 12)
158 return month < 7 ? 31 : month < 12 || isLeapYear(year) ? 30 : 29;
159
160 return 0;
161}
