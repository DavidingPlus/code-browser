26/*
27 Qt represents n BCE as -n, whereas struct tm's tm_year field represents a
28 year by the number of years after (negative for before) 1900, so that 1+m
29 BCE is -1900 -m; so treating 1 BCE as 0 CE. We thus shift by different
30 offsets depending on whether the year is BCE or CE.
31*/
32constexpr int tmYearFromQYear(int year) { return year - (year < 0 ? 1899 : 1900); }
33constexpr int qYearFromTmYear(int year) { return year + (year < -1899 ? 1899 : 1900); }
34
35constexpr inline qint64 tmSecsWithinDay(const struct tm &when)
36{
37 return (when.tm_hour * MINS_PER_HOUR + when.tm_min) * SECS_PER_MIN + when.tm_sec;
38}

40/* If mktime() returns -1, is it really an error ?
41
42 It might return -1 because we're looking at the last second of 1969 and
43 mktime does support times before 1970 (POSIX says "If the year is <1970 or
44 the value is negative, the relationship is undefined" and MS rejects the
45 value, consistent with that; so we don't call mktime() on MS in this case and
46 can't get -1 unless it's a real error). However, on UNIX, that's -1 UTC time
47 and all we know, aside from mktime's return, is the local time. (We could
48 check errno, but we call mktime from within a qt_scoped_lock(QBasicMutex),
49 whose unlocking and destruction of the locker might frob errno.)
50
51 We can assume the zone offset is a multiple of five minutes and less than a
52 day, so this can only arise for the last second of a minute that differs from
53 59 by a multiple of 5 on the last day of 1969 or the first day of 1970. That
54 makes for a cheap pre-test; if it holds, we can ask mktime about the first
55 second of the same minute; if it gives us -60, then the -1 we originally saw
56 is not an error (or was an error, but needn't have been).
57*/
58inline bool meansEnd1969(tm *local)
59{
60#ifdef Q_OS_WIN
61 Q_UNUSED(local);
62 return false;
63#else
64 if (local->tm_sec < 59 || local->tm_year < 69 || local->tm_year > 70
65 || local->tm_min % 5 != 4 // Assume zone offset is a multiple of 5 mins
66 || (local->tm_year == 69
67 ? local->tm_mon < 11 || local->tm_mday < 31
68 : local->tm_mon > 0 || local->tm_mday > 1)) {
69 return false;
70 }
71 tm copy = *local;
72 copy.tm_sec--; // Preceding second should get -2, not -1
73 if (qMkTime(when: &copy) != -2)
74 return false;
75 // The original call to qMkTime() may have returned -1 as failure, not
76 // updating local, even though it could have; so fake it here. Assumes there
77 // was no transition in the last minute of the day !
78 *local = copy;
79 local->tm_sec++; // Advance back to the intended second
80 return true;
81#endif
82}

84/*
85 Call mktime but bypass its fixing of denormal times.
86
87 The POSIX spec says mktime() accepts a struct tm whose fields lie outside
88 the usual ranges; the parameter is not const-qualified and will be updated
89 to have values in those ranges. However, MS's implementation doesn't do that
90 (or hasn't always done it); and the only member we actually want updated is
91 the tm_isdst flag. (Aside: MS's implementation also only works for tm_year
92 >= 70; this is, in fact, in accordance with the POSIX spec; but all known
93 UNIX libc implementations in fact have a signed time_t and Do The Sensible
94 Thing, to the best of their ability, at least for 0 <= tm_year < 70; see
95 meansEnd1969 for the handling of the last second of UTC's 1969.)
96
97 If we thought we knew tm_isdst and mktime() disagrees, it'll let us know
98 either by correcting it - in which case it adjusts the struct tm to reflect
99 the same time, but represented using the right tm_isdst, so typically an
100 hour earlier or later - or by returning -1. When this happens, the way we
101 actually use mktime(), we don't want a revised time with corrected DST, we
102 want the original time with its corrected DST; so we retry the call, this
103 time not claiming to know the DST-ness.
104
105 POSIX doesn't actually say what to do if the specified struct tm describes a
106 time in a spring-forward gap: read literally, this is an unrepresentable
107 time and it could return -1, setting errno to EOVERFLOW. However, actual
108 implementations chose a time one side or the other of the gap. For example,
109 if we claim to know DST, glibc pushes to the other side of the gap (changing
110 tm_isdst), but stays on the indicated branch of a repetition (no change to
111 tm_isdst); this matches how QTimeZonePrivate::dataForLocalTime() uses its
112 hint; in either case, if we don't claim to know DST, glibc picks the DST
113 candidate. (Experiments conducted with glibc 2.31-9.)
114*/
115inline bool callMkTime(tm *local, time_t *secs)
116{
117 constexpr time_t maybeError = -1; // mktime()'s return on error; or last second of 1969 UTC.
118 const tm copy = *local;
119 *secs = qMkTime(when: local);
120 bool good = *secs != maybeError || meansEnd1969(local);
121 if (copy.tm_isdst >= 0 && (!good || local->tm_isdst != copy.tm_isdst)) {
122 // We thought we knew DST-ness, but were wrong:
123 *local = copy;
124 local->tm_isdst = -1;
125 *secs = qMkTime(when: local);
126 good = *secs != maybeError || meansEnd1969(local);
127 }
128#if defined(Q_OS_WIN)
129 // Windows mktime for the missing hour backs up 1 hour instead of advancing
130 // 1 hour. If time differs and is standard time then this has happened, so
131 // add 2 hours to the time and 1 hour to the secs
132 if (local->tm_isdst == 0 && local->tm_hour != copy.tm_hour) {
133 local->tm_hour += 2;
134 if (local->tm_hour > 23) {
135 local->tm_hour -= 24;
136 if (++local->tm_mday > QGregorianCalendar::monthLength(
137 local->tm_mon + 1, qYearFromTmYear(local->tm_year))) {
138 local->tm_mday = 1;
139 if (++local->tm_mon > 11) {
140 local->tm_mon = 0;
141 ++local->tm_year;
142 }
143 }
144 }
145 *secs += 3600;
146 local->tm_isdst = 1;
147 }
148#endif // Q_OS_WIN
149 return good;
150}

152struct tm timeToTm(qint64 localDay, int secs, QDateTimePrivate::DaylightStatus dst)
153{
154 Q_ASSERT(0 <= secs && secs < 3600 * 24);
155 const auto ymd = QGregorianCalendar::partsFromJulian(jd: JULIAN_DAY_FOR_EPOCH + localDay);
156 struct tm local = {};
157 local.tm_year = tmYearFromQYear(year: ymd.year);
158 local.tm_mon = ymd.month - 1;
159 local.tm_mday = ymd.day;
160 local.tm_hour = secs / 3600;
161 local.tm_min = (secs % 3600) / 60;
162 local.tm_sec = (secs % 60);
163 local.tm_isdst = int(dst);
164 return local;
165}

167bool qtLocalTime(time_t utc, struct tm *local)
168{
169 // This should really be done under the environmentMutex wrapper qglobal.cpp
170 // uses in qTzSet() and friends. However, the only sane way to do that would
171 // be to move this whole function there (and replace its qTzSet() with a
172 // naked tzset(), since it'd already be mutex-protected).
173#if defined(Q_OS_WIN)
174 // The doc of localtime_s() says that localtime_s() corrects for the same
175 // things _tzset() sets the globals for, but doesn't explicitly say that it
176 // calls _tzset(), and QTBUG-109974 reveals the need for a _tzset() call.
177 qTzSet();
178 return !localtime_s(local, &utc);
179#elif QT_CONFIG(thread) && defined(_POSIX_THREAD_SAFE_FUNCTIONS)
180 // Use the reentrant version of localtime() where available, as it is
181 // thread-safe and doesn't use a shared static data area.
182 // As localtime() is specified to work as if it called tzset(), but
183 // localtime_r() does not have this constraint, make an explicit call.
184 // The explicit call should also request a re-parse of timezone info.
185 qTzSet();
186 if (tm *res = localtime_r(&utc, local)) {
187 Q_ASSERT(res == local);
188 return true;
189 }
190 return false;
191#else
192 // POSIX mandates that localtime() behaves as if it called tzset().
193 // Returns shared static data which may be overwritten at any time
194 // So copy the result asap:
195 if (tm *res = localtime(timer: &utc)) {
196 *local = *res;
197 return true;
198 }
199 return false;
200#endif
201}

204QString qt_tzname(QDateTimePrivate::DaylightStatus daylightStatus)
205{
206 int isDst = (daylightStatus == QDateTimePrivate::DaylightTime) ? 1 : 0;
207#if defined(Q_CC_MSVC)
208 size_t s = 0;
209 char name[512];
210 if (_get_tzname(&s, name, 512, isDst))
211 return QString();
212 return QString::fromLocal8Bit(name);
213#else
214 return QString::fromLocal8Bit(ba: tzname[isDst]);
215#endif // Q_OS_WIN
216}

242#ifndef QT_BOOTSTRAPPED
243// Even if local time is currently in DST, this returns the standard time offset
244// (in seconds) nominally in effect at present:
245int getCurrentStandardUtcOffset()
246{
247#ifdef Q_OS_WIN
248 TIME_ZONE_INFORMATION tzInfo;
249 GetTimeZoneInformation(&tzInfo);
250 return -tzInfo.Bias * SECS_PER_MIN;
251#else
252 qTzSet();
253 const time_t curr = time(nullptr);
254 /* Set t to the UTC representation of curr; the time whose local standard
255 time representation coincides with that differs from curr by local time's
256 standard offset. Note that gmtime() leaves the tm_isdst flag set to 0,
257 so mktime() will, even if local time is currently using DST, return the
258 time since epoch at which local standard time would have the same
259 representation as UTC's representation of curr. The fact that mktime()
260 also flips tm_isdst and updates the time fields to the DST-equivalent
261 time needn't concern us here; all that matters is that it returns the
262 time after epoch at which standard time's representation would have
263 matched UTC's, had it been in effect.
264 */
265# if defined(_POSIX_THREAD_SAFE_FUNCTIONS)
266 struct tm t;
267 if (gmtime_r(&curr, &t))
268 return curr - qMkTime(&t);
269# else
270 if (struct tm *tp = gmtime(&curr)) {
271 struct tm t = *tp; // Copy it quick, hopefully before it can get stomped
272 return curr - qMkTime(&t);
273 }
274# endif
275 // We can't tell, presume UTC.
276 return 0;
277#endif // Platform choice
278}

288#define IC(N) std::integral_constant<qint64, N>()

290// True if combining day and seconds overflows qint64; otherwise, sets *epochSeconds
291inline bool daysAndSecondsOverflow(qint64 julianDay, qint64 daySeconds, qint64 *epochSeconds)
292{
293 return mul_overflow(v1: julianDay - JULIAN_DAY_FOR_EPOCH, IC(SECS_PER_DAY), r: epochSeconds)
294 || add_overflow(v1: *epochSeconds, v2: daySeconds, r: epochSeconds);
295}
296
297// True if combining seconds and millis overflows; otherwise sets *epochMillis
298inline bool secondsAndMillisOverflow(qint64 epochSeconds, qint64 millis, qint64 *epochMillis)
299{
300 return mul_overflow(v1: epochSeconds, IC(MSECS_PER_SEC), r: epochMillis)
301 || add_overflow(v1: *epochMillis, v2: millis, r: epochMillis);
302}

306// Calls the platform variant of localtime() for the given utcMillis, and
307// returns the local milliseconds, offset from UTC and DST status.
308QDateTimePrivate::ZoneState utcToLocal(qint64 utcMillis)
309{
310 const time_t epochSeconds = QRoundingDown::qDiv(a: utcMillis, b: MSECS_PER_SEC);
311 const int msec = utcMillis - epochSeconds * MSECS_PER_SEC;
312 Q_ASSERT(msec >= 0 && msec < MSECS_PER_SEC);
313 if (qint64(epochSeconds) * MSECS_PER_SEC + msec != utcMillis) // time_t range too narrow
314 return {utcMillis};
315
316 tm local;
317 if (!qtLocalTime(utc: epochSeconds, local: &local))
318 return {utcMillis};
319
320 qint64 jd;
321 if (Q_UNLIKELY(!QGregorianCalendar::julianFromParts(qYearFromTmYear(local.tm_year),
322 local.tm_mon + 1, local.tm_mday, &jd))) {
323 return {utcMillis};
324 }
325 const qint64 daySeconds = tmSecsWithinDay(when: local);
326 Q_ASSERT(0 <= daySeconds && daySeconds < SECS_PER_DAY);
327 qint64 localSeconds, localMillis;
328 if (Q_UNLIKELY(daysAndSecondsOverflow(jd, daySeconds, &localSeconds)
329 || secondsAndMillisOverflow(localSeconds, qint64(msec), &localMillis))) {
330 return {utcMillis};
331 }
332 const auto dst
333 = local.tm_isdst ? QDateTimePrivate::DaylightTime : QDateTimePrivate::StandardTime;
334 return { localMillis, int(localSeconds - epochSeconds), dst };
335}

337QString localTimeAbbbreviationAt(qint64 local, QDateTimePrivate::DaylightStatus dst)
338{
339 const qint64 localDays = QRoundingDown::qDiv(a: local, b: MSECS_PER_DAY);
340 qint64 millis = local - localDays * MSECS_PER_DAY;
341 Q_ASSERT(0 <= millis && millis < MSECS_PER_DAY); // Definition of QRD::qDiv.
342 struct tm tmLocal = timeToTm(localDay: localDays, secs: int(millis / MSECS_PER_SEC), dst);
343 time_t utcSecs;
344 if (!callMkTime(local: &tmLocal, secs: &utcSecs))
345 return {};
346 return qt_tzname(daylightStatus: tmLocal.tm_isdst > 0 ? QDateTimePrivate::DaylightTime
347 : QDateTimePrivate::StandardTime);
348}

350QDateTimePrivate::ZoneState mapLocalTime(qint64 local, QDateTimePrivate::DaylightStatus dst)
351{
352 qint64 localSecs = local / MSECS_PER_SEC;
353 qint64 millis = local - localSecs * MSECS_PER_SEC; // 0 or with same sign as local
354 const qint64 localDays = QRoundingDown::qDiv(a: localSecs, b: SECS_PER_DAY);
355 qint64 daySecs = localSecs - localDays * SECS_PER_DAY;
356 Q_ASSERT(0 <= daySecs && daySecs < SECS_PER_DAY); // Definition of QRD::qDiv.
357
358 struct tm tmLocal = timeToTm(localDay: localDays, secs: daySecs, dst);
359 time_t utcSecs;
360 if (!callMkTime(local: &tmLocal, secs: &utcSecs))
361 return {local};
362
363 // TODO: for glibc, we could use tmLocal.tm_gmtoff
364 // That would give us offset directly, hence localSecs = offset + utcSecs
365 // Provisional offset, until we have a revised localSeconds:
366 int offset = QRoundingDown::qDiv(a: local, b: MSECS_PER_SEC) - utcSecs;
367 dst = tmLocal.tm_isdst > 0 ? QDateTimePrivate::DaylightTime : QDateTimePrivate::StandardTime;
368 qint64 jd;
369 if (Q_UNLIKELY(!QGregorianCalendar::julianFromParts(
370 qYearFromTmYear(tmLocal.tm_year), tmLocal.tm_mon + 1, tmLocal.tm_mday,
371 &jd))) {
372 return {local, offset, dst, false};
373 }
374 daySecs = tmSecsWithinDay(when: tmLocal);
375 Q_ASSERT(0 <= daySecs && daySecs < SECS_PER_DAY);
376 if (daySecs > 0 && jd < JULIAN_DAY_FOR_EPOCH) {
377 ++jd;
378 daySecs -= SECS_PER_DAY;
379 }
380 if (Q_UNLIKELY(daysAndSecondsOverflow(jd, daySecs, &localSecs)))
381 return {local, offset, dst, false};
382
383 offset = localSecs - utcSecs;
384
385 // The only way localSecs and millis can now have opposite sign is for
386 // resolution of the local time to have kicked us across the epoch, in which
387 // case there's no danger of overflow. So if overflow is in danger of
388 // happening, we're already doing the best we can to avoid it.
389 qint64 revised;
390 const bool overflow = secondsAndMillisOverflow(epochSeconds: localSecs, millis, epochMillis: &revised);
391 return {overflow ? local : revised, offset, dst, !overflow};
392}

394/*!
395 \internal
396 Determine the range of the system time_t functions.
397
398 On MS-systems (where time_t is 64-bit by default), the start-point is the
399 epoch, the end-point is the end of the year 3000 (for mktime(); for
400 _localtime64_s it's 18 days later, but we ignore that here). Darwin's range
401 runs from the beginning of 1900 to the end of its 64-bit time_t and Linux
402 uses the full range of time_t (but this might still be 32-bit on some
403 embedded systems).
404
405 (One potential constraint might appear to be the range of struct tm's int
406 tm_year, only allowing time_t to represent times from the start of year
407 1900+INT_MIN to the end of year INT_MAX. The 26-bit number of seconds in a
408 year means that a 64-bit time_t can indeed represent times outside the range
409 of 32-bit years, by a factor of 32 - but the range of representable
410 milliseconds needs ten more bits than that of seconds, so can't reach the
411 ends of the 32-bit year range.)
412
413 Given the diversity of ranges, we conservatively estimate the actual
414 supported range by experiment on the first call to qdatetime.cpp's
415 millisInSystemRange() by exploration among the known candidates, converting
416 the result to milliseconds and flagging whether each end is the qint64
417 range's bound (so millisInSystemRange will know not to try to pad beyond
418 those bounds). The probed date-times are somewhat inside the range, but
419 close enough to the relevant bound that we can be fairly sure the bound is
420 reached, if the probe succeeds.
421*/
422SystemMillisRange computeSystemMillisRange()
423{
424 // Assert this here, as this is called just once, in a static initialization.
425 [[maybe_unused]] qint64 epochJd;
426 Q_ASSERT(QGregorianCalendar::julianFromParts(1970, 1, 1, &epochJd)
427 && epochJd == JULIAN_DAY_FOR_EPOCH);
428
429 constexpr qint64 TIME_T_MAX = std::numeric_limits<time_t>::max();
430 using Bounds = std::numeric_limits<qint64>;
431 constexpr bool isNarrow = Bounds::max() / MSECS_PER_SEC > TIME_T_MAX;
432 if constexpr (isNarrow) {
433 const qint64 msecsMax = quint64(TIME_T_MAX) * MSECS_PER_SEC - 1 + MSECS_PER_SEC;
434 const qint64 msecsMin = -1 - msecsMax; // TIME_T_MIN is -1 - TIME_T_MAX
435 // If we reach back to msecsMin, use it; otherwise, assume 1970 cut-off (MS).
436 struct tm local = {};
437 local.tm_year = tmYearFromQYear(year: 1901);
438 local.tm_mon = 11;
439 local.tm_mday = 15; // A day and a bit after the start of 32-bit time_t:
440 local.tm_isdst = -1;
441 return {.min: qMkTime(when: &local) == -1 ? 0 : msecsMin, .max: msecsMax, .minClip: false, .maxClip: false};
442 } else {
443 const struct { int year; qint64 millis; } starts[] = {
444 { .year: int(QDateTime::YearRange::First) + 1, .millis: Bounds::min() },
445 // Beginning of the Common Era:
446 { .year: 1, .millis: -Q_INT64_C(62135596800000) },
447 // Invention of the Gregorian calendar:
448 { .year: 1582, .millis: -Q_INT64_C(12244089600000) },
449 // Its adoption by the anglophone world:
450 { .year: 1752, .millis: -Q_INT64_C(6879427200000) },
451 // Before this, struct tm's tm_year is negative (Darwin):
452 { .year: 1900, .millis: -Q_INT64_C(2208988800000) },
453 }, ends[] = {
454 { .year: int(QDateTime::YearRange::Last) - 1, .millis: Bounds::max() },
455 // MS's end-of-range, end of year 3000:
456 { .year: 3000, Q_INT64_C(32535215999999) },
457 };
458 // Assume we do at least reach the end of a signed 32-bit time_t (since
459 // our actual time_t is bigger than that):
460 qint64 stop =
461 quint64(std::numeric_limits<qint32>::max()) * MSECS_PER_SEC - 1 + MSECS_PER_SEC;
462 // Cleared if first pass round loop fails:
463 bool stopMax = true;
464 for (const auto c : ends) {
465 struct tm local = {};
466 local.tm_year = tmYearFromQYear(year: c.year);
467 local.tm_mon = 11;
468 local.tm_mday = 31;
469 local.tm_hour = 23;
470 local.tm_min = local.tm_sec = 59;
471 local.tm_isdst = -1;
472 if (qMkTime(when: &local) != -1) {
473 stop = c.millis;
474 break;
475 }
476 stopMax = false;
477 }
478 bool startMin = true;
479 for (const auto c : starts) {
480 struct tm local {};
481 local.tm_year = tmYearFromQYear(year: c.year);
482 local.tm_mon = 1;
483 local.tm_mday = 1;
484 local.tm_isdst = -1;
485 if (qMkTime(when: &local) != -1)
486 return {.min: c.millis, .max: stop, .minClip: startMin, .maxClip: stopMax};
487 startMin = false;
488 }
489 return {.min: 0, .max: stop, .minClip: false, .maxClip: stopMax};
490 }
491}
