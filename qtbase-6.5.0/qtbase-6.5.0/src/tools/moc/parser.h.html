54inline bool Parser::test(Token token)
55{
56 if (index < symbols.size() && symbols.at(i: index).token == token) {
57 ++index;
58 return true;
59 }
60 return false;
61}
62
63inline Token Parser::lookup(int k)
64{
65 const int l = index - 1 + k;
66 return l < symbols.size() ? symbols.at(i: l).token : NOTOKEN;
67}
68
69inline void Parser::next(Token token)
70{
71 if (!test(token))
72 error();
73}
74
75inline void Parser::next(Token token, const char *msg)
76{
77 if (!test(token))
78 error(msg);
79}
