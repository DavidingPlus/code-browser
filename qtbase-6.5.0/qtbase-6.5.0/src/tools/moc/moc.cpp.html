29bool Moc::parseClassHead(ClassDef *def)
30{
31 // figure out whether this is a class declaration, or only a
32 // forward or variable declaration.
33 int i = 0;
34 Token token;
35 do {
36 token = lookup(k: i++);
37 if (token == COLON || token == LBRACE)
38 break;
39 if (token == SEMIC || token == RANGLE)
40 return false;
41 } while (token);
42
43 // support attributes like "class [[deprecated]]] name"
44 skipCxxAttributes();
45
46 if (!test(token: IDENTIFIER)) // typedef struct { ... }
47 return false;
48 QByteArray name = lexem();
49
50 // support "class IDENT name" and "class IDENT(IDENT) name"
51 // also support "class IDENT name (final|sealed|Q_DECL_FINAL)"
52 if (test(token: LPAREN)) {
53 until(RPAREN);
54 if (!test(token: IDENTIFIER))
55 return false;
56 name = lexem();
57 } else if (test(token: IDENTIFIER)) {
58 const QByteArray lex = lexem();
59 if (lex != "final" && lex != "sealed" && lex != "Q_DECL_FINAL")
60 name = lex;
61 }
62
63 def->qualified += name;
64 while (test(token: SCOPE)) {
65 def->qualified += lexem();
66 if (test(token: IDENTIFIER)) {
67 name = lexem();
68 def->qualified += name;
69 }
70 }
71 def->classname = name;
72
73 if (test(token: IDENTIFIER)) {
74 const QByteArray lex = lexem();
75 if (lex != "final" && lex != "sealed" && lex != "Q_DECL_FINAL")
76 return false;
77 }
78
79 if (test(token: COLON)) {
80 do {
81 test(token: VIRTUAL);
82 FunctionDef::Access access = FunctionDef::Public;
83 if (test(token: PRIVATE))
84 access = FunctionDef::Private;
85 else if (test(token: PROTECTED))
86 access = FunctionDef::Protected;
87 else
88 test(token: PUBLIC);
89 test(token: VIRTUAL);
90 const QByteArray type = parseType().name;
91 // ignore the 'class Foo : BAR(Baz)' case
92 if (test(token: LPAREN)) {
93 until(RPAREN);
94 } else {
95 def->superclassList += qMakePair(value1: type, value2&: access);
96 }
97 } while (test(token: COMMA));
98
99 if (!def->superclassList.isEmpty()
100 && knownGadgets.contains(key: def->superclassList.constFirst().first)) {
101 // Q_GADGET subclasses are treated as Q_GADGETs
102 knownGadgets.insert(key: def->classname, value: def->qualified);
103 knownGadgets.insert(key: def->qualified, value: def->qualified);
104 }
105 }
106 if (!test(token: LBRACE))
107 return false;
108 def->begin = index - 1;
109 bool foundRBrace = until(RBRACE);
110 def->end = index;
111 index = def->begin + 1;
112 return foundRBrace;
113}

115Type Moc::parseType()
116{
117 Type type;
118 bool hasSignedOrUnsigned = false;
119 bool isVoid = false;
120 type.firstToken = lookup();
121 for (;;) {
122 skipCxxAttributes();
123 switch (next()) {
124 case SIGNED:
125 case UNSIGNED:
126 hasSignedOrUnsigned = true;
127 Q_FALLTHROUGH();
128 case CONST:
129 case VOLATILE:
130 type.name += lexem();
131 type.name += ' ';
132 if (lookup(k: 0) == VOLATILE)
133 type.isVolatile = true;
134 continue;
135 case Q_MOC_COMPAT_TOKEN:
136 case Q_INVOKABLE_TOKEN:
137 case Q_SCRIPTABLE_TOKEN:
138 case Q_SIGNALS_TOKEN:
139 case Q_SLOTS_TOKEN:
140 case Q_SIGNAL_TOKEN:
141 case Q_SLOT_TOKEN:
142 type.name += lexem();
143 return type;
144 case NOTOKEN:
145 return type;
146 default:
147 prev();
148 break;
149 }
150 break;
151 }
152
153 skipCxxAttributes();
154 test(token: ENUM) || test(token: CLASS) || test(token: STRUCT);
155 for(;;) {
156 skipCxxAttributes();
157 switch (next()) {
158 case IDENTIFIER:
159 // void mySlot(unsigned myArg)
160 if (hasSignedOrUnsigned) {
161 prev();
162 break;
163 }
164 Q_FALLTHROUGH();
165 case CHAR:
166 case SHORT:
167 case INT:
168 case LONG:
169 type.name += lexem();
170 // preserve '[unsigned] long long', 'short int', 'long int', 'long double'
171 if (test(token: LONG) || test(token: INT) || test(token: DOUBLE)) {
172 type.name += ' ';
173 prev();
174 continue;
175 }
176 break;
177 case FLOAT:
178 case DOUBLE:
179 case VOID:
180 case BOOL:
181 case AUTO:
182 type.name += lexem();
183 isVoid |= (lookup(k: 0) == VOID);
184 break;
185 case NOTOKEN:
186 return type;
187 default:
188 prev();
189 ;
190 }
191 if (test(token: LANGLE)) {
192 if (type.name.isEmpty()) {
193 // '<' cannot start a type
194 return type;
195 }
196 type.name += lexemUntil(RANGLE);
197 }
198 if (test(token: SCOPE)) {
199 type.name += lexem();
200 type.isScoped = true;
201 } else {
202 break;
203 }
204 }
205 while (test(token: CONST) || test(token: VOLATILE) || test(token: SIGNED) || test(token: UNSIGNED)
206 || test(token: STAR) || test(token: AND) || test(token: ANDAND)) {
207 type.name += ' ';
208 type.name += lexem();
209 if (lookup(k: 0) == AND)
210 type.referenceType = Type::Reference;
211 else if (lookup(k: 0) == ANDAND)
212 type.referenceType = Type::RValueReference;
213 else if (lookup(k: 0) == STAR)
214 type.referenceType = Type::Pointer;
215 }
216 type.rawName = type.name;
217 // transform stupid things like 'const void' or 'void const' into 'void'
218 if (isVoid && type.referenceType == Type::NoReference) {
219 type.name = "void";
220 }
221 return type;
222}

230bool Moc::parseEnum(EnumDef *def)
231{
232 bool isTypdefEnum = false; // typedef enum { ... } Foo;
233
234 if (test(token: CLASS) || test(token: STRUCT))
235 def->isEnumClass = true;
236
237 if (test(token: IDENTIFIER)) {
238 def->name = lexem();
239 } else {
240 if (lookup(k: -1) != TYPEDEF)
241 return false; // anonymous enum
242 isTypdefEnum = true;
243 }
244 if (test(token: COLON)) { // C++11 strongly typed enum
245 // enum Foo : unsigned long { ... };
246 parseType(); //ignore the result
247 }
248 if (!test(token: LBRACE))
249 return false;
250 auto handleInclude = [this]() -> IncludeState {
251 bool hadIncludeBegin = false;
252 if (test(token: MOC_INCLUDE_BEGIN)) {
253 currentFilenames.push(x: symbol().unquotedLexem());
254 // we do not return early to handle empty headers in one go
255 hadIncludeBegin = true;
256 }
257 if (test(token: NOTOKEN)) {
258 next(token: MOC_INCLUDE_END);
259 currentFilenames.pop();
260 return IncludeState::IncludeEnd;
261 }
262 if (hadIncludeBegin)
263 return IncludeState::IncludeBegin;
264 else
265 return IncludeState::NoInclude;
266 };
267 do {
268 handleInclude();
269 if (lookup() == RBRACE) // accept trailing comma
270 break;
271 next(token: IDENTIFIER);
272 def->values += lexem();
273 handleInclude();
274 skipCxxAttributes();
275 } while (test(token: EQ) ? until(COMMA) : test(token: COMMA));
276 next(token: RBRACE);
277 if (isTypdefEnum) {
278 if (!test(token: IDENTIFIER))
279 return false;
280 def->name = lexem();
281 }
282 return true;
283}

285void Moc::parseFunctionArguments(FunctionDef *def)
286{
287 Q_UNUSED(def);
288 while (hasNext()) {
289 ArgumentDef arg;
290 arg.type = parseType();
291 if (arg.type.name == "void")
292 break;
293 if (test(token: IDENTIFIER))
294 arg.name = lexem();
295 while (test(token: LBRACK)) {
296 arg.rightType += lexemUntil(RBRACK);
297 }
298 if (test(token: CONST) || test(token: VOLATILE)) {
299 arg.rightType += ' ';
300 arg.rightType += lexem();
301 }
302 arg.normalizedType = normalizeType(ba: QByteArray(arg.type.name + ' ' + arg.rightType));
303 arg.typeNameForCast = QByteArray("std::add_pointer_t<"+arg.normalizedType+">");
304 if (test(token: EQ))
305 arg.isDefault = true;
306 def->arguments += arg;
307 if (!until(COMMA))
308 break;
309 }
310
311 if (!def->arguments.isEmpty()
312 && def->arguments.constLast().normalizedType == "QPrivateSignal") {
313 def->arguments.removeLast();
314 def->isPrivateSignal = true;
315 }
316 if (def->arguments.size() == 1
317 && def->arguments.constLast().normalizedType == "QMethodRawArguments") {
318 def->arguments.removeLast();
319 def->isRawSlot = true;
320 }
321}

364QTypeRevision Moc::parseRevision()
365{
366 next(token: LPAREN);
367 QByteArray revisionString = lexemUntil(RPAREN);
368 revisionString.remove(index: 0, len: 1);
369 revisionString.chop(n: 1);
370 const QList<QByteArray> majorMinor = revisionString.split(sep: ',');
371 switch (majorMinor.size()) {
372 case 1: {
373 bool ok = false;
374 const int revision = revisionString.toInt(ok: &ok);
375 if (!ok || !QTypeRevision::isValidSegment(segment: revision))
376 error(msg: "Invalid revision");
377 return QTypeRevision::fromMinorVersion(minorVersion: revision);
378 }
379 case 2: { // major.minor
380 bool ok = false;
381 const int major = majorMinor[0].toInt(ok: &ok);
382 if (!ok || !QTypeRevision::isValidSegment(segment: major))
383 error(msg: "Invalid major version");
384 const int minor = majorMinor[1].toInt(ok: &ok);
385 if (!ok || !QTypeRevision::isValidSegment(segment: minor))
386 error(msg: "Invalid minor version");
387 return QTypeRevision::fromVersion(majorVersion: major, minorVersion: minor);
388 }
389 default:
390 error(msg: "Invalid revision");
391 return QTypeRevision();
392 }
393}

406// returns false if the function should be ignored
407bool Moc::parseFunction(FunctionDef *def, bool inMacro)
408{
409 def->isVirtual = false;
410 def->isStatic = false;
411 //skip modifiers and attributes
412 while (test(token: INLINE) || (test(token: STATIC) && (def->isStatic = true) == true) ||
413 (test(token: VIRTUAL) && (def->isVirtual = true) == true) //mark as virtual
414 || skipCxxAttributes() || testFunctionAttribute(def) || testFunctionRevision(def)) {}
415 bool templateFunction = (lookup() == TEMPLATE);
416 def->type = parseType();
417 if (def->type.name.isEmpty()) {
418 if (templateFunction)
419 error(msg: "Template function as signal or slot");
420 else
421 error();
422 }
423 bool scopedFunctionName = false;
424 if (test(token: LPAREN)) {
425 def->name = def->type.name;
426 scopedFunctionName = def->type.isScoped;
427 def->type = Type("int");
428 } else {
429 Type tempType = parseType();;
430 while (!tempType.name.isEmpty() && lookup() != LPAREN) {
431 if (testFunctionAttribute(tok: def->type.firstToken, def))
432 ; // fine
433 else if (def->type.firstToken == Q_SIGNALS_TOKEN)
434 error();
435 else if (def->type.firstToken == Q_SLOTS_TOKEN)
436 error();
437 else {
438 if (!def->tag.isEmpty())
439 def->tag += ' ';
440 def->tag += def->type.name;
441 }
442 def->type = tempType;
443 tempType = parseType();
444 }
445 next(token: LPAREN, msg: "Not a signal or slot declaration");
446 def->name = tempType.name;
447 scopedFunctionName = tempType.isScoped;
448 }
449
450 if (!test(token: RPAREN)) {
451 parseFunctionArguments(def);
452 next(token: RPAREN);
453 }
454
455 // support optional macros with compiler specific options
456 while (test(token: IDENTIFIER))
457 ;
458
459 def->isConst = test(token: CONST);
460
461 while (test(token: IDENTIFIER))
462 ;
463
464 if (inMacro) {
465 next(token: RPAREN);
466 prev();
467 } else {
468 if (test(token: THROW)) {
469 next(token: LPAREN);
470 until(RPAREN);
471 }
472
473 if (def->type.name == "auto" && test(token: ARROW))
474 def->type = parseType(); // Parse trailing return-type
475
476 if (test(token: SEMIC))
477 ;
478 else if ((def->inlineCode = test(token: LBRACE)))
479 until(RBRACE);
480 else if ((def->isAbstract = test(token: EQ)))
481 until(SEMIC);
482 else if (skipCxxAttributes())
483 until(SEMIC);
484 else
485 error();
486 }
487 if (scopedFunctionName) {
488 const QByteArray msg = "Function declaration " + def->name
489 + " contains extra qualification. Ignoring as signal or slot.";
490 warning(msg.constData());
491 return false;
492 }
493
494 QList<QByteArray> typeNameParts = normalizeType(ba: def->type.name).split(sep: ' ');
495 if (typeNameParts.contains(t: "auto")) {
496 // We expected a trailing return type but we haven't seen one
497 error(msg: "Function declared with auto as return type but missing trailing return type. "
498 "Return type deduction is not supported.");
499 }
500
501 // we don't support references as return types, it's too dangerous
502 if (def->type.referenceType == Type::Reference) {
503 QByteArray rawName = def->type.rawName;
504 def->type = Type("void");
505 def->type.rawName = rawName;
506 }
507
508 def->normalizedType = normalizeType(ba: def->type.name);
509 return true;
510}

512// like parseFunction, but never aborts with an error
513bool Moc::parseMaybeFunction(const ClassDef *cdef, FunctionDef *def)
514{
515 def->isVirtual = false;
516 def->isStatic = false;
517 //skip modifiers and attributes
518 while (test(token: EXPLICIT) || test(token: INLINE) || (test(token: STATIC) && (def->isStatic = true) == true) ||
519 (test(token: VIRTUAL) && (def->isVirtual = true) == true) //mark as virtual
520 || skipCxxAttributes() || testFunctionAttribute(def) || testFunctionRevision(def)) {}
521 bool tilde = test(token: TILDE);
522 def->type = parseType();
523 if (def->type.name.isEmpty())
524 return false;
525 bool scopedFunctionName = false;
526 if (test(token: LPAREN)) {
527 def->name = def->type.name;
528 scopedFunctionName = def->type.isScoped;
529 if (def->name == cdef->classname) {
530 def->isDestructor = tilde;
531 def->isConstructor = !tilde;
532 def->type = Type();
533 } else {
534 def->type = Type("int");
535 }
536 } else {
537 Type tempType = parseType();;
538 while (!tempType.name.isEmpty() && lookup() != LPAREN) {
539 if (testFunctionAttribute(tok: def->type.firstToken, def))
540 ; // fine
541 else if (def->type.name == "Q_SIGNAL")
542 def->isSignal = true;
543 else if (def->type.name == "Q_SLOT")
544 def->isSlot = true;
545 else {
546 if (!def->tag.isEmpty())
547 def->tag += ' ';
548 def->tag += def->type.name;
549 }
550 def->type = tempType;
551 tempType = parseType();
552 }
553 if (!test(token: LPAREN))
554 return false;
555 def->name = tempType.name;
556 scopedFunctionName = tempType.isScoped;
557 }
558
559 // we don't support references as return types, it's too dangerous
560 if (def->type.referenceType == Type::Reference) {
561 QByteArray rawName = def->type.rawName;
562 def->type = Type("void");
563 def->type.rawName = rawName;
564 }
565
566 def->normalizedType = normalizeType(ba: def->type.name);
567
568 if (!test(token: RPAREN)) {
569 parseFunctionArguments(def);
570 if (!test(token: RPAREN))
571 return false;
572 }
573 def->isConst = test(token: CONST);
574 if (scopedFunctionName
575 && (def->isSignal || def->isSlot || def->isInvokable)) {
576 const QByteArray msg = "parsemaybe: Function declaration " + def->name
577 + " contains extra qualification. Ignoring as signal or slot.";
578 warning(msg.constData());
579 return false;
580 }
581 return true;
582}

584inline void handleDefaultArguments(QList<FunctionDef> *functionList, FunctionDef &function)
585{
586 // support a function with a default argument by pretending there is an
587 // overload without the argument (the original function is the overload with
588 // all arguments present)
589 while (function.arguments.size() > 0 && function.arguments.constLast().isDefault) {
590 function.wasCloned = true;
591 function.arguments.removeLast();
592 *functionList += function;
593 }

596void Moc::parse()
597{
598 QList<NamespaceDef> namespaceList;
599 bool templateClass = false;
600 while (hasNext()) {
601 Token t = next();
602 switch (t) {
603 case NAMESPACE: {
604 int rewind = index;
605 if (test(token: IDENTIFIER)) {
606 QByteArray nsName = lexem();
607 QByteArrayList nested;
608 while (test(token: SCOPE)) {
609 /* treat (C++20's) namespace A::inline B {} as A::B
610 this is mostly to not break compilation when encountering such
611 a construct in a header; the interaction of Qt's meta-macros with
612 inline namespaces is still rather poor.
613 */
614 test(token: INLINE);
615 next(token: IDENTIFIER);
616 nested.append(t: nsName);
617 nsName = lexem();
618 }
619 if (test(token: EQ)) {
620 // namespace Foo = Bar::Baz;
621 until(SEMIC);
622 } else if (test(token: LPAREN)) {
623 // Ignore invalid code such as: 'namespace __identifier("x")' (QTBUG-56634)
624 until(RPAREN);
625 } else if (!test(token: SEMIC)) {
626 NamespaceDef def;
627 def.classname = nsName;
628 def.doGenerate = currentFilenames.size() <= 1;
629
630 next(token: LBRACE);
631 def.begin = index - 1;
632 until(RBRACE);
633 def.end = index;
634 index = def.begin + 1;
635
636 for (int i = namespaceList.size() - 1; i >= 0; --i) {
637 if (inNamespace(def: &namespaceList.at(i))) {
638 def.qualified.prepend(a: namespaceList.at(i).classname + "::");
639 }
640 }
641 for (const QByteArray &ns : nested) {
642 NamespaceDef parentNs;
643 parentNs.classname = ns;
644 parentNs.qualified = def.qualified;
645 def.qualified += ns + "::";
646 parentNs.begin = def.begin;
647 parentNs.end = def.end;
648 namespaceList += parentNs;
649 }
650
651 while (inNamespace(def: &def) && hasNext()) {
652 switch (next()) {
653 case NAMESPACE:
654 if (test(token: IDENTIFIER)) {
655 while (test(token: SCOPE))
656 next(token: IDENTIFIER);
657 if (test(token: EQ)) {
658 // namespace Foo = Bar::Baz;
659 until(SEMIC);
660 } else if (!test(token: SEMIC)) {
661 until(RBRACE);
662 }
663 }
664 break;
665 case Q_NAMESPACE_TOKEN:
666 def.hasQNamespace = true;
667 break;
668 case Q_NAMESPACE_EXPORT_TOKEN:
669 next(token: LPAREN);
670 while (test(token: IDENTIFIER))
671 {}
672 next(token: RPAREN);
673 def.hasQNamespace = true;
674 break;
675 case Q_ENUMS_TOKEN:
676 case Q_ENUM_NS_TOKEN:
677 parseEnumOrFlag(def: &def, isFlag: false);
678 break;
679 case Q_ENUM_TOKEN:
680 error(msg: "Q_ENUM can't be used in a Q_NAMESPACE, use Q_ENUM_NS instead");
681 break;
682 case Q_FLAGS_TOKEN:
683 case Q_FLAG_NS_TOKEN:
684 parseEnumOrFlag(def: &def, isFlag: true);
685 break;
686 case Q_FLAG_TOKEN:
687 error(msg: "Q_FLAG can't be used in a Q_NAMESPACE, use Q_FLAG_NS instead");
688 break;
689 case Q_DECLARE_FLAGS_TOKEN:
690 parseFlag(def: &def);
691 break;
692 case Q_CLASSINFO_TOKEN:
693 parseClassInfo(def: &def);
694 break;
695 case Q_MOC_INCLUDE_TOKEN:
696 // skip it, the namespace is parsed twice
697 next(token: LPAREN);
698 lexemUntil(RPAREN);
699 break;
700 case ENUM: {
701 EnumDef enumDef;
702 if (parseEnum(def: &enumDef))
703 def.enumList += enumDef;
704 } break;
705 case CLASS:
706 case STRUCT: {
707 ClassDef classdef;
708 if (!parseClassHead(def: &classdef))
709 continue;
710 while (inClass(def: &classdef) && hasNext())
711 next(); // consume all Q_XXXX macros from this class
712 } break;
713 default: break;
714 }
715 }
716 namespaceList += def;
717 index = rewind;
718 if (!def.hasQNamespace && (!def.classInfoList.isEmpty() || !def.enumDeclarations.isEmpty()))
719 error(msg: "Namespace declaration lacks Q_NAMESPACE macro.");
720 }
721 }
722 break;
723 }
724 case SEMIC:
725 case RBRACE:
726 templateClass = false;
727 break;
728 case TEMPLATE:
729 templateClass = true;
730 break;
731 case MOC_INCLUDE_BEGIN:
732 currentFilenames.push(x: symbol().unquotedLexem());
733 break;
734 case MOC_INCLUDE_END:
735 currentFilenames.pop();
736 break;
737 case Q_DECLARE_INTERFACE_TOKEN:
738 parseDeclareInterface();
739 break;
740 case Q_DECLARE_METATYPE_TOKEN:
741 parseDeclareMetatype();
742 break;
743 case Q_MOC_INCLUDE_TOKEN:
744 parseMocInclude();
745 break;
746 case USING:
747 if (test(token: NAMESPACE)) {
748 while (test(token: SCOPE) || test(token: IDENTIFIER))
749 ;
750 // Ignore invalid code such as: 'using namespace __identifier("x")' (QTBUG-63772)
751 if (test(token: LPAREN))
752 until(RPAREN);
753 next(token: SEMIC);
754 }
755 break;
756 case CLASS:
757 case STRUCT: {

969 }

1283void Moc::parsePropertyAttributes(PropertyDef &propDef)
1284{
1285 auto checkIsFunction = [&](const QByteArray &def, const char *name) {
1286 if (def.endsWith(c: ')')) {
1287 QByteArray msg = "Providing a function for ";
1288 msg += name;
1289 msg += " in a property declaration is not be supported in Qt 6.";
1290 error(msg: msg.constData());
1291 }
1292 };
1293
1294 while (test(token: IDENTIFIER)) {
1295 const QByteArray l = lexem();
1296 if (l[0] == 'C' && l == "CONSTANT") {
1297 propDef.constant = true;
1298 continue;
1299 } else if (l[0] == 'F' && l == "FINAL") {
1300 propDef.final = true;
1301 continue;
1302 } else if (l[0] == 'N' && l == "NAME") {
1303 next(token: IDENTIFIER);
1304 propDef.name = lexem();
1305 continue;
1306 } else if (l[0] == 'R' && l == "REQUIRED") {
1307 propDef.required = true;
1308 continue;
1309 } else if (l[0] == 'R' && l == "REVISION" && test(token: LPAREN)) {
1310 prev();
1311 propDef.revision = parseRevision().toEncodedVersion<int>();
1312 continue;
1313 }
1314
1315 QByteArray v, v2;
1316 if (test(token: LPAREN)) {
1317 v = lexemUntil(RPAREN);
1318 v = v.mid(index: 1, len: v.size() - 2); // removes the '(' and ')'
1319 } else if (test(token: INTEGER_LITERAL)) {
1320 v = lexem();
1321 if (l != "REVISION")
1322 error(rollback: 1);
1323 } else if (test(token: DEFAULT)) {
1324 v = lexem();
1325 if (l != "READ" && l != "WRITE")
1326 error(rollback: 1);
1327 } else {
1328 next(token: IDENTIFIER);
1329 v = lexem();
1330 if (test(token: LPAREN))
1331 v2 = lexemUntil(RPAREN);
1332 else if (v != "true" && v != "false")
1333 v2 = "()";
1334 }
1335 switch (l[0]) {
1336 case 'M':
1337 if (l == "MEMBER")
1338 propDef.member = v;
1339 else
1340 error(rollback: 2);
1341 break;
1342 case 'R':
1343 if (l == "READ")
1344 propDef.read = v;
1345 else if (l == "RESET")
1346 propDef.reset = v;
1347 else if (l == "REVISION") {
1348 bool ok = false;
1349 const int minor = v.toInt(ok: &ok);
1350 if (!ok || !QTypeRevision::isValidSegment(segment: minor))
1351 error(rollback: 1);
1352 propDef.revision = QTypeRevision::fromMinorVersion(minorVersion: minor).toEncodedVersion<int>();
1353 } else
1354 error(rollback: 2);
1355 break;
1356 case 'S':
1357 if (l == "SCRIPTABLE") {
1358 propDef.scriptable = v + v2;
1359 checkIsFunction(propDef.scriptable, "SCRIPTABLE");
1360 } else if (l == "STORED") {
1361 propDef.stored = v + v2;
1362 checkIsFunction(propDef.stored, "STORED");
1363 } else
1364 error(rollback: 2);
1365 break;
1366 case 'W': if (l != "WRITE") error(rollback: 2);
1367 propDef.write = v;
1368 break;
1369 case 'B': if (l != "BINDABLE") error(rollback: 2);
1370 propDef.bind = v;
1371 break;
1372 case 'D': if (l != "DESIGNABLE") error(rollback: 2);
1373 propDef.designable = v + v2;
1374 checkIsFunction(propDef.designable, "DESIGNABLE");
1375 break;
1376 case 'N': if (l != "NOTIFY") error(rollback: 2);
1377 propDef.notify = v;
1378 break;
1379 case 'U': if (l != "USER") error(rollback: 2);
1380 propDef.user = v + v2;
1381 checkIsFunction(propDef.user, "USER");
1382 break;
1383 default:
1384 error(rollback: 2);
1385 }
1386 }
1387 if (propDef.constant && !propDef.write.isNull()) {
1388 const QByteArray msg = "Property declaration " + propDef.name
1389 + " is both WRITEable and CONSTANT. CONSTANT will be ignored.";
1390 propDef.constant = false;
1391 warning(msg.constData());
1392 }
1393 if (propDef.constant && !propDef.notify.isNull()) {

1710bool Moc::until(Token target) {
1711 int braceCount = 0;
1712 int brackCount = 0;
1713 int parenCount = 0;
1714 int angleCount = 0;
1715 if (index) {
1716 switch(symbols.at(i: index-1).token) {
1717 case LBRACE: ++braceCount; break;
1718 case LBRACK: ++brackCount; break;
1719 case LPAREN: ++parenCount; break;
1720 case LANGLE: ++angleCount; break;
1721 default: break;
1722 }
1723 }
1724
1725 //when searching commas within the default argument, we should take care of template depth (anglecount)
1726 // unfortunately, we do not have enough semantic information to know if '<' is the operator< or
1727 // the beginning of a template type. so we just use heuristics.
1728 int possible = -1;
1729
1730 while (index < symbols.size()) {
1731 Token t = symbols.at(i: index++).token;
1732 switch (t) {
1733 case LBRACE: ++braceCount; break;
1734 case RBRACE: --braceCount; break;
1735 case LBRACK: ++brackCount; break;
1736 case RBRACK: --brackCount; break;
1737 case LPAREN: ++parenCount; break;
1738 case RPAREN: --parenCount; break;
1739 case LANGLE:
1740 if (parenCount == 0 && braceCount == 0)
1741 ++angleCount;
1742 break;
1743 case RANGLE:
1744 if (parenCount == 0 && braceCount == 0)
1745 --angleCount;
1746 break;
1747 case GTGT:
1748 if (parenCount == 0 && braceCount == 0) {
1749 angleCount -= 2;
1750 t = RANGLE;
1751 }
1752 break;
1753 default: break;
1754 }
1755 if (t == target
1756 && braceCount <= 0
1757 && brackCount <= 0
1758 && parenCount <= 0
1759 && (target != RANGLE || angleCount <= 0)) {
1760 if (target != COMMA || angleCount <= 0)
1761 return true;
1762 possible = index;
1763 }
1764
1765 if (target == COMMA && t == EQ && possible != -1) {
1766 index = possible;
1767 return true;
1768 }
1769
1770 if (braceCount < 0 || brackCount < 0 || parenCount < 0
1771 || (target == RANGLE && angleCount < 0)) {
1772 --index;
1773 break;
1774 }
1775
1776 if (braceCount <= 0 && t == SEMIC) {
1777 // Abort on semicolon. Allow recovering bad template parsing (QTBUG-31218)
1778 break;
1779 }
1780 }
1781
1782 if (target == COMMA && angleCount != 0 && possible != -1) {
1783 index = possible;
1784 return true;
1785 }
1786
1787 return false;
1788}
