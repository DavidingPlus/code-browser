18// transform \r\n into \n
19// \r into \n (os9 style)
20// backslash-newlines into newlines
21static QByteArray cleaned(const QByteArray &input)
22{
23 QByteArray result;
24 result.resize(size: input.size());
25 const char *data = input.constData();
26 const char *end = input.constData() + input.size();
27 char *output = result.data();
28
29 int newlines = 0;
30 while (data != end) {
31 while (data != end && is_space(s: *data))
32 ++data;
33 bool takeLine = (*data == '#');
34 if (*data == '%' && *(data+1) == ':') {
35 takeLine = true;
36 ++data;
37 }
38 if (takeLine) {
39 *output = '#';
40 ++output;
41 do ++data; while (data != end && is_space(s: *data));
42 }
43 while (data != end) {
44 // handle \\\n, \\\r\n and \\\r
45 if (*data == '\\') {
46 if (*(data + 1) == '\r') {
47 ++data;
48 }
49 if (data != end && (*(data + 1) == '\n' || (*data) == '\r')) {
50 ++newlines;
51 data += 1;
52 if (data != end && *data != '\r')
53 data += 1;
54 continue;
55 }
56 } else if (*data == '\r' && *(data + 1) == '\n') { // reduce \r\n to \n
57 ++data;
58 }
59 if (data == end)
60 break;
61
62 char ch = *data;
63 if (ch == '\r') // os9: replace \r with \n
64 ch = '\n';
65 *output = ch;
66 ++output;
67
68 if (*data == '\n') {
69 // output additional newlines to keep the correct line-numbering
70 // for the lines following the backslash-newline sequence(s)
71 while (newlines) {
72 *output = '\n';
73 ++output;
74 --newlines;
75 }
76 ++data;
77 break;
78 }
79 ++data;
80 }
81 }
82 result.resize(size: output - result.constData());
83 return result;
84}

104bool Preprocessor::skipBranch()
105{
106 while (index < symbols.size() - 1
107 && (symbols.at(i: index).token != PP_ENDIF
108 && symbols.at(i: index).token != PP_ELIF
109 && symbols.at(i: index).token != PP_ELSE)
110 ){
111 switch (symbols.at(i: index).token) {
112 case PP_IF:
113 case PP_IFDEF:
114 case PP_IFNDEF:
115 ++index;
116 skipUntilEndif();
117 break;
118 default:
119 ;
120 }
121 ++index;
122 }

127Symbols Preprocessor::tokenize(const QByteArray& input, int lineNum, Preprocessor::TokenizeMode mode)
128{
129 Symbols symbols;
130 // Preallocate some space to speed up the code below.
131 // The magic divisor value was found by calculating the average ratio between
132 // input size and the final size of symbols.
133 // This yielded a value of 16.x when compiling Qt Base.
134 symbols.reserve(size: input.size() / 16);
135 const char *begin = input.constData();
136 const char *data = begin;
137 while (*data) {
138 if (mode == TokenizeCpp || mode == TokenizeDefine) {
139 int column = 0;
140
141 const char *lexem = data;
142 int state = 0;
143 Token token = NOTOKEN;
144 for (;;) {
145 if (static_cast<signed char>(*data) < 0) {
146 ++data;
147 continue;
148 }
149 int nextindex = keywords[state].next;
150 int next = 0;
151 if (*data == keywords[state].defchar)
152 next = keywords[state].defnext;
153 else if (!state || nextindex)
154 next = keyword_trans[nextindex][(int)*data];
155 if (!next)
156 break;
157 state = next;
158 token = keywords[state].token;
159 ++data;
160 }
161
162 // suboptimal, is_ident_char should use a table
163 if (keywords[state].ident && is_ident_char(s: *data))
164 token = keywords[state].ident;
165
166 if (token == NOTOKEN) {
167 if (*data)
168 ++data;
169 // an error really, but let's ignore this input
170 // to not confuse moc later. However in pre-processor
171 // only mode let's continue.
172 if (!Preprocessor::preprocessOnly)
173 continue;
174 }
175
176 ++column;
177
178 if (token > SPECIAL_TREATMENT_MARK) {
179 switch (token) {
180 case QUOTE:
181 data = skipQuote(data);
182 token = STRING_LITERAL;
183 // concatenate multi-line strings for easier
184 // STRING_LITERAL handling in moc
185 if (!Preprocessor::preprocessOnly
186 && !symbols.isEmpty()
187 && symbols.constLast().token == STRING_LITERAL) {
188
189 const QByteArray newString
190 = '\"'
191 + symbols.constLast().unquotedLexem()
192 + input.mid(index: lexem - begin + 1, len: data - lexem - 2)
193 + '\"';
194 symbols.last() = Symbol(symbols.constLast().lineNum,
195 STRING_LITERAL,
196 newString);
197 continue;
198 }
199 break;
200 case SINGLEQUOTE:
201 while (*data && (*data != '\''
202 || (*(data-1)=='\\'
203 && *(data-2)!='\\')))
204 ++data;
205 if (*data)
206 ++data;
207 token = CHARACTER_LITERAL;
208 break;
209 case LANGLE_SCOPE:
210 // split <:: into two tokens, < and ::
211 token = LANGLE;
212 data -= 2;
213 break;
214 case DIGIT:
215 while (is_digit_char(s: *data) || *data == '\'')
216 ++data;
217 if (!*data || *data != '.') {
218 token = INTEGER_LITERAL;
219 if (data - lexem == 1 &&
220 (*data == 'x' || *data == 'X'
221 || *data == 'b' || *data == 'B')
222 && *lexem == '0') {
223 ++data;
224 while (is_hex_char(s: *data) || *data == '\'')
225 ++data;
226 }
227 break;
228 }
229 token = FLOATING_LITERAL;
230 ++data;
231 Q_FALLTHROUGH();
232 case FLOATING_LITERAL:
233 while (is_digit_char(s: *data) || *data == '\'')
234 ++data;
235 if (*data == '+' || *data == '-')
236 ++data;
237 if (*data == 'e' || *data == 'E') {
238 ++data;
239 while (is_digit_char(s: *data) || *data == '\'')
240 ++data;
241 }
242 if (*data == 'f' || *data == 'F'
243 || *data == 'l' || *data == 'L')
244 ++data;
245 break;
246 case HASH:
247 if (column == 1 && mode == TokenizeCpp) {
248 mode = PreparePreprocessorStatement;
249 while (*data && (*data == ' ' || *data == '\t'))
250 ++data;
251 if (is_ident_char(s: *data))
252 mode = TokenizePreprocessorStatement;
253 continue;
254 }
255 break;
256 case PP_HASHHASH:
257 if (mode == TokenizeCpp)
258 continue;
259 break;
260 case NEWLINE:
261 ++lineNum;
262 if (mode == TokenizeDefine) {
263 mode = TokenizeCpp;
264 // emit the newline token
265 break;
266 }
267 continue;
268 case BACKSLASH:
269 {
270 const char *rewind = data;
271 while (*data && (*data == ' ' || *data == '\t'))
272 ++data;
273 if (*data && *data == '\n') {
274 ++data;
275 continue;
276 }
277 data = rewind;
278 } break;
279 case CHARACTER:
280 while (is_ident_char(s: *data))
281 ++data;
282 token = IDENTIFIER;
283 break;
284 case C_COMMENT:
285 if (*data) {
286 if (*data == '\n')
287 ++lineNum;
288 ++data;
289 if (*data) {
290 if (*data == '\n')
291 ++lineNum;
292 ++data;
293 }
294 }
295 while (*data && (*(data-1) != '/' || *(data-2) != '*')) {
296 if (*data == '\n')
297 ++lineNum;
298 ++data;
299 }
300 token = WHITESPACE; // one comment, one whitespace
301 Q_FALLTHROUGH();
302 case WHITESPACE:
303 if (column == 1)
304 column = 0;
305 while (*data && (*data == ' ' || *data == '\t'))
306 ++data;
307 if (Preprocessor::preprocessOnly) // tokenize whitespace
308 break;
309 continue;
310 case CPP_COMMENT:
311 while (*data && *data != '\n')
312 ++data;
313 continue; // ignore safely, the newline is a separator
314 default:
315 continue; //ignore
316 }
317 }
318#ifdef USE_LEXEM_STORE
319 if (!Preprocessor::preprocessOnly
320 && token != IDENTIFIER
321 && token != STRING_LITERAL
322 && token != FLOATING_LITERAL
323 && token != INTEGER_LITERAL)
324 symbols += Symbol(lineNum, token);
325 else
326#endif
327 symbols += Symbol(lineNum, token, input, lexem-begin, data-lexem);
328
329 } else { // Preprocessor
330
331 const char *lexem = data;
332 int state = 0;
333 Token token = NOTOKEN;
334 if (mode == TokenizePreprocessorStatement) {
335 state = pp_keyword_trans[0][(int)'#'];
336 mode = TokenizePreprocessor;
337 }
338 for (;;) {
339 if (static_cast<signed char>(*data) < 0) {
340 ++data;
341 continue;
342 }
343 int nextindex = pp_keywords[state].next;
344 int next = 0;
345 if (*data == pp_keywords[state].defchar)
346 next = pp_keywords[state].defnext;
347 else if (!state || nextindex)
348 next = pp_keyword_trans[nextindex][(int)*data];
349 if (!next)
350 break;
351 state = next;
352 token = pp_keywords[state].token;
353 ++data;
354 }
355 // suboptimal, is_ident_char should use a table
356 if (pp_keywords[state].ident && is_ident_char(s: *data))
357 token = pp_keywords[state].ident;
358
359 switch (token) {
360 case NOTOKEN:
361 if (*data)
362 ++data;
363 break;
364 case PP_DEFINE:
365 mode = PrepareDefine;
366 break;
367 case PP_IFDEF:
368 symbols += Symbol(lineNum, PP_IF);
369 symbols += Symbol(lineNum, PP_DEFINED);
370 continue;
371 case PP_IFNDEF:
372 symbols += Symbol(lineNum, PP_IF);
373 symbols += Symbol(lineNum, PP_NOT);
374 symbols += Symbol(lineNum, PP_DEFINED);
375 continue;
376 case PP_INCLUDE:
377 mode = TokenizeInclude;
378 break;
379 case PP_QUOTE:
380 data = skipQuote(data);
381 token = PP_STRING_LITERAL;
382 break;
383 case PP_SINGLEQUOTE:
384 while (*data && (*data != '\''
385 || (*(data-1)=='\\'
386 && *(data-2)!='\\')))
387 ++data;
388 if (*data)
389 ++data;
390 token = PP_CHARACTER_LITERAL;
391 break;
392 case PP_DIGIT:
393 while (is_digit_char(s: *data) || *data == '\'')
394 ++data;
395 if (!*data || *data != '.') {
396 token = PP_INTEGER_LITERAL;
397 if (data - lexem == 1 &&
398 (*data == 'x' || *data == 'X')
399 && *lexem == '0') {
400 ++data;
401 while (is_hex_char(s: *data) || *data == '\'')
402 ++data;
403 }
404 break;
405 }
406 token = PP_FLOATING_LITERAL;
407 ++data;
408 Q_FALLTHROUGH();
409 case PP_FLOATING_LITERAL:
410 while (is_digit_char(s: *data) || *data == '\'')
411 ++data;
412 if (*data == '+' || *data == '-')
413 ++data;
414 if (*data == 'e' || *data == 'E') {
415 ++data;
416 while (is_digit_char(s: *data) || *data == '\'')
417 ++data;
418 }
419 if (*data == 'f' || *data == 'F'
420 || *data == 'l' || *data == 'L')
421 ++data;
422 break;
423 case PP_CHARACTER:
424 if (mode == PreparePreprocessorStatement) {
425 // rewind entire token to begin
426 data = lexem;
427 mode = TokenizePreprocessorStatement;
428 continue;
429 }
430 while (is_ident_char(s: *data))
431 ++data;
432 token = PP_IDENTIFIER;
433
434 if (mode == PrepareDefine) {
435 symbols += Symbol(lineNum, token, input, lexem-begin, data-lexem);
436 // make sure we explicitly add the whitespace here if the next char
437 // is not an opening brace, so we can distinguish correctly between
438 // regular and function macros
439 if (*data != '(')
440 symbols += Symbol(lineNum, WHITESPACE);
441 mode = TokenizeDefine;
442 continue;
443 }
444 break;
445 case PP_C_COMMENT:
446 if (*data) {
447 if (*data == '\n')
448 ++lineNum;
449 ++data;
450 if (*data) {
451 if (*data == '\n')
452 ++lineNum;
453 ++data;
454 }
455 }
456 while (*data && (*(data-1) != '/' || *(data-2) != '*')) {
457 if (*data == '\n')
458 ++lineNum;
459 ++data;
460 }
461 token = PP_WHITESPACE; // one comment, one whitespace
462 Q_FALLTHROUGH();
463 case PP_WHITESPACE:
464 while (*data && (*data == ' ' || *data == '\t'))
465 ++data;
466 continue; // the preprocessor needs no whitespace
467 case PP_CPP_COMMENT:
468 while (*data && *data != '\n')
469 ++data;
470 continue; // ignore safely, the newline is a separator
471 case PP_NEWLINE:
472 ++lineNum;
473 mode = TokenizeCpp;
474 break;
475 case PP_BACKSLASH:
476 {
477 const char *rewind = data;
478 while (*data && (*data == ' ' || *data == '\t'))
479 ++data;
480 if (*data && *data == '\n') {
481 ++data;
482 continue;
483 }
484 data = rewind;
485 } break;
486 case PP_LANGLE:
487 if (mode != TokenizeInclude)
488 break;
489 token = PP_STRING_LITERAL;
490 while (*data && *data != '\n' && *(data-1) != '>')
491 ++data;
492 break;
493 default:
494 break;
495 }
496 if (mode == PreparePreprocessorStatement)
497 continue;
498#ifdef USE_LEXEM_STORE
499 if (token != PP_IDENTIFIER
500 && token != PP_STRING_LITERAL
501 && token != PP_FLOATING_LITERAL
502 && token != PP_INTEGER_LITERAL)
503 symbols += Symbol(lineNum, token);
504 else
505#endif
506 symbols += Symbol(lineNum, token, input, lexem-begin, data-lexem);
507 }
508 }
509 symbols += Symbol(); // eof symbol
510 return symbols;
511}

513void Preprocessor::macroExpand(Symbols *into, Preprocessor *that, const Symbols &toExpand, int &index,
514 int lineNum, bool one, const QSet<QByteArray> &excludeSymbols)
515{
516 SymbolStack symbols;
517 SafeSymbols sf;
518 sf.symbols = toExpand;
519 sf.index = index;
520 sf.excludedSymbols = excludeSymbols;
521 symbols.push(t: sf);
522
523 if (toExpand.isEmpty())
524 return;
525
526 for (;;) {
527 QByteArray macro;
528 Symbols newSyms = macroExpandIdentifier(that, symbols, lineNum, macroName: &macro);
529
530 if (macro.isEmpty()) {
531 // not a macro
532 Symbol s = symbols.symbol();
533 s.lineNum = lineNum;
534 *into += s;
535 } else {
536 SafeSymbols sf;
537 sf.symbols = newSyms;
538 sf.index = 0;
539 sf.expandedMacro = macro;
540 symbols.push(t: sf);
541 }
542 if (!symbols.hasNext() || (one && symbols.size() == 1))
543 break;
544 symbols.next();
545 }
546
547 if (symbols.size())
548 index = symbols.top().index;
549 else
550 index = toExpand.size();
551}

554Symbols Preprocessor::macroExpandIdentifier(Preprocessor *that, SymbolStack &symbols, int lineNum, QByteArray *macroName)
555{
556 Symbol s = symbols.symbol();
557
558 // not a macro
559 if (s.token != PP_IDENTIFIER || !that->macros.contains(key: s) || symbols.dontReplaceSymbol(name: s.lexem())) {
560 return Symbols();
561 }
562
563 const Macro &macro = that->macros.value(key: s);
564 *macroName = s.lexem();
565
566 Symbols expansion;
567 if (!macro.isFunction) {
568 expansion = macro.symbols;
569 } else {
570 bool haveSpace = false;
571 while (symbols.test(token: PP_WHITESPACE)) { haveSpace = true; }
572 if (!symbols.test(token: PP_LPAREN)) {
573 *macroName = QByteArray();
574 Symbols syms;
575 if (haveSpace)
576 syms += Symbol(lineNum, PP_WHITESPACE);
577 syms += s;
578 syms.last().lineNum = lineNum;
579 return syms;
580 }
581 QVarLengthArray<Symbols, 5> arguments;
582 while (symbols.hasNext()) {
583 Symbols argument;
584 // strip leading space
585 while (symbols.test(token: PP_WHITESPACE)) {}
586 int nesting = 0;
587 bool vararg = macro.isVariadic && (arguments.size() == macro.arguments.size() - 1);
588 while (symbols.hasNext()) {
589 Token t = symbols.next();
590 if (t == PP_LPAREN) {
591 ++nesting;
592 } else if (t == PP_RPAREN) {
593 --nesting;
594 if (nesting < 0)
595 break;
596 } else if (t == PP_COMMA && nesting == 0) {
597 if (!vararg)
598 break;
599 }
600 argument += symbols.symbol();
601 }
602 arguments += argument;
603
604 if (nesting < 0)
605 break;
606 else if (!symbols.hasNext())
607 that->error(msg: "missing ')' in macro usage");
608 }
609
610 // empty VA_ARGS
611 if (macro.isVariadic && arguments.size() == macro.arguments.size() - 1)
612 arguments += Symbols();
613
614 // now replace the macro arguments with the expanded arguments
615 enum Mode {
616 Normal,
617 Hash,
618 HashHash
619 } mode = Normal;
620
621 for (int i = 0; i < macro.symbols.size(); ++i) {
622 const Symbol &s = macro.symbols.at(i);
623 if (s.token == HASH || s.token == PP_HASHHASH) {
624 mode = (s.token == HASH ? Hash : HashHash);
625 continue;
626 }
627 int index = macro.arguments.indexOf(t: s);
628 if (mode == Normal) {
629 if (index >= 0 && index < arguments.size()) {
630 // each argument undoergoes macro expansion if it's not used as part of a # or ##
631 if (i == macro.symbols.size() - 1 || macro.symbols.at(i: i + 1).token != PP_HASHHASH) {
632 Symbols arg = arguments.at(idx: index);
633 int idx = 1;
634 macroExpand(into: &expansion, that, toExpand: arg, index&: idx, lineNum, one: false, excludeSymbols: symbols.excludeSymbols());
635 } else {
636 expansion += arguments.at(idx: index);
637 }
638 } else {
639 expansion += s;
640 }
641 } else if (mode == Hash) {
642 if (index < 0) {
643 that->error(msg: "'#' is not followed by a macro parameter");
644 continue;
645 } else if (index >= arguments.size()) {
646 that->error(msg: "Macro invoked with too few parameters for a use of '#'");
647 continue;
648 }
649
650 const Symbols &arg = arguments.at(idx: index);
651 QByteArray stringified;
652 for (int i = 0; i < arg.size(); ++i) {
653 stringified += arg.at(i).lexem();
654 }
655 stringified.replace(before: '"', after: "\\\"");
656 stringified.prepend(c: '"');
657 stringified.append(c: '"');
658 expansion += Symbol(lineNum, STRING_LITERAL, stringified);
659 } else if (mode == HashHash){
660 if (s.token == WHITESPACE)
661 continue;
662
663 while (expansion.size() && expansion.constLast().token == PP_WHITESPACE)
664 expansion.pop_back();
665
666 Symbol next = s;
667 if (index >= 0 && index < arguments.size()) {
668 const Symbols &arg = arguments.at(idx: index);
669 if (arg.size() == 0) {
670 mode = Normal;
671 continue;
672 }
673 next = arg.at(i: 0);
674 }
675
676 if (!expansion.isEmpty() && expansion.constLast().token == s.token
677 && expansion.constLast().token != STRING_LITERAL) {
678 Symbol last = expansion.takeLast();
679
680 QByteArray lexem = last.lexem() + next.lexem();
681 expansion += Symbol(lineNum, last.token, lexem);
682 } else {
683 expansion += next;
684 }
685
686 if (index >= 0 && index < arguments.size()) {
687 const Symbols &arg = arguments.at(idx: index);
688 for (int i = 1; i < arg.size(); ++i)
689 expansion += arg.at(i);
690 }
691 }
692 mode = Normal;
693 }
694 if (mode != Normal)
695 that->error(msg: "'#' or '##' found at the end of a macro argument");
696
697 }
698
699 return expansion;
700}

702void Preprocessor::substituteUntilNewline(Symbols &substituted)
703{
704 while (hasNext()) {
705 Token token = next();
706 if (token == PP_IDENTIFIER) {
707 macroExpand(into: &substituted, that: this, toExpand: symbols, index, lineNum: symbol().lineNum, one: true);
708 } else if (token == PP_DEFINED) {
709 bool braces = test(token: PP_LPAREN);
710 next(token: PP_IDENTIFIER);
711 Symbol definedOrNotDefined = symbol();
712 definedOrNotDefined.token = macros.contains(key: definedOrNotDefined)? PP_MOC_TRUE : PP_MOC_FALSE;
713 substituted += definedOrNotDefined;
714 if (braces)
715 test(token: PP_RPAREN);
716 continue;
717 } else if (token == PP_NEWLINE) {
718 substituted += symbol();
719 break;
720 } else {
721 substituted += symbol();
722 }
723 }
724}

728{
729public:
730 int value() { index = 0; return unary_expression_lookup() ? conditional_expression() : 0; }

732 int conditional_expression();
733 int logical_OR_expression();
734 int logical_AND_expression();
735 int inclusive_OR_expression();
736 int exclusive_OR_expression();
737 int AND_expression();
738 int equality_expression();
739 int relational_expression();
740 int shift_expression();
741 int additive_expression();
742 int multiplicative_expression();
743 int unary_expression();
744 bool unary_expression_lookup();
745 int primary_expression();
746 bool primary_expression_lookup();
747};

749int PP_Expression::conditional_expression()
750{
751 int value = logical_OR_expression();
752 if (test(token: PP_QUESTION)) {
753 int alt1 = conditional_expression();
754 int alt2 = test(token: PP_COLON) ? conditional_expression() : 0;
755 return value ? alt1 : alt2;
756 }
757 return value;
758}
759
760int PP_Expression::logical_OR_expression()
761{
762 int value = logical_AND_expression();
763 if (test(token: PP_OROR))
764 return logical_OR_expression() || value;
765 return value;
766}
767
768int PP_Expression::logical_AND_expression()
769{
770 int value = inclusive_OR_expression();
771 if (test(token: PP_ANDAND))
772 return logical_AND_expression() && value;
773 return value;
774}
775
776int PP_Expression::inclusive_OR_expression()
777{
778 int value = exclusive_OR_expression();
779 if (test(token: PP_OR))
780 return value | inclusive_OR_expression();
781 return value;
782}
783
784int PP_Expression::exclusive_OR_expression()
785{
786 int value = AND_expression();
787 if (test(token: PP_HAT))
788 return value ^ exclusive_OR_expression();
789 return value;
790}
791
792int PP_Expression::AND_expression()
793{
794 int value = equality_expression();
795 if (test(token: PP_AND))
796 return value & AND_expression();
797 return value;
798}
799
800int PP_Expression::equality_expression()
801{
802 int value = relational_expression();
803 switch (next()) {
804 case PP_EQEQ:
805 return value == equality_expression();
806 case PP_NE:
807 return value != equality_expression();
808 default:
809 prev();
810 return value;
811 }

860int PP_Expression::multiplicative_expression()
861{
862 int value = unary_expression();
863 switch (next()) {
864 case PP_STAR:
865 {
866 // get well behaved overflow behavior by converting to long
867 // and then back to int
868 // NOTE: A conformant preprocessor would need to work intmax_t/
869 // uintmax_t according to [cpp.cond], 19.1 10
870 // But we're not compliant anyway
871 qint64 result = qint64(value) * qint64(multiplicative_expression());
872 return int(result);
873 }
874 case PP_PERCENT:
875 {
876 int remainder = multiplicative_expression();
877 return remainder ? value % remainder : 0;
878 }
879 case PP_SLASH:
880 {
881 int div = multiplicative_expression();
882 return div ? value / div : 0;
883 }
884 default:
885 prev();
886 return value;
887 };

963static void mergeStringLiterals(Symbols *_symbols)
964{
965 Symbols &symbols = *_symbols;
966 for (Symbols::iterator i = symbols.begin(); i != symbols.end(); ++i) {
967 if (i->token == STRING_LITERAL) {
968 Symbols::Iterator mergeSymbol = i;
969 int literalsLength = mergeSymbol->len;
970 while (++i != symbols.end() && i->token == STRING_LITERAL)
971 literalsLength += i->len - 2; // no quotes
972
973 if (literalsLength != mergeSymbol->len) {
974 QByteArray mergeSymbolOriginalLexem = mergeSymbol->unquotedLexem();
975 QByteArray &mergeSymbolLexem = mergeSymbol->lex;
976 mergeSymbolLexem.resize(size: 0);
977 mergeSymbolLexem.reserve(asize: literalsLength);
978 mergeSymbolLexem.append(c: '"');
979 mergeSymbolLexem.append(a: mergeSymbolOriginalLexem);
980 for (Symbols::iterator j = mergeSymbol + 1; j != i; ++j)
981 mergeSymbolLexem.append(s: j->lex.constData() + j->from + 1, len: j->len - 2); // append j->unquotedLexem()
982 mergeSymbolLexem.append(c: '"');
983 mergeSymbol->len = mergeSymbol->lex.size();
984 mergeSymbol->from = 0;
985 i = symbols.erase(begin: mergeSymbol + 1, end: i);
986 }
987 if (i == symbols.end())
988 break;
989 }
990 }

993static QByteArray searchIncludePaths(const QList<Parser::IncludePath> &includepaths,
994 const QByteArray &include,
995 const bool debugIncludes)
996{
997 QFileInfo fi;
998
999 if (Q_UNLIKELY(debugIncludes)) {
1000 fprintf(stderr, format: "debug-includes: searching for '%s'\n", include.constData());
1001 }
1002
1003 for (int j = 0; j < includepaths.size() && !fi.exists(); ++j) {
1004 const Parser::IncludePath &p = includepaths.at(i: j);
1005 if (p.isFrameworkPath) {
1006 const int slashPos = include.indexOf(c: '/');
1007 if (slashPos == -1)
1008 continue;
1009 fi.setFile(dir: QString::fromLocal8Bit(ba: p.path + '/' + include.left(len: slashPos) + ".framework/Headers/"),
1010 file: QString::fromLocal8Bit(ba: include.mid(index: slashPos + 1)));
1011 } else {
1012 fi.setFile(dir: QString::fromLocal8Bit(ba: p.path), file: QString::fromLocal8Bit(ba: include));
1013 }
1014
1015 if (Q_UNLIKELY(debugIncludes)) {
1016 const auto candidate = fi.filePath().toLocal8Bit();
1017 fprintf(stderr, format: "debug-includes: considering '%s'\n", candidate.constData());
1018 }
1019
1020 // try again, maybe there's a file later in the include paths with the same name
1021 // (186067)
1022 if (fi.isDir()) {
1023 fi = QFileInfo();
1024 continue;
1025 }
1026 }
1027
1028 if (!fi.exists() || fi.isDir()) {
1029 if (Q_UNLIKELY(debugIncludes)) {
1030 fprintf(stderr, format: "debug-includes: can't find '%s'\n", include.constData());
1031 }
1032 return QByteArray();
1033 }
1034
1035 const auto result = fi.canonicalFilePath().toLocal8Bit();
1036
1037 if (Q_UNLIKELY(debugIncludes)) {
1038 fprintf(stderr, format: "debug-includes: found '%s'\n", result.constData());
1039 }
1040
1041 return result;
1042}

1063void Preprocessor::preprocess(const QByteArray &filename, Symbols &preprocessed)
1064{
1065 currentFilenames.push(x: filename);
1066 preprocessed.reserve(size: preprocessed.size() + symbols.size());
1067 while (hasNext()) {
1068 Token token = next();
1069
1070 switch (token) {
1071 case PP_INCLUDE:
1072 {
1073 int lineNum = symbol().lineNum;
1074 QByteArray include;
1075 bool local = false;
1076 if (test(token: PP_STRING_LITERAL)) {
1077 local = lexem().startsWith(c: '\"');
1078 include = unquotedLexem();
1079 } else
1080 continue;
1081 until(PP_NEWLINE);
1082
1083 include = resolveInclude(include, relativeTo: local ? filename : QByteArray());
1084 if (include.isNull())
1085 continue;
1086
1087 if (Preprocessor::preprocessedIncludes.contains(value: include))
1088 continue;
1089 Preprocessor::preprocessedIncludes.insert(value: include);
1090
1091 QFile file(QString::fromLocal8Bit(ba: include.constData()));
1092 if (!file.open(flags: QFile::ReadOnly))
1093 continue;
1094
1095 QByteArray input = readOrMapFile(file: &file);
1096
1097 file.close();
1098 if (input.isEmpty())
1099 continue;
1100
1101 Symbols saveSymbols = symbols;
1102 int saveIndex = index;
1103
1104 // phase 1: get rid of backslash-newlines
1105 input = cleaned(input);
1106
1107 // phase 2: tokenize for the preprocessor
1108 symbols = tokenize(input);
1109 input.clear();
1110
1111 index = 0;
1112
1113 // phase 3: preprocess conditions and substitute macros
1114 preprocessed += Symbol(0, MOC_INCLUDE_BEGIN, include);
1115 preprocess(filename: include, preprocessed);
1116 preprocessed += Symbol(lineNum, MOC_INCLUDE_END, include);
1117
1118 symbols = saveSymbols;
1119 index = saveIndex;
1120 continue;
1121 }
1122 case PP_DEFINE:
1123 {
1124 next();
1125 QByteArray name = lexem();
1126 if (name.isEmpty() || !is_ident_start(s: name[0]))
1127 error();
1128 Macro macro;
1129 macro.isVariadic = false;
1130 if (test(token: LPAREN)) {
1131 // we have a function macro
1132 macro.isFunction = true;
1133 parseDefineArguments(m: &macro);
1134 } else {
1135 macro.isFunction = false;
1136 }
1137 int start = index;
1138 until(PP_NEWLINE);
1139 macro.symbols.reserve(size: index - start - 1);
1140
1141 // remove whitespace where there shouldn't be any:
1142 // Before and after the macro, after a # and around ##
1143 Token lastToken = HASH; // skip shitespace at the beginning
1144 for (int i = start; i < index - 1; ++i) {
1145 Token token = symbols.at(i).token;
1146 if (token == WHITESPACE) {
1147 if (lastToken == PP_HASH || lastToken == HASH ||
1148 lastToken == PP_HASHHASH ||
1149 lastToken == WHITESPACE)
1150 continue;
1151 } else if (token == PP_HASHHASH) {
1152 if (!macro.symbols.isEmpty() &&
1153 lastToken == WHITESPACE)
1154 macro.symbols.pop_back();
1155 }
1156 macro.symbols.append(t: symbols.at(i));
1157 lastToken = token;
1158 }
1159 // remove trailing whitespace
1160 while (!macro.symbols.isEmpty() &&
1161 (macro.symbols.constLast().token == PP_WHITESPACE || macro.symbols.constLast().token == WHITESPACE))
1162 macro.symbols.pop_back();
1163
1164 if (!macro.symbols.isEmpty()) {
1165 if (macro.symbols.constFirst().token == PP_HASHHASH ||
1166 macro.symbols.constLast().token == PP_HASHHASH) {
1167 error(msg: "'##' cannot appear at either end of a macro expansion");
1168 }
1169 }
1170 macros.insert(key: name, value: macro);
1171 continue;
1172 }
1173 case PP_UNDEF: {
1174 next();
1175 QByteArray name = lexem();
1176 until(PP_NEWLINE);
1177 macros.remove(key: name);
1178 continue;
1179 }
1180 case PP_IDENTIFIER: {
1181 // substitute macros
1182 macroExpand(into: &preprocessed, that: this, toExpand: symbols, index, lineNum: symbol().lineNum, one: true);
1183 continue;
1184 }
1185 case PP_HASH:
1186 until(PP_NEWLINE);
1187 continue; // skip unknown preprocessor statement
1188 case PP_IFDEF:
1189 case PP_IFNDEF:
1190 case PP_IF:
1191 while (!evaluateCondition()) {
1192 if (!skipBranch())
1193 break;
1194 if (test(token: PP_ELIF)) {
1195 } else {
1196 until(PP_NEWLINE);
1197 break;
1198 }
1199 }
1200 continue;
1201 case PP_ELIF:
1202 case PP_ELSE:
1203 skipUntilEndif();
1204 Q_FALLTHROUGH();
1205 case PP_ENDIF:
1206 until(PP_NEWLINE);
1207 continue;
1208 case PP_NEWLINE:
1209 continue;
1210 case SIGNALS:
1211 case SLOTS: {
1212 Symbol sym = symbol();
1213 if (macros.contains(key: "QT_NO_KEYWORDS"))
1214 sym.token = IDENTIFIER;
1215 else
1216 sym.token = (token == SIGNALS ? Q_SIGNALS_TOKEN : Q_SLOTS_TOKEN);
1217 preprocessed += sym;
1218 } continue;
1219 default:
1220 break;
1221 }
1222 preprocessed += symbol();
1223 }
1224
1225 currentFilenames.pop();
1226}

1270void Preprocessor::parseDefineArguments(Macro *m)
1271{
1272 Symbols arguments;
1273 while (hasNext()) {
1274 while (test(token: PP_WHITESPACE)) {}
1275 Token t = next();
1276 if (t == PP_RPAREN)
1277 break;
1278 if (t != PP_IDENTIFIER) {
1279 QByteArray l = lexem();
1280 if (l == "...") {
1281 m->isVariadic = true;
1282 arguments += Symbol(symbol().lineNum, PP_IDENTIFIER, "__VA_ARGS__");
1283 while (test(token: PP_WHITESPACE)) {}
1284 if (!test(token: PP_RPAREN))
1285 error(msg: "missing ')' in macro argument list");
1286 break;
1287 } else if (!is_identifier(s: l.constData(), len: l.size())) {
1288 error(msg: "Unexpected character in macro argument list.");
1289 }
1290 }
1291
1292 Symbol arg = symbol();
1293 if (arguments.contains(t: arg))
1294 error(msg: "Duplicate macro parameter.");
1295 arguments += symbol();
1296
1297 while (test(token: PP_WHITESPACE)) {}
1298 t = next();
1299 if (t == PP_RPAREN)
1300 break;
1301 if (t == PP_COMMA)
1302 continue;
1303 if (lexem() == "...") {
1304 //GCC extension: #define FOO(x, y...) x(y)
1305 // The last argument was already parsed. Just mark the macro as variadic.
1306 m->isVariadic = true;
1307 while (test(token: PP_WHITESPACE)) {}
1308 if (!test(token: PP_RPAREN))
1309 error(msg: "missing ')' in macro argument list");
1310 break;
1311 }
1312 error(msg: "Unexpected character in macro argument list.");
1313 }
1314 m->arguments = arguments;
1315 while (test(token: PP_WHITESPACE)) {}
1316}
