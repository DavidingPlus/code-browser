15static const char *error_msg = nullptr;

17#ifdef Q_CC_MSVC
18#define ErrorFormatString "%s(%d:%d): "
19#else
20#define ErrorFormatString "%s:%d:%d: "
21#endif
22
23void Parser::error(int rollback) {
24 index -= rollback;
25 error();
26}
27void Parser::error(const char *msg) {
28 if (msg || error_msg)
29 fprintf(stderr, ErrorFormatString "error: %s\n",
30 currentFilenames.top().constData(), symbol().lineNum, 1, msg?msg:error_msg);
31 else
32 fprintf(stderr, ErrorFormatString "error: Parse error at \"%s\"\n",
33 currentFilenames.top().constData(), symbol().lineNum, 1, symbol().lexem().data());
34 exit(EXIT_FAILURE);
35}

37void Parser::warning(const char *msg) {
38 if (displayWarnings && msg)
39 fprintf(stderr, ErrorFormatString "warning: %s\n",
40 currentFilenames.top().constData(), qMax(a: 0, b: index > 0 ? symbol().lineNum : 0), 1, msg);
41}

43void Parser::note(const char *msg) {

46 currentFilenames.top().constData(), qMax(a: 0, b: index > 0 ? symbol().lineNum : 0), 1, msg);
