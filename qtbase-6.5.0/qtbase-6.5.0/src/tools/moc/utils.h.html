21inline bool is_ident_start(char s)
22{
23 return ((s >= 'a' && s <= 'z')
24 || (s >= 'A' && s <= 'Z')
25 || s == '_' || s == '$'
26 );
27}

29inline bool is_ident_char(char s)
30{
31 return ((s >= 'a' && s <= 'z')
32 || (s >= 'A' && s <= 'Z')
33 || (s >= '0' && s <= '9')
34 || s == '_' || s == '$'
35 );
36}

38inline bool is_identifier(const char *s, int len)
39{
40 if (len < 1)
41 return false;
42 if (!is_ident_start(s: *s))
43 return false;
44 for (int i = 1; i < len; ++i)
45 if (!is_ident_char(s: s[i]))
46 return false;
47 return true;
48}

68inline const char *skipQuote(const char *data)
69{
70 while (*data && (*data != '\"')) {
71 if (*data == '\\') {
72 ++data;
73 if (!*data) break;
74 }
75 ++data;
76 }
77
78 if (*data) //Skip last quote
79 ++data;
80 return data;
81}
