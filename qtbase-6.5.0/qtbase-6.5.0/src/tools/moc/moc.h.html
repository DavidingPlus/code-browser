102struct PropertyDef
103{
104 bool stdCppSet() const {
105 QByteArray s("set");
106 s += QtMiscUtils::toAsciiUpper(ch: name[0]);
107 s += name.mid(index: 1);
108 return (s == write);
109 }
110
111 QByteArray name, type, member, read, write, bind, reset, designable, scriptable, stored, user, notify, inPrivateClass;
112 int notifyId = -1; // -1 means no notifyId, >= 0 means signal defined in this class, < -1 means signal not defined in this class
113 enum Specification { ValueSpec, ReferenceSpec, PointerSpec };
114 Specification gspec = ValueSpec;
115 int revision = 0;
116 bool constant = false;
117 bool final = false;
118 bool required = false;
119 int relativeIndex = -1; // property index in current metaobject
120
121 int location = -1; // token index, used for error reporting
122
123 QJsonObject toJson() const;
124};

276inline QByteArray noRef(const QByteArray &type)
277{
278 if (type.endsWith(c: '&')) {
279 if (type.endsWith(bv: "&&"))
280 return type.left(len: type.size()-2);
281 return type.left(len: type.size()-1);
282 }
283 return type;
284}
