26uint nameToBuiltinType(const QByteArray &name)
27{
28 if (name.isEmpty())
29 return 0;
30
31 uint tp = qMetaTypeTypeInternal(name.constData());
32 return tp < uint(QMetaType::User) ? tp : uint(QMetaType::UnknownType);

73static inline int lengthOfEscapeSequence(const QByteArray &s, int i)
74{
75 if (s.at(i) != '\\' || i >= s.size() - 1)
76 return 1;
77 const int startPos = i;
78 ++i;
79 char ch = s.at(i);
80 if (ch == 'x') {
81 ++i;
82 while (i < s.size() && is_hex_char(s: s.at(i)))
83 ++i;
84 } else if (is_octal_char(s: ch)) {
85 while (i < startPos + 4
86 && i < s.size()
87 && is_octal_char(s: s.at(i))) {
88 ++i;
89 }
90 } else { // single character escape sequence
91 i = qMin(a: i + 1, b: s.size());
92 }
93 return i - startPos;
94}

96static inline uint lengthOfEscapedString(const QByteArray &str)
97{
98 int extra = 0;
99 for (int j = 0; j < str.size(); ++j) {
100 if (str.at(i: j) == '\\') {
101 int cnt = lengthOfEscapeSequence(s: str, i: j) - 1;
102 extra += cnt;
103 j += cnt;
104 }
105 }
106 return str.size() - extra;
107}

109// Prints \a s to \a out, breaking it into lines of at most ColumnWidth. The
110// opening and closing quotes are NOT included (it's up to the caller).
111static void printStringWithIndentation(FILE *out, const QByteArray &s)
112{
113 static constexpr int ColumnWidth = 72;
114 int len = s.size();
115 int idx = 0;
116
117 do {
118 int spanLen = qMin(a: ColumnWidth - 2, b: len - idx);
119 // don't cut escape sequences at the end of a line
120 int backSlashPos = s.lastIndexOf(c: '\\', from: idx + spanLen - 1);
121 if (backSlashPos >= idx) {
122 int escapeLen = lengthOfEscapeSequence(s, i: backSlashPos);
123 spanLen = qBound(min: spanLen, val: backSlashPos + escapeLen - idx, max: len - idx);
124 }
125 fprintf(stream: out, format: "\n \"%.*s\"", spanLen, s.constData() + idx);
126 idx += spanLen;
127 } while (idx < len);
128}

143// Returns the sum of all parameters (including return type) for the given
144// \a list of methods. This is needed for calculating the size of the methods'
145// parameter type/name meta-data.
146static int aggregateParameterCount(const QList<FunctionDef> &list)
147{
148 int sum = 0;
149 for (int i = 0; i < list.size(); ++i)
150 sum += list.at(i).arguments.size() + 1; // +1 for return type
151 return sum;
152}

154bool Generator::registerableMetaType(const QByteArray &propertyType)
155{
156 if (metaTypes.contains(t: propertyType))
157 return true;
158
159 if (propertyType.endsWith(c: '*')) {
160 QByteArray objectPointerType = propertyType;
161 // The objects container stores class names, such as 'QState', 'QLabel' etc,
162 // not 'QState*', 'QLabel*'. The propertyType does contain the '*', so we need
163 // to chop it to find the class type in the known QObjects list.
164 objectPointerType.chop(n: 1);
165 if (knownQObjectClasses.contains(key: objectPointerType))
166 return true;
167 }
168
169 static const QList<QByteArray> smartPointers = QList<QByteArray>()
170#define STREAM_SMART_POINTER(SMART_POINTER) << #SMART_POINTER
171 QT_FOR_EACH_AUTOMATIC_TEMPLATE_SMART_POINTER(STREAM_SMART_POINTER)
172#undef STREAM_SMART_POINTER
173 ;
174
175 for (const QByteArray &smartPointer : smartPointers) {
176 QByteArray ba = smartPointer + "<";
177 if (propertyType.startsWith(bv: ba) && !propertyType.endsWith(bv: "&"))
178 return knownQObjectClasses.contains(key: propertyType.mid(index: smartPointer.size() + 1, len: propertyType.size() - smartPointer.size() - 1 - 1));
179 }
180
181 static const QList<QByteArray> oneArgTemplates = QList<QByteArray>()
182#define STREAM_1ARG_TEMPLATE(TEMPLATENAME) << #TEMPLATENAME
183 QT_FOR_EACH_AUTOMATIC_TEMPLATE_1ARG(STREAM_1ARG_TEMPLATE)
184#undef STREAM_1ARG_TEMPLATE
185 ;
186 for (const QByteArray &oneArgTemplateType : oneArgTemplates) {
187 QByteArray ba = oneArgTemplateType + "<";
188 if (propertyType.startsWith(bv: ba) && propertyType.endsWith(bv: ">")) {
189 const int argumentSize = propertyType.size() - oneArgTemplateType.size() - 1
190 // The closing '>'
191 - 1
192 // templates inside templates have an extra whitespace char to strip.
193 - (propertyType.at(i: propertyType.size() - 2) == ' ' ? 1 : 0 );
194 const QByteArray templateArg = propertyType.mid(index: oneArgTemplateType.size() + 1, len: argumentSize);
195 return isBuiltinType(type: templateArg) || registerableMetaType(propertyType: templateArg);
196 }
197 }
198 return false;
199}

201/* returns \c true if name and qualifiedName refers to the same name.
202 * If qualified name is "A::B::C", it returns \c true for "C", "B::C" or "A::B::C" */
203static bool qualifiedNameEquals(const QByteArray &qualifiedName, const QByteArray &name)
204{
205 if (qualifiedName == name)
206 return true;
207 int index = qualifiedName.indexOf(bv: "::");
208 if (index == -1)
209 return false;
210 return qualifiedNameEquals(qualifiedName: qualifiedName.mid(index: index+2), name);
211}

213static QByteArray generateQualifiedClassNameIdentifier(const QByteArray &identifier)
214{
215 QByteArray qualifiedClassNameIdentifier = identifier;
216
217 // Remove ':'s in the name, but be sure not to create any illegal
218 // identifiers in the process. (Don't replace with '_', because
219 // that will create problems with things like NS_::_class.)
220 qualifiedClassNameIdentifier.replace(before: "::", after: "SCOPE");
221
222 // Also, avoid any leading/trailing underscores (we'll concatenate
223 // the generated name with other prefixes/suffixes, and these latter
224 // may already include an underscore, leading to two underscores)
225 qualifiedClassNameIdentifier = "CLASS" + qualifiedClassNameIdentifier + "ENDCLASS";
226 return qualifiedClassNameIdentifier;
227}

229void Generator::generateCode()
230{
231 bool isQObject = (cdef->classname == "QObject");
232 bool isConstructible = !cdef->constructorList.isEmpty();
233
234 // filter out undeclared enumerators and sets
235 {
236 QList<EnumDef> enumList;
237 for (int i = 0; i < cdef->enumList.size(); ++i) {
238 EnumDef def = cdef->enumList.at(i);
239 if (cdef->enumDeclarations.contains(key: def.name)) {
240 enumList += def;
241 }
242 def.enumName = def.name;
243 QByteArray alias = cdef->flagAliases.value(key: def.name);
244 if (cdef->enumDeclarations.contains(key: alias)) {
245 def.name = alias;
246 enumList += def;
247 }
248 }
249 cdef->enumList = enumList;
250 }
251
252//
253// Register all strings used in data section
254//
255 strreg(s: cdef->qualified);
256 registerClassInfoStrings();
257 registerFunctionStrings(list: cdef->signalList);
258 registerFunctionStrings(list: cdef->slotList);
259 registerFunctionStrings(list: cdef->methodList);
260 registerFunctionStrings(list: cdef->constructorList);
261 registerByteArrayVector(list: cdef->nonClassSignalList);
262 registerPropertyStrings();
263 registerEnumStrings();
264
265 const bool hasStaticMetaCall =
266 (cdef->hasQObject || !cdef->methodList.isEmpty()
267 || !cdef->propertyList.isEmpty() || !cdef->constructorList.isEmpty());
268
269 const QByteArray qualifiedClassNameIdentifier = generateQualifiedClassNameIdentifier(identifier: cdef->qualified);
270
271 // ensure the qt_meta_stringdata_XXXX_t type is local
272 fprintf(stream: out, format: "namespace {\n");
273
274//
275// Build the strings using QtMocHelpers::StringData
276//
277
278 fprintf(stream: out, format: "\n#ifdef QT_MOC_HAS_STRINGDATA\n"
279 "struct qt_meta_stringdata_%s_t {};\n"
280 "static constexpr auto qt_meta_stringdata_%s = QtMocHelpers::stringData(",
281 qualifiedClassNameIdentifier.constData(), qualifiedClassNameIdentifier.constData());
282 {
283 char comma = 0;
284 for (const QByteArray &str : strings) {
285 if (comma)
286 fputc(c: comma, stream: out);
287 printStringWithIndentation(out, s: str);
288 comma = ',';
289 }
290 }
291 fprintf(stream: out, format: "\n);\n"
292 "#else // !QT_MOC_HAS_STRING_DATA\n");
293
294#if QT_VERSION >= QT_VERSION_CHECK(6, 9, 0)
295 fprintf(out, "#error \"qtmochelpers.h not found or too old.\"\n");
296#else
297//
298// Build stringdata struct
299//
300
301 fprintf(stream: out, format: "struct qt_meta_stringdata_%s_t {\n", qualifiedClassNameIdentifier.constData());
302 fprintf(stream: out, format: " uint offsetsAndSizes[%d];\n", int(strings.size() * 2));
303 for (int i = 0; i < strings.size(); ++i) {
304 int thisLength = lengthOfEscapedString(str: strings.at(i)) + 1;
305 fprintf(stream: out, format: " char stringdata%d[%d];\n", i, thisLength);
306 }
307 fprintf(stream: out, format: "};\n");
308
309 // Macro that simplifies the string data listing. The offset is calculated
310 // from the top of the stringdata object (i.e., past the uints).
311 fprintf(stream: out, format: "#define QT_MOC_LITERAL(ofs, len) \\\n"
312 " uint(sizeof(qt_meta_stringdata_%s_t::offsetsAndSizes) + ofs), len \n",
313 qualifiedClassNameIdentifier.constData());
314
315 fprintf(stream: out, format: "Q_CONSTINIT static const qt_meta_stringdata_%s_t qt_meta_stringdata_%s = {\n",
316 qualifiedClassNameIdentifier.constData(), qualifiedClassNameIdentifier.constData());
317 fprintf(stream: out, format: " {");
318 {
319 int idx = 0;
320 for (int i = 0; i < strings.size(); ++i) {
321 const QByteArray &str = strings.at(i);
322 const QByteArray comment = str.size() > 32 ? str.left(len: 29) + "..." : str;
323 const char *comma = (i != strings.size() - 1 ? "," : " ");
324 int len = lengthOfEscapedString(str);
325 fprintf(stream: out, format: "\n QT_MOC_LITERAL(%d, %d)%s // \"%s\"", idx, len, comma,
326 comment.constData());
327
328 idx += len + 1;
329 }
330 fprintf(stream: out, format: "\n }");
331 }
332
333//
334// Build stringdata arrays
335//
336 for (const QByteArray &s : std::as_const(t&: strings)) {
337 fputc(c: ',', stream: out);
338 printStringWithIndentation(out, s);
339 }
340
341// Terminate stringdata struct
342 fprintf(stream: out, format: "\n};\n");
343 fprintf(stream: out, format: "#undef QT_MOC_LITERAL\n");
344#endif // Qt 6.9
345
346 fprintf(stream: out, format: "#endif // !QT_MOC_HAS_STRING_DATA\n");
347 fprintf(stream: out, format: "} // unnamed namespace\n\n");
348
349//
350// build the data array
351//
352
353 int index = MetaObjectPrivateFieldCount;
354 fprintf(stream: out, format: "Q_CONSTINIT static const uint qt_meta_data_%s[] = {\n", qualifiedClassNameIdentifier.constData());
355 fprintf(stream: out, format: "\n // content:\n");
356 fprintf(stream: out, format: " %4d, // revision\n", int(QMetaObjectPrivate::OutputRevision));
357 fprintf(stream: out, format: " %4d, // classname\n", stridx(s: cdef->qualified));
358 fprintf(stream: out, format: " %4d, %4d, // classinfo\n", int(cdef->classInfoList.size()), int(cdef->classInfoList.size() ? index : 0));
359 index += cdef->classInfoList.size() * 2;
360
361 int methodCount = cdef->signalList.size() + cdef->slotList.size() + cdef->methodList.size();
362 fprintf(stream: out, format: " %4d, %4d, // methods\n", methodCount, methodCount ? index : 0);
363 index += methodCount * QMetaObjectPrivate::IntsPerMethod;
364 if (cdef->revisionedMethods)
365 index += methodCount;
366 int paramsIndex = index;
367 int totalParameterCount = aggregateParameterCount(list: cdef->signalList)
368 + aggregateParameterCount(list: cdef->slotList)
369 + aggregateParameterCount(list: cdef->methodList)
370 + aggregateParameterCount(list: cdef->constructorList);
371 index += totalParameterCount * 2 // types and parameter names
372 - methodCount // return "parameters" don't have names
373 - cdef->constructorList.size(); // "this" parameters don't have names
374
375 fprintf(stream: out, format: " %4d, %4d, // properties\n", int(cdef->propertyList.size()), int(cdef->propertyList.size() ? index : 0));
376 index += cdef->propertyList.size() * QMetaObjectPrivate::IntsPerProperty;
377 fprintf(stream: out, format: " %4d, %4d, // enums/sets\n", int(cdef->enumList.size()), cdef->enumList.size() ? index : 0);
378
379 int enumsIndex = index;
380 for (int i = 0; i < cdef->enumList.size(); ++i)
381 index += 5 + (cdef->enumList.at(i).values.size() * 2);
382 fprintf(stream: out, format: " %4d, %4d, // constructors\n", isConstructible ? int(cdef->constructorList.size()) : 0,
383 isConstructible ? index : 0);
384
385 int flags = 0;
386 if (cdef->hasQGadget || cdef->hasQNamespace) {
387 // Ideally, all the classes could have that flag. But this broke classes generated
388 // by qdbusxml2cpp which generate code that require that we call qt_metacall for properties
389 flags |= PropertyAccessInStaticMetaCall;
390 }
391 fprintf(stream: out, format: " %4d, // flags\n", flags);
392 fprintf(stream: out, format: " %4d, // signalCount\n", int(cdef->signalList.size()));
393
394
395//
396// Build classinfo array
397//
398 generateClassInfos();
399
400 // all property metatypes, + 1 for the type of the current class itself
401 int initialMetaTypeOffset = cdef->propertyList.size() + 1;
402
403//
404// Build signals array first, otherwise the signal indices would be wrong
405//
406 generateFunctions(list: cdef->signalList, functype: "signal", type: MethodSignal, paramsIndex, initialMetatypeOffset&: initialMetaTypeOffset);
407
408//
409// Build slots array
410//
411 generateFunctions(list: cdef->slotList, functype: "slot", type: MethodSlot, paramsIndex, initialMetatypeOffset&: initialMetaTypeOffset);
412
413//
414// Build method array
415//
416 generateFunctions(list: cdef->methodList, functype: "method", type: MethodMethod, paramsIndex, initialMetatypeOffset&: initialMetaTypeOffset);
417
418//
419// Build method version arrays
420//
421 if (cdef->revisionedMethods) {
422 generateFunctionRevisions(list: cdef->signalList, functype: "signal");
423 generateFunctionRevisions(list: cdef->slotList, functype: "slot");
424 generateFunctionRevisions(list: cdef->methodList, functype: "method");
425 }
426
427//
428// Build method parameters array
429//
430 generateFunctionParameters(list: cdef->signalList, functype: "signal");
431 generateFunctionParameters(list: cdef->slotList, functype: "slot");
432 generateFunctionParameters(list: cdef->methodList, functype: "method");
433 if (isConstructible)
434 generateFunctionParameters(list: cdef->constructorList, functype: "constructor");
435
436//
437// Build property array
438//
439 generateProperties();
440
441//
442// Build enums array
443//
444 generateEnums(index: enumsIndex);
445
446//
447// Build constructors array
448//
449 if (isConstructible)
450 generateFunctions(list: cdef->constructorList, functype: "constructor", type: MethodConstructor, paramsIndex, initialMetatypeOffset&: initialMetaTypeOffset);
451
452//
453// Terminate data array
454//
455 fprintf(stream: out, format: "\n 0 // eod\n};\n\n");
456
457//
458// Build extra array
459//
460 QList<QByteArray> extraList;
461 QMultiHash<QByteArray, QByteArray> knownExtraMetaObject(knownGadgets);
462 knownExtraMetaObject.unite(other: knownQObjectClasses);
463
464 for (int i = 0; i < cdef->propertyList.size(); ++i) {
465 const PropertyDef &p = cdef->propertyList.at(i);
466 if (isBuiltinType(type: p.type))
467 continue;
468
469 if (p.type.contains(c: '*') || p.type.contains(c: '<') || p.type.contains(c: '>'))
470 continue;
471
472 int s = p.type.lastIndexOf(bv: "::");
473 if (s <= 0)
474 continue;
475
476 QByteArray unqualifiedScope = p.type.left(len: s);
477
478 // The scope may be a namespace for example, so it's only safe to include scopes that are known QObjects (QTBUG-2151)
479 QMultiHash<QByteArray, QByteArray>::ConstIterator scopeIt;
480
481 QByteArray thisScope = cdef->qualified;
482 do {
483 int s = thisScope.lastIndexOf(bv: "::");
484 thisScope = thisScope.left(len: s);
485 QByteArray currentScope = thisScope.isEmpty() ? unqualifiedScope : thisScope + "::" + unqualifiedScope;
486 scopeIt = knownExtraMetaObject.constFind(key: currentScope);
487 } while (!thisScope.isEmpty() && scopeIt == knownExtraMetaObject.constEnd());
488
489 if (scopeIt == knownExtraMetaObject.constEnd())
490 continue;
491
492 const QByteArray &scope = *scopeIt;
493
494 if (scope == "Qt")
495 continue;
496 if (qualifiedNameEquals(qualifiedName: cdef->qualified, name: scope))
497 continue;
498
499 if (!extraList.contains(t: scope))
500 extraList += scope;
501 }
502
503 // QTBUG-20639 - Accept non-local enums for QML signal/slot parameters.
504 // Look for any scoped enum declarations, and add those to the list
505 // of extra/related metaobjects for this object.
506 for (auto it = cdef->enumDeclarations.keyBegin(),
507 end = cdef->enumDeclarations.keyEnd(); it != end; ++it) {
508 const QByteArray &enumKey = *it;
509 int s = enumKey.lastIndexOf(bv: "::");
510 if (s > 0) {
511 QByteArray scope = enumKey.left(len: s);
512 if (scope != "Qt" && !qualifiedNameEquals(qualifiedName: cdef->qualified, name: scope) && !extraList.contains(t: scope))
513 extraList += scope;
514 }
515 }
516
517//
518// Generate meta object link to parent meta objects
519//
520
521 if (!extraList.isEmpty()) {
522 fprintf(stream: out, format: "Q_CONSTINIT static const QMetaObject::SuperData qt_meta_extradata_%s[] = {\n",
523 qualifiedClassNameIdentifier.constData());
524 for (int i = 0; i < extraList.size(); ++i) {
525 fprintf(stream: out, format: " QMetaObject::SuperData::link<%s::staticMetaObject>(),\n", extraList.at(i).constData());
526 }
527 fprintf(stream: out, format: " nullptr\n};\n\n");
528 }
529
530//
531// Finally create and initialize the static meta object
532//
533 fprintf(stream: out, format: "Q_CONSTINIT const QMetaObject %s::staticMetaObject = { {\n",
534 cdef->qualified.constData());
535
536 if (isQObject)
537 fprintf(stream: out, format: " nullptr,\n");
538 else if (cdef->superclassList.size() && !cdef->hasQGadget && !cdef->hasQNamespace) // for qobject, we know the super class must have a static metaobject
539 fprintf(stream: out, format: " QMetaObject::SuperData::link<%s::staticMetaObject>(),\n", purestSuperClass.constData());
540 else if (cdef->superclassList.size()) // for gadgets we need to query at compile time for it
541 fprintf(stream: out, format: " QtPrivate::MetaObjectForType<%s>::value,\n", purestSuperClass.constData());
542 else
543 fprintf(stream: out, format: " nullptr,\n");
544 fprintf(stream: out, format: " qt_meta_stringdata_%s.offsetsAndSizes,\n"
545 " qt_meta_data_%s,\n", qualifiedClassNameIdentifier.constData(),
546 qualifiedClassNameIdentifier.constData());
547 if (hasStaticMetaCall)
548 fprintf(stream: out, format: " qt_static_metacall,\n");
549 else
550 fprintf(stream: out, format: " nullptr,\n");
551
552 if (extraList.isEmpty())
553 fprintf(stream: out, format: " nullptr,\n");
554 else
555 fprintf(stream: out, format: " qt_meta_extradata_%s,\n", qualifiedClassNameIdentifier.constData());
556
557 const char *comma = "";
558 const bool requireCompleteness = requireCompleteTypes || cdef->requireCompleteMethodTypes;
559 auto stringForType = [requireCompleteness](const QByteArray &type, bool forceComplete) -> QByteArray {
560 const char *forceCompleteType = forceComplete ? ", std::true_type>" : ", std::false_type>";
561 if (requireCompleteness)
562 return type;
563 return "QtPrivate::TypeAndForceComplete<" % type % forceCompleteType;
564 };
565 if (!requireCompleteness) {
566 fprintf(stream: out, format: " qt_incomplete_metaTypeArray<qt_meta_stringdata_%s_t", qualifiedClassNameIdentifier.constData());
567 comma = ",";
568 } else {
569 fprintf(stream: out, format: " qt_metaTypeArray<");
570 }
571 // metatypes for properties
572 for (int i = 0; i < cdef->propertyList.size(); ++i) {
573 const PropertyDef &p = cdef->propertyList.at(i);
574 fprintf(stream: out, format: "%s\n // property '%s'\n %s",
575 comma, p.name.constData(), stringForType(p.type, true).constData());
576 comma = ",";
577 }
578
579 // type name for the Q_OJBECT/GADGET itself, void for namespaces
580 auto ownType = !cdef->hasQNamespace ? cdef->classname.data() : "void";
581 fprintf(stream: out, format: "%s\n // Q_OBJECT / Q_GADGET\n %s",
582 comma, stringForType(ownType, true).constData());
583 comma = ",";
584
585 // metatypes for all exposed methods
586 // because we definitely printed something above, this section doesn't need comma control
587 for (const QList<FunctionDef> &methodContainer :
588 { cdef->signalList, cdef->slotList, cdef->methodList }) {
589 for (int i = 0; i< methodContainer.size(); ++i) {
590 const FunctionDef& fdef = methodContainer.at(i);
591 fprintf(stream: out, format: ",\n // method '%s'\n %s",
592 fdef.name.constData(), stringForType(fdef.type.name, false).constData());
593 for (const auto &argument: fdef.arguments)
594 fprintf(stream: out, format: ",\n %s", stringForType(argument.type.name, false).constData());
595 }
596 }
597
598 // but constructors have no return types, so this needs comma control again
599 for (int i = 0; i< cdef->constructorList.size(); ++i) {
600 const FunctionDef& fdef = cdef->constructorList.at(i);
601 if (fdef.arguments.isEmpty())
602 continue;
603
604 fprintf(stream: out, format: "%s\n // constructor '%s'", comma, fdef.name.constData());

606 for (const auto &argument: fdef.arguments) {
607 fprintf(stream: out, format: "%s\n %s", comma,
608 stringForType(argument.type.name, false).constData());
609 comma = ",";
610 }
611 }
612 fprintf(stream: out, format: "\n >,\n");
613
614 fprintf(stream: out, format: " nullptr\n} };\n\n");
615
616//
617// Generate internal qt_static_metacall() function
618//
619 if (hasStaticMetaCall)
620 generateStaticMetacall();

622 if (!cdef->hasQObject)
623 return;
624
625 fprintf(stream: out, format: "\nconst QMetaObject *%s::metaObject() const\n{\n return QObject::d_ptr->metaObject ? QObject::d_ptr->dynamicMetaObject() : &staticMetaObject;\n}\n",
626 cdef->qualified.constData());

628
629//
630// Generate smart cast function
631//
632 fprintf(stream: out, format: "\nvoid *%s::qt_metacast(const char *_clname)\n{\n", cdef->qualified.constData());

634 fprintf(stream: out, format: " if (!strcmp(_clname, qt_meta_stringdata_%s.stringdata0))\n"
635 " return static_cast<void*>(this);\n",
636 qualifiedClassNameIdentifier.constData());
637 for (int i = 1; i < cdef->superclassList.size(); ++i) { // for all superclasses but the first one
638 if (cdef->superclassList.at(i).second == FunctionDef::Private)
639 continue;
640 const char *cname = cdef->superclassList.at(i).first.constData();
641 fprintf(stream: out, format: " if (!strcmp(_clname, \"%s\"))\n return static_cast< %s*>(this);\n",
642 cname, cname);
643 }

645 const QList<ClassDef::Interface> &iface = cdef->interfaceList.at(i);
646 for (int j = 0; j < iface.size(); ++j) {
647 fprintf(stream: out, format: " if (!strcmp(_clname, %s))\n return ", iface.at(i: j).interfaceId.constData());

649 fprintf(stream: out, format: "static_cast< %s*>(", iface.at(i: k).className.constData());

652 }

654 QByteArray superClass = purestSuperClass;
655 fprintf(stream: out, format: " return %s::qt_metacast(_clname);\n", superClass.constData());
656 } else {
657 fprintf(stream: out, format: " return nullptr;\n");
658 }
659 fprintf(stream: out, format: "}\n");

662// Generate internal qt_metacall() function
663//
664 generateMetacall();
665
666//
667// Generate internal signal functions
668//
669 for (int signalindex = 0; signalindex < cdef->signalList.size(); ++signalindex)
670 generateSignal(def: &cdef->signalList[signalindex], index: signalindex);
671
672//
673// Generate plugin meta data
674//
675 generatePluginMetaData();
676
677//
678// Generate function to make sure the non-class signals exist in the parent classes
679//
680 if (!cdef->nonClassSignalList.isEmpty()) {
681 fprintf(stream: out, format: "// If you get a compile error in this function it can be because either\n");
682 fprintf(stream: out, format: "// a) You are using a NOTIFY signal that does not exist. Fix it.\n");
683 fprintf(stream: out, format: "// b) You are using a NOTIFY signal that does exist (in a parent class) but has a non-empty parameter list. This is a moc limitation.\n");
684 fprintf(stream: out, format: "[[maybe_unused]] static void checkNotifySignalValidity_%s(%s *t) {\n", qualifiedClassNameIdentifier.constData(), cdef->qualified.constData());
685 for (const QByteArray &nonClassSignal : std::as_const(t&: cdef->nonClassSignalList))
686 fprintf(stream: out, format: " t->%s();\n", nonClassSignal.constData());

741 int &paramsIndex, int &initialMetatypeOffset)
742{
743 if (list.isEmpty())
744 return;
745 fprintf(stream: out, format: "\n // %ss: name, argc, parameters, tag, flags, initial metatype offsets\n", functype);
746

979void Generator::generateMetacall()
980{
981 bool isQObject = (cdef->classname == "QObject");
982
983 fprintf(stream: out, format: "\nint %s::qt_metacall(QMetaObject::Call _c, int _id, void **_a)\n{\n",
984 cdef->qualified.constData());
985
986 if (!purestSuperClass.isEmpty() && !isQObject) {
987 QByteArray superClass = purestSuperClass;
988 fprintf(stream: out, format: " _id = %s::qt_metacall(_c, _id, _a);\n", superClass.constData());
989 }
990
991
992 bool needElse = false;
993 QList<FunctionDef> methodList;
994 methodList += cdef->signalList;
995 methodList += cdef->slotList;
996 methodList += cdef->methodList;
997
998 // If there are no methods or properties, we will return _id anyway, so
999 // don't emit this comparison -- it is unnecessary, and it makes coverity
1000 // unhappy.
1001 if (methodList.size() || cdef->propertyList.size()) {
1002 fprintf(stream: out, format: " if (_id < 0)\n return _id;\n");
1003 }
1004
1005 fprintf(stream: out, format: " ");
1006
1007 if (methodList.size()) {
1008 needElse = true;
1009 fprintf(stream: out, format: "if (_c == QMetaObject::InvokeMetaMethod) {\n");
1010 fprintf(stream: out, format: " if (_id < %d)\n", int(methodList.size()));
1011 fprintf(stream: out, format: " qt_static_metacall(this, _c, _id, _a);\n");
1012 fprintf(stream: out, format: " _id -= %d;\n }", int(methodList.size()));
1013
1014 fprintf(stream: out, format: " else if (_c == QMetaObject::RegisterMethodArgumentMetaType) {\n");
1015 fprintf(stream: out, format: " if (_id < %d)\n", int(methodList.size()));
1016
1017 if (methodsWithAutomaticTypesHelper(methodList).isEmpty())
1018 fprintf(stream: out, format: " *reinterpret_cast<QMetaType *>(_a[0]) = QMetaType();\n");
1019 else
1020 fprintf(stream: out, format: " qt_static_metacall(this, _c, _id, _a);\n");
1021 fprintf(stream: out, format: " _id -= %d;\n }", int(methodList.size()));
1022
1023 }
1024
1025 if (cdef->propertyList.size()) {
1026 if (needElse)
1027 fprintf(stream: out, format: "else ");
1028 fprintf(stream: out,
1029 format: "if (_c == QMetaObject::ReadProperty || _c == QMetaObject::WriteProperty\n"
1030 " || _c == QMetaObject::ResetProperty || _c == QMetaObject::BindableProperty\n"
1031 " || _c == QMetaObject::RegisterPropertyMetaType) {\n"
1032 " qt_static_metacall(this, _c, _id, _a);\n"
1033 " _id -= %d;\n }", int(cdef->propertyList.size()));
1034 }
1035 if (methodList.size() || cdef->propertyList.size())
1036 fprintf(stream: out, format: "\n ");
1037 fprintf(stream: out,format: "return _id;\n}\n");
1038}

1068void Generator::generateStaticMetacall()
1069{
1070 fprintf(stream: out, format: "void %s::qt_static_metacall(QObject *_o, QMetaObject::Call _c, int _id, void **_a)\n{\n",
1071 cdef->qualified.constData());
1072
1073 bool needElse = false;
1074 bool isUsed_a = false;
1075
1076 if (!cdef->constructorList.isEmpty()) {
1077 fprintf(stream: out, format: " if (_c == QMetaObject::CreateInstance) {\n");
1078 fprintf(stream: out, format: " switch (_id) {\n");
1079 for (int ctorindex = 0; ctorindex < cdef->constructorList.size(); ++ctorindex) {
1080 fprintf(stream: out, format: " case %d: { %s *_r = new %s(", ctorindex,
1081 cdef->classname.constData(), cdef->classname.constData());
1082 const FunctionDef &f = cdef->constructorList.at(i: ctorindex);
1083 int offset = 1;
1084
1085 int argsCount = f.arguments.size();
1086 for (int j = 0; j < argsCount; ++j) {
1087 const ArgumentDef &a = f.arguments.at(i: j);
1088 if (j)
1089 fprintf(stream: out, format: ",");
1090 fprintf(stream: out, format: "(*reinterpret_cast< %s>(_a[%d]))", a.typeNameForCast.constData(), offset++);
1091 }
1092 if (f.isPrivateSignal) {
1093 if (argsCount > 0)
1094 fprintf(stream: out, format: ", ");
1095 fprintf(stream: out, format: "%s", QByteArray("QPrivateSignal()").constData());
1096 }
1097 fprintf(stream: out, format: ");\n");
1098 fprintf(stream: out, format: " if (_a[0]) *reinterpret_cast<%s**>(_a[0]) = _r; } break;\n",
1099 (cdef->hasQGadget || cdef->hasQNamespace) ? "void" : "QObject");
1100 }
1101 fprintf(stream: out, format: " default: break;\n");
1102 fprintf(stream: out, format: " }\n");
1103 fprintf(stream: out, format: " }");
1104 needElse = true;
1105 isUsed_a = true;
1106 }
1107
1108 QList<FunctionDef> methodList;
1109 methodList += cdef->signalList;
1110 methodList += cdef->slotList;
1111 methodList += cdef->methodList;
1112
1113 if (!methodList.isEmpty()) {
1114 if (needElse)
1115 fprintf(stream: out, format: " else ");
1116 else
1117 fprintf(stream: out, format: " ");
1118 fprintf(stream: out, format: "if (_c == QMetaObject::InvokeMetaMethod) {\n");
1119 if (cdef->hasQObject) {
1120#ifndef QT_NO_DEBUG
1121 fprintf(out, " Q_ASSERT(staticMetaObject.cast(_o));\n");
1122#endif
1123 fprintf(stream: out, format: " auto *_t = static_cast<%s *>(_o);\n", cdef->classname.constData());
1124 } else {
1125 fprintf(stream: out, format: " auto *_t = reinterpret_cast<%s *>(_o);\n", cdef->classname.constData());
1126 }
1127 fprintf(stream: out, format: " (void)_t;\n");
1128 fprintf(stream: out, format: " switch (_id) {\n");
1129 for (int methodindex = 0; methodindex < methodList.size(); ++methodindex) {
1130 const FunctionDef &f = methodList.at(i: methodindex);
1131 Q_ASSERT(!f.normalizedType.isEmpty());
1132 fprintf(stream: out, format: " case %d: ", methodindex);
1133 if (f.normalizedType != "void")
1134 fprintf(stream: out, format: "{ %s _r = ", noRef(type: f.normalizedType).constData());
1135 fprintf(stream: out, format: "_t->");
1136 if (f.inPrivateClass.size())
1137 fprintf(stream: out, format: "%s->", f.inPrivateClass.constData());
1138 fprintf(stream: out, format: "%s(", f.name.constData());
1139 int offset = 1;
1140
1141 if (f.isRawSlot) {
1142 fprintf(stream: out, format: "QMethodRawArguments{ _a }");
1143 } else {
1144 int argsCount = f.arguments.size();
1145 for (int j = 0; j < argsCount; ++j) {
1146 const ArgumentDef &a = f.arguments.at(i: j);
1147 if (j)
1148 fprintf(stream: out, format: ",");
1149 fprintf(stream: out, format: "(*reinterpret_cast< %s>(_a[%d]))",a.typeNameForCast.constData(), offset++);
1150 isUsed_a = true;
1151 }
1152 if (f.isPrivateSignal) {
1153 if (argsCount > 0)
1154 fprintf(stream: out, format: ", ");
1155 fprintf(stream: out, format: "%s", "QPrivateSignal()");
1156 }
1157 }
1158 fprintf(stream: out, format: ");");
1159 if (f.normalizedType != "void") {
1160 fprintf(stream: out, format: "\n if (_a[0]) *reinterpret_cast< %s*>(_a[0]) = std::move(_r); } ",
1161 noRef(type: f.normalizedType).constData());
1162 isUsed_a = true;
