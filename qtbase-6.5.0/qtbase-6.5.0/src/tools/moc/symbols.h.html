20struct SubArray
21{
22 inline SubArray():from(0),len(-1){}
23 inline SubArray(const QByteArray &a):array(a),from(0), len(a.size()){}
24 inline SubArray(const char *s):array(s),from(0) { len = array.size(); }
25 inline SubArray(const QByteArray &a, int from, int len):array(a), from(from), len(len){}
26 QByteArray array;
27 int from, len;
28 inline bool operator==(const SubArray &other) const {
29 if (len != other.len)
30 return false;
31 for (int i = 0; i < len; ++i)
32 if (array.at(i: from + i) != other.array.at(i: other.from + i))
33 return false;
34 return true;
35 }
36};
37
38inline size_t qHash(const SubArray &key)
39{
40 return qHash(key: QLatin1StringView(key.array.constData() + key.from, key.len));
41}

44struct Symbol
45{
46
47#ifdef USE_LEXEM_STORE
48 typedef QHash<SubArray, QHashDummyValue> LexemStore;
49 static LexemStore lexemStore;
50
51 inline Symbol() : lineNum(-1),token(NOTOKEN){}
52 inline Symbol(int lineNum, Token token):
53 lineNum(lineNum), token(token){}
54 inline Symbol(int lineNum, Token token, const QByteArray &lexem):
55 lineNum(lineNum), token(token),lex(lexem){}
56 inline Symbol(int lineNum, Token token, const QByteArray &lexem, int from, int len):
57 lineNum(lineNum), token(token){
58 LexemStore::const_iterator it = lexemStore.constFind(SubArray(lexem, from, len));
59
60 if (it != lexemStore.constEnd()) {
61 lex = it.key().array;
62 } else {
63 lex = lexem.mid(from, len);
64 lexemStore.insert(lex, QHashDummyValue());
65 }
66 }
67 int lineNum;
68 Token token;
69 inline QByteArray unquotedLexem() const { return lex.mid(1, lex.length()-2); }
70 inline QByteArray lexem() const { return lex; }
71 inline operator QByteArray() const { return lex; }
72 QByteArray lex;
73
74#else
75
76 inline Symbol() : lineNum(-1),token(NOTOKEN), from(0),len(-1) {}
77 inline Symbol(int lineNum, Token token):
78 lineNum(lineNum), token(token), from(0), len(-1) {}
79 inline Symbol(int lineNum, Token token, const QByteArray &lexem):
80 lineNum(lineNum), token(token), lex(lexem), from(0) { len = lex.size(); }
81 inline Symbol(int lineNum, Token token, const QByteArray &lexem, int from, int len):
82 lineNum(lineNum), token(token),lex(lexem),from(from), len(len){}
83 int lineNum;
84 Token token;
85 inline QByteArray lexem() const { return lex.mid(index: from, len); }
86 inline QByteArray unquotedLexem() const { return lex.mid(index: from+1, len: len-2); }
87 inline operator SubArray() const { return SubArray(lex, from, len); }
88 bool operator==(const Symbol& o) const
89 {
90 return SubArray(lex, from, len) == SubArray(o.lex, o.from, o.len);
91 }
92 QByteArray lex;
93 int from, len;
94
95#endif
96};

109class SymbolStack : public QStack<SafeSymbols>
110{
111public:
112 inline bool hasNext() {
113 while (!isEmpty() && top().index >= top().symbols.size())
114 pop();
115 return !isEmpty();
116 }
117 inline Token next() {
118 while (!isEmpty() && top().index >= top().symbols.size())
119 pop();
120 if (isEmpty())
121 return NOTOKEN;
122 return top().symbols.at(i: top().index++).token;
123 }
124 bool test(Token);
125 inline const Symbol &symbol() const { return top().symbols.at(i: top().index-1); }
126 inline Token token() { return symbol().token; }
127 inline QByteArray lexem() const { return symbol().lexem(); }
128 inline QByteArray unquotedLexem() { return symbol().unquotedLexem(); }
129
130 bool dontReplaceSymbol(const QByteArray &name);
131 QSet<QByteArray> excludeSymbols();
132};

134inline bool SymbolStack::test(Token token)
135{
136 int stackPos = size() - 1;
137 while (stackPos >= 0 && at(i: stackPos).index >= at(i: stackPos).symbols.size())
138 --stackPos;
139 if (stackPos < 0)
140 return false;
141 if (at(i: stackPos).symbols.at(i: at(i: stackPos).index).token == token) {
142 next();
143 return true;
144 }
145 return false;
146}

148inline bool SymbolStack::dontReplaceSymbol(const QByteArray &name)
149{
150 for (int i = 0; i < size(); ++i) {
151 if (name == at(i).expandedMacro || at(i).excludedSymbols.contains(value: name))
152 return true;
153 }
154 return false;
155}
156
157inline QSet<QByteArray> SymbolStack::excludeSymbols()
158{
159 QSet<QByteArray> set;
160 for (int i = 0; i < size(); ++i) {
161 set << at(i).expandedMacro;
162 set += at(i).excludedSymbols;
163 }
164 return set;
165}
