31using AutoGenHeaderMap = QMap<QString, QString>;
32using AutoGenSourcesList = QList<QString>;
33
34static bool readAutogenInfoJson(AutoGenHeaderMap &headers, AutoGenSourcesList &sources,
35 QStringList &headerExts, const QString &autoGenInfoJsonPath)
36{
37 QFile file(autoGenInfoJsonPath);
38 if (!file.open(flags: QIODevice::ReadOnly | QIODevice::Text)) {
39 fprintf(stderr, format: "Could not open: %s\n", qPrintable(autoGenInfoJsonPath));
40 return false;
41 }
42
43 const QByteArray contents = file.readAll();
44 file.close();
45
46 QJsonParseError error;
47 QJsonDocument doc = QJsonDocument::fromJson(json: contents, error: &error);
48
49 if (error.error != QJsonParseError::NoError) {
50 fprintf(stderr, format: "Failed to parse json file: %s\n", qPrintable(autoGenInfoJsonPath));
51 return false;
52 }
53
54 QJsonObject rootObject = doc.object();
55 QJsonValue headersValue = rootObject.value(key: "HEADERS"_L1);
56 QJsonValue sourcesValue = rootObject.value(key: "SOURCES"_L1);
57 QJsonValue headerExtValue = rootObject.value(key: "HEADER_EXTENSIONS"_L1);
58
59 if (!headersValue.isArray() || !sourcesValue.isArray() || !headerExtValue.isArray()) {
60 fprintf(stderr,
61 format: "%s layout does not match the expected layout. This most likely means that file "
62 "format changed or this file is not a product of CMake's AutoGen process.\n",
63 qPrintable(autoGenInfoJsonPath));
64 return false;
65 }
66
67 QJsonArray headersArray = headersValue.toArray();
68 QJsonArray sourcesArray = sourcesValue.toArray();
69 QJsonArray headerExtArray = headerExtValue.toArray();
70
71 for (const QJsonValue value : headersArray) {
72 QJsonArray entry_array = value.toArray();
73 if (entry_array.size() > 2) {
74 // Array[0] : header path
75 // Array[2] : Location of the generated moc file for this header
76 // if no source file includes it
77 headers.insert(key: entry_array[0].toString(), value: entry_array[2].toString());
78 }
79 }
80
81 sources.reserve(size: sourcesArray.size());
82 for (const QJsonValue value : sourcesArray) {
83 QJsonArray entry_array = value.toArray();
84 if (entry_array.size() > 1) {
85 sources.push_back(t: entry_array[0].toString());
86 }
87 }
88
89 headerExts.reserve(size: headerExtArray.size());
90 for (const QJsonValue value : headerExtArray) {
91 headerExts.push_back(t: value.toString());
92 }
93
94 return true;
95}

97struct ParseCacheEntry
98{
99 QStringList mocFiles;
100 QStringList mocIncludes;
101};
102
103using ParseCacheMap = QMap<QString, ParseCacheEntry>;
104
105static bool readParseCache(ParseCacheMap &entries, const QString &parseCacheFilePath)
106{
107
108 QFile file(parseCacheFilePath);
109 if (!file.open(flags: QIODevice::ReadOnly | QIODevice::Text)) {
110 fprintf(stderr, format: "Could not open: %s\n", qPrintable(parseCacheFilePath));
111 return false;
112 }
113
114 QString source;
115 QStringList mocEntries;
116 QStringList mocIncludes;
117
118 // File format
119 // ....
120 // header/source path N
121 // mmc:Q_OBJECT| mcc:Q_GADGET # This file has been mocked
122 // miu:moc_....cpp # Path of the moc.cpp file generated for the above file
123 // relative to TARGET_BINARY_DIR/TARGET_autgen/include directory. Not
124 // present for headers.
125 // mid: ....moc # Path of .moc file generated for the above file relative
126 // to TARGET_BINARY_DIR/TARGET_autogen/include directory.
127 // uic: UI related info, ignored
128 // mdp: Moc dependencies, ignored
129 // udp: UI dependencies, ignored
130 // header/source path N + 1
131 // ....
132
133 QTextStream textStream(&file);
134 const QString mmcKey = QString(" mmc:"_L1);
135 const QString miuKey = QString(" miu:"_L1);
136 const QString uicKey = QString(" uic:"_L1);
137 const QString midKey = QString(" mid:"_L1);
138 const QString mdpKey = QString(" mdp:"_L1);
139 const QString udpKey = QString(" udp:"_L1);
140 QString line;
141 bool mmc_key_found = false;
142 while (textStream.readLineInto(line: &line)) {
143 if (!line.startsWith(c: u' ')) {
144 if (!mocEntries.isEmpty() || mmc_key_found || !mocIncludes.isEmpty()) {
145 entries.insert(key: source,
146 value: ParseCacheEntry { .mocFiles: std::move(mocEntries), .mocIncludes: std::move(mocIncludes) });
147 source.clear();
148 mocEntries = {};
149 mocIncludes = {};
150 mmc_key_found = false;
151 }
152 source = line;
153 } else if (line.startsWith(s: mmcKey)) {
154 mmc_key_found = true;
155 } else if (line.startsWith(s: miuKey)) {
156 mocIncludes.push_back(t: line.right(n: line.size() - miuKey.size()));
157 } else if (line.startsWith(s: midKey)) {
158 mocEntries.push_back(t: line.right(n: line.size() - midKey.size()));
159 } else if (line.startsWith(s: uicKey) || line.startsWith(s: mdpKey) || line.startsWith(s: udpKey)) {
160 // nothing to do ignore
161 continue;
162 } else {
163 fprintf(stderr, format: "Unhandled line entry \"%s\" in %s\n", qPrintable(line),
164 qPrintable(parseCacheFilePath));
165 return false;
166 }
167 }
168
169 // Check if last entry has any data left to processed
170 if (!mocEntries.isEmpty() || !mocIncludes.isEmpty() || mmc_key_found) {
171 entries.insert(key: source, value: ParseCacheEntry { .mocFiles: std::move(mocEntries), .mocIncludes: std::move(mocIncludes) });
172 }
173
174 file.close();
175 return true;
176}

178static bool writeJsonFiles(const QList<QString> &fileList, const QString &fileListFilePath,
179 const QString &timestampFilePath)
180{
181 QFile timestampFile(timestampFilePath);
182 if (!timestampFile.open(flags: QIODevice::ReadWrite)) {
183 fprintf(stderr, format: "Could not open: %s\n", qPrintable(timestampFilePath));
184 return false;
185 }
186
187 qint64 timestamp = std::numeric_limits<qint64>::min();
188 QByteArray timestampBuffer = timestampFile.readAll();
189 if (timestampBuffer.size() == sizeof(timestamp)) {
190 QDataStream istream(&timestampBuffer, QIODevice::ReadOnly);
191 istream >> timestamp;
192 }
193
194 // Check if any of the metatype json files produced by automoc is newer than the last file
195 // processed by cmake_automoc parser
196 for (const auto &jsonFile : fileList) {
197 const qint64 jsonFileLastModified =
198 QFileInfo(jsonFile).lastModified().toMSecsSinceEpoch();
199 if (jsonFileLastModified > timestamp) {
200 timestamp = jsonFileLastModified;
201 }
202 }
203
204 if (timestamp != std::numeric_limits<qint64>::min() || !QFile::exists(fileName: fileListFilePath)) {
205 QFile file(fileListFilePath);
206 if (!file.open(flags: QIODevice::WriteOnly | QIODevice::Text)) {
207 fprintf(stderr, format: "Could not open: %s\n", qPrintable(fileListFilePath));
208 return false;
209 }
210
211 QTextStream textStream(&file);
212 for (const auto &jsonFile : fileList) {
213 textStream << jsonFile << Qt::endl;
214 }
215 textStream.flush();
216
217 // Update the timestamp according the newest json file timestamp.
218 timestampBuffer.clear();
219 QDataStream ostream(&timestampBuffer, QIODevice::WriteOnly);
220 ostream << timestamp;
221 timestampFile.resize(sz: 0);
222 timestampFile.write(data: timestampBuffer);
223 }
224 return true;
225}

227int main(int argc, char **argv)
228{
229
230 QCoreApplication app(argc, argv);
231 QCommandLineParser parser;
232 parser.setApplicationDescription(QStringLiteral("Qt CMake Autogen parser tool"));
233
234 parser.addHelpOption();
235 parser.setSingleDashWordOptionMode(QCommandLineParser::ParseAsLongOptions);
236
237 QCommandLineOption outputFileOption(QStringLiteral("output-file-path"));
238 outputFileOption.setDescription(
239 QStringLiteral("Output file where the meta type file list will be written."));
240 outputFileOption.setValueName(QStringLiteral("output file"));
241 parser.addOption(commandLineOption: outputFileOption);
242
243 QCommandLineOption cmakeAutogenCacheFileOption(QStringLiteral("cmake-autogen-cache-file"));
244 cmakeAutogenCacheFileOption.setDescription(
245 QStringLiteral("Location of the CMake AutoGen ParseCache.txt file."));
246 cmakeAutogenCacheFileOption.setValueName(QStringLiteral("CMake AutoGen ParseCache.txt file"));
247 parser.addOption(commandLineOption: cmakeAutogenCacheFileOption);
248
249 QCommandLineOption cmakeAutogenInfoFileOption(QStringLiteral("cmake-autogen-info-file"));
250 cmakeAutogenInfoFileOption.setDescription(
251 QStringLiteral("Location of the CMake AutoGen AutogenInfo.json file."));
252 cmakeAutogenInfoFileOption.setValueName(QStringLiteral("CMake AutoGen AutogenInfo.json file"));
253 parser.addOption(commandLineOption: cmakeAutogenInfoFileOption);
254
255 QCommandLineOption cmakeAutogenIncludeDirOption(
256 QStringLiteral("cmake-autogen-include-dir-path"));
257 cmakeAutogenIncludeDirOption.setDescription(
258 QStringLiteral("Location of the CMake AutoGen include directory."));
259 cmakeAutogenIncludeDirOption.setValueName(QStringLiteral("CMake AutoGen include directory"));
260 parser.addOption(commandLineOption: cmakeAutogenIncludeDirOption);
261
262 QCommandLineOption isMultiConfigOption(
263 QStringLiteral("cmake-multi-config"));
264 isMultiConfigOption.setDescription(
265 QStringLiteral("Set this option when using CMake with a multi-config generator"));
266 parser.addOption(commandLineOption: isMultiConfigOption);
267
268 QCommandLineOption timestampFilePathOption(QStringLiteral("timestamp-file-path"));
269 timestampFilePathOption.setDescription(
270 QStringLiteral("The path to a timestamp file that determines whether the output"
271 " file needs to be updated."));
272 timestampFilePathOption.setValueName(QStringLiteral("timestamp file"));
273 parser.addOption(commandLineOption: timestampFilePathOption);
274
275 QStringList arguments = QCoreApplication::arguments();
276 parser.process(arguments);
277
278 if (!parser.isSet(option: outputFileOption) || !parser.isSet(option: cmakeAutogenInfoFileOption)
279 || !parser.isSet(option: cmakeAutogenCacheFileOption)
280 || !parser.isSet(option: cmakeAutogenIncludeDirOption)) {
281 parser.showHelp(exitCode: 1);
282 return EXIT_FAILURE;
283 }
284
285 // Read source files from AutogenInfo.json
286 AutoGenHeaderMap autoGenHeaders;
287 AutoGenSourcesList autoGenSources;
288 QStringList headerExtList;
289 if (!readAutogenInfoJson(headers&: autoGenHeaders, sources&: autoGenSources, headerExts&: headerExtList,
290 autoGenInfoJsonPath: parser.value(option: cmakeAutogenInfoFileOption))) {
291 return EXIT_FAILURE;
292 }
293
294 ParseCacheMap parseCacheEntries;
295 if (!readParseCache(entries&: parseCacheEntries, parseCacheFilePath: parser.value(option: cmakeAutogenCacheFileOption))) {
296 return EXIT_FAILURE;
297 }
298
299 const QString cmakeIncludeDir = parser.value(option: cmakeAutogenIncludeDirOption);
300
301 // Algorithm description
302 // 1) For each source from the AutoGenSources list check if there is a parse
303 // cache entry.
304 // 1a) If an entry was wound there exists an moc_...cpp file somewhere.
305 // Remove the header file from the AutoGenHeader files
306 // 1b) For every matched source entry, check the moc includes as it is
307 // possible for a source file to include moc files from other headers.
308 // Remove the header from AutoGenHeaders
309 // 2) For every remaining header in AutoGenHeaders, check if there is an
310 // entry for it in the parse cache. Use the value for the location of the
311 // moc.json file
312
313 QList<QString> jsonFileList;
314 QDir dir(cmakeIncludeDir);
315 jsonFileList.reserve(size: autoGenSources.size());
316
317 // 1) Process sources
318 for (const auto &source : autoGenSources) {
319 auto it = parseCacheEntries.find(key: source);
320 if (it == parseCacheEntries.end()) {
321 continue;
322 }
323
324 const QFileInfo fileInfo(source);
325 const QString base = fileInfo.path() + fileInfo.completeBaseName();
326 // 1a) erase header
327 for (const auto &ext : headerExtList) {
328 const QString headerPath = base + u'.' + ext;
329 auto it = autoGenHeaders.find(key: headerPath);
330 if (it != autoGenHeaders.end()) {
331 autoGenHeaders.erase(it);
332 break;
333 }
334 }
335 // Add extra moc files
336 for (const auto &mocFile : it.value().mocFiles)
337 jsonFileList.push_back(t: dir.filePath(fileName: mocFile) + ".json"_L1);
338 // Add main moc files
339 for (const auto &mocFile : it.value().mocIncludes) {
340 jsonFileList.push_back(t: dir.filePath(fileName: mocFile) + ".json"_L1);
341 // 1b) Locate this header and delete it
342 constexpr int mocKeyLen = 4; // length of "moc_"
343 const QString headerBaseName =
344 QFileInfo(mocFile.right(n: mocFile.size() - mocKeyLen)).completeBaseName();
345 bool breakFree = false;
346 for (auto &ext : headerExtList) {
347 const QString headerSuffix = headerBaseName + u'.' + ext;
348 for (auto it = autoGenHeaders.begin(); it != autoGenHeaders.end(); ++it) {
349 if (it.key().endsWith(s: headerSuffix)
350 && QFileInfo(it.key()).completeBaseName() == headerBaseName) {
351 autoGenHeaders.erase(it);
352 breakFree = true;
353 break;
354 }
355 }
356 if (breakFree) {
357 break;
358 }
359 }
360 }
361 }
362
363 // 2) Process headers
364 const bool isMultiConfig = parser.isSet(option: isMultiConfigOption);
365 for (auto mapIt = autoGenHeaders.begin(); mapIt != autoGenHeaders.end(); ++mapIt) {
366 auto it = parseCacheEntries.find(key: mapIt.key());
367 if (it == parseCacheEntries.end()) {
368 continue;
369 }
370 const QString pathPrefix = !isMultiConfig
371 ? QStringLiteral("../")
372 : QString();
373 const QString jsonPath = dir.filePath(fileName: pathPrefix + mapIt.value() + ".json"_L1);
374 jsonFileList.push_back(t: jsonPath);
375 }
376
377 // Sort for consistent checks across runs
378 jsonFileList.sort();
379
380 // Read Previous file list (if any)
381 if (!writeJsonFiles(fileList: jsonFileList, fileListFilePath: parser.value(option: outputFileOption),
382 timestampFilePath: parser.value(option: timestampFilePathOption))) {
383 return EXIT_FAILURE;
384 }
385
386 return EXIT_SUCCESS;
387}
