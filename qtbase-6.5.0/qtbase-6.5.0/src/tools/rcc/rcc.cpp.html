38void RCCResourceLibrary::write(const char *str, int len)
39{
40 int n = m_out.size();
41 m_out.resize(size: n + len);
42 memcpy(dest: m_out.data() + n, src: str, n: len);
43}

45void RCCResourceLibrary::writeByteArray(const QByteArray &other)
46{

145void RCCFileInfo::writeDataInfo(RCCResourceLibrary &lib)
146{
147 const bool text = lib.m_format == RCCResourceLibrary::C_Code;
148 const bool pass1 = lib.m_format == RCCResourceLibrary::Pass1;
149 const bool python = lib.m_format == RCCResourceLibrary::Python_Code;
150 //some info
151 if (text || pass1) {
152 if (m_language != QLocale::C) {
153 lib.writeString(s: " // ");
154 lib.writeByteArray(other: resourceName().toLocal8Bit());
155 lib.writeString(s: " [");
156 lib.writeByteArray(other: QByteArray::number(m_territory));
157 lib.writeString(s: "::");
158 lib.writeByteArray(other: QByteArray::number(m_language));
159 lib.writeString(s: "[\n ");
160 } else {
161 lib.writeString(s: " // ");
162 lib.writeByteArray(other: resourceName().toLocal8Bit());
163 lib.writeString(s: "\n ");
164 }
165 }
166
167 //pointer data
168 if (m_flags & RCCFileInfo::Directory) {
169 // name offset
170 lib.writeNumber4(number: m_nameOffset);
171
172 // flags
173 lib.writeNumber2(number: m_flags);
174
175 // child count
176 lib.writeNumber4(number: m_children.size());
177
178 // first child offset
179 lib.writeNumber4(number: m_childOffset);
180 } else {
181 // name offset
182 lib.writeNumber4(number: m_nameOffset);
183
184 // flags
185 lib.writeNumber2(number: m_flags);
186
187 // locale
188 lib.writeNumber2(number: m_territory);
189 lib.writeNumber2(number: m_language);
190
191 //data offset
192 lib.writeNumber4(number: m_dataOffset);
193 }
194 if (text || pass1)
195 lib.writeChar(c: '\n');
196 else if (python)
197 lib.writeString(s: "\\\n");
198
199 if (lib.formatVersion() >= 2) {
200 // last modified time stamp
201 const QDateTime lastModified = m_fileInfo.lastModified();
202 quint64 lastmod = quint64(lastModified.isValid() ? lastModified.toMSecsSinceEpoch() : 0);
203 static const quint64 sourceDate = 1000 * qgetenv(varName: "QT_RCC_SOURCE_DATE_OVERRIDE").toULongLong();
204 if (sourceDate != 0)
205 lastmod = sourceDate;
206 static const quint64 sourceDate2 = 1000 * qgetenv(varName: "SOURCE_DATE_EPOCH").toULongLong();
207 if (sourceDate2 != 0)
208 lastmod = sourceDate2;
209 lib.writeNumber8(number: lastmod);
210 if (text || pass1)
211 lib.writeChar(c: '\n');
212 else if (python)
213 lib.writeString(s: "\\\n");
214 }
215}

217qint64 RCCFileInfo::writeDataBlob(RCCResourceLibrary &lib, qint64 offset,
218 QString *errorMessage)
219{
220 const bool text = lib.m_format == RCCResourceLibrary::C_Code;
221 const bool pass1 = lib.m_format == RCCResourceLibrary::Pass1;
222 const bool pass2 = lib.m_format == RCCResourceLibrary::Pass2;
223 const bool binary = lib.m_format == RCCResourceLibrary::Binary;
224 const bool python = lib.m_format == RCCResourceLibrary::Python_Code;
225
226 //capture the offset
227 m_dataOffset = offset;
228
229 //find the data to be written
230 QFile file(m_fileInfo.absoluteFilePath());
231 if (!file.open(flags: QFile::ReadOnly)) {
232 *errorMessage = msgOpenReadFailed(fname: m_fileInfo.absoluteFilePath(), why: file.errorString());
233 return 0;
234 }
235 QByteArray data = file.readAll();
236
237 // Check if compression is useful for this file
238 if (data.size() != 0) {
239#if QT_CONFIG(zstd)
240 if (m_compressAlgo == RCCResourceLibrary::CompressionAlgorithm::Best && !m_noZstd) {
241 m_compressAlgo = RCCResourceLibrary::CompressionAlgorithm::Zstd;
242 m_compressLevel = 19; // not ZSTD_maxCLevel(), as 20+ are experimental
243 }
244 if (m_compressAlgo == RCCResourceLibrary::CompressionAlgorithm::Zstd && !m_noZstd) {
245 if (lib.m_zstdCCtx == nullptr)
246 lib.m_zstdCCtx = ZSTD_createCCtx();
247 qsizetype size = data.size();
248 size = ZSTD_COMPRESSBOUND(size);
249
250 int compressLevel = m_compressLevel;
251 if (compressLevel < 0)
252 compressLevel = CONSTANT_ZSTDCOMPRESSLEVEL_CHECK;
253
254 QByteArray compressed(size, Qt::Uninitialized);
255 char *dst = const_cast<char *>(compressed.constData());
256 size_t n = ZSTD_compressCCtx(lib.m_zstdCCtx, dst, size,
257 data.constData(), data.size(),
258 compressLevel);
259 if (n * 100.0 < data.size() * 1.0 * (100 - m_compressThreshold) ) {
260 // compressing is worth it
261 if (m_compressLevel < 0) {
262 // heuristic compression, so recompress
263 n = ZSTD_compressCCtx(lib.m_zstdCCtx, dst, size,
264 data.constData(), data.size(),
265 CONSTANT_ZSTDCOMPRESSLEVEL_STORE);
266 }
267 if (ZSTD_isError(n)) {
268 QString msg = QString::fromLatin1("%1: error: compression with zstd failed: %2\n")
269 .arg(m_name, QString::fromUtf8(ZSTD_getErrorName(n)));
270 lib.m_errorDevice->write(msg.toUtf8());
271 } else if (lib.verbose()) {
272 QString msg = QString::fromLatin1("%1: note: compressed using zstd (%2 -> %3)\n")
273 .arg(m_name).arg(data.size()).arg(n);
274 lib.m_errorDevice->write(msg.toUtf8());
275 }
276
277 lib.m_overallFlags |= CompressedZstd;
278 m_flags |= CompressedZstd;
279 data = std::move(compressed);
280 data.truncate(n);
281 } else if (lib.verbose()) {
282 QString msg = QString::fromLatin1("%1: note: not compressed\n").arg(m_name);
283 lib.m_errorDevice->write(msg.toUtf8());
284 }
285 }
286#endif
287#ifndef QT_NO_COMPRESS
288 if (m_compressAlgo == RCCResourceLibrary::CompressionAlgorithm::Best) {
289 m_compressAlgo = RCCResourceLibrary::CompressionAlgorithm::Zlib;
290 m_compressLevel = 9;
291 }
292 if (m_compressAlgo == RCCResourceLibrary::CompressionAlgorithm::Zlib) {
293 QByteArray compressed =
294 qCompress(data: reinterpret_cast<uchar *>(data.data()), nbytes: data.size(), compressionLevel: m_compressLevel);
295
296 int compressRatio = int(100.0 * (data.size() - compressed.size()) / data.size());
297 if (compressRatio >= m_compressThreshold) {
298 if (lib.verbose()) {
299 QString msg = QString::fromLatin1(ba: "%1: note: compressed using zlib (%2 -> %3)\n")
300 .arg(a: m_name).arg(a: data.size()).arg(a: compressed.size());
301 lib.m_errorDevice->write(data: msg.toUtf8());
302 }
303 data = compressed;
304 lib.m_overallFlags |= Compressed;
305 m_flags |= Compressed;
306 } else if (lib.verbose()) {
307 QString msg = QString::fromLatin1(ba: "%1: note: not compressed\n").arg(a: m_name);
308 lib.m_errorDevice->write(data: msg.toUtf8());
309 }
310 }
311#endif // QT_NO_COMPRESS
312 }
313
314 // some info
315 if (text || pass1) {
316 lib.writeString(s: " // ");
317 lib.writeByteArray(other: m_fileInfo.absoluteFilePath().toLocal8Bit());
318 lib.writeString(s: "\n ");
319 }
320
321 // write the length
322 if (text || binary || pass2 || python)
323 lib.writeNumber4(number: data.size());
324 if (text || pass1)
325 lib.writeString(s: "\n ");
326 else if (python)
327 lib.writeString(s: "\\\n");
328 offset += 4;
329
330 // write the payload
331 const char *p = data.constData();
332 if (text || python) {
333 for (int i = data.size(), j = 0; --i >= 0; --j) {
334 lib.writeHex(number: *p++);
335 if (j == 0) {
336 if (text)
337 lib.writeString(s: "\n ");
338 else
339 lib.writeString(s: "\\\n");
340 j = 16;
341 }
342 }
343 } else if (binary || pass2) {
344 lib.writeByteArray(other: data);
345 }
346 offset += data.size();
347
348 // done
349 if (text || pass1)
350 lib.writeString(s: "\n ");
351 else if (python)
352 lib.writeString(s: "\\\n");
353
354 return offset;
355}

357qint64 RCCFileInfo::writeDataName(RCCResourceLibrary &lib, qint64 offset)
358{
359 const bool text = lib.m_format == RCCResourceLibrary::C_Code;
360 const bool pass1 = lib.m_format == RCCResourceLibrary::Pass1;
361 const bool python = lib.m_format == RCCResourceLibrary::Python_Code;
362
363 // capture the offset
364 m_nameOffset = offset;
365
366 // some info
367 if (text || pass1) {
368 lib.writeString(s: " // ");
369 lib.writeByteArray(other: m_name.toLocal8Bit());
370 lib.writeString(s: "\n ");
371 }
372
373 // write the length
374 lib.writeNumber2(number: m_name.size());
375 if (text || pass1)
376 lib.writeString(s: "\n ");
377 else if (python)
378 lib.writeString(s: "\\\n");
379 offset += 2;
380
381 // write the hash
382 lib.writeNumber4(number: qt_hash(key: m_name));
383 if (text || pass1)
384 lib.writeString(s: "\n ");
385 else if (python)
386 lib.writeString(s: "\\\n");
387 offset += 4;
388
389 // write the m_name
390 const QChar *unicode = m_name.unicode();
391 for (int i = 0; i < m_name.size(); ++i) {
392 lib.writeNumber2(number: unicode[i].unicode());
393 if ((text || pass1) && i % 16 == 0)
394 lib.writeString(s: "\n ");
395 else if (python && i % 16 == 0)
396 lib.writeString(s: "\\\n");
397 }
398 offset += m_name.size()*2;
399
400 // done
401 if (text || pass1)
402 lib.writeString(s: "\n ");
403 else if (python)
404 lib.writeString(s: "\\\n");
405
406 return offset;
407}

460enum RCCXmlTag {
461 RccTag,
462 ResourceTag,
463 FileTag
464};
465Q_DECLARE_TYPEINFO(RCCXmlTag, Q_PRIMITIVE_TYPE);
466
467bool RCCResourceLibrary::interpretResourceFile(QIODevice *inputDevice,
468 const QString &fname, QString currentPath, bool listMode)
469{
470 Q_ASSERT(m_errorDevice);
471 const QChar slash = u'/';
472 if (!currentPath.isEmpty() && !currentPath.endsWith(c: slash))
473 currentPath += slash;
474
475 QXmlStreamReader reader(inputDevice);
476 QStack<RCCXmlTag> tokens;
477
478 QString prefix;
479 QLocale::Language language = QLocale::c().language();
480 QLocale::Territory territory = QLocale::c().territory();
481 QString alias;
482 auto compressAlgo = m_compressionAlgo;
483 int compressLevel = m_compressLevel;
484 int compressThreshold = m_compressThreshold;
485
486 while (!reader.atEnd()) {
487 QXmlStreamReader::TokenType t = reader.readNext();
488 switch (t) {
489 case QXmlStreamReader::StartElement:
490 if (reader.name() == m_strings.TAG_RCC) {
491 if (!tokens.isEmpty())
492 reader.raiseError(message: "expected <RCC> tag"_L1);
493 else
494 tokens.push(t: RccTag);
495 } else if (reader.name() == m_strings.TAG_RESOURCE) {
496 if (tokens.isEmpty() || tokens.top() != RccTag) {
497 reader.raiseError(message: "unexpected <RESOURCE> tag"_L1);
498 } else {
499 tokens.push(t: ResourceTag);
500
501 QXmlStreamAttributes attributes = reader.attributes();
502 language = QLocale::c().language();
503 territory = QLocale::c().territory();
504
505 if (attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_LANG)) {
506 QString attribute = attributes.value(qualifiedName: m_strings.ATTRIBUTE_LANG).toString();
507 QLocale lang = QLocale(attribute);
508 language = lang.language();
509 if (2 == attribute.size()) {
510 // Language only
511 territory = QLocale::AnyTerritory;
512 } else {
513 territory = lang.territory();
514 }
515 }
516
517 prefix.clear();
518 if (attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_PREFIX))
519 prefix = attributes.value(qualifiedName: m_strings.ATTRIBUTE_PREFIX).toString();
520 if (!prefix.startsWith(c: slash))
521 prefix.prepend(c: slash);
522 if (!prefix.endsWith(c: slash))
523 prefix += slash;
524 }
525 } else if (reader.name() == m_strings.TAG_FILE) {
526 if (tokens.isEmpty() || tokens.top() != ResourceTag) {
527 reader.raiseError(message: "unexpected <FILE> tag"_L1);
528 } else {
529 tokens.push(t: FileTag);
530
531 QXmlStreamAttributes attributes = reader.attributes();
532 alias.clear();
533 if (attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_ALIAS))
534 alias = attributes.value(qualifiedName: m_strings.ATTRIBUTE_ALIAS).toString();
535
536 compressAlgo = m_compressionAlgo;
537 compressLevel = m_compressLevel;
538 compressThreshold = m_compressThreshold;
539
540 QString errorString;
541 if (attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_COMPRESSALGO))
542 compressAlgo = parseCompressionAlgorithm(algo: attributes.value(qualifiedName: m_strings.ATTRIBUTE_COMPRESSALGO), errorMsg: &errorString);
543 if (errorString.isEmpty() && attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_COMPRESS)) {
544 QString value = attributes.value(qualifiedName: m_strings.ATTRIBUTE_COMPRESS).toString();
545 compressLevel = parseCompressionLevel(algo: compressAlgo, level: value, errorMsg: &errorString);
546 }
547
548 // Special case for -no-compress
549 if (m_compressLevel == -2)
550 compressAlgo = CompressionAlgorithm::None;
551
552 if (attributes.hasAttribute(qualifiedName: m_strings.ATTRIBUTE_THRESHOLD))
553 compressThreshold = attributes.value(qualifiedName: m_strings.ATTRIBUTE_THRESHOLD).toString().toInt();
554
555 if (!errorString.isEmpty())
556 reader.raiseError(message: errorString);
557 }
558 } else {
559 reader.raiseError(message: "unexpected tag: %1"_L1.arg(args: reader.name().toString()));
560 }
561 break;
562
563 case QXmlStreamReader::EndElement:
564 if (reader.name() == m_strings.TAG_RCC) {
565 if (!tokens.isEmpty() && tokens.top() == RccTag)
566 tokens.pop();
567 else
568 reader.raiseError(message: "unexpected closing tag"_L1);
569 } else if (reader.name() == m_strings.TAG_RESOURCE) {
570 if (!tokens.isEmpty() && tokens.top() == ResourceTag)
571 tokens.pop();
572 else
573 reader.raiseError(message: "unexpected closing tag"_L1);
574 } else if (reader.name() == m_strings.TAG_FILE) {
575 if (!tokens.isEmpty() && tokens.top() == FileTag)
576 tokens.pop();
577 else
578 reader.raiseError(message: "unexpected closing tag"_L1);
579 }
580 break;
581
582 case QXmlStreamReader::Characters:
583 if (reader.isWhitespace())
584 break;
585 if (tokens.isEmpty() || tokens.top() != FileTag) {
586 reader.raiseError(message: "unexpected text"_L1);
587 } else {
588 QString fileName = reader.text().toString();
589 if (fileName.isEmpty()) {
590 const QString msg = QString::fromLatin1(ba: "RCC: Warning: Null node in XML of '%1'\n").arg(a: fname);
591 m_errorDevice->write(data: msg.toUtf8());
592 }
593
594 if (alias.isNull())
595 alias = fileName;
596
597 alias = QDir::cleanPath(path: alias);
598 while (alias.startsWith(s: "../"_L1))
599 alias.remove(i: 0, len: 3);
600 alias = QDir::cleanPath(path: m_resourceRoot) + prefix + alias;
601
602 QString absFileName = fileName;
603 if (QDir::isRelativePath(path: absFileName))
604 absFileName.prepend(s: currentPath);
605 QFileInfo file(absFileName);
606 if (file.isDir()) {
607 QDir dir(file.filePath());
608 if (!alias.endsWith(c: slash))
609 alias += slash;
610
611 QStringList filePaths;
612 QDirIterator it(dir, QDirIterator::FollowSymlinks|QDirIterator::Subdirectories);
613 while (it.hasNext()) {
614 it.next();
615 if (it.fileName() == "."_L1 || it.fileName() == ".."_L1)
616 continue;
617 filePaths.append(t: it.filePath());
618 }
619
620 // make rcc output deterministic
621 std::sort(first: filePaths.begin(), last: filePaths.end());
622
623 for (const QString &filePath : filePaths) {
624 QFileInfo child(filePath);
625 const bool arc =
626 addFile(alias: alias + child.fileName(),
627 file: RCCFileInfo(child.fileName(), child, language, territory,
628 child.isDir() ? RCCFileInfo::Directory
629 : RCCFileInfo::NoFlags,
630 compressAlgo, compressLevel, compressThreshold,
631 m_noZstd));
632 if (!arc)
633 m_failedResources.push_back(t: child.fileName());
634 }
635 } else if (listMode || file.isFile()) {
636 const bool arc =
637 addFile(alias,
638 file: RCCFileInfo(alias.section(asep: slash, astart: -1),
639 file,
640 language,
641 territory,
642 RCCFileInfo::NoFlags,
643 compressAlgo,
644 compressLevel,
645 compressThreshold,
646 m_noZstd)
647 );
648 if (!arc)
649 m_failedResources.push_back(t: absFileName);
650 } else if (file.exists()) {
651 m_failedResources.push_back(t: absFileName);
652 const QString msg = QString::fromLatin1(ba: "RCC: Error in '%1': Entry '%2' is neither a file nor a directory\n")
653 .arg(args: fname, args&: fileName);
654 m_errorDevice->write(data: msg.toUtf8());
655 return false;
656 } else {
657 m_failedResources.push_back(t: absFileName);
658 const QString msg = QString::fromLatin1(ba: "RCC: Error in '%1': Cannot find file '%2'\n")
659 .arg(args: fname, args&: fileName);
660 m_errorDevice->write(data: msg.toUtf8());
661 return false;
662 }
663 }
664 break;
665
666 default:
667 break;
668 }
669 }
670
671 if (reader.hasError()) {
672 int errorLine = reader.lineNumber();
673 int errorColumn = reader.columnNumber();
674 QString errorMessage = reader.errorString();
675 QString msg = QString::fromLatin1(ba: "RCC Parse Error: '%1' Line: %2 Column: %3 [%4]\n").arg(a: fname).arg(a: errorLine).arg(a: errorColumn).arg(a: errorMessage);
676 m_errorDevice->write(data: msg.toUtf8());
677 return false;
678 }
679
680 if (m_root == nullptr) {

693bool RCCResourceLibrary::addFile(const QString &alias, const RCCFileInfo &file)
694{
695 Q_ASSERT(m_errorDevice);
696 if (file.m_fileInfo.size() > 0xffffffff) {
697 const QString msg = QString::fromLatin1(ba: "File too big: %1\n").arg(a: file.m_fileInfo.absoluteFilePath());
698 m_errorDevice->write(data: msg.toUtf8());
699 return false;
700 }
701 if (!m_root)
702 m_root = new RCCFileInfo(QString(), QFileInfo(), QLocale::C, QLocale::AnyTerritory, RCCFileInfo::Directory);
703
704 RCCFileInfo *parent = m_root;
705 const QStringList nodes = alias.split(sep: u'/');
706 for (int i = 1; i < nodes.size()-1; ++i) {
707 const QString node = nodes.at(i);
708 if (node.isEmpty())
709 continue;
710 if (!parent->m_children.contains(key: node)) {
711 RCCFileInfo *s = new RCCFileInfo(node, QFileInfo(), QLocale::C, QLocale::AnyTerritory, RCCFileInfo::Directory);
712 s->m_parent = parent;
713 parent->m_children.insert(key: node, value: s);
714 parent = s;
715 } else {
716 parent = *parent->m_children.constFind(key: node);
717 }
718 }
719
720 const QString filename = nodes.at(i: nodes.size()-1);
721 RCCFileInfo *s = new RCCFileInfo(file);
722 s->m_parent = parent;
723 auto cbegin = parent->m_children.constFind(key: filename);
724 auto cend = parent->m_children.constEnd();
725 for (auto it = cbegin; it != cend; ++it) {
726 if (it.key() == filename && it.value()->m_language == s->m_language &&
727 it.value()->m_territory == s->m_territory) {
728 for (const QString &name : std::as_const(t&: m_fileNames)) {
729 qWarning(msg: "%s: Warning: potential duplicate alias detected: '%s'",
730 qPrintable(name), qPrintable(filename));
731 }
732 break;
733 }
734 }
735 parent->m_children.insert(key: filename, value: s);
736 return true;
737}

886bool RCCResourceLibrary::output(QIODevice &outDevice, QIODevice &tempDevice, QIODevice &errorDevice)
887{
888 m_errorDevice = &errorDevice;
889
890 if (m_format == Pass2) {
891 const char pattern[] = { 'Q', 'R', 'C', '_', 'D', 'A', 'T', 'A' };
892 bool foundSignature = false;
893
894 while (true) {
895 char c;
896 for (int i = 0; i < 8; ) {
897 if (!tempDevice.getChar(c: &c)) {
898 if (foundSignature)
899 return true;
900 m_errorDevice->write(data: "No data signature found\n");
901 return false;
902 }
903 if (c == pattern[i]) {
904 ++i;
905 } else {
906 for (int k = 0; k < i; ++k)
907 outDevice.putChar(c: pattern[k]);
908 outDevice.putChar(c);
909 i = 0;
910 }
911 }
912
913 m_outDevice = &outDevice;
914 quint64 start = outDevice.pos();
915 writeDataBlobs();
916 quint64 len = outDevice.pos() - start;
917
918 tempDevice.seek(pos: tempDevice.pos() + len - 8);
919 foundSignature = true;
920 }
921 }
922
923 //write out
924 if (m_verbose)
925 m_errorDevice->write(data: "Outputting code\n");
926 if (!writeHeader()) {
927 m_errorDevice->write(data: "Could not write header\n");
928 return false;
929 }
930 if (m_root) {
931 if (!writeDataBlobs()) {
932 m_errorDevice->write(data: "Could not write data blobs.\n");
933 return false;
934 }
935 if (!writeDataNames()) {
936 m_errorDevice->write(data: "Could not write file names\n");
937 return false;
938 }
939 if (!writeDataStructure()) {
940 m_errorDevice->write(data: "Could not write data tree\n");
941 return false;
942 }
943 }
944 if (!writeInitializer()) {
945 m_errorDevice->write(data: "Could not write footer\n");
946 return false;
947 }
948 outDevice.write(data: m_out.constData(), len: m_out.size());
949 return true;
950}

1217{

1219 case C_Code:

1222 break;

1234

1243 file->m_childOffset = offset;

1260 m_root->writeDataInfo(lib&: *this);

1262 RCCFileInfo *file = pending.pop();
1263
1264 //sort by hash value for binary lookup

1266 std::sort(first: m_children.begin(), last: m_children.end(), comp: qt_rcc_compare_hash());

1313bool RCCResourceLibrary::writeInitializer()
1314{
1315 if (m_format == C_Code || m_format == Pass1) {
1316 //write("\nQT_BEGIN_NAMESPACE\n");
1317 QString initNameStr = m_initName;
1318 if (!initNameStr.isEmpty()) {
1319 initNameStr.prepend(c: u'_');
1320 auto isAsciiLetterOrNumber = [] (QChar c) -> bool {
1321 ushort ch = c.unicode();
1322 return (ch >= '0' && ch <= '9') ||
1323 (ch >= 'A' && ch <= 'Z') ||
1324 (ch >= 'a' && ch <= 'z') ||
1325 ch == '_';
1326 };
1327 for (QChar &c : initNameStr) {
1328 if (!isAsciiLetterOrNumber(c))
1329 c = u'_';
1330 }
1331 }
1332 QByteArray initName = initNameStr.toLatin1();
1333
1334 //init
1335 if (m_useNameSpace) {
1336 writeString(s: "#ifdef QT_NAMESPACE\n"
1337 "# define QT_RCC_PREPEND_NAMESPACE(name) ::QT_NAMESPACE::name\n"
1338 "# define QT_RCC_MANGLE_NAMESPACE0(x) x\n"
1339 "# define QT_RCC_MANGLE_NAMESPACE1(a, b) a##_##b\n"
1340 "# define QT_RCC_MANGLE_NAMESPACE2(a, b) QT_RCC_MANGLE_NAMESPACE1(a,b)\n"
1341 "# define QT_RCC_MANGLE_NAMESPACE(name) QT_RCC_MANGLE_NAMESPACE2( \\\n"
1342 " QT_RCC_MANGLE_NAMESPACE0(name), QT_RCC_MANGLE_NAMESPACE0(QT_NAMESPACE))\n"
1343 "#else\n"
1344 "# define QT_RCC_PREPEND_NAMESPACE(name) name\n"
1345 "# define QT_RCC_MANGLE_NAMESPACE(name) name\n"
1346 "#endif\n\n");
1347
