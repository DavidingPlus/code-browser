13enum class Target
14{
15 LTTNG,
16 ETW,
17 CTF,
18};
19
20static inline void usage(int status)
21{
22 printf(format: "Usage: tracegen <lttng|etw|ctf> <input file> <output file>\n");
23 exit(status: status);
24}
25
26static void parseArgs(int argc, char *argv[], Target *target, QString *inFile, QString *outFile)
27{
28 if (argc == 1)
29 usage(EXIT_SUCCESS);
30 if (argc != 4)
31 usage(EXIT_FAILURE);
32
33 const char *targetString = argv[1];
34
35 if (qstrcmp(str1: targetString, str2: "lttng") == 0) {
36 *target = Target::LTTNG;
37 } else if (qstrcmp(str1: targetString, str2: "etw") == 0) {
38 *target = Target::ETW;
39 } else if (qstrcmp(str1: targetString, str2: "ctf") == 0) {
40 *target = Target::CTF;
41 } else {
42 fprintf(stderr, format: "Invalid target: %s\n", targetString);
43 usage(EXIT_FAILURE);
44 }
45
46 *inFile = QLatin1StringView(argv[2]);
47 *outFile = QLatin1StringView(argv[3]);
48}
49
50int main(int argc, char *argv[])
51{
52 Target target = Target::LTTNG;
53 QString inFile;
54 QString outFile;
55
56 parseArgs(argc, argv, target: &target, inFile: &inFile, outFile: &outFile);
57
58 Provider p = parseProvider(filename: inFile);
59
60 QFile out(outFile);
61
62 if (!out.open(flags: QIODevice::WriteOnly | QIODevice::Truncate)) {
63 panic(fmt: "Cannot open '%s' for writing: %s",
64 qPrintable(outFile), qPrintable(out.errorString()));
65 }
66
67 switch (target) {
68 case Target::CTF:
69 writeCtf(device&: out, p);
70 break;
71 case Target::LTTNG:
72 writeLttng(device&: out, p);
73 break;
74 case Target::ETW:
75 writeEtw(device&: out, p);
76 break;
77 }
78
79 return 0;
80}
