21static void writeEtwMacro(QTextStream &stream, const Tracepoint::Field &field)
22{
23 const QString &name = field.name;
24
25 if (field.arrayLen > 0) {
26 for (int i = 0; i < field.arrayLen; i++) {
27 stream << "TraceLoggingValue(" << name << "[" << i << "], \"" << name << "[" << i << "]\")";
28 if (i + 1 < field.arrayLen)
29 stream << ",\n ";
30 }
31 return;
32 }
33
34 switch (field.backendType) {
35 case Tracepoint::Field::QtString:
36 stream << "TraceLoggingCountedWideString(reinterpret_cast<LPCWSTR>("
37 << name << ".utf16()), static_cast<ULONG>(" << name << ".size()), \""
38 << name << "\")";
39 return;
40 case Tracepoint::Field::QtByteArray:
41 stream << "TraceLoggingBinary(" << name << ".constData(), "
42 << name << ".size(), \"" << name << "\")";
43 return;
44 case Tracepoint::Field::QtUrl:
45 stream << "TraceLoggingValue(" << name << ".toEncoded().constData(), \"" << name << "\")";
46 return;
47 case Tracepoint::Field::QtRect:
48 case Tracepoint::Field::QtRectF:
49 stream << "TraceLoggingValue(" << name << ".x(), \"x\"), "
50 << "TraceLoggingValue(" << name << ".y(), \"y\"), "
51 << "TraceLoggingValue(" << name << ".width(), \"width\"), "
52 << "TraceLoggingValue(" << name << ".height(), \"height\")";
53 return;
54 case Tracepoint::Field::QtSize:
55 case Tracepoint::Field::QtSizeF:
56 stream << "TraceLoggingValue(" << name << ".width(), \"width\"), "
57 << "TraceLoggingValue(" << name << ".height(), \"height\")";
58 return;
59 case Tracepoint::Field::Pointer:
60 stream << "TraceLoggingPointer(" << name << ", \"" << name << "\")";
61 return;
62 case Tracepoint::Field::Unknown:
63 // Write down the previously stringified data (like we do for QString).
64 // The string is already created in writeWrapper().
65 // Variable name is name##Str.
66 stream << "TraceLoggingCountedWideString(reinterpret_cast<LPCWSTR>(" << name
67 << "Str.utf16()), static_cast<ULONG>(" << name << "Str.size()), \"" << name << "\")";
68 return;
69 case Tracepoint::Field::EnumeratedType:
70 case Tracepoint::Field::FlagType:
71 stream << "TraceLoggingString(trace_convert_" << typeToTypeName(type: field.paramType) << "(" << name << ").toUtf8().constData(), \"" << name << "\")";
72 return;
73 default:
74 break;
75 }
76
77 stream << "TraceLoggingValue(" << name << ", \"" << name << "\")";
78}

80static QString createGuid(const QUuid &uuid)
81{
82 QString guid;
83
84 QTextStream stream(&guid);
85
86 Qt::hex(s&: stream);
87
88 stream << "("
89 << "0x" << uuid.data1 << ", "
90 << "0x" << uuid.data2 << ", "
91 << "0x" << uuid.data3 << ", "
92 << "0x" << uuid.data4[0] << ", "
93 << "0x" << uuid.data4[1] << ", "
94 << "0x" << uuid.data4[2] << ", "
95 << "0x" << uuid.data4[3] << ", "
96 << "0x" << uuid.data4[4] << ", "
97 << "0x" << uuid.data4[5] << ", "
98 << "0x" << uuid.data4[6] << ", "
99 << "0x" << uuid.data4[7]
100 << ")";
101
102 return guid;
103}

105static void writePrologue(QTextStream &stream, const QString &fileName, const Provider &provider)
106{
107 QUuid uuid = QUuid::createUuidV5(ns: QUuid(), baseData: provider.name.toLocal8Bit());
108
109 const QString providerV = providerVar(providerName: provider.name);
110 const QString guard = includeGuard(filename: fileName);
111 const QString guid = createGuid(uuid);
112 const QString guidString = uuid.toString();
113
114 stream << "#ifndef " << guard << "\n"
115 << "#define " << guard << "\n"
116 << "\n"
117 << "#include <qt_windows.h>\n"
118 << "#include <TraceLoggingProvider.h>\n"
119 << "\n";
120
121 /* TraceLogging API macros cannot deal with UTF8
122 * source files, so we work around it like this
123 */
124 stream << "#undef _TlgPragmaUtf8Begin\n"
125 "#undef _TlgPragmaUtf8End\n"
126 "#define _TlgPragmaUtf8Begin\n"
127 "#define _TlgPragmaUtf8End\n";
128
129 stream << "\n";
130 stream << qtHeaders();
131 stream << "\n";
132
133 if (!provider.prefixText.isEmpty())
134 stream << provider.prefixText.join(sep: u'\n') << "\n\n";
135
136 stream << "#ifdef TRACEPOINT_DEFINE\n"
137 << "/* " << guidString << " */\n"
138 << "TRACELOGGING_DEFINE_PROVIDER(" << providerV << ", \""
139 << provider.name <<"\", " << guid << ");\n\n";
140
141 stream << "static inline void registerProvider()\n"
142 << "{\n"
143 << " TraceLoggingRegister(" << providerV << ");\n"
144 << "}\n\n";
145
146 stream << "static inline void unregisterProvider()\n"
147 << "{\n"
148 << " TraceLoggingUnregister(" << providerV << ");\n"
149 << "}\n";
150
151 stream << "Q_CONSTRUCTOR_FUNCTION(registerProvider)\n"
152 << "Q_DESTRUCTOR_FUNCTION(unregisterProvider)\n\n";
153
154 stream << "#else\n"
155 << "TRACELOGGING_DECLARE_PROVIDER(" << providerV << ");\n"
156 << "#endif // TRACEPOINT_DEFINE\n\n";
157}

165static void writeWrapper(QTextStream &stream, const Provider &provider, const Tracepoint &tracepoint,
166 const QString &providerName)
167{
168 const QString argList = formatFunctionSignature(args: tracepoint.args);
169 const QString paramList = formatParameterList(provider, args: tracepoint.args, fields: tracepoint.fields, type: ETW);
170 const QString &name = tracepoint.name;
171 const QString includeGuard = QStringLiteral("TP_%1_%2").arg(a: providerName).arg(a: name).toUpper();
172 const QString provar = providerVar(providerName);
173
174 stream << "\n";
175
176 stream << "inline void trace_" << name << "(" << argList << ")\n"
177 << "{\n";
178
179 // Convert all unknown types to QString's using QDebug.
180 // Note the naming convention: it's field.name##Str
181 for (const Tracepoint::Field &field : tracepoint.fields) {
182 if (field.backendType == Tracepoint::Field::Unknown) {
183 stream << " const QString " << field.name << "Str = QDebug::toString(" << field.name
184 << ");\n";
185 }
186 }
187 stream << " TraceLoggingWrite(" << provar << ", \"" << name << "\"";
188
189 for (const Tracepoint::Field &field : tracepoint.fields) {
190 stream << ",\n";
191 stream << " ";
192 writeEtwMacro(stream, field);
193 }
194
195 stream << ");\n"
196 << "}\n\n";
197
198 stream << "inline void do_trace_" << name << "(" << argList << ")\n"
199 << "{\n"
200 << " trace_" << name << "(" << paramList << ");\n"
201 << "}\n";
202
203 stream << "inline bool trace_" << name << "_enabled()\n"
204 << "{\n"
205 << " return TraceLoggingProviderEnabled(" << provar << ", 0, 0);\n"
206 << "}\n";
207}

209static void writeEnumConverter(QTextStream &stream, const TraceEnum &enumeration)
210{
211 stream << "inline QString trace_convert_" << typeToTypeName(type: enumeration.name) << "(" << enumeration.name << " val)\n";
212 stream << "{\n";
213 for (const auto &v : enumeration.values) {
214 if (v.range != 0) {
215 stream << " if (val >= " << v.value << " && val <= " << v.range << ")\n"
216 << " return QStringLiteral(\"" << v.name << " + \") + QString::number((int)val - " << v.value << ");\n";
217 }
218 }
219 stream << "\n QString ret;\n switch (val) {\n";
220
221 QList<int> handledValues;
222 for (const auto &v : enumeration.values) {
223 if (v.range == 0 && !handledValues.contains(t: v.value)) {
224 stream << " case " << v.value << ": ret = QStringLiteral(\""
225 << aggregateListValues(value: v.value, list: enumeration.values) << "\"); break;\n";
226 handledValues.append(t: v.value);
227 }
228 }
229
230 stream << " }\n return ret;\n}\n";
231}

233static void writeFlagConverter(QTextStream &stream, const TraceFlags &flag)
234{
235 stream << "inline QString trace_convert_" << typeToTypeName(type: flag.name) << "(" << flag.name << " val)\n";
236 stream << "{\n QString ret;\n";
237 for (const auto &v : flag.values) {
238 if (v.value == 0) {
239 stream << " if (val == 0)\n return QStringLiteral(\""
240 << aggregateListValues(value: v.value, list: flag.values) << "\");\n";
241 break;
242 }
243 }
244 QList<int> handledValues;
245 for (const auto &v : flag.values) {
246 if (v.value != 0 && !handledValues.contains(t: v.value)) {
247 stream << " if (val & " << (1 << (v.value - 1))
248 << ") { if (ret.length()) ret += QLatin1Char(\'|\'); ret += QStringLiteral(\"" << v.name << "\"); }\n";
249 handledValues.append(t: v.value);
250 }
251 }
252 stream << " return ret;\n}\n";
253}

255static void writeTracepoints(QTextStream &stream, const Provider &provider)
256{
257 if (provider.tracepoints.isEmpty())
258 return;
259
260 const QString includeGuard = QStringLiteral("TP_%1_PROVIDER").arg(a: provider.name).toUpper();
261
262 stream << "#if !defined(" << includeGuard << ") && !defined(TRACEPOINT_DEFINE)\n"
263 << "#define " << includeGuard << "\n"
264 << "QT_BEGIN_NAMESPACE\n"
265 << "namespace QtPrivate {\n";
266
267 for (const auto &enumeration : provider.enumerations)
268 writeEnumConverter(stream, enumeration);
269
270 for (const auto &flag : provider.flags)
271 writeFlagConverter(stream, flag);
272
273 for (const Tracepoint &t : provider.tracepoints)
274 writeWrapper(stream, provider, tracepoint: t, providerName: provider.name);
275
276 stream << "} // namespace QtPrivate\n"
277 << "QT_END_NAMESPACE\n"
278 << "#endif // " << includeGuard << "\n\n";
279}

281void writeEtw(QFile &file, const Provider &provider)
282{
283 QTextStream stream(&file);
284
285 const QString fileName = QFileInfo(file.fileName()).fileName();
286
287 writePrologue(stream, fileName, provider);
288 writeTracepoints(stream, provider);
289 writeEpilogue(stream, fileName);
290}
