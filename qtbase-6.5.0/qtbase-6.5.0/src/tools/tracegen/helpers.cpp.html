9QString typeToTypeName(const QString &name)
10{
11 QString ret = name;
12 return ret.replace(QStringLiteral("::"), QStringLiteral("_"));
13}

15QString includeGuard(const QString &filename)
16{
17 QString guard = filename.toUpper();
18
19 for (int i = 0; i < guard.size(); ++i) {
20 if (!guard.at(i).isLetterOrNumber())
21 guard[i] = u'_';
22 }
23
24 return guard;
25}

27template<typename T>
28static QString joinArguments(const QList<Tracepoint::Argument> &args, T joinFunction)
29{
30 QString ret;
31 bool first = true;
32
33 for (const Tracepoint::Argument &arg : args) {
34 if (!first)
35 ret += ", "_L1;
36
37 ret += joinFunction(arg);
38
39 first = false;
40 }
41
42 return ret;
43}

52QString formatParameterList(const Provider &provider, const QList<Tracepoint::Argument> &args, const QList<Tracepoint::Field> &fields, ParamType type)
53{
54 if (type == LTTNG) {
55 QString ret;
56
57 for (int i = 0; i < args.size(); i++) {
58 const Tracepoint::Argument &arg = args[i];
59 const Tracepoint::Field &field = fields[i];
60 if (field.backendType == Tracepoint::Field::FlagType)
61 ret += ", trace_convert_"_L1 + typeToTypeName(name: arg.type) + "("_L1 + arg.name + ")"_L1;
62 else
63 ret += ", "_L1 + arg.name;
64 }
65 return ret;
66 }
67
68 auto findEnumeration = [](const QList<TraceEnum> &enums, const QString &name) {
69 for (const auto &e : enums) {
70 if (e.name == name)
71 return e;
72 }
73 return TraceEnum();
74 };
75
76 if (type == CTF) {
77 QString ret;
78
79 for (int i = 0; i < args.size(); i++) {
80 const Tracepoint::Argument &arg = args[i];
81 const Tracepoint::Field &field = fields[i];
82 if (arg.arrayLen > 1) {
83 ret += ", trace::toByteArrayFromArray("_L1 + arg.name + ", "_L1 + QString::number(arg.arrayLen) + ") "_L1;
84 } else if (field.backendType == Tracepoint::Field::EnumeratedType) {
85 const TraceEnum &e = findEnumeration(provider.enumerations, arg.type);
86 QString integerType;
87 if (e.valueSize == 8)
88 integerType = QStringLiteral("quint8");
89 else if (e.valueSize == 16)
90 integerType = QStringLiteral("quint16");
91 else
92 integerType = QStringLiteral("quint32");
93 ret += ", trace::toByteArrayFromEnum<"_L1 + integerType + ">("_L1 + arg.name + ")"_L1;
94 } else if (field.backendType == Tracepoint::Field::FlagType) {
95 ret += ", trace::toByteArrayFromFlags("_L1 + arg.name + ")"_L1;
96 } else if (field.backendType == Tracepoint::Field::String) {
97 ret += ", trace::toByteArrayFromCString("_L1 + arg.name + ")"_L1;
98 } else {
99 ret += ", "_L1 + arg.name;
100 }
101 }
102 return ret;
103 }
104
105 return joinArguments(args, joinFunction: [](const Tracepoint::Argument &arg) { return arg.name; });
106}
