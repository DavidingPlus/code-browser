15static void writeCtfMacro(QTextStream &stream, const Provider &provider, const Tracepoint::Field &field)
16{
17 const QString &paramType = field.paramType;
18 const QString &name = field.name;
19 const QString &seqLen = field.seqLen;
20 const int arrayLen = field.arrayLen;
21
22 if (arrayLen > 0) {
23 stream << "ctf_array(" <<paramType << ", "
24 << name << ", " << name << ", " << arrayLen << ")";
25 return;
26 }
27
28 switch (field.backendType) {
29 case Tracepoint::Field::Sequence:
30 stream << "ctf_sequence(" << paramType
31 << ", " << name << ", " << name
32 << ", unsigned int, " << seqLen << ")";
33 return;
34 case Tracepoint::Field::Boolean:
35 case Tracepoint::Field::Integer:
36 stream << "ctf_integer(" << paramType << ", " << name << ", " << name << ")";
37 return;
38 case Tracepoint::Field::IntegerHex:
39 case Tracepoint::Field::Pointer:
40 stream << "ctf_integer_hex(" << paramType << ", " << name << ", " << name << ")";
41 return;
42 case Tracepoint::Field::Float:
43 stream << "ctf_float(" << paramType << ", " << name << ", " << name << ")";
44 return;
45 case Tracepoint::Field::String:
46 stream << "ctf_string(" << name << ", " << name << ")";
47 return;
48 case Tracepoint::Field::QtString:
49 stream << "ctf_string(" << name << ", " << name << ".toUtf8().constData())";
50 return;
51 case Tracepoint::Field::QtByteArray:
52 stream << "ctf_sequence(const char, " << name << ", "
53 << name << ".constData(), unsigned int, " << name << ".size())";
54 return;
55 case Tracepoint::Field::QtUrl:
56 stream << "ctf_string(" << name << ", " << name << ".toString().toUtf8().constData())";
57 return;
58 case Tracepoint::Field::QtRect:
59 stream << "ctf_integer(int, x, " << name << ".x()) "
60 << "ctf_integer(int, y, " << name << ".y()) "
61 << "ctf_integer(int, width, " << name << ".width()) "
62 << "ctf_integer(int, height, " << name << ".height()) ";
63 return;
64 case Tracepoint::Field::QtSizeF:
65 stream << "ctf_float(int, width, " << name << ".width()) "
66 << "ctf_float(int, height, " << name << ".height()) ";
67 return;
68 case Tracepoint::Field::QtRectF:
69 stream << "ctf_float(int, x, " << name << ".x()) "
70 << "ctf_float(int, y, " << name << ".y()) "
71 << "ctf_float(int, width, " << name << ".width()) "
72 << "ctf_float(int, height, " << name << ".height()) ";
73 return;
74 case Tracepoint::Field::QtSize:
75 stream << "ctf_integer(int, width, " << name << ".width()) "
76 << "ctf_integer(int, height, " << name << ".height()) ";
77 return;
78 case Tracepoint::Field::EnumeratedType:
79 stream << "ctf_enum(" << provider.name << ", " << typeToTypeName(type: paramType) << ", int, " << name << ", " << name << ") ";
80 return;
81 case Tracepoint::Field::FlagType:
82 stream << "ctf_sequence(const char , " << name << ", "
83 << name << ".constData(), unsigned int, " << name << ".size())";
84 return;
85 case Tracepoint::Field::Unknown:
86 panic(fmt: "Cannot deduce CTF type for '%s %s", qPrintable(paramType), qPrintable(name));
87 break;
88 }
89}

91static void writePrologue(QTextStream &stream, const QString &fileName, const Provider &provider)
92{
93 const QString guard = includeGuard(filename: fileName);
94
95 stream << "#undef TRACEPOINT_PROVIDER\n";
96 stream << "#define TRACEPOINT_PROVIDER " << provider.name << "\n";
97 stream << "\n";
98
99 // include prefix text or qt headers only once
100 stream << "#if !defined(" << guard << ")\n";
101 stream << qtHeaders();
102 stream << "\n";
103 if (!provider.prefixText.isEmpty())
104 stream << provider.prefixText.join(sep: u'\n') << "\n\n";
105 stream << "#endif\n\n";
106
107 /* the first guard is the usual one, the second is required
108 * by LTTNG to force the re-evaluation of TRACEPOINT_* macros
109 */
110 stream << "#if !defined(" << guard << ") || defined(TRACEPOINT_HEADER_MULTI_READ)\n";
111
112 stream << "#define " << guard << "\n\n"
113 << "#undef TRACEPOINT_INCLUDE\n"
114 << "#define TRACEPOINT_INCLUDE \"" << fileName << "\"\n\n";
115
116 stream << "#include <lttng/tracepoint.h>\n\n";
117
118 const QString namespaceGuard = guard + QStringLiteral("_USE_NAMESPACE");
119 stream << "#if !defined(" << namespaceGuard << ")\n"
120 << "#define " << namespaceGuard << "\n"
121 << "QT_USE_NAMESPACE\n"
122 << "#endif // " << namespaceGuard << "\n\n";
123}

133static void writeWrapper(QTextStream &stream,
134 const Tracepoint &tracepoint, const Provider &provider)
135{
136 const QString argList = formatFunctionSignature(args: tracepoint.args);
137 const QString paramList = formatParameterList(provider, args: tracepoint.args, fields: tracepoint.fields, type: LTTNG);
138 const QString &name = tracepoint.name;
139 const QString includeGuard = QStringLiteral("TP_%1_%2").arg(a: provider.name).arg(a: name).toUpper();
140
141 /* prevents the redefinion of the inline wrapper functions
142 * once LTTNG recursively includes this header file
143 */
144 stream << "\n"
145 << "#ifndef " << includeGuard << "\n"
146 << "#define " << includeGuard << "\n"
147 << "QT_BEGIN_NAMESPACE\n"
148 << "namespace QtPrivate {\n";
149
150 stream << "inline void trace_" << name << "(" << argList << ")\n"
151 << "{\n"
152 << " tracepoint(" << provider.name << ", " << name << paramList << ");\n"
153 << "}\n";
154
155 stream << "inline void do_trace_" << name << "(" << argList << ")\n"
156 << "{\n"
157 << " do_tracepoint(" << provider.name << ", " << name << paramList << ");\n"
158 << "}\n";
159
160 stream << "inline bool trace_" << name << "_enabled()\n"
161 << "{\n"
162 << " return tracepoint_enabled(" << provider.name << ", " << name << ");\n"
163 << "}\n";
164
165 stream << "} // namespace QtPrivate\n"
166 << "QT_END_NAMESPACE\n"
167 << "#endif // " << includeGuard << "\n\n";
168}

170static void writeTracepoint(QTextStream &stream, const Provider &provider,
171 const Tracepoint &tracepoint, const QString &providerName)
172{
173 stream << "TRACEPOINT_EVENT(\n"
174 << " " << providerName << ",\n"
175 << " " << tracepoint.name << ",\n"
176 << " TP_ARGS(";
177
178 const char *comma = nullptr;
179
180 for (int i = 0; i < tracepoint.args.size(); i++) {
181 const auto &arg = tracepoint.args[i];
182 const auto &field = tracepoint.fields[i];
183 if (field.backendType == Tracepoint::Field::FlagType)
184 stream << comma << "QByteArray, " << arg.name;
185 else
186 stream << comma << arg.type << ", " << arg.name;
187 comma = ", ";
188 }
189
190 stream << "),\n"
191 << " TP_FIELDS(";
192
193 const char *newline = nullptr;
194
195 for (const Tracepoint::Field &f : tracepoint.fields) {
196 stream << newline;
197 writeCtfMacro(stream, provider, field: f);
198 newline = "\n ";
199 }
200
201 stream << ")\n)\n\n";
202}

204static void writeEnums(QTextStream &stream, const Provider &provider)
205{
206 for (const auto &e : provider.enumerations) {
207 stream << "TRACEPOINT_ENUM(\n"
208 << " " << provider.name << ",\n"
209 << " " << typeToTypeName(type: e.name) << ",\n"
210 << " TP_ENUM_VALUES(\n";
211 QList<int> handledValues;
212 for (const auto &v : e.values) {
213 if (v.range > 0) {
214 stream << " ctf_enum_range(\"" << v.name << "\", " << v.value << ", " << v.range << ")\n";
215 } else if (!handledValues.contains(t: v.value)) {
216 stream << " ctf_enum_value(\"" << aggregateListValues(value: v.value, list: e.values) << "\", " << v.value << ")\n";
217 handledValues.append(t: v.value);
218 }
219 }
220 stream << " )\n)\n\n";
221 }
222}

224static void writeFlags(QTextStream &stream, const Provider &provider)
225{
226 for (const auto &f : provider.flags) {
227 stream << "TRACEPOINT_ENUM(\n"
228 << " " << provider.name << ",\n"
229 << " " << typeToTypeName(type: f.name) << ",\n"
230 << " TP_ENUM_VALUES(\n";
231 QList<int> handledValues;
232 for (const auto &v : f.values) {
233 if (!handledValues.contains(t: v.value)) {
234 stream << " ctf_enum_value(\"" << aggregateListValues(value: v.value, list: f.values) << "\", " << v.value << ")\n";
235 handledValues.append(t: v.value);
236 }
237 }
238 stream << " )\n)\n\n";
239 }
240
241 // converters
242 const QString includeGuard = QStringLiteral("TP_%1_CONVERTERS").arg(a: provider.name).toUpper();
243 stream << "\n"
244 << "#ifndef " << includeGuard << "\n"
245 << "#define " << includeGuard << "\n";
246 stream << "QT_BEGIN_NAMESPACE\n";
247 stream << "namespace QtPrivate {\n";
248 for (const auto &f : provider.flags) {
249 stream << "inline QByteArray trace_convert_" << typeToTypeName(type: f.name) << "(" << f.name << " val)\n";
250 stream << "{\n";
251 stream << " QByteArray ret;\n";
252 stream << " if (val == 0) { ret.append((char)0); return ret; }\n";
253
254 for (const auto &v : f.values) {
255 if (!v.value)
256 continue;
257 stream << " if (val & " << (1 << (v.value - 1)) << ") { ret.append((char)" << v.value << "); };\n";
258 }
259 stream << " return ret;\n";
260 stream << "}\n";
261
262 }
263 stream << "} // namespace QtPrivate\n"
264 << "QT_END_NAMESPACE\n\n"
265 << "#endif // " << includeGuard << "\n\n";
266}

268static void writeTracepoints(QTextStream &stream, const Provider &provider)
269{
270 for (const Tracepoint &t : provider.tracepoints) {
271 writeTracepoint(stream, provider, tracepoint: t, providerName: provider.name);
272 writeWrapper(stream, tracepoint: t, provider);
273 }

276void writeLttng(QFile &file, const Provider &provider)
277{
278 QTextStream stream(&file);
279
280 const QString fileName = QFileInfo(file.fileName()).fileName();
281
282 writePrologue(stream, fileName, provider);
283 writeEnums(stream, provider);
284 writeFlags(stream, provider);
285 writeTracepoints(stream, provider);
286 writeEpilogue(stream, fileName);
287}
