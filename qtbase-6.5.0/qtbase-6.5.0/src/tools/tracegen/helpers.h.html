23template <typename T>
24static QString aggregateListValues(int value, const QList<T> &list)
25{
26 QStringList values;
27 for (auto l : list) {
28 if (l.value == value)
29 values << l.name;
30 }
31 return values.join(sep: QLatin1Char('_'));
32}
