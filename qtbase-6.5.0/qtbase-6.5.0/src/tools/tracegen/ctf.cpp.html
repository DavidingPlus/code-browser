16static void writePrologue(QTextStream &stream, const QString &fileName, const Provider &provider)
17{
18 const QString guard = includeGuard(filename: fileName);
19
20 // include prefix text or qt headers only once
21 stream << "#if !defined(" << guard << ")\n";
22 stream << qtHeaders();
23 stream << "\n";
24 if (!provider.prefixText.isEmpty())
25 stream << provider.prefixText.join(sep: u'\n') << "\n\n";
26 stream << "#endif\n\n";
27
28 /* the first guard is the usual one, the second is required
29 * by LTTNG to force the re-evaluation of TRACEPOINT_* macros
30 */
31 stream << "#if !defined(" << guard << ") || defined(TRACEPOINT_HEADER_MULTI_READ)\n";
32
33 stream << "#define " << guard << "\n\n"
34 << "#undef TRACEPOINT_INCLUDE\n"
35 << "#define TRACEPOINT_INCLUDE \"" << fileName << "\"\n\n";
36
37 stream << "#include <private/qctf_p.h>\n\n";
38
39 const QString namespaceGuard = guard + QStringLiteral("_USE_NAMESPACE");
40 stream << "#if !defined(" << namespaceGuard << ")\n"
41 << "#define " << namespaceGuard << "\n"
42 << "QT_USE_NAMESPACE\n"
43 << "#endif // " << namespaceGuard << "\n\n";
44
45 stream << "TRACEPOINT_PROVIDER(" << provider.name << ");\n\n";
46}

48static void writeEpilogue(QTextStream &stream, const QString &fileName)
49{
50 stream << "\n";
51 stream << "#endif // " << includeGuard(filename: fileName) << "\n"
52 << "#include <private/qtrace_p.h>\n";
53}

55static void writeWrapper(QTextStream &stream,
56 const Tracepoint &tracepoint, const Provider &provider)
57{
58 const QString argList = formatFunctionSignature(args: tracepoint.args);
59 const QString paramList = formatParameterList(provider, args: tracepoint.args, fields: tracepoint.fields, type: CTF);
60 const QString &name = tracepoint.name;
61 const QString includeGuard = QStringLiteral("TP_%1_%2").arg(a: provider.name).arg(a: name).toUpper();
62
63 /* prevents the redefinion of the inline wrapper functions
64 */
65 stream << "\n"
66 << "#ifndef " << includeGuard << "\n"
67 << "#define " << includeGuard << "\n"
68 << "QT_BEGIN_NAMESPACE\n"
69 << "namespace QtPrivate {\n";
70
71 stream << "inline void trace_" << name << "(" << argList << ")\n"
72 << "{\n"
73 << " tracepoint(" << provider.name << ", " << name << paramList << ");\n"
74 << "}\n";
75
76 stream << "inline void do_trace_" << name << "(" << argList << ")\n"
77 << "{\n"
78 << " do_tracepoint(" << provider.name << ", " << name << paramList << ");\n"
79 << "}\n";
80
81 stream << "inline bool trace_" << name << "_enabled()\n"
82 << "{\n"
83 << " return tracepoint_enabled(" << provider.name << ", " << name << ");\n"
84 << "}\n";
85
86 stream << "} // namespace QtPrivate\n"
87 << "QT_END_NAMESPACE\n"
88 << "#endif // " << includeGuard << "\n\n";
89}

92static void writeMetadataGenerators(QTextStream &stream)
93{
94 stream << R"CPP(
95template <typename T>
96inline QString integerToMetadata(const QString &name)
97{
98 QString ret;
99 if (!std::is_signed<T>().value)
100 ret += QLatin1Char('u');
101 if (sizeof(T) == 8)
102 ret += QStringLiteral("int64_t ");
103 else if (sizeof(T) == 4)
104 ret += QStringLiteral("int32_t ");
105 else if (sizeof(T) == 2)
106 ret += QStringLiteral("int16_t ");
107 else if (sizeof(T) == 1)
108 ret += QStringLiteral("int8_t ");
109 ret += name + QLatin1Char(';');
110 return ret;
111}
112
113template <typename T>
114inline QString integerArrayToMetadata(const QString &size, const QString &name)
115{
116 QString ret;
117 if (!std::is_signed<T>().value)
118 ret += QLatin1Char('u');
119 if (sizeof(T) == 8)
120 ret += QStringLiteral("int64_t ");
121 else if (sizeof(T) == 4)
122 ret += QStringLiteral("int32_t ");
123 else if (sizeof(T) == 2)
124 ret += QStringLiteral("int16_t ");
125 else if (sizeof(T) == 1)
126 ret += QStringLiteral("int8_t ");
127 ret += name + QLatin1Char('[') + size + QStringLiteral("];");
128 return ret;
129}
130
131template <typename T>
132inline QString floatToMetadata(const QString &name)
133{
134 QString ret;
135 if (sizeof(T) == 8)
136 ret += QStringLiteral("double ");
137 else if (sizeof(T) == 4)
138 ret += QStringLiteral("float ");
139 ret += name + QLatin1Char(';');
140 return ret;
141}
142
143template <typename T>
144inline QString floatArrayToMetadata(const QString &size, const QString &name)
145{
146 QString ret;
147 if (sizeof(T) == 8)
148 ret += QStringLiteral("double ");
149 else if (sizeof(T) == 4)
150 ret += QStringLiteral("float ");
151 ret += name + QLatin1Char('[') + size + QStringLiteral("];");
152 return ret + QLatin1Char(';');
153}
154
155inline QString pointerToMetadata(const QString &name)
156{
157 QString ret;
158 if (QT_POINTER_SIZE == 8)
159 ret += QStringLiteral("intptr64_t ");
160 else if (QT_POINTER_SIZE == 4)
161 ret += QStringLiteral("intptr32_t ");

169static void writeTracepoint(QTextStream &stream,
170 const Tracepoint &tracepoint, const QString &providerName)
171{
172 stream << "TRACEPOINT_EVENT(\n"
173 << " " << providerName << ",\n"
174 << " " << tracepoint.name << ",\n";
175
176 const auto checkUnknownArgs = [](const Tracepoint &tracepoint) {
177 for (auto &field : tracepoint.fields) {
178 if (field.backendType == Tracepoint::Field::Unknown)
179 return true;
180 }
181 return false;
182 };
183
184 const auto formatType = [](const QString &type) {
185 QString ret = type;
186 if (type.endsWith(c: QLatin1Char('*')) || type.endsWith(c: QLatin1Char('&')))
187 ret = type.left(n: type.length() - 1).simplified();
188 if (ret.startsWith(QStringLiteral("const")))
189 ret = ret.right(n: ret.length() - 6).simplified();
190 return typeToTypeName(type: ret);
191 };
192 QString eventSize;
193 bool variableSize = false;
194 const bool emptyMetadata = checkUnknownArgs(tracepoint) || tracepoint.args.size() == 0;
195 if (!emptyMetadata) {
196 for (int i = 0; i < tracepoint.args.size(); i++) {
197 auto &arg = tracepoint.args[i];
198 auto &field = tracepoint.fields[i];
199 if (i > 0) {
200 stream << " + QStringLiteral(\"\\n\\\n \") + ";
201 eventSize += QStringLiteral(" + ");
202 }
203 const bool array = field.arrayLen > 0;
204 switch (field.backendType) {
205 case Tracepoint::Field::Boolean: {
206 stream << "QStringLiteral(\"Boolean " << arg.name << ";\")";
207 eventSize += QStringLiteral("sizeof(bool)");
208 } break;
209 case Tracepoint::Field::Integer: {
210 if (array) {
211 stream << "integerArrayToMetadata<" << formatType(arg.type)
212 << ">(QStringLiteral(\"" << field.arrayLen << "\"), QStringLiteral(\""
213 << arg.name << "\"))";
214 } else {
215 stream << "integerToMetadata<" << formatType(arg.type) << ">(QStringLiteral(\""
216 << arg.name << "\"))";
217 }
218 eventSize += QStringLiteral("sizeof(") + formatType(arg.type) + QStringLiteral(")");
219 if (array)
220 eventSize += QStringLiteral(" * ") + QString::number(field.arrayLen);
221 } break;
222 case Tracepoint::Field::Pointer:
223 case Tracepoint::Field::IntegerHex: {
224 stream << "pointerToMetadata(QStringLiteral(\"" << formatType(arg.type) << "_"
225 << arg.name << "\"))";
226 eventSize += QStringLiteral("QT_POINTER_SIZE");
227 } break;
228 case Tracepoint::Field::Float: {
229 if (array) {
230 stream << "floatArrayToMetadata<" << formatType(arg.type)
231 << ">(QStringLiteral(\"" << field.arrayLen << "\"), QStringLiteral(\""
232 << arg.name << "\"))";
233 } else {
234 stream << "floatToMetadata<" << formatType(arg.type) << ">(QStringLiteral(\""
235 << arg.name << "\"))";
236 }
237 eventSize += QStringLiteral("sizeof(") + formatType(arg.type) + QStringLiteral(")");
238 if (array)
239 eventSize += QStringLiteral(" * ") + QString::number(field.arrayLen);
240 } break;
241 case Tracepoint::Field::QtUrl:
242 case Tracepoint::Field::QtString:
243 case Tracepoint::Field::String: {
244 stream << "QStringLiteral(\"string " << arg.name << ";\")";
245 eventSize += QStringLiteral("1");
246 variableSize = true;
247 } break;
248 case Tracepoint::Field::QtRect: {
249 stream << "QStringLiteral(\"int32_t QRect_" << arg.name << "_x;\\n\\\n \")";
250 stream << " + QStringLiteral(\"int32_t QRect_" << arg.name << "_y;\\n\\\n \")";
251 stream << " + QStringLiteral(\"int32_t QRect_" << arg.name << "_width;\\n\\\n \")";
252 stream << " + QStringLiteral(\"int32_t QRect_" << arg.name << "_height;\\n\\\n \")";
253 eventSize += QStringLiteral("16");
254 } break;
255 case Tracepoint::Field::QtSize: {
256 stream << "QStringLiteral(\"int32_t QSize_" << arg.name << "_width;\\n\\\n \")";
257 stream << " + QStringLiteral(\"int32_t QSize_" << arg.name << "_height;\\n\\\n \")";
258 eventSize += QStringLiteral("8");
259 } break;
260 case Tracepoint::Field::QtRectF: {
261 stream << "QStringLiteral(\"float QRectF_" << arg.name << "_x;\\n\\\n \")";
262 stream << " + QStringLiteral(\"float QRectF_" << arg.name << "_y;\\n\\\n \")";
263 stream << " + QStringLiteral(\"float QRectF_" << arg.name << "_width;\\n\\\n \")";
264 stream << " + QStringLiteral(\"float QRectF_" << arg.name << "_height;\\n\\\n \")";
265 eventSize += QStringLiteral("16");
266 } break;
267 case Tracepoint::Field::QtSizeF: {
268 stream << "QStringLiteral(\"float QSizeF_" << arg.name << "_width;\\n\\\n \")";
269 stream << " + QStringLiteral(\"float QSizeF_" << arg.name << "_height;\\n\\\n \")";
270 eventSize += QStringLiteral("8");
271 } break;
272 case Tracepoint::Field::Unknown:
273 break;
274 case Tracepoint::Field::EnumeratedType: {
275 stream << "QStringLiteral(\"" << typeToTypeName(type: arg.type) << " " << arg.name << ";\")";
276 eventSize += QString::number(field.enumValueSize / 8);
277 variableSize = true;
278 } break;
279 case Tracepoint::Field::FlagType: {
280 stream << "QStringLiteral(\"uint8_t " << arg.name << "_length;\\n\\\n ";
281 stream << typeToTypeName(type: arg.type) << " " << arg.name << "[" << arg.name << "_length];\")";
282 eventSize += QStringLiteral("2");
283 variableSize = true;
284 } break;
285 case Tracepoint::Field::QtByteArray:
286 case Tracepoint::Field::Sequence:
287 panic(fmt: "Unhandled type '%s %s", qPrintable(arg.type), qPrintable(arg.name));
288 break;
289 }
290 }
291 }
292
293 if (emptyMetadata)
294 stream << "{},\n";
295 else
296 stream << ",\n";
297 if (eventSize.length())
298 stream << eventSize << ", \n";
299 else
300 stream << "0, \n";
301 stream << (variableSize ? "true" : "false") << "\n";
302 stream << ")\n\n";
303}

305static void writeTracepoints(QTextStream &stream, const Provider &provider)
306{
307 for (const Tracepoint &t : provider.tracepoints) {
308 writeTracepoint(stream, tracepoint: t, providerName: provider.name);
309 writeWrapper(stream, tracepoint: t, provider);
310 }
311}

313static void writeEnums(QTextStream &stream, const Provider &provider)
314{
315 for (const auto &e : provider.enumerations) {
316 QString name = e.name;
317 name.replace(QStringLiteral("::"), QStringLiteral("_"));
318 stream << "TRACEPOINT_METADATA(" << provider.name << ", " << name << ", \n";
319 stream << "QStringLiteral(\"typealias enum : integer { size = " << e.valueSize << "; } {\\n\\\n";
320
321 const auto values = e.values;
322 QList<int> handledValues;
323
324 for (const auto &v : values) {
325 if (handledValues.contains(t: v.value))
326 continue;
327 if (v.range) {
328 stream << v.name << " = " << v.value << " ... " << v.range << ", \\n\\\n";
329 } else {
330 const QString names = aggregateListValues(value: v.value, list: values);
331 stream << names << " = " << v.value << ", \\n\\\n";
332 handledValues.append(t: v.value);
333 }
334 }
335 stream << "} := " << name << ";\\n\\n\"));\n\n";
336 }
337 stream << "\n";
338}

347
348 const auto values = e.values;
349 QList<int> handledValues;
350
351 for (const auto &v : values) {
352 if (handledValues.contains(t: v.value))
353 continue;
354 const QString names = aggregateListValues(value: v.value, list: values);
355 stream << names << " = " << v.value << ", \\n\\\n";
356 handledValues.append(t: v.value);
357 }
358 stream << "} := " << name << ";\\n\\n\"));\n\n";
359 }
360 stream << "\n";
361}

363void writeCtf(QFile &file, const Provider &provider)
364{
365 QTextStream stream(&file);
366
367 const QString fileName = QFileInfo(file.fileName()).fileName();
368
369 writePrologue(stream, fileName, provider);
370 writeMetadataGenerators(stream);
371 writeEnums(stream, provider);
372 writeFlags(stream, provider);
373 writeTracepoints(stream, provider);
374 writeEpilogue(stream, fileName);
375}
