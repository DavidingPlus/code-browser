24// From the Python paths, find the component the UI file is under
25static QString pythonRoot(const QString &pythonPath, const QString &uiFileIn)
26{
27#ifdef Q_OS_WIN
28 static const Qt::CaseSensitivity fsSensitivity = Qt::CaseInsensitive;
29#else
30 static const Qt::CaseSensitivity fsSensitivity = Qt::CaseSensitive;
31#endif
32
33 if (pythonPath.isEmpty() || uiFileIn.isEmpty())
34 return {};
35 const QString uiFile = QFileInfo(uiFileIn).canonicalFilePath();
36 if (uiFile.isEmpty())
37 return {};
38 const auto uiFileSize = uiFile.size();
39 const auto paths = pythonPath.split(sep: QDir::listSeparator(), behavior: Qt::SkipEmptyParts);
40 for (const auto &path : paths) {
41 const QString canonicalPath = QFileInfo(path).canonicalFilePath();
42 const auto canonicalPathSize = canonicalPath.size();
43 if (uiFileSize > canonicalPathSize
44 && uiFile.at(i: canonicalPathSize) == u'/'
45 && uiFile.startsWith(s: canonicalPath, cs: fsSensitivity)) {
46 return canonicalPath;
47 }
48 }
49 return {};
50}

52int runUic(int argc, char *argv[])
53{
54 QHashSeed::setDeterministicGlobalSeed();
55
56 QCoreApplication app(argc, argv);
57 const QString version = QString::fromLatin1(ba: qVersion());
58 QCoreApplication::setApplicationVersion(version);
59
60 Driver driver;
61
62 // Note that uic isn't translated.
63 // If you use this code as an example for a translated app, make sure to translate the strings.
64 QCommandLineParser parser;
65 parser.setSingleDashWordOptionMode(QCommandLineParser::ParseAsLongOptions);
66 parser.setApplicationDescription(u"Qt User Interface Compiler version %1"_s.arg(a: version));
67 parser.addHelpOption();
68 parser.addVersionOption();
69
70 QCommandLineOption dependenciesOption(QStringList{u"d"_s, u"dependencies"_s});
71 dependenciesOption.setDescription(u"Display the dependencies."_s);
72 parser.addOption(commandLineOption: dependenciesOption);
73
74 QCommandLineOption outputOption(QStringList{u"o"_s, u"output"_s});
75 outputOption.setDescription(u"Place the output into <file>"_s);
76 outputOption.setValueName(u"file"_s);
77 parser.addOption(commandLineOption: outputOption);
78
79 QCommandLineOption noAutoConnectionOption(QStringList{u"a"_s, u"no-autoconnection"_s});
80 noAutoConnectionOption.setDescription(u"Do not generate a call to QObject::connectSlotsByName()."_s);
81 parser.addOption(commandLineOption: noAutoConnectionOption);
82
83 QCommandLineOption noProtOption(QStringList{u"p"_s, u"no-protection"_s});
84 noProtOption.setDescription(u"Disable header protection."_s);
85 parser.addOption(commandLineOption: noProtOption);
86
87 QCommandLineOption noImplicitIncludesOption(QStringList{u"n"_s, u"no-implicit-includes"_s});
88 noImplicitIncludesOption.setDescription(u"Disable generation of #include-directives."_s);
89 parser.addOption(commandLineOption: noImplicitIncludesOption);
90
91 QCommandLineOption postfixOption(u"postfix"_s);
92 postfixOption.setDescription(u"Postfix to add to all generated classnames."_s);
93 postfixOption.setValueName(u"postfix"_s);
94 parser.addOption(commandLineOption: postfixOption);
95
96 QCommandLineOption translateOption(QStringList{u"tr"_s, u"translate"_s});
97 translateOption.setDescription(u"Use <function> for i18n."_s);
98 translateOption.setValueName(u"function"_s);
99 parser.addOption(commandLineOption: translateOption);
100
101 QCommandLineOption includeOption(QStringList{u"include"_s});
102 includeOption.setDescription(u"Add #include <include-file> to <file>."_s);
103 includeOption.setValueName(u"include-file"_s);
104 parser.addOption(commandLineOption: includeOption);
105
106 QCommandLineOption generatorOption(QStringList{u"g"_s, u"generator"_s});
107 generatorOption.setDescription(u"Select generator."_s);
108 generatorOption.setValueName(u"python|cpp"_s);
109 parser.addOption(commandLineOption: generatorOption);
110
111 QCommandLineOption connectionsOption(QStringList{u"c"_s, u"connections"_s});
112 connectionsOption.setDescription(u"Connection syntax."_s);
113 connectionsOption.setValueName(u"pmf|string"_s);
114 parser.addOption(commandLineOption: connectionsOption);
115
116 QCommandLineOption idBasedOption(u"idbased"_s);
117 idBasedOption.setDescription(u"Use id based function for i18n"_s);
118 parser.addOption(commandLineOption: idBasedOption);
119
120 QCommandLineOption fromImportsOption(u"from-imports"_s);
121 fromImportsOption.setDescription(u"Python: generate imports relative to '.'"_s);
122 parser.addOption(commandLineOption: fromImportsOption);
123
124 QCommandLineOption absoluteImportsOption(u"absolute-imports"_s);
125 absoluteImportsOption.setDescription(u"Python: generate absolute imports"_s);
126 parser.addOption(commandLineOption: absoluteImportsOption);
127
128 // FIXME Qt 7: Flip the default?
129 QCommandLineOption rcPrefixOption(u"rc-prefix"_s);
130 rcPrefixOption.setDescription(uR"(Python: Generate "rc_file" instead of "file_rc" import)"_s);
131 parser.addOption(commandLineOption: rcPrefixOption);
132
133 // FIXME Qt 7: Remove?
134 QCommandLineOption useStarImportsOption(u"star-imports"_s);
135 useStarImportsOption.setDescription(u"Python: Use * imports"_s);
136 parser.addOption(commandLineOption: useStarImportsOption);
137
138 QCommandLineOption pythonPathOption(u"python-paths"_s);
139 pythonPathOption.setDescription(u"Python paths for --absolute-imports."_s);
140 pythonPathOption.setValueName(u"pathlist"_s);
141 parser.addOption(commandLineOption: pythonPathOption);
142
143 parser.addPositionalArgument(name: u"[uifile]"_s, description: u"Input file (*.ui), otherwise stdin."_s);
144
145 parser.process(app);
146
147 driver.option().dependencies = parser.isSet(option: dependenciesOption);
148 driver.option().outputFile = parser.value(option: outputOption);
149 driver.option().autoConnection = !parser.isSet(option: noAutoConnectionOption);
150 driver.option().headerProtection = !parser.isSet(option: noProtOption);
151 driver.option().implicitIncludes = !parser.isSet(option: noImplicitIncludesOption);
152 driver.option().idBased = parser.isSet(option: idBasedOption);
153 driver.option().postfix = parser.value(option: postfixOption);
154 driver.option().translateFunction = parser.value(option: translateOption);
155 driver.option().includeFile = parser.value(option: includeOption);
156 if (parser.isSet(option: connectionsOption)) {
157 const auto value = parser.value(option: connectionsOption);
158 if (value == "pmf"_L1)
159 driver.option().forceMemberFnPtrConnectionSyntax = 1;
160 else if (value == "string"_L1)
161 driver.option().forceStringConnectionSyntax = 1;
162 }
163
164 const QString inputFile = parser.positionalArguments().value(i: 0);
165
166 Language language = Language::Cpp;
167 if (parser.isSet(option: generatorOption)) {
168 if (parser.value(option: generatorOption).compare(other: "python"_L1) == 0)
169 language = Language::Python;
170 }
171 language::setLanguage(language);
172 if (language == Language::Python) {
173 if (parser.isSet(option: fromImportsOption))
174 driver.option().pythonResourceImport = Option::PythonResourceImport::FromDot;
175 else if (parser.isSet(option: absoluteImportsOption))
176 driver.option().pythonResourceImport = Option::PythonResourceImport::Absolute;
177 driver.option().useStarImports = parser.isSet(option: useStarImportsOption);
178 if (parser.isSet(option: rcPrefixOption))
179 driver.option().rcPrefix = 1;
180 QString pythonPaths;
181 if (parser.isSet(option: pythonPathOption))
182 pythonPaths = parser.value(option: pythonPathOption);
183 else if (qEnvironmentVariableIsSet(varName: pythonPathVar))
184 pythonPaths = QString::fromUtf8(ba: qgetenv(varName: pythonPathVar));
185 driver.option().pythonRoot = pythonRoot(pythonPath: pythonPaths, uiFileIn: inputFile);
186 }
187
188 if (inputFile.isEmpty()) // reading from stdin
189 driver.option().headerProtection = false;
190
191 if (driver.option().dependencies) {
192 return !driver.printDependencies(fileName: inputFile);
193 }
194
195 QTextStream *out = nullptr;
196 QFile f;
197 if (!driver.option().outputFile.isEmpty()) {
198 f.setFileName(driver.option().outputFile);
199 if (!f.open(flags: QIODevice::WriteOnly | QFile::Text)) {
200 fprintf(stderr, format: "Could not create output file\n");
201 return 1;
202 }
203 out = new QTextStream(&f);
204 out->setEncoding(QStringConverter::Utf8);
205 }
206
207 bool rtn = driver.uic(fileName: inputFile, output: out);
208 delete out;
209
210 if (!rtn) {
211 if (driver.option().outputFile.size()) {
212 f.close();
213 f.remove();
214 }
215 fprintf(stderr, format: "File '%s' is not valid\n", inputFile.isEmpty() ? "<stdin>" : inputFile.toLocal8Bit().constData());
216 }
217
218 return !rtn;
219}
