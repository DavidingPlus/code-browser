10static void removeOffsetRange(qsizetype begin, qsizetype end, QList<LineNumber> &offsets)
11{
12 qsizetype count = end - begin;
13 qsizetype i = 0;
14 DEBUGPRINTF2(printf("tracepointgen: removeOffsetRange: %d %d\n", begin, end));
15 while (i < offsets.size()) {
16 LineNumber &cur = offsets[i];
17 if (begin > cur.end) {
18 i++;
19 } else if (begin >= cur.begin && begin <= cur.end) {
20 cur.end = begin;
21 i++;
22 } else if (begin < cur.begin && end > cur.end) {
23 offsets.remove(i);
24 DEBUGPRINTF2(printf("tracepointgen: removeOffsetRange: %d, %d, %d\n", cur.begin, cur.end, cur.line));
25 } else if (end >= cur.begin && end <= cur.end) {
26 cur.begin = begin;
27 cur.end -= count;
28 i++;
29 } else if (end < cur.begin) {
30 cur.begin -= count;
31 cur.end -= count;
32 i++;
33 }
34 }
35}

37static bool findSpaceRange(const QString &data, qsizetype &offset, qsizetype &end) {
38 qsizetype cur = data.indexOf(c: QLatin1Char(' '), from: offset);
39 if (cur >= 0) {
40 qsizetype i = cur + 1;
41 while (data.constData()[i] == QLatin1Char(' ')) i++;
42 if (i - cur > 1) {
43 offset = cur;
44 end = i - 1;
45 return true;
46 }
47 cur = data.indexOf(c: QLatin1Char(' '), from: cur + 1);
48 }
49 return false;
50}

52static void simplifyData(QString &data, QList<LineNumber> &offsets)
53{
54 qsizetype offset = data.indexOf(QStringLiteral("//"));
55 while (offset >= 0) {
56 qsizetype endOfLine = data.indexOf(c: QLatin1Char('\n'), from: offset);
57 if (endOfLine == -1)
58 endOfLine = data.length();
59 removeOffsetRange(begin: offset, end: endOfLine, offsets);
60 data.remove(i: offset, len: endOfLine - offset);
61 offset = data.indexOf(QStringLiteral("//"), from: offset);
62 }
63 offset = data.indexOf(QStringLiteral("/*"));
64 while (offset >= 0) {
65 qsizetype endOfComment = data.indexOf(QStringLiteral("*/"), from: offset);
66 if (endOfComment == -1)
67 break;
68 removeOffsetRange(begin: offset, end: endOfComment + 2, offsets);
69 data.remove(i: offset, len: endOfComment - offset + 2);
70 offset = data.indexOf(QStringLiteral("/*"), from: offset);
71 }
72 offset = 0;
73 qsizetype end = 0;
74 data.replace(before: QLatin1Char('\n'), after: QLatin1Char(' '));
75 while (findSpaceRange(data, offset, end)) {
76 removeOffsetRange(begin: offset, end, offsets);
77 data.remove(i: offset, len: end - offset);
78 }
79}

81static void simplifyData(QString &data)
82{
83 qsizetype offset = data.indexOf(QStringLiteral("//"));
84 while (offset >= 0) {
85 qsizetype endOfLine = data.indexOf(c: QLatin1Char('\n'), from: offset);
86 if (endOfLine == -1)
87 endOfLine = data.length();
88 data.remove(i: offset, len: endOfLine - offset);
89 offset = data.indexOf(QStringLiteral("//"), from: offset);
90 }
91 offset = data.indexOf(QStringLiteral("/*"));
92 while (offset >= 0) {
93 qsizetype endOfComment = data.indexOf(QStringLiteral("*/"), from: offset);
94 if (endOfComment == -1)
95 break;
96 data.remove(i: offset, len: endOfComment - offset + 2);
97 offset = data.indexOf(QStringLiteral("/*"), from: offset);
98 }
99 offset = 0;
100 qsizetype end = 0;
101 while (findSpaceRange(data, offset, end))
102 data.remove(i: offset, len: end - offset);
103}

105static QString preprocessMetadata(const QString &in)
106{
107 DEBUGPRINTF(printf("in: %s\n", qPrintable(in)));
108 QList<QString> lines = in.split(sep: QLatin1Char('\\'));
109 QString out;
110 for (int i = 0; i < lines.size(); i++) {
111 QString l = lines.at(i).simplified();
112 DEBUGPRINTF(printf("line: %s\n", qPrintable(l)));
113 if (l.length() < 2)
114 continue;
115 if (l.startsWith(QStringLiteral("\"")))
116 l = l.right(n: l.length() - 1);
117 if (l.endsWith(QStringLiteral("\"")))
118 l = l.left(n: l.length() - 1);
119 l = l.simplified();
120
121 if (l.length() > 1) {
122 if (out.size() > 0)
123 out.append(c: QLatin1Char('\n'));
124 out.append(s: l);
125 }
126 }
127 DEBUGPRINTF(printf("out: %s\n", qPrintable(out)));
128 return out;
129}

142void Parser::parseParamReplace(const QString &data, qsizetype offset, const QString &name)
143{
144 Replace rep;
145 qsizetype beginBrace = data.indexOf(c: QLatin1Char('('), from: offset);
146 qsizetype endBrace = data.indexOf(c: QLatin1Char(')'), from: beginBrace);
147 QString params = data.mid(position: beginBrace + 1, n: endBrace - beginBrace -1);
148 int punc = params.indexOf(c: QLatin1Char(','));
149 if (punc < 0)
150 panic(fmt: "Syntax error in Q_TRACE_PARAM_REPLACE at file %s, line %d", qPrintable(name), lineNumber(offset));
151 rep.in = params.left(n: punc).simplified();
152 rep.out = params.right(n: params.length() - punc - 1).simplified();
153 if (rep.in.endsWith(c: QLatin1Char('*')) || rep.out.endsWith(c: QLatin1Char(']')))
154 rep.out.append(c: QLatin1Char(' '));
155 DEBUGPRINTF(printf("tracepointgen: replace: %s with %s\n", qPrintable(rep.in), qPrintable(rep.out)));
156 m_replaces.push_back(t: rep);
157}

159void Parser::parseInstrument(const QString &data, qsizetype offset)
160{
161 qsizetype beginOfProvider = data.indexOf(c: QLatin1Char('('), from: offset);
162 qsizetype endOfProvider = data.indexOf(c: QLatin1Char(')'), from: beginOfProvider);
163 Function func;
164 QString provider = data.mid(position: beginOfProvider + 1, n: endOfProvider - beginOfProvider - 1).simplified();
165 if (provider != m_provider)
166 return;
167
168 qsizetype classMarker = data.indexOf(QStringLiteral("::"), from: endOfProvider);
169 qsizetype beginOfFunctionMarker = data.indexOf(c: QLatin1Char('{'), from: classMarker);
170 QString begin = data.mid(position: endOfProvider + 1, n: classMarker - endOfProvider - 1);
171 QString end = data.mid(position: classMarker + 2, n: beginOfFunctionMarker - classMarker - 2);
172 int spaceIndex = begin.lastIndexOf(c: QLatin1Char(' '));
173 if (spaceIndex == -1)
174 func.className = begin;
175 else
176 func.className = begin.mid(position: spaceIndex + 1, n: begin.length() - spaceIndex - 1);
177 qsizetype braceIndex = end.indexOf(c: QLatin1Char('('));
178 spaceIndex = end.indexOf(c: QLatin1Char(' '));
179 if (spaceIndex < braceIndex)
180 func.functionName = end.left(n: spaceIndex).simplified();
181 else
182 func.functionName = end.left(n: braceIndex).simplified();
183
184 qsizetype lastBraceIndex = end.lastIndexOf(c: QLatin1Char(')'));
185 func.functionParameters = end.mid(position: braceIndex + 1, n: lastBraceIndex - braceIndex - 1).simplified();
186
187 DEBUGPRINTF(printf("tracepointgen: %s(%s)\n", qPrintable(func.functionName), qPrintable(func.functionParameters)));
188
189 m_functions.push_back(t: func);
190}

192void Parser::parsePoint(const QString &data, qsizetype offset)
193{
194 qsizetype beginOfProvider = data.indexOf(c: QLatin1Char('('), from: offset);
195 qsizetype endOfProvider = data.indexOf(c: QLatin1Char(','), from: beginOfProvider);
196 Point point;
197 QString provider = data.mid(position: beginOfProvider + 1, n: endOfProvider - beginOfProvider - 1).simplified();
198 if (provider != m_provider)
199 return;
200
201 qsizetype endOfPoint = data.indexOf(c: QLatin1Char(','), from: endOfProvider + 1);
202 qsizetype endOfPoint2 = data.indexOf(c: QLatin1Char(')'), from: endOfProvider + 1);
203 bool params = true;
204 if (endOfPoint == -1 || endOfPoint2 < endOfPoint) {
205 endOfPoint = endOfPoint2;
206 params = false;
207 }
208 point.name = data.mid(position: endOfProvider + 1, n: endOfPoint - endOfProvider - 1).simplified();
209 if (params) {
210 int endOfParams = data.indexOf(c: QLatin1Char(')'), from: endOfPoint);
211 point.parameters = data.mid(position: endOfPoint + 1, n: endOfParams - endOfPoint - 1).simplified();
212 }
213
214 DEBUGPRINTF(printf("tracepointgen: %s(%s)\n", qPrintable(point.name), qPrintable(point.parameters)));
215
216 m_points.push_back(t: point);
217}

237QStringList Parser::findEnumValues(const QString &name, const QStringList &includes)
238{
239 QStringList split = name.split(QStringLiteral("::"));
240 QString enumName = split.last();
241 DEBUGPRINTF(printf("searching for %s\n", qPrintable(name)));
242 QStringList ret;
243 for (auto filename : includes) {
244 QFile input(filename);
245 if (!input.open(flags: QIODevice::ReadOnly | QIODevice::Text)) {
246 DEBUGPRINTF(printf("Cannot open '%s' for reading: %s\n",
247 qPrintable(filename), qPrintable(input.errorString())));
248 return ret;
249 }
250 QString data;
251 QTextStream stream(&input);
252 while (!stream.atEnd()) {
253 QString line = stream.readLine().trimmed();
254 data += line + QLatin1Char('\n');
255 }
256 simplifyData(data);
257
258 int pos = 0;
259 bool valid = true;
260 for (int i = 0; i < split.size() - 1; i++) {
261 QRegularExpression macro(QStringLiteral("(struct|class|namespace) +([A-Za-z0-9_]*)? +([A-Za-z0-9]*;?)"));
262 QRegularExpressionMatchIterator m = macro.globalMatch(subject: data);
263 bool found = false;
264 while (m.hasNext() && !found) {
265 QRegularExpressionMatch match = m.next();
266 QString n = match.captured(nth: 2);
267 if (!n.endsWith(c: QLatin1Char(';')) && n == split[i] && match.capturedStart(nth: 2) > pos) {
268 pos = match.capturedStart(nth: 2);
269 found = true;
270 break;
271 }
272 if (match.hasCaptured(nth: 3)) {
273 n = match.captured(nth: 3);
274 if (!n.endsWith(c: QLatin1Char(';')) && n == split[i] && match.capturedStart(nth: 3) > pos) {
275 pos = match.capturedStart(nth: 3);
276 found = true;
277 break;
278 }
279 }
280 }
281 if (!found) {
282 valid = false;
283 break;
284 }
285 }
286
287 if (valid) {
288 QRegularExpression macro(QStringLiteral("enum +([A-Za-z0-9_]*)"));
289 QRegularExpressionMatchIterator m = macro.globalMatch(subject: data);
290 while (m.hasNext()) {
291 QRegularExpressionMatch match = m.next();
292
293 if (match.capturedStart() < pos)
294 continue;
295
296 QString n = match.captured(nth: 1);
297
298 if (n == enumName) {
299 DEBUGPRINTF(printf("Found enum: %s\n", qPrintable(n)));
300 int begin = data.indexOf(c: QLatin1Char('{'), from: match.capturedEnd());
301 int end = data.indexOf(c: QLatin1Char('}'), from: begin);
302 QString block = data.mid(position: begin + 1, n: end - begin - 1);
303 QStringList enums = block.split(sep: QLatin1Char('\n'));
304 for (auto e : enums) {
305 const auto trimmed = e.trimmed();
306 if (!trimmed.isEmpty() && !trimmed.startsWith(c: QLatin1Char('#')))
307 ret << trimmed;
308 }
309
310 break;
311 }
312 }
313 return ret;
314 }
315 }
316 return ret;
317}

319struct EnumNameValue
320{
321 QString name;
322 QString valueStr;
323 int value;
324};
325
326static QList<EnumNameValue> enumsToValues(const QStringList &values)
327{
328 int cur = 0;
329 QList<EnumNameValue> ret;
330 for (auto value : values) {
331 EnumNameValue r;
332 if (value.contains(c: QLatin1Char('='))) {
333 size_t offset = value.indexOf(c: QLatin1Char('='));
334 r.name = value.left(n: offset).trimmed();
335 QString val = value.right(n: value.length() - offset - 1).trimmed();
336 if (val.endsWith(c: QLatin1Char(',')))
337 val = val.left(n: val.length() - 1);
338 bool valid = false;
339 int integer = val.toInt(ok: &valid);
340 if (!valid)
341 integer = val.toInt(ok: &valid, base: 16);
342 if (valid) {
343 cur = r.value = integer;
344 ret << r;
345 } else {
346 auto iter = std::find_if(first: ret.begin(), last: ret.end(), pred: [&val](const EnumNameValue &elem){
347 return elem.name == val;
348 });
349 if (iter != ret.end()) {
350 cur = r.value = iter->value;
351 ret << r;
352 } else {
353 DEBUGPRINTF(printf("Invalid value: %s %s\n", qPrintable(r.name), qPrintable(value)));
354 }
355 }
356 } else {
357 r.name = value;
358 r.value = cur++;
359 ret << r;
360 }
361 }
362 return ret;
363}

365void Parser::parseMetadata(const QString &data, qsizetype offset, const QStringList &includes)
366{
367 qsizetype beginOfProvider = data.indexOf(c: QLatin1Char('('), from: offset);
368 qsizetype endOfProvider = data.indexOf(c: QLatin1Char(','), from: beginOfProvider);
369 QString metadata;
370 QString provider = data.mid(position: beginOfProvider + 1, n: endOfProvider - beginOfProvider - 1).simplified();
371 if (provider != m_provider)
372 return;
373
374 qsizetype endOfPoint = data.indexOf(c: QLatin1Char(')'), from: endOfProvider + 1);
375 metadata = data.mid(position: endOfProvider + 1, n: endOfPoint - endOfProvider - 1).simplified();
376
377 DEBUGPRINTF(printf("tracepointgen: metadata: %s", qPrintable(metadata)));
378
379 QString preprocessed = preprocessMetadata(in: metadata);
380
381 DEBUGPRINTF2(printf("preprocessed %s\n", qPrintable(preprocessed)));
382
383 QRegularExpression macro(QStringLiteral("([A-Z]*) ?{ ?([A-Za-z0-9=_,. ]*) ?} ?([A-Za-z0-9_:]*) ?;"));
384 QRegularExpressionMatchIterator i = macro.globalMatch(subject: preprocessed);
385 qsizetype prev = 0;
386 while (i.hasNext()) {
387 QRegularExpressionMatch match = i.next();
388 const QString values = match.captured(nth: 2).trimmed();
389 int cur = match.capturedStart();
390 if (cur > prev)
391 m_metadata.append(t: preprocessed.mid(position: prev, n: cur - prev));
392
393 prev = match.capturedEnd() + 1;
394 DEBUGPRINTF2(printf("values: %s\n", qPrintable(values)));
395 if (values.isEmpty() || values.startsWith(QStringLiteral("AUTO"))) {
396
397 QStringList ranges;
398 if (values.contains(QStringLiteral("RANGE"))) {
399 QRegularExpression rangeMacro(QStringLiteral("RANGE +([A-Za-z0-9_]*) +... +([A-Za-z0-9_]*)"));
400 QRegularExpressionMatchIterator r = rangeMacro.globalMatch(subject: values);
401 while (r.hasNext()) {
402 QRegularExpressionMatch rm = r.next();
403 ranges << rm.captured(nth: 1);
404 ranges << rm.captured(nth: 2);
405 DEBUGPRINTF2(printf("range: %s ... %s\n", qPrintable(rm.captured(1)), qPrintable(rm.captured(2))));
406 }
407 }
408
409 const auto enumOrFlag = match.captured(nth: 1);
410 const auto name = match.captured(nth: 3);
411 const bool flags = enumOrFlag == QStringLiteral("FLAGS");
412
413 QStringList values = findEnumValues(name, includes);
414 if (values.isEmpty()) {
415 if (flags && name.endsWith(c: QLatin1Char('s')))
416 values = findEnumValues(name: name.left(n: name.length() - 1), includes);
417 if (values.isEmpty()) {
418 DEBUGPRINTF(printf("Unable to find values for %s\n", qPrintable(name)));
419 }
420 }
421 if (!values.isEmpty()) {
422 auto moreValues = enumsToValues(values);
423 if (ranges.size()) {
424 for (int i = 0; i < ranges.size() / 2; i++) {
425 for (auto &v : moreValues) {
426 if (v.name == ranges[2 * i]) {
427 QString rangeEnd = ranges[2 * i + 1];
428 auto iter = std::find_if(first: moreValues.begin(), last: moreValues.end(), pred: [&rangeEnd](const EnumNameValue &elem){
429 return elem.name == rangeEnd;
430 });
431 if (iter != moreValues.end())
432 v.valueStr = QStringLiteral("RANGE(%1, %2 ... %3)").arg(a: v.name).arg(a: v.value).arg(a: iter->value);
433 }
434 }
435 }
436 }
437 std::sort(first: moreValues.begin(), last: moreValues.end(), comp: [](const EnumNameValue &a, const EnumNameValue &b) {
438 return a.value < b.value;
439 });
440 values.clear();
441 int prevValue = moreValues.first().value;
442 for (auto v : moreValues) {
443 QString a;
444 if (v.valueStr.isNull()) {
445 if (v.value == prevValue + 1 && !flags)
446 a = v.name;
447 else
448 a = QStringLiteral("%1 = %2").arg(a: v.name).arg(a: v.value);
449 prevValue = v.value;
450 } else {
451 a = v.valueStr;
452 }
453 values << a;
454 }
455
456 metadata = QStringLiteral("%1 {\n %2 \n} %3;").arg(a: enumOrFlag).arg(a: values.join(QStringLiteral(",\n"))).arg(a: name);
457 if (!m_metadata.contains(str: metadata))
458 m_metadata.append(t: metadata);
459 }
460 } else {
461 if (!m_metadata.contains(str: match.captured()))
462 m_metadata.append(t: match.captured());
463 }
464 }
465 if (prev < preprocessed.length())
466 m_metadata.append(t: preprocessed.mid(position: prev, n: preprocessed.length() - prev));
467}

482void Parser::addIncludesRecursive(const QString &filename, QList<QString> &includes)
483{
484 QFileInfo info(filename);
485 DEBUGPRINTF(printf("check include: %s\n", qPrintable(filename)));
486 QFile input(filename);
487 if (!input.open(flags: QIODevice::ReadOnly | QIODevice::Text)) {
488 DEBUGPRINTF(printf("Cannot open '%s' for reading: %s\n",
489 qPrintable(filename), qPrintable(input.errorString())));
490 return;
491 }
492 QString data;
493 QTextStream stream(&input);
494 while (!stream.atEnd()) {
495 QString line = stream.readLine().trimmed();
496 data += line + QLatin1Char(QLatin1Char('\n'));
497 }
498
499 QRegularExpression includeMacro(QStringLiteral("#include [\"<]([A-Za-z0-9_./]*.h)[\">]"));
500 QRegularExpressionMatchIterator i = includeMacro.globalMatch(subject: data);
501 while (i.hasNext()) {
502 QRegularExpressionMatch match = i.next();
503 QString filename = match.captured(nth: 1);
504
505 QString rinc = filename;
506 if (filename.startsWith(QStringLiteral("../"))) {
507 QFileInfo info2(info.absolutePath() + QLatin1Char('/') + filename);
508 if (!info2.exists()) {
509 DEBUGPRINTF(printf("unable to find %s\n", qPrintable(filename)));
510 continue;
511 }
512 rinc = info2.absoluteFilePath();
513 filename = info2.fileName();
514 }
515
516 // only search possible qt headers
517 if (filename.startsWith(c: QLatin1Char('q'), cs: Qt::CaseInsensitive)) {
518 QString resolved = resolveInclude(filename: rinc);
519 if (!resolved.isEmpty() && !includes.contains(str: resolved)) {
520 includes.push_back(t: resolved);
521 addIncludesRecursive(filename: resolved, includes);
522 }
523 }
524 }
525}

527void Parser::parse(QIODevice &input, const QString &name)
528{
529 QString data;
530 QTextStream stream(&input);
531 int lineNumber = 1;
532 qsizetype prev = 0;
533 while (!stream.atEnd()) {
534 QString line = stream.readLine().trimmed();
535 m_offsets.push_back(t: {.begin: prev, .end: prev + line.length(), .line: lineNumber++});
536 prev += line.length() + 1;
537 data += line + QLatin1Char(QLatin1Char('\n'));
538 }
539
540 simplifyData(data, offsets&: m_offsets);
541
542 QStringList includes;
543
544 QRegularExpression includeMacro(QStringLiteral("#include [\"<]([A-Za-z0-9_./]*.h)[\">]"));
545 QRegularExpressionMatchIterator i = includeMacro.globalMatch(subject: data);
546 while (i.hasNext()) {
547 QRegularExpressionMatch match = i.next();
548 const QString filename = match.captured(nth: 1);
549 // only search possible qt headers
550 if (filename.startsWith(c: QLatin1Char('q'), cs: Qt::CaseInsensitive)) {
551 const QString resolved = resolveInclude(filename);
552 if (!resolved.isEmpty() && !includes.contains(str: resolved)) {
553 includes.push_back(t: resolved);
554 addIncludesRecursive(filename: resolved, includes);
555 }
556 }
557 }
558
559 QRegularExpression traceMacro(QStringLiteral("Q_TRACE_([A-Z_]*)"));
560 i = traceMacro.globalMatch(subject: data);
561 while (i.hasNext()) {
562 QRegularExpressionMatch match = i.next();
563
564 QString macroType = match.captured(nth: 1);
565 if (macroType == QStringLiteral("PARAM_REPLACE"))
566 parseParamReplace(data, offset: match.capturedEnd(), name);
567 else if (macroType == QStringLiteral("INSTRUMENT"))
568 parseInstrument(data, offset: match.capturedEnd());
569 else if (macroType == QStringLiteral("POINT"))
570 parsePoint(data, offset: match.capturedEnd());
571 else if (macroType == QStringLiteral("PREFIX"))
572 parsePrefix(data, offset: match.capturedEnd());
573 else if (macroType == QStringLiteral("METADATA"))
574 parseMetadata(data, offset: match.capturedEnd(), includes);
575 }
576
577 for (auto &func : m_functions) {
578 for (auto &rep : m_replaces)
579 func.functionParameters.replace(before: rep.in, after: rep.out);
580 }
581}

583void Parser::write(QIODevice &input) const
584{
585 QTextStream out(&input);
586 if (m_prefixes.size() > 0) {
587 out << QStringLiteral("{\n");
588 for (auto prefix : m_prefixes)
