11static void usage(int status)
12{
13 printf(format: "Generates a tracepoint file for tracegen tool from input files.\n\n");
14 printf(format: "Usage: tracepointgen <output file> <input files> \n");

18static void parseArgs(int argc, char *argv[], QString &provider, QString &outFile, QList<QString> &inputFiles)
19{
20 if (argc == 1)
21 usage(status: 0);
22 if (argc < 4)
23 usage(status: -1);
24
25 provider = QLatin1StringView(argv[1]);
26 outFile = QLatin1StringView(argv[2]);
27 for (int i = 3; i < argc; i++)
28 inputFiles.append(t: QLatin1StringView(argv[i]));
29}
30
31int main(int argc, char *argv[])
32{
33 QString provider;
34 QList<QString> inputFiles;
35 QString outFile;
36
37 parseArgs(argc, argv, provider, outFile, inputFiles);
38
39 Parser parser(provider);
40
41 for (const QString &inputFile : inputFiles) {
42 if (inputFile.startsWith(c: QLatin1Char('I'))) {
43 QStringList includeDirs = inputFile.right(n: inputFile.length() - 1).split(sep: QLatin1Char(';'));
44 parser.addIncludeDirs(list: includeDirs);
45 continue;
46 }
47 QFile in(inputFile);
48 if (!in.open(flags: QIODevice::ReadOnly | QIODevice::Text)) {
49 panic(fmt: "Cannot open '%s' for reading: %s\n",
50 qPrintable(inputFile), qPrintable(in.errorString()));
51 }
52 DEBUGPRINTF(printf("tracepointgen: parse %s\n", qPrintable(inputFile)));
53 parser.parse(input&: in, name: inputFile);
54 }
55 if (parser.isEmpty())
56 panic(fmt: "empty provider %s\n", qPrintable(provider));
57
58 QFile out(outFile);
59
60 if (!out.open(flags: QIODevice::WriteOnly | QIODevice::Truncate)) {
61 panic(fmt: "Cannot open '%s' for writing: %s\n",
62 qPrintable(outFile), qPrintable(out.errorString()));
63 }
64
65 parser.write(input&: out);
66 out.close();
67
68 return 0;
69}
