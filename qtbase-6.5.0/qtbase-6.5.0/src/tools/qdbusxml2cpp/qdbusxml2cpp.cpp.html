54static QDBusIntrospection::Interfaces readInput()
55{
56 QFile input(inputFile);
57 if (inputFile.isEmpty() || inputFile == "-"_L1)
58 input.open(stdin, ioFlags: QIODevice::ReadOnly);
59 else
60 input.open(flags: QIODevice::ReadOnly);
61
62 QByteArray data = input.readAll();
63
64 // check if the input is already XML
65 data = data.trimmed();
66 if (data.startsWith(bv: "<!DOCTYPE ") || data.startsWith(bv: "<?xml") ||
67 data.startsWith(bv: "<node") || data.startsWith(bv: "<interface"))
68 // already XML
69 return QDBusIntrospection::parseInterfaces(xml: QString::fromUtf8(ba: data));
70
71 fprintf(stderr, format: "%s: Cannot process input: '%s'. Stop.\n",
72 PROGRAMNAME, qPrintable(inputFile));
73 exit(status: 1);
74}

88static bool isSupportedSuffix(QStringView suffix)
89{
90 const QLatin1StringView candidates[] = {
91 "h"_L1,
92 "cpp"_L1,

135// produce a moc name from the file name
136static QString moc(const QString &name)
137{
138 QString retval;
139 const QStringList fileNames = name.split(sep: u':');
140
141 if (fileNames.size() == 1) {
142 QFileInfo fi{fileNames.front()};
143 if (isSupportedSuffix(suffix: fi.suffix())) {
144 // Generates a file that contains the header and the implementation: include "filename.moc"
145 retval += fi.completeBaseName();
146 retval += ".moc"_L1;
147 } else {
148 // Separate source and header files are generated: include "moc_filename.cpp"
149 retval += "moc_"_L1;
150 retval += fi.fileName();
151 retval += ".cpp"_L1;
152 }
153 } else {
154 QString headerName = fileNames.front();
155 QString sourceName = fileNames.back();
156
157 if (sourceName.isEmpty() || sourceName == "-"_L1) {
158 // If only a header is generated, don't include anything
159 } else if (headerName.isEmpty() || headerName == "-"_L1) {
160 // If only source file is generated: include "moc_sourcename.cpp"
161 QFileInfo source{sourceName};
162
163 retval += "moc_"_L1;
164 retval += source.completeBaseName();
165 retval += ".cpp"_L1;
166
167 fprintf(stderr, format: "warning: no header name is provided, assuming it to be \"%s\"\n",
168 qPrintable(source.completeBaseName() + ".h"_L1));
169 } else {
170 // Both source and header generated: include "moc_headername.cpp"
171 QFileInfo header{headerName};
172
173 retval += "moc_"_L1;
174 retval += header.completeBaseName();
175 retval += ".cpp"_L1;
176 }
177 }
178
179 return retval;
180}

204enum ClassType { Proxy, Adaptor };
205static QString classNameForInterface(const QString &interface, ClassType classType)
206{
207 if (!globalClassName.isEmpty())
208 return globalClassName;
209
210 const auto parts = QStringView{interface}.split(sep: u'.');
211
212 QString retval;
213 if (classType == Proxy) {
214 for (const auto &part : parts) {
215 retval += part[0].toUpper();
216 retval += part.mid(pos: 1);
217 }
218 } else {
219 retval += parts.last()[0].toUpper() + parts.last().mid(pos: 1);
220 }
221
222 if (classType == Proxy)
223 retval += "Interface"_L1;
224 else
225 retval += "Adaptor"_L1;
226
227 return retval;

230static QByteArray qtTypeName(const QString &where, const QString &signature,
231 const QDBusIntrospection::Annotations &annotations, qsizetype paramId = -1,
232 const char *direction = "Out")
233{
234 int type = QDBusMetaType::signatureToMetaType(signature: signature.toLatin1()).id();
235 if (type == QMetaType::UnknownType) {
236 QString annotationName = QString::fromLatin1(ba: "org.qtproject.QtDBus.QtTypeName");
237 if (paramId >= 0)
238 annotationName += QString::fromLatin1(ba: ".%1%2").arg(a: QLatin1StringView(direction)).arg(a: paramId);
239 QString qttype = annotations.value(key: annotationName);
240 if (!qttype.isEmpty())
241 return std::move(qttype).toLatin1();
242
243 QString oldAnnotationName = QString::fromLatin1(ba: "com.trolltech.QtDBus.QtTypeName");
244 if (paramId >= 0)
245 oldAnnotationName += QString::fromLatin1(ba: ".%1%2").arg(a: QLatin1StringView(direction)).arg(a: paramId);
246 qttype = annotations.value(key: oldAnnotationName);
247
248 if (qttype.isEmpty()) {
249 fprintf(stderr, format: "%s: Got unknown type `%s' processing '%s'\n",
250 PROGRAMNAME, qPrintable(signature), qPrintable(inputFile));
251 fprintf(stderr,
252 format: "You should add <annotation name=\"%s\" value=\"<type>\"/> to the XML "
253 "description for '%s'\n",
254 qPrintable(annotationName), qPrintable(where));
255
256 exit(status: 1);
257 }
258
259 fprintf(stderr, format: "%s: Warning: deprecated annotation '%s' found while processing '%s'; "
260 "suggest updating to '%s'\n",
261 PROGRAMNAME, qPrintable(oldAnnotationName), qPrintable(inputFile),
262 qPrintable(annotationName));
263 return std::move(qttype).toLatin1();
264 }
265
266 return QMetaType(type).name();
267}

290static QStringList makeArgNames(const QDBusIntrospection::Arguments &inputArgs,
291 const QDBusIntrospection::Arguments &outputArgs =
292 QDBusIntrospection::Arguments())
293{
294 QStringList retval;
295 const qsizetype numInputArgs = inputArgs.size();
296 const qsizetype numOutputArgs = outputArgs.size();
297 retval.reserve(size: numInputArgs + numOutputArgs);
298 for (qsizetype i = 0; i < numInputArgs; ++i) {
299 const QDBusIntrospection::Argument &arg = inputArgs.at(i);
300 QString name = arg.name;
301 if (name.isEmpty())
302 name = u"in%1"_s.arg(a: i);
303 else
304 name.replace(before: u'-', after: u'_');
305 while (retval.contains(str: name))
306 name += "_"_L1;
307 retval << name;
308 }
309 for (qsizetype i = 0; i < numOutputArgs; ++i) {
310 const QDBusIntrospection::Argument &arg = outputArgs.at(i);
311 QString name = arg.name;
312 if (name.isEmpty())
313 name = u"out%1"_s.arg(a: i);
314 else
315 name.replace(before: u'-', after: u'_');
316 while (retval.contains(str: name))
317 name += "_"_L1;
318 retval << name;
319 }
320 return retval;
321}

323static void writeArgList(QTextStream &ts, const QStringList &argNames,
324 const QDBusIntrospection::Annotations &annotations,
325 const QDBusIntrospection::Arguments &inputArgs,
326 const QDBusIntrospection::Arguments &outputArgs = QDBusIntrospection::Arguments())
327{
328 // input args:
329 bool first = true;
330 qsizetype argPos = 0;
331 for (qsizetype i = 0; i < inputArgs.size(); ++i) {
332 const QDBusIntrospection::Argument &arg = inputArgs.at(i);
333 QString type = constRefArg(arg: qtTypeName(where: arg.name, signature: arg.type, annotations, paramId: i, direction: "In"));
334
335 if (!first)
336 ts << ", ";
337 ts << type << argNames.at(i: argPos++);
338 first = false;
339 }
340
341 argPos++;
342
343 // output args
344 // yes, starting from 1
345 for (qsizetype i = 1; i < outputArgs.size(); ++i) {
346 const QDBusIntrospection::Argument &arg = outputArgs.at(i);
347
348 if (!first)
349 ts << ", ";
350 ts << nonConstRefArg(arg: qtTypeName(where: arg.name, signature: arg.type, annotations, paramId: i, direction: "Out"))
351 << argNames.at(i: argPos++);
352 first = false;
353 }
354}

460static void writeProxy(const QString &filename, const QDBusIntrospection::Interfaces &interfaces)
461{
462 // open the file
463 QString headerName = header(name: filename);
464 QByteArray headerData;
465 QTextStream hs(&headerData);
466
467 QString cppName = cpp(name: filename);
468 QByteArray cppData;
469 QTextStream cs(&cppData);
470
471 // write the header:
472 writeHeader(ts&: hs, changesWillBeLost: true);
473 if (cppName != headerName)
474 writeHeader(ts&: cs, changesWillBeLost: false);
475
476 // include guards:
477 QString includeGuard;
478 if (!headerName.isEmpty() && headerName != "-"_L1) {
479 includeGuard = headerName.toUpper().replace(before: u'.', after: u'_');
480 qsizetype pos = includeGuard.lastIndexOf(c: u'/');
481 if (pos != -1)
482 includeGuard = includeGuard.mid(position: pos + 1);
483 } else {
484 includeGuard = u"QDBUSXML2CPP_PROXY"_s;
485 }
486
487 hs << "#ifndef " << includeGuard << Qt::endl
488 << "#define " << includeGuard << Qt::endl
489 << Qt::endl;
490
491 // include our stuff:
492 hs << "#include <QtCore/QObject>" << Qt::endl
493 << includeList;
494#if QT_VERSION < QT_VERSION_CHECK(7, 0, 0)
495 hs << "#include <QtDBus/QtDBus>" << Qt::endl;
496#else
497 hs << "#include <QtDBus/QDBusAbstractInterface>" << Qt::endl;
498 hs << "#include <QtDBus/QDBusPendingReply>" << Qt::endl;
499#endif
500
501 for (const QString &include : std::as_const(t&: includes)) {
502 hs << "#include \"" << include << "\"" << Qt::endl;
503 if (headerName.isEmpty())
504 cs << "#include \"" << include << "\"" << Qt::endl;
505 }
506
507 for (const QString &include : std::as_const(t&: globalIncludes)) {
508 hs << "#include <" << include << ">" << Qt::endl;
509 if (headerName.isEmpty())
510 cs << "#include <" << include << ">" << Qt::endl;
511 }
512
513 hs << Qt::endl;
514
515 if (cppName != headerName) {
516 if (!headerName.isEmpty() && headerName != "-"_L1)
517 cs << "#include \"" << headerName << "\"" << Qt::endl << Qt::endl;
518 }
519
520 for (const QDBusIntrospection::Interface *interface : interfaces) {
521 QString className = classNameForInterface(interface: interface->name, classType: Proxy);
522
523 // comment:
524 hs << "/*" << Qt::endl
525 << " * Proxy class for interface " << interface->name << Qt::endl
526 << " */" << Qt::endl;
527 cs << "/*" << Qt::endl
528 << " * Implementation of interface class " << className << Qt::endl
529 << " */" << Qt::endl
530 << Qt::endl;
531
532 // class header:
533 hs << "class " << className << ": public QDBusAbstractInterface" << Qt::endl
534 << "{" << Qt::endl
535 << " Q_OBJECT" << Qt::endl;
536
537 // the interface name
538 hs << "public:" << Qt::endl
539 << " static inline const char *staticInterfaceName()" << Qt::endl
540 << " { return \"" << interface->name << "\"; }" << Qt::endl
541 << Qt::endl;
542
543 // constructors/destructors:
544 hs << "public:" << Qt::endl
545 << " " << className << "(const QString &service, const QString &path, const QDBusConnection &connection, QObject *parent = nullptr);" << Qt::endl
546 << Qt::endl
547 << " ~" << className << "();" << Qt::endl
548 << Qt::endl;
549 cs << className << "::" << className << "(const QString &service, const QString &path, const QDBusConnection &connection, QObject *parent)" << Qt::endl
550 << " : QDBusAbstractInterface(service, path, staticInterfaceName(), connection, parent)" << Qt::endl
551 << "{" << Qt::endl
552 << "}" << Qt::endl
553 << Qt::endl
554 << className << "::~" << className << "()" << Qt::endl
555 << "{" << Qt::endl
556 << "}" << Qt::endl
557 << Qt::endl;
558
559 // properties:
560 for (const QDBusIntrospection::Property &property : interface->properties) {
561 QByteArray type = qtTypeName(where: property.name, signature: property.type, annotations: property.annotations);
562 QString getter = propertyGetter(property);
563 QString setter = propertySetter(property);
564
565 hs << " Q_PROPERTY(" << type << " " << property.name;
566
567 // getter:
568 if (property.access != QDBusIntrospection::Property::Write)
569 // it's readable
570 hs << " READ " << getter;
571
572 // setter
573 if (property.access != QDBusIntrospection::Property::Read)
574 // it's writeable
575 hs << " WRITE " << setter;
576
577 hs << ")" << Qt::endl;
578
579 // getter:
580 if (property.access != QDBusIntrospection::Property::Write) {
581 hs << " inline " << type << " " << getter << "() const" << Qt::endl
582 << " { return qvariant_cast< " << type << " >(property(\""
583 << property.name << "\")); }" << Qt::endl;
584 }
585
586 // setter:
587 if (property.access != QDBusIntrospection::Property::Read) {
588 hs << " inline void " << setter << "(" << constRefArg(arg: type) << "value)" << Qt::endl
589 << " { setProperty(\"" << property.name
590 << "\", QVariant::fromValue(value)); }" << Qt::endl;
591 }
592
593 hs << Qt::endl;
594 }
595
596 // methods:
597 hs << "public Q_SLOTS: // METHODS" << Qt::endl;
598 for (const QDBusIntrospection::Method &method : interface->methods) {
599 bool isDeprecated = method.annotations.value(key: "org.freedesktop.DBus.Deprecated"_L1) == "true"_L1;
600 bool isNoReply =
601 method.annotations.value(ANNOTATION_NO_WAIT ""_L1) == "true"_L1;
602 if (isNoReply && !method.outputArgs.isEmpty()) {
603 fprintf(stderr, format: "%s: warning while processing '%s': method %s in interface %s is marked 'no-reply' but has output arguments.\n",
604 PROGRAMNAME, qPrintable(inputFile), qPrintable(method.name),
605 qPrintable(interface->name));
606 continue;
607 }
608
609 hs << " inline "
610 << (isDeprecated ? "Q_DECL_DEPRECATED " : "");
611
612 if (isNoReply) {
613 hs << "Q_NOREPLY void ";
614 } else {
615 hs << "QDBusPendingReply<";
616 for (qsizetype i = 0; i < method.outputArgs.size(); ++i)
617 hs << (i > 0 ? ", " : "")
618 << templateArg(arg: qtTypeName(where: method.outputArgs.at(i).name, signature: method.outputArgs.at(i).type,
619 annotations: method.annotations, paramId: i, direction: "Out"));
620 hs << "> ";
621 }
622
623 hs << methodName(method) << "(";
624
625 QStringList argNames = makeArgNames(inputArgs: method.inputArgs);
626 writeArgList(ts&: hs, argNames, annotations: method.annotations, inputArgs: method.inputArgs);
627
628 hs << ")" << Qt::endl
629 << " {" << Qt::endl
630 << " QList<QVariant> argumentList;" << Qt::endl;
631
632 if (!method.inputArgs.isEmpty()) {
633 hs << " argumentList";
634 for (qsizetype argPos = 0; argPos < method.inputArgs.size(); ++argPos)
635 hs << " << QVariant::fromValue(" << argNames.at(i: argPos) << ')';
636 hs << ";" << Qt::endl;
637 }
638
639 if (isNoReply)
640 hs << " callWithArgumentList(QDBus::NoBlock, "
641 << "QStringLiteral(\"" << method.name << "\"), argumentList);" << Qt::endl;
642 else
643 hs << " return asyncCallWithArgumentList(QStringLiteral(\""
644 << method.name << "\"), argumentList);" << Qt::endl;
645
646 // close the function:
647 hs << " }" << Qt::endl;
648
649 if (method.outputArgs.size() > 1) {
650 // generate the old-form QDBusReply methods with multiple incoming parameters
651 hs << " inline " << (isDeprecated ? "Q_DECL_DEPRECATED " : "") << "QDBusReply<"
652 << templateArg(arg: qtTypeName(where: method.outputArgs.first().name, signature: method.outputArgs.first().type,
653 annotations: method.annotations, paramId: 0, direction: "Out"))
654 << "> ";
655 hs << method.name << "(";
656
657 QStringList argNames = makeArgNames(inputArgs: method.inputArgs, outputArgs: method.outputArgs);
658 writeArgList(ts&: hs, argNames, annotations: method.annotations, inputArgs: method.inputArgs, outputArgs: method.outputArgs);
659
660 hs << ")" << Qt::endl
661 << " {" << Qt::endl
662 << " QList<QVariant> argumentList;" << Qt::endl;
663
664 qsizetype argPos = 0;
665 if (!method.inputArgs.isEmpty()) {
666 hs << " argumentList";
667 for (argPos = 0; argPos < method.inputArgs.size(); ++argPos)

669 hs << ";" << Qt::endl;

682 << templateArg(arg: qtTypeName(where: method.outputArgs.at(i).name, signature: method.outputArgs.at(i).type,
683 annotations: method.annotations, paramId: i, direction: "Out"))

694 for (const QDBusIntrospection::Signal &signal : interface->signals_) {

697 hs << "Q_DECL_DEPRECATED ";
698
699 hs << "void " << signal.name << "(";

701 QStringList argNames = makeArgNames(inputArgs: signal.outputArgs);
702 writeSignalArgList(ts&: hs, argNames, annotations: signal.annotations, outputArgs: signal.outputArgs);
703
704 hs << ");" << Qt::endl; // finished for header
705 }
706
707 // close the class:
708 hs << "};" << Qt::endl
709 << Qt::endl;
710 }
711
712 if (!skipNamespaces) {
713 QStringList last;
714 QDBusIntrospection::Interfaces::ConstIterator it = interfaces.constBegin();
715 do
716 {
717 QStringList current;
718 QString name;
719 if (it != interfaces.constEnd()) {
720 current = it->constData()->name.split(sep: u'.');
721 name = current.takeLast();
722 }

724 qsizetype i = 0;
725 while (i < current.size() && i < last.size() && current.at(i) == last.at(i))
726 ++i;
727
728 // i parts matched
729 // close last.arguments().size() - i namespaces:
730 for (qsizetype j = i; j < last.size(); ++j)
731 hs << QString((last.size() - j - 1 + i) * 2, u' ') << "}" << Qt::endl;
732
733 // open current.arguments().size() - i namespaces
734 for (qsizetype j = i; j < current.size(); ++j)
735 hs << QString(j * 2, u' ') << "namespace " << current.at(i: j) << " {" << Qt::endl;
736
737 // add this class:
738 if (!name.isEmpty()) {
739 hs << QString(current.size() * 2, u' ')
740 << "using " << name << " = ::" << classNameForInterface(interface: it->constData()->name, classType: Proxy)
741 << ";" << Qt::endl;
742 }
743
744 if (it == interfaces.constEnd())
745 break;
746 ++it;
747 last = current;
748 } while (true);
749 }

751 // close the include guard
752 hs << "#endif" << Qt::endl;
753
754 QString mocName = moc(name: filename);
755 if (includeMocs && !mocName.isEmpty())
756 cs << Qt::endl
757 << "#include \"" << mocName << "\"" << Qt::endl;
758
759 cs.flush();
760 hs.flush();
761
762 QFile file;
763 const bool headerOpen = openFile(fileName: headerName, file);
764 if (headerOpen)
765 file.write(data: headerData);
766
767 if (headerName == cppName) {
768 if (headerOpen)
769 file.write(data: cppData);
770 } else {
771 QFile cppFile;
772 if (openFile(fileName: cppName, file&: cppFile))
773 cppFile.write(data: cppData);
