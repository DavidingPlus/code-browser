22static void help_me ()
23{
24 qerr() << "Usage: qlalr [options] [input file name]" << Qt::endl
25 << Qt::endl
26 << " --help, -h\t\tdisplay this help and exit" << Qt::endl
27 << " --verbose, -v\t\tverbose output" << Qt::endl
28 << " --no-debug\t\tno debug information" << Qt::endl
29 << " --no-lines\t\tno #line directives" << Qt::endl
30 << " --dot\t\t\tgenerate a graph" << Qt::endl
31 << " --qt\t\tadd the Qt copyright header and Qt-specific types and macros" << Qt::endl
32 << Qt::endl;
33 exit (status: 0);
34}

36int main (int argc, char *argv[])
37{
38 QCoreApplication app (argc, argv);
39
40 bool generate_dot = false;
41 bool generate_report = false;
42 bool no_lines = false;
43 bool debug_info = true;
44 bool qt_copyright = false;
45 QString file_name;
46
47 const QStringList args = app.arguments().mid(pos: 1);
48 for (const QString &arg : args) {
49 if (arg == "-h"_L1 || arg == "--help"_L1)
50 help_me ();
51
52 else if (arg == "-v"_L1 || arg == "--verbose"_L1)
53 generate_report = true;
54
55 else if (arg == "--dot"_L1)
56 generate_dot = true;
57
58 else if (arg == "--no-lines"_L1)
59 no_lines = true;
60
61 else if (arg == "--no-debug"_L1)
62 debug_info = false;
63
64 else if (arg == "--qt"_L1)
65 qt_copyright = true;
66
67 else if (file_name.isEmpty ())
68 file_name = arg;
69
70 else
71 qerr() << "*** Warning. Ignore argument `" << arg << "'" << Qt::endl;
72 }
73
74 if (file_name.isEmpty ())
75 {
76 help_me ();
77 exit (EXIT_SUCCESS);
78 }
79
80 Grammar grammar;
81 Recognizer p (&grammar, no_lines);
82
83 if (! p.parse (input_file: file_name))
84 exit (EXIT_FAILURE);
85
86 if (grammar.rules.empty())
87 {
88 qerr() << "*** Fatal. No rules!" << Qt::endl;
89 exit (EXIT_FAILURE);
90 }
91
92 else if (grammar.start == grammar.names.end ())
93 {
94 qerr() << "*** Fatal. No start symbol!" << Qt::endl;
95 exit (EXIT_FAILURE);
96 }
97
98 grammar.buildExtendedGrammar ();
99 grammar.buildRuleMap ();
100
101 Automaton aut (&grammar);
102 aut.build ();
103
104 CppGenerator gen (p, grammar, aut, generate_report);
105 gen.setDebugInfo (debug_info);
106 gen.setCopyright (qt_copyright);
107 gen ();
108
109 if (generate_dot)
110 {
111 DotGraph genDotFile (qout());
112 genDotFile (&aut);
113 }
114
115 else if (generate_report)
116 {
117 ParseTable genParseTable (qout());
118 genParseTable(&aut);
119 }
120
121 return EXIT_SUCCESS;
122}

136 code = code.replace (before: "$table"_L1, after: _M_grammar->table_name);
137 code = code.replace (before: "$parser"_L1, after: _M_grammar->table_name);
138
139 if (_M_current_rule != _M_grammar->rules.end ())
140 {
141 code = code.replace (before: "$rule_number"_L1, after: QString::number (std::distance (first: _M_grammar->rules.begin (), last: _M_current_rule)));
142 code = code.replace (before: "$rule"_L1, after: *_M_current_rule->lhs);
143 }
144
145 return code;
146}
