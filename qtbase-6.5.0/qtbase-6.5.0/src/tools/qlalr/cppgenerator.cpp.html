20void generateSeparator(int i, QTextStream &out)
21{
22 if (!(i % 10)) {
23 if (i)
24 out << ",";
25 out << Qt::endl << " ";
26 } else {
27 out << ", ";
28 }
29}

31void generateList(const QList<int> &list, QTextStream &out)
32{
33 for (int i = 0; i < list.size(); ++i) {
34 generateSeparator(i, out);
35
36 out << list[i];
37 }
38}

80void CppGenerator::operator () ()
81{
82 // action table...
83 state_count = static_cast<int>(aut.states.size());
84 terminal_count = static_cast<int>(grammar.terminals.size());
85 non_terminal_count = static_cast<int>(grammar.non_terminals.size());
86
87#define ACTION(i, j) table [(i) * terminal_count + (j)]
88#define GOTO(i, j) pgoto [(i) * non_terminal_count + (j)]
89
90 int *table = new int [state_count * terminal_count];
91 ::memset (s: table, c: 0, n: state_count * terminal_count * sizeof (int));
92
93 int *pgoto = new int [state_count * non_terminal_count];
94 ::memset (s: pgoto, c: 0, n: state_count * non_terminal_count * sizeof (int));
95
96 accept_state = -1;
97 int shift_reduce_conflict_count = 0;
98 int reduce_reduce_conflict_count = 0;
99
100 for (StatePointer state = aut.states.begin (); state != aut.states.end (); ++state)
101 {
102 int q = aut.id (state);
103
104 for (Bundle::iterator a = state->bundle.begin (); a != state->bundle.end (); ++a)
105 {
106 int symbol = aut.id (name: a.key ());
107 int r = aut.id (state: a.value ());
108
109 Q_ASSERT (r < state_count);
110
111 if (grammar.isNonTerminal (name: a.key ()))
112 {
113 Q_ASSERT(symbol >= terminal_count && symbol < static_cast<int>(grammar.names.size()));
114 GOTO (q, symbol - terminal_count) = r;
115 }
116
117 else
118 ACTION (q, symbol) = r;
119 }
120
121 for (ItemPointer item = state->closure.begin (); item != state->closure.end (); ++item)
122 {
123 if (item->dot != item->end_rhs ())
124 continue;
125
126 int r = aut.id (rule: item->rule);
127
128 const NameSet lookaheads = aut.lookaheads.value (key: item);
129
130 if (item->rule == grammar.goal)
131 accept_state = q;
132
133 for (const Name &s : lookaheads)
134 {
135 int &u = ACTION (q, aut.id (s));
136
137 if (u == 0)
138 u = - r;
139
140 else if (u < 0)
141 {
142 if (verbose)
143 qout() << "*** Warning. Found a reduce/reduce conflict in state " << q << " on token ``" << s << "'' between rule "
144 << r << " and " << -u << Qt::endl;
145
146 ++reduce_reduce_conflict_count;
147
148 u = qMax (a: u, b: -r);
149
150 if (verbose)
151 qout() << "\tresolved using rule " << -u << Qt::endl;
152 }
153
154 else if (u > 0)
155 {
156 if (item->rule->prec != grammar.names.end() && grammar.token_info.contains (key: s))
157 {
158 Grammar::TokenInfo info_r = grammar.token_info.value (key: item->rule->prec);
159 Grammar::TokenInfo info_s = grammar.token_info.value (key: s);
160
161 if (info_r.prec > info_s.prec)
162 u = -r;
163 else if (info_r.prec == info_s.prec)
164 {
165 switch (info_r.assoc) {
166 case Grammar::Left:
167 u = -r;
168 break;
169 case Grammar::Right:
170 // shift... nothing to do
171 break;
172 case Grammar::NonAssoc:
173 u = 0;
174 break;
175 } // switch
176 }
177 }
178
179 else
180 {
181 ++shift_reduce_conflict_count;
182
183 if (verbose)
184 qout() << "*** Warning. Found a shift/reduce conflict in state " << q << " on token ``" << s << "'' with rule " << r << Qt::endl;
185 }
186 }
187 }
188 }
189 }
190
191 if (shift_reduce_conflict_count || reduce_reduce_conflict_count)
192 {
193 if (shift_reduce_conflict_count != grammar.expected_shift_reduce
194 || reduce_reduce_conflict_count != grammar.expected_reduce_reduce)
195 qerr() << "*** Conflicts: " << shift_reduce_conflict_count << " shift/reduce, " << reduce_reduce_conflict_count << " reduce/reduce" << Qt::endl;
196
197 if (verbose)
198 qout() << Qt::endl << "*** Conflicts: " << shift_reduce_conflict_count << " shift/reduce, " << reduce_reduce_conflict_count << " reduce/reduce" << Qt::endl
199 << Qt::endl;
200 }
201
202 QBitArray used_rules{static_cast<int>(grammar.rules.size())};
203
204 int q = 0;
205 for (StatePointer state = aut.states.begin (); state != aut.states.end (); ++state, ++q)
206 {
207 for (int j = 0; j < terminal_count; ++j)
208 {
209 int &u = ACTION (q, j);
210
211 if (u < 0)
212 used_rules.setBit (-u - 1);
213 }
214 }
215
216 auto rule = grammar.rules.begin();
217 for (int i = 0; i < used_rules.size(); ++i, ++rule)
218 {
219 if (! used_rules.testBit (i))
220 {
221 if (rule != grammar.goal)
222 qerr() << "*** Warning: Rule ``" << *rule << "'' is useless!" << Qt::endl;
223 }
224 }
225
226 q = 0;
227 for (StatePointer state = aut.states.begin (); state != aut.states.end (); ++state, ++q)
228 {
229 for (int j = 0; j < terminal_count; ++j)
230 {
231 int &u = ACTION (q, j);
232
233 if (u >= 0)
234 continue;
235
236 RulePointer rule = std::next(x: grammar.rules.begin(), n: - u - 1);
237
238 if (state->defaultReduce == rule)
239 u = 0;
240 }
241 }
242
243 // ... compress the goto table
244 defgoto.resize (size: non_terminal_count);
245 for (int j = 0; j < non_terminal_count; ++j)
246 {
247 count.fill (t: 0, size: state_count);
248
249 int &mx = defgoto [j];
250
251 for (int i = 0; i < state_count; ++i)
252 {
253 int r = GOTO (i, j);
254
255 if (! r)
256 continue;
257
258 ++count [r];
259
260 if (count [r] > count [mx])
261 mx = r;
262 }
263 }
264
265 for (int i = 0; i < state_count; ++i)
266 {
267 for (int j = 0; j < non_terminal_count; ++j)
268 {
269 int &r = GOTO (i, j);
270
271 if (r == defgoto [j])
272 r = 0;
273 }
274 }
275
276 compressed_action (table, state_count, terminal_count);
277 compressed_goto (pgoto, state_count, non_terminal_count);
278
279 delete[] table;
280 table = nullptr;
281
282 delete[] pgoto;
283 pgoto = nullptr;
284
285#undef ACTION
286#undef GOTO
287
288 if (! grammar.merged_output.isEmpty())
289 {
290 QFile f(grammar.merged_output);
291 if (! f.open (flags: QFile::WriteOnly))
292 {
293 fprintf (stderr, format: "*** cannot create %s\n", qPrintable(grammar.merged_output));
294 return;
295 }
296
297 QTextStream out (&f);
298
299 // copyright headers must come first, otherwise the headers tests will fail
300 if (copyright)
301 {
302 out << copyrightHeader()
303 << privateCopyrightHeader()
304 << Qt::endl;
305 }
306
307 out << "// This file was generated by qlalr - DO NOT EDIT!\n";
308
309 out << startIncludeGuard(fileName: grammar.merged_output) << Qt::endl;
310
311 if (copyright) {
312 out << "#if defined(ERROR)" << Qt::endl
313 << "# undef ERROR" << Qt::endl
314 << "#endif" << Qt::endl << Qt::endl;
315 }
316
317 generateDecl (out);
318 generateImpl (out);
319 out << p.decls();
320 out << p.impls();
321 out << Qt::endl;
322
323 out << endIncludeGuard(fileName: grammar.merged_output) << Qt::endl;
324
325 return;
326 }
327
328 // default behaviour
329 QString declFileName = grammar.table_name.toLower () + "_p.h"_L1;
330 QString bitsFileName = grammar.table_name.toLower () + ".cpp"_L1;
331
332 { // decls...
333 QFile f (declFileName);
334 f.open (flags: QFile::WriteOnly);
335 QTextStream out (&f);
336
337 QString prot = declFileName.toUpper ().replace (before: QLatin1Char ('.'), after: QLatin1Char ('_'));
338
339 // copyright headers must come first, otherwise the headers tests will fail
340 if (copyright)
341 {
342 out << copyrightHeader()
343 << privateCopyrightHeader()
344 << Qt::endl;
345 }
346
347 out << "// This file was generated by qlalr - DO NOT EDIT!\n";
348
349 out << "#ifndef " << prot << Qt::endl
350 << "#define " << prot << Qt::endl
351 << Qt::endl;
352
353 if (copyright) {
354 out << "#include <QtCore/qglobal.h>" << Qt::endl << Qt::endl;
355 out << "QT_BEGIN_NAMESPACE" << Qt::endl << Qt::endl;
356 }
357 generateDecl (out);
358 if (copyright)
359 out << "QT_END_NAMESPACE" << Qt::endl;
360
361 out << "#endif // " << prot << Qt::endl << Qt::endl;
362 } // end decls
363
364 { // bits...
365 QFile f (bitsFileName);
366 f.open (flags: QFile::WriteOnly);
367 QTextStream out (&f);
368
369 // copyright headers must come first, otherwise the headers tests will fail
370 if (copyright)
371 out << copyrightHeader();
372
373 out << "// This file was generated by qlalr - DO NOT EDIT!\n";
374
375 out << "#include \"" << declFileName << "\"" << Qt::endl << Qt::endl;
376 if (copyright)
377 out << "QT_BEGIN_NAMESPACE" << Qt::endl << Qt::endl;
378 generateImpl(out);
379 if (copyright)
380 out << "QT_END_NAMESPACE" << Qt::endl;
381
382 } // end bits
383
384 if (! grammar.decl_file_name.isEmpty ())
385 {
386 QFile f (grammar.decl_file_name);
387 f.open (flags: QFile::WriteOnly);
388 QTextStream out (&f);
389 out << p.decls();
390 }
391
392 if (! grammar.impl_file_name.isEmpty ())
393 {
394 QFile f (grammar.impl_file_name);
395 f.open (flags: QFile::WriteOnly);
396 QTextStream out (&f);
397 out << p.impls();
398 }

409void CppGenerator::generateDecl (QTextStream &out)
410{
411 out << "class " << grammar.table_name << Qt::endl
412 << "{" << Qt::endl
413 << "public:" << Qt::endl
414 << " enum VariousConstants {" << Qt::endl;
415
416 for (const Name &t : std::as_const(t&: grammar.terminals))
417 {
418 QString name = *t;
419 int value = std::distance (first: grammar.names.begin (), last: t);
420
421 if (name == "$end"_L1)
422 name = "EOF_SYMBOL"_L1;
423
424 else if (name == "$accept"_L1)
425 name = "ACCEPT_SYMBOL"_L1;
426
427 else
428 name.prepend (s: grammar.token_prefix);
429
430 out << " " << name << " = " << value << "," << Qt::endl;
431 }
432
433 out << Qt::endl
434 << " ACCEPT_STATE = " << accept_state << "," << Qt::endl
435 << " RULE_COUNT = " << grammar.rules.size () << "," << Qt::endl
436 << " STATE_COUNT = " << state_count << "," << Qt::endl
437 << " TERMINAL_COUNT = " << terminal_count << "," << Qt::endl
438 << " NON_TERMINAL_COUNT = " << non_terminal_count << "," << Qt::endl
439 << Qt::endl
440 << " GOTO_INDEX_OFFSET = " << compressed_action.index.size () << "," << Qt::endl
441 << " GOTO_INFO_OFFSET = " << compressed_action.info.size () << "," << Qt::endl
442 << " GOTO_CHECK_OFFSET = " << compressed_action.check.size () << Qt::endl
443 << " };" << Qt::endl
444 << Qt::endl
445 << " static const char *const spell[];" << Qt::endl
446 << " static const short lhs[];" << Qt::endl
447 << " static const short rhs[];" << Qt::endl;
448
449 if (debug_info)
450 {
451 QString prot = debugInfoProt();
452
453 out << Qt::endl << "#ifndef " << prot << Qt::endl
454 << " static const int rule_index[];" << Qt::endl
455 << " static const int rule_info[];" << Qt::endl
456 << "#endif // " << prot << Qt::endl << Qt::endl;
457 }
458
459 out << " static const short goto_default[];" << Qt::endl
460 << " static const short action_default[];" << Qt::endl
461 << " static const short action_index[];" << Qt::endl
462 << " static const short action_info[];" << Qt::endl
463 << " static const short action_check[];" << Qt::endl
464 << Qt::endl
465 << " static inline int nt_action (int state, int nt)" << Qt::endl
466 << " {" << Qt::endl
467 << " const int yyn = action_index [GOTO_INDEX_OFFSET + state] + nt;" << Qt::endl
468 << " if (yyn < 0 || action_check [GOTO_CHECK_OFFSET + yyn] != nt)" << Qt::endl
469 << " return goto_default [nt];" << Qt::endl
470 << Qt::endl
471 << " return action_info [GOTO_INFO_OFFSET + yyn];" << Qt::endl
472 << " }" << Qt::endl
473 << Qt::endl
474 << " static inline int t_action (int state, int token)" << Qt::endl
475 << " {" << Qt::endl
476 << " const int yyn = action_index [state] + token;" << Qt::endl
477 << Qt::endl
478 << " if (yyn < 0 || action_check [yyn] != token)" << Qt::endl
479 << " return - action_default [state];" << Qt::endl
480 << Qt::endl
481 << " return action_info [yyn];" << Qt::endl
482 << " }" << Qt::endl
483 << "};" << Qt::endl
484 << Qt::endl
485 << Qt::endl;
486}

488void CppGenerator::generateImpl (QTextStream &out)
489{
490 int idx = 0;
491
492 out << "const char *const " << grammar.table_name << "::spell [] = {";
493 idx = 0;
494
495 QMap<Name, int> name_ids;
496 bool first_nt = true;
497
498 for (Name t = grammar.names.begin (); t != grammar.names.end (); ++t, ++idx)
499 {
500 bool terminal = grammar.isTerminal (name: t);
501
502 if (! (debug_info || terminal))
503 break;
504
505 name_ids.insert (key: t, value: idx);
506
507 generateSeparator(i: idx, out);
508
509 if (terminal)
510 {
511 QString spell = grammar.spells.value (key: t);
512
513 if (spell.isEmpty ())
514 out << "0";
515 else
516 out << "\"" << spell << "\"";
517 }
518 else
519 {
520 if (first_nt)
521 {
522 first_nt = false;
523 QString prot = debugInfoProt();
524 out << Qt::endl << "#ifndef " << prot << Qt::endl;
525 }
526 out << "\"" << *t << "\"";
527 }
528 }
529
530 if (debug_info)
531 out << Qt::endl << "#endif // " << debugInfoProt() << Qt::endl;
532
533 out << Qt::endl << "};" << Qt::endl << Qt::endl;
534
535 out << "const short " << grammar.table_name << "::lhs [] = {";
536 idx = 0;
537 for (RulePointer rule = grammar.rules.begin (); rule != grammar.rules.end (); ++rule, ++idx)
538 {
539 generateSeparator(i: idx, out);
540
541 out << aut.id (name: rule->lhs);
542 }
543 out << Qt::endl << "};" << Qt::endl << Qt::endl;
544
545 out << "const short " << grammar.table_name << "::rhs [] = {";
546 idx = 0;
547 for (RulePointer rule = grammar.rules.begin (); rule != grammar.rules.end (); ++rule, ++idx)
548 {
549 generateSeparator(i: idx, out);
550
551 out << rule->rhs.size ();
552 }
553 out << Qt::endl << "};" << Qt::endl << Qt::endl;
554
555 if (debug_info)
556 {
557 QString prot = debugInfoProt();
558
559 out << Qt::endl << "#ifndef " << prot << Qt::endl;
560 out << "const int " << grammar.table_name << "::rule_info [] = {";
561 idx = 0;
562 for (auto rule = grammar.rules.cbegin (); rule != grammar.rules.cend (); ++rule, ++idx)
563 {
564 generateSeparator(i: idx, out);
565
566 out << name_ids.value(key: rule->lhs);
567
568 for (const Name &n : rule->rhs)
569 out << ", " << name_ids.value (key: n);
570 }
571 out << Qt::endl << "};" << Qt::endl << Qt::endl;
572
573 out << "const int " << grammar.table_name << "::rule_index [] = {";
574 idx = 0;
575 size_t offset = 0;
576 for (RulePointer rule = grammar.rules.begin (); rule != grammar.rules.end (); ++rule, ++idx)
577 {
578 generateSeparator(i: idx, out);
579
580 out << offset;
581 offset += rule->rhs.size () + 1;
582 }
583 out << Qt::endl << "};" << Qt::endl
584 << "#endif // " << prot << Qt::endl << Qt::endl;
585 }
586
587 out << "const short " << grammar.table_name << "::action_default [] = {";
588 idx = 0;
589 for (StatePointer state = aut.states.begin (); state != aut.states.end (); ++state, ++idx)
590 {
591 generateSeparator(i: idx, out);
592
593 if (state->defaultReduce != grammar.rules.end ())
594 out << aut.id (rule: state->defaultReduce);
595 else
596 out << "0";
597 }
598 out << Qt::endl << "};" << Qt::endl << Qt::endl;
599
600 out << "const short " << grammar.table_name << "::goto_default [] = {";
601 generateList(list: defgoto, out);
602 out << Qt::endl << "};" << Qt::endl << Qt::endl;
603
604 out << "const short " << grammar.table_name << "::action_index [] = {";
605 generateList(list: compressed_action.index, out);
606 out << "," << Qt::endl;
607 generateList(list: compressed_goto.index, out);
608 out << Qt::endl << "};" << Qt::endl << Qt::endl;
609
610 out << "const short " << grammar.table_name << "::action_info [] = {";
611 generateList(list: compressed_action.info, out);
612 out << "," << Qt::endl;
613 generateList(list: compressed_goto.info, out);
614 out << Qt::endl << "};" << Qt::endl << Qt::endl;
615
616 out << "const short " << grammar.table_name << "::action_check [] = {";
617 generateList(list: compressed_action.check, out);
618 out << "," << Qt::endl;
619 generateList(list: compressed_goto.check, out);
620 out << Qt::endl << "};" << Qt::endl << Qt::endl;
621}
