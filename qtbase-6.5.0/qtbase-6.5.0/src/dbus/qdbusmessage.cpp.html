70/*!
71 \internal
72 Constructs a DBusMessage object from \a message. The returned value must be de-referenced
73 with q_dbus_message_unref. The \a capabilities flags indicates which capabilities to use.
74
75 The \a error object is set to indicate the error if anything went wrong with the
76 marshalling. Usually, this error message will be placed in the reply, as if the call failed.
77 The \a error pointer must not be null.
78*/
79DBusMessage *QDBusMessagePrivate::toDBusMessage(const QDBusMessage &message, QDBusConnection::ConnectionCapabilities capabilities,
80 QDBusError *error)
81{
82 if (!qdbus_loadLibDBus()) {
83 *error = QDBusError(QDBusError::Failed, "Could not open lidbus-1 library"_L1);
84 return nullptr;
85 }
86
87 DBusMessage *msg = nullptr;
88 const QDBusMessagePrivate *d_ptr = message.d_ptr;
89
90 switch (d_ptr->type) {
91 case QDBusMessage::InvalidMessage:
92 //qDebug() << "QDBusMessagePrivate::toDBusMessage" << "message is invalid";
93 break;
94 case QDBusMessage::MethodCallMessage:
95 // only service and interface can be empty -> path and name must not be empty
96 if (!d_ptr->parametersValidated) {
97 if (!QDBusUtil::checkBusName(name: d_ptr->service, empty: QDBusUtil::EmptyAllowed, error))
98 return nullptr;
99 if (!QDBusUtil::checkObjectPath(path: d_ptr->path, empty: QDBusUtil::EmptyNotAllowed, error))
100 return nullptr;
101 if (!QDBusUtil::checkInterfaceName(name: d_ptr->interface, empty: QDBusUtil::EmptyAllowed, error))
102 return nullptr;
103 if (!QDBusUtil::checkMemberName(name: d_ptr->name, empty: QDBusUtil::EmptyNotAllowed, error, nameType: "method"))
104 return nullptr;
105 }
106
107 msg = q_dbus_message_new_method_call(bus_name: data(arr: d_ptr->service.toUtf8()), path: d_ptr->path.toUtf8(),
108 interface: data(arr: d_ptr->interface.toUtf8()), method: d_ptr->name.toUtf8());
109 q_dbus_message_set_auto_start( message: msg, auto_start: d_ptr->autoStartService );
110 q_dbus_message_set_allow_interactive_authorization(message: msg, allow: d_ptr->interactiveAuthorizationAllowed);
111
112 break;
113 case QDBusMessage::ReplyMessage:
114 msg = q_dbus_message_new(DBUS_MESSAGE_TYPE_METHOD_RETURN);
115 if (!d_ptr->localMessage) {
116 q_dbus_message_set_destination(message: msg, destination: q_dbus_message_get_sender(message: d_ptr->reply));
117 q_dbus_message_set_reply_serial(message: msg, reply_serial: q_dbus_message_get_serial(message: d_ptr->reply));
118 }
119 break;
120 case QDBusMessage::ErrorMessage:
121 // error name can't be empty
122 if (!d_ptr->parametersValidated
123 && !QDBusUtil::checkErrorName(name: d_ptr->name, empty: QDBusUtil::EmptyNotAllowed, error))
124 return nullptr;
125
126 msg = q_dbus_message_new(DBUS_MESSAGE_TYPE_ERROR);
127 q_dbus_message_set_error_name(message: msg, name: d_ptr->name.toUtf8());
128 if (!d_ptr->localMessage) {
129 q_dbus_message_set_destination(message: msg, destination: q_dbus_message_get_sender(message: d_ptr->reply));
130 q_dbus_message_set_reply_serial(message: msg, reply_serial: q_dbus_message_get_serial(message: d_ptr->reply));
131 }
132 break;
133 case QDBusMessage::SignalMessage:
134 // only the service name can be empty here
135 if (!d_ptr->parametersValidated) {
136 if (!QDBusUtil::checkBusName(name: d_ptr->service, empty: QDBusUtil::EmptyAllowed, error))
137 return nullptr;
138 if (!QDBusUtil::checkObjectPath(path: d_ptr->path, empty: QDBusUtil::EmptyNotAllowed, error))
139 return nullptr;
140 if (!QDBusUtil::checkInterfaceName(name: d_ptr->interface, empty: QDBusUtil::EmptyAllowed, error))
141 return nullptr;
142 if (!QDBusUtil::checkMemberName(name: d_ptr->name, empty: QDBusUtil::EmptyNotAllowed, error, nameType: "method"))
143 return nullptr;
144 }
145
146 msg = q_dbus_message_new_signal(path: d_ptr->path.toUtf8(), interface: d_ptr->interface.toUtf8(),
147 name: d_ptr->name.toUtf8());
148 q_dbus_message_set_destination(message: msg, destination: data(arr: d_ptr->service.toUtf8()));
149 break;
150 }
151
152 // if we got here, the parameters validated
153 // and since the message parameters cannot be changed once the message is created
154 // we can record this fact
155 d_ptr->parametersValidated = true;
156
157 QDBusMarshaller marshaller(capabilities);
158 QVariantList::ConstIterator it = d_ptr->arguments.constBegin();
159 QVariantList::ConstIterator cend = d_ptr->arguments.constEnd();
160 q_dbus_message_iter_init_append(message: msg, iter: &marshaller.iterator);
161 if (!d_ptr->message.isEmpty())
162 // prepend the error message
163 marshaller.append(arg: d_ptr->message);
164 for ( ; it != cend; ++it)
165 marshaller.appendVariantInternal(arg: *it);
166
167 // check if everything is ok
168 if (marshaller.ok)
169 return msg;
170
171 // not ok;
172 q_dbus_message_unref(message: msg);
173 *error = QDBusError(QDBusError::Failed, "Marshalling failed: "_L1 + marshaller.errorString);
174 return nullptr;
175}

198/*!
199 \internal
200 Constructs a QDBusMessage by parsing the given DBusMessage object.
201*/
202QDBusMessage QDBusMessagePrivate::fromDBusMessage(DBusMessage *dmsg, QDBusConnection::ConnectionCapabilities capabilities)
203{
204 QDBusMessage message;
205 if (!dmsg)
206 return message;
207
208 message.d_ptr->type = QDBusMessage::MessageType(q_dbus_message_get_type(message: dmsg));
209 message.d_ptr->path = QString::fromUtf8(utf8: q_dbus_message_get_path(message: dmsg));
210 message.d_ptr->interface = QString::fromUtf8(utf8: q_dbus_message_get_interface(message: dmsg));
211 message.d_ptr->name = message.d_ptr->type == DBUS_MESSAGE_TYPE_ERROR ?
212 QString::fromUtf8(utf8: q_dbus_message_get_error_name(message: dmsg)) :
213 QString::fromUtf8(utf8: q_dbus_message_get_member(message: dmsg));
214 message.d_ptr->service = QString::fromUtf8(utf8: q_dbus_message_get_sender(message: dmsg));
215 message.d_ptr->signature = QString::fromUtf8(utf8: q_dbus_message_get_signature(message: dmsg));
216 message.d_ptr->interactiveAuthorizationAllowed = q_dbus_message_get_allow_interactive_authorization(message: dmsg);
217 message.d_ptr->msg = q_dbus_message_ref(message: dmsg);
218
219 QDBusDemarshaller demarshaller(capabilities);
220 demarshaller.message = q_dbus_message_ref(message: dmsg);
221 if (q_dbus_message_iter_init(message: demarshaller.message, iter: &demarshaller.iterator))
222 while (!demarshaller.atEnd())
223 message << demarshaller.toVariantInternal();
224 return message;
225}

232QDBusMessage QDBusMessagePrivate::makeLocal(const QDBusConnectionPrivate &conn,
233 const QDBusMessage &asSent)
234{
235 // simulate the message being sent to the bus and then received back
236 // the only field that the bus sets when delivering the message
237 // (as opposed to the message as we send it), is the sender
238 // so we simply set the sender to our unique name
239
240 // determine if we are carrying any complex types
241 QString computedSignature;
242 QVariantList::ConstIterator it = asSent.d_ptr->arguments.constBegin();
243 QVariantList::ConstIterator end = asSent.d_ptr->arguments.constEnd();
244 for ( ; it != end; ++it) {
245 QMetaType id = it->metaType();
246 const char *signature = QDBusMetaType::typeToSignature(type: id);
247 if ((id.id() != QMetaType::QStringList && id.id() != QMetaType::QByteArray &&
248 qstrlen(str: signature) != 1) || id == QMetaType::fromType<QDBusVariant>()) {
249 // yes, we are
250 // we must marshall and demarshall again so as to create QDBusArgument
251 // entries for the complex types
252 QDBusError error;
253 DBusMessage *message = toDBusMessage(message: asSent, capabilities: conn.connectionCapabilities(), error: &error);
254 if (!message) {
255 // failed to marshall, so it's a call error
256 return QDBusMessage::createError(err: error);
257 }
258
259 q_dbus_message_set_sender(message, sender: conn.baseService.toUtf8());
260
261 QDBusMessage retval = fromDBusMessage(dmsg: message, capabilities: conn.connectionCapabilities());
262 retval.d_ptr->localMessage = true;
263 q_dbus_message_unref(message);
264 if (retval.d_ptr->service.isEmpty())
265 retval.d_ptr->service = conn.baseService;
266 return retval;
267 } else {
268 computedSignature += QLatin1StringView(signature);
269 }
270 }
271
272 // no complex types seen
273 // optimize by using the variant list itself
274 QDBusMessage retval;
275 QDBusMessagePrivate *d = retval.d_ptr;
276 d->arguments = asSent.d_ptr->arguments;
277 d->path = asSent.d_ptr->path;
278 d->interface = asSent.d_ptr->interface;
279 d->name = asSent.d_ptr->name;
280 d->message = asSent.d_ptr->message;
281 d->type = asSent.d_ptr->type;
282
283 d->service = conn.baseService;
284 d->signature = computedSignature;
285 d->localMessage = true;
286 return retval;
287}

289QDBusMessage QDBusMessagePrivate::makeLocalReply(const QDBusConnectionPrivate &conn,
290 const QDBusMessage &callMsg)
291{
292 // simulate the reply (return or error) message being sent to the bus and
293 // then received back.
294 if (callMsg.d_ptr->localReply)
295 return makeLocal(conn, asSent: *callMsg.d_ptr->localReply);
296 return QDBusMessage(); // failed
297}

299/*!
300 \class QDBusMessage
301 \inmodule QtDBus
302 \since 4.2
303
304 \brief The QDBusMessage class represents one message sent or
305 received over the D-Bus bus.
306
307 This object can represent any of the four different types of
308 messages (MessageType) that can occur on the bus:
309
310 \list
311 \li Method calls
312 \li Method return values
313 \li Signal emissions
314 \li Error codes
315 \endlist
316
317 Objects of this type are created with the static createError(),
318 createMethodCall() and createSignal() functions. Use the
319 QDBusConnection::send() function to send the messages.
320*/
321
322/*!
323 \enum QDBusMessage::MessageType
324 The possible message types:
325
326 \value MethodCallMessage a message representing an outgoing or incoming method call
327 \value SignalMessage a message representing an outgoing or incoming signal emission
328 \value ReplyMessage a message representing the return values of a method call
329 \value ErrorMessage a message representing an error condition in response to a method call
330 \value InvalidMessage an invalid message: this is never set on messages received from D-Bus
331*/
332
333/*!
334 Constructs a new DBus message with the given \a path, \a interface
335 and \a name, representing a signal emission.
336
337 A DBus signal is emitted from one application and is received by
338 all applications that are listening for that signal from that
339 interface.
340
341 The QDBusMessage object that is returned can be sent using the
342 QDBusConnection::send() function.
343*/
344QDBusMessage QDBusMessage::createSignal(const QString &path, const QString &interface,
345 const QString &name)
346{
347 QDBusMessage message;
348 message.d_ptr->type = SignalMessage;
349 message.d_ptr->path = path;
350 message.d_ptr->interface = interface;
351 message.d_ptr->name = name;
352
353 return message;
354}

381/*!
382 Constructs a new DBus message representing a method call.
383 A method call always informs its destination address
384 (\a service, \a path, \a interface and \a method).
385
386 The DBus bus allows calling a method on a given remote object without specifying the
387 destination interface, if the method name is unique. However, if two interfaces on the
388 remote object export the same method name, the result is undefined (one of the two may be
389 called or an error may be returned).
390
391 When using DBus in a peer-to-peer context (i.e., not on a bus), the \a service parameter is
392 optional.
393
394 The QDBusInterface class provides a simpler abstraction to synchronous
395 method calling.
396
397 This function returns a QDBusMessage object that can be sent with
398 QDBusConnection::call().
399*/
400QDBusMessage QDBusMessage::createMethodCall(const QString &service, const QString &path,
401 const QString &interface, const QString &method)
402{
403 QDBusMessage message;
404 message.d_ptr->type = MethodCallMessage;
405 message.d_ptr->service = service;
406 message.d_ptr->path = path;
407 message.d_ptr->interface = interface;
408 message.d_ptr->name = method;
409
410 return message;
411}

440/*!
441 \fn QDBusMessage QDBusMessage::createReply(const QList<QVariant> &arguments) const
442
443 Constructs a new DBus message representing a reply, with the given
444 \a arguments.
445*/
446QDBusMessage QDBusMessage::createReply(const QVariantList &arguments) const
447{
448 QDBusMessage reply;
449 reply.setArguments(arguments);
450 reply.d_ptr->type = ReplyMessage;
451 if (d_ptr->msg)
452 reply.d_ptr->reply = q_dbus_message_ref(message: d_ptr->msg);
453 if (d_ptr->localMessage) {
454 reply.d_ptr->localMessage = true;
455 d_ptr->localReply = new QDBusMessage(reply); // keep an internal copy
456 }
457
458 // the reply must have a msg or be a local-loop optimization
459 Q_ASSERT(reply.d_ptr->reply || reply.d_ptr->localMessage);
460 return reply;
461}

612/*!
613 Returns the flag that indicates if this message should see a reply
614 or not. This is only meaningful for \l {MethodCallMessage}{method
615 call messages}: any other kind of message cannot have replies and
616 this function will always return false for them.
617*/
618bool QDBusMessage::isReplyRequired() const
619{
620 // Only method calls can have replies
621 if (d_ptr->type != QDBusMessage::MethodCallMessage)
622 return false;
623
624 if (!d_ptr->msg)
625 return d_ptr->localMessage; // if it's a local message, reply is required
626 return !q_dbus_message_get_no_reply(message: d_ptr->msg);
627}

658/*!
659 Sets the auto start flag to \a enable. This flag only makes sense
660 for method call messages, where it tells the D-Bus server to
661 either auto start the service responsible for the service name, or
662 not to auto start it.
663
664 By default this flag is true, i.e. a service is autostarted.
665 This means:
666
667 When the service that this method call is sent to is already
668 running, the method call is sent to it. If the service is not
669 running yet, the D-Bus daemon is requested to autostart the
670 service that is assigned to this service name. This is
671 handled by .service files that are placed in a directory known
672 to the D-Bus server. These files then each contain a service
673 name and the path to a program that should be executed when
674 this service name is requested.
675
676 \since 4.7
677*/
678void QDBusMessage::setAutoStartService(bool enable)
679{
680 d_ptr->autoStartService = enable;
681}

769QDBusMessage::MessageType QDBusMessage::type() const

804static void debugVariantList(QDebug dbg, const QVariantList &list)
805{
806 bool first = true;
807 QVariantList::ConstIterator it = list.constBegin();
808 QVariantList::ConstIterator end = list.constEnd();
809 for ( ; it != end; ++it) {

813 first = false;
