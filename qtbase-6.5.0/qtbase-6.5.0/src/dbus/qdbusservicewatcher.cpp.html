49void QDBusServiceWatcherPrivate::_q_serviceOwnerChanged(const QString &service, const QString &oldOwner, const QString &newOwner)
50{
51 Q_Q(QDBusServiceWatcher);
52 emit q->serviceOwnerChanged(service, oldOwner, newOwner);
53 if (oldOwner.isEmpty())
54 emit q->serviceRegistered(service);
55 else if (newOwner.isEmpty())
56 emit q->serviceUnregistered(service);
57}
58
59void QDBusServiceWatcherPrivate::setConnection(const QStringList &services,
60 const QDBusConnection &c,
61 QDBusServiceWatcher::WatchMode wm)
62{
63 if (connection.isConnected()) {
64 // remove older rules
65 for (const QString &s : std::as_const(t: watchedServicesData.value()))
66 removeService(service: s);
67 }
68
69 connection = c;
70 watchMode.setValueBypassingBindings(wm); // caller has to call notify()
71 watchedServicesData.setValueBypassingBindings(services); // caller has to call notify()
72
73 if (connection.isConnected()) {
74 // add new rules
75 for (const QString &s : std::as_const(t: watchedServicesData.value()))
76 addService(service: s);
77 }
78}

80void QDBusServiceWatcherPrivate::addService(const QString &service)
81{
82 QDBusConnectionPrivate *d = QDBusConnectionPrivate::d(q: connection);
83 if (d && d->shouldWatchService(service))
84 d->watchService(service, mode: watchMode, obj: q_func(), SLOT(_q_serviceOwnerChanged(QString,QString,QString)));
85}
86
87void QDBusServiceWatcherPrivate::removeService(const QString &service)
88{
89 QDBusConnectionPrivate *d = QDBusConnectionPrivate::d(q: connection);
90 if (d && d->shouldWatchService(service))
91 d->unwatchService(service, mode: watchMode, obj: q_func(), SLOT(_q_serviceOwnerChanged(QString,QString,QString)));
92}

94/*!
95 \class QDBusServiceWatcher
96 \since 4.6
97 \inmodule QtDBus
98
99 \brief The QDBusServiceWatcher class allows the user to watch for a bus service change.
100
101 A QDBusServiceWatcher object can be used to notify the application about
102 an ownership change of a service name on the bus. It has three watch
103 modes:
104
105 \list
106 \li Watching for service registration only.
107 \li Watching for service unregistration only.
108 \li Watching for any kind of service ownership change (the default mode).
109 \endlist
110
111 Besides being created or deleted, services may change owners without a
112 unregister/register operation happening. So the serviceRegistered()
113 and serviceUnregistered() signals may not be emitted if that
114 happens.
115
116 This class is more efficient than using the
117 QDBusConnectionInterface::serviceOwnerChanged() signal because it allows
118 one to receive only the signals for which the class is interested in.
119
120 Ending a service name with the character '*' will match all service names
121 within the specified namespace.
122
123 For example "com.example.backend1*" will match
124 \list
125 \li com.example.backend1
126 \li com.example.backend1.foo
127 \li com.example.backend1.foo.bar
128 \endlist
129 Substrings in the same domain will not be matched, i.e "com.example.backend12".
130
131 \sa QDBusConnection
132*/
133
134/*!
135 \enum QDBusServiceWatcher::WatchModeFlag
136
137 QDBusServiceWatcher supports three different watch modes, which are configured by this flag:
138
139 \value WatchForRegistration watch for service registration only, ignoring
140 any signals related to other service ownership change.
141
142 \value WatchForUnregistration watch for service unregistration only,
143 ignoring any signals related to other service ownership change.
144
145 \value WatchForOwnerChange watch for any kind of service ownership
146 change.
147*/
148
149/*!
150 \property QDBusServiceWatcher::watchMode
151 \brief the current watch mode for this QDBusServiceWatcher object.
152
153 The default value for this property is
154 QDBusServiceWatcher::WatchForOwnershipChange.
155*/
156
157/*!
158 \property QDBusServiceWatcher::watchedServices
159 \brief the list of services watched.
160
161 \note Modifying this list with setServicesWatched() is an expensive
162 operation. If you can, prefer to change it by way of addWatchedService()
163 and removeWatchedService().
164*/
165
166/*!
167 \fn void QDBusServiceWatcher::serviceRegistered(const QString &serviceName)
168
169 This signal is emitted whenever this object detects that the service \a
170 serviceName became available on the bus.
171
172 \sa serviceUnregistered(), serviceOwnerChanged()
173*/
174
175/*!
176 \fn void QDBusServiceWatcher::serviceUnregistered(const QString &serviceName)
177
178 This signal is emitted whenever this object detects that the service \a
179 serviceName was unregistered from the bus and is no longer available.
180
181 \sa serviceRegistered(), serviceOwnerChanged()
182*/
183
184/*!
185 \fn void QDBusServiceWatcher::serviceOwnerChanged(const QString &serviceName, const QString &oldOwner, const QString &newOwner)
186
187 This signal is emitted whenever this object detects that there was a
188 service ownership change relating to the \a serviceName service. The \a
189 oldOwner parameter contains the old owner name and \a newOwner is the new
190 owner. Both \a oldOwner and \a newOwner are unique connection names.
191
192 Note that this signal is also emitted whenever the \a serviceName service
193 was registered or unregistered. If it was registered, \a oldOwner will
194 contain an empty string, whereas if it was unregistered, \a newOwner will
195 contain an empty string.
196
197 If you need only to find out if the service is registered or unregistered
198 only, without being notified that the ownership changed, consider using
199 the specific modes for those operations. This class is more efficient if
200 you use the more specific modes.
201
202 \sa serviceRegistered(), serviceUnregistered()
203*/
204

249/*!
250 Sets the list of D-Bus services being watched to be \a services.
251
252 Note that setting the entire list means removing all previous rules for
253 watching services and adding new ones. This is an expensive operation and
254 should be avoided, if possible. Instead, use addWatchedService() and
255 removeWatchedService() if you can to manipulate entries in the list.
256
257 Removes any existing binding of watchedServices.
258*/
259void QDBusServiceWatcher::setWatchedServices(const QStringList &services)
260{
261 Q_D(QDBusServiceWatcher);
262 d->watchedServicesData.removeBindingUnlessInWrapper();
263 if (services == d->watchedServicesData)
264 return;
265 d->setConnection(services, c: d->connection, wm: d->watchMode);
266 d->watchedServicesData.notify();
267}

275/*!
276 Adds \a newService to the list of services to be watched by this object.
277 This function is more efficient than setWatchedServices() and should be
278 used whenever possible to add services.
279
280 Removes any existing binding of watchedServices.
281*/
282void QDBusServiceWatcher::addWatchedService(const QString &newService)
283{
284 Q_D(QDBusServiceWatcher);
285 d->watchedServicesData.removeBindingUnlessInWrapper();
286 if (d->watchedServicesData.value().contains(str: newService))
287 return;
288 d->addService(service: newService);
289
290 auto templist = d->watchedServicesData.valueBypassingBindings();
291 templist << newService;
292 d->watchedServicesData.setValueBypassingBindings(templist);
293
294 d->watchedServicesData.notify();
295}
296
297/*!
298 Removes the \a service from the list of services being watched by this
299 object. Note that D-Bus notifications are asynchronous, so there may
300 still be signals pending delivery about \a service. Those signals will
301 still be emitted whenever the D-Bus messages are processed.
302
303 Removes any existing binding of watchedServices.
304
305 This function returns \c true if any services were removed.
306*/
307bool QDBusServiceWatcher::removeWatchedService(const QString &service)
308{
309 Q_D(QDBusServiceWatcher);
310 d->watchedServicesData.removeBindingUnlessInWrapper();
311 d->removeService(service);
312 auto tempList = d->watchedServicesData.value();
313 bool result = tempList.removeOne(t: service);
314 if (result) {
315 d->watchedServicesData.setValueBypassingBindings(tempList);
316 d->watchedServicesData.notify();
317 return true;
318 } else {
319 // nothing changed
320 return false;
321 }
322}

354/*!
355 Sets the D-Bus connection that this object is attached to be \a
356 connection. All services watched will be transferred to this connection.
357
358 Note that QDBusConnection objects are reference counted:
359 QDBusServiceWatcher will keep a reference for this connection while it
360 exists. The connection is not closed until the reference count drops to
361 zero, so this will ensure that any notifications are received while this
362 QDBusServiceWatcher object exists.
363
364 \sa connection()
365*/
366void QDBusServiceWatcher::setConnection(const QDBusConnection &connection)
367{
368 Q_D(QDBusServiceWatcher);
369 if (connection.name() == d->connection.name())
370 return;
371 d->setConnection(services: d->watchedServicesData, c: connection, wm: d->watchMode);
372}
