76static QString generateSubObjectXml(QObject *object)
77{
78 QString retval;
79 const QObjectList &objs = object->children();
80 QObjectList::ConstIterator it = objs.constBegin();
81 QObjectList::ConstIterator end = objs.constEnd();
82 for ( ; it != end; ++it) {
83 QString name = (*it)->objectName();
84 if (!name.isEmpty() && QDBusUtil::isValidPartOfObjectPath(path: name))
85 retval += " <node name=\""_L1 + name + "\"/>\n"_L1;
86 }
87 return retval;
88}

92QString qDBusIntrospectObject(const QDBusConnectionPrivate::ObjectTreeNode &node, const QString &path)
93{
94 // object may be null
95
96 QString xml_data(DBUS_INTROSPECT_1_0_XML_DOCTYPE_DECL_NODE ""_L1);
97 xml_data += "<node>\n"_L1;
98
99 if (node.obj) {
100 Q_ASSERT_X(QThread::currentThread() == node.obj->thread(),
101 "QDBusConnection: internal threading error",
102 "function called for an object that is in another thread!!");
103
104 if (node.flags & (QDBusConnection::ExportScriptableContents
105 | QDBusConnection::ExportNonScriptableContents)) {
106 // create XML for the object itself
107 const QMetaObject *mo = node.obj->metaObject();
108 for ( ; mo != &QObject::staticMetaObject; mo = mo->superClass())
109 xml_data += qDBusGenerateMetaObjectXml(interface: node.interfaceName, mo, base: mo->superClass(),
110 flags: node.flags);
111 }
112
113 // does this object have adaptors?
114 QDBusAdaptorConnector *connector;
115 if (node.flags & QDBusConnection::ExportAdaptors &&
116 (connector = qDBusFindAdaptorConnector(object: node.obj))) {
117
118 // trasverse every adaptor in this object
119 QDBusAdaptorConnector::AdaptorMap::ConstIterator it = connector->adaptors.constBegin();
120 QDBusAdaptorConnector::AdaptorMap::ConstIterator end = connector->adaptors.constEnd();
121 for ( ; it != end; ++it) {
122 // add the interface:
123 QString ifaceXml = QDBusAbstractAdaptorPrivate::retrieveIntrospectionXml(adaptor: it->adaptor);
124 if (ifaceXml.isEmpty()) {
125 // add the interface's contents:
126 ifaceXml += qDBusGenerateMetaObjectXml(interface: QString::fromLatin1(ba: it->interface),
127 mo: it->adaptor->metaObject(),
128 base: &QDBusAbstractAdaptor::staticMetaObject,
129 flags: QDBusConnection::ExportScriptableContents
130 | QDBusConnection::ExportNonScriptableContents);
131
132 QDBusAbstractAdaptorPrivate::saveIntrospectionXml(adaptor: it->adaptor, xml: ifaceXml);
133 }
134
135 xml_data += ifaceXml;
136 }
137 }
138
139 // is it a virtual node that handles introspection itself?
140 if (node.flags & QDBusConnectionPrivate::VirtualObject) {
141 xml_data += node.treeNode->introspect(path);
142 }
143
144 xml_data += QLatin1StringView(propertiesInterfaceXml);
145 }
146
147 xml_data += QLatin1StringView(introspectableInterfaceXml);
148 xml_data += QLatin1StringView(peerInterfaceXml);
149
150 if (node.flags & QDBusConnection::ExportChildObjects) {
151 xml_data += generateSubObjectXml(object: node.obj);
152 } else {
153 // generate from the object tree
154 QDBusConnectionPrivate::ObjectTreeNode::DataList::ConstIterator it =
155 node.children.constBegin();
156 QDBusConnectionPrivate::ObjectTreeNode::DataList::ConstIterator end =
157 node.children.constEnd();
158 for ( ; it != end; ++it)
159 if (it->obj || !it->children.isEmpty())
160 xml_data += " <node name=\""_L1 + it->name + "\"/>\n"_L1;
161 }
162
163 xml_data += "</node>\n"_L1;
164 return xml_data;
165}

170{
171 return msg.createErrorReply(type: QDBusError::UnknownInterface,
172 msg: "Interface %1 was not found in object %2"_L1
173 .arg(args: interface_name, args: msg.path()));

176static inline QDBusMessage

178{
179 return msg.createErrorReply(type: QDBusError::UnknownProperty,

184 args: msg.path()));

187QDBusMessage qDBusPropertyGet(const QDBusConnectionPrivate::ObjectTreeNode &node,
188 const QDBusMessage &msg)
189{
190 Q_ASSERT(msg.arguments().size() == 2);
191 Q_ASSERT_X(!node.obj || QThread::currentThread() == node.obj->thread(),
192 "QDBusConnection: internal threading error",
193 "function called for an object that is in another thread!!");
194
195 QString interface_name = msg.arguments().at(i: 0).toString();
196 QByteArray property_name = msg.arguments().at(i: 1).toString().toUtf8();
197
198 QDBusAdaptorConnector *connector;
199 QVariant value;
200 bool interfaceFound = false;
201 if (node.flags & QDBusConnection::ExportAdaptors &&
202 (connector = qDBusFindAdaptorConnector(object: node.obj))) {
203
204 // find the class that implements interface_name or try until we've found the property
205 // in case of an empty interface
206 if (interface_name.isEmpty()) {
207 for (QDBusAdaptorConnector::AdaptorMap::ConstIterator it = connector->adaptors.constBegin(),
208 end = connector->adaptors.constEnd(); it != end; ++it) {
209 const QMetaObject *mo = it->adaptor->metaObject();
210 int pidx = mo->indexOfProperty(name: property_name);
211 if (pidx != -1) {
212 value = mo->property(index: pidx).read(obj: it->adaptor);
213 break;
214 }
215 }
216 } else {
217 QDBusAdaptorConnector::AdaptorMap::ConstIterator it;
218 it = std::lower_bound(connector->adaptors.constBegin(), connector->adaptors.constEnd(),
219 interface_name);
220 if (it != connector->adaptors.constEnd() && interface_name == QLatin1StringView(it->interface)) {
221 interfaceFound = true;
222 value = it->adaptor->property(name: property_name);
223 }
224 }
225 }
226
227 if (!interfaceFound && !value.isValid()
228 && node.flags & (QDBusConnection::ExportAllProperties |
229 QDBusConnection::ExportNonScriptableProperties)) {
230 // try the object itself
231 if (!interface_name.isEmpty())
232 interfaceFound = qDBusInterfaceInObject(obj: node.obj, interface_name);
233
234 if (interfaceFound) {
235 int pidx = node.obj->metaObject()->indexOfProperty(name: property_name);
236 if (pidx != -1) {
237 QMetaProperty mp = node.obj->metaObject()->property(index: pidx);
238 if ((mp.isScriptable() && (node.flags & QDBusConnection::ExportScriptableProperties)) ||
239 (!mp.isScriptable() && (node.flags & QDBusConnection::ExportNonScriptableProperties)))
240 value = mp.read(obj: node.obj);
241 }
242 }
243 }
244
245 if (!value.isValid()) {
246 // the property was not found
247 if (!interfaceFound)
248 return interfaceNotFoundError(msg, interface_name);
249 return propertyNotFoundError(msg, interface_name, property_name);
250 }
251
252 return msg.createReply(argument: QVariant::fromValue(value: QDBusVariant(value)));
253}

255enum PropertyWriteResult {
256 PropertyWriteSuccess = 0,
257 PropertyNotFound,
258 PropertyTypeMismatch,
259 PropertyReadOnly,
260 PropertyWriteFailed
261};
262
263static QDBusMessage propertyWriteReply(const QDBusMessage &msg, const QString &interface_name,
264 const QByteArray &property_name, int status)
265{
266 switch (status) {
267 case PropertyNotFound:
268 return propertyNotFoundError(msg, interface_name, property_name);
269 case PropertyTypeMismatch:
270 return msg.createErrorReply(type: QDBusError::InvalidArgs,
271 msg: "Invalid arguments for writing to property %1%2%3"_L1
272 .arg(args: interface_name,
273 args: interface_name.isEmpty() ? ""_L1 : "."_L1,
274 args: QLatin1StringView(property_name)));
275 case PropertyReadOnly:
276 return msg.createErrorReply(type: QDBusError::PropertyReadOnly,
277 msg: "Property %1%2%3 is read-only"_L1
278 .arg(args: interface_name,
279 args: interface_name.isEmpty() ? ""_L1 : "."_L1,
280 args: QLatin1StringView(property_name)));
281 case PropertyWriteFailed:
282 return msg.createErrorReply(type: QDBusError::InternalError,
283 msg: QString::fromLatin1(ba: "Internal error"));
284
285 case PropertyWriteSuccess:
286 return msg.createReply();
287 }
288 Q_ASSERT_X(false, "", "Should not be reached");
289 return QDBusMessage();
290}

292static int writeProperty(QObject *obj, const QByteArray &property_name, QVariant value,
293 int propFlags = QDBusConnection::ExportAllProperties)
294{
295 const QMetaObject *mo = obj->metaObject();
296 int pidx = mo->indexOfProperty(name: property_name);
297 if (pidx == -1) {
298 // this object has no property by that name
299 return PropertyNotFound;
300 }
301
302 QMetaProperty mp = mo->property(index: pidx);
303
304 // check if this property is writable
305 if (!mp.isWritable())
306 return PropertyReadOnly;
307
308 // check if this property is exported
309 bool isScriptable = mp.isScriptable();
310 if (!(propFlags & QDBusConnection::ExportScriptableProperties) && isScriptable)
311 return PropertyNotFound;
312 if (!(propFlags & QDBusConnection::ExportNonScriptableProperties) && !isScriptable)
313 return PropertyNotFound;
314
315 // we found our property
316 // do we have the right type?
317 QMetaType id = mp.metaType();
318 if (!id.isValid()){
319 // type not registered or invalid / void?
320 qWarning(msg: "QDBusConnection: Unable to handle unregistered datatype '%s' for property '%s::%s'",
321 mp.typeName(), mo->className(), property_name.constData());
322 return PropertyWriteFailed;
323 }
324
325 if (id.id() != QMetaType::QVariant && value.metaType() == QDBusMetaTypeId::argument()) {
326 // we have to demarshall before writing
327 QVariant other{QMetaType(id)};
328 if (!QDBusMetaType::demarshall(qvariant_cast<QDBusArgument>(v: value), id: other.metaType(), data: other.data())) {
329 qWarning(msg: "QDBusConnection: type '%s' (%d) is not registered with QtDBus. "
330 "Use qDBusRegisterMetaType to register it",
331 mp.typeName(), id.id());
332 return PropertyWriteFailed;
333 }
334
335 value = other;
336 }
337
338 if (mp.metaType() == QMetaType::fromType<QDBusVariant>())
339 value = QVariant::fromValue(value: QDBusVariant(value));
340
341 // the property type here should match
342 return mp.write(obj, value) ? PropertyWriteSuccess : PropertyWriteFailed;
343}

345QDBusMessage qDBusPropertySet(const QDBusConnectionPrivate::ObjectTreeNode &node,
346 const QDBusMessage &msg)
347{
348 Q_ASSERT(msg.arguments().size() == 3);
349 Q_ASSERT_X(!node.obj || QThread::currentThread() == node.obj->thread(),
350 "QDBusConnection: internal threading error",
351 "function called for an object that is in another thread!!");
352
353 QString interface_name = msg.arguments().at(i: 0).toString();
354 QByteArray property_name = msg.arguments().at(i: 1).toString().toUtf8();
355 QVariant value = qvariant_cast<QDBusVariant>(v: msg.arguments().at(i: 2)).variant();
356
357 QDBusAdaptorConnector *connector;
358 if (node.flags & QDBusConnection::ExportAdaptors &&
359 (connector = qDBusFindAdaptorConnector(object: node.obj))) {
360
361 // find the class that implements interface_name or try until we've found the property
362 // in case of an empty interface
363 if (interface_name.isEmpty()) {
364 for (QDBusAdaptorConnector::AdaptorMap::ConstIterator it = connector->adaptors.constBegin(),
365 end = connector->adaptors.constEnd(); it != end; ++it) {
366 int status = writeProperty(obj: it->adaptor, property_name, value);
367 if (status == PropertyNotFound)
368 continue;
369 return propertyWriteReply(msg, interface_name, property_name, status);
370 }
371 } else {
372 QDBusAdaptorConnector::AdaptorMap::ConstIterator it;
373 it = std::lower_bound(connector->adaptors.constBegin(), connector->adaptors.constEnd(),
374 interface_name);
375 if (it != connector->adaptors.cend() && interface_name == QLatin1StringView(it->interface)) {
376 return propertyWriteReply(msg, interface_name, property_name,
377 status: writeProperty(obj: it->adaptor, property_name, value));
378 }
379 }
380 }
381
382 if (node.flags & (QDBusConnection::ExportScriptableProperties |
383 QDBusConnection::ExportNonScriptableProperties)) {
384 // try the object itself
385 bool interfaceFound = true;
386 if (!interface_name.isEmpty())
387 interfaceFound = qDBusInterfaceInObject(obj: node.obj, interface_name);
388
389 if (interfaceFound) {
390 return propertyWriteReply(msg, interface_name, property_name,
391 status: writeProperty(obj: node.obj, property_name, value, propFlags: node.flags));
392 }
393 }
394
395 // the property was not found
396 if (!interface_name.isEmpty())
397 return interfaceNotFoundError(msg, interface_name);
398 return propertyWriteReply(msg, interface_name, property_name, status: PropertyNotFound);
399}

408 return lhs;

411static QVariantMap readAllProperties(QObject *object, int flags)
412{
413 QVariantMap result;
414 const QMetaObject *mo = object->metaObject();
415
416 // QObject has properties, so don't start from 0
417 for (int i = QObject::staticMetaObject.propertyCount(); i < mo->propertyCount(); ++i) {
418 QMetaProperty mp = mo->property(index: i);
419
420 // is it readable?
421 if (!mp.isReadable())
422 continue;
423
424 // is it a registered property?
425 QMetaType type = mp.metaType();
426 if (!type.isValid())
427 continue;
428 const char *signature = QDBusMetaType::typeToSignature(type);
429 if (!signature)
430 continue;
431
432 // is this property visible from the outside?
433 if ((mp.isScriptable() && flags & QDBusConnection::ExportScriptableProperties) ||
434 (!mp.isScriptable() && flags & QDBusConnection::ExportNonScriptableProperties)) {
435 // yes, it's visible
436 QVariant value = mp.read(obj: object);
437 if (value.isValid())
438 result.insert(key: QString::fromLatin1(ba: mp.name()), value);
439 }
440 }
441
442 return result;
443}

445QDBusMessage qDBusPropertyGetAll(const QDBusConnectionPrivate::ObjectTreeNode &node,
446 const QDBusMessage &msg)
447{
448 Q_ASSERT(msg.arguments().size() == 1);
449 Q_ASSERT_X(!node.obj || QThread::currentThread() == node.obj->thread(),
450 "QDBusConnection: internal threading error",
451 "function called for an object that is in another thread!!");
452
453 QString interface_name = msg.arguments().at(i: 0).toString();
454
455 bool interfaceFound = false;
456 QVariantMap result;
457
458 QDBusAdaptorConnector *connector;
459 if (node.flags & QDBusConnection::ExportAdaptors &&
460 (connector = qDBusFindAdaptorConnector(object: node.obj))) {
461
462 if (interface_name.isEmpty()) {
463 // iterate over all interfaces
464 for (QDBusAdaptorConnector::AdaptorMap::ConstIterator it = connector->adaptors.constBegin(),
465 end = connector->adaptors.constEnd(); it != end; ++it) {
466 result += readAllProperties(object: it->adaptor, flags: QDBusConnection::ExportAllProperties);
467 }
468 } else {
469 // find the class that implements interface_name
470 QDBusAdaptorConnector::AdaptorMap::ConstIterator it;
471 it = std::lower_bound(connector->adaptors.constBegin(), connector->adaptors.constEnd(),
472 interface_name);
473 if (it != connector->adaptors.constEnd() && interface_name == QLatin1StringView(it->interface)) {
474 interfaceFound = true;
475 result = readAllProperties(object: it->adaptor, flags: QDBusConnection::ExportAllProperties);
476 }
477 }
478 }
479
480 if (node.flags & QDBusConnection::ExportAllProperties &&
481 (!interfaceFound || interface_name.isEmpty())) {
482 // try the object itself
483 result += readAllProperties(object: node.obj, flags: node.flags);
484 interfaceFound = true;
485 }
486
487 if (!interfaceFound && !interface_name.isEmpty()) {
488 // the interface was not found
489 return interfaceNotFoundError(msg, interface_name);
490 }
491
492 return msg.createReply(argument: QVariant::fromValue(value: result));
493}
