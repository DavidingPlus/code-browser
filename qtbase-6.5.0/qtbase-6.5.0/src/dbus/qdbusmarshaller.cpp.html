15static void qIterAppend(DBusMessageIter *it, QByteArray *ba, int type, const void *arg)
16{
17 if (ba)
18 *ba += char(type);
19 else
20 q_dbus_message_iter_append_basic(iter: it, type, value: arg);
21}

28void QDBusMarshaller::unregisteredTypeError(QMetaType id)
29{
30 const char *name = id.name();
31 qWarning(msg: "QDBusMarshaller: type '%s' (%d) is not registered with D-BUS. "
32 "Use qDBusRegisterMetaType to register it",
33 name ? name : "", id.id());
34 error(message: "Unregistered type %1 passed in arguments"_L1
35 .arg(args: QLatin1StringView(id.name())));
36}

143inline void QDBusMarshaller::append(const QByteArray &arg)
144{
145 if (ba) {
146 if (!skipSignature)
147 *ba += DBUS_TYPE_ARRAY_AS_STRING DBUS_TYPE_BYTE_AS_STRING;
148 return;
149 }
150
151 const char* cdata = arg.constData();
152 DBusMessageIter subiterator;
153 q_dbus_message_iter_open_container(iter: &iterator, DBUS_TYPE_ARRAY, DBUS_TYPE_BYTE_AS_STRING,
154 sub: &subiterator);
155 q_dbus_message_iter_append_fixed_array(iter: &subiterator, DBUS_TYPE_BYTE, value: &cdata, n_elements: arg.size());
156 q_dbus_message_iter_close_container(iter: &iterator, sub: &subiterator);
157}

159inline bool QDBusMarshaller::append(const QDBusVariant &arg)
160{
161 if (ba) {
162 if (!skipSignature)
163 *ba += DBUS_TYPE_VARIANT_AS_STRING;
164 return true;
165 }
166
167 const QVariant &value = arg.variant();
168 QMetaType id = value.metaType();
169 if (!id.isValid()) {
170 qWarning(msg: "QDBusMarshaller: cannot add a null QDBusVariant");
171 error(message: "Invalid QVariant passed in arguments"_L1);
172 return false;
173 }
174
175 QByteArray tmpSignature;
176 const char *signature = nullptr;
177 if (id == QDBusMetaTypeId::argument()) {
178 // take the signature from the QDBusArgument object we're marshalling
179 tmpSignature =
180 qvariant_cast<QDBusArgument>(v: value).currentSignature().toLatin1();
181 signature = tmpSignature.constData();
182 } else {
183 // take the signatuer from the metatype we're marshalling
184 signature = QDBusMetaType::typeToSignature(type: id);
185 }
186 if (!signature) {
187 unregisteredTypeError(id);
188 return false;
189 }
190
191 QDBusMarshaller sub(capabilities);
192 open(sub, DBUS_TYPE_VARIANT, signature);
193 bool isOk = sub.appendVariantInternal(arg: value);
194 // don't call sub.close(): it auto-closes
195
196 return isOk;
197}

199inline void QDBusMarshaller::append(const QStringList &arg)
200{
201 if (ba) {
202 if (!skipSignature)
203 *ba += DBUS_TYPE_ARRAY_AS_STRING DBUS_TYPE_STRING_AS_STRING;
204 return;
205 }
206
207 QDBusMarshaller sub(capabilities);
208 open(sub, DBUS_TYPE_ARRAY, DBUS_TYPE_STRING_AS_STRING);
209 QStringList::ConstIterator it = arg.constBegin();
210 QStringList::ConstIterator end = arg.constEnd();
211 for ( ; it != end; ++it)
212 sub.append(arg: *it);
213 // don't call sub.close(): it auto-closes
214}

232inline QDBusMarshaller *QDBusMarshaller::beginMap(QMetaType kid, QMetaType vid)
233{
234 const char *ksignature = QDBusMetaType::typeToSignature(type: kid);
235 if (!ksignature) {
236 unregisteredTypeError(id: kid);
237 return this;
238 }
239 if (ksignature[1] != 0 || !QDBusUtil::isValidBasicType(c: *ksignature)) {
240QT_WARNING_PUSH
241QT_WARNING_DISABLE_GCC("-Wformat-overflow")
242 qWarning(msg: "QDBusMarshaller: type '%s' (%d) cannot be used as the key type in a D-BUS map.",
243 kid.name(), kid.id());
244QT_WARNING_POP
245 error(message: "Type %1 passed in arguments cannot be used as a key in a map"_L1
246 .arg(args: QLatin1StringView(kid.name())));
247 return this;
248 }
249
250 const char *vsignature = QDBusMetaType::typeToSignature(type: vid);
251 if (!vsignature) {
252 unregisteredTypeError(id: vid);
253 return this;
254 }
255
256 QByteArray signature;
257 signature = DBUS_DICT_ENTRY_BEGIN_CHAR_AS_STRING;
258 signature += ksignature;
259 signature += vsignature;
260 signature += DBUS_DICT_ENTRY_END_CHAR_AS_STRING;
261 return beginCommon(DBUS_TYPE_ARRAY, signature);
262}

269void QDBusMarshaller::open(QDBusMarshaller &sub, int code, const char *signature)
270{
271 sub.parent = this;
272 sub.ba = ba;
273 sub.ok = true;
274 sub.capabilities = capabilities;
275 sub.skipSignature = skipSignature;
276
277 if (ba) {
278 if (!skipSignature) {
279 switch (code) {
280 case DBUS_TYPE_ARRAY:
281 *ba += char(code);
282 *ba += signature;
283 Q_FALLTHROUGH();
284
285 case DBUS_TYPE_DICT_ENTRY:
286 sub.closeCode = 0;
287 sub.skipSignature = true;
288 break;
289
290 case DBUS_TYPE_STRUCT:
291 *ba += DBUS_STRUCT_BEGIN_CHAR;
292 sub.closeCode = DBUS_STRUCT_END_CHAR;
293 break;
294 }
295 }
296 } else {
297 q_dbus_message_iter_open_container(iter: &iterator, type: code, contained_signature: signature, sub: &sub.iterator);
298 }
299}

346bool QDBusMarshaller::appendVariantInternal(const QVariant &arg)
347{
348 QMetaType id = arg.metaType();
349 if (!id.isValid()) {
350 qWarning(msg: "QDBusMarshaller: cannot add an invalid QVariant");
351 error(msg: "Invalid QVariant passed in arguments"_L1);
352 return false;
353 }
354
355 // intercept QDBusArgument parameters here
356 if (id == QDBusMetaTypeId::argument()) {
357 QDBusArgument dbusargument = qvariant_cast<QDBusArgument>(v: arg);
358 QDBusArgumentPrivate *d = QDBusArgumentPrivate::d(q&: dbusargument);
359 if (!d->message)
360 return false; // can't append this one...
361
362 QDBusDemarshaller demarshaller(capabilities);
363 demarshaller.message = q_dbus_message_ref(message: d->message);
364
365 if (d->direction == Demarshalling) {
366 // it's demarshalling; just copy
367 demarshaller.iterator = static_cast<QDBusDemarshaller *>(d)->iterator;
368 } else {
369 // it's marshalling; start over
370 if (!q_dbus_message_iter_init(message: demarshaller.message, iter: &demarshaller.iterator))
371 return false; // error!
372 }
373
374 return appendCrossMarshalling(arg: &demarshaller);
375 }
376
377 const char *signature = QDBusMetaType::typeToSignature(type: id);
378 if (!signature) {
379 unregisteredTypeError(id);
380 return false;
381 }
382
383 switch (*signature) {
384#ifdef __OPTIMIZE__
385 case DBUS_TYPE_BYTE:
386 case DBUS_TYPE_INT16:
387 case DBUS_TYPE_UINT16:
388 case DBUS_TYPE_INT32:
389 case DBUS_TYPE_UINT32:
390 case DBUS_TYPE_INT64:
391 case DBUS_TYPE_UINT64:
392 case DBUS_TYPE_DOUBLE:
393 qIterAppend(it: &iterator, ba, type: *signature, arg: arg.constData());
394 return true;
395 case DBUS_TYPE_BOOLEAN:
396 append( arg: arg.toBool() );
397 return true;
398#else
399 case DBUS_TYPE_BYTE:
400 append( qvariant_cast<uchar>(arg) );
401 return true;
402 case DBUS_TYPE_BOOLEAN:
403 append( arg.toBool() );
404 return true;
405 case DBUS_TYPE_INT16:
406 append( qvariant_cast<short>(arg) );
407 return true;
408 case DBUS_TYPE_UINT16:
409 append( qvariant_cast<ushort>(arg) );
410 return true;
411 case DBUS_TYPE_INT32:
412 append( static_cast<dbus_int32_t>(arg.toInt()) );
413 return true;
414 case DBUS_TYPE_UINT32:
415 append( static_cast<dbus_uint32_t>(arg.toUInt()) );
416 return true;
417 case DBUS_TYPE_INT64:
418 append( arg.toLongLong() );
419 return true;
420 case DBUS_TYPE_UINT64:
421 append( arg.toULongLong() );
422 return true;
423 case DBUS_TYPE_DOUBLE:
424 append( arg.toDouble() );
425 return true;
426#endif
427
428 case DBUS_TYPE_STRING:
429 append( arg: arg.toString() );
430 return true;
431 case DBUS_TYPE_OBJECT_PATH:
432 append( arg: qvariant_cast<QDBusObjectPath>(v: arg) );
433 return true;
434 case DBUS_TYPE_SIGNATURE:
435 append( arg: qvariant_cast<QDBusSignature>(v: arg) );
436 return true;
437
438 // compound types:
439 case DBUS_TYPE_VARIANT:
440 // nested QVariant
441 return append( arg: qvariant_cast<QDBusVariant>(v: arg) );
442
443 case DBUS_TYPE_ARRAY:
444 // could be many things
445 // find out what kind of array it is
446 switch (arg.metaType().id()) {
447 case QMetaType::QStringList:
448 append( arg: arg.toStringList() );
449 return true;
450
451 case QMetaType::QByteArray:
452 append( arg: arg.toByteArray() );
453 return true;
454
455 default:
456 ;
457 }
458 Q_FALLTHROUGH();
459
460 case DBUS_TYPE_STRUCT:
461 case DBUS_STRUCT_BEGIN_CHAR:
462 return appendRegisteredType( arg );
463
464 case DBUS_TYPE_DICT_ENTRY:
465 case DBUS_DICT_ENTRY_BEGIN_CHAR:
466 qFatal(msg: "QDBusMarshaller::appendVariantInternal got a DICT_ENTRY!");
467 return false;
468
469 case DBUS_TYPE_UNIX_FD:
470 if (capabilities & QDBusConnection::UnixFileDescriptorPassing || ba) {
471 append(arg: qvariant_cast<QDBusUnixFileDescriptor>(v: arg));
472 return true;
473 }
474 Q_FALLTHROUGH();
475
476 default:
477 qWarning(msg: "QDBusMarshaller::appendVariantInternal: Found unknown D-BUS type '%s'",
478 signature);
479 return false;
480 }
481
482 return true;
483}

492bool QDBusMarshaller::appendCrossMarshalling(QDBusDemarshaller *demarshaller)
493{
494 int code = q_dbus_message_iter_get_arg_type(iter: &demarshaller->iterator);
495 if (QDBusUtil::isValidBasicType(c: code)) {
496 // easy: just append
497 // do exactly like the D-BUS docs suggest
498 // (see apidocs for q_dbus_message_iter_get_basic)
499
500 qlonglong value;
501 q_dbus_message_iter_get_basic(iter: &demarshaller->iterator, value: &value);
502 q_dbus_message_iter_next(iter: &demarshaller->iterator);
503 q_dbus_message_iter_append_basic(iter: &iterator, type: code, value: &value);
504 return true;
505 }
506
507 if (code == DBUS_TYPE_ARRAY) {
508 int element = q_dbus_message_iter_get_element_type(iter: &demarshaller->iterator);
509 if (QDBusUtil::isValidFixedType(c: element) && element != DBUS_TYPE_UNIX_FD) {
510 // another optimization: fixed size arrays
511 // code is exactly like QDBusDemarshaller::toByteArray
512 DBusMessageIter sub;
513 q_dbus_message_iter_recurse(iter: &demarshaller->iterator, sub: &sub);
514 q_dbus_message_iter_next(iter: &demarshaller->iterator);
515 int len;
516 void* data;
517 q_dbus_message_iter_get_fixed_array(iter: &sub,value: &data,n_elements: &len);
518
519 char signature[2] = { char(element), 0 };
520 q_dbus_message_iter_open_container(iter: &iterator, DBUS_TYPE_ARRAY, contained_signature: signature, sub: &sub);
521 q_dbus_message_iter_append_fixed_array(iter: &sub, element_type: element, value: &data, n_elements: len);
522 q_dbus_message_iter_close_container(iter: &iterator, sub: &sub);
523
524 return true;
525 }
526 }
527
528 // We have to recurse
529 QDBusDemarshaller *drecursed = demarshaller->beginCommon();
530
531 QDBusMarshaller mrecursed(capabilities); // create on the stack makes it autoclose
532 QByteArray subSignature;
533 const char *sig = nullptr;
534 if (code == DBUS_TYPE_VARIANT || code == DBUS_TYPE_ARRAY) {
535 subSignature = drecursed->currentSignature().toLatin1();
536 if (!subSignature.isEmpty())
537 sig = subSignature.constData();
538 }
539 open(sub&: mrecursed, code, signature: sig);
540
541 while (!drecursed->atEnd()) {
542 if (!mrecursed.appendCrossMarshalling(demarshaller: drecursed)) {
543 delete drecursed;
544 return false;
545 }
546 }
547
548 delete drecursed;
549 return true;
550}
