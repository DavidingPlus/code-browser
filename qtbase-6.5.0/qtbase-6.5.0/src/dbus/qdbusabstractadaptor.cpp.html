27int QDBusAdaptorConnector::relaySlotMethodIndex()
28{
29 if (cachedRelaySlotMethodIndex == 0) {
30 cachedRelaySlotMethodIndex = staticMetaObject.indexOfMethod(method: "relaySlot()");
31 Q_ASSERT(cachedRelaySlotMethodIndex != 0); // 0 should be deleteLater() or destroyed()
32 }
33 return cachedRelaySlotMethodIndex;
34}

36QDBusAdaptorConnector *qDBusFindAdaptorConnector(QObject *obj)
37{
38 if (!obj)
39 return nullptr;
40 const QObjectList &children = obj->children();
41 QObjectList::ConstIterator it = children.constBegin();
42 QObjectList::ConstIterator end = children.constEnd();
43 for ( ; it != end; ++it) {
44 QDBusAdaptorConnector *connector = qobject_cast<QDBusAdaptorConnector *>(object: *it);
45 if (connector) {
46 connector->polish();
47 return connector;
48 }
49 }
50 return nullptr;
51}

77/*!
78 \class QDBusAbstractAdaptor
79 \inmodule QtDBus
80 \since 4.2
81
82 \brief The QDBusAbstractAdaptor class is the base class of D-Bus adaptor classes.
83
84 The QDBusAbstractAdaptor class is the starting point for all objects intending to provide
85 interfaces to the external world using D-Bus. This is accomplished by attaching a one or more
86 classes derived from QDBusAbstractAdaptor to a normal QObject and then registering that QObject
87 with QDBusConnection::registerObject. QDBusAbstractAdaptor objects are intended to be
88 light-weight wrappers, mostly just relaying calls into the real object (its parent) and the
89 signals from it.
90
91 Each QDBusAbstractAdaptor-derived class should define the D-Bus interface it is implementing
92 using the Q_CLASSINFO macro in the class definition. Note that only one interface can be
93 exposed in this way.
94
95 QDBusAbstractAdaptor uses the standard QObject mechanism of signals, slots and properties to
96 determine what signals, methods and properties to export to the bus. Any signal emitted by
97 QDBusAbstractAdaptor-derived classes will be automatically be relayed through any D-Bus
98 connections the object is registered on.
99
100 Classes derived from QDBusAbstractAdaptor must be created on the heap using the \a new operator
101 and must not be deleted by the user (they will be deleted automatically when the object they are
102 connected to is also deleted).
103
104 \sa {usingadaptors.html}{Using adaptors}, QDBusConnection
105*/
106

129/*!
130 Toggles automatic signal relaying from the real object (see object()).
131
132 Automatic signal relaying consists of signal-to-signal connection of the signals on the parent
133 that have the exact same method signature in both classes.
134
135 If \a enable is set to true, connect the signals; if set to false, disconnect all signals.
136*/
137void QDBusAbstractAdaptor::setAutoRelaySignals(bool enable)
138{
139 const QMetaObject *us = metaObject();
140 const QMetaObject *them = parent()->metaObject();
141 bool connected = false;
142 for (int idx = staticMetaObject.methodCount(); idx < us->methodCount(); ++idx) {
143 QMetaMethod mm = us->method(index: idx);
144
145 if (mm.methodType() != QMetaMethod::Signal)
146 continue;
147
148 // try to connect/disconnect to a signal on the parent that has the same method signature
149 QByteArray sig = QMetaObject::normalizedSignature(method: mm.methodSignature().constData());
150 if (them->indexOfSignal(signal: sig) == -1)
151 continue;
152 sig.prepend(QSIGNAL_CODE + '0');
153 parent()->disconnect(signal: sig, receiver: this, member: sig);
154 if (enable)
155 connected = connect(asender: parent(), asignal: sig, amember: sig) || connected;
156 }
157 d_func()->autoRelaySignals = connected;
158}

163
164 \sa setAutoRelaySignals()

180void QDBusAdaptorConnector::addAdaptor(QDBusAbstractAdaptor *adaptor)
181{
182 // find the interface name
183 const QMetaObject *mo = adaptor->metaObject();
184 int ciid = mo->indexOfClassInfo(QCLASSINFO_DBUS_INTERFACE);
185 if (ciid != -1) {
186 QMetaClassInfo mci = mo->classInfo(index: ciid);
187 if (*mci.value()) {
188 // find out if this interface exists first
189 const char *interface = mci.value();
190 AdaptorMap::Iterator it = std::lower_bound(adaptors.begin(), adaptors.end(),
191 QByteArray(interface));
192 if (it != adaptors.end() && qstrcmp(str1: interface, str2: it->interface) == 0) {
193 // exists. Replace it (though it's probably the same)
194 if (it->adaptor != adaptor) {
195 // reconnect the signals
196 disconnectAllSignals(object: it->adaptor);
197 connectAllSignals(object: adaptor);
198 }
199 it->adaptor = adaptor;
200 } else {
201 // create a new one
202 AdaptorData entry;
203 entry.interface = interface;
204 entry.adaptor = adaptor;
205 adaptors << entry;
206
207 // connect the adaptor's signals to our relaySlot slot
208 connectAllSignals(object: adaptor);
209 }
210 }
211 }

224void QDBusAdaptorConnector::polish()
225{
226 if (!waitingForPolish)
227 return; // avoid working multiple times if multiple adaptors were added
228
229 waitingForPolish = false;
230 const QObjectList &objs = parent()->children();
231 QObjectList::ConstIterator it = objs.constBegin();
232 QObjectList::ConstIterator end = objs.constEnd();
233 for ( ; it != end; ++it) {
234 QDBusAbstractAdaptor *adaptor = qobject_cast<QDBusAbstractAdaptor *>(object: *it);
235 if (adaptor)
236 addAdaptor(adaptor);
237 }
238
239 // sort the adaptor list
240 std::sort(first: adaptors.begin(), last: adaptors.end());
241}

243void QDBusAdaptorConnector::relaySlot(QMethodRawArguments argv)
244{
245 QObject *sndr = sender();
246 if (Q_LIKELY(sndr)) {
247 relay(sender: sndr, id: senderSignalIndex(), argv.arguments);
248 } else {
249 qWarning(msg: "QtDBus: cannot relay signals from parent %s(%p \"%s\") unless they are emitted in the object's thread %s(%p \"%s\"). "
250 "Current thread is %s(%p \"%s\").",
251 parent()->metaObject()->className(), parent(), qPrintable(parent()->objectName()),
252 parent()->thread()->metaObject()->className(), parent()->thread(), qPrintable(parent()->thread()->objectName()),
253 QThread::currentThread()->metaObject()->className(), QThread::currentThread(), qPrintable(QThread::currentThread()->objectName()));
254 }
255}

257void QDBusAdaptorConnector::relay(QObject *senderObj, int lastSignalIdx, void **argv)
258{
259 if (lastSignalIdx < QObject::staticMetaObject.methodCount())
260 // QObject signal (destroyed(QObject *)) -- ignore
261 return;
262
263 const QMetaObject *senderMetaObject = senderObj->metaObject();
264 QMetaMethod mm = senderMetaObject->method(index: lastSignalIdx);
265
266 QObject *realObject = senderObj;
267 if (qobject_cast<QDBusAbstractAdaptor *>(object: senderObj))
268 // it's an adaptor, so the real object is in fact its parent
269 realObject = realObject->parent();
270
271 // break down the parameter list
272 QList<QMetaType> types;
273 QString errorMsg;
274 int inputCount = qDBusParametersForMethod(mm, metaTypes&: types, errorMsg);
275 if (inputCount == -1) {
276 // invalid signal signature
277 qWarning(msg: "QDBusAbstractAdaptor: Cannot relay signal %s::%s: %s",
278 senderMetaObject->className(), mm.methodSignature().constData(),
279 qPrintable(errorMsg));
280 return;
281 }
282 if (inputCount + 1 != types.size() ||
283 types.at(i: inputCount) == QDBusMetaTypeId::message()) {
284 // invalid signal signature
285 qWarning(msg: "QDBusAbstractAdaptor: Cannot relay signal %s::%s",
286 senderMetaObject->className(), mm.methodSignature().constData());
287 return;
288 }
289
290 QVariantList args;
291 const int numTypes = types.size();
292 args.reserve(size: numTypes - 1);
293 for (int i = 1; i < numTypes; ++i)
294 args << QVariant(QMetaType(types.at(i)), argv[i]);
295
296 // now emit the signal with all the information
297 emit relaySignal(obj: realObject, metaObject: senderMetaObject, sid: lastSignalIdx, args);
298}
