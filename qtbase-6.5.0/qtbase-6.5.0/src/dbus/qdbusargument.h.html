134template<typename T> inline T qdbus_cast(const QVariant &v)
135{
136 if (v.metaType() == QMetaType::fromType<QDBusArgument>())
137 return qdbus_cast<T>(qvariant_cast<QDBusArgument>(v));
138 else
139 return qvariant_cast<T>(v);
140}

193template<template <typename> class Container, typename T,
194 typename = typename Container<T>::iterator>
195inline QDBusArgument &operator<<(QDBusArgument &arg, const Container<T> &list)
196{
197 arg.beginArray(QMetaType::fromType<T>());
198 typename Container<T>::const_iterator it = list.begin();
199 typename Container<T>::const_iterator end = list.end();
200 for ( ; it != end; ++it)
201 arg << *it;
202 arg.endArray();
203 return arg;
204}
205
206template<template <typename> class Container, typename T,
207 typename = typename Container<T>::iterator>
208inline const QDBusArgument &operator>>(const QDBusArgument &arg, Container<T> &list)
209{
210 arg.beginArray();
211 list.clear();
212 while (!arg.atEnd()) {
213 T item;
214 arg >> item;
215 list.push_back(item);
216 }
217
218 arg.endArray();
219 return arg;
220}

233// Specializations for associative containers
234template <template <typename, typename> class Container, typename Key, typename T,
235 QtPrivate::IfAssociativeIteratorHasKeyAndValue<typename Container<Key, T>::iterator> = true>
236inline QDBusArgument &operator<<(QDBusArgument &arg, const Container<Key, T> &map)
237{
238 arg.beginMap(QMetaType::fromType<Key>(), QMetaType::fromType<T>());
239 auto it = map.begin();
240 auto end = map.end();
241 for ( ; it != end; ++it) {
242 arg.beginMapEntry();
243 arg << it.key() << it.value();
244 arg.endMapEntry();
245 }
246 arg.endMap();
247 return arg;
248}

250template <template <typename, typename> class Container, typename Key, typename T,
251 QtPrivate::IfAssociativeIteratorHasFirstAndSecond<typename Container<Key, T>::iterator> = true>
252inline QDBusArgument &operator<<(QDBusArgument &arg, const Container<Key, T> &map)
253{
254 arg.beginMap(QMetaType::fromType<Key>(), QMetaType::fromType<T>());
255 auto it = map.begin();
256 auto end = map.end();
257 for ( ; it != end; ++it) {
258 arg.beginMapEntry();
259 arg << it->first << it->second;
260 arg.endMapEntry();
261 }
262 arg.endMap();
263 return arg;
264}
265
266template <template <typename, typename> class Container, typename Key, typename T,
267 QtPrivate::IfAssociativeIteratorHasKeyAndValue<typename Container<Key, T>::iterator> = true>
268inline const QDBusArgument &operator>>(const QDBusArgument &arg, Container<Key, T> &map)
269{
270 arg.beginMap();
271 map.clear();
272 while (!arg.atEnd()) {
273 Key key;
274 T value;
275 arg.beginMapEntry();
276 arg >> key >> value;
277 map.insert(key, value);
278 arg.endMapEntry();
279 }
280 arg.endMap();
281 return arg;
282}
