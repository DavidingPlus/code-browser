96static int registerComplexDBusType(const QByteArray &typeName)
97{
98 struct QDBusRawTypeHandler : QtPrivate::QMetaTypeInterface
99 {
100 const QByteArray name;
101 QDBusRawTypeHandler(const QByteArray &name)
102 : QtPrivate::QMetaTypeInterface {
103 .revision: 0, .alignment: sizeof(void *), .size: sizeof(void *), .flags: QMetaType::RelocatableType, .typeId: 0, .metaObjectFn: nullptr,
104 .name: name.constData(),
105 .defaultCtr: nullptr, .copyCtr: nullptr, .moveCtr: nullptr, .dtor: nullptr,
106 .equals: nullptr, .lessThan: nullptr, .debugStream: nullptr,
107 .dataStreamOut: nullptr, .dataStreamIn: nullptr, .legacyRegisterOp: nullptr
108 },
109 name(name)
110 {}
111 };
112
113 Q_CONSTINIT static QBasicMutex mutex;
114 Q_CONSTINIT static struct Hash : QHash<QByteArray, QMetaType>
115 {
116 ~Hash()
117 {
118 for (QMetaType entry : *this)
119 QMetaType::unregisterMetaType(type: std::move(entry));
120 }
121 } hash;
122 QMutexLocker lock(&mutex);
123 QMetaType &metatype = hash[typeName];
124 if (!metatype.isValid())
125 metatype = QMetaType(new QDBusRawTypeHandler(typeName));
126 return metatype.id();
127}

131QDBusMetaObjectGenerator::Type
132QDBusMetaObjectGenerator::findType(const QByteArray &signature,
133 const QDBusIntrospection::Annotations &annotations,
134 const char *direction, int id)
135{
136 Type result;
137 result.id = QMetaType::UnknownType;
138
139 int type = QDBusMetaType::signatureToMetaType(signature).id();
140 if (type == QMetaType::UnknownType && !qt_dbus_metaobject_skip_annotations) {
141 // it's not a type normally handled by our meta type system
142 // it must contain an annotation
143 QString annotationName = QString::fromLatin1(ba: "org.qtproject.QtDBus.QtTypeName");
144 if (id >= 0)
145 annotationName += QString::fromLatin1(ba: ".%1%2")
146 .arg(a: QLatin1StringView(direction))
147 .arg(a: id);
148
149 // extract from annotations:
150 QByteArray typeName = annotations.value(key: annotationName).toLatin1();
151
152 // verify that it's a valid one
153 if (typeName.isEmpty()) {
154 // try the old annotation from Qt 4
155 annotationName = QString::fromLatin1(ba: "com.trolltech.QtDBus.QtTypeName");
156 if (id >= 0)
157 annotationName += QString::fromLatin1(ba: ".%1%2")
158 .arg(a: QLatin1StringView(direction))
159 .arg(a: id);
160 typeName = annotations.value(key: annotationName).toLatin1();
161 }
162
163 if (!typeName.isEmpty()) {
164 // type name found
165 type = QMetaType::fromName(name: typeName).id();
166 }
167
168 if (type == QMetaType::UnknownType || signature != QDBusMetaType::typeToSignature(type: QMetaType(type))) {
169 // type is still unknown or doesn't match back to the signature that it
170 // was expected to, so synthesize a fake type
171 typeName = "QDBusRawType<0x" + signature.toHex() + ">*";
172 type = registerComplexDBusType(typeName);
173 }
174
175 result.name = typeName;
176 } else if (type == QMetaType::UnknownType) {
177 // this case is used only by the qdbus command-line tool
178 // invalid, let's create an impossible type that contains the signature
179
180 if (signature == "av") {
181 result.name = "QVariantList";
182 type = QMetaType::QVariantList;
183 } else if (signature == "a{sv}") {
184 result.name = "QVariantMap";
185 type = QMetaType::QVariantMap;
186 } else if (signature == "a{ss}") {
187 result.name = "QMap<QString,QString>";
188 type = qMetaTypeId<QMap<QString, QString> >();
189 } else if (signature == "aay") {
190 result.name = "QByteArrayList";
191 type = qMetaTypeId<QByteArrayList>();
192 } else {
193 result.name = "{D-Bus type \"" + signature + "\"}";
194 type = registerComplexDBusType(typeName: result.name);
195 }
196 } else {
197 result.name = QMetaType(type).name();
198 }
199
200 result.id = type;
201 return result; // success
202}

204void QDBusMetaObjectGenerator::parseMethods()
205{
206 //
207 // TODO:
208 // Add cloned methods when the remote object has return types
209 //
210
211 QDBusIntrospection::Methods::ConstIterator method_it = data->methods.constBegin();
212 QDBusIntrospection::Methods::ConstIterator method_end = data->methods.constEnd();
213 for ( ; method_it != method_end; ++method_it) {
214 const QDBusIntrospection::Method &m = *method_it;
215 Method mm;
216
217 mm.name = m.name.toLatin1();
218 QByteArray prototype = mm.name;
219 prototype += '(';
220
221 bool ok = true;
222
223 // build the input argument list
224 for (qsizetype i = 0; i < m.inputArgs.size(); ++i) {
225 const QDBusIntrospection::Argument &arg = m.inputArgs.at(i);
226
227 Type type = findType(signature: arg.type.toLatin1(), annotations: m.annotations, direction: "In", id: i);
228 if (type.id == QMetaType::UnknownType) {
229 ok = false;
230 break;
231 }
232
233 mm.inputTypes.append(t: type.id);
234
235 mm.parameterNames.append(t: arg.name.toLatin1());
236
237 prototype.append(a: type.name);
238 prototype.append(c: ',');
239 }
240 if (!ok) continue;
241
242 // build the output argument list:
243 for (qsizetype i = 0; i < m.outputArgs.size(); ++i) {
244 const QDBusIntrospection::Argument &arg = m.outputArgs.at(i);
245
246 Type type = findType(signature: arg.type.toLatin1(), annotations: m.annotations, direction: "Out", id: i);
247 if (type.id == QMetaType::UnknownType) {
248 ok = false;
249 break;
250 }
251
252 mm.outputTypes.append(t: type.id);
253
254 if (i == 0 && type.id == -1) {
255 mm.rawReturnType = type.name;
256 }
257 if (i != 0) {
258 // non-const ref parameter
259 mm.parameterNames.append(t: arg.name.toLatin1());
260
261 prototype.append(a: type.name);
262 prototype.append(s: "&,");
263 }
264 }
265 if (!ok) continue;
266
267 // convert the last commas:
268 if (!mm.parameterNames.isEmpty())
269 prototype[prototype.size() - 1] = ')';
270 else
271 prototype.append(c: ')');
272
273 // check the async tag
274 if (m.annotations.value(ANNOTATION_NO_WAIT ""_L1) == "true"_L1)
275 mm.tag = "Q_NOREPLY";
276
277 // meta method flags
278 mm.flags = AccessPublic | MethodSlot | MethodScriptable;
279
280 // add
281 methods.insert(key: QMetaObject::normalizedSignature(method: prototype), value: mm);
282 }
283}

285void QDBusMetaObjectGenerator::parseSignals()
286{
287 QDBusIntrospection::Signals::ConstIterator signal_it = data->signals_.constBegin();
288 QDBusIntrospection::Signals::ConstIterator signal_end = data->signals_.constEnd();
289 for ( ; signal_it != signal_end; ++signal_it) {
290 const QDBusIntrospection::Signal &s = *signal_it;
291 Method mm;
292
293 mm.name = s.name.toLatin1();
294 QByteArray prototype = mm.name;
295 prototype += '(';
296
297 bool ok = true;
298
299 // build the output argument list
300 for (qsizetype i = 0; i < s.outputArgs.size(); ++i) {
301 const QDBusIntrospection::Argument &arg = s.outputArgs.at(i);
302
303 Type type = findType(signature: arg.type.toLatin1(), annotations: s.annotations, direction: "Out", id: i);
304 if (type.id == QMetaType::UnknownType) {
305 ok = false;
306 break;
307 }
308
309 mm.inputTypes.append(t: type.id);
310
311 mm.parameterNames.append(t: arg.name.toLatin1());
312
313 prototype.append(a: type.name);
314 prototype.append(c: ',');
315 }
316 if (!ok) continue;
317
318 // convert the last commas:
319 if (!mm.parameterNames.isEmpty())
320 prototype[prototype.size() - 1] = ')';
321 else
322 prototype.append(c: ')');
323
324 // meta method flags
325 mm.flags = AccessPublic | MethodSignal | MethodScriptable;
326
327 // add
328 signals_.insert(key: QMetaObject::normalizedSignature(method: prototype), value: mm);
329 }

332void QDBusMetaObjectGenerator::parseProperties()
333{
334 QDBusIntrospection::Properties::ConstIterator prop_it = data->properties.constBegin();
335 QDBusIntrospection::Properties::ConstIterator prop_end = data->properties.constEnd();
336 for ( ; prop_it != prop_end; ++prop_it) {
337 const QDBusIntrospection::Property &p = *prop_it;
338 Property mp;
339 Type type = findType(signature: p.type.toLatin1(), annotations: p.annotations);
340 if (type.id == QMetaType::UnknownType)
341 continue;
342
343 QByteArray name = p.name.toLatin1();
344 mp.signature = p.type.toLatin1();
345 mp.type = type.id;
346 mp.typeName = type.name;
347
348 // build the flags:
349 mp.flags = StdCppSet | Scriptable | Stored | Designable;
350 if (p.access != QDBusIntrospection::Property::Write)
351 mp.flags |= Readable;
352 if (p.access != QDBusIntrospection::Property::Read)
353 mp.flags |= Writable;
354
355 // add the property:
356 properties.insert(key: name, value: mp);
357 }
358}

360// Returns the sum of all parameters (including return type) for the given
361// \a map of methods. This is needed for calculating the size of the methods'
362// parameter type/name meta-data.
363qsizetype QDBusMetaObjectGenerator::aggregateParameterCount(const QMap<QByteArray, Method> &map)
364{
365 qsizetype sum = 0;
366 QMap<QByteArray, Method>::const_iterator it;
367 for (it = map.constBegin(); it != map.constEnd(); ++it) {
368 const Method &m = it.value();
369 sum += m.inputTypes.size() + qMax(a: qsizetype(1), b: m.outputTypes.size());
370 }
371 return sum;
372}

374void QDBusMetaObjectGenerator::write(QDBusMetaObject *obj)
375{
376 // this code here is mostly copied from qaxbase.cpp
377 // with a few modifications to make it cleaner
378
379 QString className = interface;
380 className.replace(c: u'.', after: "::"_L1);
381 if (className.isEmpty())
382 className = "QDBusInterface"_L1;
383
384 QVarLengthArray<uint> idata;
385 idata.resize(sz: sizeof(QDBusMetaObjectPrivate) / sizeof(uint));
386
387 qsizetype methodParametersDataSize =
388 ((aggregateParameterCount(map: signals_)
389 + aggregateParameterCount(map: methods)) * 2) // types and parameter names
390 - signals_.size() // return "parameters" don't have names
391 - methods.size(); // ditto
392
393 QDBusMetaObjectPrivate *header = reinterpret_cast<QDBusMetaObjectPrivate *>(idata.data());
394 static_assert(QMetaObjectPrivate::OutputRevision == 11, "QtDBus meta-object generator should generate the same version as moc");
395 header->revision = QMetaObjectPrivate::OutputRevision;
396 header->className = 0;
397 header->classInfoCount = 0;
398 header->classInfoData = 0;
399 header->methodCount = int(signals_.size() + methods.size());
400 header->methodData = int(idata.size());
401 header->propertyCount = int(properties.size());
402 header->propertyData = int(header->methodData + header->methodCount *
403 QMetaObjectPrivate::IntsPerMethod + methodParametersDataSize);
404 header->enumeratorCount = 0;
405 header->enumeratorData = 0;
406 header->constructorCount = 0;
407 header->constructorData = 0;
408 header->flags = RequiresVariantMetaObject;
409 header->signalCount = signals_.size();
410 // These are specific to QDBusMetaObject:
411 header->propertyDBusData = int(header->propertyData + header->propertyCount
412 * QMetaObjectPrivate::IntsPerProperty);
413 header->methodDBusData = int(header->propertyDBusData + header->propertyCount * intsPerProperty);
414
415 qsizetype data_size = idata.size() +
416 (header->methodCount * (QMetaObjectPrivate::IntsPerMethod+intsPerMethod)) + methodParametersDataSize +
417 (header->propertyCount * (QMetaObjectPrivate::IntsPerProperty+intsPerProperty));
418 for (const Method &mm : std::as_const(t&: signals_))
419 data_size += 2 + mm.inputTypes.size() + mm.outputTypes.size();
420 for (const Method &mm : std::as_const(t&: methods))
421 data_size += 2 + mm.inputTypes.size() + mm.outputTypes.size();
422 idata.resize(sz: data_size + 1);
423
424 QMetaStringTable strings(className.toLatin1());
425
426 qsizetype offset = header->methodData;
427 qsizetype parametersOffset = offset + header->methodCount * QMetaObjectPrivate::IntsPerMethod;
428 qsizetype signatureOffset = header->methodDBusData;
429 qsizetype typeidOffset = header->methodDBusData + header->methodCount * intsPerMethod;
430 idata[typeidOffset++] = 0; // eod
431
432 qsizetype totalMetaTypeCount = properties.size();
433 ++totalMetaTypeCount; // + 1 for metatype of dynamic metaobject
434 for (const auto& methodContainer: {signals_, methods}) {
435 for (const auto& method: methodContainer) {
436 qsizetype argc = method.inputTypes.size() + qMax(a: qsizetype(0), b: method.outputTypes.size() - 1);
437 totalMetaTypeCount += argc + 1;
438 }
439 }
440 QMetaType *metaTypes = new QMetaType[totalMetaTypeCount];
441 int propertyId = 0;
442
443 // add each method:
444 qsizetype currentMethodMetaTypeOffset = properties.size() + 1;
445 for (int x = 0; x < 2; ++x) {
446 // Signals must be added before other methods, to match moc.
447 QMap<QByteArray, Method> &map = (x == 0) ? signals_ : methods;
448 for (QMap<QByteArray, Method>::ConstIterator it = map.constBegin();
449 it != map.constEnd(); ++it) {
450 const Method &mm = it.value();
451
452 qsizetype argc = mm.inputTypes.size() + qMax(a: qsizetype(0), b: mm.outputTypes.size() - 1);
453
454 idata[offset++] = strings.enter(value: mm.name);
455 idata[offset++] = argc;
456 idata[offset++] = parametersOffset;
457 idata[offset++] = strings.enter(value: mm.tag);
458 idata[offset++] = mm.flags;
459 idata[offset++] = currentMethodMetaTypeOffset;
460
461 // Parameter types
462 for (qsizetype i = -1; i < argc; ++i) {
463 int type;
464 QByteArray typeName;
465 if (i < 0) { // Return type
466 if (!mm.outputTypes.isEmpty()) {
467 type = mm.outputTypes.first();
468 if (type == -1) {
469 type = IsUnresolvedType | strings.enter(value: mm.rawReturnType);
470 }
471 } else {
472 type = QMetaType::Void;
473 }
474 } else if (i < mm.inputTypes.size()) {
475 type = mm.inputTypes.at(idx: i);
476 } else {
477 Q_ASSERT(mm.outputTypes.size() > 1);
478 type = mm.outputTypes.at(idx: i - mm.inputTypes.size() + 1);
479 // Output parameters are references; type id not available
480 typeName = QMetaType(type).name();
481 typeName.append(c: '&');
482 type = QMetaType::UnknownType;
483 }
484 int typeInfo;
485 if (!typeName.isEmpty())
486 typeInfo = IsUnresolvedType | strings.enter(value: typeName);
487 else
488 typeInfo = type;
489 metaTypes[currentMethodMetaTypeOffset++] = QMetaType(type);
490 idata[parametersOffset++] = typeInfo;
491 }
492 // Parameter names
493 for (qsizetype i = 0; i < argc; ++i)
494 idata[parametersOffset++] = strings.enter(value: mm.parameterNames.at(i));
495
496 idata[signatureOffset++] = typeidOffset;
497 idata[typeidOffset++] = mm.inputTypes.size();
498 memcpy(dest: idata.data() + typeidOffset, src: mm.inputTypes.data(), n: mm.inputTypes.size() * sizeof(uint));
499 typeidOffset += mm.inputTypes.size();
500
501 idata[signatureOffset++] = typeidOffset;
502 idata[typeidOffset++] = mm.outputTypes.size();
503 memcpy(dest: idata.data() + typeidOffset, src: mm.outputTypes.data(), n: mm.outputTypes.size() * sizeof(uint));
504 typeidOffset += mm.outputTypes.size();
505 }
506 }
507
508 Q_ASSERT(offset == header->methodData + header->methodCount * QMetaObjectPrivate::IntsPerMethod);
509 Q_ASSERT(parametersOffset == header->propertyData);
510 Q_ASSERT(signatureOffset == header->methodDBusData + header->methodCount * intsPerMethod);
511 Q_ASSERT(typeidOffset == idata.size());
512 offset += methodParametersDataSize;
513 Q_ASSERT(offset == header->propertyData);
514
515 // add each property
516 signatureOffset = header->propertyDBusData;
517 for (QMap<QByteArray, Property>::ConstIterator it = properties.constBegin();
518 it != properties.constEnd(); ++it) {
519 const Property &mp = it.value();
520
521 // form is name, typeinfo, flags
522 idata[offset++] = strings.enter(value: it.key()); // name
523 Q_ASSERT(mp.type != QMetaType::UnknownType);
524 idata[offset++] = mp.type;
525 idata[offset++] = mp.flags;
526 idata[offset++] = -1; // notify index
527 idata[offset++] = 0; // revision
528
529 idata[signatureOffset++] = strings.enter(value: mp.signature);
530 idata[signatureOffset++] = mp.type;
531
532 metaTypes[propertyId++] = QMetaType(mp.type);
533 }
534 metaTypes[propertyId] = QMetaType(); // we can't know our own metatype
535
536 Q_ASSERT(offset == header->propertyDBusData);
537 Q_ASSERT(signatureOffset == header->methodDBusData);
538
539 char *string_data = new char[strings.blobSize()];
540 strings.writeBlob(out: string_data);
541
542 uint *uint_data = new uint[idata.size()];
543 memcpy(dest: uint_data, src: idata.data(), n: idata.size() * sizeof(uint));
544
545 // put the metaobject together
546 obj->d.data = uint_data;
547 obj->d.relatedMetaObjects = nullptr;
548 obj->d.static_metacall = nullptr;
549 obj->d.extradata = nullptr;
550 obj->d.stringdata = reinterpret_cast<const uint *>(string_data);
551 obj->d.superdata = &QDBusAbstractInterface::staticMetaObject;
552 obj->d.metaTypes = reinterpret_cast<QtPrivate::QMetaTypeInterface *const *>(metaTypes);
553}

584QDBusMetaObject *QDBusMetaObject::createMetaObject(const QString &interface, const QString &xml,
585 QHash<QString, QDBusMetaObject *> &cache,
586 QDBusError &error)
587{
588 error = QDBusError();
589 QDBusIntrospection::Interfaces parsed = QDBusIntrospection::parseInterfaces(xml);
590
591 QDBusMetaObject *we = nullptr;
592 QDBusIntrospection::Interfaces::ConstIterator it = parsed.constBegin();
593 QDBusIntrospection::Interfaces::ConstIterator end = parsed.constEnd();
594 for ( ; it != end; ++it) {
595 // check if it's in the cache
596 bool us = it.key() == interface;
597
598 QDBusMetaObject *obj = cache.value(key: it.key(), defaultValue: 0);
599 if (!obj && (us || !interface.startsWith(s: "local."_L1 ))) {
600 // not in cache; create
601 obj = new QDBusMetaObject;
602 QDBusMetaObjectGenerator generator(it.key(), it.value().constData());
603 generator.write(obj);
604
605 if ((obj->cached = !it.key().startsWith(s: "local."_L1)))
606 // cache it
607 cache.insert(key: it.key(), value: obj);
608 else if (!us)
609 delete obj;
610
611 }
612
613 if (us)
614 // it's us
615 we = obj;
616 }
617
618 if (we)
619 return we;
620 // still nothing?
621
622 if (parsed.isEmpty()) {
623 // object didn't return introspection
624 we = new QDBusMetaObject;
625 QDBusMetaObjectGenerator generator(interface, nullptr);
626 generator.write(obj: we);
627 we->cached = false;
628 return we;
629 } else if (interface.isEmpty()) {
630 // merge all interfaces
631 it = parsed.constBegin();
632 QDBusIntrospection::Interface merged = *it.value().constData();
633
634 for (++it; it != end; ++it) {
635 merged.annotations.insert(map: it.value()->annotations);
636 merged.methods.unite(other: it.value()->methods);
637 merged.signals_.unite(other: it.value()->signals_);
638 merged.properties.insert(map: it.value()->properties);
639 }
640
641 merged.name = "local.Merged"_L1;
642 merged.introspection.clear();
643
644 we = new QDBusMetaObject;
645 QDBusMetaObjectGenerator generator(merged.name, &merged);
646 generator.write(obj: we);
647 we->cached = false;
648 return we;
649 }
650
651 // mark as an error
652 error = QDBusError(QDBusError::UnknownInterface,
653 "Interface '%1' was not found"_L1.arg(args: interface));
654 return nullptr;
655}

666const int *QDBusMetaObject::inputTypesForMethod(int id) const
667{
668 //id -= methodOffset();
