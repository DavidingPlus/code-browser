42QDBusConnectionPrivate *QDBusConnectionManager::busConnection(QDBusConnection::BusType type)
43{
44 static_assert(int(QDBusConnection::SessionBus) + int(QDBusConnection::SystemBus) == 1);
45 Q_ASSERT(type == QDBusConnection::SessionBus || type == QDBusConnection::SystemBus);
46
47 if (!qdbus_loadLibDBus())
48 return nullptr;
49
50 // we'll start in suspended delivery mode if we're in the main thread
51 // (the event loop will resume delivery)
52 bool suspendedDelivery = qApp && qApp->thread() == QThread::currentThread();
53
54 const auto locker = qt_scoped_lock(mutex&: defaultBusMutex);
55 if (defaultBuses[type])
56 return defaultBuses[type];
57
58 QString name = QStringLiteral("qt_default_session_bus");
59 if (type == QDBusConnection::SystemBus)
60 name = QStringLiteral("qt_default_system_bus");
61 return defaultBuses[type] = connectToBus(type, name, suspendedDelivery);
62}

69void QDBusConnectionManager::removeConnection(const QString &name)
70{
71 QDBusConnectionPrivate *d = nullptr;
72 d = connectionHash.take(key: name);
73 if (d && !d->ref.deref())
74 d->deleteLater();
75
76 // Static objects may be keeping the connection open.
77 // However, it is harmless to have outstanding references to a connection that is
78 // closing as long as those references will be soon dropped without being used.
79
80 // ### Output a warning if connections are being used after they have been removed.
81}

121void QDBusConnectionManager::run()
122{
123 exec();
124
125 // cleanup:
126 const auto locker = qt_scoped_lock(mutex);
127 for (QHash<QString, QDBusConnectionPrivate *>::const_iterator it = connectionHash.constBegin();
128 it != connectionHash.constEnd(); ++it) {
129 QDBusConnectionPrivate *d = it.value();
130 if (!d->ref.deref()) {
131 delete d;
132 } else {
133 d->closeConnection();
134 d->moveToThread(thread: nullptr); // allow it to be deleted in another thread
135 }
136 }
137 connectionHash.clear();
138
139 // allow deletion from any thread without warning
140 moveToThread(thread: nullptr);
141}

143QDBusConnectionPrivate *QDBusConnectionManager::connectToBus(QDBusConnection::BusType type, const QString &name,
144 bool suspendedDelivery)
145{
146 ConnectionRequestData data;
147 data.type = ConnectionRequestData::ConnectToStandardBus;
148 data.busType = type;
149 data.name = &name;
150 data.suspendedDelivery = suspendedDelivery;
151
152 emit connectionRequested(&data);
153 if (suspendedDelivery && data.result->connection) {
154 data.result->ref.ref();
155 QDBusConnectionDispatchEnabler *o = new QDBusConnectionDispatchEnabler(data.result);
156 QTimer::singleShot(msec: 0, receiver: o, SLOT(execute()));
157 o->moveToThread(qApp->thread()); // qApp was checked in the caller
158 }
159 return data.result;
160}

186void QDBusConnectionManager::executeConnectionRequest(QDBusConnectionManager::ConnectionRequestData *data)
187{
188 const auto locker = qt_scoped_lock(mutex);
189 const QString &name = *data->name;
190 QDBusConnectionPrivate *&d = data->result;
191
192 // check if the connection exists by name
193 d = connection(name);
194 if (d || name.isEmpty())
195 return;
196
197 d = new QDBusConnectionPrivate;
198 DBusConnection *c = nullptr;
199 QDBusErrorInternal error;
200 switch (data->type) {
201 case ConnectionRequestData::ConnectToStandardBus:
202 switch (data->busType) {
203 case QDBusConnection::SystemBus:
204 c = q_dbus_bus_get_private(type: DBUS_BUS_SYSTEM, error);
205 break;
206 case QDBusConnection::SessionBus:
207 c = q_dbus_bus_get_private(type: DBUS_BUS_SESSION, error);
208 break;
209 case QDBusConnection::ActivationBus:
210 c = q_dbus_bus_get_private(type: DBUS_BUS_STARTER, error);
211 break;
212 }
213 break;
214
215 case ConnectionRequestData::ConnectToBusByAddress:
216 case ConnectionRequestData::ConnectToPeerByAddress:
217 c = q_dbus_connection_open_private(address: data->busAddress->toUtf8().constData(), error);
218 if (c && data->type == ConnectionRequestData::ConnectToBusByAddress) {
219 // register on the bus
220 if (!q_dbus_bus_register(connection: c, error)) {
221 q_dbus_connection_unref(connection: c);
222 c = nullptr;
223 }
224 }
225 break;
226 }
227
228 setConnection(name, c: d);
229 if (data->type == ConnectionRequestData::ConnectToPeerByAddress) {
230 d->setPeer(connection: c, error);
231 } else {
232 // create the bus service
233 // will lock in QDBusConnectionPrivate::connectRelay()
234 d->setConnection(connection: c, error);
235 d->createBusService();
236 if (c && data->suspendedDelivery)
237 d->setDispatchEnabled(false);
238 }
239}

249/*!
250 \class QDBusConnection
251 \inmodule QtDBus
252 \since 4.2
253
254 \brief The QDBusConnection class represents a connection to the D-Bus bus daemon.
255
256 This class is the initial point in a D-Bus session. Using it, you
257 can get access to remote objects, interfaces; connect remote
258 signals to your object's slots; register objects, etc.
259
260 D-Bus connections are created using the connectToBus() function,
261 which opens a connection to the server daemon and does the initial
262 handshaking, associating that connection with a name. Further
263 attempts to connect using the same name will return the same
264 connection.
265
266 The connection is then torn down using the disconnectFromBus()
267 function.
268
269 Once disconnected, calling connectToBus() will not reestablish a
270 connection, you must create a new QDBusConnection instance.
271
272 As a convenience for the two most common connection types, the
273 sessionBus() and systemBus() functions return open connections to
274 the session server daemon and the system server daemon,
275 respectively. Those connections are opened when first used and are
276 closed when the QCoreApplication destructor is run.
277
278 D-Bus also supports peer-to-peer connections, without the need for
279 a bus server daemon. Using this facility, two applications can
280 talk to each other and exchange messages. This can be achieved by
281 passing an address to connectToBus() function, which was opened by
282 another D-Bus application using QDBusServer.
283*/
284
285/*!
286 \enum QDBusConnection::BusType
287 Specifies the type of the bus connection. The valid bus types are:
288
289 \value SessionBus the session bus, associated with the running desktop session
290 \value SystemBus the system bus, used to communicate with system-wide processes
291 \value ActivationBus the activation bus, the "alias" for the bus that started the
292 service
293
294 On the Session Bus, one can find other applications by the same user that are sharing the same
295 desktop session (hence the name). On the System Bus, however, processes shared for the whole
296 system are usually found.
297*/

514/*!
515 Sends the \a message over this connection, without waiting for a
516 reply. This is suitable for errors, signals, and return values as
517 well as calls whose return values are not necessary.
518
519 Returns \c true if the message was queued successfully, false otherwise.
520*/
521bool QDBusConnection::send(const QDBusMessage &message) const
522{
523 if (!d || !d->connection) {
524 QDBusError err = QDBusError(QDBusError::Disconnected,
525 QDBusUtil::disconnectedErrorMessage());
526 if (d)
527 d->lastError = err;
528 return false;
529 }
530 return d->send(message);
531}

533/*!
534 Sends the \a message over this connection and returns immediately.
535 When the reply is received, the method \a returnMethod is called in
536 the \a receiver object. If an error occurs, the method \a errorMethod
537 will be called instead.
538
539 If no reply is received within \a timeout milliseconds, an automatic
540 error will be delivered indicating the expiration of the call.
541 The default \a timeout is -1, which will be replaced with an
542 implementation-defined value that is suitable for inter-process
543 communications (generally, 25 seconds).
544
545 This function is suitable for method calls only. It is guaranteed
546 that the slot will be called exactly once with the reply, as long
547 as the parameter types match and no error occurs.
548
549 Returns \c true if the message was sent, or false if the message could
550 not be sent.
551*/
552bool QDBusConnection::callWithCallback(const QDBusMessage &message, QObject *receiver,
553 const char *returnMethod, const char *errorMethod,
554 int timeout) const
555{
556 if (!d || !d->connection) {
557 QDBusError err = QDBusError(QDBusError::Disconnected,
558 QDBusUtil::disconnectedErrorMessage());
559 if (d)
560 d->lastError = err;
561 return false;
562 }
563 return d->sendWithReplyAsync(message, receiver, returnMethod, errorMethod, timeout) != nullptr;
564}

589/*!
590 Sends the \a message over this connection and blocks, waiting for
591 a reply, for at most \a timeout milliseconds. This function is
592 suitable for method calls only. It returns the reply message as
593 its return value, which will be either of type
594 QDBusMessage::ReplyMessage or QDBusMessage::ErrorMessage.
595
596 If no reply is received within \a timeout milliseconds, an automatic
597 error will be delivered indicating the expiration of the call.
598 The default \a timeout is -1, which will be replaced with an
599 implementation-defined value that is suitable for inter-process
600 communications (generally, 25 seconds).
601
602 See the QDBusInterface::call() function for a more friendly way
603 of placing calls.
604
605 \warning If \a mode is QDBus::BlockWithGui, this function will
606 reenter the Qt event loop in order to wait for the
607 reply. During the wait, it may deliver signals and other
608 method calls to your application. Therefore, it must be
609 prepared to handle a reentrancy whenever a call is
610 placed with call().
611*/
612QDBusMessage QDBusConnection::call(const QDBusMessage &message, QDBus::CallMode mode, int timeout) const
613{
614 if (!d || !d->connection) {
615 QDBusError err = QDBusError(QDBusError::Disconnected,
616 QDBusUtil::disconnectedErrorMessage());
617 if (d)
618 d->lastError = err;
619
620 return QDBusMessage::createError(err);
621 }
622
623 if (mode != QDBus::NoBlock)
624 return d->sendWithReply(message, mode, timeout);
625
626 d->send(message);
627 QDBusMessage retval;
628 retval << QVariant(); // add one argument (to avoid .at(0) problems)
629 return retval;
630}

700/*!
701 \overload
702 \since 4.6
703
704 Connects the signal to the slot \a slot in object \a
705 receiver. Unlike the previous connect() overload, this function
706 allows one to specify the parameter signature to be connected
707 using the \a signature variable. The function will then verify
708 that this signature can be delivered to the slot specified by \a
709 slot and return false otherwise.
710
711 The \a argumentMatch parameter lists the string parameters to be matched,
712 in sequential order. Note that, to match an empty string, you need to
713 pass a QString that is empty but not null (i.e., QString("")). A null
714 QString skips matching at that position.
715
716 Returns \c true if the connection was successful.
717
718 \note This function verifies that the signal signature matches the
719 slot's parameters, but it does not verify that the actual
720 signal exists with the given signature in the remote
721 service.
722*/
723bool QDBusConnection::connect(const QString &service, const QString &path, const QString& interface,
724 const QString &name, const QStringList &argumentMatch, const QString &signature,
725 QObject *receiver, const char *slot)
726{
727
728 if (!receiver || !slot || !d || !d->connection)
729 return false;
730 if (interface.isEmpty() && name.isEmpty())
731 return false;

830/*!
831 \overload
832 \since 5.5
833
834 Registers the object \a object at path \a path with interface name \a interface
835 and returns \c true if the registration was successful. The \a options parameter
836 specifies how much of the object \a object will be exposed through
837 D-Bus.
838
839 This function does not replace existing objects: if there is already an object registered at
840 path \a path, this function will return false. Use unregisterObject() to unregister it first.
841
842 The ExportChildObjects flag exports child objects on D-Bus based on the
843 path of the registered objects and the QObject::objectName of the child.
844 Therefore, it is important for the child object to have an object name.
845
846 You cannot register an object as a child object of an object that
847 was registered with ExportChildObjects.
848*/
849bool QDBusConnection::registerObject(const QString &path, const QString &interface, QObject *object, RegisterOptions options)
850{
851 Q_ASSERT_X(QDBusUtil::isValidObjectPath(path), "QDBusConnection::registerObject",
852 "Invalid object path given");
853 if (!d || !d->connection || !object || !options || !QDBusUtil::isValidObjectPath(path))
854 return false;
855
856 auto pathComponents = QStringView{path}.split(sep: u'/');
857 if (pathComponents.constLast().isEmpty())
858 pathComponents.removeLast();
859 QDBusWriteLocker locker(RegisterObjectAction, d);
860
861 // lower-bound search for where this object should enter in the tree
862 QDBusConnectionPrivate::ObjectTreeNode *node = &d->rootNode;
863 int i = 1;
864 while (node) {
865 if (pathComponents.size() == i) {
866 // this node exists
867 // consider it free if there's no object here and the user is not trying to
868 // replace the object sub-tree
869 if (node->obj)
870 return false;
871
872 if (options & QDBusConnectionPrivate::VirtualObject) {
873 if (options & SubPath && !node->children.isEmpty())
874 return false;
875 } else {
876 if ((options & ExportChildObjects && !node->children.isEmpty()))
877 return false;
878 }
879 // we can add the object here
880 node->obj = object;
881 node->flags = options;
882 node->interfaceName = interface;
883
884 d->registerObject(node);
885 //qDebug("REGISTERED FOR %s", path.toLocal8Bit().constData());
886 return true;
887 }
888
889 // if a virtual object occupies this path, return false
890 if (node->obj && (node->flags & QDBusConnectionPrivate::VirtualObject) && (node->flags & QDBusConnection::SubPath)) {
891 //qDebug("Cannot register object at %s because QDBusVirtualObject handles all sub-paths.",
892 // qPrintable(path));
893 return false;
894 }
895
896 // find the position where we'd insert the node
897 QDBusConnectionPrivate::ObjectTreeNode::DataList::Iterator it =
898 std::lower_bound(node->children.begin(), node->children.end(), pathComponents.at(i));
899 if (it != node->children.end() && it->name == pathComponents.at(i)) {
900 // match: this node exists
901 node = &(*it);
902
903 // are we allowed to go deeper?
904 if (node->flags & ExportChildObjects) {
905 // we're not
906 //qDebug("Cannot register object at %s because %s exports its own child objects",
907 // qPrintable(path), qPrintable(pathComponents.at(i)));
908 return false;
909 }
910 } else {
911 // add entry
912 it = node->children.insert(before: it, t: pathComponents.at(i).toString());
913 node = &(*it);
914 }
915
916 // iterate
917 ++i;
918 }
919
920 Q_ASSERT_X(false, "QDBusConnection::registerObject", "The impossible happened");
921 return false;
922}

954/*!
955 Return the object that was registered with the registerObject() at the object path given by
956 \a path.
957*/
958QObject *QDBusConnection::objectRegisteredAt(const QString &path) const
959{
960 Q_ASSERT_X(QDBusUtil::isValidObjectPath(path), "QDBusConnection::registeredObject",
961 "Invalid object path given");
962 if (!d || !d->connection || !QDBusUtil::isValidObjectPath(path))
963 return nullptr;
964
965 auto pathComponents = QStringView{path}.split(sep: u'/');
966 if (pathComponents.constLast().isEmpty())
967 pathComponents.removeLast();
968
969 // lower-bound search for where this object should enter in the tree
970 QDBusReadLocker lock(ObjectRegisteredAtAction, d);
971 const QDBusConnectionPrivate::ObjectTreeNode *node = &d->rootNode;
972
973 int i = 1;
974 while (node) {
975 if (pathComponents.size() == i)
976 return node->obj;
977 if ((node->flags & QDBusConnectionPrivate::VirtualObject) && (node->flags & QDBusConnection::SubPath))
978 return node->obj;
979
980 QDBusConnectionPrivate::ObjectTreeNode::DataList::ConstIterator it =
981 std::lower_bound(node->children.constBegin(), node->children.constEnd(), pathComponents.at(i));
982 if (it == node->children.constEnd() || it->name != pathComponents.at(i))
983 break; // node not found
984
985 node = &(*it);
986 ++i;
987 }
988 return nullptr;
989}

1230static void preventDllUnload()
1231{
1232 // Thread termination is really wacky on Windows. For some reason we don't
1233 // understand, exiting from the thread may try to unload the DLL. Since the
1234 // QDBusConnectionManager thread runs until the DLL is unloaded, we've got
1235 // a deadlock: the main thread is waiting for the manager thread to exit,
1236 // but the manager thread is attempting to acquire a lock to unload the DLL.
1237 //
1238 // We work around the issue by preventing the unload from happening in the
1239 // first place.
1240 //
1241 // For this trick, see
1242 // https://blogs.msdn.microsoft.com/oldnewthing/20131105-00/?p=2733
1243
1244 static HMODULE self;
1245 GetModuleHandleEx(GET_MODULE_HANDLE_EX_FLAG_FROM_ADDRESS |
1246 GET_MODULE_HANDLE_EX_FLAG_PIN,
1247 reinterpret_cast<const wchar_t *>(&self), // any address in this DLL
1248 &self);
1249}
