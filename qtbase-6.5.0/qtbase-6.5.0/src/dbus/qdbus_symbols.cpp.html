27void qdbus_unloadLibDBus()
28{
29 if (qdbus_libdbus) {
30 if (qEnvironmentVariableIsSet("QDBUS_FORCE_SHUTDOWN"))
31 qdbus_libdbus->resolve("dbus_shutdown")();
32 qdbus_libdbus->unload();
33 }
34 delete qdbus_libdbus;
35 qdbus_libdbus = nullptr;
36}
37#endif

39bool qdbus_loadLibDBus()
40{
41#if QT_CONFIG(library)
42#ifdef QT_BUILD_INTERNAL
43 // this is to simulate a library load failure for our autotest suite.
44 if (!qEnvironmentVariableIsEmpty("QT_SIMULATE_DBUS_LIBFAIL"))
45 return false;
46#endif
47
48 Q_CONSTINIT static bool triedToLoadLibrary = false;
49 Q_CONSTINIT static QBasicMutex mutex;
50 const auto locker = qt_scoped_lock(mutex);
51
52 QLibrary *&lib = qdbus_libdbus;
53 if (triedToLoadLibrary)
54 return lib && lib->isLoaded();
55
56 lib = new QLibrary;
57 lib->setLoadHints(QLibrary::ExportExternalSymbolsHint); // make libdbus symbols available for apps that need more advanced control over the dbus
58 triedToLoadLibrary = true;
59
60 static int majorversions[] = { 3, 2, -1 };
61 const QString baseNames[] = {
62#ifdef Q_OS_WIN
63 "dbus-1"_L1,
64#endif
65 "libdbus-1"_L1
66 };
67
68 lib->unload();
69 for (const int majorversion : majorversions) {
70 for (const QString &baseName : baseNames) {
71#ifdef Q_OS_WIN
72 QString suffix;
73 if (majorversion != -1)
74 suffix = QString::number(- majorversion); // negative so it prepends the dash
75 lib->setFileName(baseName + suffix);
76#else
77 lib->setFileNameAndVersion(baseName, majorversion);
78#endif
79 if (lib->load() && lib->resolve("dbus_connection_open_private"))
80 return true;
81
82 lib->unload();
83 }
84 }
85
86 delete lib;
87 lib = nullptr;
88 return false;
89#else
90 return true;
91#endif
92}

94void (*qdbus_resolve_conditionally(const char *name))()
95{
96#if QT_CONFIG(library)
97 if (qdbus_loadLibDBus())
98 return qdbus_libdbus->resolve(name);
99#else
100 Q_UNUSED(name);
101#endif
102 return nullptr;
103}
104
105void (*qdbus_resolve_me(const char *name))()
106{
107#if QT_CONFIG(library)
108 if (Q_UNLIKELY(!qdbus_loadLibDBus()))
109 qFatal("Cannot find libdbus-1 in your system to resolve symbol '%s'.", name);
110
111 QFunctionPointer ptr = qdbus_libdbus->resolve(name);
112 if (Q_UNLIKELY(!ptr))
113 qFatal("Cannot resolve '%s' in your libdbus-1.", name);
114
115 return ptr;
116#else
117 Q_UNUSED(name);
118 return nullptr;
119#endif
120}
