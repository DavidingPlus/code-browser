49void QDBusMetaTypeId::init()
50{
51 Q_CONSTINIT static QBasicAtomicInt initialized = Q_BASIC_ATOMIC_INITIALIZER(false);
52
53 // reentrancy is not a problem since everything else is locked on their own
54 // set the guard variable at the end
55 if (!initialized.loadRelaxed()) {
56 // register our types with Qt Core (calling qMetaTypeId<T>() does this implicitly)
57 (void)message();
58 (void)argument();
59 (void)variant();
60 (void)objectpath();
61 (void)signature();
62 (void)error();
63 (void)unixfd();
64
65#ifndef QDBUS_NO_SPECIALTYPES
66 // and register Qt Core's with us
67 qDBusRegisterMetaType<QDate>();
68 qDBusRegisterMetaType<QTime>();
69 qDBusRegisterMetaType<QDateTime>();
70 qDBusRegisterMetaType<QRect>();
71 qDBusRegisterMetaType<QRectF>();
72 qDBusRegisterMetaType<QSize>();
73 qDBusRegisterMetaType<QSizeF>();
74 qDBusRegisterMetaType<QPoint>();
75 qDBusRegisterMetaType<QPointF>();
76 qDBusRegisterMetaType<QLine>();
77 qDBusRegisterMetaType<QLineF>();
78 qDBusRegisterMetaType<QVariantList>();
79 qDBusRegisterMetaType<QVariantMap>();
80 qDBusRegisterMetaType<QVariantHash>();
81 qDBusRegisterMetaType<QDBusObjectPath>();
82
83 qDBusRegisterMetaType<QList<bool> >();
84 qDBusRegisterMetaType<QList<short> >();
85 qDBusRegisterMetaType<QList<ushort> >();
86 qDBusRegisterMetaType<QList<int> >();
87 qDBusRegisterMetaType<QList<uint> >();
88 qDBusRegisterMetaType<QList<qlonglong> >();

92 qDBusRegisterMetaType<QList<QDBusSignature> >();

94#endif
95
96 initialized.storeRelaxed(newValue: true);
97 }

104/*!
105 \class QDBusMetaType
106 \inmodule QtDBus
107 \brief Meta-type registration system for the Qt D-Bus module.
108 \internal
109
110 The QDBusMetaType class allows you to register class types for
111 marshalling and demarshalling over D-Bus. D-Bus supports a very
112 limited set of primitive types, but allows one to extend the type
113 system by creating compound types, such as arrays (lists) and
114 structs. In order to use them with Qt D-Bus, those types must be
115 registered.
116
117 See \l {qdbustypesystem.html}{Qt D-Bus Type System} for more
118 information on the type system and how to register additional
119 types.
120
121 \sa {qdbustypesystem.html}{Qt D-Bus Type System},
122 qDBusRegisterMetaType(), QMetaType, QVariant, QDBusArgument
123*/
124
125/*!
126 \fn int qDBusRegisterMetaType()
127 \relates QDBusArgument
128 \threadsafe
129 \since 4.2
130
131 Registers \c{T} with the
132 \l {qdbustypesystem.html}{Qt D-Bus Type System} and the Qt \l
133 {QMetaType}{meta-type system}, if it's not already registered.
134
135 To register a type, it must be declared as a meta-type with the
136 Q_DECLARE_METATYPE() macro, and then registered as in the
137 following example:
138
139 \snippet code/src_qdbus_qdbusmetatype.cpp 0-0
140 \codeline
141 \snippet code/src_qdbus_qdbusmetatype.cpp 0-1
142
143 If \c{T} isn't one of
144 Qt's \l{container classes}, the \c{operator<<} and
145 \c{operator>>} streaming operators between \c{T} and QDBusArgument
146 must be already declared. See the \l {qdbustypesystem.html}{Qt D-Bus
147 Type System} page for more information on how to declare such
148 types.
149
150 This function returns the Qt meta type id for the type (the same
151 value that is returned from qRegisterMetaType()).
152
153 \note The feature that a \c{T} inheriting a streamable type (including
154 the containers QList, QHash or QMap) can be streamed without providing
155 custom \c{operator<<} and \c{operator>>} is deprecated as of Qt 5.7,
156 because it ignores everything in \c{T} except the base class. There is
157 no diagnostic. You should always provide these operators for all types
158 you wish to stream and not rely on Qt-provided stream operators for
159 base classes.
160
161 \sa {qdbustypesystem.html}{Qt D-Bus Type System}, qRegisterMetaType(), QMetaType
162*/
163

174/*!
175 \internal
176 Registers the marshalling and demarshalling functions for meta
177 type \a metaType.
178*/
179void QDBusMetaType::registerMarshallOperators(QMetaType metaType, MarshallFunction mf,
180 DemarshallFunction df)
181{
182 int id = metaType.id();
183 auto *ct = customTypes();
184 if (id < 0 || !mf || !df || !ct)
185 return; // error!
186
187 QWriteLocker locker(customTypesLock());
188 QDBusCustomTypeInfo &info = (*ct)[id];
189 info.marshall = mf;
190 info.demarshall = df;
191}
192
193/*!
194 \internal
195 Executes the marshalling of type \a metaType (whose data is contained in
196 \a data) to the D-Bus marshalling argument \a arg. Returns \c true if
197 the marshalling succeeded, or false if an error occurred.
198*/
199bool QDBusMetaType::marshall(QDBusArgument &arg, QMetaType metaType, const void *data)
200{
201 int id = metaType.id();
202 QDBusMetaTypeId::init();
203
204 MarshallFunction mf;
205 {
206 QReadLocker locker(customTypesLock());
207 auto *ct = customTypes();
208 auto it = ct->constFind(key: id);
209 if (it == ct->cend())
210 return false; // non-existent
211
212 const QDBusCustomTypeInfo &info = *it;
213 if (!info.marshall) {
214 mf = nullptr; // make gcc happy
215 return false;
216 } else
217 mf = info.marshall;
218 }
219
220 mf(arg, data);
221 return true;
222}
223
224/*!
225 \internal
226 Executes the demarshalling of type \a metaType (whose data will be placed in
227 \a data) from the D-Bus marshalling argument \a arg. Returns \c true if
228 the demarshalling succeeded, or false if an error occurred.
229*/
230bool QDBusMetaType::demarshall(const QDBusArgument &arg, QMetaType metaType, void *data)
231{
232 int id = metaType.id();
233 QDBusMetaTypeId::init();
234
235 DemarshallFunction df;
236 {
237 QReadLocker locker(customTypesLock());
238 auto *ct = customTypes();
239 auto it = ct->constFind(key: id);
240 if (it == ct->cend())
241 return false; // non-existent
242
243 const QDBusCustomTypeInfo &info = *it;
244 if (!info.demarshall) {
245 df = nullptr; // make gcc happy
246 return false;
247 } else
248 df = info.demarshall;
249 }
250#ifndef QT_BOOTSTRAPPED
251 QDBusArgument copy = arg;
252 df(copy, data);
253#else
254 Q_UNUSED(arg);
255 Q_UNUSED(data);
256 Q_UNUSED(df);
257#endif
258 return true;
259}

261/*!
262 \fn QDBusMetaType::signatureToType(const char *signature)
263 \internal
264
265 Returns the Qt meta type id for the given D-Bus signature for exactly one full type, given
266 by \a signature.
267
268 Note: this function only handles the basic D-Bus types.
269
270 \sa QDBusUtil::isValidSingleSignature(), typeToSignature(),
271 QVariant::metaType()
272*/
273QMetaType QDBusMetaType::signatureToMetaType(const char *signature)
274{
275 if (!signature)
276 return QMetaType(QMetaType::UnknownType);
277
278 QDBusMetaTypeId::init();
279 switch (signature[0])
280 {
281 case DBUS_TYPE_BOOLEAN:
282 return QMetaType(QMetaType::Bool);
283
284 case DBUS_TYPE_BYTE:
285 return QMetaType(QMetaType::UChar);
286
287 case DBUS_TYPE_INT16:
288 return QMetaType(QMetaType::Short);
289
290 case DBUS_TYPE_UINT16:
291 return QMetaType(QMetaType::UShort);
292
293 case DBUS_TYPE_INT32:
294 return QMetaType(QMetaType::Int);
295
296 case DBUS_TYPE_UINT32:
297 return QMetaType(QMetaType::UInt);
298
299 case DBUS_TYPE_INT64:
300 return QMetaType(QMetaType::LongLong);
301
302 case DBUS_TYPE_UINT64:
303 return QMetaType(QMetaType::ULongLong);
304
305 case DBUS_TYPE_DOUBLE:
306 return QMetaType(QMetaType::Double);
307
308 case DBUS_TYPE_STRING:
309 return QMetaType(QMetaType::QString);
310
311 case DBUS_TYPE_OBJECT_PATH:
312 return QDBusMetaTypeId::objectpath();
313
314 case DBUS_TYPE_SIGNATURE:
315 return QDBusMetaTypeId::signature();
316
317 case DBUS_TYPE_UNIX_FD:
318 return QDBusMetaTypeId::unixfd();
319
320 case DBUS_TYPE_VARIANT:
321 return QDBusMetaTypeId::variant();
322
323 case DBUS_TYPE_ARRAY: // special case
324 switch (signature[1]) {
325 case DBUS_TYPE_BYTE:
326 return QMetaType(QMetaType::QByteArray);
327
328 case DBUS_TYPE_STRING:
329 return QMetaType(QMetaType::QStringList);
330
331 case DBUS_TYPE_VARIANT:
332 return QMetaType(QMetaType::QVariantList);
333
334 case DBUS_TYPE_OBJECT_PATH:
335 return QMetaType::fromType<QList<QDBusObjectPath> >();
336
337 case DBUS_TYPE_SIGNATURE:
338 return QMetaType::fromType<QList<QDBusSignature> >();
339
340 }
341 Q_FALLTHROUGH();
342 default:
343 return QMetaType(QMetaType::UnknownType);
344 }
345}

355void QDBusMetaType::registerCustomType(QMetaType type, const QByteArray &signature)
356{
357 auto *ct = customTypes();
358 QWriteLocker locker(customTypesLock());
359 auto &info = (*ct)[type.id()];
360 info.signature = signature;
361 // note how marshall/demarshall are not set, the type is never used at runtime
362}

364/*!
365 \fn QDBusMetaType::typeToSignature(int type)
366 \internal
367
368 Returns the D-Bus signature equivalent to the supplied meta type id \a type.
369
370 More types can be registered with the qDBusRegisterMetaType() function.
371
372 \sa QDBusUtil::isValidSingleSignature(), signatureToType(),
373 QVariant::metaType()
374*/
375const char *QDBusMetaType::typeToSignature(QMetaType type)
376{
377 // check if it's a static type
378 switch (type.id())
379 {
380 case QMetaType::UChar:
381 return DBUS_TYPE_BYTE_AS_STRING;
382
383 case QMetaType::Bool:
384 return DBUS_TYPE_BOOLEAN_AS_STRING;
385
386 case QMetaType::Short:
387 return DBUS_TYPE_INT16_AS_STRING;
388
389 case QMetaType::UShort:
390 return DBUS_TYPE_UINT16_AS_STRING;
391
392 case QMetaType::Int:
393 return DBUS_TYPE_INT32_AS_STRING;
394
395 case QMetaType::UInt:
396 return DBUS_TYPE_UINT32_AS_STRING;
397
398 case QMetaType::LongLong:
399 return DBUS_TYPE_INT64_AS_STRING;
400
401 case QMetaType::ULongLong:
402 return DBUS_TYPE_UINT64_AS_STRING;
403
404 case QMetaType::Double:
405 return DBUS_TYPE_DOUBLE_AS_STRING;
406
407 case QMetaType::QString:
408 return DBUS_TYPE_STRING_AS_STRING;
409
410 case QMetaType::QStringList:
411 return DBUS_TYPE_ARRAY_AS_STRING
412 DBUS_TYPE_STRING_AS_STRING; // as
413
414 case QMetaType::QByteArray:
415 return DBUS_TYPE_ARRAY_AS_STRING
416 DBUS_TYPE_BYTE_AS_STRING; // ay
417 }
418
419 QDBusMetaTypeId::init();
420 if (type == QDBusMetaTypeId::variant())
421 return DBUS_TYPE_VARIANT_AS_STRING;
422 else if (type == QDBusMetaTypeId::objectpath())
423 return DBUS_TYPE_OBJECT_PATH_AS_STRING;
424 else if (type == QDBusMetaTypeId::signature())
425 return DBUS_TYPE_SIGNATURE_AS_STRING;
426 else if (type == QDBusMetaTypeId::unixfd())
427 return DBUS_TYPE_UNIX_FD_AS_STRING;
428
429 // try the database
430 auto *ct = customTypes();
431 {
432 QReadLocker locker(customTypesLock());
433 auto it = ct->constFind(key: type.id());
434 if (it == ct->end())
435 return nullptr;
436
437 const QDBusCustomTypeInfo &info = *it;
438
439 if (!info.signature.isNull())
440 return info.signature;
441
442 if (!info.marshall)
443 return nullptr; // type not registered with us
444 }
445
446 // call to user code to construct the signature type
447 QDBusCustomTypeInfo *info;
448 {
449 // createSignature will never return a null QByteArray
450 // if there was an error, it'll return ""
451 QByteArray signature = QDBusArgumentPrivate::createSignature(id: type.id());
452
453 // re-acquire lock
454 QWriteLocker locker(customTypesLock());
455 info = &(*ct)[type.id()];
456 info->signature = signature;
457 }
458 return info->signature;
459}
