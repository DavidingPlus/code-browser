34/*!
35 \class QNetworkDiskCache
36 \since 4.5
37 \inmodule QtNetwork
38
39 \brief The QNetworkDiskCache class provides a very basic disk cache.
40
41 QNetworkDiskCache stores each url in its own file inside of the
42 cacheDirectory using QDataStream. Files with a text MimeType
43 are compressed using qCompress. Data is written to disk only in insert()
44 and updateMetaData().
45
46 Currently you cannot share the same cache files with more than
47 one disk cache.
48
49 QNetworkDiskCache by default limits the amount of space that the cache will
50 use on the system to 50MB.
51
52 Note you have to set the cache directory before it will work.
53
54 A network disk cache can be enabled by:
55
56 \snippet code/src_network_access_qnetworkdiskcache.cpp 0
57
58 When sending requests, to control the preference of when to use the cache
59 and when to use the network, consider the following:
60
61 \snippet code/src_network_access_qnetworkdiskcache.cpp 1
62
63 To check whether the response came from the cache or from the network, the
64 following can be applied:
65
66 \snippet code/src_network_access_qnetworkdiskcache.cpp 2
67*/
68

96/*!
97 Sets the directory where cached files will be stored to \a cacheDir
98
99 QNetworkDiskCache will create this directory if it does not exists.
100
101 Prepared cache items will be stored in the new cache directory when
102 they are inserted.
103
104 \sa QStandardPaths::CacheLocation
105*/
106void QNetworkDiskCache::setCacheDirectory(const QString &cacheDir)
107{
108#if defined(QNETWORKDISKCACHE_DEBUG)
109 qDebug() << "QNetworkDiskCache::setCacheDirectory()" << cacheDir;
110#endif
111 Q_D(QNetworkDiskCache);
112 if (cacheDir.isEmpty())
113 return;
114 d->cacheDirectory = cacheDir;
115 QDir dir(d->cacheDirectory);
116 d->cacheDirectory = dir.absolutePath();
117 if (!d->cacheDirectory.endsWith(c: u'/'))
118 d->cacheDirectory += u'/';
119
120 d->dataDirectory = d->cacheDirectory + DATA_DIR + QString::number(CACHE_VERSION) + u'/';
121 d->prepareLayout();
122}

142/*!
143 \reimp
144*/
145QIODevice *QNetworkDiskCache::prepare(const QNetworkCacheMetaData &metaData)
146{
147#if defined(QNETWORKDISKCACHE_DEBUG)
148 qDebug() << "QNetworkDiskCache::prepare()" << metaData.url();
149#endif
150 Q_D(QNetworkDiskCache);
151 if (!metaData.isValid() || !metaData.url().isValid() || !metaData.saveToDisk())
152 return nullptr;
153
154 if (d->cacheDirectory.isEmpty()) {
155 qWarning(msg: "QNetworkDiskCache::prepare() The cache directory is not set");
156 return nullptr;
157 }
158
159 const auto headers = metaData.rawHeaders();
160 for (const auto &header : headers) {
161 if (header.first.compare(a: "content-length", cs: Qt::CaseInsensitive) == 0) {
162 const qint64 size = header.second.toLongLong();
163 if (size > (maximumCacheSize() * 3)/4)
164 return nullptr;
165 break;
166 }
167 }
168 std::unique_ptr<QCacheItem> cacheItem = std::make_unique<QCacheItem>();
169 cacheItem->metaData = metaData;
170
171 QIODevice *device = nullptr;
172 if (cacheItem->canCompress()) {
173 cacheItem->data.open(openMode: QBuffer::ReadWrite);
174 device = &(cacheItem->data);
175 } else {
176 QString templateName = d->tmpCacheFileName();
177 QT_TRY {
178 cacheItem->file = new QTemporaryFile(templateName, &cacheItem->data);
179 } QT_CATCH(...) {
180 cacheItem->file = nullptr;
181 }
182 if (!cacheItem->file || !cacheItem->file->open()) {
183 qWarning(msg: "QNetworkDiskCache::prepare() unable to open temporary file");
184 cacheItem.reset();
185 return nullptr;
186 }
187 cacheItem->writeHeader(device: cacheItem->file);
188 device = cacheItem->file;
189 }
190 d->inserting[device] = cacheItem.release();
191 return device;
192}

194/*!
195 \reimp
196*/
197void QNetworkDiskCache::insert(QIODevice *device)
198{
199#if defined(QNETWORKDISKCACHE_DEBUG)
200 qDebug() << "QNetworkDiskCache::insert()" << device;
201#endif
202 Q_D(QNetworkDiskCache);
203 const auto it = d->inserting.constFind(key: device);
204 if (Q_UNLIKELY(it == d->inserting.cend())) {
205 qWarning() << "QNetworkDiskCache::insert() called on a device we don't know about" << device;
206 return;
207 }
208
209 d->storeItem(item: it.value());
210 delete it.value();
211 d->inserting.erase(it);
212}

215/*!
216 Create subdirectories and other housekeeping on the filesystem.
217 Prevents too many files from being present in any single directory.
218*/
219void QNetworkDiskCachePrivate::prepareLayout()
220{
221 QDir helper;
222 helper.mkpath(dirPath: cacheDirectory + PREPARED_SLASH);
223
224 //Create directory and subdirectories 0-F
225 helper.mkpath(dirPath: dataDirectory);
226 for (uint i = 0; i < 16 ; i++) {
227 QString str = QString::number(i, base: 16);
228 QString subdir = dataDirectory + str;
229 helper.mkdir(dirName: subdir);
230 }
231}

234void QNetworkDiskCachePrivate::storeItem(QCacheItem *cacheItem)
235{
236 Q_Q(QNetworkDiskCache);
237 Q_ASSERT(cacheItem->metaData.saveToDisk());
238
239 QString fileName = cacheFileName(url: cacheItem->metaData.url());
240 Q_ASSERT(!fileName.isEmpty());
241
242 if (QFile::exists(fileName)) {
243 if (!removeFile(file: fileName)) {
244 qWarning() << "QNetworkDiskCache: couldn't remove the cache file " << fileName;
245 return;
246 }
247 }
248
249 currentCacheSize = q->expire();
250 if (!cacheItem->file) {
251 QString templateName = tmpCacheFileName();
252 cacheItem->file = new QTemporaryFile(templateName, &cacheItem->data);
253 if (cacheItem->file->open()) {
254 cacheItem->writeHeader(device: cacheItem->file);
255 cacheItem->writeCompressedData(device: cacheItem->file);
256 }
257 }
258
259 if (cacheItem->file
260 && cacheItem->file->isOpen()
261 && cacheItem->file->error() == QFile::NoError) {
262 cacheItem->file->setAutoRemove(false);
263 // ### use atomic rename rather then remove & rename
264 if (cacheItem->file->rename(newName: fileName))
265 currentCacheSize += cacheItem->file->size();
266 else
267 cacheItem->file->setAutoRemove(true);
268 }
269 if (cacheItem->metaData.url() == lastItem.metaData.url())
270 lastItem.reset();
271}

273/*!
274 \reimp
275*/
276bool QNetworkDiskCache::remove(const QUrl &url)
277{
278#if defined(QNETWORKDISKCACHE_DEBUG)
279 qDebug() << "QNetworkDiskCache::remove()" << url;
280#endif
281 Q_D(QNetworkDiskCache);
282
283 // remove is also used to cancel insertions, not a common operation
284 for (auto it = d->inserting.cbegin(), end = d->inserting.cend(); it != end; ++it) {
285 QCacheItem *item = it.value();
286 if (item && item->metaData.url() == url) {
287 delete item;
288 d->inserting.erase(it);
289 return true;
290 }
291 }
292
293 if (d->lastItem.metaData.url() == url)
294 d->lastItem.reset();
295 return d->removeFile(file: d->cacheFileName(url));
296}

298/*!
299 Put all of the misc file removing into one function to be extra safe
300 */
301bool QNetworkDiskCachePrivate::removeFile(const QString &file)
302{
303#if defined(QNETWORKDISKCACHE_DEBUG)
304 qDebug() << "QNetworkDiskCache::removFile()" << file;
305#endif
306 if (file.isEmpty())
307 return false;
308 QFileInfo info(file);
309 QString fileName = info.fileName();
310 if (!fileName.endsWith(CACHE_POSTFIX))
311 return false;
312 qint64 size = info.size();
313 if (QFile::remove(fileName: file)) {
314 currentCacheSize -= size;
315 return true;
316 }
317 return false;
318}

339QNetworkCacheMetaData QNetworkDiskCache::fileMetaData(const QString &fileName) const

356/*!
357 \reimp
358*/
359QIODevice *QNetworkDiskCache::data(const QUrl &url)
360{
361#if defined(QNETWORKDISKCACHE_DEBUG)
362 qDebug() << "QNetworkDiskCache::data()" << url;
363#endif
364 Q_D(QNetworkDiskCache);
365 std::unique_ptr<QBuffer> buffer;
366 if (!url.isValid())
367 return nullptr;
368 if (d->lastItem.metaData.url() == url && d->lastItem.data.isOpen()) {
369 buffer.reset(p: new QBuffer);
370 buffer->setData(d->lastItem.data.data());
371 } else {
372 QScopedPointer<QFile> file(new QFile(d->cacheFileName(url)));
373 if (!file->open(flags: QFile::ReadOnly | QIODevice::Unbuffered))
374 return nullptr;
375
376 if (!d->lastItem.read(device: file.data(), readData: true)) {
377 file->close();
378 remove(url);
379 return nullptr;
380 }
381 if (d->lastItem.data.isOpen()) {
382 // compressed
383 buffer.reset(p: new QBuffer);
384 buffer->setData(d->lastItem.data.data());
385 } else {
386 buffer.reset(p: new QBuffer);
387 buffer->setData(file->readAll());
388 }
389 }
390 buffer->open(openMode: QBuffer::ReadOnly);
391 return buffer.release();
392}

394/*!
395 \reimp
396*/
397void QNetworkDiskCache::updateMetaData(const QNetworkCacheMetaData &metaData)
398{
399#if defined(QNETWORKDISKCACHE_DEBUG)
400 qDebug() << "QNetworkDiskCache::updateMetaData()" << metaData.url();
401#endif
402 QUrl url = metaData.url();
403 QIODevice *oldDevice = data(url);
404 if (!oldDevice) {
405#if defined(QNETWORKDISKCACHE_DEBUG)
406 qDebug("QNetworkDiskCache::updateMetaData(), no device!");
407#endif
408 return;
409 }
410
411 QIODevice *newDevice = prepare(metaData);
412 if (!newDevice) {
413#if defined(QNETWORKDISKCACHE_DEBUG)
414 qDebug() << "QNetworkDiskCache::updateMetaData(), no new device!" << url;
415#endif
416 return;
417 }
418 char data[1024];
419 while (!oldDevice->atEnd()) {
420 qint64 s = oldDevice->read(data, maxlen: 1024);
421 newDevice->write(data, len: s);
422 }
423 delete oldDevice;
424 insert(device: newDevice);
425}

454/*!
455 Cleans the cache so that its size is under the maximum cache size.
456 Returns the current size of the cache.
457
458 When the current size of the cache is greater than the maximumCacheSize()
459 older cache files are removed until the total size is less then 90% of
460 maximumCacheSize() starting with the oldest ones first using the file
461 creation date to determine how old a cache file is.
462
463 Subclasses can reimplement this function to change the order that cache
464 files are removed taking into account information in the application
465 knows about that QNetworkDiskCache does not, for example the number of times
466 a cache is accessed.
467
468 \note cacheSize() calls expire if the current cache size is unknown.
469
470 \sa maximumCacheSize(), fileMetaData()
471 */
472qint64 QNetworkDiskCache::expire()
473{
474 Q_D(QNetworkDiskCache);
475 if (d->currentCacheSize >= 0 && d->currentCacheSize < maximumCacheSize())
476 return d->currentCacheSize;
477
478 if (cacheDirectory().isEmpty()) {
479 qWarning(msg: "QNetworkDiskCache::expire() The cache directory is not set");
480 return 0;
481 }
482
483 // close file handle to prevent "in use" error when QFile::remove() is called
484 d->lastItem.reset();
485
486 QDir::Filters filters = QDir::AllDirs | QDir:: Files | QDir::NoDotAndDotDot;
487 QDirIterator it(cacheDirectory(), filters, QDirIterator::Subdirectories);
488
489 QMultiMap<QDateTime, QString> cacheItems;
490 qint64 totalSize = 0;
491 while (it.hasNext()) {
492 QFileInfo info = it.nextFileInfo();
493 QString path = info.filePath();
494 QString fileName = info.fileName();
495 if (fileName.endsWith(CACHE_POSTFIX)) {
496 const QDateTime birthTime = info.fileTime(time: QFile::FileBirthTime);
497 cacheItems.insert(key: birthTime.isValid() ? birthTime
498 : info.fileTime(time: QFile::FileMetadataChangeTime), value: path);
499 totalSize += info.size();
500 }
501 }
502
503 [[maybe_unused]] int removedFiles = 0; // used under QNETWORKDISKCACHE_DEBUG
504 qint64 goal = (maximumCacheSize() * 9) / 10;
505 QMultiMap<QDateTime, QString>::const_iterator i = cacheItems.constBegin();
506 while (i != cacheItems.constEnd()) {
507 if (totalSize < goal)
508 break;
509 QString name = i.value();
510 QFile file(name);
511
512 if (name.contains(PREPARED_SLASH)) {
513 for (QCacheItem *item : std::as_const(t&: d->inserting)) {
514 if (item && item->file && item->file->fileName() == name) {
515 delete item->file;
516 item->file = nullptr;
517 break;
518 }
519 }
520 }
521
522 qint64 size = file.size();
523 file.remove();
524 totalSize -= size;
525 ++removedFiles;
526 ++i;
527 }
528#if defined(QNETWORKDISKCACHE_DEBUG)
529 if (removedFiles > 0) {
530 qDebug() << "QNetworkDiskCache::expire()"
531 << "Removed:" << removedFiles
532 << "Kept:" << cacheItems.count() - removedFiles;
533 }
534#endif
535 return totalSize;
536}

553/*!
554 Given a URL, generates a unique enough filename (and subdirectory)
555 */
556QString QNetworkDiskCachePrivate::uniqueFileName(const QUrl &url)
557{
558 QUrl cleanUrl = url;
559 cleanUrl.setPassword(password: QString());
560 cleanUrl.setFragment(fragment: QString());
561
562 const QByteArray hash = QCryptographicHash::hash(data: cleanUrl.toEncoded(), method: QCryptographicHash::Sha1);
563 // convert sha1 to base36 form and return first 8 bytes for use as string
564 const QByteArray id = QByteArray::number(*(qlonglong*)hash.data(), base: 36).left(len: 8);
565 // generates <one-char subdir>/<8-char filename.d>
566 uint code = (uint)id.at(i: id.size()-1) % 16;
567 QString pathFragment = QString::number(code, base: 16) + u'/' + QLatin1StringView(id) + CACHE_POSTFIX;
568
569 return pathFragment;
570}

590/*!
591 We compress small text and JavaScript files.
592 */
593bool QCacheItem::canCompress() const
594{
595 bool sizeOk = false;
596 bool typeOk = false;
597 const auto headers = metaData.rawHeaders();
598 for (const auto &header : headers) {
599 if (header.first.compare(a: "content-length", cs: Qt::CaseInsensitive) == 0) {
600 qint64 size = header.second.toLongLong();
601 if (size > MAX_COMPRESSION_SIZE)
602 return false;
603 else
604 sizeOk = true;
605 }
606
607 if (header.first.compare(a: "content-type", cs: Qt::CaseInsensitive) == 0) {
608 QByteArray type = header.second;
609 if (type.startsWith(bv: "text/")
610 || (type.startsWith(bv: "application/")
611 && (type.endsWith(bv: "javascript") || type.endsWith(bv: "ecmascript"))))
612 typeOk = true;
613 else
614 return false;
615 }
616 if (sizeOk && typeOk)
617 return true;
618 }
619 return false;
620}

628void QCacheItem::writeHeader(QFile *device) const
629{
630 QDataStream out(device);
631
632 out << qint32(CacheMagic);
633 out << qint32(CurrentCacheVersion);
634 out << static_cast<qint32>(out.version());
635 out << metaData;
636 bool compressed = canCompress();
637 out << compressed;
638}

642 QDataStream out(device);

644 out << qCompress(data: data.data());

647/*!
648 Returns \c false if the file is a cache file,
649 but is an older version and should be removed otherwise true.
650 */
651bool QCacheItem::read(QFile *device, bool readData)
652{
653 reset();
654
655 QDataStream in(device);
656
657 qint32 marker;
658 qint32 v;
659 in >> marker;
660 in >> v;
661 if (marker != CacheMagic)
662 return true;
663
664 // If the cache magic is correct, but the version is not we should remove it
665 if (v != CurrentCacheVersion)
666 return false;
667
668 qint32 streamVersion;
669 in >> streamVersion;
670 // Default stream version is also the highest we can handle
671 if (streamVersion > in.version())
672 return false;
673 in.setVersion(streamVersion);
674
675 bool compressed;
676 QByteArray dataBA;
677 in >> metaData;
678 in >> compressed;
679 if (readData && compressed) {
680 in >> dataBA;
681 data.setData(qUncompress(data: dataBA));
682 data.open(openMode: QBuffer::ReadOnly);
683 }
684
685 // quick and dirty check if metadata's URL field and the file's name are in synch
686 QString expectedFilename = QNetworkDiskCachePrivate::uniqueFileName(url: metaData.url());
687 if (!device->fileName().endsWith(s: expectedFilename))
688 return false;
689
690 return metaData.isValid();
691}
