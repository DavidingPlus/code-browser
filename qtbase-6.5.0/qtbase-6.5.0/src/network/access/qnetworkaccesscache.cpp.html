33struct QNetworkAccessCache::Node
34{
35 QDeadlineTimer timer;
36 QByteArray key;
37
38 Node *previous = nullptr; // "previous" nodes expire "previous"ly (before us)
39 Node *next = nullptr; // "next" nodes expire "next" (after us)
40 CacheableObject *object = nullptr;
41
42 int useCount = 0;
43};

75void QNetworkAccessCache::clear()
76{
77 NodeHash hashCopy = hash;
78 hash.clear();
79
80 // remove all entries
81 NodeHash::Iterator it = hashCopy.begin();
82 NodeHash::Iterator end = hashCopy.end();
83 for ( ; it != end; ++it) {
84 (*it)->object->key.clear();
85 (*it)->object->dispose();
86 delete (*it);
87 }
88
89 // now delete:
90 hashCopy.clear();
91
92 timer.stop();
93
94 firstExpiringNode = lastExpiringNode = nullptr;
95}

97/*!
98 Appends the entry given by \a key to the end of the linked list.
99 (i.e., makes it the newest entry)
100 */
101void QNetworkAccessCache::linkEntry(const QByteArray &key)
102{
103 Node * const node = hash.value(key);
104 if (!node)
105 return;
106
107 Q_ASSERT(node != firstExpiringNode && node != lastExpiringNode);
108 Q_ASSERT(node->previous == nullptr && node->next == nullptr);
109 Q_ASSERT(node->useCount == 0);
110
111
112 node->timer.setPreciseRemainingTime(secs: node->object->expiryTimeoutSeconds);
113#ifdef DEBUG_ACCESSCACHE
114 qDebug() << "QNetworkAccessCache case trying to insert=" << QString::fromUtf8(key)
115 << node->timer.remainingTime() << "milliseconds";
116 Node *current = lastExpiringNode;
117 while (current) {
118 qDebug() << "QNetworkAccessCache item=" << QString::fromUtf8(current->key)
119 << current->timer.remainingTime() << "milliseconds"
120 << (current == lastExpiringNode ? "[last to expire]" : "")
121 << (current == firstExpiringNode ? "[first to expire]" : "");
122 current = current->previous;
123 }
124#endif
125
126 if (lastExpiringNode) {
127 Q_ASSERT(lastExpiringNode->next == nullptr);
128 if (lastExpiringNode->timer < node->timer) {
129 // Insert as new last-to-expire node.
130 node->previous = lastExpiringNode;
131 lastExpiringNode->next = node;
132 lastExpiringNode = node;
133 } else {
134 // Insert in a sorted way, as different nodes might have had different expiryTimeoutSeconds set.
135 Node *current = lastExpiringNode;
136 while (current->previous != nullptr && current->previous->timer >= node->timer)
137 current = current->previous;
138 node->previous = current->previous;
139 if (node->previous)
140 node->previous->next = node;
141 node->next = current;
142 current->previous = node;
143 if (node->previous == nullptr)
144 firstExpiringNode = node;
145 }
146 } else {
147 // no current last-to-expire node
148 lastExpiringNode = node;
149 }
150 if (!firstExpiringNode) {
151 // there are no entries, so this is the next-to-expire too
152 firstExpiringNode = node;
153 }
154 Q_ASSERT(firstExpiringNode->previous == nullptr);
155 Q_ASSERT(lastExpiringNode->next == nullptr);
156}

158/*!
159 Removes the entry pointed by \a key from the linked list.
160 Returns \c true if the entry removed was the next to expire.
161 */
162bool QNetworkAccessCache::unlinkEntry(const QByteArray &key)
163{
164 Node * const node = hash.value(key);
165 if (!node)
166 return false;
167
168 bool wasFirst = false;
169 if (node == firstExpiringNode) {
170 firstExpiringNode = node->next;
171 wasFirst = true;
172 }
173 if (node == lastExpiringNode)
174 lastExpiringNode = node->previous;
175 if (node->previous)
176 node->previous->next = node->next;
177 if (node->next)
178 node->next->previous = node->previous;
179
180 node->next = node->previous = nullptr;
181 return wasFirst;
182}

184void QNetworkAccessCache::updateTimer()
185{
186 timer.stop();
187
188 if (!firstExpiringNode)
189 return;
190
191 qint64 interval = firstExpiringNode->timer.remainingTime();
192 if (interval <= 0) {
193 interval = 0;
194 }
195
196 // Plus 10 msec so we don't spam timer events if date comparisons are too fuzzy.
197 // This code used to do (broken) rounding, but for ConnectionCacheExpiryTimeoutSecondsAttribute
198 // to work we cannot do this.
199 // See discussion in https://codereview.qt-project.org/c/qt/qtbase/+/337464
200 timer.start(msec: interval + 10, obj: this);
201}

203bool QNetworkAccessCache::emitEntryReady(Node *node, QObject *target, const char *member)
204{
205 if (!connect(sender: this, SIGNAL(entryReady(QNetworkAccessCache::CacheableObject*)),
206 receiver: target, member, Qt::QueuedConnection))
207 return false;
208
209 emit entryReady(node->object);
210 disconnect(SIGNAL(entryReady(QNetworkAccessCache::CacheableObject*)));
211
212 return true;
213}

215void QNetworkAccessCache::timerEvent(QTimerEvent *)
216{
217 while (firstExpiringNode && firstExpiringNode->timer.hasExpired()) {
218 Node *next = firstExpiringNode->next;
219 firstExpiringNode->object->dispose();
220 hash.remove(key: firstExpiringNode->key); // `firstExpiringNode` gets deleted
221 delete firstExpiringNode;
222 firstExpiringNode = next;
223 }
224
225 // fixup the list
226 if (firstExpiringNode)
227 firstExpiringNode->previous = nullptr;
228 else
229 lastExpiringNode = nullptr;
230
231 updateTimer();
232}

234void QNetworkAccessCache::addEntry(const QByteArray &key, CacheableObject *entry, qint64 connectionCacheExpiryTimeoutSeconds)
235{
236 Q_ASSERT(!key.isEmpty());
237
238 if (unlinkEntry(key))
239 updateTimer();
240
241 Node *node = hash.value(key);
242 if (!node) {
243 node = new Node;
244 hash.insert(key, value: node);
245 }
246
247 if (node->useCount)
248 qWarning(msg: "QNetworkAccessCache::addEntry: overriding active cache entry '%s'", key.constData());
249 if (node->object)
250 node->object->dispose();
251 node->object = entry;
252 node->object->key = key;
253 if (connectionCacheExpiryTimeoutSeconds > -1) {
254 node->object->expiryTimeoutSeconds = connectionCacheExpiryTimeoutSeconds; // via ConnectionCacheExpiryTimeoutSecondsAttribute
255 } else {
256 node->object->expiryTimeoutSeconds = ExpiryTime;
257 }
258 node->key = key;
259 node->useCount = 1;
260
261 // It gets only put into the expiry list in linkEntry (from releaseEntry), when it is not used anymore.
262}

269QNetworkAccessCache::CacheableObject *QNetworkAccessCache::requestEntryNow(const QByteArray &key)
270{
271 Node *node = hash.value(key);
272 if (!node)
273 return nullptr;
274
275 if (node->useCount > 0) {
276 if (node->object->shareable) {
277 ++node->useCount;
278 return node->object;
279 }
280
281 // object in use and not shareable
282 return nullptr;
283 }
284
285 // entry not in use, let the caller have it
286 bool wasNext = unlinkEntry(key);
287 ++node->useCount;
288
289 if (wasNext)
290 updateTimer();
291 return node->object;
292}

294void QNetworkAccessCache::releaseEntry(const QByteArray &key)
295{
296 Node *node = hash.value(key);
297 if (!node) {
298 qWarning(msg: "QNetworkAccessCache::releaseEntry: trying to release key '%s' that is not in cache", key.constData());
299 return;
300 }
301
302 Q_ASSERT(node->useCount > 0);
303
304 if (!--node->useCount) {
305 // no objects waiting; add it back to the expiry list
306 if (node->object->expires)
307 linkEntry(key);
308
309 if (firstExpiringNode == node)
310 updateTimer();
311 }
312}
313
314void QNetworkAccessCache::removeEntry(const QByteArray &key)
315{
316 Node *node = hash.value(key);
317 if (!node) {
318 qWarning(msg: "QNetworkAccessCache::removeEntry: trying to remove key '%s' that is not in cache", key.constData());
319 return;
320 }
321
322 if (unlinkEntry(key))
323 updateTimer();
324 if (node->useCount > 1)
325 qWarning(msg: "QNetworkAccessCache::removeEntry: removing active cache entry '%s'",
326 key.constData());
327
328 node->object->key.clear();
329 hash.remove(key: node->key);
330 delete node;
331}
