29class QFile;
30
31class QCacheItem
32{
33public:
34 QCacheItem() : file(nullptr)
35 {
36 }
37 ~QCacheItem()
38 {
39 reset();
40 }
41
42 QNetworkCacheMetaData metaData;
43 QBuffer data;
44 QTemporaryFile *file;
45 inline qint64 size() const
46 { return file ? file->size() : data.size(); }
47
48 inline void reset() {
49 metaData = QNetworkCacheMetaData();
50 data.close();
51 delete file;
52 file = nullptr;
53 }
54 void writeHeader(QFile *device) const;
55 void writeCompressedData(QFile *device) const;
56 bool read(QFile *device, bool readData);
57
58 bool canCompress() const;
59};
