40static inline bool isSeparator(char c)
41{
42 static const char separators[] = "()<>@,;:\\\"/[]?={}";
43 return isLWS(c) || strchr(s: separators, c: c) != nullptr;
44}

47static QHash<QByteArray, QByteArray> parseHttpOptionHeader(const QByteArray &header)
48{
49 // The HTTP header is of the form:
50 // header = #1(directives)
51 // directives = token | value-directive
52 // value-directive = token "=" (token | quoted-string)
53 QHash<QByteArray, QByteArray> result;
54
55 int pos = 0;
56 while (true) {
57 // skip spaces
58 pos = nextNonWhitespace(text: header, from: pos);
59 if (pos == header.size())
60 return result; // end of parsing
61
62 // pos points to a non-whitespace
63 int comma = header.indexOf(c: ',', from: pos);
64 int equal = header.indexOf(c: '=', from: pos);
65 if (comma == pos || equal == pos)
66 // huh? Broken header.
67 return result;
68
69 // The key name is delimited by either a comma, an equal sign or the end
70 // of the header, whichever comes first
71 int end = comma;
72 if (end == -1)
73 end = header.size();
74 if (equal != -1 && end > equal)
75 end = equal; // equal sign comes before comma/end
76 QByteArray key = QByteArray(header.constData() + pos, end - pos).trimmed().toLower();
77 pos = end + 1;
78
79 if (uint(equal) < uint(comma)) {
80 // case: token "=" (token | quoted-string)
81 // skip spaces
82 pos = nextNonWhitespace(text: header, from: pos);
83 if (pos == header.size())
84 // huh? Broken header
85 return result;
86
87 QByteArray value;
88 value.reserve(asize: header.size() - pos);
89 if (header.at(i: pos) == '"') {
90 // case: quoted-string
91 // quoted-string = ( <"> *(qdtext | quoted-pair ) <"> )
92 // qdtext = <any TEXT except <">>
93 // quoted-pair = "\" CHAR
94 ++pos;
95 while (pos < header.size()) {
96 char c = header.at(i: pos);
97 if (c == '"') {
98 // end of quoted text
99 break;
100 } else if (c == '\\') {
101 ++pos;
102 if (pos >= header.size())
103 // broken header
104 return result;
105 c = header.at(i: pos);
106 }
107
108 value += c;
109 ++pos;
110 }
111 } else {
112 // case: token
113 while (pos < header.size()) {
114 char c = header.at(i: pos);
115 if (isSeparator(c))
116 break;
117 value += c;
118 ++pos;
119 }
120 }
121
122 result.insert(key, value);
123
124 // find the comma now:
125 comma = header.indexOf(c: ',', from: pos);
126 if (comma == -1)
127 return result; // end of parsing
128 pos = comma + 1;
129 } else {
130 // case: token
131 // key is already set
132 result.insert(key, value: QByteArray());
133 }
134 }

137QNetworkReplyHttpImpl::QNetworkReplyHttpImpl(QNetworkAccessManager* const manager,
138 const QNetworkRequest& request,
139 QNetworkAccessManager::Operation& operation,
140 QIODevice* outgoingData)
141 : QNetworkReply(*new QNetworkReplyHttpImplPrivate, manager)
142{
143 Q_D(QNetworkReplyHttpImpl);
144 Q_ASSERT(manager);
145 d->manager = manager;
146 d->managerPrivate = manager->d_func();
147 d->request = request;
148 d->originalRequest = request;
149 d->operation = operation;
150 d->outgoingData = outgoingData;
151 d->url = request.url();
152#ifndef QT_NO_SSL
153 if (request.url().scheme() == "https"_L1)
154 d->sslConfiguration.reset(new QSslConfiguration(request.sslConfiguration()));
155#endif
156
157 // FIXME Later maybe set to Unbuffered, especially if it is zerocopy or from cache?
158 QIODevice::open(mode: QIODevice::ReadOnly);
159
160
161 // Internal code that does a HTTP reply for the synchronous Ajax
162 // in Qt WebKit.
163 QVariant synchronousHttpAttribute = request.attribute(
164 code: static_cast<QNetworkRequest::Attribute>(QNetworkRequest::SynchronousRequestAttribute));
165 if (synchronousHttpAttribute.isValid()) {
166 d->synchronous = synchronousHttpAttribute.toBool();
167 if (d->synchronous && outgoingData) {
168 // The synchronous HTTP is a corner case, we will put all upload data in one big QByteArray in the outgoingDataBuffer.
169 // Yes, this is not the most efficient thing to do, but on the other hand synchronous XHR needs to die anyway.
170 d->outgoingDataBuffer = std::make_shared<QRingBuffer>();
171 qint64 previousDataSize = 0;
172 do {
173 previousDataSize = d->outgoingDataBuffer->size();
174 d->outgoingDataBuffer->append(qba: d->outgoingData->readAll());
175 } while (d->outgoingDataBuffer->size() != previousDataSize);
176 d->_q_startOperation();
177 return;
178 }
179 }
180
181
182 if (outgoingData) {
183 // there is data to be uploaded, e.g. HTTP POST.
184
185 if (!d->outgoingData->isSequential()) {
186 // fixed size non-sequential (random-access)
187 // just start the operation
188 QMetaObject::invokeMethod(obj: this, member: "_q_startOperation", c: Qt::QueuedConnection);
189 // FIXME make direct call?
190 } else {
191 bool bufferingDisallowed =
192 request.attribute(code: QNetworkRequest::DoNotBufferUploadDataAttribute,
193 defaultValue: false).toBool();
194
195 if (bufferingDisallowed) {
196 // if a valid content-length header for the request was supplied, we can disable buffering
197 // if not, we will buffer anyway
198 if (request.header(header: QNetworkRequest::ContentLengthHeader).isValid()) {
199 QMetaObject::invokeMethod(obj: this, member: "_q_startOperation", c: Qt::QueuedConnection);
200 // FIXME make direct call?
201 } else {
202 d->state = d->Buffering;
203 QMetaObject::invokeMethod(obj: this, member: "_q_bufferOutgoingData", c: Qt::QueuedConnection);
204 }
205 } else {
206 // _q_startOperation will be called when the buffering has finished.
207 d->state = d->Buffering;
208 QMetaObject::invokeMethod(obj: this, member: "_q_bufferOutgoingData", c: Qt::QueuedConnection);
209 }
210 }
211 } else {
212 // No outgoing data (POST, ..)
213 d->_q_startOperation();
214 }
215}

223void QNetworkReplyHttpImpl::close()
224{
225 Q_D(QNetworkReplyHttpImpl);
226
227 if (d->state == QNetworkReplyPrivate::Aborted ||
228 d->state == QNetworkReplyPrivate::Finished)
229 return;
230
231 // According to the documentation close only stops the download
232 // by closing we can ignore the download part and continue uploading.
233 QNetworkReply::close();
234
235 // call finished which will emit signals
236 // FIXME shouldn't this be emitted Queued?
237 d->error(code: OperationCanceledError, errorString: tr(s: "Operation canceled"));
238 d->finished();

262qint64 QNetworkReplyHttpImpl::bytesAvailable() const
263{
264 Q_D(const QNetworkReplyHttpImpl);
265
266 // if we load from cache device
267 if (d->cacheLoadDevice) {
268 return QNetworkReply::bytesAvailable() + d->cacheLoadDevice->bytesAvailable();
269 }
270
271 // zerocopy buffer
272 if (d->downloadZerocopyBuffer) {
273 return QNetworkReply::bytesAvailable() + d->downloadBufferCurrentSize - d->downloadBufferReadPosition;
274 }
275
276 if (d->decompressHelper.isValid()) {
277 if (d->decompressHelper.isCountingBytes())
278 return QNetworkReply::bytesAvailable() + d->decompressHelper.uncompressedSize();
279 if (d->decompressHelper.hasData())
280 return QNetworkReply::bytesAvailable() + 1;
281 }
282
283 // normal buffer
284 return QNetworkReply::bytesAvailable();

300qint64 QNetworkReplyHttpImpl::readData(char* data, qint64 maxlen)
301{
302 Q_D(QNetworkReplyHttpImpl);
303
304 // cacheload device
305 if (d->cacheLoadDevice) {
306 // FIXME bytesdownloaded, position etc?
307
308 qint64 ret = d->cacheLoadDevice->read(data, maxlen);
309 return ret;
310 }
311
312 // zerocopy buffer
313 if (d->downloadZerocopyBuffer) {
314 // FIXME bytesdownloaded, position etc?
315
316 qint64 howMuch = qMin(a: maxlen, b: (d->downloadBufferCurrentSize - d->downloadBufferReadPosition));
317 memcpy(dest: data, src: d->downloadZerocopyBuffer + d->downloadBufferReadPosition, n: howMuch);
318 d->downloadBufferReadPosition += howMuch;
319 return howMuch;
320
321 }
322
323 if (d->decompressHelper.isValid() && (d->decompressHelper.hasData() || !isFinished())) {
324 if (maxlen == 0 || !d->decompressHelper.hasData())
325 return 0;
326 const qint64 bytesRead = d->decompressHelper.read(data, maxSize: maxlen);
327 if (!d->decompressHelper.isValid()) {
328 d->error(code: QNetworkReplyImpl::NetworkError::UnknownContentError,
329 errorString: QCoreApplication::translate(context: "QHttp", key: "Decompression failed: %1")
330 .arg(a: d->decompressHelper.errorString()));
331 d->decompressHelper.clear();
332 return -1;
333 }
334 if (d->cacheSaveDevice) {
335 // Need to write to the cache now that we have the data
336 d->cacheSaveDevice->write(data, len: bytesRead);
337 // ... and if we've read everything then the cache can be closed.
338 if (isFinished() && !d->decompressHelper.hasData())
339 d->completeCacheSave();
340 }
341 // In case of buffer size restriction we need to emit that it has been emptied
342 qint64 wasBuffered = d->bytesBuffered;
343 d->bytesBuffered = 0;
344 if (readBufferSize())
345 emit readBufferFreed(size: wasBuffered);
346 return bytesRead;
347 }
348
349 // normal buffer
350 if (d->state == d->Finished || d->state == d->Aborted)
351 return -1;
352
353 qint64 wasBuffered = d->bytesBuffered;
354 d->bytesBuffered = 0;
355 if (readBufferSize())
356 emit readBufferFreed(size: wasBuffered);
357 return 0;
358}

464/*
465 For a given httpRequest
466 1) If AlwaysNetwork, return
467 2) If we have a cache entry for this url populate headers so the server can return 304
468 3) Calculate if response_is_fresh and if so send the cache and set loadedFromCache to true
469 */
470bool QNetworkReplyHttpImplPrivate::loadFromCacheIfAllowed(QHttpNetworkRequest &httpRequest)
471{
472 QNetworkRequest::CacheLoadControl CacheLoadControlAttribute =
473 (QNetworkRequest::CacheLoadControl)request.attribute(code: QNetworkRequest::CacheLoadControlAttribute, defaultValue: QNetworkRequest::PreferNetwork).toInt();
474 if (CacheLoadControlAttribute == QNetworkRequest::AlwaysNetwork) {
475 // If the request does not already specify preferred cache-control
476 // force reload from the network and tell any caching proxy servers to reload too
477 if (!request.rawHeaderList().contains(t: "Cache-Control")) {
478 httpRequest.setHeaderField(name: "Cache-Control", data: "no-cache");
479 httpRequest.setHeaderField(name: "Pragma", data: "no-cache");
480 }
481 return false;
482 }
483
484 // The disk cache API does not currently support partial content retrieval.
485 // That is why we don't use the disk cache for any such requests.
486 if (request.hasRawHeader(headerName: "Range"))
487 return false;
488
489 QAbstractNetworkCache *nc = managerPrivate->networkCache;
490 if (!nc)
491 return false; // no local cache
492
493 QNetworkCacheMetaData metaData = nc->metaData(url: httpRequest.url());
494 if (!metaData.isValid())
495 return false; // not in cache
496
497 if (!metaData.saveToDisk())
498 return false;
499
500 QNetworkHeadersPrivate cacheHeaders;
501 QNetworkHeadersPrivate::RawHeadersList::ConstIterator it;
502 cacheHeaders.setAllRawHeaders(metaData.rawHeaders());
503
504 it = cacheHeaders.findRawHeader(key: "etag");
505 if (it != cacheHeaders.rawHeaders.constEnd())
506 httpRequest.setHeaderField(name: "If-None-Match", data: it->second);
507
508 QDateTime lastModified = metaData.lastModified();
509 if (lastModified.isValid())
510 httpRequest.setHeaderField(name: "If-Modified-Since", data: QNetworkHeadersPrivate::toHttpDate(dt: lastModified));
511
512 it = cacheHeaders.findRawHeader(key: "Cache-Control");
513 if (it != cacheHeaders.rawHeaders.constEnd()) {
514 QHash<QByteArray, QByteArray> cacheControl = parseHttpOptionHeader(header: it->second);
515 if (cacheControl.contains(key: "must-revalidate"))
516 return false;
517 if (cacheControl.contains(key: "no-cache"))
518 return false;
519 }
520
521 QDateTime currentDateTime = QDateTime::currentDateTimeUtc();
522 QDateTime expirationDate = metaData.expirationDate();
523
524 bool response_is_fresh;
525 if (!expirationDate.isValid()) {
526 /*
527 * age_value
528 * is the value of Age: header received by the cache with
529 * this response.
530 * date_value
531 * is the value of the origin server's Date: header
532 * request_time
533 * is the (local) time when the cache made the request
534 * that resulted in this cached response
535 * response_time
536 * is the (local) time when the cache received the
537 * response
538 * now
539 * is the current (local) time
540 */
541 qint64 age_value = 0;
542 it = cacheHeaders.findRawHeader(key: "age");
543 if (it != cacheHeaders.rawHeaders.constEnd())
544 age_value = it->second.toLongLong();
545
546 QDateTime dateHeader;
547 qint64 date_value = 0;
548 it = cacheHeaders.findRawHeader(key: "date");
549 if (it != cacheHeaders.rawHeaders.constEnd()) {
550 dateHeader = QNetworkHeadersPrivate::fromHttpDate(value: it->second);
551 date_value = dateHeader.toSecsSinceEpoch();
552 }
553
554 qint64 now = currentDateTime.toSecsSinceEpoch();
555 qint64 request_time = now;
556 qint64 response_time = now;
557
558 // Algorithm from RFC 2616 section 13.2.3
559 qint64 apparent_age = qMax<qint64>(a: 0, b: response_time - date_value);
560 qint64 corrected_received_age = qMax(a: apparent_age, b: age_value);
561 qint64 response_delay = response_time - request_time;
562 qint64 corrected_initial_age = corrected_received_age + response_delay;
563 qint64 resident_time = now - response_time;
564 qint64 current_age = corrected_initial_age + resident_time;
565
566 qint64 freshness_lifetime = 0;
567
568 // RFC 2616 13.2.4 Expiration Calculations
569 if (lastModified.isValid() && dateHeader.isValid()) {
570 qint64 diff = lastModified.secsTo(dateHeader);
571 freshness_lifetime = diff / 10;
572 if (httpRequest.headerField(name: "Warning").isEmpty()) {
573 QDateTime dt = currentDateTime.addSecs(secs: current_age);
574 if (currentDateTime.daysTo(dt) > 1)
575 httpRequest.setHeaderField(name: "Warning", data: "113");
576 }
577 }
578
579 // the cache-saving code below sets the freshness_lifetime with (dateHeader - last_modified) / 10
580 // if "last-modified" is present, or to Expires otherwise
581 response_is_fresh = (freshness_lifetime > current_age);
582 } else {
583 // expiration date was calculated earlier (e.g. when storing object to the cache)
584 response_is_fresh = currentDateTime.secsTo(expirationDate) >= 0;
585 }
586
587 if (!response_is_fresh)
588 return false;
589
590#if defined(QNETWORKACCESSHTTPBACKEND_DEBUG)
591 qDebug() << "response_is_fresh" << CacheLoadControlAttribute;
592#endif
593 return sendCacheContents(metaData);
594}

609void QNetworkReplyHttpImplPrivate::postRequest(const QNetworkRequest &newHttpRequest)
610{
611 Q_Q(QNetworkReplyHttpImpl);
612
613 QThread *thread = nullptr;
614 if (synchronous) {
615 // A synchronous HTTP request uses its own thread
616 thread = new QThread();
617 thread->setObjectName(QStringLiteral("Qt HTTP synchronous thread"));
618 QObject::connect(sender: thread, SIGNAL(finished()), receiver: thread, SLOT(deleteLater()));
619 thread->start();
620 } else {
621 // We use the manager-global thread.
622 // At some point we could switch to having multiple threads if it makes sense.
623 thread = managerPrivate->createThread();
624 }
625
626 QUrl url = newHttpRequest.url();
627 httpRequest.setUrl(url);
628 httpRequest.setRedirectCount(newHttpRequest.maximumRedirectsAllowed());
629
630 QString scheme = url.scheme();
631 bool ssl = (scheme == "https"_L1 || scheme == "preconnect-https"_L1);
632 q->setAttribute(code: QNetworkRequest::ConnectionEncryptedAttribute, value: ssl);
633 httpRequest.setSsl(ssl);
634
635 bool preConnect = (scheme == "preconnect-http"_L1 || scheme == "preconnect-https"_L1);
636 httpRequest.setPreConnect(preConnect);
637
638#ifndef QT_NO_NETWORKPROXY
639 QNetworkProxy transparentProxy, cacheProxy;
640
641 // FIXME the proxy stuff should be done in the HTTP thread
642 const auto proxies = managerPrivate->queryProxy(query: QNetworkProxyQuery(newHttpRequest.url()));
643 for (const QNetworkProxy &p : proxies) {
644 // use the first proxy that works
645 // for non-encrypted connections, any transparent or HTTP proxy
646 // for encrypted, only transparent proxies
647 if (!ssl
648 && (p.capabilities() & QNetworkProxy::CachingCapability)
649 && (p.type() == QNetworkProxy::HttpProxy ||
650 p.type() == QNetworkProxy::HttpCachingProxy)) {
651 cacheProxy = p;
652 transparentProxy = QNetworkProxy::NoProxy;
653 break;
654 }
655 if (p.isTransparentProxy()) {
656 transparentProxy = p;
657 cacheProxy = QNetworkProxy::NoProxy;
658 break;
659 }
660 }
661
662 // check if at least one of the proxies
663 if (transparentProxy.type() == QNetworkProxy::DefaultProxy &&
664 cacheProxy.type() == QNetworkProxy::DefaultProxy) {
665 // unsuitable proxies
666 QMetaObject::invokeMethod(obj: q, member: "_q_error", c: synchronous ? Qt::DirectConnection : Qt::QueuedConnection,
667 Q_ARG(QNetworkReply::NetworkError, QNetworkReply::ProxyNotFoundError),
668 Q_ARG(QString, QNetworkReplyHttpImpl::tr("No suitable proxy found")));
669 QMetaObject::invokeMethod(obj: q, member: "_q_finished", c: synchronous ? Qt::DirectConnection : Qt::QueuedConnection);
670 return;
671 }
672#endif
673
674 auto redirectPolicy = QNetworkRequest::NoLessSafeRedirectPolicy;
675 const QVariant value = newHttpRequest.attribute(code: QNetworkRequest::RedirectPolicyAttribute);
676 if (value.isValid())
677 redirectPolicy = qvariant_cast<QNetworkRequest::RedirectPolicy>(v: value);
678
679 httpRequest.setRedirectPolicy(redirectPolicy);
680
681 httpRequest.setPriority(convert(prio: newHttpRequest.priority()));
682 loadingFromCache = false;
683
684 switch (operation) {
685 case QNetworkAccessManager::GetOperation:
686 httpRequest.setOperation(QHttpNetworkRequest::Get);
687 if (loadFromCacheIfAllowed(httpRequest))
688 return; // no need to send the request! :)
689 break;
690
691 case QNetworkAccessManager::HeadOperation:
692 httpRequest.setOperation(QHttpNetworkRequest::Head);
693 if (loadFromCacheIfAllowed(httpRequest))
694 return; // no need to send the request! :)
695 break;
696
697 case QNetworkAccessManager::PostOperation:
698 invalidateCache();
699 httpRequest.setOperation(QHttpNetworkRequest::Post);
700 createUploadByteDevice();
701 break;
702
703 case QNetworkAccessManager::PutOperation:
704 invalidateCache();

1032void QNetworkReplyHttpImplPrivate::replyDownloadData(QByteArray d)
1033{
1034 Q_Q(QNetworkReplyHttpImpl);
1035
1036 // If we're closed just ignore this data
1037 if (!q->isOpen())
1038 return;
1039
1040 // cache this, we need it later and it's invalidated when dealing with compressed data
1041 auto dataSize = d.size();
1042
1043 if (cacheEnabled && isCachingAllowed() && !cacheSaveDevice)
1044 initCacheSaveDevice();
1045
1046 if (decompressHelper.isValid()) {
1047 qint64 uncompressedBefore = -1;
1048 if (decompressHelper.isCountingBytes())
1049 uncompressedBefore = decompressHelper.uncompressedSize();
1050
1051 decompressHelper.feed(data: std::move(d));
1052
1053 if (!decompressHelper.isValid()) {
1054 error(code: QNetworkReplyImpl::NetworkError::UnknownContentError,
1055 errorString: QCoreApplication::translate(context: "QHttp", key: "Decompression failed: %1")
1056 .arg(a: decompressHelper.errorString()));
1057 decompressHelper.clear();
1058 return;
1059 }
1060
1061 if (!isHttpRedirectResponse()) {
1062 if (decompressHelper.isCountingBytes())
1063 bytesDownloaded += (decompressHelper.uncompressedSize() - uncompressedBefore);
1064 setupTransferTimeout();
1065 }
1066
1067 if (synchronous) {
1068 d = QByteArray();
1069 const qsizetype increments = 16 * 1024;
1070 qint64 bytesRead = 0;
1071 while (decompressHelper.hasData()) {
1072 quint64 nextSize = quint64(d.size()) + quint64(increments);
1073 if (nextSize > quint64(std::numeric_limits<QByteArray::size_type>::max())) {
1074 error(code: QNetworkReplyImpl::NetworkError::UnknownContentError,
1075 errorString: QCoreApplication::translate(context: "QHttp",
1076 key: "Data downloaded is too large to store"));
1077 decompressHelper.clear();
1078 return;
1079 }
1080 d.resize(size: nextSize);
1081 bytesRead += decompressHelper.read(data: d.data() + bytesRead, maxSize: increments);
1082 if (!decompressHelper.isValid()) {
1083 error(code: QNetworkReplyImpl::NetworkError::UnknownContentError,
1084 errorString: QCoreApplication::translate(context: "QHttp", key: "Decompression failed: %1")
1085 .arg(a: decompressHelper.errorString()));
1086 decompressHelper.clear();
1087 return;
1088 }
1089 }
1090 d.resize(size: bytesRead);
1091 // we're synchronous so we're not calling this function again; reset the decompressHelper
1092 decompressHelper.clear();
1093 }
1094 }
1095
1096 // This is going to look a little strange. When downloading data while a
1097 // HTTP redirect is happening (and enabled), we write the redirect
1098 // response to the cache. However, we do not append it to our internal
1099 // buffer as that will contain the response data only for the final
1100 // response
1101 // Note: For compressed data this is done in readData()
1102 if (cacheSaveDevice && !decompressHelper.isValid()) {
1103 cacheSaveDevice->write(data: d);
1104 }
1105
1106 // if decompressHelper is valid then we have compressed data, and this is handled above
1107 if (!decompressHelper.isValid() && !isHttpRedirectResponse()) {
1108 buffer.append(qba: d);
1109 bytesDownloaded += dataSize;
1110 setupTransferTimeout();
1111 }
1112 bytesBuffered += dataSize;
1113
1114 int pendingSignals = pendingDownloadDataEmissions->fetchAndSubAcquire(valueToAdd: 1) - 1;
1115 if (pendingSignals > 0) {
1116 // Some more signal emissions to this slot are pending.
1117 // Instead of writing the downstream data, we wait
1118 // and do it in the next call we get
1119 // (signal comppression)
1120 return;
1121 }
1122
1123 if (isHttpRedirectResponse())
1124 return;
1125
1126 // This can occur when downloading compressed data as some of the data may be the content
1127 // encoding's header. Don't emit anything for this.
1128 if (lastReadyReadEmittedSize == bytesDownloaded) {
1129 if (readBufferMaxSize)
1130 emit q->readBufferFreed(size: dataSize);
1131 return;
1132 }
1133 lastReadyReadEmittedSize = bytesDownloaded;
1134
1135 QVariant totalSize = cookedHeaders.value(key: QNetworkRequest::ContentLengthHeader);
1136
1137 emit q->readyRead();
1138 // emit readyRead before downloadProgress in case this will cause events to be
1139 // processed and we get into a recursive call (as in QProgressDialog).
1140 if (downloadProgressSignalChoke.elapsed() >= progressSignalInterval
1141 && (!decompressHelper.isValid() || decompressHelper.isCountingBytes())) {
1142 downloadProgressSignalChoke.restart();
1143 emit q->downloadProgress(bytesReceived: bytesDownloaded,
1144 bytesTotal: totalSize.isNull() ? Q_INT64_C(-1) : totalSize.toLongLong());
1145 }
1146}

1194void QNetworkReplyHttpImplPrivate::onRedirected(const QUrl &redirectUrl, int httpStatus, int maxRedirectsRemaining)
1195{
1196 Q_Q(QNetworkReplyHttpImpl);
1197 Q_ASSERT(manager);
1198 Q_ASSERT(managerPrivate);
1199
1200 if (isFinished)
1201 return;
1202
1203 const QString schemeBefore(url.scheme());
1204 if (httpRequest.isFollowRedirects()) // update the reply's url as it could've changed
1205 url = redirectUrl;
1206
1207 if (managerPrivate->stsEnabled && managerPrivate->stsCache.isKnownHost(url)) {
1208 // RFC6797, 8.3:
1209 // The UA MUST replace the URI scheme with "https" [RFC2818],
1210 // and if the URI contains an explicit port component of "80",
1211 // then the UA MUST convert the port component to be "443", or
1212 // if the URI contains an explicit port component that is not
1213 // equal to "80", the port component value MUST be preserved;
1214 // otherwise, if the URI does not contain an explicit port
1215 // component, the UA MUST NOT add one.
1216 url.setScheme("https"_L1);
1217 if (url.port() == 80)
1218 url.setPort(443);
1219 }
1220
1221 const bool isLessSafe = schemeBefore == "https"_L1 && url.scheme() == "http"_L1;
1222 if (httpRequest.redirectPolicy() == QNetworkRequest::NoLessSafeRedirectPolicy
1223 && isLessSafe) {
1224 error(code: QNetworkReply::InsecureRedirectError,
1225 errorString: QCoreApplication::translate(context: "QHttp", key: "Insecure redirect"));
1226 return;
1227 }
1228
1229 redirectRequest = createRedirectRequest(originalRequest, url, maxRedirectsRemaining);
1230 operation = getRedirectOperation(currentOp: operation, httpStatus);
1231
1232 // Clear stale headers, the relevant ones get set again later
1233 httpRequest.clearHeaders();
1234 if (operation == QNetworkAccessManager::GetOperation
1235 || operation == QNetworkAccessManager::HeadOperation) {
1236 // possibly changed from not-GET/HEAD to GET/HEAD, make sure to get rid of upload device
1237 uploadByteDevice.reset();
1238 uploadByteDevicePosition = 0;
1239 if (outgoingData) {
1240 QObject::disconnect(sender: outgoingData, SIGNAL(readyRead()), receiver: q,
1241 SLOT(_q_bufferOutgoingData()));
1242 QObject::disconnect(sender: outgoingData, SIGNAL(readChannelFinished()), receiver: q,
1243 SLOT(_q_bufferOutgoingDataFinished()));
1244 }
1245 outgoingData = nullptr;
1246 outgoingDataBuffer.reset();
1247 // We need to explicitly unset these headers so they're not reapplied to the httpRequest
1248 redirectRequest.setHeader(header: QNetworkRequest::ContentLengthHeader, value: QVariant());
1249 redirectRequest.setHeader(header: QNetworkRequest::ContentTypeHeader, value: QVariant());
1250 }
1251
1252 if (const QNetworkCookieJar *const cookieJar = manager->cookieJar()) {
1253 auto cookies = cookieJar->cookiesForUrl(url);
1254 if (!cookies.empty()) {
1255 redirectRequest.setHeader(header: QNetworkRequest::KnownHeaders::CookieHeader,
1256 value: QVariant::fromValue(value: cookies));
1257 }
1258 }
1259
1260 if (httpRequest.redirectPolicy() != QNetworkRequest::UserVerifiedRedirectPolicy)
1261 followRedirect();
1262
1263 emit q->redirected(url);
1264}

1282void QNetworkReplyHttpImplPrivate::checkForRedirect(const int statusCode)
1283{
1284 Q_Q(QNetworkReplyHttpImpl);
1285 switch (statusCode) {
1286 case 301: // Moved Permanently
1287 case 302: // Found
1288 case 303: // See Other
1289 case 307: // Temporary Redirect
1290 case 308: // Permanent Redirect
1291 // What do we do about the caching of the HTML note?
1292 // The response to a 303 MUST NOT be cached, while the response to
1293 // all of the others is cacheable if the headers indicate it to be

1302void QNetworkReplyHttpImplPrivate::replyDownloadMetaData(const QList<QPair<QByteArray,QByteArray> > &hm,
1303 int sc, const QString &rp, bool pu,
1304 QSharedPointer<char> db,
1305 qint64 contentLength,
1306 qint64 removedContentLength,
1307 bool h2Used, bool isCompressed)
1308{
1309 Q_Q(QNetworkReplyHttpImpl);
1310 Q_UNUSED(contentLength);
1311
1312 statusCode = sc;
1313 reasonPhrase = rp;
1314
1315#ifndef QT_NO_SSL
1316 // We parse this header only if we're using secure transport:
1317 //
1318 // RFC6797, 8.1
1319 // If an HTTP response is received over insecure transport, the UA MUST
1320 // ignore any present STS header field(s).
1321 if (url.scheme() == "https"_L1 && managerPrivate->stsEnabled)
1322 managerPrivate->stsCache.updateFromHeaders(hm, url);
1323#endif
1324 // Download buffer
1325 if (!db.isNull()) {
1326 downloadBufferPointer = db;
1327 downloadZerocopyBuffer = downloadBufferPointer.data();
1328 downloadBufferCurrentSize = 0;
1329 q->setAttribute(code: QNetworkRequest::DownloadBufferAttribute, value: QVariant::fromValue<QSharedPointer<char> > (value: downloadBufferPointer));
1330 }
1331
1332 q->setAttribute(code: QNetworkRequest::HttpPipeliningWasUsedAttribute, value: pu);
1333 q->setAttribute(code: QNetworkRequest::Http2WasUsedAttribute, value: h2Used);
1334
1335 // A user having manually defined which encodings they accept is, for
1336 // somwehat unknown (presumed legacy compatibility) reasons treated as
1337 // disabling our decompression:
1338 const bool autoDecompress = request.rawHeader(headerName: "accept-encoding").isEmpty();
1339 const bool shouldDecompress = isCompressed && autoDecompress;
1340 // reconstruct the HTTP header
1341 QList<QPair<QByteArray, QByteArray> > headerMap = hm;
1342 QList<QPair<QByteArray, QByteArray> >::ConstIterator it = headerMap.constBegin(),
1343 end = headerMap.constEnd();
1344 for (; it != end; ++it) {
1345 QByteArray value = q->rawHeader(headerName: it->first);
1346
1347 // Reset any previous "location" header set in the reply. In case of
1348 // redirects, we don't want to 'append' multiple location header values,
1349 // rather we keep only the latest one
1350 if (it->first.toLower() == "location")
1351 value.clear();
1352
1353 if (shouldDecompress && !decompressHelper.isValid()
1354 && it->first.compare(a: "content-encoding", cs: Qt::CaseInsensitive) == 0) {
1355
1356 if (!synchronous) // with synchronous all the data is expected to be handled at once
1357 decompressHelper.setCountingBytesEnabled(true);
1358
1359 if (!decompressHelper.setEncoding(it->second)) {
1360 error(code: QNetworkReplyImpl::NetworkError::UnknownContentError,
1361 errorString: QCoreApplication::translate(context: "QHttp", key: "Failed to initialize decompression: %1")
1362 .arg(a: decompressHelper.errorString()));
1363 return;
1364 }
1365 decompressHelper.setDecompressedSafetyCheckThreshold(
1366 request.decompressedSafetyCheckThreshold());
1367 }
1368
1369 if (!value.isEmpty()) {
1370 // Why are we appending values for headers which are already
1371 // present?
1372 if (it->first.compare(a: "set-cookie", cs: Qt::CaseInsensitive) == 0)
1373 value += '\n';
1374 else
1375 value += ", ";
1376 }
1377 value += it->second;
1378 q->setRawHeader(headerName: it->first, value);
1379 }
1380
1381 q->setAttribute(code: QNetworkRequest::HttpStatusCodeAttribute, value: statusCode);
1382 q->setAttribute(code: QNetworkRequest::HttpReasonPhraseAttribute, value: reasonPhrase);
1383 if (removedContentLength != -1)
1384 q->setAttribute(code: QNetworkRequest::OriginalContentLengthAttribute, value: removedContentLength);
1385
1386 // is it a redirection?
1387 if (!isHttpRedirectResponse())
1388 checkForRedirect(statusCode);
1389
1390 if (statusCode >= 500 && statusCode < 600) {
1391 QAbstractNetworkCache *nc = managerPrivate->networkCache;
1392 if (nc) {
1393 QNetworkCacheMetaData metaData = nc->metaData(url: httpRequest.url());
1394 QNetworkHeadersPrivate cacheHeaders;
1395 cacheHeaders.setAllRawHeaders(metaData.rawHeaders());
1396 QNetworkHeadersPrivate::RawHeadersList::ConstIterator it;
1397 it = cacheHeaders.findRawHeader(key: "Cache-Control");
1398 bool mustReValidate = false;
1399 if (it != cacheHeaders.rawHeaders.constEnd()) {
1400 QHash<QByteArray, QByteArray> cacheControl = parseHttpOptionHeader(header: it->second);
1401 if (cacheControl.contains(key: "must-revalidate"))
1402 mustReValidate = true;
1403 }
1404 if (!mustReValidate && sendCacheContents(metaData))
1405 return;
1406 }
1407 }
1408
1409 if (statusCode == 304) {
1410#if defined(QNETWORKACCESSHTTPBACKEND_DEBUG)
1411 qDebug() << "Received a 304 from" << request.url();
1412#endif
1413 QAbstractNetworkCache *nc = managerPrivate->networkCache;
1414 if (nc) {
1415 QNetworkCacheMetaData oldMetaData = nc->metaData(url: httpRequest.url());
1416 QNetworkCacheMetaData metaData = fetchCacheMetaData(metaData: oldMetaData);
1417 if (oldMetaData != metaData)
1418 nc->updateMetaData(metaData);
1419 if (sendCacheContents(metaData))
1420 return;
1421 }
1422 }
1423
1424
1425 if (statusCode != 304 && statusCode != 303) {
1426 if (!isCachingEnabled())
1427 setCachingEnabled(true);
1428 }
1429
1430 _q_metaDataChanged();
1431}

1682QNetworkCacheMetaData QNetworkReplyHttpImplPrivate::fetchCacheMetaData(const QNetworkCacheMetaData &oldMetaData) const
1683{
1684 Q_Q(const QNetworkReplyHttpImpl);
1685
1686 QNetworkCacheMetaData metaData = oldMetaData;
1687
1688 QNetworkHeadersPrivate cacheHeaders;
1689 cacheHeaders.setAllRawHeaders(metaData.rawHeaders());
1690 QNetworkHeadersPrivate::RawHeadersList::ConstIterator it;
1691
1692 const QList<QByteArray> newHeaders = q->rawHeaderList();
1693 for (QByteArray header : newHeaders) {
1694 QByteArray originalHeader = header;
1695 header = header.toLower();
1696 bool hop_by_hop =
1697 (header == "connection"
1698 || header == "keep-alive"
1699 || header == "proxy-authenticate"
1700 || header == "proxy-authorization"
1701 || header == "te"
1702 || header == "trailers"
1703 || header == "transfer-encoding"
1704 || header == "upgrade");
1705 if (hop_by_hop)
1706 continue;
1707
1708 if (header == "set-cookie")
1709 continue;
1710
1711 // for 4.6.0, we were planning to not store the date header in the
1712 // cached resource; through that we planned to reduce the number
1713 // of writes to disk when using a QNetworkDiskCache (i.e. don't
1714 // write to disk when only the date changes).
1715 // However, without the date we cannot calculate the age of the page
1716 // anymore.
1717 //if (header == "date")
1718 //continue;
1719
1720 // Don't store Warning 1xx headers
1721 if (header == "warning") {
1722 QByteArray v = q->rawHeader(headerName: header);
1723 if (v.size() == 3
1724 && v[0] == '1'
1725 && isAsciiDigit(c: v[1])
1726 && isAsciiDigit(c: v[2]))
1727 continue;
1728 }
1729
1730 it = cacheHeaders.findRawHeader(key: header);
1731 if (it != cacheHeaders.rawHeaders.constEnd()) {
1732 // Match the behavior of Firefox and assume Cache-Control: "no-transform"
1733 if (header == "content-encoding"
1734 || header == "content-range"
1735 || header == "content-type")
1736 continue;
1737 }
1738
1739 // IIS has been known to send "Content-Length: 0" on 304 responses, so
1740 // ignore this too
1741 if (header == "content-length" && statusCode == 304)
1742 continue;
1743
1744#if defined(QNETWORKACCESSHTTPBACKEND_DEBUG)
1745 QByteArray n = q->rawHeader(header);
1746 QByteArray o;
1747 if (it != cacheHeaders.rawHeaders.constEnd())
1748 o = (*it).second;
1749 if (n != o && header != "date") {
1750 qDebug() << "replacing" << header;
1751 qDebug() << "new" << n;
1752 qDebug() << "old" << o;
1753 }
1754#endif
1755 cacheHeaders.setRawHeader(key: originalHeader, value: q->rawHeader(headerName: header));
1756 }
1757 metaData.setRawHeaders(cacheHeaders.rawHeaders);
1758
1759 bool checkExpired = true;
1760
1761 QHash<QByteArray, QByteArray> cacheControl;
1762 it = cacheHeaders.findRawHeader(key: "Cache-Control");
1763 if (it != cacheHeaders.rawHeaders.constEnd()) {
1764 cacheControl = parseHttpOptionHeader(header: it->second);
1765 QByteArray maxAge = cacheControl.value(key: "max-age");
1766 if (!maxAge.isEmpty()) {
1767 checkExpired = false;
1768 QDateTime dt = QDateTime::currentDateTimeUtc();
1769 dt = dt.addSecs(secs: maxAge.toInt());
1770 metaData.setExpirationDate(dt);
1771 }
1772 }
1773 if (checkExpired) {
1774 it = cacheHeaders.findRawHeader(key: "expires");
1775 if (it != cacheHeaders.rawHeaders.constEnd()) {
1776 QDateTime expiredDateTime = QNetworkHeadersPrivate::fromHttpDate(value: it->second);
1777 metaData.setExpirationDate(expiredDateTime);
1778 }
1779 }
1780
1781 it = cacheHeaders.findRawHeader(key: "last-modified");
1782 if (it != cacheHeaders.rawHeaders.constEnd())
1783 metaData.setLastModified(QNetworkHeadersPrivate::fromHttpDate(value: it->second));
1784
1785 bool canDiskCache;
1786 // only cache GET replies by default, all other replies (POST, PUT, DELETE)
1787 // are not cacheable by default (according to RFC 2616 section 9)
1788 if (httpRequest.operation() == QHttpNetworkRequest::Get) {
1789
1790 canDiskCache = true;
1791 // HTTP/1.1. Check the Cache-Control header
1792 if (cacheControl.contains(key: "no-store"))
1793 canDiskCache = false;
1794
1795 // responses to POST might be cacheable
1796 } else if (httpRequest.operation() == QHttpNetworkRequest::Post) {
1797
1798 canDiskCache = false;
1799 // some pages contain "expires:" and "cache-control: no-cache" field,
1800 // so we only might cache POST requests if we get "cache-control: max-age ..."
1801 if (cacheControl.contains(key: "max-age"))
1802 canDiskCache = true;
1803
1804 // responses to PUT and DELETE are not cacheable
1805 } else {
1806 canDiskCache = false;
1807 }
1808
1809 metaData.setSaveToDisk(canDiskCache);
1810 QNetworkCacheMetaData::AttributesMap attributes;
1811 if (statusCode != 304) {
1812 // update the status code
1813 attributes.insert(key: QNetworkRequest::HttpStatusCodeAttribute, value: statusCode);
1814 attributes.insert(key: QNetworkRequest::HttpReasonPhraseAttribute, value: reasonPhrase);
1815 } else {
1816 // this is a redirection, keep the attributes intact
1817 attributes = oldMetaData.attributes();
1818 }
1819 metaData.setAttributes(attributes);
1820 return metaData;
1821}

1872void QNetworkReplyHttpImplPrivate::_q_cacheLoadReadyRead()
1873{
1874 Q_Q(QNetworkReplyHttpImpl);
1875
1876 if (state != Working)
1877 return;
1878 if (!cacheLoadDevice || !q->isOpen() || !cacheLoadDevice->bytesAvailable())
1879 return;
1880
1881 // FIXME Optimize to use zerocopy download buffer if it is a QBuffer.
1882 // Needs to be done where sendCacheContents() (?) of HTTP is emitting
1883 // metaDataChanged ?
1884
1885
1886 QVariant totalSize = cookedHeaders.value(key: QNetworkRequest::ContentLengthHeader);
1887
1888 // emit readyRead before downloadProgress in case this will cause events to be
1889 // processed and we get into a recursive call (as in QProgressDialog).
1890
1891 if (!(isHttpRedirectResponse())) {
1892 // This readyRead() goes to the user. The user then may or may not read() anything.
1893 emit q->readyRead();
1894
1895 if (downloadProgressSignalChoke.elapsed() >= progressSignalInterval) {
1896 downloadProgressSignalChoke.restart();
1897 emit q->downloadProgress(bytesReceived: bytesDownloaded,
1898 bytesTotal: totalSize.isNull() ? Q_INT64_C(-1) : totalSize.toLongLong());
1899 }
1900 }
1901
1902 // A signal we've emitted might be handled by a slot that aborts,
1903 // so we need to check for that and bail out if it's happened:
1904 if (!q->isOpen())
1905 return;
1906
1907 // If there are still bytes available in the cacheLoadDevice then the user did not read
1908 // in response to the readyRead() signal. This means we have to load from the cacheLoadDevice
1909 // and buffer that stuff. This is needed to be able to properly emit finished() later.
1910 while (cacheLoadDevice->bytesAvailable() && !isHttpRedirectResponse())
1911 buffer.append(qba: cacheLoadDevice->readAll());
1912
1913 if (cacheLoadDevice->isSequential()) {
1914 // check if end and we can read the EOF -1
1915 char c;
1916 qint64 actualCount = cacheLoadDevice->read(data: &c, maxlen: 1);
1917 if (actualCount < 0) {
1918 cacheLoadDevice->deleteLater();
1919 cacheLoadDevice = nullptr;
1920 QMetaObject::invokeMethod(obj: q, member: "_q_finished", c: Qt::QueuedConnection);
1921 } else if (actualCount == 1) {
1922 // This is most probably not happening since most QIODevice returned something proper for bytesAvailable()
1923 // and had already been "emptied".
1924 cacheLoadDevice->ungetChar(c);
1925 }
1926 } else if ((!cacheLoadDevice->isSequential() && cacheLoadDevice->atEnd())) {
1927 // This codepath is in case the cache device is a QBuffer, e.g. from QNetworkDiskCache.

1959void QNetworkReplyHttpImplPrivate::_q_bufferOutgoingData()
1960{
1961 Q_Q(QNetworkReplyHttpImpl);
1962
1963 if (!outgoingDataBuffer) {
1964 // first call, create our buffer
1965 outgoingDataBuffer = std::make_shared<QRingBuffer>();
1966
1967 QObject::connect(sender: outgoingData, SIGNAL(readyRead()), receiver: q, SLOT(_q_bufferOutgoingData()));
1968 QObject::connect(sender: outgoingData, SIGNAL(readChannelFinished()), receiver: q, SLOT(_q_bufferOutgoingDataFinished()));
1969 }
1970
1971 qint64 bytesBuffered = 0;
1972 qint64 bytesToBuffer = 0;
1973
1974 // read data into our buffer
1975 forever {
1976 bytesToBuffer = outgoingData->bytesAvailable();
1977 // unknown? just try 2 kB, this also ensures we always try to read the EOF
1978 if (bytesToBuffer <= 0)
1979 bytesToBuffer = 2*1024;
1980
1981 char *dst = outgoingDataBuffer->reserve(bytes: bytesToBuffer);
1982 bytesBuffered = outgoingData->read(data: dst, maxlen: bytesToBuffer);
1983
1984 if (bytesBuffered == -1) {
1985 // EOF has been reached.
1986 outgoingDataBuffer->chop(bytes: bytesToBuffer);
1987
1988 _q_bufferOutgoingDataFinished();
1989 break;
1990 } else if (bytesBuffered == 0) {
1991 // nothing read right now, just wait until we get called again
1992 outgoingDataBuffer->chop(bytes: bytesToBuffer);
1993
1994 break;
1995 } else {
1996 // don't break, try to read() again
1997 outgoingDataBuffer->chop(bytes: bytesToBuffer - bytesBuffered);
1998 }
1999 }
