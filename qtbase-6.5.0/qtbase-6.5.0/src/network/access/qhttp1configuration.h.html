18class QHttp1ConfigurationPrivate;
19class QHttp1Configuration
20{
21public:
22 Q_NETWORK_EXPORT QHttp1Configuration();
23 Q_NETWORK_EXPORT QHttp1Configuration(const QHttp1Configuration &other);

25 : u{other.u} { other.u.d = nullptr; }
26
27 Q_NETWORK_EXPORT QHttp1Configuration &operator=(const QHttp1Configuration &other);
28 QT_MOVE_ASSIGNMENT_OPERATOR_IMPL_VIA_PURE_SWAP(QHttp1Configuration)

30 Q_NETWORK_EXPORT ~QHttp1Configuration();

32 Q_NETWORK_EXPORT void setNumberOfConnectionsPerHost(qsizetype amount);

34
35 void swap(QHttp1Configuration &other) noexcept
36 { std::swap(a&: u, b&: other.u); }
37
38private:
39 struct ShortData {
40 std::uint8_t numConnectionsPerHost;
41 char reserved[sizeof(void*) - sizeof(numConnectionsPerHost)];
42 };
43 union U {
44 U(ShortData _data) : data(_data) {}
45 QHttp1ConfigurationPrivate *d;
46 ShortData data;
47 } u;
48
49 Q_NETWORK_EXPORT bool equals(const QHttp1Configuration &other) const noexcept;
50 Q_NETWORK_EXPORT size_t hash(size_t seed) const noexcept;
51
52 friend bool operator==(const QHttp1Configuration &lhs, const QHttp1Configuration &rhs) noexcept
53 { return lhs.equals(other: rhs); }
54 friend bool operator!=(const QHttp1Configuration &lhs, const QHttp1Configuration &rhs) noexcept
55 { return !lhs.equals(other: rhs); }
56
57 friend size_t qHash(const QHttp1Configuration &key, size_t seed = 0) noexcept { return key.hash(seed); }
58};
