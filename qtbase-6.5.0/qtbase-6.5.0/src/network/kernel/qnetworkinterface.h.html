71class QNetworkInterfacePrivate;
72class Q_NETWORK_EXPORT QNetworkInterface
73{
74 Q_GADGET
75public:
76 enum InterfaceFlag {
77 IsUp = 0x1,
78 IsRunning = 0x2,
79 CanBroadcast = 0x4,
80 IsLoopBack = 0x8,
81 IsPointToPoint = 0x10,
82 CanMulticast = 0x20
83 };
84 Q_DECLARE_FLAGS(InterfaceFlags, InterfaceFlag)
85 Q_FLAG(InterfaceFlags)
86
87 enum InterfaceType {
88 Loopback = 1,
89 Virtual,
90 Ethernet,
91 Slip,
92 CanBus,
93 Ppp,
94 Fddi,
95 Wifi,
96 Ieee80211 = Wifi, // alias
97 Phonet,
98 Ieee802154,
99 SixLoWPAN, // 6LoWPAN, but we can't start with a digit
100 Ieee80216,
101 Ieee1394,
102
103 Unknown = 0
104 };
105 Q_ENUM(InterfaceType)
