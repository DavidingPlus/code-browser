41typedef int (*dn_expand_proto)(const unsigned char *, const unsigned char *, const unsigned char *, char *, int);
42static dn_expand_proto local_dn_expand = nullptr;
43typedef void (*res_nclose_proto)(res_state);
44static res_nclose_proto local_res_nclose = nullptr;
45typedef int (*res_ninit_proto)(res_state);
46static res_ninit_proto local_res_ninit = nullptr;
47typedef int (*res_nquery_proto)(res_state, const char *, int, int, unsigned char *, int);
48static res_nquery_proto local_res_nquery = nullptr;

50// Custom deleter to close resolver state.
51
52struct QDnsLookupStateDeleter
53{
54 static inline void cleanup(struct __res_state *pointer)
55 {
56 local_res_nclose(pointer);
57 }
58};

60static QFunctionPointer resolveSymbol(QLibrary &lib, const char *sym)
61{
62 if (lib.isLoaded())
63 return lib.resolve(symbol: sym);
64
65#if defined(RTLD_DEFAULT) && (defined(Q_OS_FREEBSD) || QT_CONFIG(dlopen))
66 return reinterpret_cast<QFunctionPointer>(dlsym(RTLD_DEFAULT, name: sym));
67#else
68 return nullptr;
69#endif
70}

72static bool resolveLibraryInternal()
73{
74 QLibrary lib;
75#ifdef LIBRESOLV_SO
76 lib.setFileName(QStringLiteral(LIBRESOLV_SO));
77 if (!lib.load())
78#endif
79 {
80 lib.setFileName("resolv"_L1);
81 lib.load();
82 }
83
84 local_dn_expand = dn_expand_proto(resolveSymbol(lib, sym: "__dn_expand"));
85 if (!local_dn_expand)
86 local_dn_expand = dn_expand_proto(resolveSymbol(lib, sym: "dn_expand"));
87
88 local_res_nclose = res_nclose_proto(resolveSymbol(lib, sym: "__res_nclose"));
89 if (!local_res_nclose)
90 local_res_nclose = res_nclose_proto(resolveSymbol(lib, sym: "res_9_nclose"));
91 if (!local_res_nclose)
92 local_res_nclose = res_nclose_proto(resolveSymbol(lib, sym: "res_nclose"));
93
94 local_res_ninit = res_ninit_proto(resolveSymbol(lib, sym: "__res_ninit"));
95 if (!local_res_ninit)
96 local_res_ninit = res_ninit_proto(resolveSymbol(lib, sym: "res_9_ninit"));
97 if (!local_res_ninit)
98 local_res_ninit = res_ninit_proto(resolveSymbol(lib, sym: "res_ninit"));
99
100 local_res_nquery = res_nquery_proto(resolveSymbol(lib, sym: "__res_nquery"));
101 if (!local_res_nquery)
102 local_res_nquery = res_nquery_proto(resolveSymbol(lib, sym: "res_9_nquery"));
103 if (!local_res_nquery)
104 local_res_nquery = res_nquery_proto(resolveSymbol(lib, sym: "res_nquery"));
105
106 return true;
107}

110void QDnsLookupRunnable::query(const int requestType, const QByteArray &requestName, const QHostAddress &nameserver, QDnsLookupReply *reply)
111{
112 // Load dn_expand, res_ninit and res_nquery on demand.
113 resolveLibrary();
114
115 // If dn_expand, res_ninit or res_nquery is missing, fail.
116 if (!local_dn_expand || !local_res_nclose || !local_res_ninit || !local_res_nquery) {
117 reply->error = QDnsLookup::ResolverError;
118 reply->errorString = tr(s: "Resolver functions not found");
119 return;
120 }
121
122 // Initialize state.
123 struct __res_state state;
124 std::memset(s: &state, c: 0, n: sizeof(state));
125 if (local_res_ninit(&state) < 0) {
126 reply->error = QDnsLookup::ResolverError;
127 reply->errorString = tr(s: "Resolver initialization failed");
128 return;
129 }
130
131 //Check if a nameserver was set. If so, use it
132 if (!nameserver.isNull()) {
133 if (nameserver.protocol() == QAbstractSocket::IPv4Protocol) {
134 state.nsaddr_list[0].sin_addr.s_addr = htonl(nameserver.toIPv4Address());
135 state.nscount = 1;
136 } else if (nameserver.protocol() == QAbstractSocket::IPv6Protocol) {
137#if defined(Q_OS_LINUX)
138 struct sockaddr_in6 *ns;
139 ns = state._u._ext.nsaddrs[0];
140 // nsaddrs will be NULL if no nameserver is set in /etc/resolv.conf
141 if (!ns) {
142 // Memory allocated here will be free'd in res_close() as we
143 // have done res_init() above.
144 ns = (struct sockaddr_in6*) calloc(nmemb: 1, size: sizeof(struct sockaddr_in6));
145 Q_CHECK_PTR(ns);
146 state._u._ext.nsaddrs[0] = ns;
147 }
148#ifndef __UCLIBC__
149 // Set nsmap[] to indicate that nsaddrs[0] is an IPv6 address
150 // See: https://sourceware.org/ml/libc-hacker/2002-05/msg00035.html
151 state._u._ext.nsmap[0] = MAXNS + 1;
152#endif
153 state._u._ext.nscount6 = 1;
154 ns->sin6_family = AF_INET6;
155 ns->sin6_port = htons(53);
156 SetSALen::set(ns, sizeof(*ns));
157
158 Q_IPV6ADDR ipv6Address = nameserver.toIPv6Address();
159 for (int i=0; i<16; i++) {
160 ns->sin6_addr.s6_addr[i] = ipv6Address[i];
161 }
162#else
163 qWarning("%s", QDnsLookupPrivate::msgNoIpV6NameServerAdresses);
164 reply->error = QDnsLookup::ResolverError;
165 reply->errorString = tr(QDnsLookupPrivate::msgNoIpV6NameServerAdresses);
166 return;
167#endif
168 }
169 }
170#ifdef QDNSLOOKUP_DEBUG
171 state.options |= RES_DEBUG;
172#endif
173 QScopedPointer<struct __res_state, QDnsLookupStateDeleter> state_ptr(&state);
174
175 // Perform DNS query.
176 QVarLengthArray<unsigned char, PACKETSZ> buffer(PACKETSZ);
177 std::memset(s: buffer.data(), c: 0, n: buffer.size());
178 int responseLength = local_res_nquery(&state, requestName, C_IN, requestType, buffer.data(), buffer.size());
179 if (Q_UNLIKELY(responseLength > PACKETSZ)) {
180 buffer.resize(sz: responseLength);
181 std::memset(s: buffer.data(), c: 0, n: buffer.size());
182 responseLength = local_res_nquery(&state, requestName, C_IN, requestType, buffer.data(), buffer.size());
183 if (Q_UNLIKELY(responseLength > buffer.size())) {
184 // Ok, we give up.
185 reply->error = QDnsLookup::ResolverError;
186 reply->errorString.clear(); // We cannot be more specific, alas.
187 return;
188 }
189 }
190
191 unsigned char *response = buffer.data();
192 // Check the response header. Though res_nquery returns -1 as a
193 // responseLength in case of error, we still can extract the
194 // exact error code from the response.
195 HEADER *header = (HEADER*)response;
196 const int answerCount = ntohs(header->ancount);
197 switch (header->rcode) {
198 case NOERROR:
199 break;
200 case FORMERR:
201 reply->error = QDnsLookup::InvalidRequestError;
202 reply->errorString = tr(s: "Server could not process query");
203 return;
204 case SERVFAIL:
205 reply->error = QDnsLookup::ServerFailureError;
206 reply->errorString = tr(s: "Server failure");
207 return;
208 case NXDOMAIN:
209 reply->error = QDnsLookup::NotFoundError;
210 reply->errorString = tr(s: "Non existent domain");
211 return;
212 case REFUSED:
213 reply->error = QDnsLookup::ServerRefusedError;
214 reply->errorString = tr(s: "Server refused to answer");
215 return;
216 default:
217 reply->error = QDnsLookup::InvalidReplyError;
218 reply->errorString = tr(s: "Invalid reply received");
219 return;
220 }
221
222 // Check the reply is valid.
223 if (responseLength < int(sizeof(HEADER))) {
224 reply->error = QDnsLookup::InvalidReplyError;
225 reply->errorString = tr(s: "Invalid reply received");
226 return;
227 }
228
229 // Skip the query host, type (2 bytes) and class (2 bytes).
230 char host[PACKETSZ], answer[PACKETSZ];
231 unsigned char *p = response + sizeof(HEADER);
232 int status = local_dn_expand(response, response + responseLength, p, host, sizeof(host));
233 if (status < 0) {
234 reply->error = QDnsLookup::InvalidReplyError;
235 reply->errorString = tr(s: "Could not expand domain name");
236 return;
237 }
238 p += status + 4;
239
240 // Extract results.
241 int answerIndex = 0;
242 while ((p < response + responseLength) && (answerIndex < answerCount)) {
243 status = local_dn_expand(response, response + responseLength, p, host, sizeof(host));
244 if (status < 0) {
245 reply->error = QDnsLookup::InvalidReplyError;
246 reply->errorString = tr(s: "Could not expand domain name");
247 return;
248 }
249 const QString name = QUrl::fromAce(domain: host);
250
251 p += status;
252 const quint16 type = (p[0] << 8) | p[1];
253 p += 2; // RR type
254 p += 2; // RR class
255 const quint32 ttl = (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
256 p += 4;
257 const quint16 size = (p[0] << 8) | p[1];
258 p += 2;
259
260 if (type == QDnsLookup::A) {
261 if (size != 4) {
262 reply->error = QDnsLookup::InvalidReplyError;
263 reply->errorString = tr(s: "Invalid IPv4 address record");
264 return;
265 }
266 const quint32 addr = (p[0] << 24) | (p[1] << 16) | (p[2] << 8) | p[3];
267 QDnsHostAddressRecord record;
268 record.d->name = name;
269 record.d->timeToLive = ttl;
270 record.d->value = QHostAddress(addr);
271 reply->hostAddressRecords.append(t: record);
272 } else if (type == QDnsLookup::AAAA) {
273 if (size != 16) {
274 reply->error = QDnsLookup::InvalidReplyError;
275 reply->errorString = tr(s: "Invalid IPv6 address record");
276 return;
277 }
278 QDnsHostAddressRecord record;
279 record.d->name = name;
280 record.d->timeToLive = ttl;
281 record.d->value = QHostAddress(p);
282 reply->hostAddressRecords.append(t: record);
283 } else if (type == QDnsLookup::CNAME) {
284 status = local_dn_expand(response, response + responseLength, p, answer, sizeof(answer));
285 if (status < 0) {
286 reply->error = QDnsLookup::InvalidReplyError;
287 reply->errorString = tr(s: "Invalid canonical name record");
288 return;
289 }
290 QDnsDomainNameRecord record;
291 record.d->name = name;
292 record.d->timeToLive = ttl;
293 record.d->value = QUrl::fromAce(domain: answer);
294 reply->canonicalNameRecords.append(t: record);
295 } else if (type == QDnsLookup::NS) {
296 status = local_dn_expand(response, response + responseLength, p, answer, sizeof(answer));
297 if (status < 0) {
298 reply->error = QDnsLookup::InvalidReplyError;
299 reply->errorString = tr(s: "Invalid name server record");
300 return;
301 }
302 QDnsDomainNameRecord record;
303 record.d->name = name;
304 record.d->timeToLive = ttl;
305 record.d->value = QUrl::fromAce(domain: answer);
306 reply->nameServerRecords.append(t: record);
307 } else if (type == QDnsLookup::PTR) {
308 status = local_dn_expand(response, response + responseLength, p, answer, sizeof(answer));
309 if (status < 0) {
310 reply->error = QDnsLookup::InvalidReplyError;
311 reply->errorString = tr(s: "Invalid pointer record");
312 return;
313 }
314 QDnsDomainNameRecord record;
315 record.d->name = name;
316 record.d->timeToLive = ttl;
317 record.d->value = QUrl::fromAce(domain: answer);
318 reply->pointerRecords.append(t: record);
319 } else if (type == QDnsLookup::MX) {
320 const quint16 preference = (p[0] << 8) | p[1];
321 status = local_dn_expand(response, response + responseLength, p + 2, answer, sizeof(answer));
322 if (status < 0) {
323 reply->error = QDnsLookup::InvalidReplyError;
324 reply->errorString = tr(s: "Invalid mail exchange record");
325 return;
326 }
327 QDnsMailExchangeRecord record;
328 record.d->exchange = QUrl::fromAce(domain: answer);
329 record.d->name = name;
330 record.d->preference = preference;
331 record.d->timeToLive = ttl;
332 reply->mailExchangeRecords.append(t: record);
333 } else if (type == QDnsLookup::SRV) {
334 const quint16 priority = (p[0] << 8) | p[1];
335 const quint16 weight = (p[2] << 8) | p[3];
336 const quint16 port = (p[4] << 8) | p[5];
337 status = local_dn_expand(response, response + responseLength, p + 6, answer, sizeof(answer));
338 if (status < 0) {
339 reply->error = QDnsLookup::InvalidReplyError;
340 reply->errorString = tr(s: "Invalid service record");
341 return;
342 }
343 QDnsServiceRecord record;
344 record.d->name = name;
345 record.d->target = QUrl::fromAce(domain: answer);
346 record.d->port = port;
347 record.d->priority = priority;
348 record.d->timeToLive = ttl;
349 record.d->weight = weight;
350 reply->serviceRecords.append(t: record);
351 } else if (type == QDnsLookup::TXT) {
352 unsigned char *txt = p;
353 QDnsTextRecord record;
354 record.d->name = name;
355 record.d->timeToLive = ttl;
356 while (txt < p + size) {
357 const unsigned char length = *txt;
358 txt++;
359 if (txt + length > p + size) {
360 reply->error = QDnsLookup::InvalidReplyError;
361 reply->errorString = tr(s: "Invalid text record");
362 return;
363 }
364 record.d->values << QByteArray((char*)txt, length);
365 txt += length;
366 }
367 reply->textRecords.append(t: record);
368 }
369 p += size;
370 answerIndex++;
371 }
372}
