64 // lookupHost to a QObject slot
65 template <typename Func>
66 static inline int lookupHost(const QString &name,
67 const typename QtPrivate::FunctionPointer<Func>::Object *receiver,
68 Func slot)
69 {
70 typedef QtPrivate::FunctionPointer<Func> SlotType;
71
72 typedef QtPrivate::FunctionPointer<void (*)(QHostInfo)> SignalType;
73 static_assert(int(SignalType::ArgumentCount) >= int(SlotType::ArgumentCount),
74 "The slot requires more arguments than the signal provides.");
75 static_assert((QtPrivate::CheckCompatibleArguments<typename SignalType::Arguments,
76 typename SlotType::Arguments>::value),
77 "Signal and slot arguments are not compatible.");
78 static_assert((QtPrivate::AreArgumentsCompatible<typename SlotType::ReturnType,
79 typename SignalType::ReturnType>::value),
80 "Return type of the slot is not compatible "
81 "with the return type of the signal.");
82
83 auto slotObj = new QtPrivate::QSlotObject<Func, typename SlotType::Arguments, void>(slot);
84 return lookupHostImpl(name, receiver, slotObj, member: nullptr);
85 }
86
87 // lookupHost to a callable (without context)
88 template <typename Func>
89 static inline typename std::enable_if<!QtPrivate::FunctionPointer<Func>::IsPointerToMemberFunction &&
90 !std::is_same<const char *, Func>::value, int>::type
91 lookupHost(const QString &name, Func slot)
92 {
93 return lookupHost(name, nullptr, std::move(slot));
94 }
95
96 // lookupHost to a functor or function pointer (with context)
97 template <typename Func1>
98 static inline typename std::enable_if<!QtPrivate::FunctionPointer<Func1>::IsPointerToMemberFunction &&
99 !std::is_same<const char*, Func1>::value, int>::type
100 lookupHost(const QString &name, QObject *context, Func1 slot)
101 {
102 typedef QtPrivate::FunctionPointer<Func1> SlotType;
103
104 static_assert(int(SlotType::ArgumentCount) <= 1,
105 "The slot must not require more than one argument");
106
107 auto slotObj = new QtPrivate::QFunctorSlotObject<Func1, 1,
108 typename QtPrivate::List<QHostInfo>,
109 void>(std::move(slot));
110 return lookupHostImpl(name, receiver: context, slotObj, member: nullptr);
111 }
112#endif // Q_QDOC
