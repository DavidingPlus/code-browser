43enum LibResolvFeature {
44 NeedResInit,
45 NeedResNInit
46};
47
48typedef struct __res_state *res_state_ptr;
49
50typedef int (*res_init_proto)(void);
51static res_init_proto local_res_init = nullptr;
52typedef int (*res_ninit_proto)(res_state_ptr);
53static res_ninit_proto local_res_ninit = nullptr;
54typedef void (*res_nclose_proto)(res_state_ptr);
55static res_nclose_proto local_res_nclose = nullptr;
56static res_state_ptr local_res = nullptr;
57
58#if QT_CONFIG(library) && !defined(Q_OS_QNX)
59namespace {
60struct LibResolv
61{
62 enum {
63#ifdef RES_NORELOAD
64 // If RES_NORELOAD is defined, then the libc is capable of watching
65 // /etc/resolv.conf for changes and reloading as necessary. So accept
66 // whatever is configured.
67 ReinitNecessary = false
68#else
69 ReinitNecessary = true
70#endif
71 };
72
73 QLibrary lib;
74 LibResolv();
75 ~LibResolv() { lib.unload(); }
76};
77}

79static QFunctionPointer resolveSymbol(QLibrary &lib, const char *sym)
80{
81 if (lib.isLoaded())
82 return lib.resolve(symbol: sym);
83
84#if defined(RTLD_DEFAULT) && (defined(Q_OS_FREEBSD) || QT_CONFIG(dlopen))
85 return reinterpret_cast<QFunctionPointer>(dlsym(RTLD_DEFAULT, name: sym));
86#else
87 return nullptr;
88#endif
89}

91LibResolv::LibResolv()
92{
93#ifdef LIBRESOLV_SO
94 lib.setFileName(QStringLiteral(LIBRESOLV_SO));
95 if (!lib.load())
96#endif
97 {
98 lib.setFileName("resolv"_L1);
99 lib.load();
100 }
101
102 // res_ninit is required for localDomainName()
103 local_res_ninit = res_ninit_proto(resolveSymbol(lib, sym: "__res_ninit"));
104 if (!local_res_ninit)
105 local_res_ninit = res_ninit_proto(resolveSymbol(lib, sym: "res_ninit"));
106 if (local_res_ninit) {
107 // we must now find res_nclose
108 local_res_nclose = res_nclose_proto(resolveSymbol(lib, sym: "res_nclose"));
109 if (!local_res_nclose)
110 local_res_nclose = res_nclose_proto(resolveSymbol(lib, sym: "__res_nclose"));
111 if (!local_res_nclose)
112 local_res_ninit = nullptr;
113 }
114
115 if (ReinitNecessary || !local_res_ninit) {
116 local_res_init = res_init_proto(resolveSymbol(lib, sym: "__res_init"));
117 if (!local_res_init)
118 local_res_init = res_init_proto(resolveSymbol(lib, sym: "res_init"));
119
120 if (local_res_init && !local_res_ninit) {
121 // if we can't get a thread-safe context, we have to use the global _res state
122 local_res = res_state_ptr(resolveSymbol(lib, sym: "_res"));
123 }
124 }
125}

140QHostInfo QHostInfoAgent::fromName(const QString &hostName)
141{
142 QHostInfo results;
143
144#if defined(QHOSTINFO_DEBUG)
145 qDebug("QHostInfoAgent::fromName(%s) looking up...",
146 hostName.toLatin1().constData());
147#endif
148
149 // Load res_init on demand.
150 resolveLibrary(f: NeedResInit);
151
152 // If res_init is available, poll it.
153 if (local_res_init)
154 local_res_init();
155
156 QHostAddress address;
157 if (address.setAddress(hostName))
158 return reverseLookup(address);
159
160 return lookup(hostName);
161}

163QString QHostInfo::localDomainName()
164{
165#if !defined(Q_OS_VXWORKS) && !defined(Q_OS_ANDROID)
166 resolveLibrary(f: NeedResNInit);
167 if (local_res_ninit) {
168 // using thread-safe version
169 res_state_ptr state = res_state_ptr(malloc(size: sizeof(*state)));
170 Q_CHECK_PTR(state);
171 memset(s: state, c: 0, n: sizeof(*state));
172 local_res_ninit(state);
173 QString domainName = QUrl::fromAce(domain: state->defdname);
174 if (domainName.isEmpty())
175 domainName = QUrl::fromAce(domain: state->dnsrch[0]);
176 local_res_nclose(state);
177 free(ptr: state);
178
179 return domainName;
180 }
181
182 if (local_res_init && local_res) {
183 // using thread-unsafe version
184
185 local_res_init();
186 QString domainName = QUrl::fromAce(domain: local_res->defdname);
187 if (domainName.isEmpty())
188 domainName = QUrl::fromAce(domain: local_res->dnsrch[0]);
189 return domainName;
190 }
191#endif
192 // nothing worked, try doing it by ourselves:
193 QFile resolvconf;
194#if defined(_PATH_RESCONF)
195 resolvconf.setFileName(QFile::decodeName(_PATH_RESCONF));
196#else
197 resolvconf.setFileName("/etc/resolv.conf"_L1);
198#endif
199 if (!resolvconf.open(flags: QIODevice::ReadOnly))
200 return QString(); // failure
201
202 QString domainName;
203 while (!resolvconf.atEnd()) {
204 QByteArray line = resolvconf.readLine().trimmed();
205 if (line.startsWith(bv: "domain "))
206 return QUrl::fromAce(domain: line.mid(index: sizeof "domain " - 1).trimmed());
207
208 // in case there's no "domain" line, fall back to the first "search" entry
209 if (domainName.isEmpty() && line.startsWith(bv: "search ")) {
210 QByteArray searchDomain = line.mid(index: sizeof "search " - 1).trimmed();
211 int pos = searchDomain.indexOf(c: ' ');
212 if (pos != -1)
213 searchDomain.truncate(pos);
214 domainName = QUrl::fromAce(domain: searchDomain);
215 }
216 }
217
218 // return the fallen-back-to searched domain
219 return domainName;
220}
