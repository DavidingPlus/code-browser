44class QHostInfoResult : public QObject
45{
46 Q_OBJECT
47public:
48 QHostInfoResult(const QObject *receiver, QtPrivate::QSlotObjectBase *slotObj)
49 : receiver(receiver), slotObj(slotObj),
50 withContextObject(slotObj && receiver)
51 {
52 if (receiver)
53 moveToThread(thread: receiver->thread());
54 }
55
56 void postResultsReady(const QHostInfo &info);
57
58Q_SIGNALS:
59 void resultsReady(const QHostInfo &info);
60
61protected:
62 bool event(QEvent *event) override;
63
64private:
65 QHostInfoResult(const QHostInfoResult *other)
66 : receiver(other->receiver), slotObj(other->slotObj),
67 withContextObject(other->withContextObject)
68 {
69 // cleanup if the application terminates before results are delivered
70 connect(sender: QCoreApplication::instance(), signal: &QCoreApplication::aboutToQuit,
71 receiver: this, slot: &QObject::deleteLater);
72 // maintain thread affinity
73 moveToThread(thread: other->thread());
74 }
75
76 QPointer<const QObject> receiver = nullptr;
77 QtPrivate::QSlotObjectBase *slotObj = nullptr;
78 const bool withContextObject = false;
79};
