22static bool qt_qdnsmailexchangerecord_less_than(const QDnsMailExchangeRecord &r1, const QDnsMailExchangeRecord &r2)
23{
24 // Lower numbers are more preferred than higher ones.
25 return r1.preference() < r2.preference();
26}
27
28/*
29 Sorts a list of QDnsMailExchangeRecord objects according to RFC 5321.
30*/
31
32static void qt_qdnsmailexchangerecord_sort(QList<QDnsMailExchangeRecord> &records)
33{
34 // If we have no more than one result, we are done.
35 if (records.size() <= 1)
36 return;
37
38 // Order the records by preference.
39 std::sort(first: records.begin(), last: records.end(), comp: qt_qdnsmailexchangerecord_less_than);
40
41 int i = 0;
42 while (i < records.size()) {
43
44 // Determine the slice of records with the current preference.
45 QList<QDnsMailExchangeRecord> slice;
46 const quint16 slicePreference = records.at(i).preference();
47 for (int j = i; j < records.size(); ++j) {
48 if (records.at(i: j).preference() != slicePreference)
49 break;
50 slice << records.at(i: j);
51 }
52
53 // Randomize the slice of records.
54 while (!slice.isEmpty()) {
55 const unsigned int pos = QRandomGenerator::global()->bounded(highest: slice.size());
56 records[i++] = slice.takeAt(i: pos);
57 }
58 }
59}

61static bool qt_qdnsservicerecord_less_than(const QDnsServiceRecord &r1, const QDnsServiceRecord &r2)
62{
63 // Order by priority, or if the priorities are equal,
64 // put zero weight records first.
65 return r1.priority() < r2.priority()
66 || (r1.priority() == r2.priority()
67 && r1.weight() == 0 && r2.weight() > 0);
68}

70/*
71 Sorts a list of QDnsServiceRecord objects according to RFC 2782.
72*/
73
74static void qt_qdnsservicerecord_sort(QList<QDnsServiceRecord> &records)
75{
76 // If we have no more than one result, we are done.
77 if (records.size() <= 1)
78 return;
79
80 // Order the records by priority, and for records with an equal
81 // priority, put records with a zero weight first.
82 std::sort(first: records.begin(), last: records.end(), comp: qt_qdnsservicerecord_less_than);
83
84 int i = 0;
85 while (i < records.size()) {
86
87 // Determine the slice of records with the current priority.
88 QList<QDnsServiceRecord> slice;
89 const quint16 slicePriority = records.at(i).priority();
90 unsigned int sliceWeight = 0;
91 for (int j = i; j < records.size(); ++j) {
92 if (records.at(i: j).priority() != slicePriority)
93 break;
94 sliceWeight += records.at(i: j).weight();
95 slice << records.at(i: j);
96 }
97#ifdef QDNSLOOKUP_DEBUG
98 qDebug("qt_qdnsservicerecord_sort() : priority %i (size: %i, total weight: %i)",
99 slicePriority, slice.size(), sliceWeight);
100#endif
101
102 // Order the slice of records.
103 while (!slice.isEmpty()) {
104 const unsigned int weightThreshold = QRandomGenerator::global()->bounded(highest: sliceWeight + 1);
105 unsigned int summedWeight = 0;
106 for (int j = 0; j < slice.size(); ++j) {
107 summedWeight += slice.at(i: j).weight();
108 if (summedWeight >= weightThreshold) {
109#ifdef QDNSLOOKUP_DEBUG
110 qDebug("qt_qdnsservicerecord_sort() : adding %s %i (weight: %i)",
111 qPrintable(slice.at(j).target()), slice.at(j).port(),
112 slice.at(j).weight());
113#endif
114 // Adjust the slice weight and take the current record.
115 sliceWeight -= slice.at(i: j).weight();
116 records[i++] = slice.takeAt(i: j);
117 break;
118 }
119 }
120 }
121 }

130 \since 5.0

132 \inmodule QtNetwork
133 \ingroup network

135 QDnsLookup uses the mechanisms provided by the operating system to perform
136 DNS lookups. To perform a lookup you need to specify a \l name and \l type
137 then invoke the \l{QDnsLookup::lookup()}{lookup()} slot. The
138 \l{QDnsLookup::finished()}{finished()} signal will be emitted upon
139 completion.
140
141 For example, you can determine which servers an XMPP chat client should
142 connect to for a given domain with:
143
144 \snippet code/src_network_kernel_qdnslookup.cpp 0
145
146 Once the request finishes you can handle the results with:
147
148 \snippet code/src_network_kernel_qdnslookup.cpp 1
149
150 \note If you simply want to find the IP address(es) associated with a host
151 name, or the host name associated with an IP address you should use
152 QHostInfo instead.
153*/

158 Indicates all possible error conditions found during the
159 processing of the DNS lookup.
160
161 \value NoError no error condition.
162
163 \value ResolverError there was an error initializing the system's
164 DNS resolver.
165
166 \value OperationCancelledError the lookup was aborted using the abort()

168
169 \value InvalidRequestError the requested DNS lookup was invalid.

171 \value InvalidReplyError the reply returned by the server was invalid.

449/*!
450 Aborts the DNS lookup operation.
451
452 If the lookup is already finished, does nothing.
453*/
454
455void QDnsLookup::abort()
456{
457 Q_D(QDnsLookup);
458 if (d->runnable) {
459 d->runnable = nullptr;
460 d->reply = QDnsLookupReply();
461 d->reply.error = QDnsLookup::OperationCancelledError;
462 d->reply.errorString = tr(s: "Operation cancelled");
463 d->isFinished = true;
464 emit finished();
465 }
466}
467
468/*!
469 Performs the DNS lookup.
470
471 The \l{QDnsLookup::finished()}{finished()} signal is emitted upon completion.
472*/
473
474void QDnsLookup::lookup()
475{
476 Q_D(QDnsLookup);
477 d->isFinished = false;
478 d->reply = QDnsLookupReply();
479 d->runnable = new QDnsLookupRunnable(d->type, QUrl::toAce(domain: d->name), d->nameserver);
480 connect(sender: d->runnable, SIGNAL(finished(QDnsLookupReply)),
481 receiver: this, SLOT(_q_lookupFinished(QDnsLookupReply)),
482 Qt::BlockingQueuedConnection);
483#if QT_CONFIG(thread)
484 theDnsLookupThreadPool()->start(runnable: d->runnable);
485#endif
486}

965 if (runnable == q->sender()) {
966#ifdef QDNSLOOKUP_DEBUG

975
976void QDnsLookupRunnable::run()
977{
978 QDnsLookupReply reply;
979
980 // Validate input.
981 if (requestName.isEmpty()) {
982 reply.error = QDnsLookup::InvalidRequestError;
983 reply.errorString = tr(s: "Invalid domain name");
984 emit finished(reply);
985 return;
986 }
987
988 // Perform request.
989 query(requestType, requestName, nameserver, reply: &reply);
990
991 // Sort results.
992 qt_qdnsmailexchangerecord_sort(records&: reply.mailExchangeRecords);
993 qt_qdnsservicerecord_sort(records&: reply.serviceRecords);
994
995 emit finished(reply);
996}

1007{

1009 if (!signalsConnected) {

1013 if (!app) {
1014 qWarning(msg: "QDnsLookup requires a QCoreApplication");
1015 delete runnable;
1016 return;
1017 }
1018
1019 moveToThread(thread: app->thread());
1020 connect(asender: app, SIGNAL(destroyed()),
1021 SLOT(_q_applicationDestroyed()), atype: Qt::DirectConnection);
1022 signalsConnected = true;
1023 }
1024 }
