37void QHostAddressPrivate::setAddress(quint32 a_)
38{
39 a = a_;
40 protocol = QHostAddress::IPv4Protocol;
41
42 //create mapped address, except for a_ == 0 (any)
43 a6_64.c[0] = 0;
44 if (a) {
45 a6_32.c[2] = qToBigEndian(source: 0xffff);
46 a6_32.c[3] = qToBigEndian(source: a);
47 } else {
48 a6_64.c[1] = 0;
49 }

52/// parses v4-mapped addresses or the AnyIPv6 address and stores in \a a;
53/// returns true if the address was one of those
54static bool convertToIpv4(quint32& a, const Q_IPV6ADDR &a6, const QHostAddress::ConversionMode mode)
55{
56 if (mode == QHostAddress::StrictConversion)
57 return false;
58
59 const uchar *ptr = a6.c;
60 if (qFromUnaligned<quint64>(src: ptr) != 0)
61 return false;
62
63 const quint32 mid = qFromBigEndian<quint32>(src: ptr + 8);
64 if ((mid == 0xffff) && (mode & QHostAddress::ConvertV4MappedToIPv4)) {
65 a = qFromBigEndian<quint32>(src: ptr + 12);
66 return true;
67 }
68 if (mid != 0)
69 return false;
70
71 const quint32 low = qFromBigEndian<quint32>(src: ptr + 12);
72 if ((low == 0) && (mode & QHostAddress::ConvertUnspecifiedAddress)) {
73 a = 0;
74 return true;
75 }
76 if ((low == 1) && (mode & QHostAddress::ConvertLocalHost)) {
77 a = INADDR_LOOPBACK;
78 return true;
79 }
80 if ((low != 1) && (mode & QHostAddress::ConvertV4CompatToIPv4)) {
81 a = low;
82 return true;
83 }
84 return false;
85}

101static bool parseIp6(const QString &address, QIPAddressUtils::IPv6Address &addr, QString *scopeId)
102{
103 QStringView tmp(address);
104 qsizetype scopeIdPos = tmp.lastIndexOf(c: u'%');
105 if (scopeIdPos != -1) {
106 *scopeId = tmp.mid(pos: scopeIdPos + 1).toString();
107 tmp.chop(n: tmp.size() - scopeIdPos);
108 } else {
109 scopeId->clear();
110 }
111 return QIPAddressUtils::parseIp6(address&: addr, begin: tmp.begin(), end: tmp.end()) == nullptr;
112}

114bool QHostAddressPrivate::parse(const QString &ipString)
115{
116 protocol = QHostAddress::UnknownNetworkLayerProtocol;
117 QString a = ipString.simplified();
118 if (a.isEmpty())
119 return false;
120
121 // All IPv6 addresses contain a ':', and may contain a '.'.
122 if (a.contains(c: u':')) {
123 quint8 maybeIp6[16];
124 if (parseIp6(address: a, addr&: maybeIp6, scopeId: &scopeId)) {
125 setAddress(maybeIp6);
126 return true;
127 }
128 }
129
130 quint32 maybeIp4 = 0;
131 if (QIPAddressUtils::parseIp4(address&: maybeIp4, begin: a.constBegin(), end: a.constEnd())) {
132 setAddress(maybeIp4);
133 return true;
134 }
135
136 return false;
137}

146AddressClassification QHostAddressPrivate::classify() const
147{
148 if (a) {
149 // This is an IPv4 address or an IPv6 v4-mapped address includes all
150 // IPv6 v4-compat addresses, except for ::ffff:0.0.0.0 (because `a' is
151 // zero). See setAddress(quint8*) below, which calls convertToIpv4(),
152 // for details.
153 // Source: RFC 5735
154 if ((a & 0xff000000U) == 0x7f000000U) // 127.0.0.0/8
155 return LoopbackAddress;
156 if ((a & 0xf0000000U) == 0xe0000000U) // 224.0.0.0/4
157 return MulticastAddress;
158 if ((a & 0xffff0000U) == 0xa9fe0000U) // 169.254.0.0/16
159 return LinkLocalAddress;
160 if ((a & 0xff000000U) == 0) // 0.0.0.0/8 except 0.0.0.0 (handled below)
161 return LocalNetAddress;
162 if ((a & 0xf0000000U) == 0xf0000000U) { // 240.0.0.0/4
163 if (a == 0xffffffffU) // 255.255.255.255
164 return BroadcastAddress;
165 return UnknownAddress;
166 }
167
168 // Not testing for PrivateNetworkAddress and TestNetworkAddress
169 // since we don't need them yet.
170 return GlobalAddress;
171 }
172
173 // As `a' is zero, this address is either ::ffff:0.0.0.0 or a non-v4-mapped IPv6 address.
174 // Source: https://www.iana.org/assignments/ipv6-address-space/ipv6-address-space.xhtml
175 if (a6_64.c[0]) {
176 quint32 high16 = qFromBigEndian(source: a6_32.c[0]) >> 16;
177 switch (high16 >> 8) {
178 case 0xff: // ff00::/8
179 return MulticastAddress;
180 case 0xfe:
181 switch (high16 & 0xffc0) {
182 case 0xfec0: // fec0::/10
183 return SiteLocalAddress;
184
185 case 0xfe80: // fe80::/10
186 return LinkLocalAddress;
187
188 default: // fe00::/9
189 return UnknownAddress;
190 }
191 case 0xfd: // fc00::/7
192 case 0xfc:
193 return UniqueLocalAddress;
194 default:
195 return GlobalAddress;
196 }
197 }
198
199 quint64 low64 = qFromBigEndian(source: a6_64.c[1]);
200 if (low64 == 1) // ::1
201 return LoopbackAddress;
202 if (low64 >> 32 == 0xffff) { // ::ffff:0.0.0.0/96
203 Q_ASSERT(quint32(low64) == 0);
204 return LocalNetAddress;
205 }
206 if (low64) // not ::
207 return GlobalAddress;
208
209 if (protocol == QHostAddress::UnknownNetworkLayerProtocol)
210 return UnknownAddress;
211
212 // only :: and 0.0.0.0 remain now
213 return LocalNetAddress;
214}

216bool QNetmask::setAddress(const QHostAddress &address)
217{
218 static const quint8 zeroes[16] = { 0 };
219 union {
220 quint32 v4;
221 quint8 v6[16];
222 } ip;
223
224 int netmask = 0;
225 quint8 *ptr = ip.v6;
226 quint8 *end;
227 length = 255;
228
229 if (address.protocol() == QHostAddress::IPv4Protocol) {
230 ip.v4 = qToBigEndian(source: address.toIPv4Address());
231 end = ptr + 4;
232 } else if (address.protocol() == QHostAddress::IPv6Protocol) {
233 memcpy(dest: ip.v6, src: address.toIPv6Address().c, n: 16);
234 end = ptr + 16;
235 } else {
236 return false;
237 }
238
239 while (ptr < end) {
240 switch (*ptr) {
241 case 255:
242 netmask += 8;
243 ++ptr;
244 continue;
245
246 default:
247 return false; // invalid IP-style netmask
248
249 case 254:
250 ++netmask;
251 Q_FALLTHROUGH();
252 case 252:
253 ++netmask;
254 Q_FALLTHROUGH();
255 case 248:
256 ++netmask;
257 Q_FALLTHROUGH();
258 case 240:
259 ++netmask;
260 Q_FALLTHROUGH();
261 case 224:
262 ++netmask;
263 Q_FALLTHROUGH();
264 case 192:
265 ++netmask;
266 Q_FALLTHROUGH();
267 case 128:
268 ++netmask;
269 Q_FALLTHROUGH();
270 case 0:
271 break;
272 }
273 break;
274 }
275
276 // confirm that the rest is only zeroes
277 if (ptr < end && memcmp(s1: ptr + 1, s2: zeroes, n: end - ptr - 1) != 0)
278 return false;
279
280 length = netmask;
281 return true;
282}

284static void clearBits(quint8 *where, int start, int end)
285{
286 Q_ASSERT(end == 32 || end == 128);
287 if (start == end)
288 return;
289
290 // for the byte where 'start' is, clear the lower bits only
291 quint8 bytemask = 256 - (1 << (8 - (start & 7)));
292 where[start / 8] &= bytemask;
293
294 // for the tail part, clear everything
295 memset(s: where + (start + 7) / 8, c: 0, n: end / 8 - (start + 7) / 8);
296}

298QHostAddress QNetmask::address(QHostAddress::NetworkLayerProtocol protocol) const
299{
300 if (length == 255 || protocol == QHostAddress::AnyIPProtocol ||
301 protocol == QHostAddress::UnknownNetworkLayerProtocol) {
302 return QHostAddress();
303 } else if (protocol == QHostAddress::IPv4Protocol) {
304 quint32 a;
305 if (length == 0)
306 a = 0;
307 else if (length == 32)
308 a = quint32(0xffffffff);
309 else
310 a = quint32(0xffffffff) >> (32 - length) << (32 - length);
311 return QHostAddress(a);
312 } else {
313 Q_IPV6ADDR a6;
314 memset(s: a6.c, c: 0xFF, n: sizeof(a6));
315 clearBits(where: a6.c, start: length, end: 128);
316 return QHostAddress(a6);
317 }
318}

320/*!
321 \class QHostAddress
322 \brief The QHostAddress class provides an IP address.
323 \ingroup network
324 \ingroup shared
325 \inmodule QtNetwork
326
327 This class holds an IPv4 or IPv6 address in a platform- and
328 protocol-independent manner.
329
330 QHostAddress is normally used with the QTcpSocket, QTcpServer,
331 and QUdpSocket to connect to a host or to set up a server.
332
333 A host address is set with setAddress(), and retrieved with
334 toIPv4Address(), toIPv6Address(), or toString(). You can check the
335 type with protocol().
336
337 \note Please note that QHostAddress does not do DNS lookups.
338 QHostInfo is needed for that.
339
340 The class also supports common predefined addresses: \l Null, \l
341 LocalHost, \l LocalHostIPv6, \l Broadcast, and \l Any.
342
343 \sa QHostInfo, QTcpSocket, QTcpServer, QUdpSocket
344*/
345
346/*! \enum QHostAddress::SpecialAddress
347
348 \value Null The null address object. Equivalent to QHostAddress(). See also QHostAddress::isNull().
349 \value LocalHost The IPv4 localhost address. Equivalent to QHostAddress("127.0.0.1").
350 \value LocalHostIPv6 The IPv6 localhost address. Equivalent to QHostAddress("::1").
351 \value Broadcast The IPv4 broadcast address. Equivalent to QHostAddress("255.255.255.255").
352 \value AnyIPv4 The IPv4 any-address. Equivalent to QHostAddress("0.0.0.0"). A socket bound with this address will listen only on IPv4 interfaces.
353 \value AnyIPv6 The IPv6 any-address. Equivalent to QHostAddress("::"). A socket bound with this address will listen only on IPv6 interfaces.
354 \value Any The dual stack any-address. A socket bound with this address will listen on both IPv4 and IPv6 interfaces.
355*/
356
357/*! \enum QHostAddress::ConversionModeFlag
358
359 \since 5.8
360
361 \value StrictConversion Don't convert IPv6 addresses to IPv4 when comparing two QHostAddress objects of different protocols, so they will always be considered different.
362 \value ConvertV4MappedToIPv4 Convert IPv4-mapped IPv6 addresses (RFC 4291 sect. 2.5.5.2) when comparing. Therefore QHostAddress("::ffff:192.168.1.1") will compare equal to QHostAddress("192.168.1.1").
363 \value ConvertV4CompatToIPv4 Convert IPv4-compatible IPv6 addresses (RFC 4291 sect. 2.5.5.1) when comparing. Therefore QHostAddress("::192.168.1.1") will compare equal to QHostAddress("192.168.1.1").
364 \value ConvertLocalHost Convert the IPv6 loopback addresses to its IPv4 equivalent when comparing. Therefore e.g. QHostAddress("::1") will compare equal to QHostAddress("127.0.0.1").
365 \value ConvertUnspecifiedAddress All unspecified addresses will compare equal, namely AnyIPv4, AnyIPv6 and Any.
366 \value TolerantConversion Sets all three preceding flags.
367
368 \sa isEqual()
369 */

589/*!
590 \overload
591 \since 5.8
592
593 Sets the special address specified by \a address.
594*/
595void QHostAddress::setAddress(SpecialAddress address)
596{
597 clear();
598
599 Q_IPV6ADDR ip6;
600 memset(s: &ip6, c: 0, n: sizeof ip6);
601 quint32 ip4 = INADDR_ANY;
602
603 switch (address) {
604 case Null:
605 return;
606
607 case Broadcast:
608 ip4 = INADDR_BROADCAST;
609 break;
610 case LocalHost:
611 ip4 = INADDR_LOOPBACK;
612 break;
613 case AnyIPv4:
614 break;
615
616 case LocalHostIPv6:
617 ip6[15] = 1;
618 Q_FALLTHROUGH();
619 case AnyIPv6:
620 d->setAddress(ip6);
621 return;
622
623 case Any:
624 d->protocol = QHostAddress::AnyIPProtocol;
625 return;
626 }
627
628 // common IPv4 part
629 d->setAddress(ip4);
630}

632/*!
633 Returns the IPv4 address as a number.
634
635 For example, if the address is 127.0.0.1, the returned value is
636 2130706433 (i.e. 0x7f000001).
637
638 This value is valid if the protocol() is
639 \l{QAbstractSocket::}{IPv4Protocol},
640 or if the protocol is
641 \l{QAbstractSocket::}{IPv6Protocol},
642 and the IPv6 address is an IPv4 mapped address (RFC4291). In those
643 cases, \a ok will be set to true. Otherwise, it will be set to false.
644
645 \sa toString()
646*/
647quint32 QHostAddress::toIPv4Address(bool *ok) const
648{
649 quint32 dummy;
650 if (ok)
651 *ok = d->protocol == QHostAddress::IPv4Protocol || d->protocol == QHostAddress::AnyIPProtocol
652 || (d->protocol == QHostAddress::IPv6Protocol
653 && convertToIpv4(a&: dummy, a6: d->a6, mode: ConversionMode(QHostAddress::ConvertV4MappedToIPv4
654 | QHostAddress::ConvertUnspecifiedAddress)));
655 return d->a;
656}

710/*!
711 \since 4.1
712
713 Returns the scope ID of an IPv6 address. For IPv4 addresses, or if the
714 address does not contain a scope ID, an empty QString is returned.
715
716 The IPv6 scope ID specifies the scope of \e reachability for non-global
717 IPv6 addresses, limiting the area in which the address can be used. All
718 IPv6 addresses are associated with such a reachability scope. The scope ID
719 is used to disambiguate addresses that are not guaranteed to be globally
720 unique.
721
722 IPv6 specifies the following four levels of reachability:
723
724 \list
725
726 \li Node-local: Addresses that are only used for communicating with
727 services on the same interface (e.g., the loopback interface "::1").
728
729 \li Link-local: Addresses that are local to the network interface
730 (\e{link}). There is always one link-local address for each IPv6 interface
731 on your host. Link-local addresses ("fe80...") are generated from the MAC
732 address of the local network adaptor, and are not guaranteed to be unique.
733
734 \li Global: For globally routable addresses, such as public servers on the
735 Internet.
736
737 \endlist
738
739 When using a link-local or site-local address for IPv6 connections, you
740 must specify the scope ID. The scope ID for a link-local address is
741 usually the same as the interface name (e.g., "eth0", "en1") or number
742 (e.g., "1", "2").
743
744 \sa setScopeId(), QNetworkInterface, QNetworkInterface::interfaceFromName
745*/
746QString QHostAddress::scopeId() const
747{
748 return (d->protocol == QHostAddress::IPv6Protocol) ? d->scopeId : QString();
749}

751/*!
752 \since 4.1
753
754 Sets the IPv6 scope ID of the address to \a id. If the address protocol is
755 not IPv6, this function does nothing. The scope ID may be set as an
756 interface name (such as "eth0" or "en1") or as an integer representing the

781/*!
782 \since 5.8
783
784 Returns \c true if this host address is the same as the \a other address
785 given; otherwise returns \c false.
786
787 The parameter \a mode controls which conversions are performed between addresses
788 of differing protocols. If no \a mode is given, \c TolerantConversion is performed
789 by default.
790
791 \sa ConversionMode, operator==()
792 */
793bool QHostAddress::isEqual(const QHostAddress &other, ConversionMode mode) const
794{
795 if (d == other.d)
796 return true;
797
798 if (d->protocol == QHostAddress::IPv4Protocol) {
799 switch (other.d->protocol) {
800 case QHostAddress::IPv4Protocol:
801 return d->a == other.d->a;
802 case QHostAddress::IPv6Protocol:
803 quint32 a4;
804 return convertToIpv4(a&: a4, a6: other.d->a6, mode) && (a4 == d->a);
805 case QHostAddress::AnyIPProtocol:
806 return (mode & QHostAddress::ConvertUnspecifiedAddress) && d->a == 0;
807 case QHostAddress::UnknownNetworkLayerProtocol:
808 return false;
809 }
810 }
811
812 if (d->protocol == QHostAddress::IPv6Protocol) {
813 switch (other.d->protocol) {
814 case QHostAddress::IPv4Protocol:
815 quint32 a4;
816 return convertToIpv4(a&: a4, a6: d->a6, mode) && (a4 == other.d->a);
817 case QHostAddress::IPv6Protocol:
818 return memcmp(s1: &d->a6, s2: &other.d->a6, n: sizeof(Q_IPV6ADDR)) == 0;
819 case QHostAddress::AnyIPProtocol:
820 return (mode & QHostAddress::ConvertUnspecifiedAddress)
821 && (d->a6_64.c[0] == 0) && (d->a6_64.c[1] == 0);
822 case QHostAddress::UnknownNetworkLayerProtocol:
823 return false;
824 }
825 }
826
827 if ((d->protocol == QHostAddress::AnyIPProtocol)
828 && (mode & QHostAddress::ConvertUnspecifiedAddress)) {
829 switch (other.d->protocol) {
830 case QHostAddress::IPv4Protocol:
831 return other.d->a == 0;
832 case QHostAddress::IPv6Protocol:
833 return (other.d->a6_64.c[0] == 0) && (other.d->a6_64.c[1] == 0);
834 default:
835 break;
836 }
837 }
838
839 return d->protocol == other.d->protocol;
840}

842/*!
843 Returns \c true if this host address is the same as the \a other
844 address given; otherwise returns \c false.
845*/
846bool QHostAddress::operator ==(SpecialAddress other) const
847{
848 quint32 ip4 = INADDR_ANY;
849 switch (other) {
850 case Null:
851 return d->protocol == QHostAddress::UnknownNetworkLayerProtocol;

853 case Broadcast:

855 break;

892/*!
893 \since 4.5
894
895 Returns \c true if this IP is in the subnet described by the network
896 prefix \a subnet and netmask \a netmask.
897
898 An IP is considered to belong to a subnet if it is contained
899 between the lowest and the highest address in that subnet. In the
900 case of IP version 4, the lowest address is the network address,
901 while the highest address is the broadcast address.
902
903 The \a subnet argument does not have to be the actual network
904 address (the lowest address in the subnet). It can be any valid IP
905 belonging to that subnet. In particular, if it is equal to the IP
906 address held by this object, this function will always return true
907 (provided the netmask is a valid value).
908
909 \sa parseSubnet()
910*/
911bool QHostAddress::isInSubnet(const QHostAddress &subnet, int netmask) const
912{
913 if (subnet.protocol() != d->protocol || netmask < 0)
914 return false;
915
916 union {
917 quint32 ip;
918 quint8 data[4];
919 } ip4, net4;
920 const quint8 *ip;
921 const quint8 *net;
922 if (d->protocol == QHostAddress::IPv4Protocol) {
923 if (netmask > 32)
924 netmask = 32;
925 ip4.ip = qToBigEndian(source: d->a);
926 net4.ip = qToBigEndian(source: subnet.d->a);
927 ip = ip4.data;
928 net = net4.data;
929 } else if (d->protocol == QHostAddress::IPv6Protocol) {
930 if (netmask > 128)
931 netmask = 128;
932 ip = d->a6.c;
933 net = subnet.d->a6.c;
934 } else {
935 return false;
936 }
937
938 if (netmask >= 8 && memcmp(s1: ip, s2: net, n: netmask / 8) != 0)
939 return false;
940 if ((netmask & 7) == 0)
941 return true;
942
943 // compare the last octet now
944 quint8 bytemask = 256 - (1 << (8 - (netmask & 7)));
945 quint8 ipbyte = ip[netmask / 8];
946 quint8 netbyte = net[netmask / 8];
947 return (ipbyte & bytemask) == (netbyte & bytemask);
948}

965/*!
966 \since 4.5
967
968 Parses the IP and subnet information contained in \a subnet and
969 returns the network prefix for that network and its prefix length.
970
971 The IP address and the netmask must be separated by a slash
972 (/).
973
974 This function supports arguments in the form:
975 \list
976 \li 123.123.123.123/n where n is any value between 0 and 32
977 \li 123.123.123.123/255.255.255.255
978 \li <ipv6-address>/n where n is any value between 0 and 128
979 \endlist
980
981 For IP version 4, this function accepts as well missing trailing
982 components (i.e., less than 4 octets, like "192.168.1"), followed
983 or not by a dot. If the netmask is also missing in that case, it
984 is set to the number of octets actually passed (in the example
985 above, it would be 24, for 3 octets).
986
987 \sa isInSubnet()
988*/
989QPair<QHostAddress, int> QHostAddress::parseSubnet(const QString &subnet)
990{
991 // We support subnets in the form:
992 // ddd.ddd.ddd.ddd/nn
993 // ddd.ddd.ddd/nn
994 // ddd.ddd/nn
995 // ddd/nn
996 // ddd.ddd.ddd.
997 // ddd.ddd.ddd
998 // ddd.ddd.
999 // ddd.ddd
1000 // ddd.
1001 // ddd
1002 // <ipv6-address>/nn
1003 //
1004 // where nn can be an IPv4-style netmask for the IPv4 forms
1005
1006 const QPair<QHostAddress, int> invalid = qMakePair(value1: QHostAddress(), value2: -1);
1007 if (subnet.isEmpty())
1008 return invalid;
1009
1010 qsizetype slash = subnet.indexOf(c: u'/');
1011 QStringView netStr(subnet);
1012 if (slash != -1)
1013 netStr.truncate(n: slash);
1014
1015 int netmask = -1;
1016 bool isIpv6 = netStr.contains(c: u':');
1017
1018 if (slash != -1) {
1019 // is the netmask given in IP-form or in bit-count form?
1020 if (!isIpv6 && subnet.indexOf(c: u'.', from: slash + 1) != -1) {
1021 // IP-style, convert it to bit-count form
1022 QHostAddress mask;
1023 QNetmask parser;
1024 if (!mask.setAddress(subnet.mid(position: slash + 1)))
1025 return invalid;
1026 if (!parser.setAddress(mask))
1027 return invalid;
1028 netmask = parser.prefixLength();
1029 } else {
1030 bool ok;
1031 netmask = QStringView{subnet}.mid(pos: slash + 1).toUInt(ok: &ok);
1032 if (!ok)
1033 return invalid; // failed to parse the subnet
1034 }
1035 }
1036
1037 if (isIpv6) {
1038 // looks like it's an IPv6 address
1039 if (netmask > 128)
1040 return invalid; // invalid netmask
1041 if (netmask < 0)
1042 netmask = 128;
1043
1044 QHostAddress net;
1045 if (!net.setAddress(netStr.toString()))
1046 return invalid; // failed to parse the IP
1047
1048 clearBits(where: net.d->a6.c, start: netmask, end: 128);
1049 return qMakePair(value1&: net, value2&: netmask);
1050 }
1051
1052 if (netmask > 32)
1053 return invalid; // invalid netmask
1054
1055 // parse the address manually
1056 auto parts = netStr.split(sep: u'.');
1057 if (parts.isEmpty() || parts.size() > 4)
1058 return invalid; // invalid IPv4 address
1059
1060 if (parts.constLast().isEmpty())
1061 parts.removeLast();
1062
1063 quint32 addr = 0;
1064 for (int i = 0; i < parts.size(); ++i) {
1065 bool ok;
1066 uint byteValue = parts.at(i).toUInt(ok: &ok);
1067 if (!ok || byteValue > 255)
1068 return invalid; // invalid IPv4 address
1069
1070 addr <<= 8;
1071 addr += byteValue;
1072 }
1073 addr <<= 8 * (4 - parts.size());
1074 if (netmask == -1) {
1075 netmask = 8 * parts.size();
1076 } else if (netmask == 0) {
1077 // special case here
1078 // x86's instructions "shr" and "shl" do not operate when
1079 // their argument is 32, so the code below doesn't work as expected
1080 addr = 0;
1081 } else if (netmask != 32) {
1082 // clear remaining bits
1083 quint32 mask = quint32(0xffffffff) >> (32 - netmask) << (32 - netmask);
1084 addr &= mask;
1085 }
1086
1087 return qMakePair(value1: QHostAddress(addr), value2&: netmask);
1088}
