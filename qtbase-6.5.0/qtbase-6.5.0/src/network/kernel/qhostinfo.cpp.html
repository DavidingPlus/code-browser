55template <typename InputIt, typename OutputIt1, typename OutputIt2, typename UnaryPredicate>
56std::pair<OutputIt1, OutputIt2> separate_if(InputIt first, InputIt last, OutputIt1 dest1, OutputIt2 dest2, UnaryPredicate p)
57{
58 while (first != last) {
59 if (p(*first)) {
60 *dest1 = *first;
61 ++dest1;
62 } else {
63 *dest2 = *first;
64 ++dest2;
65 }
66 ++first;
67 }
68 return std::make_pair(dest1, dest2);
69}

75/*
76 The calling thread is likely the one that executes the lookup via
77 QHostInfoRunnable. Unless we operate with a queued connection already,
78 posts the QHostInfo to a dedicated QHostInfoResult object that lives in
79 the same thread as the user-provided receiver, or (if there is none) in
80 the thread that made the call to lookupHost. That QHostInfoResult object
81 then calls the user code in the correct thread.
82
83 The 'result' object deletes itself (via deleteLater) when the metacall
84 event is received.
85*/
86void QHostInfoResult::postResultsReady(const QHostInfo &info)
87{
88 // queued connection will take care of dispatching to right thread
89 if (!slotObj) {
90 emit resultsReady(info);
91 return;
92 }
93 // we used to have a context object, but it's already destroyed
94 if (withContextObject && !receiver)
95 return;
96
97 static const int signal_index = []() -> int {
98 auto senderMetaObject = &QHostInfoResult::staticMetaObject;
99 auto signal = &QHostInfoResult::resultsReady;
100 int signal_index = -1;
101 void *args[] = { &signal_index, &signal };
102 senderMetaObject->static_metacall(QMetaObject::IndexOfMethod, 0, args);
103 return signal_index + QMetaObjectPrivate::signalOffset(m: senderMetaObject);
104 }();
105
106 // a long-living version of this
107 auto result = new QHostInfoResult(this);
108 Q_CHECK_PTR(result);
109
110 const int nargs = 2;
111 auto metaCallEvent = new QMetaCallEvent(slotObj, nullptr, signal_index, nargs);
112 Q_CHECK_PTR(metaCallEvent);
113 void **args = metaCallEvent->args();
114 QMetaType *types = metaCallEvent->types();
115 auto voidType = QMetaType::fromType<void>();
116 auto hostInfoType = QMetaType::fromType<QHostInfo>();
117 types[0] = voidType;
118 types[1] = hostInfoType;
119 args[0] = nullptr;
120 args[1] = hostInfoType.create(copy: &info);
121 Q_CHECK_PTR(args[1]);
122 qApp->postEvent(receiver: result, event: metaCallEvent);
123}
124
125/*
126 Receives the event posted by postResultsReady, and calls the functor.
127*/
128bool QHostInfoResult::event(QEvent *event)
129{
130 if (event->type() == QEvent::MetaCall) {
131 Q_ASSERT(slotObj);
132 auto metaCallEvent = static_cast<QMetaCallEvent *>(event);
133 auto args = metaCallEvent->args();
134 // we didn't have a context object, or it's still alive
135 if (!withContextObject || receiver)
136 slotObj->call(r: const_cast<QObject*>(receiver.data()), a: args);
137 slotObj->destroyIfLastRef();
138
139 deleteLater();
140 return true;
141 }
142 return QObject::event(event);
143}

145/*!
146 \class QHostInfo
147 \brief The QHostInfo class provides static functions for host name lookups.
148
149 \reentrant
150 \inmodule QtNetwork
151 \ingroup network
152
153 QHostInfo finds the IP address(es) associated with a host name,
154 or the host name associated with an IP address.
155 The class provides two static convenience functions: one that
156 works asynchronously and emits a signal once the host is found,
157 and one that blocks and returns a QHostInfo object.
158
159 To look up a host's IP addresses asynchronously, call lookupHost(),
160 which takes the host name or IP address, a receiver object, and a slot
161 signature as arguments and returns an ID. You can abort the
162 lookup by calling abortHostLookup() with the lookup ID.
163
164 Example:
165
166 \snippet code/src_network_kernel_qhostinfo.cpp 0
167
168
169 The slot is invoked when the results are ready. The results are
170 stored in a QHostInfo object. Call
171 addresses() to get the list of IP addresses for the host, and
172 hostName() to get the host name that was looked up.
173
174 If the lookup failed, error() returns the type of error that
175 occurred. errorString() gives a human-readable description of the
176 lookup error.
177
178 If you want a blocking lookup, use the QHostInfo::fromName() function:
179
180 \snippet code/src_network_kernel_qhostinfo.cpp 1
181
182 QHostInfo supports Internationalized Domain Names (IDNs) through the
183 IDNA and Punycode standards.
184
185 To retrieve the name of the local host, use the static
186 QHostInfo::localHostName() function.
187
188 QHostInfo uses the mechanisms provided by the operating system
189 to perform the lookup. As per \l {RFC 6724}
190 there is no guarantee that all IP addresses registered for a domain or
191 host will be returned.
192
193 \note Since Qt 4.6.1 QHostInfo is using multiple threads for DNS lookup
194 instead of one dedicated DNS thread. This improves performance,
195 but also changes the order of signal emissions when using lookupHost()
196 compared to previous versions of Qt.
197 \note Since Qt 4.6.3 QHostInfo is using a small internal 60 second DNS cache
198 for performance improvements.
199
200 \sa QAbstractSocket, {RFC 3492}, {RFC 6724}
201*/
202
203static int nextId()
204{
205 Q_CONSTINIT static QBasicAtomicInt counter = Q_BASIC_ATOMIC_INITIALIZER(0);
206 return 1 + counter.fetchAndAddRelaxed(valueToAdd: 1);
207}

360QHostInfo QHostInfoAgent::reverseLookup(const QHostAddress &address)
361{
362 QHostInfo results;
363 // Reverse lookup
364 sockaddr_in sa4;
365 sockaddr_in6 sa6;
366 sockaddr *sa = nullptr;
367 QT_SOCKLEN_T saSize;
368 if (address.protocol() == QAbstractSocket::IPv4Protocol) {
369 sa = reinterpret_cast<sockaddr *>(&sa4);
370 saSize = sizeof(sa4);
371 memset(s: &sa4, c: 0, n: sizeof(sa4));
372 sa4.sin_family = AF_INET;
373 sa4.sin_addr.s_addr = htonl(address.toIPv4Address());
374 } else {
375 sa = reinterpret_cast<sockaddr *>(&sa6);
376 saSize = sizeof(sa6);
377 memset(s: &sa6, c: 0, n: sizeof(sa6));
378 sa6.sin6_family = AF_INET6;
379 memcpy(dest: &sa6.sin6_addr, src: address.toIPv6Address().c, n: sizeof(sa6.sin6_addr));
380 }
381
382 char hbuf[NI_MAXHOST];
383 if (sa && getnameinfo(sa: sa, salen: saSize, host: hbuf, hostlen: sizeof(hbuf), serv: nullptr, servlen: 0, flags: 0) == 0)
384 results.setHostName(QString::fromLatin1(ba: hbuf));
385
386 if (results.hostName().isEmpty())
387 results.setHostName(address.toString());
388 results.setAddresses(QList<QHostAddress>() << address);
389
390 return results;
391}

393/*
394 Call getaddrinfo, and returns the results as QHostInfo::addresses
395*/
396QHostInfo QHostInfoAgent::lookup(const QString &hostName)
397{
398 QHostInfo results;
399
400 // IDN support
401 QByteArray aceHostname = QUrl::toAce(domain: hostName);
402 results.setHostName(hostName);
403 if (aceHostname.isEmpty()) {
404 results.setError(QHostInfo::HostNotFound);
405 results.setErrorString(hostName.isEmpty() ?
406 QCoreApplication::translate(context: "QHostInfoAgent", key: "No host name given") :
407 QCoreApplication::translate(context: "QHostInfoAgent", key: "Invalid hostname"));
408 return results;
409 }
410
411 addrinfo *res = nullptr;
412 struct addrinfo hints;
413 memset(s: &hints, c: 0, n: sizeof(hints));
414 hints.ai_family = PF_UNSPEC;
415#ifdef Q_ADDRCONFIG
416 hints.ai_flags = Q_ADDRCONFIG;
417#endif
418
419 int result = getaddrinfo(name: aceHostname.constData(), service: nullptr, req: &hints, pai: &res);
420# ifdef Q_ADDRCONFIG
421 if (result == EAI_BADFLAGS) {
422 // if the lookup failed with AI_ADDRCONFIG set, try again without it
423 hints.ai_flags = 0;
424 result = getaddrinfo(name: aceHostname.constData(), service: nullptr, req: &hints, pai: &res);
425 }
426# endif
427
428 if (result == 0) {
429 addrinfo *node = res;
430 QList<QHostAddress> addresses;
431 while (node) {
432#ifdef QHOSTINFO_DEBUG
433 qDebug() << "getaddrinfo node: flags:" << node->ai_flags << "family:" << node->ai_family
434 << "ai_socktype:" << node->ai_socktype << "ai_protocol:" << node->ai_protocol
435 << "ai_addrlen:" << node->ai_addrlen;
436#endif
437 switch (node->ai_family) {
438 case AF_INET: {
439 QHostAddress addr;
440 addr.setAddress(ntohl(((sockaddr_in *) node->ai_addr)->sin_addr.s_addr));
441 if (!addresses.contains(t: addr))
442 addresses.append(t: addr);
443 break;
444 }
445 case AF_INET6: {
446 QHostAddress addr;
447 sockaddr_in6 *sa6 = (sockaddr_in6 *) node->ai_addr;
448 addr.setAddress(sa6->sin6_addr.s6_addr);
449 if (sa6->sin6_scope_id)
450 addr.setScopeId(QString::number(sa6->sin6_scope_id));
451 if (!addresses.contains(t: addr))
452 addresses.append(t: addr);
453 break;
454 }
455 default:
456 results.setError(QHostInfo::UnknownError);
457 results.setErrorString(QCoreApplication::translate(context: "QHostInfoAgent", key: "Unknown address type"));
458 }
459 node = node->ai_next;
460 }
461 if (addresses.isEmpty()) {
462 // Reached the end of the list, but no addresses were found; this
463 // means the list contains one or more unknown address types.
464 results.setError(QHostInfo::UnknownError);
465 results.setErrorString(QCoreApplication::translate(context: "QHostInfoAgent", key: "Unknown address type"));
466 }
467
468 results.setAddresses(addresses);
469 freeaddrinfo(ai: res);
470 } else {
471 switch (result) {
472#ifdef Q_OS_WIN
473 case WSAHOST_NOT_FOUND: //authoritative not found
474 case WSATRY_AGAIN: //non authoritative not found
475 case WSANO_DATA: //valid name, no associated address
476#else
477 case EAI_NONAME:
478 case EAI_FAIL:
479# ifdef EAI_NODATA // EAI_NODATA is deprecated in RFC 3493
480 case EAI_NODATA:
481# endif
482#endif
483 results.setError(QHostInfo::HostNotFound);
484 results.setErrorString(QCoreApplication::translate(context: "QHostInfoAgent", key: "Host not found"));
485 break;
486 default:
487 results.setError(QHostInfo::UnknownError);
488#ifdef Q_OS_WIN
489 results.setErrorString(QString::fromWCharArray(gai_strerror(result)));
490#else
491 results.setErrorString(QString::fromLocal8Bit(ba: gai_strerror(ecode: result)));
492#endif
493 break;
494 }
495 }
496
497#if defined(QHOSTINFO_DEBUG)
498 if (results.error() != QHostInfo::NoError) {
499 qDebug("QHostInfoAgent::fromName(): error #%d %s",
500 h_errno, results.errorString().toLatin1().constData());
501 } else {
502 QString tmp;
503 QList<QHostAddress> addresses = results.addresses();
504 for (int i = 0; i < addresses.count(); ++i) {
505 if (i != 0) tmp += ", "_L1;
506 tmp += addresses.at(i).toString();
507 }
508 qDebug("QHostInfoAgent::fromName(): found %i entries for \"%s\": {%s}",
509 addresses.count(), aceHostname.constData(),
510 tmp.toLatin1().constData());
511 }
512#endif
513
514 return results;
515}

733/*!
734 \internal
735 Called by the various lookupHost overloads to perform the lookup.
736
737 Signals either the functor encapuslated in the \a slotObj in the context
738 of \a receiver, or the \a member slot of the \a receiver.
739
740 \a receiver might be the nullptr, but only if a \a slotObj is provided.
741*/
742int QHostInfo::lookupHostImpl(const QString &name,
743 const QObject *receiver,
744 QtPrivate::QSlotObjectBase *slotObj,
745 const char *member)
746{
747#if defined QHOSTINFO_DEBUG
748 qDebug("QHostInfo::lookupHostImpl(\"%s\", %p, %p, %s)",
749 name.toLatin1().constData(), receiver, slotObj, member ? member + 1 : 0);
750#endif
751 Q_ASSERT(!member != !slotObj); // one of these must be set, but not both
752 Q_ASSERT(receiver || slotObj);
753
754 if (!QAbstractEventDispatcher::instance(thread: QThread::currentThread())) {
755 qWarning(msg: "QHostInfo::lookupHost() called with no event dispatcher");
756 return -1;
757 }
758
759 qRegisterMetaType<QHostInfo>();
760
761 int id = nextId(); // generate unique ID
762
763 if (Q_UNLIKELY(name.isEmpty())) {
764 QHostInfo hostInfo(id);
765 hostInfo.setError(QHostInfo::HostNotFound);
766 hostInfo.setErrorString(QCoreApplication::translate(context: "QHostInfo", key: "No host name given"));
767
768 QHostInfoResult result(receiver, slotObj);
769 if (receiver && member)
770 QObject::connect(sender: &result, SIGNAL(resultsReady(QHostInfo)),
771 receiver, member, Qt::QueuedConnection);
772 result.postResultsReady(info: hostInfo);

774 return id;
775 }

777#ifdef Q_OS_WASM
778 // Resolve the host name directly without using a thread or cache,
779 // since Emscripten's host lookup is fast. Emscripten maintains an internal
780 // mapping of hosts and addresses for the purposes of WebSocket socket
781 // tunnelling, and does not perform an actual host lookup.
782 QHostInfo hostInfo = QHostInfoAgent::lookup(name);
783 hostInfo.setLookupId(id);
784
785 QHostInfoResult result(receiver, slotObj);
786 if (receiver && member)
787 QObject::connect(&result, SIGNAL(resultsReady(QHostInfo)),
788 receiver, member, Qt::QueuedConnection);
789 result.postResultsReady(hostInfo);
790#else
791 QHostInfoLookupManager *manager = theHostInfoLookupManager();
792
793 if (Q_LIKELY(manager)) {
794 // the application is still alive
795 if (manager->cache.isEnabled()) {
796 // check cache first
797 bool valid = false;
798 QHostInfo info = manager->cache.get(name, valid: &valid);
799 if (valid) {
800 info.setLookupId(id);
801 QHostInfoResult result(receiver, slotObj);
802 if (receiver && member)
803 QObject::connect(sender: &result, SIGNAL(resultsReady(QHostInfo)),
804 receiver, member, Qt::QueuedConnection);
805 result.postResultsReady(info);
806 return id;
807 }
808 }
809
810 // cache is not enabled or it was not in the cache, do normal lookup
811 QHostInfoRunnable *runnable = new QHostInfoRunnable(name, id, receiver, slotObj);
812 if (receiver && member)
813 QObject::connect(sender: &runnable->resultEmitter, SIGNAL(resultsReady(QHostInfo)),
814 receiver, member, Qt::QueuedConnection);
815 manager->scheduleLookup(r: runnable);
816 }
817#endif // Q_OS_WASM
818 return id;
819}

828// the QHostInfoLookupManager will at some point call this via a QThreadPool
829void QHostInfoRunnable::run()
830{
831 QHostInfoLookupManager *manager = theHostInfoLookupManager();
832 const auto sg = qScopeGuard(f: [&] { manager->lookupFinished(r: this); });
833 // check aborted
834 if (manager->wasAborted(id))
835 return;
836
837 QHostInfo hostInfo;
838
839 // QHostInfo::lookupHost already checks the cache. However we need to check
840 // it here too because it might have been cache saved by another QHostInfoRunnable
841 // in the meanwhile while this QHostInfoRunnable was scheduled but not running
842 if (manager->cache.isEnabled()) {
843 // check the cache first
844 bool valid = false;
845 hostInfo = manager->cache.get(name: toBeLookedUp, valid: &valid);
846 if (!valid) {
847 // not in cache, we need to do the lookup and store the result in the cache
848 hostInfo = QHostInfoAgent::fromName(hostName: toBeLookedUp);
849 manager->cache.put(name: toBeLookedUp, info: hostInfo);
850 }
851 } else {
852 // cache is not enabled, just do the lookup and continue
853 hostInfo = QHostInfoAgent::fromName(hostName: toBeLookedUp);
854 }
855
856 // check aborted again
857 if (manager->wasAborted(id))
858 return;
859
860 // signal emission
861 hostInfo.setLookupId(id);
862 resultEmitter.postResultsReady(info: hostInfo);
863
864#if QT_CONFIG(thread)
865 // now also iterate through the postponed ones
866 {
867 QMutexLocker locker(&manager->mutex);
868 const auto partitionBegin = std::stable_partition(first: manager->postponedLookups.rbegin(), last: manager->postponedLookups.rend(),
869 pred: ToBeLookedUpEquals(toBeLookedUp)).base();
870 const auto partitionEnd = manager->postponedLookups.end();
871 for (auto it = partitionBegin; it != partitionEnd; ++it) {
872 QHostInfoRunnable* postponed = *it;
873 // we can now emit
874 hostInfo.setLookupId(postponed->id);
875 postponed->resultEmitter.postResultsReady(info: hostInfo);
876 delete postponed;
877 }
878 manager->postponedLookups.erase(begin: partitionBegin, end: partitionEnd);
879 }
880
881#endif
882 // thread goes back to QThreadPool
883}

925// assumes mutex is locked by caller
926void QHostInfoLookupManager::rescheduleWithMutexHeld()
927{
928 if (wasDeleted)
929 return;
930
931 // goals of this function:
932 // - launch new lookups via the thread pool
933 // - make sure only one lookup per host/IP is in progress
934
935 if (!finishedLookups.isEmpty()) {
936 // remove ID from aborted if it is in there
937 for (int i = 0; i < finishedLookups.size(); i++) {
938 abortedLookups.removeAll(t: finishedLookups.at(i)->id);
939 }
940
941 finishedLookups.clear();
942 }
943
944#if QT_CONFIG(thread)
945 auto isAlreadyRunning = [this](QHostInfoRunnable *lookup) {
946 return std::any_of(first: currentLookups.cbegin(), last: currentLookups.cend(), pred: ToBeLookedUpEquals(lookup->toBeLookedUp));
947 };
948
949 // Transfer any postponed lookups that aren't currently running to the scheduled list, keeping already-running lookups:
950 postponedLookups.erase(begin: separate_if(first: postponedLookups.begin(),
951 last: postponedLookups.end(),
952 dest1: postponedLookups.begin(),
953 dest2: std::front_inserter(x&: scheduledLookups), // prepend! we want to finish it ASAP
954 p: isAlreadyRunning).first,
955 end: postponedLookups.end());
956
957 // Unschedule and postpone any that are currently running:
958 scheduledLookups.erase(begin: separate_if(first: scheduledLookups.begin(),
959 last: scheduledLookups.end(),
960 dest1: std::back_inserter(x&: postponedLookups),
961 dest2: scheduledLookups.begin(),
962 p: isAlreadyRunning).second,
963 end: scheduledLookups.end());
964
965 const int availableThreads = threadPool.maxThreadCount() - currentLookups.size();
966 if (availableThreads > 0) {
967 int readyToStartCount = qMin(a: availableThreads, b: scheduledLookups.size());
968 auto it = scheduledLookups.begin();
969 while (readyToStartCount--) {
970 // runnable now running in new thread, track this in currentLookups
971 threadPool.start(runnable: *it);
972 currentLookups.push_back(t: std::move(*it));
973 ++it;
974 }
975 scheduledLookups.erase(begin: scheduledLookups.begin(), end: it);
976 }
977#else
978 if (!scheduledLookups.isEmpty())
979 scheduledLookups.takeFirst()->run();
980#endif
981}

996void QHostInfoLookupManager::abortLookup(int id)
997{
998 QMutexLocker locker(&this->mutex);
999
1000 if (wasDeleted)
1001 return;
1002
1003#if QT_CONFIG(thread)
1004 // is postponed? delete and return
1005 for (int i = 0; i < postponedLookups.size(); i++) {
1006 if (postponedLookups.at(i)->id == id) {
1007 delete postponedLookups.takeAt(i);
1008 return;
1009 }
1010 }
1011#endif
1012
1013 // is scheduled? delete and return
1014 for (int i = 0; i < scheduledLookups.size(); i++) {
1015 if (scheduledLookups.at(i)->id == id) {
1016 delete scheduledLookups.takeAt(i);
1017 return;
1018 }
1019 }
1020
1021 if (!abortedLookups.contains(t: id))
1022 abortedLookups.append(t: id);

1052QHostInfo qt_qhostinfo_lookup(const QString &name, QObject *receiver, const char *member, bool *valid, int *id)
1053{

1056
1057 // check cache
1058 QHostInfoLookupManager* manager = theHostInfoLookupManager();
1059 if (manager && manager->cache.isEnabled()) {
1060 QHostInfo info = manager->cache.get(name, valid);
1061 if (*valid) {
1062 return info;
1063 }
1064 }
1065
1066 // was not in cache, trigger lookup
1067 *id = QHostInfo::lookupHostImpl(name, receiver, slotObj: nullptr, member);
