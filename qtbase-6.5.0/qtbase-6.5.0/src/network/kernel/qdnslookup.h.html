141class Q_NETWORK_EXPORT QDnsLookup : public QObject
142{
143 Q_OBJECT
144 Q_PROPERTY(Error error READ error NOTIFY finished)
145 Q_PROPERTY(QString errorString READ errorString NOTIFY finished)
146 Q_PROPERTY(QString name READ name WRITE setName NOTIFY nameChanged BINDABLE bindableName)
147 Q_PROPERTY(Type type READ type WRITE setType NOTIFY typeChanged BINDABLE bindableType)
148 Q_PROPERTY(QHostAddress nameserver READ nameserver WRITE setNameserver NOTIFY nameserverChanged
149 BINDABLE bindableNameserver)
150
151public:
152 enum Error
153 {
154 NoError = 0,
155 ResolverError,
156 OperationCancelledError,
157 InvalidRequestError,
158 InvalidReplyError,
159 ServerFailureError,
160 ServerRefusedError,
161 NotFoundError
162 };
163 Q_ENUM(Error)
164
165 enum Type
166 {
167 A = 1,
168 AAAA = 28,
169 ANY = 255,
170 CNAME = 5,
171 MX = 15,
172 NS = 2,
173 PTR = 12,
174 SRV = 33,
175 TXT = 16
176 };
177 Q_ENUM(Type)
178
179 explicit QDnsLookup(QObject *parent = nullptr);
180 QDnsLookup(Type type, const QString &name, QObject *parent = nullptr);
181 QDnsLookup(Type type, const QString &name, const QHostAddress &nameserver, QObject *parent = nullptr);
182 ~QDnsLookup();
183
184 Error error() const;
185 QString errorString() const;
186 bool isFinished() const;
187
188 QString name() const;
189 void setName(const QString &name);
190 QBindable<QString> bindableName();
191
192 Type type() const;
193 void setType(QDnsLookup::Type);
194 QBindable<Type> bindableType();
195
196 QHostAddress nameserver() const;
197 void setNameserver(const QHostAddress &nameserver);
198 QBindable<QHostAddress> bindableNameserver();
199
200 QList<QDnsDomainNameRecord> canonicalNameRecords() const;
201 QList<QDnsHostAddressRecord> hostAddressRecords() const;
202 QList<QDnsMailExchangeRecord> mailExchangeRecords() const;
203 QList<QDnsDomainNameRecord> nameServerRecords() const;
204 QList<QDnsDomainNameRecord> pointerRecords() const;
205 QList<QDnsServiceRecord> serviceRecords() const;
206 QList<QDnsTextRecord> textRecords() const;
207
208
209public Q_SLOTS:
210 void abort();
211 void lookup();
212
213Q_SIGNALS:
214 void finished();
215 void nameChanged(const QString &name);
216 void typeChanged(Type type);
217 void nameserverChanged(const QHostAddress &nameserver);
218
219private:
220 Q_DECLARE_PRIVATE(QDnsLookup)
221 Q_PRIVATE_SLOT(d_func(), void _q_lookupFinished(const QDnsLookupReply &reply))
222};
