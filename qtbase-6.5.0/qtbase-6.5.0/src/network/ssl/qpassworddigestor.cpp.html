23/*!
24 \since 5.12
25
26 Returns a hash computed using the PBKDF1-algorithm as defined in
27 \l {RFC 8018, section 5.1}.
28
29 The function takes the \a data and \a salt, and then hashes it repeatedly
30 for \a iterations iterations using the specified hash \a algorithm. If the
31 resulting hash is longer than \a dkLen then it is truncated before it is
32 returned.
33
34 This function only supports SHA-1 and MD5! The max output size is 160 bits
35 (20 bytes) when using SHA-1, or 128 bits (16 bytes) when using MD5.
36 Specifying a value for \a dkLen which is greater than this results in a
37 warning and an empty QByteArray is returned. To programmatically check this
38 limit you can use \l {QCryptographicHash::hashLength}. Furthermore: the
39 \a salt must always be 8 bytes long!
40
41 \note This function is provided for use with legacy applications and all
42 new applications are recommended to use \l {deriveKeyPbkdf2} {PBKDF2}.
43
44 \sa deriveKeyPbkdf2, QCryptographicHash, QCryptographicHash::hashLength
45*/
46Q_NETWORK_EXPORT QByteArray deriveKeyPbkdf1(QCryptographicHash::Algorithm algorithm,
47 const QByteArray &data, const QByteArray &salt,
48 int iterations, quint64 dkLen)
49{
50 // https://tools.ietf.org/html/rfc8018#section-5.1
51
52 if (algorithm != QCryptographicHash::Sha1
53#ifndef QT_CRYPTOGRAPHICHASH_ONLY_SHA1
54 && algorithm != QCryptographicHash::Md5
55#endif
56 ) {
57 qWarning(msg: "The only supported algorithms for pbkdf1 are SHA-1 and MD5!");
58 return QByteArray();
59 }
60
61 if (salt.size() != 8) {
62 qWarning(msg: "The salt must be 8 bytes long!");
63 return QByteArray();
64 }
65 if (iterations < 1 || dkLen < 1)
66 return QByteArray();
67
68 if (dkLen > quint64(QCryptographicHash::hashLength(method: algorithm))) {
69 qWarning() << "Derived key too long:\n"
70 << algorithm << "was chosen which produces output of length"
71 << QCryptographicHash::hashLength(method: algorithm) << "but" << dkLen
72 << "was requested.";
73 return QByteArray();
74 }
75
76 QCryptographicHash hash(algorithm);
77 hash.addData(data);
78 hash.addData(data: salt);
79 QByteArray key = hash.result();
80
81 for (int i = 1; i < iterations; i++) {
82 hash.reset();
83 hash.addData(data: key);
84 key = hash.result();
85 }
86 return key.left(len: dkLen);
87}

89/*!
90 \since 5.12
91
92 Derive a key using the PBKDF2-algorithm as defined in
93 \l {RFC 8018, section 5.2}.
94
95 This function takes the \a data and \a salt, and then applies HMAC-X, where
96 the X is \a algorithm, repeatedly. It internally concatenates intermediate
97 results to the final output until at least \a dkLen amount of bytes have
98 been computed and it will execute HMAC-X \a iterations times each time a
99 concatenation is required. The total number of times it will execute HMAC-X
100 depends on \a iterations, \a dkLen and \a algorithm and can be calculated
101 as
102 \c{iterations * ceil(dkLen / QCryptographicHash::hashLength(algorithm))}.
103
104 \sa deriveKeyPbkdf1, QMessageAuthenticationCode, QCryptographicHash
105*/
106Q_NETWORK_EXPORT QByteArray deriveKeyPbkdf2(QCryptographicHash::Algorithm algorithm,
107 const QByteArray &data, const QByteArray &salt,
108 int iterations, quint64 dkLen)
109{
110 // https://tools.ietf.org/html/rfc8018#section-5.2
111
112 // The RFC recommends checking that 'dkLen' is not greater than '(2^32 - 1) * hLen'
113 int hashLen = QCryptographicHash::hashLength(method: algorithm);
114 const quint64 maxLen = quint64(std::numeric_limits<quint32>::max() - 1) * hashLen;
115 if (dkLen > maxLen) {
116 qWarning().nospace() << "Derived key too long:\n"
117 << algorithm << " was chosen which produces output of length "
118 << maxLen << " but " << dkLen << " was requested.";
119 return QByteArray();
120 }
121
122 if (iterations < 1 || dkLen < 1)
123 return QByteArray();
124
125 QByteArray key;
126 quint32 currentIteration = 1;
127 QMessageAuthenticationCode hmac(algorithm, data);
128 QByteArray index(4, Qt::Uninitialized);
129 while (quint64(key.size()) < dkLen) {
130 hmac.addData(data: salt);
131
132 qToBigEndian(src: currentIteration, dest: index.data());
133 hmac.addData(data: index);
134
135 QByteArray u = hmac.result();
136 hmac.reset();
137 QByteArray tkey = u;
138 for (int iter = 1; iter < iterations; iter++) {
139 hmac.addData(data: u);
140 u = hmac.result();
141 hmac.reset();
142 std::transform(first1: tkey.cbegin(), last1: tkey.cend(), first2: u.cbegin(), result: tkey.begin(),
143 binary_op: std::bit_xor<char>());
144 }
145 key += tkey;
146 currentIteration++;
147 }
148 return key.left(len: dkLen);
149}
