163 /*! \internal
164 Sets \a address and \a port in the \a aa sockaddr structure and the size in \a sockAddrSize.
165 The address \a is converted to IPv6 if the current socket protocol is also IPv6.
166 */
167 void setPortAndAddress(quint16 port, const QHostAddress &address, qt_sockaddr *aa, QT_SOCKLEN_T *sockAddrSize)
168 {
169 if (address.protocol() == QAbstractSocket::IPv6Protocol
170 || address.protocol() == QAbstractSocket::AnyIPProtocol
171 || socketProtocol == QAbstractSocket::IPv6Protocol
172 || socketProtocol == QAbstractSocket::AnyIPProtocol) {
173 memset(s: &aa->a6, c: 0, n: sizeof(sockaddr_in6));
174 aa->a6.sin6_family = AF_INET6;
175#if QT_CONFIG(networkinterface)
176 aa->a6.sin6_scope_id = scopeIdFromString(scopeid: address.scopeId());
177#endif
178 aa->a6.sin6_port = htons(port);
179 Q_IPV6ADDR tmp = address.toIPv6Address();
180 memcpy(dest: &aa->a6.sin6_addr, src: &tmp, n: sizeof(tmp));
181 *sockAddrSize = sizeof(sockaddr_in6);
182 SetSALen::set(&aa->a, sizeof(sockaddr_in6));
183 } else {
184 memset(s: &aa->a, c: 0, n: sizeof(sockaddr_in));
185 aa->a4.sin_family = AF_INET;
186 aa->a4.sin_port = htons(port);
187 aa->a4.sin_addr.s_addr = htonl(address.toIPv4Address());
188 *sockAddrSize = sizeof(sockaddr_in);
189 SetSALen::set(&aa->a, sizeof(sockaddr_in));
190 }
191 }
