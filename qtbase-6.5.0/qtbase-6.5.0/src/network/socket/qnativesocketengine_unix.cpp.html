46/*
47 Extracts the port and address from a sockaddr, and stores them in
48 \a port and \a addr if they are non-null.
49*/
50static inline void qt_socket_getPortAndAddress(const qt_sockaddr *s, quint16 *port, QHostAddress *addr)
51{
52 if (s->a.sa_family == AF_INET6) {
53 Q_IPV6ADDR tmp;
54 memcpy(dest: &tmp, src: &s->a6.sin6_addr, n: sizeof(tmp));
55 if (addr) {
56 QHostAddress tmpAddress;
57 tmpAddress.setAddress(tmp);
58 *addr = tmpAddress;
59#if QT_CONFIG(networkinterface)
60 if (s->a6.sin6_scope_id)
61 addr->setScopeId(QNetworkInterface::interfaceNameFromIndex(index: s->a6.sin6_scope_id));
62#endif
63 }
64 if (port)
65 *port = ntohs(s->a6.sin6_port);
66 return;
67 }
68
69 if (port)
70 *port = ntohs(s->a4.sin_port);
71 if (addr) {
72 QHostAddress tmpAddress;
73 tmpAddress.setAddress(ntohl(s->a4.sin_addr.s_addr));
74 *addr = tmpAddress;
75 }
76}

78static void convertToLevelAndOption(QNativeSocketEngine::SocketOption opt,
79 QAbstractSocket::NetworkLayerProtocol socketProtocol, int &level, int &n)
80{
81 n = -1;
82 level = SOL_SOCKET; // default
83
84 switch (opt) {
85 case QNativeSocketEngine::NonBlockingSocketOption: // fcntl, not setsockopt
86 case QNativeSocketEngine::BindExclusively: // not handled on Unix
87 case QNativeSocketEngine::MaxStreamsSocketOption:
88 Q_UNREACHABLE();
89
90 case QNativeSocketEngine::BroadcastSocketOption:
91 n = SO_BROADCAST;
92 break;
93 case QNativeSocketEngine::ReceiveBufferSocketOption:
94 n = SO_RCVBUF;
95 break;
96 case QNativeSocketEngine::SendBufferSocketOption:
97 n = SO_SNDBUF;
98 break;
99 case QNativeSocketEngine::AddressReusable:
100 n = SO_REUSEADDR;
101 break;
102 case QNativeSocketEngine::ReceiveOutOfBandData:
103 n = SO_OOBINLINE;
104 break;
105 case QNativeSocketEngine::LowDelayOption:
106 level = IPPROTO_TCP;
107 n = TCP_NODELAY;
108 break;
109 case QNativeSocketEngine::KeepAliveOption:
110 n = SO_KEEPALIVE;
111 break;
112 case QNativeSocketEngine::MulticastTtlOption:
113 if (socketProtocol == QAbstractSocket::IPv6Protocol || socketProtocol == QAbstractSocket::AnyIPProtocol) {
114 level = IPPROTO_IPV6;
115 n = IPV6_MULTICAST_HOPS;
116 } else
117 {
118 level = IPPROTO_IP;
119 n = IP_MULTICAST_TTL;
120 }
121 break;
122 case QNativeSocketEngine::MulticastLoopbackOption:
123 if (socketProtocol == QAbstractSocket::IPv6Protocol || socketProtocol == QAbstractSocket::AnyIPProtocol) {
124 level = IPPROTO_IPV6;
125 n = IPV6_MULTICAST_LOOP;
126 } else
127 {
128 level = IPPROTO_IP;
129 n = IP_MULTICAST_LOOP;
130 }
131 break;
132 case QNativeSocketEngine::TypeOfServiceOption:
133 if (socketProtocol == QAbstractSocket::IPv4Protocol) {
134 level = IPPROTO_IP;
135 n = IP_TOS;
136 }
137 break;
138 case QNativeSocketEngine::ReceivePacketInformation:
139 if (socketProtocol == QAbstractSocket::IPv6Protocol || socketProtocol == QAbstractSocket::AnyIPProtocol) {
140 level = IPPROTO_IPV6;
141 n = IPV6_RECVPKTINFO;
142 } else if (socketProtocol == QAbstractSocket::IPv4Protocol) {
143 level = IPPROTO_IP;
144#ifdef IP_PKTINFO
145 n = IP_PKTINFO;
146#elif defined(IP_RECVDSTADDR)
147 // variant found in QNX and FreeBSD; it will get us only the
148 // destination address, not the interface; we need IP_RECVIF for that.
149 n = IP_RECVDSTADDR;
150#endif
151 }
152 break;
153 case QNativeSocketEngine::ReceiveHopLimit:
154 if (socketProtocol == QAbstractSocket::IPv6Protocol || socketProtocol == QAbstractSocket::AnyIPProtocol) {
155 level = IPPROTO_IPV6;
156 n = IPV6_RECVHOPLIMIT;
157 } else if (socketProtocol == QAbstractSocket::IPv4Protocol) {
158#ifdef IP_RECVTTL // IP_RECVTTL is a non-standard extension supported on some OS
159 level = IPPROTO_IP;
160 n = IP_RECVTTL;
161#endif
162 }
163 break;
164
165 case QNativeSocketEngine::PathMtuInformation:
166 if (socketProtocol == QAbstractSocket::IPv6Protocol || socketProtocol == QAbstractSocket::AnyIPProtocol) {
167#ifdef IPV6_MTU
168 level = IPPROTO_IPV6;
169 n = IPV6_MTU;
170#endif
171 } else {
172#ifdef IP_MTU
173 level = IPPROTO_IP;
174 n = IP_MTU;
175#endif
176 }
177 break;
178 }
179}

181/*! \internal
182
183 Creates and returns a new socket descriptor of type \a socketType
184 and \a socketProtocol. Returns -1 on failure.
185*/
186bool QNativeSocketEnginePrivate::createNewSocket(QAbstractSocket::SocketType socketType,
187 QAbstractSocket::NetworkLayerProtocol &socketProtocol)
188{
189#ifndef QT_NO_SCTP
190 int protocol = (socketType == QAbstractSocket::SctpSocket) ? IPPROTO_SCTP : 0;
191#else
192 if (socketType == QAbstractSocket::SctpSocket) {
193 setError(error: QAbstractSocket::UnsupportedSocketOperationError,
194 errorString: ProtocolUnsupportedErrorString);
195#if defined (QNATIVESOCKETENGINE_DEBUG)
196 qDebug("QNativeSocketEnginePrivate::createNewSocket(%d, %d): unsupported protocol",
197 socketType, socketProtocol);
198#endif
199 return false;
200 }
201 int protocol = 0;
202#endif // QT_NO_SCTP
203 int domain = (socketProtocol == QAbstractSocket::IPv6Protocol
204 || socketProtocol == QAbstractSocket::AnyIPProtocol) ? AF_INET6 : AF_INET;
205 int type = (socketType == QAbstractSocket::UdpSocket) ? SOCK_DGRAM : SOCK_STREAM;
206
207 int socket = qt_safe_socket(domain, type, protocol, O_NONBLOCK);
208 if (socket < 0 && socketProtocol == QAbstractSocket::AnyIPProtocol && errno == EAFNOSUPPORT) {
209 domain = AF_INET;
210 socket = qt_safe_socket(domain, type, protocol, O_NONBLOCK);
211 socketProtocol = QAbstractSocket::IPv4Protocol;
212 }
213
214 if (socket < 0) {
215 int ecopy = errno;
216 switch (ecopy) {
217 case EPROTONOSUPPORT:
218 case EAFNOSUPPORT:
219 case EINVAL:
220 setError(error: QAbstractSocket::UnsupportedSocketOperationError, errorString: ProtocolUnsupportedErrorString);
221 break;
222 case ENFILE:
223 case EMFILE:
224 case ENOBUFS:
225 case ENOMEM:
226 setError(error: QAbstractSocket::SocketResourceError, errorString: ResourceErrorString);
227 break;
228 case EACCES:
229 setError(error: QAbstractSocket::SocketAccessError, errorString: AccessErrorString);
230 break;
231 default:
232 break;
233 }
234
235#if defined (QNATIVESOCKETENGINE_DEBUG)
236 qDebug("QNativeSocketEnginePrivate::createNewSocket(%d, %d) == false (%s)",
237 socketType, socketProtocol,
238 strerror(ecopy));
239#endif
240
241 return false;
242 }
243
244#if defined (QNATIVESOCKETENGINE_DEBUG)
245 qDebug("QNativeSocketEnginePrivate::createNewSocket(%d, %d) == true",
246 socketType, socketProtocol);
247#endif
248
249 socketDescriptor = socket;
250 this->socketProtocol = socketProtocol;
251 this->socketType = socketType;
252 return true;
253}

255/*
256 Returns the value of the socket option \a opt.
257*/
258int QNativeSocketEnginePrivate::option(QNativeSocketEngine::SocketOption opt) const
259{
260 Q_Q(const QNativeSocketEngine);
261 if (!q->isValid())
262 return -1;
263
264 // handle non-getsockopt and specific cases first
265 switch (opt) {
266 case QNativeSocketEngine::BindExclusively:
267 case QNativeSocketEngine::NonBlockingSocketOption:
268 case QNativeSocketEngine::BroadcastSocketOption:
269 return -1;
270 case QNativeSocketEngine::MaxStreamsSocketOption: {
271#ifndef QT_NO_SCTP
272 sctp_initmsg sctpInitMsg;
273 QT_SOCKOPTLEN_T sctpInitMsgSize = sizeof(sctpInitMsg);
274 if (::getsockopt(socketDescriptor, SOL_SCTP, SCTP_INITMSG, &sctpInitMsg,
275 &sctpInitMsgSize) == 0)
276 return int(qMin(sctpInitMsg.sinit_num_ostreams, sctpInitMsg.sinit_max_instreams));
277#endif
278 return -1;
279 }
280
281 case QNativeSocketEngine::PathMtuInformation:
282#if defined(IPV6_PATHMTU) && !defined(IPV6_MTU)
283 // Prefer IPV6_MTU (handled by convertToLevelAndOption), if available
284 // (Linux); fall back to IPV6_PATHMTU otherwise (FreeBSD):
285 if (socketProtocol == QAbstractSocket::IPv6Protocol) {
286 ip6_mtuinfo mtuinfo;
287 QT_SOCKOPTLEN_T len = sizeof(mtuinfo);
288 if (::getsockopt(socketDescriptor, IPPROTO_IPV6, IPV6_PATHMTU, &mtuinfo, &len) == 0)
289 return int(mtuinfo.ip6m_mtu);
290 return -1;
291 }
292#endif
293 break;
294
295 default:
296 break;
297 }
298
299 int n, level;
300 int v = 0;
301 QT_SOCKOPTLEN_T len = sizeof(v);
302
303 convertToLevelAndOption(opt, socketProtocol, level, n);
304 if (n != -1 && ::getsockopt(fd: socketDescriptor, level: level, optname: n, optval: (char *) &v, optlen: &len) != -1)
305 return len == 1 ? qFromUnaligned<quint8>(src: &v) : v;
306
307 return -1;
308}

311/*
312 Sets the socket option \a opt to \a v.
313*/
314bool QNativeSocketEnginePrivate::setOption(QNativeSocketEngine::SocketOption opt, int v)
315{
316 Q_Q(QNativeSocketEngine);
317 if (!q->isValid())
318 return false;
319
320 // handle non-setsockopt and specific cases first
321 switch (opt) {
322 case QNativeSocketEngine::NonBlockingSocketOption: {
323 // Make the socket nonblocking.
324#if !defined(Q_OS_VXWORKS)
325 int flags = ::fcntl(fd: socketDescriptor, F_GETFL, 0);
326 if (flags == -1) {
327#ifdef QNATIVESOCKETENGINE_DEBUG
328 perror("QNativeSocketEnginePrivate::setOption(): fcntl(F_GETFL) failed");
329#endif
330 return false;
331 }
332 if (::fcntl(fd: socketDescriptor, F_SETFL, flags | O_NONBLOCK) == -1) {
333#ifdef QNATIVESOCKETENGINE_DEBUG
334 perror("QNativeSocketEnginePrivate::setOption(): fcntl(F_SETFL) failed");
335#endif
336 return false;
337 }
338#else // Q_OS_VXWORKS
339 int onoff = 1;
340
341 if (qt_safe_ioctl(socketDescriptor, FIONBIO, &onoff) < 0) {
342
343#ifdef QNATIVESOCKETENGINE_DEBUG
344 perror("QNativeSocketEnginePrivate::setOption(): ioctl(FIONBIO, 1) failed");
345#endif
346 return false;
347 }
348#endif // Q_OS_VXWORKS
349 return true;
350 }
351 case QNativeSocketEngine::BindExclusively:
352 return true;
353
354 case QNativeSocketEngine::MaxStreamsSocketOption: {
355#ifndef QT_NO_SCTP
356 sctp_initmsg sctpInitMsg;
357 QT_SOCKOPTLEN_T sctpInitMsgSize = sizeof(sctpInitMsg);
358 if (::getsockopt(socketDescriptor, SOL_SCTP, SCTP_INITMSG, &sctpInitMsg,
359 &sctpInitMsgSize) == 0) {
360 sctpInitMsg.sinit_num_ostreams = sctpInitMsg.sinit_max_instreams = uint16_t(v);
361 return ::setsockopt(socketDescriptor, SOL_SCTP, SCTP_INITMSG, &sctpInitMsg,
362 sctpInitMsgSize) == 0;
363 }
364#endif
365 return false;
366 }
367
368 default:
369 break;
370 }
371
372 int n, level;
373 convertToLevelAndOption(opt, socketProtocol, level, n);
374#if defined(SO_REUSEPORT) && !defined(Q_OS_LINUX)
375 if (opt == QNativeSocketEngine::AddressReusable) {
376 // on OS X, SO_REUSEADDR isn't sufficient to allow multiple binds to the
377 // same port (which is useful for multicast UDP). SO_REUSEPORT is, but
378 // we most definitely do not want to use this for TCP. See QTBUG-6305.
379 if (socketType == QAbstractSocket::UdpSocket)
380 n = SO_REUSEPORT;
381 }
382#endif
383
384 if (n == -1)
385 return false;
386 return ::setsockopt(fd: socketDescriptor, level: level, optname: n, optval: (char *) &v, optlen: sizeof(v)) == 0;
387}

389bool QNativeSocketEnginePrivate::nativeConnect(const QHostAddress &addr, quint16 port)
390{
391#ifdef QNATIVESOCKETENGINE_DEBUG
392 qDebug() << "QNativeSocketEnginePrivate::nativeConnect() " << socketDescriptor;
393#endif
394
395 qt_sockaddr aa;
396 QT_SOCKLEN_T sockAddrSize;
397 setPortAndAddress(port, address: addr, aa: &aa, sockAddrSize: &sockAddrSize);
398
399 int connectResult = qt_safe_connect(sockfd: socketDescriptor, addr: &aa.a, addrlen: sockAddrSize);
400#if defined (QNATIVESOCKETENGINE_DEBUG)
401 int ecopy = errno;
402#endif
403 if (connectResult == -1) {
404 switch (errno) {
405 case EISCONN:
406 socketState = QAbstractSocket::ConnectedState;
407 break;
408 case ECONNREFUSED:
409 case EINVAL:
410 setError(error: QAbstractSocket::ConnectionRefusedError, errorString: ConnectionRefusedErrorString);
411 socketState = QAbstractSocket::UnconnectedState;
412 break;
413 case ETIMEDOUT:
414 setError(error: QAbstractSocket::NetworkError, errorString: ConnectionTimeOutErrorString);
415 break;
416 case EHOSTUNREACH:
417 setError(error: QAbstractSocket::NetworkError, errorString: HostUnreachableErrorString);
418 socketState = QAbstractSocket::UnconnectedState;
419 break;
420 case ENETUNREACH:
421 setError(error: QAbstractSocket::NetworkError, errorString: NetworkUnreachableErrorString);
422 socketState = QAbstractSocket::UnconnectedState;
423 break;
424 case EADDRINUSE:
425 setError(error: QAbstractSocket::NetworkError, errorString: AddressInuseErrorString);
426 break;
427 case EINPROGRESS:
428 case EALREADY:
429 setError(error: QAbstractSocket::UnfinishedSocketOperationError, errorString: InvalidSocketErrorString);
430 socketState = QAbstractSocket::ConnectingState;
431 break;
432 case EAGAIN:
433 setError(error: QAbstractSocket::UnfinishedSocketOperationError, errorString: InvalidSocketErrorString);
434 break;
435 case EACCES:
436 case EPERM:
437 setError(error: QAbstractSocket::SocketAccessError, errorString: AccessErrorString);
438 socketState = QAbstractSocket::UnconnectedState;
439 break;
440 case EAFNOSUPPORT:
441 case EBADF:
442 case EFAULT:
443 case ENOTSOCK:
444 socketState = QAbstractSocket::UnconnectedState;
445 default:
446 break;
447 }
448
449 if (socketState != QAbstractSocket::ConnectedState) {
450#if defined (QNATIVESOCKETENGINE_DEBUG)
451 qDebug("QNativeSocketEnginePrivate::nativeConnect(%s, %i) == false (%s)",
452 addr.toString().toLatin1().constData(), port,
453 socketState == QAbstractSocket::ConnectingState
454 ? "Connection in progress" : strerror(ecopy));
455#endif
456 return false;
457 }
458 }
459
460#if defined (QNATIVESOCKETENGINE_DEBUG)
461 qDebug("QNativeSocketEnginePrivate::nativeConnect(%s, %i) == true",
462 addr.toString().toLatin1().constData(), port);
463#endif
464
465 socketState = QAbstractSocket::ConnectedState;
466 return true;
467}

469bool QNativeSocketEnginePrivate::nativeBind(const QHostAddress &address, quint16 port)
470{
471 qt_sockaddr aa;
472 QT_SOCKLEN_T sockAddrSize;
473 setPortAndAddress(port, address, aa: &aa, sockAddrSize: &sockAddrSize);
474
475#ifdef IPV6_V6ONLY
476 if (aa.a.sa_family == AF_INET6) {
477 int ipv6only = 0;
478 if (address.protocol() == QAbstractSocket::IPv6Protocol)
479 ipv6only = 1;
480 //default value of this socket option varies depending on unix variant (or system configuration on BSD), so always set it explicitly
481 ::setsockopt(fd: socketDescriptor, IPPROTO_IPV6, IPV6_V6ONLY, optval: (char*)&ipv6only, optlen: sizeof(ipv6only) );
482 }
483#endif
484
485 int bindResult = QT_SOCKET_BIND(fd: socketDescriptor, addr: &aa.a, len: sockAddrSize);
486 if (bindResult < 0 && errno == EAFNOSUPPORT && address.protocol() == QAbstractSocket::AnyIPProtocol) {
487 // retry with v4
488 aa.a4.sin_family = AF_INET;
489 aa.a4.sin_port = htons(port);
490 aa.a4.sin_addr.s_addr = htonl(address.toIPv4Address());
491 sockAddrSize = sizeof(aa.a4);
492 bindResult = QT_SOCKET_BIND(fd: socketDescriptor, addr: &aa.a, len: sockAddrSize);
493 }
494
495 if (bindResult < 0) {
496#if defined (QNATIVESOCKETENGINE_DEBUG)
497 int ecopy = errno;
498#endif
499 switch(errno) {
500 case EADDRINUSE:
501 setError(error: QAbstractSocket::AddressInUseError, errorString: AddressInuseErrorString);
502 break;
503 case EACCES:
504 setError(error: QAbstractSocket::SocketAccessError, errorString: AddressProtectedErrorString);
505 break;
506 case EINVAL:
507 setError(error: QAbstractSocket::UnsupportedSocketOperationError, errorString: OperationUnsupportedErrorString);
508 break;
509 case EADDRNOTAVAIL:
510 setError(error: QAbstractSocket::SocketAddressNotAvailableError, errorString: AddressNotAvailableErrorString);
511 break;
512 default:
513 break;
514 }
515
516#if defined (QNATIVESOCKETENGINE_DEBUG)
517 qDebug("QNativeSocketEnginePrivate::nativeBind(%s, %i) == false (%s)",
518 address.toString().toLatin1().constData(), port, strerror(ecopy));
519#endif
520
521 return false;
522 }
523
524#if defined (QNATIVESOCKETENGINE_DEBUG)
525 qDebug("QNativeSocketEnginePrivate::nativeBind(%s, %i) == true",
526 address.toString().toLatin1().constData(), port);
527#endif
528 socketState = QAbstractSocket::BoundState;
529 return true;
530}

579#if !defined(Q_OS_OPENBSD)
580 case EPROTO:
581#endif
582 case EAFNOSUPPORT:
583 case EINVAL:
584 setError(error: QAbstractSocket::UnsupportedSocketOperationError, errorString: ProtocolUnsupportedErrorString);
585 break;
586 case ENFILE:
587 case EMFILE:
588 case ENOBUFS:
589 case ENOMEM:
590 setError(error: QAbstractSocket::SocketResourceError, errorString: ResourceErrorString);
591 break;
592 case EACCES:
593 case EPERM:
594 setError(error: QAbstractSocket::SocketAccessError, errorString: AccessErrorString);

605 }
606 }

613static bool multicastMembershipHelper(QNativeSocketEnginePrivate *d,
614 int how6,
615 int how4,
616 const QHostAddress &groupAddress,
617 const QNetworkInterface &interface)
618{
619 int level = 0;
620 int sockOpt = 0;
621 void *sockArg;
622 int sockArgSize;
623
624 ip_mreq mreq4;
625 ipv6_mreq mreq6;
626
627 if (groupAddress.protocol() == QAbstractSocket::IPv6Protocol) {
628 level = IPPROTO_IPV6;
629 sockOpt = how6;
630 sockArg = &mreq6;
631 sockArgSize = sizeof(mreq6);
632 memset(s: &mreq6, c: 0, n: sizeof(mreq6));
633 Q_IPV6ADDR ip6 = groupAddress.toIPv6Address();
634 memcpy(dest: &mreq6.ipv6mr_multiaddr, src: &ip6, n: sizeof(ip6));
635 mreq6.ipv6mr_interface = interface.index();
636 } else if (groupAddress.protocol() == QAbstractSocket::IPv4Protocol) {
637 level = IPPROTO_IP;
638 sockOpt = how4;
639 sockArg = &mreq4;
640 sockArgSize = sizeof(mreq4);
641 memset(s: &mreq4, c: 0, n: sizeof(mreq4));
642 mreq4.imr_multiaddr.s_addr = htonl(groupAddress.toIPv4Address());
643
644 if (interface.isValid()) {
645 const QList<QNetworkAddressEntry> addressEntries = interface.addressEntries();
646 bool found = false;
647 for (const QNetworkAddressEntry &entry : addressEntries) {
648 const QHostAddress ip = entry.ip();
649 if (ip.protocol() == QAbstractSocket::IPv4Protocol) {
650 mreq4.imr_interface.s_addr = htonl(ip.toIPv4Address());
651 found = true;
652 break;
653 }
654 }
655 if (!found) {
656 d->setError(error: QAbstractSocket::NetworkError,
657 errorString: QNativeSocketEnginePrivate::NetworkUnreachableErrorString);
658 return false;
659 }
660 } else {
661 mreq4.imr_interface.s_addr = INADDR_ANY;
662 }
663 } else {
664 // unreachable
665 d->setError(error: QAbstractSocket::UnsupportedSocketOperationError,
666 errorString: QNativeSocketEnginePrivate::ProtocolUnsupportedErrorString);
667 return false;
668 }
669
670 int res = setsockopt(fd: d->socketDescriptor, level: level, optname: sockOpt, optval: sockArg, optlen: sockArgSize);
671 if (res == -1) {
672 switch (errno) {
673 case ENOPROTOOPT:
674 d->setError(error: QAbstractSocket::UnsupportedSocketOperationError,
675 errorString: QNativeSocketEnginePrivate::OperationUnsupportedErrorString);
676 break;
677 case EADDRNOTAVAIL:
678 d->setError(error: QAbstractSocket::SocketAddressNotAvailableError,
679 errorString: QNativeSocketEnginePrivate::AddressNotAvailableErrorString);
680 break;
681 default:
682 d->setError(error: QAbstractSocket::UnknownSocketError,

797bool QNativeSocketEnginePrivate::nativeHasPendingDatagrams() const
798{
799 // Peek 1 bytes into the next message.
800 ssize_t readBytes;
801 char c;
802 EINTR_LOOP(readBytes, ::recv(socketDescriptor, &c, 1, MSG_PEEK));
803
804 // If there's no error, or if our buffer was too small, there must be a
805 // pending datagram.
806 bool result = (readBytes != -1) || errno == EMSGSIZE;
807
808#if defined (QNATIVESOCKETENGINE_DEBUG)
809 qDebug("QNativeSocketEnginePrivate::nativeHasPendingDatagrams() == %s",
810 result ? "true" : "false");
811#endif
812 return result;
813}
814
815qint64 QNativeSocketEnginePrivate::nativePendingDatagramSize() const
816{
817 ssize_t recvResult = -1;
818#ifdef Q_OS_LINUX
819 // Linux can return the actual datagram size if we use MSG_TRUNC
820 char c;
821 EINTR_LOOP(recvResult, ::recv(socketDescriptor, &c, 1, MSG_PEEK | MSG_TRUNC));
822#elif defined(SO_NREAD)
823 // macOS can return the actual datagram size if we use SO_NREAD
824 int value;
825 socklen_t valuelen = sizeof(value);
826 recvResult = getsockopt(socketDescriptor, SOL_SOCKET, SO_NREAD, &value, &valuelen);
827 if (recvResult != -1)
828 recvResult = value;
829#else
830 // We need to grow the buffer to fit the entire datagram.
831 // We start at 1500 bytes (the MTU for Ethernet V2), which should catch
832 // almost all uses (effective MTU for UDP under IPv4 is 1468), except
833 // for localhost datagrams and those reassembled by the IP layer.
834 char udpMessagePeekBuffer[1500];
835 struct msghdr msg;
836 struct iovec vec;
837
838 memset(&msg, 0, sizeof(msg));
839 msg.msg_iov = &vec;
840 msg.msg_iovlen = 1;
841 vec.iov_base = udpMessagePeekBuffer;
842 vec.iov_len = sizeof(udpMessagePeekBuffer);
843
844 for (;;) {
845 // the data written to udpMessagePeekBuffer is discarded, so
846 // this function is still reentrant although it might not look
847 // so.
848 recvResult = ::recvmsg(socketDescriptor, &msg, MSG_PEEK);
849 if (recvResult == -1 && errno == EINTR)
850 continue;
851
852 // was the result truncated?
853 if ((msg.msg_flags & MSG_TRUNC) == 0)
854 break;
855
856 // grow by 16 times
857 msg.msg_iovlen *= 16;
858 if (msg.msg_iov != &vec)
859 delete[] msg.msg_iov;
860 msg.msg_iov = new struct iovec[msg.msg_iovlen];
861 std::fill_n(msg.msg_iov, msg.msg_iovlen, vec);
862 }
863
864 if (msg.msg_iov != &vec)
865 delete[] msg.msg_iov;
866#endif
867
868#if defined (QNATIVESOCKETENGINE_DEBUG)
869 qDebug("QNativeSocketEnginePrivate::nativePendingDatagramSize() == %zd", recvResult);
870#endif
871
872 return qint64(recvResult);
873}

875qint64 QNativeSocketEnginePrivate::nativeReceiveDatagram(char *data, qint64 maxSize, QIpPacketHeader *header,
876 QAbstractSocketEngine::PacketHeaderOptions options)
877{
878 // we use quintptr to force the alignment
879 quintptr cbuf[(CMSG_SPACE(sizeof(struct in6_pktinfo)) + CMSG_SPACE(sizeof(int))
880#if !defined(IP_PKTINFO) && defined(IP_RECVIF) && defined(Q_OS_BSD4)
881 + CMSG_SPACE(sizeof(sockaddr_dl))
882#endif
883#ifndef QT_NO_SCTP
884 + CMSG_SPACE(sizeof(struct sctp_sndrcvinfo))
885#endif
886 + sizeof(quintptr) - 1) / sizeof(quintptr)];
887
888 struct msghdr msg;
889 struct iovec vec;
890 qt_sockaddr aa;
891 char c;
892 memset(s: &msg, c: 0, n: sizeof(msg));
893 memset(s: &aa, c: 0, n: sizeof(aa));
894
895 // we need to receive at least one byte, even if our user isn't interested in it
896 vec.iov_base = maxSize ? data : &c;
897 vec.iov_len = maxSize ? maxSize : 1;
898 msg.msg_iov = &vec;
899 msg.msg_iovlen = 1;
900 if (options & QAbstractSocketEngine::WantDatagramSender) {
901 msg.msg_name = &aa;
902 msg.msg_namelen = sizeof(aa);
903 }
904 if (options & (QAbstractSocketEngine::WantDatagramHopLimit | QAbstractSocketEngine::WantDatagramDestination
905 | QAbstractSocketEngine::WantStreamNumber)) {
906 msg.msg_control = cbuf;
907 msg.msg_controllen = sizeof(cbuf);
908 }
909
910 ssize_t recvResult = 0;
911 do {
912 recvResult = ::recvmsg(fd: socketDescriptor, message: &msg, flags: 0);
913 } while (recvResult == -1 && errno == EINTR);
914
915 if (recvResult == -1) {
916 switch (errno) {
917#if defined(EWOULDBLOCK) && EWOULDBLOCK != EAGAIN
918 case EWOULDBLOCK:
919#endif
920 case EAGAIN:
921 // No datagram was available for reading
922 recvResult = -2;
923 break;
924 case ECONNREFUSED:
925 setError(error: QAbstractSocket::ConnectionRefusedError, errorString: ConnectionRefusedErrorString);
926 break;
927 default:
928 setError(error: QAbstractSocket::NetworkError, errorString: ReceiveDatagramErrorString);
929 }
930 if (header)
931 header->clear();
932 } else if (options != QAbstractSocketEngine::WantNone) {
933 Q_ASSERT(header);
934 qt_socket_getPortAndAddress(s: &aa, port: &header->senderPort, addr: &header->senderAddress);
935 header->destinationPort = localPort;
936 header->endOfRecord = (msg.msg_flags & MSG_EOR) != 0;
937
938 // parse the ancillary data
939 struct cmsghdr *cmsgptr;
940 QT_WARNING_PUSH
941 QT_WARNING_DISABLE_CLANG("-Wsign-compare")
942 for (cmsgptr = CMSG_FIRSTHDR(&msg); cmsgptr != nullptr;
943 cmsgptr = CMSG_NXTHDR(&msg, cmsgptr)) {
944 QT_WARNING_POP
945 if (cmsgptr->cmsg_level == IPPROTO_IPV6 && cmsgptr->cmsg_type == IPV6_PKTINFO
946 && cmsgptr->cmsg_len >= CMSG_LEN(sizeof(in6_pktinfo))) {
947 in6_pktinfo *info = reinterpret_cast<in6_pktinfo *>(CMSG_DATA(cmsgptr));
948
949 header->destinationAddress.setAddress(reinterpret_cast<quint8 *>(&info->ipi6_addr));
950 header->ifindex = info->ipi6_ifindex;
951 if (header->ifindex)
952 header->destinationAddress.setScopeId(QString::number(info->ipi6_ifindex));
953 }
954
955#ifdef IP_PKTINFO
956 if (cmsgptr->cmsg_level == IPPROTO_IP && cmsgptr->cmsg_type == IP_PKTINFO
957 && cmsgptr->cmsg_len >= CMSG_LEN(sizeof(in_pktinfo))) {
958 in_pktinfo *info = reinterpret_cast<in_pktinfo *>(CMSG_DATA(cmsgptr));
959
960 header->destinationAddress.setAddress(ntohl(info->ipi_addr.s_addr));
961 header->ifindex = info->ipi_ifindex;
962 }
963#else
964# ifdef IP_RECVDSTADDR
965 if (cmsgptr->cmsg_level == IPPROTO_IP && cmsgptr->cmsg_type == IP_RECVDSTADDR
966 && cmsgptr->cmsg_len >= CMSG_LEN(sizeof(in_addr))) {
967 in_addr *addr = reinterpret_cast<in_addr *>(CMSG_DATA(cmsgptr));
968
969 header->destinationAddress.setAddress(ntohl(addr->s_addr));
970 }
971# endif
972# if defined(IP_RECVIF) && defined(Q_OS_BSD4)
973 if (cmsgptr->cmsg_level == IPPROTO_IP && cmsgptr->cmsg_type == IP_RECVIF
974 && cmsgptr->cmsg_len >= CMSG_LEN(sizeof(sockaddr_dl))) {
975 sockaddr_dl *sdl = reinterpret_cast<sockaddr_dl *>(CMSG_DATA(cmsgptr));
976 header->ifindex = sdl->sdl_index;
977 }
978# endif
979#endif
980
981 if (cmsgptr->cmsg_len == CMSG_LEN(sizeof(int))
982 && ((cmsgptr->cmsg_level == IPPROTO_IPV6 && cmsgptr->cmsg_type == IPV6_HOPLIMIT)
983 || (cmsgptr->cmsg_level == IPPROTO_IP && cmsgptr->cmsg_type == IP_TTL))) {
984 static_assert(sizeof(header->hopLimit) == sizeof(int));
985 memcpy(dest: &header->hopLimit, CMSG_DATA(cmsgptr), n: sizeof(header->hopLimit));
986 }
987
988#ifndef QT_NO_SCTP
989 if (cmsgptr->cmsg_level == IPPROTO_SCTP && cmsgptr->cmsg_type == SCTP_SNDRCV
990 && cmsgptr->cmsg_len >= CMSG_LEN(sizeof(sctp_sndrcvinfo))) {
991 sctp_sndrcvinfo *rcvInfo = reinterpret_cast<sctp_sndrcvinfo *>(CMSG_DATA(cmsgptr));
992
993 header->streamNumber = int(rcvInfo->sinfo_stream);
994 }
995#endif
996 }
997 }
998
999#if defined (QNATIVESOCKETENGINE_DEBUG)
1000 qDebug("QNativeSocketEnginePrivate::nativeReceiveDatagram(%p \"%s\", %lli, %s, %i) == %lli",
1001 data, QtDebugUtils::toPrintable(data, recvResult, 16).constData(), maxSize,
1002 (recvResult != -1 && options != QAbstractSocketEngine::WantNone)
1003 ? header->senderAddress.toString().toLatin1().constData() : "(unknown)",
1004 (recvResult != -1 && options != QAbstractSocketEngine::WantNone)
1005 ? header->senderPort : 0, (qint64) recvResult);
1006#endif
1007
1008 return qint64((maxSize || recvResult < 0) ? recvResult : Q_INT64_C(0));
1009}

1011qint64 QNativeSocketEnginePrivate::nativeSendDatagram(const char *data, qint64 len, const QIpPacketHeader &header)
1012{
1013 // we use quintptr to force the alignment
1014 quintptr cbuf[(CMSG_SPACE(sizeof(struct in6_pktinfo)) + CMSG_SPACE(sizeof(int))
1015#ifndef QT_NO_SCTP
1016 + CMSG_SPACE(sizeof(struct sctp_sndrcvinfo))
1017#endif
1018 + sizeof(quintptr) - 1) / sizeof(quintptr)];
1019
1020 struct cmsghdr *cmsgptr = reinterpret_cast<struct cmsghdr *>(cbuf);
1021 struct msghdr msg;
1022 struct iovec vec;
1023 qt_sockaddr aa;
1024
1025 memset(s: &msg, c: 0, n: sizeof(msg));
1026 memset(s: &aa, c: 0, n: sizeof(aa));
1027 vec.iov_base = const_cast<char *>(data);
1028 vec.iov_len = len;
1029 msg.msg_iov = &vec;
1030 msg.msg_iovlen = 1;
1031 msg.msg_control = &cbuf;
1032
1033 if (header.destinationPort != 0) {
1034 msg.msg_name = &aa.a;
1035 setPortAndAddress(port: header.destinationPort, address: header.destinationAddress,
1036 aa: &aa, sockAddrSize: &msg.msg_namelen);
1037 }
1038
1039 if (msg.msg_namelen == sizeof(aa.a6)) {
1040 if (header.hopLimit != -1) {
1041 msg.msg_controllen += CMSG_SPACE(sizeof(int));
1042 cmsgptr->cmsg_len = CMSG_LEN(sizeof(int));
1043 cmsgptr->cmsg_level = IPPROTO_IPV6;
1044 cmsgptr->cmsg_type = IPV6_HOPLIMIT;
1045 memcpy(CMSG_DATA(cmsgptr), src: &header.hopLimit, n: sizeof(int));
1046 cmsgptr = reinterpret_cast<cmsghdr *>(reinterpret_cast<char *>(cmsgptr) + CMSG_SPACE(sizeof(int)));
1047 }
1048 if (header.ifindex != 0 || !header.senderAddress.isNull()) {
1049 struct in6_pktinfo *data = reinterpret_cast<in6_pktinfo *>(CMSG_DATA(cmsgptr));
1050 memset(s: data, c: 0, n: sizeof(*data));
1051 msg.msg_controllen += CMSG_SPACE(sizeof(*data));
1052 cmsgptr->cmsg_len = CMSG_LEN(sizeof(*data));
1053 cmsgptr->cmsg_level = IPPROTO_IPV6;
1054 cmsgptr->cmsg_type = IPV6_PKTINFO;
1055 data->ipi6_ifindex = header.ifindex;
1056
1057 QIPv6Address tmp = header.senderAddress.toIPv6Address();
1058 memcpy(dest: &data->ipi6_addr, src: &tmp, n: sizeof(tmp));
1059 cmsgptr = reinterpret_cast<cmsghdr *>(reinterpret_cast<char *>(cmsgptr) + CMSG_SPACE(sizeof(*data)));
1060 }
1061 } else {
1062 if (header.hopLimit != -1) {
1063 msg.msg_controllen += CMSG_SPACE(sizeof(int));
1064 cmsgptr->cmsg_len = CMSG_LEN(sizeof(int));
1065 cmsgptr->cmsg_level = IPPROTO_IP;
1066 cmsgptr->cmsg_type = IP_TTL;
1067 memcpy(CMSG_DATA(cmsgptr), src: &header.hopLimit, n: sizeof(int));
1068 cmsgptr = reinterpret_cast<cmsghdr *>(reinterpret_cast<char *>(cmsgptr) + CMSG_SPACE(sizeof(int)));
1069 }
1070
1071#if defined(IP_PKTINFO) || defined(IP_SENDSRCADDR)
1072 if (header.ifindex != 0 || !header.senderAddress.isNull()) {
1073# ifdef IP_PKTINFO
1074 struct in_pktinfo *data = reinterpret_cast<in_pktinfo *>(CMSG_DATA(cmsgptr));
1075 memset(s: data, c: 0, n: sizeof(*data));
1076 cmsgptr->cmsg_type = IP_PKTINFO;
1077 data->ipi_ifindex = header.ifindex;
1078 data->ipi_addr.s_addr = htonl(header.senderAddress.toIPv4Address());
1079# elif defined(IP_SENDSRCADDR)
1080 struct in_addr *data = reinterpret_cast<in_addr *>(CMSG_DATA(cmsgptr));
1081 cmsgptr->cmsg_type = IP_SENDSRCADDR;
1082 data->s_addr = htonl(header.senderAddress.toIPv4Address());
1083# endif
1084 cmsgptr->cmsg_level = IPPROTO_IP;
1085 msg.msg_controllen += CMSG_SPACE(sizeof(*data));
1086 cmsgptr->cmsg_len = CMSG_LEN(sizeof(*data));
1087 cmsgptr = reinterpret_cast<cmsghdr *>(reinterpret_cast<char *>(cmsgptr) + CMSG_SPACE(sizeof(*data)));
1088 }
1089#endif
1090 }
1091
1092#ifndef QT_NO_SCTP
1093 if (header.streamNumber != -1) {
1094 struct sctp_sndrcvinfo *data = reinterpret_cast<sctp_sndrcvinfo *>(CMSG_DATA(cmsgptr));
1095 memset(data, 0, sizeof(*data));
1096 msg.msg_controllen += CMSG_SPACE(sizeof(sctp_sndrcvinfo));
1097 cmsgptr->cmsg_len = CMSG_LEN(sizeof(sctp_sndrcvinfo));
1098 cmsgptr->cmsg_level = IPPROTO_SCTP;
1099 cmsgptr->cmsg_type = SCTP_SNDRCV;
1100 data->sinfo_stream = uint16_t(header.streamNumber);
1101 cmsgptr = reinterpret_cast<cmsghdr *>(reinterpret_cast<char *>(cmsgptr) + CMSG_SPACE(sizeof(*data)));
1102 }
1103#endif
1104
1105 if (msg.msg_controllen == 0)
1106 msg.msg_control = nullptr;
1107 ssize_t sentBytes = qt_safe_sendmsg(sockfd: socketDescriptor, msg: &msg, flags: 0);
1108
1109 if (sentBytes < 0) {
1110 switch (errno) {
1111#if defined(EWOULDBLOCK) && EWOULDBLOCK != EAGAIN
1112 case EWOULDBLOCK:
1113#endif
1114 case EAGAIN:
1115 sentBytes = -2;
1116 break;
1117 case EMSGSIZE:
1118 setError(error: QAbstractSocket::DatagramTooLargeError, errorString: DatagramTooLargeErrorString);
1119 break;
1120 case ECONNRESET:
1121 setError(error: QAbstractSocket::RemoteHostClosedError, errorString: RemoteHostClosedErrorString);
1122 break;
1123 default:
1124 setError(error: QAbstractSocket::NetworkError, errorString: SendDatagramErrorString);
1125 }
1126 }
1127
1128#if defined (QNATIVESOCKETENGINE_DEBUG)
1129 qDebug("QNativeSocketEngine::sendDatagram(%p \"%s\", %lli, \"%s\", %i) == %lli", data,
1130 QtDebugUtils::toPrintable(data, len, 16).constData(), len,
1131 header.destinationAddress.toString().toLatin1().constData(),
1132 header.destinationPort, (qint64) sentBytes);
1133#endif
1134
1135 return qint64(sentBytes);
1136}
1137
1138bool QNativeSocketEnginePrivate::fetchConnectionParameters()
1139{
1140 localPort = 0;
1141 localAddress.clear();
1142 peerPort = 0;
1143 peerAddress.clear();
1144 inboundStreamCount = outboundStreamCount = 0;
1145
1146 if (socketDescriptor == -1)
1147 return false;
1148
1149 qt_sockaddr sa;
1150 QT_SOCKLEN_T sockAddrSize = sizeof(sa);
1151
1152 // Determine local address
1153 memset(s: &sa, c: 0, n: sizeof(sa));
1154 if (::getsockname(fd: socketDescriptor, addr: &sa.a, len: &sockAddrSize) == 0) {
1155 qt_socket_getPortAndAddress(s: &sa, port: &localPort, addr: &localAddress);
1156
1157 // Determine protocol family
1158 switch (sa.a.sa_family) {
1159 case AF_INET:
1160 socketProtocol = QAbstractSocket::IPv4Protocol;
1161 break;
1162 case AF_INET6:
1163 socketProtocol = QAbstractSocket::IPv6Protocol;
1164 break;
1165 default:
1166 socketProtocol = QAbstractSocket::UnknownNetworkLayerProtocol;
1167 break;
1168 }
1169
1170 } else if (errno == EBADF) {
1171 setError(error: QAbstractSocket::UnsupportedSocketOperationError, errorString: InvalidSocketErrorString);
1172 return false;
1173 }
1174
1175#if defined (IPV6_V6ONLY)
1176 // determine if local address is dual mode
1177 // On linux, these are returned as "::" (==AnyIPv6)
1178 // On OSX, these are returned as "::FFFF:0.0.0.0" (==AnyIPv4)
1179 // in either case, the IPV6_V6ONLY option is cleared
1180 int ipv6only = 0;
1181 socklen_t optlen = sizeof(ipv6only);
1182 if (socketProtocol == QAbstractSocket::IPv6Protocol
1183 && (localAddress == QHostAddress::AnyIPv4 || localAddress == QHostAddress::AnyIPv6)
1184 && !getsockopt(fd: socketDescriptor, IPPROTO_IPV6, IPV6_V6ONLY, optval: (char*)&ipv6only, optlen: &optlen )) {
1185 if (optlen != sizeof(ipv6only))
1186 qWarning(msg: "unexpected size of IPV6_V6ONLY socket option");
1187 if (!ipv6only) {
1188 socketProtocol = QAbstractSocket::AnyIPProtocol;
1189 localAddress = QHostAddress::Any;
1190 }
1191 }
1192#endif
1193
1194 // Determine the remote address
1195 bool connected = ::getpeername(fd: socketDescriptor, addr: &sa.a, len: &sockAddrSize) == 0;
1196 if (connected) {
1197 qt_socket_getPortAndAddress(s: &sa, port: &peerPort, addr: &peerAddress);
1198 inboundStreamCount = outboundStreamCount = 1;
1199 }
1200
1201 // Determine the socket type (UDP/TCP/SCTP)
1202 int value = 0;
1203 QT_SOCKOPTLEN_T valueSize = sizeof(int);
1204 if (::getsockopt(fd: socketDescriptor, SOL_SOCKET, SO_TYPE, optval: &value, optlen: &valueSize) == 0) {
1205 if (value == SOCK_STREAM) {
1206#ifndef QT_NO_SCTP
1207 if (option(QNativeSocketEngine::MaxStreamsSocketOption) != -1) {
1208 socketType = QAbstractSocket::SctpSocket;
1209 if (connected) {
1210 sctp_status sctpStatus;
1211 QT_SOCKOPTLEN_T sctpStatusSize = sizeof(sctpStatus);
1212 sctp_event_subscribe sctpEvents;
1213
1214 memset(&sctpEvents, 0, sizeof(sctpEvents));
1215 sctpEvents.sctp_data_io_event = 1;
1216 if (::getsockopt(socketDescriptor, SOL_SCTP, SCTP_STATUS, &sctpStatus,
1217 &sctpStatusSize) == 0 &&
1218 ::setsockopt(socketDescriptor, SOL_SCTP, SCTP_EVENTS, &sctpEvents,
1219 sizeof(sctpEvents)) == 0) {
1220 inboundStreamCount = int(sctpStatus.sstat_instrms);
1221 outboundStreamCount = int(sctpStatus.sstat_outstrms);
1222 } else {
1223 setError(QAbstractSocket::UnsupportedSocketOperationError,
1224 InvalidSocketErrorString);
1225 return false;
1226 }
1227 }
1228 } else {
1229 socketType = QAbstractSocket::TcpSocket;
