113/*
114 inserts the host address in buf at pos and updates pos.
115 if the func fails the data in buf and the value of pos is undefined
116*/
117static bool qt_socks5_set_host_address_and_port(const QHostAddress &address, quint16 port, QByteArray *pBuf)
118{
119 QSOCKS5_DEBUG << "setting [" << address << ':' << port << ']';
120
121 union {
122 quint16 port;
123 quint32 ipv4;
124 QIPv6Address ipv6;
125 char ptr;
126 } data;
127
128 // add address
129 if (address.protocol() == QAbstractSocket::IPv4Protocol) {
130 data.ipv4 = qToBigEndian<quint32>(source: address.toIPv4Address());
131 pBuf->append(S5_IP_V4);
132 pBuf->append(a: QByteArray::fromRawData(data: &data.ptr, size: sizeof data.ipv4));
133 } else if (address.protocol() == QAbstractSocket::IPv6Protocol) {
134 data.ipv6 = address.toIPv6Address();
135 pBuf->append(S5_IP_V6);
136 pBuf->append(a: QByteArray::fromRawData(data: &data.ptr, size: sizeof data.ipv6));
137 } else {
138 return false;
139 }
140
141 // add port
142 data.port = qToBigEndian<quint16>(source: port);
143 pBuf->append(a: QByteArray::fromRawData(data: &data.ptr, size: sizeof data.port));
144 return true;
145}

176/*
177 retrieves the host address in buf at pos and updates pos.
178 return 1 if OK, 0 if need more data, -1 if error
179 if the func fails the value of the address and the pos is undefined
180*/
181static int qt_socks5_get_host_address_and_port(const QByteArray &buf, QHostAddress *pAddress, quint16 *pPort, int *pPos)
182{
183 int ret = -1;
184 int pos = *pPos;
185 const unsigned char *pBuf = reinterpret_cast<const unsigned char*>(buf.constData());
186 QHostAddress address;
187 quint16 port = 0;
188
189 if (buf.size() - pos < 1) {
190 QSOCKS5_DEBUG << "need more data address/port";
191 return 0;
192 }
193 if (pBuf[pos] == S5_IP_V4) {
194 pos++;
195 if (buf.size() - pos < 4) {
196 QSOCKS5_DEBUG << "need more data for ip4 address";
197 return 0;
198 }
199 address.setAddress(qFromBigEndian<quint32>(src: &pBuf[pos]));
200 pos += 4;
201 ret = 1;
202 } else if (pBuf[pos] == S5_IP_V6) {
203 pos++;
204 if (buf.size() - pos < 16) {
205 QSOCKS5_DEBUG << "need more data for ip6 address";
206 return 0;
207 }
208 QIPv6Address add;
209 for (int i = 0; i < 16; ++i)
210 add[i] = buf[pos++];
211 address.setAddress(add);
212 ret = 1;
213 } else if (pBuf[pos] == S5_DOMAINNAME){
214 // just skip it
215 pos++;
216 qDebug() << "skipping hostname of len" << uint(pBuf[pos]);
217 pos += uchar(pBuf[pos]);
218 } else {
219 QSOCKS5_DEBUG << "invalid address type" << (int)pBuf[pos];
220 ret = -1;
221 }
222
223 if (ret == 1) {
224 if (buf.size() - pos < 2) {
225 QSOCKS5_DEBUG << "need more data for port";
226 return 0;
227 }
228 port = qFromBigEndian<quint16>(src: &pBuf[pos]);
229 pos += 2;
230 }
231
232 if (ret == 1) {
233 QSOCKS5_DEBUG << "got [" << address << ':' << port << ']';
234 *pAddress = address;
235 *pPort = port;
236 *pPos = pos;
237 }
238
239 return ret;
240}

331QSocks5BindData *QSocks5BindStore::retrieve(qintptr socketDescriptor)
332{
333 QMutexLocker lock(&mutex);
334 const auto it = store.constFind(key: socketDescriptor);
335 if (it == store.cend())
336 return nullptr;
337 QSocks5BindData *bindData = it.value();
338 store.erase(it);
339 if (bindData) {
340 if (bindData->controlSocket->thread() != QThread::currentThread()) {
341 qWarning(msg: "Cannot access socks5 bind data from different thread");
342 return nullptr;
343 }
344 } else {
345 QSOCKS5_DEBUG << "__ERROR__ binddata == 0";
346 }
347 // stop the sweep timer if not needed
348 if (store.isEmpty()) {
349 killTimer(id: sweepTimerId);
350 sweepTimerId = -1;
351 }
352 return bindData;

427{
428 *completed = false;
429 QByteArray uname = userName.toLatin1();
430 QByteArray passwd = password.toLatin1();

432 char *buf = dataBuf.data();

434 buf[pos++] = S5_PASSWORDAUTH_VERSION;
435 buf[pos++] = uname.size();
436 memcpy(dest: &buf[pos], src: uname.data(), n: uname.size());
437 pos += uname.size();
438 buf[pos++] = passwd.size();
439 memcpy(dest: &buf[pos], src: passwd.data(), n: passwd.size());
440 return socket->write(data: dataBuf) == dataBuf.size();
441}

542void QSocks5SocketEnginePrivate::setErrorState(Socks5State state, const QString &extraMessage)
543{
544 Q_Q(QSocks5SocketEngine);
545
546 switch (state) {
547 case Uninitialized:
548 case Authenticating:
549 case AuthenticationMethodsSent:
550 case RequestMethodSent:
551 case Connected:
552 case UdpAssociateSuccess:
553 case BindSuccess:
554 // these aren't error states
555 return;
556
557 case ConnectError:
558 case ControlSocketError: {
559 QAbstractSocket::SocketError controlSocketError = data->controlSocket->error();
560 if (socks5State != Connected) {
561 switch (controlSocketError) {
562 case QAbstractSocket::ConnectionRefusedError:
563 q->setError(error: QAbstractSocket::ProxyConnectionRefusedError,
564 errorString: QSocks5SocketEngine::tr(s: "Connection to proxy refused"));
565 break;
566 case QAbstractSocket::RemoteHostClosedError:
567 q->setError(error: QAbstractSocket::ProxyConnectionClosedError,
568 errorString: QSocks5SocketEngine::tr(s: "Connection to proxy closed prematurely"));
569 break;
570 case QAbstractSocket::HostNotFoundError:
571 q->setError(error: QAbstractSocket::ProxyNotFoundError,
572 errorString: QSocks5SocketEngine::tr(s: "Proxy host not found"));
573 break;
574 case QAbstractSocket::SocketTimeoutError:
575 if (state == ConnectError) {
576 q->setError(error: QAbstractSocket::ProxyConnectionTimeoutError,
577 errorString: QSocks5SocketEngine::tr(s: "Connection to proxy timed out"));
578 break;
579 }
580 Q_FALLTHROUGH();
581 default:
582 q->setError(error: controlSocketError, errorString: data->controlSocket->errorString());
583 break;
584 }
585 } else {
586 q->setError(error: controlSocketError, errorString: data->controlSocket->errorString());
587 }
588 break;
589 }
590
591 case AuthenticatingError:
592 q->setError(error: QAbstractSocket::ProxyAuthenticationRequiredError,
593 errorString: extraMessage.isEmpty() ?
594 QSocks5SocketEngine::tr(s: "Proxy authentication failed") :
595 QSocks5SocketEngine::tr(s: "Proxy authentication failed: %1").arg(a: extraMessage));
596 break;
597
598 case RequestError:
599 // error code set by caller (overload)
600 break;
601
602 case SocksError:
603 q->setError(error: QAbstractSocket::ProxyProtocolError,
604 errorString: QSocks5SocketEngine::tr(s: "SOCKS version 5 protocol error"));
605 break;
606
607 case HostNameLookupError:
608 q->setError(error: QAbstractSocket::HostNotFoundError,
609 errorString: QAbstractSocket::tr(s: "Host not found"));
610 break;
611 }
612
613 q->setState(QAbstractSocket::UnconnectedState);
614 socks5State = state;
615}

663void QSocks5SocketEnginePrivate::reauthenticate()
664{
665 Q_Q(QSocks5SocketEngine);
666
667 // we require authentication
668 QAuthenticator auth;
669 q->proxyAuthenticationRequired(proxy: proxyInfo, authenticator: &auth);
670
671 if (!auth.user().isEmpty() || !auth.password().isEmpty()) {
672 // we have new credentials, let's try again
673 QSOCKS5_DEBUG << "authentication failure: retrying connection";
674 socks5State = QSocks5SocketEnginePrivate::Uninitialized;
675
676 delete data->authenticator;
677 proxyInfo.setUser(auth.user());
678 proxyInfo.setPassword(auth.password());
679 data->authenticator = new QSocks5PasswordAuthenticator(proxyInfo.user(), proxyInfo.password());
680
681 {
682 const QSignalBlocker blocker(data->controlSocket);
683 data->controlSocket->abort();
684 }
685 data->controlSocket->connectToHost(hostName: proxyInfo.hostName(), port: proxyInfo.port());
686 } else {
687 // authentication failure
688
689 setErrorState(state: AuthenticatingError);
690 data->controlSocket->close();
691 emitConnectionNotification();
692 }

696{
697 // not enough data to begin
698 if (data->controlSocket->bytesAvailable() < 2)
699 return;
700
701 QByteArray buf = data->controlSocket->read(maxlen: 2);
702 if (buf.at(i: 0) != S5_VERSION_5) {
703 QSOCKS5_D_DEBUG << "Socks5 version incorrect";
704 setErrorState(state: SocksError);
705 data->controlSocket->close();
706 emitConnectionNotification();
707 return;
708 }
709
710 bool authComplete = false;
711 if (uchar(buf.at(i: 1)) == S5_AUTHMETHOD_NONE) {
712 authComplete = true;
713 } else if (uchar(buf.at(i: 1)) == S5_AUTHMETHOD_NOTACCEPTABLE) {
714 reauthenticate();
715 return;
716 } else if (buf.at(i: 1) != data->authenticator->methodId()
717 || !data->authenticator->beginAuthenticate(socket: data->controlSocket, completed: &authComplete)) {
718 setErrorState(state: AuthenticatingError, extraMessage: "Socks5 host did not support authentication method."_L1);
719 socketError = QAbstractSocket::SocketAccessError; // change the socket error
720 emitConnectionNotification();
721 return;
722 }
723
724 if (authComplete)
725 sendRequestMethod();
726 else
727 socks5State = Authenticating;
728}

786void QSocks5SocketEnginePrivate::parseRequestMethodReply()
787{
788 Q_Q(QSocks5SocketEngine);
789 QSOCKS5_DEBUG << "parseRequestMethodReply()";
790
791 QByteArray inBuf;
792 if (!data->authenticator->unSeal(sealedSocket: data->controlSocket, buf: &inBuf)) {
793 // ### check error and not just not enough data
794 QSOCKS5_DEBUG << "unSeal failed, needs more data";
795 return;
796 }
797
798 inBuf.prepend(a: receivedHeaderFragment);
799 receivedHeaderFragment.clear();
800 QSOCKS5_DEBUG << dump(inBuf);
801 if (inBuf.size() < 3) {
802 QSOCKS5_DEBUG << "need more data for request reply header .. put this data somewhere";
803 receivedHeaderFragment = inBuf;
804 return;
805 }
806
807 QHostAddress address;
808 quint16 port = 0;
809
810 if (inBuf.at(i: 0) != S5_VERSION_5 || inBuf.at(i: 2) != 0x00) {
811 QSOCKS5_DEBUG << "socks protocol error";
812 setErrorState(state: SocksError);
813 } else if (inBuf.at(i: 1) != S5_SUCCESS) {
814 Socks5Error socks5Error = Socks5Error(inBuf.at(i: 1));
815 QSOCKS5_DEBUG << "Request error :" << socks5Error;
816 if ((socks5Error == SocksFailure || socks5Error == ConnectionNotAllowed)
817 && !peerName.isEmpty()) {
818 // Dante seems to use this error code to indicate hostname resolution failure
819 setErrorState(state: HostNameLookupError);
820 } else {
821 setErrorState(state: RequestError, socks5error: socks5Error);
822 }
823 } else {
824 // connection success, retrieve the remote addresses
825 int pos = 3;
826 int err = qt_socks5_get_host_address_and_port(buf: inBuf, pAddress: &address, pPort: &port, pPos: &pos);
827 if (err == -1) {
828 QSOCKS5_DEBUG << "error getting address";
829 setErrorState(state: SocksError);
830 } else if (err == 0) {
831 //need more data
832 receivedHeaderFragment = inBuf;
833 return;
834 } else {
835 inBuf.remove(index: 0, len: pos);
836 for (int i = inBuf.size() - 1; i >= 0 ; --i)
837 data->controlSocket->ungetChar(c: inBuf.at(i));
838 }
839 }
840
841 if (socks5State == RequestMethodSent) {
842 // no error
843 localAddress = address;
844 localPort = port;
845
846 if (mode == ConnectMode) {
847 inboundStreamCount = outboundStreamCount = 1;
848 socks5State = Connected;
849 // notify the upper layer that we're done
850 q->setState(QAbstractSocket::ConnectedState);
851 emitConnectionNotification();
852 } else if (mode == BindMode) {
853 socks5State = BindSuccess;
854 q->setState(QAbstractSocket::ListeningState);
855 } else {
856 socks5State = UdpAssociateSuccess;
857 }
858 } else if (socks5State == BindSuccess) {
859 // no error and we got a connection
860 bindData->peerAddress = address;
861 bindData->peerPort = port;
862
863 emitReadNotification();
864 } else {
865 // got an error
866 data->controlSocket->close();
867 emitConnectionNotification();
868 }
869}

1133{

1135 << "bytes available" << data->controlSocket->bytesAvailable();

1284bool QSocks5SocketEngine::bind(const QHostAddress &addr, quint16 port)
1285{
1286 Q_D(QSocks5SocketEngine);
1287
1288 // when bind we will block until the bind is finished as the info from the proxy server is needed
1289
1290 QHostAddress address;
1291 if (addr.protocol() == QAbstractSocket::AnyIPProtocol)
1292 address = QHostAddress::AnyIPv4; //SOCKS5 doesn't support dual stack, and there isn't any implementation of udp on ipv6 yet
1293 else
1294 address = addr;
1295
1296 if (!d->data) {
1297 if (socketType() == QAbstractSocket::TcpSocket) {
1298 d->initialize(socks5Mode: QSocks5SocketEnginePrivate::BindMode);
1299#ifndef QT_NO_UDPSOCKET
1300 } else if (socketType() == QAbstractSocket::UdpSocket) {
1301 d->initialize(socks5Mode: QSocks5SocketEnginePrivate::UdpAssociateMode);
1302#endif
1303 } else {
1304 //### something invalid
1305 return false;
1306 }
1307 }
1308
1309#ifndef QT_NO_UDPSOCKET
1310 if (d->mode == QSocks5SocketEnginePrivate::UdpAssociateMode) {
1311 if (!d->udpData->udpSocket->bind(address, port)) {
1312 QSOCKS5_Q_DEBUG << "local udp bind failed";
1313 setError(error: d->udpData->udpSocket->error(), errorString: d->udpData->udpSocket->errorString());
1314 return false;
1315 }
1316 d->localAddress = d->udpData->udpSocket->localAddress();
1317 d->localPort = d->udpData->udpSocket->localPort();
1318 } else
1319#endif
1320 if (d->mode == QSocks5SocketEnginePrivate::BindMode) {
1321 d->localAddress = address;
1322 d->localPort = port;
1323 } else {
1324 //### something invalid
1325 return false;
1326 }
1327
1328 int msecs = SOCKS5_BLOCKING_BIND_TIMEOUT;
1329 QElapsedTimer stopWatch;
1330 stopWatch.start();
1331 d->data->controlSocket->connectToHost(hostName: d->proxyInfo.hostName(), port: d->proxyInfo.port());
1332 if (!d->waitForConnected(msecs, timedOut: nullptr) ||
1333 d->data->controlSocket->state() == QAbstractSocket::UnconnectedState) {
1334 // waitForConnected sets the error state and closes the socket
1335 QSOCKS5_Q_DEBUG << "waitForConnected to proxy server" << d->data->controlSocket->errorString();
1336 return false;
1337 }
1338 if (d->socks5State == QSocks5SocketEnginePrivate::BindSuccess) {
1339 setState(QAbstractSocket::BoundState);
1340 return true;
1341#ifndef QT_NO_UDPSOCKET
1342 } else if (d->socks5State == QSocks5SocketEnginePrivate::UdpAssociateSuccess) {
1343 setState(QAbstractSocket::BoundState);
1344 d->udpData->associateAddress = d->localAddress;
1345 d->localAddress = QHostAddress();
1346 d->udpData->associatePort = d->localPort;
1347 d->localPort = 0;
1348 return true;
1349#endif // QT_NO_UDPSOCKET
1350 }
1351
1352 // binding timed out

1714
1715 QElapsedTimer stopWatch;
1716 stopWatch.start();
