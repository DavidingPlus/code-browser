15class QOpenGLTextureCacheWrapper
16{
17public:
18 QOpenGLTextureCacheWrapper()
19 {
20 QImagePixmapCleanupHooks::instance()->addPlatformPixmapModificationHook(cleanupTexturesForPixmapData);
21 QImagePixmapCleanupHooks::instance()->addPlatformPixmapDestructionHook(cleanupTexturesForPixmapData);
22 QImagePixmapCleanupHooks::instance()->addImageHook(cleanupTexturesForCacheKey);
23 }
24
25 ~QOpenGLTextureCacheWrapper()
26 {
27 QImagePixmapCleanupHooks::instance()->removePlatformPixmapModificationHook(cleanupTexturesForPixmapData);
28 QImagePixmapCleanupHooks::instance()->removePlatformPixmapDestructionHook(cleanupTexturesForPixmapData);
29 QImagePixmapCleanupHooks::instance()->removeImageHook(cleanupTexturesForCacheKey);
30 }

32 QOpenGLTextureCache *cacheForContext(QOpenGLContext *context) {
33 QMutexLocker lock(&m_mutex);
34 return m_resource.value<QOpenGLTextureCache>(context);
35 }

37 static void cleanupTexturesForCacheKey(qint64 key);
38 static void cleanupTexturesForPixmapData(QPlatformPixmap *pmd);
39
40private:
41 QOpenGLMultiGroupSharedResource m_resource;
42 QMutex m_mutex;
43};

52void QOpenGLTextureCacheWrapper::cleanupTexturesForCacheKey(qint64 key)
53{
54 QList<QOpenGLSharedResource *> resources = qt_texture_caches()->m_resource.resources();
55 for (QList<QOpenGLSharedResource *>::iterator it = resources.begin(); it != resources.end(); ++it)
56 static_cast<QOpenGLTextureCache *>(*it)->invalidate(key);
57}

59void QOpenGLTextureCacheWrapper::cleanupTexturesForPixmapData(QPlatformPixmap *pmd)
60{
61 cleanupTexturesForCacheKey(key: pmd->cacheKey());
62}

64static quint64 cacheSize()
65{
66 bool ok = false;
67 const int envCacheSize = qEnvironmentVariableIntValue(varName: "QT_OPENGL_TEXTURE_CACHE_SIZE", ok: &ok);
68 if (ok)
69 return envCacheSize;
70
71 return 1024 * 1024; // 1024 MB cache default
72}

84GLuint QOpenGLTextureCache::bindTexture(QOpenGLContext *context, const QPixmap &pixmap, QOpenGLTextureUploader::BindOptions options)
85{
86 if (pixmap.isNull())
87 return 0;
88 QMutexLocker locker(&m_mutex);
89 qint64 key = pixmap.cacheKey();
90
91 // A QPainter is active on the image - take the safe route and replace the texture.
92 if (!pixmap.paintingActive()) {
93 QOpenGLCachedTexture *entry = m_cache.object(key);
94 if (entry && entry->options() == options) {
95 context->functions()->glBindTexture(GL_TEXTURE_2D, texture: entry->id());
96 return entry->id();
97 }
98 }
99
100 GLuint id = bindTexture(context, key, image: pixmap.toImage(), options);
101 if (id > 0)
102 QImagePixmapCleanupHooks::enableCleanupHooks(pixmap);
103
104 return id;
105}

107GLuint QOpenGLTextureCache::bindTexture(QOpenGLContext *context, const QImage &image, QOpenGLTextureUploader::BindOptions options)
108{
109 if (image.isNull())
110 return 0;
111 QMutexLocker locker(&m_mutex);
112 qint64 key = image.cacheKey();
113
114 // A QPainter is active on the image - take the safe route and replace the texture.
115 if (!image.paintingActive()) {
116 QOpenGLCachedTexture *entry = m_cache.object(key);
117 if (entry && entry->options() == options) {
118 context->functions()->glBindTexture(GL_TEXTURE_2D, texture: entry->id());
119 return entry->id();
120 }
121 }
122
123 QImage img = image;
124 if (!context->functions()->hasOpenGLFeature(feature: QOpenGLFunctions::NPOTTextures))
125 options |= QOpenGLTextureUploader::PowerOfTwoBindOption;
126
127 GLuint id = bindTexture(context, key, image: img, options);
128 if (id > 0)
129 QImagePixmapCleanupHooks::enableCleanupHooks(image);
130
131 return id;
132}

137GLuint QOpenGLTextureCache::bindTexture(QOpenGLContext *context, qint64 key, const QImage &image, QOpenGLTextureUploader::BindOptions options)
138{
139 Q_TRACE_SCOPE(QOpenGLTextureCache_bindTexture, context, key, image.bits(), options);
140
141 GLuint id;
142 QOpenGLFunctions *funcs = context->functions();
143 funcs->glGenTextures(n: 1, textures: &id);
144 funcs->glBindTexture(GL_TEXTURE_2D, texture: id);
145
146 int cost = QOpenGLTextureUploader::textureImage(GL_TEXTURE_2D, image, options);
147
148 m_cache.insert(key, object: new QOpenGLCachedTexture(id, options, context), cost: cost / 1024);
149
150 return id;
151}

153void QOpenGLTextureCache::invalidate(qint64 key)
154{
155 QMutexLocker locker(&m_mutex);
156 m_cache.remove(key);
157}

169static void freeTexture(QOpenGLFunctions *funcs, GLuint id)
170{
171 funcs->glDeleteTextures(n: 1, textures: &id);
172}
