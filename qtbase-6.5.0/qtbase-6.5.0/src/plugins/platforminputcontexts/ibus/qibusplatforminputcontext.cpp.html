85QIBusPlatformInputContext::QIBusPlatformInputContext ()
86 : d(new QIBusPlatformInputContextPrivate())
87{
88 if (!d->usePortal) {
89 QString socketPath = QIBusPlatformInputContextPrivate::getSocketPath();
90 QFile file(socketPath);
91 if (file.open(flags: QFile::ReadOnly)) {
92#if QT_CONFIG(filesystemwatcher)
93 qCDebug(qtQpaInputMethods) << "socketWatcher.addPath" << socketPath;
94 // If KDE session save is used or restart ibus-daemon,
95 // the applications could run before ibus-daemon runs.
96 // We watch the getSocketPath() to get the launching ibus-daemon.
97 m_socketWatcher.addPath(file: socketPath);
98 connect(sender: &m_socketWatcher, SIGNAL(fileChanged(QString)), receiver: this, SLOT(socketChanged(QString)));
99#endif
100 }
101 m_timer.setSingleShot(true);
102 connect(sender: &m_timer, SIGNAL(timeout()), receiver: this, SLOT(connectToBus()));
103 }
104
105 QObject::connect(sender: &d->serviceWatcher, SIGNAL(serviceRegistered(QString)), receiver: this, SLOT(busRegistered(QString)));
106 QObject::connect(sender: &d->serviceWatcher, SIGNAL(serviceUnregistered(QString)), receiver: this, SLOT(busUnregistered(QString)));
107
108 connectToContextSignals();
109
110 QInputMethod *p = qApp->inputMethod();
111 connect(sender: p, SIGNAL(cursorRectangleChanged()), receiver: this, SLOT(cursorRectChanged()));
112 m_eventFilterUseSynchronousMode = false;
113 if (qEnvironmentVariableIsSet(varName: "IBUS_ENABLE_SYNC_MODE")) {
114 bool ok;
115 int enableSync = qEnvironmentVariableIntValue(varName: "IBUS_ENABLE_SYNC_MODE", ok: &ok);

185void QIBusPlatformInputContext::update(Qt::InputMethodQueries q)
186{
187 QObject *input = qApp->focusObject();
188
189 if (d->needsSurroundingText && input
190 && (q.testFlag(flag: Qt::ImSurroundingText)
191 || q.testFlag(flag: Qt::ImCursorPosition)
192 || q.testFlag(flag: Qt::ImAnchorPosition))) {
193
194 QInputMethodQueryEvent query(Qt::ImSurroundingText | Qt::ImCursorPosition | Qt::ImAnchorPosition);
195
196 QCoreApplication::sendEvent(receiver: input, event: &query);
197
198 QString surroundingText = query.value(query: Qt::ImSurroundingText).toString();
199 uint cursorPosition = query.value(query: Qt::ImCursorPosition).toUInt();
200 uint anchorPosition = query.value(query: Qt::ImAnchorPosition).toUInt();
201
202 QIBusText text;
203 text.text = surroundingText;
204
205 QVariant variant;
206 variant.setValue(text);
207 QDBusVariant dbusText(variant);
208
209 d->context->SetSurroundingText(text: dbusText, cursor_pos: cursorPosition, anchor_pos: anchorPosition);
210 }

213void QIBusPlatformInputContext::cursorRectChanged()
214{
215 if (!d->busConnected)
216 return;
217
218 QRect r = qApp->inputMethod()->cursorRectangle().toRect();
219 if (!r.isValid())
220 return;
221
222 QWindow *inputWindow = qApp->focusWindow();
223 if (!inputWindow)
224 return;
225 if (!inputWindow->screen())
226 return;
227
228 if (QGuiApplication::platformName().startsWith(s: "wayland"_L1)) {
229 auto margins = inputWindow->frameMargins();
230 r.translate(dx: margins.left(), dy: margins.top());
231 qreal scale = inputWindow->devicePixelRatio();
232 QRect newRect = QRect(r.x() * scale, r.y() * scale, r.width() * scale, r.height() * scale);
233 if (debug)
234 qDebug() << "microFocus" << newRect;
235 d->context->SetCursorLocationRelative(x: newRect.x(), y: newRect.y(),
236 w: newRect.width(), h: newRect.height());
237 return;
238 }
239
240 // x11/xcb
241 auto screenGeometry = inputWindow->screen()->geometry();
242 auto point = inputWindow->mapToGlobal(pos: r.topLeft());
243 qreal scale = inputWindow->devicePixelRatio();
244 auto native = (point - screenGeometry.topLeft()) * scale + screenGeometry.topLeft();
245 QRect newRect(native, r.size() * scale);
246 if (debug)
247 qDebug() << "microFocus" << newRect;
248 d->context->SetCursorLocation(x: newRect.x(), y: newRect.y(),
249 w: newRect.width(), h: newRect.height());
250}

320void QIBusPlatformInputContext::forwardKeyEvent(uint keyval, uint keycode, uint state)
321{
322 if (!qApp)
323 return;
324
325 QObject *input = qApp->focusObject();
326 if (!input)
327 return;
328
329 QEvent::Type type = QEvent::KeyPress;
330 if (state & IBUS_RELEASE_MASK)
331 type = QEvent::KeyRelease;
332
333 state &= ~IBUS_RELEASE_MASK;
334 keycode += 8;
335
336 Qt::KeyboardModifiers modifiers = Qt::NoModifier;
337 if (state & IBUS_SHIFT_MASK)
338 modifiers |= Qt::ShiftModifier;
339 if (state & IBUS_CONTROL_MASK)
340 modifiers |= Qt::ControlModifier;
341 if (state & IBUS_MOD1_MASK)
342 modifiers |= Qt::AltModifier;
343 if (state & IBUS_META_MASK)
344 modifiers |= Qt::MetaModifier;
345
346 int qtcode = QXkbCommon::keysymToQtKey(keysym: keyval, modifiers);
347 QString text = QXkbCommon::lookupStringNoKeysymTransformations(keysym: keyval);
348
349 if (debug)
350 qDebug() << "forwardKeyEvent" << keyval << keycode << state << modifiers << qtcode << text;
351
352 QKeyEvent event(type, qtcode, modifiers, keycode, keyval, state, text);
353 QCoreApplication::sendEvent(receiver: input, event: &event);
354}

399bool QIBusPlatformInputContext::filterEvent(const QEvent *event)
400{
401 if (!d->busConnected)
402 return false;
403
404 if (!inputMethodAccepted())
405 return false;
406
407 const QKeyEvent *keyEvent = static_cast<const QKeyEvent *>(event);
408 quint32 sym = keyEvent->nativeVirtualKey();
409 quint32 code = keyEvent->nativeScanCode();
410 quint32 state = keyEvent->nativeModifiers();
411 quint32 ibusState = state;
412
413 if (keyEvent->type() != QEvent::KeyPress)
414 ibusState |= IBUS_RELEASE_MASK;
415
416 QDBusPendingReply<bool> reply = d->context->ProcessKeyEvent(keyval: sym, keycode: code - 8, state: ibusState);
417
418 if (m_eventFilterUseSynchronousMode || reply.isFinished()) {
419 bool filtered = reply.value();
420 qCDebug(qtQpaInputMethods) << "filterEvent return" << code << sym << state << filtered;
421 return filtered;
422 }
423
424 Qt::KeyboardModifiers modifiers = keyEvent->modifiers();
425 const int qtcode = keyEvent->key();
426
427 // From QKeyEvent::modifiers()
428 switch (qtcode) {
429 case Qt::Key_Shift:
430 modifiers ^= Qt::ShiftModifier;
431 break;
432 case Qt::Key_Control:
433 modifiers ^= Qt::ControlModifier;
434 break;
435 case Qt::Key_Alt:
436 modifiers ^= Qt::AltModifier;
437 break;
438 case Qt::Key_Meta:
439 modifiers ^= Qt::MetaModifier;
440 break;
441 case Qt::Key_AltGr:
442 modifiers ^= Qt::GroupSwitchModifier;
443 break;
444 }
445
446 QVariantList args;
447 args << QVariant::fromValue(value: keyEvent->timestamp());
448 args << QVariant::fromValue(value: static_cast<uint>(keyEvent->type()));
449 args << QVariant::fromValue(value: qtcode);
450 args << QVariant::fromValue(value: code) << QVariant::fromValue(value: sym) << QVariant::fromValue(value: state);
451 args << QVariant::fromValue(value: keyEvent->text());
452 args << QVariant::fromValue(value: keyEvent->isAutoRepeat());
453
454 QIBusFilterEventWatcher *watcher = new QIBusFilterEventWatcher(reply, this, QGuiApplication::focusWindow(), modifiers, args);
455 QObject::connect(sender: watcher, signal: &QDBusPendingCallWatcher::finished, receiver: this, slot: &QIBusPlatformInputContext::filterEventFinished);
456
457 return true;
458}

460void QIBusPlatformInputContext::filterEventFinished(QDBusPendingCallWatcher *call)
461{
462 QIBusFilterEventWatcher *watcher = (QIBusFilterEventWatcher *) call;
463 QDBusPendingReply<bool> reply = *call;
464
465 if (reply.isError()) {
466 call->deleteLater();
467 return;
468 }
469
470 // Use watcher's window instead of the current focused window
471 // since there is a time lag until filterEventFinished() returns.
472 QWindow *window = watcher->window();
473
474 if (!window) {
475 call->deleteLater();
476 return;
477 }
478
479 Qt::KeyboardModifiers modifiers = watcher->modifiers();
480 QVariantList args = watcher->arguments();
481 const ulong time = static_cast<ulong>(args.at(i: 0).toUInt());
482 const QEvent::Type type = static_cast<QEvent::Type>(args.at(i: 1).toUInt());
483 const int qtcode = args.at(i: 2).toInt();
484 const quint32 code = args.at(i: 3).toUInt();
485 const quint32 sym = args.at(i: 4).toUInt();
486 const quint32 state = args.at(i: 5).toUInt();
487 const QString string = args.at(i: 6).toString();
488 const bool isAutoRepeat = args.at(i: 7).toBool();
489
490 // copied from QXcbKeyboard::handleKeyEvent()
491 bool filtered = reply.value();
492 qCDebug(qtQpaInputMethods) << "filterEventFinished return" << code << sym << state << filtered;
493 if (!filtered) {
494#ifndef QT_NO_CONTEXTMENU
495 if (type == QEvent::KeyPress && qtcode == Qt::Key_Menu
496 && window != nullptr) {
497 const QPoint globalPos = window->screen()->handle()->cursor()->pos();
498 const QPoint pos = window->mapFromGlobal(pos: globalPos);
499 QWindowSystemInterfacePrivate::ContextMenuEvent contextMenuEvent(window, false, pos,
500 globalPos, modifiers);
501 QGuiApplicationPrivate::processWindowSystemEvent(e: &contextMenuEvent);
502 }
503#endif
504 QWindowSystemInterfacePrivate::KeyEvent keyEvent(window, time, type, qtcode, modifiers,
505 code, sym, state, string, isAutoRepeat);
506 QGuiApplicationPrivate::processWindowSystemEvent(e: &keyEvent);
507 }
508 call->deleteLater();
509}

639void QIBusPlatformInputContextPrivate::createBusProxy()
640{
641 QDBusConnection connection("QIBusProxy"_L1);
642 if (!connection.isConnected())
643 return;
644
645 const char* ibusService = usePortal ? "org.freedesktop.portal.IBus" : "org.freedesktop.IBus";
646 QDBusReply<QDBusObjectPath> ic;
647 if (usePortal) {
648 portalBus = std::make_unique<QIBusProxyPortal>(args: QLatin1StringView(ibusService),
649 args: "/org/freedesktop/IBus"_L1,
650 args&: connection);
651 if (!portalBus->isValid()) {
652 qWarning(msg: "QIBusPlatformInputContext: invalid portal bus.");
653 return;
654 }
655
656 ic = portalBus->CreateInputContext(name: "QIBusInputContext"_L1);
657 } else {
658 bus = std::make_unique<QIBusProxy>(args: QLatin1StringView(ibusService),
659 args: "/org/freedesktop/IBus"_L1,
660 args&: connection);
661 if (!bus->isValid()) {
662 qWarning(msg: "QIBusPlatformInputContext: invalid bus.");
663 return;
664 }
665
666 ic = bus->CreateInputContext(name: "QIBusInputContext"_L1);
667 }
668
669 serviceWatcher.removeWatchedService(service: ibusService);
670 serviceWatcher.setConnection(connection);
671 serviceWatcher.addWatchedService(newService: ibusService);
672
673 if (!ic.isValid()) {
674 qWarning(msg: "QIBusPlatformInputContext: CreateInputContext failed.");
675 return;
676 }
677
678 context = std::make_unique<QIBusInputContextProxy>(args: QLatin1StringView(ibusService), args: ic.value().path(), args&: connection);
679
680 if (!context->isValid()) {
681 qWarning(msg: "QIBusPlatformInputContext: invalid input context.");
682 return;
683 }

700QString QIBusPlatformInputContextPrivate::getSocketPath()
