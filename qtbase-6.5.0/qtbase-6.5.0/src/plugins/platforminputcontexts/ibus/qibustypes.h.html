44 UnderlineError = 4,
45 };
46
47 QIBusAttribute();
48
49 QTextCharFormat format() const;
50
51 void serializeTo(QDBusArgument &argument) const;
52 void deserializeFrom(const QDBusArgument &argument);
53
54 Type type;
55 quint32 value;
56 quint32 start;
57 quint32 end;
58};
